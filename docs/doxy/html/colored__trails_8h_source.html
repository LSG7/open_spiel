26// A simple bargaining game [1].
27//
28// This code currently implements the three-player imperfect information game
29// from these papers [2,3] but we would like this to be a generic implementation
30// that can handle the several variants of the classic game.
31//
32// [1] Ya'akov Gal, Barbara Grosz, Sarit Kraus, Avi Pfeffer, and Stuart Shieber.
33// 2010. Agent decision-making in open mixed networks. Artificial
34// Intelligence 174(18): 1460-1480.
35// [2] de Jong et al. '11, Metastrategies in the Colored Trails Game
36// https://www.ifaamas.org/Proceedings/aamas2011/papers/C4_R57.pdf
37// [3] S. G. Ficici and A. Pfeffer. Modeling how Humans Reason about Others with
38// Partial Information. In Proceedings of the Seventh International
39// Conference on Autonomous Agents and Multiagent Systems (AAMAS), 2008.
40//
41// Parameters:
42// "boards_file" string The file containing the boards (default: "")
43// "board_size" int number of rows / columns (default = 4)
44// "num_colors" int number of colors (default = 5)
45// "players" int number of players (default = 3)
46

203class ColoredTrailsGame : public Game {

207 int NumDistinctActions() const override;
208 std::unique_ptr<State> NewInitialState() const override {
209 return std::unique_ptr<State>(
210 new ColoredTrailsState(shared_from_this(), board_size_, num_colors_));
211 }
212 int MaxChanceOutcomes() const override { return all_boards_.size(); }
213
214 int MaxGameLength() const override { return 3; }
215 int MaxChanceNodesInHistory() const override { return MaxGameLength(); }
216
217 int NumPlayers() const override { return num_players_; }
218 double MaxUtility() const override {
219 // Get max chips, then do a 1-for-8 trade, and only use 1 chip.
220 // = 0 (for reaching goal) + (8 - 1 + 8) * leftover_chip_value
221 return kLeftoverChipScore * (kNumChipsUpperBound - 1 + kNumChipsUpperBound);
222 }
223 double MinUtility() const override {
224 // No chips left and as far away from the goal as possible.
225 return board_size_ * board_size_ * kFlagPenaltyPerCell;
226 }
227 std::vector<int> ObservationTensorShape() const override;
228 std::vector<int> InformationStateTensorShape() const override;
229
230 const std::vector<Board>& AllBoards() const { return all_boards_; }
231
232 const Trade& LookupTrade(int trade_id) const {
233 if (trade_id == PassAction()) {
234 return pass_trade_;
235 } else {
236 return *(trade_info_.possible_trades.at(trade_id));
237 }
238 }
239
240 Action ResponderTradeWithPlayerAction(Player player) const {
241 SPIEL_CHECK_GE(player, 0);
242 SPIEL_CHECK_LE(player, 1);
243 return NumDistinctActions() - 3 + player;
244 }
245
246 Action PassAction() const { return NumDistinctActions() - 1; }
247
248 int LookupTradeId(const std::string& trade_str) const {
249 return trade_info_.trade_str_to_id.at(trade_str);
