146 // A tabular policy represented internally as a map. Note that this
147 // implementation is not directly compatible with the Python TabularPolicy
148 // implementation; the latter is implemented as a table of size
149 // [num_states, num_actions], while this is implemented as a map. It is
150 // non-trivial to convert between the two, but we have a function that does so
151 // in the open_spiel/python/policy.py file.
152 class TabularPolicy : public Policy {
153  public:
154  TabularPolicy() = default;
155  TabularPolicy(const Game& game); // Construct a uniform random policy.
156  TabularPolicy(const TabularPolicy& other) = default;
157  TabularPolicy(const std::unordered_map<std::string, ActionsAndProbs>& table)
158  : policy_table_(table) {}
159 
160  // Converts a policy to a TabularPolicy.
161  TabularPolicy(const Game& game, const Policy& policy) : TabularPolicy(game) {
162  for (auto& [infostate, is_policy] : policy_table_) {
163  is_policy = policy.GetStatePolicy(infostate);
164  }
165  }
166 
167  // Creates a new TabularPolicy from a deterministic policy encoded as a
168  // {info_state_str -> action} dict. The dummy_policy is used to initialize
169  // the initial mapping.
170  TabularPolicy(const TabularPolicy& dummy_policy,
171  const std::unordered_map<std::string, Action>& action_map)
172  : policy_table_(dummy_policy.policy_table_) {
173  for (const auto& entry : action_map) {
174  std::string info_state = entry.first;
175  Action action_taken = action_map.at(entry.first);
176  for (auto& action_and_prob : policy_table_[info_state]) {
177  action_and_prob.second =
178  (action_and_prob.first == action_taken ? 1.0 : 0.0);
179  }
180  }
181  }
182 
183  ActionsAndProbs GetStatePolicy(const std::string& info_state) const override {
184  auto iter = policy_table_.find(info_state);
185  if (iter == policy_table_.end()) {
186  return {};
187  } else {
188  return iter->second;
189  }
190  }
191 
192  std::string Serialize(int double_precision = -1,
193  std::string delimiter = "<~>") const override {
194  SPIEL_CHECK_GE(double_precision, -1);
195  if (delimiter == "," || delimiter == "=") {
196  // The two delimiters are used for de/serialization of policy_table_
197  SpielFatalError(
198  "Please select a different delimiter,"
199  "invalid values are \",\" and \"=\".");
200  }
201  std::string str = "TabularPolicy:";
202  if (policy_table_.empty()) return str;
203 
204  for (auto const& [info_state, policy] : policy_table_) {
205  if (info_state.find(delimiter) != std::string::npos) {
206  SpielFatalError(absl::StrCat(
207  "Info state contains delimiter \"", delimiter,
208  "\", please fix the info state or select a different delimiter."));
209  }
210 
211  std::string policy_str;
212  if (double_precision == -1) {
213  policy_str =
214  absl::StrJoin(policy, ",",
215  absl::PairFormatter(absl::AlphaNumFormatter(), "=",
216  HexDoubleFormatter()));
217  } else {
218  policy_str = absl::StrJoin(
219  policy, ",",
220  absl::PairFormatter(absl::AlphaNumFormatter(), "=",
221  SimpleDoubleFormatter(double_precision)));
222  }
223  absl::StrAppend(&str, info_state, delimiter, policy_str, delimiter);
224  }
225  // Remove the trailing delimiter
226  str.erase(str.length() - delimiter.length());
227  return str;
228  }

230  // Set (overwrite) all the state policies contained in another policy within
231  // this policy. Does not change other state policies not contained in this
232  // policy.
233  void ImportPolicy(const TabularPolicy& other_policy) {
234  for (const auto& [info_state, actions_and_probs] :
235  other_policy.policy_table_) {
236  SetStatePolicy(info_state, actions_and_probs);
237  }
238  }

240  // Set the probability for action at the info state. If the info state is not
241  // in the policy, it is added. If the action is not in the info state policy,
242  // it is added. Otherwise it is modified.
243  void SetProb(const std::string& info_state, Action action, double prob) {
244  auto iter = policy_table_.find(info_state);
245  if (iter == policy_table_.end()) {
246  auto iter_and_bool = policy_table_.insert({info_state, {}});
247  iter = iter_and_bool.first;
248  }
249  open_spiel::SetProb(&(iter->second), action, prob);
250  }

252  void SetStatePolicy(const std::string& info_state,
253  const ActionsAndProbs& state_policy) {
254  policy_table_[info_state] = state_policy;
255  }
