123void FlatJointactionTest() {
124 FlatJointActionTestState state;
125 auto legal_flat_joint_actions = state.LegalActions(kSimultaneousPlayerId);
126 SPIEL_CHECK_EQ(legal_flat_joint_actions.size(), 18);
127 for (int i = 0; i < 18; ++i) {
128 std::cerr << "Joint action " << i << " expands to "
129 << state.ActionToString(kSimultaneousPlayerId, i) << std ::endl;
130 }
131 // Last-but-one joint action --> last action for everyone except p0 (which
132 // takes its last-but-one action).
133 SPIEL_CHECK_EQ(state.ActionToString(kSimultaneousPlayerId, 16),
134 "[(p=0,a=4), (p=1,a=5), (p=2,a=100)]");
135 state.ApplyAction(16);
136 std::vector<Action> expected_joint_action{4, 5, 100};
137 SPIEL_CHECK_EQ(state.JointAction(), expected_joint_action);
138}

168void LeducPokerDeserializeTest() {
169 // Example Leduc state: player 1 gets the 0th card, player 2 gets the 3rd card
170 // and the first two actions are: check, check.
171 std::string serialized_game_and_state =
172 "# Automatically generated by OpenSpiel SerializeGameAndState\n"
173 "[Meta]\n"
174 "Version: 1\n"
175 "\n"
176 "[Game]\n"
177 "leduc_poker()\n"
178 "[State]\n"
179 "0\n" // first chance event (deal to first player)
180 "3\n" // second chance event (deal to second player)
181 "1\n" // check
182 "1\n" // check
183 "\n";
184
185 std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>
186 game_and_state =
187 open_spiel::DeserializeGameAndState(serialized_game_and_state);
188
189 // Should be at round 2 deal (chance node).
190 SPIEL_CHECK_TRUE(game_and_state.second->IsChanceNode());
191
192 // Check that the game got deserialized properly.
193 SPIEL_CHECK_EQ(game_and_state.first->ToString(),
194 LoadGame("leduc_poker")->ToString());
195
196 // And now check that serializing this game and state gives the same string as
197 // above.
198 SPIEL_CHECK_EQ(
199 SerializeGameAndState(*game_and_state.first, *game_and_state.second),
200 serialized_game_and_state);
201}

203void GameParametersTest() {
204 // Basic types
205 SPIEL_CHECK_TRUE(GameParameter(1).has_int_value());
206 SPIEL_CHECK_TRUE(GameParameter(1.0).has_double_value());
207 SPIEL_CHECK_TRUE(GameParameter(true).has_bool_value());
208 SPIEL_CHECK_TRUE(GameParameter(std::string("1")).has_string_value());
209 SPIEL_CHECK_TRUE(GameParameter("1").has_string_value()); // See issue #380.
210
211 // Writing to string
212 SPIEL_CHECK_EQ(GameParameter("1").ToString(), "1");
213 SPIEL_CHECK_EQ(GameParameter(1).ToString(), "1");
214 // -- Currently we serialize doubles with 10 digits after the point.
215 SPIEL_CHECK_EQ(GameParameter(1.0).ToString(), "1.0");
216 SPIEL_CHECK_EQ(GameParameter(1.).ToString(), "1.0");
217 SPIEL_CHECK_EQ(GameParameter(1.5).ToString(), "1.5");
218 SPIEL_CHECK_EQ(GameParameter(001.0485760000).ToString(), "1.048576");
219 SPIEL_CHECK_EQ(GameParameter(1e-9).ToString(), "0.000000001");
220
221 // Parsing from string
222 //
223 // XXX: Game parameter parsing from string is a bit quirky at the
224 // moment. For example, the strings "+" or "-" make the parser
225 // throw since the parses eagerly tries to parse those as integers and
226 // passes them to std::stoi.
227 //
228 // Similarly, "." would be parsed using std::stod with a similar outcome.
229 //
230 // Doubles must contain a point . inside, or they would be parsed as
231 // integer, and exponential notation is not allowed for now.
232 //
233 // Leading or trailing whitespace is not stripped before parsing, so " 1"
234 // would be parsed as a string instead of an integer.
235 //
236 // See also: #382.
237 //
238 //
239 // The next few tests are not always intended to check the long term desired
240 // behavior, but rather that no accidental regression is introduced in the
241 // current behavior.
242
243 // -- Quirks
244 // TODO: find a way to test the failures. These four fail (on purpose).
245 // GameParameterFromString("+");
246 // GameParameterFromString("---");
247 // GameParameterFromString(".");
248 // GameParameterFromString("...");
249 SPIEL_CHECK_TRUE(GameParameterFromString("1.2e-1").has_string_value());
250
251 // -- Whitespace related
252 SPIEL_CHECK_TRUE(GameParameterFromString(" 1").has_string_value());
253 SPIEL_CHECK_TRUE(GameParameterFromString("1 ").has_string_value());
254
255 // -- Intended behavior
256 SPIEL_CHECK_TRUE(GameParameterFromString("true").has_bool_value());
257 SPIEL_CHECK_TRUE(GameParameterFromString("True").has_bool_value());
258 SPIEL_CHECK_TRUE(GameParameterFromString("false").has_bool_value());
259 SPIEL_CHECK_TRUE(GameParameterFromString("False").has_bool_value());
260 SPIEL_CHECK_TRUE(GameParameterFromString("1").has_int_value());
261 SPIEL_CHECK_TRUE(GameParameterFromString("1.0").has_double_value());
262 SPIEL_CHECK_TRUE(GameParameterFromString("1. 0").has_string_value());
263
264 // Tests for GameParametersFromString
265 // Empty string
266 auto params = GameParametersFromString("");
267 SPIEL_CHECK_TRUE(params.empty());
268
269 // Bare name
270 params = GameParametersFromString("game_one");
271 SPIEL_CHECK_EQ(params.size(), 1);
272 SPIEL_CHECK_EQ(params["name"].string_value(), "game_one");
273
274 // Name with empty list
275 params = GameParametersFromString("game_two()");
276 SPIEL_CHECK_EQ(params.size(), 1);
277 SPIEL_CHECK_EQ(params["name"].string_value(), "game_two");
278
279 // Single string parameter
280 params = GameParametersFromString("game_three(foo=bar)");
281 SPIEL_CHECK_EQ(params.size(), 2);
282 SPIEL_CHECK_EQ(params["name"].string_value(), "game_three");
283 SPIEL_CHECK_EQ(params["foo"].string_value(), "bar");
284
285 // Every type of parameter
286 params = GameParametersFromString(
287 "game_four(str=strval,int=42,float=-1.2,game1=nested(),"
288 "game2=nested2(param=val),bool1=True,bool2=False)");
289 SPIEL_CHECK_EQ(params.size(), 8);
290 SPIEL_CHECK_EQ(params["name"].string_value(), "game_four");
291 SPIEL_CHECK_EQ(params["str"].string_value(), "strval");
292 SPIEL_CHECK_EQ(params["int"].int_value(), 42);
293 SPIEL_CHECK_EQ(params["float"].double_value(), -1.2);
294 SPIEL_CHECK_EQ(params["bool1"].bool_value(), true);
295 SPIEL_CHECK_EQ(params["bool2"].bool_value(), false);
296
297 auto game1 = params["game1"].game_value();
298 SPIEL_CHECK_EQ(game1.size(), 1);
299 SPIEL_CHECK_EQ(game1["name"].string_value(), "nested");
300
301 auto game2 = params["game2"].game_value();
302 SPIEL_CHECK_EQ(game2.size(), 2);
303 SPIEL_CHECK_EQ(game2["name"].string_value(), "nested2");
304 SPIEL_CHECK_EQ(game2["param"].string_value(), "val");
305}

307void PolicySerializationTest() {
308 // Check empty tabular policy
309 auto policy = std::make_unique<TabularPolicy>();
310 std::shared_ptr<Policy> deserialized_policy =
311 DeserializePolicy(policy->Serialize());
312 auto deserialized =
313 std::static_pointer_cast<TabularPolicy>(deserialized_policy);
314 SPIEL_CHECK_EQ(policy->PolicyTable().size(), 0);
315 SPIEL_CHECK_EQ(deserialized->PolicyTable().size(), 0);
316
317 // Check non-empty tabular policy
318 auto game = LoadGame("tic_tac_toe");
319 policy = std::make_unique<TabularPolicy>(*game);
320 deserialized_policy = DeserializePolicy(policy->Serialize(6));
321 deserialized = std::static_pointer_cast<TabularPolicy>(deserialized_policy);
322 SPIEL_CHECK_EQ(policy->PolicyTable().size(),
323 deserialized->PolicyTable().size());
324 for (const auto& [info_state, policy] : policy->PolicyTable()) {
325 for (int i = 0; i < policy.size(); i++) {
326 auto original_val = policy.at(i);
327 auto deserialized_val = deserialized->PolicyTable().at(info_state).at(i);
328 SPIEL_CHECK_EQ(original_val.first, deserialized_val.first);
329 SPIEL_CHECK_FLOAT_NEAR(original_val.second, deserialized_val.second,
330 1e-6);
331 }
332 }
333
334 // Check uniform policy
335 DeserializePolicy(std::make_unique<UniformPolicy>()->Serialize());
336}
