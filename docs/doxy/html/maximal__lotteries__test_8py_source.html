33 def test_maximal_lotteries_basic_run(self):
34 # "a" is a dominant strategy of the margin game, so it should be chosen with
35 # probablity 1.
36 votes = [["a", "b", "c"], ["a", "c", "b"], ["b", "a", "c"]]
37 profile = base.PreferenceProfile(votes=votes)
38 method = maximal_lotteries.MaximalLotteriesVoting()
39 outcome = method.run_election(profile)
40 with self.subTest("Top-rank the condorcet winner"):
41 self.assertEqual(outcome.ranking[0], "a")
42 with self.subTest("Check extreme scores"):
43 self.assertAlmostEqual(outcome.scores[0], 1.0)
44 self.assertAlmostEqual(outcome.scores[1], 0.0)
45 self.assertAlmostEqual(outcome.scores[2], 0.0)

47 def test_maximal_lotteries_basic_iterative(self):
48 votes = [["a", "b", "c"], ["a", "c", "b"], ["b", "a", "c"]]
49 profile = base.PreferenceProfile(votes=votes)
50 # "a" is a dominant strategy, so in the iterative version it should be
51 # chosen first, leading to a new matrix with the first row and column
52 # deleted. This then means that "b" is dominant in the subgame.
53 expected_margin_matrix = np.array([
54 [0, 1, 3],
55 [-1, 0, 1],
56 [-3, -1, 0]])
57 with self.subTest("Check margin matrix"):
58 self.assertTrue(np.array_equal(profile.margin_matrix(),
59 expected_margin_matrix))
60 method = maximal_lotteries.MaximalLotteriesVoting(iterative=True)
61 outcome = method.run_election(profile)
62 with self.subTest("Check ranking"):
63 self.assertListEqual(outcome.ranking, ["a", "b", "c"])
64 with self.subTest("Check scores"):
65 self.assertAlmostEqual(outcome.scores[0], 3.0)
66 self.assertAlmostEqual(outcome.scores[1], 2.0)
67 self.assertAlmostEqual(outcome.scores[2], 1.0)

69 def test_maximal_lotteries_cycle(self):
70 # Cyclical profile leads to a Rock, Paper, Scissors margin game.
71 votes = [["a", "b", "c"], ["b", "c", "a"], ["c", "a", "b"]]
72 profile = base.PreferenceProfile(votes=votes)
73 method = maximal_lotteries.MaximalLotteriesVoting()
74 outcome = method.run_election(profile)
75 with self.subTest("Check prob 1/3"):
76 self.assertAlmostEqual(outcome.scores[0], 1.0 / 3.0)
77 with self.subTest("Check uniform"):
78 self.assertAlmostEqual(outcome.scores[0], outcome.scores[1])
79 self.assertAlmostEqual(outcome.scores[1], outcome.scores[2])
