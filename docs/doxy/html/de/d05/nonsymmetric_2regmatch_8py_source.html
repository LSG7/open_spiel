40 def init_vars(self, num_strats, num_players):
41 """Initialize solver parameters."""
42 self.num_players = num_players
43 if len(num_strats) != num_players:
44 raise ValueError('Must specify num strategies for each player')
45 init_dist = []
46 for num_strats_i in num_strats:
47 if self.rnd_init:
48 init_dist_i = self.random.rand(num_strats_i)
49 else:
50 init_dist_i = np.ones(num_strats_i)
51 init_dist_i /= init_dist_i.sum()
52 init_dist.append(init_dist_i)
53 init_regret = [np.zeros_like(dist_i) for dist_i in init_dist]
54 return (init_dist, init_regret)

56 def record_aux_errors(self, grads):
57 """Record errors for the auxiliary variables."""
58 grad_regret = grads[1]
59 grad_regret_flat = np.concatenate(grad_regret)
60 self.aux_errors.append([np.linalg.norm(grad_regret_flat)])

92 def update(self, params, grads, t):
93 """Update cumulative regret and strategy (dist).
94
95 Args:
96 params: tuple of variables to be updated (dist, regret)
97 grads: tuple of variable gradients (grad_dist, grad_regret)
98 t: int, solver iteration (not used)
99 Returns:
100 new_params: tuple of update params (new_dist, new_regret)
101 """
102 dist, regret = params
103 regret_delta = grads[1]
104 if self.discount:
105 gamma = t / float(t + 1)
106 else:
107 gamma = 1
108
109 new_dist = []
110 new_regret = []
111 for dist_i, regret_i, regret_delta_i in zip(dist, regret, regret_delta):
112 new_regret_i = gamma * regret_i + regret_delta_i
113 new_clipped_regrets_i = np.clip(
114 new_regret_i + self.optimism * regret_delta_i, 0., np.inf)
115 if np.sum(new_clipped_regrets_i) > 0:
116 new_dist_i = new_clipped_regrets_i / new_clipped_regrets_i.sum()
117 else:
118 new_dist_i = np.ones_like(dist_i) / dist_i.size
119 new_dist.append(new_dist_i)
120 new_regret.append(new_regret_i)
121
122 new_params = (new_dist, new_regret)
123 return new_params

126def gradients(dist, regret, payoff_matrices, num_players):
127 """Computes regret delta to be added to regret in update.
128
129 Args:
130 dist: list of 1-d np.arrays, current estimate of nash distribution
131 regret: list of 1-d np.arrays (same as dist), current estimate of regrets
132 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
133 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
134 are sorted and arrays should be indexed in the same order
135 num_players: int, number of players, in case payoff_matrices is abbreviated
136 Returns:
137 deltas w.r.t. (dist, regret) as tuple
138 unregularized exploitability (stochastic estimate)
139 solver exploitability (stochastic estimate) - NaN
140 """
141 del regret
142
143 # first compute best responses and payoff gradients
144 grad_dist = []
145 grad_regret = []
146 unreg_exp = []
147 for i in range(num_players):
148
149 nabla_i = np.zeros_like(dist[i])
150 # TODO(imgemp): decide if averaging over nablas provides best comparison
151 for j in range(num_players):
152 if j == i:
153 continue
154 if i < j:
155 hess_i_ij = payoff_matrices[(i, j)][0]
156 else:
157 hess_i_ij = payoff_matrices[(j, i)][1].T
158
159 nabla_ij = hess_i_ij.dot(dist[j])
160 nabla_i += nabla_ij / float(num_players - 1)
161
162 grad_dist_i = np.NaN * np.ones_like(nabla_i)
163 grad_dist.append(grad_dist_i)
164
165 utility_i = nabla_i.dot(dist[i])
166 grad_regret_i = nabla_i - utility_i
167 grad_regret.append(grad_regret_i)
168
169 unreg_exp.append(np.max(nabla_i) - nabla_i.dot(dist[i]))
170
171 return (grad_dist, grad_regret), np.mean(unreg_exp), np.NaN
