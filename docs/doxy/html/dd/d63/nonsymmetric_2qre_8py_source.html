29 def __init__(self, temperature=0., proj_grad=True, euclidean=False,
30 cheap=False, lrs=(1e-2, 1e-1), rnd_init=False, seed=None,
31 **kwargs):
32 """Ctor."""
33 del kwargs
34 if temperature < 0.:
35 raise ValueError('temperature must be non-negative')
36 self.num_players = None
37 self.temperature = temperature
38 self.proj_grad = proj_grad
39 self.cheap = cheap
40 self.rnd_init = rnd_init
41 self.lrs = lrs
42 self.has_aux = True
43 self.aux_errors = []
44
45 self.euclidean = euclidean
46 if euclidean:
47 self.update = self.euc_descent_step
48 else:
49 self.update = self.mirror_descent_step
50
51 self.seed = seed
52 self.random = np.random.RandomState(seed)

54 def init_vars(self, num_strats, num_players):
55 """Initialize solver parameters."""
56 self.num_players = num_players
57 if len(num_strats) != num_players:
58 raise ValueError('Must specify num strategies for each player')
59 init_dist = []
60 for num_strats_i in num_strats:
61 if self.rnd_init:
62 init_dist_i = self.random.rand(num_strats_i)
63 else:
64 init_dist_i = np.ones(num_strats_i)
65 init_dist_i /= init_dist_i.sum()
66 init_dist.append(init_dist_i)
67 init_y = [np.zeros_like(dist_i) for dist_i in init_dist]
68 return (init_dist, init_y)

70 def record_aux_errors(self, grads):
71 """Record errors for the auxiliary variables."""
72 grad_y = grads[1]
73 # call ravel in case use y to track entire payoff matrices in future
74 grad_y_flat = np.concatenate([np.ravel(g) for g in grad_y])
75 self.aux_errors.append([np.linalg.norm(grad_y_flat)])

77 def compute_gradients(self, params, payoff_matrices):
78 """Compute and return gradients (and exploitabilities) for all parameters.
79
80 Args:
81 params: tuple of params (dist, y), see ate.gradients
82 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
83 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
84 are sorted and arrays should be indexed in the same order
85 Returns:
86 tuple of gradients (grad_dist, grad_y), see ate.gradients
87 unregularized exploitability (stochastic estimate)
88 tsallis regularized exploitability (stochastic estimate)
89 """
90 if self.cheap:
91 return cheap_gradients(self.random, *params, payoff_matrices,
92 self.num_players, self.temperature, self.proj_grad)
93 else:
94 return gradients(*params, payoff_matrices, self.num_players,
95 self.temperature, self.proj_grad)

97 def exploitability(self, params, payoff_matrices):
98 """Compute and return tsallis entropy regularized exploitability.
99
100 Args:
101 params: tuple of params (dist, y), see ate.gradients
102 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
103 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
104 are sorted and arrays should be indexed in the same order
105 Returns:
106 float, exploitability of current dist
107 """
108 return exp.qre_exploitability(params, payoff_matrices, self.temperature)

110 def euc_descent_step(self, params, grads, t):
111 """Projected gradient descent on exploitability using Euclidean projection.
112
113 Args:
114 params: tuple of variables to be updated (dist, y)
115 grads: tuple of variable gradients (grad_dist, grad_y)
116 t: int, solver iteration (unused)
117 Returns:
118 new_params: tuple of update params (new_dist, new_y)
119 """
120 lr_dist, lr_y = self.lrs
121 new_dist = []
122 for dist_i, dist_grad_i in zip(params[0], grads[0]):
123 new_dist_i = dist_i - lr_dist * dist_grad_i
124 new_dist_i = simplex.euclidean_projection_onto_simplex(new_dist_i)
125 new_dist.append(new_dist_i)
126 lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
127 new_y = []
128 for y_i, y_grad_i in zip(params[1], grads[1]):
129 new_y_i = y_i - lr_y * y_grad_i
130 new_y_i = np.clip(new_y_i, 0., np.inf)
131 new_y.append(new_y_i)
132 return (new_dist, new_y)

134 def mirror_descent_step(self, params, grads, t):
135 """Entropic mirror descent on exploitability.
136
137 Args:
138 params: tuple of variables to be updated (dist, y)
139 grads: tuple of variable gradients (grad_dist, grad_y)
140 t: int, solver iteration (unused)
141 Returns:
142 new_params: tuple of update params (new_dist, new_y)
143 """
144 lr_dist, lr_y = self.lrs
145 new_dist = []
146 for dist_i, dist_grad_i in zip(params[0], grads[0]):
147 new_dist_i = np.log(np.clip(dist_i, 0., np.inf)) - lr_dist * dist_grad_i
148 new_dist_i = special.softmax(new_dist_i)
149 new_dist.append(new_dist_i)
150 lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
151 new_y = []
152 for y_i, y_grad_i in zip(params[1], grads[1]):
153 new_y_i = y_i - lr_y * y_grad_i
154 new_y_i = np.clip(new_y_i, 0., np.inf)
155 new_y.append(new_y_i)
156 return (new_dist, new_y)

159def gradients(dist, y, payoff_matrices, num_players, temperature=0.,
160 proj_grad=True):
161 """Computes exploitablity gradient and aux variable gradients.
162
163 Args:
164 dist: list of 1-d np.arrays, current estimate of nash distribution
165 y: list 1-d np.arrays (same shape as dist), current est. of payoff gradient
166 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
167 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
168 are sorted and arrays should be indexed in the same order
169 num_players: int, number of players, in case payoff_matrices is abbreviated
170 temperature: non-negative float, default 0.
171 proj_grad: bool, if True, projects dist gradient onto simplex
172 Returns:
173 gradient of exploitability w.r.t. (dist, y) as tuple
174 unregularized exploitability (stochastic estimate)
175 shannon regularized exploitability (stochastic estimate)
176 """
177 # first compute policy gradients and player effects (fx)
178 policy_gradient = []
179 other_player_fx = []
180 grad_y = []
181 unreg_exp = []
182 reg_exp = []
183 for i in range(num_players):
184
185 nabla_i = np.zeros_like(dist[i])
186 for j in range(num_players):
187 if j == i:
188 continue
189 if i < j:
190 hess_i_ij = payoff_matrices[(i, j)][0]
191 else:
192 hess_i_ij = payoff_matrices[(j, i)][1].T
193
194 nabla_ij = hess_i_ij.dot(dist[j])
195 nabla_i += nabla_ij / float(num_players - 1)
196
197 grad_y.append(y[i] - nabla_i)
198
199 if temperature > 0:
200 br_i = special.softmax(y[i] / temperature)
201 br_i_policy_gradient = nabla_i - temperature * (np.log(br_i) + 1)
202 else:
203 power = np.inf
204 s_i = np.linalg.norm(y[i], ord=power)
205 br_i = np.zeros_like(dist[i])
206 maxima_i = (y[i] == s_i)
207 br_i[maxima_i] = 1. / maxima_i.sum()
208 br_i_policy_gradient = np.zeros_like(br_i)
209
210 policy_gradient_i = nabla_i
211 if temperature > 0:
212 policy_gradient_i -= temperature * (np.log(dist[i]) + 1)
213 policy_gradient.append(policy_gradient_i)
214
215 unreg_exp.append(np.max(y[i]) - y[i].dot(dist[i]))
216
217 entr_br_i = temperature * special.entr(br_i).sum()
218 entr_dist_i = temperature * special.entr(dist[i]).sum()
219
220 reg_exp.append(y[i].dot(br_i - dist[i]) + entr_br_i - entr_dist_i)
221
222 other_player_fx_i = (br_i - dist[i])
223 if temperature > 0:
224 # much faster to avoid constructing br_i_mat and then computing
225 # br_i_mat.dot(br_policy_gradient) -- instead, expand out and only compute
226 # inner products
227 temp = (br_i_policy_gradient - br_i.dot(br_i_policy_gradient))
228 other_player_fx_i += br_i / temperature * temp
229 other_player_fx.append(other_player_fx_i)
230
231 # then construct exploitability gradient
232 grad_dist = []
233 for i in range(num_players):
234
235 grad_dist_i = -policy_gradient[i]
236 for j in range(num_players):
237 if j == i:
238 continue
239 if i < j:
240 hess_j_ij = payoff_matrices[(i, j)][1]
241 else:
242 hess_j_ij = payoff_matrices[(j, i)][0].T
243
244 grad_dist_i += hess_j_ij.dot(other_player_fx[j])
245
246 if proj_grad:
247 grad_dist_i = simplex.project_grad(grad_dist_i)
248
249 grad_dist.append(grad_dist_i)
250
251 return (grad_dist, grad_y), np.mean(unreg_exp), np.mean(reg_exp)

254def cheap_gradients(random, dist, y, payoff_matrices, num_players,
255 temperature=0., proj_grad=True):
256 """Computes exploitablity gradient and aux variable gradients with samples.
257
258 This implementation takes payoff_matrices as input so technically uses O(d^2)
259 compute but only a single column of payoff_matrices is used to perform the
260 update so can be re-implemented in O(d) if needed.
261
262 Args:
263 random: random number generator, np.random.RandomState(seed)
264 dist: list of 1-d np.arrays, current estimate of nash distribution
265 y: list 1-d np.arrays (same shape as dist), current est. of payoff gradient
266 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
267 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
268 are sorted and arrays should be indexed in the same order
269 num_players: int, number of players, in case payoff_matrices is abbreviated
270 temperature: non-negative float, default 0.
271 proj_grad: bool, if True, projects dist gradient onto simplex
272 Returns:
273 gradient of exploitability w.r.t. (dist, y) as tuple
274 unregularized exploitability (stochastic estimate)
275 shannon regularized exploitability (stochastic estimate)
276 """
277 # first compute policy gradients and player effects (fx)
278 policy_gradient = []
279 other_player_fx = []
280 grad_y = []
281 unreg_exp = []
282 reg_exp = []
283 for i in range(num_players):
284
285 others = list(range(num_players))
286 others.remove(i)
287 j = np.random.choice(others)
288 action_j = random.choice(dist[j].size, p=dist[j])
289 if i < j:
290 hess_i_ij = payoff_matrices[(i, j)][0]
291 else:
292 hess_i_ij = payoff_matrices[(j, i)][1].T
293 nabla_i = hess_i_ij[:, action_j]
294
295 grad_y.append(y[i] - nabla_i)
296
297 if temperature > 0:
298 br_i = special.softmax(y[i] / temperature)
299 br_i_policy_gradient = nabla_i - temperature * (np.log(br_i) + 1)
300 else:
301 power = np.inf
302 s_i = np.linalg.norm(y[i], ord=power)
303 br_i = np.zeros_like(dist[i])
304 maxima_i = (y[i] == s_i)
305 br_i[maxima_i] = 1. / maxima_i.sum()
306 br_i_policy_gradient = np.zeros_like(br_i)
307
308 policy_gradient_i = nabla_i
309 if temperature > 0:
310 policy_gradient_i -= temperature * (np.log(dist[i]) + 1)
311 policy_gradient.append(policy_gradient_i)
312
313 unreg_exp.append(np.max(y[i]) - y[i].dot(dist[i]))
314
315 entr_br_i = temperature * special.entr(br_i).sum()
316 entr_dist_i = temperature * special.entr(dist[i]).sum()
317
318 reg_exp.append(y[i].dot(br_i - dist[i]) + entr_br_i - entr_dist_i)
319
320 other_player_fx_i = (br_i - dist[i])
321 if temperature > 0:
322 # much faster to avoid constructing br_i_mat and then computing
323 # br_i_mat.dot(br_policy_gradient) -- instead, expand out and only compute
324 # inner products
325 temp = (br_i_policy_gradient - br_i.dot(br_i_policy_gradient))
326 other_player_fx_i += br_i / temperature * temp
327 other_player_fx.append(other_player_fx_i)
328
329 # then construct exploitability gradient
330 grad_dist = []
331 for i in range(num_players):
332
333 grad_dist_i = -policy_gradient[i]
334 for j in range(num_players):
335 if j == i:
336 continue
337 if i < j:
338 hess_j_ij = payoff_matrices[(i, j)][1]
339 else:
340 hess_j_ij = payoff_matrices[(j, i)][0].T
341
342 action_u = random.choice(dist[j].size) # uniform, ~importance sampling
343 other_player_fx_j = dist[j].size * other_player_fx[j][action_u]
344 grad_dist_i += hess_j_ij[:, action_u] * other_player_fx_j
345
346 if proj_grad:
347 grad_dist_i = simplex.project_grad(grad_dist_i)
348
349 grad_dist.append(grad_dist_i)
350
351 return (grad_dist, grad_y), np.mean(unreg_exp), np.mean(reg_exp)
