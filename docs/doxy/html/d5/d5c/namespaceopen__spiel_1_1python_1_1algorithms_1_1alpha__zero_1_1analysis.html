55 padding = 2

59 rows = math.ceil(len(strings) / cols)
60 chunks = [strings[i:i + rows] for i in range(0, len(strings), rows)]

62 if sum(col_widths) + 2 * padding * len(col_widths) <= max_width:

64 for r in range(rows):
65 for c in range(cols):
