44class NFSPPolicies(policy.Policy):
45 """Joint policy to be evaluated."""
46
47 def __init__(self, env, nfsp_policies, mode):
48 game = env.game
49 player_ids = [0, 1]
50 super(NFSPPolicies, self).__init__(game, player_ids)
51 self._policies = nfsp_policies
52 self._mode = mode
53 self._obs = {"info_state": [None, None], "legal_actions": [None, None]}
54
55 def action_probabilities(self, state, player_id=None):
56 cur_player = state.current_player()
57 legal_actions = state.legal_actions(cur_player)
58
59 self._obs["current_player"] = cur_player
60 self._obs["info_state"][cur_player] = (
61 state.information_state_tensor(cur_player))
62 self._obs["legal_actions"][cur_player] = legal_actions
63
64 info_state = rl_environment.TimeStep(
65 observations=self._obs, rewards=None, discounts=None, step_type=None)
66
67 with self._policies[cur_player].temp_mode_as(self._mode):
68 p = self._policies[cur_player].step(info_state, is_evaluation=True).probs
69 prob_dict = {action: p[action] for action in legal_actions}
70 return prob_dict

73def main(unused_argv):
74 game = "kuhn_poker"
75 num_players = 2
76
77 env_configs = {"players": num_players}
78 env = rl_environment.Environment(game, **env_configs)
79 info_state_size = env.observation_spec()["info_state"][0]
80 num_actions = env.action_spec()["num_actions"]
81
82 hidden_layers_sizes = [int(l) for l in FLAGS.hidden_layers_sizes]
83 kwargs = {
84 "replay_buffer_capacity": FLAGS.replay_buffer_capacity,
85 "epsilon_decay_duration": FLAGS.num_train_episodes,
86 "epsilon_start": 0.06,
87 "epsilon_end": 0.001,
88 }
89
90 with tf.Session() as sess:
91 # pylint: disable=g-complex-comprehension
92 agents = [
93 nfsp.NFSP(sess, idx, info_state_size, num_actions, hidden_layers_sizes,
94 FLAGS.reservoir_buffer_capacity, FLAGS.anticipatory_param,
95 **kwargs) for idx in range(num_players)
96 ]
97 expl_policies_avg = NFSPPolicies(env, agents, nfsp.MODE.average_policy)
98
99 sess.run(tf.global_variables_initializer())
100 for ep in range(FLAGS.num_train_episodes):
101 if (ep + 1) % FLAGS.eval_every == 0:
102 losses = [agent.loss for agent in agents]
103 logging.info("Losses: %s", losses)
104 expl = exploitability.exploitability(env.game, expl_policies_avg)
105 logging.info("[%s] Exploitability AVG %s", ep + 1, expl)
106 logging.info("_____________________________________________")
107
108 time_step = env.reset()
109 while not time_step.last():
110 player_id = time_step.observations["current_player"]
111 agent_output = agents[player_id].step(time_step)
112 action_list = [agent_output.action]
113 time_step = env.step(action_list)
114
115 # Episode is over, step all agents with final info state.
116 for agent in agents:
117 agent.step(time_step)
118
119
120if __name__ == "__main__":
121 app.run(main)
