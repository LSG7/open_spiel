170 void RemoveExtraAsciiWhitespace(std::string* str) {
171  auto stripped = StripAsciiWhitespace(*str);
172 
173  if (stripped.empty()) {
174  str->clear();
175  return;
176  }
177 
178  auto input_it = stripped.begin();
179  auto input_end = stripped.end();
180  auto output_it = &(*str)[0];
181  bool is_ws = false;
182 
183  for (; input_it < input_end; ++input_it) {
184  if (is_ws) {
185  // Consecutive whitespace? Keep only the last.
186  is_ws = absl::ascii_isspace(static_cast<unsigned char>(*input_it));
187  if (is_ws) --output_it;
188  } else {
189  is_ws = absl::ascii_isspace(static_cast<unsigned char>(*input_it));
190  }
191 
192  *output_it = *input_it;
193  ++output_it;
194  }
195 
196  str->erase(static_cast<size_t>(output_it - &(*str)[0]));
197 }
