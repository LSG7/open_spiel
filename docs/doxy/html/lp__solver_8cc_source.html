30std::pair<opres::MPVariable*, opres::MPObjective*> SetupVariablesAndObjective(
31 opres::MPSolver* solver, std::vector<opres::MPVariable*>* variables,
32 int num_strategy_variables, double min_utility, double max_utility) {
33 // Value and strategy probability variables
34 opres::MPVariable* v = solver->MakeNumVar(min_utility, max_utility, "v");
35 variables->reserve(num_strategy_variables);
36 for (int i = 0; i < num_strategy_variables; ++i) {
37 variables->push_back(solver->MakeNumVar(0.0, 1.0, absl::StrCat("var ", i)));
38 }
39
40 // Strategy probs sum to one
41 opres::MPConstraint* const sum_to_one =
42 solver->MakeRowConstraint(1.0, 1.0, "sum_to_one");
43 for (int i = 0; i < num_strategy_variables; ++i) {
44 sum_to_one->SetCoefficient((*variables)[i], 1.0);
45 }
46
47 opres::MPObjective* objective = solver->MutableObjective();
48 objective->SetCoefficient(v, 1.0);
49 objective->SetMaximization();
50
51 return {v, objective};
52}

54ZeroSumGameSolution SolveZeroSumMatrixGame(
55 const matrix_game::MatrixGame& matrix_game) {
56 SPIEL_CHECK_EQ(matrix_game.GetType().information,
57 GameType::Information::kOneShot);
58 SPIEL_CHECK_EQ(matrix_game.GetType().utility, GameType::Utility::kZeroSum);
59 int num_rows = matrix_game.NumRows();
60 int num_cols = matrix_game.NumCols();
61 double min_utility = matrix_game.MinUtility();
62 double max_utility = matrix_game.MaxUtility();
63
64 // Solving a game for player i (e.g. row player) requires finding a mixed
65 // policy over player i's pure strategies (actions) such that a value of the
66 // mixed strategy against every opponent pure strategy is maximized.
67 //
68 // For more detail, please refer to Sec 4.1 of Shoham & Leyton-Brown, 2009:
69 // Multiagent Systems: Algorithmic, Game-Theoretic, and Logical Foundations
70 // http://www.masfoundations.org/mas.pdf
71 //
72 // For the row player the LP looks like:
73 // max V
74 // st. sigma_a1 \dot col_0 >= V
75 // sigma_a2 \dot col_1 >= V
76 // .
77 // .
78 // sigma_am \cot col_n >= V
79 // for all i, sigma_ai >= 0
80 // sigma \dot 1 = 1
81
82 ZeroSumGameSolution solution{
83 {0, 0},
84 {std::vector<double>(num_rows, 0), std::vector<double>(num_cols, 0)}};
85
86 // First, the row player (player 0).
87 opres::MPSolver p0_solver("solver", opres::MPSolver::GLOP_LINEAR_PROGRAMMING);
88 std::vector<opres::MPVariable*> p0_vars;
89 auto [p0_v, p0_objective] = SetupVariablesAndObjective(
90 &p0_solver, &p0_vars, num_rows, min_utility, max_utility);
91
92 // Utility constriants
93 for (int c = 0; c < num_cols; ++c) {
94 opres::MPConstraint* const constraint = p0_solver.MakeRowConstraint();
95 constraint->SetLB(0.0);
96 constraint->SetCoefficient(p0_v, -1.0);
97 for (int r = 0; r < num_rows; ++r) {
98 constraint->SetCoefficient(p0_vars[r],
99 matrix_game.PlayerUtility(0, r, c));
100 }
101 }
102
103 p0_solver.Solve();
104 solution.values[0] = p0_objective->Value();
105 for (int r = 0; r < num_rows; ++r) {
106 solution.strategies[0][r] = p0_vars[r]->solution_value();
107 }
108
109 // Now, the column player.
110 opres::MPSolver p1_solver("solver", opres::MPSolver::GLOP_LINEAR_PROGRAMMING);
111 std::vector<opres::MPVariable*> p1_vars;
112 auto [p1_v, p1_objective] = SetupVariablesAndObjective(
113 &p1_solver, &p1_vars, num_cols, min_utility, max_utility);
114
115 // Utility constriants
116 for (int r = 0; r < num_rows; ++r) {
117 opres::MPConstraint* const constraint = p1_solver.MakeRowConstraint();
118 constraint->SetLB(0.0);
119 constraint->SetCoefficient(p1_v, -1.0);
120 for (int c = 0; c < num_cols; ++c) {
121 constraint->SetCoefficient(p1_vars[c],
122 matrix_game.PlayerUtility(1, r, c));
123 }
124 }
125
126 p1_solver.Solve();
127 solution.values[1] = p1_objective->Value();
128 for (int c = 0; c < num_cols; ++c) {
129 solution.strategies[1][c] = p1_vars[c]->solution_value();
130 }
131
132 return solution;
133}

135NormalFormCorrelationDevice ComputeCorrelatedEquilibrium(
136 const NormalFormGame& normal_form_game, CorrEqObjType obj_type,
137 double social_welfare_lower_bound) {
138 // Implements an LP solver as explained in Section 4.6 of Shoham and
139 // Leyton-Brown '09: http://masfoundations.org/
140
141 // The NormalFormState inherits from SimultaneousGame, which conveniently
142 // provides a flattened joint action space, which is useful for setting up
143 // the LP.
144 std::unique_ptr<State> initial_state = normal_form_game.NewInitialState();
145 NFGState* nfg_state = static_cast<NFGState*>(initial_state.get());
146 std::vector<Action> flat_joint_actions = nfg_state->LegalActions();
147
148 opres::MPSolver solver("solver", opres::MPSolver::GLOP_LINEAR_PROGRAMMING);
149 std::vector<opres::MPVariable*> variables;
150 variables.reserve(flat_joint_actions.size());
151
152 // Probability and distribution constraints.
153 opres::MPConstraint* const sum_to_one =
154 solver.MakeRowConstraint(1.0, 1.0, "sum_to_one");
155 for (int i = 0; i < flat_joint_actions.size(); ++i) {
156 variables.push_back(solver.MakeNumVar(0.0, 1.0, absl::StrCat("var ", i)));
157 sum_to_one->SetCoefficient(variables[i], 1.0);
158 }
159
160 // Utility constraints.
161 for (Player p = 0; p < normal_form_game.NumPlayers(); ++p) {
162 // This player's legal actions a_i
163 for (Action a_i : nfg_state->LegalActions(p)) {
164 // This player's alternative legal actions a_i'
165 for (Action a_ip : nfg_state->LegalActions(p)) {
166 // Consider only alternatives a_i' != a_i
167 if (a_ip == a_i) {
168 continue;
169 }
170
171 // Now add the constraint:
172 // \sum_{a \in A | a_i \in a} [u_i(a) - u_i(a_i', a_{-i})] p(a) >= 0.
173 opres::MPConstraint* const constraint = solver.MakeRowConstraint();
174 constraint->SetLB(0.0);
175
176 for (int ja_idx = 0; ja_idx < flat_joint_actions.size(); ++ja_idx) {
177 std::vector<Action> joint_action =
178 nfg_state->FlatJointActionToActions(flat_joint_actions[ja_idx]);
179 // Skip this joint action if a_i is not taken for this player.
180 if (joint_action[p] != a_i) {
181 continue;
182 }
183
184 std::vector<Action> alternative_joint_action = joint_action;
185 alternative_joint_action[p] = a_ip;
186
187 double coeff =
188 normal_form_game.GetUtility(p, joint_action) -
189 normal_form_game.GetUtility(p, alternative_joint_action);
190 constraint->SetCoefficient(variables[ja_idx], coeff);
191 }
192 }
193 }
194 }
195
196 opres::MPObjective* objective = solver.MutableObjective();
197 objective->SetMaximization();
198
199 // Objective depends on the type.
200 if (obj_type == CorrEqObjType::kSocialWelfareAtLeast) {
201 // Add constraint expected SW >= k.
202 opres::MPConstraint* constraint = solver.MakeRowConstraint();
203 constraint->SetLB(social_welfare_lower_bound);
204 for (int i = 0; i < variables.size(); ++i) {
205 std::vector<Action> joint_action =
206 nfg_state->FlatJointActionToActions(flat_joint_actions[i]);
207 std::vector<double> utilities =
208 normal_form_game.GetUtilities(joint_action);
209 constraint->SetCoefficient(
210 variables[i],
211 std::accumulate(utilities.begin(), utilities.end(), 0.0));
212 }
213 } else if (obj_type == CorrEqObjType::kSocialWelfareMax) {
214 // Set the objective to the max social welfare
215 for (int i = 0; i < variables.size(); ++i) {
216 std::vector<Action> joint_action =
217 nfg_state->FlatJointActionToActions(flat_joint_actions[i]);
218 std::vector<double> utilities =
219 normal_form_game.GetUtilities(joint_action);
220 objective->SetCoefficient(
221 variables[i],
222 std::accumulate(utilities.begin(), utilities.end(), 0.0));
223 }
224 }
225
226 solver.Solve();
227
228 NormalFormCorrelationDevice mu;
229 mu.reserve(variables.size());
230 for (int i = 0; i < variables.size(); ++i) {
231 mu.push_back({variables[i]->solution_value(), // probability, actions
232 nfg_state->FlatJointActionToActions(flat_joint_actions[i])});
233 }
234
235 return mu;
236}
