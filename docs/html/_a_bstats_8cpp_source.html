302 void ABstats::PrintStats(ofstream& fout)
303 {
304  ABtracker ABsidesSum;
305  ABsidesSum.sum = ABsides[1].sum + ABsides[0].sum;
306  ABsidesSum.sumCum = ABsides[1].sumCum + ABsides[0].sumCum;
307 
308  if (ABsidesSum.sum)
309  {
310  // First table: By side and position.
311 
312  ABstats::PrintHeaderPosition(fout);
313 
314  ABstats::PrintStatsPosition(fout, -1, "Side1", ABsides[1], ABsidesSum);
315  ABstats::PrintStatsPosition(fout, -1, "Side0", ABsides[0], ABsidesSum);
316  fout << "\n";
317 
318  for (int p = 0; p < AB_SIZE; p++)
319  ABstats::PrintStatsPosition(fout, p, name[p], ABplaces[p], ABsidesSum);
320  fout << "\n";
321  }
322 
323  ABstats::PrintHeaderDepth(fout);
324 
325 
326  // Second table: By depth.
327 
328  int c = 0;
329  for (int d = DDS_MAXDEPTH - 1; d >= 0; d--)
330  {
331  if (ABnodesCum.list[d] == 0)
332  continue;
333 
334  c += ABnodesCum.list[d];
335  ABstats::PrintStatsDepth(fout, d, c);
336  }
337 
338  ABstats::PrintAverageDepth(fout, ABsidesSum);
339 
340 
341 #ifdef DDS_AB_DETAILS
342  // Third table: All the detail.
343 
344  ABstats::PrintHeaderDetail(fout);
345 
346  for (int d = DDS_MAXDEPTH - 1; d >= 0; d--)
347  ABstats::PrintStatsDetail(fout, d);
348 
349  ABstats::PrintSumDetail(fout);
350 #endif
351 }
