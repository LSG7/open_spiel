33  def test_stationary_distribution(self):
34  """Tests stationary distribution using payoffs from Han et al., 2013."""
35  r = 1.
36  t = 2.
37  p = 0.
38  s = -1.
39  delta = 4.
40  eps = 0.25
41  payoff_tables = [
42  np.asarray([[r - eps / 2., r - eps, 0, s + delta - eps, r - eps],
43  [r, r, s, s, s], [0, t, p, p, p], [t - delta, t, p, p, p],
44  [r, t, p, p, p]])
45  ]
46 
47  m = 20
48  alpha = 0.1
49  expected_pi = np.asarray(
50  [0.40966787, 0.07959841, 0.20506998, 0.08505983, 0.2206039])
51 
52  # Test payoffs in matrix format
53  _, _, pi_matrix, _, _ = alpharank.compute(
54  payoff_tables, m=m, alpha=alpha, use_local_selection_model=False)
55  np.testing.assert_array_almost_equal(pi_matrix, expected_pi, decimal=4)
56 
57  # Test payoffs in HPT format
58  hpts = [heuristic_payoff_table.from_matrix_game(payoff_tables[0])]
59  _, _, pi_hpts, _, _ = alpharank.compute(
60  hpts, m=m, alpha=alpha, use_local_selection_model=False)
61  np.testing.assert_array_almost_equal(pi_hpts, expected_pi, decimal=4)
