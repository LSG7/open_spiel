50bool ParseCommand(const std::string& line, const Game& game, const State* state,
51 const std::vector<Action>& legal_actions) {
52 if (line == "#h") {
53 std::cout << "History: " << absl::StrJoin(state->History(), ", ")
54 << std::endl;
55 return true;
56 } else if (line == "#l") {
57 PrintLegals(legal_actions, state);
58 return true;
59 } else {
60 return false;

66void ConsolePlayTest(
67 const Game& game, const State* start_state,
68 const std::vector<Action>* start_history,
69 const std::unordered_map<Player, std::unique_ptr<Bot>>* bots) {
70 // Sampled stochastic and simultaneous move games are not yet supported.
71 GameType type = game.GetType();
72 SPIEL_CHECK_NE(type.chance_mode, GameType::ChanceMode::kSampledStochastic);
73 SPIEL_CHECK_NE(type.dynamics, GameType::Dynamics::kSimultaneous);
74
75 std::unique_ptr<State> state;
76 if (start_state != nullptr) {
77 state = start_state->Clone();
78 } else {
79 state = game.NewInitialState();
80 if (start_history != nullptr) {
81 for (Action action : *start_history) {
82 state->ApplyAction(action);
83 }
84 }
85 }
86
87 bool applied_action = true;
88 std::unique_ptr<State> new_state;
89
90 while (true) {
91 if (applied_action) {
92 std::cout << state->ToString() << std::endl << std::endl;
93 }
94 applied_action = false;
95 Player player = state->CurrentPlayer();
96 std::vector<Action> legal_actions = state->LegalActions();
97
98 if (state->IsTerminal()) {
99 std::cout << "Warning! State is terminal. Returns: ";
100 for (Player p = 0; p < game.NumPlayers(); ++p) {
101 std::cout << state->PlayerReturn(p) << " ";
102 }
103 std::cout << std::endl;
104 }
105
106 if (bots != nullptr && bots->at(player) != nullptr) {
107 Action action = bots->at(player)->Step(*state);
108 std::cout << "Bot chose action: " << state->ActionToString(player, action)
109 << std::endl;
110 state->ApplyAction(action);
111 applied_action = true;
112 } else {
113 std::cout << "[Enter move, or press enter for help menu]> ";
114 std::string line = "";
115 std::getline(std::cin, line);
116 absl::StripAsciiWhitespace(&line);
117 if (line.empty()) {
118 PrintHelpMenu();
119 } else if (line == "#b") {
120 Action last_action = state->History().back();
121 new_state = game.NewInitialState();
122 std::vector<Action> history = state->History();
123 for (int i = 0; i < history.size() - 1; ++i) {
124 new_state->ApplyAction(history[i]);
125 }
126 state = std::move(new_state);
127 std::cout << "Popped action: " << last_action << std::endl;
128 applied_action = true;
129 } else if (line == "#q") {
130 return;
131 } else if (ParseCommand(line, game, state.get(), legal_actions)) {
132 // Do nothing, was already handled.
133 } else {
134 Action action;
135 bool valid_integer = absl::SimpleAtoi(line, &action);
136 if (valid_integer) {
137 auto iter = absl::c_find(legal_actions, action);
138 SPIEL_CHECK_TRUE(iter != legal_actions.end());
139 state->ApplyAction(action);
140 applied_action = true;
141 } else {
142 // Search for the move string.
143 for (Action action : legal_actions) {
144 if (line == state->ActionToString(player, action)) {
145 state->ApplyAction(action);
146 applied_action = true;
147 break;
148 }
149 }
150 }
151 }
152 }
153 }
154
155 std::cout << "Terminal state:" << std::endl
156 << std::endl
157 << state->ToString() << std::endl;
158 std::cout << "Returns: ";
159 std::vector<double> returns = state->Returns();
160 for (Player p = 0; p < game.NumPlayers(); ++p) {
161 std::cout << returns[p] << " ";
162 }
163 std::cout << std::endl;
164}
