28void SingleRankHandTest() {
29 std::array<int, kNumRanks> hand1{}, hand2{};
30 hand1[6] = 3;
31 int action_id1 = SingleRankHandToActionId(hand1);
32 SPIEL_CHECK_EQ(FormatSingleHand(SingleRankHand(action_id1)), "999");
33
34 hand2[13] = 1;
35 int action_id2 = SingleRankHandToActionId(hand2);
36 SPIEL_CHECK_EQ(FormatSingleHand(SingleRankHand(action_id2)), "(BWJ)");
37
38 // 558999TJJJJKKK
39 std::array<int, kNumRanks> current_hand = {0, 0, 2, 0, 0, 1, 3, 1, 4, 0, 3};
40
41 std::vector<Action> actions1, actions2, actions3;
42
43 // The only hands that are greater than 999 are JJJ and KKK
44 SearchSingleRankActions(&actions1, current_hand, /*prev_action=*/action_id1);
45 SPIEL_CHECK_EQ(static_cast<int>(actions1.size()), 2);
46
47 // No hands greater than BWJ
48 SearchSingleRankActions(&actions2, current_hand, /*prev_action=*/action_id2);
49 SPIEL_CHECK_EQ(static_cast<int>(actions2.size()), 0);
50
51 // 6 solos + 4 pairs + 3 trios + 1 bomb = 14
52 SearchSingleRankActions(&actions3, current_hand,
53 /*prev_action=*/kInvalidAction);
54 SPIEL_CHECK_EQ(static_cast<int>(actions3.size()), 14);
55}

57void ChainOnlyHandTest() {

59 int action_id1 = ChainOnlyHandToActionId(hand1);
60
61 SPIEL_CHECK_EQ(FormatSingleHand(ChainOnlyHand(action_id1)), "666777888");
62
63 std::array<int, kNumRanks> hand2 = {2, 2, 2, 2, 2, 2, 2, 2, 2};
64
65 int action_id2 = ChainOnlyHandToActionId(hand2);
66 SPIEL_CHECK_EQ(FormatSingleHand(ChainOnlyHand(action_id2)),
67 "33445566778899TTJJ");
68
69 // 5566777888999TTTJJQQKKAA22(BWJ)(CJ)
70 std::array<int, kNumRanks> current_hand = {0, 0, 2, 2, 3, 3, 3, 3,
71 2, 2, 2, 2, 2, 1, 1};
72
73 std::vector<Action> actions1, actions2, actions3;
74 SearchChainOnlyActions(&actions1, current_hand, /*prev_action=*/action_id1);
75
76 // The only hands greater than 666777888 are 777888999 and 888999TTT
77 SPIEL_CHECK_EQ(static_cast<int>(actions1.size()), 2);
78
79 SearchChainOnlyActions(&actions2, current_hand, /*prev_action=*/action_id2);

90void SingleTrioCombHandTest() {
91 std::array<int, kNumRanks> hand1{}, hand2{};
92
93 // 999-(CJ)
94 hand1[6] = 3;
95 hand1[14] = 1;
96 int action_id1 = SingleTrioCombHandToActionId(hand1);
97 SPIEL_CHECK_EQ(FormatSingleHand(SingleTrioCombHand(action_id1)), "999(CJ)");
98
99 // 333-22
100 hand2[12] = 2;
101 hand2[0] = 3;
102
103 int action_id2 = SingleTrioCombHandToActionId(hand2);
104 SPIEL_CHECK_EQ(FormatSingleHand(SingleTrioCombHand(action_id2)), "33322");
105
106 // 666777TTTQQQ222(BWJ)(CJ)
107 std::array<int, kNumRanks> current_hand = {0, 0, 0, 3, 3, 0, 0, 3,
108 0, 3, 0, 0, 3, 1, 1};
109
110 std::vector<Action> actions1, actions2, actions3;
111
112 // The hands that are greater than 333222 uses trios 666, 777, TTT, QQQ, 222
113 // And we just enuemerate all possible pairs
114 SearchSingleTrioCombActions(&actions1, current_hand,
115 /*prev_action=*/action_id1);
116 SPIEL_CHECK_EQ(static_cast<int>(actions1.size()), 18);
117
118 SearchSingleTrioCombActions(&actions2, current_hand,
119 /*prev_action=*/action_id2);
120 SPIEL_CHECK_EQ(static_cast<int>(actions2.size()), 20);
121
122 SearchSingleTrioCombActions(&actions3, current_hand, kInvalidAction);
123 SPIEL_CHECK_EQ(static_cast<int>(actions3.size()), 50);
124}

126void AirplaneCombHandTest() {
127 // 888999TTTJJJQQQ-7772(CJ)
128 std::array<int, kNumRanks> hand1 = {0, 0, 0, 0, 3, 3, 3, 3,
129 3, 3, 0, 0, 1, 0, 1};
130
131 int action_id1 = AirplaneCombHandToActionId(hand1, /*chain_head=*/5,
132 /*kicker_type=*/kSolo);
133 SPIEL_CHECK_EQ(FormatSingleHand(AirplaneCombHand(action_id1)),
134 "777888999TTTJJJQQQ2(CJ)");
135
136 // TTTJJJQQQKKK-33445522
137 std::array<int, kNumRanks> hand2 = {2, 2, 2, 0, 0, 0, 0, 3,
138 3, 3, 3, 0, 2, 0, 0};
139 int action_id2 = AirplaneCombHandToActionId(hand2, /*chain_head=*/7,
140 /*kicker_type=*/kPair);
141 SPIEL_CHECK_EQ(FormatSingleHand(AirplaneCombHand(action_id2)),
142 "334455TTTJJJQQQKKK22");
143
144 // 667899TTTJJJJQQQKKKAAA222(BWJ)(CJ)
145 std::array<int, kNumRanks> current_hand = {0, 0, 0, 2, 1, 1, 2, 3,
146 4, 3, 3, 3, 3, 1, 1};
147 std::vector<Action> actions1, actions2, actions3;
148 SearchAirplaneCombActions(&actions1, current_hand,
149 /*prev_action=*/action_id1);
150 // C(7, 5) - C(5, 3) + 3*(C(6, 3) - C(4, 1)) + C(3, 2) * 5 + 2 + C(6, 2) - 1 =
151 // 90
152 SPIEL_CHECK_EQ(static_cast<int>(actions1.size()), 90);
153
154 // The only hand that is greater than TTTJJJQQQKKK-33445522 is
155 // JJJQQQKKKAAA-6699TT22
156 SearchAirplaneCombActions(&actions2, current_hand,
157 /*prev_action=*/action_id2);
158 SPIEL_CHECK_EQ(static_cast<int>(actions2.size()), 1);
159
160 SearchAirplaneCombActions(&actions3, current_hand,
161 /*prev_action=*/kInvalidAction);
162 SPIEL_CHECK_EQ(static_cast<int>(actions3.size()), 1052);
163}
