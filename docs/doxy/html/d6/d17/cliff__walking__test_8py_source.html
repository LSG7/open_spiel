62 def test_many_runs(self):
63 random.seed(1234)
64 for _ in range(30):
65 height = random.randint(3, 10)
66 width = random.randint(3, 10)
67 env = cliff_walking.Environment(height=height, width=width)
68
69 time_step = env.reset()

71 self.assertIsNone(time_step.rewards)
72
73 action_int = cliff_walking.UP
74 time_step = env.step(action_int)
75 self.assertEqual(time_step.step_type, rl_environment.StepType.MID)
76 self.assertEqual(time_step.rewards, [-1.0])

78 action_int = cliff_walking.RIGHT
