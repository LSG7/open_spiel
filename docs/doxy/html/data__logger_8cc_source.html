34void DataLoggerJsonLines::Write(DataLogger::Record record) {
35 static absl::TimeZone utc = absl::UTCTimeZone();
36 absl::Time now = absl::Now();
37 record.insert({
38 {"time_str", absl::FormatTime("%Y-%m-%d %H:%M:%E6S %z", now, utc)},
39 {"time_abs", absl::ToUnixMicros(now) / 1000000.},
40 {"time_rel", absl::ToDoubleSeconds(now - start_time_)},
41 });
42 fd_.Write(json::ToString(record));
43 fd_.Write("\n");
44 if (flush_) {
45 Flush();
46 }
47}
