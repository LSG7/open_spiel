42 def _rollout_until_timeout(game_name,
43  time_limit,
44  give_up_after,
45  if_simultaneous_convert_to_turn_based=False):
46  """Run rollouts on the specified game until the time limit.
47 
48  Args:
49  game_name: str
50  time_limit: In number of seconds
51  give_up_after: Cuts off trajectories longer than specified
52  if_simultaneous_convert_to_turn_based: if the game is simultaneous and this
53  boolean is true, then the game is loaded as a turn based game.
54 
55  Returns:
56  A dict of collected statistics.
57  """
58  game = pyspiel.load_game(game_name)
59  if game.get_type().dynamics == pyspiel.GameType.Dynamics.MEAN_FIELD:
60  raise NotImplementedError(
61  "Benchmark on mean field games is not available yet.")
62  if (game.get_type().dynamics == pyspiel.GameType.Dynamics.SIMULTANEOUS and
63  if_simultaneous_convert_to_turn_based):
64  game = pyspiel.convert_to_turn_based(game)
65  is_time_out = lambda t: time.time() - t > time_limit
66  num_rollouts = 0
67  num_giveups = 0
68  num_moves = 0
69  start = time.time()
70  while not is_time_out(start):
71  state = game.new_initial_state()
72  while not state.is_terminal():
73  if len(state.history()) > give_up_after:
74  num_giveups += 1
75  break
76  if state.is_simultaneous_node():
77 
78  def random_choice(actions):
79  if actions:
80  return random.choice(actions)
81  return 0
82 
83  actions = [
84  random_choice(state.legal_actions(i))
85  for i in range(state.num_players())
86  ]
87  state.apply_actions(actions)
88  else:
89  action = random.choice(state.legal_actions(state.current_player()))
90  state.apply_action(action)
91  num_moves += 1
92  num_rollouts += 1
93  time_elapsed = time.time() - start
94  return dict(
95  game_name=game_name,
96  ms_per_rollouts=time_elapsed / num_rollouts * 1000,
97  ms_per_moves=time_elapsed / num_moves * 1000,
98  giveups_per_rollout=num_giveups / num_rollouts,
