29 // 8 adjacent directions.
30 //
31 // 405
32 // 1 2
33 // 637
34 //
35 // The order is important because it is used to index 3x3 patterns!
36 //
37 inline constexpr std::array<int, 9> Dir8 = {{
38  kVirtualBoardSize, // new line
39  -1, // new line
40  +1, // new line
41  -static_cast<int>(kVirtualBoardSize),
42  +static_cast<int>(kVirtualBoardSize) - 1,
43  +static_cast<int>(kVirtualBoardSize) + 1,
44  -static_cast<int>(kVirtualBoardSize) - 1,
45  -static_cast<int>(kVirtualBoardSize) + 1,
46  0 // Dummy element.
47 }};

49 // Calls f for all 4 direct neighbours of p.
50 // f should have type void f(VirtualPoint n), but is passed as a template so we
51 // can elide the function call overhead.
52 template <typename F>
53 void Neighbours(VirtualPoint p, const F &f) {
54  f(p + kVirtualBoardSize);
55  f(p + 1);
56  f(p - 1);
57  f(p - kVirtualBoardSize);
58 }

93 std::string MoveAsAscii(VirtualPoint p, GoColor c) {
94  static std::string code = "0123456789abcdefghijklmnopqrstuvwxyz";
95  static int mask = 31;
96  // 1 bit for color, 9 bits for the point.
97  uint16_t value = static_cast<int>(c) | (p << 1);
98  // Encode in 2 characters of 5 bit each.
99  std::string encoded;
100  encoded.push_back(code[(value >> 5) & mask]);
101  encoded.push_back(code[value & mask]);
102  return encoded;
103 }

145 // Internally, the board is *always* 21*21 with a border of guard stones around
146 // all sides of the board. Thus we need to map a coordinate in that space
147 // to a coordinate in the normal board.
148 Action VirtualActionToAction(int virtual_action, int board_size) {
149  if (virtual_action == kVirtualPass) return board_size * board_size;
150  const int virtual_row = static_cast<int>(virtual_action) / kVirtualBoardSize;
151  const int virtual_col = static_cast<int>(virtual_action) % kVirtualBoardSize;
152  return board_size * (virtual_row - 1) + (virtual_col - 1);
153 }

155 int ActionToVirtualAction(Action action, int board_size) {
156  if (action == board_size * board_size) return kVirtualPass;
157  int row = action / board_size;
158  int column = action % board_size;
159  return (row + 1) * kVirtualBoardSize + (column + 1);
160 }

249 VirtualPoint MakePoint(std::string s) {
250  std::transform(s.begin(), s.end(), s.begin(), ::tolower);
251 
252  if (s == "pass") return kVirtualPass;
253  if (s.size() < 2 || s.size() > 3) return kInvalidPoint;
254 
255  int col = s[0] < 'i' ? s[0] - 'a' : s[0] - 'a' - 1;
256  int row = s[1] - '0';
257  if (s.size() == 3) {
258  row *= 10;
259  row += s[2] - '0';
260  }
261  return VirtualPointFrom2DPoint({row - 1, col});
262 }

275 void PhantomGoBoard::Clear() {
276  zobrist_hash_ = 0;
277 
278  for (int i = 0; i < board_size_ * board_size_; i++) {
279  observations_[(uint8_t)GoColor::kBlack][i] = GoColor::kEmpty;
280  observations_[(uint8_t)GoColor::kWhite][i] = GoColor::kEmpty;
281  }
282 
283  stone_count_ = {0, 0};
284 
285  last_move_valid = true;
286  last_move_pass = false;
287  last_move_captured = 0;
288 
289  for (int i = 0; i < board_.size(); ++i) {
290  Vertex &v = board_[i];
291  v.color = GoColor::kGuard;
292  v.chain_head = static_cast<VirtualPoint>(i);
293  v.chain_next = static_cast<VirtualPoint>(i);
294  chains_[i].reset_border();
295  }
296 
297  for (VirtualPoint p : BoardPoints(board_size_)) {
298  board_[p].color = GoColor::kEmpty;
299  chains_[p].reset();
300  }
301 
302  for (VirtualPoint p : BoardPoints(board_size_)) {
303  Neighbours(p, [this, p](VirtualPoint n) {
304  if (IsEmpty(n)) chain(p).add_liberty(n);
305  });
306  }
307 
308  for (int i = 0; i < last_captures_.size(); ++i) {
309  last_captures_[i] = kInvalidPoint;
310  }
311 
312  last_ko_point_ = kInvalidPoint;
313 }

315 bool PhantomGoBoard::PlayMove(VirtualPoint p, GoColor c) {
316  if (p == kVirtualPass) {
317  last_ko_point_ = kInvalidPoint;
318  last_move_captured = 0;
319  last_move_pass = true;
320  last_move_valid = true;
321  return true;
322  } else {
323  last_move_pass = false;
324  }
325 
326  observations_[(uint8_t)c][VirtualPointToBoardPoint(p, board_size_)] =
327  board_[p].color;
328 
329  // playing illegal moves will occur during phantom go, it is even desired
330  if (!IsLegalMoveObserver(p, c)) {
331  last_move_captured = 0;
332  last_move_valid = false; // was a observational move
333  return false;
334  }
335 
336  last_move_valid = true;
337 
338  stone_count_[(uint8_t)c]++;
339 
340  // Preparation for ko checking.
341  bool played_in_enemy_eye = true;
342  Neighbours(p, [this, c, &played_in_enemy_eye](VirtualPoint n) {
343  GoColor s = PointColor(n);
344  if (s == c || s == GoColor::kEmpty) {
345  played_in_enemy_eye = false;
346  }
347  });
348 
349  JoinChainsAround(p, c);
350  SetStone(p, c);
351  RemoveLibertyFromNeighbouringChains(p);
352  int stones_captured = CaptureDeadChains(p, c);
353 
354  stone_count_[(uint8_t)OppColor(c)] -= stones_captured;
355  last_move_captured = stones_captured;
356 
357  observations_[(uint8_t)c][VirtualPointToBoardPoint(p, board_size_)] = c;
358 
359  if (played_in_enemy_eye && stones_captured == 1) {
360  last_ko_point_ = last_captures_[0];
361  } else {
362  last_ko_point_ = kInvalidPoint;
363  }
364 
365  if (stones_captured != 0) {
366  for (int point = 0; point < board_size_ * board_size_; point++) {
367  VirtualPoint vpoint = VirtualPointFromBoardPoint(point, board_size_);
368 
369  if (observations_[(uint8_t)OppColor(c)][point] == OppColor(c) &&
370  board_[vpoint].color == GoColor::kEmpty) {
371  observations_[(uint8_t)GoColor::kBlack][point] = GoColor::kEmpty;
372  observations_[(uint8_t)GoColor::kWhite][point] = GoColor::kEmpty;
373  }
374  }
375  }
376 
377  SPIEL_CHECK_GT(chain(p).num_pseudo_liberties, 0);
378 
379  return true;
380 }

382 VirtualPoint PhantomGoBoard::SingleLiberty(VirtualPoint p) const {
383  VirtualPoint head = ChainHead(p);
384  VirtualPoint liberty = chain(p).single_liberty();
385 
386  // Check it is really a liberty.
387  SPIEL_CHECK_TRUE(IsInBoardArea(liberty));
388  SPIEL_CHECK_TRUE(IsEmpty(liberty));
389 
390  // Make sure the liberty actually borders the group.
391  for (auto n = Neighbours4(liberty); n; ++n) {
392  if (ChainHead(*n) == head) return liberty;
393  }
394 
395  SpielFatalError(
396  absl::StrCat("liberty", liberty, " does not actually border group ", p));
397 }

458 // Combines the groups around the newly placed stone at vertex. If no groups
459 // are available for joining, the new stone is placed as a new group.
460 void PhantomGoBoard::JoinChainsAround(VirtualPoint p, GoColor c) {
461  VirtualPoint largest_chain_head = kInvalidPoint;
462  int largest_chain_size = 0;
463  Neighbours(
464  p, [this, c, &largest_chain_head, &largest_chain_size](VirtualPoint n) {
465  if (PointColor(n) == c) {
466  Chain &c = chain(n);
467  if (c.num_stones > largest_chain_size) {
468  largest_chain_size = c.num_stones;
469  largest_chain_head = ChainHead(n);
470  }
471  }
472  });
473  if (largest_chain_size == 0) {
474  InitNewChain(p);
475  return;
476  }
477 
478  Neighbours(p, [this, c, &largest_chain_head](VirtualPoint n) {
479  if (PointColor(n) == c) {
480  VirtualPoint chain_head = ChainHead(n);
481  if (chain_head != largest_chain_head) {
482  chain(largest_chain_head).merge(chain(n));
483 
484  // Set all stones in the smaller string to be part of the larger
485  // chain.
486  VirtualPoint cur = n;
487  do {
488  board_[cur].chain_head = largest_chain_head;
489  cur = board_[cur].chain_next;
490  } while (cur != n);
491 
492  // Connect the 2 linked lists representing the stones in the two
493  // chains.
494  std::swap(board_[largest_chain_head].chain_next, board_[n].chain_next);
495  }
496  }
497  });
498 
499  board_[p].chain_next = board_[largest_chain_head].chain_next;
500  board_[largest_chain_head].chain_next = p;
501  board_[p].chain_head = largest_chain_head;
502  chain(largest_chain_head).num_stones += 1;
503 
504  Neighbours(p, [this, largest_chain_head](VirtualPoint n) {
505  if (IsEmpty(n)) {
506  chain(largest_chain_head).add_liberty(n);
507  }
508  });
509 }

515 int PhantomGoBoard::CaptureDeadChains(VirtualPoint p, GoColor c) {
516  int stones_captured = 0;
517  int capture_index = 0;
518  Neighbours(p, [this, c, &capture_index, &stones_captured](VirtualPoint n) {
519  if (PointColor(n) == OppColor(c) && chain(n).num_pseudo_liberties == 0) {
520  last_captures_[capture_index++] = ChainHead(n);
521  stones_captured += chain(n).num_stones;
522  RemoveChain(n);
523  }
524  });
525 
526  for (; capture_index < last_captures_.size(); ++capture_index) {
527  last_captures_[capture_index] = kInvalidPoint;
528  }
529 
530  return stones_captured;
531 }

533 void PhantomGoBoard::RemoveChain(VirtualPoint p) {
534  VirtualPoint this_chain_head = ChainHead(p);
535  VirtualPoint cur = p;
536  do {
537  VirtualPoint next = board_[cur].chain_next;
538 
539  SetStone(cur, GoColor::kEmpty);
540  InitNewChain(cur);
541 
542  Neighbours(cur, [this, this_chain_head, cur](VirtualPoint n) {
543  if (ChainHead(n) != this_chain_head || IsEmpty(n)) {
544  chain(n).add_liberty(cur);
545  }
546  });
547 
548  cur = next;
549  } while (cur != p);
550 }

573 bool PhantomGoBoard::IsLegalMoveObserver(VirtualPoint p, GoColor c) const {
574  if (p == kVirtualPass) return true;
575  if (!IsInBoardArea(p)) return false;
576  if (!IsEmpty(p) || p == LastKoPoint()) return false;
577  if (chain(p).num_pseudo_liberties > 0) return true;
578 
579  // For all checks below, the newly placed stone is completely surrounded by
580  // enemy and friendly stones.
581 
582  // Allow to play if the placed stones connects to a group that still has at
583  // least one other liberty after connecting.
584  bool has_liberty = false;
585  Neighbours(p, [this, c, &has_liberty](VirtualPoint n) {
586  has_liberty |= (PointColor(n) == c && !chain(n).in_atari());
587  });
588  if (has_liberty) return true;
589 
590  // Allow to play if the placed stone will kill at least one group.
591  bool kills_group = false;
592  Neighbours(p, [this, c, &kills_group](VirtualPoint n) {
593  kills_group |= (PointColor(n) == OppColor(c) && chain(n).in_atari());
594  });
595  if (kills_group) return true;
596 
597  return false;
598 }

646 VirtualPoint PhantomGoBoard::Chain::single_liberty() const {
647  SPIEL_CHECK_TRUE(in_atari());
648  // A point is in Atari if it has only a single liberty, i.e. all pseudo
649  // liberties are for the same point.
650  // This is true exactly when
651  // liberty_vertex_sum**2 == liberty_vertex_sum_squared * num_pseudo_liberties
652  // Since all pseudo liberties are for the same point, this is equivalent to
653  // (taking n = num_pseudo_liberties):
654  // (n * p)**2 = (n * p**2) * n
655  // Thus to obtain p, we simple need to divide out the number of pseudo
656  // liberties.
657  SPIEL_CHECK_EQ(liberty_vertex_sum % num_pseudo_liberties, 0);
658  return static_cast<VirtualPoint>(liberty_vertex_sum / num_pseudo_liberties);

729 // Returns the number of points surrounded entirely by one color.
730 // Aborts early and returns 0 if the area borders both black and white stones.
731 int NumSurroundedPoints(const PhantomGoBoard &board, const VirtualPoint p,
732  std::array<bool, kVirtualBoardPoints> *marked,
733  bool *reached_black, bool *reached_white) {
734  if ((*marked)[p]) return 0;
735  (*marked)[p] = true;
736 
737  int num_points = 1;
738  Neighbours(p, [&board, &num_points, marked, reached_black,
739  reached_white](VirtualPoint n) {
740  switch (board.PointColor(n)) {
741  case GoColor::kBlack:
742  *reached_black = true;
743  break;
744  case GoColor::kWhite:
745  *reached_white = true;
746  break;
747  case GoColor::kEmpty:
748  num_points +=
749  NumSurroundedPoints(board, n, marked, reached_black, reached_white);
750  break;
751  case GoColor::kGuard:
752  // Ignore the border.
753  break;
754  }
755  });
756 
757  return num_points;
758 }

760 float TrompTaylorScore(const PhantomGoBoard &board, float komi, int handicap) {
761  // The delta of how many points on the board black and white have occupied,
762  // from black's point of view, i.e. Black points - White points.
763  int occupied_delta = 0;
764 
765  // We need to keep track of which empty points we've already counted as part
766  // of a larger territory.
767  std::array<bool, kVirtualBoardPoints> marked;
768  marked.fill(false);
769 
770  for (VirtualPoint p : BoardPoints(board.board_size())) {
771  switch (board.PointColor(p)) {
772  case GoColor::kBlack:
773  ++occupied_delta;
774  break;
775  case GoColor::kWhite:
776  --occupied_delta;
777  break;
778  case GoColor::kEmpty: {
779  if (marked[p]) continue;
780  // If some empty points are surrounded entirely by one player, they
781  // count as that player's territory.
782  bool reached_black = false, reached_white = false;
783  int n = NumSurroundedPoints(board, p, &marked, &reached_black,
784  &reached_white);
785  if (reached_black && !reached_white) {
786  occupied_delta += n;
787  } else if (!reached_black && reached_white) {
788  occupied_delta -= n;
789  }
790  break;
791  }
792  case GoColor::kGuard:
793  SpielFatalError("unexpected color");
794  }
795  }
796 
797  float score = occupied_delta - komi;
798  if (handicap >= 2) {
799  score -= handicap;
800  }
801  return score;
802 }

804 PhantomGoBoard CreateBoard(const std::string &initial_stones) {
805  PhantomGoBoard board(9);
806 
807  int row = 0;
808  for (const auto &line : absl::StrSplit(initial_stones, '\n')) {
809  int col = 0;
810  bool stones_started = false;
811  for (const auto &c : line) {
812  if (c == ' ') {
813  if (stones_started) {
814  SpielFatalError(
815  "Whitespace is only allowed at the start of "
816  "the line. To represent empty intersections, "
817  "use +");
818  }
819  continue;
820  } else if (c == 'X') {
821  stones_started = true;
822  SPIEL_CHECK_TRUE(board.PlayMove(VirtualPointFrom2DPoint({row, col}),
823  GoColor::kBlack));
824  } else if (c == 'O') {
825  stones_started = true;
826  SPIEL_CHECK_TRUE(board.PlayMove(VirtualPointFrom2DPoint({row, col}),
827  GoColor::kWhite));
828  }
829  col++;
830  }
831  row++;
832  }
833 
834  return board;
