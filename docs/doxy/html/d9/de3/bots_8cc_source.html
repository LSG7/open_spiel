57 .def("step", &Bot::Step)
58 .def("restart", &Bot::Restart)
59 .def("restart_at", &Bot::RestartAt)
60 .def("provides_force_action", &Bot::ProvidesForceAction)
61 .def("force_action", &Bot::ForceAction)
62 .def("inform_action", &Bot::InformAction)
63 .def("inform_actions", &Bot::InformActions)
64 .def("provides_policy", &Bot::ProvidesPolicy)
65 .def("get_policy", &Bot::GetPolicy)
66 .def("step_with_policy", &Bot::StepWithPolicy)
67 .def("is_clonable", &Bot::IsClonable)
68 .def("clone", &Bot::Clone);

70 m.def(
71 "load_bot",
72 py::overload_cast<const std::string&, const std::shared_ptr<const Game>&,
73 Player>(&open_spiel::LoadBot),
74 py::arg("bot_name"), py::arg("game"), py::arg("player"),
75 "Returns a new bot object for the specified bot name using default "
76 "parameters");
77 m.def(
78 "load_bot",
79 py::overload_cast<const std::string&, const std::shared_ptr<const Game>&,
80 Player, const GameParameters&>(&open_spiel::LoadBot),
81 py::arg("bot_name"), py::arg("game"), py::arg("player"),
82 py::arg("params"),
83 "Returns a new bot object for the specified bot name using given "
84 "parameters");
85 m.def("is_bot_registered", &IsBotRegistered,
86 "Checks if a bot under the given name is registered.");
87 m.def("registered_bots", &RegisteredBots,
88 "Returns a list of registered bot names.");
89 m.def(
90 "bots_that_can_play_game",
91 [](std::shared_ptr<const Game> game, int player) {
92 return BotsThatCanPlayGame(*game, player);
93 },
94 py::arg("game"), py::arg("player"),
95 "Returns a list of bot names that can play specified game for the "
96 "given player.");
97 m.def(
98 "bots_that_can_play_game",
99 [](std::shared_ptr<const Game> game) {
100 return BotsThatCanPlayGame(*game);

102 py::arg("game"),
103 "Returns a list of bot names that can play specified game for any "
104 "player.");

142 py::arg("game"), py::arg("evaluator"), py::arg("uct_c"),
143 py::arg("max_simulations"), py::arg("max_memory_mb"),
144 py::arg("solve"), py::arg("seed"), py::arg("verbose"),
145 py::arg("child_selection_policy") =
146 algorithms::ChildSelectionPolicy::UCT)
147 .def("step", &algorithms::MCTSBot::Step)
148 .def("mcts_search", &algorithms::MCTSBot::MCTSearch);
149

161 py::arg("max_simulations"),
162 py::arg("max_world_samples") = algorithms::kUnlimitedNumWorldSamples,
163 py::arg("final_policy_type") =
164 algorithms::ISMCTSFinalPolicyType::kNormalizedVisitCount,
165 py::arg("use_observation_string") = false,
166 py::arg("allow_inconsistent_action_sets") = false)

168 .def("provides_policy", &algorithms::MCTSBot::ProvidesPolicy)
169 .def("get_policy", &algorithms::ISMCTSBot::GetPolicy)

171 .def("restart", &algorithms::ISMCTSBot::Restart)
172 .def("restart_at", &algorithms::ISMCTSBot::RestartAt);
