104// Game object.
105class ConnectFourGame : public Game {
106 public:
107 explicit ConnectFourGame(const GameParameters& params);
108 int NumDistinctActions() const override { return kCols; }
109 std::unique_ptr<State> NewInitialState() const override {
110 return std::unique_ptr<State>(new ConnectFourState(shared_from_this()));
111 }
112 int NumPlayers() const override { return kNumPlayers; }
113 double MinUtility() const override { return -1; }
114 absl::optional<double> UtilitySum() const override { return 0; }
115 double MaxUtility() const override { return 1; }
116 std::vector<int> ObservationTensorShape() const override {
117 return {kCellStates, kRows, kCols};
118 }
119 int MaxGameLength() const override { return kNumCells; }
120};

122inline std::ostream& operator<<(std::ostream& stream, const CellState& state) {
123 switch (state) {
124 case CellState::kEmpty:
125 return stream << "Empty";
126 case CellState::kNought:
127 return stream << "O";
128 case CellState::kCross:
129 return stream << "X";
130 default:
131 SpielFatalError("Unknown cell state");
132 }
133}
