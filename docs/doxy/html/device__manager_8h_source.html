27// Keeps track of a bunch of VPNet models, intended to be one per device, and
28// gives them out based on usage. When you request a device you specify how much
29// work you're going to give it, which is assumed done once the loan is
30// returned.
31class DeviceManager {
32 public:
33 DeviceManager() {
34 learning_ = false;
35 multiple_devices_ = false;
36 }
37
38 void AddDevice(VPNetModel model) { // Not thread safe.
39 devices.emplace_back(Device{std::move(model)});
40 multiple_devices_ = devices.size() > 1;
41 }
42
43 // Acts as a pointer to the model, but lets the manager know when you're done.
44 class DeviceLoan {
45 public:
46 // DeviceLoan is not public constructible and is move only.
47 DeviceLoan(DeviceLoan&& other) = default;
48 DeviceLoan& operator=(DeviceLoan&& other) = default;
49 DeviceLoan(const DeviceLoan&) = delete;
50 DeviceLoan& operator=(const DeviceLoan&) = delete;
51
52 ~DeviceLoan() { manager_->Return(device_id_, requests_); }
53 VPNetModel* operator->() { return model_; }
54
55 private:
56 DeviceLoan(DeviceManager* manager, VPNetModel* model, int device_id,
57 int requests)
58 : manager_(manager),
59 model_(model),
60 device_id_(device_id),
61 requests_(requests) {}
62 DeviceManager* manager_;
63 VPNetModel* model_;
64 int device_id_;
65 int requests_;
66 friend DeviceManager;
67 };

69 // Gives the device with the fewest outstanding requests.
70 DeviceLoan Get(int requests, int device_id = -1) {
71 absl::MutexLock lock(&m_);
72 if (device_id < 0) {
73 // The starting device changes depending on if we are allowed to
74 // use the first device or not.
75 device_id = 0 + (learning_ && multiple_devices_);
76 for (int i = 1 + (learning_ && multiple_devices_); i < devices.size();
77 ++i) {
78 if (devices[i].requests < devices[device_id].requests) {
79 device_id = i;
80 }
81 }
82 }
83 devices[device_id].requests += requests;
84 return DeviceLoan(this, &devices[device_id].model, device_id, requests);
85 }
86
87 // A member to ensure that when device:0 is learning and there are
88 // multiple devices available, that device:0 does not take on any
89 // inference requests from the actors and evaluators. These inference
90 // requests should be dealt with by the other available devices.
91 void SetLearning(bool value) { learning_ = value; }
92
93 int Count() const { return devices.size(); }
94
95 private:
96 void Return(int device_id, int requests) {
97 absl::MutexLock lock(&m_);
98 devices[device_id].requests -= requests;
99 }
100
101 struct Device {
102 VPNetModel model;
103 int requests = 0;
104 };
105
106 bool learning_;
107 bool multiple_devices_;
108 std::vector<Device> devices;
109 absl::Mutex m_;
110};
