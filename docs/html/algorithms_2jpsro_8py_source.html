147 # Helper Functions - Dominated strategy elimination.
148 def _eliminate_dominated_payoff(
149  payoff, epsilon, action_labels=None, action_repeats=None, weakly=False):
150  """Eliminate epsilon dominated strategies."""
151  num_players = payoff.shape[0]
152  eliminated = True
153  if action_labels is None:
154  action_labels = [np.arange(na, dtype=np.int32) for na in payoff.shape[1:]]
155  if action_repeats is not None:
156  action_repeats = [ar for ar in action_repeats]
157  while eliminated:
158  eliminated = False
159  for p in range(num_players):
160  if epsilon > 0.0:
161  continue
162  num_actions = payoff.shape[1:]
163  if num_actions[p] <= 1:
164  continue
165  for a in range(num_actions[p]):
166  index = [slice(None) for _ in range(num_players)]
167  index[p] = slice(a, a+1)
168  if weakly:
169  diff = payoff[p] <= payoff[p][tuple(index)]
170  else:
171  diff = payoff[p] < payoff[p][tuple(index)]
172  axis = tuple(range(p)) + tuple(range(p+1, num_players))
173  less = np.all(diff, axis=axis)
174  less[a] = False # Action cannot eliminate itself.
175  if np.any(less):
176  nonzero = np.nonzero(less)
177  payoff = np.delete(payoff, nonzero, axis=p+1)
178  action_labels[p] = np.delete(action_labels[p], nonzero)
179  if action_repeats is not None:
180  action_repeats[p] = np.delete(action_repeats[p], nonzero)
181  eliminated = True
182  break
183  return payoff, action_labels, action_repeats

186 def _reconstruct_dist(eliminated_dist, action_labels, num_actions):
187  """Returns reconstructed dist from eliminated_dist and action_labels.
188 
189  Redundant dist elements are given values 0.
190 
191  Args:
192  eliminated_dist: Array of shape [A0E, A1E, ...].

194  num_actions: List of length N and values [A0, A1, ...].
195 
196  Returns:
197  reconstructed_dist: Array of shape [A0, A1, ...].
198  """
199  reconstructed_payoff = np.zeros(num_actions)

201  return reconstructed_payoff

205  """Wrap eliminate dominated."""
206  def wrapper(payoff, per_player_repeats, *args, eliminate_dominated=True,
207  **kwargs):
208  epsilon = getattr(kwargs, "epsilon", 0.0)
209  if not eliminate_dominated:

260 def _cce_constraints(payoff, epsilons, remove_null=True, zero_tolerance=1e-8):
261  """Returns the coarse correlated constraints.
262 
263  Args:
264  payoff: A [NUM_PLAYER, NUM_ACT_0, NUM_ACT_1, ...] shape payoff tensor.
265  epsilons: Per player floats corresponding to the epsilon.
266  remove_null: Remove null rows of the constraint matrix.
267  zero_tolerance: Zero out elements with small value.
268 
269  Returns:
270  a_mat: The gain matrix for deviting to an action or shape [SUM(A), PROD(A)].
271  meta: Dictionary containing meta information.
272  """
273  num_players = payoff.shape[0]
274  num_actions = payoff.shape[1:]
275  num_dists = int(np.prod(num_actions))
276 
277  cor_cons = int(np.sum(num_actions))
278 
279  a_mat = np.zeros([cor_cons] + list(num_actions))
280  p_vec = np.zeros([cor_cons], dtype=np.int32)
281  i_vec = np.zeros([cor_cons], dtype=np.int32)
282  con = 0
283  for p in range(num_players):
284  for a1 in range(num_actions[p]):
285  a1_inds = tuple(_indices(p, a1, num_players))
286  for a0 in range(num_actions[p]):
287  a0_inds = tuple(_indices(p, a0, num_players))
288  a_mat[con][a0_inds] += payoff[p][a1_inds]
289  a_mat[con] -= payoff[p]
290  a_mat[con] -= epsilons[p]
291 
292  p_vec[con] = p
293  i_vec[con] = a0
294 
295  con += 1
296 
297  a_mat = np.reshape(a_mat, [cor_cons, num_dists])
298  a_mat[np.abs(a_mat) < zero_tolerance] = 0.0
299  if remove_null:
300  null_cons = np.any(a_mat != 0.0, axis=-1)
301  redundant_cons = np.max(a_mat, axis=1) >= 0
302  nonzero_mask = null_cons & redundant_cons
303  a_mat = a_mat[nonzero_mask, :].copy()
304  p_vec = p_vec[nonzero_mask].copy()
305  i_vec = i_vec[nonzero_mask].copy()
306 
307  meta = dict(
308  p_vec=p_vec,
309  i_vec=i_vec,
310  epsilons=epsilons,
311  )
312 
313  return a_mat, meta

316 def _ace_constraints(payoff, epsilons, remove_null=True, zero_tolerance=0.0):
317  """Returns sparse alternate ce constraints Ax - epsilon <= 0.
318 
319  Args:
320  payoff: Dense payoff tensor.
321  epsilons: Scalar epsilon approximation.
322  remove_null: Whether to remove null row constraints.
323  zero_tolerance: Smallest absolute value.
324 
325  Returns:
326  a_csr: Sparse gain matrix from switching from one action to another.
327  e_vec: Epsilon vector.
328  meta: Dictionary containing meta information.
329  """
330  num_players = payoff.shape[0]
331  num_actions = payoff.shape[1:]
332  num_dists = int(np.prod(num_actions))
333 
334  num_cons = 0
335  for p in range(num_players):
336  num_cons += num_actions[p] * (num_actions[p] - 1)
337 
338  a_dok = sp.sparse.dok_matrix((num_cons, num_dists))
339  e_vec = np.zeros([num_cons])
340  p_vec = np.zeros([num_cons], dtype=np.int32)
341  i_vec = np.zeros([num_cons, 2], dtype=np.int32)
342 
343  num_null_cons = None
344  num_redundant_cons = None
345  num_removed_cons = None
346 
347  if num_cons > 0:
348  con = 0
349  for p in range(num_players):
350  generator = itertools.permutations(range(num_actions[p]), 2)
351  for a0, a1 in generator:
352  a0_inds = _sparse_indices_generator(p, a0, num_actions)
353  a1_inds = _sparse_indices_generator(p, a1, num_actions)
354 
355  for a0_ind, a1_ind in zip(a0_inds, a1_inds):
356  a0_ind_flat = np.ravel_multi_index(a0_ind, num_actions)
357  val = payoff[p][a1_ind] - payoff[p][a0_ind]
358  if abs(val) > zero_tolerance:
359  a_dok[con, a0_ind_flat] = val
360 
361  e_vec[con] = epsilons[p]
362  p_vec[con] = p
363  i_vec[con] = [a0, a1]
364  con += 1
365 
366  a_csr = a_dok.tocsr()
367  if remove_null:
368  null_cons = np.logical_or(
369  a_csr.max(axis=1).todense() != 0.0,
370  a_csr.min(axis=1).todense() != 0.0)
371  null_cons = np.ravel(null_cons)
372  redundant_cons = np.ravel(a_csr.max(axis=1).todense()) >= e_vec
373  nonzero_mask = null_cons & redundant_cons
374  a_csr = a_csr[nonzero_mask, :]
375  e_vec = e_vec[nonzero_mask].copy()
376  p_vec = p_vec[nonzero_mask].copy()
377  i_vec = i_vec[nonzero_mask].copy()
378  num_null_cons = np.sum(~null_cons)
379  num_redundant_cons = np.sum(~redundant_cons)
380  num_removed_cons = np.sum(~nonzero_mask)
381 
382  else:
383  a_csr = a_dok.tocsr()
384 
385  meta = dict(
386  p_vec=p_vec,
387  i_vec=i_vec,
388  epsilons=epsilons,
389  num_null_cons=num_null_cons,
390  num_redundant_cons=num_redundant_cons,
391  num_removed_cons=num_removed_cons,
392  )
393 
394  return a_csr, e_vec, meta

397 def _get_repeat_factor(action_repeats):
398  """Returns the repeat factors for the game."""
399  num_players = len(action_repeats)
400  out_labels = string.ascii_lowercase[:len(action_repeats)]
401  in_labels = ",".join(out_labels)
402  repeat_factor = np.ravel(np.einsum(
403  "{}->{}".format(in_labels, out_labels), *action_repeats))
404  indiv_repeat_factors = []
405  for player in range(num_players):
406  action_repeats_ = [
407  np.ones_like(ar) if player == p else ar
408  for p, ar in enumerate(action_repeats)]
409  indiv_repeat_factor = np.ravel(np.einsum(
410  "{}->{}".format(in_labels, out_labels), *action_repeats_))
411  indiv_repeat_factors.append(indiv_repeat_factor)
412  return repeat_factor, indiv_repeat_factors

416 def _linear(
417  payoff,
418  a_mat,
419  e_vec,
420  action_repeats=None,
421  solver_kwargs=None,
422  cost=None):
423  """Returns linear solution.
424 
425  This is a linear program.
426 
427  Args:
428  payoff: A [NUM_PLAYER, NUM_ACT_0, NUM_ACT_1, ...] shape payoff tensor.
429  a_mat: Constaint matrix.
430  e_vec: Epsilon vector.
431  action_repeats: List of action repeat counts.
432  solver_kwargs: Solver kwargs.
433  cost: Cost function of same shape as payoff.
434 
435  Returns:
436  An epsilon-correlated equilibrium.
437  """
438  num_players = payoff.shape[0]
439  num_actions = payoff.shape[1:]
440  num_dists = int(np.prod(num_actions))
441 
442  if solver_kwargs is None:
443  solver_kwargs = DEFAULT_ECOS_SOLVER_KWARGS
444 
445  if a_mat.shape[0] > 0:
446  # Variables.
447  x = cp.Variable(num_dists, nonneg=True)
448 
449  # Classifier.
450  epsilon_dists = cp.matmul(a_mat, x) - e_vec
451 
452  # Constraints.
453  dist_eq_con = cp.sum(x) == 1
454  cor_lb_con = epsilon_dists <= 0
455 
456  # Objective.
457  if cost is None:
458  player_totals = [
459  cp.sum(cp.multiply(payoff[p].flat, x)) for p in range(num_players)]
460  reward = cp.sum(player_totals)
461  else:
462  reward = cp.sum(cp.multiply(cost.flat, x))
463  obj = cp.Maximize(reward)
464 
465  prob = cp.Problem(obj, [
466  dist_eq_con,
467  cor_lb_con,
468  ])
469 
470  # Solve.
471  prob.solve(**solver_kwargs)
472  status = prob.status
473 
474  # Distribution.
475  dist = np.reshape(x.value, num_actions)
476 
477  # Other.
478  val = reward.value
479  else:
480  if action_repeats is not None:
481  repeat_factor, _ = _get_repeat_factor(action_repeats)
482  x = repeat_factor / np.sum(repeat_factor)
483  else:
484  x = np.ones([num_dists]) / num_dists
485  val = 0.0 # Fix me.
486  dist = np.reshape(x, num_actions)
487  status = None
488 
489  meta = dict(
490  x=x,
491  a_mat=a_mat,
492  val=val,
493  status=status,
494  payoff=payoff,
495  consistent=True,
496  unique=False,
497  )
498 
499  return dist, meta

502 def _qp_cce(
503  payoff,
504  a_mats,
505  e_vecs,
506  assume_full_support=False,
507  action_repeats=None,
508  solver_kwargs=None,
509  min_epsilon=False):
510  """Returns the correlated equilibrium with maximum Gini impurity.
511 
512  Args:
513  payoff: A [NUM_PLAYER, NUM_ACT_0, NUM_ACT_1, ...] shape payoff tensor.
514  a_mats: A [NUM_CON, PROD(A)] shape gain tensor.
515  e_vecs: Epsilon vector.
516  assume_full_support: Whether to ignore beta values.
517  action_repeats: Vector of action repeats for each player.
518  solver_kwargs: Additional kwargs for solver.
519  min_epsilon: Whether to minimize epsilon.
520 
521  Returns:
522  An epsilon-correlated equilibrium.
523  """
524  num_players = payoff.shape[0]
525  num_actions = payoff.shape[1:]
526  num_dists = int(np.prod(num_actions))
527 
528  if solver_kwargs is None:
529  solver_kwargs = DEFAULT_OSQP_SOLVER_KWARGS
530 
531  epsilon = None
532  nonzero_cons = [a_mat.shape[0] > 0 for a_mat in a_mats if a_mat is not None]
533  if any(nonzero_cons):
534  x = cp.Variable(num_dists, nonneg=(not assume_full_support))
535  if min_epsilon:
536  epsilon = cp.Variable(nonpos=True)
537  e_vecs = [epsilon] * num_players
538 
539  if action_repeats is not None:
540  repeat_factor, _ = _get_repeat_factor(action_repeats)
541  x_repeated = cp.multiply(x, repeat_factor)
542  dist_eq_con = cp.sum(x_repeated) == 1
543  cor_lb_cons = [
544  cp.matmul(a_mat, cp.multiply(x, repeat_factor)) <= e_vec
545  for a_mat, e_vec in
546  zip(a_mats, e_vecs) if a_mat.size > 0]
547  eye = sp.sparse.diags(repeat_factor)
548  else:
549  repeat_factor = 1
550  x_repeated = x
551  dist_eq_con = cp.sum(x_repeated) == 1
552  cor_lb_cons = [
553  cp.matmul(a_mat, x) <= e_vec for a_mat, e_vec in
554  zip(a_mats, e_vecs) if a_mat.size > 0]
555  eye = sp.sparse.eye(num_dists)
556 
557  # This is more memory efficient than using cp.sum_squares.
558  cost = 1 - cp.quad_form(x, eye)
559  if min_epsilon:
560  cost -= cp.multiply(2, epsilon)
561 
562  obj = cp.Maximize(cost)
563  prob = cp.Problem(obj, [dist_eq_con] + cor_lb_cons)
564  cost_value = prob.solve(**solver_kwargs)
565  status = prob.status
566  alphas = [cor_lb_con.dual_value for cor_lb_con in cor_lb_cons]
567  lamb = dist_eq_con.dual_value
568 
569  val = cost.value
570  x = x_repeated.value
571  dist = np.reshape(x, num_actions)
572  else:
573  cost_value = 0.0
574  val = 1 - 1 / num_dists
575  if action_repeats is not None:
576  repeat_factor, _ = _get_repeat_factor(action_repeats)
577  x = repeat_factor / np.sum(repeat_factor)
578  else:
579  x = np.ones([num_dists]) / num_dists
580  dist = np.reshape(x, num_actions)
581  status = None
582  alphas = [np.zeros([])]
583  lamb = None
584 
585  meta = dict(
586  x=x,
587  a_mats=a_mats,
588  status=status,
589  cost=cost_value,
590  val=val,
591  alphas=alphas,
592  lamb=lamb,
593  unique=True,
594  min_epsilon=None if epsilon is None else epsilon.value,
595  )
596  return dist, meta

639  x_repeated = cp.multiply(x, repeat_factor)
640  dist_eq_con = cp.sum(x_repeated) == 1
641  cor_lb_cons = [
642  cp.matmul(a_mat, cp.multiply(x, rf)) <= e_vec for a_mat, e_vec, rf in
643  zip(a_mats, e_vecs, indiv_repeat_factors) if a_mat.size > 0]
644  eye = sp.sparse.diags(repeat_factor)
645  else:
646  repeat_factor = 1
647  x_repeated = x
648  dist_eq_con = cp.sum(x_repeated) == 1
649  cor_lb_cons = [
650  cp.matmul(a_mat, x) <= e_vec for a_mat, e_vec in
651  zip(a_mats, e_vecs) if a_mat.size > 0]
652  eye = sp.sparse.eye(num_dists)
653 
654  # This is more memory efficient than using cp.sum_squares.
655  cost = 1 - cp.quad_form(x, eye)
656  if min_epsilon:
657  cost -= cp.multiply(2, epsilon)
658 
659  obj = cp.Maximize(cost)
660  prob = cp.Problem(obj, [dist_eq_con] + cor_lb_cons)
661  cost_value = prob.solve(**solver_kwargs)
662  status = prob.status
663  alphas = [cor_lb_con.dual_value for cor_lb_con in cor_lb_cons]
664  lamb = dist_eq_con.dual_value
665 
666  val = cost.value
667  x = x_repeated.value
668  dist = np.reshape(x, num_actions)
669  else:
670  cost_value = 0.0
671  val = 1 - 1 / num_dists
672  if action_repeats is not None:
673  repeat_factor, indiv_repeat_factors = _get_repeat_factor(
674  action_repeats)
675  x = repeat_factor / np.sum(repeat_factor)
676  else:
677  x = np.ones([num_dists]) / num_dists
678  dist = np.reshape(x, num_actions)
679  status = None
680  alphas = [np.zeros([])]
681  lamb = None
682 
683  meta = dict(
684  x=x,
685  a_mats=a_mats,
686  status=status,
687  cost=cost_value,
688  val=val,
689  alphas=alphas,
690  lamb=lamb,
691  unique=True,
692  min_epsilon=None if epsilon is None else epsilon.value,
693  )
694  return dist, meta
