28 void TestGameValueAndExploitability(const std::string& game_name,
29  double expected_game_value) {
30  std::shared_ptr<const Game> game = LoadGame(game_name);
31  SequenceFormLpSpecification specification(*game);
32  specification.SpecifyLinearProgram(0);
33  double actual_game_value = specification.Solve();
34  SPIEL_CHECK_FLOAT_NEAR(actual_game_value, expected_game_value,
35  kErrorTolerance);
36 
37  // Compute policy for the opponent.
38  TabularPolicy policy0 = specification.OptimalPolicy(0);
39  specification.SpecifyLinearProgram(1);
40  double opponent_game_value = specification.Solve();
41  SPIEL_CHECK_FLOAT_NEAR(actual_game_value + opponent_game_value, 0.,
42  kErrorTolerance);
43  TabularPolicy policy1 = specification.OptimalPolicy(1);
44 
45  // Test exploitability -- this is implemented only for turn-based games.
46  if (game->GetType().dynamics == GameType::Dynamics::kSimultaneous) return;
47 
48  // Merge the two tables.
49  std::unordered_map<std::string, ActionsAndProbs> profile_table =
50  policy0.PolicyTable();
51  profile_table.insert(policy1.PolicyTable().begin(),
52  policy1.PolicyTable().end());
53  TabularPolicy optimal_profile(profile_table);
54  SPIEL_CHECK_FLOAT_NEAR(Exploitability(*game, optimal_profile), 0.,
55  kErrorTolerance);
56 }
