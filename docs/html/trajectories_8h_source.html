33 // The content of a trajectory. The idea is to represent a batch of trajectories
34 // of shape [B, T]. Each trajectory will be padded to have the same length,
35 // which is equal to the length of the longest episode in the batch.
36 struct BatchedTrajectory {
37  // Initializes a BatchedTrajectory of size [batch_size, T].
38  BatchedTrajectory(int batch_size);
39 
40  // Moves the trajectory fields into the current trajectory.
41  void MoveTrajectory(int index, BatchedTrajectory* trajectory);
42 
43  // Pads fields to make sure that they're all the same shape, i.e. [B, T, N],
44  // where N = the size of each field. If size is -1, i.e. the default, then
45  // we resize to the max trajectory length in the batch.
46  void ResizeFields(int length = -1);
47 
48  int batch_size;
49 
50  // Observations is an optional field that corresponds to the results of
51  // calling State::InformationStateTensor. Only one of observations
52  // and state_indices will be filled out for any given instance of
53  // BatchedTrajectory.
54  std::vector<std::vector<std::vector<float>>> observations;
55 
56  // The indices corresponding to the viewed state.
57  std::vector<std::vector<int>> state_indices;
58 
59  // Stores the result of open_spiel::State::LegalActionMask.
60  std::vector<std::vector<std::vector<int>>> legal_actions;
61  std::vector<std::vector<Action>> actions;
62  std::vector<std::vector<std::vector<double>>> player_policies;
63  std::vector<std::vector<int>> player_ids;
64 
65  // This is a tensor of shape [B, T], where rewards[b][n] is the terminal
66  // reward for episode b for player n.
67  std::vector<std::vector<double>> rewards;
68 
69  // Tensor of shape [B, T]. valid[b][n] is true if actions[b][n] was actually
70  // taken during a rollout, and false if it is just padding.
71  std::vector<std::vector<int>> valid;
72 
73  // This is false everywhere except for the last state of the trajectory.
74  std::vector<std::vector<int>> next_is_terminal;
75  uint64_t max_trajectory_length = 0;
76 };

110 // Stateful version of RecordTrajectory. There are several optimisations that
111 // this allows. Currently, the only optimisation is preventing making multiple
112 // copies of the state_to_index class. When state_to_index.empty() is false,
113 // then we default to setting the full observations field and not setting the
114 // state_indices field.
115 class TrajectoryRecorder {
116  public:
117  TrajectoryRecorder(const Game& game,
118  const std::unordered_map<std::string, int>& state_to_index,
119  int seed)
120  : game_(game.shared_from_this()),
121  state_to_index_(state_to_index),
122  rng_(std::mt19937(seed)) {}
123 
124  BatchedTrajectory RecordBatch(const std::vector<TabularPolicy>& policies,
125  int batch_size, int max_unroll_length) {
126  const bool include_full_observations = state_to_index_.empty();
127  std::unique_ptr<State> root = game_->NewInitialState();
128  return RecordBatchedTrajectory(*game_, policies, *root, state_to_index_,
129  batch_size, include_full_observations, &rng_,
130  max_unroll_length);
131  }
132 
133  private:
134  std::shared_ptr<const Game> game_;
135 
136  // Note: The key here depends on the game, and is implemented by the
137  // StateKey method.
138  std::unordered_map<std::string, int> state_to_index_;
139 
140  std::mt19937 rng_;
141 };
