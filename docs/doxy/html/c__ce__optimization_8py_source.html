23def get_proba_constraints_positivity(nus):
24 A = np.zeros((nus.shape[0], 1 + nus.shape[0]))
25 A[:, 1:] = -np.eye(nus.shape[0])
26 return A, np.zeros(A.shape[0])

29def get_proba_constraint_sum_eq(nus):
30 A = np.ones((1, 1 + nus.shape[0]))

35def compress_internal_weights(nus, regrets, rewards, lbd=0.0):
36 """Computes distribution over `nus` while minimizing internal regret.
37
38 Args:
39 nus: [T, P] array, T the number of different population distributions, P the
40 number of different policies.
41 regrets: [T, P, P] array, regrets[t, i, j] = payoff for switching from
42 policy i to j at time t.
43 rewards: [T, P] array, T the number of different population distributions, P
44 the number of different policies
45 lbd: Sparsity argument.
46
47 Returns:
48 Computed distribution over `nus`.
49 """
50
51 def get_c(nus):
52 return np.concatenate(
53 (np.array([1.0]), -lbd * np.sum(rewards * nus, axis=1))
54 )
55
56 def get_max_constraint(regrets):
57 regrets = np.transpose(np.array(regrets), axes=[0, 2, 1])
58 regrets = regrets.reshape(-1, regrets.shape[-1])
59 A = np.zeros((regrets.shape[0], 1 + regrets.shape[1]))
60 A[:, 1:] = regrets
61 A[:, 0] = -1.0
62
63 b = np.zeros(A.shape[0])
64 return A, b
65
66 def get_a_ub(nus, regrets):
67 Amax, bmax = get_max_constraint(regrets)
68 Apos, bpos = get_proba_constraints_positivity(nus)
69 return np.concatenate((Amax, Apos), axis=0), np.concatenate(
70 (bmax, bpos), axis=0
71 )
72
73 c = get_c(nus)
74
75 A_ub, b_ub = get_a_ub(nus, regrets)
76 A_eq, b_eq = get_proba_constraint_sum_eq(nus)
77
78 res = scipy.optimize.linprog(
79 c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq, options={'tol': 1e-10}
80 )
81 new_weights = res.x
82 return new_weights[1:]

85def compress_external_weights(nus, regrets, rewards, lbd=0.0):
86 """Computes distribution over `nus` while minimizing external regret.
87
88 Args:
89 nus: [T, P] array, T the number of different population distributions, P the
90 number of different policies.
91 regrets: [T, P] array, regrets[t, i] = payoff for switching from current
92 policy to i at time t.
93 rewards: [T, P] array, reward for playing policy P at time T.
94 lbd: Sparsity argument.
95
96 Returns:
97 Computed distribution over `nus`.
98 """
99
100 def get_c(nus):
101 return np.concatenate(
102 (np.array([1.0]), -lbd * np.sum(rewards * nus, axis=1))
103 )

105 def get_max_constraints(nus, regrets, lbd):
106 A = np.zeros((regrets.shape[1], 1 + nus.shape[0]))
107 A[:, 0] = -1.0
108 A[:, 1:] = np.transpose(
109 regrets
110 - np.sum(regrets * nus, axis=1).reshape(-1, 1)
111 - lbd * np.abs(regrets)
112 )
113 return A, np.zeros(A.shape[0])

116 Amax, bmax = get_max_constraints(nus, regrets, lbd)

118 return np.concatenate((Amax, Apos), axis=0), np.concatenate(
119 (bmax, bpos), axis=0
120 )

125 A_eq, b_eq = get_proba_constraint_sum_eq(nus)
