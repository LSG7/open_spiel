56chess::ObservationTable ComputePrivateInfoTable(
57 const chess::ChessBoard& board, chess::Color color,
58 const chess::ObservationTable& public_info_table) {
59 const int board_size = board.BoardSize();
60 chess::ObservationTable observability_table{false};
61 board.GenerateLegalMoves(
62 [&](const chess::Move& move) -> bool {
63 size_t to_index = chess::SquareToIndex(move.to, board_size);
64 if (!public_info_table[to_index]) observability_table[to_index] = true;
65
66 if (move.to == board.EpSquare() &&
67 move.piece.type == chess::PieceType::kPawn) {
68 int8_t reversed_y_direction = color == chess::Color::kWhite ? -1 : 1;
69 chess::Square en_passant_capture =
70 move.to + chess::Offset{0, reversed_y_direction};
71 size_t index = chess::SquareToIndex(en_passant_capture, board_size);
72 if (!public_info_table[index]) observability_table[index] = true;
73 }
74 return true;
75 },
76 color);
77
78 for (int8_t y = 0; y < board_size; ++y) {
79 for (int8_t x = 0; x < board_size; ++x) {
80 chess::Square sq{x, y};
81 const auto& piece = board.at(sq);
82 if (piece.color == color) {
83 size_t index = chess::SquareToIndex(sq, board_size);
84 if (!public_info_table[index]) observability_table[index] = true;
85 }
86 }
87 }
88 return observability_table;
89}

91// Checks whether the defender is under attack from the attacker,
92// for the special case when we already know that attacker is under attack
93// from the defender.
94// I.e. D -> A, but D <-? A (where arrow is the "under attack relation")
95// This is used for computation of the public info table.
96bool IsUnderAttack(const chess::Square defender_sq,
97 const chess::Piece defender_piece,
98 const chess::Square attacker_sq,
99 const chess::Piece attacker_piece) {
100 // Identity: i.e. we only check distinct piece types from now on.
101 if (defender_piece.type == attacker_piece.type) {
102 return true;
103 }
104 // No need to check empty attackers from now on.
105 if (attacker_piece.type == chess::PieceType::kEmpty) {
106 return false;
107 }
108
109 const auto pawn_attack = [&]() {
110 int8_t y_dir = attacker_piece.color == chess::Color::kWhite ? 1 : -1;
111 return defender_sq == attacker_sq + chess::Offset{1, y_dir} ||
112 defender_sq == attacker_sq + chess::Offset{-1, y_dir};
113 };
114 const auto king_attack = [&]() {
115 return abs(attacker_sq.x - defender_sq.x) <= 1 &&
116 abs(attacker_sq.y - defender_sq.y) <= 1;
117 };
118 const auto rook_attack = [&]() {
119 return abs(attacker_sq.x - defender_sq.x) == 0 ||
120 abs(attacker_sq.y - defender_sq.y) == 0;
121 };
122 const auto bishop_attack = [&]() {
123 return abs(attacker_sq.x - defender_sq.x) >= 1 &&
124 abs(attacker_sq.y - defender_sq.y) >= 1;
125 };
126
127 switch (defender_piece.type) {
128 case chess::PieceType::kEmpty:
129 SpielFatalError("Empty squares cannot be already attacking.");
130
131 case chess::PieceType::kKing:
132 switch (attacker_piece.type) {
133 case chess::PieceType::kQueen:
134 return true;
135 case chess::PieceType::kRook:
136 return rook_attack();
137 case chess::PieceType::kBishop:
138 return bishop_attack();
139 case chess::PieceType::kKnight:
140 return false;
141 case chess::PieceType::kPawn:
142 return pawn_attack();
143 default:
144 SpielFatalError("Exhausted match");
145 }
146
147 case chess::PieceType::kQueen:
148 switch (attacker_piece.type) {
149 case chess::PieceType::kKing:
150 return king_attack();
151 case chess::PieceType::kRook:
152 return rook_attack();
153 case chess::PieceType::kBishop:
154 return bishop_attack();
155 case chess::PieceType::kKnight:
156 return false;
157 case chess::PieceType::kPawn:
158 return pawn_attack();
159 default:
160 SpielFatalError("Exhausted match");
161 }
162
163 case chess::PieceType::kRook:
164 switch (attacker_piece.type) {
165 case chess::PieceType::kKing:
166 return king_attack();
167 case chess::PieceType::kQueen:
168 return true;
169 default:
170 return false;
171 }
172
173 case chess::PieceType::kBishop:
174 switch (attacker_piece.type) {
175 case chess::PieceType::kKing:
176 return king_attack();
177 case chess::PieceType::kQueen:
178 return true;
179 case chess::PieceType::kPawn:
180 return pawn_attack();
181 default:
182 return false;
183 }
184
185 case chess::PieceType::kKnight:
186 return false;
187
188 case chess::PieceType::kPawn:
189 return attacker_piece.type == chess::PieceType::kKing ||
190 attacker_piece.type == chess::PieceType::kQueen ||
191 attacker_piece.type == chess::PieceType::kBishop;
192
193 default:
194 // This should not happen, we cover all the possibilities.
195 SpielFatalError("Exhausted pattern match in dark_chess::IsUnderAttack()");
196 }
197}

199// Computes which squares are public information. It does not recognize all of
200// them. Only squares of two opponent pieces of the same type attacking each
201// other.
202chess::ObservationTable ComputePublicInfoTable(const chess::ChessBoard& board) {
203 const int board_size = board.BoardSize();
204 std::array<bool, chess::k2dMaxBoardSize> observability_table{false};
205 board.GenerateLegalMoves(
206 [&](const chess::Move& move) -> bool {
207 const chess::Piece& from_piece = board.at(move.from);
208 const chess::Piece& to_piece = board.at(move.to);
209
210 if (IsUnderAttack(move.from, from_piece, move.to, to_piece)) {
211 size_t from_index = chess::SquareToIndex(move.from, board_size);
212 observability_table[from_index] = true;
213
214 size_t to_index = chess::SquareToIndex(move.to, board_size);
215 observability_table[to_index] = true;
216
217 // Fill the table also between the indices.
218 if (from_piece.type != chess::PieceType::kKnight) {
219 int offset_x = 0;
220 int offset_y = 0;
221
222 int diff_x = move.to.x - move.from.x;
223 if (diff_x > 0)
224 offset_x = 1;
225 else if (diff_x < 0)
226 offset_x = -1;
227
228 int diff_y = move.to.y - move.from.y;
229 if (diff_y > 0)
230 offset_y = 1;
231 else if (diff_y < 0)
232 offset_y = -1;
233 chess::Offset offset_step = {
234 static_cast<int8_t>(offset_x),
235 static_cast<int8_t>(offset_y)
236 };
237
238 for (chess::Square dest = move.from + offset_step; dest != move.to;
239 dest += offset_step) {
240 size_t dest_index = chess::SquareToIndex(dest, board_size);
241 observability_table[dest_index] = true;
242 }
243 }
244 }
245 return true;
246 },
247 chess::Color::kWhite);
248
249 return observability_table;
250}

265 explicit DarkChessObserver(IIGObservationType iig_obs_type)

270 void WriteTensor(const State& observed_state, int player,
271 Allocator* allocator) const override {
272 auto& state = open_spiel::down_cast<const DarkChessState&>(observed_state);
273 auto& game = open_spiel::down_cast<const DarkChessGame&>(*state.GetGame());
274 SPIEL_CHECK_GE(player, 0);
275 SPIEL_CHECK_LT(player, game.NumPlayers());
276
277 if (iig_obs_type_.perfect_recall) {
278 SpielFatalError(
279 "DarkChessObserver: tensor with perfect recall not implemented.");
280 }
281
282 const auto public_info_table = ComputePublicInfoTable(state.Board());
283
284 if (iig_obs_type_.public_info) {
285 WritePublicInfoTensor(state, public_info_table, allocator);
286 }
287 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
288 std::string prefix = "private";
289 WritePrivateInfoTensor(state, public_info_table, player, prefix,
290 allocator);
291 } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
292 for (int i = 0; i < chess::NumPlayers(); ++i) {
293 chess::Color color = chess::PlayerToColor(player);
294 std::string prefix = chess::ColorToString(color);
295 WritePrivateInfoTensor(state, public_info_table, i, prefix, allocator);
296 }
297 }
298 }
299
300 std::string StringFrom(const State& observed_state,
301 int player) const override {
302 auto& state = open_spiel::down_cast<const DarkChessState&>(observed_state);
303 auto& game = open_spiel::down_cast<const DarkChessGame&>(*state.GetGame());
304 SPIEL_CHECK_GE(player, 0);
305 SPIEL_CHECK_LT(player, game.NumPlayers());
306
307 if (iig_obs_type_.perfect_recall) {
308 SpielFatalError(
309 "DarkChessObserver: string with perfect recall is not supported");
310 }
311
312 if (iig_obs_type_.public_info &&
313 iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
314 chess::Color color = chess::PlayerToColor(player);
315 chess::ObservationTable empty_public_info_table{};
316 auto obs_table = ComputePrivateInfoTable(state.Board(), color,
317 empty_public_info_table);
318 return state.Board().ToDarkFEN(obs_table, color);
319 } else {
320 SpielFatalError(
321 "DarkChessObserver: string with imperfect recall is implemented only"
322 " for the (default) observation type.");
323 }
324 }

326 private:
327 void WritePieces(chess::Color color, chess::PieceType piece_type,
328 const chess::ChessBoard& board,
329 const chess::ObservationTable& observability_table,
330 const std::string& prefix, Allocator* allocator) const {
331 const std::string type_string =
332 color == chess::Color::kEmpty
333 ? "empty"
334 : chess::PieceTypeToString(
335 piece_type,
336 /*uppercase=*/color == chess::Color::kWhite);
337 const int board_size = board.BoardSize();
338
339 auto out = allocator->Get(prefix + "_" + type_string + "_pieces",
340 {board_size, board_size});
341 for (int8_t y = 0; y < board_size; ++y) {
342 for (int8_t x = 0; x < board_size; ++x) {
343 const chess::Square square{x, y};
344 const chess::Piece& piece_on_board = board.at(square);
345 const bool write_square =
346 piece_on_board.color == color &&
347 piece_on_board.type == piece_type &&
348 observability_table[chess::SquareToIndex(square, board_size)];
349 out.at(x, y) = write_square ? 1.0f : 0.0f;
350 }
351 }
352 }

354 void WriteUnknownSquares(const chess::ChessBoard& board,
355 chess::ObservationTable& observability_table,
356 const std::string& prefix,
357 Allocator* allocator) const {
358 const int board_size = board.BoardSize();
359 auto out = allocator->Get(prefix + "_unknown_squares",
360 {board.BoardSize(), board.BoardSize()});
361 for (int8_t y = 0; y < board_size; ++y) {
362 for (int8_t x = 0; x < board_size; ++x) {
363 const chess::Square square{x, y};
364 const bool write_square =
365 observability_table[chess::SquareToIndex(square, board_size)];
366 out.at(x, y) = write_square ? 0.0f : 1.0f;
367 }
368 }

386 void WritePrivateInfoTensor(const DarkChessState& state,
387 const chess::ObservationTable& public_info_table,
388 int player, const std::string& prefix,
389 Allocator* allocator) const {
390 chess::Color color = chess::PlayerToColor(player);
391 chess::ObservationTable private_info_table =
392 ComputePrivateInfoTable(state.Board(), color, public_info_table);
393
394 // Piece configuration.
395 for (const chess::PieceType& piece_type : chess::kPieceTypes) {
396 WritePieces(chess::Color::kWhite, piece_type, state.Board(),
397 private_info_table, prefix, allocator);
398 WritePieces(chess::Color::kBlack, piece_type, state.Board(),
399 private_info_table, prefix, allocator);
400 }
401 WritePieces(chess::Color::kEmpty, chess::PieceType::kEmpty, state.Board(),
402 private_info_table, prefix, allocator);
403 WriteUnknownSquares(state.Board(), private_info_table, prefix, allocator);
404
405 // Castling rights.
406 WriteBinary(
407 state.Board().CastlingRight(color, chess::CastlingDirection::kLeft),
408 prefix + "_left_castling", allocator);
409 WriteBinary(
410 state.Board().CastlingRight(color, chess::CastlingDirection::kRight),
411 prefix + "_right_castling", allocator);
412 }
413
414 void WritePublicInfoTensor(const DarkChessState& state,
415 const chess::ObservationTable& public_info_table,
416 Allocator* allocator) const {
417 const auto entry = state.repetitions_.find(state.Board().HashValue());
418 SPIEL_CHECK_FALSE(entry == state.repetitions_.end());
419 int repetitions = entry->second;
420
421 // Piece configuration.
422 std::string prefix = "public";
423 for (const chess::PieceType& piece_type : chess::kPieceTypes) {
424 WritePieces(chess::Color::kWhite, piece_type, state.Board(),
425 public_info_table, prefix, allocator);
426 WritePieces(chess::Color::kBlack, piece_type, state.Board(),
427 public_info_table, prefix, allocator);
428 }
429 WritePieces(chess::Color::kEmpty, chess::PieceType::kEmpty, state.Board(),
430 public_info_table, prefix, allocator);
431
432 // Num repetitions for the current board.
433 WriteScalar(/*val=*/repetitions, /*min=*/1, /*max=*/3, "repetitions",
434 allocator);
435
436 // Side to play.
437 WriteScalar(/*val=*/ColorToPlayer(state.Board().ToPlay()),
438 /*min=*/0, /*max=*/1, "side_to_play", allocator);
439
440 // Irreversible move counter.
441 auto out = allocator->Get("irreversible_move_counter", {1});
442 out.at(0) = state.Board().IrreversibleMoveCounter() / 100.;
443 }

445 IIGObservationType iig_obs_type_;
446};

457void DarkChessState::DoApplyAction(Action action) {
458 chess::Move move = ActionToMove(action, Board());
459 moves_history_.push_back(move);

465void DarkChessState::MaybeGenerateLegalActions() const {
466 if (!cached_legal_actions_) {
467 cached_legal_actions_ = std::vector<Action>();
468 Board().GenerateLegalMoves([this](const chess::Move& move) -> bool {
469 cached_legal_actions_->push_back(MoveToAction(move, BoardSize()));
470 return true;
471 });
472 absl::c_sort(*cached_legal_actions_);
473 }

533absl::optional<std::vector<double>> DarkChessState::MaybeFinalReturns() const {
534 const auto to_play_color = Board().ToPlay();
535 const auto opp_color = chess::OppColor(to_play_color);
536
537 const auto to_play_king =
538 chess::Piece{to_play_color, chess::PieceType::kKing};
539 const auto opp_king = chess::Piece{opp_color, chess::PieceType::kKing};
540
541 if (Board().find(to_play_king) == chess::kInvalidSquare) {
542 std::vector<double> returns(NumPlayers());
543 returns[chess::ColorToPlayer(to_play_color)] = LossUtility();
544 returns[chess::ColorToPlayer(opp_color)] = WinUtility();
545 return returns;
546
547 } else if (Board().find(opp_king) == chess::kInvalidSquare) {
548 std::vector<double> returns(NumPlayers());
549 returns[chess::ColorToPlayer(to_play_color)] = WinUtility();
550 returns[chess::ColorToPlayer(opp_color)] = LossUtility();
551 return returns;
552 }
553
554 if (!Board().HasSufficientMaterial()) {
555 return std::vector<double>{DrawUtility(), DrawUtility()};
556 }
557
558 if (IsRepetitionDraw()) {
559 return std::vector<double>{DrawUtility(), DrawUtility()};
560 }
561 // Compute and cache the legal actions.
562 MaybeGenerateLegalActions();
563 SPIEL_CHECK_TRUE(cached_legal_actions_);
564 const bool have_legal_moves = !cached_legal_actions_->empty();
565
566 // If we don't have legal moves we are stalemated
567 if (!have_legal_moves) {
568 return std::vector<double>{DrawUtility(), DrawUtility()};
569 }
570
571 if (Board().IrreversibleMoveCounter() >= kNumReversibleMovesToDraw) {
572 // This is theoretically a draw that needs to be claimed, but we implement
573 // it as a forced draw for now.
574 return std::vector<double>{DrawUtility(), DrawUtility()};
575 }
576
577 return absl::nullopt;
578}
