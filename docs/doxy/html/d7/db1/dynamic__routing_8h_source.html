15// Implementation of a mean field routing game.
16//
17// The game is derived from https://arxiv.org/abs/2110.11943.
18// This game is also implemented in python, see
19// open_spiel/python/mfg/games/dynamic_routing.py.
20// The list of vehicles decribing the N player of the dynamic routing game is
21// replaced by a list of OriginDestinationDemand. One OriginDestinationDemand
22// corresponds to one population of vehicles (with the same origin, destination
23// and departure time).
24//
25// This game is a variant of the mean field route choice game
26// (https://ieeexplore.ieee.org/abstract/document/8619448) as the vehicle
27// movement depends on the current network congestion. In the mean field route
28// choice game, the number of time steps to reach the destination is constant
29// and does not depend on the network congestion, neither of the vehicle cost
30// function. In the dynamic driving and routing game
31// (https://doi.org/10.1016/j.trc.2021.103189), the vehicle choose its
32// speed to travel on each link in order to minimize its cost function.
33// Therefore the congestion is encoded in the cost function.
34//
35// More context can be found on the docstring of the python_dynamic_routing
36// class.

57// time is reassigned based on the number of players on the new link during the

121 // Returns the list of states for which we need to know the distribution of
122 // players over to update the current representative player state.
123 // The distribution of the vehicle's states is used to determined the number
124 // of cars on the new link location link of the representative vehicle in
125 // order to define their waiting time of the representative vehicle when they
126 // join this link. Therefore, If the representative vehicle does not move at
127 // this time step, then no states are useful. If the representative vehicle
128 // moves at this time step, then only the states corresponding to be on the
129 // new link of the representative vehicle are needed to compute the
130 // representative vehicle new waiting time.
131 // Returns:
132 // An array of the string representation of all OD_DEMANDs.
133 std::vector<std::string> DistributionSupport() override;

135 // Updates the travel time from the distribution.
136 // Using the distribution `distribution` of vehicles over the states in
137 // `DistributionSupport()`, computes the number of cars on the same link as
138 // the representative player if they has moved during the last time step and
139 // store it internally to assign a new waiting time to the player. If the
140 // player has not moved during the last time step, do nothing.
141 // Args:
142 // `distribution`: the probability for a vehicle to be in the states in
143 // distribution_support. The distribution is a list of probabilities.

247// In the implementation of the mean field routing game, the representative
248// vehicle/player is represented as a tuple current location, current waiting
249// time and destination. When the waiting time is negative, the vehicle chooses
250// the successor link it would like to go. When arriving on the link, a
251// waiting time is assigned to the player based on the distribution of players
252// on the link. The vehicle arrival time is equal to the time step when they
253// first reach their destination. See module docstring for more information.
254class MeanFieldRoutingGame : public Game {
255 public:
256 // Constructor of the game.
257 // Args:
258 // `params`: game parameters. It should define max_num_time_step,
259 // time_step_length, network and perform_sanity_checks.
260 explicit MeanFieldRoutingGame(const GameParameters& params);
261
262 // There is only 1 chance node (the initial node).
263 int MaxChanceNodesInHistory() const override { return 1; }
264 // Maximum number of possible actions.
265 // This is equal to the number of links + 1
266 // (corresponding to having no possible action kNoPossibleAction).
267 int NumDistinctActions() const override {
268 return game_info_.num_distinct_actions;
269 }
270 // The number of vehicles.
271 // Should be 1 as this mean field game is a one population game.
272 int NumPlayers() const override {
273 SPIEL_CHECK_EQ(game_info_.num_players, 1);
274 return game_info_.num_players;
275 }
276 // Minimum utility is the opposite of the maximum arrival time.
277 // Set to - max_game_length - 1.
278 double MinUtility() const override {
279 SPIEL_CHECK_EQ(game_info_.min_utility, -1 * game_info_.max_game_length - 1);
280 return game_info_.min_utility;
281 }
282 // Maximum utility is the opposite of the minimum arrival time. Set to 0.
283 double MaxUtility() const override { return game_info_.max_utility; }
284 // Maximum number of time step played. Passed during construction.
285 int MaxGameLength() const override { return game_info_.max_game_length; }
286 // Maximum number of chance actions. Set to the length of
287 // od_demand_, i.e. the number of `OriginDestinationDemand`s.
288 int MaxChanceOutcomes() const override {
289 return game_info_.max_chance_outcomes;
290 }
291 // If true, sanity checks are done during the game, should be set to false to
292 // speed up the game.
293 bool perform_sanity_checks() const { return perform_sanity_checks_; }
294
295 // Creates a new initial state of the MeanFieldRoutingGame.
296 std::unique_ptr<State> NewInitialState() const override {
297 return MeanFieldRoutingGameState::CreateNewInitialState(
298 shared_from_this(), time_step_length_, od_demand_.get(), network_.get(),
299 perform_sanity_checks_);
300 }
301
302 // Returns the tensor shape for observation.
303 std::vector<int> ObservationTensorShape() const override {
304 int num_locations = network_->num_actions();
305 int max_num_time_step = MaxGameLength();
306 return {num_locations * 2 + max_num_time_step + 1 + 1};
307 }
308
309 // Deserialize a formatted string to MeanFieldRoutingGameState.
310 std::unique_ptr<State> DeserializeState(
311 const std::string& str) const override;
312
313 private:
314 std::string network_name_;
315 std::unique_ptr<Network> network_;
316 // A list of the vehicle. Their origin and their destination should be road
317 // sections of the game.
318 std::unique_ptr<std::vector<OriginDestinationDemand>> od_demand_;
319 // If true, sanity checks are done during the game, should be set to false to
320 // speed up the game.
321 bool perform_sanity_checks_;
322 // Is used to convert travel times into number of game time steps.
323 double time_step_length_;
324 GameInfo game_info_;
325};
