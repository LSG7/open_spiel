51 void CorrDevBuilder::AddMixedJointPolicy(const TabularPolicy& policy,
52  double weight) {
53  std::vector<int> action_indices(policy.PolicyTable().size(), 0);
54  bool done = false;
55  double total_prob = 0.0;
56 
57  while (!done) {
58  // Construct the joint policy and add it.
59  TabularPolicy deterministic_policy;
60  double prob = 1.0;
61  int info_state_idx = 0;
62  for (const auto& iter : policy.PolicyTable()) {
63  Action action = iter.second[action_indices[info_state_idx]].first;
64  prob *= GetProb(iter.second, action);
65  if (prob == 0.0) {
66  break;
67  }
68  deterministic_policy.SetStatePolicy(
69  iter.first, ToDeterministicPolicy(iter.second, action));
70  info_state_idx++;
71  }
72 
73  SPIEL_CHECK_PROB(prob);
74  if (prob > 0.0) {
75  AddDeterminsticJointPolicy(deterministic_policy, prob * weight);
76  total_prob += prob;
77  }
78 
79  // Now, try to move to the next joint policy.
80  info_state_idx = 0;
81  done = true;
82  for (const auto& iter : policy.PolicyTable()) {
83  if (++action_indices[info_state_idx] < iter.second.size()) {
84  done = false;
85  break;
86  } else {
87  action_indices[info_state_idx] = 0;
88  }
89  info_state_idx++;
90  }
91  }
92 
93  SPIEL_CHECK_TRUE(Near(total_prob, 1.0, 1e-10));
94 }

96 CorrelationDevice CorrDevBuilder::GetCorrelationDevice() const {
97  SPIEL_CHECK_GT(total_weight_, 0);
98  CorrelationDevice corr_dev;
99  double sum_weight = 0;
100  for (const auto& key_and_policy : policy_map_) {
101  double weight = policy_weights_.at(key_and_policy.first);
102  sum_weight += weight;
103  corr_dev.push_back({weight / total_weight_, key_and_policy.second});
104  }
105  SPIEL_CHECK_TRUE(Near(sum_weight, total_weight_));
106  return corr_dev;
107 }
