37std::vector<absl::string_view> ProcessStringParam(
38 const std::string& string_param_str, int max_size) {
39 // ProcessStringParam takes a parameter string and split it is a sequence of
40 // substring. Example:
41 // "" -> {}
42 // "[0|0;0|1]" -> {"0|0", "0|1"}
43 // "[0.5;0.5]" -> {"0.5", "0.5"}
44 absl::string_view string_param = absl::StripAsciiWhitespace(string_param_str);
45 SPIEL_CHECK_TRUE(absl::ConsumePrefix(&string_param, "["));
46 SPIEL_CHECK_TRUE(absl::ConsumeSuffix(&string_param, "]"));
47
48 std::vector<absl::string_view> split_string_list;
49 if (!string_param.empty()) {
50 split_string_list = absl::StrSplit(string_param, ';');
51 }
52 SPIEL_CHECK_GE(split_string_list.size(), 0);
53 SPIEL_CHECK_LE(split_string_list.size(), max_size * max_size);
54 return split_string_list;
55}

99std::string StateToString(int x, int y, int t, Player player_id,
100 bool is_chance_init) {
101 if (is_chance_init) {
102 return "initial";
103 }
104 if (player_id == 0) {
105 return absl::Substitute("($0, $1, $2)", x, y, t);
106 }
107 if (player_id == kMeanFieldPlayerId) {
108 return absl::Substitute("($0, $1, $2)_a", x, y, t);
109 }
110 if (player_id == kChancePlayerId) {
111 return absl::Substitute("($0, $1, $2)_a_mu", x, y, t);
112 }
113 SpielFatalError(
114 absl::Substitute("Unexpected state (player_id: $0, is_chance_init: $1)",
115 player_id, is_chance_init));
116}

118std::vector<std::pair<int, int>> StringListToPairs(
119 std::vector<absl::string_view> strings) {
120 // Transforms a list of strings and returns a list of pairs
121 // {} -> {}
122 // {"0|0", "0|1"} -> {(0, 0), (0, 1)}
123 std::vector<std::pair<int, int>> pairs;
124 for (int i = 0; i < strings.size(); ++i) {
125 std::vector<absl::string_view> xy = absl::StrSplit(strings[i], '|');
126 int xx;
127 int yy;
128 SPIEL_CHECK_TRUE(absl::SimpleAtoi(xy[0], &xx));
129 SPIEL_CHECK_TRUE(absl::SimpleAtoi(xy[1], &yy));
130 pairs.push_back({xx, yy});
131 }
132 return pairs;
133}
134std::vector<float> StringListToFloats(std::vector<absl::string_view> strings) {
135 // Transforms a list of strings and returns a list of float
136 // {} -> {}
137 // {"0.5","0.5"} -> {0.5, 0.5}
138 std::vector<float> floats;
139 floats.reserve(strings.size());
140 for (int i = 0; i < strings.size(); ++i) {
141 float ff;
142 SPIEL_CHECK_TRUE(absl::SimpleAtof(strings[i], &ff));
143 floats.push_back(ff);
144 }
145 return floats;
146}

150int GetY(int i, int size) { return i / size; }

152int MergeXY(int xx, int yy, int size) {
153 SPIEL_CHECK_GE(xx, 0);
154 SPIEL_CHECK_LE(xx, size - 1);
155 SPIEL_CHECK_GE(yy, 0);
156 SPIEL_CHECK_LE(yy, size - 1);
157 return yy + xx * size;
158}

160bool ComparisonPair(const std::pair<int, int>& a,
161 const std::pair<int, int>& b) {
162 return a.first < b.first;

165std::vector<int> StringListToInts(std::vector<absl::string_view> strings,
166 int size) {
167 // Transforms a list of strings and returns a list of pairs
168 // {} -> {}
169 // {"0|0", "0|1"} -> {(0, 0), (0, 1)}
170 std::vector<int> ints;
171 for (int i = 0; i < strings.size(); ++i) {
172 std::vector<absl::string_view> xy = absl::StrSplit(strings[i], '|');
173 int xx;
174 int yy;
175 SPIEL_CHECK_TRUE(absl::SimpleAtoi(xy[0], &xx));
176 SPIEL_CHECK_TRUE(absl::SimpleAtoi(xy[1], &yy));
177 ints.push_back(MergeXY(xx, yy, size));
178 }
179 return ints;
180}

188CrowdModelling2dState::CrowdModelling2dState(
189 std::shared_ptr<const Game> game, int size, int horizon,
190 bool only_distribution_reward, const std::string& forbidden_states,
191 const std::string& initial_distribution,
192 const std::string& initial_distribution_value,
193 const std::string& positional_reward,
194 const std::string& positional_reward_value, bool with_congestion,
195 double noise_intensity, double crowd_aversion_coef)
196 : State(game),
197 size_(size),
198 horizon_(horizon),
199 only_distribution_reward_(only_distribution_reward),
200 with_congestion_(with_congestion),
201 noise_intensity_(noise_intensity),
202 crowd_aversion_coef_(crowd_aversion_coef),
203 distribution_(size_ * size_, 1. / (size_ * size_)) {
204 std::vector<absl::string_view> initial_distribution_list =
205 ProcessStringParam(initial_distribution, size_);
206 std::vector<absl::string_view> initial_distribution_value_list =
207 ProcessStringParam(initial_distribution_value, size_);
208 SPIEL_CHECK_EQ(initial_distribution_list.size(),
209 initial_distribution_value_list.size());
210
211 auto initial_distribution_pair = StringListToPairs(initial_distribution_list);
212 auto initial_distribution_value_f =
213 StringListToFloats(initial_distribution_value_list);
214
215 int initial_distribution_action_prob_size = initial_distribution_list.size();
216
217 if (initial_distribution_action_prob_size == 0) {
218 for (int i = 0; i < size_ * size_; ++i) {
219 initial_distribution_action_prob_.push_back({i, 1. / (size_ * size_)});
220 }
221 } else {
222 for (int i = 0; i < initial_distribution_action_prob_size; ++i) {
223 int kk = MergeXY(initial_distribution_pair[i].first,
224 initial_distribution_pair[i].second, size_);
225 initial_distribution_action_prob_.push_back(
226 {kk, initial_distribution_value_f[i]});
227 }
228 }
229
230 std::sort(initial_distribution_action_prob_.begin(),
231 initial_distribution_action_prob_.end(), ComparisonPair);
232
233 std::vector<absl::string_view> forbidden_states_list =
234 ProcessStringParam(forbidden_states, size_);
235 forbidden_states_xy_ = StringListToPairs(forbidden_states_list);
236 for (const auto& forbidden_state_xy : forbidden_states_xy_) {
237 SPIEL_CHECK_GE(forbidden_state_xy.first, 0);
238 SPIEL_CHECK_LE(forbidden_state_xy.first, size_ - 1);
239 SPIEL_CHECK_GE(forbidden_state_xy.second, 0);
240 SPIEL_CHECK_LE(forbidden_state_xy.second, size_ - 1);
241 }
242
243 std::vector<absl::string_view> positional_reward_list =
244 ProcessStringParam(positional_reward, size_);
245 std::vector<absl::string_view> positional_reward_value_list =
246 ProcessStringParam(positional_reward_value, size_);
247 positional_reward_xy_ = StringListToPairs(positional_reward_list);
248 positional_reward_value_ = StringListToFloats(positional_reward_value_list);
249 // There should be a reward for each positional reward XY pair.
250 SPIEL_CHECK_EQ(positional_reward_xy_.size(), positional_reward_value_.size());
251 for (const auto& positional_reward_xy : positional_reward_xy_) {
252 SPIEL_CHECK_GE(positional_reward_xy.first, 0);
253 SPIEL_CHECK_LE(positional_reward_xy.first, size_ - 1);
254 SPIEL_CHECK_GE(positional_reward_xy.second, 0);
255 SPIEL_CHECK_LE(positional_reward_xy.second, size_ - 1);
256 }
257
258 if (positional_reward_xy_.empty()) {
259 // Use the center point as the reward position.
260 positional_reward_xy_.push_back({size_ / 2, size_ / 2});
261 positional_reward_value_.push_back(1.0);
262 }
263
264 // Forbid to the initial distribution and the forbidden states to overlap.
265 auto forbidden_states_int = StringListToInts(forbidden_states_list, size_);
266 auto initial_distribution_int =
267 StringListToInts(initial_distribution_list, size_);
268 std::vector<int> intersection;
269 std::sort(forbidden_states_int.begin(), forbidden_states_int.end());
270 std::sort(initial_distribution_int.begin(), initial_distribution_int.end());
271
272 std::set_intersection(
273 forbidden_states_int.begin(), forbidden_states_int.end(),
274 initial_distribution_int.begin(), initial_distribution_int.end(),
275 back_inserter(intersection));
276 SPIEL_CHECK_EQ(intersection.size(), 0);
277}

314std::vector<Action> CrowdModelling2dState::LegalActions() const {
315 if (IsTerminal()) return {};
316 if (IsChanceNode()) return LegalChanceOutcomes();
317 if (IsMeanFieldNode()) return {};
318 SPIEL_CHECK_TRUE(IsPlayerNode());
319 return LegalPlayerActions();
320}
321
322ActionsAndProbs CrowdModelling2dState::ChanceOutcomes() const {
323 if (is_chance_init_) {
324 return initial_distribution_action_prob_;
325 }
326 const std::vector<Action> legal_actions = LegalPlayerActions();
327 ActionsAndProbs outcomes;
328 if (legal_actions.empty()) {
329 return outcomes;
330 }
331 // Neutral action will always be present in the legal actions.
332 const double prob = noise_intensity_ / legal_actions.size();
333 outcomes.reserve(legal_actions.size());
334 for (const Action action : legal_actions) {
335 if (action == kNeutralAction) {
336 outcomes.emplace_back(action, 1.0 - noise_intensity_ + prob);
337 } else {
338 outcomes.emplace_back(action, prob);
339 }
340 }
341 return outcomes;
342}

344bool CrowdModelling2dState::IsForbidden(Action action) const {
345 int xx = (x_ + kActionToMoveX.at(action) + size_) % size_;
346 int yy = (y_ + kActionToMoveY.at(action) + size_) % size_;
347 return IsForbiddenPosition(xx, yy);
348}

350bool CrowdModelling2dState::IsForbiddenPosition(int x, int y) const {
351 for (const auto& forbidden_xy : forbidden_states_xy_) {
352 if (x == forbidden_xy.first && y == forbidden_xy.second) {
353 return true;
354 }
355 }
356 return false;

359void CrowdModelling2dState::DoApplyAction(Action action) {
360 SPIEL_CHECK_NE(current_player_, kMeanFieldPlayerId);
361 return_value_ += Rewards()[0];
362 int xx;
363 int yy;
364 // Compute the next state
365 if (is_chance_init_) {
366 SPIEL_CHECK_GE(action, 0);
367 SPIEL_CHECK_LT(action, (size_ * size_));
368 SPIEL_CHECK_EQ(current_player_, kChancePlayerId);
369 xx = GetX(action, size_);
370 yy = GetY(action, size_);
371 is_chance_init_ = false;
372 current_player_ = 0;
373 } else if (current_player_ == kChancePlayerId) {
374 xx = (x_ + kActionToMoveX.at(action) + size_) % size_;
375 yy = (y_ + kActionToMoveY.at(action) + size_) % size_;
376 ++t_;
377 current_player_ = kMeanFieldPlayerId;
378 } else {
379 SPIEL_CHECK_EQ(current_player_, 0);
380 xx = (x_ + kActionToMoveX.at(action) + size_) % size_;
381 yy = (y_ + kActionToMoveY.at(action) + size_) % size_;
382 last_action_ = action;
383 current_player_ = kChancePlayerId;
384 }
385 // Assign the new (x,y) position if it isn't forbidden.
386 if (!IsForbiddenPosition(xx, yy) || is_chance_init_) {
387 x_ = xx;
388 y_ = yy;
389 }
390}

422std::vector<double> CrowdModelling2dState::Rewards() const {
423 if (current_player_ != 0) {
424 return {0.};
425 }
426 // Distribution-based reward
427 double r_mu = -crowd_aversion_coef_ *
428 std::log(distribution_[MergeXY(x_, y_, size_)] + kEpsilon);
429 if (only_distribution_reward_) {
430 return {r_mu};
431 }
432 // Positional reward
433 double r_x = 1;
434 double r_y = 1;
435 for (int i = 0; i < positional_reward_xy_.size(); ++i) {
436 double val_r = 2.0 * positional_reward_value_[i] / size_;
437 r_x -= val_r * std::abs(x_ - positional_reward_xy_[i].first);
438 r_y -= val_r * std::abs(y_ - positional_reward_xy_[i].second);
439 }
440 double r_a = -1.0 *
441 (std::abs(kActionToMoveX.at(last_action_)) +
442 std::abs(kActionToMoveY.at(last_action_))) /
443 size_;
444 if (with_congestion_) {
445 // Congestion effect: higher penalty when moving in a high-density area
446 r_a *= distribution_[MergeXY(x_, y_, size_)];
447 }
448 return {r_x + r_y + r_a + r_mu};
449}

471void CrowdModelling2dState::ObservationTensor(Player player,
472 absl::Span<float> values) const {
473 SPIEL_CHECK_GE(player, 0);
474 SPIEL_CHECK_LT(player, num_players_);
475 SPIEL_CHECK_EQ(values.size(), 2 * size_ + horizon_ + 1);
476 SPIEL_CHECK_LT(x_, size_);
477 SPIEL_CHECK_LT(y_, size_);
478 SPIEL_CHECK_GE(t_, 0);
479 // Allow t_ == horizon_.
480 SPIEL_CHECK_LE(t_, horizon_);
481 std::fill(values.begin(), values.end(), 0.);
482 if (x_ >= 0 && y_ >= 0) {
483 values[x_] = 1.;
484 values[y_ + size_] = 1.;
485 } else {
486 // x_ and y_ equal -1 for the initial (blank) state, don't set any position
487 // bit in that case.
488 SPIEL_CHECK_EQ(x_, -1);
489 SPIEL_CHECK_EQ(y_, -1);
490 }
491 values[2 * size_ + t_] = 1.;
492}

534std::unique_ptr<State> CrowdModelling2dGame::DeserializeState(
535 const std::string& str) const {
536 std::vector<std::string> lines = absl::StrSplit(str, '\n');
537 if (lines.size() != 2) {
538 SpielFatalError(absl::StrCat("Expected 2 lines in serialized state, got: ",
539 lines.size()));
540 }
541 Player current_player;
542 int is_chance_init, x, y, t, last_action;
543 double return_value;
544 std::vector<double> distribution;
545
546 std::vector<std::string> properties = absl::StrSplit(lines[0], ',');
547 if (properties.size() != 7) {
548 SpielFatalError(
549 absl::StrCat("Expected 7 properties for serialized state, got: ",
550 properties.size()));
551 }
552 SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[0], &current_player));
553 SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[1], &is_chance_init));
554 SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[2], &x));
555 SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[3], &y));
556 SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[4], &t));
557 SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[5], &last_action));
558 SPIEL_CHECK_TRUE(absl::SimpleAtod(properties[6], &return_value));
559 std::vector<std::string> serialized_distrib = absl::StrSplit(lines[1], ',');
560 distribution.reserve(serialized_distrib.size());
561 for (std::string& v : serialized_distrib) {
562 double parsed_weight;
563 SPIEL_CHECK_TRUE(absl::SimpleAtod(v, &parsed_weight));
564 distribution.push_back(parsed_weight);
565 }
566 return absl::make_unique<CrowdModelling2dState>(
567 shared_from_this(), size_, horizon_, only_distribution_reward_,
568 forbidden_states_, initial_distribution_, initial_distribution_value_,
569 positional_reward_, positional_reward_value_, current_player,
570 is_chance_init, x, y, t, last_action, return_value, distribution,
571 with_congestion_, noise_intensity_, crowd_aversion_coef_);
572}
