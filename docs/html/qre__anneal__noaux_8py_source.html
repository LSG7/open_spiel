106  temperature=0., proj_grad=True):

108 
109  Args:
110  dist: 1-d np.array, current estimate of nash distribution
111  anneal_steps: int, elapsed num steps since last anneal
112  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
113  num_players: int, number of players, in case payoff_matrices is
114  abbreviated

116  proj_grad: bool, if True, projects dist gradient onto simplex

119  unregularized exploitability (stochastic estimate)
120  tsallis regularized exploitability (stochastic estimate)

123  if temperature >= 1e-3:
124  br = special.softmax(y / temperature)

126  log_br_safe = np.clip(np.log(br), -1e5, 0)
127  br_policy_gradient = nabla - temperature * (log_br_safe + 1)
128  else:
129  power = np.inf
130  s = np.linalg.norm(y, ord=power)
131  br = np.zeros_like(dist)
132  maxima = (y == s)
133  br[maxima] = 1. / maxima.sum()

135  br_policy_gradient = np.zeros_like(br)
136 
137  unreg_exp = np.max(y) - y.dot(dist)
138  entr_br = temperature * special.entr(br).sum()
139  entr_dist = temperature * special.entr(dist).sum()
140  reg_exp = y.dot(br - dist) + entr_br - entr_dist

143  if temperature > 0:
144  log_dist_safe = np.clip(np.log(dist), -1e5, 0)
145  policy_gradient -= temperature * (log_dist_safe + 1)
146  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)

149  grad_dist = -policy_gradient

164  def cheap_gradients(self, random, dist, anneal_steps, payoff_matrices,
165  num_players, temperature=0., proj_grad=True):
166  """Computes exploitablity gradient and aux variable gradients with samples.
167 
168  This implementation takes payoff_matrices as input so technically uses
169  O(d^2) compute but only a single column of payoff_matrices is used to
170  perform the update so can be re-implemented in O(d) if needed.
171 
172  Args:
173  random: random number generator, np.random.RandomState(seed)
174  dist: 1-d np.array, current estimate of nash distribution
175  anneal_steps: int, elapsed num steps since last anneal
176  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
177  num_players: int, number of players, in case payoff_matrices is
178  abbreviated
179  temperature: non-negative float, default 0.
180  proj_grad: bool, if True, projects dist gradient onto simplex
181  Returns:
182  gradient of exploitability w.r.t. (dist, anneal_steps) as tuple
183  unregularized exploitability (stochastic estimate)
184  tsallis regularized exploitability (stochastic estimate)
185  """
186  del anneal_steps
187  action_1 = random.choice(dist.size, p=dist)
188  y = nabla = payoff_matrices[0][:, action_1]
189  if temperature >= 1e-3:
190  br = special.softmax(y / temperature)
191  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
192  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
193  else:
194  power = np.inf
195  s = np.linalg.norm(y, ord=power)
196  br = np.zeros_like(dist)
197  maxima = (y == s)
198  br[maxima] = 1. / maxima.sum()
199  br_mat = np.zeros((br.size, br.size))
200  br_policy_gradient = np.zeros_like(br)
201 
202  unreg_exp = np.max(y) - y.dot(dist)
203  entr_br = temperature * special.entr(br).sum()
204  entr_dist = temperature * special.entr(dist).sum()
205  reg_exp = y.dot(br - dist) + entr_br - entr_dist
206 
207  policy_gradient = nabla - temperature * (np.log(dist) + 1)
208  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
209

211  other_player_fx = dist.size * other_player_fx[action_u]
212  other_player_fx_translat = payoff_matrices[1, :, action_u] * other_player_fx
213  grad_dist = -policy_gradient + (num_players - 1) * other_player_fx_translat

238  version: int, default 0, two options for variance reduction

242  tsallis regularized exploitability (stochastic estimate)
243  """
244  del anneal_steps
245  if pm_vr is None:
246  raise ValueError("pm_vr must be np.array of shape (num_strats,) * 2")
247  if (not isinstance(version, int)) or (version < 0) or (version > 1):
248  raise ValueError("version must be non-negative int < 2")
249 
250  action_1 = random.choice(dist.size, p=dist)
251  y = nabla = payoff_matrices[0][:, action_1]
252  if temperature >= 1e-3:
253  br = special.softmax(y / temperature)
254  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
255  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
256  else:
257  power = np.inf
258  s = np.linalg.norm(y, ord=power)
259  br = np.zeros_like(dist)
260  maxima = (y == s)
261  br[maxima] = 1. / maxima.sum()
262  br_mat = np.zeros((br.size, br.size))
263  br_policy_gradient = np.zeros_like(br)
264 
265  unreg_exp = np.max(y) - y.dot(dist)
266  entr_br = temperature * special.entr(br).sum()
267  entr_dist = temperature * special.entr(dist).sum()
268  reg_exp = y.dot(br - dist) + entr_br - entr_dist
269 
270  policy_gradient = nabla - temperature * (np.log(dist) + 1)
271  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
272 
273  if version == 0:
274  other_player_fx_translated = pm_vr.dot(other_player_fx)
275  action_u = random.choice(dist.size) # uniform, ~importance sampling
276  other_player_fx = other_player_fx[action_u]

283  other_player_fx = other_player_fx[action_u]

290  if proj_grad:

292 
293  if version == 0:
294  pm_vr[:, action_u] = payoff_matrices[1, :, action_u]
295  elif version == 1:
296  pm_vr[:, action_u] = payoff_matrices[1, :, action_u] * other_player_fx

301  """Projected gradient descent on exploitability using Euclidean projection.
302

304  params: tuple of variables to be updated (dist, anneal_steps)

308  new_params: tuple of update params (new_dist, new_anneal_steps)

310  del t

312  new_params = [params[0] - lr_dist * grads[0]]
313  new_params = euc_project(*new_params)
