73 def compute_next_policy_invariants(
74  infostates: typing.InfostateMapping, all_actions: List[int],
75  infostate_map: typing.InfostateMapping
76 ) -> tuple[Dict[str, jnp.ndarray], Dict[str, List[int]]]:
77  """Computes information needed to calculate next policy.
78 
79  This function computes one hot encodings of infostates and returns mappings
80  from infostate strings to one hot representations of infostates as well as
81  illegal actions.
82 
83  Args:
84  infostates: List of infostate mappings.
85  all_actions: List of actions.
86  infostate_map: Mapping from infostate string to infostate.
87 
88  Returns:
89  Returns mappings of infostate strings to one hot representation for
90  infostates and illegal actions
91  """
92  one_hot_representations = {}
93  illegal_actions = {}
94 
95  for (infostate_str, infostate) in infostates.items():
96  if infostate.is_terminal():
97  continue
98 
99  legal_actions = infostate.get_actions()
100 
101  if len(legal_actions) == 1:
102  infostate.policy[infostate.get_actions()[0]] = 1
103  continue
104  infostate_str_one_hot = jax.nn.one_hot(infostate_map[infostate_str],
105  len(infostates))
106  one_hot_representations[infostate_str] = infostate_str_one_hot
107  illegal_actions[infostate_str] = [
108  i for i, a in enumerate(all_actions) if a not in legal_actions
109  ]
110  return one_hot_representations, illegal_actions

113 def compute_next_policy(infostates: typing.InfostateMapping,
114  net_apply: typing.ApplyFn, net_params: typing.Params,
115  epoch: int, all_actions: List[int],
116  one_hot_representations: Dict[str, jnp.ndarray],
117  illegal_actions: Dict[str,
118  List[int]], key: hk.PRNGSequence):
119  """Computes next step policy from output of the model.
120 
121  Args:
122  infostates: List of infostate mappings.
123  net_apply: Apply function.
124  net_params: Model params.
125  epoch: epoch.
126  all_actions: List of actions.
127  one_hot_representations: Dictionary from infostate string to infostate.
128  illegal_actions: Dictionary from infostate string to the list of illegal
129  actions.
130  key: Haiku Pseudo random number generator.
131  """
132 
133  infostate_lst = []
134  input_lst = []
135  illegal_action_lst = []
136 
137  batched_net_output = []
138  for (infostate_str, infostate) in infostates.items():
139  if infostate.is_terminal():
140  continue
141 
142  legal_actions = infostate.get_actions()
143  if len(legal_actions) == 1:
144  infostate.policy[infostate.get_actions()[0]] = 1
145  continue
146  regret_vec = np.array([
147  infostate.regret[a] /
148  (epoch + 1) if a in infostate.get_actions() else 0
149  for a in all_actions
150  ])
151  if FLAGS.use_infostate_representation:
152  one_hot_representation = one_hot_representations[infostate_str]
153  net_input = jnp.concatenate([regret_vec, one_hot_representation])
154  else:
155  net_input = regret_vec
156  input_lst.append(net_input)
157  infostate_lst.append(infostate)
158  illegal_action_lst.append(illegal_actions[infostate_str])
159  batched_inputs, output_mappings, relevant_illegal_actions = (
160  utils.get_batched_input(
161  input_lst, infostate_lst, illegal_action_lst, FLAGS.batch_size
162  )
163  )
164  idx = 0
165 
166  for _ in range(int(len(batched_inputs) / FLAGS.batch_size)):
167  batched_input, output_mapping, relevant_illegal_action = batched_inputs[
168  idx:idx + FLAGS.batch_size], output_mappings[
169  idx:idx +
170  FLAGS.batch_size], relevant_illegal_actions[idx:idx +
171  FLAGS.batch_size]
172  idx += FLAGS.batch_size
173 
174  batched_input_jnp = jnp.array(
175  np.expand_dims(np.array(batched_input), axis=1))
176  batched_net_output = utils.get_network_output_batched(
177  net_apply, net_params,
178  batched_input_jnp,
179  relevant_illegal_action, key)
180  for i, infostate in enumerate(output_mapping):
181  net_output = jnp.squeeze(batched_net_output[i])
182  for ai, action in enumerate(infostate.get_actions()):
183  infostate.policy[action] = float(net_output[ai])

196  Args:
197  history_tree_node: Game tree HistoryTreeNode which is the root of the game

222  all_infostates_map[1]

234  illegal_actions_player1, key)

236  cfr.compute_reach_probabilities(history_tree_node, all_infostates_map)
237  cfr.cumulate_average_policy(list(all_infostates_map[1].values()))
238  cfr.compute_best_response_policy(infostate_nodes[2])
239  cfr.compute_reach_probabilities(history_tree_node, all_infostates_map)
240  cfr.compute_counterfactual_values(infostate_nodes[1])
241  cfr.update_regrets(list(all_infostates_map[1].values()))
242  append_counterfactual_values(
243  list(all_infostates_map[1].values()), counterfactual_values_player1)
244  cfr.normalize_average_policy(all_infostates_map[1].values())
245  cfr.compute_reach_probabilities(history_tree_node, all_infostates_map)
246  player_2_last_best_response_values.append(
247  float(cfr.compute_best_response_values(infostate_nodes[2]))
248  )
249 
250  logging.info(
251  "Epoch %d: player_2 best response value is %f",
252  epoch,
253  player_2_last_best_response_values[-1],
254  )
255 
256  return ( # pytype: disable=bad-return-type # jax-ndarray
257  counterfactual_values_player1,
258  counterfactual_values_player2,
259  player_2_last_best_response_values,
