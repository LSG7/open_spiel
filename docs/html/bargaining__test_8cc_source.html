50 void BargainingMaxTurnsTest() {
51  std::shared_ptr<const Game> game = LoadGame("bargaining(max_turns=200)");
52  std::unique_ptr<State> state = game->NewInitialState();
53  int num_turns = 200;
54  while (num_turns > 0) {
55  if (state->IsChanceNode()) {
56  ActionsAndProbs chance_outcomes = state->ChanceOutcomes();
57  state->ApplyAction(chance_outcomes[0].first);
58  } else {
59  SPIEL_CHECK_TRUE(!state->IsTerminal());
60  num_turns--;
61  std::vector<Action> legal_actions = state->LegalActions();
62  state->ApplyAction(legal_actions[0]);
63  }
64  }
65  SPIEL_CHECK_TRUE(state->IsTerminal());
66 }

68 void BargainingDiscountTest() {
69  std::shared_ptr<const Game> game = LoadGame("bargaining(discount=0.9)");
70  std::unique_ptr<State> state = game->NewInitialState();
71  BargainingState* bargaining_state =
72  static_cast<BargainingState*>(state.get());
73  ActionsAndProbs chance_outcomes = state->ChanceOutcomes();
74  state->ApplyAction(chance_outcomes[0].first);
75  std::vector<Action> legal_actions = state->LegalActions();
76  state->ApplyAction(legal_actions[0]);
77  state->ApplyAction(legal_actions[0]);
78  state->ApplyAction(legal_actions[0]);
79  state->ApplyAction(legal_actions[0]);
80  state->ApplyAction(bargaining_state->AgreeAction());
81  // P0 offers [0,0,0] then P1, then P0, then P1, then P0 agrees.
82  // P0 would get 10, but it's discounted by 0.9 three times
83  SPIEL_CHECK_FLOAT_EQ(state->PlayerReturn(0), 0.9 * 0.9 * 0.9 * 10);
84  SPIEL_CHECK_FLOAT_EQ(state->PlayerReturn(1), 0.0);
85 }
