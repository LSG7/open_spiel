61 VPNetModel::VPNetModel(const Game& game, const std::string& path,
62  const std::string& file_name, const std::string& device)
63  : device_(device),
64  path_(path),
65  flat_input_size_(game.ObservationTensorSize()),
66  num_actions_(game.NumDistinctActions()) {
67  // Some assumptions that we can remove eventually. The value net returns
68  // a single value in terms of player 0 and the game is assumed to be zero-sum,
69  // so player 1 can just be -value.
70  SPIEL_CHECK_EQ(game.NumPlayers(), 2);
71  SPIEL_CHECK_EQ(game.GetType().utility, GameType::Utility::kZeroSum);
72 
73  std::string model_path = absl::StrCat(path, "/", file_name);
74  model_meta_graph_contents_ = file::ReadContentsFromFile(model_path, "r");
75 
76  TF_CHECK_OK(
77  ReadBinaryProto(tf::Env::Default(), model_path, &meta_graph_def_));
78 
79  tf::graph::SetDefaultDevice(device, meta_graph_def_.mutable_graph_def());
80 
81  if (tf_session_ != nullptr) {
82  TF_CHECK_OK(tf_session_->Close());
83  }
84 
85  // create a new session
86  TF_CHECK_OK(NewSession(tf_opts_, &tf_session_));
87 
88  // Load graph into session
89  TF_CHECK_OK(tf_session_->Create(meta_graph_def_.graph_def()));
90 
91  // Initialize our variables
92  TF_CHECK_OK(tf_session_->Run({}, {}, {"init_all_vars_op"}, nullptr));
93 }

95 std::string VPNetModel::SaveCheckpoint(int step) {
96  std::string full_path = absl::StrCat(path_, "/checkpoint-", step);
97  tensorflow::Tensor checkpoint_path(tf::DT_STRING, tf::TensorShape());
98  checkpoint_path.scalar<tensorflow::tstring>()() = full_path;
99  TF_CHECK_OK(tf_session_->Run(
100  {{meta_graph_def_.saver_def().filename_tensor_name(), checkpoint_path}},
101  {}, {meta_graph_def_.saver_def().save_tensor_name()}, nullptr));
102  // Writing a checkpoint from python writes the metagraph file, but c++
103  // doesn't, so do it manually to make loading checkpoints easier.
104  file::File(absl::StrCat(full_path, ".meta"), "w").Write(
105  model_meta_graph_contents_);
106  return full_path;
107 }

117 std::vector<VPNetModel::InferenceOutputs> VPNetModel::Inference(
118  const std::vector<InferenceInputs>& inputs) {
119  int inference_batch_size = inputs.size();
120 
121  // Fill the inputs and mask
122  tensorflow::Tensor tf_inf_inputs(
123  tf::DT_FLOAT, tf::TensorShape({inference_batch_size, flat_input_size_}));
124  tensorflow::Tensor tf_inf_legal_mask(
125  tf::DT_BOOL, tf::TensorShape({inference_batch_size, num_actions_}));
126 
127  TensorMap inputs_matrix = tf_inf_inputs.matrix<float>();
128  TensorMapBool mask_matrix = tf_inf_legal_mask.matrix<bool>();
129 
130  for (int b = 0; b < inference_batch_size; ++b) {
131  // Zero initialize the sparse inputs.
132  for (int a = 0; a < num_actions_; ++a) {
133  mask_matrix(b, a) = 0;
134  }
135  for (Action action : inputs[b].legal_actions) {
136  mask_matrix(b, action) = 1;
137  }
138  for (int i = 0; i < inputs[b].observations.size(); ++i) {
139  inputs_matrix(b, i) = inputs[b].observations[i];
140  }
141  }
142 
143  // Run the inference
144  std::vector<tensorflow::Tensor> tf_outputs;
145  TF_CHECK_OK(tf_session_->Run(
146  {{"input", tf_inf_inputs}, {"legals_mask", tf_inf_legal_mask},
147  {"training", tensorflow::Tensor(false)}},
148  {"policy_softmax", "value_out"}, {}, &tf_outputs));
149 
150  TensorMap policy_matrix = tf_outputs[0].matrix<float>();
151  TensorMap value_matrix = tf_outputs[1].matrix<float>();
152 
153  std::vector<InferenceOutputs> out;
154  out.reserve(inference_batch_size);
155  for (int b = 0; b < inference_batch_size; ++b) {
156  double value = value_matrix(b, 0);
157 
158  ActionsAndProbs state_policy;
159  state_policy.reserve(inputs[b].legal_actions.size());
160  for (Action action : inputs[b].legal_actions) {
161  state_policy.push_back({action, policy_matrix(b, action)});
162  }
163 
164  out.push_back({value, state_policy});
165  }
166 
167  return out;
168 }

170 VPNetModel::LossInfo VPNetModel::Learn(const std::vector<TrainInputs>& inputs) {
171  int training_batch_size = inputs.size();
172 
173  tensorflow::Tensor tf_train_inputs(
174  tf::DT_FLOAT, tf::TensorShape({training_batch_size, flat_input_size_}));
175  tensorflow::Tensor tf_train_legal_mask(
176  tf::DT_BOOL, tf::TensorShape({training_batch_size, num_actions_}));
177  tensorflow::Tensor tf_policy_targets(
178  tf::DT_FLOAT, tf::TensorShape({training_batch_size, num_actions_}));
179  tensorflow::Tensor tf_value_targets(
180  tf::DT_FLOAT, tf::TensorShape({training_batch_size, 1}));
181 
182  // Fill the inputs and mask
183  TensorMap inputs_matrix = tf_train_inputs.matrix<float>();
184  TensorMapBool mask_matrix = tf_train_legal_mask.matrix<bool>();
185  TensorMap policy_targets_matrix = tf_policy_targets.matrix<float>();
186  TensorMap value_targets_matrix = tf_value_targets.matrix<float>();
187 
188  for (int b = 0; b < training_batch_size; ++b) {
189  // Zero initialize the sparse inputs.
190  for (int a = 0; a < num_actions_; ++a) {
191  mask_matrix(b, a) = 0;
192  policy_targets_matrix(b, a) = 0;
193  }
194 
195  for (Action action : inputs[b].legal_actions) {
196  mask_matrix(b, action) = 1;
197  }
198 
199  for (int a = 0; a < inputs[b].observations.size(); ++a) {
200  inputs_matrix(b, a) = inputs[b].observations[a];
201  }
202 
203  for (const auto& [action, prob] : inputs[b].policy) {
204  policy_targets_matrix(b, action) = prob;
205  }
206 
207  value_targets_matrix(b, 0) = inputs[b].value;
208  }
209 
210  // Run a training step and get the losses.
211  std::vector<tensorflow::Tensor> tf_outputs;
212  TF_CHECK_OK(tf_session_->Run({{"input", tf_train_inputs},
213  {"legals_mask", tf_train_legal_mask},
214  {"policy_targets", tf_policy_targets},
215  {"value_targets", tf_value_targets},
216  {"training", tensorflow::Tensor(true)}},
217  {"policy_loss", "value_loss", "l2_reg_loss"},
218  {"train"}, &tf_outputs));
219 
220  return LossInfo(
221  tf_outputs[0].scalar<float>()(0),
222  tf_outputs[1].scalar<float>()(0),
223  tf_outputs[2].scalar<float>()(0));
224 }
