24// A simple jeopardy dice game that includes chance nodes.
25// See http://cs.gettysburg.edu/projects/pig/index.html for details.
26// Also https://en.wikipedia.org/wiki/Pig_(dice_game)
27//
28// Piglet variant: Instead of increasing the running total by the roll results,
29// it is always increased by a fixed step size of 1 upon rolling anything higher
30// than a 1. [Note: Internally, this behaviour is modelled with only two chance
31// outcomes, rolling a 1 or rolling anything higher than that.]
32// Divide winscore by the average dice outcome != 1 (i.e. by diceoutcomes/2 + 1)
33// when enabling Piglet to play a game that's roughly equivalent to the
34// corresponding Pig game. The main advantage of this variant is thus a greatly
35// reduced state space, making the game accessible to tabular methods.
36// See also http://cs.gettysburg.edu/~tneller/papers/pig.zip. The original
37// Piglet variant described there is played with a fair coin and a winscore
38// of 10. This behaviour can be achieved by setting diceoutcomes = 2, winscore =
39// 10, piglet = true.
40//
41// Parameters:
42// "diceoutcomes" int number of outcomes of the dice (default = 6)
43// "horizon" int max number of moves before draw (default = 1000)
44// "players" int number of players (default = 2)
45// "winscore" int number of points needed to win (default = 100)
46// "piglet" bool is piglet variant enabled? (default = false)

95class PigGame : public Game {

98
99 int NumDistinctActions() const override { return 2; }
100 std::unique_ptr<State> NewInitialState() const override {
101 return std::unique_ptr<State>(new PigState(
102 shared_from_this(), dice_outcomes_, horizon_, win_score_, piglet_));
103 }
104 int MaxChanceOutcomes() const override { return dice_outcomes_; }
105
106 // There is arbitrarily chosen number to ensure the game is finite.
107 int MaxGameLength() const override { return horizon_; }
108
109 // Every chance node is preceded by a decision node (roll)
110 // -> At most as many chance nodes as decision nodes.
111 // -> Up to as many chance nodes as decision nodes, if
112 // every action is "roll" and player never 'falls'.
113 int MaxChanceNodesInHistory() const override { return MaxGameLength(); }
114
115 int NumPlayers() const override { return num_players_; }
116 double MinUtility() const override { return -1; }
117 absl::optional<double> UtilitySum() const override { return 0; }
118 double MaxUtility() const override { return +1; }
119 std::vector<int> ObservationTensorShape() const override;
120
121 private:
122 // Number of different dice outcomes, i.e. 6.
123 int dice_outcomes_;
124
125 // Maximum number of moves before draw.
126 int horizon_;
127
128 // Number of players in this game.
129 int num_players_;
130
131 // The amount needed to win.
132 int win_score_;
133
134 // Whether Piglet variant is enabled (always move only 1 step forward)
135 bool piglet_;
136};
