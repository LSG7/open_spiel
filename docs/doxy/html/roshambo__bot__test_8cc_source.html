40// This matchup is deterministic and both bots utilize the match history so
41// we can test that the bots are perceiving the game correctly.
42void RoshamboBotHistoryTest() {
43 GameParameters params;
44 params["num_repetitions"] = GameParameter(roshambo::kNumThrows);
45 std::shared_ptr<const Game> game = CreateRepeatedGame("matrix_rps", params);
46
47 std::vector<std::unique_ptr<Bot>> bots;
48 bots.push_back(roshambo::MakeRoshamboBot(0, "rotatebot"));
49 bots.push_back(roshambo::MakeRoshamboBot(1, "copybot"));
50 std::unique_ptr<State> state = game->NewInitialState();
51
52 const int num_players = bots.size();
53 std::vector<Action> joint_actions(bots.size());
54 for (int i = 0; i < roshambo::kNumThrows; ++i) {
55 for (Player p = 0; p < num_players; ++p)
56 joint_actions[p] = bots[p]->Step(*state);
57 state->ApplyActions(joint_actions);
58 if (i == 0) {
59 // Copybot wins the first round.
60 SPIEL_CHECK_EQ(state->PlayerReturn(0), -1);
61 SPIEL_CHECK_EQ(state->PlayerReturn(1), 1);
62 } else {
63 // All subsequent rounds are draws.
64 SPIEL_CHECK_EQ(state->PlayerReward(0), 0);
65 SPIEL_CHECK_EQ(state->PlayerReward(1), 0);
66 }
67 }
68 SPIEL_CHECK_TRUE(state->IsTerminal());
69 SPIEL_CHECK_EQ(state->PlayerReturn(0), -1);
70 SPIEL_CHECK_EQ(state->PlayerReturn(1), 1);
71}

73// Matchup between 1999 and 2000 tournament winners!
74void RoshamboBotBasicPlayGame() {
75 int num_games = 5;
76 std::mt19937 rng(Seed());
77 GameParameters params;
78 params["num_repetitions"] = GameParameter(roshambo::kNumThrows);
79 std::shared_ptr<const Game> game = CreateRepeatedGame("matrix_rps", params);
80 std::vector<std::unique_ptr<Bot>> bots;
81
82 bots.push_back(roshambo::MakeRoshamboBot(0, "greenberg"));
83 bots.push_back(roshambo::MakeRoshamboBot(1, "iocainebot"));
84
85 for (int i = 0; i < num_games; i++) {
86 // Set seed for the underlying C code
87 srandom(Seed());
88 std::unique_ptr<State> state = game->NewInitialState();
89
90 const int num_players = bots.size();
91 std::vector<Action> joint_actions(bots.size());
92 for (Player p = 0; p < num_players; ++p) bots[p]->Restart();
93 while (!state->IsTerminal()) {
94 for (Player p = 0; p < num_players; ++p)
95 joint_actions[p] = bots[p]->Step(*state);
96 state->ApplyActions(joint_actions);
97 }
98 }
99}
