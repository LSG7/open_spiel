6 // Type conversion utilities for pybind11 and absl data structures.
7 //
8 // Usage: Just include this file in the .cc file with your bindings and add the
9 // appropriate dependency. Any functions which take or return the supported
10 // types will have those types automatically converted.
11 //
12 // Supported types:
13 // - absl::Duration- converted to/from python datetime.timedelta
14 // - absl::CivilTime- converted to/from python datetime.datetime and from date.
15 // - absl::Time- converted to/from python datetime.datetime and from date.
16 // - absl::TimeZone- converted to/from python str and from int.
17 // - absl::Span- converted to python sequences and from python buffers,
18 // opaque std::vectors and/or sequences.
19 // - absl::string_view
20 // - absl::optional- converts absl::nullopt to/from python None, otherwise
21 // converts the contained value.
22 // - absl::flat_hash_map- converts to/from python dict.
23 // - absl::flat_hash_set- converst to/from python set.
24 // - absl::btree_map- converts to/from python dict.
25 //
26 // For details, see the README.md.
27 //

62 // Given a python date or datetime object, figure out an appropriate
63 // absl::TimeZone with a fixed offset, or default to the local timezone.
64 inline absl::TimeZone GetTimeZone(handle src) {
65  if (!hasattr(src, "tzinfo")) {
66  // datetime.date objects lack this property, so assume local.
67  return absl::LocalTimeZone();
68  }
69  object tzinfo = src.attr("tzinfo");
70  if (tzinfo.is_none()) {
71  // non-tz aware datetime, again assume local time.
72  return absl::LocalTimeZone();
73  }
74  object utc_offset = tzinfo.attr("utcoffset")(src);
75  if (utc_offset.is_none()) {
76  return absl::LocalTimeZone();
77  }
78  int64_t offset_seconds =
79  std::lround(utc_offset.attr("total_seconds")().cast<double>());
80  return absl::FixedTimeZone(offset_seconds);
81 }

108 // Convert between absl::Duration and python datetime.timedelta.
109 template <>
110 struct type_caster<absl::Duration> {
111  public:
112  // This macro establishes the name 'absl::Duration' in function signatures
113  // and declares a local variable 'value' of type absl::Duration.
114  PYBIND11_TYPE_CASTER(absl::Duration, const_name("absl::Duration"));
115 
116  // Conversion part 1 (Python->C++)
117  bool load(handle src, bool convert) {
118  if (!convert) {
119  return false;
120  }
121  if (PyFloat_Check(src.ptr())) {
122  value = absl::Seconds(src.cast<double>());
123  return true;
124  } else if (PyLong_Check(src.ptr())) {
125  value = absl::Seconds(src.cast<int64_t>());
126  return true;
127  } else {
128  // Ensure that absl::Duration is converted from a Python
129  // datetime.timedelta.
130  if (!hasattr(src, "days") || !hasattr(src, "seconds") ||
131  !hasattr(src, "microseconds")) {
132  return false;
133  }
134  value = absl::Hours(24 * GetInt64Attr(src, "days")) +
135  absl::Seconds(GetInt64Attr(src, "seconds")) +
136  absl::Microseconds(GetInt64Attr(src, "microseconds"));
137  return true;
138  }
139  return false;
140  }
141 
142  // Conversion part 2 (C++ -> Python)
143  static handle cast(const absl::Duration& src, return_value_policy, handle) {
144  absl::Duration remainder;
145  int64_t secs = absl::IDivDuration(src, absl::Seconds(1), &remainder);
146  int64_t microsecs = absl::ToInt64Microseconds(remainder);
147  auto py_duration_t = module::import("datetime").attr("timedelta");
148  auto py_duration =
149  py_duration_t(arg("seconds") = secs, arg("microseconds") = microsecs);
150  return py_duration.release();
151  }
152 };

154 // Convert between absl::Time and python datetime.date, datetime.
155 template <>
156 struct type_caster<absl::Time> {
157  public:
158  // This macro establishes the name 'absl::Time' in function signatures
159  // and declares a local variable 'value' of type absl::Time.
160  PYBIND11_TYPE_CASTER(absl::Time, const_name("absl::Time"));
161 
162  // Conversion part 1 (Python->C++)
163  bool load(handle src, bool convert) {
164  // Ensure that absl::Duration is converted from a Python datetime.date.
165  if (!convert || !hasattr(src, "year") || !hasattr(src, "month") ||
166  !hasattr(src, "day")) {
167  return false;
168  }
169  if (hasattr(src, "timestamp")) {
170  // python datetime.datetime object
171  double timestamp = src.attr("timestamp")().cast<double>();
172  int64_t as_micros = static_cast<int64_t>(timestamp * 1e6);
173  value = absl::FromUnixMicros(as_micros);
174  }
175 #if PY_MAJOR_VERSION < 3
176  else if (hasattr(src, "microsecond")) { // NOLINT(readability/braces)
177  // python datetime.datetime object
178  // This doesn't rely on datetime.datetime.timestamp(), which is not
179  // available in Python 2.
180  auto utc = module::import("dateutil.tz").attr("UTC");
181  auto datetime = module_::import("datetime").attr("datetime");
182  auto gettz = module::import("dateutil.tz").attr("gettz");
183 
184  auto epoch_dt = datetime.attr("fromtimestamp")(0, utc);
185  auto tz = (src.attr("tzinfo").is_none()) ? gettz() : src.attr("tzinfo");
186  auto src_with_tz = src.attr("replace")("tzinfo"_a = tz);
187 
188  double timestamp =
189  (src_with_tz - epoch_dt).attr("total_seconds")().cast<double>();
190  int64_t as_micros = static_cast<int64_t>(timestamp * 1e6);
191  value = absl::FromUnixMicros(as_micros);
192  }
193 #endif
194  else { // NOLINT(readability/braces)
195  // python datetime.date object
196  absl::CivilDay civil_day(GetInt64Attr(src, "year"),
197  GetInt64Attr(src, "month"),
198  GetInt64Attr(src, "day"));
199  value = absl::FromCivil(civil_day, GetTimeZone(src));
200  }
201  return true;
202  }
203 
204  // Conversion part 2 (C++ -> Python)
205  static handle cast(const absl::Time& src, return_value_policy, handle) {
206  // This function truncates fractional microseconds as the python datetime
207  // objects cannot support a resolution higher than this.
208  auto py_datetime_t = module::import("datetime").attr("datetime");
209  auto py_from_timestamp = py_datetime_t.attr("fromtimestamp");
210  auto py_timezone_t = module::import("dateutil.tz").attr("gettz");
211  auto py_timezone = py_timezone_t(absl::LocalTimeZone().name());
212  double as_seconds = static_cast<double>(absl::ToUnixMicros(src)) / 1e6;
213  auto py_datetime = py_from_timestamp(as_seconds, "tz"_a = py_timezone);
214  return py_datetime.release();
215  }
216 };

218 template <typename CivilTimeType>
219 struct absl_civil_time_caster {
220  public:
221  PYBIND11_TYPE_CASTER(CivilTimeType, const_name("CivilTimeType"));
222 
223  bool load(handle src, bool convert) {
224  if (!convert || !hasattr(src, "year") || !hasattr(src, "month") ||
225  !hasattr(src, "day")) {
226  return false;
227  }
228  int64_t hour = 0, minute = 0, second = 0;
229  if (hasattr(src, "hour") && hasattr(src, "minute") &&
230  hasattr(src, "second")) {
231  hour = GetInt64Attr(src, "hour");
232  minute = GetInt64Attr(src, "minute");
233  second = GetInt64Attr(src, "second");
234  }
235  value = CivilTimeType(GetInt64Attr(src, "year"), GetInt64Attr(src, "month"),
236  GetInt64Attr(src, "day"), hour, minute, second);
237  return true;
238  }
239 
240  static handle cast(const CivilTimeType& src, return_value_policy, handle) {
241  auto py_datetime_t = module::import("datetime").attr("datetime");
242  auto py_datetime = py_datetime_t(src.year(), src.month(), src.day(),
243  src.hour(), src.minute(), src.second());
244  return py_datetime.release();
245  }

272 // Returns {true, a span referencing the data contained by src} without copying
273 // or converting the data if possible. Otherwise returns {false, an empty span}.
274 template <typename T, typename std::enable_if<std::is_arithmetic<T>::value,
275  bool>::type = true>
276 std::tuple<bool, absl::Span<T>> LoadSpanFromBuffer(handle src) {
277  Py_buffer view;
278  int flags = PyBUF_STRIDES | PyBUF_FORMAT;
279  if (!std::is_const<T>::value) flags |= PyBUF_WRITABLE;
280  if (PyObject_GetBuffer(src.ptr(), &view, flags) == 0) {
281  auto cleanup = absl::MakeCleanup([&view] { PyBuffer_Release(&view); });
282  if (view.ndim == 1 && view.strides[0] == sizeof(T) &&
283  view.format[0] == format_descriptor<T>::c) {
284  return {true, absl::MakeSpan(static_cast<T*>(view.buf), view.shape[0])};
285  }
286  } else {
287  // Clear the buffer error (failure is reported in the return value).
288  PyErr_Clear();
289  }
290  return {false, absl::Span<T>()};
291 }
292 // If T is not a numeric type, the buffer interface cannot be used.
293 template <typename T, typename std::enable_if<!std::is_arithmetic<T>::value,
294  bool>::type = true>
295 constexpr std::tuple<bool, absl::Span<T>> LoadSpanFromBuffer(handle src) {
296  return {false, absl::Span<T>()};
297 }

305 // Convert between absl::Span and sequence types.
306 // See http://g3doc/pybind11_abseil/README.md#abslspan
307 template <typename T>
308 struct type_caster<absl::Span<T>> {
309  public:
310  // The type referenced by the span, with const removed.
311  using value_type = typename std::remove_cv<T>::type;
312  static_assert(!is_absl_span<value_type>::value,
313  "Nested absl spans are not supported.");
314 
315  type_caster() = default;
316  // Copy and Move operations must ensure the span points to the copied or
317  // moved vector (if any), not the original one. Allows implicit conversions.
318  template <typename U>
319  type_caster(const type_caster<absl::Span<U>>& other) {
320  *this = other;
321  }
322  template <typename U>
323  type_caster(type_caster<absl::Span<U>>&& other) {
324  *this = std::move(other);
325  }
326  template <typename U>
327  type_caster& operator=(const type_caster<absl::Span<U>>& other) {
328  list_caster_ = other.list_caster_;
329  value_ = list_caster_ ? get_value(*list_caster_) : other.value_;
330  return *this;
331  }
332  template <typename U>
333  type_caster& operator=(type_caster<absl::Span<U>>&& other) {
334  list_caster_ = std::move(other.list_caster_);
335  value_ = list_caster_ ? get_value(*list_caster_) : other.value_;
336  return *this;
337  }
338 
339  static constexpr auto name = _("Span[") + make_caster<T>::name + _("]");
340 
341  // We do not allow moving because 1) spans are super lightweight, so there's
342  // no advantage to moving and 2) the span cannot exist without the caster,
343  // so moving leaves an implicit dependency (while a reference or pointer
344  // make that dependency explicit).
345  operator absl::Span<T>*() { return &value_; }
346  operator absl::Span<T>&() { return value_; }
347  template <typename T_>
348  using cast_op_type = cast_op_type<T_>;
349 
350  bool load(handle src, bool convert) {
351  // Attempt to reference a buffer, including np.ndarray and array.arrays.
352  bool loaded;
353  std::tie(loaded, value_) = LoadSpanFromBuffer<T>(src);
354  if (loaded) return true;
355 
356  // Attempt to unwrap an opaque std::vector.
357  type_caster_base<std::vector<value_type>> caster;
358  if (caster.load(src, false)) {
359  value_ = get_value(caster);
360  return true;
361  }
362 
363  // Attempt to convert a native sequence. If the is_base_of_v check passes,
364  // the elements do not require converting and pointers do not reference a
365  // temporary object owned by the element caster. Pointers to converted
366  // types are not allowed because they would result a dangling reference
367  // when the element caster is destroyed.
368  if (convert && std::is_const<T>::value &&
369  (!std::is_pointer<T>::value ||
370  std::is_base_of<type_caster_generic, make_caster<T>>::value)) {
371  list_caster_.emplace();
372  if (list_caster_->load(src, convert)) {
373  value_ = get_value(*list_caster_);
374  return true;
375  } else {
376  list_caster_.reset();
377  }
378  }
379 
380  return false; // Python type cannot be loaded into a span.
381  }
382 
383  template <typename CType>
384  static handle cast(CType&& src, return_value_policy policy, handle parent) {
385  return ListCaster::cast(src, policy, parent);
386  }
387 
388  private:
389  template <typename Caster>
390  absl::Span<T> get_value(Caster& caster) {
391  return absl::MakeSpan(static_cast<std::vector<value_type>&>(caster));
392  }
393 
394  using ListCaster = list_caster<std::vector<value_type>, value_type>;
395  absl::optional<ListCaster> list_caster_;
396  absl::Span<T> value_;
397 };
