148 logic::CardSet HoleCards(Player player) const {
149 logic::CardSet hole_cards;
150 const int num_players = acpc_game_->GetNbPlayers();
151 const int num_cards_dealt_to_all = hole_cards_dealt_ / num_players;
152 int num_cards_dealt_to_player = num_cards_dealt_to_all;
153 // We deal to players in order from 0 to n - 1. So if the number of cards
154 // dealt % num_players is > the player, we haven't dealt them a card yet;
155 // otherwise we have.
156 if (player < (hole_cards_dealt_ % num_players) &&
157 num_cards_dealt_to_all < acpc_game_->GetNbHoleCardsRequired()) {
158 ++num_cards_dealt_to_player;
159 }
160 SPIEL_CHECK_LT(player, acpc_game_->GetNbPlayers());
161 SPIEL_CHECK_LE(num_cards_dealt_to_player,
162 static_cast<int>(acpc_game_->GetNbHoleCardsRequired()));
163 for (int i = 0; i < num_cards_dealt_to_player; ++i) {
164 hole_cards.AddCard(acpc_state_.hole_cards(player, i));
165 }
166 return hole_cards;
167 }

169 logic::CardSet BoardCards() const {

173 static_cast<int>(acpc_game_->GetTotalNbBoardCards()));
174 for (int i = 0; i < num_board_cards; ++i) {
175 board_cards.AddCard(acpc_state_.board_cards(i));

242// Only supported for UniversalPoker. Randomly plays an action from a fixed list
243// of actions. If none of the actions are legal, checks/calls.
244class UniformRestrictedActions : public Policy {
245 public:
246 // Actions will be restricted to this list when legal. If no such action is
247 // legal, checks/calls.
248 explicit UniformRestrictedActions(absl::Span<const ActionType> actions)
249 : actions_(actions.begin(), actions.end()),
250 max_action_(*absl::c_max_element(actions)) {}
251
252 ActionsAndProbs GetStatePolicy(const State &state) const {
253 ActionsAndProbs policy;
254 policy.reserve(actions_.size());
255 const std::vector<Action> legal_actions = state.LegalActions();
256 for (Action action : legal_actions) {
257 if (actions_.contains(static_cast<ActionType>(action))) {
258 policy.emplace_back(action, 1.);
259 }
260 if (policy.size() >= actions_.size() || action > max_action_) break;
261 }
262
263 // It is always legal to check/call.
264 if (policy.empty()) {
265 SPIEL_DCHECK_TRUE(absl::c_find(legal_actions, ActionType::kCall) !=
266 legal_actions.end());
267 policy.push_back({static_cast<Action>(ActionType::kCall), 1.});
268 }
269
270 // If we have a non-empty policy, normalize it!
271 if (policy.size() > 1) NormalizePolicy(&policy);
272 return policy;
273 }
274
275 private:
276 const absl::flat_hash_set<ActionType> actions_;
277 const ActionType max_action_;
278};

285// Get hole card index within the array of reach probabilities, as specified
286// in https://github.com/Sandholm-Lab/LibratusEndgames :
287//
288// The probability, according to the Libratus blueprint strategy, of each player
289// reaching this endgame with each hand. There are a total of 2,652
290// probabilities in this list. The first 1,326 are for the "out of position"
291// player (the first player to act on the round), while the remaining 1,326 are
292// for the "button" player. Each of the 1,326 probabilities corresponds to a
293// poker hand, ordered as follows:
