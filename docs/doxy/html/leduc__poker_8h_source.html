15// A generalized version of a Leduc poker, a simple but non-trivial poker game
16// described in http://poker.cs.ualberta.ca/publications/UAI05.pdf .
17//
18// Taken verbatim from the linked paper above: "In Leduc hold'em, the deck
19// consists of two suits with three cards in each suit. There are two rounds.
20// In the first round a single private card is dealt to each player. In the
21// second round a single board card is revealed. There is a two-bet maximum,
22// with raise amounts of 2 and 4 in the first and second round, respectively.
23// Both players start the first round with 1 already in the pot.
24//
25// So the maximin sequence is of the form:
26// private card player 0, private card player 1, [bets], public card, [bets]
27//
28// Parameters:
29// "players" int number of players (default = 2)
30// "action_mapping" bool regard all actions as legal and internally
31// map otherwise illegal actions to check/call
32// (default = false)
33// "suit_isomorphism" bool player observations do not distinguish
34// between cards of different suits with
35// the same rank (default = false)
36

216 constexpr int MaxBetsPerRound() const {
217 // E.g. longest round for 4-player is 10 bets:
218 // check, check, check, bet, call, call, raise, call, call, call
219 // = 1 bet + 1 raise + (num_players_-1)*2 calls + (num_players_-2) calls
220 return 3 * num_players_ - 2;
221 }
222 int MaxGameLength() const override {
223 // 2 rounds.
224 return 2 * MaxBetsPerRound();
225 }
226 int MaxChanceNodesInHistory() const override { return 3; }
227 int NumObservableCards() const {
228 return suit_isomorphism_ ? total_cards_ / 2 : total_cards_;
229 }
