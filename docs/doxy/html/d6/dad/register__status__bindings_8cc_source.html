21// Returns false if status_or represents a non-ok status object, and true in all
22// other cases (including the case that this is passed a non-status object).
23bool IsOk(handle status_or) {
24 detail::make_caster<absl::Status> caster;
25 // "Not a status" means "ok" for StatusOr.
26 if (!caster.load(status_or, true)) return true;
27 return static_cast<absl::Status &>(caster).ok();
28}

96object BuildStatusNotOk(handle PyStatusNotOk, const StatusNotOk& e) {
97 object py_e = PyStatusNotOk(e.what());
98 py_e.attr("status") = cast(google::NoThrowStatus<absl::Status>(e.status()));
99 // code is int by choice. Sorry it would be a major API break to make this an
100 // enum. Use status.code() to obtain the enum.
101 py_e.attr("code") = cast(e.status().raw_code());
102 py_e.attr("message") = cast(e.status().message());
103 return py_e;
104}

110void RegisterStatusBindings(module m) {
111 enum_<absl::StatusCode>(m, "StatusCode")
112 .value("OK", absl::StatusCode::kOk)
113 .value("CANCELLED", absl::StatusCode::kCancelled)
114 .value("UNKNOWN", absl::StatusCode::kUnknown)
115 .value("INVALID_ARGUMENT", absl::StatusCode::kInvalidArgument)
116 .value("DEADLINE_EXCEEDED", absl::StatusCode::kDeadlineExceeded)
117 .value("NOT_FOUND", absl::StatusCode::kNotFound)
118 .value("ALREADY_EXISTS", absl::StatusCode::kAlreadyExists)
119 .value("PERMISSION_DENIED", absl::StatusCode::kPermissionDenied)
120 .value("RESOURCE_EXHAUSTED", absl::StatusCode::kResourceExhausted)
121 .value("FAILED_PRECONDITION", absl::StatusCode::kFailedPrecondition)
122 .value("ABORTED", absl::StatusCode::kAborted)
123 .value("OUT_OF_RANGE", absl::StatusCode::kOutOfRange)
124 .value("UNIMPLEMENTED", absl::StatusCode::kUnimplemented)
125 .value("INTERNAL", absl::StatusCode::kInternal)
126 .value("UNAVAILABLE", absl::StatusCode::kUnavailable)
127 .value("DATA_LOSS", absl::StatusCode::kDataLoss)
128 .value("UNAUTHENTICATED", absl::StatusCode::kUnauthenticated);
129
130 class_<absl::Status>(m, "Status")
131 .def(init())
132 .def(init<absl::StatusCode, std::string>())
133 .def("ok", &absl::Status::ok)
134 .def("code", &absl::Status::code)
135 .def("code_int", [](const absl::Status& self) {
136 return static_cast<int>(self.code());
137 })
138 .def("message", &absl::Status::message)
139 .def(
140 "update",
141 (void (absl::Status::*)(const absl::Status &)) & absl::Status::Update,
142 arg("other"))
143 .def("to_string", [](const absl::Status& s) { return s.ToString(); })
144 .def("__repr__", [](const absl::Status& s) { return s.ToString(); })
145 .def_static("OkStatus", DoNotThrowStatus(&absl::OkStatus))
146 .def("raw_code", &absl::Status::raw_code)
147 .def("CanonicalCode", [](const absl::Status& self) {
148 return static_cast<int>(self.code());
149 })
150 .def("error_message", &absl::Status::message)
151 .def("IgnoreError", &absl::Status::IgnoreError);
152
153 m.def("is_ok", &IsOk, arg("status_or"),
154 "Returns false only if passed a non-ok status; otherwise returns true. "
155 "This can be used on the return value of a function which returns a "
156 "StatusOr without raising an exception. The .ok() method cannot be "
157 "used in this case because an ok status is never returned; instead, a "
158 "non-status object is returned, which doesn't have a .ok() method.");
159
160 // Return canonical errors.
161 def_status_factory(m, "aborted_error", WrapAbortedError);

179 // Generate the Python exception type.
180 static exception<StatusNotOk> PyStatusNotOk(m, "StatusNotOk");

182 // Register a custom handler which converts a C++ StatusNotOk to a
183 // PyStatusNotOk.
184 register_exception_translator([](std::exception_ptr p) {
185 try {
186 if (p) std::rethrow_exception(p);
187 } catch (const StatusNotOk& e) {
188 PyErr_SetObject(PyStatusNotOk.ptr(),
189 BuildStatusNotOk(PyStatusNotOk, e).ptr());
190 }
191 });
192
193 m.def("BuildStatusNotOk", [](int code, const std::string& msg) {
194 return BuildStatusNotOk(
195 PyStatusNotOk,
196 StatusNotOk(absl::Status(static_cast<absl::StatusCode>(code), msg)));
197 });
198}
