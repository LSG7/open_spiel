70 void ExternalSamplingMCCFRSolver::RunIteration(std::mt19937* rng) {
71  for (auto p = Player{0}; p < game_->NumPlayers(); ++p) {
72  UpdateRegrets(*game_->NewInitialState(), p, rng);
73  }
74 
75  if (avg_type_ == AverageType::kFull) {
76  std::vector<double> reach_probs(game_->NumPlayers(), 1.0);
77  FullUpdateAverage(*game_->NewInitialState(), reach_probs);
78  }
79 }

99  // RNG section
100  absl::StrAppend(&str, kSerializeSolverRNGSectionHeader, "\n");
101  std::ostringstream rng_stream;
102  rng_stream << *rng_;
103  absl::StrAppend(&str, rng_stream.str(), "\n");
104  // Average type section
105  absl::StrAppend(&str, kSerializeSolverAverageTypeSectionHeader, "\n");
106  if (avg_type_ == AverageType::kFull)
107  absl::StrAppend(&str, "FullAverageType", "\n");
108  else if (avg_type_ == AverageType::kSimple)
109  absl::StrAppend(&str, "SimpleAverageType", "\n");
110  // Default policy section
111  absl::StrAppend(&str, kSerializeSolverDefaultPolicySectionHeader, "\n");
112  absl::StrAppend(&str, default_policy_->Serialize(double_precision, delimiter),
113  "\n");
114  // Info state values table section
115  absl::StrAppend(&str, kSerializeSolverValuesTableSectionHeader, "\n");
116  SerializeCFRInfoStateValuesTable(info_states_, &str, double_precision,
117  delimiter);

121 double ExternalSamplingMCCFRSolver::UpdateRegrets(const State& state,
122  Player player,
123  std::mt19937* rng) {
124  if (state.IsTerminal()) {
125  return state.PlayerReturn(player);
126  } else if (state.IsChanceNode()) {
127  Action action = SampleAction(state.ChanceOutcomes(), dist_(*rng)).first;
128  return UpdateRegrets(*state.Child(action), player, rng);
129  } else if (state.IsSimultaneousNode()) {
130  SpielFatalError(
131  "Simultaneous moves not supported. Use "
132  "TurnBasedSimultaneousGame to convert the game first.");
133  }
134 
135  Player cur_player = state.CurrentPlayer();
136  std::string is_key = state.InformationStateString(cur_player);
137  std::vector<Action> legal_actions = state.LegalActions();
138 
139  // The insert here only inserts the default value if the key is not found,
140  // otherwise returns the entry in the map.
141  auto iter_and_result = info_states_.insert(
142  {is_key, CFRInfoStateValues(legal_actions, kInitialTableValues)});
143 
144  CFRInfoStateValues info_state_copy = iter_and_result.first->second;
145  info_state_copy.ApplyRegretMatching();
146 
147  double value = 0;
148  std::vector<double> child_values(legal_actions.size(), 0);
149 
150  if (cur_player != player) {
151  // Sample at opponent nodes.
152  int aidx = info_state_copy.SampleActionIndex(0.0, dist_(*rng));
153  value = UpdateRegrets(*state.Child(legal_actions[aidx]), player, rng);
154  } else {
155  // Walk over all actions at my nodes
156  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
157  child_values[aidx] =
158  UpdateRegrets(*state.Child(legal_actions[aidx]), player, rng);
159  value += info_state_copy.current_policy[aidx] * child_values[aidx];
160  }
161  }
162 
163  // Now the regret and avg strategy updates.
164  CFRInfoStateValues& info_state = info_states_[is_key];
165 
166  if (cur_player == player) {
167  // Update regrets
168  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
169  info_state.cumulative_regrets[aidx] += (child_values[aidx] - value);
170  }
171  }
172 
173  // Simple average does averaging on the opponent node. To do this in a game
174  // with more than two players, we only update the player + 1 mod num_players,
175  // which reduces to the standard rule in 2 players.
176  if (avg_type_ == AverageType::kSimple &&
177  cur_player == ((player + 1) % game_->NumPlayers())) {
178  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
179  info_state.cumulative_policy[aidx] +=
180  info_state_copy.current_policy[aidx];
181  }
182  }
183 
184  return value;
185 }

187 void ExternalSamplingMCCFRSolver::FullUpdateAverage(
188  const State& state, const std::vector<double>& reach_probs) {
189  if (state.IsTerminal()) {
190  return;
191  } else if (state.IsChanceNode()) {
192  for (Action action : state.LegalActions()) {
193  FullUpdateAverage(*state.Child(action), reach_probs);
194  }
195  return;
196  } else if (state.IsSimultaneousNode()) {
197  SpielFatalError(
198  "Simultaneous moves not supported. Use "
199  "TurnBasedSimultaneousGame to convert the game first.");
200  }
201 
202  // If all the probs are zero, no need to keep going.
203  double sum = std::accumulate(reach_probs.begin(), reach_probs.end(), 0.0);
204  if (sum == 0.0) return;
205 
206  Player cur_player = state.CurrentPlayer();
207  std::string is_key = state.InformationStateString(cur_player);
208  std::vector<Action> legal_actions = state.LegalActions();
209 
210  // The insert here only inserts the default value if the key is not found,
211  // otherwise returns the entry in the map.
212  auto iter_and_result = info_states_.insert(
213  {is_key, CFRInfoStateValues(legal_actions, kInitialTableValues)});
214 
215  CFRInfoStateValues info_state_copy = iter_and_result.first->second;
216  info_state_copy.ApplyRegretMatching();
217 
218  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
219  std::vector<double> new_reach_probs = reach_probs;
220  new_reach_probs[cur_player] *= info_state_copy.current_policy[aidx];
221  FullUpdateAverage(*state.Child(legal_actions[aidx]), new_reach_probs);
222  }
223 
224  // Now update the cumulative policy.
225  CFRInfoStateValues& info_state = info_states_[is_key];
226  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
227  info_state.cumulative_policy[aidx] +=
228  (reach_probs[cur_player] * info_state_copy.current_policy[aidx]);
229  }
230 }

232 std::unique_ptr<ExternalSamplingMCCFRSolver>
233 DeserializeExternalSamplingMCCFRSolver(const std::string& serialized,
234  std::string delimiter) {
235  auto partial = PartiallyDeserializeCFRSolver(serialized);
236  SPIEL_CHECK_EQ(partial.solver_type, "ExternalSamplingMCCFRSolver");
237 
238  enum Section {
239  kInvalid = -1,
240  kRNG = 0,
241  kAverageType = 1,
242  kDefaultPolicy = 2
243  };
244  std::array<std::string, 3> section_strings = {"", "", ""};
245  Section current_section = kInvalid;
246 
247  std::vector<absl::string_view> lines =
248  absl::StrSplit(partial.solver_specific_state, '\n');
249  for (int i = 0; i < lines.size(); i++) {
250  if (lines[i] == kSerializeSolverRNGSectionHeader) {
251  SPIEL_CHECK_EQ(current_section, kInvalid);
252  current_section = kRNG;
253  } else if (lines[i] == kSerializeSolverAverageTypeSectionHeader) {
254  SPIEL_CHECK_EQ(current_section, kRNG);
255  current_section = kAverageType;
256  } else if (lines[i] == kSerializeSolverDefaultPolicySectionHeader) {
257  SPIEL_CHECK_EQ(current_section, kAverageType);
258  current_section = kDefaultPolicy;
259  } else {
260  SPIEL_CHECK_NE(current_section, kInvalid);
261  absl::StrAppend(&section_strings[current_section], lines[i], "\n");
262  }
263  }
264  // Remove trailing newline characters
265  for (int i = 0; i < section_strings.size(); i++) {
266  section_strings[i].erase(section_strings[i].length() - 1);
267  }
268 
269  // Do the actual deserialization for all internal state values
270  std::unique_ptr<std::mt19937> rng = std::make_unique<std::mt19937>();
271  std::istringstream rng_stream(section_strings[kRNG]);
272  rng_stream >> *rng;
273 
274  AverageType average_type;
275  if (section_strings[kAverageType] == "FullAverageType")
276  average_type = AverageType::kFull;
277  else if (section_strings[kAverageType] == "SimpleAverageType")
278  average_type = AverageType::kSimple;
279 
280  auto solver = std::make_unique<ExternalSamplingMCCFRSolver>(
281  partial.game, DeserializePolicy(section_strings[kDefaultPolicy]),
282  std::move(rng), average_type);
283  DeserializeCFRInfoStateValuesTable(partial.serialized_cfr_values_table,
284  &solver->InfoStateValuesTable(),
285  delimiter);
286  return solver;
287 }
