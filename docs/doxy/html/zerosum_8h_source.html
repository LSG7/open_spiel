23// Transforms a general sum game into a zero sum one by subtracting the mean
24// of the rewards and final returns.
25
26namespace open_spiel {
27
28inline std::vector<double> SubtractMean(std::vector<double>&& vec) {
29 double mean = std::accumulate(vec.begin(), vec.end(), 0.0) / vec.size();
30 std::vector<double> result = std::move(vec);
31 for (auto& item : result) item -= mean;
32 return result;
33}

54class ZeroSumGame : public WrappedGame {
55 public:
56 ZeroSumGame(std::shared_ptr<const Game> game, GameType game_type,
57 GameParameters game_parameters);
58 ZeroSumGame(const ZeroSumGame& other) = default;
59
60 std::unique_ptr<State> NewInitialState() const override {
61 return std::unique_ptr<State>(
62 new ZeroSumState(shared_from_this(), game_->NewInitialState()));
63 }
64
65 double MaxUtility() const override {
66 // The maximum utility is obtained if, in the original game,
67 // one player gains game_->MaxUtility() while all other players
68 // obtain game_->MinUtility(), because the mean is subtracted.
69 double n = static_cast<double>(game_->NumPlayers());
70 return (game_->MaxUtility() - game_->MinUtility()) * (n - 1) / n;
71 }
72 double MinUtility() const override {
73 // By symmetry:
74 return - MaxUtility();
75 }
76 absl::optional<double> UtilitySum() const override {
77 return 0.0;
78 }
