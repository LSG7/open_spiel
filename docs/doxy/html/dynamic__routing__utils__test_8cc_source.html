84std::unique_ptr<Network> InitNetwork() {
85 absl::flat_hash_map<std::string, std::vector<std::string>> adjacency_list;
86 adjacency_list["O"] = std::vector<std::string>{"A"};

92void TestNetworkAdjacencyListInit() {
93 auto network = InitNetwork();
94 SPIEL_CHECK_EQ(network->GetActionIdFromMovement("O", "A"), 2);
95 SPIEL_CHECK_EQ(network->GetActionIdFromMovement("A", "D"), 1);
96 SPIEL_CHECK_EQ(network->num_links(), 2);
97 SPIEL_CHECK_EQ(network->GetSuccessors("O"), std::vector<std::string>{"A"});
98 SPIEL_CHECK_EQ(network->GetSuccessors("A"), std::vector<std::string>{"D"});
99 SPIEL_CHECK_EQ(network->GetSuccessors("D"), std::vector<std::string>{});
100 SPIEL_CHECK_TRUE(network->IsLocationASinkNode("A->D"));
101 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("O->A"));
102 SPIEL_CHECK_EQ(network->GetRoadSectionFromActionId(2), "O->A");
103 SPIEL_CHECK_EQ(network->GetRoadSectionFromActionId(1), "A->D");
104}
