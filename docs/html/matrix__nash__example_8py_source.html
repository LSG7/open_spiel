15 """Find Nash equilibria for constant- or general-sum 2-player games.
16 
17 Non-matrix games are handled by computing the normal (bimatrix) form.
18 
19 The algorithms used are:
20 * direct computation of pure equilibria.
21 * linear programming to find equilibria for constant-sum games.
22 * iterated dominance to reduce the action space.
23 * reverse search vertex enumeration (if using lrsnash) to find all general-sum
24  equilibria.
25 * support enumeration (if using nashpy) to find all general-sum equilibria.
26 * Lemke-Howson enumeration (if using nashpy) to find one general-sum
27  equilibrium.
28 
29 The general-sum mixed-equilibrium algorithms are likely to work well for tens of
30 actions, but less likely to scale beyond that.
31 
32 
33 Example usage:
34 ```
35 matrix_nash_example --game kuhn_poker
36 ```
37 """

71 def main(_):
72  game = pyspiel.load_game(FLAGS.game)
73  print("loaded game")
74 
75  # convert game to matrix form if it isn't already a matrix game
76  if not isinstance(game, pyspiel.MatrixGame):
77  game = pyspiel.extensive_to_matrix_game(game)
78  num_rows, num_cols = game.num_rows(), game.num_cols()
79  print("converted to matrix form with shape (%d, %d)" % (num_rows, num_cols))
80 
81  # use iterated dominance to reduce the space unless the solver is LP (fast)
82  if FLAGS.solver != "linear":
83  if FLAGS.mode == "all":
84  game, _ = lp_solver.iterated_dominance(
85  game, tol=FLAGS.tol, mode=lp_solver.DOMINANCE_STRICT)
86  num_rows, num_cols = game.num_rows(), game.num_cols()
87  print("discarded strictly dominated actions yielding shape (%d, %d)" %
88  (num_rows, num_cols))
89  if FLAGS.mode == "one":
90  game, _ = lp_solver.iterated_dominance(
91  game, tol=FLAGS.tol, mode=lp_solver.DOMINANCE_VERY_WEAK)
92  num_rows, num_cols = game.num_rows(), game.num_cols()
93  print("discarded very weakly dominated actions yielding shape (%d, %d)" %
94  (num_rows, num_cols))
95 
96  # game is now finalized
97  num_rows, num_cols = game.num_rows(), game.num_cols()
98  row_actions = [game.row_action_name(row) for row in range(num_rows)]
99  col_actions = [game.col_action_name(col) for col in range(num_cols)]
100  row_payoffs, col_payoffs = utils.game_payoffs_array(game)
101  pure_nash = list(
102  zip(*((row_payoffs >= row_payoffs.max(0, keepdims=True) - FLAGS.tol)
103  & (col_payoffs >= col_payoffs.max(1, keepdims=True) - FLAGS.tol)
104  ).nonzero()))
105  if pure_nash:
106  print("found %d pure equilibria" % len(pure_nash))
107  if FLAGS.mode == "pure":
108  if not pure_nash:
109  print("found no pure equilibria")
110  return
111  print("pure equilibria:")
112  for row, col in pure_nash:
113  print("payoffs %f, %f:" % (row_payoffs[row, col], col_payoffs[row, col]))
114  print("row action:")
115  print(row_actions[row])
116  print("col action:")
117  print(col_actions[col])
118  print("")
119  return
120  if FLAGS.mode == "one" and pure_nash:
121  print("pure equilibrium:")
122  row, col = pure_nash[0]
123  print("payoffs %f, %f:" % (row_payoffs[row, col], col_payoffs[row, col]))
124  print("row action:")
125  print(row_actions[row])
126  print("col action:")
127  print(col_actions[col])
128  print("")
129  return
130  for row, action in enumerate(row_actions):
131  print("row action %s:" % row)
132  print(action)
133  print("--")
134  for col, action in enumerate(col_actions):
135  print("col action %s:" % col)
136  print(action)
137  print("--")
138  if num_rows == 1 or num_cols == 1:
139  equilibria = itertools.product(np.eye(num_rows), np.eye(num_cols))
140  elif FLAGS.solver == "linear":
141  if FLAGS.mode != "one" or (row_payoffs + col_payoffs).max() > (
142  row_payoffs + col_payoffs).min() + FLAGS.tol:
143  raise ValueError("can't use linear solver for non-constant-sum game or "
144  "for finding all optima!")
145  print("using linear solver")
146 
147  def gen():
148  p0_sol, p1_sol, _, _ = lp_solver.solve_zero_sum_matrix_game(
149  pyspiel.create_matrix_game(row_payoffs - col_payoffs,
150  col_payoffs - row_payoffs))
151  yield (np.squeeze(p0_sol, 1), np.squeeze(p1_sol, 1))
152 
153  equilibria = gen()
154  elif FLAGS.solver == "lrsnash":
155  print("using lrsnash solver")
156  equilibria = matrix_nash.lrs_solve(row_payoffs, col_payoffs,
157  FLAGS.lrsnash_max_denom,
158  FLAGS.lrsnash_path)
159  elif FLAGS.solver == "nashpy":
160  if FLAGS.mode == "all":
161  print("using nashpy vertex enumeration")
162  equilibria = nashpy.Game(row_payoffs, col_payoffs).vertex_enumeration()
163  else:
164  print("using nashpy Lemke-Howson solver")
165  equilibria = matrix_nash.lemke_howson_solve(row_payoffs, col_payoffs)
166  print("equilibria:" if FLAGS.mode == "all" else "an equilibrium:")
167  equilibria = iter(equilibria)
168  # check that there's at least one equilibrium
169  try:
170  equilibria = itertools.chain([next(equilibria)], equilibria)
171  except StopIteration:
172  print("not found!")
173  for row_mixture, col_mixture in equilibria:
174  print("payoffs %f, %f for %s, %s" %
175  (row_mixture.dot(row_payoffs.dot(col_mixture)),
176  row_mixture.dot(
177  col_payoffs.dot(col_mixture)), row_mixture, col_mixture))
178  if FLAGS.mode == "one":
179  return
