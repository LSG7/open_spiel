39 class Subprocess {
40  pid_t child_pid_;
41  int in_pipe_[2];
42  int out_pipe_[2];
43  int err_pipe_[2];
44 
45  public:
46  Subprocess(const std::string& shell_command, bool should_block = false) {
47  // Create pipes for input/output/error communication.
48  RUN(pipe, in_pipe_);
49  RUN(pipe, out_pipe_);
50  RUN(pipe, err_pipe_);
51 
52  // Make sure to set all file descriptors of the pipes to be non-blocking.
53  if (!should_block) {
54  RUN(fcntl, in_pipe_[WRITE], F_SETFL, O_NONBLOCK);
55  RUN(fcntl, out_pipe_[READ], F_SETFL, O_NONBLOCK);
56  RUN(fcntl, err_pipe_[READ], F_SETFL, O_NONBLOCK);
57  }
58 
59  // Clone the calling process, creating an exact copy.
60  // Returns -1 for errors, 0 to the new process,
61  // and the process ID of the new process to the old process.
62  child_pid_ = fork();
63  if (child_pid_ == -1) {
64  std::perror("subprocess: fork failed");
65  std::exit(1);
66  }
67  if (child_pid_ == 0) child(shell_command);
68 
69  // The code below will be executed only by parent.
70  RUN(close, in_pipe_[READ]);
71  RUN(close, out_pipe_[WRITE]);
72  RUN(close, err_pipe_[WRITE]);
73  }
74 
75  int stdin() { return in_pipe_[WRITE]; }
76  int stdout() { return out_pipe_[READ]; }
77  int stderr() { return err_pipe_[READ]; }
78  pid_t child_pid() const { return child_pid_; }
79 
80  private:
81  enum ends_of_pipe { READ = 0, WRITE = 1 };
82 
83  // Code run only by the child process.
84  void child(const std::string& shell_command) {
85  // Connect the pipe ends to STDIO for the child.
86  RUN(dup2, in_pipe_[READ], STDIN_FILENO)
87  RUN(dup2, out_pipe_[WRITE], STDOUT_FILENO)
88  RUN(dup2, err_pipe_[WRITE], STDERR_FILENO)
89 
90  // Close all parent pipes, as they have been rerouted.
91  for (auto& pipe : {in_pipe_, out_pipe_, err_pipe_}) {
92  RUN(close, pipe[READ]);
93  RUN(close, pipe[WRITE]);
94  }
95 
96  std::vector<std::string> cargs;
97  cargs.push_back("/bin/sh");
98  cargs.push_back("-c");
99  std::string command = shell_command; // Drop const.
100  cargs.push_back(command.data());
101 
102  char **argv = new char* [cargs.size()+1];
103  argv[cargs.size()] = nullptr;
104  for (int i = 0; i < cargs.size(); ++i) {
105  argv[i] = const_cast<char*>(cargs[i].c_str());
106  }
107 
108  // Execute the command.
109  RUN(execvp, argv[0], argv);
110  delete [] argv;
111  }
112 };
