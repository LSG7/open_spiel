29// The AFCCE auxiliary game is very similar to the EFCCE auxiliary game. The
30// only difference is that the number of deviations is limited to 1. Once the
31// player has deviated, they can no longer deviate and must follow for the
32// rest of the game.
33class AFCCEState : public WrappedState {
34 public:
35 AFCCEState(std::shared_ptr<const Game> game, std::unique_ptr<State> state,
36 CorrDistConfig config, const CorrelationDevice& mu,
37 Action follow_action, Action defect_action);
38
39 std::unique_ptr<State> Clone() const override {
40 return std::make_unique<AFCCEState>(*this);
41 }
42
43 // Need to override this because otherwise WrappedState forwards the
44 // implementation to the underlying state, which calls the wrong
45 // ChanceOutcomes
46 std::vector<Action> LegalChanceOutcomes() const override {
47 return State::LegalChanceOutcomes();
48 }
49
50 Player CurrentPlayer() const override;
51 ActionsAndProbs ChanceOutcomes() const override;
52 std::vector<Action> LegalActions() const override;
53 std::string InformationStateString(Player player) const override;
54 std::string ToString() const override;
55
56 bool HasDefected(Player player) const;
57
58 protected:
59 Action CurRecommendation() const;
60 void DoApplyAction(Action action_id) override;
61
62 private:
63 const CorrDistConfig config_;
64 const CorrelationDevice& mu_;
65
66 Action follow_action_;
67 Action defect_action_;
68
69 // Which joint policy was chosen?
70 int rec_index_;
71
72 // Has the player defected?
73 std::vector<int> defected_;
74
75 // Where did the player defect? This is the information set of the original
76 // game. Indexed by player.
77 std::vector<absl::optional<std::string>> defection_infoset_;
78
79 // The sequence of recommendations, indexed by player
80 std::vector<std::vector<Action>> recommendation_seq_;
81};

83class AFCCEGame : public WrappedGame {
84 public:
85 AFCCEGame(std::shared_ptr<const Game> game, CorrDistConfig config,
86 const CorrelationDevice& mu)
87 : WrappedGame(game, game->GetType(), game->GetParameters()),
88 config_(config),
89 mu_(mu),
90 orig_num_distinct_actions_(game->NumDistinctActions()) {}
91
92 std::unique_ptr<State> NewInitialState() const override {
93 return std::make_unique<AFCCEState>(shared_from_this(),
94 game_->NewInitialState(), config_, mu_,
95 FollowAction(), DefectAction());
96 }
97
98 int NumDistinctActions() const override {
99 // 2 extra actions: cooperate/follow or defect
100 return orig_num_distinct_actions_ + 2;
101 }
102
103 int FollowAction() const { return orig_num_distinct_actions_; }
104 int DefectAction() const { return orig_num_distinct_actions_ + 1; }
105
106 private:
107 const CorrDistConfig config_;
108 const CorrelationDevice& mu_;
109
110 // Number of distinct actions in the original game.
111 int orig_num_distinct_actions_;
112};
