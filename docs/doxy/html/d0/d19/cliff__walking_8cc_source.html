96std::string CliffWalkingState::ToString() const {
97 std::string str;
98 str.reserve(height_ * (width_ + 1));
99 for (int r = 0; r < height_; ++r) {
100 for (int c = 0; c < width_; ++c) {
101 if (r == player_row_ && c == player_col_)
102 str += 'P';
103 else if (IsCliff(r, c))
104 str += 'X';
105 else if (IsGoal(r, c))
106 str += 'G';
107 else
108 str += '.';
109 }
110 str += '\n';
111 }
112 return str;
113}

115bool CliffWalkingState::IsTerminal() const {

143void CliffWalkingState::ObservationTensor(Player player,
144 absl::Span<float> values) const {
145 SPIEL_CHECK_GE(player, 0);
146 SPIEL_CHECK_LT(player, num_players_);
147 SPIEL_CHECK_EQ(values.size(), height_ * width_);

166void CliffWalkingState::DoApplyAction(Action move) {
167 switch (move) {
168 case RIGHT:
169 ++player_col_;
170 break;
171 case UP:
172 --player_row_;
173 break;
174 case LEFT:
175 --player_col_;
176 break;
177 case DOWN:
178 ++player_row_;
179 break;
180 default:
181 SpielFatalError("Unexpected action");
182 }
183 player_row_ = std::min(std::max(player_row_, 0), height_ - 1);
184 player_col_ = std::min(std::max(player_col_, 0), width_ - 1);
185 ++time_counter_;
186}

188bool CliffWalkingState::IsCliff(int row, int col) const {
189 return col > 0 && col < width_ - 1 && row == height_ - 1;
190}
191
192bool CliffWalkingState::IsGoal(int row, int col) const {
193 return row == height_ - 1 && col == width_ - 1;
194}
