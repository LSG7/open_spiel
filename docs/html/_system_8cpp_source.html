179 void System::GetHardware(
180  int& ncores,
181  unsigned long long& kilobytesFree) const
182 {
183  kilobytesFree = 0;
184  ncores = 1;
185  (void) System::GetCores(ncores);
186 
187 #if defined(_WIN32) || defined(__CYGWIN__)
188  // Using GlobalMemoryStatusEx instead of GlobalMemoryStatus
189  // was suggested by Lorne Anderson.
190  MEMORYSTATUSEX statex;
191  statex.dwLength = sizeof(statex);
192  GlobalMemoryStatusEx(&statex);
193  kilobytesFree = static_cast<unsigned long long>(
194  statex.ullTotalPhys / 1024);
195 
196  SYSTEM_INFO sysinfo;
197  GetSystemInfo(&sysinfo);
198  ncores = static_cast<int>(sysinfo.dwNumberOfProcessors);
199  return;
200 #endif
201 
202 #ifdef __APPLE__
203  // The code for Mac OS X was suggested by Matthew Kidd.
204 
205  // This is physical memory, rather than "free" memory as below
206  // for Linux. Always leave 0.5 GB for the OS and other stuff.
207  // It would be better to find free memory (how?) but in practice
208  // the number of cores rather than free memory is almost certainly
209  // the limit for Macs which have standardized hardware (whereas
210  // say a 32 core Linux server is hardly unusual).
211  FILE * fifo = popen("sysctl -n hw.memsize", "r");
212  fscanf(fifo, "%lld", &kilobytesFree);
213  fclose(fifo);
214 
215  kilobytesFree /= 1024;
216  if (kilobytesFree > 500000)
217  {
218  kilobytesFree -= 500000;
219  }
220 
221  ncores = sysconf(_SC_NPROCESSORS_ONLN);
222  return;
223 #endif
224 
225 #ifdef __linux__
226  // The code for linux was suggested by Antony Lee.
227  FILE * fifo = popen(
228  "free -k | tail -n+3 | head -n1 | awk '{print $NF}'", "r");
229  int ignore = fscanf(fifo, "%llu", &kilobytesFree);
230  fclose(fifo);
231 
232  ncores = sysconf(_SC_NPROCESSORS_ONLN);
233  return;
234 #endif
235 }

340 int System::RunThreadsWinAPI()
341 {
342 #ifdef DDS_THREADS_WINAPI
343  HANDLE * solveAllEvents = static_cast<HANDLE * >(
344  malloc(static_cast<unsigned>(numThreads) * sizeof(HANDLE)));
345 
346  for (int k = 0; k < numThreads; k++)
347  {
348  solveAllEvents[k] = CreateEvent(NULL, FALSE, FALSE, 0);
349  if (solveAllEvents[k] == 0)
350  return RETURN_THREAD_CREATE;
351  }
352 
353  vector<WinWrapType> winWrap;
354  const unsigned nt = static_cast<unsigned>(numThreads);
355  winWrap.resize(nt);
356 
357  for (unsigned k = 0; k < nt; k++)
358  {
359  winWrap[k].thrId = static_cast<int>(k);
360  winWrap[k].fptr = fptr;
361  winWrap[k].waitPtr = solveAllEvents;
362 
363  int res = QueueUserWorkItem(WinCallback,
364  static_cast<void *>(&winWrap[k]), WT_EXECUTELONGFUNCTION);
365  if (res != 1)
366  return res;
367  }
368 
369  DWORD solveAllWaitResult;
370  solveAllWaitResult = WaitForMultipleObjects(
371  static_cast<unsigned>(numThreads), solveAllEvents, TRUE, INFINITE);
372 
373  if (solveAllWaitResult != WAIT_OBJECT_0)
374  return RETURN_THREAD_WAIT;
375 
376  for (int k = 0; k < numThreads; k++)
377  CloseHandle(solveAllEvents[k]);
378 
379  free(solveAllEvents);
380 #endif
381 
382  return RETURN_NO_FAULT;
383 }

490 int System::RunThreadsSTLIMPL()
491 {
492 #ifdef DDS_THREADS_STLIMPL
493  vector<int> uniques;
494  vector<int> crossrefs;
495  (* CallbackDuplList[runCat])(* bop, uniques, crossrefs);
496 
497  static atomic<int> thrIdNext = 0;
498  bool err = false;
499 
500  threadMgr.Reset(numThreads);
501 
502  for_each(std::execution::par, uniques.begin(), uniques.end(),
503  [&](int &bno)
504  {
505  thread_local int thrId = -1;
506  thread_local int realThrId;
507  if (thrId == -1)
508  thrId = thrIdNext++;
509 
510  realThrId = threadMgr.Occupy(thrId);
511 
512  if (realThrId == -1)
513  err = true;
514  else
515  (* CallbackSingleList[runCat])(realThrId, bno);
516 
517  if (! threadMgr.Release(thrId))
518  err = true;
519  });
520 
521  if (err)
522  {
523  cout << "Too many threads, numThreads " << numThreads << endl;
524  return RETURN_THREAD_INDEX;
525  }
526 
527  (* CallbackCopyList[runCat])(crossrefs);
528 #endif
529 
530  return RETURN_NO_FAULT;
