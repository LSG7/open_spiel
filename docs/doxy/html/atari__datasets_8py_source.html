41 def get_column(self, agent_name: str) -> Dict[str, float]:
42 column_dict = {}
43 agent_idx = self.agent_names.index(agent_name)
44 assert 0 <= agent_idx < len(self.agent_names)
45 for game_name, scores in self.table_data.items():
46 column_dict[game_name] = scores[agent_idx]

61 def add_column(self, column, agent_name):
62 """Add a column.
63
64 Args:
65 column: a dictionary of game_name -> score,
66 agent_name: name for the new agent.
67
68 Note: beware! This can delete rows within this data set, in order to keep
69 data complete, i.e. it deletes rows if you don't have this agent's score for
70 that game.
71 """
72 self.agent_names.append(agent_name)
73 game_names_copy = self.game_names[:]
74 for game_name in game_names_copy:
75 if game_name not in column:
76 logging.warning("Warning: deleting game {%s}", game_name)
77 self.delete_game(game_name)
78 else:
79 self.table_data[game_name].append(column[game_name])

81 def to_task_by_agent_matrix(self) -> np.ndarray:

118def parse_atari_table(filename: str) -> DataSet:
119 """Parse an Atari data file.
120
121 The files are created by copy/paste from the papers.
122
123 Args:
124 filename: the file that contains the dataset.
125
126 Returns:
127 a DataSet object referring to the Atari data.
128 """
129 with open(filename, "r") as f:
130 string_data = f.read()
131
132 # First line is a comment
133 # Second line format is column descriptions, e.g.:
134 # "# game <agent1 name> <agent2 name> ..."
135 # Rest of the lines are copy/paste from the paper tables.
136 lines = string_data.split("\n")
137 assert lines[1].startswith("# game ")
138 agent_names = lines[1].split()[2:]
139 num_agents = len(agent_names)
140 game_names = []
141 table_data = {}
142 for i in range(2, len(lines)):
143 if lines[i].strip():
144 parts = lines[i].split()
145 game_name = parts[0]
146 game_names.append(game_name)
147 str_scores = parts[1:]
148 assert len(str_scores) == num_agents, f"Error line: {lines[i]}"
149 scores = parse_values(str_scores)
150 table_data[game_name] = scores
151 return DataSet(agent_names, game_names, table_data)
