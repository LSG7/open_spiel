15 """An example of building and exporting a Tensorflow graph.
16 
17 Adapted from the Travis Ebesu's blog post:
18 https://tebesu.github.io/posts/Training-a-TensorFlow-graph-in-C++-API
19 """
20 
21 from absl import app
22 from absl import flags
23 import numpy as np
24 import tensorflow.compat.v1 as tf
25 import pyspiel
26 
27 FLAGS = flags.FLAGS
28 flags.DEFINE_string("game", "breakthrough", "Name of the game")
29 flags.DEFINE_string("dir", "/tmp", "Directory to save graph")
30 flags.DEFINE_string("filename", "graph.pb", "Filename for the graph")
31 
32 
33 def main(_):
34  game = pyspiel.load_game(FLAGS.game)
35 
36  # Information state length
37  info_state_shape = game.observation_tensor_shape()
38  flat_info_state_length = np.prod(info_state_shape)
39 
40  # Output
41  num_actions = game.num_distinct_actions()
42 
43  with tf.Session() as sess:
44  net_input = tf.placeholder(
45  tf.float32, [None, flat_info_state_length], name="input")
46 
47  # pylint: disable=unused-variable
48  output = tf.placeholder(tf.float32, [None, num_actions], name="output")
49  legals_mask = tf.placeholder(
50  tf.float32, [None, num_actions], name="legals_mask")
51 
52  policy_net = tf.layers.dense(net_input, 128, activation=tf.nn.relu)
53  policy_net = tf.layers.dense(policy_net, 128, activation=tf.nn.relu)
54  policy_net = tf.layers.dense(policy_net, num_actions)
55 
56  # Note: subtracting the max here is to help with numerical stability.
57  # However, there can still be numerical problems. If you are doing a softmax
58  # here, it can return NaN when the max for the policy net is high on one of
59  # the illegal actions, because policy_net - max will be small for legal
60  # actions, giving all exp(small) == 0 in the denominator, returning NaN at
61  # the end. One fix is to set the logits to -inf and define a custom cross
62  # entropy op that ignores over the illegal actions.
63  policy_net = policy_net - tf.reduce_max(policy_net, axis=-1, keepdims=True)
64 
65  masked_exp_logit = tf.multiply(tf.exp(policy_net), legals_mask)
66  renormalizing_factor = tf.reduce_sum(
67  masked_exp_logit, axis=-1, keepdims=True)
68  # pylint: disable=unused-variable
69  policy_softmax = tf.where(
70  tf.equal(legals_mask, 0.),
71  tf.zeros_like(masked_exp_logit),
72  tf.divide(masked_exp_logit, renormalizing_factor),
73  name="policy_softmax")
74 
75  policy_targets = tf.placeholder(shape=[None, num_actions], dtype=tf.float32)
76 
77  policy_cost = tf.reduce_mean(
78  tf.nn.softmax_cross_entropy_with_logits_v2(
79  logits=policy_net, labels=policy_targets),
80  axis=0)
81 
82  # We make one sample.
83  sampled_actions = tf.random.categorical(
84  tf.log(policy_softmax), 1, name="sampled_actions")
85 
86  # pylint: disable=unused-variable
87  optimizer = tf.train.AdamOptimizer(0.0001).minimize(
88  policy_cost, name="train")
89 
90  # pylint: disable=unused-variable
91  init = tf.variables_initializer(tf.global_variables(),
92  name="init_all_vars_op")
93 
94  print("Writing file: {}/{}".format(FLAGS.dir, FLAGS.filename))
95  tf.train.write_graph(
96  sess.graph_def, FLAGS.dir, FLAGS.filename, as_text=False)
97 
98 
99 if __name__ == "__main__":
100  app.run(main)
