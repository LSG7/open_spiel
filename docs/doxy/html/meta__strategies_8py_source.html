28def uniform_strategy(solver, return_joint=False):
29 """Returns a Random Uniform distribution on policies.
30
31 Args:
32 solver: GenPSROSolver instance.
33 return_joint: If true, only returns marginals. Otherwise marginals as well
34 as joint probabilities.
35
36 Returns:
37 uniform distribution on strategies.
38 """
39 policies = solver.get_policies()
40 policy_lengths = [len(pol) for pol in policies]
41 result = [np.ones(pol_len) / pol_len for pol_len in policy_lengths]
42 if not return_joint:
43 return result
44 else:
45 joint_strategies = get_joint_strategy_from_marginals(result)
46 return result, joint_strategies

49def softmax_on_range(number_policies):
50 x = np.array(list(range(number_policies)))
51 x = np.exp(x-x.max())
52 x /= np.sum(x)
53 return x

56def uniform_biased_strategy(solver, return_joint=False):
57 """Returns a Biased Random Uniform distribution on policies.
58
59 The uniform distribution is biased to prioritize playing against more recent
60 policies (Policies that were appended to the policy list later in training)
61 instead of older ones.
62
63 Args:
64 solver: GenPSROSolver instance.
65 return_joint: If true, only returns marginals. Otherwise marginals as well
66 as joint probabilities.
67
68 Returns:
69 uniform distribution on strategies.
70 """
71 policies = solver.get_policies()
72 if not isinstance(policies[0], list):
73 policies = [policies]
74 policy_lengths = [len(pol) for pol in policies]

83def renormalize(probabilities):
84 """Replaces all negative entries with zeroes and normalizes the result.
85
86 Args:
87 probabilities: probability vector to renormalize. Has to be one-dimensional.
88
89 Returns:
90 Renormalized probabilities.
91 """
92 probabilities[probabilities < 0] = 0
93 probabilities = probabilities / np.sum(probabilities)
94 return probabilities

97def get_joint_strategy_from_marginals(probabilities):
98 """Returns a joint strategy tensor from a list of marginals.
99
100 Args:
101 probabilities: list of probabilities.
102
103 Returns:
104 A flat joint strategy from a list of marginals.
105 """
106 res = np.ones((1,), dtype=np.float64)
107 for prob in probabilities:
108 res = res[..., None] @ np.asarray(prob).reshape((1,) * res.ndim + (-1,))
109 return res.reshape(-1)

112def nash_strategy(solver, return_joint=False):
113 """Returns nash distribution on meta game matrix.
114
115 This method only works for two player zero-sum games.
116
117 Args:
118 solver: GenPSROSolver instance.
119 return_joint: If true, only returns marginals. Otherwise marginals as well
120 as joint probabilities.
121
122 Returns:
123 Nash distribution on strategies.
124 """
125 meta_games = solver.get_meta_game()
126 if not isinstance(meta_games, list):
127 meta_games = [meta_games, -meta_games]
128 meta_games = [x.tolist() for x in meta_games]
129 if len(meta_games) != 2:
130 raise NotImplementedError(
131 "nash_strategy solver works only for 2p zero-sum"
132 "games, but was invoked for a {} player game".format(len(meta_games)))
133 nash_prob_1, nash_prob_2, _, _ = (
134 lp_solver.solve_zero_sum_matrix_game(
135 pyspiel.create_matrix_game(*meta_games)))
136 result = [
137 renormalize(np.array(nash_prob_1).reshape(-1)),
138 renormalize(np.array(nash_prob_2).reshape(-1))
139 ]
140
141 if not return_joint:
142 return result
143 else:
144 joint_strategies = get_joint_strategy_from_marginals(result)
145 return result, joint_strategies
