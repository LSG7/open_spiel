25 //
26 // ABSL_INTERNAL_CAT(a, b)
27 //
28 // This macro takes two arguments and concatenates them together via ## after
29 // expansion.
30 //
31 // Example:
32 //
33 // ABSL_INTERNAL_CAT(foo_, bar)
34 //
35 // Results in:
36 //
37 // foo_bar
38 #define ABSL_INTERNAL_CAT(a, b) ABSL_INTERNAL_CAT_IMPL(a, b)
39 #define ABSL_INTERNAL_CAT_IMPL(a, b) a##b
40 
41 //
42 // ABSL_INTERNAL_TRANSFORM_ARGS(m, ...)
43 //
44 // This macro takes another macro as an argument followed by a trailing series
45 // of additional parameters (up to 32 additional arguments). It invokes the
46 // passed-in macro once for each of the additional arguments, with the
47 // expansions separated by commas.
48 //
49 // Example:
50 //
51 // ABSL_INTERNAL_TRANSFORM_ARGS(MY_MACRO, a, b, c)
52 //
53 // Results in:
54 //
55 // MY_MACRO(a), MY_MACRO(b), MY_MACRO(c)
56 //
57 // TODO(calabrese) Handle no arguments as a special case.
58 #define ABSL_INTERNAL_TRANSFORM_ARGS(m, ...) \
59  ABSL_INTERNAL_CAT(ABSL_INTERNAL_TRANSFORM_ARGS, \
60  ABSL_INTERNAL_NUM_ARGS(__VA_ARGS__)) \
61  (m, __VA_ARGS__)
