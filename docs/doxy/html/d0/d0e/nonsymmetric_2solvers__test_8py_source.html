35def numerical_gradient(fun, x, eps=np.sqrt(np.finfo(float).eps)):
36 fun_0 = fun(x)
37 num_grad = [np.zeros_like(xi) for xi in x]
38 x_plus_dx = [np.copy(xi) for xi in x]
39 for i, xi in enumerate(x):
40 for j, xij in enumerate(xi):
41 x_plus_dx[i][j] = xij + eps
42 num_grad[i][j] = (fun(x_plus_dx) - fun_0) / eps
43 x_plus_dx[i][j] = xij
44 return num_grad

47def prep_params(dist, pt, num_params):

69 def test_exploitability_gradient_on_nonsymmetric_three_player_matrix_games(
70 self, solver_tuple, trials=100, max_num_strats=3, atol=1e-1, rtol=1e-1,
71 seed=1234):
72 num_players = 3
73 solver = solver_tuple[0].Solver(*solver_tuple[1:])
74
75 if hasattr(solver, "num_estimates"):
76 num_estimates = solver.num_estimates
77 else:
78 num_estimates = 1
79
80 random = np.random.RandomState(seed)
81
82 successes = []
83 for _ in range(trials):
84 num_strats = random.randint(low=2, high=max_num_strats + 1,
85 size=num_players)
86 num_strats = tuple([int(ns) for ns in num_strats])
87 payoff_tensor = random.rand(num_players, *num_strats)
88
89 num_params = len(solver.init_vars(num_strats, num_players))
90 dirichlet_alpha = [np.ones(num_strats_i) for num_strats_i in num_strats]
91 dist = [random.dirichlet(alpha_i) for alpha_i in dirichlet_alpha]
92 params = prep_params(dist, payoff_tensor, num_params)
93
94 payoff_matrices = {}
95 for pi, pj in itertools.combinations(range(num_players), 2):
96 key = (pi, pj)
97 pt_i = misc.pt_reduce(payoff_tensor[pi], dist, [pi, pj])
98 pt_j = misc.pt_reduce(payoff_tensor[pj], dist, [pi, pj])
99 payoff_matrices[key] = np.stack((pt_i, pt_j), axis=0)
100 if num_estimates > 1:
101 payoff_matrices = [payoff_matrices] * num_estimates
102 grad = solver.compute_gradients(params, payoff_matrices)[0][0]
103 grad = np.concatenate(grad) / float(num_players)
104
105 exp = lambda x: solver.exploitability(x, payoff_tensor) # pylint: disable=cell-var-from-loop
106 num_grad = np.concatenate(numerical_gradient(exp, dist))
107
108 successes += [np.logical_and(np.allclose(grad, num_grad, rtol, atol),
109 distance.cosine(grad, num_grad) <= atol)]
110
111 perc = 100 * np.mean(successes)
112 logging.info("gradient accuracy success rate out of %d is %f", trials, perc)
113 self.assertGreaterEqual(
114 perc, 95., "exploitability gradient accuracy is too poor")
