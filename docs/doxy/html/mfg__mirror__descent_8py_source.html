46def main(argv: Sequence[str]) -> None:
47 if len(argv) > 1:

50 game = factory.create_game_with_setting(_GAME_NAME.value, _SETTING.value)
51
52 # Metrics writer will also log the metrics to stderr.
53 just_logging = _LOGDIR.value is None
54 writer = metrics.create_default_writer(
55 logdir=_LOGDIR.value, just_logging=just_logging)
56
57 # Save the parameters.
58 learning_rate = _LEARNING_RATE.value
59 writer.write_hparams({'learning_rate': learning_rate})
60
61 md = mirror_descent.MirrorDescent(game, lr=learning_rate)
62
63 for it in range(_NUM_ITERATIONS.value):

67 writer.write_scalars(it, {'exploitability': exploitability})
68 if _LOG_DISTRIBUTION.value and not just_logging:
69 filename = os.path.join(_LOGDIR.value, f'distribution_{it}.pkl')
