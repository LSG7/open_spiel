28 std::vector<double> EvaluateBots(State* state, const std::vector<Bot*>& bots,
29  int seed) {
30  const int num_players = bots.size();
31  std::mt19937 rng(seed);
32  std::vector<Action> joint_actions(bots.size());
33  if (state->History().empty()) {
34  for (auto bot : bots) bot->Restart();
35  } else {
36  for (auto bot : bots) bot->RestartAt(*state);
37  }
38  while (!state->IsTerminal()) {
39  if (state->IsChanceNode()) {
40  Action action = SampleAction(state->ChanceOutcomes(), rng).first;
41  for (auto bot : bots) bot->InformAction(*state, kChancePlayerId, action);
42  state->ApplyAction(action);
43  } else if (state->IsSimultaneousNode()) {
44  for (Player p = 0; p < num_players; ++p) {
45  if (state->LegalActions(p).empty()) {
46  joint_actions[p] = kInvalidAction;
47  } else {
48  joint_actions[p] = bots[p]->Step(*state);
49  }
50  }
51  state->ApplyActions(joint_actions);
52  } else {
53  Player current_player = state->CurrentPlayer();
54  Action action = bots[current_player]->Step(*state);
55  for (Player p = 0; p < num_players; ++p) {
56  if (p != current_player) {
57  bots[p]->InformAction(*state, current_player, action);
58  }
59  }
60  state->ApplyAction(action);
61  }
62  }
63 
64  // Return terminal utility.
65  return state->Returns();
66 }
