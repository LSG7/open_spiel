47 // Implements the Go Text Protocol, GTP, which is a text based protocol for
48 // communication with computer Go programs
49 // (https://www.lysator.liu.se/~gunnar/gtp/). This offers the open_spiel games
50 // and the mcts bot as a command line gtp server, which can be played against
51 // third party programs, or used on the command line directly.
52 int main(int argc, char** argv) {
53  absl::ParseCommandLine(argc, argv);
54 
55  std::string game_name = absl::GetFlag(FLAGS_game);
56  std::shared_ptr<const open_spiel::Game> game =
57  open_spiel::LoadGame(game_name);
58 
59  std::unique_ptr<open_spiel::State> state = game->NewInitialState();
60 
61  auto evaluator =
62  std::make_shared<open_spiel::algorithms::RandomRolloutEvaluator>(
63  /*n_rollouts=*/1, /*seed=*/0);
64  std::unique_ptr<open_spiel::Bot> bot = MakeBot(*game, evaluator);
65 
66  using Args = std::vector<std::string>;
67  std::map<std::string, std::function<std::string(const Args&)>> cmds = {
68  {"name", [](const Args&) { return Success("open_spiel"); }},
69  {"version", [](const Args&) { return Success("unknown"); }},
70  {"protocol_version", [](const Args&) { return Success("2"); }},
71  {"quit", [](const Args&) { return Success(); }},
72  {"list_commands", [&cmds](const Args& args) {
73  std::vector<std::string> keys;
74  keys.reserve(cmds.size());
75  for (auto const& item : cmds) {
76  keys.push_back(item.first);
77  }
78  return Success(absl::StrJoin(keys, " "));
79  }},
80  {"known_command", [&cmds](const Args& args) {
81  if (args.empty()) {
82  return Failure("Not enough args");
83  }
84  return Success(cmds.find(args[0]) == cmds.end() ? "false" : "true");
85  }},
86  {"known_games", [](const Args& args) {
87  return Success(absl::StrJoin(open_spiel::RegisteredGames(), " "));
88  }},
89  {"game", [&bot, &game, &state, &evaluator](const Args& args) {
90  if (args.empty()) {
91  return Success(game->ToString());
92  }
93  game = open_spiel::LoadGame(args[0]);
94  state = game->NewInitialState();
95  bot = MakeBot(*game, evaluator);
96  return Success(game->ToString());
97  }},
98  {"boardsize", [&bot, &game, &state, &evaluator](const Args& args) {
99  open_spiel::GameParameters params = game->GetParameters();
100  if (params.find("board_size") == params.end()) {
101  return Failure("Game doesn't support setting the board size");
102  }
103  if (args.empty()) {
104  return Success(params["board_size"].ToString());
105  }
106  int board_size;
107  if (!absl::SimpleAtoi(args[0], &board_size)) {
108  return Failure("Failed to parse first arg as an int");
109  }
110  params["board_size"] = open_spiel::GameParameter(board_size);
111  game = open_spiel::LoadGame(game->GetType().short_name, params);
112  state = game->NewInitialState();
113  bot = MakeBot(*game, evaluator);
114  return Success();
115  }},
116  {"play", [&bot, &state](const Args& args) {
117  if (args.size() < 2) {
118  return Failure("Not enough args");
119  }
120  // Ignore player arg, assume it's always the current player.
121  const std::string& action_str = args[1];
122  for (const open_spiel::Action action : state->LegalActions()) {
123  if (action_str == state->ActionToString(action)) {
124  bot->InformAction(*state, state->CurrentPlayer(), action);
125  state->ApplyAction(action);
126  return Success();
127  }
128  }
129  return Failure("Invalid action");
130  }},
131  {"genmove", [&bot, &state](const Args& args) {
132  if (state->IsTerminal()) {
133  return Failure("Game is already over");
134  }
135  // Ignore player arg, assume it's always the current player.
136  open_spiel::Action action = bot->Step(*state);
137  std::string action_str = state->ActionToString(action);
138  state->ApplyAction(action);
139  return Success(action_str);
140  }},
141  {"clear_board", [&bot, &game, &state](const Args& args) {
142  state = game->NewInitialState();
143  bot->Restart();
144  return Success();
145  }},
146  {"undo", [&bot, &game, &state](const Args& args) {
147  std::vector<open_spiel::Action> history = state->History();
148  int count = 1;
149  if (!args.empty() && !absl::SimpleAtoi(args[0], &count)) {
150  return Failure("Failed to parse first arg as an int");
151  }
152  if (history.size() < count) {
153  return Failure(absl::StrCat(
154  "Can't undo ", count, " moves from game of length ",
155  history.size()));
156  }
157  state = game->NewInitialState();
158  bot->Restart();
159  for (int i = 0; i < history.size() - count; ++i) {
160  bot->InformAction(*state, state->CurrentPlayer(), history[i]);
161  state->ApplyAction(history[i]);
162  }
163  return Success();
164  }},
165  {"showboard", [&state](const Args& args) {
166  return Success("\n" + state->ToString());
167  }},
168  {"history", [&state](const Args& args) {
169  return Success(state->HistoryString());
170  }},
171  {"is_terminal", [&state](const Args& args) {
172  return Success(state->IsTerminal() ? "true" : "false");
173  }},
174  {"current_player", [&state](const Args& args) {
175  return Success(absl::StrCat(state->CurrentPlayer()));
176  }},
177  {"returns", [&state](const Args& args) {
178  return Success(absl::StrJoin(state->Returns(), " "));
179  }},
180  {"legal_actions", [&state](const Args& args) {
181  std::vector<std::string> actions;
182  std::vector<open_spiel::Action> legal_actions = state->LegalActions();
183  actions.reserve(legal_actions.size());
184  for (const open_spiel::Action action : legal_actions) {
185  actions.push_back(state->ActionToString(action));
186  }
187  return Success(absl::StrJoin(actions, " "));
188  }},
189  };
190 
191  std::cerr << "Welcome to OpenSpiel GTP interface. Try `list_commands`."
192  << std::endl << std::endl;
193  for (std::string line; std::getline(std::cin, line);) {
194  std::vector<std::string> parts = absl::StrSplit(line, ' ');
195  if (parts.empty()) continue;
196  std::string& cmd = parts[0];
197 
198  auto cmd_it = cmds.find(cmd);
199  if (cmd_it == cmds.end()) {
200  std::cout << Failure("unknown command");
201  continue;
202  }
203 
204  Args args(parts.begin() + 1, parts.end());
205  std::cout << cmd_it->second(args);
206  if (cmd == "quit") {
207  break;
208  }
209  }
210  return 0;
211 }
