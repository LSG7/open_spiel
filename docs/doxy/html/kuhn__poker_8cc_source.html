72 void WriteTensor(const State& observed_state, int player,
73 Allocator* allocator) const override {
74 const KuhnState& state =
75 open_spiel::down_cast<const KuhnState&>(observed_state);
76 SPIEL_CHECK_GE(player, 0);
77 SPIEL_CHECK_LT(player, state.num_players_);
78 const int num_players = state.num_players_;
79 const int num_cards = num_players + 1;
80
81 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
82 { // Observing player.
83 auto out = allocator->Get("player", {num_players});
84 out.at(player) = 1;
85 }
86 { // The player's card, if one has been dealt.
87 auto out = allocator->Get("private_card", {num_cards});
88 if (state.history_.size() > player)
89 out.at(state.history_[player].action) = 1;
90 }
91 }
92
93 // Betting sequence.
94 if (iig_obs_type_.public_info) {
95 if (iig_obs_type_.perfect_recall) {
96 auto out = allocator->Get("betting", {2 * num_players - 1, 2});
97 for (int i = num_players; i < state.history_.size(); ++i) {
98 out.at(i - num_players, state.history_[i].action) = 1;
99 }
100 } else {
101 auto out = allocator->Get("pot_contribution", {num_players});
102 for (auto p = Player{0}; p < state.num_players_; p++) {
103 out.at(p) = state.ante_[p];
104 }
105 }
106 }
107 }

109 std::string StringFrom(const State& observed_state,
110 int player) const override {
111 const KuhnState& state =
112 open_spiel::down_cast<const KuhnState&>(observed_state);
113 SPIEL_CHECK_GE(player, 0);
114 SPIEL_CHECK_LT(player, state.num_players_);
115 std::string result;
116
117 // Private card
118 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
119 if (iig_obs_type_.perfect_recall || iig_obs_type_.public_info) {
120 if (state.history_.size() > player) {
121 absl::StrAppend(&result, state.history_[player].action);
122 }
123 } else {
124 if (state.history_.size() == 1 + player) {
125 absl::StrAppend(&result, "Received card ",
126 state.history_[player].action);
127 }
128 }
129 }
130
131 // Betting.
132 // TODO(author11) Make this more self-consistent.
133 if (iig_obs_type_.public_info) {
134 if (iig_obs_type_.perfect_recall) {
135 // Perfect recall public info.
136 for (int i = state.num_players_; i < state.history_.size(); ++i)
137 result.push_back(state.history_[i].action ? 'b' : 'p');
138 } else {
139 // Imperfect recall public info - two different formats.
140 if (iig_obs_type_.private_info == PrivateInfoType::kNone) {
141 if (state.history_.empty()) {
142 absl::StrAppend(&result, "start game");
143 } else if (state.history_.size() > state.num_players_) {
144 absl::StrAppend(&result,
145 state.history_.back().action ? "Bet" : "Pass");
146 }
147 } else {
148 if (state.history_.size() > player) {
149 for (auto p = Player{0}; p < state.num_players_; p++) {
150 absl::StrAppend(&result, state.ante_[p]);
151 }
152 }
153 }
154 }
155 }
156
157 // Fact that we're dealing a card.
158 if (iig_obs_type_.public_info &&
159 iig_obs_type_.private_info == PrivateInfoType::kNone &&
160 !state.history_.empty() &&
161 state.history_.size() <= state.num_players_) {
162 int currently_dealing_to_player = state.history_.size() - 1;
163 absl::StrAppend(&result, "Deal to player ", currently_dealing_to_player);
164 }
165 return result;
166 }
167
168 private:
169 IIGObservationType iig_obs_type_;
170};

190void KuhnState::DoApplyAction(Action move) {
191 // Additional book-keeping
192 if (history_.size() < num_players_) {
193 // Give card `move` to player `history_.size()` (CurrentPlayer will return
194 // kChancePlayerId, so we use that instead).
195 card_dealt_[move] = history_.size();
196 } else if (move == ActionType::kBet) {
197 if (first_bettor_ == kInvalidPlayer) first_bettor_ = CurrentPlayer();
198 pot_ += 1;
199 ante_[CurrentPlayer()] += kAnte;
200 }
201
202 // We undo that before exiting the method.
203 // This is used in `DidBet`.
204 history_.push_back({CurrentPlayer(), move});
205
206 // Check for the game being over.
207 const int num_actions = history_.size() - num_players_;
208 if (first_bettor_ == kInvalidPlayer && num_actions == num_players_) {
209 // Nobody bet; the winner is the person with the highest card dealt,
210 // which is either the highest or the next-highest card.
211 // Losers lose 1, winner wins 1 * (num_players - 1)
212 winner_ = card_dealt_[num_players_];
213 if (winner_ == kInvalidPlayer) winner_ = card_dealt_[num_players_ - 1];
214 } else if (first_bettor_ != kInvalidPlayer &&
215 num_actions == num_players_ + first_bettor_) {
216 // There was betting; so the winner is the person with the highest card
217 // who stayed in the hand.
218 // Check players in turn starting with the highest card.
219 for (int card = num_players_; card >= 0; --card) {
220 const Player player = card_dealt_[card];
221 if (player != kInvalidPlayer && DidBet(player)) {
222 winner_ = player;
223 break;
224 }
225 }
226 SPIEL_CHECK_NE(winner_, kInvalidPlayer);
227 }
228 history_.pop_back();
229}

231std::vector<Action> KuhnState::LegalActions() const {
232 if (IsTerminal()) return {};
233 if (IsChanceNode()) {
234 std::vector<Action> actions;
235 for (int card = 0; card < card_dealt_.size(); ++card) {
236 if (card_dealt_[card] == kInvalidPlayer) actions.push_back(card);
237 }
238 return actions;
239 } else {
240 return {ActionType::kPass, ActionType::kBet};
241 }

253std::string KuhnState::ToString() const {
254 // The deal: space separated card per player
255 std::string str;
256 for (int i = 0; i < history_.size() && i < num_players_; ++i) {
257 if (!str.empty()) str.push_back(' ');
258 absl::StrAppend(&str, history_[i].action);
259 }
260
261 // The betting history: p for Pass, b for Bet
262 if (history_.size() > num_players_) str.push_back(' ');
263 for (int i = num_players_; i < history_.size(); ++i) {
264 str.push_back(history_[i].action ? 'b' : 'p');
265 }
266
267 return str;
268}

272std::vector<double> KuhnState::Returns() const {
273 if (!IsTerminal()) {
274 return std::vector<double>(num_players_, 0.0);
275 }
276
277 std::vector<double> returns(num_players_);
278 for (auto player = Player{0}; player < num_players_; ++player) {
279 const int bet = DidBet(player) ? 2 : 1;
280 returns[player] = (player == winner_) ? (pot_ - bet) : -bet;
281 }
282 return returns;
283}

313void KuhnState::UndoAction(Player player, Action move) {
314 if (history_.size() <= num_players_) {
315 // Undoing a deal move.
316 card_dealt_[move] = kInvalidPlayer;
317 } else {
318 // Undoing a bet / pass.
319 if (move == ActionType::kBet) {
320 pot_ -= 1;
321 if (player == first_bettor_) first_bettor_ = kInvalidPlayer;
322 }
323 winner_ = kInvalidPlayer;
324 }
325 history_.pop_back();
326 --move_number_;
327}

329std::vector<std::pair<Action, double>> KuhnState::ChanceOutcomes() const {
330 SPIEL_CHECK_TRUE(IsChanceNode());
331 std::vector<std::pair<Action, double>> outcomes;
332 const double p = 1.0 / (num_players_ + 1 - history_.size());
333 for (int card = 0; card < card_dealt_.size(); ++card) {
334 if (card_dealt_[card] == kInvalidPlayer) outcomes.push_back({card, p});
335 }
336 return outcomes;
337}

339bool KuhnState::DidBet(Player player) const {
340 if (first_bettor_ == kInvalidPlayer) {
341 return false;
342 } else if (player == first_bettor_) {
343 return true;
344 } else if (player > first_bettor_) {
345 return history_[num_players_ + player].action == ActionType::kBet;
346 } else {
347 return history_[num_players_ * 2 + player].action == ActionType::kBet;
348 }
349}

351std::unique_ptr<State> KuhnState::ResampleFromInfostate(
352 int player_id, std::function<double()> rng) const {
353 std::unique_ptr<State> state = game_->NewInitialState();
354 Action player_chance = history_.at(player_id).action;
355 for (int p = 0; p < game_->NumPlayers(); ++p) {
356 if (p == history_.size()) return state;
357 if (p == player_id) {
358 state->ApplyAction(player_chance);
359 } else {
360 Action other_chance = player_chance;
361 while (other_chance == player_chance) {
362 other_chance = SampleAction(state->ChanceOutcomes(), rng()).first;
363 }
364 state->ApplyAction(other_chance);
365 }
366 }
367 SPIEL_CHECK_GE(state->CurrentPlayer(), 0);
368 if (game_->NumPlayers() == history_.size()) return state;
369 for (int i = game_->NumPlayers(); i < history_.size(); ++i) {
370 state->ApplyAction(history_.at(i).action);
371 }
372 return state;
373}

395std::vector<int> KuhnGame::InformationStateTensorShape() const {
396 // One-hot for whose turn it is.
397 // One-hot encoding for the single private card. (n+1 cards = n+1 bits)
398 // Followed by 2 (n - 1 + n) bits for betting sequence (longest sequence:
399 // everyone except one player can pass and then everyone can bet/pass).
400 // n + n + 1 + 2 (n-1 + n) = 6n - 1.
401 return {6 * num_players_ - 1};
402}
403
404std::vector<int> KuhnGame::ObservationTensorShape() const {
405 // One-hot for whose turn it is.
406 // One-hot encoding for the single private card. (n+1 cards = n+1 bits)
407 // Followed by the contribution of each player to the pot (n).
408 // n + n + 1 + n = 3n + 1.
409 return {3 * num_players_ + 1};
410}

412double KuhnGame::MaxUtility() const {
413 // In poker, the utility is defined as the money a player has at the end
414 // of the game minus then money the player had before starting the game.
415 // Everyone puts a chip in at the start, and then they each have one more
416 // chip. Most that a player can gain is (#opponents)*2.
417 return (num_players_ - 1) * 2;
418}
419
420double KuhnGame::MinUtility() const {
421 // In poker, the utility is defined as the money a player has at the end
422 // of the game minus then money the player had before starting the game.
423 // In Kuhn, the most any one player can lose is the single chip they paid
424 // to play and the single chip they paid to raise/call.
425 return -2;
426}

451TabularPolicy GetOptimalPolicy(double alpha) {
452 SPIEL_CHECK_GE(alpha, 0.);
453 SPIEL_CHECK_LE(alpha, 1. / 3);
454 const double three_alpha = 3 * alpha;
455 std::unordered_map<std::string, ActionsAndProbs> policy;
456
457 // All infostates have two actions: Pass (0) and Bet (1).
458 // Player 0
459 policy["0"] = {{0, 1 - alpha}, {1, alpha}};
460 policy["0pb"] = {{0, 1}, {1, 0}};
461 policy["1"] = {{0, 1}, {1, 0}};
462 policy["1pb"] = {{0, 2. / 3. - alpha}, {1, 1. / 3. + alpha}};
463 policy["2"] = {{0, 1 - three_alpha}, {1, three_alpha}};
464 policy["2pb"] = {{0, 0}, {1, 1}};
465
466 // Player 1
467 policy["0p"] = {{0, 2. / 3.}, {1, 1. / 3.}};
468 policy["0b"] = {{0, 1}, {1, 0}};
469 policy["1p"] = {{0, 1}, {1, 0}};
470 policy["1b"] = {{0, 2. / 3.}, {1, 1. / 3.}};
471 policy["2p"] = {{0, 0}, {1, 1}};
472 policy["2b"] = {{0, 0}, {1, 1}};
473 return TabularPolicy(policy);
474}
