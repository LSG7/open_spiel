35 TEST(CrcCordState, Normalize) {
36  absl::crc_internal::CrcCordState state;
37  auto* rep = state.mutable_rep();
38  rep->prefix_crc.push_back(
39  absl::crc_internal::CrcCordState::PrefixCrc(1000, absl::crc32c_t{1000}));
40  rep->prefix_crc.push_back(
41  absl::crc_internal::CrcCordState::PrefixCrc(2000, absl::crc32c_t{2000}));
42  rep->removed_prefix =
43  absl::crc_internal::CrcCordState::PrefixCrc(500, absl::crc32c_t{500});
44 
45  // The removed_prefix means state is not normalized.
46  EXPECT_FALSE(state.IsNormalized());
47 
48  absl::crc32c_t crc = state.Checksum();
49  state.Normalize();
50  EXPECT_TRUE(state.IsNormalized());
51 
52  // The checksum should not change as a result of calling Normalize().
53  EXPECT_EQ(state.Checksum(), crc);
54  EXPECT_EQ(rep->removed_prefix.length, 0);
55 }
