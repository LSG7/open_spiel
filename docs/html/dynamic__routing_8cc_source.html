77 MeanFieldRoutingGame::MeanFieldRoutingGame(const GameParameters& params)
78  : Game(kGameType, params) {
79  // Maps data name from string to the enum.
80  const absl::flat_hash_map<std::string, DynamicRoutingDataName>
81  data_name_string_to_enum = {{"line", DynamicRoutingDataName::kLine},
82  {"braess", DynamicRoutingDataName::kBraess}};
83 
84  int max_num_time_step =
85  ParameterValue<int>("max_num_time_step", kDefaultMaxTimeStep);
86  SPIEL_CHECK_NE(max_num_time_step, 0);
87  time_step_length_ =
88  ParameterValue<double>("time_step_length", kDefaultTimeStepLength);
89  network_name_ = ParameterValue<std::string>("network", kDefaultNetworkName);
90  SPIEL_CHECK_NE(network_name_, "");
91  perform_sanity_checks_ = ParameterValue<bool>("perform_sanity_checks", true);
92  std::unique_ptr<DynamicRoutingData> data =
93  DynamicRoutingData::Create(data_name_string_to_enum.at(network_name_));
94  network_ = std::move(data->network_);
95  od_demand_ = std::move(data->od_demand_);
96  network_->CheckListOfOdDemandIsCorrect(od_demand_.get());
97  game_info_.num_distinct_actions = network_->num_actions();
98  game_info_.max_chance_outcomes = static_cast<int>(od_demand_->size());
99  game_info_.num_players = kNumPlayers;
100  game_info_.min_utility = static_cast<double>(-max_num_time_step - 1);
101  game_info_.max_utility = 0;
102  game_info_.max_game_length = max_num_time_step;
103 }

138 std::unique_ptr<MeanFieldRoutingGameState> MeanFieldRoutingGameState::Create(
139  std::shared_ptr<const Game> game, double time_step_length,
140  std::vector<OriginDestinationDemand>* od_demand, Network* network,
141  bool perform_sanity_checks, int current_time_step,
142  open_spiel::PlayerId player_id, bool is_chance_init, bool is_terminal,
143  bool vehicle_at_destination, bool vehicle_without_legal_action,
144  int waiting_time, double vehicle_final_travel_time,
145  std::string vehicle_location, std::string vehicle_destination) {
146  double total_num_vehicle = 0;
147  for (const OriginDestinationDemand& od_demand_item : *od_demand) {
148  total_num_vehicle += od_demand_item.counts;
149  }
150  int i = 0;
151  ActionsAndProbs chance_outcomes;
152  for (const auto& od_demand_item : *od_demand) {
153  chance_outcomes.emplace_back(
154  std::pair(i++, od_demand_item.counts / total_num_vehicle));
155  }
156  return absl::WrapUnique<MeanFieldRoutingGameState>(
157  new MeanFieldRoutingGameState(
158  game, time_step_length, od_demand, network, perform_sanity_checks,
159  current_time_step, player_id, is_chance_init, is_terminal,
160  vehicle_at_destination, vehicle_without_legal_action, waiting_time,
161  vehicle_final_travel_time, vehicle_location, vehicle_destination,
162  total_num_vehicle, chance_outcomes));
163 }

212 std::string MeanFieldRoutingGameState::StateToString(
213  std::string location, int time_step, int player_id, int waiting_time,
214  std::string destination, double ret) const {
215  std::string time;
216  if (destination.empty()) {
217  destination = vehicle_destination_;
218  }
219  if (is_chance_init_) {
220  return "initial chance node";
221  }
222  if (player_id == PlayerId::kDefaultPlayerId ||
223  player_id == PlayerId::kTerminalPlayerId) {
224  time = absl::StrCat(time_step);
225  } else if (player_id == PlayerId::kMeanFieldPlayerId) {
226  time = absl::StrFormat("%d_mean_field", time_step);
227  } else if (player_id == PlayerId::kChancePlayerId) {
228  time = absl::StrFormat("%d_chance", time_step);
229  } else {
230  SpielFatalError(
231  "Player id should be DEFAULT_PLAYER_ID, MEAN_FIELD or CHANCE");
232  }
233  if (vehicle_final_travel_time_ != 0.0) {
234  return absl::StrFormat("Arrived at %s, with arrival time %.2f, t=%s",
235  location, vehicle_final_travel_time_, time);
236  }
237  return absl::StrFormat("Location=%s, waiting time=%d, t=%s, destination=%s",
238  location, waiting_time, time, destination);
239 }

241 std::vector<Action> MeanFieldRoutingGameState::LegalActions() const {
242  if (is_terminal_) {
243  return {};
244  }
245  SPIEL_CHECK_NE(CurrentPlayer(), kMeanFieldPlayerId);
246  if (CurrentPlayer() == kChancePlayerId) {
247  return LegalChanceOutcomes();
248  }
249  if (perform_sanity_checks_) {
250  SPIEL_CHECK_EQ(CurrentPlayer(), kDefaultPlayerId);
251  }
252  if (waiting_time_ > 0) {
253  return {kNoPossibleAction};
254  }
255  if (vehicle_without_legal_action_) {
256  return {kNoPossibleAction};
257  }
258  std::string end_section_node = NodesFromRoadSection(vehicle_location_)[1];
259  std::vector<std::string> successors =
260  network_->GetSuccessors(end_section_node);
261  if (perform_sanity_checks_) {
262  SPIEL_CHECK_TRUE(!successors.empty());
263  }
264  std::vector<Action> actions;
265  for (const auto& d : successors) {
266  Action action = network_->GetActionIdFromMovement(end_section_node, d);
267  network_->AssertValidAction(action);
268  actions.push_back(action);
269  }
270  std::sort(actions.begin(), actions.end());
271  return actions;
272 }

274 void MeanFieldRoutingGameState::DoApplyAction(Action action) {
275  if (perform_sanity_checks_) {
276  SPIEL_CHECK_TRUE(!IsTerminal());
277  SPIEL_CHECK_NE(current_player_id_, PlayerId::kMeanFieldPlayerId);
278  }
279  switch (current_player_id_) {
280  case PlayerId::kChancePlayerId: {
281  current_player_id_ = PlayerId::kDefaultPlayerId;
282  SPIEL_CHECK_EQ(is_chance_init_, true);
283  auto od_demand = od_demand_->at(action);
284  vehicle_destination_ = od_demand.vehicle.destination;
285  vehicle_location_ = od_demand.vehicle.origin;
286  waiting_time_ = static_cast<int>(od_demand.vehicle.departure_time /
287  time_step_length_);
288  is_chance_init_ = false;
289  break;
290  }
291  case PlayerId::kDefaultPlayerId: {
292  current_player_id_ = PlayerId::kMeanFieldPlayerId;
293  if (!vehicle_without_legal_action_) {
294  if (waiting_time_ > 0) {
295  waiting_time_ -= 1;
296  } else {
297  if (perform_sanity_checks_) {
298  network_->AssertValidAction(action, vehicle_location_);
299  }
300  vehicle_location_ = network_->GetRoadSectionFromActionId(action);
301  if (vehicle_location_ == vehicle_destination_) {
302  vehicle_final_travel_time_ = current_time_step_;
303  vehicle_at_destination_ = true;
304  vehicle_without_legal_action_ = true;
305  } else if (network_->IsLocationASinkNode(vehicle_location_)) {
306  vehicle_without_legal_action_ = true;
307  vehicle_final_travel_time_ = -1 * GetGame()->MinUtility();
308  } else {
309  waiting_time_ = kWaitingTimeNotAssigned;
310  }
311  }
312  }
313  current_time_step_ += 1;
314  break;
315  }
316  default:
317  SpielFatalError(absl::StrCat("Unsupported Player ID in DoApplyAction(): ",
318  current_player_id_));
319  }
320 
321  if (current_time_step_ >= GetGame()->MaxGameLength()) {
322  is_terminal_ = true;
323  current_player_id_ = PlayerId::kTerminalPlayerId;
324  if (!vehicle_at_destination_) {
325  vehicle_final_travel_time_ = -1 * GetGame()->MinUtility();
326  }
327  }
328 }

330 std::string MeanFieldRoutingGameState::ActionToString(Player player,
331  Action action) const {
332  SPIEL_CHECK_NE(player, PlayerId::kMeanFieldPlayerId);
333  if (player == PlayerId::kChancePlayerId) {
334  SPIEL_CHECK_TRUE(is_chance_init_);
335  return absl::StrFormat("Vehicle is assigned to population %d", action);
336  }
337  if (perform_sanity_checks_) {
338  SPIEL_CHECK_EQ(player, kDefaultPlayerId);
339  }
340 
341  if (action == kNoPossibleAction) {
342  return absl::StrFormat("Vehicle %d reach a sink node or its destination.",
343  player);
344  }
345  if (perform_sanity_checks_) {
346  network_->AssertValidAction(action);
347  }
348  return absl::StrFormat("Vehicle %d would like to move to %s.", player,
349  network_->GetRoadSectionFromActionId(action));
350 }

382 std::vector<std::string> MeanFieldRoutingGameState::DistributionSupport() {
383  if (vehicle_without_legal_action_) {
384  return {};
385  }
386  std::vector<std::string> dist;
387  for (int waiting_time = kWaitingTimeNotAssigned;
388  waiting_time < max_travel_time_; waiting_time++) {
389  for (const auto& od : *(od_demand_)) {
390  std::string destination = od.vehicle.destination;
391  std::string value =
392  StateToString(vehicle_location_, current_time_step_,
393  PlayerId::kMeanFieldPlayerId, waiting_time, destination,
394  /*ret = */ 0.0);
395  dist.push_back(value);
396  }
397  }
398  absl::btree_set<std::string> dist_set(dist.begin(), dist.end());
399  SPIEL_CHECK_EQ(dist_set.size(), dist.size());
400  return dist;
401 }

403 void MeanFieldRoutingGameState::UpdateDistribution(
404  const std::vector<double>& distribution) {
405  if (current_player_id_ == PlayerId::kTerminalPlayerId) {
406  return;
407  }
408  if (perform_sanity_checks_) {
409  SPIEL_CHECK_EQ(current_player_id_, PlayerId::kMeanFieldPlayerId);
410  }
411  current_player_id_ = PlayerId::kDefaultPlayerId;
412 
413  if (!vehicle_without_legal_action_) {
414  double normed_density_on_vehicle_link = 0;
415  for (const double& d : distribution) {
416  normed_density_on_vehicle_link += d;
417  }
418  if (perform_sanity_checks_) {
419  SPIEL_CHECK_GE(normed_density_on_vehicle_link, 0);
420  SPIEL_CHECK_LE(normed_density_on_vehicle_link, 1 + kEpsilon);
421  }
422  if (waiting_time_ == kWaitingTimeNotAssigned) {
423  double volume = total_num_vehicle_ * normed_density_on_vehicle_link;
424  waiting_time_ =
425  static_cast<int>(network_->GetTravelTime(vehicle_location_, volume) /
426  time_step_length_) -
427  1;
428  waiting_time_ = std::max(0, waiting_time_);
429  }
430  }
431 }
