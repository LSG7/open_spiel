14"""Boltzmann Q learning agent.
15
16This algorithm is a variation of Q learning that uses action selection
17based on boltzmann probability interpretation of Q-values.
18
19For more details, see equation (2) page 2 in
20 https://arxiv.org/pdf/1109.1528.pdf
21"""
22
23import numpy as np

25from open_spiel.python import rl_tools
26from open_spiel.python.algorithms import tabular_qlearner

29class BoltzmannQLearner(tabular_qlearner.QLearner):
30 """Tabular Boltzmann Q-Learning agent.
31
32 See open_spiel/python/examples/tic_tac_toe_qlearner.py for an usage example.
33
34 The tic_tac_toe example uses the standard Qlearner. Using the
35 BoltzmannQlearner is
36 identical and only differs in the initialization of the agents.
37 """
38
39 def __init__(self,
40 player_id,
41 num_actions,
42 step_size=0.1,
43 discount_factor=1.0,
44 temperature_schedule=rl_tools.ConstantSchedule(.5),
45 centralized=False):
46 super().__init__(
47 player_id,
48 num_actions,
49 step_size=step_size,
50 discount_factor=discount_factor,
51 epsilon_schedule=temperature_schedule,

54 def _softmax(self, info_state, legal_actions, temperature):
55 """Action selection based on boltzmann probability interpretation of Q-values.
56
57 For more details, see equation (2) page 2 in
58 https://arxiv.org/pdf/1109.1528.pdf
59
60 Args:
61 info_state: hashable representation of the information state.
62 legal_actions: list of actions at `info_state`.
63 temperature: temperature used for softmax.
64
65 Returns:
66 A valid soft-max selected action and valid action probabilities.
67 """
68 probs = np.zeros(self._num_actions_num_actions)
69
70 if temperature > 0.0:
71 probs += [
72 np.exp((1 / temperature) * self._q_values[info_state][i])
73 for i in range(self._num_actions_num_actions)
74 ]
75 probs /= np.sum(probs)
76 else:
77 # Temperature = 0 causes normal greedy action selection
78 greedy_q = max([self._q_values[info_state][a] for a in legal_actions])
79 greedy_actions = [
80 a for a in legal_actions if self._q_values[info_state][a] == greedy_q
81 ]
82
83 probs[greedy_actions] += 1 / len(greedy_actions)
84
85 action = np.random.choice(range(self._num_actions_num_actions), p=probs)
86 return action, probs
87
88 def _get_action_probs(self, info_state, legal_actions, epsilon):
89 """Returns a selected action and the probabilities of legal actions."""
90 return self._softmax(info_state, legal_actions, temperature=epsilon)
