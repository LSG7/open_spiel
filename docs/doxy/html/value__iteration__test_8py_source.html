30 initial_state = "...\n...\n..."

37 def test_solve_small_goofspiel(self):
38 # TODO(author5): This test fails with num_cards = 4 with a new version of
39 # LAPACK (3.10.0), which is used by cvxopt. Might be a bug or bad assumption
40 # about the handling of numerical error. Look into this.
41 game = pyspiel.load_game("goofspiel", {"num_cards": 3})
42 values = value_iteration.value_iteration(
43 game, depth_limit=-1, threshold=1e-6)
44
45 initial_state = game.new_initial_state()
46 assert initial_state.is_chance_node()
47 root_value = 0
48 for action, action_prob in initial_state.chance_outcomes():
49 next_state = initial_state.child(action)
50 root_value += action_prob * values[str(next_state)]
51
52 # Symmetric game: value is 0
53 self.assertAlmostEqual(root_value, 0)
