91class MFGPeriodicAversionGame(pyspiel.Game):
92 """A Mean-Field Game on periodic domain with crowd aversion cost.
93
94 A game starts by an initial chance node that select the initial state
95 of the player in the MFG.
96 Then the game sequentially alternates between:
97 - An action selection node (where the player id is >= 0)
98 - A chance node (the player id is pyspiel.PlayerId.CHANCE)
99 - A Mean Field node (the player id is pyspiel.PlayerId.MEAN_FIELD)
100 """
101
102 # pylint:disable=dangerous-default-value
103 def __init__(self, params: Mapping[str, Any] = _DEFAULT_PARAMS):
104 self.size = params.get("size", _SIZE) # number of states
105 self.horizon = params.get("horizon", _HORIZON) # number of time steps
106 self.dt = params.get("dt", _DELTA_T) # size of one step in time
107 self.xmin = params.get("xmin", _X_MIN) # smallest position
108 self.xmax = params.get("xmax", _X_MAX) # largest position
109 self.dx = (self.xmax - self.xmin) / (
110 self.size - 1
111 ) # size of one step in space
112 self.n_actions_per_side = params.get(
113 "n_actions_per_side", _N_ACTIONS_PER_SIDE
114 ) # number of actions on each side, for both players and noise
115 self.volatility = params.get("volatility", _VOLATILITY)
116 self.coef_aversion = params.get("coef_aversion", _COEF_AVERSION)
117
118 game_info = pyspiel.GameInfo(
119 num_distinct_actions=2 * self.n_actions_per_side + 1,
120 max_chance_outcomes=2 * self.n_actions_per_side + 1,
121 num_players=_NUM_PLAYERS,
122 min_utility=-np.inf,
123 max_utility=+np.inf,
124 utility_sum=0.0,
125 max_game_length=self.horizon,
126 )
127 super().__init__(_GAME_TYPE, game_info, params)
128
129 def new_initial_state(self):
130 """Returns a state corresponding to the start of a game."""
131 return MFGPeriodicAversionState(self)

133 def make_py_observer(self, iig_obs_type=None, params=None):
134 """Returns an object used for observing game state."""
135 if (iig_obs_type is None) or (
136 iig_obs_type.public_info and not iig_obs_type.perfect_recall

138 return Observer(params, self)
139 return observation.IIGObserverForPublicInfoGame(iig_obs_type, params)

141 def max_chance_nodes_in_history(self):
142 """Maximun chance nodes in game history."""
143 return self.horizon + 1

146class MFGPeriodicAversionState(pyspiel.State):
147 """A Mean Field Normal-Form state.
148
149 In this class, x and action are integers. They are converted, when needed, to
150 spatial variables by using a scaling factor representing the size of a step in
151 space and by shifting them depending on the minimal allowed value.
152 """
153
154 def __init__(self, game):
155 """Constructor; should only be called by Game.new_initial_state."""
156 super().__init__(game)
157 # Initial state where the initial position is chosen according to
158 # an initial distribution.
159 self._player_id = pyspiel.PlayerId.CHANCE
160
161 self._last_action = game.n_actions_per_side # neutral action
162 self.tick = 0
163 self.x = None
164 self.return_value = 0.0
165
166 self.game = game
167
168 self.size = game.size
169 self.horizon = game.horizon
170 self.dt = game.dt
171 self.xmin = game.xmin
172 self.xmax = game.xmax
173 self.dx = game.dx
174 self.da = game.dx
175 self.n_actions_per_side = game.n_actions_per_side
176 self.volatility = game.volatility
177 self.coef_aversion = game.coef_aversion
178
179 # Represents the current probability distribution over game states.
180 # Initialized with a uniform distribution.
181 self._distribution = [1.0 / self.size for _ in range(self.size)]
182
183 def to_string(self):
184 return self.state_to_str(self.x, self.tick)
185
186 def state_to_str(self, x, tick, player_id=pyspiel.PlayerId.DEFAULT_PLAYER_ID):
187 """A string that uniquely identify a triplet x, t, player_id."""
188 if self.x is None:
189 return "initial"
190 if self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
191 return "({}, {})".format(x, tick)
192 elif self._player_id == pyspiel.PlayerId.MEAN_FIELD:
193 return "({}, {})_a".format(x, tick)
194 elif self._player_id == pyspiel.PlayerId.CHANCE:
195 return "({}, {})_a_mu".format(x, tick)
196 raise ValueError(
197 "player_id is not mean field, chance or default player id."
198 )
199
200 # OpenSpiel (PySpiel) API functions are below. This is the standard set that
201 # should be implemented by every perfect-information sequential-move game.
202
203 @property
204 def n_actions(self):
205 return 2 * self.n_actions_per_side + 1
206
207 def _legal_actions(self, player):
208 """Returns a list of legal actions for player and MFG nodes."""
209 if player == pyspiel.PlayerId.MEAN_FIELD:
210 return []
211 if (
212 player == pyspiel.PlayerId.DEFAULT_PLAYER_ID
213 and player == self.current_player()
214 ):
215 return list(range(self.n_actionsn_actions))
216 raise ValueError(
217 f"Unexpected player {player}. "
218 "Expected a mean field or current player 0."
219 )
220
221 def _apply_action(self, action):
222 """Applies the specified action to the state."""
223 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
224 raise ValueError(
225 "_apply_action should not be called at a MEAN_FIELD state."
226 )
227 self.return_value = self._rewards()
228
229 assert (
230 self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID
231 or self._player_id == pyspiel.PlayerId.CHANCE
232 )
233
234 if self.x is None:
235 self.x = action
236 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
237 return
238
239 if action < 0 or action >= self.n_actionsn_actions:
240 raise ValueError(
241 "The action is between 0 and {} at any node".format(self.n_actionsn_actions)
242 )
243
244 self.x = (self.x + action - self.n_actions_per_side) % self.size
245 if self._player_id == pyspiel.PlayerId.CHANCE:
246 self._player_id = pyspiel.PlayerId.MEAN_FIELD
247 self.tick += 1
248 elif self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
249 self._last_action = action
250 self._player_id = pyspiel.PlayerId.CHANCE
251
252 def _action_to_string(self, player, action):
253 """Action -> string."""
254 del player
255 return str(action - self.n_actions_per_side)
256
257 def action_to_move(self, action):
258 return (action - self.n_actions_per_side) * self.da
259
260 def state_to_position(self, state):
261 return state * self.dx + self.xmin
262
263 def position_to_state(self, position):
264 return round((position - self.xmin) / self.dx)
265
266 def chance_outcomes(self):
267 """Returns the possible chance outcomes and their probabilities."""
268 if self.x is None:
269 # Initial distribution
270 return list(enumerate(self._distribution))
271 actions = np.array(
272 [(a - self.n_actions_per_side) * self.da for a in range(self.n_actionsn_actions)]
273 )
274 stddev = self.volatility * math.sqrt(self.dt)
275 probas = scipy.stats.norm.pdf(actions, scale=stddev)
276 probas /= np.sum(probas)
277 return [(act, p) for act, p in zip(list(range(self.n_actionsn_actions)), probas)]
278
279 def distribution_support(self):
280 """return a list of state string."""
281 return [
282 self.state_to_str(i, self.tick, player_id=pyspiel.PlayerId.MEAN_FIELD)
283 for i in range(self.size)
284 ]

286 def get_state_proba(self, state: int) -> float:
287 """Gets the probability of a position in the current distrib.
288
289 Args:
290 state: state requested.
291
292 Returns:
293 The probability for the provided position.
294 """
295 assert state >= 0, state
296 assert state < self.size, state
297 # This logic needs to match the ordering defined in distribution_support().
298 index = state
299 assert 0 <= index < len(self._distribution), (
300 f"Invalid index {index} vs dist length:"
301 f" {len(self._distribution)}, state={state},"
302 f" state={self}"
303 )
304 return self._distribution[index]
305
306 def update_distribution(self, distribution):
307 """This function is central and specific to the logic of the MFG.
308
309 Args:
310 distribution: a distribution to register. - function should be called
311 when the node is in MEAN_FIELD state. - distribution are probabilities
312 that correspond to each game state given by distribution_support.
313 """
314 if self._player_id != pyspiel.PlayerId.MEAN_FIELD:
315 raise ValueError(
316 "update_distribution should only be called at a MEAN_FIELD state."
317 )
318 self._distribution = distribution.copy()
319 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
320
321 @property
322 def t(self):
323 return self.tick
324
325 def is_terminal(self):
326 """Returns True if the game is over."""
327 return self.t >= self.horizon
328
329 def current_player(self):
330 """Returns id of the next player to move, or TERMINAL if game is over."""
331 if self.is_terminal():
332 return pyspiel.PlayerId.TERMINAL
333 return self._player_id
334
335 def _rewards(self):
336 """Reward for the player for this state."""
337 if self._player_id != pyspiel.PlayerId.DEFAULT_PLAYER_ID:
338 return 0.0
339 assert self.x is not None
340 velocity = self.action_to_move(self._last_action) / self.dt
341 action_r = -0.5 * velocity**2
342 eps = 1e-15
343 mu_x = self.get_state_proba(self.x) / self.dx # represents the density
344 # The density should have an integral equal to 1; here sum_x mu_x * dx = 1
345 aversion_r = -np.log(mu_x + eps)
346 pos = self.state_to_position(self.x)
347 pix2 = 2 * np.pi * pos
348 geom_r = (
349 self.volatility * 2 * np.pi**2 * np.sin(pix2)
350 - 2 * np.pi**2 * np.cos(pix2) ** 2
351 + (2 / self.volatility**2) * np.sin(pix2)
352 )
353 return (action_r + self.coef_aversion * aversion_r + geom_r) * self.dt
354
355 def rewards(self) -> List[float]:
356 """Rewards for all players."""
357 # For now, only single-population (single-player) mean field games
358 # are supported.
359 return [self._rewards()]
360
361 def _returns(self):
362 """Returns is the sum of all payoffs collected so far."""
363 return self.return_value + self._rewards()
364
365 def returns(self) -> List[float]:
366 """Returns for all players."""
367 # For now, only single-population (single-player) mean field games
368 # are supported.
369 return [self._returns()]

371 def __str__(self):
372 """A string that uniquely identify the current state."""
373 return self.state_to_str(
374 x=self.x, tick=self.tick, player_id=self._player_id

378class Observer:
379 """Observer, conforming to the PyObserver interface (see observation.py)."""
380
381 def __init__(self, params, game):
382 """Initializes an empty observation tensor."""
383 del params
384
385 self.size = game.size
386 self.horizon = game.horizon
387 # +1 to allow t == horizon.
388 self.tensor = np.zeros(self.size + self.horizon + 1, np.float32)
389 self.dict = {"x": self.tensor[: self.size], "t": self.tensor[self.size :]}
390
391 def set_from(self, state, player: int):
392 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
393 del player
394 # We update the observation via the shaped tensor since indexing is more
395 # convenient than with the 1-D tensor. Both are views onto the same memory.
396 self.tensor.fill(0)
397 # state.x is None for the initial (blank) state, don't set any
398 # position bit in that case.
399 if state.x is not None:
400 if state.x < 0 or state.x > self.size:
401 raise ValueError(
402 f"Expected {state} positions to be in [0, {self.size})"
403 )
404 self.dict["x"][state.x] = 1
405 if not 0 <= state.tick <= self.horizon:
406 raise ValueError(f"Expected {state} time to be in [0, {self.horizon}]")
407 self.dict["t"][state.tick] = 1
408
409 def string_from(self, state, player):
410 """Observation of `state` from the PoV of `player`, as a string."""
411 del player
412 return state.to_string()
