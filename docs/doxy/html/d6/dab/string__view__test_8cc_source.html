46// A minimal allocator that uses malloc().
47template <typename T>
48struct Mallocator {
49 typedef T value_type;
50 typedef size_t size_type;
51 typedef ptrdiff_t difference_type;
52 typedef T* pointer;
53 typedef const T* const_pointer;
54 typedef T& reference;
55 typedef const T& const_reference;
56
57 size_type max_size() const {
58 return size_t(std::numeric_limits<size_type>::max()) / sizeof(value_type);
59 }
60 template <typename U>
61 struct rebind {
62 typedef Mallocator<U> other;
63 };
64 Mallocator() = default;
65 template <class U>
66 Mallocator(const Mallocator<U>&) {} // NOLINT(runtime/explicit)
67
68 T* allocate(size_t n) { return static_cast<T*>(std::malloc(n * sizeof(T))); }
69 void deallocate(T* p, size_t) { std::free(p); }
70};
71template <typename T, typename U>
72bool operator==(const Mallocator<T>&, const Mallocator<U>&) {
73 return true;
74}
75template <typename T, typename U>
76bool operator!=(const Mallocator<T>&, const Mallocator<U>&) {
77 return false;
78}
