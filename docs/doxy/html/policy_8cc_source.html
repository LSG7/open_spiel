41void SetProb(ActionsAndProbs* actions_and_probs, Action action, double prob) {
42 for (auto& iter : *actions_and_probs) {
43 if (iter.first == action) {
44 iter.second = prob;
45 return;
46 }
47 }
48 actions_and_probs->push_back({action, prob});

51double GetProb(const ActionsAndProbs& action_and_probs, Action action) {
52 auto it = absl::c_find_if(action_and_probs,
53 [&action](const std::pair<Action, double>& p) {
54 return p.first == action;

56 if (it == action_and_probs.end()) return -1.;
57 return it->second;

79bool StatePoliciesEqual(const ActionsAndProbs& state_policy1,
80 const ActionsAndProbs& state_policy2,
81 double float_tolerance) {
82 if (state_policy1.size() != state_policy2.size()) {
83 return false;
84 }
85
86 for (int i = 0; i < state_policy1.size(); ++i) {
87 if (state_policy1[i].first != state_policy2[i].first) {
88 return false;
89 }
90
91 if (!Near(state_policy1[i].second, state_policy2[i].second,
92 float_tolerance)) {
93 return false;
94 }
95 }
96
97 return true;
98}

110ActionsAndProbs UniformStatePolicy(const std::vector<Action>& actions) {

161std::unique_ptr<TabularPolicy> DeserializeTabularPolicy(
162 const std::string& serialized, std::string delimiter) {
163 // Class’s identity is the very first line, see Policy::Serialize
164 // for more info.
165 std::pair<std::string, absl::string_view> cls_and_content =
166 absl::StrSplit(serialized, absl::MaxSplits(':', 1));
167 SPIEL_CHECK_EQ(cls_and_content.first, "TabularPolicy");
168
169 std::unique_ptr<TabularPolicy> res = std::make_unique<TabularPolicy>();
170 if (cls_and_content.second.empty()) return res;
171
172 std::vector<absl::string_view> splits =
173 absl::StrSplit(cls_and_content.second, delimiter);
174
175 // Insert the actual values
176 Action action;
177 double prob;
178 for (int i = 0; i < splits.size(); i += 2) {
179 std::vector<absl::string_view> policy_values =
180 absl::StrSplit(splits.at(i + 1), ',');
181 ActionsAndProbs res_policy;
182 res_policy.reserve(policy_values.size());
183
184 for (absl::string_view policy_value : policy_values) {
185 std::pair<absl::string_view, absl::string_view> action_and_prob =
186 absl::StrSplit(policy_value, '=');
187 SPIEL_CHECK_TRUE(absl::SimpleAtoi(action_and_prob.first, &action));
188 absl::from_chars(
189 action_and_prob.second.data(),
190 action_and_prob.second.data() + action_and_prob.second.size(), prob);
191 res_policy.push_back({action, prob});
192 }
193 res->SetStatePolicy(std::string(splits.at(i)), res_policy);
194 }
195 return res;
196}

275TabularPolicy GetEmptyTabularPolicy(const Game& game,
276 bool initialize_to_uniform,
277 Player player) {
278 std::unordered_map<std::string, ActionsAndProbs> policy;
279 if (game.GetType().dynamics != GameType::Dynamics::kSequential) {
280 SpielFatalError("Game is not sequential.");
281 return TabularPolicy(policy);
282 }
283 std::list<std::unique_ptr<State>> to_visit;
284 to_visit.push_back(game.NewInitialState());
285 while (!to_visit.empty()) {
286 std::unique_ptr<State> state = std::move(to_visit.back());
287 to_visit.pop_back();
288 if (state->IsTerminal()) {
289 continue;
290 }
291 if (state->IsChanceNode()) {
292 for (const auto& outcome_and_prob : state->ChanceOutcomes()) {
293 to_visit.emplace_back(state->Child(outcome_and_prob.first));
294 }
295 } else {
296 ActionsAndProbs infostate_policy;
297 std::vector<Action> legal_actions = state->LegalActions();
298 const int num_legal_actions = legal_actions.size();
299 SPIEL_CHECK_GT(num_legal_actions, 0.);
300 for (Action action : legal_actions) {
301 to_visit.push_back(state->Child(action));
302 }
303 if (player < 0 || state->IsPlayerActing(player)) {
304 double action_probability = 1.;
305 if (initialize_to_uniform) {
306 action_probability = 1. / num_legal_actions;
307 }
308 ActionsAndProbs infostate_policy;
309 infostate_policy.reserve(num_legal_actions);
310 for (Action action : legal_actions) {
311 infostate_policy.push_back({action, action_probability});
312 }
313 if (infostate_policy.empty()) {
314 SpielFatalError("State has zero legal actions.");
315 }
316 policy.insert({state->InformationStateString(), infostate_policy});
317 }
318 }
319 }
320 return TabularPolicy(policy);
321}

327template <typename RandomNumberDistribution>
328TabularPolicy SamplePolicy(
329 const Game& game, int seed, RandomNumberDistribution& dist, Player player) {
330 std::mt19937 gen(seed);
331 TabularPolicy policy = GetEmptyTabularPolicy(game, false, player);
332 std::unordered_map<std::string, ActionsAndProbs>& policy_table =
333 policy.PolicyTable();
334 for (auto& kv : policy_table) {
335 ActionsAndProbs state_policy;
336 if (kv.second.empty()) {
337 SpielFatalError("State has zero legal actions.");
338 }
339 state_policy.reserve(kv.second.size());
340 double sum = 0;
341 double prob;
342 for (const auto& action_and_prob : kv.second) {
343 // We multiply the original probability by a random number greater than
344 // 0. We then normalize. This has the effect of randomly permuting the
345 // policy but all illegal actions still have zero probability.
346 prob = dist(gen) * action_and_prob.second;
347 sum += prob;
348 state_policy.push_back({action_and_prob.first, prob});
349 }
350 // We normalize the policy to ensure it sums to 1.
351 for (auto& action_and_prob : state_policy) {
352 action_and_prob.second /= sum;
353 }
354 // This is included as a sanity check.
355 double normalized_sum = 0;
356 for (auto& action_and_prob : state_policy) {
357 normalized_sum += action_and_prob.second;
358 }
359 SPIEL_CHECK_FLOAT_EQ(normalized_sum, 1.0);
360 kv.second = state_policy;
361 }
362 return policy;
363}

376TabularPolicy GetRandomDeterministicPolicy(
377 const Game& game, int seed, Player player) {
378 std::mt19937 gen(seed);
379 absl::node_hash_map<int, std::uniform_int_distribution<int>> dists;
380 std::unordered_map<std::string, ActionsAndProbs> policy;
381 if (game.GetType().dynamics != GameType::Dynamics::kSequential) {
382 SpielFatalError("Game is not sequential.");
383 return TabularPolicy(policy);
384 }
385 const GameType::Information information = game.GetType().information;
386 std::list<std::unique_ptr<State>> to_visit;
387 to_visit.push_back(game.NewInitialState());
388 while (!to_visit.empty()) {
389 std::unique_ptr<State> state = std::move(to_visit.back());
390 to_visit.pop_back();
391 if (state->IsTerminal()) {
392 continue;
393 } else if (state->IsChanceNode()) {
394 for (const auto& outcome_and_prob : state->ChanceOutcomes()) {
395 to_visit.emplace_back(state->Child(outcome_and_prob.first));
396 }
397 } else if (player < 0 || state->IsPlayerActing(player)) {
398 std::vector<Action> legal_actions = state->LegalActions();
399 const int num_legal_actions = legal_actions.size();
400 SPIEL_CHECK_GT(num_legal_actions, 0.);
401 if (dists.count(num_legal_actions) == 0) {
402 std::uniform_int_distribution<int> dist(0, num_legal_actions - 1);
403 dists.insert({num_legal_actions, std::move(dist)});
404 }
405 const int legal_action_index = dists[num_legal_actions](gen);
406 SPIEL_CHECK_GE(legal_action_index, 0);
407 SPIEL_CHECK_LT(legal_action_index, num_legal_actions);
408 const int action = legal_actions[legal_action_index];
409 ActionsAndProbs infostate_policy;
410 infostate_policy.reserve(1);
411 infostate_policy.push_back({action, 1.0});
412 policy.insert({state->InformationStateString(), infostate_policy});
413 if (information == GameType::Information::kPerfectInformation) {
414 to_visit.push_back(state->Child(action));
415 } else {
416 for (Action action : legal_actions) {
417 to_visit.push_back(state->Child(action));
418 }
419 }
420 } else {
421 std::vector<Action> legal_actions = state->LegalActions();
422 const int num_legal_actions = legal_actions.size();
423 SPIEL_CHECK_GT(num_legal_actions, 0.);
424 for (Action action : legal_actions) {
425 to_visit.push_back(state->Child(action));
426 }
427 }
428 }
429 return TabularPolicy(policy);
430}

432TabularPolicy GetFirstActionPolicy(const Game& game) {
433 std::unordered_map<std::string, ActionsAndProbs> policy;
434 if (game.GetType().dynamics != GameType::Dynamics::kSequential) {
435 SpielFatalError("Game is not sequential.");
436 return TabularPolicy(policy);
437 }
438 std::vector<std::unique_ptr<State>> to_visit;
439 to_visit.push_back(game.NewInitialState());
440 while (!to_visit.empty()) {
441 std::unique_ptr<State> state = std::move(to_visit.back());
442 to_visit.pop_back();
443 if (state->IsTerminal()) {
444 continue;
445 }
446 if (state->IsChanceNode()) {
447 for (const auto& outcome_and_prob : state->ChanceOutcomes()) {
448 to_visit.emplace_back(state->Child(outcome_and_prob.first));
449 }
450 } else {
451 ActionsAndProbs infostate_policy;
452 std::vector<Action> legal_actions = state->LegalActions();
453 const int num_legal_actions = legal_actions.size();
454 SPIEL_CHECK_GT(num_legal_actions, 0.);
455 bool first_legal_action_found = false;
456
457 infostate_policy.reserve(num_legal_actions);
458 for (Action action : legal_actions) {
459 to_visit.push_back(state->Child(action));
460 if (!first_legal_action_found) {
461 first_legal_action_found = true;
462 infostate_policy.push_back({action, 1.});
463
464 } else {
465 infostate_policy.push_back({action, 0.});
466 }
467 }
468 if (infostate_policy.empty()) {
469 SpielFatalError("State has zero legal actions.");
470 }
471 policy[state->InformationStateString()] = std::move(infostate_policy);
472 }
473 }
474 return TabularPolicy(policy);
475}

488TabularPolicy ToTabularPolicy(const Game& game, const Policy* policy) {
489 TabularPolicy tabular_policy;
490 std::vector<std::unique_ptr<State>> to_visit;
491 to_visit.push_back(game.NewInitialState());
492 for (int idx = 0; idx < to_visit.size(); ++idx) {
493 const State* state = to_visit[idx].get();
494 if (state->IsTerminal()) {
495 continue;
496 }
497
498 if (!state->IsChanceNode()) {
499 std::vector<Player> players(game.NumPlayers());
500 if (state->IsSimultaneousNode()) {
501 absl::c_iota(players, 0);
502 } else {
503 players = {state->CurrentPlayer()};
504 }
505
506 for (Player player : players) {
507 ActionsAndProbs state_policy = policy->GetStatePolicy(*state);
508 tabular_policy.SetStatePolicy(state->InformationStateString(player),
509 state_policy);
510 }
511 }
512
513 for (Action action : state->LegalActions()) {
514 to_visit.push_back(state->Child(action));
515 }
516 }
517 return tabular_policy;
518}
