36// For simplicity and speed, we store the board in terms of a "virtual board",
37// with a border of guard stones around all sides of the board.
38// This allows us to skip bounds checking.
39// In Virtual mode, an action (row, col) is row * 21 + col, and pass is 21*21+1.
40// All functions in this file (except stated otherwise) use these virtual
41// coordinates.
42//
43// However, in the OpenSpiel API (in go.{h, cc}), the actions are still exposed
44// as actions within 0, board_size*boardsize) (with pass = board_size **2.
45//
46// We support boards up to size 19.
47inline constexpr int kMaxBoardSize = 19;
48inline constexpr int kVirtualBoardSize = kMaxBoardSize + 2;
49inline constexpr int kVirtualBoardPoints =
50 kVirtualBoardSize * kVirtualBoardSize;
51

53
54inline constexpr VirtualPoint kInvalidPoint = 0;
55inline constexpr VirtualPoint kVirtualPass = kVirtualBoardPoints + 1;

140 // Count of pseudo-liberties, i.e. each liberty is counted between 1 and 4
141 // times, once for each stone of the group that borders it.
142 // This is much faster than realLiberty(), so prefer it if possible.
143 inline int PseudoLiberty(VirtualPoint p) const {
144 return chain(p).num_pseudo_liberties == 0
145 ? 0
146 : (chain(p).in_atari() ? 1 : chain(p).num_pseudo_liberties);
147 }

149 inline bool InAtari(VirtualPoint p) const { return chain(p).in_atari(); }
150
151 // If a chain has a single liberty (it is in Atari), return that liberty.
152 VirtualPoint SingleLiberty(VirtualPoint p) const;
153
154 // Actual liberty count, i.e. each liberty is counted exactly once.
155 // This is computed on the fly by actually walking the group and checking the
156 // neighbouring stones.
157 inline int RealLiberty(VirtualPoint p) const {
158 int num_lib = 0;
159 for (auto it = LibIter(p); it; ++it) {
160 ++num_lib;
161 }
162 return num_lib;
163 }
