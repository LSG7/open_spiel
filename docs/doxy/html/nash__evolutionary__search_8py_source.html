27def softmax(x):
28 e = np.exp(x - np.max(x))
29 return e / np.sum(e, axis=-1, keepdims=True)

32class NashCMAES(regret_minimizer.RegretMinimizer):
33 """Base class for Regret Minimizers.
34
35 Implements base functions for regret minimizers to implement.
36
37 Attributes:
38 _game: Pyspiel game.
39 _regret_steps_per_step: Number of regret steps per `step` call (Maximum
40 number in case `stop_early` is true)
41 _rho_tol: If `_compress_nus` is true, minimum probability threshold (
42 Probabilities below `rho_tol` will be filtered out).
43 _compress_nus: Whether to compress nus (Remove nus with low selection
44 probability) or not.
45 _compress_lbd: Penalty term in L1 minimization when compressing nus.
46 _stop_early: Whether to stop regret computation when average regret is lower
47 than `_stop_regret_threshold` or to keep going until
48 `_regret_steps_per_step` steps have been accomplished.
49 _stop_regret_threshold: If `stop_early` is true, average regret threshold
50 under which the algorithm will stop.
51 _policies: List of Policies
52 _value_estimator: Value estimation function.
53 _value_estimation_n: Number of runs to average _value_estimator's result on.
54 """
