69Suit GetSuit(int action) {
70 SPIEL_CHECK_GE(action, 0);
71 SPIEL_CHECK_LT(action, kNumCards);
72
73 return static_cast<Suit>(action % kNumSuits);
74}
75
76int GetRank(int action) {
77 SPIEL_CHECK_GE(action, 0);
78 SPIEL_CHECK_LT(action, kNumCards);
79
80 return action / kNumSuits;
81}

83int GetAction(Suit suit, int rank) {
84 SPIEL_CHECK_LE(rank, kNumRanks);
85 return rank * kNumSuits + static_cast<int>(suit);

88std::string GetCardStr(int action) {
89 SPIEL_CHECK_GE(action, 0);
90 SPIEL_CHECK_LT(action, kNumCards);
91 int rank = GetRank(action);
92 int suit = static_cast<int>(GetSuit(action));

105CrazyEightsState::CrazyEightsState(std::shared_ptr<const Game> game,
106 int num_players, int max_draw_cards,
107 bool use_special_cards, bool reshuffle)
108 : State(game),
109 reshuffle_(reshuffle),
110 num_players_(num_players),
111 max_draw_cards_(max_draw_cards),
112 use_special_cards_(use_special_cards) {
113 num_initial_cards_ =
114 num_players == 2 ? kNumInitialCardsForTwoPlayers : kNumInitialCards;
115 num_decks_ = num_players > 5 ? 2 : 1;
116 num_cards_left_ = num_decks_ * kNumCards;
117 absl::c_fill(dealer_deck_, num_decks_);
118 for (int i = 0; i < num_players; ++i) {
119 hands_.push_back(std::vector<int>(kNumCards, 0));
120 returns_.push_back(0);
121 }
122}

124std::string CrazyEightsState::ActionToString(Player player,
125 Action action) const {
126 if (player == kChancePlayerId) {
127 if (action < kDraw) {
128 return absl::StrFormat("Deal %s", GetCardStr(action));
129 } else if (action < kDecideDealerActionBase + num_players_) {
130 return absl::StrFormat("Decide Player %d to be the dealer",
131 action - kDecideDealerActionBase);
132 } else {
133 SpielFatalError(
134 absl::StrFormat("Non action valid Id %d for chance player", action));
135 }
136 }
137
138 if (action < kDraw) {
139 return absl::StrFormat("Play %s", GetCardStr(action));
140 } else if (action == kDraw) {
141 return "Draw";
142 } else if (action == kPass) {
143 return "Pass";
144 } else if (action < kNominateSuitActionBase + kNumSuits) {
145 return absl::StrFormat("Nominate suit %c",
146 kSuitChar[action - kNominateSuitActionBase]);
147 } else {
148 SpielFatalError(
149 absl::StrFormat("Non valid Id %d for player: %d", action, player));
150 }
151}

153std::vector<std::string> CrazyEightsState::FormatHand(Player player) const {
154 std::vector<std::string> hand_str(kNumSuits,
155 std::string(num_decks_ * kNumRanks, ' '));
156 for (int suit = 0; suit < kNumSuits; ++suit) {
157 for (int rank = 0; rank < kNumRanks; ++rank) {
158 int card = GetAction(static_cast<Suit>(suit), rank);
159 for (int i = 0; i < hands_[player][card]; ++i) {
160 hand_str[suit][rank * num_decks_ + i] = kRankChar[rank];
161 }
162 }
163 }
164 return hand_str;
165}

167std::string CrazyEightsState::FormatAllHands() const {
168 std::string hands_str;
169 std::vector<std::vector<std::string>> all_hands;
170 all_hands.reserve(num_players_);
171 for (int player = 0; player < num_players_; ++player) {
172 all_hands.push_back(FormatHand(player));
173 }
174 constexpr int kLongWidth = 40;
175
176 for (int player = 0; player < num_players_; ++player) {
177 std::string player_str = absl::StrFormat("Player %d:", player);
178 if (player != num_players_ - 1) {
179 absl::StrAppend(&player_str,
180 std::string(kLongWidth - player_str.length(), ' '));
181 } else {
182 absl::StrAppend(&player_str, "\n");
183 }
184 absl::StrAppend(&hands_str, player_str);
185 }
186
187 for (int suit = 0; suit < kNumSuits; ++suit) {
188 std::string suit_row;
189 for (int player = 0; player < num_players_; ++player) {
190 std::string player_row;
191 absl::StrAppend(&player_row,
192 absl::StrFormat("Suit %c: %s", kSuitChar[suit],
193 all_hands[player][suit]));
194 SPIEL_CHECK_GE(kLongWidth, player_row.length());
195 if (player != num_players_ - 1) {
196 absl::StrAppend(&player_row,
197 std::string(kLongWidth - player_row.length(), ' '));
198 } else {
199 absl::StrAppend(&player_row, "\n");
200 }
201 absl::StrAppend(&suit_row, player_row);
202 }
203 absl::StrAppend(&hands_str, suit_row);
204 }
205 return hands_str;
206}

208std::string CrazyEightsState::ToString() const {
209 std::string str;
210 int playing_player = dealer_;
211 for (int i = 0; i < history_.size(); ++i) {
212 if (i == 0) {
213 absl::StrAppend(
214 &str, absl::StrFormat("Player %d becomes the dealer\n", dealer_));
215 } else if (i <= num_players_ * num_initial_cards_) {
216 int player = (dealer_ + i) % num_players_;
217 absl::StrAppend(&str, absl::StrFormat("Player %d is dealt %s\n", player,
218 GetCardStr(history_[i].action)));
219 } else {
220 if (history_[i].player == kChancePlayerId) {
221 absl::StrAppend(&str,
222 absl::StrFormat("Player %d draws %s\n", playing_player,
223 GetCardStr(history_[i].action)));
224 } else if (history_[i].player != kTerminalPlayerId) {
225 playing_player = history_[i].player;
226 if (history_[i].action == kDraw) {
227 absl::StrAppend(&str, absl::StrFormat("Player %d starts drawing\n",
228 playing_player));
229 } else if (history_[i].action == kPass) {
230 absl::StrAppend(
231 &str, absl::StrFormat("Player %d passes\n", playing_player));
232 } else if (history_[i].action >= kNominateSuitActionBase &&
233 history_[i].action < kNominateSuitActionBase + kNumSuits) {
234 int suit = history_[i].action - kNominateSuitActionBase;
235 absl::StrAppend(&str,
236 absl::StrFormat("Player %d nominates suit %c\n",
237 playing_player, kSuitChar[suit]));
238 } else {
239 SPIEL_CHECK_GE(history_[i].action, 0);
240 SPIEL_CHECK_LT(history_[i].action, kNumCards);
241 absl::StrAppend(
242 &str, absl::StrFormat("Player %d plays %s\n", playing_player,
243 GetCardStr(history_[i].action)));
244 }
245 } else {
246 absl::StrAppend(&str, "Final scores\n");
247 for (int player = 0; player < num_players_; ++player) {
248 absl::StrAppend(&str, absl::StrFormat("Player %d gets score %f\n",
249 player, returns_[player]));
250 }
251 }
252 }
253 }
254 if (last_card_ != kInvalidAction) {
255 absl::StrAppend(&str,
256 absl::StrFormat("Last card: %s\n", GetCardStr(last_card_)));
257 absl::StrAppend(&str,
258 absl::StrFormat("Last suit: %c\n", kSuitChar[last_suit_]));
259 }
260 absl::StrAppend(&str, absl::StrFormat("Number of cards left in deck: %d\n",
261 num_cards_left_));
262 absl::StrAppend(&str, FormatAllHands());
263 return str;
264}

266std::string CrazyEightsState::ObservationString(Player player) const {
267 SPIEL_CHECK_GE(player, 0);
268 SPIEL_CHECK_LT(player, num_players_);
269 std::string str;
270 if (phase_ == Phase::kDeal) return str;
271 absl::StrAppend(&str, "Currently I have: \n");
272 std::vector<std::string> hands = FormatHand(player);
273 for (int suit = 0; suit < kNumSuits; ++suit) {
274 absl::StrAppend(
275 &str, absl::StrFormat("Suit %c: %s\n", kSuitChar[suit], hands[suit]));
276 }
277 absl::StrAppend(
278 &str, absl::StrFormat("Previous card: %s\n", GetCardStr(last_card_)));
279 absl::StrAppend(
280 &str, absl::StrFormat("Previous suit: %c\n", kSuitChar[last_suit_]));
281 absl::StrAppend(&str, "Starting counterclockwise, other players have: ");
282 for (int i = 0; i <= num_players_ - 1; ++i) {
283 int player_idx = (player + i) % num_players_;
284 int player_num_cards = 0;
285 for (int card = 0; card < kNumCards; ++card) {
286 player_num_cards += hands_[player_idx][card];
287 }
288 if (i != num_players_ - 1) {
289 absl::StrAppend(&str, absl::StrFormat("%d, ", player_num_cards));
290 } else {
291 absl::StrAppend(&str, absl::StrFormat("%d cards.\n", player_num_cards));
292 }
293 }
294 if (use_special_cards_) {
295 absl::StrAppend(&str, absl::StrFormat("The direction is %s\n",
296 direction_ == 1 ? "counterclockwise"
297 : "clockwise"));
298 }
299 return str;
300}

308void CrazyEightsState::WriteObservationTensor(Player player,
309 absl::Span<float> values) const {
310 SPIEL_CHECK_GE(player, 0);
311 SPIEL_CHECK_LT(player, num_players_);
312
313 absl::c_fill(values, 0.);
314 if (phase_ == Phase::kDeal) return;
315
316 for (int card = 0; card < kNumCards; ++card) {
317 values[card * (num_decks_ + 1) + hands_[player][card]] = 1;
318 }
319 values[(num_decks_ + 1) * kNumCards + last_card_] = 1;
320 values[(num_decks_ + 1) * kNumCards + kNumCards + last_suit_] = 1;
321 int tmp_base = (num_decks_ + 1) * kNumCards + kNumCards + kNumSuits;
322 for (int i = 1; i <= num_players_ - 1; ++i) {
323 int num_cards = 0;
324 for (int card = 0; card < kNumCards; ++card) {
325 num_cards += hands_[(player + i) % num_players_][card];
326 }
327 values[tmp_base + (i - 1) * (num_decks_ * kNumCards + 1) + num_cards] = 1;
328 }
329
330 if (use_special_cards_) {
331 tmp_base += (num_decks_ * kNumCards + 1) * (num_players_ - 1);
332 values[tmp_base] = (direction_ + 1) / 2;
333 }

347std::vector<std::pair<Action, double>> CrazyEightsState::ChanceOutcomes()
348 const {
349 std::vector<std::pair<Action, double>> outcomes;
350 if (history_.empty()) {
351 for (int player = 0; player < num_players_; ++player) {
352 outcomes.emplace_back(player + kDecideDealerActionBase,
353 1.0 / num_players_);
354 }
355 } else {
356 int num_cards_remaining = 0;
357 for (int card = 0; card < kNumCards; ++card) {
358 SPIEL_CHECK_GE(dealer_deck_[card], 0);
359 SPIEL_CHECK_LE(dealer_deck_[card], num_decks_);
360 num_cards_remaining += dealer_deck_[card];
361 }
362 outcomes.reserve(num_cards_remaining);
363 for (int card = 0; card < kNumCards; ++card) {
364 if (dealer_deck_[card]) {
365 outcomes.emplace_back(card, static_cast<double>(dealer_deck_[card]) /
366 num_cards_remaining);
367 }
368 }
369 }
370 return outcomes;
371}

402void CrazyEightsState::Reshuffle() {
403 SPIEL_CHECK_NE(last_card_, kInvalidAction);
404 for (int card = 0; card < kNumCards; ++card) {
405 dealer_deck_[card] = num_decks_;
406 for (int player = 0; player < num_players_; ++player) {
407 dealer_deck_[card] -= hands_[player][card];
408 }
409 if (card == last_card_) dealer_deck_[card]--;
410 SPIEL_CHECK_GE(dealer_deck_[card], 0);
411 SPIEL_CHECK_LE(dealer_deck_[card], num_decks_);
412 num_cards_left_ += dealer_deck_[card];
413 }
414}

416void CrazyEightsState::ApplyDealAction(int action) {
417 // determine the dealer
418 if (history_.empty()) {
419 dealer_ = action - kDecideDealerActionBase;
420 current_player_ = (dealer_ + 1) % num_players_;
421 return;
422 }
423
424 SPIEL_CHECK_GE(action, 0);
425 SPIEL_CHECK_LT(action, kDraw);
426
427 num_cards_left_--;
428 dealer_deck_[action]--;
429 hands_[current_player_][action]++;
430
431 SPIEL_CHECK_GE(dealer_deck_[action], 0);
432 SPIEL_CHECK_LE(dealer_deck_[action], num_decks_);
433
434 // reshuffle the discarded cards
435 if (!num_cards_left_ && reshuffle_) {
436 Reshuffle();
437 }
438
439 // redraw=true if we are examining the first card turned face up after the
440 // initial dealing round, which cannot be Eights
441 if (redraw_) {
442 SPIEL_CHECK_EQ(current_player_, dealer_);
443 int rank = GetRank(action);
444 if (rank != kEightRank) {
445 phase_ = Phase::kPlay;
446 redraw_ = false;
447 last_card_ = action;
448 last_suit_ = GetSuit(action);
449 // if it is special card, act as if the dealer played this card
450 if (use_special_cards_) {
451 if (rank == kSkipRank) {
452 current_player_ = (current_player_ + 2) % num_players_;
453 return;
454 } else if (rank == kReverseRank) {
455 current_player_ = (current_player_ - 1 + num_players_) %
456 num_players_;
457 direction_ *= -1;
458 return;
459 } else if (rank == kDrawTwoRank) {
460 num_draws_from_twos_left_ += 2;
461 current_player_ = (current_player_ + 1) % num_players_;
462 return;
463 }
464 }
465 current_player_ = (current_player_ + 1) % num_players_;
466 return;
467 } else {
468 // put back
469 dealer_deck_[action]++;
470 num_cards_left_++;
471 hands_[current_player_][action]--;
472 return;
473 }
474 }
475
476 SPIEL_CHECK_FALSE(redraw_);
477
478 if (history_.size() < num_players_ * num_initial_cards_) {
479 current_player_ = (current_player_ + 1) % num_players_;
480 return;
481 }
482
483 if (history_.size() == num_players_ * num_initial_cards_) {
484 SPIEL_CHECK_EQ(current_player_, dealer_);
485 redraw_ = true;
486 return;
487 }
488
489 if (!num_cards_left_) can_pass_action_ = true;
490
491 // if has accumlated 2s and has decided to draw these 2s from previous plays
492 if (start_draw_twos_) {
493 SPIEL_CHECK_TRUE(use_special_cards_);
494 num_draws_from_twos_left_--;
495 // assume if there is no card in the pile then the liability is cleared
496 if (!num_cards_left_) {
497 // if it is due to that the pile is exhausted during drawing +2s,
498 // counted as a pass
499 if (!num_draws_from_twos_left_) num_passes_++;
500 num_draws_from_twos_left_ = 0;
501 }
502 if (!num_draws_from_twos_left_) {
503 start_draw_twos_ = false;
504 phase_ = Phase::kPlay;
505 current_player_ = (current_player_ + direction_ +
506 num_players_) % num_players_;
507 }
508 return;
509 }
510
511 // lastly, consider when the player draws card without having a previous +2
512 // card
513 num_draws_before_play_++;
514 phase_ = Phase::kPlay;
515
516 if (!num_cards_left_) num_draws_before_play_ = max_draw_cards_;
517 if (num_draws_before_play_ == max_draw_cards_) {
518 can_pass_action_ = true;
519 }
520}

522void SearchLegalCards(std::vector<Action>* legal_actions,
523 const std::vector<int>& hand, int last_rank,
524 int last_suit) {
525 for (int card = 0; card < kNumCards; ++card) {
526 if (hand[card] == 0) continue;
527 Suit suit = GetSuit(card);
528 int rank = GetRank(card);
529 if (rank == kEightRank) {
530 legal_actions->push_back(card);
531 } else if (last_suit == suit || last_rank == rank) {
532 legal_actions->push_back(card);
533 }
534 }
535}

537std::vector<Action> CrazyEightsState::PlayLegalActions() const {
538 std::vector<Action> legal_actions;
539 if (nominate_suits_) {
540 for (int suit = kClubs; suit <= kSpades; ++suit) {
541 legal_actions.push_back(suit + kNominateSuitActionBase);
542 }
543 return legal_actions;
544 }
545
546 if (can_pass_action_ || !num_cards_left_) {
547 SPIEL_CHECK_TRUE(!start_draw_twos_);
548 legal_actions.push_back(kPass);
549 }
550
551 if (num_draws_from_twos_left_) {
552 SPIEL_CHECK_GT(num_cards_left_, 0);
553
554 legal_actions.push_back(kDraw);
555 // since we are able to draw
556 SPIEL_CHECK_FALSE(can_pass_action_);
557 SPIEL_CHECK_TRUE(use_special_cards_);
558
559 if (!start_draw_twos_) {
560 for (int suit = kClubs; suit <= kSpades; ++suit) {
561 int duo_card = GetAction(static_cast<Suit>(suit), kDrawTwoRank);
562 if (hands_[current_player_][duo_card])
563 legal_actions.push_back(duo_card);
564 }
565 }
566 } else {
567 for (int card = 0; card < kNumCards; ++card) {
568 if (hands_[current_player_][card] == 0) continue;
569 Suit suit = GetSuit(card);
570 int rank = GetRank(card);
571 if (rank == kEightRank) {
572 legal_actions.push_back(card);
573 } else if (last_suit_ == suit || GetRank(last_card_) == rank) {
574 legal_actions.push_back(card);
575 }
576 }
577 if (num_cards_left_ && num_draws_before_play_ != max_draw_cards_) {
578 SPIEL_CHECK_FALSE(can_pass_action_);
579 legal_actions.push_back(kDraw);
580 }
581 }
582 absl::c_sort(legal_actions);
583 return legal_actions;
584}

586bool CrazyEightsState::CheckAllCardsPlayed(int action) {
587 SPIEL_CHECK_GT(hands_[current_player_][action], 0);
588 hands_[current_player_][action]--;
589 bool all_played = true;
590 for (int card = 0; card < kNumCards; ++card) {
591 all_played &= !hands_[current_player_][card];
592 }
593 return all_played;
594}

596void CrazyEightsState::ApplyPlayAction(int action) {
597 if (action == kPass) {
598 if (!num_cards_left_) {
599 num_passes_++;
600 } else {
601 num_passes_ = 0;
602 }
603
604 if (num_passes_ == num_players_ + 1) {
605 phase_ = kGameOver;
606 ScoreUp();
607 return;
608 }
609
610 if (max_draw_cards_ == num_draws_before_play_) {
611 num_draws_before_play_ = 0;
612 }
613 current_player_ =
614 (current_player_ + direction_ + num_players_) % num_players_;
615 if (num_cards_left_) can_pass_action_ = false;
616 return;
617 } else {
618 num_passes_ = 0;
619 }
620
621 if (action == kDraw) {
622 SPIEL_CHECK_FALSE(can_pass_action_);
623 phase_ = kDeal;
624 if (num_draws_from_twos_left_) { start_draw_twos_ = true; }
625 return;
626 } else if (nominate_suits_) {
627 SPIEL_CHECK_LT(action, kNominateSuitActionBase + kNumSuits);
628 SPIEL_CHECK_GE(action, kNominateSuitActionBase);
629 last_suit_ = action - kNominateSuitActionBase;
630 current_player_ =
631 (current_player_ + direction_ + num_players_) % num_players_;
632 nominate_suits_ = false;
633 return;
634 } else {
635 num_plays++;
636 can_pass_action_ = false;
637 num_draws_before_play_ = 0;
638 bool all_played = CheckAllCardsPlayed(action);
639 if (all_played || num_plays >= kMaxTurnLimit) {
640 phase_ = kGameOver;
641 ScoreUp();
642 }
643
644 last_card_ = action;
645 last_suit_ = GetSuit(action);
646
647 if (!num_cards_left_ && reshuffle_) {
648 Reshuffle();
649 }
650
651 int rank = GetRank(action);
652
653 if (rank == kEightRank) {

656 }

668 }

671 // the next player doesn't have to draw cards in the next round,
672 // and just view it played a normal card
673 if (num_cards_left_) num_draws_from_twos_left_ += 2;
674 current_player_ =
675 (current_player_ + direction_ + num_players_) % num_players_;
676 return;
677 }
678 }

682 }

695void CrazyEightsState::ScoreUp() {
696 for (int player = 0; player < num_players_; ++player) {
697 for (int card = 0; card < kNumCards; ++card) {
698 if (!hands_[player][card]) continue;
699 int rank = GetRank(card);
700 if (rank == kEightRank) {
701 returns_[player] -= 50 * hands_[player][card];
702 } else if (rank >= 9) {
703 returns_[player] -= 10 * hands_[player][card];
704 } else {
705 returns_[player] -= (card + 2) * hands_[player][card];
706 }
707 }
708 }
