52std::unique_ptr<open_spiel::Bot> InitBot(
53 std::string type, const open_spiel::Game& game, open_spiel::Player player) {
54 if (type == "random") {
55 return open_spiel::MakeUniformRandomBot(player, Seed());
56 }
57 if (type == "simple_gin_rummy_bot") {
58 return std::make_unique<open_spiel::gin_rummy::SimpleGinRummyBot>(
59 game.GetParameters(), player);
60 }
61 open_spiel::SpielFatalError(
62 "Bad player type. Known types: simple_gin_rummy_bot, random");
63}

65std::vector<double> PlayGame(const open_spiel::Game& game,
66 const std::vector<std::unique_ptr<open_spiel::Bot>>& bots,
67 std::mt19937* rng, std::ostream& os, bool verbose, bool show_legals,
68 bool log_histories, bool log_returns, bool log_reach_probs) {
69 std::unique_ptr<open_spiel::State> state = game.NewInitialState();
70 for (open_spiel::Player p = 0; p < open_spiel::gin_rummy::kNumPlayers; ++p)
71 bots[p]->Restart();
72 std::vector<double> players_reach(2, 1.0);
73 double chance_reach = 1.0;
74
75 while (!state->IsTerminal()) {
76 open_spiel::Player player = state->CurrentPlayer();
77
78 if (verbose) os << "Player turn: " << player << std::endl;
79 if (show_legals) {
80 os << "Legal moves for player " << player << ":" << std::endl;
81 for (open_spiel::Action action : state->LegalActions(player))
82 os << " " << state->ActionToString(player, action) << std::endl;
83 }
84
85 open_spiel::Action action;
86 if (state->IsChanceNode()) {
87 std::pair<open_spiel::Action, double> outcome_and_prob =
88 open_spiel::SampleAction(state->ChanceOutcomes(), *rng);
89 action = outcome_and_prob.first;
90 SPIEL_CHECK_PROB(outcome_and_prob.second);
91 SPIEL_CHECK_GT(outcome_and_prob.second, 0);
92 SPIEL_CHECK_PROB(chance_reach);
93 chance_reach *= outcome_and_prob.second;
94 if (verbose) {
95 os << "Sampled action: " << state->ActionToString(player, action)
96 << std::endl;
97 }
98 } else {
99 std::pair<open_spiel::Action, double> outcome_and_prob =
100 open_spiel::SampleAction(bots[player]->GetPolicy(*state), *rng);
101 action = outcome_and_prob.first;
102 SPIEL_CHECK_PROB(outcome_and_prob.second);
103 SPIEL_CHECK_GT(outcome_and_prob.second, 0);
104 SPIEL_CHECK_PROB(players_reach[player]);
105 players_reach[player] *= outcome_and_prob.second;
106 if (verbose) {
107 os << "Chose action: " << state->ActionToString(player, action)
108 << std::endl;
109 }
110 }
111 if (!absl::c_binary_search(state->LegalActions(), action)) {
112 std::cerr << "State: " << std::endl << state->ToString() << std::endl
113 << "History: " << absl::StrJoin(state->History(), " ")
114 << std::endl << "Legal actions: "
115 << absl::StrJoin(state->LegalActions(), " ") << std::endl;
116 open_spiel::SpielFatalError("Illegal bot action.");
117 }
118 state->ApplyAction(action);
119 if (verbose) os << "State: " << std::endl << state->ToString() << std::endl;
120 }
121 if (verbose) {
122 os << "Returns: " << absl::StrJoin(state->Returns(), ",") << std::endl
123 << "History: " << absl::StrJoin(state->History(), " ") << std::endl;
124 } else if (log_histories) {
125 os << absl::StrJoin(state->History(), " ") << std::endl;
126 } else if (log_returns) {
127 os << absl::StrJoin(state->Returns(), " ") << " ";
128 if (log_reach_probs) {
129 os << absl::StrJoin(players_reach, " ") << " " << chance_reach;
130 }
131 os << std::endl;
132 }
133 return state->Returns();
134}

136int main(int argc, char** argv) {
137 std::vector<char*> positional_args = absl::ParseCommandLine(argc, argv);
138 std::mt19937 rng(Seed());
139
140 std::string game_string = absl::GetFlag(FLAGS_game_string);
141 std::cout << "Game string: " << game_string << std::endl;
142 std::shared_ptr<const open_spiel::Game> game =
143 open_spiel::LoadGame(game_string);
144
145 std::vector<std::unique_ptr<open_spiel::Bot>> bots;
146 bots.push_back(InitBot(absl::GetFlag(FLAGS_player0), *game, 0));
147 bots.push_back(InitBot(absl::GetFlag(FLAGS_player1), *game, 1));
148
149 int num_games = absl::GetFlag(FLAGS_num_games);
150 bool show_legals = absl::GetFlag(FLAGS_show_legals);
151 bool verbose = absl::GetFlag(FLAGS_verbose) || show_legals;
152 bool log_histories = absl::GetFlag(FLAGS_log_histories);
153 bool log_returns = absl::GetFlag(FLAGS_log_returns);
154 bool log_reach_probs = absl::GetFlag(FLAGS_log_reach_probs);
155 std::string path = absl::GetFlag(FLAGS_path);
156
157 std::ofstream os(path);
158 std::vector<double> overall_returns(2, 0);
159 std::vector<int> overall_wins(2, 0);
160 int percent = 0;
161 int refresh_threshold = 0;
162 absl::Time start = absl::Now();
163 for (int game_num = 0; game_num < num_games; ++game_num) {
164 percent = (100 * (game_num + 1)) / num_games;
165 if (percent >= refresh_threshold) {
166 // Progress bar.
167 std::cout << "\r" << "[" << std::string(percent / 5, '=')
168 << std::string(100 / 5 - percent / 5, ' ') << "]" << percent
169 << "%" << " [Game " << game_num + 1 << " of " << num_games
170 << "]";
171 std::cout.flush();
172 ++refresh_threshold;
173 }
174 std::vector<double> returns = PlayGame(*game, bots, &rng, os, verbose,
175 show_legals, log_histories, log_returns, log_reach_probs);
176 for (int i = 0; i < returns.size(); ++i) {
177 double v = returns[i];
178 overall_returns[i] += v;
179 if (v > 0) overall_wins[i] += 1;
180 }
181 }
182 absl::Time end = absl::Now();
183 double seconds = absl::ToDoubleSeconds(end - start);
184
185 std::cout << std::endl << "Number of games played: " << num_games << std::endl
186 << "Overall wins: " << absl::StrJoin(overall_wins, ",") << std::endl
187 << "Overall returns: " << absl::StrJoin(overall_returns, ",")
188 << std::endl << "Seconds: " << seconds << std::endl;
189 if (verbose || log_histories || log_returns)
190 std::cout << "Game histories logged to " << path << std::endl;
191}
