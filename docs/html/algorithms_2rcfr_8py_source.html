15 """Regression counterfactual regret minimization (RCFR) [Waugh et al., 2015; Morrill, 2016].
16 
17 In contrast to (tabular) counterfactual regret minimization (CFR)
18 [Zinkevich et al., 2007], RCFR replaces the table of regrets that generate the
19 current policy profile with a profile of regression models. The average
20 policy is still tracked exactly with a full game-size table. The exploitability
21 of the average policy in zero-sum games decreases as the model accuracy and
22 the number of iterations increase [Waugh et al., 2015; Morrill, 2016]. As long
23 as the regression model errors decrease across iterations, the average policy
24 converges toward a Nash equilibrium in zero-sum games.
25 
26 # References
27 
28 Dustin Morrill. Using Regret Estimation to Solve Games Compactly.
29  M.Sc. thesis, Computing Science Department, University of Alberta,
30  Apr 1, 2016, Edmonton Alberta, Canada.
31 Kevin Waugh, Dustin Morrill, J. Andrew Bagnell, and Michael Bowling.
32  Solving Games with Functional Regret Estimation. At the Twenty-Ninth AAAI
33  Conference on Artificial Intelligence, January 25-29, 2015, Austin Texas,
34  USA. Pages 2138-2145.
35 Martin Zinkevich, Michael Johanson, Michael Bowling, and Carmelo Piccione.
36  Regret Minimization in Games with Incomplete Information.
37  At Advances in Neural Information Processing Systems 20 (NeurIPS). 2007.
38 """

47 def tensor_to_matrix(tensor):
48  """Converts `tensor` to a matrix (a rank-2 tensor) or raises an exception.
49 
50  Args:
51  tensor: The tensor to convert.
52 
53  Returns:
54  A TensorFlow matrix (rank-2 `tf.Tensor`).
55  Raises:
56  ValueError: If `tensor` cannot be trivially converted to a matrix, i.e.
57  `tensor` has a rank > 2.
58  """
59  tensor = tf.convert_to_tensor(tensor)
60  rank = tensor.shape.rank
61  if rank > 2:
62  raise ValueError(
63  ("Tensor {} cannot be converted into a matrix as it is rank "
64  "{} > 2.").format(tensor, rank))
65  elif rank < 2:
66  num_columns = 1 if rank == 0 else tensor.shape[0].value
67  tensor = tf.reshape(tensor, [1, num_columns])
68  return tensor

71 def with_one_hot_action_features(state_features, legal_actions,
72  num_distinct_actions):
73  """Constructs features for each sequence by extending state features.
74 
75  Sequences features are constructed by concatenating one-hot features
76  indicating each action to the information state features and stacking them.
77 
78  Args:
79  state_features: The features for the information state alone. Must be a
80  `tf.Tensor` with a rank less than or equal to (if batched) 2.
81  legal_actions: The list of legal actions in this state. Determines the
82  number of rows in the returned feature matrix.
83  num_distinct_actions: The number of globally distinct actions in the game.
84  Determines the length of the action feature vector concatenated onto the
85  state features.
86 
87  Returns:
88  A `tf.Tensor` feature matrix with one row for each sequence and # state
89  features plus `num_distinct_actions`-columns.
90 
91  Raises:
92  ValueError: If `state_features` has a rank > 2.
93  """
94  state_features = tensor_to_matrix(state_features)
95  with_action_features = []
96  for action in legal_actions:
97  action_features = tf.one_hot([action], num_distinct_actions)
98  action_features = tf.tile(action_features, [tf.shape(state_features)[0], 1])
99  all_features = tf.concat([state_features, action_features], axis=1)
100  with_action_features.append(all_features)
101  return tf.concat(with_action_features, axis=0)

258  def counterfactual_regrets_and_reach_weights(self, regret_player,
259  reach_weight_player,
260  *sequence_weights):
261  """Returns counterfactual regrets and reach weights as a tuple.
262 
263  Args:
264  regret_player: The player for whom counterfactual regrets are computed.
265  reach_weight_player: The player for whom reach weights are computed.
266  *sequence_weights: A list of non-negative sequence weights for each player
267  determining the policy profile. Behavioral policies are generated by
268  normalizing sequence weights corresponding to actions in each
269  information state by their sum.
270 
271  Returns:
272  The counterfactual regrets and reach weights as an `np.array`-`np.array`
273  tuple.
274 
275  Raises:
276  ValueError: If there are too few sequence weights at any information state
277  for any player.
278  """
279  num_players = len(sequence_weights)
280  regrets = np.zeros(self.num_player_sequencesnum_player_sequences[regret_player])
281  reach_weights = np.zeros(self.num_player_sequencesnum_player_sequences[reach_weight_player])

283  def _walk_descendants(state, reach_probabilities, chance_reach_probability):
284  """Compute `state`'s counterfactual regrets and reach weights.
285 
286  Args:
287  state: An OpenSpiel `State`.
288  reach_probabilities: The probability that each player plays to reach
289  `state`'s history.
290  chance_reach_probability: The probability that all chance outcomes in
291  `state`'s history occur.
292 
293  Returns:
294  The counterfactual value of `state`'s history.
295  Raises:
296  ValueError if there are too few sequence weights at any information
297  state for any player.
298  """
299 
300  if state.is_terminal():
301  player_reach = (
302  np.prod(reach_probabilities[:regret_player]) *
303  np.prod(reach_probabilities[regret_player + 1:]))
304 
305  counterfactual_reach_prob = player_reach * chance_reach_probability
306  u = self.terminal_valuesterminal_values[state.history_str()]
307  return u[regret_player] * counterfactual_reach_prob
308 
309  elif state.is_chance_node():
310  v = 0.0
311  for action, action_prob in state.chance_outcomes():
312  v += _walk_descendants(
313  state.child(action), reach_probabilities,
314  chance_reach_probability * action_prob)
315  return v
316 
317  player = state.current_player()
318  info_state = state.information_state_string(player)
319  sequence_idx_offset = self.info_state_to_sequence_idxinfo_state_to_sequence_idx[info_state]
320  actions = state.legal_actions(player)
321 
322  sequence_idx_end = sequence_idx_offset + len(actions)
323  my_sequence_weights = sequence_weights[player][
324  sequence_idx_offset:sequence_idx_end]
325 
326  if len(my_sequence_weights) < len(actions):
327  raise ValueError(
328  ("Invalid policy: Policy {player} at sequence offset "
329  "{sequence_idx_offset} has only {policy_len} elements but there "
330  "are {num_actions} legal actions.").format(
331  player=player,
332  sequence_idx_offset=sequence_idx_offset,
333  policy_len=len(my_sequence_weights),
334  num_actions=len(actions)))
335 
336  policy = normalized_by_sum(my_sequence_weights)
337  action_values = np.zeros(len(actions))
338  state_value = 0.0
339 
340  is_reach_weight_player_node = player == reach_weight_player
341  is_regret_player_node = player == regret_player
342 
343  reach_prob = reach_probabilities[player]
344  for action_idx, action in enumerate(actions):
345  action_prob = policy[action_idx]
346  next_reach_prob = reach_prob * action_prob
347 
348  if is_reach_weight_player_node:
349  reach_weight_player_plays_down_this_line = next_reach_prob > 0
350  if not reach_weight_player_plays_down_this_line:
351  continue
352  sequence_idx = sequence_idx_offset + action_idx
353  reach_weights[sequence_idx] += next_reach_prob
354 
355  reach_probabilities[player] = next_reach_prob
356 
357  action_value = _walk_descendants(
358  state.child(action), reach_probabilities, chance_reach_probability)
359 
360  if is_regret_player_node:
361  state_value = state_value + action_prob * action_value
362  else:
363  state_value = state_value + action_value
364  action_values[action_idx] = action_value
365 
366  reach_probabilities[player] = reach_prob
367 
368  if is_regret_player_node:
369  regrets[sequence_idx_offset:sequence_idx_end] += (
370  action_values - state_value)
371  return state_value
372 
373  # End of _walk_descendants
374 
375  _walk_descendants(self.rootroot, np.ones(num_players), 1.0)
376  return regrets, reach_weights

379 def normalized_by_sum(v, axis=0, mutate=False):
380  """Divides each element of `v` along `axis` by the sum of `v` along `axis`.
381 
382  Assumes `v` is non-negative. Sets of `v` elements along `axis` that sum to
383  zero are normalized to `1 / v.shape[axis]` (a uniform distribution).
384 
385  Args:
386  v: Non-negative array of values.
387  axis: An integer axis.
388  mutate: Whether or not to store the result in `v`.
389 
390  Returns:
391  The normalized array.
392  """
393  v = np.asarray(v)
394  denominator = v.sum(axis=axis, keepdims=True)
395  denominator_is_zero = denominator == 0
396 
397  # Every element of `denominator_is_zero` that is true corresponds to a
398  # set of elements in `v` along `axis` that are all zero. By setting these
399  # denominators to `v.shape[axis]` and adding 1 to each of the corresponding
400  # elements in `v`, these elements are normalized to `1 / v.shape[axis]`
401  # (a uniform distribution).
402  denominator += v.shape[axis] * denominator_is_zero
403  if mutate:
404  v += denominator_is_zero
405  v /= denominator
406  else:
407  v = (v + denominator_is_zero) / denominator
408  return v

416 def _descendant_states(state, depth_limit, depth, include_terminals,
417  include_chance_states):
418  """Recursive descendant state generator.
419 
420  Decision states are always yielded.
421 
422  Args:
423  state: The current state.
424  depth_limit: The descendant depth limit. Zero will ensure only
425  `initial_state` is generated and negative numbers specify the absence of a
426  limit.
427  depth: The current descendant depth.
428  include_terminals: Whether or not to include terminal states.
429  include_chance_states: Whether or not to include chance states.
430 
431  Yields:
432  `State`, a state that is `initial_state` or one of its descendants.
433  """
434  if state.is_terminal():
435  if include_terminals:
436  yield state
437  return
438 
439  if depth > depth_limit >= 0:
440  return
441 
442  if not state.is_chance_node() or include_chance_states:
443  yield state
444 
445  for action in state.legal_actions():
446  state_for_search = state.child(action)
447  for substate in _descendant_states(state_for_search, depth_limit, depth + 1,
448  include_terminals,
449  include_chance_states):
450  yield substate

485 def sequence_weights_to_tabular_profile(root, policy_fn):
486  """Returns the `dict` of `list`s of action-prob pairs-form of `policy_fn`."""
487  tabular_policy = {}
488  players = range(root.num_players())
489  for state in all_states(root):
490  for player in players:
491  legal_actions = state.legal_actions(player)
492  if len(legal_actions) < 1:
493  continue

495  if info_state in tabular_policy:

498  tabular_policy[info_state] = list(zip(legal_actions, my_policy))

502 @tf.function
503 def feedforward_evaluate(layers,
504  x,
505  use_skip_connections=False,
506  hidden_are_factored=False):
507  """Evaluates `layers` as a feedforward neural network on `x`.
508 
509  Args:
510  layers: The neural network layers (`tf.Tensor` -> `tf.Tensor` callables).
511  x: The array-like input to evaluate. Must be trivially convertible to a
512  matrix (tensor rank <= 2).
513  use_skip_connections: Whether or not to use skip connections between layers.
514  If the layer input has too few features to be added to the layer output,
515  then the end of input is padded with zeros. If it has too many features,
516  then the input is truncated.
517  hidden_are_factored: Whether or not hidden logical layers are factored into
518  two separate linear transformations stored as adjacent elements of
519  `layers`.
520 
521  Returns:
522  The `tf.Tensor` evaluation result.
523 
524  Raises:
525  ValueError: If `x` has a rank greater than 2.
526  """
527  x = tensor_to_matrix(x)
528  i = 0
529  while i < len(layers) - 1:
530  y = layers[i](x)
531  i += 1
532  if hidden_are_factored:
533  y = layers[i](y)
534  i += 1
535  if use_skip_connections:
536  my_num_features = x.shape[1].value
537  padding = y.shape[1].value - my_num_features
538  if padding > 0:
539  zeros = tf.zeros([tf.shape(x)[0], padding])
540  x = tf.concat([x, zeros], axis=1)
541  elif padding < 0:
542  x = tf.strided_slice(x, [0, 0], [tf.shape(x)[0], y.shape[1].value])
543  y = x + y
544  x = y
545  return layers[-1](x)

548 class DeepRcfrModel(object):
549  """A flexible deep feedforward RCFR model class.
550 
551  Properties:
552  layers: The `tf.keras.Layer` layers describing this model.
553  trainable_variables: The trainable `tf.Variable`s in this model's `layers`.

556 
557  def __init__(self,
558  game,
559  num_hidden_units,
560  num_hidden_layers=1,
561  num_hidden_factors=0,
562  hidden_activation=tf.nn.relu,
563  use_skip_connections=False,
564  regularizer=None):
565  """Creates a new `DeepRcfrModel.
566 
567  Args:
568  game: The OpenSpiel game being solved.
569  num_hidden_units: The number of units in each hidden layer.
570  num_hidden_layers: The number of hidden layers. Defaults to 1.
571  num_hidden_factors: The number of hidden factors or the matrix rank of the
572  layer. If greater than zero, hidden layers will be split into two
573  separate linear transformations, the first with
574  `num_hidden_factors`-columns and the second with
575  `num_hidden_units`-columns. The result is that the logical hidden layer
576  is a rank-`num_hidden_units` matrix instead of a rank-`num_hidden_units`
577  matrix. When `num_hidden_units < num_hidden_units`, this is effectively
578  implements weight sharing. Defaults to 0.
579  hidden_activation: The activation function to apply over hidden layers.
580  Defaults to `tf.nn.relu`.
581  use_skip_connections: Whether or not to apply skip connections (layer
582  output = layer(x) + x) on hidden layers. Zero padding or truncation is
583  used to match the number of columns on layer inputs and outputs.
584  regularizer: A regularizer to apply to each layer. Defaults to `None`.
585  """
586  self._use_skip_connections_use_skip_connections = use_skip_connections
587  self._hidden_are_factored_hidden_are_factored = num_hidden_factors > 0
588 
589  self.layerslayers = []
590  for _ in range(num_hidden_layers):
591  if self._hidden_are_factored_hidden_are_factored:
592  self.layerslayers.append(
593  tf.keras.layers.Dense(
594  num_hidden_factors,
595  use_bias=True,
596  kernel_regularizer=regularizer))
597 
598  self.layerslayers.append(
599  tf.keras.layers.Dense(
600  num_hidden_units,
601  use_bias=True,
602  activation=hidden_activation,
603  kernel_regularizer=regularizer))
604 
605  self.layerslayers.append(
606  tf.keras.layers.Dense(1, use_bias=True, kernel_regularizer=regularizer))
607 
608  # Construct variables for all layers by exercising the network.
609  x = tf.zeros([1, num_features(game)])
610  for layer in self.layerslayers:
611  x = layer(x)
612 
613  self.trainable_variablestrainable_variables = sum(
614  [layer.trainable_variables for layer in self.layerslayers], [])
615  self.losseslosses = sum([layer.losses for layer in self.layerslayers], [])
616 
617  def __call__(self, x):
618  """Evaluates this model on `x`."""
619  return feedforward_evaluate(
620  layers=self.layerslayers,
621  x=x,
622  use_skip_connections=self._use_skip_connections_use_skip_connections,
623  hidden_are_factored=self._hidden_are_factored_hidden_are_factored)

783 class ReservoirBuffer(object):
784  """A generic reservoir buffer data structure.
785 
786  After every insertion, its contents represents a `size`-size uniform
787  random sample from the stream of candidates that have been encountered.
788  """
789 
790  def __init__(self, size):
791  self.sizesize = size
792  self.num_elementsnum_elements = 0
793  self._buffer_buffer = np.full([size], None, dtype=object)
794  self._num_candidates_num_candidates = 0
795 
796  @property
797  def buffer(self):
798  return self._buffer_buffer[:self.num_elementsnum_elements]
799 
800  def insert(self, candidate):
801  """Consider this `candidate` for inclusion in this sampling buffer."""
802  self._num_candidates_num_candidates += 1
803  if self.num_elementsnum_elements < self.sizesize:
804  self._buffer_buffer[self.num_elementsnum_elements] = candidate
805  self.num_elementsnum_elements += 1
806  return
807  idx = np.random.choice(self._num_candidates_num_candidates)
808  if idx < self.sizesize:
809  self._buffer_buffer[idx] = candidate
810 
811  def insert_all(self, candidates):
812  """Consider all `candidates` for inclusion in this sampling buffer."""
813  for candidate in candidates:
814  self.insertinsert(candidate)
815 
816  def num_available_spaces(self):
817  """The number of freely available spaces in this buffer."""
818  return self.sizesize - self.num_elementsnum_elements
