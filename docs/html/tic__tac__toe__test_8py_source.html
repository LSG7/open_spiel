135  def test_consistent(self):
136  """Checks the Python and C++ game implementations are the same."""
137  py_game = pyspiel.load_game("python_tic_tac_toe")
138  cc_game = pyspiel.load_game("tic_tac_toe")
139  py_obs = make_observation(py_game)
140  cc_obs = make_observation(cc_game)
141  py_states = get_all_states(py_game, to_string=str)
142  cc_states = get_all_states(cc_game, to_string=str)
143  self.assertCountEqual(list(cc_states), list(py_states))
144  for key, cc_state in cc_states.items():
145  py_state = py_states[key]
146  np.testing.assert_array_equal(py_state.history(), cc_state.history())
147  np.testing.assert_array_equal(py_state.returns(), cc_state.returns())
148  py_obs.set_from(py_state, 0)
149  cc_obs.set_from(cc_state, 0)
150  np.testing.assert_array_equal(py_obs.tensor, cc_obs.tensor)
