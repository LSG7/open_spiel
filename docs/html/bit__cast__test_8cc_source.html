28 template <int N>
29 struct marshall { char buf[N]; };

31 template <typename T>
32 void TestMarshall(const T values[], int num_values) {
33  for (int i = 0; i < num_values; ++i) {
34  T t0 = values[i];
35  marshall<sizeof(T)> m0 = absl::bit_cast<marshall<sizeof(T)> >(t0);
36  T t1 = absl::bit_cast<T>(m0);
37  marshall<sizeof(T)> m1 = absl::bit_cast<marshall<sizeof(T)> >(t1);
38  ASSERT_EQ(0, memcmp(&t0, &t1, sizeof(T)));
39  ASSERT_EQ(0, memcmp(&m0, &m1, sizeof(T)));
40  }
41 }

43 // Convert back and forth to an integral type. The C++ standard does
44 // not guarantee this will work, but we test that this works on all the
45 // platforms we support.
46 //
47 // Likewise, we below make assumptions about sizeof(float) and
48 // sizeof(double) which the standard does not guarantee, but which hold on the
49 // platforms we support.
50 
51 template <typename T, typename I>
52 void TestIntegral(const T values[], int num_values) {
53  for (int i = 0; i < num_values; ++i) {
54  T t0 = values[i];
55  I i0 = absl::bit_cast<I>(t0);
56  T t1 = absl::bit_cast<T>(i0);
57  I i1 = absl::bit_cast<I>(t1);
58  ASSERT_EQ(0, memcmp(&t0, &t1, sizeof(T)));
59  ASSERT_EQ(i0, i1);
60  }
61 }
