57class NFGGameParser {
58 public:
59 explicit NFGGameParser(const std::string& data)
60 : string_data_(data), pos_(0) {}

62 shared_ptr<const Game> ParseGame() {
63 // Skip any initial whitespace.
64 while (IsWhiteSpace(string_data_.at(pos_))) {
65 AdvancePosition();
66 }
67 SPIEL_CHECK_LT(pos_, string_data_.length());
68
69 ParsePrologue();
70 InitializeMetaInformation();
71 ParseUtilities();
72
73 if (num_players_ == 2) {
74 return matrix_game::CreateMatrixGame(
75 "matrix_nfg", name_, matrix_row_action_names_,
76 matrix_col_action_names_, matrix_row_utilities_,
77 matrix_col_utilities_);
78 } else {
79 return tensor_game::CreateTensorGame(
80 "tensor_nfg", name_, tensor_action_names_, tensor_utilities_);
81 }
82 }
83
84 private:
85 void ParsePrologue() {
86 // Parse the first part of the header "NFG 1 R "
87 SPIEL_CHECK_TRUE(NextToken() == "NFG");
88 SPIEL_CHECK_TRUE(NextToken() == "1");
89 // Older versions of .nfg format use D
90 std::string data_type = NextToken();
91 SPIEL_CHECK_TRUE(data_type == "R" || data_type == "D");
92 SPIEL_CHECK_EQ(string_data_.at(pos_), '"');
93 name_ = NextToken();
94 // Player names
95 std::string token = NextToken();
96 SPIEL_CHECK_TRUE(token == "{");
97 SPIEL_CHECK_EQ(string_data_.at(pos_), '"');
98 token = NextToken();
99 while (token != "}") {
100 player_names_.push_back(token);
101 token = NextToken();
102 }
103 num_players_ = player_names_.size();
104 // Number of actions
105 token = NextToken();
106 SPIEL_CHECK_TRUE(token == "{");
107 token = NextToken();
108 while (token != "}") {
109 int num = 0;
110 SPIEL_CHECK_TRUE(absl::SimpleAtoi(token, &num));
111 num_actions_.push_back(num);
112 token = NextToken();
113 }
114 SPIEL_CHECK_EQ(num_actions_.size(), num_players_);
115 }
116
117 void InitializeMetaInformation() {
118 total_entries_ = std::accumulate(num_actions_.begin(), num_actions_.end(),
119 1ULL, std::multiplies<uint64_t>());
120 SPIEL_CHECK_GT(total_entries_, 0);
121
122 // Fill some of the meta information.
123 if (num_players_ == 2) {
124 matrix_row_action_names_.reserve(num_actions_[0]);
125 matrix_col_action_names_.reserve(num_actions_[1]);
126 matrix_row_utilities_ = std::vector<double>(total_entries_, 0);
127 matrix_col_utilities_ = std::vector<double>(total_entries_, 0);
128 for (int a = 0; a < num_actions_[0]; ++a) {
129 matrix_row_action_names_.push_back(absl::StrCat("", a));
130 }
131 for (int a = 0; a < num_actions_[1]; ++a) {
132 matrix_col_action_names_.push_back(absl::StrCat("", a));
133 }
134 } else {
135 tensor_action_names_.reserve(num_players_);
136 tensor_utilities_.reserve(num_players_);
137 for (int p = 0; p < num_players_; ++p) {
138 tensor_utilities_.push_back(std::vector<double>(total_entries_, 0));
139 tensor_action_names_.push_back({});
140 tensor_action_names_.back().reserve(num_actions_[p]);
141 for (int a = 0; a < num_actions_[p]; ++a) {
142 tensor_action_names_[p].push_back(absl::StrCat("", a));
143 }
144 }
145 }
146 }
147
148 int RowMajorIndex(const std::vector<int>& num_actions,
149 const std::vector<int>& actions) {
150 int index = 0;
151 int base_value = 1;
152 for (int p = actions.size() - 1; p >= 0; --p) {
153 if (p + 1 < actions.size()) {
154 base_value *= num_actions[p + 1];
155 }
156 index += actions[p] * base_value;
157 }
158 return index;
159 }
160
161 void ParseUtilities() {
162 // Parse all the utilities.
163 std::string token;
164 std::vector<int> actions(num_players_, 0);
165 for (uint64_t entry = 0; entry < total_entries_; ++entry) {
166 double value = 0;
167 int row_major_index = RowMajorIndex(num_actions_, actions);
168 for (int p = 0; p < num_players_; ++p) {
169 // Check that the position has not reached the end for every value we
170 // read, except the very last one.
171 bool check_end = entry != total_entries_ - 1 && p != num_players_ - 1;
172 std::string token = NextToken(check_end);
173 ParseDoubleValue(token, &value);
174
175 if (num_players_ == 2) {
176 if (p == 0) {
177 matrix_row_utilities_[row_major_index] = value;
178 } else {
179 matrix_col_utilities_[row_major_index] = value;
180 }
181 } else {
182 tensor_utilities_[p][row_major_index] = value;
183 }
184 }
185
186 // next action indices, in column-major order.
187 for (int i = 0; i < actions.size(); ++i) {
188 if (++actions[i] < num_actions_[i]) {
189 break;
190 } else {
191 actions[i] = 0;
192 }
193 }
194 }
195
196 // After reading all the utilities, we should reach the end of the file.
197 SPIEL_CHECK_EQ(pos_, string_data_.length());
198 }

200 bool ParseDoubleValue(const std::string& str, double* value) const {
201 if (str.find('/') != std::string::npos) {
202 // Check for rational number of the form X/Y
203 std::vector<std::string> parts = absl::StrSplit(str, '/');
204 SPIEL_CHECK_EQ(parts.size(), 2);
205 int numerator = 0, denominator = 0;
206 bool success = absl::SimpleAtoi(parts[0], &numerator);
207 if (!success) {
208 return false;
209 }
210 success = absl::SimpleAtoi(parts[1], &denominator);
211 if (!success) {
212 return false;
213 }
214 SPIEL_CHECK_FALSE(denominator == 0);
215 *value = static_cast<double>(numerator) / denominator;
216 return true;
217 } else {
218 // Otherwise, parse as a double.
219 return absl::SimpleAtod(str, value);
220 }
221 }

223 bool IsWhiteSpace(char c) const {
224 return (c == ' ' || c == '\r' || c == '\n');
225 }
226
227 void AdvancePosition() { pos_++; }
228
229 // Get the next token, and then advance the position to the start of the next
230 // token. If check_not_end is true, then a check is done to ensure that the
231 // position has not reached the end of the string.
232 std::string NextToken(bool check_not_end = true) {
233 std::string str = "";
234 bool reading_quoted_string = false;
235
236 if (string_data_.at(pos_) == '"') {
237 reading_quoted_string = true;
238 AdvancePosition();
239 }
240
241 while (true) {
242 // Check stopping condition:
243 if (pos_ >= string_data_.length() ||
244 (reading_quoted_string && string_data_.at(pos_) == '"') ||
245 (!reading_quoted_string && IsWhiteSpace(string_data_.at(pos_)))) {
246 break;
247 }
248
249 str.push_back(string_data_.at(pos_));
250 AdvancePosition();
251 }
252
253 if (reading_quoted_string) {
254 SPIEL_CHECK_EQ(string_data_.at(pos_), '"');
255 }
256 AdvancePosition();
257
258 // Advance the position to the next token.
259 while (pos_ < string_data_.length() &&
260 IsWhiteSpace(string_data_.at(pos_))) {
261 AdvancePosition();
262 }
263
264 if (check_not_end) {
265 SPIEL_CHECK_LT(pos_, string_data_.length());
266 }
267
268 return str;
269 }

271 const std::string& string_data_;
272 int pos_;
273 int num_players_;
274 std::string name_;
275 std::vector<std::string> player_names_;
276 std::vector<int> num_actions_;
277 std::vector<std::vector<double>> utilities_;
278
279 // Information needed to construct the matrix / tensor games.
280 uint64_t total_entries_;
281 // MatrixGame case.
282 std::vector<std::string> matrix_row_action_names_;
283 std::vector<std::string> matrix_col_action_names_;
284 std::vector<double> matrix_row_utilities_;
285 std::vector<double> matrix_col_utilities_;
286 // TensorGame case.
287 std::vector<std::vector<std::string>> tensor_action_names_;
288 std::vector<std::vector<double>> tensor_utilities_;
289};
