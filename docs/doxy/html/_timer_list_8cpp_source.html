80void TimerList::PrintStats(ofstream& fout) const
81{
82 if (! TimerList::Used())
83 return;
84
85 // Approximate the exclusive times of each function.
86 // The ABsearch*() functions are recursively nested,
87 // so subtract out the one below.
88 // The other ones are subtracted out based on knowledge
89 // of the functions.
90
91 TimerGroup ABGroup;
92 ABGroup = timerGroups[0];
93 ABGroup.Differentiate();
94 for (unsigned g = 1; g < TIMER_NO_SIZE; g++)
95 ABGroup -= timerGroups[g];
96
97 Timer ABTotal;
98 ABGroup.SetNames("AB");
99 ABGroup.Sum(ABTotal);
100 ABTotal.SetName("Sum");
101
102 Timer sumTotal = ABTotal;
103 for (unsigned g = 1; g < TIMER_NO_SIZE; g++)
104 {
105 Timer t;
106 timerGroups[g].Sum(t);
107 sumTotal += t;
108 }
109
110 fout << timerGroups[0].Header();
111 fout << ABGroup.SumLine(sumTotal);
112 for (unsigned g = 1; g < TIMER_NO_SIZE; g++)
113 fout << timerGroups[g].SumLine(sumTotal);
114 fout << timerGroups[0].DashLine();
115 fout << sumTotal.SumLine(sumTotal) << endl;
116
117 if (ABGroup.Used())
118 {
119 fout << ABGroup.Header();
120 fout << ABGroup.TimerLines(ABTotal);
121 fout << ABGroup.DashLine();
122 fout << ABTotal.SumLine(ABTotal) << endl;
123 }
124
125#ifdef DDS_TIMING_DETAILS
126 fout << timerGroups[0].DetailHeader();
127 for (unsigned g = 0; g < TIMER_NO_SIZE; g++)
128 fout << timerGroups[g].DetailLines();
129 fout << endl;
130#endif
131}
