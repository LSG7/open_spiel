27 // A helper class for maintaining and building correlation devices
28 // (distributions over joint deterministic policies).
29 //
30 // This helper exists to serve algorithms interact with the CorrDist functions
31 // (see corr_dist.h), which require distributions over joint deterministic
32 // policies. Algorithms like CFR produce stochastic policies, so they have
33 // to either be converted or sampled before they can be evaluated by the
34 // CorrDist functions.
35 //
36 // This helper class maintains weights over joint determinstic tabular policies,
37 // updating each as new policies are added. A correlation device is obtained by
38 // normalizing the weights over all the deterministic policies being tracked.
39 class CorrDevBuilder {
40  public:
41  CorrDevBuilder(int seed = 0);
42 
43  // Add a joint policy with the specified weight.
44  void AddDeterminsticJointPolicy(const TabularPolicy& policy,
45  double weight = 1.0);
46 
47  // Take a number of sampled joint policies and add each one with a weight
48  // of 1.0 / num_samples. The mixed policy should be complete: a policy should
49  // be defined for every information state, otherwise the distribution will not
50  // be properly built (joint policies will be incomplete).
51  void AddSampledJointPolicy(const TabularPolicy& policy, int num_samples,
52  double weight = 1.0);
53 
54  // This function adds a mixed joint policy to the correlation device. It does
55  // so by computing the probability of each deterministic joint policy by
56  // enumerating all possible actions that the policy is mixing over and
57  // computing the weight of each joint policy as a product of these
58  // probabilities. The mixed policy should be complete: a policy should be
59  // defined for every information state, otherwise the distribution will not
60  // be properly built (joint policies will be incomplete).
61  // Important note: this is computationally expensive and should only be used
62  // for small games. For larger games, used the sampled version above.
63  void AddMixedJointPolicy(const TabularPolicy& policy, double weight = 1.0);
64 
65  // Return the correlation device represented by this builder.
66  CorrelationDevice GetCorrelationDevice() const;
67 
68  private:
69  std::mt19937 rng_;
70  double total_weight_;
71 
72  // Each of these uses keys that have a canonical stringified policy as the
73  // key (e.g. complete policies with sorted keys).
74  absl::flat_hash_map<std::string, double> policy_weights_;
75  absl::flat_hash_map<std::string, TabularPolicy> policy_map_;
76 };
