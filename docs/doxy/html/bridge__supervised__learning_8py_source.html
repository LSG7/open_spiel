58def _no_play_trajectory(line: str):
59 """Returns the deal and bidding actions only given a text trajectory."""
60 actions = [int(x) for x in line.split(' ')]
61 # Usually a trajectory is NUM_CARDS chance events for the deal, plus one
62 # action for every bid of the auction, plus NUM_CARDS actions for the play
63 # phase. Exceptionally, if all NUM_PLAYERS players Pass, there is no play
64 # phase and the trajectory is just of length NUM_CARDS + NUM_PLAYERS.
65 if len(actions) == NUM_CARDS + NUM_PLAYERS:
66 return tuple(actions)
67 else:
68 return tuple(actions[:-NUM_CARDS])

71def make_dataset(file: str):
72 """Creates dataset as a generator of single examples."""
73 all_trajectories = [_no_play_trajectory(line) for line in open(file)]
74 while True:
75 np.random.shuffle(all_trajectories)
76 for trajectory in all_trajectories:
77 action_index = np.random.randint(52, len(trajectory))
78 state = GAME.new_initial_state()
79 for action in trajectory[:action_index]:
80 state.apply_action(action)
81 yield (state.observation_tensor(), trajectory[action_index] - MIN_ACTION)

84def batch(dataset, batch_size: int):
85 """Creates a batched dataset from a one-at-a-time dataset."""
86 observations = np.zeros([batch_size] + GAME.observation_tensor_shape(),
87 np.float32)
88 labels = np.zeros(batch_size, dtype=np.int32)
89 while True:
90 for batch_index in range(batch_size):
91 observations[batch_index], labels[batch_index] = next(dataset)
92 yield observations, labels

95def one_hot(x, k):
96 """Returns a one-hot encoding of `x` of size `k`."""
97 return jnp.array(x[..., jnp.newaxis] == jnp.arange(k), dtype=np.float32)

100def net_fn(x):
101 """Haiku module for our network."""
102 net = hk.Sequential([
103 hk.Linear(1024),
104 jax.nn.relu,
105 hk.Linear(1024),
106 jax.nn.relu,
107 hk.Linear(1024),
108 jax.nn.relu,
109 hk.Linear(1024),
110 jax.nn.relu,
111 hk.Linear(NUM_ACTIONS),
112 jax.nn.log_softmax,
113 ])
114 return net(x)

117def main(argv):
118 if len(argv) > 1:
119 raise app.UsageError('Too many command-line arguments.')

121 # Make the network.
122 net = hk.without_apply_rng(hk.transform(net_fn))
123
124 # Make the optimiser.
125 opt = optax.adam(1e-4)
126
127 @jax.jit
128 def loss(
129 params: Params,
130 inputs: np.ndarray,
131 targets: np.ndarray,
132 ) -> jax.Array:
133 """Cross-entropy loss."""
134 assert targets.dtype == np.int32
135 log_probs = net.apply(params, inputs)
136 return -jnp.mean(one_hot(targets, NUM_ACTIONS) * log_probs)
137
138 @jax.jit
139 def accuracy(
140 params: Params,
141 inputs: np.ndarray,
142 targets: np.ndarray,
143 ) -> jax.Array:
144 """Classification accuracy."""
145 predictions = net.apply(params, inputs)
146 return jnp.mean(jnp.argmax(predictions, axis=-1) == targets)
147
148 @jax.jit
149 def update(
150 params: Params,
151 opt_state: OptState,
152 inputs: np.ndarray,
153 targets: np.ndarray,
154 ) -> Tuple[Params, OptState]:
155 """Learning rule (stochastic gradient descent)."""
156 _, gradient = jax.value_and_grad(loss)(params, inputs, targets)
157 updates, opt_state = opt.update(gradient, opt_state)
158 new_params = optax.apply_updates(params, updates)
159 return new_params, opt_state

161 def output_samples(params: Params, max_samples: int):
162 """Output some cases where the policy disagrees with the dataset action."""
163 if max_samples == 0:
164 return
165 count = 0
166 with open(os.path.join(FLAGS.data_path, 'test.txt')) as f:
167 lines = list(f)
168 np.random.shuffle(lines)
169 for line in lines:
170 state = GAME.new_initial_state()
171 actions = _no_play_trajectory(line)
172 for action in actions:
173 if not state.is_chance_node():
174 observation = np.array(state.observation_tensor(), np.float32)
175 policy = np.exp(net.apply(params, observation))
176 probs_actions = [(p, a + MIN_ACTION) for a, p in enumerate(policy)]
177 pred = max(probs_actions)[1]
178 if pred != action:
179 print(state)
180 for p, a in reversed(sorted(probs_actions)[-TOP_K_ACTIONS:]):
181 print('{:7} {:.2f}'.format(state.action_to_string(a), p))
182 print('Ground truth {}\n'.format(state.action_to_string(action)))
183 count += 1
184 break
185 state.apply_action(action)
186 if count >= max_samples:
187 return

189 # Make datasets.
190 if FLAGS.data_path is None:
191 raise app.UsageError(
192 'Please generate your own supervised training data or download from '
193 'https://console.cloud.google.com/storage/browser/openspiel-data/bridge'
194 ' and supply the local location as --data_path')
195 train = batch(
196 make_dataset(os.path.join(FLAGS.data_path, 'train.txt')),
197 FLAGS.train_batch)
198 test = batch(
199 make_dataset(os.path.join(FLAGS.data_path, 'test.txt')), FLAGS.eval_batch)
200
201 # Initialize network and optimiser.
202 rng = jax.random.PRNGKey(FLAGS.rng_seed) # seed used for network weights
203 inputs, unused_targets = next(train)
204 params = net.init(rng, inputs)
205 opt_state = opt.init(params)
206
207 # Train/eval loop.
208 for step in range(FLAGS.iterations):
209 # Do SGD on a batch of training examples.
210 inputs, targets = next(train)
211 params, opt_state = update(params, opt_state, inputs, targets)
212
213 # Periodically evaluate classification accuracy on the test set.
214 if (1 + step) % FLAGS.eval_every == 0:
215 inputs, targets = next(test)
216 test_accuracy = accuracy(params, inputs, targets)
217 print(f'After {1+step} steps, test accuracy: {test_accuracy}.')
218 if FLAGS.save_path:
219 filename = os.path.join(FLAGS.save_path, f'params-{1 + step}.pkl')
220 with open(filename, 'wb') as pkl_file:
221 pickle.dump(params, pkl_file)
222 output_samples(params, FLAGS.num_examples)
