15"""Implements Deep CFR Algorithm.
16
17See https://arxiv.org/abs/1811.00164.
18
19The algorithm defines an `advantage` and `strategy` networks that compute
20advantages used to do regret matching across information sets and to approximate
21the strategy profiles of the game. To train these networks a fixed ring buffer
22(other data structures may be used) memory is used to accumulate samples to
23train the networks.
24"""

45class SonnetLinear(nn.Module):
46 """A Sonnet linear module.
47
48 Always includes biases and only supports ReLU activations.
49 """
50
51 def __init__(self, in_size, out_size, activate_relu=True):
52 """Creates a Sonnet linear layer.
53
54 Args:
55 in_size: (int) number of inputs
56 out_size: (int) number of outputs
57 activate_relu: (bool) whether to include a ReLU activation layer
58 """
59 super(SonnetLinear, self).__init__()
60 self._activate_relu = activate_relu
61 self._in_size = in_size
62 self._out_size = out_size
63 # stddev = 1.0 / math.sqrt(self._in_size)
64 # mean = 0
65 # lower = (-2 * stddev - mean) / stddev
66 # upper = (2 * stddev - mean) / stddev
67 # # Weight initialization inspired by Sonnet's Linear layer,
68 # # which cites https://arxiv.org/abs/1502.03167v3
69 # # pytorch default: initialized from
70 # # uniform(-sqrt(1/in_features), sqrt(1/in_features))
71 self._weight = None
72 self._bias = None
73 self.reset()
74
75 def forward(self, tensor):
76 y = F.linear(tensor, self._weight, self._bias)
77 return F.relu(y) if self._activate_relu else y
78
79 def reset(self):
80 stddev = 1.0 / math.sqrt(self._in_size)
81 mean = 0
82 lower = (-2 * stddev - mean) / stddev
83 upper = (2 * stddev - mean) / stddev
84 # Weight initialization inspired by Sonnet's Linear layer,
85 # which cites https://arxiv.org/abs/1502.03167v3
86 # pytorch default: initialized from
87 # uniform(-sqrt(1/in_features), sqrt(1/in_features))
88 self._weight = nn.Parameter(
89 torch.Tensor(
90 stats.truncnorm.rvs(
91 lower,
92 upper,
93 loc=mean,
94 scale=stddev,
95 size=[self._out_size, self._in_size])))
96 self._bias = nn.Parameter(torch.zeros([self._out_size]))

99class MLP(nn.Module):
100 """A simple network built from nn.linear layers."""
101
102 def __init__(self,
103 input_size,
104 hidden_sizes,
105 output_size,
106 activate_final=False):
107 """Create the MLP.
108
109 Args:
110 input_size: (int) number of inputs
111 hidden_sizes: (list) sizes (number of units) of each hidden layer
112 output_size: (int) number of outputs
113 activate_final: (bool) should final layer should include a ReLU
114 """
115
116 super(MLP, self).__init__()
117 self._layers = []
118 # Hidden layers
119 for size in hidden_sizes:
120 self._layers.append(SonnetLinear(in_size=input_size, out_size=size))
121 input_size = size
122 # Output layer
123 self._layers.append(
124 SonnetLinear(
125 in_size=input_size,
126 out_size=output_size,
127 activate_relu=activate_final))
128
129 self.model = nn.ModuleList(self._layers)
130
131 def forward(self, x):
132 for layer in self.model:
133 x = layer(x)
134 return x
135
136 def reset(self):
137 for layer in self._layers:
138 layer.reset()

141class ReservoirBuffer(object):
142 """Allows uniform sampling over a stream of data.
143
144 This class supports the storage of arbitrary elements, such as observation
145 tensors, integer actions, etc.
146 See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
147 """
148
149 def __init__(self, reservoir_buffer_capacity):
150 self._reservoir_buffer_capacity = reservoir_buffer_capacity
151 self._data = []
152 self._add_calls = 0
153
154 def add(self, element):
155 """Potentially adds `element` to the reservoir buffer.
156
157 Args:
158 element: data to be added to the reservoir buffer.
159 """
160 if len(self._data) < self._reservoir_buffer_capacity:
161 self._data.append(element)
162 else:
163 idx = np.random.randint(0, self._add_calls + 1)
164 if idx < self._reservoir_buffer_capacity:
165 self._data[idx] = element
166 self._add_calls += 1
167
168 def sample(self, num_samples):
169 """Returns `num_samples` uniformly sampled from the buffer.
170
171 Args:
172 num_samples: `int`, number of samples to draw.
173
174 Returns:
175 An iterable over `num_samples` random elements of the buffer.
176 Raises:
177 ValueError: If there are less than `num_samples` elements in the buffer
178 """
179 if len(self._data) < num_samples:
180 raise ValueError("{} elements could not be sampled from size {}".format(
181 num_samples, len(self._data)))
182 return random.sample(self._data, num_samples)
183
184 def clear(self):
185 self._data = []
186 self._add_calls = 0
187
188 def __len__(self):
189 return len(self._data)
190
191 def __iter__(self):
192 return iter(self._data)

195class DeepCFRSolver(policy.Policy):
196 """Implements a solver for the Deep CFR Algorithm with PyTorch.
197
198 See https://arxiv.org/abs/1811.00164.
199
200 Define all networks and sampling buffers/memories. Derive losses & learning
201 steps. Initialize the game state and algorithmic variables.
202
203 Note: batch sizes default to `None` implying that training over the full
204 dataset in memory is done by default. To sample from the memories you
205 may set these values to something less than the full capacity of the
206 memory.
207 """
208
209 def __init__(self,
210 game,
211 policy_network_layers=(256, 256),
212 advantage_network_layers=(128, 128),
213 num_iterations: int = 100,
214 num_traversals: int = 20,
215 learning_rate: float = 1e-4,
216 batch_size_advantage=None,
217 batch_size_strategy=None,
218 memory_capacity: int = int(1e6),
219 policy_network_train_steps: int = 1,
220 advantage_network_train_steps: int = 1,
221 reinitialize_advantage_networks: bool = True):
222 """Initialize the Deep CFR algorithm.
223
224 Args:
225 game: Open Spiel game.
226 policy_network_layers: (list[int]) Layer sizes of strategy net MLP.
227 advantage_network_layers: (list[int]) Layer sizes of advantage net MLP.
228 num_iterations: (int) Number of training iterations.
229 num_traversals: (int) Number of traversals per iteration.
230 learning_rate: (float) Learning rate.
231 batch_size_advantage: (int or None) Batch size to sample from advantage
232 memories.
233 batch_size_strategy: (int or None) Batch size to sample from strategy
234 memories.
235 memory_capacity: Number af samples that can be stored in memory.
236 policy_network_train_steps: Number of policy network training steps (per
237 iteration).
238 advantage_network_train_steps: Number of advantage network training steps
239 (per iteration).
240 reinitialize_advantage_networks: Whether to re-initialize the advantage
241 network before training on each iteration.
242 """
243 all_players = list(range(game.num_players()))
244 super(DeepCFRSolver, self).__init__(game, all_players)
245 self._game = game
246 if game.get_type().dynamics == pyspiel.GameType.Dynamics.SIMULTANEOUS:
247 # `_traverse_game_tree` does not take into account this option.
248 raise ValueError("Simulatenous games are not supported.")
249 self._batch_size_advantage = batch_size_advantage
250 self._batch_size_strategy = batch_size_strategy
251 self._policy_network_train_steps = policy_network_train_steps
252 self._advantage_network_train_steps = advantage_network_train_steps
253 self._num_players = game.num_players()
254 self._root_node = self._game.new_initial_state()
255 self._embedding_size = len(self._root_node.information_state_tensor(0))
256 self._num_iterations = num_iterations
257 self._num_traversals = num_traversals
258 self._reinitialize_advantage_networks = reinitialize_advantage_networks
259 self._num_actions = game.num_distinct_actions()
260 self._iteration = 1
261
262 # Define strategy network, loss & memory.
263 self._strategy_memories = ReservoirBuffer(memory_capacity)
264 self._policy_network = MLP(self._embedding_size,
265 list(policy_network_layers),
266 self._num_actions)
267 # Illegal actions are handled in the traversal code where expected payoff
268 # and sampled regret is computed from the advantage networks.
269 self._policy_sm = nn.Softmax(dim=-1)
270 self._loss_policy = nn.MSELoss()
271 self._optimizer_policy = torch.optim.Adam(
272 self._policy_network.parameters(), lr=learning_rate)
273
274 # Define advantage network, loss & memory. (One per player)
275 self._advantage_memories = [
276 ReservoirBuffer(memory_capacity) for _ in range(self._num_players)
277 ]
278 self._advantage_networks = [
279 MLP(self._embedding_size, list(advantage_network_layers),
280 self._num_actions) for _ in range(self._num_players)
281 ]
282 self._loss_advantages = nn.MSELoss(reduction="mean")
283 self._optimizer_advantages = []
284 for p in range(self._num_players):
285 self._optimizer_advantages.append(
286 torch.optim.Adam(
287 self._advantage_networks[p].parameters(), lr=learning_rate))
288 self._learning_rate = learning_rate
289
290 @property
291 def advantage_buffers(self):
292 return self._advantage_memories
293
294 @property
295 def strategy_buffer(self):
296 return self._strategy_memories
297
298 def clear_advantage_buffers(self):
299 for p in range(self._num_players):
300 self._advantage_memories[p].clear()
301
302 def reinitialize_advantage_network(self, player):
303 self._advantage_networks[player].reset()
304 self._optimizer_advantages[player] = torch.optim.Adam(
305 self._advantage_networks[player].parameters(), lr=self._learning_rate)
306
307 def reinitialize_advantage_networks(self):
308 for p in range(self._num_players):
309 self.reinitialize_advantage_network(p)

311 def solve(self):
312 """Solution logic for Deep CFR.
313
314 Traverses the game tree, while storing the transitions for training
315 advantage and policy networks.
316
317 Returns:
318 1. (nn.Module) Instance of the trained policy network for inference.
319 2. (list of floats) Advantage network losses for
320 each player during each iteration.
321 3. (float) Policy loss.
322 """
323 advantage_losses = collections.defaultdict(list)
324 for _ in range(self._num_iterations):
325 for p in range(self._num_players):
326 for _ in range(self._num_traversals):
327 self._traverse_game_tree(self._root_node, p)
328 if self._reinitialize_advantage_networks:
329 # Re-initialize advantage network for player and train from scratch.
330 self.reinitialize_advantage_network(p)
331 # Re-initialize advantage networks and train from scratch.
332 advantage_losses[p].append(self._learn_advantage_network(p))
333 self._iteration += 1
334 # Train policy network.
335 policy_loss = self._learn_strategy_network()
336 return self._policy_network, advantage_losses, policy_loss
337
338 def _traverse_game_tree(self, state, player):
339 """Performs a traversal of the game tree.
340
341 Over a traversal the advantage and strategy memories are populated with
342 computed advantage values and matched regrets respectively.
343
344 Args:
345 state: Current OpenSpiel game state.
346 player: (int) Player index for this traversal.
347
348 Returns:
349 (float) Recursively returns expected payoffs for each action.
350 """
351 expected_payoff = collections.defaultdict(float)
352 if state.is_terminal():
353 # Terminal state get returns.
354 return state.returns()[player]
355 elif state.is_chance_node():
356 # If this is a chance node, sample an action
357 chance_outcome, chance_proba = zip(*state.chance_outcomes())
358 action = np.random.choice(chance_outcome, p=chance_proba)
359 return self._traverse_game_tree(state.child(action), player)
360 elif state.current_player() == player:
361 sampled_regret = collections.defaultdict(float)
362 # Update the policy over the info set & actions via regret matching.
363 _, strategy = self._sample_action_from_advantage(state, player)
364 for action in state.legal_actions():
365 expected_payoff[action] = self._traverse_game_tree(
366 state.child(action), player)
367 cfv = 0
368 for a_ in state.legal_actions():
369 cfv += strategy[a_] * expected_payoff[a_]
370 for action in state.legal_actions():
371 sampled_regret[action] = expected_payoff[action]
372 sampled_regret[action] -= cfv
373 sampled_regret_arr = [0] * self._num_actions
374 for action in sampled_regret:
375 sampled_regret_arr[action] = sampled_regret[action]
376 self._advantage_memories[player].add(
377 AdvantageMemory(state.information_state_tensor(), self._iteration,
378 sampled_regret_arr, action))
379 return cfv
380 else:
381 other_player = state.current_player()
382 _, strategy = self._sample_action_from_advantage(state, other_player)
383 # Recompute distribution for numerical errors.
384 probs = np.array(strategy)
385 probs /= probs.sum()
386 sampled_action = np.random.choice(range(self._num_actions), p=probs)
387 self._strategy_memories.add(
388 StrategyMemory(
389 state.information_state_tensor(other_player), self._iteration,
390 strategy))
391 return self._traverse_game_tree(state.child(sampled_action), player)

393 def _sample_action_from_advantage(self, state, player):
394 """Returns an info state policy by applying regret-matching.
395
396 Args:
397 state: Current OpenSpiel game state.
398 player: (int) Player index over which to compute regrets.
399
400 Returns:
401 1. (list) Advantage values for info state actions indexed by action.
402 2. (list) Matched regrets, prob for actions indexed by action.
403 """
404 info_state = state.information_state_tensor(player)
405 legal_actions = state.legal_actions(player)
406 with torch.no_grad():
407 state_tensor = torch.FloatTensor(np.expand_dims(info_state, axis=0))
408 raw_advantages = self._advantage_networks[player](state_tensor)[0].numpy()
409 advantages = [max(0., advantage) for advantage in raw_advantages]
410 cumulative_regret = np.sum([advantages[action] for action in legal_actions])
411 matched_regrets = np.array([0.] * self._num_actions)
412 if cumulative_regret > 0.:
413 for action in legal_actions:
414 matched_regrets[action] = advantages[action] / cumulative_regret
415 else:
416 matched_regrets[max(legal_actions, key=lambda a: raw_advantages[a])] = 1
417 return advantages, matched_regrets

419 def action_probabilities(self, state):
420 """Computes action probabilities for the current player in state.
421
422 Args:
423 state: (pyspiel.State) The state to compute probabilities for.
424
425 Returns:
426 (dict) action probabilities for a single batch.
427 """
428 cur_player = state.current_player()
429 legal_actions = state.legal_actions(cur_player)
430 info_state_vector = np.array(state.information_state_tensor())
431 if len(info_state_vector.shape) == 1:
432 info_state_vector = np.expand_dims(info_state_vector, axis=0)
433 with torch.no_grad():
434 logits = self._policy_network(torch.FloatTensor(info_state_vector))
435 probs = self._policy_sm(logits).numpy()
436 return {action: probs[0][action] for action in legal_actions}

438 def _learn_advantage_network(self, player):
439 """Compute the loss on sampled transitions and perform a Q-network update.
440
441 If there are not enough elements in the buffer, no loss is computed and
442 `None` is returned instead.
443
444 Args:
445 player: (int) player index.
446
447 Returns:
448 (float) The average loss over the advantage network.
449 """
450 for _ in range(self._advantage_network_train_steps):
451
452 if self._batch_size_advantage:
453 if self._batch_size_advantage > len(self._advantage_memories[player]):
454
455 return None
456 samples = self._advantage_memories[player].sample(
457 self._batch_size_advantage)
458 else:
459 samples = self._advantage_memories[player]
460 info_states = []
461 advantages = []
462 iterations = []
463 for s in samples:
464 info_states.append(s.info_state)
465 advantages.append(s.advantage)
466 iterations.append([s.iteration])
467 # Ensure some samples have been gathered.
468 if not info_states:
469 return None
470 self._optimizer_advantages[player].zero_grad()
471 advantages = torch.FloatTensor(np.array(advantages))
472 iters = torch.FloatTensor(np.sqrt(np.array(iterations)))
473 outputs = self._advantage_networks[player](
474 torch.FloatTensor(np.array(info_states)))
475 loss_advantages = self._loss_advantages(iters * outputs,
476 iters * advantages)
477 loss_advantages.backward()
478 self._optimizer_advantages[player].step()
479
480 return loss_advantages.detach().numpy()

482 def _learn_strategy_network(self):
483 """Compute the loss over the strategy network.
484
485 Returns:
486 (float) The average loss obtained on this batch of transitions or `None`.
487 """
488 for _ in range(self._policy_network_train_steps):
489 if self._batch_size_strategy:
490 if self._batch_size_strategy > len(self._strategy_memories):
491
492 return None
493 samples = self._strategy_memories.sample(self._batch_size_strategy)
494 else:
495 samples = self._strategy_memories
496 info_states = []
497 action_probs = []
498 iterations = []
499 for s in samples:
500 info_states.append(s.info_state)
501 action_probs.append(s.strategy_action_probs)
502 iterations.append([s.iteration])
503
504 self._optimizer_policy.zero_grad()
505 iters = torch.FloatTensor(np.sqrt(np.array(iterations)))
506 ac_probs = torch.FloatTensor(np.array(np.squeeze(action_probs)))
507 logits = self._policy_network(torch.FloatTensor(np.array(info_states)))
508 outputs = self._policy_sm(logits)
509 loss_strategy = self._loss_policy(iters * outputs, iters * ac_probs)
510 loss_strategy.backward()
511 self._optimizer_policy.step()
512
513 return loss_strategy.detach().numpy()
