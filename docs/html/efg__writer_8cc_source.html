41 void EFGWriter::Write() {
42  std::ofstream efg_file(filename_);
43  efg_file << "EFG 2 R";
44  GameParameters params = game_.GetParameters();
45  efg_file << " \"" << game_.ToString() << "\" { ";
46  for (int i = 1; i <= game_.NumPlayers(); i++) {
47  // EFG player index starts at 1.
48  efg_file << '"' << "Player " << i << "\" ";
49  infostate_numbers_.push_back(std::map<std::string, int>());
50  }
51  efg_file << "}\n";
52 
53  // Get the root state.
54  Write(efg_file, *game_.NewInitialState());
55  efg_file.close();
56 }

58 void EFGWriter::Write(std::ostream& f, const State& state) {
59  if (state.IsTerminal()) {
60  f << "t \"\" ";
61  f << terminal_node_counter_;
62  terminal_node_counter_++;
63  f << " \"\" ";
64  f << "{ ";
65  for (auto r : state.Returns()) {
66  f << r << " ";
67  }
68  f << "}\n";
69  return;
70  } else if (state.IsChanceNode()) {
71  f << "c \"\" ";
72  f << chance_node_counter_;
73  chance_node_counter_++;
74  f << " \"\" ";
75  f << "{ ";
76  for (auto action_and_probs : state.ChanceOutcomes()) {
77  if (action_names_) {
78  f << '"' << state.ActionToString(action_and_probs.first) << "\" ";
79  } else {
80  f << '"' << action_and_probs.first << "\" ";
81  }
82  f << std::setprecision(10) << action_and_probs.second << " ";
83  }
84  f << "} 0\n";
85  } else {
86  int p = state.CurrentPlayer();
87  f << "p \"\" " << p + 1 << " "; // EFG player index starts at 1.
88 
89  std::string key = state.InformationStateString();
90  int idx = state.CurrentPlayer();
91  if (!separate_infostate_numbers_) idx = 0; // Only use one map.
92 
93  if (infostate_numbers_[idx].find(key) == infostate_numbers_[idx].end()) {
94  infostate_numbers_[idx][key] = infostate_numbers_[idx].size();
95  }
96  f << infostate_numbers_[idx][key] + 1; // Infostate numbering starts at 1.
97  f << " \"\" { ";
98  for (auto action : state.LegalActions()) {
99  if (action_names_) {
100  f << '"' << state.ActionToString(action) << "\" ";
101  } else {
102  f << '"' << action << "\" ";
103  }
104  }
105  f << "} 0\n";
106  }
107  for (auto action : state.LegalActions()) {
108  Write(f, *state.Child(action));
109  }
110 }
