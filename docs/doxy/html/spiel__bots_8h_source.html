26// Bots are things which can play a game. Here we define the interface for
27// various features of a bot, and some trivial uniform and fixed action bots.
28
29// Different use-cases include:
30// - play a bot versus another bot (we just need an action). This should be
31// general enough to support simultaneous games (in which case the bot needs
32// to know a player_id).
33// - restart the game (to the initial state, and an arbitrary state)
34// - interact with a bot and study its behavior, for example by looking at its
35// policy in specific states, or by accessing its action distribution. This
36// implies being able to set the bot into a specific game state.
37
38// Bots can differ, in particular with respect to:
39//
40// 1. Bot determinism.
41// - deterministic: the (state -> action) suggestion is deterministic
42// - Explicit Stochastic: the (state-> distribution over actions) is
43// deterministic and the bot exposes it
44// - Implicitly stochastic: even though the (state -> actions distribution)
45// may exist in theory, it's intractable or not implemented. Thus, the
46// (state -> action) suggestion is stochastic.
47//
48// 2. Bot statefulness. A bot can be stateless, or stateful (the policy can
49// depend on the history of states, observations and/or actions).
50

53// A simple bot that can play moves and be restarted. The bot may be stateful,
54// thus, one should restart it to provide states from a different history line.
55//
56// For simultaneous games, or for bots playing as a single player, the
57// implementation should take the player_id in the constructor.
58//
59// Optionally, the Bot can provide additional functionality (see
60// `IsOverridable` and `ProvidesPolicy`).
61// In Python, the simplest way to implement such a bot is:
62//
63// class MyBot(pyspiel.Bot):
64//
65// def __init__(self):
66// pyspiel.Bot.__init__(self)
67// # If you do implement get_policy and step_with_policy
68// def provides_force_action(self):
69// return True
70// def force_action(self, state, action):
71// ...
72class Bot {
73 public:
74 // Constructs a Bot that only supports `Step` and `Restart` (maybe RestartAt).
75 virtual ~Bot() = default;
76
77 // Asks the bot to decide on an action to play. The bot should be able to
78 // safely assumes the action was played.
79 virtual Action Step(const State& state) = 0;
80
81 // Let the bot know that a different player made an action at a given state.
82 //
83 // The state is the state at which the `player_id` player decided to take
84 // the given `action` (but before it is applied to the state). Some usage
85 // example looks like:
86 //
87 // Player current_player = state->CurrentPlayer();
88 // Action action = bots[current_player]->Step(*state);
89 // for (Player p = 0; p < num_players; ++p) {
90 // if (p != current_player) {
91 // bots[p]->InformAction(*state, current_player, action);
92 // }
93 // }
94 // state->ApplyAction(action); # We apply the action after informing bots.
95 //
96 // This is useful for stateful bots so they know that the state of the game
97 // has advanced. This should not be called for the bot that generated the
98 // action as it already knows the action it took. As most bots are not
99 // stateful, the default implementation is a no-op.
100 // This is more explicit and less error prone than having bots inspect and
101 // potentially replay the history of actions.
102 virtual void InformAction(const State& state, Player player_id,
103 Action action) {}
104 // In simultaneous move games the bot receives a vector containing the
105 // actions taken by all players in the given state.
106 virtual void InformActions(const State& state,
107 const std::vector<Action>& actions) {}
108
109 // Restarts the bot to its initial state, ready to start a new trajectory.
110 virtual void Restart() {}
111 // Configure the bot to be on the given `state` which can be arbitrary.
112 // Bot not supporting this feature can raise an error.
113 virtual void RestartAt(const State& state) {
114 SpielFatalError("RestartAt(state) not implemented.");
115 }
116
117 // Returns `true` if it is possible to force the Bot to take a specific
118 // action on playable states. In case of a stateful bot, it should correctly
119 // update its internal state.
120 virtual bool ProvidesForceAction() { return false; }
121 // Notifies the bot that it should consider that it took action action in
122 // the given state.
123 virtual void ForceAction(const State& state, Action action) {
124 if (ProvidesForceAction()) {
125 SpielFatalError(
126 "ForceAction not implemented but should because the bot is "
127 "registered as overridable.");
128 } else {
129 SpielFatalError(
130 "ForceAction not implemented because the bot is not overridable");
131 }
132 }
133
134 // Extends a bot to support explicit stochasticity, meaning that it can
135 // return a distribution over moves.
136 virtual bool ProvidesPolicy() { return false; }
137 virtual ActionsAndProbs GetPolicy(const State& state) {
138 if (ProvidesPolicy()) {
139 SpielFatalError(
140 "GetPolicy not implemented but should because the bot is registered "
141 "as exposing its policy.");
142 } else {
143 SpielFatalError(
144 "GetPolicy not implemented because the bot is not exposing any "
145 "policy.");
146 }
147 }
148 virtual std::pair<ActionsAndProbs, Action> StepWithPolicy(
149 const State& state) {
150 if (ProvidesPolicy()) {
151 SpielFatalError(
152 "StepWithPolicy not implemented but should because the bot is "
153 "registered as exposing its policy.");
154 } else {
155 SpielFatalError(
156 "StepWithPolicy not implemented because the bot is not exposing any "
157 "policy.");
158 }
159 }
160
161 // Creates a clone of the bot with an independent copy of its internal state.
162 // The original bot and the clone are completely independent.
163 // The Clone method should be as cheap to execute as possible.
164 //
165 // Important: the cloned bot must sample actions independently and differently
166 // from the original bot. I.e. if the bot uses any randomness controlling key,
167 // that key *must* be reseeded when cloning the bot.
168 // The typical use-case for cloning is generating multiple continuations
169 // of a game. The cloned bot should produce the same policy as the original
170 // bot, but there *must* be no correllation between action sampling of
171 // the original bot and its clone.
172 // Note that bot clones must also sample actions independently.
173 virtual bool IsClonable() const { return false; }
174 virtual std::unique_ptr<Bot> Clone() {
175 SpielFatalError("Clone method not implemented.");
176 }
177};
