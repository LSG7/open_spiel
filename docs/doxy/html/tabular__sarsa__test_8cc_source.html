31Action GetOptimalAction(
32 absl::flat_hash_map<std::pair<std::string, Action>, double> q_values,
33 const std::unique_ptr<State> &state) {
34 std::vector<Action> legal_actions = state->LegalActions();
35 Action optimal_action = open_spiel::kInvalidAction;
36
37 double value = -1;
38 const auto state_str = state->ToString();
39 for (const Action &action : legal_actions) {
40 double q_val = q_values[{state_str, action}];
41 if (q_val >= value) {
42 value = q_val;
43 optimal_action = action;
44 }
45 }
46 return optimal_action;
47}

49Action GetRandomAction(const std::unique_ptr<State> &state, int seed) {
50 std::vector<Action> legal_actions = state->LegalActions();
51 if (legal_actions.empty()) {
52 return kInvalidAction;
53 }
54 std::mt19937 rng(seed);
55 return legal_actions[absl::Uniform<int>(rng, 0, legal_actions.size())];
56}

58double PlayCatch(
59 absl::flat_hash_map<std::pair<std::string, Action>, double> q_values,
60 const std::unique_ptr<State> &state, double seed) {
61 // First action determines the starting column. Do the first action before the
62 // main loop, where the optimal action is chosen.
63 // Example: Initial state with random seed 42
64 // ...o.
65 // .....
66 // .....
67 // .....
68 // .....
69 // .....
70 // .....
71 // .....
72 // .....
73 // ..x..
74 std::mt19937 gen(seed);
75 std::uniform_int_distribution<int> distribution(0,
76 catch_::kDefaultColumns - 1);
77 int ball_starting_column = distribution(gen);
78 state->ApplyAction(ball_starting_column);
79
80 while (!state->IsTerminal()) {
81 Action optimal_action = GetOptimalAction(q_values, state);
82 state->ApplyAction(optimal_action);
83 }
84
85 return state->Rewards()[0];
86}

99void TabularSarsaTest_Catch_Lambda00_Loss() {
100 // Classic SARSA. No bootstraping (lambda=0.0)
101 // Player loses after only 1 train iteration.
102 std::shared_ptr<const Game> game = LoadGame("catch");
103 auto tabular_sarsa_solver = SarsaSolver(game, 0);
104
105 tabular_sarsa_solver->RunIteration();
106 const absl::flat_hash_map<std::pair<std::string, Action>, double> &q_values =
107 tabular_sarsa_solver->GetQValueTable();
108 std::unique_ptr<State> state = game->NewInitialState();
109
110 double reward = PlayCatch(q_values, state, 42);
111 SPIEL_CHECK_EQ(reward, -1);
112}

147void TabularSarsaTest_Catch_Lambda01FasterThanLambda00() {
148 // Eligibility traces (lambda > 0.0) always achieves victory with less
149 // training steps w.r.t. SARSA(lambda=0.0)
150 std::shared_ptr<const Game> game = LoadGame("catch");
151 auto tabular_sarsa_solve_lambda00 = SarsaSolver(game, 0);
152 auto tabular_sarsa_solve_lambda01 = SarsaSolver(game, 0.1);
153
154 for (int seed = 0; seed < 100; seed++) {
155 int lambda_00_train_iter = 0;
156 int lambda_01_train_iter = 0;
157 double lambda_00_reward = -1.0;
158 double lambda_01_reward = -1.0;
159
160 while (lambda_00_reward == -1.0) {
161 tabular_sarsa_solve_lambda00->RunIteration();
162 std::unique_ptr<State> state = game->NewInitialState();
163 lambda_00_reward = PlayCatch(
164 tabular_sarsa_solve_lambda00->GetQValueTable(), state, seed);
165 lambda_00_train_iter++;
166 }
167 while (lambda_01_reward == -1.0) {
168 tabular_sarsa_solve_lambda01->RunIteration();
169 std::unique_ptr<State> state = game->NewInitialState();
170 lambda_01_reward = PlayCatch(
171 tabular_sarsa_solve_lambda01->GetQValueTable(), state, seed);
172 lambda_01_train_iter++;
173 }
174 SPIEL_CHECK_GE(lambda_00_train_iter, lambda_01_train_iter);
175 }
176}

178void TabularSarsaTest_TicTacToe_Lambda01_Win() {
179 std::shared_ptr<const Game> game = open_spiel::LoadGame("tic_tac_toe");
180 auto tabular_sarsa_solver = SarsaSolver(game, 0.1);
181
182 for (int i = 1; i < 100; i++) {
183 tabular_sarsa_solver->RunIteration();
184 }
185
186 const absl::flat_hash_map<std::pair<std::string, Action>, double> &q_values =
187 tabular_sarsa_solver->GetQValueTable();
188 std::unique_ptr<State> state = game->NewInitialState();
189
190 while (!state->IsTerminal()) {
191 Action random_action = GetRandomAction(state, 42);
192 state->ApplyAction(random_action); // player 0
193 if (random_action == kInvalidAction) break;
194 state->ApplyAction(GetOptimalAction(q_values, state)); // player 1
195 }
196
197 SPIEL_CHECK_EQ(state->Rewards()[0], -1);
198}
