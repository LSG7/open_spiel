30void PrintLegalActions(const open_spiel::State& state,
31 open_spiel::Player player,
32 const std::vector<open_spiel::Action>& movelist) {
33 std::cerr << "Legal moves for player " << player << ":" << std::endl;
34 for (open_spiel::Action action : movelist) {

39int main(int argc, char** argv) {
40 absl::ParseCommandLine(argc, argv);
41
42 std::string game_name = absl::GetFlag(FLAGS_game);

48 // Print out registered games.

51 for (const std::string& name : names) {
52 std::cerr << name << std::endl;
53 }
54
55 // Random number generator.
56 std::mt19937 rng(seed ? seed : time(0));
57
58 // Create the game.
59 std::cerr << "Creating game..\n" << std::endl;
60
61 // Add any specified parameters to override the defaults.
62 open_spiel::GameParameters params;
63 if (players > 0) {
64 params["players"] = open_spiel::GameParameter(players);
65 }
66 std::shared_ptr<const open_spiel::Game> game =
67 open_spiel::LoadGame(game_name, params);
68
69 if (!game) {
70 std::cerr << "problem with loading game, exiting..." << std::endl;
71 return -1;
72 }
73
74 std::cerr << "Starting new game..." << std::endl;
75 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
76
77 std::cerr << "Initial state:" << std::endl;
78 std::cerr << "State:" << std::endl << state->ToString() << std::endl;
79
80 while (!state->IsTerminal()) {
81 std::cerr << "player " << state->CurrentPlayer() << std::endl;
82
83 if (state->IsChanceNode()) {
84 // Chance node; sample one according to underlying distribution.
85 std::vector<std::pair<open_spiel::Action, double>> outcomes =
86 state->ChanceOutcomes();
87 open_spiel::Action action = open_spiel::SampleAction(outcomes, rng).first;
88 std::cerr << "sampled outcome: "
89 << state->ActionToString(open_spiel::kChancePlayerId, action)
90 << std::endl;
91 state->ApplyAction(action);
92 } else if (state->IsMeanFieldNode()) {
93 int num_states_distribution = state->DistributionSupport().size();
94 state->UpdateDistribution(std::vector<double>(
95 num_states_distribution,
96 num_states_distribution > 0 ? 1.0 / num_states_distribution : 1.0));
97 std::cerr << "Call update distribution on a uniform distribution of "
98 << num_states_distribution << " states (length of "
99 << "DistributionSupport" << std::endl;
100 } else if (state->IsSimultaneousNode()) {
101 // open_spiel::Players choose simultaneously?
102 std::vector<open_spiel::Action> joint_action;
103 std::vector<float> infostate(game->InformationStateTensorSize());
104
105 // Sample a action for each player
106 for (auto player = open_spiel::Player{0}; player < game->NumPlayers();
107 ++player) {
108 if (show_infostate) {
109 if (game->GetType().provides_information_state_tensor) {
110 state->InformationStateTensor(player, absl::MakeSpan(infostate));
111 std::cerr << "player " << player << ": "
112 << absl::StrJoin(infostate, " ") << std::endl;
113 }
114 if (game->GetType().provides_information_state_string) {
115 std::cerr << "player " << player << ": "
116 << state->InformationStateString(player) << std::endl;
117 }
118 }
119
120 std::vector<open_spiel::Action> actions = state->LegalActions(player);
121 if (show_legals) {
122 PrintLegalActions(*state, player, actions);
123 }
124
125 open_spiel::Action action = 0;
126 if (!actions.empty()) {
127 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
128 action = actions[dis(rng)];
129 }
130 joint_action.push_back(action);
131 std::cerr << "player " << player << " chose "
132 << state->ActionToString(player, action) << std::endl;
133 }
134
135 state->ApplyActions(joint_action);
136 } else {
137 // Decision node, sample one uniformly.
138 auto player = state->CurrentPlayer();
139 if (show_infostate) {
140 if (game->GetType().provides_information_state_tensor) {
141 std::vector<float> infostate;
142 state->InformationStateTensor(player, absl::MakeSpan(infostate));
143 std::cerr << "player " << player << ": "
144 << absl::StrJoin(infostate, " ") << std::endl;
145 }
146 if (game->GetType().provides_information_state_string) {
147 std::cerr << "player " << player << ": "
148 << state->InformationStateString(player) << std::endl;
149 }
150 }
151
152 std::vector<open_spiel::Action> actions = state->LegalActions();
153 if (show_legals) {
154 PrintLegalActions(*state, player, actions);
155 }
156
157 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
158 auto action = actions[dis(rng)];
159 std::cerr << "chose action: " << state->ActionToString(player, action)
160 << std::endl;
161 state->ApplyAction(action);
162 }
163
164 std::cerr << "State: " << std::endl << state->ToString() << std::endl;
165 }
166
167 auto returns = state->Returns();
168 for (auto p = open_spiel::Player{0}; p < game->NumPlayers(); p++) {
169 std::cerr << "Final return to player " << p << " is " << returns[p]
170 << std::endl;
171 }
