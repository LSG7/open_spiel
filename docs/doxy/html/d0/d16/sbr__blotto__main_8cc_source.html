54BaseSamplerType GetBaseSamplerType(const std::string& str) {
55 if (str == "uniform") {
56 return BaseSamplerType::kBaseUniform;
57 } else if (str == "latest") {
58 return BaseSamplerType::kBaseLatest;
59 } else {
60 open_spiel::SpielFatalError("Unrecognized base sampler type.");
61 }
62}
63
64CandidatesSamplerType GetCandidatesSamplerType(const std::string& str) {
65 if (str == "initial") {
66 return CandidatesSamplerType::kCandidatesInitial;
67 } else if (str == "uniform") {
68 return CandidatesSamplerType::kCandidatesUniform;
69 } else if (str == "latest") {
70 return CandidatesSamplerType::kCandidatesLatest;
71 } else if (str == "mixedIU") {
72 return CandidatesSamplerType::kCandidatesInitialUniform;
73 } else if (str == "mixedIL") {
74 return CandidatesSamplerType::kCandidatesInitialLatest;
75 } else {
76 open_spiel::SpielFatalError("Unrecognized candidates sampler type.");
77 }
78}

80int main(int argc, char** argv) {
81 open_spiel::Init(argv[0], &argc, &argv, /*remove_flags=*/true);
82
83 int players = absl::GetFlag(FLAGS_players);
84 int coins = absl::GetFlag(FLAGS_coins);
85 int fields = absl::GetFlag(FLAGS_fields);
86
87 absl::ParseCommandLine(argc, argv);
88
89 std::string game_string = "";
90 if (absl::GetFlag(FLAGS_game).empty()) {
91 game_string = absl::StrCat(
92 "turn_based_simultaneous_game(game=blotto(players=", players,
93 ",coins=", coins, ",fields=", fields, "))");
94 } else {
95 game_string = absl::StrCat(
96 "turn_based_simultaneous_game(game=", absl::GetFlag(FLAGS_game), ")");
97 }
98
99 std::cout << "game string: " << game_string << std::endl;
100 std::shared_ptr<const open_spiel::Game> game =
101 open_spiel::LoadGame(game_string);
102 SPIEL_CHECK_TRUE(game != nullptr);
103
104 std::cout << "NumDistinctActions: " << game->NumDistinctActions()
105 << std::endl;
106 std::cout << "Size of game: " << std::fixed
107 << std::pow(game->NumDistinctActions(), game->NumPlayers())
108 << std::endl;
109
110 FictitiousPlayProcess fpp(game, absl::GetFlag(FLAGS_seed),
111 absl::GetFlag(FLAGS_randomize_initial_policies));
112 int iterations = absl::GetFlag(FLAGS_iterations);
113 std::string algo = absl::GetFlag(FLAGS_algorithm);
114 int sbr_num_base = absl::GetFlag(FLAGS_sbr_b);
115 int sbr_num_candidates = absl::GetFlag(FLAGS_sbr_c);
116 int brpi_n = absl::GetFlag(FLAGS_brpi_n);
117 double lambda = absl::GetFlag(FLAGS_lambda);
118 BaseSamplerType base_sampler_type =
119 GetBaseSamplerType(absl::GetFlag(FLAGS_base_sampler));
120 CandidatesSamplerType candidates_sampler_type =
121 GetCandidatesSamplerType(absl::GetFlag(FLAGS_candidates_sampler));
122
123 std::unique_ptr<open_spiel::file::File> logfile = nullptr;
124 if (absl::GetFlag(FLAGS_enable_log)) {
125 std::cout << "Opening log file.." << std::endl;
126
127 std::string dir = absl::StrCat(absl::GetFlag(FLAGS_logdirpref), "/",
128 absl::GetFlag(FLAGS_run_name));
129
130 std::string filename = absl::StrCat(dir, "/blotto_", players, "_", coins,
131 "_", fields, "_", algo);
132
133 if (absl::GetFlag(FLAGS_randomize_initial_policies)) {
134 absl::StrAppend(&filename, "_rip");
135 }
136
137 absl::StrAppend(&filename, "_seed", absl::GetFlag(FLAGS_seed));
138
139 if (algo == "sbr") {
140 absl::StrAppend(&filename, "_", sbr_num_base, "_", sbr_num_candidates);
141 } else if (algo == "sfp") {
142 absl::StrAppend(&filename, "_lambda", lambda);
143 } else if (algo == "brpi") {
144 absl::StrAppend(&filename, "_", sbr_num_base, "_", sbr_num_candidates,
145 "_", absl::GetFlag(FLAGS_base_sampler), "_",
146 absl::GetFlag(FLAGS_candidates_sampler));
147 }
148
149 if (!open_spiel::file::Exists(dir)) {
150 std::cout << "Creating log directory " << dir << std::endl;
151 SPIEL_CHECK_TRUE(open_spiel::file::Mkdir(dir));
152 }
153
154 logfile = std::make_unique<open_spiel::file::File>(filename, "w");
155 }
156
157 std::cout << "Starting." << std::endl;
158
159 int next_br_iter = 1;
160 for (int i = 1; i < iterations || iterations < 0; ++i) {
161 if (algo == "fp") {
162 fpp.FullFPIteration();
163 } else if (algo == "sbr") {
164 fpp.SBRIteration(sbr_num_base, sbr_num_candidates);
165 } else if (algo == "ibr") {
166 fpp.IBRIteration();
167 } else if (algo == "meibr") {
168 fpp.MaxEntIBRIteration();
169 } else if (algo == "sfp") {
170 fpp.SFPIteration(lambda);
171 } else if (algo == "brpi") {
172 fpp.BRPIIteration(base_sampler_type, candidates_sampler_type,
173 sbr_num_base, sbr_num_candidates, brpi_n);
174 } else {
175 std::cerr << "Unrecognized algorithm. Exiting...";
176 exit(-1);
177 }
178
179 if (i == next_br_iter) {
180 double cce_dist = fpp.CCEDist();
181 double nash_conv = fpp.NashConv();
182
183 absl::Duration total_time = fpp.TotalTime();
184
185 std::string outline =
186 absl::StrCat(i, " ", absl::ToDoubleSeconds(total_time), " ", cce_dist,
187 " ", nash_conv, "\n");
188 std::cout << outline;
189 if (logfile != nullptr) {
190 SPIEL_CHECK_TRUE(logfile->Write(outline));
191 SPIEL_CHECK_TRUE(logfile->Flush());
192 }
193
194 next_br_iter *= 2;
195 }
196 }
197}
