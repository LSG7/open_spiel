75  // Returns a signal id, which corresponds to a specific policy used by the
76  // specified player in the joint policy at the specified recommendation index
77  // in the correlation device. This method makes use of a table that maintains
78  // these mappings initialized at construction time.
79  int GetSignalId(int rec_index, Player player) const {
80  const auto iter = recidx_player_to_signal_id_.find({rec_index, player});
81  SPIEL_CHECK_TRUE(iter != recidx_player_to_signal_id_.end());
82  return iter->second;
83  }
84 
85  int NumDistinctActions() const override { return orig_num_distinct_actions_; }
86 
87  private:
88  const CorrDistConfig config_;
89  const CorrelationDevice& mu_;
90 
91  // Number of distinct actions in the original game.
92  int orig_num_distinct_actions_;
93 
94  // To compute a correlated equilibria, we need to map individual player
95  // strategies to signal ids that are handed out by the recommender at the
96  // start of the game. These signal ids get tacked onto the information state
97  // strings so that the best response is computed conditionally on the signal.
98  //
99  // Keeps track of the number of signal id's per player.
