24class DistributionTest(absltest.TestCase):
25
26 def test_basic(self):
27 game = pyspiel.load_game("python_mfg_crowd_modelling")
28 uniform_policy = policy.UniformRandomPolicy(game)
29 dist = distribution.DistributionPolicy(game, uniform_policy)
30 state = game.new_initial_state().child(0)
31 self.assertAlmostEqual(dist.value(state), 1 / game.size)
32

36 "initial_distribution_value": "[1.]",
37 })
38 uniform_policy = policy.UniformRandomPolicy(game)

41 def test_multi_pop(self):
42 game = pyspiel.load_game("python_mfg_predator_prey")
43 self.assertEqual(game.num_players(), 3)
44 uniform_policy = policy.UniformRandomPolicy(game)
45 dist = distribution.DistributionPolicy(game, uniform_policy)

47 self.assertAlmostEqual(
