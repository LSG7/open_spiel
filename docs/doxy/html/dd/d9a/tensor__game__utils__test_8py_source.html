17from absl.testing import absltest

22class TensorGamesUtilsTest(absltest.TestCase):

24 def test_extensive_to_tensor_game_type(self):
25 game = pyspiel.extensive_to_tensor_game(
26 pyspiel.load_game(
27 "turn_based_simultaneous_game(game=blotto(players=3,coins=5))"))
28 game_type = game.get_type()
29 self.assertEqual(game_type.dynamics, pyspiel.GameType.Dynamics.SIMULTANEOUS)
30 self.assertEqual(game_type.chance_mode,
31 pyspiel.GameType.ChanceMode.DETERMINISTIC)
32 self.assertEqual(game_type.information,
33 pyspiel.GameType.Information.ONE_SHOT)
34 self.assertEqual(game_type.utility, pyspiel.GameType.Utility.ZERO_SUM)

36 def test_extensive_to_tensor_game_payoff_tensor(self):
37 turn_based_game = pyspiel.load_game_as_turn_based(
38 "blotto(players=3,coins=5)")
39 tensor_game1 = pyspiel.extensive_to_tensor_game(turn_based_game)
40 tensor_game2 = pyspiel.load_tensor_game("blotto(players=3,coins=5)")
41 self.assertEqual(tensor_game1.shape(), tensor_game2.shape())
42 s0 = turn_based_game.new_initial_state()
43 self.assertEqual(tensor_game1.shape()[0], s0.num_distinct_actions())
44 for a0 in range(s0.num_distinct_actions()):
45 s1 = s0.child(a0)
46 self.assertEqual(tensor_game1.shape()[1], s1.num_distinct_actions())
47 for a1 in range(s1.num_distinct_actions()):
48 s2 = s1.child(a1)
49 self.assertEqual(tensor_game1.shape()[2], s2.num_distinct_actions())
50 for a2 in range(s2.num_distinct_actions()):
51 s3 = s2.child(a2)
52 self.assertTrue(s3.is_terminal())
53 for player in range(3):
54 self.assertEqual(
55 s3.returns()[player],
56 tensor_game1.player_utility(player, (a0, a1, a2)))
57 self.assertEqual(
58 s3.returns()[player],
59 tensor_game2.player_utility(player, (a0, a1, a2)))
