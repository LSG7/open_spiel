121 BreakthroughState::BreakthroughState(std::shared_ptr<const Game> game, int rows,
122  int cols)
123  : State(game), rows_(rows), cols_(cols) {
124  SPIEL_CHECK_GT(rows_, 1);
125  SPIEL_CHECK_GT(cols_, 1);
126 
127  board_ = std::vector<CellState>(rows_ * cols_, CellState::kEmpty);
128  for (int r = 0; r < rows_; r++) {
129  for (int c = 0; c < cols_; c++) {
130  // Only use two rows if there are at least 6 rows.
131  if (r == 0 || (rows_ >= kNumRowsForFullPieces && r == 1)) {
132  SetBoard(r, c, CellState::kBlack);
133  } else if (r == (rows_ - 1) ||
134  (rows_ >= kNumRowsForFullPieces && r == (rows_ - 2))) {
135  SetBoard(r, c, CellState::kWhite);
136  }
137  }
138  }
139 
140  winner_ = kInvalidPlayer;
141  pieces_[0] = pieces_[1] = cols_ * (rows_ >= kNumRowsForFullPieces ? 2 : 1);
142  cur_player_ = 0;
143  total_moves_ = 0;
144 }

154 void BreakthroughState::DoApplyAction(Action action) {
155  std::vector<int> values =
156  UnrankActionMixedBase(action, {rows_, cols_, kNumDirections, 2});
157  int r1 = values[0];
158  int c1 = values[1];
159  int dir = values[2];
160  bool capture = values[3] == 1;
161  int r2 = r1 + kDirRowOffsets[dir];
162  int c2 = c1 + kDirColOffsets[dir];
163 
164  SPIEL_CHECK_TRUE(InBounds(r1, c1));
165  SPIEL_CHECK_TRUE(InBounds(r2, c2));
166 
167  // Remove pieces if captured.
168  if (board(r2, c2) == CellState::kWhite) {
169  pieces_[StateToPlayer(CellState::kWhite)]--;
170  SPIEL_CHECK_EQ(board(r1, c1), CellState::kBlack);
171  SPIEL_CHECK_EQ(cur_player_, StateToPlayer(CellState::kBlack));
172  } else if (board(r2, c2) == CellState::kBlack) {
173  pieces_[StateToPlayer(CellState::kBlack)]--;
174  SPIEL_CHECK_EQ(board(r1, c1), CellState::kWhite);
175  SPIEL_CHECK_EQ(cur_player_, StateToPlayer(CellState::kWhite));
176  }
177 
178  // Move the piece.
179  if (capture) {
180  SPIEL_CHECK_EQ(board(r2, c2), OpponentState(board(r1, c1)));
181  }
182  SetBoard(r2, c2, board(r1, c1));
183  SetBoard(r1, c1, CellState::kEmpty);
184 
185  // Check for winner.
186  if (cur_player_ == 0 && r2 == (rows_ - 1)) {
187  winner_ = 0;
188  } else if (cur_player_ == 1 && r2 == 0) {
189  winner_ = 1;
190  }
191 
192  cur_player_ = NextPlayerRoundRobin(cur_player_, kNumPlayers);
193  total_moves_++;
194 }

219 std::vector<Action> BreakthroughState::LegalActions() const {
220  std::vector<Action> movelist;
221  if (IsTerminal()) return movelist;
222  const Player player = CurrentPlayer();
223  CellState mystate = PlayerToState(player);
224  std::vector<int> action_bases = {rows_, cols_, kNumDirections, 2};
225  std::vector<int> action_values = {0, 0, 0, 0};
226 
227  for (int r = 0; r < rows_; r++) {
228  for (int c = 0; c < cols_; c++) {
229  if (board(r, c) == mystate) {
230  for (int o = 0; o < kNumDirections / 2; o++) {
231  int dir = player * kNumDirections / 2 + o;
232  int rp = r + kDirRowOffsets[dir];
233  int cp = c + kDirColOffsets[dir];
234 
235  if (InBounds(rp, cp)) {
236  action_values[0] = r;
237  action_values[1] = c;
238  action_values[2] = dir;
239  if (board(rp, cp) == CellState::kEmpty) {
240  // Regular move.
241  action_values[3] = 0;
242  movelist.push_back(
243  RankActionMixedBase(action_bases, action_values));
244  } else if ((o == 0 || o == 2) &&
245  board(rp, cp) == OpponentState(mystate)) {
246  // Capture move (can only capture diagonally)
247  action_values[3] = 1;
248  movelist.push_back(
249  RankActionMixedBase(action_bases, action_values));
250  }
251  }
252  }
253  }
254  }
255  }
256 
257  return movelist;
258 }

344 void BreakthroughState::UndoAction(Player player, Action action) {
345  std::vector<int> values =
346  UnrankActionMixedBase(action, {rows_, cols_, kNumDirections, 2});
347  int r1 = values[0];
348  int c1 = values[1];
349  int dir = values[2];
350  bool capture = values[3] == 1;
351  int r2 = r1 + kDirRowOffsets[dir];
352  int c2 = c1 + kDirColOffsets[dir];
353 
354  cur_player_ = PreviousPlayerRoundRobin(cur_player_, 2);
355  total_moves_--;
356 
357  // Undo win status.
358  winner_ = kInvalidPlayer;
359 
360  // Move back the piece, and put back the opponent's piece if necessary.
361  // The move is (r1, c1) -> (r2, c2) where r is row and c is column.
362  SetBoard(r1, c1, board(r2, c2));
363  SetBoard(r2, c2, CellState::kEmpty);
364  if (capture) {
365  if (board(r1, c1) == CellState::kWhite) {
366  // It was a white move: put back the black piece.
367  SetBoard(r2, c2, CellState::kBlack);
368  pieces_[kBlackPlayerId]++;
369  } else if (board(r1, c1) == CellState::kBlack) {
370  // It was a black move: put back the white piece.
371  SetBoard(r2, c2, CellState::kWhite);
372  pieces_[kWhitePlayerId]++;
373  }
374  }
375  history_.pop_back();
376  --move_number_;
377 }

402 std::unique_ptr<State> BreakthroughGame::DeserializeState(
403  const std::string& str) const {
404  std::unique_ptr<State> state = NewInitialState();
405 
406  if (str.length() != rows_ * cols_) {
407  SpielFatalError("Incorrect number of characters in string.");
408  return std::unique_ptr<State>();
409  }
410 
411  BreakthroughState* bstate = dynamic_cast<BreakthroughState*>(state.get());
412 
413  bstate->SetPieces(0, 0);
414  bstate->SetPieces(1, 0);
415  int i = 0;
416  for (int r = 0; r < rows_; r++) {
417  for (int c = 0; c < cols_; c++) {
418  if (str.at(i) == 'b') {
419  bstate->SetPieces(0, bstate->pieces(0) + 1);
420  bstate->SetBoard(r, c, CellState::kBlack);
421  } else if (str.at(i) == 'w') {
422  bstate->SetPieces(1, bstate->pieces(1) + 1);
423  bstate->SetBoard(r, c, CellState::kWhite);
424  } else if (str.at(i) == '.') {
425  bstate->SetBoard(r, c, CellState::kEmpty);
426  } else {
427  std::string error = "Invalid character in std::string: ";
428  error += str.at(i);
429  SpielFatalError(error);
430  return std::unique_ptr<State>();
431  }
432 
433  i++;
434  }
435  }
436 
437  return state;
438 }
