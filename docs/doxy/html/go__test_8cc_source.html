52void ConcreteActionsAreUsedInTheAPI() {
53 int board_size = 13;
54 std::shared_ptr<const Game> game =
55 LoadGame("go", {{"board_size", open_spiel::GameParameter(board_size)}});
56 std::unique_ptr<State> state = game->NewInitialState();
57
58 SPIEL_CHECK_EQ(state->NumDistinctActions(), board_size * board_size + 1);
59 SPIEL_CHECK_EQ(state->LegalActions().size(), state->NumDistinctActions());
60 for (Action action : state->LegalActions()) {
61 SPIEL_CHECK_GE(action, 0);
62 SPIEL_CHECK_LE(action, board_size * board_size);
63 }
64}
