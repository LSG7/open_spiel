44void XinxinBot_GenerateGames(int num_games, std::string path, int uct_num_runs,
45 double uct_c_val, int iimc_num_worlds,
46 bool use_threads) {
47 std::mt19937 rng(Seed());
48 std::ofstream game_logs(path);
49 auto game = open_spiel::LoadGame("hearts");
50 std::vector<std::unique_ptr<open_spiel::Bot>> bots;
51 std::vector<Bot *> bot_ptrs;
52
53 for (int i = 0; i < hearts::kNumPlayers; i++) {
54 bots.push_back(open_spiel::hearts::MakeXinxinBot(
55 game->GetParameters(), uct_num_runs, uct_c_val, iimc_num_worlds,
56 use_threads));
57 bot_ptrs.push_back(bots.back().get());
58 }
59
60 for (int i = 0; i < num_games; i++) {
61 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
62 const int num_players = bots.size();
63 for (auto bot : bot_ptrs) bot->Restart();
64 while (!state->IsTerminal()) {
65 if (state->IsChanceNode()) {
66 Action action = SampleAction(state->ChanceOutcomes(), rng).first;
67 for (auto bot : bot_ptrs)
68 bot->InformAction(*state, kChancePlayerId, action);
69 state->ApplyAction(action);
70 } else {
71 Player current_player = state->CurrentPlayer();
72 Action action = bots[current_player]->Step(*state);
73 for (Player p = 0; p < num_players; ++p) {
74 if (p != current_player) {
75 bots[p]->InformAction(*state, current_player, action);
76 }
77 }
78 state->ApplyAction(action);
79 }
80 }
81 for (Player p = 0; p < num_players; ++p) {
82 // allows checking for differences in the returns
83 bots[p]->InformAction(*state, kTerminalPlayerId, kInvalidAction);
84 }
85 game_logs << absl::StrJoin(state->History(), " ") << "\n";
86 }
87 game_logs.close();
88}
