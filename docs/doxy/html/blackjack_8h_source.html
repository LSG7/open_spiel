97 double MinUtility() const override { return -1; }
98 double MaxUtility() const override { return +1; }
99 std::vector<int> ObservationTensorShape() const override {
100 return {
101 NumPlayers() + 1 + // turn (incl. chance)
102 1 + // is terminal?
103 (kNumSuits + 1) * (NumPlayers() + 1) + // num_aces_ for every player
104 kDeckSize * (NumPlayers() + 1) // many-hot of the cards for each player
105 };
106 };
