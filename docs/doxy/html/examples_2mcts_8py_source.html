77def _init_bot(bot_type, game, player_id):
78 """Initializes a bot by type."""
79 rng = np.random.RandomState(FLAGS.seed)
80 if bot_type == "mcts":
81 evaluator = mcts.RandomRolloutEvaluator(FLAGS.rollout_count, rng)
82 return mcts.MCTSBot(
83 game,
84 FLAGS.uct_c,
85 FLAGS.max_simulations,
86 evaluator,
87 random_state=rng,
88 solve=FLAGS.solve,
89 verbose=FLAGS.verbose)
90 if bot_type == "az":
91 model = az_model.Model.from_checkpoint(FLAGS.az_path)
92 evaluator = az_evaluator.AlphaZeroEvaluator(game, model)
93 return mcts.MCTSBot(
94 game,
95 FLAGS.uct_c,
96 FLAGS.max_simulations,
97 evaluator,
98 random_state=rng,
99 child_selection_fn=mcts.SearchNode.puct_value,
100 solve=FLAGS.solve,
101 verbose=FLAGS.verbose)
102 if bot_type == "random":
103 return uniform_random.UniformRandomBot(player_id, rng)
104 if bot_type == "human":
105 return human.HumanBot()
106 if bot_type == "gtp":
107 bot = gtp.GTPBot(game, FLAGS.gtp_path)
108 for cmd in FLAGS.gtp_cmd:
109 bot.gtp_cmd(cmd)
110 return bot
111 raise ValueError("Invalid bot type: %s" % bot_type)

114def _get_action(state, action_str):
115 for action in state.legal_actions():
116 if action_str == state.action_to_string(state.current_player(), action):
117 return action
118 return None

121def _play_game(game, bots, initial_actions):
122 """Plays one game."""
123 state = game.new_initial_state()
124 _opt_print("Initial state:\n{}".format(state))
125
126 history = []
127
128 if FLAGS.random_first:
129 assert not initial_actions
130 initial_actions = [state.action_to_string(
131 state.current_player(), random.choice(state.legal_actions()))]
132
133 for action_str in initial_actions:
134 action = _get_action(state, action_str)
135 if action is None:
136 sys.exit("Invalid action: {}".format(action_str))
137
138 history.append(action_str)
139 for bot in bots:
140 bot.inform_action(state, state.current_player(), action)
141 state.apply_action(action)
142 _opt_print("Forced action", action_str)
143 _opt_print("Next state:\n{}".format(state))
144
145 while not state.is_terminal():
146 current_player = state.current_player()
147 # The state can be three different types: chance node,
148 # simultaneous node, or decision node
149 if state.is_chance_node():
150 # Chance node: sample an outcome
151 outcomes = state.chance_outcomes()
152 num_actions = len(outcomes)
153 _opt_print("Chance node, got " + str(num_actions) + " outcomes")
154 action_list, prob_list = zip(*outcomes)
155 action = np.random.choice(action_list, p=prob_list)
156 action_str = state.action_to_string(current_player, action)
157 _opt_print("Sampled action: ", action_str)
158 elif state.is_simultaneous_node():
159 raise ValueError("Game cannot have simultaneous nodes.")
160 else:
161 # Decision node: sample action for the single current player
162 bot = bots[current_player]
163 action = bot.step(state)
164 action_str = state.action_to_string(current_player, action)
165 _opt_print("Player {} sampled action: {}".format(current_player,
166 action_str))
167
168 for i, bot in enumerate(bots):
169 if i != current_player:
170 bot.inform_action(state, current_player, action)
171 history.append(action_str)
172 state.apply_action(action)
173
174 _opt_print("Next state:\n{}".format(state))
175
176 # Game is now done. Print return for each player
177 returns = state.returns()
178 print("Returns:", " ".join(map(str, returns)), ", Game actions:",
179 " ".join(history))
180
181 for bot in bots:
182 bot.restart()
183
184 return returns, history

187def main(argv):
188 game = pyspiel.load_game(FLAGS.game)
189 if game.num_players() > 2:
190 sys.exit("This game requires more players than the example can handle.")
191 bots = [
192 _init_bot(FLAGS.player1, game, 0),
193 _init_bot(FLAGS.player2, game, 1),
194 ]
195 histories = collections.defaultdict(int)
196 overall_returns = [0, 0]
197 overall_wins = [0, 0]
198 game_num = 0
199 try:
200 for game_num in range(FLAGS.num_games):
201 returns, history = _play_game(game, bots, argv[1:])
202 histories[" ".join(history)] += 1
203 for i, v in enumerate(returns):
204 overall_returns[i] += v
205 if v > 0:
206 overall_wins[i] += 1
207 except (KeyboardInterrupt, EOFError):
208 game_num -= 1
209 print("Caught a KeyboardInterrupt, stopping early.")
210 print("Number of games played:", game_num + 1)
211 print("Number of distinct games played:", len(histories))
212 print("Players:", FLAGS.player1, FLAGS.player2)
213 print("Overall wins", overall_wins)
214 print("Overall returns", overall_returns)
