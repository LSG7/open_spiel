16"""Wraps third-party bridge bots to make them usable in OpenSpiel.
17
18This code enables OpenSpiel interoperation for bots which implement the BlueChip
19bridge protocol. This is widely used, e.g. in the World computer bridge
20championships. For a rough outline of the protocol, see:
21http://www.bluechipbridge.co.uk/protocol.htm
22
23No formal specification is available. This implementation has been verified
24to work correctly with WBridge5.
25
26This bot controls a single player in the full game of bridge, including both the
27bidding and play phase. It chooses its actions by invoking an external bot which
28plays the full game of bridge. This means that each time the bot is asked for an
29action, it sends up to three actions (one for each other player) to the external
30bridge bot, and obtains an action in return.
31"""

58# Template regular expressions for messages we receive
59_CONNECT = 'Connecting "(?P<client_name>.*)" as ANYPL using protocol version 18'
60_PLAYER_ACTION = ("(?P<seat>NORTH|SOUTH|EAST|WEST) "
61 "((?P<pass>PASSES)|(?P<dbl>DOUBLES)|(?P<rdbl>REDOUBLES)|bids "
62 "(?P<bid>[^ ]*)|(plays (?P<play>[23456789tjqka][cdhs])))"
63 "(?P<alert> Alert.)?")
64_READY_FOR_OTHER = ("{seat} ready for "
65 "(((?P<other>[^']*)'s ((bid)|(card to trick \\d+)))"
66 "|(?P<dummy>dummy))")

69_READY_FOR_TEAMS = "{seat} ready for teams"
70_READY_TO_START = "{seat} ready to start"
71_READY_FOR_DEAL = "{seat} ready for deal"
72_READY_FOR_CARDS = "{seat} ready for cards"
73_READY_FOR_BID = "{seat} ready for {other}'s bid"

75# Templates for messages we send
76_SEATED = '{seat} ("{client_name}") seated'
77_TEAMS = 'Teams: N/S "north-south" E/W "east-west"'
78_START_BOARD = "start of board"
79_DEAL = "Board number {board}. Dealer NORTH. Neither vulnerable."
80_CARDS = "{seat}'s cards: {hand}"
81_OTHER_PLAYER_ACTION = "{player} {action}"
82_PLAYER_TO_LEAD = "{seat} to lead"
83_DUMMY_CARDS = "Dummy's cards: {}"

85# BlueChip bridge protocol message constants
86_SEATS = ["NORTH", "EAST", "SOUTH", "WEST"]
87_TRUMP_SUIT = ["C", "D", "H", "S", "NT"]
88_NUMBER_TRUMP_SUITS = len(_TRUMP_SUIT)
89_SUIT = _TRUMP_SUIT[:4]
90_NUMBER_SUITS = len(_SUIT)
91_RANKS = ["2", "3", "4", "5", "6", "7", "8", "9", "T", "J", "Q", "K", "A"]
92_LSUIT = [x.lower() for x in _SUIT]
93_LRANKS = [x.lower() for x in _RANKS]

95# OpenSpiel action ids
96_ACTION_PASS = 52
97_ACTION_DBL = 53
98_ACTION_RDBL = 54
99_ACTION_BID = 55 # First bid, i.e. 1C

102def _bid_to_action(action_str):
103 """Returns an OpenSpiel action id (an integer) from a BlueChip bid string."""
104 level = int(action_str[0])
105 trumps = _TRUMP_SUIT.index(action_str[1:])
106 return _ACTION_BID + (level - 1) * _NUMBER_TRUMP_SUITS + trumps
107
108
109def _play_to_action(action_str):
110 """Returns an OpenSpiel action id (an integer) from a BlueChip card string."""
111 rank = _LRANKS.index(action_str[0])
112 suit = _LSUIT.index(action_str[1])
113 return rank * _NUMBER_SUITS + suit

116def _action_to_string(action):
117 """Converts OpenSpiel action id (an integer) to a BlueChip action string.
118
119 Args:
120 action: an integer action id corresponding to a bid.
121
122 Returns:
123 A string in BlueChip format, e.g. 'PASSES' or 'bids 1H', or 'plays ck'.
124 """
125 if action == _ACTION_PASS:
126 return "PASSES"
127 elif action == _ACTION_DBL:
128 return "DOUBLES"
129 elif action == _ACTION_RDBL:
130 return "REDOUBLES"
131 elif action >= _ACTION_BID:
132 level = str((action - _ACTION_BID) // _NUMBER_TRUMP_SUITS + 1)
133 trumps = _TRUMP_SUIT[(action - _ACTION_BID) % _NUMBER_TRUMP_SUITS]
134 return "bids " + level + trumps
135 else:
136 rank = action // _NUMBER_SUITS
137 suit = action % _NUMBER_SUITS
138 return "plays " + _LRANKS[rank] + _LSUIT[suit]

141def _expect_regex(controller, regex):
142 """Reads a line from the controller, parses it using the regular expression."""
143 line = controller.read_line()
144 match = re.match(regex, line)
145 if not match:
146 raise ValueError("Received '{}' which does not match regex '{}'".format(
147 line, regex))
148 return match.groupdict()

151def _expect(controller, expected):
152 """Reads a line from the controller, checks it matches expected line exactly."""
153 line = controller.read_line()
154 if expected != line:
155 raise ValueError("Received '{}' but expected '{}'".format(line, expected))

158def _hand_string(cards):
159 """Returns the hand of the to-play player in the state in BlueChip format."""
160 if len(cards) != 13:
161 raise ValueError("Must have 13 cards")
162 suits = [[] for _ in range(4)]
163 for card in reversed(sorted(cards)):
164 suit = card % 4
165 rank = card // 4
166 suits[suit].append(_RANKS[rank])
167 for i in range(4):
168 if suits[i]:
169 suits[i] = _TRUMP_SUIT[i] + " " + " ".join(suits[i]) + "."
170 else:
171 suits[i] = _TRUMP_SUIT[i] + " -."
172 return " ".join(suits)

175def _connect(controller, seat):
176 """Performs the initial handshake with a BlueChip bot."""
177 client_name = _expect_regex(controller, _CONNECT)["client_name"]
178 controller.send_line(_SEATED.format(seat=seat, client_name=client_name))
179 _expect(controller, _READY_FOR_TEAMS.format(seat=seat))
180 controller.send_line(_TEAMS)
181 _expect(controller, _READY_TO_START.format(seat=seat))

184def _new_deal(controller, seat, hand, board):
185 """Informs a BlueChip bots that there is a new deal."""
186 controller.send_line(_START_BOARD)
187 _expect(controller, _READY_FOR_DEAL.format(seat=seat))
188 controller.send_line(_DEAL.format(board=board))
189 _expect(controller, _READY_FOR_CARDS.format(seat=seat))
190 controller.send_line(_CARDS.format(seat=seat, hand=hand))

193class BlueChipBridgeBot(pyspiel.Bot):
194 """An OpenSpiel bot, wrapping a BlueChip bridge bot implementation."""

196 def __init__(self, game, player_id, controller_factory):
197 """Initializes an OpenSpiel `Bot` wrapping a BlueChip-compatible bot.
198
199 Args:
200 game: The OpenSpiel game object, should be an instance of
201 `bridge(use_double_dummy_result=false)`.
202 player_id: The id of the player the bot will act as, 0 = North (dealer), 1
203 = East, 2 = South, 3 = West.
204 controller_factory: Callable that returns new BlueChip controllers which
205 must support methods `read_line` and `send_line`, and `terminate`.
206 """
207 pyspiel.Bot.__init__(self)
208 if str(game) != GAME_STR:
209 raise ValueError(f"BlueChipBridgeBot invoked with {game}")
210 self._game = game
211 self._player_id = player_id
212 self._controller_factory = controller_factory
213 self._seat = _SEATS[player_id]
214 self._num_actions = 52
215 self.dummy = None
216 self.is_play_phase = False
217 self.cards_played = 0
218 self._board = 0
219 self._state = self._game.new_initial_state()
220 self._controller = None
221
222 def player_id(self):
223 return self._player_id

225 def restart(self):
226 """Indicates that we are starting a new episode."""
227 # If we already have a fresh state, there is nothing to do.
228 if not self._state.history():
229 return
230 self._num_actions = 52
231 self.dummy = None
232 self.is_play_phase = False
233 self.cards_played = 0
234 # We didn't see the end of the episode, so the external bot will still
235 # be expecting it. If we can autoplay other people's actions to the end
236 # (e.g. everyone passes or players play their last card), then do that.
237 if not self._state.is_terminal():
238 state = self._state.clone()
239 while (not state.is_terminal()
240 and state.current_player() != self._player_id):
241 legal_actions = state.legal_actions()
242 if _ACTION_PASS in legal_actions:
243 state.apply(_ACTION_PASS)
244 elif len(legal_actions) == 1:
245 state.apply_action(legal_actions[0])
246 if state.is_terminal():
247 self.inform_state(state)
248 # Otherwise, we will have to restart the external bot, because
249 # the protocol makes no provision for this case.
250 if not self._state.is_terminal():
251 self._controller.terminate()
252 self._controller = None
253 self._state = self._game.new_initial_state()
254
255 def _update_for_state(self):
256 """Called for all non-chance nodes, whether or not we have to act."""
257 # Get the actions in the game so far.
258 actions = self._state.history()
259 self.is_play_phase = (not self._state.is_terminal() and
260 max(self._state.legal_actions()) < 52)
261 self.cards_played = sum(1 if a < 52 else 0 for a in actions) - 52
262
263 # If this is the first time we've seen the deal, send our hand.
264 if len(actions) == 52:
265 self._board += 1
266 _new_deal(self._controller, self._seat,
267 _hand_string(actions[self._player_id:52:4]), self._board)
268
269 # Send actions since last `step` call.
270 for other_player_action in actions[self._num_actions:]:
271 other = _expect_regex(self._controller,
272 _READY_FOR_OTHER.format(seat=self._seat))
273 other_player = other["other"]
274 if other_player == "Dummy":
275 other_player = _SEATS[self.dummy]
276 self._controller.send_line(
277 _OTHER_PLAYER_ACTION.format(
278 player=other_player,
279 action=_action_to_string(other_player_action)))
280 self._num_actions = len(actions)
281
282 # If the opening lead has just been made, give the dummy.
283 if self.is_play_phase and self.cards_played == 1:
284 self.dummy = self._state.current_player() ^ 2
285 if self._player_id != self.dummy:
286 other = _expect_regex(self._controller,
287 _READY_FOR_OTHER.format(seat=self._seat))
288 dummy_cards = _hand_string(actions[self.dummy:52:4])
289 self._controller.send_line(_DUMMY_CARDS.format(dummy_cards))
290
291 # If the episode is terminal, send (fake) timing info.
292 if self._state.is_terminal():
293 self._controller.send_line(
294 "Timing - N/S : this board [1:15], total [0:11:23]. "
295 "E/W : this board [1:18], total [0:10:23]"
296 )
297 self.dummy = None
298 self.is_play_phase = False
299 self.cards_played = 0

301 def inform_action(self, state, player, action):
302 del player, action
303 self.inform_state(state)

305 def inform_state(self, state):
306 # Connect if we need to.
307 if self._controller is None:
308 self._controller = self._controller_factory()
309 _connect(self._controller, self._seat)
310
311 full_history = state.history()
312 known_history = self._state.history()
313 if full_history[:len(known_history)] != known_history:
314 raise ValueError(
315 "Supplied state is inconsistent with bot's internal state\n"
316 f"Supplied state:\n{state}\n"
317 f"Internal state:\n{self._state}\n")
318 for action in full_history[len(known_history):]:
319 self._state.apply_action(action)
320 if not self._state.is_chance_node():
321 self._update_for_state()
322
323 def step(self, state):
324 """Returns an action for the given state."""
325 # Bring the external bot up-to-date.
326 self.inform_state(state)
327
328 # If we're on a new trick, tell the bot it is its turn.
329 if self.is_play_phase and self.cards_played % 4 == 0:
330 self._controller.send_line(_PLAYER_TO_LEAD.format(seat=self._seat))
331
332 # Get our action from the bot.
333 our_action = _expect_regex(self._controller, _PLAYER_ACTION)
334 self._num_actions += 1
335 if our_action["pass"]:
336 return _ACTION_PASS
337 elif our_action["dbl"]:
338 return _ACTION_DBL
339 elif our_action["rdbl"]:
340 return _ACTION_RDBL
341 elif our_action["bid"]:
342 return _bid_to_action(our_action["bid"])
343 elif our_action["play"]:
344 return _play_to_action(our_action["play"])
345
346 def terminate(self):
347 self._controller.terminate()
348 self._controller = None
