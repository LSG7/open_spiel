26std::shared_ptr<const TensorGame> LoadTensorGame(const std::string& name) {
27 std::shared_ptr<const Game> game = LoadGame(name);
28 // Make sure it is indeed a tensor game.
29 const TensorGame* tensor_game = dynamic_cast<const TensorGame*>(game.get());
30 if (tensor_game == nullptr) {
31 // If it is not already a tensor game, check if it is an NFG.
32 // If so, convert it.
33 const NormalFormGame* nfg = dynamic_cast<const NormalFormGame*>(game.get());
34 if (nfg != nullptr) {
35 return AsTensorGame(nfg);
36 } else {
37 SpielFatalError(absl::StrCat("Cannot load ", name, " as a tensor game."));
38 }
39 }
40 return std::static_pointer_cast<const TensorGame>(game);
41}

49std::shared_ptr<const TensorGame> AsTensorGame(const NormalFormGame* game) {
50 const int num_players = game->NumPlayers();
51 std::unique_ptr<State> initial_state = game->NewInitialState();
52 std::vector<std::vector<Action>> legal_actions(num_players);
53 std::vector<std::vector<std::string>> action_names(num_players);
54 for (Player player = 0; player < num_players; ++player) {
55 legal_actions[player] = initial_state->LegalActions(player);
56 for (const Action& action : legal_actions[player]) {
57 action_names[player].push_back(
58 initial_state->ActionToString(player, action));
59 }
60 }
61 std::vector<std::vector<double>> utils(num_players);
62
63 GameType type = game->GetType();
64 type.min_num_players = num_players;
65 type.max_num_players = num_players;
66
67 std::vector<Action> actions(num_players);
68 bool last_entry;
69 do {
70 std::unique_ptr<State> clone = initial_state->Clone();
71 clone->ApplyActions(actions);
72 SPIEL_CHECK_TRUE(clone->IsTerminal());
73 std::vector<double> returns = clone->Returns();
74 SPIEL_CHECK_EQ(returns.size(), num_players);
75 for (Player player = 0; player < num_players; ++player) {
76 utils[player].push_back(returns[player]);
77 }
78 last_entry = true;
79 for (Player player = num_players - 1; player >= 0; --player) {
80 if (++actions[player] < legal_actions[player].size()) {
81 last_entry = false;
82 break;
83 } else {
84 actions[player] = 0;
85 }
86 }
87 } while (!last_entry);
88
89 return std::shared_ptr<TensorGame>(
90 new TensorGame(type, {}, action_names, utils));
91}
