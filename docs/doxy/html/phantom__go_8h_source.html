30// Game of Phantom Go:
31// https://www.chessprogramming.org/Phantom_Go
32//
33// Parameters:
34// "komi" float compensation for white (default
35// = 7.5) "board_size" int rows of the board, usually 9, 13 or 19
36// (default = 9) "handicap" int number of handicap stones for black
37// (default = 0) "max_game_length" int maximal lenght of a game (default =
38// board_size * board_size * 4)

139class PhantomGoGame : public Game {
140 public:
141 explicit PhantomGoGame(const GameParameters& params);
142
143 std::shared_ptr<PhantomGoObserver> default_observer_;
144
145 int NumDistinctActions() const override {
146 return phantom_go::NumDistinctActions(board_size_);
147 }
148
149 std::unique_ptr<State> NewInitialState() const override {
150 return std::unique_ptr<State>(
151 new PhantomGoState(shared_from_this(), board_size_, komi_, handicap_));
152 }
153
154 std::vector<int> ObservationTensorShape() const override {
155 // Planes: black, white, empty, and a bias plane indicating komi (whether
156 // white is to play)
157 // and 2 for stone count of white and black
158 return {2 + board_size_ * board_size_ * (CellStates() + 1)};
159 }
160
161 TensorLayout ObservationTensorLayout() const override {
162 return TensorLayout::kCHW;
163 }
164
165 int NumPlayers() const override { return phantom_go::NumPlayers(); }
166
167 double MinUtility() const override { return LossUtility(); }
168 absl::optional<double> UtilitySum() const override {
169 return LossUtility() + WinUtility();
170 }
171 double MaxUtility() const override { return WinUtility(); }
172
173 int MaxGameLength() const override { return max_game_length_; }
174
175 private:
176 const float komi_;
177 const int board_size_;
178 const int handicap_;
179 const int max_game_length_;
180};
