174 void CatchState::DoApplyAction(Action move) {
175  if (!initialized_) {
176  initialized_ = true;
177  ball_col_ = move;
178  ball_row_ = 0;
179  paddle_col_ = num_columns_ / 2;
180  } else {
181  ball_row_++;
182  int direction = move - 1;
183  paddle_col_ =
184  std::min(std::max(paddle_col_ + direction, 0), num_columns_ - 1);
185  }
186 }
