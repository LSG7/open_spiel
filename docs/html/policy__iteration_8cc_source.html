38 // Adds transitions and transition probability from a given state
39 void AddTransition(
40  absl::flat_hash_map<Action, vector<std::pair<std::string, double>>>*
41  transitions,
42  const std::string& key, const std::unique_ptr<State>& state) {
43  for (Action action : state->LegalActions()) {
44  std::unique_ptr<State> next_state = state->Child(action);
45  vector<std::pair<std::string, double>> possibilities;
46  if (next_state->IsChanceNode()) {
47  // For a chance node, record the transition probabilities
48  for (const auto& actionprob : next_state->ChanceOutcomes()) {
49  std::unique_ptr<State> realized_next_state = next_state->Child(action);
50  possibilities.emplace_back(realized_next_state->ToString(),
51  actionprob.second);
52  }
53  } else {
54  // A non-chance node is equivalent to transition with probability 1
55  possibilities.emplace_back(next_state->ToString(), 1.0);
56  }
57  (*transitions)[action] = possibilities;
58  }
59 }

61 // Initialize transition map and value map
62 void InitializeMaps(
63  const std::map<std::string, std::unique_ptr<State>>& states,
64  absl::flat_hash_map<std::string, MDPState>* mdp_state_nodes) {
65  for (const auto& kv : states) {
66  const std::string& key = kv.first;
67  if (kv.second->IsTerminal()) {
68  // For both 1-player and 2-player zero sum games, suffices to look at
69  // player 0's utility
70  (*mdp_state_nodes)[key].value = kv.second->PlayerReturn(Player{0});
71  // No action possible from a terminal state.
72  (*mdp_state_nodes)[key].optimal_action = kInvalidAction;
73  } else {
74  absl::flat_hash_map<Action, std::vector<std::pair<std::string, double>>>&
75  transitions = (*mdp_state_nodes)[key].transitions;
76  AddTransition(&transitions, key, kv.second);
77  (*mdp_state_nodes)[key].value = 0;
78  // Assign any random action as the optimal action, initially.
79  (*mdp_state_nodes)[key].optimal_action = kv.second->LegalActions()[0];
80  }
81  }
82 }

84 double QValue(const absl::flat_hash_map<std::string, MDPState>& mdp_state_nodes,
85  const std::unique_ptr<State>& state, const Action& action) {
86  if (!mdp_state_nodes.contains(state->ToString()) ||
87  !mdp_state_nodes.at(state->ToString()).transitions.contains(action)) {
88  // This action is not possible from this state.
89  return 0;
90  }
91 
92  double value = 0;
93  const vector<std::pair<std::string, double>>& possibilities =
94  mdp_state_nodes.at(state->ToString()).transitions.at(action);
95  for (const auto& outcome : possibilities) {
96  if (mdp_state_nodes.contains(outcome.first)) {
97  value += outcome.second * mdp_state_nodes.at(outcome.first).value;
98  }
99  }
100  return value;
101 }

103 // Given a player and a state, gets the best possible action from this state
104 Action GetBestAction(
105  const absl::flat_hash_map<std::string, MDPState>& mdp_state_nodes,
106  const std::unique_ptr<State>& state, const Player& player,
107  const double& min_utility, const double& max_utility) {
108  vector<Action> legal_actions = state->LegalActions();
109  Action optimal_action = kInvalidAction;
110 
111  // Initialize value to be the minimum utility if current player
112  // is the maximizing player (i.e. player 0), and to maximum utility
113  // if current player is the minimizing player (i.e. player 1).
114  double value = (player == Player{0}) ? min_utility : max_utility;
115  for (Action action : legal_actions) {
116  double q_val = QValue(mdp_state_nodes, state, action);
117  bool is_best_so_far = (player == Player{0} && q_val >= value) ||
118  (player == Player{1} && q_val <= value);
119  if (is_best_so_far) {
120  value = q_val;
121  optimal_action = action;
122  }
123  }
124  return optimal_action;
125 }

129 absl::flat_hash_map<std::string, double> PolicyIteration(const Game& game,
130  int depth_limit,
131  double threshold) {
132  // Currently only supports 1-player or 2-player zero sum games
133  SPIEL_CHECK_TRUE(game.NumPlayers() == 1 || game.NumPlayers() == 2);
134  if (game.NumPlayers() == 2) {
135  SPIEL_CHECK_EQ(game.GetType().utility, GameType::Utility::kZeroSum);
136  }
137 
138  // No support for simultaneous games (needs an LP solver). And so also must
139  // be a perfect information game.
140  SPIEL_CHECK_EQ(game.GetType().dynamics, GameType::Dynamics::kSequential);
141  SPIEL_CHECK_EQ(game.GetType().information,
142  GameType::Information::kPerfectInformation);
143 
144  std::map<std::string, std::unique_ptr<State>> states =
145  GetAllStates(game, depth_limit, /*include_terminals=*/true,
146  /*include_chance_states=*/false);
147  absl::flat_hash_map<std::string, MDPState> mdp_state_nodes;
148 
149  InitializeMaps(states, &mdp_state_nodes);
150 
151  bool policy_stable;
152  do {
153  // Policy evaluation done in place
154  double error;
155  do {
156  error = 0;
157  for (const auto& kv : states) {
158  const std::string& key = kv.first;
159  if (kv.second->IsTerminal()) continue;
160 
161  // Evaluate the state value function
162  Action curr_optimal_action = mdp_state_nodes.at(key).optimal_action;
163  double value = QValue(mdp_state_nodes, kv.second, curr_optimal_action);
164 
165  double* stored_value = &mdp_state_nodes.at(key).value;
166  error = std::max(std::abs(*stored_value - value), error);
167  *stored_value = value;
168  }
169  } while (error > threshold);
170 
171  // Policy improvement
172  double min_utility = game.MinUtility();
173  double max_utility = game.MaxUtility();
174  policy_stable = true;
175  for (const auto& kv : states) {
176  const std::string& key = kv.first;
177  if (kv.second->IsTerminal()) continue;
178 
179  Player player = kv.second->CurrentPlayer();
180 
181  // Choose the action with the highest possible action value function
182  Action curr_optimal_action = GetBestAction(
183  mdp_state_nodes, kv.second, player, min_utility, max_utility);
184 
185  double curr_value =
186  QValue(mdp_state_nodes, kv.second, curr_optimal_action);
187 
188  double* stored_value = &mdp_state_nodes.at(key).value;
189  Action* stored_optimal_action = &mdp_state_nodes.at(key).optimal_action;
190  if (std::abs(*stored_value - curr_value) > threshold) {
191  policy_stable = false;
192  *stored_optimal_action = curr_optimal_action;
193  }
194  }
195  } while (!policy_stable);
196 
197  absl::flat_hash_map<std::string, double> values;
198  for (const auto& kv : states) {
199  std::string state_string = kv.first;
200  values[state_string] = mdp_state_nodes[state_string].value;
201  }
202  return values;
203 }
