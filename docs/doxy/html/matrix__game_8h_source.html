67 // Implementation of Game interface
68 int NumDistinctActions() const override {
69 return std::max(NumRows(), NumCols());

76 double MinUtility() const override {

82 double MaxUtility() const override {

88 absl::optional<double> UtilitySum() const override;
89
90 std::string ActionToString(Player player, Action action) const override {
91 switch (player) {
92 case 0: {
93 SPIEL_CHECK_GE(action, 0);
94 SPIEL_CHECK_LT(action, row_action_names_.size());
95 return row_action_names_[action];
96 }
97
98 case 1: {
99 SPIEL_CHECK_GE(action, 0);
100 SPIEL_CHECK_LT(action, col_action_names_.size());
101 return col_action_names_[action];
102 }
103
104 default:
105 SpielFatalError("Unknown player");
106 }
107 }

137 std::vector<double> GetUtilities(const std::vector<Action>& joint_action)
138 const override {
139 int index = Index(joint_action[0], joint_action[1]);
140 return {row_utilities_[index], col_utilities_[index]};
141 }
142
143 double GetUtility(Player player, const std::vector<Action>& joint_action)

146 }
147
148 bool operator==(const Game& other_game) const override {
149 const auto& other = down_cast<const MatrixGame&>(other_game);
150 return (row_action_names_.size() == other.row_action_names_.size() &&
151 col_action_names_.size() == other.col_action_names_.size() &&
152 row_utilities_ == other.row_utilities_ &&
153 col_utilities_ == other.col_utilities_);
154 }
155
156 bool ApproxEqual(const Game& other_game, double tolerance) const {
157 const auto& other = down_cast<const MatrixGame&>(other_game);
158 return (row_action_names_.size() == other.row_action_names_.size() &&
159 col_action_names_.size() == other.col_action_names_.size() &&
160 AllNear(row_utilities_, other.row_utilities_, tolerance) &&
161 AllNear(col_utilities_, other.col_utilities_, tolerance));
162 }

164 private:
165 int Index(int row, int col) const { return row * NumCols() + col; }
166 std::vector<std::string> row_action_names_;
167 std::vector<std::string> col_action_names_;
168 std::vector<double> row_utilities_;
169 std::vector<double> col_utilities_;

172class MatrixState : public NFGState {
173 public:
174 explicit MatrixState(std::shared_ptr<const Game> game);
175 explicit MatrixState(const MatrixState&) = default;
176
177 std::vector<Action> LegalActions(Player player) const override {
178 if (IsTerminal()) return {};
179 if (player == kSimultaneousPlayerId) {
180 return LegalFlatJointActions();
181 } else {
182 std::vector<Action> moves(player == kRowPlayer ? matrix_game_->NumRows()
183 : matrix_game_->NumCols());
184 std::iota(moves.begin(), moves.end(), 0); // fill with values 0...n-1
185 return moves;
186 }
187 }
188
189 std::string ToString() const override;
190
191 std::string ActionToString(Player player, Action action_id) const override {
192 if (player == kSimultaneousPlayerId)
193 return FlatJointActionToString(action_id);
194 else if (player == kRowPlayer)
195 return matrix_game_->RowActionName(action_id);
196 else
197 return matrix_game_->ColActionName(action_id);
198 }
199
200 bool IsTerminal() const override { return !joint_move_.empty(); }
201
202 std::vector<double> Returns() const override {
203 if (IsTerminal()) {
204 return {matrix_game_->RowUtility(joint_move_[0], joint_move_[1]),
205 matrix_game_->ColUtility(joint_move_[0], joint_move_[1])};
206 } else {
207 return {0, 0};
208 }
209 }
210
211 std::unique_ptr<State> Clone() const override {
212 return std::unique_ptr<State>(new MatrixState(*this));
213 }
214
215 protected:
216 void DoApplyActions(const std::vector<Action>& moves) override {
217 SPIEL_CHECK_EQ(moves.size(), 2);
218 SPIEL_CHECK_GE(moves[kRowPlayer], 0);
219 SPIEL_CHECK_LT(moves[kRowPlayer], matrix_game_->NumRows());
220 SPIEL_CHECK_GE(moves[kColPlayer], 0);
221 SPIEL_CHECK_LT(moves[kColPlayer], matrix_game_->NumCols());
222 joint_move_ = moves;
223 }
224
225 private:
226 std::vector<Action> joint_move_{}; // joint move that was chosen
227 const MatrixGame* matrix_game_;
228};
