14"""An example that computes the Nash bargaining score from negotiations.
15
16This uses the bargaining game that was introduced in:
17
18[1] Lewis et al., Deal or no deal? End-to-end learning of negotiation
19 dialogues, 2017. https://arxiv.org/abs/1706.05125
20[2] David DeVault, Johnathan Mell, and Jonathan Gratch.
21 2015. Toward Natural Turn-taking in a Virtual Human Negotiation Agent
22
23It computes the empirical Nash bargaining score (NBS) from three sources:
24 - Human play
25 - IS-MCTS in self-play
26 - A theoretical maximum NBS if the players had full information and can see
27 each other's utilities and then maximize their NBS.
28
29These are all run on a data set extracted from the Lewis et al. '17 data set:
30https://github.com/facebookresearch/end-to-end-negotiator/blob/master/src/data/negotiate/data.txt
31
32This example is inspired by the paper (Iwasa and Fujita, "Prediction of Nash
33Bargaining Solution in Negotiation Dialogue", 2018).
34"""

50class Instance(object):
51 """An instance of a bargaining problem."""
52
53 def __init__(self, pool, p1values, p2values):
54 self.pool = np.array(pool)
55 self.p1values = np.array(p1values)
56 self.p2values = np.array(p2values)
57 assert 5 <= sum(pool) <= 7
58 assert np.dot(pool, p1values) == 10
59 assert np.dot(pool, p2values) == 10
60
61 def __str__(self):
62 return (",".join([str(x) for x in self.pool]) + " " +
63 ",".join([str(x) for x in self.p1values]) + " " +
64 ",".join([str(x) for x in self.p2values]))

80def dialogue_matches_prev_line(line1, line2):
81 """Checks if the dialogue matches the previous line's."""
82 parts1 = line1.split(" ")
83 parts2 = line2.split(" ")
84 for i in range(6, min(len(parts1), len(parts2))):
85 if parts1[i] == "YOU:" or parts1[i] == "THEM:":
86 if parts1[i] == "YOU:" and parts2[i] != "THEM:":
87 return False
88 if parts1[i] == "THEM:" and parts2[i] != "YOU:":
89 return False
90 elif parts1[i] != parts2[i]:
91 return False
92 if parts1[i] == "<selection>":
93 break
94 return True

98def parse_dataset(filename):
99 """Parse the Lewis et al. '17 data file."""
100 # book, hat, ball
101 # Example format
102 # 1 0 4 2 1 2 YOU: i would like 4 hats and you can have the rest . <eos> THEM: deal <eos> YOU: <selection> item0=0 item1=4 item2=0 <eos> reward=8 agree 1 4 4 1 1 2
103 # 1 4 4 1 1 2 THEM: i would like 4 hats and you can have the rest . <eos> YOU: deal <eos> THEM: <selection> item0=1 item1=0 item2=1 <eos> reward=6 agree 1 0 4 2 1 2
104 # 1 6 3 0 2 2 YOU: you can have all the hats if i get the book and basketballs . <eos> THEM: <selection> item0=1 item1=3 item2=2 <eos> reward=10 disagree 1 2 3 2 2 1
105 # 1 10 3 0 1 0 YOU: hi i would like the book and ball and you can have the hats <eos> THEM: i can give you either the book or the ball <eos> YOU: ill take the book <eos> THEM: ok i will take the hats and ball <eos> YOU: deal <eos> THEM: <selection> item0=1 item1=0 item2=0 <eos> reward=10 agree 1 2 3 2 1 2
106 # 1 2 3 2 1 2 THEM: hi i would like the book and ball and you can have the hats <eos> YOU: i can give you either the book or the ball <eos> THEM: ill take the book <eos> YOU: ok i will take the hats and ball <eos> THEM: deal <eos> YOU: <selection> item0=0 item1=3 item2=1 <eos> reward=8 agree 1 10 3 0 1 0
107 contents = pyspiel.read_contents_from_file(filename, "r")
108 lines = contents.split("\n")
109 cur_nego = None
110 negotiations = []
111 instances = []
112
113 for line_no in range(len(lines)):
114 line = lines[line_no]
115 if line:
116 parts = line.split(" ")
117 # parse the line to add a new negotiation
118 pool = [int(parts[0]), int(parts[2]), int(parts[4])]
119 my_values = [int(parts[1]), int(parts[3]), int(parts[5])]
120 pool2 = [int(parts[-6]), int(parts[-4]), int(parts[-2])]
121 other_values = [int(parts[-5]), int(parts[-3]), int(parts[-1])]
122 assert pool == pool2
123 rewards = [0, 0]
124 add_nego = False
125 outcome_str = parts[-7] # this will be "agree" or "disagree"
126 if parts[6] == "YOU:":
127 player_id = 0
128 instance = Instance(pool, my_values, other_values)
129 elif parts[6] == "THEM:":
130 player_id = 1
131 instance = Instance(pool, other_values, my_values)
132 else:
133 assert False, parts[6]
134 outcome = False
135 my_reward = 0
136 instances.append(instance)
137 if "disconnect" in line:
138 continue
139 # sometimes there is a "no agreement" in the rewards section
140 if (outcome_str == "disagree" or
141 (parts[-9] + " " + parts[-8]) == "reward=no agreement" or
142 parts[-8] == "reward=disconnect"):
143 # do not parse the reward, but must still parse the next line
144 add_nego = False
145 elif outcome_str == "agree":
146 outcome = True
147 reward_parts = parts[-8].split("=")
148 assert len(reward_parts) == 2, f"reward parts str: {parts[-8]}"
149 assert reward_parts[0] == "reward"
150 my_reward = int(reward_parts[1])
151 else:
152 assert False, f"Bad outcome: {outcome_str}"
153 if cur_nego is None:
154 rewards[player_id] = my_reward
155 if player_id == 0:
156 cur_nego = Negotiation(instance, outcome, rewards)
157 else:
158 cur_nego = Negotiation(instance, outcome, rewards)
159 else:
160 # There are some in the data set that are incomplete (i.e. are missing the second perspective).
161 # We should not count these.
162 if dialogue_matches_prev_line(line, lines[line_no - 1]):
163 assert list(cur_nego.instance.pool) == pool
164 if player_id == 1:
165 assert list(cur_nego.instance.p2values) == my_values
166 assert list(cur_nego.instance.p1values) == other_values
167 elif player_id == 0:
168 assert list(cur_nego.instance.p1values) == my_values
169 assert list(cur_nego.instance.p2values) == other_values
170 cur_nego.rewards[player_id] = my_reward
171 add_nego = True
172 else:
173 # not matching, treat as new negotiation
174 rewards[player_id] = my_reward
175 if player_id == 0:
176 cur_nego = Negotiation(instance, outcome, rewards)
177 else:
178 cur_nego = Negotiation(instance, outcome, rewards)
179 add_nego = False
180 if add_nego or outcome_str == "disagree":
181 negotiations.append(cur_nego)
182 print(str(cur_nego))
183 print(len(negotiations))
184 cur_nego = None
185 if outcome_str != "disagree":
186 # same instance was added twice, so remove the last one
187 instances.pop()
188 return instances, negotiations

198def compute_nbs_from_simulations(game, num_games, bots):
199 """Compute empirical NBS from simulations."""
200 avg_returns = np.zeros(game.num_players())
201 for _ in range(num_games):
202 state = game.new_initial_state()
203 while not state.is_terminal():
204 if state.is_chance_node():
205 # Chance node: sample an outcome
206 outcomes = state.chance_outcomes()
207 action_list, prob_list = zip(*outcomes)
208 action = np.random.choice(action_list, p=prob_list)
209 state.apply_action(action)
210 else:
211 player = state.current_player()
212 action = bots[player].step(state)
213 state.apply_action(action)
214 returns = np.asarray(state.returns())
215 avg_returns += returns
216 avg_returns /= num_games
217 return np.prod(avg_returns)

220class MaxBot(object):
221 """Finds the single (deterministic) trade offer that maximizes the NBS."""
222
223 def __init__(self):
224 pass
225
226 def step(self, state):
227 """Returns the NBS-maximizing action.
228
229 If i'm player 0, then search over all possible moves, assume player 2
230 takes the agree action, and choose the action that maximizes the NBS
231 Player 1 just always agrees.
232
233 Args:
234 state: the OpenSpiel state to act from.
235 """
236 player = state.current_player()
237 if player == 1:
238 return state.agree_action()
239 max_nbs = -1
240 max_action = -1
241 for action in state.legal_actions():
242 state_clone = state.clone()
243 state_clone.apply_action(action)
244 state_clone.apply_action(state.agree_action())
245 returns = state_clone.returns()
246 nbs = np.prod(returns)
247 if nbs > max_nbs:
248 max_nbs = nbs
249 max_action = action
250 assert max_action >= 0
251 return max_action
