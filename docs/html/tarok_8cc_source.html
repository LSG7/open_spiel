93 std::unique_ptr<State> TarokGame::DeserializeState(
94  const std::string& str) const {
95  std::unique_ptr<TarokState> state = NewInitialTarokState();
96  if (str.empty()) return state;
97 
98  std::vector<std::string> lines = absl::StrSplit(str, '\n');
99  for (int i = 0; i < lines.size(); i++) {
100  if (i == 0) {
101  // chance node where we presisted the card dealing seed, see
102  // TarokState::DoApplyActionInCardDealing for more info
103  std::tie(state->talon_, state->players_cards_) =
104  DealCards(num_players_, std::stoi(lines.at(i)));
105  state->current_game_phase_ = GamePhase::kBidding;
106  state->current_player_ = 1;
107  state->AddPrivateCardsToInfoStates();
108  } else {
109  state->ApplyAction(std::stol(lines.at(i)));
110  }
111  }
112  return state;
113 }

192 std::vector<Action> TarokState::LegalActions() const {
193  // all card actions are encoded as 0, 1, ..., 52, 53 and correspond to card
194  // indices wrt. tarok_parent_game_->card_deck_, card actions are returned:
195  // - in the king calling phase
196  // - by LegalActionsInTalonExchange() after the talon set is selected (i.e.
197  // when discarding the cards)
198  // - by LegalActionsInTricksPlaying()
199  switch (current_game_phase_) {
200  case GamePhase::kCardDealing:
201  // return a dummy action due to implicit stochasticity
202  return {0};
203  case GamePhase::kBidding:
204  return LegalActionsInBidding();
205  case GamePhase::kKingCalling:
206  return {kKingOfHeartsAction, kKingOfDiamondsAction, kKingOfSpadesAction,
207  kKingOfClubsAction};
208  case GamePhase::kTalonExchange:
209  return LegalActionsInTalonExchange();
210  case GamePhase::kTricksPlaying:
211  return LegalActionsInTricksPlaying();
212  case GamePhase::kFinished:
213  return {};
214  }
215 }

217 std::vector<Action> TarokState::LegalActionsInBidding() const {
218  // actions 1 - 12 correspond to contracts in tarok_parent_game_->contracts_
219  // respectively, action 0 means pass
220  auto it = std::max_element(players_bids_.begin(), players_bids_.end());
221  int max_bid = *it;
222  int max_bid_player = it - players_bids_.begin();
223 
224  std::vector<Action> actions;
225  if (current_player_ == 0 &&
226  players_bids_.at(current_player_) == kInvalidBidAction &&
227  AllButCurrentPlayerPassedBidding()) {
228  // no bidding has happened before so forehand can
229  // bid any contract but can't pass
230  actions.insert(actions.end(), {kBidKlopAction, kBidThreeAction});
231  } else if (!AllButCurrentPlayerPassedBidding()) {
232  // other players still playing
233  actions.push_back(kBidPassAction);
234  }
235 
236  for (int action = 3; action <= 12; action++) {
237  if (num_players_ == 3 && action >= kBidSoloThreeAction &&
238  action <= kBidSoloOneAction) {
239  // skip solo contracts for three players
240  continue;
241  }
242  if (action < max_bid) {
243  continue;
244  }
245  if ((action > max_bid) ||
246  (action == max_bid && current_player_ <= max_bid_player)) {
247  actions.push_back(action);
248  }
249  }
250  return actions;
251 }

253 std::vector<Action> TarokState::LegalActionsInTalonExchange() const {
254  if (talon_.size() == 6) {
255  // choosing one of the talon card sets where actions are encoded as
256  // 0, 1, 2, etc. from left to right, i.e. 0 is the leftmost talon set
257  // as returned by TalonSets()
258  std::vector<Action> actions(6 / selected_contract_->num_talon_exchanges);
259  std::iota(actions.begin(), actions.end(), 0);
260  return actions;
261  }
262  // prevent discarding of taroks and kings
263  std::vector<Action> actions;
264  for (auto const& action : players_cards_.at(current_player_)) {
265  const Card& card = ActionToCard(action);
266  if (card.suit != CardSuit::kTaroks && card.points != 5)
267  actions.push_back(action);
268  }
269  // allow discarding of taroks (except of trula) if player has no other choice
270  if (actions.empty()) {
271  for (auto const& action : players_cards_.at(current_player_)) {
272  if (ActionToCard(action).points != 5) actions.push_back(action);
273  }
274  }
275  return actions;
276 }

278 std::vector<Action> TarokState::LegalActionsInTricksPlaying() const {
279  if (trick_cards_.empty()) {
280  // trick opening, i.e. the current player is choosing
281  // the first card for this trick

283  return RemovePagatIfNeeded(players_cards_.at(current_player_));

291 std::vector<Action> TarokState::LegalActionsInTricksPlayingFollowing() const {
292  auto [can_follow_suit, cant_follow_suit_but_has_tarok] =
293  CanFollowSuitOrCantButHasTarok();
294 
295  CardSuit take_suit;
296  if (can_follow_suit) {
297  take_suit = ActionToCard(trick_cards_.front()).suit;
298  } else if (cant_follow_suit_but_has_tarok) {
299  take_suit = CardSuit::kTaroks;
300  } else {
301  // can't follow suit and doesn't have taroks so any card can be played
302  return players_cards_.at(current_player_);
303  }
304 
305  if (selected_contract_->is_negative)
306  return TakeSuitFromPlayerCardsInNegativeContracts(take_suit);
307  else
308  return TakeSuitFromPlayerCardsInPositiveContracts(take_suit);
309 }

311 std::tuple<bool, bool> TarokState::CanFollowSuitOrCantButHasTarok() const {
312  const Card& opening_card = ActionToCard(trick_cards_.front());
313  bool has_taroks = false;
314  for (auto const& action : players_cards_.at(current_player_)) {
315  const Card& current_card = ActionToCard(action);
316  if (current_card.suit == opening_card.suit) {
317  // note that the second return value is irrelevant in this case
318  return {true, false};
319  }
320  if (current_card.suit == CardSuit::kTaroks) {
321  has_taroks = true;
322  }
323  }
324  return {false, has_taroks};
325 }

327 std::vector<Action> TarokState::TakeSuitFromPlayerCardsInNegativeContracts(
328  CardSuit suit) const {
329  bool player_has_pagat =
330  ActionInActions(kPagatAction, players_cards_.at(current_player_));
331  if (player_has_pagat && ActionInActions(kMondAction, trick_cards_) &&
332  ActionInActions(kSkisAction, trick_cards_)) {
333  // the emperor trick, i.e. pagat has to be played as it is the only card
334  // that will win the trick
335  return {kPagatAction};
336  }
337 
338  absl::optional<Action> action_to_beat = ActionToBeatInNegativeContracts(suit);
339  std::vector<Action> actions;
340 
341  if (action_to_beat) {
342  const Card& card_to_beat = ActionToCard(*action_to_beat);
343  auto const& player_cards = players_cards_.at(current_player_);
344  // a higher card only has to be played when the player actually has a higher
345  // card otherwise any card of the suit can be played
346  bool has_higher_card = false;
347  for (auto const& action : player_cards) {
348  const Card& current_card = ActionToCard(action);
349  if (current_card.suit == suit && current_card.rank > card_to_beat.rank) {
350  has_higher_card = true;
351  break;
352  }
353  }
354  // collect the actual cards
355  for (auto const& action : player_cards) {
356  const Card& current_card = ActionToCard(action);
357  if (current_card.suit == suit &&
358  (!has_higher_card || current_card.rank > card_to_beat.rank)) {
359  actions.push_back(action);
360  }
361  }
362  } else {
363  // no need to beat any card so simply return all cards of the correct suit
364  actions = TakeSuitFromPlayerCardsInPositiveContracts(suit);
365  }
366 
367  if (player_has_pagat)
368  return RemovePagatIfNeeded(actions);
369  else
370  return actions;
371 }
372 
373 absl::optional<Action> TarokState::ActionToBeatInNegativeContracts(
374  CardSuit suit) const {
375  // there are two cases where no card has to be beaten; the player is following
376  // a colour suit and there is already at least one tarok in trick_cards_ or
377  // the player is forced to play a tarok and there are no taroks in
378  // trick_cards_
379  bool tarok_in_trick_cards = false;
380  for (auto const& action : trick_cards_) {
381  if (ActionToCard(action).suit == CardSuit::kTaroks) {
382  tarok_in_trick_cards = true;
383  break;
384  }
385  }
386  if ((suit != CardSuit::kTaroks && tarok_in_trick_cards) ||
387  (suit == CardSuit::kTaroks && !tarok_in_trick_cards)) {
388  return {};
389  }
390  // the specified suit should be present in trick_cards_ from here on because
391  // it is either a suit of the opening card or CardSuit::kTaroks with existing
392  // taroks in trick_cards_
393  Action action_to_beat = trick_cards_.front();
394  for (int i = 1; i < trick_cards_.size(); i++) {
395  const Card& card_to_beat = ActionToCard(action_to_beat);
396  const Card& current_card = ActionToCard(trick_cards_.at(i));
397  if (current_card.suit == suit && current_card.rank > card_to_beat.rank)
398  action_to_beat = trick_cards_.at(i);
399  }
400  return action_to_beat;
401 }

403 std::vector<Action> TarokState::RemovePagatIfNeeded(
404  const std::vector<Action>& actions) const {
405  if (actions.size() > 1) {
406  // mustn't play pagat unless it's the only card, note that actions
407  // can be all player's cards or a subset already filtered by the caller
408  std::vector<Action> actions_no_pagat;
409  for (auto const& action : actions) {
410  if (action != kPagatAction) actions_no_pagat.push_back(action);
411  }
412  return actions_no_pagat;
413  }
414  return actions;

458 void TarokState::DoApplyAction(Action action_id) {
459  if (!ActionInActions(action_id, LegalActions())) {
460  SpielFatalError(absl::StrCat("Action ", action_id,
461  " is not valid in the current state."));
462  }
463  switch (current_game_phase_) {
464  case GamePhase::kCardDealing:
465  DoApplyActionInCardDealing();
466  break;
467  case GamePhase::kBidding:
468  DoApplyActionInBidding(action_id);
469  break;
470  case GamePhase::kKingCalling:
471  DoApplyActionInKingCalling(action_id);
472  break;
473  case GamePhase::kTalonExchange:
474  DoApplyActionInTalonExchange(action_id);
475  break;
476  case GamePhase::kTricksPlaying:
477  DoApplyActionInTricksPlaying(action_id);
478  break;
479  case GamePhase::kFinished:
480  SpielFatalError("Calling DoApplyAction in a terminal state.");
481  }
482 }

484 void TarokState::DoApplyActionInCardDealing() {
485  // do the actual sampling here due to implicit stochasticity
486  do {
487  // seed is persisted for serialization purposes
488  card_dealing_seed_ = tarok_parent_game_->RNG();
489  // hands without taroks are illegal
490  std::tie(talon_, players_cards_) =
491  DealCards(num_players_, card_dealing_seed_);
492  } while (AnyPlayerWithoutTaroks());
493  current_game_phase_ = GamePhase::kBidding;
494  // lower player indices correspond to higher bidding priority,
495  // i.e. 0 is the forehand, num_players - 1 is the dealer
496  current_player_ = 1;
497  AddPrivateCardsToInfoStates();
498 }

517 void TarokState::DoApplyActionInBidding(Action action_id) {
518  players_bids_.at(current_player_) = action_id;
519  AppendToAllInformationStates(std::to_string(action_id));
520  if (AllButCurrentPlayerPassedBidding()) {

527  AppendToAllInformationStates(",");
528  }

539 void TarokState::FinishBiddingPhase(Action action_id) {
540  declarer_ = current_player_;
541  selected_contract_ = &tarok_parent_game_->contracts_.at(action_id - 1);
542  if (num_players_ == 4 && selected_contract_->needs_king_calling)
543  current_game_phase_ = GamePhase::kKingCalling;
544  else if (selected_contract_->NeedsTalonExchange())
545  current_game_phase_ = GamePhase::kTalonExchange;
546  else
547  StartTricksPlayingPhase();
548 }

550 void TarokState::DoApplyActionInKingCalling(Action action_id) {
551  called_king_ = action_id;
552  if (ActionInActions(action_id, talon_)) {
553  called_king_in_talon_ = true;
554  } else {
555  for (int i = 0; i < num_players_; i++) {
556  if (i == current_player_) {
557  continue;
558  } else if (ActionInActions(action_id, players_cards_.at(i))) {
559  declarer_partner_ = i;
560  break;
561  }
562  }
563  }
564  current_game_phase_ = GamePhase::kTalonExchange;
565  AppendToAllInformationStates(absl::StrCat(action_id, ";"));
566 }

568 void TarokState::DoApplyActionInTalonExchange(Action action_id) {
569  auto& player_cards = players_cards_.at(current_player_);
570 
571  if (talon_.size() == 6) {
572  // add all talon cards to info states
573  AppendToAllInformationStates(absl::StrCat(absl::StrJoin(talon_, ","), ";"));
574 
575  // choosing one of the talon card sets
576  int set_begin = action_id * selected_contract_->num_talon_exchanges;
577  int set_end = set_begin + selected_contract_->num_talon_exchanges;
578 
579  bool mond_in_talon = ActionInActions(kMondAction, talon_);
580  bool mond_in_selected_talon_set = false;
581  for (int i = set_begin; i < set_end; i++) {
582  player_cards.push_back(talon_.at(i));
583  if (talon_.at(i) == kMondAction) mond_in_selected_talon_set = true;
584  }
585  if (mond_in_talon && !mond_in_selected_talon_set) {
586  // the captured mond penalty applies if mond is in talon and not part of
587  // the selected set
588  captured_mond_player_ = current_player_;
589  }
590 
591  // add the selected talon set to info states
592  AppendToAllInformationStates(absl::StrCat(action_id, ";"));
593 
594  std::sort(player_cards.begin(), player_cards.end());
595  talon_.erase(talon_.begin() + set_begin, talon_.begin() + set_end);
596  } else {
597  // discarding the cards
598  MoveActionFromTo(action_id, &player_cards,
599  &players_collected_cards_.at(current_player_));
600 
601  bool talon_exchange_finished = player_cards.size() == 48 / num_players_;
602  std::string info_state_delimiter = talon_exchange_finished ? ";" : ",";
603 
604  // note that all players see discarded tarok cards but only the discarder
605  // knows about discarded non-taroks
606  if (ActionToCard(action_id).suit == CardSuit::kTaroks) {
607  AppendToAllInformationStates(
608  absl::StrCat(action_id, info_state_delimiter));
609  } else {
610  AppendToInformationState(current_player_,
611  absl::StrCat(action_id, info_state_delimiter));

653 void TarokState::ResolveTrick() {
654  auto [trick_winner, winning_action] = ResolveTrickWinnerAndWinningAction();
655  auto& trick_winner_collected_cards =
656  players_collected_cards_.at(trick_winner);
657 
658  for (auto const& action : trick_cards_) {
659  trick_winner_collected_cards.push_back(action);
660  }
661 
662  if (selected_contract_->name == ContractName::kKlop && !talon_.empty()) {
663  // add the "gift" talon card in klop
664  trick_winner_collected_cards.push_back(talon_.front());
665  AppendToAllInformationStates(absl::StrCat(",", talon_.front()));
666  talon_.erase(talon_.begin());
667  } else if (winning_action == called_king_ && called_king_in_talon_) {
668  // declearer won the trick with the called king that was in talon so all
669  // of the talon cards belong to the declearer (note that this is only
670  // possible when talon exchange actually happened in the past)
671  bool mond_in_talon = false;
672  for (auto const& action : talon_) {
673  trick_winner_collected_cards.push_back(action);
674  if (action == kMondAction) mond_in_talon = true;
675  }
676  if (mond_in_talon) {
677  // the called king and mond were in different parts of the talon and
678  // declearer selected the set with the king plus won the mond as
679  // part of the obtained talon remainder, negating the captured mond
680  // penalty obtained during DoApplyActionInTalonExchange()
681  captured_mond_player_ = kInvalidPlayer;
682  }
683  talon_.clear();
684  } else if ((selected_contract_->NeedsTalonExchange() ||
685  selected_contract_->name == ContractName::kSoloWithout) &&
686  (winning_action == kSkisAction ||
687  winning_action == kPagatAction)) {
688  // check if mond is captured by skis or pagat (emperor's trick) and
689  // penalise the player of the mond in certain contracts
690  for (int i = 0; i < trick_cards_.size(); i++) {
691  if (trick_cards_.at(i) == kMondAction) {
692  captured_mond_player_ = TrickCardsIndexToPlayer(i);
693  }
694  }
695  }
696 
697  trick_cards_.clear();
698  current_player_ = trick_winner;
699 }

701 TrickWinnerAndAction TarokState::ResolveTrickWinnerAndWinningAction() const {
702  // compute the winning action index within trick_cards_
703  int winning_action_i;
704  if ((ActionInActions(kPagatAction, trick_cards_) &&
705  ActionInActions(kMondAction, trick_cards_) &&
706  ActionInActions(kSkisAction, trick_cards_)) &&
707  (selected_contract_->name != ContractName::kColourValatWithout ||
708  ActionToCard(trick_cards_.front()).suit == CardSuit::kTaroks)) {
709  // the emperor trick, i.e. pagat wins over mond and skis in all cases but
710  // not in Contract::kColourValatWithout when a non-trump is led
711  winning_action_i =
712  std::find(trick_cards_.begin(), trick_cards_.end(), kPagatAction) -
713  trick_cards_.begin();
714  } else {
715  winning_action_i = 0;
716  for (int i = 1; i < trick_cards_.size(); i++) {
717  const Card& winning_card =
718  ActionToCard(trick_cards_.at(winning_action_i));
719  const Card& current_card = ActionToCard(trick_cards_.at(i));
720 
721  if (((current_card.suit == CardSuit::kTaroks &&
722  selected_contract_->name != ContractName::kColourValatWithout) ||
723  current_card.suit == winning_card.suit) &&
724  current_card.rank > winning_card.rank) {
725  winning_action_i = i;
726  }
727  }
728  }
729  return {TrickCardsIndexToPlayer(winning_action_i),
730  trick_cards_.at(winning_action_i)};
731 }

773 std::vector<int> TarokState::ScoresInKlop() const {

793  for (int i = 0; i < num_players_; i++) {
794  if (std::abs(scores.at(i)) != 70) scores.at(i) = 0;
795  }

800 std::vector<int> TarokState::ScoresInNormalContracts() const {
801  auto [collected_cards, opposite_collected_cards] =
802  SplitCollectedCardsPerTeams();
803 
804  int score;
805  if (collected_cards.size() == 48) {
806  // valat won
807  score = 250;
808  } else if (opposite_collected_cards.size() == 48) {
809  // valat lost
810  score = -250;
811  } else {
812  int card_points =
813  CardPoints(collected_cards, tarok_parent_game_->card_deck_);
814  score = card_points - 35;
815 
816  if (card_points > 35)
817  score += selected_contract_->score;
818  else
819  score -= selected_contract_->score;
820 
821  // bonuses could be positive, negative or 0
822  int bonuses = NonValatBonuses(collected_cards, opposite_collected_cards);
823  score += bonuses;
824  }
825 
826  std::vector<int> scores(num_players_, 0);
827  scores.at(declarer_) = score;
828  if (declarer_partner_ != kInvalidPlayer) scores.at(declarer_partner_) = score;

849 int TarokState::NonValatBonuses(
850  const std::vector<Action>& collected_cards,
851  const std::vector<Action>& opposite_collected_cards) const {
852  int bonuses = 0;
853 
854  // last trick winner is the current player
855  auto const& last_trick_winner_cards =
856  players_collected_cards_.at(current_player_);
857  // king ultimo and pagat ultimo
858  int ultimo_bonus = 0;
859  if (std::find(last_trick_winner_cards.end() - num_players_,
860  last_trick_winner_cards.end(),
861  called_king_) != last_trick_winner_cards.end()) {
862  // king ultimo
863  ultimo_bonus = 10;
864  } else if (std::find(last_trick_winner_cards.end() - num_players_,
865  last_trick_winner_cards.end(),
866  0) != last_trick_winner_cards.end()) {
867  // pagat ultimo
868  ultimo_bonus = 25;
869  }
870 
871  if (ultimo_bonus > 0 &&
872  (current_player_ == declarer_ || current_player_ == declarer_partner_)) {
873  bonuses = ultimo_bonus;
874  } else if (ultimo_bonus > 0) {
875  bonuses = -ultimo_bonus;
876  }
877 
878  // collected kings or trula
879  auto [collected_kings, collected_trula] =
880  CollectedKingsAndOrTrula(collected_cards);
881  auto [opposite_collected_kings, opposite_collected_trula] =
882  CollectedKingsAndOrTrula(opposite_collected_cards);
883 
884  if (collected_kings)
885  bonuses += 10;
886  else if (opposite_collected_kings)
887  bonuses -= 10;
888  if (collected_trula)
889  bonuses += 10;
890  else if (opposite_collected_trula)
891  bonuses -= 10;
892  return bonuses;
893 }

895 std::tuple<bool, bool> TarokState::CollectedKingsAndOrTrula(
896  const std::vector<Action>& collected_cards) const {
897  int num_kings = 0, num_trula = 0;

900  action == kKingOfSpadesAction || kKingOfClubsAction == 53) {
901  num_kings += 1;
902  } else if (action == kPagatAction || action == kMondAction ||
903  action == kSkisAction) {
904  num_trula += 1;

906  }
