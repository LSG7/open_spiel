58 def meta_loss(opt_params, net_apply, payoff, steps, rng):
59  """Meta loss function."""
60 
61  regret_sum_x = np.zeros(shape=[FLAGS.batch_size, 1, FLAGS.num_actions])
62  regret_sum_y = np.zeros(shape=[FLAGS.batch_size, 1, FLAGS.num_actions])
63  total_loss = 0
64 
65  @jax.jit
66  def body_fun(s, total_loss):
67  nonlocal regret_sum_x
68  nonlocal regret_sum_y
69  x = net_apply(opt_params, rng, regret_sum_x / (s + 1))
70  y = net_apply(opt_params, rng, regret_sum_y / (s + 1))
71 
72  strategy_x = jax.nn.softmax(x)
73  strategy_y = jnp.transpose(jax.nn.softmax(y), [0, 2, 1])
74 
75  values_x = jnp.matmul(payoff, strategy_y)
76  values_y = -jnp.matmul(strategy_x, payoff)
77 
78  value_x = jnp.matmul(jnp.matmul(strategy_x, payoff), strategy_y)
79  value_y = -value_x
80 
81  curren_regret_x = values_x - value_x
82  curren_regret_y = values_y - value_y
83  curren_regret_x = jnp.transpose(curren_regret_x, [0, 2, 1])
84 
85  regret_sum_x += curren_regret_x
86  regret_sum_y += curren_regret_y
87 
88  current_loss = jnp.max(
89  jax.numpy.concatenate([curren_regret_x, curren_regret_y], axis=2),
90  axis=[1, 2])
91  total_loss += current_loss
92  return total_loss
93  def fori_loop(lower, steps, body_fun, total_loss):
94  val = total_loss
95  for i in range(lower, steps):
96  val = body_fun(i, total_loss)
97  return val
98  total_loss = fori_loop(0, steps, body_fun, total_loss)
99  return jnp.mean(total_loss)
