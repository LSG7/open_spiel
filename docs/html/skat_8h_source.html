22 // A slightly simplified version of Skat.
23 // See https://en.wikipedia.org/wiki/Skat_(card_game)
24 // This is a 3 player trick-based card game. After cards are dealed a bidding
25 // phase decides which player is the solo player and what game type is played.
26 // After the bidding phase, there is a phase where the solo players takes up the
27 // two cards in the Skat and discards two cards (whose points are secured for
28 // the solo player). After that the playing phase starts.
29 //
30 // Currently the bidding is vastly simplified. The players are allowed to make
31 // bids or not in order. The first player who makes a bid is the solo player.
32 // Allowed bids are only the 6 game types (4 suits, Grand & Null). This means
33 // Hand and Ouvert games are currently not implemented.
34 //
35 // The play phase consists of 10 tricks. The utility is the points made minus 60
36 // and divided by 120 for the solo player and 240 for the team players. This
37 // makes it a zero-sum game and since there are 120 points in total, each side
38 // gets a positive score if they get more than half the points.
39 //
40 // The action space is as follows:
41 // 0..31 Cards, used for dealing, discarding and playing cards.
42 // 32+ Bidding, currently you can only bid for a game type.

108 // This is the information about one trick, i.e. up to three cards where each
109 // card was played by one player.
110 class Trick {
111  public:
112  Trick() : Trick{-1} {}
113  Trick(Player leader) { leader_ = leader; }
114  int FirstCard() const;
115  Player Leader() const { return leader_; }
116  // How many cards have been played in the trick. Between 0 and 3.
117  int CardsPlayed() const { return cards_.size(); }
118  // Returns a vector of the cards played in this trick. These are ordered by
119  // the order of play, i.e. the first card is not necessarily played by player
120  // 1 but by the player who played first in this trick.
121  std::vector<int> GetCards() const { return cards_; }
122  // Adds `card` to the trick as played by player with id `player`.
123  void PlayCard(int card);
124  // Returns the player id of the player who was at position `position` in this
125  // trick. Position is 0 based here, i.e. PlayerAtPosition(0) returns the
126  // player who played the first card in this trick. This method fails if no
127  // cards have been played yet.
128  int PlayerAtPosition(int position) const;
129  // Returns the sum of the values of the cards in the trick.
130  int Points() const;
131  std::string ToString() const;
132 
133  private:
134  std::vector<int> cards_{};
135  Player leader_;
136  Suit led_suit_;
137 };
