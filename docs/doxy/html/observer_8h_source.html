18// This class is the primary method for getting observations from games.
19// Each Game object has a MakeObserver() method which returns an Observer
20// object given a specification of the required observation type.
21
22// To access observation from C++, first initialize an observer and observation
23// for the game (one time only).
24//
25// auto observer = game->MakeObserver(iig_obs_type, params);
26// Observation observation(*game, observer);
27//
28// Then for states in a trajectory, get a tensor observation using:
29//
30// observation.SetFrom(state, player); // populates observation.Tensor()
31//
32// The resultant tensor is accessible from observation.Tensor(). Note that
33// the decomposition of the tensor into named pieces is not currently available
34// through this API (it is available in Python).
35//
36// To obtain a string observation:
37//
38// std::string string_obs = observation.StringFrom(state, player);
39//
40// Access from Python follows a similar pattern, with the addition of support
41// for accessing pieces of the observation tensor by name. See `observation.py`
42// and `observation_test.py`.
43

73 : name_(name), shape_(shape) {}
74
75 inline const std::string& name() const { return name_; }
76 inline const Shape& shape() const { return shape_; }
77
78 // Convenience accessor for the shape as a plain vector of ints.
79 template <typename int_type = int32_t>
80 inline std::vector<int_type> vector_shape() const {
81 return {shape_.begin(), shape_.end()};
82 }
83
84 // Number of floats in a tensor.
85 int32_t size() const {
86 return std::accumulate(shape_.begin(), shape_.end(), 1,
87 std::multiplies<int32_t>());
88 }

91 int32_t nbytes() const { return size() * sizeof(float); }

93 std::string DebugString() const {
94 return absl::StrCat("SpanTensor(name='", name(), "', shape=(",
95 absl::StrJoin(shape_, ","), "), nbytes=", nbytes(),
96 ")");
97 }
98
99 private:
100 std::string name_;
101 Shape shape_;
102};

104// A tensor backed up by a data buffer *not* owned by SpanTensor.
105//
106// This is a view class that points to some externally owned data buffer
107// and helps with accessing and modifying the data via its `at` methods.
108//
109// The class has the pointer semantics, akin to `std::unique_ptr` or a raw
110// pointer, where `SpanTensor` just "points" to an array.
111// In particular helper accessor methods like `data` or `at` are marked as const
112// but still give access to mutable data.
113class SpanTensor {
114 public:
115 SpanTensor(SpanTensorInfo info, absl::Span<float> data)
116 : info_(std::move(info)), data_(data) {
117 SPIEL_CHECK_EQ(info_.size(), data_.size());
118 }
119
120 const SpanTensorInfo& info() const { return info_; }
121
122 absl::Span<float> data() const { return data_; }
123
124 std::string DebugString() const { return info_.DebugString(); }
125
126 // Mutators of data.
127 float& at() const {
128 SPIEL_DCHECK_EQ(info_.shape().size(), 0);
129 return data_[0];
130 }
131
132 float& at(int idx) const {
133 SPIEL_DCHECK_EQ(info_.shape().size(), 1);
134 return data_[idx];
135 }
136
137 float& at(int idx1, int idx2) const {
138 SPIEL_DCHECK_EQ(info_.shape().size(), 2);
139 return data_[idx1 * info_.shape()[1] + idx2];
140 }
141
142 float& at(int idx1, int idx2, int idx3) const {
143 SPIEL_DCHECK_EQ(info_.shape().size(), 3);
144 return data_[(idx1 * info_.shape()[1] + idx2) * info_.shape()[2] + idx3];
145 }
146
147 float& at(int idx1, int idx2, int idx3, int idx4) const {
148 SPIEL_DCHECK_EQ(info_.shape().size(), 4);
149 return data_[((idx1 * info_.shape()[1] + idx2) * info_.shape()[2] + idx3) *
150 info_.shape()[3] +
151 idx4];
152 }
153
154 private:
155 SpanTensorInfo info_;
156 absl::Span<float> data_;
157};

220// Observation types for imperfect-information games.
221
222// The public / private observations factorize observations into their
223// (mostly) non-overlapping public and private parts. They may overlap only for
224// the start of the game and time.
225//
226// The public observations correspond to information that all the players know
227// that all the players know, like upward-facing cards on a table.
228// Perfect information games, like Chess, have only public observations.
229//
230// All games have non-empty public observations. The minimum public
231// information is time: we assume that all the players can perceive absolute
232// time (which can be accessed via the MoveNumber() method). The implemented
233// games must be 1-timeable, a property that is trivially satisfied with all
234// human-played board games, so you typically don't have to worry about this.
235// (You'd have to knock players out / consider Einstein's time-relativistic
236// effects to make non-timeable games.).
237//
238// The public observations are used to create a sequence of observations:
239// a public observation history. Because of the sequential structure, when you
240// return any non-empty public observation, you implicitly encode time as well
241// within this sequence.
242//
243// Public observations are not required to be "common knowledge" observations.
244// Example: In imperfect-info version of card game Goofspiel, players make
245// bets with cards on their hand, and their imperfect information consists of
246// not knowing exactly what cards the opponent currently holds, as the players
247// only learn public information whether they have won/lost/draw the bet.
248// However, when the player bets a card "5" and learns it drew the round,
249// it can infer that the opponent must have also bet the card "5", just as the
250// player did. In principle we could ask the game to make this inference
251// automatically, and return observation "draw-5". We do not require this, as
252// it is in general expensive to compute. Returning public observation "draw"
253// is sufficient.
254
255// The private observations correspond to the part of the observation that
256// is not public. In Poker, this would be the cards the player holds in his
257// hand. Note that this does not imply that other players don't have access
258// to this information.
259//
260// For example, consider there is a mirror behind an unaware player, betraying
261// his hand in the reflection. Even if everyone was aware of the mirror, then
262// this information still may not be public, because the players do not know
263// for certain that everyone is aware of this. It would become public if and
264// only if all the players were aware of the mirror, and they also knew that
265// indeed everyone else knows about it too. Then this would effectively make
266// it the same as if the player just placed his cards on the table for
267// everyone to see.
268//
269// If there is no private observation available, the implementation should
270// return an empty string.
271struct IIGObservationType {
272 // If true, include public information in the observation.
273 bool public_info;
274
275 // Whether the observation is perfect recall (identical to an info state).
276 // If true, the observation must be sufficient to reconstruct the complete
277 // history of actions and observations for the observing player.
278 bool perfect_recall;
279
280 // Which players' private information to include in the observation.
281 PrivateInfoType private_info;
282
283 bool operator==(const IIGObservationType&);
284};

318// An Observer is something which can produce an observation of a State,
319// e.g. a Tensor or collection of Tensors or a string.
320// Observers are game-specific. They are created by a Game object, and
321// may only be applied to a State class generated from the same Game instance.
322class Observer {
323 public:
324 Observer(bool has_string, bool has_tensor)
325 : has_string_(has_string), has_tensor_(has_tensor) {
326 SPIEL_CHECK_TRUE(has_string || has_tensor);
327 }
328
329 // Write a tensor observation to the memory returned by the Allocator.
330 virtual void WriteTensor(const State& state, int player,
331 Allocator* allocator) const = 0;
332
333 // Return a string observation. For human-readability or for tabular
334 // algorithms on small games.
335 virtual std::string StringFrom(const State& state, int player) const = 0;
336
337 // What observations do we support?
338 bool HasString() const { return has_string_; }
339 bool HasTensor() const { return has_tensor_; }
340
341 virtual ~Observer() = default;
342
343 protected:
344 // TODO(author11) Remove when all games support both types of observations.
345 bool has_string_;
346 bool has_tensor_;
347};

392 // TODO(author11) Remove when all games support both types of observations.

396 public:

398
