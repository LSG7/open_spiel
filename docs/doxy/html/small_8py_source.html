24class MatrixGame(object):
25 """Matrix Game."""
26
27 def __init__(self, pt, seed=None):
28 """Ctor. Inits payoff tensor (players x actions x ... np.array).
29
30 Args:
31 pt: payoff tensor, np.array
32 seed: seed for random number generator, used if computing best responses
33 """
34 if np.any(pt < 0.):
35 raise ValueError("Payoff tensor must contain non-negative values")
36 self.pt = pt
37
38 self.seed = seed
39 self.random = np.random.RandomState(seed)
40
41 def num_players(self):
42 return self.pt.shape[0]
43
44 def num_strategies(self):
45 return self.pt.shape[1:]
46
47 def payoff_tensor(self):
48 return self.pt
49
50 def get_payoffs_for_strategies(self, policies):
51 """Return vector of payoffs for all players given list of strategies.
52
53 Args:
54 policies: list of integers indexing strategies for each player
55 Returns:
56 np.array (length num players) of payoffs
57 """
58 return self.pt[:, policies[0], policies[1]]
59
60 def best_response(self, mixed_strategy, return_exp=False):
61 """Return best response and its superiority over the current strategy.
62
63 Args:
64 mixed_strategy: np.ndarray (distribution over strategies)
65 return_exp: bool, whether to return how much best response exploits the
66 given mixed strategy (default is False)
67 Returns:
68 br: int, index of strategy (ties split randomly)
69 exp: u(br) - u(mixed_strategy)
70 """
71 logging.warn("Assumes symmetric game! Returns br for player 0.")
72 gradient = self.pt[0].dot(mixed_strategy)
73 br = misc.argmax(self.random, gradient)
74 exp = gradient.max() - gradient.dot(mixed_strategy)
75 if return_exp:
76 return br, exp
77 else:
78 return br
79
80 def best_population_response(self, dist, policies):
81 """Returns the best response to the current population of policies.
82
83 Args:
84 dist: np.ndarray, distribution over policies
85 policies: list of integers indexing strategies for each player
86 Returns:
87 best response, exploitability tuple (see best_response)
88 """
89 ns = self.num_strategies()
90 mixed_strat = np.zeros(ns)
91 for pure_strat, prob in zip(policies, dist):
92 mixed_strat[pure_strat] += prob
93 return self.best_response(mixed_strat)

96class BiasedGame(MatrixGame):
97 """2-Player, 3-Action symmetric game with biased stochastic best responses."""
98
99 def __init__(self, seed=None):
100 """Ctor. Initializes payoff tensor (2 x 3 x 3 np.array).
101
102 Args:
103 seed: seed for random number generator, used if computing best responses
104 """
105 # pylint:disable=bad-whitespace
106 pt_r = np.array([[0, 0, 0 ],
107 [1, -2, .5],
108 [-2, 1, -1]]) + 2.
109 # pylint:enable=bad-whitespace
110 pt_c = pt_r.T # symmetric game
111 pt = np.stack((pt_r, pt_c), axis=0).astype(float)
112 pt /= pt.max() # arbitrary design choice to upper bound entries to 1
113 super().__init__(pt, seed)

162class SpiralGame(MatrixGame):
163 """2-Player, 3-Action symmetric game with spiral dynamics on simplex."""
164
165 def __init__(self, center=None, seed=None):
166 """Ctor. Initializes payoff tensor (2 x 3 x 3 np.array).
167
168 Args:
169 center: center of cycle given in [x, y, z] Euclidean coordinates
170 seed: seed for random number generator, used if computing best responses
171 """
172 if center is None:
173 center = np.ones(3) / 3.
174 else:
175 if not ((np.sum(center) <= 1 + 1e-8) and np.all(center >= -1e-8)):
176 raise ValueError("center must lie on simplex")
177 self.center = center
178 center = center.reshape((3, 1))
179
180 # define coordinate frame for simplex; basis vectors on columns of transform
181 transform = np.array([[.5, -.5, 0], [-.5, -.5, 1], [1, 1, 1]]).T
182 transform /= np.linalg.norm(transform, axis=0)
183 transform_inv = np.linalg.inv(transform)
184
185 # canonical cycle matrix in 2-d
186 cycle = 0.1 * np.array([[0, 1, 0], [1, 0, 0], [0, 0, 0]])
187
188 # payoff tensor maps euclidean to simplex frame, applies cycle, maps back
189 pt_r = transform.dot(cycle.dot(transform_inv))
190 # subtracting off a column vector effectively offsets the vector field
191 # because [[c c c], ...] [[x], [y], [z]] = [c * (x + y + z), ...] = [c, ...]
192 pt_r -= pt_r.dot(center)
193 # shift tensor to ensure positivity required for ATE
194 if pt_r.min() < 0:
195 pt_r -= pt_r.min()
196
197 pt_c = pt_r.T # symmetric game
198 pt = np.stack((pt_r, pt_c), axis=0).astype(float)
199 super().__init__(pt, seed)

224class Shapleys(MatrixGame):

226
227 def __init__(self, beta=1., seed=None):
228 """Ctor. Initializes payoff tensor (2 x 2 x 2 np.array).
229
230 See Eqn 4 in https://arxiv.org/pdf/1308.4049.pdf.
231
232 Args:
233 beta: float, modifies the game so that the utilities @ Nash are now
234 u_1(Nash) = (1 + beta) / 3 and u_2(Nash) = (1 - beta) / 3
235 where Nash is the joint uniform distribution
236 seed: seed for random number generator, used if computing best responses
237 """
238 # pylint:disable=bad-whitespace
239 pt_r = np.array([[1, 0, beta],
240 [beta, 1, 0],
241 [0, beta, 1]])
242 pt_c = np.array([[-beta, 1, 0],
243 [0, -beta, 1],
244 [1, 0, -beta]])
245 # pylint:enable=bad-whitespace
246 pt = np.stack((pt_r, pt_c), axis=0).astype(float)
247 # shift tensor to ensure positivity required for ATE
248 pt -= pt.min()
