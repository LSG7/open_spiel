206  for info_state in player_info_states:

214  check_avg_policy_is_uniform_random()

218  # The acting player in 1b is player 1 and they have not acted before, so
219  # the probability this player plays to this information state is 1, and
220  # the sequence probability of any action is just the probability of that
221  # action given the information state. On the first iteration, this
222  # probability is 0.5 for both actions. On the second iteration, the
223  # current policy is [0, 1], so the average cumulants should be
224  # [0.5, 1.5]. Normalizing this gives the average policy.
225  normalization = 0.5 + 0.5 + 1
226  np.testing.assert_allclose(cfr_solver.average_policy().policy_for_key("1b"),
227  [0.5 / normalization, (0.5 + 1) / normalization])
