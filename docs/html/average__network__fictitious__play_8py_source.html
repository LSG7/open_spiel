14 """Implementation of the Deep Average-network Fictitious Play.
15 
16 Coupled with agents that compute a best-response (BR) at each iteration, instead
17 of keeping in memory all the BRs from past iterations Deep Average-network
18 Fictitious Play learns along the way the policy generating the average
19 distribution. This is done by keeping a buffer of state-action pairs generated
20 by past BRs and learning the average policy (represented by a neural network) by
21 minimizing a categorical loss. This approach is inspired by the Neural
22 Fictitious Self Play (NFSP) method (Heinrich & Silver, 2016), developed
23 initially for imperfect information games with a finite number of players, and
24 adapted here to the MFG setting.
25 """

145  def _act(self, info_state, legal_actions) -> Tuple[int, np.ndarray]:
146  """Returns an action and the action probabilities."""
147  info_state = np.reshape(info_state, [1, -1])
148  action_probs = self._avg_network_policy_avg_network_policy(self._params_avg_network_params_avg_network,
149  info_state)

152  action_probs = np.asarray(action_probs)
153  probs[legal_actions] = action_probs[0][legal_actions]
154  probs /= sum(probs)

216  avg_logit = self.avg_networkavg_network.apply(param_avg, info_states)
217  loss_value = -jnp.sum(
