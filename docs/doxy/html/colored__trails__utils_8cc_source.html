31void InitChipCombosRec(TradeInfo* trade_info, int num_colors,
32 std::string cur_combo_str) {
33 if (cur_combo_str.length() > 0 &&
34 cur_combo_str.length() <= kNumChipsUpperBound) {
35 trade_info->chip_combinations.push_back(
36 ComboStringToCombo(cur_combo_str, num_colors));
37 } else if (cur_combo_str.length() > kNumChipsUpperBound) {
38 return;
39 }
40
41 int last_color =
42 (cur_combo_str.empty() ? 0 : CharToColor(cur_combo_str.back()));
43 for (int c = last_color; c < num_colors; ++c) {
44 std::string child = cur_combo_str;
45 child.push_back(ColorToChar(c));
46 InitChipCombosRec(trade_info, num_colors, child);
47 }
48}

50int ManhattanDistance(const Board& board, int pos1, int pos2) {
51 int r1 = pos1 / board.size, c1 = pos1 % board.size;
52 int r2 = pos2 / board.size, c2 = pos2 % board.size;
53 return std::abs(r2 - r1) + std::abs(c2 - c1);
54}

56int CurrentScore(Player p, const Board& board) {
57 int score = std::accumulate(board.chips[p].begin(), board.chips[p].end(), 0) *
58 kLeftoverChipScore;
59 score += kFlagPenaltyPerCell *
60 ManhattanDistance(board, board.positions[p], board.positions.back());
61 return score;
62}

64int ScoreRec(Player player, const Board& board, bool* solved) {
65 int score = CurrentScore(player, board);
66 int row = board.positions[player] / board.size;
67 int col = board.positions[player] % board.size;
68
69 if (board.positions.back() == board.positions[player]) {
70 // We found the goal. This has to be the maximal score: terminate recursion.
71 *solved = true;
72 return score;
73 }
74
75 for (int dir = 0; dir < kNumDirections; ++dir) {
76 int rp = row + kRowOffsets[dir];
77 int cp = col + kColumnOffsets[dir];
78 if (board.InBounds(rp, cp)) { // Check this position is in bounds.
79 int pos = rp * board.size + cp;
80 int color = board.board[pos];
81 if (board.chips[player][color] > 0) {
82 // If this player has a chip to travel here, then move them and call
83 // score on the child board.
84 Board child_board = board;
85 child_board.chips[player][color]--;
86 child_board.positions[player] = pos;
87 int child_score = ScoreRec(player, child_board, solved);
88 score = std::max(score, child_score);
89 }
90 }
91 }
92
93 return score;
94}

98ChipComboIterator::ChipComboIterator(const std::vector<int>& chips)
99 : chips_(chips), cur_combo_(chips.size(), 0) {
100 SPIEL_CHECK_GT(std::accumulate(chips_.begin(), chips_.end(), 0), 0);
101}

103bool ChipComboIterator::IsFinished() const {
104 // If every digit is maximized, we are done.
105 return cur_combo_ == chips_;
106}

108std::vector<int> ChipComboIterator::Next() {
109 // Try to increase the left-most non-maximized chip with non-zero chips. Then
110 // reset every digit to the left of it with nonzero chips.
111 for (int inc_idx = 0; inc_idx < chips_.size(); ++inc_idx) {
112 if (cur_combo_[inc_idx] < chips_[inc_idx]) {
113 cur_combo_[inc_idx]++;
114 for (int j = inc_idx - 1; j >= 0; --j) {
115 cur_combo_[j] = 0;
116 }
117 break;
118 }
119 }
120 return cur_combo_;
121}

123std::vector<int> ComboStringToCombo(const std::string& combo_str,
124 int num_colors) {
125 std::vector<int> combo(num_colors, 0);
126 for (int i = 0; i < combo_str.length(); ++i) {
127 int color = CharToColor(combo_str[i]);
128 combo[color]++;
129 }
130 return combo;
131}

133std::string ComboToString(const std::vector<int>& combo) {
134 std::string combo_str;
135 for (int i = 0; i < combo.size(); ++i) {
136 for (int k = 0; k < combo[i]; ++k) {
137 combo_str.push_back(ColorToChar(i));
138 }
139 }
140 return combo_str;
141}

147void InitTradeInfo(TradeInfo* trade_info, int num_colors) {
148 InitChipCombosRec(trade_info, num_colors, "");
149 for (int i = 0; i < trade_info->chip_combinations.size(); ++i) {
150 for (int j = 0; j < trade_info->chip_combinations.size(); ++j) {
151 Trade candidate(trade_info->chip_combinations[i],
152 trade_info->chip_combinations[j]);
153 bool valid = candidate.reduce();
154 if (!valid) {
155 continue;
156 }
157
158 std::string candidate_str = candidate.ToString();
159
160 if (trade_info->trade_str_to_id.find(candidate_str) ==
161 trade_info->trade_str_to_id.end()) {
162 // std::cout << "Valid trade: " << candidate_str << std::endl;
163 trade_info->possible_trades.push_back(
164 std::make_unique<Trade>(candidate));
165 trade_info->trade_str_to_id[candidate_str] =
166 trade_info->possible_trades.size() - 1;
167 }
168 }
169 }
170}

178void ParseBoardsString(std::vector<Board>* boards,
179 const std::string& boards_string,
180 int num_colors, int board_size, int num_players) {
181 std::vector<std::string> lines = absl::StrSplit(boards_string, '\n');
182 SPIEL_CHECK_GT(lines.size(), 1);
183 for (const std::string& line : lines) {
184 if (!line.empty()) {
185 Board board(board_size, num_colors, num_players);
186 board.ParseFromLine(line);
187 boards->push_back(board);
188 }
189 }
190}
