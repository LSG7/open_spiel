40class EntropySchedule:
41 """An increasing list of steps where the regularisation network is updated.
42
43 Example
44 EntropySchedule([3, 5, 10], [2, 4, 1])
45 => [0, 3, 6, 11, 16, 21, 26, 36]
46 | 3 x2 | 5 x4 | 10 x1
47 """
48
49 def __init__(self, *, sizes: Sequence[int], repeats: Sequence[int]):
50 """Constructs a schedule of entropy iterations.
51
52 Args:
53 sizes: the list of iteration sizes.
54 repeats: the list, parallel to sizes, with the number of times for each
55 size from `sizes` to repeat.
56 """
57 try:
58 if len(repeats) != len(sizes):
59 raise ValueError("`repeats` must be parallel to `sizes`.")
60 if not sizes:
61 raise ValueError("`sizes` and `repeats` must not be empty.")
62 if any([(repeat <= 0) for repeat in repeats]):
63 raise ValueError("All repeat values must be strictly positive")
64 if repeats[-1] != 1:
65 raise ValueError("The last value in `repeats` must be equal to 1, "
66 "ince the last iteration size is repeated forever.")
67 except ValueError as e:
68 raise ValueError(
69 f"Entropy iteration schedule: repeats ({repeats}) and sizes"
70 f" ({sizes})."
71 ) from e
72
73 schedule = [0]
74 for size, repeat in zip(sizes, repeats):
75 schedule.extend([schedule[-1] + (i + 1) * size for i in range(repeat)])
76
77 self.schedule = np.array(schedule, dtype=np.int32)
78
79 def __call__(self, learner_step: int) -> Tuple[float, bool]:
80 """Entropy scheduling parameters for a given `learner_step`.
81
82 Args:
83 learner_step: The current learning step.
84
85 Returns:
86 alpha: The mixing weight (from [0, 1]) of the previous policy with
87 the one before for computing the intrinsic reward.
88 update_target_net: A boolean indicator for updating the target network
89 with the current network.
90 """
91
92 # The complexity below is because at some point we might go past
93 # the explicit schedule, and then we'd need to just use the last step
94 # in the schedule and apply the logic of
95 # ((learner_step - last_step) % last_iteration) == 0)
96
97 # The schedule might look like this:
98 # X----X-------X--X--X--X--------X
99 # learner_step | might be here ^ |
100 # or there ^ |
101 # or even past the schedule ^
102
103 # We need to deal with two cases below.
104 # Instead of going for the complicated conditional, let's just
105 # compute both and then do the A * s + B * (1 - s) with s being a bool
106 # selector between A and B.
107
108 # 1. assume learner_step is past the schedule,
109 # ie schedule[-1] <= learner_step.
110 last_size = self.schedule[-1] - self.schedule[-2]
111 last_start = self.schedule[-1] + (
112 learner_step - self.schedule[-1]) // last_size * last_size
113 # 2. assume learner_step is within the schedule.
114 start = jnp.amax(self.schedule * (self.schedule <= learner_step))
115 finish = jnp.amin(
116 self.schedule * (learner_step < self.schedule),
117 initial=self.schedule[-1],
118 where=(learner_step < self.schedule))
119 size = finish - start
120
121 # Now select between the two.
122 beyond = (self.schedule[-1] <= learner_step) # Are we past the schedule?
123 iteration_start = (last_start * beyond + start * (1 - beyond))
124 iteration_size = (last_size * beyond + size * (1 - beyond))
125
126 update_target_net = jnp.logical_and(
127 learner_step > 0,
128 jnp.sum(learner_step == iteration_start + iteration_size - 1),
129 )
130 alpha = jnp.minimum(
131 (2.0 * (learner_step - iteration_start)) / iteration_size, 1.0)
132
133 return alpha, update_target_net # pytype: disable=bad-return-type # jax-types

136@chex.dataclass(frozen=True)
137class FineTuning:
138 """Fine tuning options, aka policy post-processing.
139
140 Even when fully trained, the resulting softmax-based policy may put
141 a small probability mass on bad actions. This results in an agent
142 waiting for the opponent (itself in self-play) to commit an error.
143
144 To address that the policy is post-processed using:
145 - thresholding: any action with probability smaller than self.threshold
146 is simply removed from the policy.
147 - discretization: the probability values are rounded to the closest
148 multiple of 1/self.discretization.
149
150 The post-processing is used on the learner, and thus must be jit-friendly.
151 """
152 # The learner step after which the policy post processing (aka finetuning)
153 # will be enabled when learning. A strictly negative value is equivalent
154 # to infinity, ie disables finetuning completely.
155 from_learner_steps: int = -1
156 # All policy probabilities below `threshold` are zeroed out. Thresholding
157 # is disabled if this value is non-positive.
158 policy_threshold: float = 0.03
159 # Rounds the policy probabilities to the "closest"
160 # multiple of 1/`self.discretization`.
161 # Discretization is disabled for non-positive values.
162 policy_discretization: int = 32
163
164 def __call__(self, policy: chex.Array, mask: chex.Array,
165 learner_steps: int) -> chex.Array:
166 """A configurable fine tuning of a policy."""
167 chex.assert_equal_shape((policy, mask))
168 do_finetune = jnp.logical_and(self.from_learner_stepsfrom_learner_steps >= 0,
169 learner_steps > self.from_learner_stepsfrom_learner_steps)
170
171 return jnp.where(do_finetune, self.post_process_policy(policy, mask),
172 policy)
173
174 def post_process_policy(
175 self,
176 policy: chex.Array,
177 mask: chex.Array,
178 ) -> chex.Array:
179 """Unconditionally post process a given masked policy."""
180 chex.assert_equal_shape((policy, mask))
181 policy = self._threshold(policy, mask)
182 policy = self._discretize(policy)
183 return policy
184
185 def _threshold(self, policy: chex.Array, mask: chex.Array) -> chex.Array:
186 """Remove from the support the actions 'a' where policy(a) < threshold."""
187 chex.assert_equal_shape((policy, mask))
188 if self.policy_threshold <= 0:
189 return policy
190
191 mask = mask * (
192 # Values over the threshold.
193 (policy >= self.policy_threshold) +
194 # Degenerate case is when policy is less than threshold *everywhere*.
195 # In that case we just keep the policy as-is.
196 (jnp.max(policy, axis=-1, keepdims=True) < self.policy_threshold))
197 return mask * policy / jnp.sum(mask * policy, axis=-1, keepdims=True)
198
199 def _discretize(self, policy: chex.Array) -> chex.Array:
200 """Round all action probabilities to a multiple of 1/self.discretize."""
201 if self.policy_discretization <= 0:
202 return policy
203
204 # The unbatched/single policy case:
205 if len(policy.shape) == 1:
206 return self._discretize_single(policy)
207
208 # policy may be [B, A] or [T, B, A], etc. Thus add hk.BatchApply.
209 dims = len(policy.shape) - 1
210
211 # TODO(author18): avoid mixing vmap and BatchApply since the two could
212 # be folded into either a single BatchApply or a sequence of vmaps, but
213 # not the mix.
214 vmapped = jax.vmap(self._discretize_single)
215 policy = hk.BatchApply(vmapped, num_dims=dims)(policy)
216
217 return policy
218
219 def _discretize_single(self, mu: chex.Array) -> chex.Array:
220 """A version of self._discretize but for the unbatched data."""
221 # TODO(author18): try to merge _discretize and _discretize_single
222 # into one function that handles both batched and unbatched cases.
223 if len(mu.shape) == 2:
224 mu_ = jnp.squeeze(mu, axis=0)
225 else:
226 mu_ = mu
227 n_actions = mu_.shape[-1]
228 roundup = jnp.ceil(mu_ * self.policy_discretization).astype(jnp.int32)
229 result = jnp.zeros_like(mu_)
230 order = jnp.argsort(-mu_) # Indices of descending order.
231 weight_left = self.policy_discretization
232
233 def f_disc(i, order, roundup, weight_left, result):
234 x = jnp.minimum(roundup[order[i]], weight_left)
235 result = jax.numpy.where(weight_left >= 0, result.at[order[i]].add(x),
236 result)
237 weight_left -= x
238 return i + 1, order, roundup, weight_left, result
239
240 def f_scan_scan(carry, x):
241 i, order, roundup, weight_left, result = carry
242 i_next, order_next, roundup_next, weight_left_next, result_next = f_disc(
243 i, order, roundup, weight_left, result)
244 carry_next = (i_next, order_next, roundup_next, weight_left_next,
245 result_next)
246 return carry_next, x
247
248 (_, _, _, weight_left_next, result_next), _ = jax.lax.scan(
249 f_scan_scan,
250 init=(jnp.asarray(0), order, roundup, weight_left, result),
251 xs=None,
252 length=n_actions)
253
254 result_next = jnp.where(weight_left_next > 0,
255 result_next.at[order[0]].add(weight_left_next),
256 result_next)
257 if len(mu.shape) == 2:
258 result_next = jnp.expand_dims(result_next, axis=0)
259 return result_next / self.policy_discretization

262def _legal_policy(logits: chex.Array, legal_actions: chex.Array) -> chex.Array:
263 """A soft-max policy that respects legal_actions."""
264 chex.assert_equal_shape((logits, legal_actions))
265 # Fiddle a bit to make sure we don't generate NaNs or Inf in the middle.
266 l_min = logits.min(axis=-1, keepdims=True) # 1. logits .
267 logits = jnp.where(legal_actions, logits, l_min) # 2. logits min
268 logits -= logits.max(axis=-1, keepdims=True)
269 logits *= legal_actions
270 exp_logits = jnp.where(legal_actions, jnp.exp(logits),
271 0) # Illegal actions become 0.
272 exp_logits_sum = jnp.sum(exp_logits, axis=-1, keepdims=True)
273 return exp_logits / exp_logits_sum
274

276def legal_log_policy(logits: chex.Array,
277 legal_actions: chex.Array) -> chex.Array:
278 """Return the log of the policy on legal action, 0 on illegal action."""
279 chex.assert_equal_shape((logits, legal_actions))
280 # logits_masked has illegal actions set to -inf.
281 logits_masked = logits + jnp.log(legal_actions)
282 max_legal_logit = logits_masked.max(axis=-1, keepdims=True)
283 logits_masked = logits_masked - max_legal_logit
284 # exp_logits_masked is 0 for illegal actions.
285 exp_logits_masked = jnp.exp(logits_masked)
286
287 baseline = jnp.log(jnp.sum(exp_logits_masked, axis=-1, keepdims=True))
288 # Subtract baseline from logits. We do not simply return
289 # logits_masked - baseline
290 # because that has -inf for illegal actions, or
291 # legal_actions * (logits_masked - baseline)
292 # because that leads to 0 * -inf == nan for illegal actions.
293 log_policy = jnp.multiply(legal_actions,
294 (logits - max_legal_logit - baseline))
295 return log_policy

298def _player_others(player_ids: chex.Array, valid: chex.Array,
299 player: int) -> chex.Array:
300 """A vector of 1 for the current player and -1 for others.
301
302 Args:
303 player_ids: Tensor [...] containing player ids (0 <= player_id < N).
304 valid: Tensor [...] containing whether these states are valid.
305 player: The player id as int.
306
307 Returns:
308 player_other: is 1 for the current player and -1 for others [..., 1].
309 """
310 chex.assert_equal_shape((player_ids, valid))
311 current_player_tensor = (player_ids == player).astype(jnp.int32) # pytype: disable=attribute-error # numpy-scalars
312
313 res = 2 * current_player_tensor - 1
314 res = res * valid
315 return jnp.expand_dims(res, axis=-1)

318def _policy_ratio(pi: chex.Array, mu: chex.Array, actions_oh: chex.Array,
319 valid: chex.Array) -> chex.Array:
320 """Returns a ratio of policy pi/mu when selecting action a.
321
322 By convention, this ratio is 1 on non valid states
323 Args:
324 pi: the policy of shape [..., A].
325 mu: the sampling policy of shape [..., A].
326 actions_oh: a one-hot encoding of the current actions of shape [..., A].
327 valid: 0 if the state is not valid and else 1 of shape [...].
328
329 Returns:
330 pi/mu on valid states and 1 otherwise. The shape is the same
331 as pi, mu or actions_oh but without the last dimension A.
332 """
333 chex.assert_equal_shape((pi, mu, actions_oh))
334 chex.assert_shape((valid,), actions_oh.shape[:-1])
335
336 def _select_action_prob(pi):
337 return (jnp.sum(actions_oh * pi, axis=-1, keepdims=False) * valid +
338 (1 - valid))
339
340 pi_actions_prob = _select_action_prob(pi)
341 mu_actions_prob = _select_action_prob(mu)
342 return pi_actions_prob / mu_actions_prob

345def _where(pred: chex.Array, true_data: chex.ArrayTree,
346 false_data: chex.ArrayTree) -> chex.ArrayTree:
347 """Similar to jax.where but treats `pred` as a broadcastable prefix."""
348
349 def _where_one(t, f):
350 chex.assert_equal_rank((t, f))
351 # Expand the dimensions of pred if true_data and false_data are higher rank.
352 p = jnp.reshape(pred, pred.shape + (1,) * (len(t.shape) - len(pred.shape)))
353 return jnp.where(p, t, f)
354
355 return tree.tree_map(_where_one, true_data, false_data)

358def _has_played(valid: chex.Array, player_id: chex.Array,
359 player: int) -> chex.Array:
360 """Compute a mask of states which have a next state in the sequence."""
361 chex.assert_equal_shape((valid, player_id))
362
363 def _loop_has_played(carry, x):
364 valid, player_id = x
365 chex.assert_equal_shape((valid, player_id))
366
367 our_res = jnp.ones_like(player_id)
368 opp_res = carry
369 reset_res = jnp.zeros_like(carry)
370
371 our_carry = carry
372 opp_carry = carry
373 reset_carry = jnp.zeros_like(player_id)
374
375 # pyformat: disable
376 return _where(valid, _where((player_id == player),
377 (our_carry, our_res),
378 (opp_carry, opp_res)),
379 (reset_carry, reset_res))
380 # pyformat: enable
381
382 _, result = lax.scan(
383 f=_loop_has_played,
384 init=jnp.zeros_like(player_id[-1]),
385 xs=(valid, player_id),
386 reverse=True)
387 return result

390# V-Trace
391#
392# Custom implementation of VTrace to handle trajectories having a mix of
393# different player steps. The standard rlax.vtrace can't be applied here
394# out of the box because a trajectory could look like '121211221122'.
395
396
397def v_trace(
398 v: chex.Array,
399 valid: chex.Array,
400 player_id: chex.Array,
401 acting_policy: chex.Array,
402 merged_policy: chex.Array,
403 merged_log_policy: chex.Array,
404 player_others: chex.Array,
405 actions_oh: chex.Array,
406 reward: chex.Array,
407 player: int,
408 # Scalars below.
409 eta: float,
410 lambda_: float,
411 c: float,
412 rho: float,
413) -> Tuple[Any, Any, Any]:
414 """Custom VTrace for trajectories with a mix of different player steps."""
415 gamma = 1.0
416
417 has_played = _has_played(valid, player_id, player)
418
419 policy_ratio = _policy_ratio(merged_policy, acting_policy, actions_oh, valid)
420 inv_mu = _policy_ratio(
421 jnp.ones_like(merged_policy), acting_policy, actions_oh, valid)
422
423 eta_reg_entropy = (-eta *
424 jnp.sum(merged_policy * merged_log_policy, axis=-1) *
425 jnp.squeeze(player_others, axis=-1))
426 eta_log_policy = -eta * merged_log_policy * player_others
427
428 @chex.dataclass(frozen=True)
429 class LoopVTraceCarry:
430 """The carry of the v-trace scan loop."""
431 reward: chex.Array
432 # The cumulated reward until the end of the episode. Uncorrected (v-trace).
433 # Gamma discounted and includes eta_reg_entropy.
434 reward_uncorrected: chex.Array
435 next_value: chex.Array
436 next_v_target: chex.Array
437 importance_sampling: chex.Array
438
439 init_state_v_trace = LoopVTraceCarry(
440 reward=jnp.zeros_like(reward[-1]),
441 reward_uncorrected=jnp.zeros_like(reward[-1]),
442 next_value=jnp.zeros_like(v[-1]),
443 next_v_target=jnp.zeros_like(v[-1]),
444 importance_sampling=jnp.ones_like(policy_ratio[-1]))
445
446 def _loop_v_trace(carry: LoopVTraceCarry, x) -> Tuple[LoopVTraceCarry, Any]:
447 (cs, player_id, v, reward, eta_reg_entropy, valid, inv_mu, actions_oh,
448 eta_log_policy) = x
449
450 reward_uncorrected = (
451 reward + gamma * carry.reward_uncorrected + eta_reg_entropy)
452 discounted_reward = reward + gamma * carry.reward
453
454 # V-target:
455 our_v_target = (
456 v + jnp.expand_dims(
457 jnp.minimum(rho, cs * carry.importance_sampling), axis=-1) *
458 (jnp.expand_dims(reward_uncorrected, axis=-1) +
459 gamma * carry.next_value - v) + lambda_ * jnp.expand_dims(
460 jnp.minimum(c, cs * carry.importance_sampling), axis=-1) * gamma *
461 (carry.next_v_target - carry.next_value))
462
463 opp_v_target = jnp.zeros_like(our_v_target)
464 reset_v_target = jnp.zeros_like(our_v_target)
465
466 # Learning output:
467 our_learning_output = (
468 v + # value
469 eta_log_policy + # regularisation
470 actions_oh * jnp.expand_dims(inv_mu, axis=-1) *
471 (jnp.expand_dims(discounted_reward, axis=-1) + gamma * jnp.expand_dims(
472 carry.importance_sampling, axis=-1) * carry.next_v_target - v))
473
474 opp_learning_output = jnp.zeros_like(our_learning_output)
475 reset_learning_output = jnp.zeros_like(our_learning_output)
476
477 # State carry:
478 our_carry = LoopVTraceCarry(
479 reward=jnp.zeros_like(carry.reward),
480 next_value=v,
481 next_v_target=our_v_target,
482 reward_uncorrected=jnp.zeros_like(carry.reward_uncorrected),
483 importance_sampling=jnp.ones_like(carry.importance_sampling))
484 opp_carry = LoopVTraceCarry(
485 reward=eta_reg_entropy + cs * discounted_reward,
486 reward_uncorrected=reward_uncorrected,
487 next_value=gamma * carry.next_value,
488 next_v_target=gamma * carry.next_v_target,
489 importance_sampling=cs * carry.importance_sampling)
490 reset_carry = init_state_v_trace
491
492 # Invalid turn: init_state_v_trace and (zero target, learning_output)
493 # pyformat: disable
494 return _where(valid, # pytype: disable=bad-return-type # numpy-scalars
495 _where((player_id == player),
496 (our_carry, (our_v_target, our_learning_output)),
497 (opp_carry, (opp_v_target, opp_learning_output))),
498 (reset_carry, (reset_v_target, reset_learning_output)))
499 # pyformat: enable
500
501 _, (v_target, learning_output) = lax.scan(
502 f=_loop_v_trace,
503 init=init_state_v_trace,
504 xs=(policy_ratio, player_id, v, reward, eta_reg_entropy, valid, inv_mu,
505 actions_oh, eta_log_policy),
506 reverse=True)
507
508 return v_target, has_played, learning_output

511def get_loss_v(v_list: Sequence[chex.Array],
512 v_target_list: Sequence[chex.Array],
513 mask_list: Sequence[chex.Array]) -> chex.Array:
514 """Define the loss function for the critic."""
515 chex.assert_trees_all_equal_shapes(v_list, v_target_list)
516 # v_list and v_target_list come with a degenerate trailing dimension,
517 # which mask_list tensors do not have.
518 chex.assert_shape(mask_list, v_list[0].shape[:-1])
519 loss_v_list = []
520 for (v_n, v_target, mask) in zip(v_list, v_target_list, mask_list):
521 assert v_n.shape[0] == v_target.shape[0]
522
523 loss_v = jnp.expand_dims(
524 mask, axis=-1) * (v_n - lax.stop_gradient(v_target))**2
525 normalization = jnp.sum(mask)
526 loss_v = jnp.sum(loss_v) / (normalization + (normalization == 0.0))
527
528 loss_v_list.append(loss_v)
529 return sum(loss_v_list)

532def apply_force_with_threshold(decision_outputs: chex.Array, force: chex.Array,
533 threshold: float,
534 threshold_center: chex.Array) -> chex.Array:
535 """Apply the force with below a given threshold."""
536 chex.assert_equal_shape((decision_outputs, force, threshold_center))
537 can_decrease = decision_outputs - threshold_center > -threshold
538 can_increase = decision_outputs - threshold_center < threshold
539 force_negative = jnp.minimum(force, 0.0)
540 force_positive = jnp.maximum(force, 0.0)
541 clipped_force = can_decrease * force_negative + can_increase * force_positive
542 return decision_outputs * lax.stop_gradient(clipped_force)

545def renormalize(loss: chex.Array, mask: chex.Array) -> chex.Array:
546 """The `normalization` is the number of steps over which loss is computed."""
547 chex.assert_equal_shape((loss, mask))
548 loss = jnp.sum(loss * mask)
549 normalization = jnp.sum(mask)
550 return loss / (normalization + (normalization == 0.0))

553def get_loss_nerd(logit_list: Sequence[chex.Array],
554 policy_list: Sequence[chex.Array],
555 q_vr_list: Sequence[chex.Array],
556 valid: chex.Array,
557 player_ids: Sequence[chex.Array],
558 legal_actions: chex.Array,
559 importance_sampling_correction: Sequence[chex.Array],
560 clip: float = 100,
561 threshold: float = 2) -> chex.Array:
562 """Define the nerd loss."""
563 assert isinstance(importance_sampling_correction, list)
564 loss_pi_list = []
565 num_valid_actions = jnp.sum(legal_actions, axis=-1, keepdims=True)
566 for k, (logit_pi, pi, q_vr, is_c) in enumerate(
567 zip(logit_list, policy_list, q_vr_list, importance_sampling_correction)):
568 assert logit_pi.shape[0] == q_vr.shape[0]
569 # loss policy
570 adv_pi = q_vr - jnp.sum(pi * q_vr, axis=-1, keepdims=True)
571 adv_pi = is_c * adv_pi # importance sampling correction
572 adv_pi = jnp.clip(adv_pi, a_min=-clip, a_max=clip)
573 adv_pi = lax.stop_gradient(adv_pi)
574
575 valid_logit_sum = jnp.sum(logit_pi * legal_actions, axis=-1, keepdims=True)
576 mean_logit = valid_logit_sum / num_valid_actions
577
578 # Subtract only the mean of the valid logits
579 logits = logit_pi - mean_logit
580
581 threshold_center = jnp.zeros_like(logits)
582
583 nerd_loss = jnp.sum(
584 legal_actions *
585 apply_force_with_threshold(logits, adv_pi, threshold, threshold_center),
586 axis=-1)
587 nerd_loss = -renormalize(nerd_loss, valid * (player_ids == k))
588 loss_pi_list.append(nerd_loss)
589 return sum(loss_pi_list)

809 def loss(self, params: Params, params_target: Params, params_prev: Params,
810 params_prev_: Params, ts: TimeStep, alpha: float,
811 learner_steps: int) -> float:
812
813 #self.network = transformed
814 rollout = jax.vmap(self.network.apply, (None, 0), 0)
815 pi, v, log_pi, logit = rollout(params, ts.env)
816
817 policy_pprocessed = self.config.finetune(pi, ts.env.legal, learner_steps)
818
819 _, v_target, _, _ = rollout(params_target, ts.env)
820 _, _, log_pi_prev, _ = rollout(params_prev, ts.env)
821 _, _, log_pi_prev_, _ = rollout(params_prev_, ts.env)
822 # This line creates the reward transform log(pi(a|x)/pi_reg(a|x)).
823 # For the stability reasons, reward changes smoothly between iterations.
824 # The mixing between old and new reward transform is a convex combination
825 # parametrised by alpha.
826 log_policy_reg = log_pi - (alpha * log_pi_prev + (1 - alpha) * log_pi_prev_)
827
828 v_target_list, has_played_list, v_trace_policy_target_list = [], [], []
829 for player in range(self._game.num_players()):
830 reward = ts.actor.rewards[:, :, player] # [T, B, Player]
831 v_target_, has_played, policy_target_ = v_trace(
832 v_target,
833 ts.env.valid,
834 ts.env.player_id,
835 ts.actor.policy,
836 policy_pprocessed,
837 log_policy_reg,
838 _player_others(ts.env.player_id, ts.env.valid, player),
839 ts.actor.action_oh,
840 reward,
841 player,
842 lambda_=1.0,
843 c=self.config.c_vtrace,
844 rho=np.inf,
845 eta=self.config.eta_reward_transform)
846 v_target_list.append(v_target_)
847 has_played_list.append(has_played)
848 v_trace_policy_target_list.append(policy_target_)
849 loss_v = get_loss_v([v] * self._game.num_players(), v_target_list,
850 has_played_list)
851
852 is_vector = jnp.expand_dims(jnp.ones_like(ts.env.valid), axis=-1)
853 importance_sampling_correction = [is_vector] * self._game.num_players()
854 # Uses v-trace to define q-values for Nerd
855 loss_nerd = get_loss_nerd(
856 [logit] * self._game.num_players(), [pi] * self._game.num_players(),
857 v_trace_policy_target_list,
858 ts.env.valid,
859 ts.env.player_id,
860 ts.env.legal,
861 importance_sampling_correction,
862 clip=self.config.nerd.clip,
863 threshold=self.config.nerd.beta)
864 return loss_v + loss_nerd # pytype: disable=bad-return-type # numpy-scalars
