34 double SolveState(const State& state,
35  absl::flat_hash_map<std::string, int>& cache,
36  std::vector<VPNetModel::TrainInputs>& train_inputs) {
37  std::string state_str = state.ToString();
38  if (cache.find(state_str) != cache.end()) {
39  return train_inputs[cache[state_str]].value;
40  }
41  if (state.IsTerminal()) {
42  return state.PlayerReturn(0);
43  }
44 
45  bool max_player = state.CurrentPlayer() == 0;
46  std::vector<float> obs = state.ObservationTensor();
47  std::vector<Action> legal_actions = state.LegalActions();
48 
49  Action best_action = kInvalidAction;
50  double best_value = -2;
51  for (Action action : legal_actions) {
52  double value = SolveState(*state.Child(action), cache, train_inputs);
53  if (best_action == kInvalidAction ||
54  (max_player ? value > best_value : value < best_value)) {
55  best_action = action;
56  best_value = value;
57  }
58  }
59  ActionsAndProbs policy({{best_action, 1}});
60 
61  cache[state_str] = train_inputs.size();
62  train_inputs.push_back(
63  VPNetModel::TrainInputs{legal_actions, obs, policy, best_value});
64  return best_value;
65 }

67 std::vector<VPNetModel::TrainInputs> SolveGame() {
68  std::shared_ptr<const open_spiel::Game> game =
69  open_spiel::LoadGame("tic_tac_toe");
70  std::unique_ptr<open_spiel::State> state = game->NewInitialState();
71 
72  // Store them directly into a vector so they are returned in order so
73  // given a static initialization the model trains identically.
74  absl::flat_hash_map<std::string, int> cache;
75  std::vector<VPNetModel::TrainInputs> train_inputs;
76  train_inputs.reserve(4520);
77  SolveState(*state, cache, train_inputs);
78  return train_inputs;
79 }

124 // Can learn a single trajectory
125 void TestModelLearnsSimple(const std::string& nn_model) {
126  std::cout << "TestModelLearnsSimple: " << nn_model << std::endl;
127  std::shared_ptr<const Game> game = LoadGame("tic_tac_toe");
128  VPNetModel model = BuildModel(*game, nn_model, false);
129 
130  std::vector<VPNetModel::TrainInputs> train_inputs;
131  std::unique_ptr<open_spiel::State> state = game->NewInitialState();
132 
133  while (!state->IsTerminal()) {
134  std::vector<float> obs = state->ObservationTensor();
135  std::vector<Action> legal_actions = state->LegalActions();
136  Action action = legal_actions[0];
137  ActionsAndProbs policy({{action, 1}});
138 
139  train_inputs.emplace_back(
140  VPNetModel::TrainInputs{legal_actions, obs, policy, 1});
141 
142  VPNetModel::InferenceInputs inputs = {legal_actions, obs};
143  std::vector<VPNetModel::InferenceOutputs> out =
144  model.Inference(std::vector{inputs});
145  SPIEL_CHECK_EQ(out.size(), 1);
146  SPIEL_CHECK_EQ(out[0].policy.size(), legal_actions.size());
147 
148  state->ApplyAction(action);
149  }
150 
151  std::cout << "states: " << train_inputs.size() << std::endl;
152  std::vector<VPNetModel::LossInfo> losses;
153  for (int i = 0; i < 1000; i++) {
154  VPNetModel::LossInfo loss = model.Learn(train_inputs);
155  std::cout << absl::StrFormat(
156  "%d: Losses(total: %.3f, policy: %.3f, value: %.3f, l2: %.3f)\n", i,
157  loss.Total(), loss.Policy(), loss.Value(), loss.L2());
158  losses.push_back(loss);
159  if (loss.Policy() < 0.05 && loss.Value() < 0.05) {
160  break;
161  }
162  }
163  SPIEL_CHECK_GT(losses.front().Total(), losses.back().Total());
164  SPIEL_CHECK_GT(losses.front().Policy(), losses.back().Policy());
165  SPIEL_CHECK_GT(losses.front().Value(), losses.back().Value());
166  SPIEL_CHECK_LT(losses.back().Value(), 0.05);
167  SPIEL_CHECK_LT(losses.back().Policy(), 0.05);
168 }
