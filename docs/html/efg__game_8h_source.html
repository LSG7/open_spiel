129  // Gets the player / decision node action associated to this label.

131  auto iter = action_ids_.find(label);

143  Action AddOrGetAction(const std::string& label) {
144  auto iter = action_ids_.find(label);
145  if (iter != action_ids_.end()) {
146  return iter->second;
147  }
148  Action new_action = action_ids_.size();
149  action_ids_[label] = new_action;
150  return new_action;
151  }
