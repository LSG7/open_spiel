67 void UltimateTTTState::DoApplyAction(Action move) {
68  if (current_state_ < 0) {
69  // Choosing a board.
70  SPIEL_CHECK_GE(move, 0);
71  SPIEL_CHECK_LT(move, ttt::kNumCells);
72  current_state_ = move;
73  } else {
74  // Apply action to local state, then apply that move.
75  SPIEL_CHECK_FALSE(local_states_[current_state_]->IsTerminal());
76  local_states_[current_state_]->ApplyAction(move);
77  // Check if it's terminal and mark the outcome in the meta-game.
78  if (local_states_[current_state_]->IsTerminal()) {
79  Player local_outcome = local_state(current_state_)->outcome();
80  if (local_outcome < 0) {
81  meta_board_[current_state_] = ttt::CellState::kEmpty;
82  } else {
83  meta_board_[current_state_] = ttt::PlayerToState(local_outcome);
84  }
85  }
86  // Set the next potential local state.
87  current_state_ = move;
88  // Check for a win or no more moves in the meta-game.
89  if (ttt::BoardHasLine(meta_board_, current_player_)) {
90  outcome_ = current_player_;
91  } else if (AllLocalStatesTerminal()) {
92  outcome_ = kInvalidPlayer; // draw.
93  } else {
94  // Does the next board done? If not, set current_state_ to less than 0 to
95  // indicate that the next board is a choice.
96  if (local_states_[current_state_]->IsTerminal()) {
97  current_state_ = -1;
98  }
99  current_player_ = NextPlayerRoundRobin(current_player_, ttt::kNumPlayers);
100  // Need to set the current player in the local board.
101  if (current_state_ >= 0) {
102  local_state(current_state_)->SetCurrentPlayer(current_player_);
103  }
104  }
105  }
106 }

108 std::vector<Action> UltimateTTTState::LegalActions() const {
109  if (IsTerminal()) return {};
110  if (current_state_ < 0) {
111  std::vector<Action> actions;
112  // Choosing the next state to play: any one that is not finished.
113  for (int i = 0; i < local_states_.size(); ++i) {
114  if (!local_states_[i]->IsTerminal()) {
115  actions.push_back(i);
116  }
117  }
118  return actions;
119  } else {
120  return local_states_[current_state_]->LegalActions();
121  }
122 }

146 std::string UltimateTTTState::ToString() const {
147  std::string str;
148  const int rows = ttt::kNumRows * 3;
149  const int cols = ttt::kNumCols * 3;
150  int meta_row = 0;
151  int meta_col = 0;
152  for (int r = 0; r < rows; ++r) {
153  meta_row = r / 3;
154  int local_row = r % 3;
155  for (int c = 0; c < cols; ++c) {
156  meta_col = c / 3;
157  int local_col = c % 3;
158  int state_idx = meta_row * 3 + meta_col;
159  SPIEL_CHECK_GE(state_idx, 0);
160  SPIEL_CHECK_LT(state_idx, local_states_.size());
161  absl::StrAppend(&str, ttt::StateToString(local_state(state_idx)->BoardAt(
162  local_row, local_col)));
163  if (local_col == 2) {
164  absl::StrAppend(&str, c == 8 ? "\n" : " ");
165  }
166  if (local_row == 2 && r < 8 && c == 8) {
167  absl::StrAppend(&str, "\n");
168  }
169  }
170  }
171  return str;
172 }

197 void UltimateTTTState::ObservationTensor(Player player,
198  absl::Span<float> values) const {
199  SPIEL_CHECK_GE(player, 0);
200  SPIEL_CHECK_LT(player, num_players_);
201 
202  // Treat `values` as a 3-d tensor: 3 x 9 x 9:
203  // - empty versus x versus o, then
204  // - local state index, then
205  // - then 3x3 position within the local board.
206  TensorView<3> view(values, {ttt::kCellStates, ttt::kNumCells, ttt::kNumCells},
207  /*reset*/true);
208  for (int state = 0; state < ttt::kNumCells; ++state) {
209  for (int cell = 0; cell < ttt::kNumCells; ++cell) {
210  view[{static_cast<int>(local_state(state)->BoardAt(cell)),
211  state, cell}] = 1.0;
212  }
213  }
214 }
