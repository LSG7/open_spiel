24def type_from_states(states):
25 """Get node type of a list of states and assert they are the same."""
26 types = [state.get_type() for state in states]
27 assert len(set(types)) == 1, f"types: {types}"
28 return types[0]

31def _check_distribution_sum(distribution: DistributionDict, expected_sum: int):
32 """Sanity check that the distribution sums to a given value."""
33 sum_state_probabilities = sum(distribution.values())
34 assert abs(sum_state_probabilities - expected_sum) < 1e-4, (
35 "Sum of probabilities of all possible states should be the number of "
36 f"population, it is {sum_state_probabilities}.")

39class DistributionPolicy(tabular_distribution.TabularDistribution):
40 """Computes the distribution of a specified strategy."""
41
42 def __init__(self,
43 game: pyspiel.Game,
44 policy: policy_module.Policy,
45 root_state: pyspiel.State = None):
46 """Initializes the distribution calculation.
47
48 Args:
49 game: The game to analyze.
50 policy: The policy we compute the distribution of.
51 root_state: The state of the game at which to start analysis. If `None`,
52 the game root states are used.
53 """
54 super().__init__(game)
55 self._policy = policy
56 if root_state is None:
57 self._root_states = game.new_initial_states()
58 else:
59 self._root_states = [root_state]
60 self.evaluate()
61
62 def evaluate(self):
63 """Evaluate the distribution over states of self._policy."""
64 # List of all game states that have a non-zero probability at the current
65 # timestep and player ID.
66 current_states = self._root_states.copy()
67 # Distribution at the current timestep. Maps state strings to
68 # floats. For each group of states for a given population, these
69 # floats represent a probability distribution.
70 current_distribution = {
71 self.state_to_str(state): 1 for state in current_states
72 }
73 # List of all distributions computed so far.
74 all_distributions = [current_distribution]
75
76 while type_from_states(current_states) != pyspiel.StateType.TERMINAL:
77 new_states, new_distribution = self._one_forward_step(
78 current_states, current_distribution, self._policy)
79 _check_distribution_sum(new_distribution, self.game.num_players())
80 current_distribution = new_distribution
81 current_states = new_states
82 all_distributions.append(new_distribution)
83
84 # Merge all per-timestep distributions into `self.distribution`.
85 for dist in all_distributions:
86 for state_str, prob in dist.items():
87 if state_str in self.distributiondistribution:
88 raise ValueError(
89 f"{state_str} has already been seen in distribution.")
90 self.distributiondistribution[state_str] = prob
91
92 def _forward_actions(
93 self, current_states: List[pyspiel.State], distribution: DistributionDict,
94 actions_and_probs_fn) -> Tuple[List[pyspiel.State], DistributionDict]:
95 """Applies one action to each current state.
96
97 Args:
98 current_states: The states to apply actions on.
99 distribution: Current distribution.
100 actions_and_probs_fn: Function that maps one state to the corresponding
101 list of (action, proba). For decision nodes, this should be the policy,
102 and for chance nodes, this should be chance outcomes.
103
104 Returns:
105 A pair:
106 - new_states: List of new states after applying one action on
107 each input state.
108 - new_distribution: Probabilities for each of these states.
109 """
110 new_states = []
111 new_distribution = collections.defaultdict(float)
112 for state in current_states:
113 state_str = self.state_to_str(state)
114 for action, prob in actions_and_probs_fn(state):
115 new_state = state.child(action)
116 new_state_str = self.state_to_str(new_state)
117 if new_state_str not in new_distribution:
118 new_states.append(new_state)
119 new_distribution[new_state_str] += prob * distribution[state_str]
120 return new_states, new_distribution

122 def _one_forward_step(self, current_states: List[pyspiel.State],
123 distribution: DistributionDict,
124 policy: policy_module.Policy):
125 """Performs one step of the forward equation.
126
127 Namely, this takes as input a list of current state, the current
128 distribution, and performs one step of the forward equation, using
129 actions coming from the policy or from the chance node
130 probabilities, or propagating the distribution to the MFG nodes.
131
132 Args:
133 current_states: The states to perform the forward step on. All states are
134 assumed to be of the same type.
135 distribution: Current distribution.
136 policy: Policy that will be used if states
137
138 Returns:
139 A pair:
140 - new_states: List of new states after applying one step of the
141 forward equation (either performing one action or doing one
142 distribution update).
143 - new_distribution: Probabilities for each of these states.
144 """
145 state_types = type_from_states(current_states)
146 if state_types == pyspiel.StateType.CHANCE:
147 return self._forward_actions(current_states, distribution,
148 lambda state: state.chance_outcomes())
149
150 if state_types == pyspiel.StateType.MEAN_FIELD:
151 new_states = []
152 new_distribution = {}
153 for state in current_states:
154 dist = [
155 # We need to default to 0, since the support requested by
156 # the state in `state.distribution_support()` might have
157 # states that we might not have reached yet. A probability
158 # of 0. should be given for them.
159 distribution.get(str_state, 0.)
160 for str_state in state.distribution_support()
161 ]
162 new_state = state.clone()
163 new_state.update_distribution(dist)
164 new_state_str = self.state_to_str(new_state)
165 if new_state_str not in new_distribution:
166 new_states.append(new_state)
167 new_distribution[new_state_str] = 0.0
168 new_distribution[new_state_str] += distribution.get(
169 self.state_to_str(state), 0)
170 return new_states, new_distribution
171
172 if state_types == pyspiel.StateType.DECISION:
173 return self._forward_actions(
174 current_states, distribution,
175 lambda state: policy.action_probabilities(state).items())
176
177 raise ValueError(
178 f"Unpexpected state_stypes: {state_types}, states: {current_states}")
