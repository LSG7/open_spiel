25using std::map;
26using std::vector;
27using state_pointer = std::unique_ptr<State>;
28using state_action = std::pair<std::string, Action>;
29using state_prob = std::pair<std::string, double>;
30
31// Adds transitions and transition probability from a given state
32void AddTransition(map<state_action, vector<state_prob>>* transitions,
33 std::string key, const state_pointer& state) {
34 for (auto action : state->LegalActions()) {
35 auto next_state = state->Clone();
36 next_state->ApplyAction(action);
37 vector<state_prob> possibilities;
38 if (next_state->IsChanceNode()) {
39 // For a chance node, record the transition probabilities
40 for (const auto& actionprob : next_state->ChanceOutcomes()) {
41 auto realized_next_state = next_state->Clone();
42 realized_next_state->ApplyAction(actionprob.first);
43 possibilities.emplace_back(realized_next_state->ToString(),
44 actionprob.second);
45 }
46 } else {
47 // A non-chance node is equivalent to transition with probability 1
48 possibilities.emplace_back(next_state->ToString(), 1.0);
49 }
50 (*transitions)[std::make_pair(key, action)] = possibilities;
51 }
52}

54// Initialize transition map and value map
55void InitializeMaps(const map<std::string, state_pointer>& states,
56 map<std::string, double>* values,
57 map<state_action, vector<state_prob>>* transitions) {
58 for (const auto& kv : states) {
59 auto key = kv.first;
60 if (kv.second->IsTerminal()) {
61 // For both 1-player and 2-player zero sum games, suffices to look at
62 // player 0's utility
63 (*values)[key] = kv.second->PlayerReturn(Player{0});
64 } else {
65 (*values)[key] = 0;
66 AddTransition(transitions, key, kv.second);
67 }
68 }
69}

73std::map<std::string, double> ValueIteration(const Game& game, int depth_limit,
74 double threshold) {
75 using state_action = std::pair<std::string, Action>;
76 using state_prob = std::pair<std::string, double>;
77
78 // Currently only supports 1-player or 2-player zero sum games
79 SPIEL_CHECK_TRUE(game.NumPlayers() == 1 || game.NumPlayers() == 2);
80 if (game.NumPlayers() == 2) {
81 SPIEL_CHECK_EQ(game.GetType().utility, GameType::Utility::kZeroSum);
82 }
83
84 // No support for simultaneous games (needs an LP solver). And so also must
85 // be a perfect information game.
86 SPIEL_CHECK_EQ(game.GetType().dynamics, GameType::Dynamics::kSequential);
87 SPIEL_CHECK_EQ(game.GetType().information,
88 GameType::Information::kPerfectInformation);
89
90 auto states = GetAllStates(game, depth_limit, /*include_terminals=*/true,
91 /*include_chance_states=*/false,
92 /*stop_at_duplicates*/true);
93 std::map<std::string, double> values;
94 std::map<state_action, std::vector<state_prob>> transitions;
95
96 InitializeMaps(states, &values, &transitions);
97
98 double error;
99 double min_utility = game.MinUtility();
100 double max_utility = game.MaxUtility();
101 do {
102 error = 0;
103 for (const auto& kv : states) {
104 auto key = kv.first;
105
106 if (kv.second->IsTerminal()) continue;
107
108 auto player = kv.second->CurrentPlayer();
109
110 // Initialize value to be the minimum utility if current player
111 // is the maximizing player (i.e. player 0), and to maximum utility
112 // if current player is the minimizing player (i.e. player 1).
113 double value = (player == Player{0}) ? min_utility : max_utility;
114 for (auto action : kv.second->LegalActions()) {
115 auto possibilities = transitions[std::make_pair(key, action)];
116 double q_value = 0;
117 for (const auto& outcome : possibilities) {
118 q_value += outcome.second * values[outcome.first];
119 }
120 // Player 0 is maximizing the value (which is w.r.t. player 0)
121 // Player 1 is minimizing the value
122 if (player == Player{0})
123 value = std::max(value, q_value);
124 else
125 value = std::min(value, q_value);
126 }
127
128 double* stored_value = &values[key];
129 error = std::max(std::abs(*stored_value - value), error);
130 *stored_value = value;
131 }
132 } while (error > threshold);
133
134 return values;
135}
