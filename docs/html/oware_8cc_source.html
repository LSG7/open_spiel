76 std::vector<Action> OwareState::LegalActions() const {
77  std::vector<Action> actions;
78  if (IsTerminal()) return actions;
79  const Player lower = PlayerLowerHouse(board_.current_player);
80  const Player upper = PlayerUpperHouse(board_.current_player);
81  if (OpponentSeeds() == 0) {
82  // In case the opponent does not have any seeds, a player must make
83  // a move which gives the opponent seeds.
84  for (int house = lower; house <= upper; house++) {
85  const int first_seeds_in_own_row = upper - house;
86  if (board_.seeds[house] - first_seeds_in_own_row > 0) {
87  actions.push_back(HouseToAction(house));
88  }
89  }
90  } else {
91  for (int house = lower; house <= upper; house++) {
92  if (board_.seeds[house] > 0) {
93  actions.push_back(HouseToAction(house));
94  }
95  }
96  }
97  return actions;
98 }

150 bool OwareState::IsTerminal() const {
151  // Terminate when one player has more than half of the seeds
152  // (works both for even and odd number of seeds), or when all seeds
153  // are equally shared.
154  const int limit = total_seeds_ / 2;
155  return board_.score[0] > limit || board_.score[1] > limit ||
156  (board_.score[0] == limit && board_.score[1] == limit);
157 }

177 int OwareState::DistributeSeeds(int house) {
178  int to_distribute = board_.seeds[house];
179  SPIEL_CHECK_NE(to_distribute, 0);
180  board_.seeds[house] = 0;
181  int index = house;
182  while (to_distribute > 0) {
183  index = (index + 1) % NumHouses();
184  // Seeds are never sown into the house they were drawn from.
185  if (index != house) {
186  board_.seeds[index]++;
187  to_distribute--;
188  }
189  }
190  return index;
191 }

197 bool OwareState::IsGrandSlam(int house) const {
198  // If there are seeds beyond the house in which the last seed was dropped,
199  // it is not a Grand Slam.
200  for (int index = UpperHouse(house); index > house; index--) {
201  if (board_.seeds[index] > 0) {
202  return false;
203  }
204  }
205  // If not all houses are captured starting from the house in which the last
206  // seed was dropped, it is not a Grand Slam. It means the opponent will still
207  // have some seeds left because none of these houses can be empty due to
208  // the way seeds are sown.
209  const int lower = LowerHouse(house);
210  for (int index = house; index >= lower; index--) {
211  SPIEL_CHECK_GT(board_.seeds[index], 0);
212  if (!ShouldCapture(board_.seeds[index])) {
213  return false;
214  }
215  }
216  return true;
217 }

219 int OwareState::OpponentSeeds() const {
220  int count = 0;
221  const Player opponent = 1 - board_.current_player;
222  const int lower = PlayerLowerHouse(opponent);
223  const int upper = PlayerUpperHouse(opponent);
224  for (int house = lower; house <= upper; house++) {
225  count += board_.seeds[house];
226  }
227  return count;

230 int OwareState::DoCaptureFrom(int house) {
231  const int lower = LowerHouse(house);
232  int captured = 0;
233  for (int index = house; index >= lower; index--) {
234  if (ShouldCapture(board_.seeds[index])) {
235  captured += board_.seeds[index];
236  board_.seeds[index] = 0;
237  } else {
238  break;
239  }
240  }
241  board_.score[board_.current_player] += captured;
242  return captured;
243 }

245 void OwareState::DoApplyAction(Action action) {
246  SPIEL_CHECK_LT(history_.size(), kMaxGameLength);
247 
248  int last_house = DistributeSeeds(ActionToHouse(CurrentPlayer(), action));
249 
250  if (InOpponentRow(last_house) && !IsGrandSlam(last_house)) {
251  const int captured = DoCaptureFrom(last_house);
252  if (captured > 0) {
253  // No need to keep previous boards for checking game repetition because
254  // captured seeds do not re-enter the game.
255  boards_since_last_capture_.clear();
256  }
257  }
258  board_.current_player = 1 - board_.current_player;
259 
260  if (!boards_since_last_capture_.insert(board_).second) {
261  // We have game repetition, the game is ended.
262  CollectAndTerminate();
263  }
264 
265  if (LegalActions().empty()) {
266  CollectAndTerminate();
267  }
268 }

270 void OwareState::CollectAndTerminate() {
271  for (int house = 0; house < NumHouses(); house++) {
272  const Player player = house / num_houses_per_player_;
273  board_.score[player] += board_.seeds[house];
274  board_.seeds[house] = 0;
275  }
276 }
