31 def name(self) -> str:
32 return "kemeny_young"

34 def _score(
35 self,
36 pref_mat: np.ndarray,
37 perm: Tuple[int, ...],
38 ) -> np.ndarray:
39 # The score of alternative a_i in a ranking R is defined to be:
40 # KemenyScore(a_i) = sum_{a_j s.t. R(a_i) >= R(a_j)} N(a_i, a_j)
41 # The score of ranking R is then sum_i KemenyScore(a_i).
42 num_alts = len(perm)
43 scores = np.zeros(num_alts, dtype=np.int32)
44 for i in range(num_alts):
45 for j in range(i+1, num_alts):
46 scores[i] += pref_mat[perm[i], perm[j]]
47 return scores
48
49 def _permutation_to_ranking(
50 self,
51 alternatives: List[base.AlternativeId],
52 permutation: Tuple[base.AlternativeId, ...]) -> List[base.AlternativeId]:
53 assert len(permutation) == len(alternatives)
54 return [alternatives[permutation[i]] for i in range(len(alternatives))]
55
56 def run_election(self, profile: base.PreferenceProfile) -> base.RankOutcome:
57 assert self.is_valid_profile(profile)
58 pref_mat = profile.pref_matrix()
59 alternatives = profile.alternatives
60 m = profile.num_alternatives()
61 best_permutation = None
62 best_score = -1
63 best_score_array = None
64 for permutation in itertools.permutations(range(m)):
65 scores = self._score(pref_mat, permutation)
66 total_score = scores.sum()
67 if total_score > best_score:
68 best_score = total_score
69 best_score_array = scores
70 best_permutation = permutation
71 best_ranking = self._permutation_to_ranking(alternatives, best_permutation)
72 outcome = base.RankOutcome(rankings=best_ranking,
73 scores=list(best_score_array))
74 return outcome
