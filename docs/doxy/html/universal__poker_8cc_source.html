207UniversalPokerState::UniversalPokerState(std::shared_ptr<const Game> game)
208 : State(game),
209 acpc_game_(
210 static_cast<const UniversalPokerGame *>(game.get())->GetACPCGame()),
211 acpc_state_(acpc_game_),
212 deck_(/*num_suits=*/acpc_game_->NumSuitsDeck(),
213 /*num_ranks=*/acpc_game_->NumRanksDeck()),
214 cur_player_(kChancePlayerId),
215 possibleActions_(ACTION_DEAL),
216 betting_abstraction_(static_cast<const UniversalPokerGame *>(game.get())
217 ->betting_abstraction()) {
218 // -- Optionally apply subgame parameters. -----------------------------------
219 // Pot size.
220 const int pot_size = game->GetParameters().at("potSize").int_value();
221 if (pot_size > 0) {
222 acpc_state_.SetPotSize(pot_size);
223 }
224 // Board cards.
225 const std::string board_cards =
226 game->GetParameters().at("boardCards").string_value();
227 if (!board_cards.empty()) {
228 // Add the cards.
229 logic::CardSet cs(board_cards);
230 int before_add = deck_.NumCards();
231 for (uint8_t card : cs.ToCardArray()) {
232 AddBoardCard(card);
233 deck_.RemoveCard(card);
234 }
235 SPIEL_CHECK_EQ(deck_.NumCards(), before_add - cs.NumCards());
236 // Advance the round according to the number of board cards.
237 int num_cards = cs.NumCards();
238 int round = 0;
239 do {
240 num_cards -= acpc_game_->NumBoardCards(round);
241 round++;
242 } while (round < acpc_game_->NumRounds() && num_cards > 0);
243 acpc_state_.mutable_state()->round = round - 1;
244 }
245 // Set specific hand reach probabilities.
246 const std::string handReaches =
247 game->GetParameters().at("handReaches").string_value();
248 if (!handReaches.empty()) {
249 std::stringstream iss(handReaches);
250 double number;
251 while ( iss >> number ) {
252 handReaches_.push_back(number);
253 }
254 SPIEL_CHECK_EQ(handReaches_.size(), kSubgameUniqueHands * 2);
255 }

258std::string UniversalPokerState::ToString() const {
259 std::string str =
260 absl::StrCat(BettingAbstractionToString(betting_abstraction_), "\n");
261 for (int p = 0; p < acpc_game_->GetNbPlayers(); ++p) {
262 absl::StrAppend(&str, "P", p, " Cards: ", HoleCards(p).ToString(), "\n");
263 }
264 absl::StrAppend(&str, "BoardCards ", BoardCards().ToString(), "\n");
265
266 if (IsChanceNode()) {
267 absl::StrAppend(&str, "PossibleCardsToDeal ", deck_.ToString(), "\n");
268 }
269 if (IsTerminal()) {
270 for (int p = 0; p < acpc_game_->GetNbPlayers(); ++p) {
271 absl::StrAppend(&str, "P", p, " Reward: ", GetTotalReward(p), "\n");
272 }
273 }
274 absl::StrAppend(&str, "Node type?: ");
275 if (IsChanceNode()) {
276 absl::StrAppend(&str, "Chance node\n");
277 } else if (IsTerminal()) {
278 absl::StrAppend(&str, "Terminal Node!\n");
279 } else {
280 absl::StrAppend(&str, "Player node for player ", cur_player_, "\n");
281 }
282
283 if (betting_abstraction_ == BettingAbstraction::kFC ||
284 betting_abstraction_ == BettingAbstraction::kFCPA) {
285 absl::StrAppend(&str, "PossibleActions (", GetPossibleActionCount(),
286 "): [");
287 for (StateActionType action : ALL_ACTIONS) {
288 if (action & possibleActions_) {
289 if (action == ACTION_ALL_IN) absl::StrAppend(&str, " ACTION_ALL_IN ");
290 if (action == ACTION_BET) absl::StrAppend(&str, " ACTION_BET ");
291 if (action == ACTION_CHECK_CALL) {
292 absl::StrAppend(&str, " ACTION_CHECK_CALL ");
293 }
294 if (action == ACTION_FOLD) absl::StrAppend(&str, " ACTION_FOLD ");
295 if (action == ACTION_DEAL) absl::StrAppend(&str, " ACTION_DEAL ");
296 }
297 }
298 }
299 absl::StrAppend(&str, "]", "\nRound: ", acpc_state_.GetRound(),
300 "\nACPC State: ", acpc_state_.ToString(),
301 "\nAction Sequence: ", actionSequence_);
302 return str;
303}

305std::string UniversalPokerState::ActionToString(Player player,
306 Action move) const {
307 std::string move_str;
308 if (IsChanceNode()) {
309 move_str = absl::StrCat("Deal(", move, ")");
310 } else if (static_cast<ActionType>(move) == ActionType::kFold) {
311 move_str = "Fold";
312 } else if (static_cast<ActionType>(move) == ActionType::kCall) {
313 move_str = "Call";
314 } else if (static_cast<ActionType>(move) == ActionType::kHalfPot &&
315 // Avoids an edge case where we interpret a bet size that's
316 // literally meant to be '4' as a half pot bet (since that's the
317 // actual value of ActionTye::kHalfPot).
318 betting_abstraction_ != BettingAbstraction::kFULLGAME) {
319 move_str = "HalfPot";
320 } else if (betting_abstraction_ == BettingAbstraction::kFULLGAME) {
321 SPIEL_CHECK_GE(move, 2);
322 move_str = absl::StrCat("Bet", move);
323 } else if (static_cast<ActionType>(move) == ActionType::kBet) {
324 move_str = "Bet";
325 } else if (static_cast<ActionType>(move) == ActionType::kAllIn) {
326 move_str = "AllIn";
327 } else if (move > kBet) {
328 SPIEL_CHECK_EQ(betting_abstraction_, BettingAbstraction::kFCHPA);
329 move_str = absl::StrCat("r", move);
330 } else {
331 SpielFatalError(absl::StrCat("Unknown action: ", move));
332 }
333 return absl::StrCat("player=", player, " move=", move_str);
334}

371void UniversalPokerState::InformationStateTensor(
372 Player player, absl::Span<float> values) const {
373 SPIEL_CHECK_GE(player, 0);
374 SPIEL_CHECK_LT(player, num_players_);
375
376 SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorShape()[0]);
377 std::fill(values.begin(), values.end(), 0.);
378
379 // Layout:
380 // my player number: num_players bits
381 // my cards: Initial deck size bits (1 means you have the card), i.e.
382 // MaxChanceOutcomes() = NumSuits * NumRanks
383 // public cards: Same as above, but for the public cards.
384 // action sequence: (max game length)*2 bits (fold/raise/call/all-in)
385 // action sequence sizings: (max game length) integers with value >= 0,
386 // 0 when corresponding to 'deal' or 'check'.
387 int offset = 0;
388
389 // Mark who I am.
390 values[player] = 1;
391 offset += NumPlayers();
392
393 const logic::CardSet full_deck(acpc_game_->NumSuitsDeck(),
394 acpc_game_->NumRanksDeck());
395 const std::vector<uint8_t> deckCards = full_deck.ToCardArray();
396 logic::CardSet holeCards = HoleCards(player);
397 logic::CardSet boardCards = BoardCards();
398
399 // Mark my private cards
400 // (Note: it should be way more efficient to iterate over the cards of the
401 // player, rather than iterating over all the cards. We may want to change
402 // this in the future.)
403 for (uint32_t i = 0; i < full_deck.NumCards(); i++) {
404 values[i + offset] = holeCards.ContainsCards(deckCards[i]) ? 1.0 : 0.0;
405 }
406 offset += full_deck.NumCards();
407
408 // Mark the public cards
409 for (int i = 0; i < full_deck.NumCards(); ++i) {
410 values[i + offset] = boardCards.ContainsCards(deckCards[i]) ? 1.0 : 0.0;
411 }
412 offset += full_deck.NumCards();
413
414 const std::string actionSeq = GetActionSequence();
415 const int length = actionSeq.length();
416 SPIEL_CHECK_LT(length, game_->MaxGameLength());
417
418 // Mark the action sequence (abstracted).
419 for (int i = 0; i < length; ++i) {
420 SPIEL_CHECK_LT(offset + i + 1, values.size());
421 if (actionSeq[i] == 'c') {
422 // Encode call as 10.
423 values[offset + (2 * i)] = 1;
424 values[offset + (2 * i) + 1] = 0;
425 } else if (actionSeq[i] == 'p') {
426 // Encode raise as 01.
427 values[offset + (2 * i)] = 0;
428 values[offset + (2 * i) + 1] = 1;
429 } else if (actionSeq[i] == 'a') {
430 // Encode all-in as 11.
431 values[offset + (2 * i)] = 1;
432 values[offset + (2 * i) + 1] = 1;
433 } else if (actionSeq[i] == 'f') {
434 // Encode fold as 00.
435 // TODO(author2): Should this be 11?
436 values[offset + (2 * i)] = 0;
437 values[offset + (2 * i) + 1] = 0;
438 } else if (actionSeq[i] == 'd') {
439 values[offset + (2 * i)] = 0;
440 values[offset + (2 * i) + 1] = 0;
441 } else {
442 SPIEL_CHECK_EQ(actionSeq[i], 'd');
443 }
444 }
445 // Move offset to the end of the abstracted betting sequence (since 2 entries
446 // per move).
447 offset += game_->MaxGameLength() * 2;
448
449 // Mark the action sequence sizings.
450 const std::vector<int> action_sequence_sizings = GetActionSequenceSizings();
451 SPIEL_CHECK_EQ(length, action_sequence_sizings.size());
452 for (int i = 0; i < length; ++i) {
453 values[offset + i] = action_sequence_sizings[i];
454 }
455 // Move offset to the end of the un-abstracted betting sequence.
456 offset += game_->MaxGameLength();
457
458 SPIEL_CHECK_EQ(offset, game_->InformationStateTensorShape()[0]);
459}

461void UniversalPokerState::ObservationTensor(Player player,
462 absl::Span<float> values) const {
463 SPIEL_CHECK_GE(player, 0);
464 SPIEL_CHECK_LT(player, NumPlayers());
465
466 SPIEL_CHECK_EQ(values.size(), game_->ObservationTensorShape()[0]);
467 std::fill(values.begin(), values.end(), 0.);
468
469 // Layout of observation:
470 // my player number: num_players bits
471 // my cards: Initial deck size bits (1 means you have the card), i.e.
472 // MaxChanceOutcomes() = NumSuits * NumRanks
473 // public cards: Same as above, but for the public cards.
474 // the contribution of each player to the pot. num_players integers.
475 int offset = 0;
476
477 // Mark who I am.
478 values[player] = 1;
479 offset += NumPlayers();
480
481 const logic::CardSet full_deck(acpc_game_->NumSuitsDeck(),
482 acpc_game_->NumRanksDeck());
483 const std::vector<uint8_t> all_cards = full_deck.ToCardArray();
484 logic::CardSet holeCards = HoleCards(player);
485 logic::CardSet boardCards = BoardCards();
486
487 for (uint32_t i = 0; i < full_deck.NumCards(); i++) {
488 values[i + offset] = holeCards.ContainsCards(all_cards[i]) ? 1.0 : 0.0;
489 }
490 offset += full_deck.NumCards();
491
492 for (uint32_t i = 0; i < full_deck.NumCards(); i++) {
493 values[i + offset] = boardCards.ContainsCards(all_cards[i]) ? 1.0 : 0.0;
494 }
495 offset += full_deck.NumCards();
496
497 // Adding the contribution of each players to the pot.
498 for (auto p = Player{0}; p < NumPlayers(); p++) {
499 values[offset + p] = acpc_state_.Ante(p);
500 }
501 offset += NumPlayers();
502 SPIEL_CHECK_EQ(offset, game_->ObservationTensorShape()[0]);
503}

562int GetHoleCardsReachIndex(int card_a, int card_b,
563 int num_suits, int num_ranks) {
564 // CardSet uses kSuitChars of "cdhs", but we need the inverse i.e. "shdc"
565 // according to https://github.com/Sandholm-Lab/LibratusEndgames
566 const int a_suit = num_suits - 1 - suitOfCard(card_a);
567 const int b_suit = num_suits - 1 - suitOfCard(card_b);
568 const int a_rank = rankOfCard(card_a);
569 const int b_rank = rankOfCard(card_b);
570 // Order by rank, then order by suit.
571 const int lesser_card = a_rank < b_rank
572 || (a_rank == b_rank && a_suit < b_suit)
573 ? card_a : card_b;
574 // We use ints, so this is ok.
575 const int higher_card = card_a + card_b - lesser_card;
576
577 const int lesser_suit = num_suits - 1 - suitOfCard(lesser_card);
578 const int higher_suit = num_suits - 1 - suitOfCard(higher_card);
579 const int lesser_rank = rankOfCard(lesser_card);
580 const int higher_rank = rankOfCard(higher_card);
581
582 // There is an ordering on the cards, which indexes an upper triangular matrix
583 // like so (example for num_cards_in_deck=4)
584 //
585 // j ->
586 // 0 1 2 3
587 // 0 # 0 1 2
588 // i 1 # 3 4
589 // | 2 # 5
590 // v 3 #
591 //
592 const int i = lesser_rank * num_suits + lesser_suit;
593 const int j = higher_rank * num_suits + higher_suit;
594 const int n = num_suits * num_ranks;
595 // Compute index k in the triangle.
596 return i * (2*n - i - 3) / 2 + j - 1;
597}

599// Normally distribution of cards happens in a tree, one card at a time to each
600// player. We flatten this distribution and omit repetitions.
601// The yield function is called for each unique tuples of cards the players can
602// receive.
603void DistributeHands(const logic::CardSet& full_deck,
604 uint8_t num_players,
605 uint8_t num_hole_cards,
606 std::function<void(/*per_player_hole_cards=*/
607 std::vector<std::vector<uint8_t>>)> yield) {
608 // Taken and modified from CardSet::SampleCards()
609 int nbCards = num_hole_cards * num_players;
610
611 uint64_t p = 0;
612 for (int i = 0; i < nbCards; ++i) {
613 p += (1 << i);
614 }
615 // Enumerates all the uint64_t integers that with nbCards 1-bits.
616 // The final n is ignored. It is fine as long as the rank < 16 (a property
617 // satisfied by CardSet, as ranks >= 16 are not representable).
618 int n_choose_k = 0;
619 for (uint64_t n = logic::bit_twiddle_permute(p); n > p;
620 p = n, n = logic::bit_twiddle_permute(p)) {
621 // Checks whether the CardSet represented by p is inside the CardSet.
622 uint64_t combo = p & full_deck.cs.cards;
623 if (__builtin_popcountl(combo) == nbCards) {
624 // Generate all partitions of size num_hole_cards:
625 logic::CardSet c;
626 c.cs.cards = combo;
627 std::vector<uint8_t> x = c.ToCardArray();
628
629 yield({{x[0], x[1]}, {x[2], x[3]}});
630 yield({{x[0], x[2]}, {x[1], x[3]}});
631 yield({{x[0], x[3]}, {x[1], x[2]}});
632 yield({{x[2], x[3]}, {x[0], x[1]}});
633 yield({{x[1], x[3]}, {x[0], x[2]}});
634 yield({{x[1], x[2]}, {x[0], x[3]}});
635 ++n_choose_k;
636 }
637 }
638 // 52 choose 4
639 SPIEL_CHECK_EQ(n_choose_k, 270725);
640}

642const std::vector <int> UniversalPokerState::GetEncodingBase() const {
643 const int num_hole_cards = acpc_game_->GetNbHoleCardsRequired();
644 SPIEL_CHECK_EQ(num_hole_cards, 2); // Only this case is implemented.
645 const int num_distribute_cards = num_hole_cards * num_players_;
646 const int full_deck_cards = acpc_game_->NumSuitsDeck()
647 * acpc_game_->NumRanksDeck();
648 return std::vector <int>(num_distribute_cards, full_deck_cards);
649}

651// Distribute the hole cards to each player in one large chance node.
652// Modify the chance probs to be a well-formed public belief state.
653std::vector<std::pair<Action, double>>
654UniversalPokerState::DistributeHandCardsInSubgame() const {
655 const int num_hole_cards = acpc_game_->GetNbHoleCardsRequired();
656 // Only this case is supported. Generalizing the code to other pokers
657 // should not be too difficult.
658 SPIEL_CHECK_EQ(num_hole_cards, 2);
659 SPIEL_CHECK_EQ(num_players_, 2);
660
661 const logic::CardSet full_deck(acpc_game_->NumSuitsDeck(),
662 acpc_game_->NumRanksDeck());
663
664 int possible_hands = kSubgameUniqueHands; // 52*51/2
665 const int possible_hand_pairs = 270725 * 6; // (52 choose 4) * choose_hands
666 const double hole_cards_chance_prob = 1. / possible_hand_pairs;
667
668 const int reach_offset = possible_hands;
669 SPIEL_DCHECK_EQ(reach_offset*num_players_, handReaches_.size());
670
671 std::vector<std::pair<Action, double>> outcomes;
672 outcomes.reserve(possible_hand_pairs);
673 double normalizer = 0.; // We need to normalize the probs.
674 const std::vector <int> encoding_bases = GetEncodingBase();
675 const auto add_outcome = [&](const std::vector<std::vector<uint8_t>>& cards) {
676 // Encode OpenSpiel action.
677 // logic::CardSet uses 64 bits for its representation, the same size as
678 // open_spiel::Action. This means that we cannot easily encode distribution
679 // of the 4 cards along with their partition assignment to each player:
680 // as the name suggests, it's used for set representations, not for list
681 // representation. This is unfortunate, so instead we compute specific
682 // action numbers via multiplication of RankActionMixedBase.
683 SPIEL_CHECK_EQ(cards.size(), num_players_);
684 SPIEL_CHECK_EQ(cards[0].size(), num_hole_cards);
685 std::vector<int> flatten_hole_cards;
686 flatten_hole_cards.reserve(num_players_ * num_hole_cards);
687 for (int i = 0; i < num_players_; ++i) {
688 for (int j = 0; j < num_hole_cards; ++j) {
689 flatten_hole_cards.push_back(cards[i][j]);
690 }
691 }
692 Action encoded = RankActionMixedBase(encoding_bases, flatten_hole_cards);
693
694 // Compute outcome probability.
695 double p = hole_cards_chance_prob;
696 for (int pl = 0; pl < num_players_; ++pl) {
697 const int hole_idx = GetHoleCardsReachIndex(cards[pl][0], cards[pl][1],
698 acpc_game_->NumSuitsDeck(),
699 acpc_game_->NumRanksDeck());
700 const double player_reach = handReaches_[pl*reach_offset + hole_idx];
701 p *= player_reach;
702 }
703
704 // We generate all hands, however not all them are necessarily feasible:
705 // some hands are prohibited if there are cards on the board. If we used
706 // those hands, the board cards could not then appear on the board.
707 for (int card : flatten_hole_cards) {
708 if (!deck_.ContainsCards(card)) {
709 p *= 0;
710 break;
711 }
712 }
713
714 outcomes.push_back({encoded, p});
715 normalizer += p;
716 };
717 DistributeHands(full_deck, NumPlayers(), num_hole_cards, add_outcome);
718
719 SPIEL_CHECK_GT(normalizer, 0.);
720 for (auto&[action, p] : outcomes) p /= normalizer;
721 SPIEL_CHECK_EQ(outcomes.size(), possible_hand_pairs);
722 return outcomes;
723}

730std::vector<std::pair<Action, double>> UniversalPokerState::ChanceOutcomes()
731 const {
732 SPIEL_CHECK_TRUE(IsChanceNode());
733 if (IsDistributingSingleCard()) {
734 auto available_cards = LegalActions();
735 const int num_cards = available_cards.size();
736 const double p = 1.0 / num_cards;

739 std::vector<std::pair<Action, double>> outcomes;
740 outcomes.reserve(num_cards);
741 for (const auto &card : available_cards) {

745 } else {
746 return DistributeHandCardsInSubgame();

750std::vector<Action> UniversalPokerState::LegalActions() const {
751 if (IsChanceNode()) {
752 if (IsDistributingSingleCard()) {
753 const logic::CardSet full_deck(acpc_game_->NumSuitsDeck(),
754 acpc_game_->NumRanksDeck());
755 const std::vector<uint8_t> all_cards = full_deck.ToCardArray();
756 std::vector<Action> actions;
757 actions.reserve(deck_.NumCards());
758 for (uint32_t i = 0; i < full_deck.NumCards(); i++) {
759 if (deck_.ContainsCards(all_cards[i])) actions.push_back(i);
760 }
761 return actions;
762 } else {
763 // Strip away probability outcomes.
764 std::vector<std::pair<Action, double>> outcomes =
765 DistributeHandCardsInSubgame();
766 std::vector <Action> actions;
767 actions.reserve(outcomes.size());
768 for (auto&[action, p] : outcomes) actions.push_back(action);
769 return actions;
770 }
771 }
772
773 std::vector<Action> legal_actions;
774
775 if (betting_abstraction_ != BettingAbstraction::kFULLGAME) {
776 if (ACTION_FOLD & possibleActions_) legal_actions.push_back(kFold);
777 if (ACTION_CHECK_CALL & possibleActions_) legal_actions.push_back(kCall);
778 if (ACTION_BET & possibleActions_) legal_actions.push_back(kBet);
779 if (ACTION_ALL_IN & possibleActions_) legal_actions.push_back(kAllIn);
780
781 // For legacy reasons, kHalfPot is the biggest action (in terms of the
782 // action representation).
783 // Note that FCHPA only tells the players about HalfPot + FCPA, but it will
784 // accept most of the other ones.
785 if (ACTION_BET & possibleActions_ && betting_abstraction_ == kFCHPA) {
786 legal_actions.push_back(kHalfPot);
787 }
788
789 return legal_actions;
790 } else {
791 if (acpc_state_.IsFinished()) {
792 return legal_actions;
793 }
794 if (acpc_state_.IsValidAction(
795 acpc_cpp::ACPCState::ACPCActionType::ACPC_FOLD, 0)) {
796 legal_actions.push_back(kFold);
797 }
798 if (acpc_state_.IsValidAction(
799 acpc_cpp::ACPCState::ACPCActionType::ACPC_CALL, 0)) {
800 legal_actions.push_back(kCall);
801 }
802 int32_t min_bet_size = 0;
803 int32_t max_bet_size = 0;
804 if (acpc_state_.RaiseIsValid(&min_bet_size, &max_bet_size)) {
805 const int original_size = legal_actions.size();
806 legal_actions.resize(original_size + max_bet_size - min_bet_size + 1);
807 std::iota(legal_actions.begin() + original_size, legal_actions.end(),
808 min_bet_size);
809 }
810 }
811 return legal_actions;
812}

814int UniversalPokerState::PotSize(double multiple) const {
815 const project_acpc_server::State &state = acpc_state_.raw_state();
816 const project_acpc_server::Game &game = acpc_state_.game()->Game();
817 const int pot_size = absl::c_accumulate(
818 absl::Span<const int32_t>(state.spent, game.numPlayers), 0);
819 const int amount_to_call =
820 state.maxSpent -
821 state.spent[project_acpc_server::currentPlayer(&game, &state)];
822 const int pot_after_call = amount_to_call + pot_size;
823 return std::round(state.maxSpent + multiple * pot_after_call);
824}

833// We first deal the cards to each player, dealing all the cards to the first
834// player first, then the second player, until all players have their private
835// cards.
836void UniversalPokerState::DoApplyAction(Action action_id) {
837 if (IsChanceNode()) {
838 if (IsDistributingSingleCard()) {
839 // In chance nodes, the action_id is an index into the full deck.
840 uint8_t card =
841 logic::CardSet(acpc_game_->NumSuitsDeck(), acpc_game_->NumRanksDeck())
842 .ToCardArray()[action_id];
843 deck_.RemoveCard(card);
844 actionSequence_ += 'd';
845 actionSequenceSizings_.push_back(0);
846
847 // Check where to add this card
848 if (hole_cards_dealt_ <
849 acpc_game_->GetNbPlayers() * acpc_game_->GetNbHoleCardsRequired()) {
850 AddHoleCard(card);
851 _CalculateActionsAndNodeType();
852 return;
853 }
854
855 if (board_cards_dealt_ <
856 acpc_game_->GetNbBoardCardsRequired(acpc_state_.GetRound())) {
857 AddBoardCard(card);
858 _CalculateActionsAndNodeType();
859 return;
860 }
861 } else {
862 // We are creating the subgame: therefore we distribute the hole cards
863 // to each player.
864 std::vector<int> base = GetEncodingBase();
865 std::vector<int> cards = UnrankActionMixedBase(action_id, base);
866 int num_hole_cards = acpc_game_->GetNbHoleCardsRequired();
867 int num_cards_before_dist = deck_.NumCards();
868 for (int pl = 0; pl < num_players_; ++pl) {
869 for (int i = 0; i < num_hole_cards; ++i) {
870 int card = cards.at(pl*num_hole_cards + i);
871 SPIEL_CHECK_GE(rankOfCard(card), 0);
872 SPIEL_CHECK_LT(rankOfCard(card), MAX_RANKS);
873 SPIEL_CHECK_GE(suitOfCard(card), 0);
874 SPIEL_CHECK_LT(suitOfCard(card), MAX_SUITS);
875 SPIEL_CHECK_TRUE(deck_.ContainsCards(card));
876
877 acpc_state_.AddHoleCard(pl, i, card);
878 deck_.RemoveCard(card);
879 SPIEL_CHECK_FALSE(deck_.ContainsCards(card));
880 ++hole_cards_dealt_;
881 }
882 }
883 SPIEL_CHECK_EQ(deck_.NumCards(), num_cards_before_dist - cards.size());
884 _CalculateActionsAndNodeType();
885 }
886 } else {
887 int action_int = static_cast<int>(action_id);
888 if (action_int == kFold) {
889 ApplyChoiceAction(ACTION_FOLD, 0);
890 return;
891 }
892 if (action_int == kCall) {
893 ApplyChoiceAction(ACTION_CHECK_CALL, 0);
894 return;
895 }
896 if (betting_abstraction_ == BettingAbstraction::kFC) {
897 SpielFatalError(absl::StrCat(
898 "Tried to apply action that was not fold or call. Action: ",
899 State::ActionToString(action_id)));
900 }
901 if (betting_abstraction_ != BettingAbstraction::kFULLGAME) {
902 if (action_int == kHalfPot) {
903 ApplyChoiceAction(ACTION_BET, PotSize(0.5));
904 return;
905 }
906 if (action_int == kBet && acpc_game_->IsLimitGame()) {
907 // In a limit game, the bet size is fixed, so the ACPC code expects size
908 // to be 0.
909 ApplyChoiceAction(ACTION_BET, /*size=*/0);
910 return;
911 }
912 if (action_int == kBet && !acpc_game_->IsLimitGame()) {
913 ApplyChoiceAction(ACTION_BET, PotSize());
914 return;
915 }
916 if (action_int == kAllIn) {
917 ApplyChoiceAction(ACTION_ALL_IN, AllInSize());
918 return;
919 }
920 // FCHPA allows for arbitrary bets.
921 if (betting_abstraction_ == BettingAbstraction::kFCHPA) {
922 SPIEL_CHECK_LE(action_int, acpc_game_->Game().stack[0]);
923 ApplyChoiceAction(ACTION_BET, action_int);
924 return;
925 }
926 }
927 if (betting_abstraction_ != BettingAbstraction::kFULLGAME) {
928 SpielFatalError(absl::StrCat(
929 "Tried to apply action that was not allowed by the betting "
930 "abstraction. Action: ",
931 State::ActionToString(action_id),
932 ", abstraction: ", betting_abstraction_));
933 }
934 if (action_int >= static_cast<int>(kBet) &&
935 action_int <= NumDistinctActions()) {
936 ApplyChoiceAction(ACTION_BET, action_int);
937 return;
938 }
939 SpielFatalError(absl::StrFormat("Action not recognized: %i", action_id));
940 }
941}

956
957std::unique_ptr<HistoryDistribution>
958UniversalPokerState::GetHistoriesConsistentWithInfostate(int player_id) const {
959 // This is only implemented for 2 players.
960 if (acpc_game_->GetNbPlayers() != 2) return {};
961
962 logic::CardSet is_cards;
963 logic::CardSet our_cards = HoleCards(player_id);
964 for (uint8_t card : our_cards.ToCardArray()) is_cards.AddCard(card);
965 for (uint8_t card : BoardCards().ToCardArray()) is_cards.AddCard(card);
966 logic::CardSet fresh_deck(/*num_suits=*/acpc_game_->NumSuitsDeck(),
967 /*num_ranks=*/acpc_game_->NumRanksDeck());
968 for (uint8_t card : is_cards.ToCardArray()) fresh_deck.RemoveCard(card);
969 auto dist = absl::make_unique<HistoryDistribution>();
970
971 // We only consider half the possible hands as we only look at each pair of
972 // hands once, i.e. order does not matter.
973 const int num_hands =
974 0.5 * fresh_deck.NumCards() * (fresh_deck.NumCards() - 1);
975 dist->first.reserve(num_hands);
976 for (uint8_t hole_card1 : fresh_deck.ToCardArray()) {
977 logic::CardSet subset_deck = fresh_deck;
978 subset_deck.RemoveCard(hole_card1);
979 for (uint8_t hole_card2 : subset_deck.ToCardArray()) {
980 if (hole_card1 < hole_card2) continue;
981 std::unique_ptr<State> root = game_->NewInitialState();
982 if (player_id == 0) {
983 for (uint8_t card : our_cards.ToCardArray()) root->ApplyAction(card);
984 root->ApplyAction(hole_card1);
985 root->ApplyAction(hole_card2);
986 } else if (player_id == 1) {
987 root->ApplyAction(hole_card1);
988 root->ApplyAction(hole_card2);
989 for (uint8_t card : our_cards.ToCardArray()) root->ApplyAction(card);
990 }
991 SPIEL_CHECK_FALSE(root->IsChanceNode());
992 dist->first.push_back(std::move(root));
993 }
994 }
995 SPIEL_DCHECK_EQ(dist->first.size(), num_hands);
996 const double divisor = 1. / static_cast<double>(dist->first.size());
997 dist->second.assign(dist->first.size(), divisor);
998 return dist;
999}

1059double UniversalPokerGame::MaxCommitment() const {
1060 int max_commit = 0;
1061 if (acpc_game_.IsLimitGame()) {
1062 // The most a player can put into the pot is the raise amounts on each round
1063 // times the maximum number of raises, plus the original chips they put in
1064 // to play, which has the big blind as an upper bound.
1065 const auto &acpc_game = acpc_game_.Game();
1066 max_commit = big_blind();
1067 for (int i = 0; i < acpc_game_.NumRounds(); ++i) {
1068 max_commit += acpc_game.maxRaises[i] * acpc_game.raiseSize[i];
1069 }
1070 } else {
1071 // In No-Limit games, this isn't true, as there is no maximum raise value,
1072 // so the limit is the number of chips that the player has.
1073 max_commit = acpc_game_.StackSize(0);
1074 }
1075 return static_cast<double>(max_commit);
1076}

1078double UniversalPokerGame::MaxUtility() const {
1079 // In poker, the utility is defined as the money a player has at the end of
1080 // the game minus then money the player had before starting the game.
1081 // The most a player can win *per opponent* is the most each player can put
1082 // into the pot,
1083 // The maximum amount of money a player can win is the maximum bet any player
1084 // can make, times the number of players (excluding the original player).
1085 return MaxCommitment() * (acpc_game_.GetNbPlayers() - 1);
1086}

1088double UniversalPokerGame::MinUtility() const {
1089 // In poker, the utility is defined as the money a player has at the end of
1090 // the game minus then money the player had before starting the game. As such,
1091 // the most a player can lose is the maximum amount they can bet.
1092 return -1 * MaxCommitment();
1093}

1112int UniversalPokerGame::MaxGameLength() const {
1113 // We cache this as this is very slow to calculate.
1114 if (max_game_length_) return *max_game_length_;
1115
1116 // Make a good guess here because bruteforcing the tree is far too slow
1117 // One Terminal Action
1118 int length = 1;
1119
1120 // Deal Actions
1121 length += acpc_game_.GetTotalNbBoardCards() +
1122 acpc_game_.GetNbHoleCardsRequired() * acpc_game_.GetNbPlayers();
1123
1124 // The longest game (with a single betting round, for simplicity) consists of:
1125 // n-1 players checking,
1126 // 1 player betting, n-2 players calling,
1127 // 1 player raising, n-2 players calling,
1128 // etc...,
1129 // 1 player raising, n-1 players calling
1130
1131 // Check Actions
1132 length += (NumPlayers() * acpc_game_.NumRounds());
1133
1134 // Bet/Raise/Call Actions
1135 double maxStack = 0;
1136 double maxBlind = 0;
1137 for (uint32_t p = 0; p < NumPlayers(); p++) {
1138 maxStack =
1139 acpc_game_.StackSize(p) > maxStack ? acpc_game_.StackSize(p) : maxStack;
1140 maxBlind =
1141 acpc_game_.BlindSize(p) > maxBlind ? acpc_game_.BlindSize(p) : maxBlind;
1142 }
1143
1144 int max_num_raises = 0;
1145 if (betting_abstraction_ == BettingAbstraction::kFC) {
1146 // no raises
1147 } else if (betting_abstraction_ == BettingAbstraction::kFCPA) {
1148 double pot_size = maxBlind * NumPlayers();
1149 while (pot_size / NumPlayers() < maxStack) {
1150 max_num_raises++;
1151 pot_size += pot_size * NumPlayers();
1152 }
1153 } else if (betting_abstraction_ == BettingAbstraction::kFCHPA) {
1154 double pot_size = maxBlind * NumPlayers();
1155 while (pot_size / NumPlayers() < maxStack) {
1156 max_num_raises++;
1157 pot_size += NumPlayers() * pot_size/2;
1158 }
1159 } else if (betting_abstraction_ == BettingAbstraction::kFULLGAME) {
1160 max_num_raises = (maxStack + maxBlind - 1)/maxBlind; // ceil divide
1161 } else {
1162 SpielFatalError("Unknown Betting Abstraction");
1163 }

1260void UniversalPokerState::_CalculateActionsAndNodeType() {
1261 possibleActions_ = 0;
1262
1263 if (acpc_state_.IsFinished()) {
1264 if (acpc_state_.NumFolded() >= acpc_game_->GetNbPlayers() - 1) {
1265 // All players except one has fold.
1266 cur_player_ = kTerminalPlayerId;
1267 } else {
1268 if (board_cards_dealt_ <
1269 acpc_game_->GetNbBoardCardsRequired(acpc_state_.GetRound())) {
1270 cur_player_ = kChancePlayerId;
1271 possibleActions_ = ACTION_DEAL;
1272 return;
1273 }
1274 // Showdown!
1275 cur_player_ = kTerminalPlayerId;
1276 }
1277
1278 } else {
1279 // Check if we need to deal cards. We assume all cards are dealt at the
1280 // start of the game.
1281 if (hole_cards_dealt_ <
1282 acpc_game_->GetNbHoleCardsRequired() * acpc_game_->GetNbPlayers()) {
1283 cur_player_ = kChancePlayerId;
1284 possibleActions_ = ACTION_DEAL;
1285 return;
1286 }
1287 // 2. We need to deal a public card.
1288 if (board_cards_dealt_ <
1289 acpc_game_->GetNbBoardCardsRequired(acpc_state_.GetRound())) {
1290 cur_player_ = kChancePlayerId;
1291 possibleActions_ = ACTION_DEAL;
1292 return;
1293 }
1294
1295 // Check for CHOICE Actions
1296 cur_player_ = acpc_state_.CurrentPlayer();
1297 if (acpc_state_.IsValidAction(
1298 acpc_cpp::ACPCState::ACPCActionType::ACPC_FOLD, 0)) {
1299 possibleActions_ |= ACTION_FOLD;
1300 }
1301 if (acpc_state_.IsValidAction(
1302 acpc_cpp::ACPCState::ACPCActionType::ACPC_CALL, 0)) {
1303 possibleActions_ |= ACTION_CHECK_CALL;
1304 }
1305
1306 int potSize = 0;
1307 int allInSize = 0;
1308 // We have to call this as this sets potSize and allInSize_.
1309 bool valid_to_raise = acpc_state_.RaiseIsValid(&potSize, &allInSize);
1310 if (betting_abstraction_ == BettingAbstraction::kFC) return;
1311 if (valid_to_raise) {
1312 if (acpc_game_->IsLimitGame()) {
1313 potSize = 0;
1314 // There's only one "bet" allowed in Limit, which is "all-in or fixed
1315 // bet".
1316 possibleActions_ |= ACTION_BET;
1317 } else {
1318 int cur_spent = acpc_state_.CurrentSpent(acpc_state_.CurrentPlayer());
1319 int pot_raise_to =
1320 acpc_state_.TotalSpent() + 2 * acpc_state_.MaxSpend() - cur_spent;
1321
1322 if (pot_raise_to >= potSize && pot_raise_to <= allInSize) {
1323 potSize = pot_raise_to;
1324 possibleActions_ |= ACTION_BET;
1325 }
1326
1327 if (pot_raise_to != allInSize) {
1328 // If the raise to amount happens to match the number of chips I have,
1329 // then this action was already added as a pot-bet.
1330 possibleActions_ |= ACTION_ALL_IN;
1331 }
1332 }
1333 }
1334 }

1342open_spiel::Action ACPCActionToOpenSpielAction(
1343 const project_acpc_server::Action &action,
1344 const UniversalPokerState &state) {
1345 // We assign this here as we cannot initialize a variable within a switch
1346 // statement.
1347 const auto &up_game =
1348 static_cast<const UniversalPokerGame &>(*state.GetGame());
1349 switch (action.type) {
1350 case project_acpc_server::ActionType::a_fold:
1351 return ActionType::kFold;
1352 case project_acpc_server::ActionType::a_call:
1353 return ActionType::kCall;
1354 case project_acpc_server::ActionType::a_raise:
1355 SPIEL_CHECK_NE(state.betting(), BettingAbstraction::kFC);
1356 // The maximum utility is exactly equal to the all-in amount for both
1357 // players.
1358 if (action.size == up_game.MaxCommitment() * up_game.NumPlayers()) {
1359 return ActionType::kCall;
1360 }
1361 if (action.size == state.PotSize(0.5)) {
1362 return ActionType::kHalfPot;
1363 }
1364 if (action.size == state.PotSize()) return ActionType::kBet;
1365 if (action.size == state.AllInSize()) return ActionType::kAllIn;
1366 if (state.betting() == BettingAbstraction::kFCHPA) {
1367 return action.size;
1368 }
1369 if (state.betting() != BettingAbstraction::kFULLGAME) {
1370 SpielFatalError(absl::StrCat(
1371 "Unsupported bet size: ", action.size, ", pot: ", state.PotSize(),
1372 ", all_in: ", state.AllInSize(),
1373 ", max_commitment: ", state.acpc_state().raw_state().maxSpent,
1374 ", state: ", state.ToString(),
1375 ", history: ", state.HistoryString()));
1376 }
1377 SPIEL_CHECK_EQ(state.betting(), BettingAbstraction::kFULLGAME);
1378 return ActionType::kBet + action.size;
1379 case project_acpc_server::ActionType::a_invalid:
1380 SpielFatalError("Invalid action type.");
1381 default:
1382 SpielFatalError(absl::StrCat("Type not found. Type: ", action.type));
1383 }
1384 // Will never get called.
1385 return kInvalidAction;
1386}

1393std::shared_ptr<const Game> MakeRandomSubgame(std::mt19937 &rng, int pot_size,
1394 std::string board_cards,
1395 std::vector<double> hand_reach) {
1396 constexpr const char* base_game =
1397 "universal_poker("
1398 "betting=nolimit,"
1399 "numPlayers=2,"
1400 "numRounds=4,"
1401 "blind=100 50,"
1402 "firstPlayer=2 1 1 1,"
1403 "numSuits=4,"
1404 "numRanks=13,"
1405 "numHoleCards=2,"
1406 "numBoardCards=0 3 1 1,"
1407 "stack=20000 20000,"
1408 "bettingAbstraction=fcpa,"
1409 "potSize=%d,"
1410 "boardCards=%s,"
1411 "handReaches=%s"
1412 ")";
1413
1414 if (pot_size == -1) {
1415 // 40k is total money in the game -- sum of the players stacks (20k+20k).
1416 // 50 is the size of the small blind, 2*50 is big blind = minimum pot size.
1417 // As both players need to match bets, the pot size must be divisible by 2.
1418 std::uniform_int_distribution<int> dist(50, 20000);
1419 pot_size = dist(rng) * 2;
1420 }
1421 if (board_cards.empty()) {
1422 // Pick a round, i.e. 3/4/5 cards.
1423 std::uniform_int_distribution<int> rnd_cards(3, 5);
1424 std::uniform_int_distribution<int> rnd_rank(0, MAX_RANKS);
1425 std::uniform_int_distribution<int> rnd_suit(0, MAX_SUITS);
1426 // Populate random non-repeating cards.
1427 int num_cards = rnd_cards(rng);
1428 std::vector<int> cards;
1429 cards.reserve(num_cards);
1430 while (cards.size() != num_cards) {
1431 int rank = rnd_rank(rng);
1432 int suit = rnd_suit(rng);
1433 int card = makeCard(rank, suit);
1434 if (std::find(cards.begin(), cards.end(), card) == cards.end()) {
1435 cards.push_back(card);
1436 }
1437 }
1438 logic::CardSet set(cards);
1439 board_cards = set.ToString();
1440 }
1441 if (hand_reach.empty()) {
1442 // Normally uniform_real_distribution is defined on open interval [0, 1)
1443 // We make it into a closed interval [0, 1] thanks to std::nextafter.
1444 double next_after_one = std::nextafter(1.0, 2.0);
1445 SPIEL_CHECK_NE(1.0, next_after_one);
1446 SPIEL_CHECK_LT(1.0, next_after_one);
1447 std::uniform_real_distribution<double> dist(0.0, next_after_one);
1448 for (int i = 0; i < 2*kSubgameUniqueHands; ++i) {
1449 hand_reach.push_back(dist(rng));
1450 }
1451 }
1452 std::string reach = absl::StrJoin(hand_reach.begin(), hand_reach.end(), " ");
1453 return LoadGame(absl::StrFormat(base_game, pot_size, board_cards, reach));
1454}
