56 Action UCIBot::Step(const State& state) {
57  std::string move_str;
58  auto chess_state = down_cast<const chess::ChessState&>(state);
59  if (ponder_ && ponder_move_) {
60  if (!was_ponder_hit_) {
61  Stop();
62  Position(chess_state.Board().ToFEN());
63  tie(move_str, ponder_move_) = Go();
64  } else {
65  tie(move_str, ponder_move_) = ReadBestMove();
66  }
67  } else {
68  Position(chess_state.Board().ToFEN());
69  tie(move_str, ponder_move_) = Go();
70  }
71  was_ponder_hit_ = false;
72  auto move = chess_state.Board().ParseLANMove(move_str);
73  if (!move) {
74  SpielFatalError("Uci sub-process returned an illegal or invalid move");
75  }
76 
77  if (ponder_ && ponder_move_) {
78  Position(chess_state.Board().ToFEN(), {move_str, *ponder_move_});
79  GoPonder();
80  }
81 
82  Action action = chess::MoveToAction(*move);
83  return action;
84 }

109 void UCIBot::StartProcess(const std::string& bot_binary_path) {
110  int output_pipe[2];
111  int input_pipe[2];
112 
113  if (pipe(output_pipe) || pipe(input_pipe)) {
114  SpielFatalError("Creating pipes failed");
115  }
116 
117  pid_ = fork();
118  if (pid_ < 0) {
119  SpielFatalError("Forking failed");
120  }
121 
122  if (pid_ > 0) { // parent
123  close(output_pipe[0]);
124  close(input_pipe[1]);
125 
126  output_fd_ = output_pipe[1];
127  input_fd_ = input_pipe[0];
128  } else { // child
129  dup2(output_pipe[0], STDIN_FILENO);
130  dup2(input_pipe[1], STDOUT_FILENO);
131  dup2(input_pipe[1], STDERR_FILENO);
132 
133  close(output_pipe[1]);
134  close(input_pipe[0]);
135 
136  execlp(bot_binary_path.c_str(), bot_binary_path.c_str(), (char *)nullptr);
137  // See /usr/include/asm-generic/errno-base.h for error codes.
138  switch (errno) {
139  case ENOENT:
140  SpielFatalError(
141  absl::StrCat("Executing uci bot sub-process failed: file '",
142  bot_binary_path, "' not found."));
143  default:
144  SpielFatalError(absl::StrCat(
145  "Executing uci bot sub-process failed: Error ", errno));
146  }
147  }
148 }

213 std::pair<std::string, absl::optional<std::string>> UCIBot::ReadBestMove() {
214  while (true) {
215  auto response = Read(true);
216  std::istringstream response_stream(response);
217  std::string line;
218  while (getline(response_stream, line)) {
219  std::istringstream line_stream(line);
220  std::string token;
221  std::string move_str;
222  absl::optional<std::string> ponder_str = absl::nullopt;
223  line_stream >> std::skipws;
224  while (line_stream >> token) {
225  if (token == "bestmove") {
226  line_stream >> move_str;
227  } else if (token == "ponder") {
228  line_stream >> token;
229  ponder_str = token;
230  }
231  }
232  if (!move_str.empty()) {
233  return std::make_pair(move_str, ponder_str);
234  }
235  }
236  }
237 }

246 std::string UCIBot::Read(bool wait) const {
247  char *buff;
248  int count = 0;
249  std::string response;
250 
251  fd_set fds;
252  FD_ZERO(&fds);
253  FD_SET(input_fd_, &fds);
254  timeval timeout = {5, 0}; // 5 second timeout.
255 
256  int ready_fd = select(/*nfds=*/input_fd_ + 1,
257  /*readfds=*/&fds,
258  /*writefds=*/nullptr,
259  /*exceptfds*/ nullptr, wait ? nullptr : &timeout);
260  if (ready_fd == -1) {
261  SpielFatalError("Failed to read from uci sub-process");
262  }
263  if (ready_fd == 0) {
264  SpielFatalError("Response from uci sub-process not received in time");
265  }
266  if (ioctl(input_fd_, FIONREAD, &count) == -1) {
267  SpielFatalError("Failed to read input size.");
268  }
269  if (count == 0) {
270  return "";
271  }
272  buff = (char*)malloc(count);
273  if (read(input_fd_, buff, count) != count) {
274  SpielFatalError("Read wrong number of bytes");
275  }
276  response.assign(buff, count);
277  free(buff);
278  return response;
279 }
