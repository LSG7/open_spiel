33 // Calculates the best response to every state in the game against the given
34 // policy, where the best responder plays as player_id.
35 // This only works for two player, zero- or constant-sum sequential games, and
36 // raises a SpielFatalError if an incompatible game is passed to it.
37 //
38 // This implementation requires that InformationStateString for the game has
39 // perfect recall. Otherwise, the algorithm will still run, but the value
40 // returned will be wrong.
41 //
42 // A partially computed best-response can be computed when using a
43 // prob_cut_threshold >= 0.
44 //
45 // The max-entropy best-response policy is computed if a non-negative
46 // `action_value_tolerance` is used.
47 // Support is equally split between actions whose values are within
48 // `action_value_tolerance` of the max-value action.
49 //
50 // NOTE: if `action_value_tolerance` is negative, the first action with max
51 // value is selected and a biased determinisitc BR is computed. This may
52 // implicitly simplify coordination games by introducing a convention in games
53 // that require coordination.
54

69  // Returns the action that maximizes utility for the agent at the given
70  // infostate. The infostate must correspond to a decision node for
71  // best_responder.
72  Action BestResponseAction(const std::string& infostate);
73  Action BestResponseAction(const State& state) {
74  SPIEL_CHECK_EQ(state.CurrentPlayer(), best_responder_);
75  return BestResponseAction(state.InformationStateString(best_responder_));
76  }

78  // Returns all the actions that maximize utility for the agent at the given
79  // infostate. The infostate must correspond to a decision node for

81  std::vector<Action> BestResponseActions(const std::string& infostate,
82  double tolerance);

101  // Returns a map of infostates to best responses, for all information states
102  // that have been calculated so far. If no best responses have been
103  // calculated, then we calculate them for every state in the game.
104  // When two actions have the same value, we
105  // return the action with the lowest number (as an int).
106  std::unordered_map<std::string, Action> GetBestResponseActions() {
107  if (action_value_tolerance_ >= 0.0)
108  SpielFatalError(
109  "TabularBestResponse is returning the max-entropy best-response but "
110  "deterministic best-response is requested.");
111  // If the best_response_policy_ cache is empty, we fill it by
112  // calculating all best responses, starting at the root.
113  if (best_response_actions_.empty()) Value(*root_);
114  return best_response_actions_;
115  }
116 
117  // Returns the computed best response as a policy object.
118  TabularPolicy GetBestResponsePolicy() {
119  // If the best_response_policy_ cache is empty, we fill it by calculating
120  // all best responses, starting at the root.
121  if (best_response_policy_.empty()) Value(*root_);
122  return TabularPolicy(best_response_policy_);
123  }

125  // Returns the expected utility for best_responder when playing the game
126  // beginning at history.
127  double Value(const std::string& history);
128  double Value(const State& state) { return Value(state.HistoryString()); }
129 
130  // Changes the policy that we are calculating a best response to. This is
131  // useful as a large amount of the data structures can be reused, causing
132  // the calculation to be quicker than if we had to re-initialize the class.
133  void SetPolicy(const Policy* policy) {
134  policy_ = policy;
135  value_cache_.clear();
136  best_response_actions_.clear();
137  best_response_policy_.clear();
138  // TODO(author1): Replace this with something that traverses the tree
139  // and rebuilds the probabilities.
140  infosets_ =
141  GetAllInfoSets(root_->Clone(), best_responder_, policy_, &tree_);
142  }
143 
144  // Set the policy given a policy table. This stores the table internally.
145  void SetPolicy(
146  const std::unordered_map<std::string, ActionsAndProbs>& policy_table) {
147  tabular_policy_container_ = TabularPolicy(policy_table);
148  SetPolicy(&tabular_policy_container_);
149  }

151  private:
152  // For chance nodes, we recursively calculate the value of each child node,
153  // and weight them by the probability of reaching each child.
154  double HandleChanceCase(HistoryNode* node);
155 
156  // Calculates the value of the HistoryNode when we have to make a decision.
157  // Does this by calculating the value of each possible child node and then
158  // setting the value of the current node equal to the maximum (as we can just
159  // choose the best child).
160  double HandleDecisionCase(HistoryNode* node);
161 
162  // Calculates the value of the HistoryNode when the node is a terminal node.
163  // Conveniently, the game tells us the value of every terminal node, so we
164  // have nothing to do.
165  double HandleTerminalCase(const HistoryNode& node) const;

167  Player best_responder_;
168 
169  // Used to store a specific policy if not passed in from the caller.
170  TabularPolicy tabular_policy_container_;
171 
172  // The actual policy that we are computing a best response to.
173  const Policy* policy_;
174 
175  HistoryTree tree_;
176  int num_players_;
177 
178  // The probability tolerance for truncating value estimation.
179  float prob_cut_threshold_;
180 
181  // The tolerance in terms of action values deciding if a maxent BR is
182  // requested.
183  float action_value_tolerance_;
184 
185  // Maps infoset strings (from the State::InformationState method) to
186  // the HistoryNodes that represent all histories with
187  // the same information state, along with the counter-factual probability of
188  // doing so. If the information state is a chance node, the probability comes
189  // from the State::ChanceOutcomes method. If the information state is a
190  // decision node for best_responder, the probability is one, following the
191  // definition of counter-factual probability. Finally, if the information
192  // state is a decision node for a player other than best_responder, the
193  // probabilities come from their policy (i.e. policy_).
194  absl::flat_hash_map<std::string, std::vector<std::pair<HistoryNode*, double>>>
195  infosets_;
196 
197  // Caches all best responses calculated so far (for each infostate).
198  std::unordered_map<std::string, ActionsAndProbs> best_response_policy_;
199 
200  // Caches all best responses calculated so far (for each infostate) in case of
201  // biased deterministic best-response.
202  std::unordered_map<std::string, Action> best_response_actions_;
203 
204  // Caches all values calculated so far (for each history).
205  std::unordered_map<std::string, double> value_cache_;
206  std::unique_ptr<State> root_;
207 
208  // Keep a cache of an empty policy to avoid recomputing it.
209  std::unique_ptr<TabularPolicy> dummy_policy_;
210 };
