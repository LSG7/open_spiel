26// Track the count, min max, avg and standard deviation.
27class BasicStats {
28 public:
29 BasicStats() { Reset(); }
30
31 // Reset all the stats to 0.
32 void Reset() {
33 num_ = 0;
34 min_ = std::numeric_limits<double>::max();
35 max_ = std::numeric_limits<double>::min();
36 sum_ = 0;
37 sum_sq_ = 0;
38 }
39
40 // Merge two BasicStats. Useful for merging per thread stats before printing.
41 BasicStats& operator+=(const BasicStats& o) {
42 num_ += o.num_;
43 sum_ += o.sum_;
44 sum_sq_ += o.sum_sq_;
45 min_ = std::min(min_, o.min_);
46 max_ = std::max(max_, o.max_);
47 return *this;
48 }
49
50 void Add(double val) {
51 min_ = std::min(min_, val);
52 max_ = std::max(max_, val);
53 sum_ += val;
54 sum_sq_ += val * val;
55 num_ += 1;
56 }
57
58 int64_t Num() const { return num_; }
59 double Min() const { return (num_ == 0 ? 0 : min_); }
60 double Max() const { return (num_ == 0 ? 0 : max_); }
61 double Avg() const { return (num_ == 0 ? 0 : sum_ / num_); }
62 double StdDev() const {
63 if (num_ <= 1) return 0;
64 // Numerical precision can cause variance to be negative, leading to NaN's.
65 double variance = (sum_sq_ - sum_ * sum_ / num_) / (num_ - 1);
66 return variance <= 0 ? 0 : std::sqrt(variance);
67 }
68
69 json::Object ToJson() const {
70 return {
71 {"num", Num()},
72 {"min", Min()},
73 {"max", Max()},
74 {"avg", Avg()},
75 {"std_dev", StdDev()},
76 };
77 }
78
79 private:
80 int64_t num_;
81 double min_;
82 double max_;
83 double sum_;
84 double sum_sq_;
85};

87// Track the occurences for `count` buckets. You need to decide how to map your
88// data into the buckets. Mainly useful for scalar values.
89class HistogramNumbered {
90 public:
91 explicit HistogramNumbered(int num_buckets) : counts_(num_buckets, 0) {}
92 void Reset() { absl::c_fill(counts_, 0); }
93 void Add(int bucket_id) { counts_[bucket_id] += 1; }
94 json::Array ToJson() const { return json::CastToArray(counts_); }
95
96 private:
97 std::vector<int> counts_;
98};
99
100// Same as HistogramNumbered, but each bucket has a name associated with it
101// and is returned in the json output. Mainly useful for categorical values.
102class HistogramNamed {
103 public:
104 explicit HistogramNamed(std::vector<std::string> names)
105 : counts_(names.size(), 0), names_(names) {}
106 void Reset() { absl::c_fill(counts_, 0); }
107 void Add(int bucket_id) { counts_[bucket_id] += 1; }
108 json::Object ToJson() const {
109 return {
110 {"counts", json::CastToArray(counts_)},
111 {"names", json::CastToArray(names_)},
112 };
113 }
114
115 private:
116 std::vector<int> counts_;
117 std::vector<std::string> names_;
118};
