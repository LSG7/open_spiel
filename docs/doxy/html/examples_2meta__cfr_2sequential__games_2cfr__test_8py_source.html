33 @parameterized.named_parameters(('kuhn_poker_test', 'kuhn_poker'),
34 ('leduc_poker_test', 'leduc_poker'))
35 def test_zero_policy_is_uniform(self, game):
36 config = {'players': 2}
37 cfr_game_tree = trees.build_game_tree(
38 openspiel_api.WorldState(
39 game_name=game, config=config, perturbation=False))
40 cfr.compute_cfr_values(cfr_game_tree, 1)
41 infostates_p1 = list(cfr_game_tree.all_infostates_map[1].values())
42 infostates_p2 = list(cfr_game_tree.all_infostates_map[2].values())
43 with self.subTest('player_1_initial_policy'):
44 for i in range(len(infostates_p1)):
45 self.assertListEqual(
46 list(infostates_p1[i].policy.values()),
47 _uniform_policy(len(infostates_p1[i].policy.values())))
48 with self.subTest('player_2_initial_policy'):
49 for i in range(len(infostates_p2)):
50 self.assertListEqual(
51 list(infostates_p2[i].policy.values()),
52 _uniform_policy(len(infostates_p2[i].policy.values())))

54 def test_cfr_leduc_poker(self):
55 config = {'players': 2}
56 exploitability_error = 0.2
57 cfr_game_tree = trees.build_game_tree(
58 openspiel_api.WorldState(
59 game_name='leduc_poker', config=config, perturbation=False))
60 best_response_value_p1, best_response_value_p2 = cfr.compute_cfr_values(
61 cfr_game_tree, 20)
62 last_best_response_value_player_1 = best_response_value_p1[-1]
63 last_best_response_value_player_2 = best_response_value_p2[-1]
64 exploitability = (last_best_response_value_player_1 +
65 last_best_response_value_player_2) / 2
66 # Exploitability values are computed using OpenSpiel cfr
67 self.assertLessEqual(exploitability, 0.59 + exploitability_error)
