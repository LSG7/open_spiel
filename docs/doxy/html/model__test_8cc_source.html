49void TestModelInference() {
50 std::cout << "\n~-~-~-~- TestModelInference -~-~-~-~" << std::endl;
51
52 const int channels = 3;
53 const int rows = 8;
54 const int columns = 8;
55 std::string game_string =
56 absl::StrCat("clobber(rows=", std::to_string(rows),
57 ",columns=", std::to_string(columns), ")");
58
59 std::shared_ptr<const Game> game = LoadGame(game_string);
60 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
61
62 ModelConfig net_config = {
63 /*observation_tensor_shape=*/game->ObservationTensorShape(),
64 /*number_of_actions=*/game->NumDistinctActions(),
65 /*nn_depth=*/rows + 1,
66 /*nn_width=*/128,
67 /*learning_rate=*/0.001,
68 /*weight_decay=*/0.001};
69 Model net(net_config, "cpu:0");
70
71 std::vector<float> observation_vector = state->ObservationTensor();
72 torch::Tensor observation_tensor = torch::from_blob(
73 observation_vector.data(), {1, channels * rows * columns});
74 torch::Tensor mask = torch::full({1, game->NumDistinctActions()}, false,
75 torch::TensorOptions().dtype(torch::kByte));
76
77 for (Action action : state->LegalActions()) {
78 mask[0][action] = true;
79 }
80
81 std::cout << "Input:\n"
82 << observation_tensor.view({channels, rows, columns}) << std::endl;
83 std::cout << "Mask:\n" << mask << std::endl;
84
85 std::vector<torch::Tensor> output = net(observation_tensor, mask);
86
87 std::cout << "Output:\n" << output << std::endl;
88
89 // Check value and policy.
90 SPIEL_CHECK_EQ((int)output.size(), 2);
91 SPIEL_CHECK_EQ(output[0].numel(), 1);
92 SPIEL_CHECK_EQ(output[1].numel(), game->NumDistinctActions());
93
94 // Check mask's influence on the policy.
95 for (int i = 0; i < game->NumDistinctActions(); i++) {
96 if (mask[0][i].item<bool>()) {
97 SPIEL_CHECK_GT(output[1][0][i].item<float>(), 0.0);
98 } else {
99 SPIEL_CHECK_EQ(output[1][0][i].item<float>(), 0.0);
100 }
101 }
102
103 std::cout << "Value:\n" << output[0] << std::endl;
104 std::cout << "Policy:\n" << output[1] << std::endl;
105}
