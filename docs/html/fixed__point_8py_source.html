27 class FixedPoint(object):
28  """The fixed point algorithm.
29 
30  This algorithm is based on Banach-Picard iterations for the fixed point
31  operator characterizing the Nash equilibrium. At each iteration, the policy is
32  updated by computing a best response against the current mean-field or a
33  regularized version that is obtained by taking a softmax with respect to the
34  optimal Q-function, and the mean-field is updated by taking the mean-field
35  induced by the current policy.
36  """
37 
38  def __init__(self, game: pyspiel.Game, temperature: Optional[float] = None):
39  """Initializes the algorithm.
40 
41  Args:
42  game: The game to analyze.
43  temperature: If set, then instead of the greedy policy a softmax policy
44  with the specified temperature will be used to update the policy at each
45  iteration.
46  """
47  self._game_game = game
48  self._temperature_temperature = temperature
49  self._policy_policy = policy_lib.UniformRandomPolicy(self._game_game)
50  self._distribution_distribution = distribution.DistributionPolicy(game, self._policy_policy)
51 
52  def iteration(self):
53  """An itertion of Fixed Point."""
54  # Calculate the current distribution and the best response.
55  distrib = distribution.DistributionPolicy(self._game_game, self._policy_policy)
56  br_value = best_response_value.BestResponse(
57  self._game_game, distrib, value.TabularValueFunction(self._game_game))
58 
59  # Policy is either greedy or softmax with respect to the best response if
60  # temperature is specified.
61  player_ids = list(range(self._game_game.num_players()))
62  if self._temperature_temperature is None:
63  self._policy_policy = greedy_policy.GreedyPolicy(self._game_game, player_ids,
64  br_value)
65  else:
66  self._policy_policy = softmax_policy.SoftmaxPolicy(self._game_game, player_ids,
67  self._temperature_temperature, br_value)
68 
69  self._distribution_distribution = distribution.DistributionPolicy(
70  self._game_game, self._policy_policy)

73  return self._policy_policy
