22 // Transforms a game into its Misere version by inverting the sign of the
23 // rewards / utilities. This is a self-inverse operation.
24 // https://en.wikipedia.org/wiki/Mis%C3%A8re

28 // Flips the sign of a vector.
29 inline std::vector<double> Negative(std::vector<double>&& vector) {
30  std::vector<double> neg = std::move(vector);
31  for (auto& item : neg) item = -item;
32  return neg;
33 }
