92// This exception class is used to forward errors from Spiel to Python.
93// Do not create exceptions of this type directly! Instead, call
94// SpielFatalError, which will raise a Python exception when called from
95// Python, and exit the process otherwise.
96class SpielException : public std::exception {
97 public:
98 explicit SpielException(std::string message) : message_(message) {}
99 const char* what() const noexcept override { return message_.c_str(); }
100
101 private:
102 std::string message_;
103};
