28_OPTIONAL_GAMES = frozenset(["hanabi", "universal_poker"])
29_AVAILABLE_GAMES = set(pyspiel.registered_names())

31# Games for which we do not have playthroughs. Please don't add new games
32# here if you can avoid it. Adding a playthrough is easy and very useful!
33# Run `generate_new_playthrough.sh $GAME` to add a playthrough.
34_MISSING_GAMES = set(["nfg_game", "efg_game", "restricted_nash_response"])
35
36# Regex to find the game name in a playthrough. This will return the name of the
37# transform for wrapped games, e.g. goofspiel --> turn_based_simultaneous_game
38_SHORTNAME = r'^GameType\.short_name = "(.*)"$'

41def _is_optional_game(basename):
42 """Returns (bool, game_name or None).
43
44 Args:
45 basename: The basename of the file. It is assumed it starts with the game
46 name.
47 """
48 for game_name in _OPTIONAL_GAMES:
49 if basename.startswith(game_name):
50 return True, game_name
51 return False, None

54def _playthrough_match(filename, regex):
55 """Returns the specified value fromm the playthrough."""
56 with open(filename, "r", encoding="utf-8") as f:
57 data = f.read()
58 return re.search(regex, data, re.MULTILINE)

61class PlaythroughTest(absltest.TestCase):
62
63 def run_test(self, path, basename):
64 """Instantiated for each test case in main, below."""
65
66 # We check whether the game is optional, and if it is, whether we do
67 # have the game.
68 is_optional, game_name = _is_optional_game(basename)
69 if is_optional:
70 if game_name not in _AVAILABLE_GAMES:
71 logging.info("Skipping %s because %s is not built in.", basename,
72 game_name)
73 return
74
75 file_path = os.path.join(path, basename)
76 expected, actual = generate_playthrough.replay(file_path)
77 for line_num, (expected_line, actual_line) in enumerate(
78 zip(expected.split("\n"), actual.split("\n"))):
79 self.assertEqual(
80 expected_line,
81 actual_line,
82 msg="Wrong line {} in {}".format(line_num, basename))
83 self.assertMultiLineEqual(expected, actual)

85 def test_all_games_tested(self):
86 """Verify that every game is present in the playthroughs."""
87 test_srcdir = os.environ.get("TEST_SRCDIR", "")
88 path = os.path.join(test_srcdir, _DATA_DIR)
89 basenames = set(os.listdir(path))
90 missing_games = set(_AVAILABLE_GAMES) - set(_MISSING_GAMES) - set(
91 _playthrough_match(os.path.join(path, basename), _SHORTNAME)[1]
92 for basename in basenames)
93 self.assertEmpty(
94 missing_games,
95 msg="These games do not have playthroughs."
96 "Create playthroughs using generate_new_playthrough.sh")

99def _add_tests():
100 """Adds a test for each playthrough to the test class (above)."""
101 test_srcdir = os.environ.get("TEST_SRCDIR", "")
102 path = os.path.join(test_srcdir, _DATA_DIR)
103 basenames = sorted(os.listdir(path))
104 if len(basenames) < 40:
105 raise ValueError(f"Playthroughs are missing from {path}")
106 for basename in basenames:
107 test_name = f"test_playthrough_{basename}"
108 test_func = lambda self, basename=basename: self.run_test(path, basename)
109 test_func.__name__ = test_name
110 setattr(PlaythroughTest, test_name, test_func)
