29 def _partial_multi_dot(player_payoff_tensor, strategies, index_avoided):
30  """Computes a generalized dot product avoiding one dimension.
31 
32  This is used to directly get the expected return of a given action, given
33  other players' strategies, for the player indexed by index_avoided.
34  Note that the numpy.dot function is used to compute this product, as it ended
35  up being (Slightly) faster in performance tests than np.tensordot. Using the
36  reduce function proved slower for both np.dot and np.tensordot.
37 
38  Args:
39  player_payoff_tensor: payoff tensor for player[index_avoided], of dimension
40  (dim(vector[0]), dim(vector[1]), ..., dim(vector[-1])).
41  strategies: Meta strategy probabilities for each player.
42  index_avoided: Player for which we do not compute the dot product.
43 
44  Returns:
45  Vector of expected returns for each action of player [the player indexed by
46  index_avoided].
47  """
48  new_axis_order = [index_avoided] + [
49  i for i in range(len(strategies)) if (i != index_avoided)
50  ]
51  accumulator = np.transpose(player_payoff_tensor, new_axis_order)
52  for i in range(len(strategies) - 1, -1, -1):
53  if i != index_avoided:
54  accumulator = np.dot(accumulator, strategies[i])
55  return accumulator

58 def _regret_matching_step(payoff_tensors, strategies, regrets, gamma):
59  """Does one step of the projected replicator dynamics algorithm.
60 
61  Args:
62  payoff_tensors: List of payoff tensors for each player.
63  strategies: List of the strategies used by each player.
64  regrets: List of cumulative regrets used by each player.
65  gamma: Minimum exploratory probability term.
66 
67  Returns:
68  A list of updated strategies for each player.
69  """
70 
71  # TODO(author4): Investigate whether this update could be fully vectorized.
72  new_strategies = []
73  for player in range(len(payoff_tensors)):
74  current_payoff_tensor = payoff_tensors[player]
75  current_strategy = strategies[player]
76 
77  values_per_strategy = _partial_multi_dot(current_payoff_tensor, strategies,
78  player)
79  average_return = np.dot(values_per_strategy, current_strategy)
80  regrets[player] += values_per_strategy - average_return
81 
82  updated_strategy = regrets[player].copy()
83  updated_strategy[updated_strategy < 0] = 0.0
84  sum_regret = updated_strategy.sum()
85  uniform_strategy = np.ones(len(updated_strategy)) / len(updated_strategy)
86 
87  if sum_regret > 0:
88  updated_strategy /= sum_regret
89  updated_strategy = gamma * uniform_strategy + (1 -
90  gamma) * updated_strategy
91  else:
92  updated_strategy = uniform_strategy
93 
94  new_strategies.append(updated_strategy)
95  return new_strategies

98 def regret_matching(payoff_tensors,
99  initial_strategies=None,
100  iterations=int(1e5),
101  gamma=1e-6,
102  average_over_last_n_strategies=None,
103  **unused_kwargs):
104  """Runs regret-matching for the stated number of iterations.
105 
106  Args:
107  payoff_tensors: List of payoff tensors for each player.
108  initial_strategies: Initial list of the strategies used by each player, if
109  any. Could be used to speed up the search by providing a good initial
110  solution.
111  iterations: Number of algorithmic steps to take before returning an answer.
112  gamma: Minimum exploratory probability term.
113  average_over_last_n_strategies: Running average window size for average
114  policy computation. If None, use the whole trajectory.
115  **unused_kwargs: Convenient way of exposing an API compatible with other
116  methods with possibly different arguments.
117 
118  Returns:
119  RM-computed strategies.
120  """
121  number_players = len(payoff_tensors)
122  # Number of actions available to each player.
123  action_space_shapes = payoff_tensors[0].shape
124 
125  # If no initial starting position is given, start with uniform probabilities.
126  new_strategies = initial_strategies or [
127  np.ones(action_space_shapes[k]) / action_space_shapes[k]
128  for k in range(number_players)
129  ]
130 
131  regrets = [
132  np.ones(action_space_shapes[k]) / INITIAL_REGRET_DENOM
133  for k in range(number_players)
134  ]
135 
136  averager = nfg_utils.StrategyAverager(number_players, action_space_shapes,
137  average_over_last_n_strategies)
138  averager.append(new_strategies)
139 
140  for _ in range(iterations):
141  new_strategies = _regret_matching_step(payoff_tensors, new_strategies,
142  regrets, gamma)
143  averager.append(new_strategies)
144  return averager.average_strategies()
