43std::vector<double> RandomRolloutEvaluator::Evaluate(const State& state) {
44 std::vector<double> result;
45 for (int i = 0; i < n_rollouts_; ++i) {
46 std::unique_ptr<State> working_state = state.Clone();
47 while (!working_state->IsTerminal()) {
48 if (working_state->IsChanceNode()) {
49 ActionsAndProbs outcomes = working_state->ChanceOutcomes();
50 working_state->ApplyAction(SampleAction(outcomes, rng_).first);
51 } else {
52 std::vector<Action> actions = working_state->LegalActions();
53 working_state->ApplyAction(
54 actions[absl::Uniform(rng_, 0u, actions.size())]);
55 }
56 }
57
58 std::vector<double> returns = working_state->Returns();
59 if (result.empty()) {
60 result.swap(returns);
61 } else {
62 SPIEL_CHECK_EQ(returns.size(), result.size());
63 for (int i = 0; i < result.size(); ++i) {
64 result[i] += returns[i];
65 }
66 }
67 }
68 for (int i = 0; i < result.size(); ++i) {
69 result[i] /= n_rollouts_;
70 }
71 return result;
72}

74ActionsAndProbs RandomRolloutEvaluator::Prior(const State& state) {
75 // Returns equal probability for all actions.
76 if (state.IsChanceNode()) {
77 return state.ChanceOutcomes();
78 } else {
79 std::vector<Action> legal_actions = state.LegalActions();
80 ActionsAndProbs prior;
81 prior.reserve(legal_actions.size());
82 for (const Action& action : legal_actions) {
83 prior.emplace_back(action, 1.0 / legal_actions.size());
84 }
85 return prior;
86 }
87}

89// UCT value of given child
90double SearchNode::UCTValue(int parent_explore_count, double uct_c) const {
91 if (!outcome.empty()) {
92 return outcome[player];
93 }
94
95 if (explore_count == 0) return std::numeric_limits<double>::infinity();
96
97 // The "greedy-value" of choosing a given child is always with respect to
98 // the current player for this node.
99 return total_reward / explore_count +
100 uct_c * std::sqrt(std::log(parent_explore_count) / explore_count);
101}
102
103double SearchNode::PUCTValue(int parent_explore_count, double uct_c) const {
104 // Returns the PUCT value of this node.
105 if (!outcome.empty()) {
106 return outcome[player];
107 }
108
109 return ((explore_count != 0 ? total_reward / explore_count : 0) +
110 uct_c * prior * std::sqrt(parent_explore_count) /
111 (explore_count + 1));
112}

114bool SearchNode::CompareFinal(const SearchNode& b) const {
115 double out = (outcome.empty() ? 0 : outcome[player]);
116 double out_b = (b.outcome.empty() ? 0 : b.outcome[b.player]);
117 if (out != out_b) {
118 return out < out_b;
119 }
120 if (explore_count != b.explore_count) {
121 return explore_count < b.explore_count;
122 }
123 return total_reward < b.total_reward;
124}

126const SearchNode& SearchNode::BestChild() const {
127 // Returns the best action from this node, either proven or most visited.
128 //
129 // This ordering leads to choosing:
130 // - Highest proven score > 0 over anything else, including a promising but
131 // unproven action.
132 // - A proven draw only if it has higher exploration than others that are
133 // uncertain, or the others are losses.
134 // - Uncertain action with most exploration over loss of any difficulty
135 // - Hardest loss if everything is a loss
136 // - Highest expected reward if explore counts are equal (unlikely).
137 // - Longest win, if multiple are proven (unlikely due to early stopping).
138 return *std::max_element(children.begin(), children.end(),
139 [](const SearchNode& a, const SearchNode& b) {
140 return a.CompareFinal(b);
141 });
142}

144std::string SearchNode::ChildrenStr(const State& state) const {
145 std::string out;
146 if (!children.empty()) {
147 std::vector<const SearchNode*> refs; // Sort a list of refs, not a copy.
148 refs.reserve(children.size());
149 for (const SearchNode& child : children) {
150 refs.push_back(&child);
151 }
152 std::sort(refs.begin(), refs.end(),
153 [](const SearchNode* a, const SearchNode* b) {
154 return b->CompareFinal(*a);
155 });
156 for (const SearchNode* child : refs) {
157 absl::StrAppend(&out, child->ToString(state), "\n");
158 }
159 }
160 return out;
161}

187std::vector<double> dirichlet_noise(int count, double alpha,
188 std::mt19937* rng) {
189 std::vector<double> noise;
190 noise.reserve(count);
191
192 std::gamma_distribution<double> gamma(alpha, 1.0);
193 for (int i = 0; i < count; ++i) {
194 noise.emplace_back(gamma(*rng));
195 }
196
197 double sum = absl::c_accumulate(noise, 0.0);
198 for (double& v : noise) {
199 v /= sum;
200 }
201 return noise;
202}

231Action MCTSBot::Step(const State& state) {
232 absl::Time start = absl::Now();
233 std::unique_ptr<SearchNode> root = MCTSearch(state);
234
235 if (max_simulations_ <= 1) {
236 // sample from prior
237 return root->SampleFromPrior(state, evaluator_.get(), &rng_);
238 } else {
239 // return best action
240 const SearchNode& best = root->BestChild();
241
242 if (verbose_) {
243 double seconds = absl::ToDoubleSeconds(absl::Now() - start);
244 std::cerr << absl::StrFormat(
245 ("Finished %d sims in %.3f secs, %.1f sims/s, "
246 "tree size: %d nodes / %d mb."),
247 root->explore_count, seconds,
248 (root->explore_count / seconds), nodes_,
249 MemoryUsedMb(nodes_))
250 << std::endl;
251 std::cerr << "Root:" << std::endl;
252 std::cerr << root->ToString(state) << std::endl;
253 std::cerr << "Children:" << std::endl;
254 std::cerr << root->ChildrenStr(state) << std::endl;
255 if (!best.children.empty()) {
256 std::unique_ptr<State> chosen_state = state.Clone();
257 chosen_state->ApplyAction(best.action);
258 std::cerr << "Children of chosen:" << std::endl;
259 std::cerr << best.ChildrenStr(*chosen_state) << std::endl;
260 }
261 }
262 return best.action;
263 }

271std::unique_ptr<State> MCTSBot::ApplyTreePolicy(
272 SearchNode* root, const State& state,
273 std::vector<SearchNode*>* visit_path) {
274 visit_path->push_back(root);
275 std::unique_ptr<State> working_state = state.Clone();
276 SearchNode* current_node = root;
277 while ((!working_state->IsTerminal() && current_node->explore_count > 0) ||
278 (working_state->IsChanceNode() && dont_return_chance_node_)) {
279 if (current_node->children.empty()) {
280 // For a new node, initialize its state, then choose a child as normal.
281 ActionsAndProbs legal_actions = evaluator_->Prior(*working_state);
282 if (current_node == root && dirichlet_alpha_ > 0) {
283 std::vector<double> noise =
284 dirichlet_noise(legal_actions.size(), dirichlet_alpha_, &rng_);
285 for (int i = 0; i < legal_actions.size(); i++) {
286 legal_actions[i].second =
287 (1 - dirichlet_epsilon_) * legal_actions[i].second +
288 dirichlet_epsilon_ * noise[i];
289 }
290 }
291 // Reduce bias from move generation order.
292 std::shuffle(legal_actions.begin(), legal_actions.end(), rng_);
293 Player player = working_state->CurrentPlayer();
294 current_node->children.reserve(legal_actions.size());
295 for (auto [action, prior] : legal_actions) {
296 current_node->children.emplace_back(action, player, prior);
297 }
298 nodes_ += current_node->children.capacity();
299 }
300
301 Action selected_action;
302 if (current_node->children.empty()) {
303 // no children, sample from prior
304 selected_action = current_node->SampleFromPrior(state, evaluator_.get(),
305 &rng_);
306 } else {
307 // look at children
308 SearchNode* chosen_child = nullptr;
309 if (working_state->IsChanceNode()) {
310 // For chance nodes, rollout according to chance node's probability
311 // distribution
312 Action chosen_action =
313 SampleAction(working_state->ChanceOutcomes(), rng_).first;
314
315 for (SearchNode& child : current_node->children) {
316 if (child.action == chosen_action) {
317 chosen_child = &child;
318 break;
319 }
320 }
321 } else {
322 // Otherwise choose node with largest UCT value.
323 double max_value = -std::numeric_limits<double>::infinity();
324 for (SearchNode& child : current_node->children) {
325 double val;
326 switch (child_selection_policy_) {
327 case ChildSelectionPolicy::UCT:
328 val = child.UCTValue(current_node->explore_count, uct_c_);
329 break;
330 case ChildSelectionPolicy::PUCT:
331 val = child.PUCTValue(current_node->explore_count, uct_c_);
332 break;
333 }
334 if (val > max_value) {
335 max_value = val;
336 chosen_child = &child;
337 }
338 }
339 }
340 selected_action = chosen_child->action;
341 current_node = chosen_child;
342 }
343
344 working_state->ApplyAction(selected_action);
345 visit_path->push_back(current_node);
346 }
347
348 return working_state;
349}

351std::unique_ptr<SearchNode> MCTSBot::MCTSearch(const State& state) {
352 nodes_ = 1;
353 gc_limit_ = MIN_GC_LIMIT;
354 auto root = std::make_unique<SearchNode>(kInvalidAction,
355 state.CurrentPlayer(), 1);
356 std::vector<SearchNode*> visit_path;
357 std::vector<double> returns;
358 visit_path.reserve(64);
359 for (int i = 0; i < max_simulations_; ++i) {
360 visit_path.clear();
361 returns.clear();
362
363 std::unique_ptr<State> working_state =
364 ApplyTreePolicy(root.get(), state, &visit_path);
365
366 bool solved;
367 if (working_state->IsTerminal()) {
368 returns = working_state->Returns();
369 visit_path[visit_path.size() - 1]->outcome = returns;
370 solved = solve_;
371 } else {
372 returns = evaluator_->Evaluate(*working_state);
373 solved = false;
374 }
375
376 // Propagate values back.
377 while (!visit_path.empty()) {
378 int decision_node_idx = visit_path.size() - 1;
379 SearchNode* node = visit_path[decision_node_idx];
380
381 // If it's a chance node, find the parent player id.
382 while (visit_path[decision_node_idx]->player == kChancePlayerId) {
383 decision_node_idx--;
384 }
385
386 node->total_reward += returns[visit_path[decision_node_idx]->player];
387 node->explore_count += 1;
388 visit_path.pop_back();
389
390 // Back up solved results as well.
391 if (solved && !node->children.empty()) {
392 Player player = node->children[0].player;
393 if (player == kChancePlayerId) {
394 // Only back up chance nodes if all have the same outcome.
395 // An alternative would be to back up the weighted average of
396 // outcomes if all children are solved, but that is less clear.
397 const std::vector<double>& outcome = node->children[0].outcome;
398 if (!outcome.empty() &&
399 std::all_of(node->children.begin() + 1, node->children.end(),
400 [&outcome](const SearchNode& c) {
401 return c.outcome == outcome;
402 })) {
403 node->outcome = outcome;
404 } else {
405 solved = false;
406 }
407 } else {
408 // If any have max utility (won?), or all children are solved,
409 // choose the one best for the player choosing.
410 const SearchNode* best = nullptr;
411 bool all_solved = true;
412 for (const SearchNode& child : node->children) {
413 if (child.outcome.empty()) {
414 all_solved = false;
415 } else if (best == nullptr ||
416 child.outcome[player] > best->outcome[player]) {
417 best = &child;
418 }
419 }
420 if (best != nullptr &&
421 (all_solved || best->outcome[player] == max_utility_)) {
422 node->outcome = best->outcome;
423 } else {
424 solved = false;
425 }
426 }
427 }
428 }
429
430 if (!root->outcome.empty() || // Full game tree is solved.
431 root->children.size() == 1) {
432 break;
433 }
434 if (max_nodes_ > 1 && nodes_ >= max_nodes_) {
435 // Note that actual memory used as counted by ps/top might exceed the
436 // counted value here, possibly by a significant margin (1.5x even!). Part
437 // of that is not counting the outcome array, but most of that is due to
438 // memory fragmentation and is out of our control without writing our own
439 // memory manager.
440 if (verbose_) {
441 std::cerr << absl::StrFormat(
442 ("Approx %d mb in %d nodes after %d sims, garbage collecting with "
443 "limit %d ... "),
444 MemoryUsedMb(nodes_), nodes_, i, gc_limit_);
445 }
446 GarbageCollect(root.get());
447
448 // Slowly increase or decrease to target releasing half the memory.
449 gc_limit_ *= (nodes_ > max_nodes_ / 2 ? 1.25 : 0.9);
450 gc_limit_ = std::max(MIN_GC_LIMIT, gc_limit_);
451 if (verbose_) {
452 std::cerr << absl::StrFormat(
453 "%d mb in %d nodes remaining\n",
454 MemoryUsedMb(nodes_), nodes_);
455 }
456 }
457 }
458
459 return root;
460}

462void MCTSBot::GarbageCollect(SearchNode* node) {
463 if (node->children.empty()) {
464 return;
465 }
466 bool clear_children = node->explore_count < gc_limit_;
467 for (SearchNode& child : node->children) {
468 GarbageCollect(&child);
469 }
470 if (clear_children) {
471 nodes_ -= node->children.capacity();
472 node->children.clear();
473 node->children.shrink_to_fit(); // release the memory
474 }
475}
