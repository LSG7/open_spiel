114 def _regret_matching(self, regrets, num_legal_actions):
115 """Applies regret matching to get a policy.
116
117 Args:
118 regrets: numpy array of regrets for each action.
119 num_legal_actions: number of legal actions at this state.
120
121 Returns:
122 numpy array of the policy indexed by the index of legal action in the
123 list.
124 """
125 positive_regrets = np.maximum(regrets,
126 np.zeros(num_legal_actions, dtype=np.float64))
127 sum_pos_regret = positive_regrets.sum()

129 return np.ones(num_legal_actions, dtype=np.float64) / num_legal_actions
130 else:
131 return positive_regrets / sum_pos_regret
