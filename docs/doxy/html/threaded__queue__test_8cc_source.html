25void TestThreadedQueue() {

27
28 auto CheckPopEq = [&q](int expected) {
29 absl::optional<int> v = q.Pop();
30 SPIEL_CHECK_TRUE(v);
31 SPIEL_CHECK_EQ(*v, expected);
32 };
33
34 SPIEL_CHECK_TRUE(q.Empty());
35 SPIEL_CHECK_EQ(q.Size(), 0);
36
37 SPIEL_CHECK_FALSE(q.Pop(absl::Milliseconds(1)));
