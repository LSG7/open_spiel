88// Cards are represented suit * number of cards per suit + rank
89Suit CardSuit(int card) { return Suit(card / 13); }
90int CardRank(int card) { return card % 13; }
91int Card(Suit suit, int rank) {
92 return static_cast<int>(suit) * kNumCardsPerSuit + rank;
93}

108// There are two partnerships: players 0 and 2 versus players 1 and 3.
109// We call 0 and 2 partnership 0, and 1 and 3 partnership 1.
110int Partnership(Player player) { return player & 1; }
111int Partner(Player player) { return (player + 2) % 4; }

144std::array<std::string, kNumSuits> FormatHand(
145 int player, bool mark_voids,
146 const std::array<absl::optional<Player>, kNumCards>& deal) {
147 std::array<std::string, kNumSuits> cards;
148 for (int suit = 0; suit < kNumSuits; ++suit) {
149 cards[suit].push_back(kSuitChar[suit]);
150 cards[suit].push_back(' ');
151 bool is_void = true;
152 for (int rank = kNumCardsPerSuit - 1; rank >= 0; --rank) {
153 if (player == deal[Card(Suit(suit), rank)]) {
154 cards[suit].push_back(kRankChar[rank]);
155 is_void = false;
156 }
157 }
158 if (is_void && mark_voids) absl::StrAppend(&cards[suit], "none");
159 }
160 return cards;
161}

163std::string SpadesState::ObservationString(Player player) const {
164 SPIEL_CHECK_GE(player, 0);
165 SPIEL_CHECK_LT(player, num_players_);
166 if (IsTerminal()) return ToString();
167 std::string rv = "";
168 auto cards = FormatHand(player, /*mark_voids=*/true, holder_);
169 for (int suit = kNumSuits - 1; suit >= 0; --suit)
170 absl::StrAppend(&rv, cards[suit], "\n");
171 if (history_.size() > kNumCards)
172 absl::StrAppend(
173 &rv, FormatAuction(/*trailing_query=*/phase_ == Phase::kAuction &&

175 if (num_cards_played_ > 0) absl::StrAppend(&rv, FormatPlay());
176 return rv;

188std::string SpadesState::FormatDeal() const {
189 std::array<std::array<std::string, kNumSuits>, kNumPlayers> cards;
190 if (IsTerminal()) {
191 // Include all cards in the terminal state to make reviewing the deal easier
192 auto deal = OriginalDeal();
193 for (auto player : {kNorth, kEast, kSouth, kWest}) {
194 cards[player] = FormatHand(player, /*mark_voids=*/false, deal);
195 }
196 } else {
197 for (auto player : {kNorth, kEast, kSouth, kWest}) {
198 cards[player] = FormatHand(player, /*mark_voids=*/false, holder_);
199 }
200 }
201 constexpr int kColumnWidth = 8;
202 std::string padding(kColumnWidth, ' ');
203 std::string rv;
204 for (int suit = kNumSuits - 1; suit >= 0; --suit)
205 absl::StrAppend(&rv, padding, cards[kNorth][suit], "\n");
206 for (int suit = kNumSuits - 1; suit >= 0; --suit)
207 absl::StrAppend(&rv, absl::StrFormat("%-8s", cards[kWest][suit]), padding,
208 cards[kEast][suit], "\n");
209 for (int suit = kNumSuits - 1; suit >= 0; --suit)
210 absl::StrAppend(&rv, padding, cards[kSouth][suit], "\n");
211 return rv;
212}

214std::string SpadesState::FormatAuction(bool trailing_query) const {
215 SPIEL_CHECK_GT(history_.size(), kNumCards);
216 std::string rv = "\nNorth East South West ";
217 for (int i = kNumCards; i < history_.size() - num_cards_played_; ++i) {
218 if (i % kNumPlayers == 0) rv.push_back('\n');
219 absl::StrAppend(
220 &rv, absl::StrFormat(
221 "%-6s", BidString(history_[i].action - kBiddingActionBase)));
222 }
223 if (trailing_query) {
224 if ((history_.size() - num_cards_played_) % kNumPlayers == kNumPlayers - 1)
225 rv.push_back('\n');
226 rv.push_back('?');
227 }
228 return rv;
229}

231std::string SpadesState::FormatPlay() const {
232 SPIEL_CHECK_GT(num_cards_played_, 0);
233 std::string rv = "\n\nN E S W N E S";
234 Trick trick{kInvalidPlayer, 0};
235 Player player = kFirstPlayer;
236 for (int i = 0; i < num_cards_played_; ++i) {
237 if (i % kNumPlayers == 0) {
238 if (i > 0) player = trick.Winner();
239 absl::StrAppend(&rv, "\n", std::string(3 * player, ' '));
240 } else {
241 player = (1 + player) % kNumPlayers;
242 }
243 const int card = history_[history_.size() - num_cards_played_ + i].action;
244 if (i % kNumPlayers == 0) {
245 trick = Trick(player, card);
246 } else {
247 trick.Play(player, card);
248 }
249 absl::StrAppend(&rv, CardString(card), " ");
250 }
251 absl::StrAppend(&rv, "\n\nTricks taken:\n\n", "North East South West\n",
252 absl::StrFormat("%-6d", num_player_tricks_[0]),
253 absl::StrFormat("%-6d", num_player_tricks_[1]),
254 absl::StrFormat("%-6d", num_player_tricks_[2]),
255 absl::StrFormat("%-6d", num_player_tricks_[3]), "\n");
256 return rv;
257}

273void SpadesState::WriteObservationTensor(Player player,
274 absl::Span<float> values) const {
275 SPIEL_CHECK_GE(player, 0);
276 SPIEL_CHECK_LT(player, num_players_);
277
278 std::fill(values.begin(), values.end(), 0.0);
279 if (phase_ == Phase::kDeal) return;
280 auto ptr = values.begin();
281
282 // Mark bidding or playing phase
283 ptr[static_cast<int>(phase_) - 1] = 1;
284 ptr += kPhaseInfoSize;
285
286 if (num_cards_played_ > 0) {
287 // Observation for play phase
288
289 // Contracts
290 for (int i = 0; i < kNumPlayers; i++) {
291 ptr[contracts_[i]] = 1;
292 ptr += kNumBids;
293 }
294
295 // Our remaining cards.
296 for (int i = 0; i < kNumCards; ++i)
297 if (holder_[i] == player) ptr[i] = 1;
298 ptr += kNumCards;
299
300 // Indexing into history for recent tricks.
301 int current_trick = num_cards_played_ / kNumPlayers;
302 int this_trick_cards_played = num_cards_played_ % kNumPlayers;
303 int this_trick_start = history_.size() - this_trick_cards_played;
304
305 // Current trick
306 if (phase_ != Phase::kGameOver) {
307 int leader = tricks_[current_trick].Leader();
308 for (int i = 0; i < this_trick_cards_played; ++i) {
309 int card = history_[this_trick_start + i].action;
310 int relative_player = (i + leader + kNumPlayers - player) % kNumPlayers;
311 ptr[relative_player * kNumCards + card] = 1;
312 }
313 }
314
315 ptr += kNumPlayers * kNumCards;
316
317 // Previous tricks
318 for (int j = current_trick - 1;
319 j >= std::max(0, current_trick - num_tricks_ + 1); --j) {
320 int leader = tricks_[j].Leader();
321 for (int i = 0; i < kNumPlayers; ++i) {
322 int card =
323 history_[this_trick_start - kNumPlayers * (current_trick - j) + i]
324 .action;
325 int relative_player = (i + leader + kNumPlayers - player) % kNumPlayers;
326 ptr[relative_player * kNumCards + card] = 1;
327 }
328 ptr += kNumPlayers * kNumCards;
329 }
330
331 // Move pointer for future tricks to have a fixed size tensor
332 if (num_tricks_ > current_trick + 1) {
333 ptr += kNumPlayers * kNumCards * (num_tricks_ - current_trick - 1);
334 }
335
336 // Number of tricks taken by each side.
337 for (int i = 0; i < kNumPlayers; i++) {
338 ptr[num_player_tricks_[i]] = 1;
339 ptr += kNumTricks;
340 }
341
342 int kPlayTensorSize = SpadesGame::GetPlayTensorSize(num_tricks_);
343 SPIEL_CHECK_EQ(std::distance(values.begin(), ptr),
344 kPlayTensorSize + kPhaseInfoSize);
345 SPIEL_CHECK_LE(std::distance(values.begin(), ptr), values.size());
346 } else {
347 // Observation for auction
348
349 // Bids made so far
350 for (int i = 0; i < kNumPlayers; i++) {
351 // If player has bid, mark it
352 if (contracts_[i] >= 0) {
353 ptr[contracts_[i]] = 1;
354 }
355 ptr += kNumBids;
356 }
357
358 // Our cards.
359 for (int i = 0; i < kNumCards; ++i)
360 if (holder_[i] == player) ptr[i] = 1;
361 ptr += kNumCards;
362 SPIEL_CHECK_EQ(std::distance(values.begin(), ptr),
363 kAuctionTensorSize + kPhaseInfoSize);
364 SPIEL_CHECK_LE(std::distance(values.begin(), ptr), values.size());
365 }
366}

403std::vector<Action> SpadesState::DealLegalActions() const {

405 legal_actions.reserve(kNumCards - history_.size());

412std::vector<Action> SpadesState::BiddingLegalActions() const {
413 std::vector<Action> legal_actions;
414 legal_actions.reserve(kNumBids);
415 int partner_bid = contracts_[Partner(current_player_)];
416
417 if (partner_bid >= 0) {
418 // Combined bid between partners cannot be more than 13
419 for (int bid = 0; bid < kNumBids - partner_bid; ++bid) {
420 legal_actions.push_back(kBiddingActionBase + bid);
421 }
422 } else {
423 for (int bid = 0; bid < kNumBids; ++bid) {
424 legal_actions.push_back(kBiddingActionBase + bid);
425 }
426 }
427
428 return legal_actions;
429}

431std::vector<Action> SpadesState::PlayLegalActions() const {
432 std::vector<Action> legal_actions;
433 legal_actions.reserve(kNumCardsPerHand - num_cards_played_ / kNumPlayers);
434
435 // Check if we can follow suit.
436 if (num_cards_played_ % kNumPlayers != 0) {
437 auto suit = CurrentTrick().LedSuit();
438 for (int rank = 0; rank < kNumCardsPerSuit; ++rank) {
439 if (holder_[Card(suit, rank)] == current_player_) {
440 legal_actions.push_back(Card(suit, rank));
441 }
442 }
443 } else if (num_cards_played_ % kNumPlayers == 0 && !is_spades_broken_) {
444 // If leading, and spades have not been broken, play any other suit if
445 // possible.
446 for (int suit = 0 /*kClubs*/; suit < 3 /*kSpades*/; ++suit) {
447 for (int rank = 0; rank < kNumCardsPerSuit; ++rank) {
448 if (holder_[Card(Suit(suit), rank)] == current_player_) {
449 legal_actions.push_back(Card(Suit(suit), rank));
450 }
451 }
452 }
453 }
454 if (!legal_actions.empty()) return legal_actions;
455
456 // Otherwise, we can play any of our cards.
457 for (int card = 0; card < kNumCards; ++card) {
458 if (holder_[card] == current_player_) legal_actions.push_back(card);
459 }
460 return legal_actions;
461}

463std::vector<std::pair<Action, double>> SpadesState::ChanceOutcomes() const {
464 std::vector<std::pair<Action, double>> outcomes;
465 int num_cards_remaining = kNumCards - history_.size();
466 outcomes.reserve(num_cards_remaining);
467 const double p = 1.0 / static_cast<double>(num_cards_remaining);
468 for (int card = 0; card < kNumCards; ++card) {
469 if (!holder_[card].has_value()) outcomes.emplace_back(card, p);
470 }
471 return outcomes;
472}

474void SpadesState::DoApplyAction(Action action) {
475 switch (phase_) {
476 case Phase::kDeal:
477 return ApplyDealAction(action);
478 case Phase::kAuction:
479 return ApplyBiddingAction(action - kBiddingActionBase);
480 case Phase::kPlay:
481 return ApplyPlayAction(action);
482 case Phase::kGameOver:
483 SpielFatalError("Cannot act in terminal states");
484 }

495void SpadesState::ApplyBiddingAction(int bid) {
496 // A bid was made.
497 const int partner = Partner(current_player_);
498 SPIEL_CHECK_TRUE(contracts_[partner] == -1 ||
499 bid + contracts_[partner] <= 13);
500 contracts_[current_player_] = bid;
501
502 // Mark off possible_contracts for this player's other bids
503 std::fill(
504 possible_contracts_.begin() + (current_player_ * kNumBids),
505 possible_contracts_.begin() + (current_player_ * kNumBids) + kNumBids,
506 false);
507 // If partner hasn't bid, mark off partner's possible bids that would go past
508 // 13
509 if (contracts_[partner] == -1 && bid > 0) {
510 std::fill(
511 possible_contracts_.begin() + (partner * kNumBids) + kNumBids - bid,
512 possible_contracts_.begin() + (partner * kNumBids) + kNumBids, false);
513 }
514
515 // And now mark this bid as the player's contract
516 possible_contracts_[current_player_ * kNumBids + bid] = true;
517
518 current_player_ = (current_player_ + 1) % kNumPlayers;
519
520 // After 4 bids, end the auction.
521 if (std::all_of(contracts_.begin(), contracts_.end(),
522 [](int x) { return x != -1; })) {
523 phase_ = Phase::kPlay;
524 }
525}

527void SpadesState::ApplyPlayAction(int card) {
528 SPIEL_CHECK_TRUE(holder_[card] == current_player_);
529 holder_[card] = absl::nullopt;
530 if (num_cards_played_ % kNumPlayers == 0) {
531 CurrentTrick() = Trick(current_player_, card);
532 } else {
533 CurrentTrick().Play(current_player_, card);
534 }
535 const Player winner = CurrentTrick().Winner();
536 ++num_cards_played_;
537 if (num_cards_played_ % kNumPlayers == 0) {
538 current_player_ = winner;
539 ++num_player_tricks_[current_player_];
540 } else {
541 current_player_ = (current_player_ + 1) % kNumPlayers;
542 }
543 if (num_cards_played_ == kNumCards) {
544 phase_ = Phase::kGameOver;
545 ScoreUp();
546 }
547}

559void SpadesState::ScoreUp() {
560 std::array<int, kNumPartnerships> scores =
561 Score(contracts_, num_player_tricks_, current_scores_);
562 for (int pl = 0; pl < kNumPlayers; ++pl) {
563 returns_[pl] = scores[Partnership(pl)];

574void Trick::Play(Player player, int card) {
575 if (CardSuit(card) == winning_suit_) {
576 if (CardRank(card) > winning_rank_) {
577 winning_rank_ = CardRank(card);
578 winning_player_ = player;
579 }
580 } else if (CardSuit(card) == Suit(3) /*kSpades*/) {
581 winning_suit_ = Suit(3) /*kSpades*/;
582 winning_rank_ = CardRank(card);
583 winning_player_ = player;
584 }
