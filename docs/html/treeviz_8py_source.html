15 """Visualizing game trees with graphviz.
16 
17 GameTree builds a `pygraphviz.AGraph` reprensentation of the game tree. The
18 resulting tree can be directly visualized in Jupyter notebooks or Google Colab
19 via SVG plotting - or written to a file by calling `draw(filename, prog="dot")`.
20 
21 See `examples/treeviz_example.py` for a more detailed example.
22 
23 This module relies on external dependencies, which need to be installed before
24 use. On a debian system follow these steps:
25 ```
26 sudo apt-get install graphviz libgraphviz-dev
27 pip install pygraphviz
28 ```
29 """

52 def default_node_decorator(state):
53  """Decorates a state-node of the game tree.
54 
55  This method can be called by a custom decorator to prepopulate the attributes
56  dictionary. Then only relevant attributes need to be changed, or added.
57 
58  Args:
59  state: The state.
60 
61  Returns:
62  `dict` with graphviz node style attributes.
63  """
64  player = state.current_player()
65  attrs = {
66  "label": "",
67  "fontsize": _FONTSIZE,
68  "width": _WIDTH,
69  "height": _HEIGHT,
70  "margin": _MARGIN
71  }
72  if state.is_terminal():
73  attrs["label"] = ", ".join(map(str, state.returns()))
74  attrs["shape"] = "diamond"
75  elif state.is_chance_node():
76  attrs["shape"] = "point"
77  attrs["width"] = _WIDTH / 2.
78  attrs["height"] = _HEIGHT / 2.
79  else:
80  attrs["label"] = str(state.information_state_string())
81  attrs["shape"] = _PLAYER_SHAPES.get(player, "ellipse")
82  attrs["color"] = _PLAYER_COLORS.get(player, "black")
83  return attrs

207  def _build_tree(self, state, depth, depth_limit):
208  """Recursively builds the game tree."""
209  state_str = self.state_to_strstate_to_str(state)
210 
211  if state.is_terminal():
212  self._terminal_nodes_terminal_nodes.append(state_str)
213  return
214  if depth > depth_limit >= 0:
215  return
216 
217  for action in state.legal_actions():
218  child = state.child(action)
219  child_str = self.state_to_strstate_to_str(child)
220  self.add_node(child_str, **self._node_decorator_node_decorator(child))
221  self.add_edge(state_str, child_str,
222  **self._edge_decorator_edge_decorator(state, child, action))
