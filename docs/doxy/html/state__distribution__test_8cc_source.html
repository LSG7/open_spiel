27void KuhnStateDistributionTest() {
28 std::shared_ptr<const Game> game = LoadGame("kuhn_poker");
29 std::unique_ptr<State> state = game->NewInitialState();
30 TabularPolicy uniform_policy = GetUniformPolicy(*game);
31
32 // Construct the state 1b
33 state->ApplyAction(0); // p0 card: jack
34 state->ApplyAction(1); // p1 card: queen
35 state->ApplyAction(1); // player 0 bet
36 SPIEL_CHECK_EQ(state->CurrentPlayer(), 1);
37 SPIEL_CHECK_EQ(state->InformationStateString(), "1b");
38 HistoryDistribution dist = GetStateDistribution(*state, uniform_policy);
39
40 SPIEL_CHECK_EQ(dist.first.size(), 2);
41 SPIEL_CHECK_EQ(dist.second.size(), 2);
42
43 // Check that sampled states have private cards jack and king for the opponent
44 SPIEL_CHECK_TRUE(dist.first[0]->InformationStateString(0) == "0b" ||
45 dist.first[0]->InformationStateString(0) == "2b");
46 SPIEL_CHECK_TRUE(dist.first[1]->InformationStateString(0) == "0b" ||
47 dist.first[1]->InformationStateString(0) == "2b");
48 SPIEL_CHECK_NE(dist.first[0]->InformationStateString(0),
49 dist.first[1]->InformationStateString(0));
50
51 // Check that they are equally likely, and sum to 1
52 SPIEL_CHECK_EQ(dist.second[0], 0.5);
53 SPIEL_CHECK_EQ(dist.second[0], 0.5);
54}

56void CompareDists(const HistoryDistribution& lhs,
57 const HistoryDistribution& rhs) {
58 for (int i = 0; i < lhs.first.size(); ++i) {
59 std::cerr << "lhs[" << i << "]: " << lhs.first[i]->HistoryString()
60 << ", p: " << lhs.second[i] << std::endl;
61 std::cerr << "rhs[" << i << "]: " << rhs.first[i]->HistoryString()
62 << ", p: " << rhs.second[i] << std::endl;
63 }
64 for (int i = 0; i < lhs.first.size(); ++i) {
65 for (int j = 0; j < rhs.first.size(); ++j) {
66 if (lhs.first[i]->History() == rhs.first[j]->History()) {
67 SPIEL_CHECK_FLOAT_EQ(lhs.second[i], rhs.second[j]);
68 break;
69 }
70 }
71 }
72}

74void CheckDistHasSameInfostate(const HistoryDistribution& dist,
75 const State& state, int player_id) {
76 for (int i = 0; i < dist.first.size(); ++i) {
77 if (dist.second[i] > 0) {
78 SPIEL_CHECK_EQ(dist.first[i]->InformationStateString(player_id),
79 state.InformationStateString(player_id));
80 }
81 }

84void LeducStateDistributionTest() {
85 std::shared_ptr<const Game> game = LoadGame("leduc_poker");
86 std::unique_ptr<State> state = game->NewInitialState();
87 TabularPolicy uniform_policy = GetUniformPolicy(*game);
88 state->ApplyAction(0); // p0 card: jack of first suit
89 state->ApplyAction(1); // p1 card: queen of first suit
90 state->ApplyAction(1); // player 0 bet
91 std::string info_state_string = state->InformationStateString();
92 std::string state_history_string = state->HistoryString();
93 SPIEL_CHECK_EQ(state->CurrentPlayer(), 1);
94 HistoryDistribution dist = GetStateDistribution(*state, uniform_policy);
95 std::cerr << "Check infostates..." << std::endl;
96 CheckDistHasSameInfostate(dist, *state, /*player_id=*/1);
97
98 std::unique_ptr<HistoryDistribution> incremental_dist =
99 UpdateIncrementalStateDistribution(*state, uniform_policy,
100 /*player_id=*/1, nullptr);
101 std::cerr << "Comparing dists 1..." << std::endl;
102 SPIEL_CHECK_TRUE(incremental_dist);
103 CompareDists(dist, *incremental_dist);
104 CompareDists(dist, *CloneBeliefs(dist));
105 std::cerr << "Check infostates2..." << std::endl;
106 CheckDistHasSameInfostate(*incremental_dist, *state, /*player_id=*/1);
107
108 std::vector<double> correct_distribution(5, 0.2);
109 SPIEL_CHECK_EQ(dist.first.size(), 5);
110 SPIEL_CHECK_EQ(dist.second, correct_distribution);
111
112 // Check that none of the states are equal, that one of them is equal to the
113 // state used to generate the distribution, and that they are all equally
114 // likely with probability 0.2.
115 int state_matches = 0;
116 for (int i = 0; i < dist.first.size(); ++i) {
117 SPIEL_CHECK_EQ(dist.first[i]->InformationStateString(), info_state_string);
118 if (dist.first[i]->HistoryString() == state_history_string) {
119 state_matches++;
120 }
121 for (int j = i + 1; j < dist.first.size(); ++j) {
122 SPIEL_CHECK_NE(dist.first[i]->HistoryString(),
123 dist.first[j]->HistoryString());
124 }
125 }
126 SPIEL_CHECK_EQ(state_matches, 1);
127
128 // Now, it's a chance node...
129 state->ApplyAction(state->LegalActions()[0]);
130 incremental_dist = UpdateIncrementalStateDistribution(
131 *state, uniform_policy,
132 /*player_id=*/1, std::move(incremental_dist));
133 std::cerr << "Check infostates2a..." << std::endl;
134 CheckDistHasSameInfostate(*incremental_dist, *state, /*player_id=*/1);
135 state->ApplyAction(state->LegalActions()[0]);
136 dist = GetStateDistribution(*state, uniform_policy);
137 incremental_dist = UpdateIncrementalStateDistribution(
138 *state, uniform_policy,
139 /*player_id=*/1, std::move(incremental_dist));
140 std::cerr << "Check infostates3..." << std::endl;
141 CheckDistHasSameInfostate(*incremental_dist, *state, /*player_id=*/1);
142
143 std::cerr << "Comparing dists 2..." << std::endl;
144 CompareDists(dist, *incremental_dist);
145 CompareDists(dist, *CloneBeliefs(dist));
146}

154void HUNLIncrementalTest() {
155 // universal_poker requires ACPC, which is an optional dependency.
156 // Skip this test if the game is not registered.
157 if (!IsGameRegistered(std::string(kHUNLGameString))) { return; }
158 std::shared_ptr<const Game> game = LoadGame(std::string(kHUNLGameString));
159 std::unique_ptr<State> state = game->NewInitialState();
160 state->ApplyAction(14); // p0 card: 5h
161 state->ApplyAction(46); // p0 card: Kh5h
162 state->ApplyAction(7); // p1 card: 3s
163 state->ApplyAction(19); // p1 cards: 6s3s
164 UniformPolicy uniform_policy;
165 std::cerr << "Checking first call..." << std::endl;
166 std::unique_ptr<HistoryDistribution> incremental_dist =
167 UpdateIncrementalStateDistribution(*state, uniform_policy,
168 /*player_id=*/0, /*previous=*/nullptr);
169 CheckDistHasSameInfostate(*incremental_dist, *state, /*player_id=*/0);
170 std::cerr << "First call passed!" << std::endl;
171 state->ApplyAction(1); // p0 bets pot.
172 incremental_dist = UpdateIncrementalStateDistribution(
173 *state, uniform_policy, /*player_id=*/0, std::move(incremental_dist));
174 CheckDistHasSameInfostate(*incremental_dist, *state, /*player_id=*/0);
175}

177void LeducRegressionTest() {
178 std::shared_ptr<const Game> game = LoadGame("leduc_poker");
179 std::unique_ptr<State> state = game->NewInitialState();
180 UniformPolicy opponent_policy;
181 const int player_id = 1;
182 std::unique_ptr<HistoryDistribution> dist;
183 for (const Action action : {0, 5, 1, 2, 1, 4}) {
184 if (state->CurrentPlayer() == player_id) {
185 dist = UpdateIncrementalStateDistribution(*state, opponent_policy,
186 player_id, std::move(dist));
187 algorithms::CheckBeliefs(*state, *dist, player_id);
188 }
189 state->ApplyAction(action);
190 }
191 dist = UpdateIncrementalStateDistribution(*state, opponent_policy, player_id,
192 std::move(dist));
193 algorithms::CheckBeliefs(*state, *dist, player_id);
194}

196void LeducRegressionTestPerPlayer(int player_id) {
197 std::shared_ptr<const Game> game = LoadGame("leduc_poker");
198 std::unique_ptr<State> state = game->NewInitialState();
199 UniformPolicy opponent_policy;
200 std::unique_ptr<HistoryDistribution> dist;
201
202 // The first two actions are chance actions, then both players call. This was
203 // found to cause CheckBeliefs to fail previously, so we add a test verifying
204 // that doesn't happen.
205 for (const Action action : {4, 0, 2, 2}) {
206 if (state->CurrentPlayer() == player_id) {
207 dist = UpdateIncrementalStateDistribution(*state, opponent_policy,
208 player_id, std::move(dist));
209 algorithms::CheckBeliefs(*state, *dist, player_id);
210 }
211 state->ApplyAction(action);
212 }
213 dist = UpdateIncrementalStateDistribution(*state, opponent_policy, player_id,
214 std::move(dist));
215 algorithms::CheckBeliefs(*state, *dist, player_id);
216}
