57// Action encoding (must be changed to support larger boards):
58// - 0 - 129 represents lines with direction [0, 1]
59// - 130 - 259 represents lines with direction [1, 0]
60// - 260 - 359 represents lines with direction [1, -1]
61// - 360 - 459 represents lines with direction [1, 1]
62Line::Line(Action action) {
63 int row;
64 int base;
65 Point point1;
66 Point point2;
67 if (action >= 0 && action <= 129) {
68 // [0, 1]
69 row = action / 10;
70 point1 = Point(row, action - row * 10);
71 point2 = Point(row, (action - row * 10) + 3);
72 } else if (action >= 130 && action <= 259) {
73 // [1, 0]
74 base = action - 130;
75 row = (base) / 13;
76 point1 = Point(row, base - row * 13);
77 point2 = Point(row + 3, (base - row * 13));
78 } else if (action >= 260 && action <= 359) {
79 // [1, -1]
80 base = action - 260;
81 row = (base) / 10;
82 point1 = Point(row, base - row * 10);
83 point2 = Point(row + 3, (base - row * 10) + 3);
84 } else if (action >= 360 && action <= 459) {
85 // [1, 1]
86 base = action - 360;
87 row = (base) / 10;
88 point1 = Point(row + 3, base - row * 10);
89 point2 = Point(row, (base - row * 10) + 3);
90 } else {
91 SpielFatalError("action provided does not correspond with a move");
92 }
93 Init(point1, point2);
94}

96void Line::Init(Point point1, Point point2) {
97 if (point1 < point2) {
98 endpoint1_ = point1;
99 endpoint2_ = point2;
100 } else {
101 endpoint1_ = point2;
102 endpoint2_ = point1;
103 }
104 // Categorize line in one of four directions ([0, 1], [1, 1], [1, -1], [1,
105 // 0]).
106 direction_[0] = static_cast<int>((endpoint2_.x - endpoint1_.x) / 3);
107 direction_[1] = static_cast<int>((endpoint2_.y - endpoint1_.y) / 3);
108 // Get all points in line (beyond the two initial endpoints) and sort.
109 for (int i = 0; i < 4; i++) {
110 line_points_.emplace_back(endpoint1_.x + i * direction_[0],
111 endpoint1_.y + i * direction_[1]);
112 }
113 std::sort(line_points_.begin(), line_points_.end());
114}

116bool Line::CheckOverlap(Line l) {
117 // Only check for overlapping points for lines in the same direction.
118 if (direction_ != l.GetDirection()) {
119 return false;
120 }
121 // Check if it's the same line.
122 if ((endpoint1_ == l.GetEndpoints()[0]) &&
123 (endpoint2_ == l.GetEndpoints()[1])) {
124 return false;
125 }
126 // Check for overlapping points between the two lines.
127 std::vector<Point> intersect = {};
128 std::vector<Point> l_points = l.GetAllPoints();
129 std::set_intersection(l_points.begin(), l_points.end(), line_points_.begin(),
130 line_points_.end(), std::back_inserter(intersect));
131 if (!intersect.empty()) { // Line is overlapping if intersection.size() >=1
132 // in 4D version.
133 return true;
134 }
135 return false;
136}

143// Getters
144Action Line::GetAction() {
145 int dirCode;
146 if ((direction_[0] == 0) && (direction_[1] == 1)) {
147 dirCode = 1;
148 } else if ((direction_[0] == 1) && (direction_[1] == 0)) {
149 dirCode = 2;
150 } else if ((direction_[0] == 1) && (direction_[1] == 1)) {
151 dirCode = 3;
152 } else {
153 dirCode = 4;
154 }
155 // Get action encoding from line endpoints
156 switch (dirCode) {
157 // [0, 1] 0 ... 129
158 case 1:
159 return endpoint1_.x * 10 + endpoint1_.y;
160
161 // [1, 0] 130 ... 259
162 case 2:
163 return endpoint1_.x * 13 + endpoint1_.y + 130;
164
165 // [1, 1] 260 ... 359
166 case 3:
167 return endpoint1_.x * 10 + endpoint1_.y + 260;
168
169 // [1, -1] 360 ... 459
170 case 4:
171 return (endpoint2_.x - 3) * 10 + endpoint2_.y + 360;
172
173 default:
174 SpielFatalError(absl::StrCat("Unhandled case in Line::GetAction()",
175 ", dirCode = ", dirCode));
176 }
177}

192void MorpionState::DoApplyAction(Action move) {
193 Line newMove = *action_map_.at(move);
194 Point newPoint;
195 int pos;
196 for (Point p : newMove.GetAllPoints()) {
197 pos = p.y + (p.x * kNumRows);
198 if (board_[pos] == 0) {
199 board_[pos] = 1;
200 newPoint = p;
201 break;
202 }
203 }
204 move_history_.emplace_back(newMove, newPoint);
205 num_moves_ += 1;
206 current_returns_ += 1;
207}

229MorpionState::MorpionState(std::shared_ptr<const Game> game) : State(game) {
230 // Initialize 4D starting points and find all possible lines on the board
231 for (int i = 0; i < kNumRows; i++) {
232 for (int j = 0; j < kNumCols; j++) {
233 // Initialize starting points on board
234 if ((i == 3 || i == 9) && j > 4 && j < 8) {
235 board_[j + (i * kNumRows)] = 1;
236 }
237 if ((i == 4 || i == 8) && (j == 5 || j == 7)) {
238 board_[j + (i * kNumRows)] = 1;
239 }
240 if ((i == 5 || i == 7) && ((j > 2 && j < 6) || (j > 6 && j < 10))) {
241 board_[j + (i * kNumRows)] = 1;
242 }
243 if (i == 6 && ((j == 3) || (j == 9))) {
244 board_[j + (i * kNumRows)] = 1;
245 }
246 // Get all possible lines on board (460)
247 if (j + 3 < kNumCols) {
248 all_lines_.emplace_back(Point(i, j), Point(i, j + 3));
249 }
250 if ((j + 3 < kNumCols) && (i + 3 < kNumRows)) {
251 all_lines_.emplace_back(Point(i, j), Point(i + 3, j + 3));
252 }
253 if (i + 3 < kNumRows) {
254 all_lines_.emplace_back(Point(i, j), Point(i + 3, j));
255 }
256 if ((j >= 3) && (i + 3 < kNumRows)) {
257 all_lines_.emplace_back(Point(i, j), Point(i + 3, j - 3));
258 }
259 }
260 }
261 // For each line, store in a map of action # -> line object.
262 for (Line& line : all_lines_) {
263 action_map_[line.GetAction()] = &line;
264 }
265}

267// Generate all valid lines / moves in current board state.
268void MorpionState::getAllValidMoves() const {
269 current_valid_moves_.clear();
270 for (Line l : all_lines_) {
271 // Check that exactly one point is empty.
272 int count = 0;
273 for (Point p : l.GetAllPoints()) {
274 if (board_[p.y + (p.x * kNumRows)] == 1) {
275 count++;
276 }
277 }
278 if (count != 3) {
279 continue;
280 }
281 // Check that line does not overlap any existing moves / lines.
282 bool overlaps = false;
283 for (const std::pair<Line, Point>& m : move_history_) {
284 overlaps = l.CheckOverlap(m.first);
285 if (overlaps) {
286 break;
287 }
288 }
289 if (overlaps) {
290 continue;
291 }
292 current_valid_moves_.push_back(l);
293 }
294}

334void MorpionState::UndoAction(Player player, Action move) {
335 std::pair<Line, Point> last_move = move_history_.back();
336 board_[last_move.second.x * kNumRows + last_move.second.y] = 0;
337 move_history_.pop_back();
338 num_moves_ -= 1;
339 history_.pop_back();
340 --move_number_;
341}
