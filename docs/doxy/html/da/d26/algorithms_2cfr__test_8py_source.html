38 def test__update_current_policy(self):
39 game = pyspiel.load_game("kuhn_poker")
40 tabular_policy = policy.TabularPolicy(game)
41
42 cumulative_regrets = np.arange(0, 12 * 2).reshape((12, 2))
43 expected_policy = cumulative_regrets / np.sum(
44 cumulative_regrets, axis=-1, keepdims=True)
45 nodes_indices = {
46 u"0": 0,
47 u"0pb": 1,
48 u"1": 2,
49 u"1pb": 3,
50 u"2": 4,
51 u"2pb": 5,
52 u"1p": 6,
53 u"1b": 7,
54 u"2p": 8,
55 u"2b": 9,
56 u"0p": 10,
57 u"0b": 11,
58 }
59 # pylint: disable=g-complex-comprehension
60 info_state_nodes = {
61 key: cfr._InfoStateNode(
62 legal_actions=[0, 1],
63 index_in_tabular_policy=None,
64 cumulative_regret=dict(enumerate(cumulative_regrets[index])),
65 cumulative_policy=None) for key, index in nodes_indices.items()
66 }
67 # pylint: enable=g-complex-comprehension
68
69 cfr._update_current_policy(tabular_policy, info_state_nodes)

132 def test_cfr_cce_ce_dist_goofspiel(self):
133 """Copy of the TestCCEDistCFRGoofSpiel in corr_dist_test.cc."""
134 game = pyspiel.load_game(
135 "turn_based_simultaneous_game(game=goofspiel(num_cards=3,points_order="
136 "descending,returns_type=total_points))")
137 for num_iterations in [1, 10, 100]:
138 policies = []
139 cfr_solver = cfr.CFRSolver(game)
140 for _ in range(num_iterations):
141 cfr_solver.evaluate_and_update_policy()
142 policies.append(
143 policy.python_policy_to_pyspiel_policy(cfr_solver.current_policy()))
144 mu = pyspiel.uniform_correlation_device(policies)
145 cce_dist_info = pyspiel.cce_dist(game, mu)
146 print("goofspiel, cce test num_iters: {}, cce_dist: {}, per player: {}"
147 .format(num_iterations, cce_dist_info.dist_value,
148 cce_dist_info.deviation_incentives))
149 # Try converting one of the BR policies:
150 _ = policy.pyspiel_policy_to_python_policy(
151 game, cce_dist_info.best_response_policies[0])
152
153 # Assemble the same correlation device manually, just as an example for
154 # how to do non-uniform distributions of them and to test the python
155 # bindings for lists of tuples works properly
156 uniform_prob = 1.0 / len(policies)
157 mu2 = [(uniform_prob, policy) for policy in policies]
158 cce_dist_info2 = pyspiel.cce_dist(game, mu2)
159 self.assertAlmostEqual(cce_dist_info2.dist_value,
160 sum(cce_dist_info.deviation_incentives))
161 # Test the CEDist function too, why not. Disable the exact one, as it
162 # takes too long for a test.
163 # ce_dist_info = pyspiel.ce_dist(game, pyspiel.determinize_corr_dev(mu))
164 ce_dist_info = pyspiel.ce_dist(
165 game, pyspiel.sampled_determinize_corr_dev(mu, 100))
166 print("goofspiel, ce test num_iters: {}, ce_dist: {}, per player: {}"
167 .format(num_iterations, ce_dist_info.dist_value,
168 ce_dist_info.deviation_incentives))
169 print("number of conditional best responses per player:")
170 for p in range(game.num_players()):
171 print(" player {}, num: {}".format(
172 p, len(ce_dist_info.conditional_best_response_policies[p])))

194 @parameterized.parameters(list(itertools.product([False, True])))
195 def test_simultaneous_two_step_avg_1b_seq_in_kuhn_poker(
196 self, regret_matching_plus):
197 num_players = 2
198 game = pyspiel.load_game("kuhn_poker", {"players": num_players})
199 cfr_solver = cfr._CFRSolver(
200 game,
201 regret_matching_plus=regret_matching_plus,
202 linear_averaging=False,
203 alternating_updates=False)
204
205 def check_avg_policy_is_uniform_random():
206 avg_policy = cfr_solver.average_policy()
207 for player_info_states in avg_policy.states_per_player:
208 for info_state in player_info_states:
209 state_policy = avg_policy.policy_for_key(info_state)
210 np.testing.assert_allclose(state_policy, [1.0 / len(state_policy)] *
211 len(state_policy))
212
213 check_avg_policy_is_uniform_random()
214
215 cfr_solver.evaluate_and_update_policy()
216 check_avg_policy_is_uniform_random()
217
218 cfr_solver.evaluate_and_update_policy()
219
220 # The acting player in 1b is player 1 and they have not acted before, so
221 # the probability this player plays to this information state is 1, and
222 # the sequence probability of any action is just the probability of that
223 # action given the information state. On the first iteration, this
224 # probability is 0.5 for both actions. On the second iteration, the
225 # current policy is [0, 1], so the average cumulants should be
226 # [0.5, 1.5]. Normalizing this gives the average policy.
227 normalization = 0.5 + 0.5 + 1
228 np.testing.assert_allclose(cfr_solver.average_policy().policy_for_key("1b"),
229 [0.5 / normalization, (0.5 + 1) / normalization])

241 @parameterized.parameters([
242 (pyspiel.load_game("kuhn_poker"), pyspiel.CFRSolver, cfr.CFRSolver),

249 def test_cpp_algorithms_identical_to_python_algorithm(self, game, cpp_class,
250 python_class):
251 cpp_solver = cpp_class(game)
252 python_solver = python_class(game)
253
254 for _ in range(5):
255 cpp_solver.evaluate_and_update_policy()
256 python_solver.evaluate_and_update_policy()
257
258 cpp_avg_policy = cpp_solver.average_policy()
259 python_avg_policy = python_solver.average_policy()
260
261 # We do not compare the policy directly as we do not have an easy way to
262 # convert one to the other, so we use the exploitability as a proxy.
263 cpp_expl = pyspiel.nash_conv(game, cpp_avg_policy)
264 python_expl = exploitability.nash_conv(game, python_avg_policy)
265 self.assertAlmostEqual(cpp_expl, python_expl, places=10)
266 # Then we also check the CurrentPolicy, just to check it is giving the same
267 # results too
268 cpp_current_policy = cpp_solver.current_policy()
269 python_current_policy = python_solver.current_policy()
270 cpp_expl = pyspiel.nash_conv(game, cpp_current_policy)
271 python_expl = exploitability.nash_conv(game, python_current_policy)
272 self.assertAlmostEqual(cpp_expl, python_expl, places=10)
