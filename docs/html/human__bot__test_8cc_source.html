31 Action GetActionFromString(const State &state,
32  const std::string &action_string) {
33  for (Action action : state.LegalActions(state.CurrentPlayer())) {
34  if (action_string == state.ActionToString(action)) {
35  return action;
36  }
37  }
38  return kInvalidAction;
