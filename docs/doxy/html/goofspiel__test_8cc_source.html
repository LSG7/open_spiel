62void EgocentricViewOfSymmetricActions() {
63 GameParameters params;
64 params["imp_info"] = GameParameter(true);
65 params["egocentric"] = GameParameter(true);
66 params["num_cards"] = GameParameter(4);
67 params["players"] = GameParameter(3);
68 params["points_order"] = GameParameter(std::string("descending"));
69 std::shared_ptr<const Game> game = LoadGame("goofspiel", params);
70
71 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
72
73 // Three action sequences each played by one player.
74 std::vector<Action> seq1{3, 2, 0 /*, 1 */};
75 std::vector<Action> seq2{0, 1, 2 /*, 3 */};
76 std::vector<Action> seq3{2, 3, 1 /*, 0 */};
77
78 // Accumulate info state histories form the perspective of `seq1` when playing
79 // as one of the three players.
80 std::vector<std::vector<std::vector<float>>> info_state_histories(
81 game->NumPlayers());
82 for (int as_player = 0; as_player < game->NumPlayers(); as_player++) {
83 for (int t = 0; t < game->MaxGameLength() - 1; t++) {
84 std::vector<Action> joint_actions(game->NumPlayers(), -1);
85 joint_actions[as_player] = seq1[t];
86 joint_actions[(as_player + 1) % game->NumPlayers()] = seq2[t];
87 joint_actions[(as_player + 2) % game->NumPlayers()] = seq3[t];
88 state->ApplyActions(std::move(joint_actions));
89 auto info_state = state->InformationStateTensor(as_player);
90 info_state_histories[as_player].push_back(std::move(info_state));
91 }
92 state = game->NewInitialState();
93 }
94
95 // Verify that the observations remain identical regardless of which player
96 // `seq1` was executed for.
97 SPIEL_CHECK_EQ(info_state_histories.size(), game->NumPlayers());
98 SPIEL_CHECK_EQ(info_state_histories[0], info_state_histories[1]);
99 SPIEL_CHECK_EQ(info_state_histories[1], info_state_histories[2]);
100}
