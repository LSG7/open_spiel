36def _memoize_method(key_fn=lambda x: x):
37 """Memoize a single-arg instance method using an on-object cache."""
38
39 def memoizer(method):
40 cache_name = "cache_" + method.__name__
41
42 def wrap(self, arg):
43 key = key_fn(arg)
44 cache = vars(self).setdefault(cache_name, {})
45 if key not in cache:
46 cache[key] = method(self, arg)
47 return cache[key]
48
49 return wrap
50
51 return memoizer

54def compute_states_and_info_states_if_none(game,
55 all_states=None,
56 state_to_information_state=None):
57 """Returns all_states and/or state_to_information_state for the game.
58
59 To recompute everything, pass in None for both all_states and
60 state_to_information_state. Otherwise, this function will use the passed in
61 values to reconstruct either of them.
62
63 Args:
64 game: The open_spiel game.
65 all_states: The result of calling get_all_states.get_all_states. Cached for
66 improved performance.
67 state_to_information_state: A dict mapping state.history_str() to
68 state.information_state for every state in the game. Cached for improved
69 performance.
70 """
71 if all_states is None:
72 all_states = get_all_states.get_all_states(
73 game,
74 depth_limit=-1,
75 include_terminals=False,
76 include_chance_states=False)
77
78 if state_to_information_state is None:
79 state_to_information_state = {
80 state: all_states[state].information_state_string()
81 for state in all_states
82 }
83
84 return all_states, state_to_information_state

87class BestResponsePolicy(openspiel_policy.Policy):
88 """Computes the best response to a specified strategy."""
89
90 def __init__(self,
91 game,
92 player_id,
93 policy,
94 root_state=None,
95 cut_threshold=0.0):
96 """Initializes the best-response calculation.
97
98 Args:
99 game: The game to analyze.
100 player_id: The player id of the best-responder.
101 policy: A `policy.Policy` object.
102 root_state: The state of the game at which to start analysis. If `None`,
103 the game root state is used.
104 cut_threshold: The probability to cut when calculating the value.
105 Increasing this value will trade off accuracy for speed.
106 """
107 self._num_players = game.num_players()
108 self._player_id = player_id
109 self._policy = policy
110 if root_state is None:
111 root_state = game.new_initial_state()
112 self._root_state = root_state
113 self.infosets = self.info_sets(root_state)
114
115 self._cut_threshold = cut_threshold
116
117 def info_sets(self, state):
118 """Returns a dict of infostatekey to list of (state, cf_probability)."""
119 infosets = collections.defaultdict(list)
120 for s, p in self.decision_nodes(state):
121 infosets[s.information_state_string(self._player_id)].append((s, p))
122 return dict(infosets)
123
124 def decision_nodes(self, parent_state):
125 """Yields a (state, cf_prob) pair for each descendant decision node."""
126 if not parent_state.is_terminal():
127 if (parent_state.current_player() == self._player_id or
128 parent_state.is_simultaneous_node()):
129 yield (parent_state, 1.0)
130 for action, p_action in self.transitions(parent_state):
131 for state, p_state in self.decision_nodes(
132 openspiel_policy.child(parent_state, action)):
133 yield (state, p_state * p_action)
134
135 def joint_action_probabilities_counterfactual(self, state):
136 """Get list of action, probability tuples for simultaneous node.
137
138 Counterfactual reach probabilities exclude the best-responder's actions,
139 the sum of the probabilities is equal to the number of actions of the
140 player _player_id.
141 Args:
142 state: the current state of the game.
143
144 Returns:
145 list of action, probability tuples. An action is a tuple of individual
146 actions for each player of the game.
147 """
148 actions_per_player, probs_per_player = (
149 openspiel_policy.joint_action_probabilities_aux(state, self._policy))
150 probs_per_player[self._player_id] = [
151 1.0 for _ in probs_per_player[self._player_id]
152 ]
153 return [(list(actions), np.prod(probs)) for actions, probs in zip(
154 itertools.product(
155 *actions_per_player), itertools.product(*probs_per_player))]
156
157 def transitions(self, state):
158 """Returns a list of (action, cf_prob) pairs from the specified state."""
159 if state.current_player() == self._player_id:
160 # Counterfactual reach probabilities exclude the best-responder's actions,
161 # hence return probability 1.0 for every action.
162 return [(action, 1.0) for action in state.legal_actions()]
163 elif state.is_chance_node():
164 return state.chance_outcomes()
165 elif state.is_simultaneous_node():
166 return self.joint_action_probabilities_counterfactual(state)
167 else:
168 return list(self._policy.action_probabilities(state).items())
169
170 @_memoize_method(key_fn=lambda state: state.history_str())
171 def value(self, state):
172 """Returns the value of the specified state to the best-responder."""
173 if state.is_terminal():
174 return state.player_return(self._player_id)
175 elif (state.current_player() == self._player_id or
176 state.is_simultaneous_node()):
177 action = self.best_response_action(
178 state.information_state_string(self._player_id))
179 return self.q_value(state, action)
180 else:
181 return sum(p * self.q_value(state, a)
182 for a, p in self.transitions(state)
183 if p > self._cut_threshold)
184
185 def q_value(self, state, action):
186 """Returns the value of the (state, action) to the best-responder."""
187 if state.is_simultaneous_node():
188
189 def q_value_sim(sim_state, sim_actions):
190 child = sim_state.clone()
191 # change action of _player_id
192 sim_actions[self._player_id] = action
193 child.apply_actions(sim_actions)
194 return self.value(child)
195
196 actions, probabilities = zip(*self.transitions(state))
197 return sum(p * q_value_sim(state, a)
198 for a, p in zip(actions, probabilities / sum(probabilities))
199 if p > self._cut_threshold)
200 else:
201 return self.value(state.child(action))
202
203 @_memoize_method()
204 def best_response_action(self, infostate):
205 """Returns the best response for this information state."""
206 infoset = self.infosets[infostate]
207 # Get actions from the first (state, cf_prob) pair in the infoset list.
208 # Return the best action by counterfactual-reach-weighted state-value.
209 return max(
210 infoset[0][0].legal_actions(self._player_id),
211 key=lambda a: sum(cf_p * self.q_value(s, a) for s, cf_p in infoset))
212
213 def action_probabilities(self, state, player_id=None):
214 """Returns the policy for a player in a state.
215
216 Args:
217 state: A `pyspiel.State` object.
218 player_id: Optional, the player id for whom we want an action. Optional
219 unless this is a simultaneous state at which multiple players can act.
220
221 Returns:
222 A `dict` of `{action: probability}` for the specified player in the
223 supplied state.
224 """
225 if player_id is None:
226 if state.is_simultaneous_node():
227 player_id = self._player_id
228 else:
229 player_id = state.current_player()
230 return {
231 self.best_response_action(state.information_state_string(player_id)): 1
232 }

235class CPPBestResponsePolicy(openspiel_policy.Policy):
236 """Computes best response action_probabilities using open_spiel's C++ backend.
237
238 May have better performance than best_response.py for large games.
239 """
240
241 def __init__(self,
242 game,
243 best_responder_id,
244 policy,
245 all_states=None,
246 state_to_information_state=None,
247 best_response_processor=None,
248 cut_threshold=0.0):
249 """Constructor.
250
251 Args:
252 game: The game to analyze.
253 best_responder_id: The player id of the best-responder.
254 policy: A policy.Policy object representing the joint policy, taking a
255 state and returning a list of (action, probability) pairs. This could be
256 aggr_policy, for instance.
257 all_states: The result of calling get_all_states.get_all_states. Cached
258 for improved performance.
259 state_to_information_state: A dict mapping state.history_str to
260 state.information_state for every state in the game. Cached for improved
261 performance.
262 best_response_processor: A TabularBestResponse object, used for processing
263 the best response actions.
264 cut_threshold: The probability to cut when calculating the value.
265 Increasing this value will trade off accuracy for speed.
266 """
267 (self.all_states, self.state_to_information_state) = (
268 compute_states_and_info_states_if_none(game, all_states,
269 state_to_information_state))
270
271 policy_to_dict = policy_utils.policy_to_dict(
272 policy, game, self.all_states, self.state_to_information_state)
273
274 # pylint: disable=g-complex-comprehension
275 # Cache TabularBestResponse for players, due to their costly construction
276 # TODO(b/140426861): Use a single best-responder once the code supports
277 # multiple player ids.
278 if not best_response_processor:
279 best_response_processor = pyspiel.TabularBestResponse(
280 game, best_responder_id, policy_to_dict)
281
282 self._policy = policy
283 self.game = game
284 self.best_responder_id = best_responder_id
285 self.tabular_best_response_map = (
286 best_response_processor.get_best_response_actions())
287
288 self._cut_threshold = cut_threshold
289
290 def decision_nodes(self, parent_state):
291 """Yields a (state, cf_prob) pair for each descendant decision node."""
292 if not parent_state.is_terminal():
293 if parent_state.current_player() == self.best_responder_id:
294 yield (parent_state, 1.0)
295 for action, p_action in self.transitions(parent_state):
296 for state, p_state in self.decision_nodes(parent_state.child(action)):
297 yield (state, p_state * p_action)
298
299 def transitions(self, state):
300 """Returns a list of (action, cf_prob) pairs from the specified state."""
301 if state.current_player() == self.best_responder_id:
302 # Counterfactual reach probabilities exclude the best-responder's actions,
303 # hence return probability 1.0 for every action.
304 return [(action, 1.0) for action in state.legal_actions()]
305 elif state.is_chance_node():
306 return state.chance_outcomes()
307 else:
308 return list(self._policy.action_probabilities(state).items())
309
310 @_memoize_method(key_fn=lambda state: state.history_str())
311 def value(self, state):
312 """Returns the value of the specified state to the best-responder."""
313 if state.is_terminal():
314 return state.player_return(self.best_responder_id)
315 elif state.current_player() == self.best_responder_id:
316 action = self.best_response_action(
317 state.information_state_string(self.best_responder_id))
318 return self.q_value(state, action)
319 else:
320 return sum(p * self.q_value(state, a)
321 for a, p in self.transitions(state)
322 if p > self._cut_threshold)
323
324 def q_value(self, state, action):
325 """Returns the value of the (state, action) to the best-responder."""
326 return self.value(state.child(action))
327
328 @_memoize_method()
329 def best_response_action(self, infostate):
330 """Returns the best response for this information state."""
331 action = self.tabular_best_response_map[infostate]
332 return action

334 def action_probabilities(self, state, player_id=None):
335 """Returns the policy for a player in a state.
336
337 Args:
338 state: A `pyspiel.State` object.
339 player_id: Optional, the player id for whom we want an action. Optional
340 unless this is a simultabeous state at which multiple players can act.
341
342 Returns:
343 A `dict` of `{action: probability}` for the specified player in the
344 supplied state.
345 """
346 # Send the best-response probabilities for the best-responder
347 if state.current_player() == self.best_responder_id:
348 probs = {action_id: 0. for action_id in state.legal_actions()}
349 info_state = self.state_to_information_state[state.history_str()]
350 probs[self.tabular_best_response_map[info_state]] = 1.
351 return probs
352
353 # Send the default probabilities for all other players
354 return self._policy.action_probabilities(state, player_id)

360 def copy_with_noise(self, alpha=0.0, beta=0.0):
361 """Copies this policy and adds noise, making it a Noisy Best Response.
362
363 The policy's new probabilities P' on each state s become
364 P'(s) = alpha * epsilon + (1-alpha) * P(s)
365
366 With P the former policy's probabilities, and epsilon ~ Softmax(beta *
367 Uniform)
368
369 Args:
370 alpha: First mixture component
371 beta: Softmax 1/temperature component
372
373 Returns:
374 Noisy copy of best response.
375 """
376 return noisy_policy.NoisyPolicy(self, alpha, beta, self.all_states)
