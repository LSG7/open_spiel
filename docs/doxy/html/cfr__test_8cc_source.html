36void CheckNashKuhnPoker(const Game& game, const Policy& policy) {
37 const std::vector<double> game_value =
38 ExpectedReturns(*game.NewInitialState(), policy, -1);
39
40 // 1/18 is the Nash value. See https://en.wikipedia.org/wiki/Kuhn_poker
41 constexpr float nash_value = 1.0 / 18.0;
42 constexpr float eps = 1e-3;
43
44 SPIEL_CHECK_EQ(2, game_value.size());
45 SPIEL_CHECK_FLOAT_NEAR((float)game_value[0], -nash_value, eps);
46 SPIEL_CHECK_FLOAT_NEAR((float)game_value[1], nash_value, eps);
47}

49void CheckExploitabilityKuhnPoker(const Game& game, const Policy& policy) {
50 SPIEL_CHECK_LE(Exploitability(game, policy), 0.05);
51}

105void CFRTest_KuhnPokerRunsWithThreePlayers(bool linear_averaging,
106 bool regret_matching_plus,
107 bool alternating_updates) {
108 int num_players = 3;
109 std::shared_ptr<const Game> game =
110 LoadGame("kuhn_poker", {{"players", GameParameter(num_players)}});
111 CFRSolverBase solver(*game, alternating_updates, linear_averaging,
112 regret_matching_plus);
113 for (int i = 0; i < 10; i++) {
114 solver.EvaluateAndUpdatePolicy();
115 }
116 std::shared_ptr<Policy> average_policy = solver.AveragePolicy();
117 // Value upper-bounds inspired by Fig 2 of (Srinivasan et al., Actor-Critic
118 // Policy Optimization in Partially Observable Multiagent Environments, 2018)
119 // https://arxiv.org/abs/1810.09026
120 SPIEL_CHECK_LE(NashConv(*game, *average_policy), 1);
121}

144void CFRTest_OneShotGameTest(int iterations, std::string one_shot_game,
145 double nashconv_upper_bound) {
146 // Note: this is a 3-player general sum game with a unique uniform mixed
147 // strategy. However, CFR is not guaranteed to converge, and indeed fails to,
148 // just like was shown for fictitious play.
149 std::cout << one_shot_game << " convergence test." << std::endl;
150 std::shared_ptr<const Game> game = LoadGameAsTurnBased(one_shot_game);
151 CFRSolverBase solver(*game,
152 /*alternating_updates=*/true,
153 /*linear_averaging=*/false,
154 /*regret_matching_plus=*/false);
155 double nash_conv = 0;
156 for (int i = 0; i < iterations; i++) {
157 solver.EvaluateAndUpdatePolicy();
158 if (i % 10 == 0) {
159 std::shared_ptr<Policy> average_policy = solver.AveragePolicy();
160 nash_conv = NashConv(*game, *average_policy);
161 std::cout << "iter " << i << ", nashconv = " << nash_conv << std::endl;
162
163 if (game->GetType().utility == GameType::Utility::kConstantSum ||
164 game->GetType().utility == GameType::Utility::kZeroSum) {
165 double expl = Exploitability(*game, *average_policy);
166 SPIEL_CHECK_FLOAT_NEAR(expl, nash_conv / game->NumPlayers(), 1e-10);
167 }
168 }
169 }
170 SPIEL_CHECK_LE(nash_conv, nashconv_upper_bound);
171}

191void CFRTest_InfoStateValuesTableSerialization() {
192 // Check empty
193 CFRInfoStateValuesTable info_state_values_table = {};
194 std::string serialized0 = "";
195 SerializeCFRInfoStateValuesTable(info_state_values_table, &serialized0, -1);
196 CFRInfoStateValuesTable deserialized0;
197 DeserializeCFRInfoStateValuesTable(serialized0, &deserialized0);
198 SPIEL_CHECK_TRUE(deserialized0.empty());
199
200 // Check non-empty
201 info_state_values_table = {
202 {"", CFRInfoStateValues({0}, 1.0)},
203 {"0:0,0;0", CFRInfoStateValues({0, 1, 2}, 0.1)},
204 {"<->\n<->", CFRInfoStateValues({0, 1, 2}, 0.1)},
205 {"1:1,1;1", CFRInfoStateValues({0, 1, 2, 3}, 0.2)}};
206 std::string serialized1 = "";
207 SerializeCFRInfoStateValuesTable(info_state_values_table, &serialized1, -1);
208 CFRInfoStateValuesTable deserialized1;
209 DeserializeCFRInfoStateValuesTable(serialized1, &deserialized1);
210
211 SPIEL_CHECK_EQ(info_state_values_table.size(),
212 info_state_values_table.size());
213 for (const auto& [info_state, values] : info_state_values_table) {
214 for (int i = 0; i < values.legal_actions.size(); i++) {
215 SPIEL_CHECK_EQ(values.legal_actions.at(i),
216 deserialized1.at(info_state).legal_actions.at(i));
217 SPIEL_CHECK_FLOAT_NEAR(
218 values.cumulative_regrets.at(i),
219 deserialized1.at(info_state).cumulative_regrets.at(i), 1e-15);
220 SPIEL_CHECK_FLOAT_NEAR(
221 values.cumulative_policy.at(i),
222 deserialized1.at(info_state).cumulative_policy.at(i), 1e-15);
223 SPIEL_CHECK_FLOAT_NEAR(values.current_policy.at(i),
224 deserialized1.at(info_state).current_policy.at(i),
225 1e-15);
226 }
227 }
228}

230void CFRTest_CFRSolverSerialization() {
231 auto game = LoadGame("kuhn_poker");
232 CFRSolver solver = CFRSolver(*game);
233 double exploitability0 = Exploitability(*game, *solver.AveragePolicy());
234
235 for (int i = 0; i < 50; i++) {
236 solver.EvaluateAndUpdatePolicy();
237 }
238 double exploitability1 = Exploitability(*game, *solver.AveragePolicy());
239 SPIEL_CHECK_GT(exploitability0, exploitability1);
240
241 std::string serialized = solver.Serialize();
242 std::unique_ptr<CFRSolver> deserialized_solver =
243 DeserializeCFRSolver(serialized);
244 SPIEL_CHECK_EQ(solver.InfoStateValuesTable().size(),
245 deserialized_solver->InfoStateValuesTable().size());
246 double exploitability2 =
247 Exploitability(*game, *deserialized_solver->AveragePolicy());
248 SPIEL_CHECK_FLOAT_NEAR(exploitability1, exploitability2, 1e-15);
249
250 for (int i = 0; i < 50; i++) {
251 deserialized_solver->EvaluateAndUpdatePolicy();
252 }
253 double exploitability3 =
254 Exploitability(*game, *deserialized_solver->AveragePolicy());
255 SPIEL_CHECK_GT(exploitability2, exploitability3);
256}
