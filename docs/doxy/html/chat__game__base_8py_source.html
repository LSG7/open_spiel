199 def _names_from_validated_receiver(self, receiver: int, speaker: int
200 ) -> Tuple[Tuple[str, str, str], int]:
201 """Modify receiver if sending to self. Then return names of all roles.
202
203 Args:
204 receiver: integer action indicating receiver to send message to
205 speaker: integer representing current message sender
206 Returns:
207 names: tuple of strings, (speaker_name, receiver_name, others_names)
208 receiver: integer representing validated receiver
209 """
210 if (receiver >= self.get_game().num_players()
211 or speaker >= self.get_game().num_players()):
212 logging.info('Warning: rolling receiver/speaker to valid id.')
213 receiver = receiver % self.get_game().num_players()
214 speaker = speaker % self.get_game().num_players()
215 # overwrite speaking to self as speaking to all in header
216 receiver_name = ''
217 if receiver == speaker:
218 if len(self._names) > 2:
219 receiver_name = ALL_PLAYERS
220 receiver = -1
221 else:
222 receiver = (receiver + 1) % self.get_game().num_players()
223 speaker_name = ''
224 others = []
225 for idx, name in enumerate(self._names):
226 if idx == speaker:
227 speaker_name = name
228 elif idx == receiver:
229 receiver_name = name
230 elif receiver > -1:
231 others.append(name)
232 others_names = ', '.join(others)
233 names = (speaker_name, receiver_name, others_names)
234 return names, receiver

296 def action_to_prompt(self,
297 action: int,
298 seed: int,
299 header: header_utils.Header
300 ) -> Tuple[str, str]:
301 """Unravel action int to multidimensional action tuple and construct prompt.
302
303 Args:
304 action: int, the action taken in the game
305 seed: int, llm seed
306 header: header_utils.Header, used to format a prompt
307 Returns:
308 prompt: str, formatted prompt to feed the LLM to generate a new message
309 header_plain: str, the formatted header without any private info / actions
310 """
311 speaker = int(self._current_speaker)
312 action_dict = self.unravel_flat_action_to_dict(speaker, action)
313 receiver = action_dict['receiver']
314 opts = {**action_dict['action'], **action_dict['info']}
315
316 names, _ = self._names_from_validated_receiver(receiver, speaker)
317 speaker_name, receiver_name, others_names = names
318 header_plain = header.plain.format(sender=speaker_name,
319 receiver=receiver_name,
320 others=others_names)
321
322 header_w_opts = header.w_opts.format(sender=speaker_name,
323 receiver=receiver_name,
324 others=others_names,
325 **opts)
326 # provide header with opts to llm for response
327 logging.info('Generating message (speaker=%d:%s)...',
328 speaker,
329 speaker_name)
330
331 prompt = header.context + '\n\n' + self.dialogue_str + header_w_opts
332
333 return prompt, header_plain

335 def action_to_msg(self, action: int, seed: int) -> str:
336 """Unravel action int to multidimensional action tuple and construct msg.
337
338 Args:
339 action: int, the action taken in the game
340 seed: int, llm seed
341 Returns:
342 speaker_msg: str
343 """
344 header = self.get_game().header
345 prompt, header_plain = self.action_to_prompt(action, seed, header)

348 response = self.get_game().generate_response(
349 prompt=prompt,
350 seed=seed,
351 num_output_tokens=LLM_LENGTH_MESSAGE_TOKENS

354 logging.info('LLM response:\n%s', response)

356 first_special_char = text.first_special_char(
357 response, len(response), self.get_game().header.special_chars)
358 speaker_msg = header_plain + response[:first_special_char]

360 return speaker_msg

380 def compute_rewards(self, dialogue: str) -> np.ndarray:
381 """Compute rewards for each player from a given dialogue string.
382
383 Args:
384 dialogue: str, a single string with the entire dialogue thus far
385 Returns:
386 rewards: np.ndarray, len-num_players vector of floats
387 """
388 ct.set_color(logging_utils.GREEN)
389
390 rewards = np.zeros(self.get_game().num_players(), dtype=float)
391
392 if (not self.is_terminal() and
393 self.get_game().reward_type == pyspiel.GameType.RewardModel.TERMINAL):
394 return rewards
395
396 # gather private info to compute true underlying rewards
397 info_prefix = []
398 for player, name in enumerate(self._names):
399 extra_info_strs = [pi[player] for pi in self._private_info.values()]
400 info_prefix_p = [
401 f'{k}:\n{v}' for k, v in zip(self.get_game().header.info_keys,
402 extra_info_strs)
403 ]
404 info_prefix_p = name + '\n' + '\n'.join(info_prefix_p)
405 info_prefix.append(info_prefix_p)
406 info_prefix = '\n\n'.join(info_prefix)
407
408 # compute rewards
409 for player, name in enumerate(self._names):
410 player_payoffs = []
411 for p, payoff in enumerate(self.get_game().payoffs):
412 if payoff.obs_trans_prefix or payoff.obs_trans_postfix:
413 payoff_obs_prompt = (payoff.obs_trans_prefix +
414 dialogue +
415 payoff.obs_trans_postfix)
416 logging.info(ct.color('Scoring payoff (speaker=%d:%s)...'),
417 player, name)
418 logging.info(ct.color('LLM prompt:\n%s'), payoff_obs_prompt)
419 response = self.get_game().generate_response(
420 prompt=payoff_obs_prompt,
421 seed=DEFAULT_LLM_SEED,
422 num_output_tokens=LLM_LENGTH_PAYOFF_OBS_TOKENS
423 )
424 payoff_obs = response[:LLM_LENGTH_PAYOFF_OBS_CHARS]
425 else:
426 payoff_obs = dialogue
427 payoff_obs = info_prefix + '\n\n' + payoff_obs
428 query = self._build_payoff_query(payoff.query, payoff_obs, name)
429 logging.info(ct.color('Calculating payoff %d (player=%d:%s)...'),
430 p, player, name)
431 logging.info(ct.color('LLM prompt:\n%s'), query)
432 response = self.get_game().generate_response(
433 prompt=query,
434 seed=DEFAULT_LLM_SEED,
435 num_output_tokens=LLM_LENGTH_SCORE_TOKENS
436 )
437 logging.info(ct.color('LLM response:\n%s'), response)
438
439 logging.info(ct.color('Extracting payoff %d (player=%d:%s)...'),
440 p, player, name)
441 query = (f'Extract out the final value for {name} as a single ' +
442 'numeric value from the following payoff valuation. Do ' +
443 'NOT show your work:\n\n' +
444 f'{response}\n\nResult: ')
445 logging.info(ct.color('LLM prompt:\n%s'), query)
446 response = self.get_game().generate_response(
447 prompt=query,
448 seed=DEFAULT_LLM_SEED,
449 num_output_tokens=LLM_LENGTH_SCORE_TOKENS
450 )
451 logging.info(ct.color('LLM response:\n%s'), response)
452
453 player_payoff = 0 # payoff defaults to 0 if LLM parsing fails
454 if text.retrieve_numeric_block(response):
455 player_payoff = int(text.retrieve_numeric_block(response))
456 player_payoff = min(max(player_payoff, payoff.min), payoff.max)
457 else:
458 logging.warning(
459 ct.color('Payoff extraction from response failed:\n\n%s.'),
460 response)
461 logging.info(ct.color('Extracted integer payoff (%s): %d'),
462 name, player_payoff)
463 player_payoffs.append(player_payoff)
464 rewards[player] = self.get_game().aggregate_payoffs(player_payoffs)
465
466 ct.reset()
467
468 return rewards.astype(float)

683 state.obs[player].obs_trans_postfix)

705 player,
706 state.names[player],
