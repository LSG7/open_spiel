31 Action RoshamboBot::Step(const State& state) {
32  // Every step must synchronize histories between the OpenSpiel wrapper
33  // bot and the RoShamBo bot.
34  std::vector<Action> history = state.History();
35  if (history.empty()) {
36  SPIEL_CHECK_EQ(bot_->CurrentMatchLength(), 0);
37  } else {
38  const int throw_num = history.size() / 2;
39  SPIEL_CHECK_EQ(bot_->CurrentMatchLength() + 1, throw_num);
40  bot_->RecordTrial(history[((throw_num - 1) * 2) + player_id_],
41  history[((throw_num - 1) * 2) + opponent_id_]);
42  }
43  return bot_->GetAction();
44 }
