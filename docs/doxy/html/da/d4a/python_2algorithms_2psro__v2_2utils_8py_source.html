34def random_choice(outcomes, probabilities):
35 """Samples from discrete probability distribution.
36
37 `numpy.choice` does not seem optimized for repeated calls, this code
38 had higher performance.
39
40 Args:
41 outcomes: List of categorical outcomes.
42 probabilities: Discrete probability distribtuion as list of floats.
43
44 Returns:
45 Entry of `outcomes` sampled according to the distribution.
46 """
47 cumsum = np.cumsum(probabilities)
48 return outcomes[np.searchsorted(cumsum/cumsum[-1], random.random())]

51def sample_strategy(total_policies,
52 probabilities_of_playing_policies,
53 probs_are_marginal=True):
54 """Samples strategies given probabilities.
55
56 Uses independent sampling if probs_are_marginal, and joint sampling otherwise.
57
58 Args:
59 total_policies: if probs_are_marginal, this is a list, each element a list
60 of each player's policies. If not, this is a list of joint policies. In
61 both cases the policy orders must match that of
62 probabilities_of_playing_policies.
63 probabilities_of_playing_policies: if probs_are_marginal, this is a list,
64 with the k-th element also a list specifying the play probabilities of the
65 k-th player's policies. If not, this is a list of play probabilities of
66 the joint policies specified by total_policies.
67 probs_are_marginal: a boolean indicating if player-wise marginal
68 probabilities are provided in probabilities_of_playing_policies. If False,
69 then play_probabilities is assumed to specify joint distribution.
70
71 Returns:
72 sampled_policies: A list specifying a single sampled joint strategy.
73 """
74
75 if probs_are_marginal:
76 return sample_strategy_marginal(total_policies,
77 probabilities_of_playing_policies)
78 else:
79 return sample_strategy_joint(total_policies,
80 probabilities_of_playing_policies)

83def sample_strategy_marginal(total_policies, probabilities_of_playing_policies):
84 """Samples strategies given marginal probabilities.
85
86 Uses independent sampling if probs_are_marginal, and joint sampling otherwise.
87
88 Args:
89 total_policies: A list, each element a list of each player's policies.
90 probabilities_of_playing_policies: This is a list, with the k-th element
91 also a list specifying the play probabilities of the k-th player's
92 policies.
93
94 Returns:
95 sampled_policies: A list specifying a single sampled joint strategy.
96 """
97
98 num_players = len(total_policies)
99 sampled_policies = []
100 for k in range(num_players):
101 current_policies = total_policies[k]
102 current_probabilities = probabilities_of_playing_policies[k]
103 sampled_policy_k = random_choice(current_policies, current_probabilities)
104 sampled_policies.append(sampled_policy_k)
105 return sampled_policies

117def sample_strategy_joint(total_policies, probabilities_of_playing_policies):
118 """Samples strategies given joint probabilities.
119
120 Uses independent sampling if probs_are_marginal, and joint sampling otherwise.
121
122 Args:
123 total_policies: A list, each element a list of each player's policies.
124 probabilities_of_playing_policies: This is a list of play probabilities of
125 the joint policies specified by total_policies.
126
127 Returns:
128 sampled_policies: A list specifying a single sampled joint strategy.
129 """
130
131 sampled_index = sample_random_tensor_index(probabilities_of_playing_policies)
132 sampled_policies = []
133 for player in range(len(sampled_index)):
134 ind = sampled_index[player]
135 sampled_policies.append(total_policies[player][ind])
136 return sampled_policies

143def round_maintain_sum(x):
144 """Returns element-wise rounded version y of a vector x, with sum(x)==sum(y).
145
146 E.g., if x = array([3.37625333, 2.27920304, 4.34454364]), note sum(x) == 10.
147 However, naively doing y = np.round(x) yields sum(y) == 9. In this function,
148 however, the rounded counterpart y will have sum(y) == 10.
149
150 Args:
151 x: a vector.
152 """
153 y = np.floor(x)
154 sum_diff = round(sum(x)) - sum(y) # Difference of original vs. floored sum
155 indices = np.argsort(y - x)[:int(sum_diff)] # Indices with highest difference
156 y[indices] += 1 # Add the missing mass to the elements with the most missing
157 return y

160def get_alpharank_marginals(payoff_tables, pi):
161 """Returns marginal strategy rankings for each player given joint rankings pi.
162
163 Args:
164 payoff_tables: List of meta-game payoff tables for a K-player game, where
165 each table has dim [n_strategies_player_1 x ... x n_strategies_player_K].
166 These payoff tables may be asymmetric.
167 pi: The vector of joint rankings as computed by alpharank. Each element i
168 corresponds to a unique integer ID representing a given strategy profile,
169 with profile_to_id mappings provided by
170 alpharank_utils.get_id_from_strat_profile().
171
172 Returns:
173 pi_marginals: List of np.arrays of player-wise marginal strategy masses,
174 where the k-th player's np.array has shape [n_strategies_player_k].
175 """
176 num_populations = len(payoff_tables)
177
178 if num_populations == 1:
179 return pi
180 else:
181 num_strats_per_population = alpharank_utils.get_num_strats_per_population(
182 payoff_tables, payoffs_are_hpt_format=False)
183 num_profiles = alpharank_utils.get_num_profiles(num_strats_per_population)
184 pi_marginals = [np.zeros(n) for n in num_strats_per_population]
185 for i_strat in range(num_profiles):
186 strat_profile = (
187 alpharank_utils.get_strat_profile_from_id(num_strats_per_population,
188 i_strat))
189 for i_player in range(num_populations):
190 pi_marginals[i_player][strat_profile[i_player]] += pi[i_strat]
191 return pi_marginals

194def remove_epsilon_negative_probs(probs, epsilon=1e-9):
195 """Removes negative probabilities that occur due to precision errors."""
196 if len(probs[probs < 0]) > 0: # pylint: disable=g-explicit-length-test
197 # Ensures these negative probabilities aren't large in magnitude, as that is
198 # unexpected and likely not due to numerical precision issues
199 print("Probabilities received were: {}".format(probs[probs < 0]))
200 assert np.all(np.min(probs[probs < 0]) > -1.*epsilon), (
201 "Negative Probabilities received were: {}".format(probs[probs < 0]))
202
203 probs[probs < 0] = 0
204 probs = probs / np.sum(probs)
205 return probs

208def get_joint_strategy_from_marginals(probabilities):
209 """Returns a joint strategy tensor from a list of marginals.
210
211 Args:
212 probabilities: list of list of probabilities, one for each player.
213
214 Returns:
215 A joint strategy from a list of marginals.
216 """
217 probas = []
218 for i in range(len(probabilities)):
219 probas_shapes = [1] * len(probabilities)
220 probas_shapes[i] = -1
221 probas.append(np.array(probabilities[i]).reshape(probas_shapes))
222 return np.prod(probas)

225def alpharank_strategy(solver, return_joint=False, **unused_kwargs):
226 """Returns AlphaRank distribution on meta game matrix.
227
228 This method works for general games.
229
230 Args:
231 solver: GenPSROSolver instance.
232 return_joint: a boolean specifying whether to return player-wise
233 marginals.
234
235 Returns:
236 marginals: a list, specifying for each player the alpharank marginal
237 distributions on their strategies.
238 joint_distr: a list, specifying the joint alpharank distributions for all
239 strategy profiles.
240 """
241 meta_games = solver.get_meta_game()
242 meta_games = [np.asarray(x) for x in meta_games]
243
244 if solver.symmetric_game:
245 meta_games = [meta_games[0]]
246
247 # Get alpharank distribution via alpha-sweep
248 joint_distr = alpharank.sweep_pi_vs_epsilon(
249 meta_games)
250 joint_distr = remove_epsilon_negative_probs(joint_distr)
251
252 marginals = 2 * [joint_distr]
253 joint_distr = get_joint_strategy_from_marginals(marginals)
254 if return_joint:
255 return marginals, joint_distr
256 else:
257 return joint_distr
258
259 else:
260 joint_distr = alpharank.sweep_pi_vs_epsilon(meta_games)
261 joint_distr = remove_epsilon_negative_probs(joint_distr)
262
263 if return_joint:
264 marginals = get_alpharank_marginals(meta_games, joint_distr)
265 return marginals, joint_distr
266 else:
267 return joint_distr

277def get_joint_policies_from_id_list(payoff_tables, policies, profile_id_list):
278 """Returns a list of joint policies, given a list of integer IDs.
279
280 Args:
281 payoff_tables: List of payoff tables, one per player.
282 policies: A list of policies, one per player.
283 profile_id_list: list of integer IDs, each corresponding to a joint policy.
284 These integers correspond to those in get_strategy_profile_ids().
285
286 Returns:
287 selected_joint_policies: A list, with each element being a joint policy
288 instance (i.e., a list of policies, one per player).
289 """
290 num_strats_per_population = (
291 alpharank_utils.get_num_strats_per_population(
292 payoff_tables, payoffs_are_hpt_format=False))
293 np.testing.assert_array_equal(num_strats_per_population,
294 [len(p) for p in policies])
295 num_players = len(policies)
296
297 selected_joint_policies = []
298 for profile_id in profile_id_list:
299 # Compute the profile associated with the integer profile_id
300 policy_profile = alpharank_utils.get_strat_profile_from_id(
301 num_strats_per_population, profile_id)
302 # Append the joint policy corresponding to policy_profile
303 selected_joint_policies.append(
304 [policies[k][policy_profile[k]] for k in range(num_players)])
305 return selected_joint_policies

308def compute_states_and_info_states_if_none(game,
309 all_states=None,
310 state_to_information_state=None):
311 """Returns all_states and/or state_to_information_state for the game.
312
313 To recompute everything, pass in None for both all_states and
314 state_to_information_state. Otherwise, this function will use the passed in
315 values to reconstruct either of them.
316
317 Args:
318 game: The open_spiel game.
319 all_states: The result of calling get_all_states.get_all_states. Cached for
320 improved performance.
321 state_to_information_state: A dict mapping str(state) to
322 state.information_state for every state in the game. Cached for improved
323 performance.
324 """
325 if all_states is None:
326 all_states = get_all_states.get_all_states(
327 game,
328 depth_limit=-1,
329 include_terminals=False,
330 include_chance_states=False)
331
332 if state_to_information_state is None:
333 state_to_information_state = {
334 state: all_states[state].information_state_string()
335 for state in all_states
336 }
337
338 return all_states, state_to_information_state

341def aggregate_policies(game, total_policies, probabilities_of_playing_policies):
342 """Aggregate the players' policies.
343
344 Specifically, returns a single callable policy object that is
345 realization-equivalent to playing total_policies with
346 probabilities_of_playing_policies. I.e., aggr_policy is a joint policy that
347 can be called at any information state [via
348 action_probabilities(state, player_id)].
349
350 Args:
351 game: The open_spiel game.
352 total_policies: A list of list of all policy.Policy strategies used for
353 training, where the n-th entry of the main list is a list of policies
354 available to the n-th player.
355 probabilities_of_playing_policies: A list of arrays representing, per
356 player, the probabilities of playing each policy in total_policies for the
357 same player.
358
359 Returns:
360 A callable object representing the policy.
361 """
362 aggregator = policy_aggregator.PolicyAggregator(game)
363
364 return aggregator.aggregate(
365 range(len(probabilities_of_playing_policies)), total_policies,
366 probabilities_of_playing_policies)

369def marginal_to_joint(policies):
370 """Marginal policies to joint policies.
371
372 Args:
373 policies: List of list of policies, one list per player.
374
375 Returns:
376 Joint policies in the right order (np.reshape compatible).
377 """
378 shape = tuple([len(a) for a in policies])
379 num_players = len(shape)
380 total_length = np.prod(shape)
381 indexes = np.array(list(range(total_length)))
382 joint_indexes = np.unravel_index(indexes, shape)
383
384 joint_policies = []
385 for joint_index in zip(*joint_indexes):
386 joint_policies.append([
387 policies[player][joint_index[player]] for player in range(num_players)
388 ])
389 return joint_policies
