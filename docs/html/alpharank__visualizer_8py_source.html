180  def compute_and_draw_network(self):
181  """Computes the various node/edge connections of the graph and draws it."""
182 
183  if np.max(self.rhosrhos) < self.rho_mrho_m:
184  print("All node-to-node fixation probabilities (not including self-cycles"
185  " are lower than neutral. Thus, no graph will be drawn.")
186  return
187 
188  self.gg = nx.MultiDiGraph()
189  self.edge_labelsedge_labels = {}
190  self.edge_alphasedge_alphas = []
191  rho_max = np.max(self.rhosrhos / self.rho_mrho_m)
192  rho_m_alpha = 0.1 # Transparency of neutral selection edges
193 
194  for i in range(self.num_profilesnum_profiles):
195  for j in range(self.num_profilesnum_profiles):
196  # Do not draw edge if any node involved is skipped
197  if j not in self.nodes_to_skipnodes_to_skip and i not in self.nodes_to_skipnodes_to_skip:
198  rate = self.rhosrhos[i][j] / self.rho_mrho_m
199  # Draws edges when fixation from one strategy to another occurs (i.e.,
200  # rate > 1), or with fixation equal to neutral selection probability
201  # (i.e., rate == 1). This is consistent with visualizations used in
202  # finite-population literature.
203  if rate > 1:
204  # Compute alphas. Clip needed due to numerical precision.
205  alpha = np.clip(rho_m_alpha + (1 - rho_m_alpha) * rate / rho_max,
206  None, 1.)
207  self.gg.add_edge(i, j, weight=alpha, label="{:.01f}".format(rate))
208  self.edge_alphasedge_alphas.append(alpha)
209  elif np.isclose(rate, 1):
210  alpha = rho_m_alpha
211  self.gg.add_edge(i, j, weight=alpha, label="{:.01f}".format(rate))
212  self.edge_alphasedge_alphas.append(alpha)
213  # Label edges for non-self-loops with sufficient flowrate
214  if i != j and rate > 1:
215  edge_string = "$" + str(np.round(rate, decimals=2)) + "\\rho_m$"
216  else:
217  edge_string = ""
218  self.edge_labelsedge_labels[(i, j)] = edge_string
219 
220  # MultiDiGraph nodes are not ordered, so order the node colors accordingly
221  self.node_colorsnode_colors = [self.pipi[node] for node in self.gg.nodes()]
222 
223  self.cyclescycles = list(nx.simple_cycles(self.gg))
224  self.num_cyclesnum_cycles = len(self.cyclescycles)
225 
226  # Color the edges of cycles if user requested it
227  if self.i_cycle_to_showi_cycle_to_show >= 0:
228  all_cycle_edges = [
229  zip(nodes, (nodes[1:] + nodes[:1])) for nodes in self.cyclescycles
230  ]
231  cur_cycle_edges = all_cycle_edges[self.i_cycle_to_showi_cycle_to_show]
232  self.edge_colorsedge_colors = []
233  for u, v in self.gg.edges():
234  if (u, v) in cur_cycle_edges:
235  self.edge_colorsedge_colors.append([1., 0., 0.])
236  else:
237  self.edge_colorsedge_colors.append([1. - self.gg[u][v][0]["weight"]] * 3)
238  else:
239  self.edge_colorsedge_colors = [
240  [1. - self.gg[u][v][0]["weight"]] * 3 for u, v in self.gg.edges()
241  ]
242  self.edge_alphasedge_alphas = [self.gg[u][v][0]["weight"] for u, v in self.gg.edges()]
243 
244  ax = plt.gca()
245 
246  # Centered circular pose
247  self.pospos = nx.layout.circular_layout(self.gg)
248  all_x = [node_pos[0] for node, node_pos in self.pospos.items()]
249  all_y = [node_pos[1] for node, node_pos in self.pospos.items()]
250  min_x = np.min(all_x)
251  max_x = np.max(all_x)
252  min_y = np.min(all_y)
253  max_y = np.max(all_y)
254  for _, node_pos in self.pospos.items():
255  node_pos[0] -= (max_x + min_x) / 2
256  node_pos[1] -= (max_y + min_y) / 2
257 
258  # Rendering

260  if self.first_runfirst_run:

262  ax.set_axis_off()
263  ax.set_aspect("equal")
264  plt.ylim(-1.3, 1.3)
265  plt.xlim(-1.3, 1.3)
