79 // Reads a bitfield within action, with LSB at offset, and length bits long (up
80 // to 8).
81 inline uint8_t GetField(Action action, int offset, int length) {
82  return (action >> offset) & ((1ULL << length) - 1);
83 }
84 
85 // Sets a bitfield within action, with LSB at offset, and length bits long (up
86 // to 8) to value.
87 inline void SetField(int offset, int length, uint8_t value, Action* action) {
88  uint32_t mask = ((1ULL << length) - 1) << offset;
89  *action &= ~mask;
90  *action |= static_cast<Action>(value) << offset;
91 }
