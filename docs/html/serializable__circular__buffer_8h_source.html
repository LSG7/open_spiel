34 // A serializable circular buffer of fixed size.
35 template <class T>
36 class SerializableCircularBuffer : public CircularBuffer<T> {
37  public:
38  explicit SerializableCircularBuffer(int max_size)
39  : CircularBuffer<T>(max_size) {}

41  // Serialize the data of the buffer to a file.
42  void SaveBuffer(const std::string& path) const {
43  nop::Serializer<nop::StreamWriter<std::ofstream>> serializer{path};
44  serializer.Write(this->max_size_);
45  serializer.Write(this->total_added_);
46  serializer.Write(this->data_);
47  }
48 
49  // Populate the buffer with data from a saved buffer's file.
50  void LoadBuffer(const std::string& path) {
51  nop::Deserializer<nop::StreamReader<std::ifstream>> deserializer{path};
52 
53  // Ensure this buffer's max size equals the max size of the saved buffer.
54  int max_size;
55  deserializer.Read(&max_size);
56  if (max_size != this->max_size_) {
57  SpielFatalError(absl::StrFormat("Cannot load data from a buffer with max"
58  "size %d into a buffer with max size %d.",
59  max_size,
60  this->max_size_));
61  }
62 
63  deserializer.Read(&(this->total_added_));
64  deserializer.Read(&(this->data_));
65  }
66 };
