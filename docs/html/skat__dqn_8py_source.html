55 def eval_against_random_bots(env, trained_agents, random_agents, num_episodes):
56  """Evaluates `trained_agents` against `random_agents` for `num_episodes`."""
57  num_players = len(trained_agents)
58  sum_episode_rewards = np.zeros(num_players)
59  for player_pos in range(num_players):
60  for _ in range(num_episodes):
61  cur_agents = random_agents[:]
62  if FLAGS.randomize_positions:
63  eval_player_pos = random.randrange(num_players)
64  else:
65  eval_player_pos = player_pos
66  cur_agents[eval_player_pos] = trained_agents[player_pos]
67  cur_agents[eval_player_pos].player_id = eval_player_pos
68  time_step = env.reset()
69  episode_rewards = 0
70  while not time_step.last():
71  player_id = time_step.observations["current_player"]
72  agent_output = cur_agents[player_id].step(
73  time_step, is_evaluation=True)
74  action_list = [agent_output.action]
75  time_step = env.step(action_list)
76  episode_rewards += time_step.rewards[eval_player_pos]
77  sum_episode_rewards[player_pos] += episode_rewards
78  return sum_episode_rewards / num_episodes

102  agents = [

105  player_id=idx,

107  num_actions=num_actions,

115  for ep in range(FLAGS.num_train_episodes):

117  r_mean = eval_against_random_bots(env, agents, random_agents,
118  FLAGS.num_eval_games)
