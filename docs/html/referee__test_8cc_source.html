36 void SayHelloViaSubprocess() {
37  Subprocess s("echo Hello", /*should_block=*/true);
38  char buf[5];
39  auto bytes_read = read(s.stdout(), &buf, 5);
40  SPIEL_CHECK_EQ(bytes_read, 5);
41  char expected[5] = {'H', 'e', 'l', 'l', 'o'};
42  for (int i = 0; i < 5; ++i) SPIEL_CHECK_EQ(buf[i], expected[i]);
43 }

45 void SayHelloViaChannel() {
46  // Bot channels are asynchronous -- we read from a different thread.
47  std::unique_ptr<BotChannel> channel = MakeBotChannel(0, "echo Hello");
48  std::thread read(ReadLineFromChannelStdout, channel.get());
49  channel->StartRead(/*time_limit=*/500);
50  sleep_ms(1000);
51  channel->ShutDown();
52  read.join();
53  SPIEL_CHECK_EQ(channel->response(), "Hello");
54 }

105 void PlayWithFailingBots() {
106  std::vector<std::string> failing_cases = {
107  "/non_existing_bot", "/test_bot_with_non_exec_flag",
108  "/test_bot_break_pipe.sh", "/test_bot_sleep.sh",
109  "/test_bot_ready.sh", "/test_bot_start.sh",
110  "/test_bot_illegal_action.sh",
111  // "/test_bot_buffer_overflow.sh",
112  };
113 
114  for (int i = 0; i < failing_cases.size(); ++i) {
115  const std::string& failing_case = failing_cases[i];
116  std::string failing_bot =
117  absl::StrCat(absl::GetFlag(FLAGS_bots_dir), failing_case);
118  std::cout << "\n\nFailing bot: " << failing_bot << std::endl;
119 
120  // Use a single-player game.
121  open_spiel::higc::Referee ref(
122  "cliff_walking", {failing_bot}, /*seed=*/42,
123  /*settings=*/
124  TournamentSettings{// Disqualify after the 2nd failing match.
125  .disqualification_rate = 0.5});
126  std::unique_ptr<TournamentResults> results = ref.PlayTournament(2);
127  SPIEL_CHECK_EQ(results->disqualified[0], true);
128  if (i < 4) {
129  // No matches are played, if the bot can't even start properly.
130  SPIEL_CHECK_EQ(results->num_matches(), 0);
131  } else {
132  SPIEL_CHECK_EQ(results->num_matches(), 2);
133  }
134  }
135 }
