127 std::unique_ptr<State> PyState::Clone() const {
128  // Create a new State of the right type.
129  auto rv = game_->NewInitialState();
130 
131  // Copy the Python-side properties of the state.
132  py::function deepcopy = py::module::import("copy").attr("deepcopy");
133  py::object py_state = py::cast(*rv);
134  for (auto [k, v] : PyDict(*this)) {
135  py_state.attr(k) = deepcopy(v);
136  }
137 
138  // Copy the C++-side properties of the state (all on the parent class).
139  // Since we started with a valid initial state, we only need to copy
140  // properties that change during the life of the state - hence num_players,
141  // num_distinct_actions are omitted.
142  PyState* state = open_spiel::down_cast<PyState*>(rv.get());
143  state->history_ = history_;
144  state->move_number_ = move_number_;
145 
146  return rv;
147 }

160  GameRegisterer::RegisterGame(

199 void PyObserver::WriteTensor(const State& state, int player,
200  Allocator* allocator) const {
201  using Array = py::array_t<float, py::array::c_style | py::array::forcecast>;
202  const PyState& py_state = open_spiel::down_cast<const PyState&>(state);
203  set_from_(py_state, player);
204  py::dict dict = py_observer_.attr("dict");
205  for (auto [k, v] : dict) {
206  auto a = py::cast<Array>(v);
207  const int dims = a.ndim();
208  absl::InlinedVector<int, 4> shape(dims);
209  for (int i = 0; i < dims; ++i) shape[i] = a.shape(i);
210  SpanTensor out = allocator->Get(k.cast<std::string>(), shape);
211  std::copy(a.data(), a.data() + a.size(), out.data().begin());
212  }
213 }

312 py::dict decode_dict(const absl::string_view str) {
313  std::string bytes;
314  SPIEL_CHECK_TRUE(absl::Base64Unescape(str, &bytes));
315  py::function pickle_loads = py::module::import("pickle").attr("loads");
316  return pickle_loads(py::bytes(bytes));

329 void PyState::Deserialize(const std::string& str) {
330  std::vector<absl::string_view> pieces =
331  absl::StrSplit(str, absl::MaxSplits('\n', 2));
332  SPIEL_CHECK_EQ(pieces.size(), 3);
333 
334  SPIEL_CHECK_EQ(pieces[0].substr(0, kTagHistory.size()), kTagHistory);
335  auto history_str = pieces[0].substr(kTagHistory.size());
336  if (!history_str.empty()) {
337  for (auto& h : absl::StrSplit(history_str, ',')) {
338  std::vector<absl::string_view> p = absl::StrSplit(h, ':');
339  SPIEL_CHECK_EQ(p.size(), 2);
340  int player, action;
341  SPIEL_CHECK_TRUE(absl::SimpleAtoi(p[0], &player));
342  SPIEL_CHECK_TRUE(absl::SimpleAtoi(p[1], &action));
343  history_.push_back({player, action});
344  }
345  }
346 
347  SPIEL_CHECK_EQ(pieces[1].substr(0, kTagMoveNumber.size()), kTagMoveNumber);
348  SPIEL_CHECK_TRUE(
349  absl::SimpleAtoi(pieces[1].substr(kTagMoveNumber.size()), &move_number_));
350 
351  SPIEL_CHECK_EQ(pieces[2].substr(0, kTagDict.size()), kTagDict);
352  py::object py_state = py::cast(*this);
353  for (const auto& [k, v] : decode_dict(pieces[2].substr(kTagDict.size()))) {
354  py_state.attr(k) = v;
355  }
356 }

358 std::string PyState::Serialize() const {
359  return absl::StrCat(
360  // C++ Attributes
361  kTagHistory,
362  absl::StrJoin(history_, ",",
363  [](std::string* out, const PlayerAction& pa) {
364  absl::StrAppend(out, pa.player, ":", pa.action);
