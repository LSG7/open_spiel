154 bool CheckMap(const absl::flat_hash_map<int, int>& map,
155  const std::vector<std::pair<int, int>>& keys_and_values) {
156  for (const auto& kvp : keys_and_values) {
157  auto found = map.find(kvp.first);
158  if (found == map.end()) {
159  return false;
160  }
161  if (found->second != kvp.second) {
162  return false;
163  }
164  }
165  return true;
166 }

202 bool CheckSpan(absl::Span<const int> span, const std::vector<int>& values) {
203  if (span.size() != values.size()) return false;
204  for (size_t i = 0; i < span.size(); ++i) {
205  if (span[i] != values[i]) return false;
206  }
207  return true;
208 }

256 int VariantToInt(AOrB value) {
257  if (absl::holds_alternative<A>(value)) {
