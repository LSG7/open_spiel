31 @parameterized.named_parameters(('cpp', 'mfg_crowd_modelling'),
32 ('python', 'python_mfg_crowd_modelling'))
33 def test_train(self, name):
34 """Checks that the training works."""
35 game = pyspiel.load_game(name)
36 assert game.num_players() == 1
37 uniform_policy = policy.UniformRandomPolicy(game)
38 uniform_dist = distribution.DistributionPolicy(game, uniform_policy)
39 env = rl_environment.Environment(
40 game, mfg_distribution=uniform_dist, mfg_population=0)
41 info_state_size = env.observation_spec()['info_state'][0]
42 num_actions = env.action_spec()['num_actions']
43 np.random.seed(0)
44 args = {
45 'alpha': 0.9,
46 'batch_size': 128,
47 'discount_factor': 1.0,
48 'epsilon_decay_duration': 20000000,
49 'epsilon_end': 0.1,
50 'epsilon_start': 0.1,
51 'gradient_clipping': 40,
52 'hidden_layers_sizes': [128, 128],
53 'learn_every': 64,
54 'learning_rate': 0.01,
55 'loss': 'mse',
56 'min_buffer_size_to_learn': 500,
57 'optimizer': 'adam',
58 'replay_buffer_capacity': 2000,
59 'tau': 10,
60 'update_target_network_every': 50
61 }
62 agent = munchausen_deep_mirror_descent.MunchausenDQN(
63 0, info_state_size, num_actions, **args)
64 md = munchausen_deep_mirror_descent.DeepOnlineMirrorDescent(
65 game, [env], [agent], num_episodes_per_iteration=100)
66 for _ in range(10):
67 md.iteration()
68 nash_conv_md = nash_conv.NashConv(game, md.policy)
69 self.assertLessEqual(nash_conv_md.nash_conv(), 3)
