25 def test_ranked_pairs_wikipedia_example1(self):
26 alternatives = ["w", "x", "y", "z"]
27 votes = [
28 base.WeightedVote(7, ["w", "x", "z", "y"]),
29 base.WeightedVote(2, ["w", "y", "x", "z"]),
30 base.WeightedVote(4, ["x", "y", "z", "w"]),
31 base.WeightedVote(5, ["x", "z", "w", "y"]),
32 base.WeightedVote(1, ["y", "w", "x", "z"]),
33 base.WeightedVote(8, ["y", "z", "w", "x"]),
34 ]
35 profile = base.PreferenceProfile(votes=votes, alternatives=alternatives)
36 method = ranked_pairs.RankedPairsVoting()
37 outcome = method.run_election(profile)
38 with self.subTest("Ranking and scores"):
39 self.assertListEqual(outcome.ranking, ["w", "x", "y", "z"])
40 self.assertListEqual(outcome.scores, [29, 19, 3, 0])
41 with self.subTest("Check the graph"):
42 expected_graph = np.array(
43 [[0, 1, 1, 0], [0, 0, 1, 1], [0, 0, 0, 1], [0, 0, 0, 0]]
44 )
45 self.assertTrue(np.array_equal(outcome.graph, expected_graph))
