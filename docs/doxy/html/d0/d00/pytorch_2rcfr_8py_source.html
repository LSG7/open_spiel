15"""Regression counterfactual regret minimization (RCFR) [Waugh et al., 2015; Morrill, 2016].
16
17In contrast to (tabular) counterfactual regret minimization (CFR)
18[Zinkevich et al., 2007], RCFR replaces the table of regrets that generate the
19current policy profile with a profile of regression models. The average
20policy is still tracked exactly with a full game-size table. The exploitability
21of the average policy in zero-sum games decreases as the model accuracy and
22the number of iterations increase [Waugh et al., 2015; Morrill, 2016]. As long
23as the regression model errors decrease across iterations, the average policy
24converges toward a Nash equilibrium in zero-sum games.
25
26# References
27
28Dustin Morrill. Using Regret Estimation to Solve Games Compactly.
29 M.Sc. thesis, Computing Science Department, University of Alberta,
30 Apr 1, 2016, Edmonton Alberta, Canada.
31Kevin Waugh, Dustin Morrill, J. Andrew Bagnell, and Michael Bowling.
32 Solving Games with Functional Regret Estimation. At the Twenty-Ninth AAAI
33 Conference on Artificial Intelligence, January 25-29, 2015, Austin Texas,
34 USA. Pages 2138-2145.
35Martin Zinkevich, Michael Johanson, Michael Bowling, and Carmelo Piccione.
36 Regret Minimization in Games with Incomplete Information.
37 At Advances in Neural Information Processing Systems 20 (NeurIPS). 2007.
38"""

46def tensor_to_matrix(tensor):
47 """Converts `tensor` to a matrix (a rank-2 tensor) or raises an exception.
48
49 Args:
50 tensor: The tensor to convert.
51
52 Returns:
53 A PyTorch matrix (rank-2 `torch.Tensor`).
54
55 Raises:
56 ValueError: If `tensor` cannot be trivially converted to a matrix, i.e.
57 `tensor` has a rank > 2.
58 """
59 tensor = torch.Tensor(tensor)
60 rank = tensor.ndim
61 # rank = len(list(tensor.shape))
62 if rank > 2:
63 raise ValueError(
64 ("Tensor {} cannot be converted into a matrix as it is rank "
65 "{} > 2.").format(tensor, rank))
66 elif rank < 2:
67 num_columns = 1 if rank == 0 else tensor.shape[0]
68 tensor = torch.reshape(tensor, [1, num_columns])
69 return tensor

72def with_one_hot_action_features(state_features, legal_actions,
73 num_distinct_actions):
74 """Constructs features for each sequence by extending state features.
75
76 Sequences features are constructed by concatenating one-hot features
77 indicating each action to the information state features and stacking them.
78
79 Args:
80 state_features: The features for the information state alone. Must be a
81 `torch.Tensor` with a rank less than or equal to (if batched) 2.
82 legal_actions: The list of legal actions in this state. Determines the
83 number of rows in the returned feature matrix.
84 num_distinct_actions: The number of globally distinct actions in the game.
85 Determines the length of the action feature vector concatenated onto the
86 state features.
87
88 Returns:
89 A `torch.Tensor` feature matrix with one row for each sequence and # state
90 features plus `num_distinct_actions`-columns.
91
92 Raises:
93 ValueError: If `state_features` has a rank > 2.
94 """
95 state_features = tensor_to_matrix(state_features)
96 with_action_features = []
97 for action in legal_actions:
98 action_features = F.one_hot(
99 torch.tensor([action]), num_classes=num_distinct_actions)
100 all_features = torch.cat([state_features, action_features], axis=1)
101 with_action_features.append(all_features)
102 return torch.cat(with_action_features, axis=0)

134class RootStateWrapper(object):
135 """Analyzes the subgame at a given root state.
136
137 It enumerates features for each player sequence, creates a mapping between
138 information states to sequence index offsets, and caches terminal values
139 in a dictionary with history string keys.
140
141 Properties:
142 root: An OpenSpiel `State`.
143 sequence_features: A `list` of sequence feature matrices, one for each
144 player. This list uses depth-first, information state-major ordering, so
145 sequences are grouped by information state. I.e. the first legal action
146 in the first state has index 0, the second action in the same information
147 state has index 1, the third action will have index 3, and so on.
148 Sequences in the next information state descendant of the first action
149 will begin indexing its sequences at the number of legal actions in the
150 ancestor information state.
151 num_player_sequences: The number of sequences for each player.
152 info_state_to_sequence_idx: A `dict` mapping each information state string
153 to the `sequence_features` index of the first sequence in the
154 corresponding information state.
155 terminal_values: A `dict` mapping history strings to terminal values for
156 each player.
157 """
158
159 def __init__(self, state):
160 self.root = state
161 self._num_distinct_actions = len(state.legal_actions_mask(0))
162
163 self.sequence_features = [[] for _ in range(state.num_players())]
164 self.num_player_sequences = [0] * state.num_players()
165 self.info_state_to_sequence_idx = {}
166 self.terminal_values = {}
167 self._walk_descendants(state)
168 self.sequence_features = [
169 torch.cat(rows, axis=0) for rows in self.sequence_features
170 ]
171
172 def _walk_descendants(self, state):
173 """Records information about `state` and its descendants."""
174 if state.is_terminal():
175 self.terminal_values[state.history_str()] = np.array(state.returns())
176 return
177
178 elif state.is_chance_node():
179 for action, _ in state.chance_outcomes():
180 self._walk_descendants(state.child(action))
181 return
182
183 player = state.current_player()
184 info_state = state.information_state_string(player)
185 actions = state.legal_actions()
186
187 if info_state not in self.info_state_to_sequence_idx:
188 n = self.num_player_sequences[player]
189 self.info_state_to_sequence_idx[info_state] = n
190 self.sequence_features[player].append(
191 sequence_features(state, self._num_distinct_actions))
192 self.num_player_sequences[player] += len(actions)
193
194 for action in actions:
195 self._walk_descendants(state.child(action))

197 def sequence_weights_to_policy(self, sequence_weights, state):
198 """Returns a behavioral policy at `state` from sequence weights.
199
200 Args:
201 sequence_weights: An array of non-negative weights, one for each of
202 `state.current_player()`'s sequences in `state`'s game.
203 state: An OpenSpiel `State` that represents an information state in an
204 alternating-move game.
205
206 Returns:
207 A `np.array<double>` probability distribution representing the policy in
208 `state` encoded by `sequence_weights`. Weights corresponding to actions
209 in `state` are normalized by their sum.
210
211 Raises:
212 ValueError: If there are too few sequence weights at `state`.
213 """
214 info_state = state.information_state_string()
215 sequence_offset = self.info_state_to_sequence_idx[info_state]
216 actions = state.legal_actions()
217
218 sequence_idx_end = sequence_offset + len(actions)
219 weights = sequence_weights[sequence_offset:sequence_idx_end]
220
221 if len(weights) < len(actions):
222 raise ValueError(
223 ("Invalid policy: Policy {player} at sequence offset "
224 "{sequence_offset} has only {policy_len} elements but there "
225 "are {num_actions} legal actions.").format(
226 player=state.current_player(),
227 sequence_offset=sequence_offset,
228 policy_len=len(weights),
229 num_actions=len(actions)))
230 return normalized_by_sum(weights)
231
232 def sequence_weights_to_policy_fn(self, player_sequence_weights):
233 """Returns a policy function based on sequence weights for each player.
234
235 Args:
236 player_sequence_weights: A list of weight arrays, one for each player.
237 Each array should have a weight for each of that player's sequences in
238 `state`'s game.
239
240 Returns:
241 A `State` -> `np.array<double>` function. The output of this function is
242 a probability distribution that represents the policy at the given
243 `State` encoded by `player_sequence_weights` according to
244 `sequence_weights_to_policy`.
245 """
246
247 def policy_fn(state):
248 player = state.current_player()
249 return self.sequence_weights_to_policy(player_sequence_weights[player],
250 state)
251
252 return policy_fn

259 def counterfactual_regrets_and_reach_weights(self, regret_player,
260 reach_weight_player,
261 *sequence_weights):
262 """Returns counterfactual regrets and reach weights as a tuple.
263
264 Args:
265 regret_player: The player for whom counterfactual regrets are computed.
266 reach_weight_player: The player for whom reach weights are computed.
267 *sequence_weights: A list of non-negative sequence weights for each player
268 determining the policy profile. Behavioral policies are generated by
269 normalizing sequence weights corresponding to actions in each
270 information state by their sum.
271
272 Returns:
273 The counterfactual regrets and reach weights as an `np.array`-`np.array`
274 tuple.
275
276 Raises:
277 ValueError: If there are too few sequence weights at any information state
278 for any player.
279 """
280 num_players = len(sequence_weights)
281 regrets = np.zeros(self.num_player_sequences[regret_player])
282 reach_weights = np.zeros(self.num_player_sequences[reach_weight_player])
283
284 def _walk_descendants(state, reach_probabilities, chance_reach_probability):
285 """Compute `state`'s counterfactual regrets and reach weights.
286
287 Args:
288 state: An OpenSpiel `State`.
289 reach_probabilities: The probability that each player plays to reach
290 `state`'s history.
291 chance_reach_probability: The probability that all chance outcomes in
292 `state`'s history occur.
293
294 Returns:
295 The counterfactual value of `state`'s history.
296 Raises:
297 ValueError if there are too few sequence weights at any information
298 state for any player.
299 """
300
301 if state.is_terminal():
302 player_reach = (
303 np.prod(reach_probabilities[:regret_player]) *
304 np.prod(reach_probabilities[regret_player + 1:]))
305
306 counterfactual_reach_prob = player_reach * chance_reach_probability
307 u = self.terminal_values[state.history_str()]
308 return u[regret_player] * counterfactual_reach_prob
309
310 elif state.is_chance_node():
311 v = 0.0
312 for action, action_prob in state.chance_outcomes():
313 v += _walk_descendants(
314 state.child(action), reach_probabilities,
315 chance_reach_probability * action_prob)
316 return v
317
318 player = state.current_player()
319 info_state = state.information_state_string(player)
320 sequence_idx_offset = self.info_state_to_sequence_idx[info_state]
321 actions = state.legal_actions(player)
322
323 sequence_idx_end = sequence_idx_offset + len(actions)
324 my_sequence_weights = sequence_weights[player][
325 sequence_idx_offset:sequence_idx_end]
326
327 if len(my_sequence_weights) < len(actions):
328 raise ValueError(
329 ("Invalid policy: Policy {player} at sequence offset "
330 "{sequence_idx_offset} has only {policy_len} elements but there "
331 "are {num_actions} legal actions.").format(
332 player=player,
333 sequence_idx_offset=sequence_idx_offset,
334 policy_len=len(my_sequence_weights),
335 num_actions=len(actions)))
336
337 policy = normalized_by_sum(my_sequence_weights)
338 action_values = np.zeros(len(actions))
339 state_value = 0.0
340
341 is_reach_weight_player_node = player == reach_weight_player
342 is_regret_player_node = player == regret_player
343
344 reach_prob = reach_probabilities[player]
345 for action_idx, action in enumerate(actions):
346 action_prob = policy[action_idx]
347 next_reach_prob = reach_prob * action_prob
348
349 if is_reach_weight_player_node:
350 reach_weight_player_plays_down_this_line = next_reach_prob > 0
351 if not reach_weight_player_plays_down_this_line:
352 continue
353 sequence_idx = sequence_idx_offset + action_idx
354 reach_weights[sequence_idx] += next_reach_prob
355
356 reach_probabilities[player] = next_reach_prob
357
358 action_value = _walk_descendants(
359 state.child(action), reach_probabilities, chance_reach_probability)
360
361 if is_regret_player_node:
362 state_value = state_value + action_prob * action_value
363 else:
364 state_value = state_value + action_value
365 action_values[action_idx] = action_value
366
367 reach_probabilities[player] = reach_prob
368
369 if is_regret_player_node:
370 regrets[sequence_idx_offset:sequence_idx_end] += (
371 action_values - state_value)
372 return state_value
373
374 # End of _walk_descendants
375
376 _walk_descendants(self.root, np.ones(num_players), 1.0)
377 return regrets, reach_weights

380def normalized_by_sum(v, axis=0, mutate=False):
381 """Divides each element of `v` along `axis` by the sum of `v` along `axis`.
382
383 Assumes `v` is non-negative. Sets of `v` elements along `axis` that sum to
384 zero are normalized to `1 / v.shape[axis]` (a uniform distribution).
385
386 Args:
387 v: Non-negative array of values.
388 axis: An integer axis.
389 mutate: Whether or not to store the result in `v`.
390
391 Returns:
392 The normalized array.
393 """
394 v = np.asarray(v)
395 denominator = v.sum(axis=axis, keepdims=True)
396 denominator_is_zero = denominator == 0
397
398 # Every element of `denominator_is_zero` that is true corresponds to a
399 # set of elements in `v` along `axis` that are all zero. By setting these
400 # denominators to `v.shape[axis]` and adding 1 to each of the corresponding
401 # elements in `v`, these elements are normalized to `1 / v.shape[axis]`
402 # (a uniform distribution).
403 denominator += v.shape[axis] * denominator_is_zero
404 if mutate:
405 v += denominator_is_zero
406 v /= denominator
407 else:
408 v = (v + denominator_is_zero) / denominator
409 return v

412def relu(v):
413 """Returns the element-wise maximum between `v` and 0."""
414 return np.maximum(v, 0)

417def _descendant_states(state, depth_limit, depth, include_terminals,
418 include_chance_states):
419 """Recursive descendant state generator.
420
421 Decision states are always yielded.
422
423 Args:
424 state: The current state.
425 depth_limit: The descendant depth limit. Zero will ensure only
426 `initial_state` is generated and negative numbers specify the absence of a
427 limit.
428 depth: The current descendant depth.
429 include_terminals: Whether or not to include terminal states.
430 include_chance_states: Whether or not to include chance states.
431
432 Yields:
433 `State`, a state that is `initial_state` or one of its descendants.
434 """
435 if state.is_terminal():
436 if include_terminals:
437 yield state
438 return
439
440 if depth > depth_limit >= 0:
441 return
442
443 if not state.is_chance_node() or include_chance_states:
444 yield state
445
446 for action in state.legal_actions():
447 state_for_search = state.child(action)
448 for substate in _descendant_states(state_for_search, depth_limit, depth + 1,
449 include_terminals,
450 include_chance_states):
451 yield substate

454def all_states(initial_state,
455 depth_limit=-1,
456 include_terminals=False,
457 include_chance_states=False):
458 """Generates states from `initial_state`.
459
460 Generates the set of states that includes only the `initial_state` and its
461 descendants that satisfy the inclusion criteria specified by the remaining
462 parameters. Decision states are always included.
463
464 Args:
465 initial_state: The initial state from which to generate states.
466 depth_limit: The descendant depth limit. Zero will ensure only
467 `initial_state` is generated and negative numbers specify the absence of a
468 limit. Defaults to no limit.
469 include_terminals: Whether or not to include terminal states. Defaults to
470 `False`.
471 include_chance_states: Whether or not to include chance states. Defaults to
472 `False`.
473
474 Returns:
475 A generator that yields the `initial_state` and its descendants that
476 satisfy the inclusion criteria specified by the remaining parameters.
477 """
478 return _descendant_states(
479 state=initial_state,
480 depth_limit=depth_limit,
481 depth=0,
482 include_terminals=include_terminals,
483 include_chance_states=include_chance_states)

486def sequence_weights_to_tabular_profile(root, policy_fn):
487 """Returns the `dict` of `list`s of action-prob pairs-form of `policy_fn`."""
488 tabular_policy = {}
489 players = list(range(root.num_players()))
490 for state in all_states(root):
491 for player in players:
492 legal_actions = state.legal_actions(player)
493 if len(legal_actions) < 1:
494 continue
495 info_state = state.information_state_string(player)
496 if info_state in tabular_policy:
497 continue
498 my_policy = policy_fn(state)
499 tabular_policy[info_state] = list(zip(legal_actions, my_policy))
500 return tabular_policy

503def feedforward_evaluate(layers,
504 x,
505 use_skip_connections=False,
506 hidden_are_factored=False,
507 hidden_activation=nn.ReLU):
508 """Evaluates `layers` as a feedforward neural network on `x`.
509
510 Args:
511 layers: The neural network layers (`torch.Tensor` -> `torch.Tensor`
512 callables).
513 x: The array-like input to evaluate. Must be trivially convertible to a
514 matrix (tensor rank <= 2).
515 use_skip_connections: Whether or not to use skip connections between layers.
516 If the layer input has too few features to be added to the layer output,
517 then the end of input is padded with zeros. If it has too many features,
518 then the input is truncated.
519 hidden_are_factored: Whether or not hidden logical layers are factored into
520 two separate linear transformations stored as adjacent elements of
521 `layers`.
522 hidden_activation: the activation function following the hidden layers.
523
524 Returns:
525 The `torch.Tensor` evaluation result.
526
527 Raises:
528 ValueError: If `x` has a rank greater than 2.
529 """
530 x = tensor_to_matrix(x)
531 i = 0
532 while i < len(layers) - 1:
533 if isinstance(layers[i], hidden_activation):
534 x = layers[i](x)
535 i += 1
536 continue
537 y = layers[i](x)
538 i += 1
539 if hidden_are_factored:
540 y = layers[i](y)
541 i += 1
542 if use_skip_connections:
543 my_num_features = x.shape[1]
544 padding = y.shape[1] - my_num_features
545 if padding > 0:
546 zeros = torch.zeros([x.shape[0], padding])
547 x = torch.cat([x, zeros], axis=1)
548 elif padding < 0:
549 x = x[0:x.shape[0], 0:y.shape[1]]
550 y = x + y
551 x = y
552 return layers[-1](x)

555class DeepRcfrModel(nn.Module):
556 """A flexible deep feedforward RCFR model class.
557
558 Properties:
559 layers: The `torch.keras.Layer` layers describing this model.
560 """
561
562 def __init__(self,
563 game,
564 num_hidden_units,
565 num_hidden_layers=1,
566 num_hidden_factors=0,
567 hidden_activation=nn.ReLU,
568 use_skip_connections=False,
569 regularizer=None):
570 """Creates a new `DeepRcfrModel.
571
572 Args:
573 game: The OpenSpiel game being solved.
574 num_hidden_units: The number of units in each hidden layer.
575 num_hidden_layers: The number of hidden layers. Defaults to 1.
576 num_hidden_factors: The number of hidden factors or the matrix rank of the
577 layer. If greater than zero, hidden layers will be split into two
578 separate linear transformations, the first with
579 `num_hidden_factors`-columns and the second with
580 `num_hidden_units`-columns. The result is that the logical hidden layer
581 is a rank-`num_hidden_units` matrix instead of a rank-`num_hidden_units`
582 matrix. When `num_hidden_units < num_hidden_units`, this is effectively
583 implements weight sharing. Defaults to 0.
584 hidden_activation: The activation function to apply over hidden layers.
585 Defaults to `torch.nn.ReLU`.
586 use_skip_connections: Whether or not to apply skip connections (layer
587 output = layer(x) + x) on hidden layers. Zero padding or truncation is
588 used to match the number of columns on layer inputs and outputs.
589 regularizer: A regularizer to apply to each layer. Defaults to `None`.
590 """
591 super(DeepRcfrModel, self).__init__()
592 self._use_skip_connections = use_skip_connections
593 self._hidden_are_factored = num_hidden_factors > 0
594 self._hidden_activation = hidden_activation
595 input_size = num_features(game)
596
597 self.layers = []
598 for _ in range(num_hidden_layers):
599 if self._hidden_are_factored:
600 self.layers.append(nn.Linear(input_size, num_hidden_factors, bias=True))
601
602 self.layers.append(
603 nn.Linear(
604 num_hidden_factors if self._hidden_are_factored else input_size,
605 num_hidden_units,
606 bias=True))
607 if hidden_activation:
608 self.layers.append(hidden_activation())
609
610 self.layers.append(nn.Linear(num_hidden_units, 1, bias=True))
611
612 self.layers = nn.ModuleList(self.layers)
613 # Construct variables for all layers by exercising the network.
614 x = torch.zeros([1, num_features(game)])
615 for layer in self.layers:
616 x = layer(x)
617
618 def __call__(self, x):
619 """Evaluates this model on `x`."""
620 return feedforward_evaluate(
621 layers=self.layers,
622 x=x,
623 use_skip_connections=self._use_skip_connections,
624 hidden_are_factored=self._hidden_are_factored,
625 hidden_activation=self._hidden_activation)

628class _RcfrSolver(object):
629 """An abstract RCFR solver class.
630
631 Requires that subclasses implement `evaluate_and_update_policy`.
632 """
633
634 def __init__(self, game, models, truncate_negative=False):
635 """Creates a new `_RcfrSolver`.
636
637 Args:
638 game: An OpenSpiel `Game`.
639 models: Current policy models (optimizable array-like -> `torch.Tensor`
640 callables) for both players.
641 truncate_negative: Whether or not to truncate negative (approximate)
642 cumulative regrets to zero to implement RCFR+. Defaults to `False`.
643 """
644 self._game = game
645 self._models = models
646 self._truncate_negative = truncate_negative
647 self._root_wrapper = RootStateWrapper(game.new_initial_state())
648
649 self._cumulative_seq_probs = [
650 np.zeros(n) for n in self._root_wrapper.num_player_sequences
651 ]
652
653 def _sequence_weights(self, player=None):
654 """Returns regret-like weights for each sequence as an `np.array`.
655
656 Negative weights are truncated to zero.
657
658 Args:
659 player: The player to compute weights for, or both if `player` is `None`.
660 Defaults to `None`.
661 """
662 if player is None:
663 return [
664 self._sequence_weights(player)
665 for player in range(self._game.num_players())
666 ]
667 else:
668 tensor = F.relu(
669 torch.squeeze(self._models[player](
670 self._root_wrapper.sequence_features[player])))
671 return tensor.detach().numpy()
672
673 def evaluate_and_update_policy(self, train_fn):
674 """Performs a single step of policy evaluation and policy improvement.
675
676 Args:
677 train_fn: A (model, `torch.data.Dataset`) function that trains the given
678 regression model to accurately reproduce the x to y mapping given x-y
679 data.
680
681 Raises:
682 NotImplementedError: If not overridden by child class.
683 """
684 raise NotImplementedError()

696 def average_policy(self):
697 """Returns the average of all policies iterated.
698
699 This average policy converges toward a Nash policy as the number of
700 iterations increases as long as the regret prediction error decreases
701 continually [Morrill, 2016].
702
703 The policy is computed using the accumulated policy probabilities computed
704 using `evaluate_and_update_policy`.
705
706 Returns:
707 A `dict<info state, list<Action, probability>>` that maps info state
708 strings to (Action, probability) pairs describing each player's policy.
709 """
710 return self._root_wrapper.sequence_weights_to_tabular_profile(
711 self._cumulative_seq_probs)
712
713 def _previous_player(self, player):
714 """The previous player in the turn ordering."""
715 return player - 1 if player > 0 else self._game.num_players() - 1
716
717 def _average_policy_update_player(self, regret_player):
718 """The player for whom the average policy should be updated."""
719 return self._previous_player(regret_player)

722class RcfrSolver(_RcfrSolver):
723 """RCFR with an effectively infinite regret data buffer.
724
725 Exact or bootstrapped cumulative regrets are stored as if an infinitely
726 large data buffer. The average strategy is updated and stored in a full
727 game-size table. Reproduces the RCFR versions used in experiments by
728 Waugh et al. [2015] and Morrill [2016] except that this class does not
729 restrict the user to regression tree models.
730 """
731
732 def __init__(self, game, models, bootstrap=None, truncate_negative=False):
733 self._bootstrap = bootstrap
734 super(RcfrSolver, self).__init__(
735 game, models, truncate_negative=truncate_negative)
736
737 self._regret_targets = [
738 np.zeros(n) for n in self._root_wrapper.num_player_sequences
739 ]
740
741 def evaluate_and_update_policy(self, train_fn):
742 """Performs a single step of policy evaluation and policy improvement.
743
744 Args:
745 train_fn: A (model, `torch.data.Dataset`) function that trains the given
746 regression model to accurately reproduce the x to y mapping given x-y
747 data.
748 """
749 sequence_weights = self._sequence_weights()
750 player_seq_features = self._root_wrapper.sequence_features
751 for regret_player in range(self._game.num_players()):
752 seq_prob_player = self._average_policy_update_player(regret_player)
753
754 regrets, seq_probs = (
755 self._root_wrapper.counterfactual_regrets_and_reach_weights(
756 regret_player, seq_prob_player, *sequence_weights))
757
758 if self._bootstrap:
759 self._regret_targets[regret_player][:] = sequence_weights[regret_player]
760 if self._truncate_negative_truncate_negative:
761 regrets = np.maximum(-relu(self._regret_targets[regret_player]),
762 regrets)
763
764 self._regret_targets[regret_player] += regrets
765 self._cumulative_seq_probs[seq_prob_player] += seq_probs
766
767 targets = torch.unsqueeze(
768 torch.Tensor(self._regret_targets[regret_player]), axis=1)
769 data = torch.utils.data.TensorDataset(player_seq_features[regret_player],
770 targets)
771
772 regret_player_model = self._models[regret_player]
773 train_fn(regret_player_model, data)
774 sequence_weights[regret_player] = self._sequence_weights(regret_player)

777class ReservoirBuffer(object):
778 """A generic reservoir buffer data structure.
779
780 After every insertion, its contents represents a `size`-size uniform
781 random sample from the stream of candidates that have been encountered.
782 """
783
784 def __init__(self, size):
785 self.size = size
786 self.num_elements = 0
787 self._buffer = np.full([size], None, dtype=object)
788 self._num_candidates = 0
789
790 @property
791 def buffer(self):
792 return self._buffer[:self.num_elements]
793
794 def insert(self, candidate):
795 """Consider this `candidate` for inclusion in this sampling buffer."""
796 self._num_candidates += 1
797 if self.num_elements < self.size:
798 self._buffer[self.num_elements] = candidate
799 self.num_elements += 1
800 return
801 idx = np.random.choice(self._num_candidates)
802 if idx < self.size:
803 self._buffer[idx] = candidate
804
805 def insert_all(self, candidates):
806 """Consider all `candidates` for inclusion in this sampling buffer."""
807 for candidate in candidates:
808 self.insert(candidate)
809
810 def num_available_spaces(self):
811 """The number of freely available spaces in this buffer."""
812 return self.size - self.num_elements
