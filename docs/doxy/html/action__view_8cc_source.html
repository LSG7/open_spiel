24std::vector<std::vector<Action>> CollectActions(const State& state) {
25 std::vector<std::vector<Action>> legal_actions;
26 if (state.IsSimultaneousNode()) {
27 legal_actions = std::vector<std::vector<Action>>(state.NumPlayers());
28 for (int i = 0; i < state.NumPlayers(); ++i) {
29 legal_actions[i] = state.LegalActions(i);
30 }
31 } else {
32 legal_actions = std::vector<std::vector<Action>>{state.LegalActions()};
33 }
34 return legal_actions;
35}

78FixedActions ActionView::fixed_action(Player player, int action_index) const {
79 SPIEL_CHECK_EQ(current_player, kSimultaneousPlayerId);
80 int prod_after = 1;
81 for (int pl = player + 1; pl < legal_actions.size(); pl++) {
82 const std::vector<Action>& actions = legal_actions[pl];
83 if (!actions.empty()) prod_after *= actions.size();
84 }
85 int prod_before = 1;
86 for (int pl = 0; pl < player; pl++) {
87 const std::vector<Action>& actions = legal_actions[pl];
88 if (!actions.empty()) prod_before *= actions.size();
89 }
90 int num_actions = legal_actions[player].size();
91 return FixedActions{action_index, num_actions, prod_before, prod_after};
92}

105// This essentially imitates a generator that uses a nested for loop:
106//
107// for i in range(prod_after):
108// for j in range(prod_before):
109// yield prod_before * (fixed_action + i * num_actions) + j
110FixedActionsIterator& FixedActionsIterator::operator++() {
111 if (j_ + 1 < prod_before_) {
112 ++j_;
113 return *this;
114 } else {
115 j_ = 0;
116 ++i_;
117 SPIEL_CHECK_LE(i_, prod_after_);
118 return *this;
119 }
120}
121Action FixedActionsIterator::operator*() const {
122 return prod_before_ * (fixed_action_ + i_ * num_actions_) + j_;
123}
