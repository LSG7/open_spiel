34void PrintColumns(const std::vector<std::string> &strings) {
35 std::string padding_string(kPadding, ' ');
36
37 int longest_string_length = 0;
38 for (const std::string &string : strings) {
39 if (string.length() > longest_string_length) {
40 longest_string_length = string.length();
41 }
42 }
43
44 int max_columns = (kMaxWidth - 1) / (longest_string_length + 2 * kPadding);
45 int rows = ceil((float)strings.size() / (float)max_columns);
46 int columns = ceil((float)strings.size() / (float)rows);
47 for (int row = 0; row < rows; ++row) {
48 for (int column = 0; column < columns; ++column) {
49 int index = row + column * rows;
50 if (index < strings.size()) {
51 std::cout << std::left << std::setw(longest_string_length + kPadding)
52 << padding_string << strings[index];
53 }
54 }
55 std::cout << std::endl;
56 }
57}

61Action HumanBot::Step(const State &state) {
62 std::vector<Action> legal_actions = state.LegalActions(state.CurrentPlayer());
63
64 if (legal_actions.empty()) {
65 return kInvalidAction;
66 }
67
68 absl::flat_hash_map<std::string, Action> action_map;
69 for (Action legal_action : legal_actions) {
70 action_map[state.ActionToString(legal_action)] = legal_action;
71 }
72
73 while (true) {
74 Action action;
75 std::string action_string = "";
76
77 std::cout << "Choose an action (empty to print legal actions): ";
78 std::getline(std::cin, action_string);
79
80 // Print the legal actions if no action is given.
81 if (action_string.empty()) {
82 std::cout << "Legal action(s):" << std::endl;
83
84 std::vector<std::string> legal_action_strings;
85 std::vector<std::pair<std::string, Action>> sorted_action_map(
86 action_map.begin(), action_map.end());
87
88 std::sort(sorted_action_map.begin(), sorted_action_map.end(),
89 [](const auto &left, const auto &right) {
90 return left.first < right.first;
91 });
92
93 int longest_action_length = 0;
94 for (const Action &legal_action : legal_actions) {
95 int action_length = std::to_string(legal_action).length();
96 if (action_length > longest_action_length) {
97 longest_action_length = action_length;
98 }
99 }
100
101 for (const auto &string_action_pair : sorted_action_map) {
102 std::string action_string = string_action_pair.first;
103 std::string action_int_string =
104 std::to_string(string_action_pair.second);
105 std::string action_padding(
106 longest_action_length - action_int_string.length(), ' ');
107 legal_action_strings.push_back(absl::StrCat(
108 action_padding, action_int_string, ": ", action_string));
109 }
110 PrintColumns(legal_action_strings);
111 continue;
112 }
113
114 // Return the action if a valid string is given.
115 if (action_map.find(action_string) != action_map.end()) {
116 return action_map[action_string];
117 }
118
119 // Return the action if a valid integer is given.
120 bool parse_succeeded = absl::SimpleAtoi(action_string, &action);
121 if (!parse_succeeded) {
122 std::cout << "Could not parse the action: " << action_string << std::endl;
123 continue;
124 }
125
126 for (Action legal_action : legal_actions) {
127 if (action == legal_action) {
128 return action;
129 }
130 }
131
132 // The input was not valid.
133 std::cout << "Illegal action selected: " << action_string << std::endl;
134 }
135}
