24def _transitions(state, policies):
25 """Returns iterator over (action, prob) from the given state."""
26 if state.is_chance_node():
27 return state.chance_outcomes()
28 elif state.is_simultaneous_node():
29 return policy.joint_action_probabilities(state, policies)
30 else:
31 player = state.current_player()
32 return policies[player].action_probabilities(state).items()

35def policy_value(state,
36 policies: Union[List[policy.Policy], policy.Policy],
37 probability_threshold: float = 0):
38 """Returns the expected values for the state for players following `policies`.
39
40 Computes the expected value of the`state` for each player, assuming player `i`
41 follows the policy given in `policies[i]`.
42
43 Args:
44 state: A `pyspiel.State`.
45 policies: A `list` of `policy.Policy` objects, one per player for sequential
46 games, one policy for simulatenous games.
47 probability_threshold: only sum over entries with prob greater than this
48 (default: 0).
49
50 Returns:
51 A `numpy.array` containing the expected value for each player.
52 """
53 if state.is_terminal():
54 return np.array(state.returns())
55 else:
56 return sum(prob * policy_value(policy.child(state, action), policies)
57 for action, prob in _transitions(state, policies)
58 if prob > probability_threshold)
