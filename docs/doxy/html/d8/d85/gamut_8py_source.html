32 def __init__(self, config_list, java_path='', seed=None):
33 """Ctor. Inits payoff tensor (players x actions x ... np.array).
34
35 Args:
36 config_list: a list or strings alternating between gamut flags and values
37 see http://gamut.stanford.edu/userdoc.pdf for more information
38 e.g., config_list = ['-g', 'CovariantGame', '-players', '6',
39 '-normalize', '-min_payoff', '0',
40 '-max_payoff', '1', '-actions', '5', '-r', '0']
41 java_path: string, java path
42 seed: random seed, some GAMUT games are randomly generated
43 """
44 self.pt = None
45 self.config_list = config_list
46
47 self.seed = seed
48 self.random = np.random.RandomState(seed)
49
50 # parse interval for rho if supplied, e.g., '[-.2,1]'
51 if '-r' in config_list:
52 idx = next(i for i, s in enumerate(config_list) if s == '-r')
53 val = config_list[idx + 1]
54 if not val.isnumeric() and val[0] in '([' and val[-1] in ')]':
55 a, b = val.strip('[]()').split(',')
56 a = float(a)
57 b = float(b)
58 rho = self.random.rand() * (b - a) + a
59 config_list[idx + 1] = str(rho)
60
61 if isinstance(seed, int):
62 self.config_list += ['-random_seed', str(seed)]
63 self.java_path = java_path
64
65 if java_path:
66 generator = pyspiel.GamutGenerator(
67 java_path,
68 'gamut/gamut_main_deploy.jar')
69 else: # use default java path as specified by pyspiel
70 generator = pyspiel.GamutGenerator(
71 'gamut.jar')
72 self.game = generator.generate_game(config_list)
73
74 def num_players(self):
75 return self.game.num_players()
76
77 def num_strategies(self):
78 return [self.game.num_distinct_actions()] * self.num_players()
79
80 def payoff_tensor(self):
81 if self.pt is None:
82 pt = np.asarray(game_payoffs_array(self.game))
83 self.pt = pt - self.game.min_utility()
84 return self.pt
85
86 def get_payoffs_for_strategies(self, policies):
87 """Return vector of payoffs for all players given list of strategies.
88
89 Args:
90 policies: list of integers indexing strategies for each player
91 Returns:
92 np.array (length num players) of payoffs
93 """
94 state = self.game.new_initial_state()
95 state.apply_actions(policies)
96 return np.asarray(state.returns()) - self.game.min_utility()
