63// All possible contracts.
64inline constexpr int kNumContracts =
65 kNumBids * kNumPlayers * kNumDoubleStates + 1;
66constexpr std::array<Contract, kNumContracts> AllContracts() {
67 std::array<Contract, kNumContracts> contracts;
68 int i = 0;
69 contracts[i++] = Contract();
70 for (int level : {1, 2, 3, 4, 5, 6, 7}) {
71 for (Denomination trumps :
72 {kClubs, kDiamonds, kHearts, kSpades, kNoTrump}) {
73 for (int declarer = 0; declarer < kNumPlayers; ++declarer) {
74 for (DoubleStatus double_status : {kUndoubled, kDoubled, kRedoubled}) {
75 contracts[i++] = Contract{level, trumps, double_status, declarer};
76 }
77 }
78 }
79 }
80 return contracts;
81}
82inline constexpr std::array<Contract, kNumContracts> kAllContracts =
83 AllContracts();
