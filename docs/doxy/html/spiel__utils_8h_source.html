101// SpielStrOut(out, a, b, c) is equivalent to:
102// out << a << b << c;
103// It is used to enable SpielStrCat, below.
104template <typename Out, typename T>
105void SpielStrOut(Out& out, const T& arg) {
106 out << arg;
107}
108
109template <typename Out, typename T, typename... Args>
110void SpielStrOut(Out& out, const T& arg1, Args&&... args) {
111 out << arg1;
112 SpielStrOut(out, std::forward<Args>(args)...);
113}

115// Builds a string from pieces:
116//
117// SpielStrCat(1, " + ", 1, " = ", 2) --> "1 + 1 = 2"
118//
119// Converting the parameters to strings is done using the stream operator<<.
120// This is only kept around to be used in the SPIEL_CHECK_* macros and should
121// not be called by any code outside of this file. Prefer absl::StrCat instead.
122// It is kept here due to support for more types, including char.
123template <typename... Args>
124std::string SpielStrCat(Args&&... args) {
125 std::ostringstream out;
126 SpielStrOut(out, std::forward<Args>(args)...);
127 return out.str();
128}

182// Returns whether the absolute difference between floating point values a and
183// b is less than or equal to.
184template <typename T>
185bool Near(T a, T b) {
186 static_assert(std::is_floating_point<T>::value,
187 "Near() is only for floating point args.");
188 return fabs(a - b) <= FloatingPointDefaultTolerance();
189}
190
191// Returns whether |a - b| <= epsilon.
192template <typename T>
193bool Near(T a, T b, T epsilon) {
194 static_assert(std::is_floating_point<T>::value,
195 "Near() is only for floating point args.");
196 return fabs(a - b) <= epsilon;
197}

199template <typename T>
200bool AllNear(const std::vector<T>& vector1, const std::vector<T>& vector2,
201 T epsilon) {
202 if (vector1.size() != vector2.size()) {
203 return false;
204 }
205 for (int i = 0; i < vector1.size(); ++i) {
206 if (!Near(vector1[i], vector2[i], epsilon)) {
207 return false;
208 }
209 }
210 return true;
211}

218// Macros to check for error conditions.
219// These trigger SpielFatalError if the condition is violated.
220// These macros are always executed. If you want to use checks
221// only for debugging, use SPIEL_DCHECK_*
222
223#define SPIEL_CHECK_OP(x_exp, op, y_exp) \
224 do { \
225 auto x = x_exp; \
226 auto y = y_exp; \
227 if (!((x)op(y))) \
228 open_spiel::SpielFatalError(open_spiel::internal::SpielStrCat( \
229 __FILE__, ":", __LINE__, " ", #x_exp " " #op " " #y_exp, \
230 "\n" #x_exp, " = ", x, ", " #y_exp " = ", y)); \
231 } while (false)
232
233#define SPIEL_CHECK_FN2(x_exp, y_exp, fn) \
234 do { \
235 auto x = x_exp; \
236 auto y = y_exp; \
237 if (!fn(x, y)) \
238 open_spiel::SpielFatalError(open_spiel::internal::SpielStrCat( \
239 __FILE__, ":", __LINE__, " ", #fn "(" #x_exp ", " #y_exp ")\n", \
240 #x_exp " = ", x, ", " #y_exp " = ", y)); \
241 } while (false)
242
243#define SPIEL_CHECK_FN3(x_exp, y_exp, z_exp, fn) \
244 do { \
245 auto x = x_exp; \
246 auto y = y_exp; \
247 auto z = z_exp; \
248 if (!fn(x, y, z)) \
249 open_spiel::SpielFatalError(open_spiel::internal::SpielStrCat( \
250 __FILE__, ":", __LINE__, " ", \
251 #fn "(" #x_exp ", " #y_exp ", " #z_exp ")\n", #x_exp " = ", x, \
252 ", " #y_exp " = ", y, ", " #z_exp " = ", z)); \
253 } while (false)
254
255#define SPIEL_CHECK_GE(x, y) SPIEL_CHECK_OP(x, >=, y)
256#define SPIEL_CHECK_GT(x, y) SPIEL_CHECK_OP(x, >, y)
257#define SPIEL_CHECK_LE(x, y) SPIEL_CHECK_OP(x, <=, y)
258#define SPIEL_CHECK_LT(x, y) SPIEL_CHECK_OP(x, <, y)
259#define SPIEL_CHECK_EQ(x, y) SPIEL_CHECK_OP(x, ==, y)
260#define SPIEL_CHECK_NE(x, y) SPIEL_CHECK_OP(x, !=, y)
261#define SPIEL_CHECK_PROB(x) \
262 SPIEL_CHECK_GE(x, 0); \
263 SPIEL_CHECK_LE(x, 1); \
264 SPIEL_CHECK_FALSE(std::isnan(x) || std::isinf(x))
265#define SPIEL_CHECK_PROB_TOLERANCE(x, tol) \
266 SPIEL_CHECK_GE(x, -(tol)); \
267 SPIEL_CHECK_LE(x, 1.0 + (tol)); \
268 SPIEL_CHECK_FALSE(std::isnan(x) || std::isinf(x))

359class UniformProbabilitySampler {

362 : seed_(seed), rng_(seed_), dist_(min, max), min_(min), max_(max) {}
363
364 UniformProbabilitySampler(double min = 0., double max = 1.)
365 : rng_(seed_), dist_(min, max), min_(min), max_(max) {}
366
367 // When copying, we reinitialize the sampler to have the initial seed.

381 absl::uniform_real_distribution<double> dist_;
382
383 const double min_;
384 const double max_;
385};

387// Utility functions intended to be used for casting
388// from a Base class to a Derived subclass.
389// These functions handle various use cases, such as pointers and const
390// references. For shared or unique pointers you can get the underlying pointer.
391// When you use debug mode, a more expensive dynamic_cast is used and it checks
392// whether the casting has been successful. In optimized builds only static_cast
393// is used when possible.
394
395// use like this: down_cast<T*>(foo);
396template <typename To, typename From>
397inline To down_cast(From* f) {
398#if !defined(NDEBUG)
399 if (f != nullptr && dynamic_cast<To>(f) == nullptr) {
400 std::string from = typeid(From).name();
401 std::string to = typeid(From).name();
402 SpielFatalError(
403 absl::StrCat("Cast failure: could not cast a pointer from '", from,
404 "' to '", to, "'"));
405 }
406#endif
407 return static_cast<To>(f);
408}
409
410// use like this: down_cast<T&>(foo);
411template <typename To, typename From>
412inline To down_cast(From& f) {
413 typedef typename std::remove_reference<To>::type* ToAsPointer;
414#if !defined(NDEBUG)
415 if (dynamic_cast<ToAsPointer>(&f) == nullptr) {
416 std::string from = typeid(From).name();
417 std::string to = typeid(From).name();
418 SpielFatalError(
419 absl::StrCat("Cast failure: could not cast a reference from '", from,
420 "' to '", to, "'"));
421 }
422#endif
423 return *static_cast<ToAsPointer>(&f);
424}
