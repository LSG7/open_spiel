30class GreedyPolicyTest(parameterized.TestCase):
31
32 @parameterized.named_parameters(('python', 'python_mfg_crowd_modelling'),
33 ('cpp', 'mfg_crowd_modelling'))
34 def test_greedy(self, name):
35 """Check if the greedy policy works as expected.
36
37 The test checks that a greedy policy with respect to an optimal value is
38 an optimal policy.
39
40 Args:
41 name: Name of the game.
42 """
43 game = pyspiel.load_game(name)
44 uniform_policy = policy.UniformRandomPolicy(game)
45 dist = distribution.DistributionPolicy(game, uniform_policy)
46 br_value = best_response_value.BestResponse(
47 game, dist, value.TabularValueFunction(game))
48 br_val = br_value(game.new_initial_state())
49
50 greedy_pi = greedy_policy.GreedyPolicy(game, None, br_value)
51 greedy_pi = greedy_pi.to_tabular()
52 pybr_value = policy_value.PolicyValue(game, dist, greedy_pi,
53 value.TabularValueFunction(game))
54 pybr_val = pybr_value(game.new_initial_state())
55 self.assertAlmostEqual(br_val, pybr_val)
