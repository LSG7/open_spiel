80 // A call is one of Pass, Double, Redouble, or a bid.
81 // Bids are a combination of a number of tricks (level + 6) and denomination
82 // (trump suit or no-trumps).
83 // The calls are represented in sequence: Pass, Dbl, RDbl, 1C, 1D, 1H, 1S, etc.
84 enum Calls { kPass = 0, kDouble = 1, kRedouble = 2 };
85 inline constexpr int kFirstBid = kRedouble + 1;
86 int Bid(int level, Denomination denomination) {
87  return (level - 1) * kNumDenominations + denomination + kFirstBid;
88 }
89 int BidLevel(int bid) { return 1 + (bid - kNumOtherCalls) / kNumDenominations; }
90 Denomination BidSuit(int bid) {
91  return Denomination((bid - kNumOtherCalls) % kNumDenominations);
92 }
93 
94 // Cards are represented as rank * kNumSuits + suit.
95 Suit CardSuit(int card) { return Suit(card % kNumSuits); }
96 int CardRank(int card) { return card / kNumSuits; }
97 int Card(Suit suit, int rank) {
98  return rank * kNumSuits + static_cast<int>(suit);
99 }

154 std::array<std::string, kNumSuits> FormatHand(
155  int player, bool mark_voids,
156  const std::array<absl::optional<Player>, kNumCards>& deal) {
157  std::array<std::string, kNumSuits> cards;
158  for (int suit = 0; suit < kNumSuits; ++suit) {
159  cards[suit].push_back(kSuitChar[suit]);
160  cards[suit].push_back(' ');
161  bool is_void = true;
162  for (int rank = kNumCardsPerSuit - 1; rank >= 0; --rank) {
163  if (player == deal[Card(Suit(suit), rank)]) {
164  cards[suit].push_back(kRankChar[rank]);
165  is_void = false;
166  }
167  }
168  if (is_void && mark_voids) absl::StrAppend(&cards[suit], "none");
169  }
170  return cards;
171 }

198 std::string BridgeState::FormatDeal() const {
199  std::array<std::array<std::string, kNumSuits>, kNumPlayers> cards;
200  if (IsTerminal()) {
201  // Include all cards in the terminal state to make reviewing the deal easier
202  auto deal = OriginalDeal();
203  for (auto player : {kNorth, kEast, kSouth, kWest}) {
204  cards[player] = FormatHand(player, /*mark_voids=*/false, deal);
205  }
206  } else {
207  for (auto player : {kNorth, kEast, kSouth, kWest}) {
208  cards[player] = FormatHand(player, /*mark_voids=*/false, holder_);
209  }
210  }
211  constexpr int kColumnWidth = 8;
212  std::string padding(kColumnWidth, ' ');
213  std::string rv;
214  for (int suit = kNumSuits - 1; suit >= 0; --suit)
215  absl::StrAppend(&rv, padding, cards[kNorth][suit], "\n");
216  for (int suit = kNumSuits - 1; suit >= 0; --suit)
217  absl::StrAppend(&rv, absl::StrFormat("%-8s", cards[kWest][suit]), padding,
218  cards[kEast][suit], "\n");
219  for (int suit = kNumSuits - 1; suit >= 0; --suit)
220  absl::StrAppend(&rv, padding, cards[kSouth][suit], "\n");
221  return rv;
222 }

231 std::string BridgeState::FormatAuction(bool trailing_query) const {
232  SPIEL_CHECK_GT(history_.size(), kNumCards);
233  std::string rv = "\nWest North East South\n ";
234  for (int i = kNumCards; i < history_.size() - num_cards_played_; ++i) {
235  if (i % kNumPlayers == kNumPlayers - 1) rv.push_back('\n');
236  absl::StrAppend(
237  &rv, absl::StrFormat(
238  "%-6s", BidString(history_[i].action - kBiddingActionBase)));
239  }
240  if (trailing_query) {
241  if ((history_.size() - num_cards_played_) % kNumPlayers == kNumPlayers - 1)
242  rv.push_back('\n');
243  rv.push_back('?');
244  }
245  return rv;

248 std::string BridgeState::FormatPlay() const {
249  SPIEL_CHECK_GT(num_cards_played_, 0);
250  std::string rv = "\n\nN E S W N E S";
251  Trick trick{kInvalidPlayer, kNoTrump, 0};
252  Player player = (1 + contract_.declarer) % kNumPlayers;
253  for (int i = 0; i < num_cards_played_; ++i) {
254  if (i % kNumPlayers == 0) {
255  if (i > 0) player = trick.Winner();
256  absl::StrAppend(&rv, "\n", std::string(3 * player, ' '));
257  } else {
258  player = (1 + player) % kNumPlayers;
259  }
260  const int card = history_[history_.size() - num_cards_played_ + i].action;
261  if (i % kNumPlayers == 0) {
262  trick = Trick(player, contract_.trumps, card);
263  } else {
264  trick.Play(player, card);
265  }
266  absl::StrAppend(&rv, CardString(card), " ");
267  }
268  absl::StrAppend(&rv, "\n\nDeclarer tricks: ", num_declarer_tricks_);
269  return rv;
270 }

289 void BridgeState::WriteObservationTensor(Player player,
290  absl::Span<float> values) const {
291  SPIEL_CHECK_GE(player, 0);
292  SPIEL_CHECK_LT(player, num_players_);
293 
294  std::fill(values.begin(), values.end(), 0.0);
295  if (phase_ == Phase::kDeal) return;
296  int partnership = Partnership(player);
297  auto ptr = values.begin();
298  if (num_cards_played_ > 0) {
299  // Observation for play phase
300  if (phase_ == Phase::kPlay) ptr[2] = 1;
301  ptr += kNumObservationTypes;
302 
303  // Contract
304  ptr[contract_.level - 1] = 1;
305  ptr += kNumBidLevels;
306 
307  // Trump suit
308  ptr[contract_.trumps] = 1;
309  ptr += kNumDenominations;
310 
311  // Double status
312  *ptr++ = contract_.double_status == DoubleStatus::kUndoubled;
313  *ptr++ = contract_.double_status == DoubleStatus::kDoubled;
314  *ptr++ = contract_.double_status == DoubleStatus::kRedoubled;
315 
316  // Identity of the declarer.
317  ptr[(contract_.declarer + kNumPlayers - player) % kNumPlayers] = 1;
318  ptr += kNumPlayers;
319 
320  // Vulnerability.
321  ptr[is_vulnerable_[Partnership(contract_.declarer)]] = 1.0;
322  ptr += kNumVulnerabilities;
323 
324  // Our remaining cards.
325  for (int i = 0; i < kNumCards; ++i)
326  if (holder_[i] == player) ptr[i] = 1;
327  ptr += kNumCards;
328 
329  // Dummy's remaining cards.
330  const int dummy = Partner(contract_.declarer);
331  for (int i = 0; i < kNumCards; ++i)
332  if (holder_[i] == dummy) ptr[i] = 1;
333  ptr += kNumCards;
334 
335  // Indexing into history for recent tricks.
336  int current_trick = num_cards_played_ / kNumPlayers;
337  int this_trick_cards_played = num_cards_played_ % kNumPlayers;
338  int this_trick_start = history_.size() - this_trick_cards_played;
339 
340  // Previous trick.
341  if (current_trick > 0) {
342  int leader = tricks_[current_trick - 1].Leader();
343  for (int i = 0; i < kNumPlayers; ++i) {
344  int card = history_[this_trick_start - kNumPlayers + i].action;
345  int relative_player = (i + leader + kNumPlayers - player) % kNumPlayers;
346  ptr[relative_player * kNumCards + card] = 1;
347  }
348  }
349  ptr += kNumPlayers * kNumCards;
350 
351  // Current trick
352  if (phase_ != Phase::kGameOver) {
353  int leader = tricks_[current_trick].Leader();
354  for (int i = 0; i < this_trick_cards_played; ++i) {
355  int card = history_[this_trick_start + i].action;
356  int relative_player = (i + leader + kNumPlayers - player) % kNumPlayers;
357  ptr[relative_player * kNumCards + card] = 1;
358  }
359  }
360  ptr += kNumPlayers * kNumCards;
361 
362  // Number of tricks taken by each side.
363  ptr[num_declarer_tricks_] = 1;
364  ptr += kNumTricks;
365  ptr[num_cards_played_ / 4 - num_declarer_tricks_] = 1;
366  ptr += kNumTricks;
367  SPIEL_CHECK_EQ(std::distance(values.begin(), ptr),
368  kPlayTensorSize + kNumObservationTypes);
369  SPIEL_CHECK_LE(std::distance(values.begin(), ptr), values.size());
370  } else {
371  // Observation for auction or opening lead.
372  ptr[phase_ == Phase::kPlay ? 1 : 0] = 1;
373  ptr += kNumObservationTypes;
374  ptr[is_vulnerable_[partnership]] = 1;
375  ptr += kNumVulnerabilities;
376  ptr[is_vulnerable_[1 - partnership]] = 1;
377  ptr += kNumVulnerabilities;
378  int last_bid = 0;
379  for (int i = kNumCards; i < history_.size(); ++i) {
380  int this_call = history_[i].action - kBiddingActionBase;
381  int relative_bidder = (i + kNumPlayers - player) % kNumPlayers;
382  if (last_bid == 0 && this_call == kPass) ptr[relative_bidder] = 1;
383  if (this_call == kDouble) {
384  ptr[kNumPlayers + (last_bid - kFirstBid) * kNumPlayers * 3 +
385  kNumPlayers + relative_bidder] = 1;
386  } else if (this_call == kRedouble) {
387  ptr[kNumPlayers + (last_bid - kFirstBid) * kNumPlayers * 3 +
388  kNumPlayers * 2 + relative_bidder] = 1;
389  } else if (this_call != kPass) {
390  last_bid = this_call;
391  ptr[kNumPlayers + (last_bid - kFirstBid) * kNumPlayers * 3 +
392  relative_bidder] = 1;
393  }
394  }
395  ptr += kNumPlayers * (1 + 3 * kNumBids);
396  for (int i = 0; i < kNumCards; ++i)
397  if (holder_[i] == player) ptr[i] = 1;
398  ptr += kNumCards;
399  SPIEL_CHECK_EQ(std::distance(values.begin(), ptr),
400  kAuctionTensorSize + kNumObservationTypes);
401  SPIEL_CHECK_LE(std::distance(values.begin(), ptr), values.size());
402  }
403 }

405 std::vector<double> BridgeState::PublicObservationTensor() const {
406  SPIEL_CHECK_TRUE(phase_ == Phase::kAuction);
407  std::vector<double> rv(kPublicInfoTensorSize);
408  auto ptr = rv.begin();
409  ptr[is_vulnerable_[0]] = 1;
410  ptr += kNumVulnerabilities;
411  ptr[is_vulnerable_[1]] = 1;
412  ptr += kNumVulnerabilities;
413  auto bidding = ptr + 2 * kNumPlayers; // initial and recent passes
414  int last_bid = 0;
415  for (int i = kNumCards; i < history_.size(); ++i) {
416  const int player = i % kNumPlayers;
417  const int this_call = history_[i].action - kBiddingActionBase;
418  if (this_call == kPass) {
419  if (last_bid == 0) ptr[player] = 1; // Leading passes
420  ptr[kNumPlayers + player] = 1; // Trailing passes
421  } else {
422  // Call is a non-Pass, so clear the trailing pass markers.
423  for (int i = 0; i < kNumPlayers; ++i) ptr[kNumPlayers + i] = 0;
424  if (this_call == kDouble) {
425  auto base = bidding + (last_bid - kFirstBid) * kNumPlayers * 3;
426  base[kNumPlayers + player] = 1;
427  } else if (this_call == kRedouble) {
428  auto base = bidding + (last_bid - kFirstBid) * kNumPlayers * 3;
429  base[kNumPlayers * 2 + player] = 1;
430  } else {
431  last_bid = this_call;
432  auto base = bidding + (last_bid - kFirstBid) * kNumPlayers * 3;
433  base[player] = 1;
434  }
435  }
436  }
437  return rv;
438 }

454 void BridgeState::ComputeDoubleDummyTricks() const {
455  if (!double_dummy_results_.has_value()) {
456  absl::MutexLock lock(&dds_mutex); // TODO(author11) Make DDS code thread-safe
457  double_dummy_results_ = ddTableResults{};
458  ddTableDeal dd_table_deal{};
459  for (int suit = 0; suit < kNumSuits; ++suit) {
460  for (int rank = 0; rank < kNumCardsPerSuit; ++rank) {
461  const int player = holder_[Card(Suit(suit), rank)].value();
462  dd_table_deal.cards[player][suit] += 1 << (2 + rank);
463  }
464  }
465  DDS_EXTERNAL(SetMaxThreads)(0);
466  const int return_code = DDS_EXTERNAL(CalcDDtable)(
467  dd_table_deal, &double_dummy_results_.value());
468  if (return_code != RETURN_NO_FAULT) {
469  char error_message[80];
470  DDS_EXTERNAL(ErrorMessage)(return_code, error_message);
471  SpielFatalError(absl::StrCat("double_dummy_solver:", error_message));
472  }
473  }
474  ComputeScoreByContract();
475 }

477 std::vector<int> BridgeState::ScoreForContracts(
478  int player, const std::vector<int>& contracts) const {
479  // Storage for the number of tricks.
480  std::array<std::array<int, kNumPlayers>, kNumDenominations> dd_tricks;
481 
482  if (double_dummy_results_.has_value()) {
483  // If we have already computed double-dummy results, use them.
484  for (int declarer = 0; declarer < kNumPlayers; ++declarer) {
485  for (int trumps = 0; trumps < kNumDenominations; ++trumps) {
486  dd_tricks[trumps][declarer] =
487  double_dummy_results_->resTable[trumps][declarer];
488  }
489  }
490  } else {
491  {
492  // This performs some sort of global initialization; unclear
493  // exactly what.
494  absl::MutexLock lock(&dds_mutex);
495  DDS_EXTERNAL(SetMaxThreads)(0);
496  }
497 
498  // Working storage for DD calculation.
499  auto thread_data = std::make_unique<ThreadData>();
500  auto transposition_table = std::make_unique<TransTableL>();
501  transposition_table->SetMemoryDefault(95); // megabytes
502  transposition_table->SetMemoryMaximum(160); // megabytes
503  transposition_table->MakeTT();
504  thread_data->transTable = transposition_table.get();
505 
506  // Which trump suits do we need to handle?
507  std::set<int> suits;
508  for (auto index : contracts) {
509  const auto& contract = kAllContracts[index];
510  if (contract.level > 0) suits.emplace(contract.trumps);
511  }
512  // Build the deal
513  ::deal dl{};
514  for (int suit = 0; suit < kNumSuits; ++suit) {
515  for (int rank = 0; rank < kNumCardsPerSuit; ++rank) {
516  const int player = holder_[Card(Suit(suit), rank)].value();
517  dl.remainCards[player][suit] += 1 << (2 + rank);
518  }
519  }
520  for (int k = 0; k <= 2; k++) {
521  dl.currentTrickRank[k] = 0;
522  dl.currentTrickSuit[k] = 0;
523  }
524 
525  // Analyze for each trump suit.
526  for (int suit : suits) {
527  dl.trump = suit;
528  transposition_table->ResetMemory(TT_RESET_NEW_TRUMP);
529 
530  // Assemble the declarers we need to consider.
531  std::set<int> declarers;
532  for (auto index : contracts) {
533  const auto& contract = kAllContracts[index];
534  if (contract.level > 0 && contract.trumps == suit)
535  declarers.emplace(contract.declarer);
536  }
537 
538  // Analyze the deal for each declarer.
539  absl::optional<Player> first_declarer;
540  absl::optional<int> first_tricks;
541  for (int declarer : declarers) {
542  ::futureTricks fut;
543  dl.first = (declarer + 1) % kNumPlayers;
544  if (!first_declarer.has_value()) {
545  // First time we're calculating this trump suit.
546  const int return_code = SolveBoardInternal(
547  thread_data.get(), dl,
548  /*target=*/-1, // Find max number of tricks
549  /*solutions=*/1, // Just the tricks (no card-by-card result)
550  /*mode=*/2, // Unclear
551  &fut // Output
552  );
553  if (return_code != RETURN_NO_FAULT) {
554  char error_message[80];
555  DDS_EXTERNAL(ErrorMessage)(return_code, error_message);
556  SpielFatalError(
557  absl::StrCat("double_dummy_solver:", error_message));
558  }
559  dd_tricks[suit][declarer] = 13 - fut.score[0];
560  first_declarer = declarer;
561  first_tricks = 13 - fut.score[0];
562  } else {
563  // Reuse data from last time.
564  const int hint = Partnership(declarer) == Partnership(*first_declarer)
565  ? *first_tricks
566  : 13 - *first_tricks;
567  const int return_code =
568  SolveSameBoard(thread_data.get(), dl, &fut, hint);
569  if (return_code != RETURN_NO_FAULT) {
570  char error_message[80];
571  DDS_EXTERNAL(ErrorMessage)(return_code, error_message);
572  SpielFatalError(
573  absl::StrCat("double_dummy_solver:", error_message));
574  }
575  dd_tricks[suit][declarer] = 13 - fut.score[0];
576  }
577  }
578  }
579  }
580 
581  // Compute the scores.
582  std::vector<int> scores;
583  scores.reserve(contracts.size());
584  for (int contract_index : contracts) {
585  const Contract& contract = kAllContracts[contract_index];

592  ? declarer_score

594  }

696 void BridgeState::ApplyBiddingAction(int call) {
697  // Track the number of consecutive passes since the last bid (if any).
698  if (call == kPass) {
699  ++num_passes_;
700  } else {
701  num_passes_ = 0;
702  }
703 
704  auto partnership = Partnership(current_player_);
705  if (call == kDouble) {
706  SPIEL_CHECK_NE(Partnership(contract_.declarer), partnership);
707  SPIEL_CHECK_EQ(contract_.double_status, kUndoubled);
708  SPIEL_CHECK_GT(contract_.level, 0);
709  possible_contracts_[contract_.Index()] = false;
710  contract_.double_status = kDoubled;
711  } else if (call == kRedouble) {
712  SPIEL_CHECK_EQ(Partnership(contract_.declarer), partnership);
713  SPIEL_CHECK_EQ(contract_.double_status, kDoubled);
714  possible_contracts_[contract_.Index()] = false;
715  contract_.double_status = kRedoubled;
716  } else if (call == kPass) {
717  if (num_passes_ == 4) {
718  // Four consecutive passes can only happen if no-one makes a bid.
719  // The hand is then over, and each side scores zero points.
720  phase_ = Phase::kGameOver;
721  possible_contracts_.fill(false);
722  possible_contracts_[0] = true;
723  } else if (num_passes_ == 3 && contract_.level > 0) {
724  // After there has been a bid, three consecutive passes end the auction.
725  possible_contracts_.fill(false);
726  possible_contracts_[contract_.Index()] = true;
727  if (use_double_dummy_result_) {
728  SPIEL_CHECK_TRUE(double_dummy_results_.has_value());
729  phase_ = Phase::kGameOver;
730  num_declarer_tricks_ =
731  double_dummy_results_
732  ->resTable[contract_.trumps][contract_.declarer];
733  ScoreUp();
734  } else {
735  phase_ = Phase::kPlay;
736  current_player_ = (contract_.declarer + 1) % kNumPlayers;
737  return;
738  }
739  }
740  } else {
741  // A bid was made.
742  SPIEL_CHECK_TRUE((BidLevel(call) > contract_.level) ||
743  (BidLevel(call) == contract_.level &&
744  BidSuit(call) > contract_.trumps));
745  contract_.level = BidLevel(call);
746  contract_.trumps = BidSuit(call);
747  contract_.double_status = kUndoubled;
748  auto partnership = Partnership(current_player_);
749  if (!first_bidder_[partnership][contract_.trumps].has_value()) {
750  // Partner cannot declare this denomination.
751  first_bidder_[partnership][contract_.trumps] = current_player_;
752  const int partner = Partner(current_player_);
753  for (int level = contract_.level + 1; level <= kNumBidLevels; ++level) {
754  for (DoubleStatus double_status : {kUndoubled, kDoubled, kRedoubled}) {
755  possible_contracts_[Contract{level, contract_.trumps, double_status,
756  partner}
757  .Index()] = false;
758  }
759  }
760  }
761  contract_.declarer = first_bidder_[partnership][contract_.trumps].value();
762  // No lower contract is possible.
763  std::fill(
764  possible_contracts_.begin(),
765  possible_contracts_.begin() +
766  Contract{contract_.level, contract_.trumps, kUndoubled, 0}.Index(),
767  false);
768  // No-one else can declare this precise contract.
769  for (int player = 0; player < kNumPlayers; ++player) {
770  if (player != current_player_) {
771  for (DoubleStatus double_status : {kUndoubled, kDoubled, kRedoubled}) {
772  possible_contracts_[Contract{contract_.level, contract_.trumps,
773  double_status, player}
774  .Index()] = false;
775  }
776  }
777  }
778  }
779  current_player_ = (current_player_ + 1) % kNumPlayers;
780 }

782 void BridgeState::ApplyPlayAction(int card) {
783  SPIEL_CHECK_TRUE(holder_[card] == current_player_);
784  holder_[card] = absl::nullopt;
785  if (num_cards_played_ % kNumPlayers == 0) {
786  CurrentTrick() = Trick(current_player_, contract_.trumps, card);
787  } else {
788  CurrentTrick().Play(current_player_, card);
789  }
790  const Player winner = CurrentTrick().Winner();
791  ++num_cards_played_;
792  if (num_cards_played_ % kNumPlayers == 0) {
793  current_player_ = winner;
794  if (Partnership(winner) == Partnership(contract_.declarer))
795  ++num_declarer_tricks_;
796  } else {
797  current_player_ = (current_player_ + 1) % kNumPlayers;
798  }
799  if (num_cards_played_ == kNumCards) {
800  phase_ = Phase::kGameOver;
801  ScoreUp();
802  }
803 }
