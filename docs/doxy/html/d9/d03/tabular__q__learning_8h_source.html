27// Tabular Q-learning algorithm: solves for the optimal action value function
28// of a game.
29// It considers all states with depth at most depth_limit from the
30// initial state (so if depth_limit is 0, only the root is considered).
31// If depth limit is negative, all states are considered.
32//
33// Currently works for sequential 1-player or 2-player zero-sum games.
34//
35// Based on the implementation in Sutton and Barto, Intro to RL. Second Edition,
36// 2018. Section 6.5.
37//
38// Includes implementation of Watkinsâ€™s Q(lambda) which can be found in
39// Sutton and Barto, Intro to RL. Second Edition, 2018. Section 12.10.
40// (E.g. https://www.andrew.cmu.edu/course/10-703/textbook/BartoSutton.pdf)
41// Eligibility traces are implemented with the "accumulate"
42// method (+1 at each iteration) instead of "replace" implementation
43// (doesn't sum trace values). Parameter lambda_ determines the level
44// of bootstraping.
45
46class TabularQLearningSolver {
47 static inline constexpr double kDefaultDepthLimit = -1;
48 static inline constexpr double kDefaultEpsilon = 0.01;
49 static inline constexpr double kDefaultLearningRate = 0.01;
50 static inline constexpr double kDefaultDiscountFactor = 0.99;
51 static inline constexpr double kDefaultLambda = 0;
52
53 public:
54 TabularQLearningSolver(std::shared_ptr<const Game> game);
55
56 TabularQLearningSolver(std::shared_ptr<const Game> game, double depth_limit,
57 double epsilon, double learning_rate,
58 double discount_factor, double lambda);
59
60 void RunIteration();
61
62 const absl::flat_hash_map<std::pair<std::string, Action>, double>&
63 GetQValueTable() const;
64
65 private:
66 // Given a player and a state, gets the best possible action from this state
67 Action GetBestAction(const State& state, double min_utility);
68
69 // Given a state, gets the best possible action value from this state
70 double GetBestActionValue(const State& state, double min_utility);
71
72 // Given a player and a state, gets the action, sampled from an epsilon-greedy
73 // policy. Returns <action, chosen_uniformly> where the second element
74 // indicates whether an action was chosen uniformly (which occurs with epsilon
75 // chance).
76 std::pair<Action, bool> SampleActionFromEpsilonGreedyPolicy(
77 const State& state, double min_utility);
78
79 // Moves a chance node to the next decision/terminal node by sampling from
80 // the legal actions repeatedly
81 void SampleUntilNextStateOrTerminal(State* state);
82
83 std::shared_ptr<const Game> game_;
84 int depth_limit_;
85 double epsilon_;
86 double learning_rate_;
87 double discount_factor_;
88 double lambda_;
89 std::mt19937 rng_;
90 absl::flat_hash_map<std::pair<std::string, Action>, double> values_;
91 absl::flat_hash_map<std::pair<std::string, Action>, double>
92 eligibility_traces_;
93};
