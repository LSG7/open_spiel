30 std::string Escape(const std::string& input) {
31  std::string out;
32  out.reserve(input.length());
33  for (const char c : input) {
34  switch (c) {
35  case '"': out.append("\\\""); break;
36  case '\\': out.append("\\\\"); break;
37  case '\b': out.append("\\b"); break;
38  case '\f': out.append("\\f"); break;
39  case '\n': out.append("\\n"); break;
40  case '\r': out.append("\\r"); break;
41  case '\t': out.append("\\t"); break;
42  default: out.push_back(c); break;
43  }
44  }
45  return out;
46 }

90 absl::optional<Value> ParseNumber(absl::string_view* str) {
91  size_t valid_double =
92  std::min(str->find_first_not_of("-+.0123456789eE"), str->size());
93  size_t valid_int =
94  std::min(str->find_first_not_of("-0123456789"), str->size());
95  if (valid_double == valid_int) {
96  if (int64_t v; absl::SimpleAtoi(str->substr(0, valid_int), &v)) {
97  str->remove_prefix(valid_int);
98  return Value(v);
99  }
100  } else {
101  if (double v; absl::SimpleAtod(str->substr(0, valid_double), &v)) {
102  str->remove_prefix(valid_double);
103  return Value(v);
104  }
105  }
106  return ParseError("Invalid number", *str);
107 }

109 absl::optional<std::string> ParseString(absl::string_view* str) {
110  if (!ConsumeToken(str, "\"")) {
111  return ParseError("Expected '\"'", *str);
112  }
113  std::string out;
114  bool escape = false;
115  for (auto p = str->begin(); p < str->end(); ++p) {
116  switch (*p) {
117  case '\\':
118  if (escape) {
119  out.push_back('\\');
120  }
121  escape = !escape;
122  break;
123  case '"':
124  if (escape) {
125  out.push_back('"');
126  escape = false;
127  break;
128  } else {
129  str->remove_prefix(p - str->begin() + 1);
130  return out;
131  }
132  default:
133  if (escape) {
134  switch (*p) {
135  case 'b': out.append("\b"); break;
136  case 'f': out.append("\f"); break;
137  case 'n': out.append("\n"); break;
138  case 'r': out.append("\r"); break;
139  case 't': out.append("\t"); break;
140  default: out.push_back(*p); break;
141  }
142  escape = false;
143  } else {
144  out.push_back(*p);
145  }
146  break;
147  }
148  }
149  return ParseError("Unfinished string", *str);
150 }

154 absl::optional<Array> ParseArray(absl::string_view* str) {
155  if (!ConsumeToken(str, "[")) {
156  return ParseError("Expected '['", *str);
157  }
158  Array out;
159  bool first = true;
160  while (!str->empty()) {
161  ConsumeWhitespace(str);
162  if (ConsumeToken(str, "]")) {
163  return out;
164  }
165  if (!first && !ConsumeToken(str, ",")) {
166  return ParseError("Expected ','", *str);
167  }
168  first = false;
169  ConsumeWhitespace(str);
170  absl::optional<Value> v = ParseValue(str);
171  if (!v) {
172  return absl::nullopt;
173  }
174  out.push_back(*v);
175  }
176  return ParseError("Unfinished array", *str);
177 }

179 absl::optional<Object> ParseObject(absl::string_view* str) {
180  if (!ConsumeToken(str, "{")) {
181  return ParseError("Expected '{'", *str);
182  }
183  Object out;
184  bool first = true;
185  while (!str->empty()) {
186  ConsumeWhitespace(str);
187  if (ConsumeToken(str, "}")) {
188  return out;
189  }
190  if (!first && !ConsumeToken(str, ",")) {
191  return ParseError("Expected ','", *str);
192  }
193  first = false;
194  ConsumeWhitespace(str);
195  absl::optional<std::string> key = ParseString(str);
196  if (!key) {
197  return absl::nullopt;
198  }
199  ConsumeWhitespace(str);
200  if (!ConsumeToken(str, ":")) {
201  return ParseError("Expected ':'", *str);
202  }
203  ConsumeWhitespace(str);
204  absl::optional<Value> v = ParseValue(str);
205  if (!v) {
206  return absl::nullopt;
207  }
208  out.emplace(*key, *v);
209  }
210  return ParseError("Unfinished object", *str);
211 }

213 absl::optional<Value> ParseValue(absl::string_view* str) {
214  ConsumeWhitespace(str);
215  if (str->empty()) {
216  return ParseError("Empty string", *str);
217  }
218  switch (str->at(0)) {
219  case '-':
220  case '0':
221  case '1':
222  case '2':
223  case '3':
224  case '4':
225  case '5':
226  case '6':
227  case '7':
228  case '8':
229  case '9': return ParseNumber(str);
230  case 'n': return ParseConstant(str, "null", Null());
231  case 't': return ParseConstant(str, "true", true);
232  case 'f': return ParseConstant(str, "false", false);
233  case '"': return ParseString(str);
234  case '[': return ParseArray(str);
235  case '{': return ParseObject(str);
236  default: return ParseError("Unexpected char: ", *str);
237  }
238 }

291 std::string ToString(const Value& value, bool wrap, int indent) {
292  if (value.IsNull()) {
293  return "null";
294  } else if (value.IsBool()) {
295  return (value.GetBool() ? "true" : "false");
296  } else if (value.IsInt()) {
297  return std::to_string(value.GetInt());
298  } else if (value.IsDouble()) {
299  double v = value.GetDouble();
300  if (std::isfinite(v)) {
301  return std::to_string(v);
302  } else {
303  // It'd be nice to show an error with a path, but at least this is
304  // debuggable by looking at the json. Crashing doesn't tell you where
305  // the problem is.
306  return absl::StrCat("\"", std::to_string(v), "\"");
307  }
308  } else if (value.IsString()) {
309  return absl::StrCat("\"", Escape(value.GetString()), "\"");
310  } else if (value.IsArray()) {
311  return ToString(value.GetArray(), wrap, indent);
312  } else if (value.IsObject()) {
313  return ToString(value.GetObject(), wrap, indent);
314  } else {
315  SpielFatalError("json::ToString is missing a type.");
316  }
317 }
