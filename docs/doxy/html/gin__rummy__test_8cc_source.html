34void MeldTests() {
35 GinRummyUtils utils = GinRummyUtils(kDefaultNumRanks, kDefaultNumSuits,
36 kDefaultHandSize);
37 // There are 185 melds of length between 3 and 5 cards. All melds of
38 // length greater than 5 can be expressed as two or more smaller melds.
39 std::vector<int> full_deck;
40 for (int i = 0; i < utils.num_cards; ++i) full_deck.push_back(i);
41 std::vector<std::vector<int>> all_melds = utils.AllMelds(full_deck);
42 SPIEL_CHECK_EQ(all_melds.size(), kNumMeldActions);
43
44 // Some simple meld tests
45 std::vector<std::string> cards;
46 cards = {"As", "2s", "3s"};
47 SPIEL_CHECK_TRUE(utils.IsSuitMeld(utils.CardStringsToCardInts(cards)));
48 SPIEL_CHECK_FALSE(utils.IsRankMeld(utils.CardStringsToCardInts(cards)));
49 cards = {"As", "Ac", "Ad"};
50 SPIEL_CHECK_TRUE(utils.IsRankMeld(utils.CardStringsToCardInts(cards)));
51 SPIEL_CHECK_FALSE(utils.IsSuitMeld(utils.CardStringsToCardInts(cards)));
52 cards = {"As", "Ac", "Ad", "2s"};
53 SPIEL_CHECK_FALSE(utils.IsRankMeld(utils.CardStringsToCardInts(cards)));
54 SPIEL_CHECK_FALSE(utils.IsSuitMeld(utils.CardStringsToCardInts(cards)));
55
56 // No "around the corner" melds
57 cards = {"As", "2s", "3s", "Ks"};
58 SPIEL_CHECK_FALSE(utils.IsRankMeld(utils.CardStringsToCardInts(cards)));
59 SPIEL_CHECK_FALSE(utils.IsSuitMeld(utils.CardStringsToCardInts(cards)));
60
61 // These cards are represented internally as consecutive ints
62 // but are not a meld.
63 cards = {"Js", "Qs", "Ks", "Ac"};
64 SPIEL_CHECK_FALSE(utils.IsRankMeld(utils.CardStringsToCardInts(cards)));
65 SPIEL_CHECK_FALSE(utils.IsSuitMeld(utils.CardStringsToCardInts(cards)));

70 meld_id = utils.meld_to_int.at(utils.CardStringsToCardInts(cards));

83 SPIEL_CHECK_EQ(meld_id, 149);

87 SPIEL_CHECK_EQ(meld_id, 184);
88 SPIEL_CHECK_EQ(utils.meld_to_int.at(utils.int_to_meld.at(184)), 184);

93 // |Ac |

131// An extremely rare situation, but one that does arise in actual gameplay.
132// Tests both layoff and undercut functionality.
133void GameplayTest1() {
134 GameParameters params;
135 // Modify undercut bonus game parameter as an additional test.
136 params["undercut_bonus"] = GameParameter(20);
137 std::shared_ptr<const open_spiel::Game> game =
138 open_spiel::LoadGame("gin_rummy", params);
139 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
140 std::vector<Action> initial_actions;
141 initial_actions = {11, 4, 5, 6, 21, 22, 23, 12, 25, 38, 1, 14,
142 27, 40, 7, 20, 33, 8, 19, 13, 36, 52, 55, 11};
143 for (auto action : initial_actions) state->ApplyAction(action);
144 std::cout << state->ToString() << std::endl;
145 // Player turn: 0
146 // Phase: Knock
147 //
148 // Player1: Deadwood=49
149 // +--------------------------+
150 // | 2s 8s9s |
151 // |Ac2c 7c8c |
152 // | 2d 8d |
153 // | 2h |
154 // +--------------------------+
155 //
156 // Stock size: 31 Upcard: XX
157 // Discard pile: Qs
158 //
159 // Player0: Deadwood=87
160 // +--------------------------+
161 // | 5s6s7s Ks|
162 // | 9cTcJc Kc|
163 // | Jd Kd|
164 // | |
165 // +--------------------------+
166 //
167 // Player0 has knocked, and after laying melds will have the Jd left for
168 // 10 points. Player 1 has two melds (2's and 8's) with 17 points remaining.
169 // Laying the hand this way gives Player0 a win of 7 points. But there's a
170 // better play! Player1 is not compelled to lay his 8's as a meld. Instead,
171 // Player1 can lay off the 8s9s on the 5s6s7s, and the 7c8c on the 9cTcJc.
172 // This leaves Player1 with only the 8d and Ac as deadwood, for a total of
173 // 9 points, less than the 10 points Player0 knocked with. By breaking the
174 // meld of 8's Player1 wins an undercut!
175
176 // Player0 lays melds.
177 state->ApplyAction(119); // KsKcKd
178 state->ApplyAction(125); // 5s6s7s
179 state->ApplyAction(140); // 9cTcJc
180 state->ApplyAction(54);
181 // Player1 layoffs.
182 std::vector<Action> legal_actions = state->LegalActions();
183 SPIEL_CHECK_TRUE(absl::c_linear_search(legal_actions, 7));
184 state->ApplyAction(7); // Lay off 8s
185 legal_actions = state->LegalActions();
186 SPIEL_CHECK_TRUE(absl::c_linear_search(legal_actions, 8));
187 state->ApplyAction(8); // Lay off 9s
188 legal_actions = state->LegalActions();
189 SPIEL_CHECK_TRUE(absl::c_linear_search(legal_actions, 20));
190 state->ApplyAction(20); // Lay off 8c
191 legal_actions = state->LegalActions();
192 SPIEL_CHECK_TRUE(absl::c_linear_search(legal_actions, 19));
193 state->ApplyAction(19); // Lay off 7c
194 state->ApplyAction(54); // Finished layoffs
195 state->ApplyAction(65); // Lay meld of 2's
196 state->ApplyAction(54);
197 // Player1 wins the difference in deadwood (10 - 9 = 1) and the undercut
198 // bonus which was set to 20, for a total of 21 points.
199 std::vector<double> returns = state->Returns();
200 SPIEL_CHECK_EQ(returns[0], -21);
201 SPIEL_CHECK_EQ(returns[1], 21);
202}

204void GameplayTest2() {
205 GameParameters params;
206 std::shared_ptr<const open_spiel::Game> game =
207 open_spiel::LoadGame("gin_rummy", params);
208 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
209 std::vector<Action> initial_actions;
210 initial_actions = {1, 4, 5, 6, 17, 18, 19, 30, 31, 32, 2, 3,
211 16, 29, 43, 44, 45, 7, 20, 33, 0, 52, 55, 1};
212 for (auto action : initial_actions) state->ApplyAction(action);
213 std::cout << state->ToString() << std::endl;
214 // Player turn: 0
215 // Phase: Knock
216 //
217 // Player1: Deadwood=57
218 // +--------------------------+
219 // | 3s4s 8s |
220 // | 4c 8c |
221 // | 4d 8d |
222 // | 5h6h7h |
223 // +--------------------------+
224 //
225 // Stock size: 31 Upcard: XX
226 // Discard pile:
227 //
228 // Player0: Deadwood=57
229 // +--------------------------+
230 // |As 5s6s7s |
231 // | 5c6c7c |
232 // | 5d6d7d |
233 // | |
234 // +--------------------------+
235 //
236 // Player0 has knocked. There are 6 different melds in Player0's hand.
237 // Because the melds overlap, the first meld layed dictates the remaining
238 // melds than can be layed.
239 // In situations where there is a choice between laying rank melds or suit
240 // melds, it is often advantageous to lay the hand as rank melds, which offer
241 // fewer layoffs. Indeed, here Player0 must lay the hand as three rank melds
242 // to avoid the undercut.
243 std::vector<Action> legal_actions = state->LegalActions();
244 SPIEL_CHECK_EQ(legal_actions.size(), 6);
245 state->ApplyAction(79); // Lay the 5s5c5d
246 legal_actions = state->LegalActions();
247 SPIEL_CHECK_EQ(legal_actions.size(), 2);
248 state->ApplyAction(84); // Lay the 6s6c6d
249 legal_actions = state->LegalActions();
250 SPIEL_CHECK_EQ(legal_actions.size(), 1);
251 state->ApplyAction(89); // Lay the 7s7c7d
252 state->ApplyAction(54);
253 // Player1 can lay off the 5h6h7h, but there's no need, as it's already
254 // a meld.
255 legal_actions = state->LegalActions();
256 SPIEL_CHECK_EQ(legal_actions.size(), 4); // 3 layoffs & kPassAction
257 state->ApplyAction(54);
258 state->ApplyAction(74);
259 state->ApplyAction(94);
260 state->ApplyAction(158);
261 state->ApplyAction(54);
262 // Player0 has 1 deadwood and Player1 has 3, so Player0 scores 2 points.
263 std::vector<double> returns = state->Returns();
264 SPIEL_CHECK_EQ(returns[0], 2);
265 SPIEL_CHECK_EQ(returns[1], -2);
266}

269void GameplayTest3() {
270 GameParameters params;
271 std::shared_ptr<const open_spiel::Game> game =
272 open_spiel::LoadGame("gin_rummy", params);
273 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
274 std::vector<Action> initial_actions;
275 initial_actions = {10, 11, 12, 22, 35, 48, 13, 26, 1, 40, 9, 8,
276 3, 16, 29, 42, 4, 17, 30, 43, 0, 52, 55, 1};
277 for (auto action : initial_actions) state->ApplyAction(action);
278 std::cout << state->ToString() << std::endl;
279 // Player turn: 0
280 // Phase: Knock
281 //
282 // Player1: Deadwood=55
283 // +--------------------------+
284 // | 4s5s 9sTs |
285 // | 4c5c |
286 // | 4d5d |
287 // | 4h5h |
288 // +--------------------------+
289 //
290 // Stock size: 31 Upcard: XX
291 // Discard pile: 2s
292 //
293 // Player0: Deadwood=65
294 // +--------------------------+
295 // |As JsQsKs|
296 // |Ac Tc |
297 // |Ad Td |
298 // | 2h Th |
299 // +--------------------------+
300 //
301 // Player0 has knocked. Player1 will have the opportunity to lay off the Ts.
302 // We want to make sure that after laying off the Ts, Player1 will then
303 // be able to lay off the 9s as well. If the Ts only gets counted as
304 // a layoff on the rank meld of three tens, then the 9s would not lay off.
305
306 // Player0 lays melds
307 state->ApplyAction(59);
308 state->ApplyAction(101);
309 state->ApplyAction(131);

313 SPIEL_CHECK_TRUE(absl::c_linear_search(legal_actions, 9));
314 state->ApplyAction(9);
315 // Assert Player1 can lay off the 9s
316 legal_actions = state->LegalActions();
317 SPIEL_CHECK_TRUE(absl::c_linear_search(legal_actions, 8));
318 state->ApplyAction(8);
319 // Player1 completes the hand and wins an undercut
320 state->ApplyAction(54);
321 state->ApplyAction(75);
322 state->ApplyAction(80);
323 state->ApplyAction(54);
324 std::vector<double> returns = state->Returns();
325 SPIEL_CHECK_EQ(returns[0], -27);
326 SPIEL_CHECK_EQ(returns[1], 27);

332 GinRummyUtils utils = GinRummyUtils(kDefaultNumRanks, kDefaultNumSuits,

334 GameParameters params;
335 std::shared_ptr<const open_spiel::Game> game =

337 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
338 std::vector<Action> legal_actions;
339 std::vector<Action> initial_actions;
340 initial_actions = {8, 9, 10, 11, 12, 13, 14, 15, 48, 49, 0, 1, 2, 3,
341 4, 5, 6, 7, 50, 51, 16, 54, 54, 53, 17, 17, 53, 18,
342 18, 53, 19, 19, 53, 20, 20, 53, 21, 21, 53, 22, 22, 53,
343 23, 23, 53, 24, 24, 53, 25, 25, 53, 26, 26, 53, 27, 27,

346 37, 37, 53, 38, 38, 53, 39, 39, 53, 40, 40, 53, 41, 41,
347 53, 42, 42, 53, 43, 43, 53, 44, 44, 53, 46, 49};
348 for (auto action : initial_actions) state->ApplyAction(action);
349 std::cout << state->ToString() << std::endl;
350 // Player turn: 1

387 state->ApplyAction(164);
388 state->ApplyAction(166);

397 legal_actions = state->LegalActions();
398 SPIEL_CHECK_TRUE(state->IsTerminal());
399 SPIEL_CHECK_EQ(legal_actions.size(), 0);
400 std::vector<double> returns = state->Returns();
401 SPIEL_CHECK_EQ(returns[1], 43); // 25 point gin bonus + 18 deadwood
402 SPIEL_CHECK_EQ(returns[0], -43);
403}

405// The rules of gin rummy do not explicitly prevent infinite action sequences.
406// Both players can keep drawing the upcard and discarding indefinitely. This
407// is poor strategy and never occurs in actual play, but we need a way of
408// ensuring the game is finite. In doing so, we don't want to prematurely
409// declare a draw and prevent legitimate lines of play. Our solution is to cap
410// the number of times the upcard can be drawn at 50. This is well above
411// anything observed in actual play, and corresponds nicely to the 50 cards in
412// play each hand.
413void MaxGameLengthTest() {
414 GameParameters params;
415 std::shared_ptr<const open_spiel::Game> game =
416 open_spiel::LoadGame("gin_rummy", params);
417 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
418 std::vector<Action> initial_actions;
419 // Deal hands
420 initial_actions = {0, 1, 2, 13, 14, 15, 26, 27, 28, 39, 9,
421 10, 11, 12, 23, 24, 25, 36, 37, 38, 40};
422 // Loop of drawing and discarding.
423 for (int i = 0; i < 16; ++i) {
424 initial_actions.push_back(52);
425 initial_actions.push_back(0);
426 initial_actions.push_back(52);
427 initial_actions.push_back(12);
428 initial_actions.push_back(52);
429 initial_actions.push_back(1);
430 }
431 initial_actions.push_back(52);
432 initial_actions.push_back(0);
433 initial_actions.push_back(52);
434 initial_actions.push_back(12);
435 // 51st time an upcard is drawn ends the game in a draw.
436 initial_actions.push_back(52);
437 for (auto action : initial_actions) {
438 state->ApplyAction(action);
439 }
440 SPIEL_CHECK_TRUE(state->IsTerminal());
441 std::vector<double> returns = state->Returns();
442 SPIEL_CHECK_EQ(returns[0], 0);
443 SPIEL_CHECK_EQ(returns[1], 0);
444}
