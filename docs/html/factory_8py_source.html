103 def create_game_with_setting(game_name: str,
104  setting: Optional[str] = None) -> pyspiel.Game:
105  """Creates an OpenSpiel game with the specified setting.
106 
107  Args:
108  game_name: Name of a registered game, e.g. mfg_crowd_modelling_2d.
109  setting: Name of the pre-defined setting. If None, game_name will be used
110  instead. The setting should be present in the GAME_SETTINGS map above.
111 
112  Returns:
113  a Game.
114  """
115  setting = setting or game_name
116  params = GAME_SETTINGS.get(setting)
117  if params is None:
118  raise ValueError(f"{setting} setting does not exist for {game_name}.")
119 
120  logging.info("Creating %s game with parameters: %r", game_name, params)
121 
122  # Dynamic routing game requires setting the network and demand explicitly.
123  if game_name == "python_mfg_dynamic_routing":
124  # Create a copy since we modify it below removing the network key.
125  params = params.copy()
126  network = params.pop("network")
127  network, od_demand = DYNAMIC_ROUTING_NETWORK[network]
128  return dynamic_routing.MeanFieldRoutingGame(
129  params, network=network, od_demand=od_demand)
130 
131  return pyspiel.load_game(game_name, params)
