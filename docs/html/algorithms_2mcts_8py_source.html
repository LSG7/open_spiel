55  """Returns evaluation on given state."""
56  result = None
57  for _ in range(self.n_rolloutsn_rollouts):

81 class SearchNode(object):
82  """A node in the search tree.
83 
84  A SearchNode represents a state and possible continuations from it. Each child
85  represents a possible action, and the expected result from doing so.
86 
87  Attributes:
88  action: The action from the parent node's perspective. Not important for the
89  root node, as the actions that lead to it are in the past.
90  player: Which player made this action.
91  prior: A prior probability for how likely this action will be selected.
92  explore_count: How many times this node was explored.
93  total_reward: The sum of rewards of rollouts through this node, from the
94  parent node's perspective. The average reward of this node is
95  `total_reward / explore_count`
96  outcome: The rewards for all players if this is a terminal node or the
97  subtree has been proven, otherwise None.
98  children: A list of SearchNodes representing the possible actions from this
99  node, along with their expected rewards.
100  """

140  def sort_key(self):

142 
143  This ordering leads to choosing:
144  - Highest proven score > 0 over anything else, including a promising but
145  unproven action.

147  uncertain, or the others are losses.
148  - Uncertain action with most exploration over loss of any difficulty

292  def _apply_tree_policy(self, root, state):
293  """Applies the UCT policy to play the game until reaching a leaf node.
294 
295  A leaf node is defined as a node that is terminal or has not been evaluated
296  yet. If it reaches a node that has been evaluated before but hasn't been
297  expanded, then expand it's children and continue.
298 
299  Args:
300  root: The root node in the search tree.
301  state: The state of the game at the root node.
302 
303  Returns:
304  visit_path: A list of nodes descending from the root node to a leaf node.
305  working_state: The state of the game at the leaf node.
306  """
307  visit_path = [root]
308  working_state = state.clone()
309  current_node = root
310  while (not working_state.is_terminal() and
311  current_node.explore_count > 0) or (
312  working_state.is_chance_node() and self.dont_return_chance_nodedont_return_chance_node):
313  if not current_node.children:
314  # For a new node, initialize its state, then choose a child as normal.
315  legal_actions = self.evaluatorevaluator.prior(working_state)
316  if current_node is root and self._dirichlet_noise_dirichlet_noise:
317  epsilon, alpha = self._dirichlet_noise_dirichlet_noise
318  noise = self._random_state_random_state.dirichlet([alpha] * len(legal_actions))
319  legal_actions = [(a, (1 - epsilon) * p + epsilon * n)
320  for (a, p), n in zip(legal_actions, noise)]
321  # Reduce bias from move generation order.
322  self._random_state_random_state.shuffle(legal_actions)
323  player = working_state.current_player()
324  current_node.children = [
325  SearchNode(action, player, prior) for action, prior in legal_actions
326  ]
327 
328  if working_state.is_chance_node():
329  # For chance nodes, rollout according to chance node's probability
330  # distribution
331  outcomes = working_state.chance_outcomes()
332  action_list, prob_list = zip(*outcomes)
333  action = self._random_state_random_state.choice(action_list, p=prob_list)
334  chosen_child = next(
335  c for c in current_node.children if c.action == action)
336  else:
337  # Otherwise choose node with largest UCT value
338  chosen_child = max(
339  current_node.children,
340  key=lambda c: self._child_selection_fn_child_selection_fn( # pylint: disable=g-long-lambda
341  c, current_node.explore_count, self.uct_cuct_c))
342 
343  working_state.apply_action(chosen_child.action)
344  current_node = chosen_child
345  visit_path.append(current_node)
346 
347  return visit_path, working_state

349  def mcts_search(self, state):
350  """A vanilla Monte-Carlo Tree Search algorithm.
351 
352  This algorithm searches the game tree from the given state.
353  At the leaf, the evaluator is called if the game state is not terminal.
354  A total of max_simulations states are explored.
355 
356  At every node, the algorithm chooses the action with the highest PUCT value,
357  defined as: `Q/N + c * prior * sqrt(parent_N) / N`, where Q is the total
358  reward after the action, and N is the number of times the action was
359  explored in this position. The input parameter c controls the balance
360  between exploration and exploitation; higher values of c encourage
361  exploration of under-explored nodes. Unseen actions are always explored
362  first.
363 
364  At the end of the search, the chosen action is the action that has been
365  explored most often. This is the action that is returned.
366 
367  This implementation supports sequential n-player games, with or without
368  chance nodes. All players maximize their own reward and ignore the other
369  players' rewards. This corresponds to max^n for n-player games. It is the
370  norm for zero-sum games, but doesn't have any special handling for
371  non-zero-sum games. It doesn't have any special handling for imperfect
372  information games.
373 
374  The implementation also supports backing up solved states, i.e. MCTS-Solver.
375  The implementation is general in that it is based on a max^n backup (each
376  player greedily chooses their maximum among proven children values, or there
377  exists one child whose proven value is game.max_utility()), so it will work
378  for multiplayer, general-sum, and arbitrary payoff games (not just win/loss/
379  draw games). Also chance nodes are considered proven only if all children
380  have the same value.
381 
382  Some references:
383  - Sturtevant, An Analysis of UCT in Multi-Player Games, 2008,
384  https://web.cs.du.edu/~sturtevant/papers/multi-player_UCT.pdf
385  - Nijssen, Monte-Carlo Tree Search for Multi-Player Games, 2013,
386  https://project.dke.maastrichtuniversity.nl/games/files/phd/Nijssen_thesis.pdf
387  - Silver, AlphaGo Zero: Starting from scratch, 2017
388  https://deepmind.com/blog/article/alphago-zero-starting-scratch
389  - Winands, Bjornsson, and Saito, "Monte-Carlo Tree Search Solver", 2008.
390  https://dke.maastrichtuniversity.nl/m.winands/documents/uctloa.pdf
391 
392  Arguments:
393  state: pyspiel.State object, state to search from
394 
395  Returns:
396  The most visited move from the root node.
397  """
398  root = SearchNode(None, state.current_player(), 1)
399  for _ in range(self.max_simulationsmax_simulations):
400  visit_path, working_state = self._apply_tree_policy_apply_tree_policy(root, state)
401  if working_state.is_terminal():
402  returns = working_state.returns()
403  visit_path[-1].outcome = returns
404  solved = self.solvesolve
405  else:
406  returns = self.evaluatorevaluator.evaluate(working_state)
407  solved = False
408 
409  while visit_path:
410  # For chance nodes, walk up the tree to find the decision-maker.
411  decision_node_idx = -1
412  while visit_path[decision_node_idx].player == pyspiel.PlayerId.CHANCE:
413  decision_node_idx -= 1
414  # Chance node targets are for the respective decision-maker.
415  target_return = returns[visit_path[decision_node_idx].player]
416  node = visit_path.pop()
417  node.total_reward += target_return
418  node.explore_count += 1
419 
420  if solved and node.children:
421  player = node.children[0].player
422  if player == pyspiel.PlayerId.CHANCE:
423  # Only back up chance nodes if all have the same outcome.
424  # An alternative would be to back up the weighted average of
425  # outcomes if all children are solved, but that is less clear.
426  outcome = node.children[0].outcome
427  if (outcome is not None and
428  all(np.array_equal(c.outcome, outcome) for c in node.children)):
429  node.outcome = outcome
430  else:
431  solved = False
432  else:
433  # If any have max utility (won?), or all children are solved,
434  # choose the one best for the player choosing.
435  best = None
436  all_solved = True
437  for child in node.children:
438  if child.outcome is None:
439  all_solved = False
440  elif best is None or child.outcome[player] > best.outcome[player]:
441  best = child
442  if (best is not None and
443  (all_solved or best.outcome[player] == self.max_utilitymax_utility)):
444  node.outcome = best.outcome
445  else:
446  solved = False
447  if root.outcome is not None:
448  break
449 
450  return root
