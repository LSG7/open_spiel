27 // Treat a `absl::Span<float>` as a tensor of fixed shape. The rank (number of
28 // dimensions) must be known at compile time, though the actual sizes of the
29 // dimensions can be supplied at construction time. It then lets you index into
30 // the vector easily without having to compute the 1d-vector's indices manually.
31 template <int Rank>
32 class TensorView {
33  public:
34  constexpr TensorView(absl::Span<float> values,
35  const std::array<int, Rank>& shape, bool reset)
36  : values_(values), shape_(shape) {
37  SPIEL_CHECK_EQ(size(), values_.size());
38  if (reset) std::fill(values.begin(), values.end(), 0);
39  }
40 
41  constexpr int size() const {
42  return std::accumulate(shape_.begin(), shape_.end(), 1,
43  std::multiplies<int>());
44  }
45 
46  void clear() { std::fill(values_.begin(), values_.end(), 0.0); }
47 
48  constexpr int index(const std::array<int, Rank>& args) const {
49  int ind = 0;
50  for (int i = 0; i < Rank; ++i) {
51  ind = ind * shape_[i] + args[i];
52  }
53  return ind;
54  }
55 
56  constexpr float& operator[](const std::array<int, Rank>& args) {
57  return values_[index(args)];
58  }
59  constexpr const float& operator[](const std::array<int, Rank>& args) const {
60  return values_[index(args)];
61  }
62 
63  constexpr int rank() const { return Rank; }
64  constexpr const std::array<int, Rank> shape() const { return shape_; }
65  constexpr int shape(int i) const { return shape_[i]; }
66 
67  private:
68  absl::Span<float> values_;
69  const std::array<int, Rank> shape_;
70 };
