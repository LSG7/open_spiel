48 // CordRepBtreeNavigatorTest is a test fixture which automatically creates a
49 // tree to test navigation logic on. The parameter `count' defines the number of
50 // data edges in the test tree.
51 class CordRepBtreeNavigatorTest : public testing::TestWithParam<size_t> {
52  public:
53  using Flats = std::vector<CordRep*>;
54  static constexpr size_t kCharsPerFlat = 3;
55 
56  CordRepBtreeNavigatorTest() {
57  data_ = CreateRandomString(count() * kCharsPerFlat);
58  flats_ = CreateFlatsFromString(data_, kCharsPerFlat);
59 
60  // Turn flat 0 or 1 into a substring to cover partial reads on substrings.
61  if (count() > 1) {
62  CordRep::Unref(flats_[1]);
63  flats_[1] = MakeSubstring(kCharsPerFlat, kCharsPerFlat, MakeFlat(data_));
64  } else {
65  CordRep::Unref(flats_[0]);
66  flats_[0] = MakeSubstring(0, kCharsPerFlat, MakeFlat(data_));
67  }
68 
69  tree_ = CordRepBtreeFromFlats(flats_);
70  }
71 
72  ~CordRepBtreeNavigatorTest() override { CordRep::Unref(tree_); }
73 
74  size_t count() const { return GetParam(); }
75  CordRepBtree* tree() { return tree_; }
76  const std::string& data() const { return data_; }
77  const std::vector<CordRep*>& flats() const { return flats_; }
78 
79  static std::string ToString(testing::TestParamInfo<size_t> param) {
80  return absl::StrCat(param.param, "_Flats");
81  }
82 
83  private:
84  std::string data_;
85  Flats flats_;
86  CordRepBtree* tree_;
87 };

276 TEST_P(CordRepBtreeNavigatorTest, Read) {
277  const Flats& flats = this->flats();
278  const std::string& data = this->data();
279 
280  for (size_t offset = 0; offset < data.size(); ++offset) {
281  for (size_t length = 1; length <= data.size() - offset; ++length) {
282  CordRepBtreeNavigator nav;
283  nav.InitFirst(tree());
284 
285  // Skip towards edge holding offset
286  size_t edge_offset = nav.Skip(offset).offset;
287 
288  // Read node
289  ReadResult result = nav.Read(edge_offset, length);
290  ASSERT_THAT(result.tree, Ne(nullptr));
291  EXPECT_THAT(result.tree->length, Eq(length));
292  if (result.tree->tag == BTREE) {
293  ASSERT_TRUE(CordRepBtree::IsValid(result.tree->btree()));
294  }
295 
296  // Verify contents
297  std::string value = CordToString(result.tree);
298  EXPECT_THAT(value, Eq(data.substr(offset, length)));
299 
300  // Verify 'partial last edge' reads.
301  size_t partial = (offset + length) % kCharsPerFlat;
302  ASSERT_THAT(result.n, Eq(partial));
303 
304  // Verify ending position if not EOF
305  if (offset + length < data.size()) {
306  size_t index = (offset + length) / kCharsPerFlat;
307  EXPECT_THAT(nav.Current(), Eq(flats[index]));
308  }
309 
310  CordRep::Unref(result.tree);
311  }
312  }
