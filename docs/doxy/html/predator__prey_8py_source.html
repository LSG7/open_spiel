16"""Mean Field predator-prey game, implemented in Python.

18This corresponds to the predator-prey game described in section 5.4 of
19"Scaling up Mean Field Games with Online Mirror Descent"
20(https://arxiv.org/abs/2103.00623)

22The environment is configurable in the following high-level ways:
23- Number of populations.
24- Reward matrix.
25- Initial distribution.
26- Geometry (torus, basic square).

187class MFGPredatorPreyGame(pyspiel.Game):
188 """Predator-prey multi-population MFG."""
189
190 # pylint:disable=dangerous-default-value
191 def __init__(self, params: Mapping[str, Any] = _DEFAULT_PARAMS):
192 self.size = get_param("size", params)
193 self.horizon = get_param("horizon", params)
194 flat_reward_matrix = np.fromstring(
195 get_param("reward_matrix", params), dtype=np.float64, sep=" "
196 )
197 num_players = get_param("players", params)
198 if len(flat_reward_matrix) != num_players**2:
199 raise ValueError(
200 "Reward matrix passed in flat representation does not represent a "
201 f"square matrix: {flat_reward_matrix}"
202 f" with number of players: {num_players}"
203 )
204 self.reward_matrix = flat_reward_matrix.reshape([num_players, num_players])
205 self.geometry = get_param("geometry", params)
206 num_states = self.size**2
207 game_info = pyspiel.GameInfo(
208 num_distinct_actions=_NUM_ACTIONS,
209 max_chance_outcomes=max(num_states, _NUM_CHANCE),
210 num_players=num_players,
211 min_utility=-np.inf,
212 max_utility=+np.inf,
213 utility_sum=None,
214 max_game_length=self.horizon,
215 )
216
217 self.noise_probability = get_param("noise_probability", params)
218 self.congestion_coeff = get_param("congestion_coeff", params)
219 # Represents the current probability distribution over game states
220 # (when grouped for each population).
221 str_init_distrib = get_param("init_distrib", params)
222 if str_init_distrib:
223 flat_init_distrib = np.fromstring(
224 str_init_distrib, dtype=np.float64, sep=" "
225 )
226 if len(flat_init_distrib) != num_players * self.size**2:
227 raise ValueError(
228 "Initial distribution matrix passed in flat representation does"
229 f" not represent a sequence of square matrices: {flat_init_distrib}"
230 f" with number of players: {num_players}"
231 f" and size: {self.size}"
232 )
233 self.initial_distribution = flat_init_distrib
234 else:
235 # Initialized with a uniform distribution.
236 self.initial_distribution = [1.0 / num_states] * (
237 num_states * num_players
238 )
239 super().__init__(_GAME_TYPE, game_info, params)

268def pos_to_merged(pos: np.ndarray, size: int) -> int:
269 """Converts a [x, y] position into a single integer."""
270 assert (pos >= 0).all(), pos
271 assert (pos < size).all(), pos
272 return pos[0] + pos[1] * size

275def merged_to_pos(merged_pos: int, size: int) -> np.ndarray:
276 """Inverse of pos_to_merged()."""

281class MFGPredatorPreyState(pyspiel.State):
282 """State for the predator-prey MFG."""
283
284 # Maps legal actions to the corresponding move on the grid of the game.
285 _ACTION_TO_MOVE = {
286 0: np.array([0, 0]),
287 1: np.array([1, 0]),
288 2: np.array([0, 1]),
289 3: np.array([0, -1]),
290 4: np.array([-1, 0]),
291 }
292 # Action that corresponds to no displacement.
293 _NEUTRAL_ACTION = 0
294
295 def __init__(self, game, population=None):
296 """Constructor; should only be called by Game.new_initial_state.*.
297
298 Args:
299 game: MFGPredatorPreyGame for which a state should be created.
300 population: ID of the population to create this state for. Must be in [0,
301 num_players()) or None. States with population=None cannot be used to
302 perform game actions.
303 """
304 super().__init__(game)
305 # Initial state where the initial position is chosen according to
306 # an initial distribution.
307 self._is_position_init = True
308 self._player_id = pyspiel.PlayerId.CHANCE
309 # Population this state corresponds to. Can be None, in which
310 # case, ApplyAction() is forbidden.
311 self._population = population
312 if self._population is not None:
313 assert 0 <= self._population < self.num_players()
314 # When set, <int>[2] numpy array representing the x, y position on the grid.
315 self._pos = None # type: Optional[np.ndarray]
316 self._t = 0
317 self.size = game.size
318 # Number of states in the grid.
319 self.num_states = self.size**2
320 self.horizon = game.horizon
321 self.reward_matrix = game.reward_matrix
322 self.geometry = game.geometry
323 self._returns = np.zeros([self.num_players()], dtype=np.float64)
324 self._distribution = shared_value.SharedValue(game.initial_distribution)
325 self.noise_probability = game.noise_probability
326 self.congestion_coeff = game.congestion_coeff

340 def state_to_str(self, pos, t, population, player_id=0):
341 """A string that uniquely identify (pos, t, population, player_id)."""
342 if self._is_position_init:
343 return f"position_init_{population}"
344 assert isinstance(pos, np.ndarray), f"Got type {type(pos)}"
345 assert len(pos.shape) == 1, f"Got {len(pos.shape)}, expected 1 (pos={pos})."
346 assert pos.shape[0] == 2, f"Got {pos.shape[0]}, expected 2 (pos={pos})."
347 return _state_to_str(pos[0], pos[1], t, population, player_id)

352 def mean_field_population(self):
353 return self._population

355 def _legal_actions(self, player):
356 """Returns a list of legal actions for player and MFG nodes."""
357 if player == pyspiel.PlayerId.MEAN_FIELD:
358 return []
359 if player >= 0 and player == self.current_player():
360 return list(self._ACTION_TO_MOVE_ACTION_TO_MOVE)
361 raise ValueError(
362 f"Unexpected player {player}."
363 "Expected a mean field or current player >=0."
364 )
365
366 def chance_outcomes(self) -> List[Tuple[int, float]]:
367 """Returns the possible chance outcomes and their probabilities."""
368 if self._is_position_init:
369 if (
370 self._population is None
371 or not 0 <= self._population < self.num_players()
372 ):
373 raise ValueError(f"Invalid population {self._population}")
374 return [
375 (i, self._distribution.value[self._population * self.num_states + i])
376 for i in range(self.num_states)
377 if self._distribution.value[self._population * self.num_states + i]
378 != 0.0
379 ]
380 return [
381 (0, 1.0 - self.noise_probability),
382 (1, self.noise_probability / 4.0),
383 (2, self.noise_probability / 4.0),
384 (3, self.noise_probability / 4.0),
385 (4, self.noise_probability / 4.0),
386 ]

388 def update_pos(self, action):
389 """Updates the position of the player given a move action."""
390 if action < 0 or action >= len(self._ACTION_TO_MOVE_ACTION_TO_MOVE):
391 raise ValueError(
392 f"The action must be between 0 and {len(self._ACTION_TO_MOVE)}, "
393 f"got {action}"
394 )
395 candidate_pos = self._pos + self._ACTION_TO_MOVE_ACTION_TO_MOVE[action]
396 if self.geometry == Geometry.TORUS:
397 candidate_pos += self.size
398 candidate_pos %= self.size
399 else:
400 assert (
401 self.geometry == Geometry.SQUARE
402 ), f"Invalid geometry {self.geometry}"
403 # Keep the position within the bounds of the square.
404 candidate_pos = np.minimum(candidate_pos, self.size - 1)
405 candidate_pos = np.maximum(candidate_pos, 0)
406 self._pos = candidate_pos
407
408 def _apply_action(self, action):
409 """Applies the specified action to the state."""
410 if self._population is None:
411 raise ValueError(
412 "Attempting to perform an action with a population-less state."
413 )
414 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
415 raise ValueError(
416 "_apply_action should not be called at a MEAN_FIELD state."
417 )
418 self._returns += np.array(self.rewards())
419 if self._is_position_init:
420 self._pos = merged_to_pos(action, self.size)
421 self._is_position_init = False
422 self._player_id = self._population
423 elif self._player_id == pyspiel.PlayerId.CHANCE:
424 self.update_pos(action)
425 self._t += 1
426 self._player_id = pyspiel.PlayerId.MEAN_FIELD
427 elif int(self._player_id) >= 0:
428 assert self._player_id == self._population, (
429 f"Invalid decision player id {self._player_id} "
430 f"expected {self._population}"
431 )
432 self.update_pos(action)
433 self._player_id = pyspiel.PlayerId.CHANCE
434 else:
435 raise ValueError(f"Unexpected state. Player id: {self._player_id}")

437 def _action_to_string(self, player, action):
438 """Action -> string."""
439 del player
440 if self.is_chance_node() and self._is_position_init:
441 return f"init_position={action}"
442 return str(self._ACTION_TO_MOVE_ACTION_TO_MOVE[action])

444 def distribution_support(self):
445 """Returns a list of state string."""
446 support = []
447 for x in range(self.size):
448 for y in range(self.size):
449 for population in range(self.num_players()):
450 support.append(
451 self.state_to_str(
452 np.array([x, y]),
453 self._t,
454 population,
455 player_id=pyspiel.PlayerId.MEAN_FIELD,
456 )
457 )
458 return support

460 def get_pos_proba(self, pos: np.ndarray, population: int) -> float:
461 """Gets the probability of a pos and population in the current distrib.
462
463 Args:
464 pos: 2D position.
465 population: Population requested.
466
467 Returns:
468 The probability for the provided position and population.
469 """
470 assert (pos >= 0).all(), pos
471 assert (pos < self.size).all(), pos
472 assert 0 <= population < self.num_players(), population
473 # This logic needs to match the ordering defined in distribution_support().
474 index = population + self.num_players() * (pos[1] + self.size * pos[0])
475 assert 0 <= index < len(self._distribution.value), (
476 f"Invalid index {index} vs dist length: {len(self._distribution.value)}"
477 f", population={population}, pos={pos}, state={self}"
478 )
479 return self._distribution.value[index]

481 def update_distribution(self, distribution):
482 """This function is central and specific to the logic of the MFG.
483
484 It should only be called when the node is in MEAN_FIELD state.
485
486 Args:
487 distribution: List of floats that should contain the probability of each
488 state returned by distribution_support().
489 """
490 expected_dist_size = self.num_states * self.num_players()
491 assert len(distribution) == expected_dist_size, (
492 "Unexpected distribution length "
493 f"{len(distribution)} != {expected_dist_size}"
494 )
495 if self._player_id != pyspiel.PlayerId.MEAN_FIELD:
496 raise ValueError(
497 "update_distribution should only be called at a MEAN_FIELD state."
498 )
499 self._distribution = shared_value.SharedValue(distribution)
500 self._player_id = self._population
501
502 def is_terminal(self):
503 """Returns True if the game is over."""
504 return self.t >= self.horizon
505
506 def current_player(self):
507 """Returns id of the next player to move, or TERMINAL if game is over."""
508 if self.is_terminal():
509 return pyspiel.PlayerId.TERMINAL
510 return self._player_id

512 def rewards(self) -> List[float]:
513 """Predator-prey rewards for all populations.
514
515 See section 5.4, paragraph Environment in https://arxiv.org/abs/2103.00623.
516
517 Returns:
518 One float per population.
519 """
520 if int(self._player_id) < 0:
521 return [0.0] * self.num_players()
522 # TODO(author15): Remove this eps once b/191064186 is fixed.
523 eps = 1e-25
524 densities = np.array(
525 [
526 self.get_pos_proba(self._pos, population)
527 for population in range(self.num_players())
528 ],
529 dtype=np.float64,
530 )
531 rew = -self.congestion_coeff * np.log(densities + eps) + np.dot(
532 self.reward_matrix, densities
533 )
534 return list(rew)
535
536 def returns(self) -> List[float]:
537 """Returns is the sum of all payoffs collected so far."""
538 return list(self._returns + np.array(self.rewards()))

540 def __str__(self):

542 return self.state_to_str(

547class Observer:
548 """Observer, conforming to the PyObserver interface (see observation.py)."""
549
550 def __init__(self, params, game):
551 """Initializes an empty observation tensor."""
552 del params
553
554 self.size = game.size
555 self.horizon = game.horizon
556 # +1 to allow t == horizon.
557 self.tensor = np.zeros(2 * self.size + self.horizon + 1, np.float32)
558 self.dict = {
559 "x": self.tensor[: self.size],
560 "y": self.tensor[self.size : self.size * 2],
561 "t": self.tensor[self.size * 2 :],
562 }
563
564 def set_from(self, state: MFGPredatorPreyState, player: int):
565 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
566 del player
567 # We update the observation via the shaped tensor since indexing is more
568 # convenient than with the 1-D tensor. Both are views onto the same memory.
569 self.tensor.fill(0)
570 # state.pos is None for the initial (blank) state, don't set any
571 # position bit in that case.
572 if state.pos is not None:
573 if not (state.pos >= 0).all() or not (state.pos < self.size).all():
574 raise ValueError(
575 f"Expected {state} positions to be in [0, {self.size})"
576 )
577 self.dict["x"][state.pos[0]] = 1
578 self.dict["y"][state.pos[1]] = 1
579 if not 0 <= state.t <= self.horizon:
580 raise ValueError(f"Expected {state} time to be in [0, {self.horizon}]")
581 self.dict["t"][state.t] = 1
582
583 def string_from(self, state, player):
584 """Observation of `state` from the PoV of `player`, as a string."""
585 del player
586 return str(state)
