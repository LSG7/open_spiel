57 // Adds a plane to the information state vector corresponding to the presence
58 // and absence of the given piece type and colour at each square.
59 void AddPieceTypePlane(Color color, PieceType piece_type,
60  const ChessBoard& board,
61  absl::Span<float>::iterator& value_it) {
62  for (int8_t y = 0; y < kMaxBoardSize; ++y) {
63  for (int8_t x = 0; x < kMaxBoardSize; ++x) {
64  Piece piece_on_board = board.at(Square{x, y});
65  *value_it++ =
66  (piece_on_board.color == color && piece_on_board.type == piece_type
67  ? 1.0
68  : 0.0);
69  }
70  }
71 }

74 template <typename T>
75 void AddScalarPlane(T val, T min, T max,
76  absl::Span<float>::iterator& value_it) {
77  double normalized_val = static_cast<double>(val - min) / (max - min);
78  for (int i = 0; i < k2dMaxBoardSize; ++i) *value_it++ = normalized_val;

152 Action MoveToAction(const Move& move, int board_size) {
153  // Special-case for pass move.
154  if (move == kPassMove) return kPassAction;
155 
156  Color color = move.piece.color;
157  // We rotate the move to be from player p's perspective.
158  Move player_move(move);
159 
160  // Rotate move to be from player p's perspective.
161  player_move.from.y = ReflectRank(color, board_size, player_move.from.y);
162  player_move.to.y = ReflectRank(color, board_size, player_move.to.y);
163 
164  // For each starting square, we enumerate 73 actions:
165  // - 9 possible underpromotions
166  // - 56 queen moves
167  // - 8 knight moves
168  // In total, this results in 64 * 73 = 4672 indices.
169  // This is the union of all possible moves, by reducing this to the number of
170  // moves actually available from each starting square this could still be
171  // reduced a little to 1816 indices.
172  int starting_index =
173  EncodeMove(player_move.from, 0, kMaxBoardSize, kNumActionDestinations);
174  int8_t x_diff = player_move.to.x - player_move.from.x;
175  int8_t y_diff = player_move.to.y - player_move.from.y;
176  Offset offset{x_diff, y_diff};
177  bool is_under_promotion = move.promotion_type != PieceType::kEmpty &&
178  move.promotion_type != PieceType::kQueen;
179  if (is_under_promotion) {
180  // We have to indicate underpromotions as special moves, because in terms of
181  // from/to they are identical to queen promotions.
182  // For a given starting square, an underpromotion can have 3 possible
183  // destination squares (straight, left diagonal, right diagonal) and 3
184  // possible piece types.
185  SPIEL_CHECK_EQ(move.piece.type, PieceType::kPawn);
186  SPIEL_CHECK_TRUE((move.piece.color == color &&
187  player_move.from.y == board_size - 2 &&
188  player_move.to.y == board_size - 1) ||
189  (move.piece.color == OppColor(color) &&
190  player_move.from.y == 1 && player_move.to.y == 0));
191 
192  int promotion_index;
193  {
194  auto itr = absl::c_find(kUnderPromotionIndexToType, move.promotion_type);
195  SPIEL_CHECK_TRUE(itr != kUnderPromotionIndexToType.end());
196  promotion_index = std::distance(kUnderPromotionIndexToType.begin(), itr);
197  }
198 
199  int direction_index;
200  {
201  auto itr = absl::c_find_if(
202  kUnderPromotionDirectionToOffset,
203  [offset](Offset o) { return o.x_offset == offset.x_offset; });
204  SPIEL_CHECK_TRUE(itr != kUnderPromotionDirectionToOffset.end());
205  direction_index =
206  std::distance(kUnderPromotionDirectionToOffset.begin(), itr);
207  }
208  return starting_index +
209  kUnderPromotionDirectionToOffset.size() * promotion_index +
210  direction_index;
211  } else {
212  // For the normal moves, we simply encode starting and destination square.
213  int destination_index =
214  OffsetToDestinationIndex(offset, kKnightOffsets, kMaxBoardSize);
215  SPIEL_CHECK_TRUE(destination_index >= 0 && destination_index < 64);
216  return starting_index + kNumUnderPromotions + destination_index;
217  }
218 }

220 std::pair<Square, int> ActionToDestination(int action, int board_size,
221  int num_actions_destinations) {
222  const int xy = action / num_actions_destinations;
223  SPIEL_CHECK_GE(xy, 0);
224  SPIEL_CHECK_LT(xy, board_size * board_size);
225  const int8_t x = xy / board_size;
226  const int8_t y = xy % board_size;
227  const int destination_index = action % num_actions_destinations;
228  SPIEL_CHECK_GE(destination_index, 0);
229  SPIEL_CHECK_LT(destination_index, num_actions_destinations);
230  return {Square{x, y}, destination_index};
231 }

233 Move ActionToMove(const Action& action, const ChessBoard& board) {
234  SPIEL_CHECK_GE(action, 0);
235  SPIEL_CHECK_LT(action, NumDistinctActions());
236 
237  // Some chess variants (e.g. RBC) allow pass moves.
238  if (board.AllowPassMove() && action == kPassAction) {
239  return kPassMove;
240  }
241 
242  // The encoded action represents an action encoded from color's perspective.
243  Color color = board.ToPlay();
244  int board_size = board.BoardSize();
245  PieceType promotion_type = PieceType::kEmpty;
246  bool is_castling = false;
247  auto [from_square, destination_index] =
248  ActionToDestination(action, kMaxBoardSize, kNumActionDestinations);
249  SPIEL_CHECK_LT(destination_index, kNumActionDestinations);
250 
251  bool is_under_promotion = destination_index < kNumUnderPromotions;
252  Offset offset;
253  if (is_under_promotion) {
254  int promotion_index = destination_index / 3;
255  int direction_index = destination_index % 3;
256  promotion_type = kUnderPromotionIndexToType[promotion_index];
257  offset = kUnderPromotionDirectionToOffset[direction_index];
258  } else {
259  destination_index -= kNumUnderPromotions;
260  offset = DestinationIndexToOffset(destination_index, kKnightOffsets,
261  kMaxBoardSize);
262  }
263  Square to_square = from_square + offset;
264 
265  from_square.y = ReflectRank(color, board_size, from_square.y);
266  to_square.y = ReflectRank(color, board_size, to_square.y);
267 
268  // This uses the current state to infer the piece type.
269  Piece piece = {board.ToPlay(), board.at(from_square).type};
270 
271  // Check for queen promotion.
272  if (!is_under_promotion && piece.type == PieceType::kPawn &&
273  ReflectRank(color, board_size, from_square.y) == board_size - 2 &&
274  ReflectRank(color, board_size, to_square.y) == board_size - 1) {
275  promotion_type = PieceType::kQueen;
276  }
277 
278  // Check for castling which is defined here just as king moves horizontally
279  // by 2 spaces.
280  // TODO(b/149092677): Chess no longer supports chess960. Distinguish between
281  // left/right castle.
282  if (piece.type == PieceType::kKing && std::abs(offset.x_offset) == 2) {
283  is_castling = true;
284  }
285  Move move(from_square, to_square, piece, promotion_type, is_castling);
286  return move;
287 }

317 void ChessState::ObservationTensor(Player player,
318  absl::Span<float> values) const {
319  SPIEL_CHECK_GE(player, 0);
320  SPIEL_CHECK_LT(player, num_players_);
321 
322  auto value_it = values.begin();
323 
324  // Piece cconfiguration.
325  for (const auto& piece_type : kPieceTypes) {
326  AddPieceTypePlane(Color::kWhite, piece_type, Board(), value_it);
327  AddPieceTypePlane(Color::kBlack, piece_type, Board(), value_it);
328  }
329 
330  AddPieceTypePlane(Color::kEmpty, PieceType::kEmpty, Board(), value_it);
331 
332  const auto entry = repetitions_.find(Board().HashValue());
333  SPIEL_CHECK_FALSE(entry == repetitions_.end());
334  int repetitions = entry->second;
335 
336  // Num repetitions for the current board.
337  AddScalarPlane(repetitions, 1, 3, value_it);
338 
339  // Side to play.
340  AddScalarPlane(ColorToPlayer(Board().ToPlay()), 0, 1, value_it);
341 
342  // Irreversible move counter.
343  AddScalarPlane(Board().IrreversibleMoveCounter(), 0, 101, value_it);
344 
345  // Castling rights.
346  AddBinaryPlane(Board().CastlingRight(Color::kWhite, CastlingDirection::kLeft),
347  value_it);
348 
349  AddBinaryPlane(
350  Board().CastlingRight(Color::kWhite, CastlingDirection::kRight),
351  value_it);
352 
353  AddBinaryPlane(Board().CastlingRight(Color::kBlack, CastlingDirection::kLeft),
354  value_it);
355 
356  AddBinaryPlane(
357  Board().CastlingRight(Color::kBlack, CastlingDirection::kRight),
358  value_it);
359 
360  SPIEL_CHECK_EQ(value_it, values.end());
361 }

386 absl::optional<std::vector<double>> ChessState::MaybeFinalReturns() const {
387  if (!Board().HasSufficientMaterial()) {
388  return std::vector<double>{DrawUtility(), DrawUtility()};
389  }
390 
391  if (IsRepetitionDraw()) {
392  return std::vector<double>{DrawUtility(), DrawUtility()};
393  }
394  // Compute and cache the legal actions.
395  MaybeGenerateLegalActions();
396  SPIEL_CHECK_TRUE(cached_legal_actions_);
397  bool have_legal_moves = !cached_legal_actions_->empty();
398 
399  // If we don't have legal moves we are either stalemated or checkmated,
400  // depending on whether we are in check or not.
401  if (!have_legal_moves) {
402  if (!Board().InCheck()) {
403  return std::vector<double>{DrawUtility(), DrawUtility()};
404  } else {
405  std::vector<double> returns(NumPlayers());
406  auto next_to_play = ColorToPlayer(Board().ToPlay());
407  returns[next_to_play] = LossUtility();
408  returns[OtherPlayer(next_to_play)] = WinUtility();
409  return returns;
410  }
411  }
412 
413  if (Board().IrreversibleMoveCounter() >= kNumReversibleMovesToDraw) {
414  // This is theoretically a draw that needs to be claimed, but we implement
415  // it as a forced draw for now.
416  return std::vector<double>{DrawUtility(), DrawUtility()};
417  }
418 
419  return absl::nullopt;
420 }
