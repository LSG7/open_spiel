117void DarkHexState::DoApplyAction(Action move) {
118 Player cur_player = CurrentPlayer(); // current player
119 auto& cur_view = (cur_player == 0 ? black_view_ : white_view_);
120
121 // Either occupied or not
122 if (game_version_ == GameVersion::kClassicalDarkHex) {
123 if (state_.BoardAt(move) == CellState::kEmpty) {
124 state_.ApplyAction(move);
125 }
126 } else {
127 SPIEL_CHECK_EQ(game_version_, GameVersion::kAbruptDarkHex);
128 if (state_.BoardAt(move) == CellState::kEmpty) {
129 state_.ApplyAction(move);
130 } else {
131 // switch the current player
132 state_.ChangePlayer();
133 }
134 }
135
136 SPIEL_CHECK_EQ(cur_view[move], CellState::kEmpty);
137 // Update the view - only using CellState::kBlack and CellState::kWhite
138 if (state_.BoardAt(move) == CellState::kBlack ||
139 state_.BoardAt(move) == CellState::kBlackNorth ||
140 state_.BoardAt(move) == CellState::kBlackSouth) {
141 cur_view[move] = CellState::kBlack;
142 } else if (state_.BoardAt(move) == CellState::kWhite ||
143 state_.BoardAt(move) == CellState::kWhiteEast ||
144 state_.BoardAt(move) == CellState::kWhiteWest) {
145 cur_view[move] = CellState::kWhite;
146 } else if (state_.BoardAt(move) == CellState::kBlackWin ||
147 state_.BoardAt(move) == CellState::kWhiteWin) {
148 cur_view[move] = state_.BoardAt(move);
149 } else {
150 SPIEL_CHECK_TRUE(false);
151 }
152 action_sequence_.push_back(std::pair<int, Action>(cur_player, move));
153}

155std::vector<Action> DarkHexState::LegalActions() const {
156 if (IsTerminal()) return {};
157 std::vector<Action> moves;
158 const Player player = CurrentPlayer();
159 const auto& cur_view = (player == 0 ? black_view_ : white_view_);
160
161 for (Action move = 0; move < num_cells_; ++move) {
162 if (cur_view[move] == CellState::kEmpty) {
163 moves.push_back(move);
164 }
165 }
166
167 return moves;
168}

170std::string DarkHexState::ViewToString(Player player) const {
171 const auto& cur_view = (player == 0 ? black_view_ : white_view_);
172 std::string str;
173
174 for (int r = 0; r < num_rows_; ++r) {
175 for (int c = 0; c < num_cols_; ++c) {
176 absl::StrAppend(&str, StateToString(cur_view[r * num_cols_ + c]));
177 }
178 if (r < (num_rows_ - 1)) {
179 absl::StrAppend(&str, "\n");
180 }
181 }
182 return str;
183}

185std::string DarkHexState::ActionSequenceToString(Player player) const {
186 SPIEL_CHECK_GE(player, 0);
187 SPIEL_CHECK_LT(player, num_players_);
188
189 std::string str;
190 for (const auto& player_with_action : action_sequence_) {
191 if (player_with_action.first == player) {
192 absl::StrAppend(&str, player_with_action.first, ",");
193 absl::StrAppend(&str, player_with_action.second, " ");
194 } else if (obs_type_ == ObservationType::kRevealNumTurns) {
195 absl::StrAppend(&str, player_with_action.first, ",? ");
196 } else {
197 SPIEL_CHECK_EQ(obs_type_, ObservationType::kRevealNothing);
198 }
199 }
200 return str;
201}

213void DarkHexState::InformationStateTensor(Player player,
214 absl::Span<float> values) const {
215 SPIEL_CHECK_GE(player, 0);
216 SPIEL_CHECK_LT(player, num_players_);
217
218 const auto& player_view = (player == 0 ? black_view_ : white_view_);
219
220 SPIEL_CHECK_EQ(values.size(), num_cells_ * kCellStates +
221 longest_sequence_ * (1 + bits_per_action_));
222 std::fill(values.begin(), values.end(), 0.);
223 for (int cell = 0; cell < num_cells_; ++cell) {
224 values[cell * kCellStates +
225 (static_cast<int>(player_view[cell]) - kMinValueCellState)] = 1.0;
226 }
227
228 // Encoding the sequence
229 int offset = num_cells_ * kCellStates;
230 for (const auto& player_with_action : action_sequence_) {
231 if (player_with_action.first == player) {
232 // Always include the observing player's actions.
233 values[offset] = player_with_action.first;
234 values[offset + 1 + player_with_action.second] = 1.0;
235 } else if (obs_type_ == ObservationType::kRevealNumTurns) {
236 // If the number of turns are revealed, then each of the other player's
237 // actions will show up as unknowns. Here, num_cells_ is used to
238 // encode "unknown".
239 values[offset] = player_with_action.first;
240 values[offset + 1 + num_cells_] = 1.0;
241 } else {
242 SPIEL_CHECK_EQ(obs_type_, ObservationType::kRevealNothing);
243 }
244 offset += (1 + bits_per_action_);
245 }
246}

258void DarkHexState::ObservationTensor(Player player,
259 absl::Span<float> values) const {
260 SPIEL_CHECK_GE(player, 0);
261 SPIEL_CHECK_LT(player, num_players_);
262 SPIEL_CHECK_EQ(values.size(), game_->ObservationTensorSize());
263 std::fill(values.begin(), values.end(), 0.);
264
265 const auto& player_view = (player == 0 ? black_view_ : white_view_);
266 for (int cell = 0; cell < num_cells_; ++cell) {
267 values[cell * kCellStates +
268 (static_cast<int>(player_view[cell]) - kMinValueCellState)] = 1.0;
269 }
270
271 if (obs_type_ == ObservationType::kRevealNumTurns) {
272 values[num_cells_ * kCellStates + action_sequence_.size()] = 1.0;
273 }
274}

280DarkHexGame::DarkHexGame(const GameParameters& params, GameType game_type)
281 : Game(game_type, params),
282 game_(std::static_pointer_cast<const hex::HexGame>(LoadGame(
283 "hex",
284 {{"num_cols", GameParameter(ParameterValue<int>(
285 "num_cols", ParameterValue<int>("board_size")))},
286 {"num_rows",
287 GameParameter(ParameterValue<int>(
288 "num_rows", ParameterValue<int>("board_size")))}}))),
289 num_cols_(
290 ParameterValue<int>("num_cols", ParameterValue<int>("board_size"))),
291 num_rows_(
292 ParameterValue<int>("num_rows", ParameterValue<int>("board_size"))),
293 num_cells_(num_cols_ * num_rows_),
294 bits_per_action_(num_cells_ + 1),
295 longest_sequence_(num_cells_ * 2 - 1) {
296 std::string obs_type = ParameterValue<std::string>("obstype");
297 if (obs_type == "reveal-nothing") {
298 obs_type_ = ObservationType::kRevealNothing;
299 } else if (obs_type == "reveal-numturns") {
300 obs_type_ = ObservationType::kRevealNumTurns;
301 } else {
302 SpielFatalError(absl::StrCat("Unrecognized observation type: ", obs_type));
303 }
304
305 std::string game_version = ParameterValue<std::string>("gameversion");
306 if (game_version == "cdh") {
307 game_version_ = GameVersion::kClassicalDarkHex;
308 } else if (game_version == "adh") {
309 game_version_ = GameVersion::kAbruptDarkHex;
310 } else {
311 SpielFatalError(absl::StrCat("Unrecognized game version: ", game_version));
312 }
313}
