15"""Discounted CFR and Linear CFR algorithms.
16
17This implements Discounted CFR and Linear CFR, from Noam Brown and Tuomas
18Sandholm, 2019, "Solving Imperfect-Information Games via Discounted Regret
19Minimization".
20See https://arxiv.org/abs/1809.04040.
21
22Linear CFR (LCFR), is identical to CFR, except on iteration `t` the updates to
23the regrets and average strategies are given weight `t`. (Equivalently, one
24could multiply the accumulated regret by t / (t+1) on each iteration.)
25
26Discounted CFR(alpha, beta, gamma) is defined by, at iteration `t`:
27- multiplying the positive accumulated regrets by (t^alpha / (t^alpha + 1))
28- multiplying the negative accumulated regrets by (t^beta / (t^beta + 1))
29- multiplying the contribution to the average strategy by t^gamma
30
31WARNING: This was contributed on Github, and the OpenSpiel team is not aware it
32has been verified we can reproduce the paper results.
33"""

93 def _compute_counterfactual_regret_for_player(self, state, policies,
94 reach_probabilities, player):
95 """Increments the cumulative regrets and policy for `player`.
96
97 Args:
98 state: The initial game state to analyze from.
99 policies: Unused. To be compatible with the `_CFRSolver` signature.
100 reach_probabilities: The probability for each player of reaching `state`
101 as a numpy array [prob for player 0, for player 1,..., for chance].
102 `player_reach_probabilities[player]` will work in all cases.
103 player: The 0-indexed player to update the values for. If `None`, the
104 update for all players will be performed.
105
106 Returns:
107 The utility of `state` for all players, assuming all players follow the
108 current policy defined by `self.Policy`.
109 """
110 if state.is_terminal():
111 return np.asarray(state.returns())
112
113 if state.is_chance_node():
114 state_value = 0.0
115 for action, action_prob in state.chance_outcomes():
116 assert action_prob > 0
117 new_state = state.child(action)
118 new_reach_probabilities = reach_probabilities.copy()
119 new_reach_probabilities[-1] *= action_prob
120 state_value += action_prob * self._compute_counterfactual_regret_for_player_compute_counterfactual_regret_for_player(
121 new_state, policies, new_reach_probabilities, player)
122 return state_value
123
124 current_player = state.current_player()
125 info_state = state.information_state_string(current_player)
126
127 # No need to continue on this history branch as no update will be performed
128 # for any player.
129 # The value we return here is not used in practice. If the conditional
130 # statement is True, then the last taken action has probability 0 of
131 # occurring, so the returned value is not impacting the parent node value.
132 if all(reach_probabilities[:-1] == 0):
133 return np.zeros(self._num_players_num_players)
134
135 state_value = np.zeros(self._num_players_num_players)
136
137 # The utilities of the children states are computed recursively. As the
138 # regrets are added to the information state regrets for each state in that
139 # information state, the recursive call can only be made once per child
140 # state. Therefore, the utilities are cached.
141 children_utilities = {}
142
143 info_state_node = self._info_state_nodes_info_state_nodes_info_state_nodes[info_state]
144 if policies is None:
145 info_state_policy = self._get_infostate_policy(info_state)
146 else:
147 info_state_policy = policies[current_player](info_state)
148 for action in state.legal_actions():
149 action_prob = info_state_policy.get(action, 0.)
150 new_state = state.child(action)
151 new_reach_probabilities = reach_probabilities.copy()
152 new_reach_probabilities[current_player] *= action_prob
153 child_utility = self._compute_counterfactual_regret_for_player_compute_counterfactual_regret_for_player(
154 new_state,
155 policies=policies,
156 reach_probabilities=new_reach_probabilities,
157 player=player)
158
159 state_value += action_prob * child_utility
160 children_utilities[action] = child_utility
161
162 # If we are performing alternating updates, and the current player is not
163 # the current_player, we skip the cumulative values update.
164 # If we are performing simultaneous updates, we do update the cumulative
165 # values.
166 simulatenous_updates = player is None
167 if not simulatenous_updates and current_player != player:
168 return state_value
169
170 reach_prob = reach_probabilities[current_player]
171 counterfactual_reach_prob = (
172 np.prod(reach_probabilities[:current_player]) *
173 np.prod(reach_probabilities[current_player + 1:]))
174 state_value_for_player = state_value[current_player]
175
176 for action, action_prob in info_state_policy.items():
177 cfr_regret = counterfactual_reach_prob * (
178 children_utilities[action][current_player] - state_value_for_player)
179
180 info_state_node = self._info_state_nodes_info_state_nodes_info_state_nodes[info_state]
181 info_state_node.cumulative_regret[action] += cfr_regret
182 if self._linear_averaging_linear_averaging:
183 info_state_node.cumulative_policy[action] += (
184 reach_prob * action_prob * (self._iteration**self.gamma))
185 else:
186 info_state_node.cumulative_policy[action] += reach_prob * action_prob
187
188 return state_value
