36class Process(object):
37 """A wrapper around `multiprocessing` that allows it to be used at google.
38
39 It spawns a subprocess from the given target function. That function should
40 take an additional argument `queue` which will get a bidirectional
41 _ProcessQueue for communicating with the parent.
42 """
43
44 def __init__(self, target, args=(), kwargs=None):
45 if kwargs is None:
46 kwargs = {}
47 elif "queue" in kwargs:
48 raise ValueError("`queue` is reserved for use by `Process`.")
49
50 q1 = multiprocessing.Queue()
51 q2 = multiprocessing.Queue()
52 self._queue = _ProcessQueue(q1, q2)
53 kwargs["queue"] = _ProcessQueue(q2, q1)
54
55 self._process = multiprocessing.Process(
56 target=target, args=args, kwargs=kwargs)
57 self._process.start()
58
59 def join(self, *args):
60 return self._process.join(*args)
61
62 @property
63 def exitcode(self):
64 return self._process.exitcode
65
66 @property
67 def queue(self):
68 return self._queue

71class _ProcessQueue(object):
72 """A bidirectional queue for talking to a subprocess.
73
74 `empty`, `get` and `get_nowait` act on the incoming queue, while
75 `full`, `put` and `put_nowait` act on the outgoing queue.
76
77 This class should only be created by the Process object.
78 """
79
80 def __init__(self, q_in, q_out):
81 self._q_in = q_in
82 self._q_out = q_out
83
84 def empty(self):
85 return self._q_in.empty()
86
87 def full(self):
88 return self._q_out.full()
89
90 def get(self, block=True, timeout=None):

92
