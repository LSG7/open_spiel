107 // n-dimensional array of uniform random numbers.
108 // Example:
109 // ZobristTable<int, 3, 4, 5> table;
110 //
111 // table[a][b][c] is a random int where a < 3, b < 4, c < 5
112 //
113 template <typename T, std::size_t InnerDim, std::size_t... OtherDims>
114 class ZobristTable {
115  public:
116  using Generator = std::mt19937_64;
117  using NestedTable = ZobristTable<T, OtherDims...>;
118 
119  ZobristTable(Generator::result_type seed) {
120  Generator generator(seed);
121  absl::uniform_int_distribution<Generator::result_type> dist;
122  data_.reserve(InnerDim);
123  for (std::size_t i = 0; i < InnerDim; ++i) {
124  data_.emplace_back(dist(generator));
125  }
126  }
127 
128  const NestedTable& operator[](std::size_t inner_index) const {
129  return data_[inner_index];
130  }
131 
132  private:
133  std::vector<NestedTable> data_;
134 };
