15"""Modular implementations of the PSRO meta algorithm.
16
17Allows the use of Restricted Nash Response, Nash Response, Uniform Response,
18and other modular matchmaking selection components users can add.
19
20This version works for N player, general sum games.
21
22One iteration of the algorithm consists of:
23
241) Computing the selection probability vector (or meta-strategy) for current
25strategies of each player, given their payoff.
262) [optional] Generating a mask over joint policies that restricts which policy
27to train against, ie. rectify the set of policies trained against. (This
28operation is designated by "rectify" in the code)
293) From every strategy used, generating a new best response strategy against the
30meta-strategy-weighted, potentially rectified, mixture of strategies using an
31oracle.
324) Updating meta game matrix with new game results.
33
34"""

52class PSROSolver(abstract_meta_trainer.AbstractMetaTrainer):
53 """A general implementation PSRO.
54
55 PSRO is the algorithm described in (Lanctot et Al., 2017,
56 https://arxiv.org/pdf/1711.00832.pdf ).
57
58 Subsequent work regarding PSRO's matchmaking and training has been performed
59 by David Balduzzi, who introduced Restricted Nash Response (RNR), Nash
60 Response (NR) and Uniform Response (UR).
61 RNR is Algorithm 4 in (Balduzzi, 2019, "Open-ended Learning in Symmetric
62 Zero-sum Games"). NR, Nash response, is algorithm 3.
63 Balduzzi et Al., 2019, https://arxiv.org/pdf/1901.08106.pdf
64
65 This implementation allows one to modularly choose different meta strategy
66 computation methods, or other user-written ones.
67 """

83 """Initialize the PSRO solver.
84
85 Arguments:
86 game: The open_spiel game object.
87 oracle: Callable that takes as input: - game - policy - policies played -
88 array representing the probability of playing policy i - other kwargs
89 and returns a new best response.
90 sims_per_entry: Number of simulations to run to estimate each element of
91 the game outcome matrix.
92 initial_policies: A list of initial policies for each player, from which
93 the optimization process will start.
94 rectifier: A string indicating the rectifying method. Can be :
95 - "" or None: Train against potentially all strategies.
96 - "rectified": Train only against strategies beaten by current
97 strategy.
98 training_strategy_selector: Callable taking (PSROSolver,
99 'number_policies_selected') and returning a list of list of selected
100 strategies to train from - this usually means copying weights and
101 rectifying with respect to the selected strategy's performance (One list
102 entry per player), or string selecting pre-implemented methods.
103 String value can be:
104 - "top_k_probabilites": selects the first
105 'number_policies_selected' policies with highest selection
106 probabilities.
107 - "probabilistic": randomly selects 'number_policies_selected'
108 with probabilities determined by the meta strategies.
109 - "exhaustive": selects every policy of every player.
110 - "rectified": only selects strategies that have nonzero chance of
111 being selected.
112 - "uniform": randomly selects 'number_policies_selected'
113 policies with uniform probabilities.
114 meta_strategy_method: String or callable taking a GenPSROSolver object and
115 returning two lists ; one list of meta strategies (One list entry per
116 player), and one list of joint strategies.
117 String value can be:
118 - alpharank: AlphaRank distribution on policies.
119 - "uniform": Uniform distribution on policies.
120 - "nash": Taking nash distribution. Only works for 2 player, 0-sum
121 games.
122 - "prd": Projected Replicator Dynamics, as described in Lanctot et
123 Al.
124 sample_from_marginals: A boolean, specifying whether to sample from
125 marginal (True) or joint (False) meta-strategy distributions.
126 number_policies_selected: Number of policies to return for each player.
127
128 n_noisy_copies: Number of noisy copies of each agent after training. 0 to
129 ignore this.
130 alpha_noise: lower bound on alpha noise value (Mixture amplitude.)
131 beta_noise: lower bound on beta noise value (Softmax temperature.)
132 **kwargs: kwargs for meta strategy computation and training strategy
133 selection.

265 def _restrict_target_training(self,
266 current_player,
267 ind,
268 total_policies,
269 probabilities_of_playing_policies,
270 restrict_target_training_bool,
271 epsilon=1e-12):
272 """Rectifies training.
273
274 Args:
275 current_player: the current player.
276 ind: Current strategy index of the player.
277 total_policies: all policies available to all players.
278 probabilities_of_playing_policies: meta strategies.
279 restrict_target_training_bool: Boolean specifying whether to restrict
280 training. If False, standard meta strategies are returned. Otherwise,
281 restricted joint strategies are returned.
282 epsilon: threshold below which we consider 0 sum of probabilities.
283
284 Returns:
285 Probabilities of playing each joint strategy (If rectifying) / probability
286 of each player playing each strategy (Otherwise - marginal probabilities)
287 """
288 true_shape = tuple([len(a) for a in total_policies])
289 if not restrict_target_training_bool:
290 return probabilities_of_playing_policies
291 else:
292 kept_probas = self._rectifier(
293 self, current_player, ind)
294 # Ensure probabilities_of_playing_policies has same shape as kept_probas.
295 probability = probabilities_of_playing_policies.reshape(true_shape)
296 probability = probability * kept_probas
297 prob_sum = np.sum(probability)
298
299 # If the rectified probabilities are too low / 0, we play against the
300 # non-rectified probabilities.
301 if prob_sum <= epsilon:
302 probability = probabilities_of_playing_policies
303 else:
304 probability /= prob_sum
305
306 return probability

308 def update_agents(self):
309 """Updates policies for each player at the same time by calling the oracle.
310
311 The resulting policies are appended to self._new_policies.
312 """
313
314 used_policies, used_indexes = self._training_strategy_selector(
315 self, self._number_policies_selected_number_policies_selected)
316
317 (sample_strategy,
318 total_policies,
319 probabilities_of_playing_policies) = self.get_policies_and_strategies()
320
321 # Contains the training parameters of all trained oracles.
322 # This is a list (Size num_players) of list (Size num_new_policies[player]),
323 # each dict containing the needed information to train a new best response.
324 training_parameters = [[] for _ in range(self._num_players_num_players)]
325
326 for current_player in range(self._num_players_num_players):
327 if self.sample_from_marginals:
328 currently_used_policies = used_policies[current_player]
329 current_indexes = used_indexes[current_player]
330 else:
331 currently_used_policies = [
332 joint_policy[current_player] for joint_policy in used_policies
333 ]
334 current_indexes = used_indexes[current_player]
335
336 for i in range(len(currently_used_policies)):
337 pol = currently_used_policies[i]
338 ind = current_indexes[i]
339
340 new_probabilities = self._restrict_target_training(
341 current_player, ind, total_policies,
342 probabilities_of_playing_policies,
343 self._rectify_training)
344
345 new_parameter = {
346 "policy": pol,
347 "total_policies": total_policies,
348 "current_player": current_player,
349 "probabilities_of_playing_policies": new_probabilities
350 }
351 training_parameters[current_player].append(new_parameter)
352
353 if self.symmetric_gamesymmetric_game:
354 # Notice that the following line returns N references to the same policy
355 # This might not be correct for certain applications.
356 # E.g., a DQN BR oracle with player_id information
357 self._policies = self._game_num_players * self._policies
358 self._num_players_num_players = self._game_num_players
359 training_parameters = [training_parameters[0]]
360
361 # List of List of new policies (One list per player)
362 self._new_policies = self._oracle(
363 self._game_game,
364 training_parameters,
365 strategy_sampler=sample_strategy,
366 using_joint_strategies=self._rectify_training or
367 not self.sample_from_marginals)
368
369 if self.symmetric_gamesymmetric_game:
370 # In a symmetric game, only one population is kept. The below lines
371 # therefore make PSRO consider only the first player during training,
372 # since both players are identical.
373 self._policies = [self._policies[0]]
374 self._num_players_num_players = 1

376 def update_empirical_gamestate(self, seed=None):
377 """Given new agents in _new_policies, update meta_games through simulations.
378
379 Args:
380 seed: Seed for environment generation.
381
382 Returns:
383 Meta game payoff matrix.
384 """
385 if seed is not None:
386 np.random.seed(seed=seed)
387 assert self._oracle is not None
388
389 if self.symmetric_gamesymmetric_game:
390 # Switch to considering the game as a symmetric game where players have
391 # the same policies & new policies. This allows the empirical gamestate
392 # update to function normally.
393 # Notice that the following line returns N references to the same policy
394 # This might not be correct for certain applications.
395 # E.g., a DQN BR oracle with player_id information
396 self._policies = self._game_num_players * self._policies
397 self._new_policies = self._game_num_players * self._new_policies
398 self._num_players_num_players = self._game_num_players
399
400 # Concatenate both lists.
401 updated_policies = [
402 self._policies[k] + self._new_policies[k]
403 for k in range(self._num_players_num_players)
404 ]
405
406 # Each metagame will be (num_strategies)^self._num_players.
407 # There are self._num_player metagames, one per player.
408 total_number_policies = [
409 len(updated_policies[k]) for k in range(self._num_players_num_players)
410 ]
411 number_older_policies = [
412 len(self._policies[k]) for k in range(self._num_players_num_players)
413 ]
414 number_new_policies = [
415 len(self._new_policies[k]) for k in range(self._num_players_num_players)
416 ]
417
418 # Initializing the matrix with nans to recognize unestimated states.
419 meta_games = [
420 np.full(tuple(total_number_policies), np.nan)
421 for k in range(self._num_players_num_players)
422 ]
423
424 # Filling the matrix with already-known values.
425 older_policies_slice = tuple(
426 [slice(len(self._policies[k])) for k in range(self._num_players_num_players)])
427 for k in range(self._num_players_num_players):
428 meta_games[k][older_policies_slice] = self._meta_games[k]
429
430 # Filling the matrix for newly added policies.
431 for current_player in range(self._num_players_num_players):
432 # Only iterate over new policies for current player ; compute on every
433 # policy for the other players.
434 range_iterators = [
435 range(total_number_policies[k]) for k in range(current_player)
436 ] + [range(number_new_policies[current_player])] + [
437 range(total_number_policies[k])
438 for k in range(current_player + 1, self._num_players_num_players)
439 ]
440 for current_index in itertools.product(*range_iterators):
441 used_index = list(current_index)
442 used_index[current_player] += number_older_policies[current_player]
443 if np.isnan(meta_games[current_player][tuple(used_index)]):
444 estimated_policies = [
445 updated_policies[k][current_index[k]]
446 for k in range(current_player)
447 ] + [
448 self._new_policies[current_player][current_index[current_player]]
449 ] + [
450 updated_policies[k][current_index[k]]
451 for k in range(current_player + 1, self._num_players_num_players)
452 ]
453
454 if self.symmetric_gamesymmetric_game:
455 # TODO(author4): This update uses ~2**(n_players-1) * sims_per_entry
456 # samples to estimate each payoff table entry. This should be
457 # brought to sims_per_entry to coincide with expected behavior.
458
459 utility_estimates = self.sample_episodes(estimated_policies,
460 self._sims_per_entry)
461
462 player_permutations = list(itertools.permutations(list(range(
463 self._num_players_num_players))))
464 for permutation in player_permutations:
465 used_tuple = tuple([used_index[i] for i in permutation])
466 for player in range(self._num_players_num_players):
467 if np.isnan(meta_games[player][used_tuple]):
468 meta_games[player][used_tuple] = 0.0
469 meta_games[player][used_tuple] += utility_estimates[
470 permutation[player]] / len(player_permutations)
471 else:
472 utility_estimates = self.sample_episodes(estimated_policies,
473 self._sims_per_entry)
474 for k in range(self._num_players_num_players):
475 meta_games[k][tuple(used_index)] = utility_estimates[k]
476
477 if self.symmetric_gamesymmetric_game:
478 # Make PSRO consider that we only have one population again, as we
479 # consider that we are in a symmetric game (No difference between players)
480 self._policies = [self._policies[0]]
481 self._new_policies = [self._new_policies[0]]
482 updated_policies = [updated_policies[0]]
483 self._num_players_num_players = 1
484
485 self._meta_games = meta_games
486 self._policies = updated_policies
487 return meta_games
