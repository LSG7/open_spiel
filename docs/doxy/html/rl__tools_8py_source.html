58class LinearSchedule(ValueSchedule):
59 """A simple linear schedule."""
60
61 def __init__(self, init_val, final_val, num_steps):
62 """A simple linear schedule.
63
64 Once the the number of steps is reached, value is always equal to the final
65 value.
66
67 Arguments:
68 init_val: the initial value.
69 final_val: the final_value
70 num_steps: the number of steps to get from the initial to final value.
71 """
72 super(LinearSchedule, self).__init__()
73 self._value = init_val
74 self._final_value = final_val
75 assert isinstance(num_steps, int)
76 self._num_steps = num_steps
77 self._steps_taken = 0
78 self._increment = (final_val - init_val) / num_steps
79
80 def step(self):
81 self._steps_taken += 1
82 if self._steps_taken < self._num_steps:
83 self._value += self._increment
84 elif self._steps_taken == self._num_steps:
85 self._value = self._final_value
86 return self._value
