62int FPSBAState::CurrentPlayer() const {
63 if (valuations_.size() < num_players_) return kChancePlayerId;
64 if (bids_.size() < num_players_) return bids_.size();
65 if (winner_ == kInvalidPlayer) return kChancePlayerId;
66 return kTerminalPlayerId;

69std::vector<Action> FPSBAState::EligibleWinners() const {
70 int max_bid = *std::max_element(bids_.begin(), bids_.end());
71 std::vector<Action> eligibles;
72 for (auto player = Player{0}; player < num_players_; player++) {
73 if (bids_[player] == max_bid) {
74 eligibles.push_back(player);
75 }
76 }
77 return eligibles;
78}

80std::vector<Action> FPSBAState::LegalActions() const {
81 if (valuations_.size() < num_players_) {
82 std::vector<Action> values(max_value_);
83 std::iota(values.begin(), values.end(), 1);
84 return values;
85 }
86 if (bids_.size() < num_players_) {
87 std::vector<Action> bids(valuations_[bids_.size()]);
88 std::iota(bids.begin(), bids.end(), 0);
89 return bids;
90 }
91 if (winner_ == kInvalidPlayer) {
92 return EligibleWinners();
93 }
94 return {};
95}

150void FPSBAState::InformationStateTensor(Player player,
151 absl::Span<float> values) const {
152 SPIEL_CHECK_GE(player, 0);
153 SPIEL_CHECK_LT(player, num_players_);
154 SPIEL_CHECK_EQ(values.size(), 2 * max_value_ + num_players_);
155 std::fill(values.begin(), values.end(), 0);
156 auto cursor = values.begin();
157 cursor[player] = 1;
158 cursor += num_players_;
159 if (valuations_.size() > player) {
160 cursor[valuations_[player] - 1] = 1;
161 }
162 cursor += max_value_;
163 if (bids_.size() > player) {
164 cursor[bids_[player]] = 1;
165 }
166 cursor += max_value_;
167 SPIEL_CHECK_EQ(cursor - values.begin(), values.size());
168}

188ActionsAndProbs FPSBAState::ChanceOutcomes() const {
189 ActionsAndProbs valuesAndProbs;
190 if (valuations_.size() < num_players_) {
191 for (int i = 1; i <= max_value_; i++) {
192 valuesAndProbs.push_back(std::make_pair(i, 1. / max_value_));
193 }
194 } else if (bids_.size() == num_players_ && winner_ == kInvalidPlayer) {
195 int max_bid = *std::max_element(bids_.begin(), bids_.end());
196 int num_tie = std::count(bids_.begin(), bids_.end(), max_bid);
197 for (auto player = Player{0}; player < num_players_; player++) {
198 if (bids_[player] == max_bid) {
199 valuesAndProbs.push_back(std::make_pair(player, 1. / num_tie));
200 }
201 }
202 } else {
203 SpielFatalError("This isn't a chance node");
204 }
205 return valuesAndProbs;
206}
