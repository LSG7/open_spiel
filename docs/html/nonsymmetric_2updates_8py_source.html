106  def mirror_descent_step(self, params, grads, t):
107  """Entropic mirror descent on exploitability.
108 
109  Args:
110  params: tuple of variables to be updated (dist - a list of np.arrays)
111  grads: tuple of variable gradients (grad_dist - a list of np.arrays)
112  t: int, solver iteration (unused)
113  Returns:
114  new_params: tuple of update params (new_dist)
115  """
116  del t
117  lr_dist = self.lrslrs[0]
118  new_params = []
119  for dist_i, dist_grad_i in zip(params[0], grads[0]):
120  new_dist_i = np.clip(dist_i, 0, np.inf)
121  new_dist_i = special.softmax(np.log(new_dist_i) - lr_dist * dist_grad_i)
122  new_params.append(new_dist_i)
