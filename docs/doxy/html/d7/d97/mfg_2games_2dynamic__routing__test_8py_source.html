39class SocialOptimumBraess(policy.Policy):
40
41 def action_probabilities(self, state, player_id=None):
42 legal_actions = state.legal_actions()
43 if not legal_actions:
44 return {dynamic_routing_utils.NO_POSSIBLE_ACTION: 1.0}
45 elif len(legal_actions) == 1:
46 return {legal_actions[0]: 1.0}
47 else:
48 if legal_actions[0] == 1:
49 return {1: 0.5, 2: 0.5}
50 elif legal_actions[0] == 3:
51 return {4: 1.0}
52 raise ValueError(f"{legal_actions} is not correct.")

55class NashEquilibriumBraess(policy.Policy):
56
57 def action_probabilities(self, state, player_id=None):
58 legal_actions = state.legal_actions()
59 if not legal_actions:
60 return {dynamic_routing_utils.NO_POSSIBLE_ACTION: 1.0}
61 elif len(legal_actions) == 1:
62 return {legal_actions[0]: 1.0}
63 else:
64 if legal_actions[0] == 1:
65 return {1: 0.75, 2: 0.25}
66 elif legal_actions[0] == 3:
67 return {3: 2 / 3, 4: 1 / 3}
68 raise ValueError(f"{legal_actions} is not correct. {state}.")

187 def test_observer_correct(self):
188 """Checks that the observer is correctly updated."""
189 game = pyspiel.load_game("python_mfg_dynamic_routing")
190 num_locations, steps = 8, 10
191 self.assertEqual(game.num_distinct_actions(), num_locations)
192 self.assertEqual(game.max_game_length(), steps)
193 py_obs = make_observation(game)
194
195 state = game.new_initial_state()
196 self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
197
198 state.apply_action(0)
199 self.assertEqual(state.current_player(), 0)
200
201 location, destination = 7, 6
202 self.assertEqual(state.get_location_as_int(), location)
203 self.assertEqual(state.get_destination_as_int(), destination)
204
205 py_obs.set_from(state, state.current_player())
206 obs_size = num_locations * 2 + steps + 2
207 expected_tensor = np.zeros(obs_size)
208 # location = 7
209 # destination + num_locations = 14
210 # time + 2 * num_locations = 16
211 # waiting bit at last index.
212 expected_tensor[[7, 14, 16]] = 1
213 npt.assert_array_equal(py_obs.tensor, expected_tensor)

240class CppVsPythonMeanFieldRoutingGameTest(parameterized.TestCase):
241
242 @parameterized.named_parameters(
243 ("python", ("python_mfg_dynamic_routing(max_num_time_step=100,"
244 "time_step_length=0.05)")),
245 ("cpp", ("mfg_dynamic_routing(max_num_time_step=100,"
246 "time_step_length=0.05,network=braess)")))
247 def test_braess_paradox_game(self, game_name):
248 """Test that Braess paradox can be reproduced with the mean field game."""
249 mfg_game = pyspiel.load_game(game_name)
250
251 ne_policy = NashEquilibriumBraess(mfg_game, 1)
252 self.assertEqual(
253 -policy_value.PolicyValue(
254 mfg_game, distribution.DistributionPolicy(mfg_game, ne_policy),
255 ne_policy).value(mfg_game.new_initial_state()), 3.75)
256 self.assertEqual(nash_conv.NashConv(mfg_game, ne_policy).nash_conv(), 0.0)
257
258 so_policy = SocialOptimumBraess(mfg_game, 1)
259 self.assertEqual(
260 -policy_value.PolicyValue(
261 mfg_game, distribution.DistributionPolicy(mfg_game, so_policy),
262 so_policy).value(mfg_game.new_initial_state()), 3.5)
263 self.assertEqual(nash_conv.NashConv(mfg_game, so_policy).nash_conv(), 0.75)
