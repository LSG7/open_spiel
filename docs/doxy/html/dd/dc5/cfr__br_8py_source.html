31class CFRBRSolver(_CFRSolverBase):
32 """Implements the Counterfactual Regret Minimization (CFR-BR) algorithm.
33
34 This is Counterfactual Regret Minimization against Best Response, from
35 Michael Johanson and al., 2012, Finding Optimal Abstract Strategies in
36 Extensive-Form Games,
37 https://poker.cs.ualberta.ca/publications/AAAI12-cfrbr.pdf).
38
39 The algorithm
40 computes an approximate Nash policy for n-player zero-sum games, but the
41 implementation is currently restricted to 2-player.
42
43 It uses an exact Best Response and full tree traversal.
44
45 One iteration for a n-player game consists of the following:
46
47 - Compute the BR of each player against the rest of the players.
48 - Then, for each player p sequentially (from player 0 to N-1):
49 - Compute the conterfactual reach probabilities and action values for player
50 p, playing against the set of the BR for all other players.
51 - Update the player `p` policy using these values.
52
53 CFR-BR should converge with high probability (see the paper), but we can also
54 compute the time-averaged strategy.
55
56 The implementation reuses the `action_values_vs_best_response` module and
57 thus uses TabularPolicies. This will run only for smallish games.
58 """
59
60 def __init__(self, game, linear_averaging=False, regret_matching_plus=False):
61 # pyformat: disable
62 """Initializer.
63
64 Args:
65 game: The `pyspiel.Game` to run on.
66 linear_averaging: Whether to use linear averaging, i.e.
67 cumulative_policy[info_state][action] += (
68 iteration_number * reach_prob * action_prob)
69
70 or not:
71
72 cumulative_policy[info_state][action] += reach_prob * action_prob
73 regret_matching_plus: Whether to use Regret Matching+:
74 cumulative_regrets = max(cumulative_regrets + regrets, 0)
75 or simply regret matching:
76 cumulative_regrets = cumulative_regrets + regrets
77 """

95 def _compute_best_responses(self):
96 """Computes each player best-response against the pool of other players."""
97
98 def policy_fn(state):
99 key = state.information_state_string()
100 return self._get_infostate_policy(key)
101
102 current_policy = policy.tabular_policy_from_callable(self._game, policy_fn)
103
104 for player_id in range(self._game.num_players()):
105 self._best_responses[player_id] = exploitability.best_response(
106 self._game, current_policy, player_id)

108 def evaluate_and_update_policy(self):
109 """Performs a single step of policy evaluation and policy improvement."""
110 self._iteration += 1
111
112 self._compute_best_responses()
113
114 for player in range(self._num_players):
115 # We do not use policies, to not have to call `state.information_state`
116 # several times (in here and within policy).
117 policies = []
118 for p in range(self._num_players):
119 # pylint: disable=g-long-lambda
120 policies.append(
121 lambda infostate_str, p=p:
122 {self._best_responses[p]["best_response_action"][infostate_str]: 1})
123 # pylint: enable=g-long-lambda
124 policies[player] = self._get_infostate_policy
125
126 self._compute_counterfactual_regret_for_player(
127 state=self._root_node,
128 policies=policies,
129 reach_probabilities=np.ones(self._num_players + 1),
130 player=player)
131
132 if self._regret_matching_plus:
133 _apply_regret_matching_plus_reset(self._info_state_nodes)
134 _update_current_policy(self._current_policy, self._info_state_nodes)
