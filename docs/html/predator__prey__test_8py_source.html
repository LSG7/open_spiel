233  'init_distrib': ' '.join(str(v) for v in init_distrib.flatten()),
234  },

236  state = game.new_initial_state_for_population(population)
237  # Initial chance node.
238  self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
239  state.apply_action(predator_prey.pos_to_merged(initial_pos, state.size))
240  self.assertEqual(state.current_player(), population)
241  npt.assert_array_equal(state.pos, initial_pos)
242  state.apply_action(state._NEUTRAL_ACTION)
243  npt.assert_array_equal(state.pos, initial_pos)
244  self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
245  state.apply_action(state._NEUTRAL_ACTION)
246  self.assertEqual(state.current_player(), pyspiel.PlayerId.MEAN_FIELD)
247 
248  # Maps states (in string representation) to their proba.
249  dist = {}
250  for x in range(state.size):
251  for y in range(state.size):
252  for pop in range(len(reward_matrix)):
253  state_str = state.state_to_str(
254  np.array([x, y]),
255  state.t,
256  pop,
257  player_id=pyspiel.PlayerId.MEAN_FIELD,
258  )
259  dist[state_str] = distributions[pop][y][x]
260  support = state.distribution_support()
