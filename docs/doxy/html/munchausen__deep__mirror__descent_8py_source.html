55 def __init__(
56 self,
57 player_id,
58 state_representation_size,
59 num_actions,
60 # Training options.
61 batch_size: int = 128,
62 learn_every: int = 64,
63 epsilon_start: float = 0.1,
64 epsilon_end: float = 0.1,
65 epsilon_decay_duration: int = int(20e6),
66 epsilon_power: float = 1.0,
67 discount_factor: float = 1.0,
68 # Replay buffer options.
69 replay_buffer_capacity: int = int(2e5),
70 min_buffer_size_to_learn: int = 1000,
71 replay_buffer_class=ReplayBuffer,
72 # Loss and optimizer options.
73 optimizer: str = "sgd",
74 learning_rate: float = 0.01,
75 loss: str = "mse",
76 huber_loss_parameter: float = 1.0,
77 # Network options.
78 update_target_network_every: int = 19200,
79 hidden_layers_sizes=128,
80 qnn_params_init=None,
81 # Munchausen options.
82 tau=0.05,
83 alpha=0.9,
84 reset_replay_buffer_on_update: bool = True,
85 gradient_clipping: Optional[float] = None,
86 with_munchausen: bool = True,
87 seed: int = 42):
88 """Initialize the Munchausen DQN agent."""
89 self.player_id = int(player_id)
90 self._num_actions = num_actions
91
92 self._batch_size = batch_size
93 self._learn_every = learn_every
94 self._epsilon_start = epsilon_start
95 self._epsilon_end = epsilon_end
96 self._epsilon_decay_duration = epsilon_decay_duration
97 self._epsilon_power = epsilon_power
98 self._discount_factor = discount_factor
99 self._reset_replay_buffer_on_update = reset_replay_buffer_on_update
100
101 self._tau = tau
102 self._alpha = alpha
103
104 # If true, the target uses Munchausen penalty terms.
105 self._with_munchausen = with_munchausen
106
107 self._prev_action = None
108 self._prev_legal_action = None
109 self._prev_time_step = None
110
111 # Used to select actions.
112 self._rs = np.random.RandomState(seed)
113
114 # Step counter to keep track of learning, eps decay and target network.
115 self._step_counter = 0
116
117 # Keep track of the last training loss achieved in an update step.
118 self._last_loss_value = None
119
120 # Create the replay buffer.
121 if not isinstance(replay_buffer_capacity, int):
122 raise ValueError("Replay buffer capacity not an integer.")
123 self._replay_buffer = replay_buffer_class(replay_buffer_capacity)
124 self._min_buffer_size_to_learn = min_buffer_size_to_learn
125
126 # Create the Q-network.
127 self._update_target_network_every = update_target_network_every
128
129 if isinstance(hidden_layers_sizes, int):
130 hidden_layers_sizes = [hidden_layers_sizes]
131
132 def network(x):
133 mlp = hk.nets.MLP(hidden_layers_sizes + [num_actions])
134 return mlp(x)
135
136 self.hk_network = hk.without_apply_rng(hk.transform(network))
137 self.hk_network_apply = jax.jit(self.hk_network.apply)
138
139 if qnn_params_init:
140 self._params_q_network = _copy_params(qnn_params_init)
141 self._params_target_q_network = _copy_params(qnn_params_init)
142 self._params_prev_q_network = _copy_params(qnn_params_init)
143 else:
144 rng = jax.random.PRNGKey(seed)
145 x = jnp.ones([1, state_representation_size])
146 self._params_q_network = self.hk_network.init(rng, x)
147 self._params_target_q_network = self.hk_network.init(rng, x)
148 self._params_prev_q_network = self.hk_network.init(rng, x)
149
150 # Create the loss function and the optimizer.
151 if loss == "mse":
152 self._loss_func = lambda x: jnp.mean(x**2)
153 elif loss == "huber":
154 self._loss_func = lambda x: jnp.mean( # pylint: disable=g-long-lambda
155 rlax.huber_loss(x, huber_loss_parameter))
156 else:
157 raise ValueError("Not implemented, choose from 'mse', 'huber'.")
158
159 if optimizer == "adam":
160 optimizer = optax.adam(learning_rate)
161 elif optimizer == "sgd":
162 optimizer = optax.sgd(learning_rate)
163 else:
164 raise ValueError("Not implemented, choose from 'adam' and 'sgd'.")
165
166 # Clipping the gradients prevent divergence and allow more stable training.
167 if gradient_clipping:
168 optimizer = optax.chain(optimizer,
169 optax.clip_by_global_norm(gradient_clipping))
170
171 opt_init, opt_update = optimizer.init, optimizer.update

173 def _stochastic_gradient_descent(params, opt_state, gradient):
174 updates, opt_state = opt_update(gradient, opt_state)
175 new_params = optax.apply_updates(params, updates)
176 return new_params, opt_state
177
178 self._opt_update_fn = _stochastic_gradient_descent
179 self._opt_state = opt_init(self._params_q_network)
180 self._loss_and_grad = jax.value_and_grad(self._loss, has_aux=False)
181 self._jit_update = jax.jit(self._get_update())

183 def step(self,
184 time_step,
185 is_evaluation=False,
186 add_transition_record=True,
187 use_softmax=False,
188 tau: Optional[float] = None):
189 """Returns the action to be taken and updates the Q-network if needed.
190
191 Args:
192 time_step: an instance of rl_environment.TimeStep.
193 is_evaluation: bool, whether this is a training or evaluation call.
194 add_transition_record: Whether to add to the replay buffer on this step.
195 use_softmax: Uses soft-max action selection.
196 tau: Tau for soft-max action selection. If None, then the training value
197 will be used.
198
199 Returns:
200 A `rl_agent.StepOutput` containing the action probs and chosen action.
201 """
202
203 # Act step: don't act at terminal info states or if its not our turn.
204 if (not time_step.last()) and (time_step.is_simultaneous_move() or
205 self.player_id == int(
206 time_step.current_player())):
207 # Act according to epsilon-greedy or soft-max for current Q-network.
208 info_state = time_step.observations["info_state"][self.player_id]
209 legal_actions = time_step.observations["legal_actions"][self.player_id]
210 if use_softmax:
211 action, probs = self._softmax(info_state, legal_actions,
212 self._tau if tau is None else tau)
213 else:
214 epsilon = self._get_epsilon(is_evaluation)
215 action, probs = self._epsilon_greedy(info_state, legal_actions, epsilon)
216 else:
217 action = None
218 probs = []
219
220 # Don't mess up with the state during evaluation.
221 if not is_evaluation:
222 self._step_counter += 1
223
224 if self._step_counter % self._learn_every == 0:
225 self._last_loss_value = self.learn()
226
227 if self._step_counter % self._update_target_network_every == 0:
228 self._params_target_q_network = _copy_params(self._params_q_network)
229
230 if self._prev_time_step and add_transition_record:
231 # We may omit record adding here if it's done elsewhere.
232 self.add_transition(self._prev_time_step, self._prev_action,
233 self._prev_legal_action, time_step)
234
235 if time_step.last(): # prepare for the next episode.
236 self._prev_time_step = None
237 self._prev_action = None
238 self._prev_legal_action = None
239 else:
240 self._prev_time_step = time_step
241 self._prev_action = action
242 self._prev_legal_action = legal_actions
243
244 return rl_agent.StepOutput(action=action, probs=probs)

246 def add_transition(self, prev_time_step, prev_action, prev_legal_actions,
247 time_step):
248 """Adds the new transition using `time_step` to the replay buffer.
249
250 Adds the transition from `self._prev_time_step` to `time_step` by
251 `self._prev_action`.
252
253 Args:
254 prev_time_step: prev ts, an instance of rl_environment.TimeStep.
255 prev_action: int, action taken at `prev_time_step`.
256 prev_legal_actions: Previous legal actions.
257 time_step: current ts, an instance of rl_environment.TimeStep.
258 """
259 assert prev_time_step is not None
260 next_legal_actions = (
261 time_step.observations["legal_actions"][self.player_id])
262 next_legal_one_hots = self._to_one_hot(next_legal_actions)
263 # Added for deep OMD: keep previous action mask.
264 prev_legal_one_hots = self._to_one_hot(prev_legal_actions)
265
266 transition = Transition(
267 info_state=(
268 prev_time_step.observations["info_state"][self.player_id][:]),
269 action=prev_action,
270 legal_one_hots=prev_legal_one_hots,
271 reward=time_step.rewards[self.player_id],
272 next_info_state=time_step.observations["info_state"][self.player_id][:],
273 is_final_step=float(time_step.last()),
274 next_legal_one_hots=next_legal_one_hots)
275 self._replay_buffer.add(transition)

277 def _get_action_probs(self, params, info_states, legal_one_hots):
278 """Returns the soft-max action probability distribution."""
279 q_values = self.hk_network.apply(params, info_states)
280 legal_q_values = q_values + (1 - legal_one_hots) * ILLEGAL_ACTION_PENALTY
281 return jax.nn.softmax(legal_q_values / self._tau)

283 def _loss(self, params, params_target, params_prev, info_states, actions,
284 legal_one_hots, rewards, next_info_states, are_final_steps,
285 next_legal_one_hots):
286 """Returns the Munchausen loss."""
287 # Target with 2 parts: reward and value for next state; each part is
288 # modified according to the Munchausen trick.
289 q_values = self.hk_network.apply(params, info_states)
290 target_q_values = self.hk_network.apply(params_target, next_info_states)
291
292 r_term = rewards
293 if self._with_munchausen:
294 probs = self._get_action_probs(params_prev, info_states, legal_one_hots)
295 prob_prev_action = jnp.sum(probs * actions, axis=-1)
296 penalty_pi = jnp.log(jnp.clip(prob_prev_action, MIN_ACTION_PROB))
297 r_term += self._alpha * self._tau * penalty_pi
298
299 if self._with_munchausen:
300 # Average value over actions + extra log term.
301 # We clip the probabilities to avoid NaNs in the log term.
302 next_probs = self._get_action_probs(params_prev, next_info_states,
303 next_legal_one_hots)
304 q_term_values = next_probs * (
305 target_q_values -
306 self._tau * jnp.log(jnp.clip(next_probs, MIN_ACTION_PROB)))
307 q_term = jnp.sum(q_term_values, axis=-1)
308 else:
309 # Maximum value.
310 max_next_q = jnp.max(
311 target_q_values + (1 - legal_one_hots) * ILLEGAL_ACTION_PENALTY,
312 axis=-1)
313 max_next_q = jax.numpy.where(
314 1 - are_final_steps, max_next_q, jnp.zeros_like(max_next_q))
315 q_term = max_next_q
316
317 target = (r_term + (1 - are_final_steps) * self._discount_factor * q_term)
318 target = jax.lax.stop_gradient(target)
319
320 predictions = jnp.sum(q_values * actions, axis=-1)
321
322 return self._loss_func(predictions - target)

327 def update(params, params_target, params_prev, opt_state, info_states,
328 actions, legal_one_hots, rewards, next_info_states,
329 are_final_steps, next_legal_one_hots):
330 loss_val, grad_val = self._loss_and_grad(params, params_target,
331 params_prev, info_states,
332 actions, legal_one_hots, rewards,
333 next_info_states,
334 are_final_steps,
335 next_legal_one_hots)
336 new_params, new_opt_state = self._opt_update_fn(params, opt_state,
337 grad_val)
338 return new_params, new_opt_state, loss_val

342 def _to_one_hot(self, a, value=1.0):
343 """Returns the one-hot encoding of the action."""
344 a_one_hot = np.zeros(self._num_actions)
345 a_one_hot[a] = value
346 return a_one_hot

348 def learn(self):
349 """Compute the loss on sampled transitions and perform a Q-network update.
350
351 If there are not enough elements in the buffer, no loss is computed and
352 `None` is returned instead.
353
354 Returns:
355 The average loss obtained on this batch of transitions or `None`.
356 """
357
358 if (len(self._replay_buffer) < self._batch_size or
359 len(self._replay_buffer) < self._min_buffer_size_to_learn):
360 return None
361
362 transitions = self._replay_buffer.sample(self._batch_size)
363 info_states = np.asarray([t.info_state for t in transitions])
364 actions = np.asarray([self._to_one_hot(t.action) for t in transitions])
365 legal_one_hots = np.asarray([t.legal_one_hots for t in transitions])
366 rewards = np.asarray([t.reward for t in transitions])
367 next_info_states = np.asarray([t.next_info_state for t in transitions])
368 are_final_steps = np.asarray([t.is_final_step for t in transitions])
369 next_legal_one_hots = np.asarray(
370 [t.next_legal_one_hots for t in transitions])
371
372 self._params_q_network, self._opt_state, loss_val = self._jit_update(
373 self._params_q_network, self._params_target_q_network,
374 self._params_prev_q_network, self._opt_state, info_states, actions,
375 legal_one_hots, rewards, next_info_states, are_final_steps,
376 next_legal_one_hots)
377
378 return loss_val

380 def _epsilon_greedy(self, info_state, legal_actions, epsilon):
381 """Returns a valid epsilon-greedy action and action probabilities.
382
383 Args:
384 info_state: hashable representation of the information state.
385 legal_actions: list of legal actions at `info_state`.
386 epsilon: float, probability of taking an exploratory action.
387
388 Returns:
389 A valid epsilon-greedy action and action probabilities.
390 """
391 if self._rs.rand() < epsilon:
392 action = self._rs.choice(legal_actions)
393 probs = self._to_one_hot(legal_actions, value=1.0 / len(legal_actions))
394 return action, probs
395
396 info_state = np.reshape(info_state, [1, -1])
397 q_values = self.hk_network_apply(self._params_q_network, info_state)[0]
398 legal_one_hot = self._to_one_hot(legal_actions)
399 legal_q_values = q_values + (1 - legal_one_hot) * ILLEGAL_ACTION_PENALTY
400 action = int(np.argmax(legal_q_values))
401 probs = self._to_one_hot(action)
402 return action, probs
403
404 def _get_epsilon(self, is_evaluation):
405 """Returns the evaluation or decayed epsilon value."""
406 if is_evaluation:
407 return 0.0
408
409 decay_steps = min(self._step_counter, self._epsilon_decay_duration)
410 decayed_epsilon = (
411 self._epsilon_end + (self._epsilon_start - self._epsilon_end) *
412 (1 - decay_steps / self._epsilon_decay_duration)**self._epsilon_power)
413 return decayed_epsilon
414
415 def _softmax(self, info_state, legal_actions,
416 tau: float) -> Tuple[int, np.ndarray]:
417 """Returns a valid soft-max action and action probabilities."""
418 info_state = np.reshape(info_state, [1, -1])
419 q_values = self.hk_network_apply(self._params_q_network, info_state)[0]
420 legal_one_hot = self._to_one_hot(legal_actions)
421 legal_q_values = q_values + (1 - legal_one_hot) * ILLEGAL_ACTION_PENALTY
422 # Apply temperature and subtract the maximum value for numerical stability.
423 temp = legal_q_values / tau
424 unnormalized = np.exp(temp - np.amax(temp))
425 probs = unnormalized / unnormalized.sum()
426 action = self._rs.choice(legal_actions, p=probs[legal_actions])
427 return action, probs

430 """Updates the parameters of the previous Q-network."""

433 # Also reset the replay buffer to avoid having transitions from the
434 # previous policy.
435 self._replay_buffer.reset()

454class DeepOnlineMirrorDescent(object):
455 """The deep online mirror descent algorithm."""
456
457 def __init__(self,
458 game,
459 envs,
460 agents,
461 eval_every=200,
462 num_episodes_per_iteration=1000,
463 logging_fn: Optional[Callable[[int, int, Dict[str, Any]],
464 None]] = None):
465 """Initializes mirror descent.
466
467 Args:
468 game: The game,
469 envs: RL environment for each player.
470 agents: Munchausen DQN agents for each player.
471 eval_every: Number of training episodes between two evaluations.
472 num_episodes_per_iteration: Number of training episodes for each
473 iiteration.
474 logging_fn: Callable for logging the metrics. The arguments will be the
475 current iteration, episode and a dictionary of metrics to log.
476 """
477 assert len(envs) == len(agents)
478 # Make sure that the agents are all MunchausenDQN.
479 for agent in agents:
480 assert isinstance(agent, MunchausenDQN)
481
482 self._game = game
483
484 self._eval_every = eval_every
485 self._num_episodes_per_iteration = num_episodes_per_iteration
486
487 self._envs = envs
488 self._agents = agents
489 self._use_observation = envs[0].use_observation
490
491 self._iteration = 0
492
493 if logging_fn is None:
494 logging_fn = lambda it, ep, vals: logging.info("%d/%d %r", it, ep, vals)
495 self._logging_fn = logging_fn
496
497 # Set the initial policy and distribution.
498 self._update_policy_and_distribution()
499
500 def _train_agents(self):
501 """Trains the agents.
502
503 This will evaluate the Q-network for current policy and distribution.
504 """
505 for ep in range(self._num_episodes_per_iteration):
506 for env, agent in zip(self._envs, self._agents):
507 time_step = env.reset()
508 while not time_step.last():
509 agent_output = agent.step(time_step, use_softmax=False)
510 action_list = [agent_output.action]
511 time_step = env.step(action_list)
512
513 # Episode is over, step all agents with final info state.
514 agent.step(time_step, use_softmax=False)
515
516 if (ep + 1) % self._eval_every == 0:
517 metrics = {}
518 for i, agent in enumerate(self._agents):
519 metrics[f"agent{i}/loss"] = agent.loss
520 self._logging_fn(self._iteration, ep + 1, metrics)
521
522 def _update_policy_and_distribution(self):
523 """Updates the current soft-max policy and the distribution."""
524 self._policy = self.get_softmax_policy()
525 self._distribution = distribution_std.DistributionPolicy(
526 self._game, self._policy)
527
528 def get_softmax_policy(self,
529 tau: Optional[float] = None
530 ) -> rl_agent_policy.JointRLAgentPolicy:
531 """Returns the softmax policy with the specified tau.
532
533 Args:
534 tau: Tau for soft-max action selection, or None to use the value set in
535 the MunchausenDQN agents.
536
537 Returns:
538 A JointRLAgentPolicy.
539 """
540 return rl_agent_policy.JointRLAgentPolicy(
541 self._game, {
542 idx: SoftMaxMunchausenDQN(agent, tau=tau)
543 for idx, agent in enumerate(self._agents)
544 }, self._use_observation)
545
546 def iteration(self):
547 """An iteration of Mirror Descent."""
548 self._train_agents()
549 self._update_policy_and_distribution()
550 self._iteration += 1
551 # Update the distributions of the environments and the previous Q-networks
552 # of the agents.
553 for env, agent in zip(self._envs, self._agents):
554 env.update_mfg_distribution(self.distributiondistribution)
555 agent.update_prev_q_network()
556
557 @property
558 def policy(self):
559 return self._policy

561 @property
562 def distribution(self):
563 return self._distribution
