30 bool IsValidProbDistribution(const ActionsAndProbs& probs) {
31  double sum_probs = 0;
32  for (const auto& [action, prob] : probs) {
33  if (prob < 0) return false;
34  if (prob > 1) return false;
35  sum_probs += prob;
36  }
37  return abs(sum_probs - 1.0) < 1e-10;
38 }

40 bool IsPositiveProbDistribution(const ActionsAndProbs& probs) {
41  double sum_probs = 0;
42  for (const auto& [action, prob] : probs) {
43  if (prob <= 0) return false;
44  if (prob > 1) return false;
45  sum_probs += prob;
46  }
47  return abs(sum_probs - 1.0) < 1e-10;
48 }

64 ActionsAndProbs ExplorativeSamplingPolicy::GetStatePolicy(
65  const std::string& info_state) const {
66  auto policy = CFRCurrentPolicy::GetStatePolicy(info_state);
67  const double unif = 1. / policy.size();
68  for (auto& [_, prob] : policy) {

97 // Negative zeros denote the banned actions. Useful for debugging,
98 // as it is immediately obvious which actions have been banned.
99 // It is not currently used for any indication of state
100 // (but in principle could be).
101 constexpr double kBannedAction = -0.;
102 
103 ActionsAndProbs TargetedPolicy::GetStatePolicy(const State& h) const {
104  // Check if current state is part of the currently built tree.
105  ActionsAndProbs policy;
106  if (h.IsChanceNode()) {
107  policy = h.ChanceOutcomes();
108  } else if (h.IsPlayerNode()) {
109  policy = CFRCurrentPolicy::GetStatePolicy(h);
110  } else {
111  SpielFatalError("Could not get policy for this state.");
112  }
113 
114  double biased_sum = 0.0;
115  for (auto& [action, prob] : policy) {
116  if (IsAllowedAction(h, action)) {
117  biased_sum += prob;
118  } else {
119  prob = kBannedAction;
120  }
121  }
122 
123  // Normalize the biased policy if some actions have been banned.
124  double bias_exploration = bias_exploration_; // Default exploration.
125  if (biased_sum > 0) {
126  for (auto& [_, prob] : policy) {
127  prob /= biased_sum;
128  }
129  } else {
130  // Do only uniform exploration when all actions are banned.
131  // This means the targeted policy has become "lost" in the game due
132  // to its imperfect information structure. Just because an action is locally
133  // allowed, it does not mean that we will always reach the target by
134  // following the (locally) allowed actions.
135  bias_exploration = 1.;
136  if (stats_) ++(stats_->missed_targets);
137  }
138 
139  // Mix in exploration.
140  const double unif = 1. / policy.size();
141  for (auto& [_, prob] : policy) {
142  prob = bias_exploration * unif + (1 - bias_exploration) * prob;
143  }
144  return policy;
145 }

147 bool TargetedPolicy::IsAllowedAction(const State& h,
148  const Action& action) const {
149  if (targeting_ == kDoNotUseTargeting) return true;
150 
151  const std::unique_ptr<const State> ha = h.Child(action);
152 
153  if (targeting_ == Targeting::kInfoStateTargeting) {
154  SPIEL_CHECK_NE(target_info_state_, kNoActionObsTargetSpecified);
155  return target_info_state_->IsExtensionOf(target_info_state_->GetPlayer(),
156  *ha);
157  }
158 
159  if (targeting_ == Targeting::kPublicStateTargeting) {
160  SPIEL_CHECK_NE(target_public_state_, kNoPublicObsTargetSpecified);
161  return target_public_state_->IsExtensionOf(*ha);
162  }
163 
164  SpielFatalError("Unknown targeting.");
165 }

167 bool TargetedPolicy::IsTargetHit(const State& h) {
168  SPIEL_CHECK_TRUE(targeting_ != kInfoStateTargeting ||
169  target_info_state_ != kNoActionObsTargetSpecified);
170  SPIEL_CHECK_TRUE(targeting_ != kPublicStateTargeting ||
171  target_public_state_ != kNoPublicObsTargetSpecified);
172  const bool hit_info_state =
173  targeting_ == kInfoStateTargeting &&
174  target_info_state_->CorrespondsTo(target_info_state_->GetPlayer(), h);
175  const bool hit_public_state = targeting_ == kPublicStateTargeting &&
176  target_public_state_->CorrespondsTo(h);
177  return hit_info_state || hit_public_state;
178 }

300 void OOSAlgorithm::RunTargetedIterations(int iterations) {
301  if (target_biasing_ == 0.) {
302  return RunUnbiasedIterations(iterations);
303  }
304 
305  for (int t = 0; t < iterations; ++t) {
306  for (Player exploringPl = 0; exploringPl < 2; ++exploringPl) {
307  std::unique_ptr<State> h = game_->NewInitialState();
308  is_biased_iteration_ = random_->RandomUniform() <= target_biasing_;
309  // We always have a target, which cannot be a root node
310  // (this was handled by publicly facing methods)
311  is_below_target_ = false;
312 
313  if (is_biased_iteration_) stats_.biased_iterations++;
314  RootIteration(h.get(), Player(exploringPl));
315  }
316  }
317 }

319 void OOSAlgorithm::RootIteration(State* h, Player exploringPl) {
320  ++stats_.root_visits;
321 
322  // Make sure we don't use mutable vars where we shouldn't.
323  // We have bunch of not-nan tests all over the code to catch any bugs.
324  u_z_ = std::numeric_limits<double>::quiet_NaN();
325  s_z_all_ = std::numeric_limits<double>::quiet_NaN();
326 
327  Iteration(h,
328  /*rm_h_pl=*/1.0, /*rm_h_opp=*/1.0,
329  /*bs_h_all=*/1.0, /*us_h_all=*/1.0,
330  /*us_h_cn=*/1.0, exploringPl);
331 }

333 double OOSAlgorithm::Iteration(State* h, double rm_h_pl, double rm_h_opp,
334  double bs_h_all, double us_h_all, double us_h_cn,
335  Player exploringPl) {
336  // Have we hit the target? And update some statistics.
337  ++stats_.state_visits;
338 
339  if (bias_policy_->IsTargetHit(*h)) {
340  is_below_target_ = true;
341 
342  ++stats_.target_visits;
343  if (is_biased_iteration_) ++stats_.target_biased_visits;
344  }
345 
346  // Dispatch to appropriate methods.
347  if (h->IsTerminal()) {
348  ++stats_.terminal_visits;
349  return IterationTerminalNode(h, bs_h_all, us_h_all, exploringPl);
350  }
351 
352  if (h->IsChanceNode()) {
353  return IterationChanceNode(h, rm_h_pl, rm_h_opp, bs_h_all, us_h_all,
354  us_h_cn, exploringPl);
355  }
356 
357  if (h->IsPlayerNode()) {
358  return IterationPlayerNode(h, rm_h_pl, rm_h_opp, bs_h_all, us_h_all,
359  us_h_cn, exploringPl);
360  }
361 
362  SpielFatalError("Unrecognized state type.");
363 }

365 double OOSAlgorithm::IterationTerminalNode(State* h, double bs_h_all,
366  double us_h_all,
367  Player exploringPl) {
368  SPIEL_DCHECK_TRUE(h->IsTerminal());
369  s_z_all_ = Bias(bs_h_all, us_h_all);
370  u_z_ = h->PlayerReturn(exploringPl);
371  return u_z_;

374 double OOSAlgorithm::IterationChanceNode(State* h, double rm_h_pl,
375  double rm_h_opp, double bs_h_all,
376  double us_h_all, double us_h_cn,
377  Player exploringPl) {
378  SPIEL_DCHECK_TRUE(h->IsChanceNode());
379 
380  const TakeAction take = SelectAction(h, IsBiasingApplicable(bs_h_all));
381  const double s_ha_all = Bias(take.bs, take.us);
382  SPIEL_DCHECK_GT(s_ha_all, 0);
383 
384  h->ApplyAction(take.action);
385  const double u_ha =
386  Iteration(h, rm_h_pl, rm_h_opp, bs_h_all * take.bs, us_h_all * take.us,
387  us_h_cn * take.us, exploringPl);
388 
389  // Compute estimate of the expected utility.
390  double u_h = u_ha * take.us / s_ha_all;
391  SPIEL_DCHECK_FALSE(std::isnan(u_h));
392  SPIEL_DCHECK_FALSE(std::isinf(u_h));
393  return u_h;
394 }

396 double OOSAlgorithm::IterationPlayerNode(State* h, double rm_h_pl,
397  double rm_h_opp, double bs_h_all,
398  double us_h_all, double us_h_cn,
399  Player exploringPl) {
400  SPIEL_DCHECK_TRUE(h->IsPlayerNode());
401 
402  bool exploring_move_in_node = h->CurrentPlayer() == exploringPl;
403  const std::string info_state = h->InformationStateString();
404 
405  const double s_h_all = Bias(bs_h_all, us_h_all);
406  SPIEL_DCHECK_GT(s_h_all, 0);
407  const auto it = values_->find(info_state);
408  bool is_leaf_state = it == values_->end();
409 
410  // Note: we cannot use h / aoh after this code executes,
411  // as it will be set to leaf values.
412  const PlayerNodeOutcome outcome =
413  is_leaf_state
414  ? IncrementallyBuildTree(h, info_state, s_h_all, exploringPl)
415  : SampleExistingTree(h, info_state, &it->second, rm_h_pl, rm_h_opp,
416  bs_h_all, us_h_all, us_h_cn, exploringPl);
417 
418  SPIEL_DCHECK_TRUE(h->IsTerminal());
419  SPIEL_DCHECK_FALSE(std::isnan(u_z_));
420  SPIEL_DCHECK_FALSE(std::isnan(outcome.u_h));
421  SPIEL_DCHECK_FALSE(std::isinf(outcome.u_h));
422 
423  // Note: the only probability that's missing here is rm_h_pl
424  // for it to be full reach probability weighted by full sampling probability.
425  double importance_sampling_ratio = rm_h_opp * us_h_cn / s_h_all;
426 
427  if (exploring_move_in_node) {
428  UpdateInfoStateCumulativeRegrets(&outcome.data, outcome.action,
429  outcome.u_ha, outcome.u_h,
430  importance_sampling_ratio);
431  } else {
432  UpdateInfoStateCumulativePolicy(&outcome.data, importance_sampling_ratio);
433  }
434 
435  return outcome.u_h;
436 }

438 PlayerNodeOutcome OOSAlgorithm::SampleExistingTree(
439  State* h, const std::string& info_state, CFRInfoStateValues* values,
440  double rm_h_pl, double rm_h_opp, double bs_h_all, double us_h_all,
441  double us_h_cn, Player exploringPl) {
442  SPIEL_DCHECK_TRUE(h->IsPlayerNode());
443  SPIEL_DCHECK_FALSE(std::isnan(rm_h_pl));
444  SPIEL_DCHECK_FALSE(std::isnan(rm_h_opp));
445  SPIEL_DCHECK_FALSE(std::isnan(bs_h_all));
446  SPIEL_DCHECK_FALSE(std::isnan(us_h_all));
447  SPIEL_DCHECK_FALSE(std::isnan(us_h_cn));
448 
449  const bool exploring_move_in_node = h->CurrentPlayer() == exploringPl;
450  const TakeAction take = SelectAction(h, IsBiasingApplicable(bs_h_all));
451 
452  const int action_index = values->GetActionIndex(take.action);
453  const double rm_ha_both = values->current_policy[action_index];
454  const double s_ha_all = Bias(take.bs, take.us);
455  SPIEL_DCHECK_GT(s_ha_all, 0);
456 
457  h->ApplyAction(take.action);
458 
459  const double u_ha =
460  Iteration(h, (exploring_move_in_node) ? rm_h_pl * rm_ha_both : rm_h_pl,
461  (exploring_move_in_node) ? rm_h_opp : rm_h_opp * rm_ha_both,
462  bs_h_all * take.bs, us_h_all * take.us, us_h_cn, exploringPl);
463 
464  double u_h = u_ha * rm_ha_both / s_ha_all;
465  SPIEL_DCHECK_FALSE(std::isnan(rm_ha_both));
466  SPIEL_DCHECK_FALSE(std::isnan(u_h));
467  return PlayerNodeOutcome{take.action, rm_ha_both, u_h, u_ha / s_ha_all,
468  *values};
469 }

471 PlayerNodeOutcome OOSAlgorithm::IncrementallyBuildTree(
472  State* h, const std::string& info_state, double s_h_all,
473  Player exploringPl) {
474  SPIEL_DCHECK_FALSE(std::isnan(s_h_all));
475  ++stats_.rollouts;
476 
477  // The current history is a leaf within the currently built look-ahead tree.
478  // By adding info state values, we make sure that next sampling from here
479  // will be into the existing tree.
480  const std::vector<Action> actions = h->LegalActions();
481  const auto [it, state_inserted] =
482  values_->emplace(info_state, CFRInfoStateValues(actions));
483  // If it was already in the values, we shouldn't be building the tree.
484  SPIEL_DCHECK_TRUE(state_inserted);
485 
486  const double rm_ha_both = 1.0 / actions.size();
487  double reach_prob = 1.0;
488  Action first_action = kInvalidAction;
489  SPIEL_DCHECK_TRUE(h->IsPlayerNode());
490  while (!h->IsTerminal()) {
491  ActionsAndProbs policy;
492  if (h->IsChanceNode()) {
493  policy = h->ChanceOutcomes();
494  } else if (h->IsPlayerNode()) {
495  policy = UniformStatePolicy(*h);
496  } else {
497  SpielFatalError("Invalid state");
498  }
499 
500  const auto [action, prob] = SampleAction(policy, random_->RandomUniform());
501 
502  if (first_action == kInvalidAction) {
503  first_action = action;
504  }
505  reach_prob *= prob;
506  h->ApplyAction(action);
507  }
508  SPIEL_DCHECK_NE(first_action, kInvalidAction);
509 
510  u_z_ = h->PlayerReturn(exploringPl);
511  s_z_all_ = s_h_all * reach_prob;
512 
513  // The expected values for u(h) must be unbiased so MCCFR can work correctly.
514  // Normally we use importance sampling, but since the strategy and sampling
515  // policy are the same, they cancel each other out. Leaving just leaf value
516  // for the current estimate.
517  const double u_h = u_z_;
518  const double u_ha = u_z_;
519 
520  return PlayerNodeOutcome{first_action, rm_ha_both, u_h, u_ha, it->second};
521 }

527 TakeAction OOSAlgorithm::SelectAction(State* h, bool do_biased_sample) {
528  const ActionsAndProbs& sample_probs = sample_policy_->GetStatePolicy(*h);
529  const ActionsAndProbs& biased_probs = bias_policy_->GetStatePolicy(*h);
530 
531  // Check what comes out of policies are proper distributions.
532  SPIEL_DCHECK_TRUE(IsValidProbDistribution(biased_probs));
533  // All leaves must be reachable under sample policy!
534  SPIEL_DCHECK_TRUE(IsPositiveProbDistribution(sample_probs));
535 
536  // When we do biased sampling, we completely ignore
537  // the sample policy for choosing any actions.
538  const ActionsAndProbs& followProbs =
539  do_biased_sample ? biased_probs : sample_probs;
540 
541  auto [action, prob] = SampleAction(followProbs, random_->RandomUniform());
542  return TakeAction{action, GetProb(sample_probs, action),
543  GetProb(biased_probs, action)};
544 }

546 void OOSAlgorithm::UpdateInfoStateCumulativePolicy(
547  CFRInfoStateValues* values, double importance_sampling_ratio) {
548  // We use stochastically weighted averaging.
549  for (int i = 0; i < values->cumulative_policy.size(); i++) {
550  SPIEL_DCHECK_GE(values->cumulative_policy[i], 0);
551  values->cumulative_policy[i] +=
552  importance_sampling_ratio * values->current_policy[i];
553  }
554 }
555 
556 void OOSAlgorithm::UpdateInfoStateCumulativeRegrets(
557  CFRInfoStateValues* values, Action a, double u_ha, double u_h,
558  double importance_sampling_ratio) {
559  SPIEL_DCHECK_FALSE(std::isnan(u_ha));
560  SPIEL_DCHECK_FALSE(std::isnan(u_h));
561  SPIEL_DCHECK_FALSE(std::isnan(importance_sampling_ratio));
562  auto& regs = values->cumulative_regrets;
563  const int action_index = values->GetActionIndex(a);
564  for (int i = 0; i < regs.size(); i++) {
565  if (i == action_index) {
566  regs[i] += (u_ha - u_h) * importance_sampling_ratio;
567  } else {
568  regs[i] += (-u_h) * importance_sampling_ratio;
569  }
570  }
571  values->ApplyRegretMatching();
572 }
