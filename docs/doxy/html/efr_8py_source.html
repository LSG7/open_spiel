18"""Python implementation of the extensive-form regret minimization algorithm.
19
20See: "Efficient Deviation Types and Learning
21 for Hindsight Rationality in Extensive-Form Games",
22 Morrill et al. 2021b,
23 https://arxiv.org/abs/2102.06973
24
25One iteration of EFR consists of:
261) Compute current strategy from regrets (e.g. using Regret Matching).
272) Compute values using the current strategy
283) Compute regrets from these values
29
30The average policy converges to a Nash Equilibrium
31rather than the current policy.
32"""

45@attr.s
46class _InfoStateNode(object):
47 """An object wrapping values associated to an information state."""
48
49 # The list of the legal actions.
50 legal_actions = attr.ib()
51 index_in_tabular_policy = attr.ib()
52 # The newly availible deviations + the old ones
53 relizable_deviations = attr.ib()
54 # Player -> state -> action -> prob
55 current_history_probs = attr.ib()
56
57 # An array representing the preceeding actions played
58 # upto this information state.
59 history = attr.ib()
60
61 cumulative_regret = attr.ib(factory=lambda: collections.defaultdict(float))
62 # The sum of all prior iteration's policies
63 cumulative_policy = attr.ib(factory=lambda: collections.defaultdict(float))
64
65 # A dictionary mapping each deviation to their "y values"
66 # for the current iteration.
67 y_values = attr.ib(factory=lambda: collections.defaultdict(float))

70class _EFRSolverBase(object):
71 """The base EFR solver class.
72
73 The main iteration loop is implemented in `evaluate_and_update_policy`:
74 ```python
75 game = pyspiel.load_game("game_name")
76 initial_state = game.new_initial_state()
77 solver = Solver(game)
78 for i in range(num_iterations):
79 solver.evaluate_and_update_policy()
80 solver.current_policy() # Access the current policy
81 solver.average_policy() # Access the average policy
82 ```
83 """

85 def __init__(self, game, deviation_gen):
86 """Initializer.
87
88 Args:
89 game: The `pyspiel.Game` to run on.
90 deviation_gen: a function that accepts (num_actions : int,
91 history, prior_legal_actions)
92 and returns a list containing`LocalDeviationWithTimeSelection` objects
93 of the realisable deviations of a described type
94 (e.g blind causal deviations) and given the information state described
95 by the function parameters.
96 """
97 # pyformat: enable
98 assert game.get_type().dynamics == pyspiel.GameType.Dynamics.SEQUENTIAL, (
99 "EFR requires sequential games. If you're trying to run it "
100 + "on a simultaneous (or normal-form) game, please first transform it "
101 + "using turn_based_simultaneous_game."
102 )
103
104 self._game = game
105 self._num_players = game.num_players()
106 self._root_node = self._game.new_initial_state()
107
108 # This is for returning the current policy and average policy to a caller
109 self._current_policy = policy.TabularPolicy(game)
110 self._average_policy = self._current_policy.__copy__()
111 self._deviation_gen = deviation_gen
112
113 self._info_state_nodes = {}
114 hist = {player: [] for player in range(self._num_players)}
115 empty_path_indices = [[] for _ in range(self._num_players)]
116
117 self._initialize_info_state_nodes(self._root_node, hist, empty_path_indices)
118
119 self._iteration = 1 # For possible linear-averaging.

121 def return_cumulative_regret(self):
122 """Returns a dictionary mapping.
123
124 The mapping is fromevery information state to its associated regret
125 (accumulated over all iterations).
126 """
127 return {
128 list(self._info_state_nodes.keys())[i]: list(
129 self._info_state_nodes.values()
130 )[i].cumulative_regret
131 for i in range(len(self._info_state_nodes.keys()))
132 }

134 def current_policy(self):
135 """Returns the current policy as a TabularPolicy.
136
137 WARNING: The same object, updated in-place will be returned! You can copy
138 it (or its `action_probability_array` field).
139
140 For EFR, this policy does not necessarily have to converge.
141 """
142 return self._current_policy

144 def average_policy(self):
145 """Returns the average of all policies iterated.
146
147 WARNING: The same object, updated in-place will be returned! You can copy it
148 (or its `action_probability_array` field).
149
150 This average policy converges to a equilibrium policy as the number
151 of iterations increases (equilibrium type depends on learning
152 deviations used).
153
154 The policy is computed using the accumulated policy probabilities computed
155 using `evaluate_and_update_policy`.
156
157 Returns:
158 A `policy.TabularPolicy` object (shared between calls) giving the (linear)
159 time averaged policy (weighted by player reach probabilities) for all
160 players.
161 """
162 _update_average_policy(self._average_policy, self._info_state_nodes)
163 return self._average_policy

165 def _initialize_info_state_nodes(self, state, history, path_indices):
166 """Initializes info_state_nodes.
167
168 Create one _InfoStateNode per infoset. We could also initialize the node
169 when we try to access it and it does not exist.
170
171 Generates all deviations that are realisable at this state and stores
172 the history and preceeding state policy information to create memory states
173 and calculate the memory reach probability for each deviation.
174
175 Args:
176 state: The current state in the tree traversal. This should be the
177 root node when we call this function from the EFR solver.
178 history: an arrays of the preceeding actions taken prior to the state
179 for each player.
180 path_indices: a 3d array [player number]x[preceeding state]x[legal actions
181 for state, index of the policy for this state in TabularPolicy].
182 """
183 if state.is_terminal():
184 return
185
186 if state.is_chance_node():
187 for action, unused_action_prob in state.chance_outcomes():
188 self._initialize_info_state_nodes(
189 state.child(action), history, path_indices
190 )
191 return
192
193 current_player = state.current_player()
194 info_state = state.information_state_string(current_player)
195 info_state_node = self._info_state_nodes.get(info_state)
196 if info_state_node is None:
197 legal_actions = state.legal_actions(current_player)
198 info_state_node = _InfoStateNode(
199 legal_actions=legal_actions,
200 index_in_tabular_policy=self._current_policy.state_lookup[info_state],
201 relizable_deviations=None,
202 history=history[current_player].copy(),
203 current_history_probs=copy.deepcopy(path_indices[current_player]),
204 )
205 prior_possible_actions = []
206 for i in range(len(info_state_node.current_history_probs)):
207 prior_possible_actions.append(
208 info_state_node.current_history_probs[i][0]
209 )
210 prior_possible_actions.append(info_state_node.legal_actions)
211
212 info_state_node.relizable_deviations = self._deviation_gen(
213 len(info_state_node.legal_actions),
214 info_state_node.history,
215 prior_possible_actions,
216 )
217 self._info_state_nodes[info_state] = info_state_node
218
219 legal_actions = state.legal_actions(current_player)
220
221 for action in info_state_node.legal_actions:
222 new_path_indices = copy.deepcopy(path_indices)
223 new_path_indices[current_player].append(
224 [legal_actions, info_state_node.index_in_tabular_policy]
225 )
226 new_history = copy.deepcopy(history)
227 new_history[current_player].append(action)
228 assert len(new_history[current_player]) == len(
229 new_path_indices[current_player]
230 )
231
232 self._initialize_info_state_nodes(
233 state.child(action), new_history, new_path_indices
234 )

236 def _update_current_policy(self, state, current_policy):
237 """Updated the current policy.
238
239 Updated in order so that memory reach probs are defined wrt to the new
240 strategy.
241
242 Note that the function is called recursively (first call should
243 be the root).
244
245 Additionally, to update the strategy for a given state we require
246 the (t+1)th strategy for all prior states.
247
248 Args:
249 state: the state of which to update the strategy.
250 current_policy: the (t+1)th strategy that is being recursively computed,
251 see the function description for more detail.
252 """
253
254 if state.is_terminal():
255 return
256 elif not state.is_chance_node():
257 current_player = state.current_player()
258 info_state = state.information_state_string(current_player)
259 info_state_node = self._info_state_nodes[info_state]
260 deviations = info_state_node.relizable_deviations
261 for devation in range(len(deviations)):
262 mem_reach_probs = create_probs_from_index(
263 info_state_node.current_history_probs, current_policy
264 )
265 deviation_reach_prob = deviations[
266 devation
267 ].player_deviation_reach_probability(mem_reach_probs)
268 y_increment = (
269 max(0, info_state_node.cumulative_regret[devation])
270 * deviation_reach_prob
271 )
272 info_state_node.y_values[deviations[devation]] = (
273 info_state_node.y_values[deviations[devation]] + y_increment
274 )
275
276 state_policy = current_policy.policy_for_key(info_state)
277 for action, value in self._regret_matching(
278 info_state_node.legal_actions, info_state_node
279 ).items():
280 state_policy[action] = value
281
282 for action in info_state_node.legal_actions:
283 new_state = state.child(action)
284 self._update_current_policy(new_state, current_policy)
285 else:
286 for action, _ in state.chance_outcomes():
287 new_state = state.child(action)
288 self._update_current_policy(new_state, current_policy)
289
290 # Path to state probability ignores chance probabilty as this is stored as
291 # new_reach_probabilities[-1]
292 def _compute_cumulative_immediate_regret_for_player(
293 self, state, policies, reach_probabilities, player
294 ):
295 """Increments the immediate regrets and policy.
296
297 Increments are done for `player` of all realisable deviations at this state.
298
299 Args:
300 state: The initial game state to analyze from.
301 policies: A list of `num_players` callables taking as input an
302 `info_state_node` and returning a {action: prob} dictionary.
303 reach_probabilities: The probability for each player of reaching `state`
304 as a numpy array [prob for player 0, for player 1,..., for chance].
305 `reach_probabilities[player]` will work in all cases.
306 player: The 0-indexed player to update the values for. If `None`, the
307 update for all players will be performed.
308
309 Returns:
310 The utility of `state` for all players, assuming all players follow the
311 current policy defined by `self.Policy`.
312 """
313 if state.is_terminal():
314 return np.asarray(state.returns())
315
316 if state.is_chance_node():
317 state_value = 0.0
318 for action, action_prob in state.chance_outcomes():
319 assert action_prob > 0
320 new_state = state.child(action)
321 new_reach_probabilities = reach_probabilities.copy()
322 new_reach_probabilities[-1] *= action_prob
323
324 state_value += (
325 action_prob
326 * self._compute_cumulative_immediate_regret_for_player(
327 new_state, policies, new_reach_probabilities, player
328 )
329 )
330 return state_value
331
332 current_player = state.current_player()
333 info_state = state.information_state_string(current_player)
334
335 # No need to continue on this history branch as no update will be performed
336 # for any player.
337 # The value we return here is not used in practice. If the conditional
338 # statement is True, then the last taken action has probability 0 of
339 # occurring, so the returned value is not impacting the parent node value.
340 if all(reach_probabilities[:-1] == 0):
341 return np.zeros(self._num_players)
342
343 state_value = np.zeros(self._num_players)
344
345 # The utilities of the children states are computed recursively. As the
346 # regrets are added to the information state regrets for each state in that
347 # information state, the recursive call can only be made once per child
348 # state. Therefore, the utilities are cached.
349 children_utilities = {}
350
351 info_state_node = self._info_state_nodes[info_state]
352 # Reset y values
353 info_state_node.y_values = collections.defaultdict(float)
354 if policies is None:
355 info_state_policy = self._get_infostate_policy(info_state)
356 else:
357 info_state_policy = policies[current_player](info_state)
358
359 reach_prob = reach_probabilities[current_player]
360 for action in state.legal_actions():
361 action_prob = info_state_policy.get(action, 0.0)
362 info_state_node.cumulative_policy[action] = (
363 info_state_node.cumulative_policy[action] + action_prob * reach_prob
364 )
365 new_state = state.child(action)
366 new_reach_probabilities = reach_probabilities.copy()
367 assert action_prob <= 1
368 new_reach_probabilities[current_player] *= action_prob
369 child_utility = self._compute_cumulative_immediate_regret_for_player(
370 new_state,
371 policies=policies,
372 reach_probabilities=new_reach_probabilities,
373 player=player,
374 )
375
376 state_value += action_prob * child_utility
377 children_utilities[action] = child_utility
378
379 counterfactual_reach_prob = np.prod(
380 reach_probabilities[:current_player]
381 ) * np.prod(reach_probabilities[current_player + 1 :])
382
383 state_value_for_player = state_value[current_player]
384 deviations = info_state_node.relizable_deviations
385 for deviation_index in range(len(deviations)):
386 deviation = deviations[deviation_index]
387 deviation_strategy = deviation.deviate(
388 strat_dict_to_array(self._get_infostate_policy(info_state))
389 )
390
391 player_child_utilities = np.array(list(children_utilities.values()))[
392 :, current_player
393 ]
394 devation_cf_value = np.inner(
395 np.transpose(deviation_strategy), player_child_utilities
396 )
397
398 memory_reach_probs = create_probs_from_index(
399 info_state_node.current_history_probs, self.current_policy()
400 )
401 player_current_memory_reach_prob = (
402 deviation.player_deviation_reach_probability(memory_reach_probs)
403 )
404
405 deviation_regret = player_current_memory_reach_prob * (
406 (devation_cf_value * counterfactual_reach_prob)
407 - (counterfactual_reach_prob * state_value_for_player)
408 )
409
410 info_state_node.cumulative_regret[deviation_index] += deviation_regret
411 return state_value

413 def _get_infostate_policy(self, info_state_str):
414 """Returns an {action: prob} dictionary for the policy on `info_state`."""
415 info_state_node = self._info_state_nodes[info_state_str]
416 prob_vec = self._current_policy.action_probability_array[
417 info_state_node.index_in_tabular_policy
418 ]
419 return {
420 action: prob_vec[action] for action in info_state_node.legal_actions
421 }

438class EFRSolver(_EFRSolver):
439 """Implements the EFR algorithm with several deviation types.
440
441 See: https://arxiv.org/abs/2102.06973
442 """
443
444 def __init__(self, game, deviations_name):
445 """Initializer.
446
447 Args:
448 game: The `pyspiel.Game` to run on.
449 deviations_name: the name of the deviation type to use for
450 accumulating regrets and calculating the strategy at the next timestep.
451
452 Deviation types implemented are "blind action", "informed action",
453 "blind cf", "informed counterfactual", "blind partial sequence",
454 "counterfactual partial sequence", "casual partial sequence",
455 "twice informed partial sequence", "single target behavioural".
456
457 See "Efficient Deviation Types and Learning for Hindsight Rationality in
458 Extensive-Form Games" by D. Morrill et al. 2021b
459 for the full definition of each type.
460 """
461
462 # external_only = True leads to a shortcut in the computation of the next
463 # timesteps strategy from the regrets
464 external_only = False
465 deviation_sets = None # pylint: disable=unused-variable
466
467 if deviations_name in {"blind action"}:
468 deviation_sets = return_blind_action
469 external_only = True
470 elif deviations_name in {"informed action"}:
471 deviation_sets = return_informed_action
472 elif deviations_name in {"blind cf", "blind counterfactual"}:
473 deviation_sets = return_blind_cf
474 external_only = True
475 elif deviations_name in {"informed cf", "informed counterfactual"}:
476 deviation_sets = return_informed_cf
477 elif deviations_name in {"bps", "blind partial sequence"}:
478 deviation_sets = return_blind_partial_sequence
479 external_only = True
480 elif deviations_name in {
481 "cfps",
482 "cf partial sequence",
483 "counterfactual partial sequence",
484 }:
485 deviation_sets = return_cf_partial_sequence
486 elif deviations_name in {"csps", "casual partial sequence"}:
487 deviation_sets = return_cs_partial_sequence
488 elif deviations_name in {"tips", "twice informed partial sequence"}:
489 deviation_sets = return_twice_informed_partial_sequence
490 elif deviations_name in {"bhv", "single target behavioural", "behavioural"}:
491 deviation_sets = return_behavourial
492 else:
493 raise ValueError(
494 "Unsupported Deviation Set Passed As "
495 " Constructor Argument"
496 )
497 super(EFRSolver, self).__init__(game, deviation_sets)
498 self._external_only = external_only

500 def _regret_matching(self, legal_actions, info_set_node):
501 """Returns an info state policy.
502
503 The info state policy returned is the one obtained by applying
504 regret-matching function over all deviations and time selection functions.
505
506 Args:
507 legal_actions: the list of legal actions at this state.
508 info_set_node: the info state node to compute the policy for.
509
510 Returns:
511 A dict of action -> prob for all legal actions.
512 """
513 z = sum(info_set_node.y_values.values())
514 info_state_policy = {}
515
516 # The fixed point solution can be directly obtained through the
517 # weighted regret matrix if only external deviations are used.
518 if self._external_only and z > 0:
519 weighted_deviation_matrix = np.zeros(
520 (len(legal_actions), len(legal_actions))
521 )
522 for dev in list(info_set_node.y_values.keys()):
523 weighted_deviation_matrix += (
524 info_set_node.y_values[dev] / z
525 ) * dev.return_transform_matrix()
526 new_strategy = weighted_deviation_matrix[:, 0]
527 for index in range(len(legal_actions)):
528 info_state_policy[legal_actions[index]] = new_strategy[index]
529
530 # Full regret matching by finding the least squares solution to the
531 # fixed point of the EFR regret matching function.
532 # Last row of matrix and the column entry minimises the solution
533 # towards a strategy.
534 elif z > 0:
535 num_actions = len(info_set_node.legal_actions)
536 weighted_deviation_matrix = -np.eye(num_actions)
537
538 for dev in list(info_set_node.y_values.keys()):
539 weighted_deviation_matrix += (
540 info_set_node.y_values[dev] / z
541 ) * dev.return_transform_matrix()
542
543 normalisation_row = np.ones(num_actions)
544 weighted_deviation_matrix = np.vstack(
545 [weighted_deviation_matrix, normalisation_row]
546 )
547 b = np.zeros(num_actions + 1)
548 b[num_actions] = 1
549 b = np.reshape(b, (num_actions + 1, 1))
550
551 strategy = linalg.lstsq(weighted_deviation_matrix, b)[0]
552
553 # Adopt same clipping strategy as paper author's code.
554 strategy[np.where(strategy < 0)] = 0
555 strategy[np.where(strategy > 1)] = 1
556
557 strategy = strategy / sum(strategy)
558 for index in range(len(strategy)):
559 info_state_policy[info_set_node.legal_actions[index]] = strategy[index]
560 # Use a uniform strategy as sum of all regrets is negative.
561 else:
562 for index in range(len(legal_actions)):
563 info_state_policy[legal_actions[index]] = 1.0 / len(legal_actions)
564 return info_state_policy

567def _update_average_policy(average_policy, info_state_nodes):
568 """Updates in place `average_policy` to the average of all policies iterated.
569
570 This function is a module level function to be reused by both CFRSolver and
571 CFRBRSolver.
572
573 Args:
574 average_policy: A `policy.TabularPolicy` to be updated in-place.
575 info_state_nodes: A dictionary {`info_state_str` -> `_InfoStateNode`}.
576 """
577 for info_state, info_state_node in info_state_nodes.items():
578 info_state_policies_sum = info_state_node.cumulative_policy
579 state_policy = average_policy.policy_for_key(info_state)
580 probabilities_sum = sum(info_state_policies_sum.values())
581 if probabilities_sum == 0:
582 num_actions = len(info_state_node.legal_actions)
583 for action in info_state_node.legal_actions:
584 state_policy[action] = 1 / num_actions
585 else:
586 for action, action_prob_sum in info_state_policies_sum.items():
587 state_policy[action] = action_prob_sum / probabilities_sum

590def strat_dict_to_array(strategy_dictionary):
591 """A helper function to convert the strategy dictionary mapping.
592
593 Conversion applies action -> prob value to an array.
594
595 Args:
596 strategy_dictionary: a dictionary action -> prob value.
597
598 Returns:
599 strategy_array: an array with the ith action's value at the i-1th index.
600 """
601 actions = list(strategy_dictionary.keys())
602 strategy_array = np.zeros((len(actions), 1))
603 for action in range(len(actions)):
604 strategy_array[action][0] = strategy_dictionary[actions[action]]
605 return strategy_array

608def array_to_strat_dict(strategy_array, legal_actions):
609 """A helper function to convert a strategy.
610
611 Converts a strategy array to an action -> prob value dictionary.
612
613 Args:
614 strategy_array: an array with the ith action's value at the i-1th index.
615 legal_actions: the list of all legal actions at the current state.
616
617 Returns:
618 strategy_dictionary: a dictionary action -> prob value.
619 """
620 strategy_dictionary = {}
621 for action in legal_actions:
622 strategy_dictionary[action] = strategy_array[action]
623 return strategy_dictionary

626def create_probs_from_index(indices, current_policy):
627 path_to_state = []
628 if indices is None or not indices:
629 return []
630 for index in indices:
631 strat_dict = array_to_strat_dict(
632 current_policy.action_probability_array[index[1]], index[0]
633 )
634 path_to_state.append(strat_dict)

909def generate_all_action_permutations(current_stem, remaining_actions):
910 """Return a List of all possible game continuations.
911
912 Return a List of all possible game continuations playing on from the
913 current stem and with playing from the set of remaining actions.
914 `current_stem` = "" generates all possible playthroughs from the current
915 information state.
916
917 Args:
918 current_stem: the prior sequence of actions to be completed by the
919 remaining actions
920 remaining_actions: a 2d array of [subsequent states]x[possible actions]
921
922 Returns:
923 An array with each element being the current stem joined with a possible
924 permuation of remaining actions
925 """

927 return [np.array(current_stem)]
928 else:
929 next_actions = remaining_actions[0]
930 permutations = []
931 for action in next_actions:
932 next_stem = current_stem.copy()
933 next_stem.append(action)
934 next_remaining_actions = remaining_actions[1:]
935 prev_permutations = generate_all_action_permutations(
936 next_stem, next_remaining_actions
937 )
938 for i in prev_permutations:
939 permutations.append(i)

1082 def player_deviation_reach_probability(
1083 self, prior_possible_action_probabilities
1084 ):
1085 """Calculate the probability of reaching the current memory state.
1086
1087 Calculate the probability of reaching the current memory state
1088 provided the player played from the start of the game to this state.
1089 This is assuming that they play with their current strategy with the
1090 deviation applied.
1091 Args:
1092 prior_possible_action_probabilities: a 2d array of length [player's
1093 history]x[number of actions at that state]. These are the current

1098 """
1099 if (
1100 self.prior_actions_weight is None
1101 or self.prior_memory_actions is None
1102 or prior_possible_action_probabilities is None
1103 ):
1104 return 1.0
1105
1106 memory_action_probabilities = np.ones(len(self.prior_actions_weight))
1107 # Reconstruct memory probabilities from history provided to the deviation

1110 if self.use_unmodified_history:

1115 self.prior_memory_actions[state]
1116 ]
1117 )
1118 else:
1119 memory_action_probabilities[state] = 1
1120 memory_weightings[state] = 1

1123 memory_weightings, memory_action_probabilities
1124 )

1126 return memory_reach_probability
