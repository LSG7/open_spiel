22// Player ids are 0, 1, 2, ...

24enum PlayerId {

32 kInvalidPlayer = -3,

37};

52// Layouts for 3-D tensors. For 2-D tensors, we assume that the layout is a
53// single spatial dimension and a channel dimension. If a 2-D tensor should be
54// interpreted as a 2-D space, report it as 3-D with a channel dimension of
55// size 1. We have no standard for higher-dimensional tensors.
56enum class TensorLayout {
57 kHWC, // indexes are in the order (height, width, channels)
58 kCHW, // indexes are in the order (channels, height, width)
