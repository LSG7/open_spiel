43inline constexpr int kMinValueCellState = -4;
44// State of a cell.
45// Describes if a cell is
46// - empty, black or white
47// - connected to N/S edges if black, or was a winning move
48// - connected to E/W edges if white, or was a winning move
49// These are used in calculation of winning connections, and may be useful
50// features for learning agents
51//
52// Convention is that black plays first (and is player 0)
53enum class CellState {
54 kEmpty = 0,
55 kWhiteWest = -3,
56 kWhiteEast = -2,
57 kWhiteWin = -4,
58 kWhite = -1, // White and not edge connected
59 kBlackNorth = 3,
60 kBlackSouth = 2,
61 kBlackWin = 4,
62 kBlack = 1, // Black and not edge connected
63};
