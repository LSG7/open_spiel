130 // Set the spent amounts uniformly for each player.
131 // Must be divisible by the number of players!
132 void SetPotSize(int pot_size) {
133 SPIEL_CHECK_GE(pot_size, 0);
134 SPIEL_CHECK_LE(pot_size, game_->TotalMoney());
135 SPIEL_CHECK_EQ(pot_size % game_->GetNbPlayers(), 0);
136 const int num_players = game_->GetNbPlayers();
137 for (int pl = 0; pl < num_players; ++pl) {
138 acpcState_.spent[pl] = pot_size / num_players;
139 }
140 }
