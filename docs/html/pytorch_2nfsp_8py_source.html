215  def _learn(self):
216  """Compute the loss on sampled transitions and perform a avg-network update.
217 
218  If there are not enough elements in the buffer, no loss is computed and
219  `None` is returned instead.
220 
221  Returns:
222  The average loss obtained on this batch of transitions or `None`.
223  """
224  if (len(self._reservoir_buffer_reservoir_buffer) < self._batch_size_batch_size or
225  len(self._reservoir_buffer_reservoir_buffer) < self._min_buffer_size_to_learn_min_buffer_size_to_learn):
226  return None
227 
228  transitions = self._reservoir_buffer_reservoir_buffer.sample(self._batch_size_batch_size)
229  info_states = torch.Tensor([t.info_state for t in transitions])
230  action_probs = torch.Tensor([t.action_probs for t in transitions])

234  torch.max(action_probs, dim=1)[1])

283 class ReservoirBuffer(object):
284  """Allows uniform sampling over a stream of data.
285 
286  This class supports the storage of arbitrary elements, such as observation
287  tensors, integer actions, etc.
288 
289  See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
290  """
291 
292  def __init__(self, reservoir_buffer_capacity):
293  self._reservoir_buffer_capacity_reservoir_buffer_capacity = reservoir_buffer_capacity
294  self._data_data = []
295  self._add_calls_add_calls = 0
296 
297  def add(self, element):
298  """Potentially adds `element` to the reservoir buffer.
299 
300  Args:
301  element: data to be added to the reservoir buffer.
302  """
303  if len(self._data_data) < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
304  self._data_data.append(element)
305  else:
306  idx = np.random.randint(0, self._add_calls_add_calls + 1)
307  if idx < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
308  self._data_data[idx] = element
309  self._add_calls_add_calls += 1
310 
311  def sample(self, num_samples):
312  """Returns `num_samples` uniformly sampled from the buffer.
313 
314  Args:
315  num_samples: `int`, number of samples to draw.
316 
317  Returns:
318  An iterable over `num_samples` random elements of the buffer.
319 
320  Raises:
321  ValueError: If there are less than `num_samples` elements in the buffer
322  """
323  if len(self._data_data) < num_samples:
324  raise ValueError("{} elements could not be sampled from size {}".format(
325  num_samples, len(self._data_data)))
326  return random.sample(self._data_data, num_samples)

329  self._data_data = []
330  self._add_calls_add_calls = 0
