34def thresholded(logits, regrets, threshold=2.0):
35 """Zeros out `regrets` where `logits` are too negative or too large."""
36 can_decrease = torch.gt(logits, -threshold).float()
37 can_increase = torch.lt(logits, threshold).float()
38 regrets_negative = torch.minimum(regrets, torch.Tensor([0.0]))
39 regrets_positive = torch.maximum(regrets, torch.Tensor([0.0]))
40 return can_decrease * regrets_negative + can_increase * regrets_positive

43def train(model,
44 data,
45 batch_size,
46 step_size=1.0,
47 threshold=2.0,
48 autoencoder_loss=None):
49 """Train NeuRD `model` on `data`."""
50 data = torch.utils.data.DataLoader(data, batch_size=batch_size, shuffle=True)
51
52 for x, regrets in data:
53 output = model(x, training=True)
54 logits = output[:, :1]
55 logits = logits - torch.mean(logits)
56
57 regrets = thresholded(logits, regrets, threshold=threshold).detach()
58 utility = torch.mean(logits * regrets)
59
60 if autoencoder_loss is not None:
61 utility = utility - autoencoder_loss(x, output[:, 1:])
62 model.zero_grad()
63 utility.backward()
64 with torch.no_grad():
65 for var in model.layers.parameters():
66 new_var = var + step_size * var.grad
67 var.copy_(new_var)

70class DeepNeurdModel(nn.Module):
71 """A flexible deep feedforward NeuRD model class.
72
73 Properties:
74 layers: The `torch.nn.Linear` layers describing this model.
75 """
76
77 def __init__(self,
78 game,
79 num_hidden_units,
80 num_hidden_layers=1,
81 num_hidden_factors=0,
82 hidden_activation=nn.ReLU,
83 use_skip_connections=False,
84 autoencode=False):
85 """Creates a new `DeepNeurdModel.
86
87 Args:
88 game: The OpenSpiel game being solved.
89 num_hidden_units: The number of units in each hidden layer.
90 num_hidden_layers: The number of hidden layers. Defaults to 1.
91 num_hidden_factors: The number of hidden factors or the matrix rank of the
92 layer. If greater than zero, hidden layers will be split into two
93 separate linear transformations, the first with
94 `num_hidden_factors`-columns and the second with
95 `num_hidden_units`-columns. The result is that the logical hidden layer
96 is a rank-`num_hidden_units` matrix instead of a rank-`num_hidden_units`
97 matrix. When `num_hidden_units < num_hidden_units`, this is effectively
98 implements weight sharing. Defaults to 0.
99 hidden_activation: The activation function to apply over hidden layers.
100 Defaults to `torch.nn.Relu`.
101 use_skip_connections: Whether or not to apply skip connections (layer
102 output = layer(x) + x) on hidden layers. Zero padding or truncation is
103 used to match the number of columns on layer inputs and outputs.
104 autoencode: Whether or not to output a reconstruction of the inputs upon
105 being called. Defaults to `False`.
106 """
107 super(DeepNeurdModel, self).__init__()
108 self._autoencode = autoencode
109 self._use_skip_connections = use_skip_connections
110 self._hidden_are_factored = num_hidden_factors > 0
111
112 self.layers = nn.ModuleList()
113 self.input_size = rcfr.num_features(game)
114 for _ in range(num_hidden_layers):
115 if self._hidden_are_factored:
116 self.layers.append(
117 nn.Linear(self.input_size, num_hidden_factors, bias=True))
118 self.input_size = num_hidden_factors
119
120 self.layers.append(
121 nn.Linear(self.input_size, num_hidden_units, bias=True))
122 if hidden_activation:
123 self.layers.append(hidden_activation())
124 self.input_size = num_hidden_units
125
126 self.layers.append(
127 nn.Linear(
128 self.input_size,
129 1 + self._autoencode * rcfr.num_features(game),
130 bias=True))
131
132 def forward(self, x, training=False):
133 """Evaluates this model on x.
134
135 Args:
136 x: Model input.
137 training: Whether or not this is being called during training. If
138 `training` and the constructor argument `autoencode` was `True`, then
139 the output will contain the estimated regrets concatenated with a
140 reconstruction of the input, otherwise only regrets will be returned.
141 Defaults to `False`.
142
143 Returns:
144 The `torch.Tensor` resulting from evaluating this model on `x`. If
145 `training` and the constructor argument `autoencode` was `True`, then
146 it will contain the estimated regrets concatenated with a
147 reconstruction of the input, otherwise only regrets will be returned.
148 """
149 y = rcfr.feedforward_evaluate(
150 layers=self.layers,
151 x=x,
152 use_skip_connections=self._use_skip_connections,
153 hidden_are_factored=self._hidden_are_factored)
154 return y if training else y[:, :1]

157class CounterfactualNeurdSolver(object):
158 """All-actions, strong NeuRD on counterfactual regrets.
159
160 No regularization bonus is applied, so the current policy likely will not
161 converge. The average policy profile is updated and stored in a full
162 game-size table and may converge to an approximate Nash equilibrium in
163 two-player, zero-sum games.
164 """
165
166 def __init__(self, game, models):
167 """Creates a new `CounterfactualNeurdSolver`.
168
169 Args:
170 game: An OpenSpiel `Game`.
171 models: Current policy models (optimizable array-like -> `torch.Tensor`
172 callables) for both players.
173 """
174 self._game = game
175 self._models = models
176 self._root_wrapper = rcfr.RootStateWrapper(game.new_initial_state())
177
178 self._cumulative_seq_probs = [
179 np.zeros(n) for n in self._root_wrapper.num_player_sequences
180 ]
181
182 def _sequence_weights(self, player=None):
183 """Returns exponentiated weights for each sequence as an `np.array`."""
184 if player is None:
185 return [
186 self._sequence_weights(player)
187 for player in range(self._game.num_players())
188 ]
189 else:
190 tensor = torch.squeeze(self._models[player](
191 self._root_wrapper.sequence_features[player]))
192 tensor = tensor - torch.max(tensor, dim=0)[0]
193 tensor = torch.exp(tensor)
194 return tensor.detach().numpy()
195
196 def current_policy(self):
197 """Returns the current policy profile.
198
199 Returns:
200 A `dict<info state, list<Action, probability>>` that maps info state
201 strings to `Action`-probability pairs describing each player's policy.
202 """
203 return self._root_wrapper.sequence_weights_to_tabular_profile(
204 self._sequence_weights())
205
206 def average_policy(self):
207 """Returns the average of all policies iterated.
208
209 The policy is computed using the accumulated policy probabilities computed
210 using `evaluate_and_update_policy`.
211
212 Returns:
213 A `dict<info state, list<Action, probability>>` that maps info state
214 strings to (Action, probability) pairs describing each player's policy.
215 """
216 return self._root_wrapper.sequence_weights_to_tabular_profile(
217 self._cumulative_seq_probs)
218
219 def _previous_player(self, player):
220 """The previous player in the turn ordering."""
221 return player - 1 if player > 0 else self._game.num_players() - 1
222
223 def _average_policy_update_player(self, regret_player):
224 """The player for whom the average policy should be updated."""
225 return self._previous_player(regret_player)
226
227 def evaluate_and_update_policy(self, train_fn):
228 """Performs a single step of policy evaluation and policy improvement.
229
230 Args:
231 train_fn: A (model, `torch.utils.data.TensorDataset`) function that trains
232 the given regression model to accurately reproduce the x to y mapping
233 given x-y data.
234 """
235 sequence_weights = self._sequence_weights()
236 player_seq_features = self._root_wrapper.sequence_features
237 for regret_player in range(self._game.num_players()):
238 seq_prob_player = self._average_policy_update_player(regret_player)
239
240 regrets, seq_probs = (
241 self._root_wrapper.counterfactual_regrets_and_reach_weights(
242 regret_player, seq_prob_player, *sequence_weights))
243
244 self._cumulative_seq_probs[seq_prob_player] += seq_probs
245 targets = torch.unsqueeze(torch.Tensor(regrets), axis=1)
246 data = torch.utils.data.TensorDataset(player_seq_features[regret_player],
247 targets)
248
249 regret_player_model = self._models[regret_player]
250 train_fn(regret_player_model, data)
251 sequence_weights[regret_player] = self._sequence_weights(regret_player)
