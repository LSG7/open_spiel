440  def test_rcfr_functions(self):
441  models = [_new_model() for _ in range(_GAME.num_players())]
442  root = rcfr.RootStateWrapper(_GAME.new_initial_state())
443 
444  num_half_iterations = 4
445  num_epochs = 100
446 
447  cumulative_regrets = [np.zeros(n) for n in root.num_player_sequences]
448  cumulative_reach_weights = [np.zeros(n) for n in root.num_player_sequences]
449 
450  average_profile = root.sequence_weights_to_tabular_profile(
451  cumulative_reach_weights)
452  self.assertGreater(pyspiel.nash_conv(_GAME, average_profile), 0.91)
453 
454  regret_player = 0
455  sequence_weights = [
456  model(root.sequence_features[player]).numpy()
457  for player, model in enumerate(models)
458  ]
459 
460  for _ in range(num_half_iterations):
461  reach_weights_player = 1 if regret_player == 0 else 0
462 
463  sequence_weights[reach_weights_player] = models[reach_weights_player](
464  root.sequence_features[reach_weights_player]).numpy()
465 
466  regrets, seq_probs = root.counterfactual_regrets_and_reach_weights(
467  regret_player, reach_weights_player, *sequence_weights)
468 
469  cumulative_regrets[regret_player] += regrets
470  cumulative_reach_weights[reach_weights_player] += seq_probs
471 
472  data = tf.data.Dataset.from_tensor_slices(
473  (root.sequence_features[regret_player],
474  tf.expand_dims(cumulative_regrets[regret_player], axis=1)))
475  data = data.shuffle(12)
476  data = data.batch(12)
477  data = data.repeat(num_epochs)
478 
479  optimizer = tf.keras.optimizers.Adam(lr=0.005, amsgrad=True)
480 
481  for x, y in data:
482  optimizer.minimize(
483  lambda: tf.losses.huber_loss(y, models[regret_player](x)), # pylint: disable=cell-var-from-loop
484  models[regret_player].trainable_variables)
485 
486  regret_player = reach_weights_player
487 
488  average_profile = root.sequence_weights_to_tabular_profile(
489  cumulative_reach_weights)
490 
491  self.assertLess(pyspiel.nash_conv(_GAME, average_profile), 0.91)

503  data = data.shuffle(12)
504  data = data.batch(12)
505  data = data.repeat(num_epochs)
506 
507  optimizer = tf.keras.optimizers.Adam(lr=0.005, amsgrad=True)
508 
509  for x, y in data:
