39def convert_param_spec(param_spec):
40 """Converts 'size:10;200' into ('size', [10, 200])."""
41 split = param_spec.split(':', 2)
42 return split[0], [int(v) for v in split[1].split(';')]

45def main(argv: Sequence[str]) -> None:
46 if len(argv) > 1:
47 raise app.UsageError('Too many command-line arguments.')
48
49 param_names, param_values = zip(
50 *[convert_param_spec(spec) for spec in FLAGS.parameters])
51 header = (['game_name'] + list(param_names) +
52 ['fictitious_play_iteration_time'])
53 timing_results = []
54 for game_name in FLAGS.games:
55 for param_tuple in itertools.product(*param_values):
56 result_line = [game_name] + [str(p) for p in param_tuple]
57 print('Computing timings for:', ' '.join(result_line))
58 param_dict = dict(zip(param_names, param_tuple))
59 game = pyspiel.load_game(game_name, param_dict)
60 t0 = time.time()
61 fp = fictitious_play.FictitiousPlay(game)
62 fp.iteration()
63 elapsed = time.time() - t0
64 result_line.append(f'{elapsed:.4f}s')
65 print(' '.join(result_line))
66 timing_results.append(result_line)
