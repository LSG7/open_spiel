16 """Kuhn Poker implemented in Python.
17 
18 This is a simple demonstration of implementing a game in Python, featuring
19 chance and imperfect information.
20 
21 Python games are significantly slower than C++, but it may still be suitable
22 for prototyping or for small games.
23 
24 It is possible to run C++ algorithms on Python implemented games, This is likely
25 to have good performance if the algorithm simply extracts a game tree and then
26 works with that. It is likely to be poor if the algorithm relies on processing
27 and updating states as it goes, e.g. MCTS.
28 """

86 class KuhnPokerState(pyspiel.State):
87  """A python version of the Kuhn poker state."""
88 
89  def __init__(self, game):
90  """Constructor; should only be called by Game.new_initial_state."""
91  super().__init__(game)
92  self.cardscards = []
93  self.betsbets = []
94  self.potpot = [1.0, 1.0]
95  self._game_over_game_over = False
96  self._next_player_next_player = 0
97 
98  # OpenSpiel (PySpiel) API functions are below. This is the standard set that
99  # should be implemented by every sequential-move game with chance.
100 
101  def current_player(self):
102  """Returns id of the next player to move, or TERMINAL if game is over."""
103  if self._game_over_game_over:
104  return pyspiel.PlayerId.TERMINAL
105  elif len(self.cardscards) < _NUM_PLAYERS:
106  return pyspiel.PlayerId.CHANCE
107  else:
108  return self._next_player_next_player
109 
110  def _legal_actions(self, player):
111  """Returns a list of legal actions, sorted in ascending order."""
112  assert player >= 0
113  return [Action.PASS, Action.BET]
114 
115  def chance_outcomes(self):
116  """Returns the possible chance outcomes and their probabilities."""
117  assert self.is_chance_node()
118  outcomes = sorted(_DECK - set(self.cardscards))
119  p = 1.0 / len(outcomes)
120  return [(o, p) for o in outcomes]
121 
122  def _apply_action(self, action):
123  """Applies the specified action to the state."""
124  if self.is_chance_node():
125  self.cardscards.append(action)
126  else:
127  self.betsbets.append(action)
128  if action == Action.BET:
129  self.potpot[self._next_player_next_player] += 1
130  self._next_player_next_player = 1 - self._next_player_next_player
131  if ((min(self.potpot) == 2) or
132  (len(self.betsbets) == 2 and action == Action.PASS) or
133  (len(self.betsbets) == 3)):
134  self._game_over_game_over = True
135 
136  def _action_to_string(self, player, action):
137  """Action -> string."""
138  if player == pyspiel.PlayerId.CHANCE:
139  return f"Deal:{action}"
140  elif action == Action.PASS:
141  return "Pass"
142  else:
143  return "Bet"
144 
145  def is_terminal(self):
146  """Returns True if the game is over."""
147  return self._game_over_game_over
148 
149  def returns(self):
150  """Total reward for each player over the course of the game so far."""
151  pot = self.potpot
152  winnings = float(min(pot))
153  if not self._game_over_game_over:
154  return [0., 0.]
155  elif pot[0] > pot[1]:
156  return [winnings, -winnings]
157  elif pot[0] < pot[1]:
158  return [-winnings, winnings]
159  elif self.cardscards[0] > self.cardscards[1]:
160  return [winnings, -winnings]
161  else:
162  return [-winnings, winnings]
163 
164  def __str__(self):
165  """String for debug purposes. No particular semantics are required."""
166  return "".join([str(c) for c in self.cardscards] + ["pb"[b] for b in self.betsbets])

169 class KuhnPokerObserver:
170  """Observer, conforming to the PyObserver interface (see observation.py)."""
171 
172  def __init__(self, iig_obs_type, params):
173  """Initializes an empty observation tensor."""
174  if params:
175  raise ValueError(f"Observation parameters not supported; passed {params}")
176 
177  # Determine which observation pieces we want to include.
178  pieces = [("player", 2, (2,))]
179  if iig_obs_type.private_info == pyspiel.PrivateInfoType.SINGLE_PLAYER:
180  pieces.append(("private_card", 3, (3,)))
181  if iig_obs_type.public_info:
182  if iig_obs_type.perfect_recall:
183  pieces.append(("betting", 6, (3, 2)))
184  else:
185  pieces.append(("pot_contribution", 2, (2,)))
186 
187  # Build the single flat tensor.
188  total_size = sum(size for name, size, shape in pieces)
189  self.tensortensor = np.zeros(total_size, np.float32)
190 
191  # Build the named & reshaped views of the bits of the flat tensor.
192  self.dictdict = {}
193  index = 0
194  for name, size, shape in pieces:
195  self.dictdict[name] = self.tensortensor[index:index + size].reshape(shape)
196  index += size
197 
198  def set_from(self, state, player):
199  """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
200  self.tensortensor.fill(0)
201  if "player" in self.dictdict:
202  self.dictdict["player"][player] = 1
203  if "private_card" in self.dictdict and len(state.cards) > player:
204  self.dictdict["private_card"][state.cards[player]] = 1
205  if "pot_contribution" in self.dictdict:
206  self.dictdict["pot_contribution"][:] = state.pot
207  if "betting" in self.dictdict:
208  for turn, action in enumerate(state.bets):
209  self.dictdict["betting"][turn, action] = 1
210 
211  def string_from(self, state, player):
212  """Observation of `state` from the PoV of `player`, as a string."""
213  pieces = []
214  if "player" in self.dictdict:
215  pieces.append(f"p{player}")
216  if "private_card" in self.dictdict and len(state.cards) > player:
217  pieces.append(f"card:{state.cards[player]}")
218  if "pot_contribution" in self.dictdict:
219  pieces.append(f"pot[{int(state.pot[0])} {int(state.pot[1])}]")
220  if "betting" in self.dictdict and state.bets:
221  pieces.append("".join("pb"[b] for b in state.bets))
222  return " ".join(str(p) for p in pieces)
