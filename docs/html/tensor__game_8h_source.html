36 class TensorGame : public NormalFormGame {
37  public:
38  // action_names[player] is the list of action names for player.
39  // utilities[player] is a flattened tensor of utilities for player, in
40  // row-major/C-style/lexicographic order of all players' actions.
41  TensorGame(GameType game_type, GameParameters game_parameters,
42  std::vector<std::vector<std::string>> action_names,
43  std::vector<std::vector<double>> utilities)
44  : NormalFormGame(std::move(game_type), std::move(game_parameters)),
45  action_names_(std::move(action_names)),
46  utilities_(std::move(utilities)),
47  shape_(utilities_.size()) {
48  int size = 1;
49  for (Player player = 0; player < action_names_.size(); ++player) {
50  size *= action_names_[player].size();
51  shape_[player] = action_names_[player].size();
52  }
53  ComputeMinMaxUtility();
54  SPIEL_CHECK_TRUE(std::all_of(utilities_.begin(), utilities_.end(),
55  [size](const auto& player_utils) {
56  return player_utils.size() == size;
57  }));
58  }
59 
60  // Implementation of Game interface
61  int NumDistinctActions() const override {
62  return *std::max_element(begin(shape_), end(shape_));
63  }

65  std::unique_ptr<State> NewInitialState() const override;

69  double MinUtility() const override { return min_utility_; }
70

74  return ActionName(player, action);
75  }

77  const std::vector<int>& Shape() const { return shape_; }
78  const double PlayerUtility(const Player player,
79  const std::vector<Action>& actions) const {

96  bool operator==(const Game& other_game) const override {
97  const auto& other = down_cast<const TensorGame&>(other_game);
98  return (shape_ == other.shape_ && utilities_ == other.utilities_);
99  }
100 
101  bool ApproxEqual(const Game& other_game, double tolerance) const {
102  const auto& other = down_cast<const TensorGame&>(other_game);
103  if (shape_ != other.shape_) {
104  return false;
105  }
106  for (Player p = 0; p < NumPlayers(); ++p) {
107  if (!AllNear(utilities_[p], other.utilities_[p], tolerance)) {
108  return false;
109  }
110  }
111  return true;
112  }

114  std::vector<double> GetUtilities(const std::vector<Action>& joint_action)
115  const override {
116  int idx = index(joint_action);
117  std::vector<double> utilities;
118  utilities.reserve(NumPlayers());
119  for (Player p = 0; p < NumPlayers(); ++p) {
120  utilities.push_back(utilities_[p][idx]);
121  }
122  return utilities;
123  }

125  double GetUtility(Player player, const std::vector<Action>& joint_action)
126  const override {
127  return PlayerUtility(player, joint_action);

130  std::shared_ptr<const matrix_game::MatrixGame> AsMatrixGame() const {
131  SPIEL_CHECK_EQ(NumPlayers(), 2);
132  const GameType& game_type = GetType();
133  return matrix_game::CreateMatrixGame(

139  private:
140  const int index(const std::vector<Action>& args) const {
141  int ind = 0;
142  for (int i = 0; i < NumPlayers(); ++i) {
143  ind = ind * shape_[i] + args[i];
144  }
145  return ind;
146  }
147 
148  void ComputeMinMaxUtility() {
149  min_utility_ = *std::min_element(begin(utilities_[0]), end(utilities_[0]));
150  for (Player player = 1; player < NumPlayers(); ++player) {
151  min_utility_ =
152  std::min(min_utility_, *std::min_element(begin(utilities_[player]),
153  end(utilities_[player])));
154  }
155 
156  max_utility_ = *std::max_element(begin(utilities_[0]), end(utilities_[0]));
157  for (Player player = 1; player < NumPlayers(); ++player) {
158  max_utility_ =
159  std::max(max_utility_, *std::max_element(begin(utilities_[player]),
160  end(utilities_[player])));
161  }
162  }
163 
164  // action_names_[player] is the list of action names for player.
165  const std::vector<std::vector<std::string>> action_names_;
166  // utilities_[player] is a flattened tensor of utilities for player, in
167  // row-major/C-style/lexicographic order of all players' actions.
168  const std::vector<std::vector<double>> utilities_;
169  std::vector<int> shape_;
170  double min_utility_;
171  double max_utility_;
172 };

179  std::vector<Action> LegalActions(Player player) const override {
180  if (IsTerminal()) return {};
181  if (player == kSimultaneousPlayerId) {
182  return LegalFlatJointActions();
183  } else {
184  std::vector<Action> moves(tensor_game_->Shape()[player]);
185  std::iota(moves.begin(), moves.end(), 0); // fill with values 0...n-1
186  return moves;

193  if (player == kSimultaneousPlayerId)
194  return FlatJointActionToString(action_id);

197  }
198 
199  bool IsTerminal() const override { return !joint_move_.empty(); }
200 
201  std::vector<double> Returns() const override {
202  std::vector<double> returns(NumPlayers());
203  if (IsTerminal()) {
204  for (Player player = 0; player < returns.size(); player++) {
205  returns[player] = tensor_game_->PlayerUtility(player, joint_move_);
206  }
207  }
208  return returns;
209  }

212  return std::unique_ptr<State>(new TensorState(*this));
213  }
214 
215  protected:
216  void DoApplyActions(const std::vector<Action>& moves) override {
217  SPIEL_CHECK_EQ(moves.size(), NumPlayers());
218  for (Player player = 0; player < NumPlayers(); player++) {
219  SPIEL_CHECK_GE(moves[player], 0);
220  SPIEL_CHECK_LT(moves[player], tensor_game_->Shape()[player]);
221  }
222  joint_move_ = moves;
223  }
224 
225  private:

227  const TensorGame* tensor_game_;
