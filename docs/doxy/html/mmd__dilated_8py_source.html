15r"""Python implementation of the magnetic mirror descent (MMD) algorithm.
16
17The algorithm operated over the sequence-from with dilated entropy.
18
19See https://arxiv.org/abs/2206.05825.
20
21One iteration of MMD consists of:
221) Compute gradients of dilated entropy
23 and payoffs for current sequence form policies.
242) Compute behavioural form policy starting from the bottom
25 of the tree and updating gradients of parent nodes along the way.
263) Convert behavioural form policy to equivalent sequence form policy.
27
28The last sequence form policy converges linearly (exponentially fast)
29to a \alpha-reduced normal-form QRE.
30"""

53def softmax(x):
54 unnormalized = np.exp(x - np.max(x))

58def divergence(x, y, psi_x, psi_y, grad_psi_y):
59 """Compute Bregman divergence between x and y, B_psi(x;y).
60
61 Args:
62 x: Numpy array.
63 y: Numpy array.
64 psi_x: Value of psi evaluated at x.
65 psi_y: Value of psi evaluated at y.
66 grad_psi_y: Gradient of psi evaluated at y.
67
68 Returns:
69 Scalar.
70 """
71 return psi_x - psi_y - np.dot(grad_psi_y, x - y)

74def dilated_dgf_divergence(mmd_1, mmd_2):
75 """Bregman divergence between two MMDDilatedEnt objects.
76
77 The value is equivalent to a sum of two Bregman divergences
78 over the sequence form, one for each player.
79
80 Args:
81 mmd_1: MMDDilatedEnt Object
82 mmd_2: MMDDilatedEnt Object
83
84 Returns:
85 Scalar.
86 """
87
88 dgf_values = [mmd_1.dgf_eval(), mmd_2.dgf_eval()]
89 dgf_grads = mmd_2.dgf_grads()
90 div = 0
91 for player in range(2):
92 div += divergence(mmd_1.sequences[player], mmd_2.sequences[player],
93 dgf_values[0][player], dgf_values[1][player],
94 dgf_grads[player])
95 return div

98class MMDDilatedEnt(object):
99 r"""Implements Magnetic Mirror Descent (MMD) with Dilated Entropy.
100
101 The implementation uses the sequence form representation.
102
103 The policies converge to a \alpha-reduced normal form QRE of a
104 two-player zero-sum extensive-form game. If \alpha is set
105 to zero then the method is equivalent to mirror descent ascent
106 over the sequence form with dilated entropy and the policies
107 will converge on average to a nash equilibrium with
108 the appropriate stepsize schedule (or approximate equilirbrium
109 for fixed stepsize).
110
111 The main iteration loop is implemented in `update_sequences`:
112
113 ```python
114 game = pyspiel.load_game("game_name")
115 mmd = MMDDilatedEnt(game, alpha=0.1)
116 for i in range(num_iterations):
117 mmd.update_sequences()
118 ```
119 The gap in the regularized game (i.e. 2x exploitability) converges
120 to zero and can be computed:
121
122 ```python
123 gap = mmd.get_gap()
124 ```
125 The policy (i.e. behavioural form policy) can be retrieved:
126 ```python
127 policies = mmd.get_policies()
128 ```
129
130 The average sequences and policies can be retrieved:
131
132 ```python
133 avg_sequences = mmd.get_avg_sequences()
134 avg_policies = mmd.get_avg_policies()
135 ```
136
137 """

142 def __init__(self, game, alpha, stepsize=None):
143 """Initialize the solver object.
144
145 Args:
146 game: a zeros-um spiel game with two players.
147 alpha: weight of dilated entropy regularization. If alpha > 0 MMD
148 will converge to an alpha-QRE. If alpha = 0 mmd will converge to
149 Nash on average.
150 stepsize: MMD stepsize. Will be set automatically if None.
151 """
152 assert game.num_players() == 2
153 assert game.get_type().utility == pyspiel.GameType.Utility.ZERO_SUM
154 assert game.get_type().dynamics == pyspiel.GameType.Dynamics.SEQUENTIAL
155 assert (game.get_type().chance_mode
156 == pyspiel.GameType.ChanceMode.DETERMINISTIC or
157 game.get_type().chance_mode
158 == pyspiel.GameType.ChanceMode.EXPLICIT_STOCHASTIC)
159 assert alpha >= 0
160
161 self.game = game
162 self.alpha = float(alpha)
163 (self.infosets, self.infoset_actions_to_seq, self.infoset_action_maps,
164 self.infoset_parent_map, self.payoff_mat,
165 self.infoset_actions_children) = construct_vars(game)
166
167 if stepsize is not None:
168 self.stepsize = stepsize
169 else:
170 self.stepsize = self.alpha / (np.max(np.abs(self.payoff_mat))**2)
171
172 if self.stepsize == 0.:
173 warnings.warn("MMD stepsize is 0, probably because alpha = 0.")
174
175 self.sequences = uniform_random_seq(game, self.infoset_actions_to_seq)
176 self.avg_sequences = copy.deepcopy(self.sequences)
177 self.iteration_count = 1

179 def get_parent_seq(self, player, infostate):
180 """Looks up the parent sequence value for a given infostate.
181
182 Args:
183 player: player number, either 0 or 1.
184 infostate: infostate id string.
185
186 Returns:
187 Scalar.
188 """
189 parent_isa_key = self.infoset_parent_map[player][infostate]
190 seq_id = self.infoset_actions_to_seq[player][parent_isa_key]
191 parent_seq = self.sequences[player][seq_id]
192 return parent_seq

194 def get_infostate_seq(self, player, infostate):
195 """Gets vector of sequence form values corresponding to a given infostate.
196
197 Args:
198 player: player number, either 0 or 1.
199 infostate: infostate id string.
200
201 Returns:
202 Numpy array.
203 """
204 seq_idx = [
205 self.infoset_actions_to_seq[player][isa_key]
206 for isa_key in self.infoset_action_maps[player][infostate]
207 ]
208 seqs = np.array([self.sequences[player][idx] for idx in seq_idx])
209 return seqs

211 def dgf_eval(self):
212 """Computes the value of dilated entropy for current sequences.
213
214 Returns:
215 List of values, one for each player.
216 """
217 dgf_value = [0., 0.]
218
219 for player in range(2):
220 for infostate in self.infosets[player]:
221
222 if is_root(infostate):
223 continue
224
225 parent_seq = self.get_parent_seq(player, infostate)
226 if parent_seq > 0:
227 children_seq = self.get_infostate_seq(player, infostate)
228 dgf_value[player] += parent_seq * neg_entropy(
229 children_seq / parent_seq)
230
231 return dgf_value

233 def dgf_grads(self):
234 """Computes gradients of dilated entropy for each player and current seqs.
235
236 Returns:
237 A list of numpy arrays.
238 """
239 grads = [np.zeros(len(self.sequences[0])), np.zeros(len(self.sequences[1]))]
240 for player in range(2):
241 for infostate in self.infosets[player]:
242
243 # infostates contain empty sequence for root variable
244 if is_root(infostate):
245 continue
246
247 parent_seq = self.get_parent_seq(player, infostate)
248 if parent_seq > 0:
249
250 for isa_key in self.infoset_action_maps[player][infostate]:
251 # compute infostate term
252 seq_idx = self.infoset_actions_to_seq[player][isa_key]
253 seq = self.sequences[player][seq_idx]
254 grads[player][seq_idx] += np.log(seq / parent_seq) + 1
255
256 # compute terms from children if there are any
257 num_children = len(self.infoset_actions_children[player].get(
258 isa_key, []))
259 grads[player][seq_idx] -= num_children
260 return grads

262 def update_sequences(self):
263 """Performs one step of MMD."""
264 self.iteration_count += 1
265 psi_grads = self.dgf_grads()
266 # pylint: disable=invalid-unary-operand-type
267 grads = [
268 (self.stepsize * self.payoff_mat @ self.sequences[1] - psi_grads[0]) /
269 ((1 + self.stepsize * self.alpha)),
270 (-self.stepsize * self.payoff_mat.T @ self.sequences[0] - psi_grads[1])
271 / (1 + self.stepsize * self.alpha)
272 ]
273
274 new_policy = policy.TabularPolicy(self.game)
275 for player in range(2):
276 self._update_state_sequences(self.empty_infoset_keys[player],
277 grads[player], player, new_policy)
278
279 self.sequences = policy_to_sequence(self.game, new_policy,
280 self.infoset_actions_to_seq)
281 self.update_avg_sequences()

283 def _update_state_sequences(self, infostate, g, player, pol):
284 """Update the state sequences."""
285
286 isa_keys = self.infoset_action_maps[player][infostate]
287 seq_idx = [
288 self.infoset_actions_to_seq[player][isa_key] for isa_key in isa_keys
289 ]
290
291 for isa_key, isa_idx in zip(isa_keys, seq_idx):
292
293 # update children first if there are any
294 children = self.infoset_actions_children[player].get(isa_key, [])
295 for child in children:
296 self._update_state_sequences(child, g, player, pol)
297 # update gradient
298 child_isa_keys = self.infoset_action_maps[player][child]
299 child_seq_idx = [
300 self.infoset_actions_to_seq[player][child_isa_key]
301 for child_isa_key in child_isa_keys
302 ]
303 g_child = np.array([g[idx] for idx in child_seq_idx])
304
305 actions_child = [
306 _get_action_from_key(child_isa_key)
307 for child_isa_key in child_isa_keys
308 ]
309 policy_child = pol.policy_for_key(child)[:]
310 policy_child = np.array([policy_child[a] for a in actions_child])
311 g[isa_idx] += np.dot(g_child, policy_child)
312 g[isa_idx] += neg_entropy(policy_child)
313
314 # no update needed for empty sequence
315 if is_root(infostate):
316 return
317
318 state_policy = pol.policy_for_key(infostate)
319 g_infostate = np.array([g[idx] for idx in seq_idx])
320 actions = [_get_action_from_key(isa_key) for isa_key in isa_keys]
321 new_state_policy = softmax(-g_infostate)
322 for action, pr in zip(actions, new_state_policy):
323 state_policy[action] = pr

325 def get_gap(self):
326 """Computes saddle point gap of the regularized game.
327
328 The gap measures convergence to the alpha-QRE.
329
330 Returns:
331 Scalar.
332 """
333 assert self.alpha > 0, "gap cannot be computed for alpha = 0"
334 grads = [(self.payoff_mat @ self.sequences[1]) / (self.alpha),
335 (-self.payoff_mat.T @ self.sequences[0]) / (self.alpha)]
336 dgf_values = self.dgf_eval()
337
338 br_policy = policy.TabularPolicy(self.game)
339 for player in range(2):
340 self._update_state_sequences(self.empty_infoset_keys[player],
341 grads[player], player, br_policy)
342
343 br_sequences = policy_to_sequence(self.game, br_policy,
344 self.infoset_actions_to_seq)
345 curr_sequences = copy.deepcopy(self.sequences)
346 self.sequences = br_sequences
347 br_dgf_values = self.dgf_eval()
348 self.sequences = curr_sequences
349
350 # gap of sequences (x,y)
351 # d(x) + max_y' x.T A y'-d(y') + d(y) - min_x' d(x') + x'.T Ay
352
353 gap = 0
354 gap += curr_sequences[0].T @ self.payoff_mat @ br_sequences[1]
355 gap += self.alpha * (dgf_values[1] - br_dgf_values[1])
356 gap += self.alpha * (dgf_values[0] - br_dgf_values[0])
357 gap += -br_sequences[0].T @ self.payoff_mat @ curr_sequences[1]
358 return gap

360 def update_avg_sequences(self):
361 for player in range(2):
362 self.avg_sequences[player] = self.avg_sequences[player] * (
363 self.iteration_count - 1) + self.sequences[player]

365 player] = self.avg_sequences[player] / self.iteration_count
