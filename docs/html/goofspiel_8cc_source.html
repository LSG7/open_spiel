104  void WriteTensor(const State& observed_state, int player,
105  Allocator* allocator) const override {
106  const GoofspielState& state =
107  open_spiel::down_cast<const GoofspielState&>(observed_state);
108  const GoofspielGame& game =
109  open_spiel::down_cast<const GoofspielGame&>(*state.GetGame());
110  SPIEL_CHECK_GE(player, 0);
111  SPIEL_CHECK_LT(player, game.NumPlayers());
112 
113  // List all predicates.
114  const bool imp_info = game.IsImpInfo();
115  const bool pub_info = iig_obs_type_.public_info;
116  const bool perf_rec = iig_obs_type_.perfect_recall;
117  const bool priv_one =
118  iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer;
119 
120  // Conditionally write each field.
121  if (pub_info && !perf_rec) {
122  WriteCurrentPointCard(game, state, allocator);
123  WriteRemainingPointCards(game, state, allocator);
124  }
125  if (pub_info) WritePointsTotal(game, state, player, allocator);
126  if (imp_info && priv_one) WritePlayerHand(game, state, player, allocator);
127  if (imp_info && pub_info) WriteWinSequence(game, state, player, allocator);
128  if (pub_info && perf_rec) WritePointCardSequence(game, state, allocator);
129  if (imp_info && perf_rec && priv_one)
130  WritePlayerActionSequence(game, state, player, allocator);
131  if (!imp_info && pub_info)
132  WriteAllPlayersHands(game, state, player, allocator);
133  }

135  std::string StringFrom(const State& observed_state,
136  int player) const override {
137  const GoofspielState& state =
138  open_spiel::down_cast<const GoofspielState&>(observed_state);
139  const GoofspielGame& game =
140  open_spiel::down_cast<const GoofspielGame&>(*state.GetGame());
141  SPIEL_CHECK_GE(player, 0);
142  SPIEL_CHECK_LT(player, game.NumPlayers());
143  std::string result;
144 
145  // List all predicates.
146  const bool imp_info = game.IsImpInfo();
147  const bool pub_info = iig_obs_type_.public_info;
148  const bool perf_rec = iig_obs_type_.perfect_recall;
149  const bool priv_one =
150  iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer;
151 
152  // Conditionally write each field.
153  // This is done in a backwards-compatible way.
154  if (imp_info && priv_one && perf_rec) { // InformationState
155  StringPlayerHand(game, state, player, &result);
156  StringActionSequence(game, state, player, &result);
157  StringPointCardSequence(state, &result);
158  StringWinSequence(state, &result);
159  StringPoints(game, state, &result);
160  StringIsTerminal(state, &result);
161  return result;
162  }
163  if (imp_info && priv_one && !perf_rec) { // Observation
164  StringCurrentPointCard(state, &result);
165  StringRemainingPointCards(state, &result);
166  StringPoints(game, state, &result);
167  StringPlayerHand(game, state, player, &result);
168  StringWinSequence(state, &result);
169  return result;
170  }
171 
172  // Remaining public observation requests.
173  if (pub_info && perf_rec) StringPointCardSequence(state, &result);
174  if (pub_info && !perf_rec) {
175  StringCurrentPointCard(state, &result);
176  StringRemainingPointCards(state, &result);
177  }
178  if (pub_info && !imp_info) StringPlayersHands(game, state, &result);
179  if (pub_info) {
180  StringWinSequence(state, &result);
181  StringPoints(game, state, &result);
182  }
183  return result;
184  }

186  private:
187  // Point totals: one-hot vector encoding points, per player.
188  // Writes this public information from the perspective
189  // of the requesting player.
190  void WritePointsTotal(const GoofspielGame& game, const GoofspielState& state,
191  int player, Allocator* allocator) const {
192  auto out = allocator->Get("point_totals",
193  {game.NumPlayers(), game.MaxPointSlots()});
194  Player p = player;
195  for (int n = 0; n < game.NumPlayers(); state.NextPlayer(&n, &p)) {
196  out.at(n, state.points_[p]) = 1.0;
197  }
198  }
199 
200  // Bit vectors encoding all players' hands.
201  // Writes this public information from the perspective
202  // of the requesting player.
203  void WriteAllPlayersHands(const GoofspielGame& game,
204  const GoofspielState& state, int player,
205  Allocator* allocator) const {
206  auto out =
207  allocator->Get("player_hands", {game.NumPlayers(), game.NumCards()});
208  Player p = player;
209  for (int n = 0; n < game.NumPlayers(); state.NextPlayer(&n, &p)) {
210  for (int c = 0; c < game.NumCards(); ++c) {
211  out.at(n, c) = state.player_hands_[p][c];
212  }
213  }
214  }

216  // Sequence of who won each trick.
217  void WriteWinSequence(const GoofspielGame& game, const GoofspielState& state,
218  int player, Allocator* allocator) const {
219  auto out =
220  allocator->Get("win_sequence", {game.NumRounds(), game.NumPlayers()});
221  for (int i = 0; i < state.win_sequence_.size(); ++i) {
222  if (state.win_sequence_[i] != kInvalidPlayer) {
223  int one_hot = state.win_sequence_[i];
224  if (egocentric_) {
225  // Positive, relative distance to the winner.
226  one_hot = ((game.NumPlayers() + state.win_sequence_[i] - player) %
227  game.NumPlayers());
228  }
229  out.at(i, one_hot) = 1.0;
230  }
231  }
232  }

365 GoofspielState::GoofspielState(std::shared_ptr<const Game> game, int num_cards,
366  int num_turns, PointsOrder points_order,
367  bool impinfo, bool egocentric,
368  ReturnsType returns_type)
369  : SimMoveState(game),
370  num_cards_(num_cards),
371  num_turns_(num_turns),
372  points_order_(points_order),
373  returns_type_(returns_type),
374  impinfo_(impinfo),
375  egocentric_(egocentric),
376  current_player_(kInvalidPlayer),
377  winners_({}),
378  current_turn_(0),
379  point_card_(-1),
380  point_card_sequence_({}),
381  win_sequence_({}),
382  actions_history_({}) {
383  SPIEL_CHECK_LE(num_turns_, num_cards_);
384 
385  // Points and point-card deck.
386  points_.resize(num_players_);
387  std::fill(points_.begin(), points_.end(), 0);
388 
389  // Player hands.
390  player_hands_.clear();
391  for (auto p = Player{0}; p < num_players_; ++p) {
392  std::vector<bool> hand(num_cards_, true);
393  player_hands_.push_back(hand);
394  }
395 
396  // Set the points card index.
397  if (points_order_ == PointsOrder::kRandom) {
398  point_card_ = -1;
399  current_player_ = kChancePlayerId;
400  } else if (points_order_ == PointsOrder::kAscending) {
401  DealPointCard(0);
402  current_player_ = kSimultaneousPlayerId;
403  } else if (points_order_ == PointsOrder::kDescending) {
404  DealPointCard(num_cards - 1);
405  current_player_ = kSimultaneousPlayerId;
406  }
407 }

418 void GoofspielState::DoApplyAction(Action action_id) {
419  if (IsSimultaneousNode()) {
420  ApplyFlatJointAction(action_id);
421  return;
422  }
423  SPIEL_CHECK_TRUE(IsChanceNode());
424  DealPointCard(action_id);
425  current_player_ = kSimultaneousPlayerId;
426 }

428 void GoofspielState::DoApplyActions(const std::vector<Action>& actions) {
429  // Check the actions are valid.
430  SPIEL_CHECK_EQ(actions.size(), num_players_);
431  for (auto p = Player{0}; p < num_players_; ++p) {
432  const int action = actions[p];
433  SPIEL_CHECK_GE(action, 0);
434  SPIEL_CHECK_LT(action, num_cards_);
435  SPIEL_CHECK_TRUE(player_hands_[p][action]);
436  }
437 
438  // Find the highest bid
439  int max_bid = -1;
440  int num_max_bids = 0;
441  int max_bidder = -1;
442 
443  for (int p = 0; p < actions.size(); ++p) {
444  if (actions[p] > max_bid) {
445  max_bid = actions[p];
446  num_max_bids = 1;
447  max_bidder = p;
448  } else if (actions[p] == max_bid) {
449  num_max_bids++;
450  }
451  }
452 
453  if (num_max_bids == 1) {
454  // Winner takes the point card.
455  points_[max_bidder] += CurrentPointValue();
456  win_sequence_.push_back(max_bidder);
457  } else {
458  // Tied among several players: discarded.
459  win_sequence_.push_back(kInvalidPlayer);
460  }
461 
462  // Add these actions to the history.
463  actions_history_.push_back(actions);
464 
465  // Remove the cards from the player's hands.
466  for (auto p = Player{0}; p < num_players_; ++p) {
467  player_hands_[p][actions[p]] = false;
468  }
469 
470  // Next player's turn.
471  current_turn_++;
472 
473  // Deal the next point card.
474  if (current_turn_ < num_turns_) {
475  if (points_order_ == PointsOrder::kRandom) {
476  current_player_ = kChancePlayerId;
477  point_card_ = -1;
478  } else if (points_order_ == PointsOrder::kAscending) {
479  if (point_card_ < num_cards_ - 1) DealPointCard(point_card_ + 1);
480  } else if (points_order_ == PointsOrder::kDescending) {
481  if (point_card_ > 0) DealPointCard(point_card_ - 1);
482  }
483  }
484 
485  // No choice at the last turn, so we can play it now
486  // We use DoApplyAction(s) not to modify the history, as these actions are
487  // not available in the tree.
488  if (current_turn_ == num_cards_ - 1) {
489  // There might be a chance event
490  if (IsChanceNode()) {
491  auto legal_actions = LegalChanceOutcomes();
492  SPIEL_CHECK_EQ(legal_actions.size(), 1);
493  DoApplyAction(legal_actions.front());
494  }
495 
496  // Each player plays their last card
497  std::vector<Action> actions(num_players_);
498  for (auto p = Player{0}; p < num_players_; ++p) {
499  auto legal_actions = LegalActions(p);
500  SPIEL_CHECK_EQ(legal_actions.size(), 1);
501  actions[p] = legal_actions[0];
502  }
503  DoApplyActions(actions);
504  } else if (current_turn_ == num_turns_) {
505  // Game over - determine winner.
506  int max_points = -1;
507  for (auto p = Player{0}; p < num_players_; ++p) {
508  if (points_[p] > max_points) {
509  winners_.clear();
510  max_points = points_[p];
511  winners_.insert(p);
512  } else if (points_[p] == max_points) {
513  winners_.insert(p);
514  }
515  }
516  current_player_ = kTerminalPlayerId;
517  }
518 }

520 std::vector<std::pair<Action, double>> GoofspielState::ChanceOutcomes() const {
521  SPIEL_CHECK_TRUE(IsChanceNode());
522  std::set<int> played(point_card_sequence_.begin(),
523  point_card_sequence_.end());
524  std::vector<std::pair<Action, double>> outcomes;
525  const int n = num_cards_ - played.size();
526  const double p = 1.0 / n;
527  outcomes.reserve(n);
528  for (int i = 0; i < num_cards_; ++i) {
529  if (played.count(i) == 0) outcomes.emplace_back(i, p);
530  }
531  SPIEL_CHECK_EQ(outcomes.size(), n);
532  return outcomes;
533 }

535 std::vector<Action> GoofspielState::LegalActions(Player player) const {
536  if (CurrentPlayer() == kTerminalPlayerId) return std::vector<Action>();
537  if (player == kSimultaneousPlayerId) return LegalFlatJointActions();
538  if (player == kChancePlayerId) return LegalChanceOutcomes();
539  SPIEL_CHECK_GE(player, 0);
540  SPIEL_CHECK_LT(player, num_players_);
541 
542  std::vector<Action> movelist;
543  for (int bid = 0; bid < player_hands_[player].size(); ++bid) {
544  if (player_hands_[player][bid]) {
545  movelist.push_back(bid);
546  }
547  }
548  return movelist;
549 }

564 std::string GoofspielState::ToString() const {
565  std::string points_line = "Points: ";
566  std::string result = "";
567 
568  for (auto p = Player{0}; p < num_players_; ++p) {
569  absl::StrAppend(&points_line, points_[p]);
570  absl::StrAppend(&points_line, " ");
571  absl::StrAppend(&result, "P");
572  absl::StrAppend(&result, p);
573  absl::StrAppend(&result, " hand: ");
574  for (int c = 0; c < num_cards_; ++c) {
575  if (player_hands_[p][c]) {
576  absl::StrAppend(&result, c + 1);
577  absl::StrAppend(&result, " ");
578  }
579  }
580  absl::StrAppend(&result, "\n");
581  }
582 
583  // In imperfect information, the full state depends on both betting sequences
584  if (impinfo_) {
585  for (auto p = Player{0}; p < num_players_; ++p) {
586  absl::StrAppend(&result, "P", p, " actions: ");
587  for (int i = 0; i < actions_history_.size(); ++i) {
588  absl::StrAppend(&result, actions_history_[i][p]);
589  absl::StrAppend(&result, " ");
590  }
591  absl::StrAppend(&result, "\n");
592  }
593  }

608 std::vector<double> GoofspielState::Returns() const {
609  if (!IsTerminal()) {
610  return std::vector<double>(num_players_, 0.0);
611  }
612 
613  if (returns_type_ == ReturnsType::kWinLoss) {
614  if (winners_.size() == num_players_) {
615  // All players have same number of points? This is a draw.
616  return std::vector<double>(num_players_, 0.0);
617  } else {
618  int num_winners = winners_.size();
619  int num_losers = num_players_ - num_winners;
620  std::vector<double> returns(num_players_, (-1.0 / num_losers));
621  for (const auto& winner : winners_) {
622  returns[winner] = 1.0 / num_winners;
623  }
624  return returns;
625  }
626  } else if (returns_type_ == ReturnsType::kPointDifference) {
627  std::vector<double> returns(num_players_, 0);
628  double sum = 0;
629  for (Player p = 0; p < num_players_; ++p) {
630  returns[p] = points_[p];
631  sum += points_[p];
632  }
633  for (Player p = 0; p < num_players_; ++p) {
634  returns[p] -= sum / num_players_;
635  }
636  return returns;
637  } else if (returns_type_ == ReturnsType::kTotalPoints) {
638  std::vector<double> returns(num_players_, 0);
639  for (Player p = 0; p < num_players_; ++p) {
640  returns[p] = points_[p];
641  }
642  return returns;
643  } else {
644  SpielFatalError(absl::StrCat("Unrecognized returns type: ", returns_type_));
645  }
646 }

737 std::vector<int> GoofspielGame::InformationStateTensorShape() const {

739  return {// 1-hot bit vector for point total per player; upper bound is 1 +
740  // 2 + ... + N = N*(N+1) / 2, but must add one to include 0 points.
741  num_players_ * ((num_cards_ * (num_cards_ + 1)) / 2 + 1) +
742  // Bit vector for my remaining cards:
743  num_cards_ +
