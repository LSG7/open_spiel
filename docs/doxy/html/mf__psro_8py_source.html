27def dict_equal(dic1, dic2):
28 return all([dic1[a] == dic2[a] for a in dic1]) and all(
29 [dic1[a] == dic2[a] for a in dic2]

33def equal_policies(pol1, pol2, all_states):
34 assert isinstance(pol1, greedy_policy.GreedyPolicy)
35 equal = True
36 for state_key in all_states:
37 state = all_states[state_key]
38 try:
39 equal = equal and dict_equal(pol1(state), pol2(state))
40 except KeyError:
41 equal = False
42 except ValueError:
43 continue
44 return equal

47def filter_policies(policies, new_policies, all_states):
48 all_policies = policies
49 no_novelty = True
50 for new_policy in new_policies:
51 if all([
52 not equal_policies(new_policy, policy, all_states)
53 for policy in all_policies
54 ]):
55 all_policies.append(new_policy)
56 no_novelty = False
57 return all_policies, no_novelty

92 def step(self):
93 """Does a best-response step."""
94 rewards = self._regret_minimizer.get_rewards()
95
96 print("Computing best response.")
97 new_policies, gap_value = self._best_responder(
98 self._game, self._policies, self._weights, self._mus, self._nus, rewards
99 )
100
101 no_novelty = False
102 if self._filter_new_policies:
103 print("Filtering best responses")
104 self._policies, no_novelty = filter_policies(
105 self._policies, new_policies, self._all_states
106 )
107 else:
108 self._policies = self._policies + new_policies
109
110 if no_novelty:
111 print("No new policy added, PSRO has terminated.")
112 if self._increase_precision_when_done_early:
113 print("Increasing precision")
114 self._regret_minimizer.increase_precision_x_fold(2.0)
115 self._regret_steps_per_step *= 2
116 self._regret_minimizer.restart()
117 self._regret_minimizer.step_for(self._regret_steps_per_step)
118 else:
119 print("Minimizing regret")
120 self._regret_minimizer.reset(self._policies)
121 self._regret_minimizer.step_for(self._regret_steps_per_step)
122
123 average_regret = self._regret_minimizer.compute_average_regret()
124 print("Average Regret : {}".format(average_regret))
125
126 self._mus, self._weights = self._regret_minimizer.get_mus_and_weights()
127 self._nus = self._regret_minimizer.get_nus()
128 return average_regret, gap_value
