60 void TFBatchTrajectoryRecorder::SampleChance(int idx) {
61  while (states_[idx]->IsChanceNode()) {
62  std::vector<std::pair<open_spiel::Action, double>> outcomes =
63  states_[idx]->ChanceOutcomes();
64  Action action = open_spiel::SampleAction(outcomes, dist_(rng_)).first;
65  states_[idx]->ApplyAction(action);
66  }
67 
68  if (states_[idx]->IsTerminal()) {
69  num_terminals_++;
70  terminal_flags_[idx] = 1;
71  }
72 }

82 void TFBatchTrajectoryRecorder::GetNextStatesUniform() {
83  for (int b = 0; b < batch_size_; ++b) {
84  if (!terminal_flags_[b]) {
85  std::vector<Action> actions = states_[b]->LegalActions();
86  absl::uniform_int_distribution<> dist(0, actions.size() - 1);
87  Action action = actions[dist(rng_)];
88  states_[b]->ApplyAction(action);
89  SampleChance(b);
90  }

117 void TFBatchTrajectoryRecorder::FillInputsAndMasks() {
118  TensorMap inputs_matrix = tf_inputs_.matrix<float>();
119  TensorMap mask_matrix = tf_legal_mask_.matrix<float>();
120 
121  std::vector<float> info_state_vector(game_->ObservationTensorSize());
122  for (int b = 0; b < batch_size_; ++b) {
123  if (!terminal_flags_[b]) {
124  std::vector<int> mask = states_[b]->LegalActionsMask();
125  // Is there a way to use a vector operation here?
126  for (int a = 0; a < mask.size(); ++a) {
127  mask_matrix(b, a) = mask[a];
128  }
129 
130  states_[b]->ObservationTensor(states_[b]->CurrentPlayer(),
131  absl::MakeSpan(info_state_vector));
132  for (int i = 0; i < info_state_vector.size(); ++i) {
133  inputs_matrix(b, i) = info_state_vector[i];

139 void TFBatchTrajectoryRecorder::ApplyActions() {
140  std::vector<double> prob_dist(num_actions_, 0.0);
141  auto sampled_action = tf_outputs_[1].matrix<tensorflow::int64>();
142  for (int b = 0; b < batch_size_; ++b) {
143  if (!terminal_flags_[b]) {
144  Action action = sampled_action(b);
145  SPIEL_CHECK_GE(action, 0);
146  SPIEL_CHECK_LT(action, num_actions_);
147  SPIEL_CHECK_EQ(tf_legal_mask_.matrix<float>()(b, action), 1);
148  states_[b]->ApplyAction(action);
149  SampleChance(b);
150  }
151  }
