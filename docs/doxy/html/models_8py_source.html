32def _mlp_forwards(mlp_hidden_sizes: List[int]) -> hk.Transformed:
33 """Returns a haiku transformation of the MLP model to be used in optimizer.
34
35 Args:
36 mlp_hidden_sizes: List containing size of linear layers.
37
38 Returns:
39 Haiku transformation of the RNN network.
40 """
41 def forward_fn(inputs):
42 mlp = hk.nets.MLP(mlp_hidden_sizes, activation=jax.nn.relu, name="mlp")
43 return mlp(inputs)
44 return hk.transform(forward_fn)

47def _make_rnn_network(lstm_hidden_sizes: List[int],
48 mlp_hidden_sizes: List[int]) -> hk.RNNCore:
49 """Returns the RNN network.
50
51 Args:
52 lstm_hidden_sizes: List containing size of lstm layers.
53 mlp_hidden_sizes: List containing size of linear layers.
54
55 Returns:
56 Returns an instance of RNN model.
57 """
58 layers = []
59 for k, hidden_size in enumerate(lstm_hidden_sizes):
60 layers += [hk.LSTM(hidden_size, name=f"lstm_layer_{k}"), jax.nn.relu]
61 layers += [hk.nets.MLP(mlp_hidden_sizes, name="mlp")]
62 return RNNModel(layers)

65def _rnn_forwards(lstm_hidden_sizes: List[int], mlp_hidden_sizes: List[int],
66 batch_size: int) -> hk.Transformed:
67 """Returns a haiku transformation of the RNN model to be used in optimizer.
68
69 Args:
70 lstm_hidden_sizes: List containing size of lstm layers.
71 mlp_hidden_sizes: List containing size of linear layers.
72 batch_size: Batch size.
73
74 Returns:
75 Haiku transformation of the RNN network.
76 """
77 def forward_fn(inputs):
78 rnn = _make_rnn_network(lstm_hidden_sizes, mlp_hidden_sizes)
79 initial_state = rnn.initial_state(batch_size=batch_size)
80 outputs, _ = hk.dynamic_unroll(rnn, inputs, initial_state, time_major=False)
81 return outputs
82
83 return hk.transform(forward_fn)

86class RNNModel(hk.RNNCore):
87 """RNN model."""
88
89 def __init__(self,
90 layers: List[Union[hk.Module, Callable[[jnp.ndarray],
91 jnp.ndarray]]],
92 name: Optional[str] = None):
93 super().__init__(name=name)
94 self._layers = layers
95
96 def __call__(self, inputs, prev_state):
97 x = inputs
98 curr_state = [None] * len(prev_state)
99 for k, layer in enumerate(self._layers):
100 if isinstance(layer, hk.RNNCore):
101 x, curr_state[k] = layer(x, prev_state[k])
102 else:
103 x = layer(x)
104 return x, tuple(curr_state)
105
106 def initial_state(self, batch_size: Optional[int]) -> Any:
107 layerwise_init_state = []
108 for layer in self._layers:
109 if isinstance(layer, hk.RNNCore):
110 layerwise_init_state.append(layer.initial_state(batch_size))
111 else:
112 layerwise_init_state.append(None)
113 return tuple(layerwise_init_state)

116class OptimizerModel:
117 """Optimizer model in l2l paradigm to learn update rules of regret minimizers.
118
119 Attributes:
120 mlp_sizes: Size of mlp layers. This is a string, containing sequence of
121 numbers, each number indicate size of a linear layer.
122 lstm_sizes: Size of lstm layers. This is a string, containing sequence of
123 numbers, each number indicate size of an lstm layer.
124 initial_learning_rate: Initial value of learning rate used in learning
125 rate scheduler.
126 batch_size: Batch size.
127 num_actions: Number of possible actions.
128 num_infostates: Total number of information states.
129 model_type: Type of model. For now it can be either MLP or RNN.
130 use_infostate_representation: Boolean value to indicate if we use
131 information state information as part of model input or not.
132 rng: Jax pseudo random number generator.
133 model: Neural network model we want to optimize.
134 opt_update: Optax optimizer update function.
135 net_params: Network parameters.
136 opt_state: Optax optimizer state.
137 net_apply: Network apply function.
138 """
139
140 def __init__(self,
141 mlp_sizes: str,
142 lstm_sizes: str,
143 initial_learning_rate: float,
144 batch_size: int,
145 num_actions: int,
146 num_infostates: int,
147 model_type: str = "MLP",
148 use_infostate_representation: bool = True):
149 self.num_actions = num_actions
150 self.num_infostates = num_infostates
151 self.initial_learning_rate = initial_learning_rate
152 self.batch_size = batch_size
153 self.use_infostate_representation = use_infostate_representation
154 self.rng = jax.random.PRNGKey(10)
155
156 mlp_sizes_list = [
157 int(mlp_size.strip()) for mlp_size in mlp_sizes.split(",")
158 ]
159 mlp_sizes_list.append(self.num_actions)
160 lstm_sizes_list = [
161 int(lstm_size.strip()) for lstm_size in lstm_sizes.split(",")
162 ]
163
164 if model_type == ModelType.MLP.value:
165 self.model = _mlp_forwards(mlp_sizes_list)
166 elif model_type == ModelType.RNN.value:
167 self.model = _rnn_forwards(lstm_sizes_list, mlp_sizes_list,
168 self.batch_size)
169 else:
170 raise ValueError(
171 f"{model_type} is not a valid model, model_type should be MLP or RNN."
172 )
173
174 self.net_apply = self.model.apply
175 self._net_init = self.model.init
176 self.opt_update, self.net_params, self.opt_state = None, None, None
177
178 def lr_scheduler(self, init_value: float) -> optax.Schedule:
179 schedule_fn = optax.polynomial_schedule(
180 init_value=init_value, end_value=0.0001, power=1., transition_steps=100)
181 return schedule_fn
182
183 def initialize_optimizer_model(self):
184 """Initializes the optax optimizer and neural network model."""
185 lr_scheduler_fn = self.lr_scheduler(self.initial_learning_rate)
186 opt_init, self.opt_update = optax.chain(
187 optax.scale_by_adam(), optax.scale_by_schedule(lr_scheduler_fn),
188 optax.scale(-self.initial_learning_rate))
189
190 input_size = self.num_actions
191 if self.use_infostate_representation:
192 input_size += self.num_infostates
193
194 dummy_input = np.zeros(shape=[self.batch_size, 1, input_size])
195
196 self.net_params = self._net_init(self.rng, dummy_input)
197 self.opt_state = opt_init(self.net_params)
