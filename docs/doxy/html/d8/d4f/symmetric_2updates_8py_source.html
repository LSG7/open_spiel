79 def euc_descent_step(self, params, grads, t, eps=0.):
80 """Projected gradient descent on exploitability using Euclidean projection.
81
82 Args:
83 params: tuple of variables to be updated (dist,)
84 grads: tuple of variable gradients (grad_dist,)
85 t: int, solver iteration
86 eps: float > 0, force all probabilities >= eps / dim(dist)
87 Returns:
88 new_params: tuple of update params (new_dist,)
89 """
90 del t
91 new_dist = params[0] - self.lrs[0] * grads[0]
92 new_dist = simplex.euclidean_projection_onto_simplex(new_dist)
93 if eps > 0:
94 new_dist = simplex.project_to_interior(new_dist, eps)
95 return (new_dist,)

97 def mirror_descent_step(self, params, grads, t, eps=0.):
98 """Entropic mirror descent on exploitability.
99
100 Args:
101 params: tuple of variables to be updated (dist)
102 grads: tuple of variable gradients (grad_dist)
103 t: int, solver iteration
104 eps: float > 0, force all probabilities >= eps / dim(dist)
105 Returns:
106 new_params: tuple of update params (new_dist)
107 """
108 del t
109 dist = np.clip(params[0], 0, np.inf)
110 new_dist = special.softmax(np.log(dist) - self.lrs[0] * grads[0])
111 if eps > 0:
112 new_dist = simplex.project_to_interior(new_dist, eps)
113 return (new_dist,)
