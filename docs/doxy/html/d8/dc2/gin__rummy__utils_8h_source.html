31struct SuitComparator {
32 explicit SuitComparator(const int num_ranks) : num_ranks(num_ranks) {}
33 int CardSuit(int card) { return card / num_ranks; }
34 bool operator()(int card_1, int card_2) {
35 if (CardSuit(card_1) == CardSuit(card_2)) {
36 return card_1 < card_2;
37 }
38 return CardSuit(card_1) < CardSuit(card_2);
39 }
40 const int num_ranks;
41};
42
43struct RankComparator {
44 explicit RankComparator(const int num_ranks) : num_ranks(num_ranks) {}
45 int CardRank(int card) { return card % num_ranks; }
46 bool operator()(int card_1, int card_2) {
47 if (CardRank(card_1) == CardRank(card_2)) {
48 return card_1 < card_2;
49 }
50 return CardRank(card_1) < CardRank(card_2);
51 }
52 const int num_ranks;
53};
