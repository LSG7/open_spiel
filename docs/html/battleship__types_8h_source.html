26 // A cell of a generic player's board.
27 struct Cell {
28  int row;
29  int col;
30 
31  std::string ToString() const { return absl::StrFormat("%d_%d", row, col); }
32  bool operator==(const Cell& other) const {
33  return row == other.row && col == other.col;
34  }
35  bool operator<(const Cell& other) const {
36  return (row < other.row) || (row == other.row && col < other.col);
37  }
38 };

63 // Ships can be placed either horizontal or vertical on the board. When the ship

122 // Returns whether there is still enough space to finish placing ships in a
123 // partially-filled-in board.
124 //
125 // This method receives a vector of current ship placements, and
126 // returns `true` if and only if there is at least one way to place the
127 // remaining ship on the board without overlapping ships.
128 //
129 // This method is used when deciding the set of placement actions a player has
130 // a given point in time, as well as to validate at construction time whether
131 // the given Battleship configuration is feasible.
132 //
133 //
134 // Inductive contract
135 // ------------------
136 //
137 // The correctness of this method relies on the following inductive contract.
138 // The preconditions are checked in debug mode.
139 // - Precondition: partial_placement contains a valid (that is, within the
140 // bounds of the board, and non-overlapping) placement of a prefix of ships
141 // defined in `conf.ships`.
142 // - Postcondition: by the time the function returns, partial_placement is
143 // exactly the same vector as at the time of entering the call.
144 bool ExistsFeasiblePlacement(const BattleshipConfiguration& conf,
145  std::vector<ShipPlacement>* partial_placement);
