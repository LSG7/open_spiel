27 def solve_stackelberg(game, is_first_leader=True):
28  """Solves the optimal mixed strategty to commit to for the leader.
29 
30  Args:
31  game: a pyspiel game,
32  is_first_leader: if true, then player 0 is the leader, o.w. player 1 is
33  the leader.
34 
35  Returns:
36  (player0 strategy, player1 strategy, player0 payoff, player1 payoff) at an
37  SSE.
38  """
39  p_mat = game_payoffs_array(game)
40  assert len(p_mat) == 2
41  if is_first_leader:
42  leader_payoff, follower_payoff = p_mat[0], p_mat[1]
43  else:
44  leader_payoff, follower_payoff = p_mat[1].T, p_mat[0].T
45 
46  num_leader_strategies, num_follower_strategies = leader_payoff.shape
47 
48  leader_eq_value = -float("inf")
49  follower_eq_value = None
50  leader_eq_strategy = None
51  follower_eq_strategy = None
52 
53  for t in range(num_follower_strategies):
54  lp = lp_solver.LinearProgram(objective=lp_solver.OBJ_MAX)
55  for s in range(num_leader_strategies):
56  lp.add_or_reuse_variable("s_{}".format(s))
57  lp.set_obj_coeff("s_{}".format(s), leader_payoff[s, t])
58 
59  for t_ in range(num_follower_strategies):
60  if t_ == t:
61  continue
62  lp.add_or_reuse_constraint("t_{}".format(t_), lp_solver.CONS_TYPE_GEQ)
63  for s in range(num_leader_strategies):
64  lp.set_cons_coeff("t_{}".format(t_), "s_{}".format(s),
65  follower_payoff[s, t] - follower_payoff[s, t_])
66  lp.set_cons_rhs("t_{}".format(t_), 0.0)
67  lp.add_or_reuse_constraint("sum_to_one", lp_solver.CONS_TYPE_EQ)
68  for s in range(num_leader_strategies):
69  lp.set_cons_coeff("sum_to_one", "s_{}".format(s), 1.0)
70  lp.set_cons_rhs("sum_to_one", 1.0)
71  try:
72  leader_strategy = np.array(lp.solve())
73  leader_strategy = _simplex_projection(
74  leader_strategy.reshape(-1)).reshape(-1, 1)
75  leader_value = leader_strategy.T.dot(leader_payoff)[0, t]
76  if leader_value > leader_eq_value:
77  leader_eq_strategy = leader_strategy
78  follower_eq_strategy = t
79  leader_eq_value = leader_value
80  follower_eq_value = leader_strategy.T.dot(follower_payoff)[0, t]
81  except: # pylint: disable=bare-except
82  continue
83  if is_first_leader:
84  return (leader_eq_strategy.reshape(-1), np.identity(
85  num_follower_strategies)[follower_eq_strategy],
86  leader_eq_value, follower_eq_value)
87  else:
88  return (np.identity(num_follower_strategies)[follower_eq_strategy],
89  leader_eq_strategy.reshape(-1), follower_eq_value, leader_eq_value)
