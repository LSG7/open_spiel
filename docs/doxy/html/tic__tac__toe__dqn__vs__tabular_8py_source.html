15"""DQN agent vs Tabular Q-Learning agents trained on Tic Tac Toe.

41def pretty_board(time_step):
42 """Returns the board in `time_step` in a human readable format."""
43 info_state = time_step.observations["info_state"][0]
44 x_locations = np.nonzero(info_state[9:18])[0]
45 o_locations = np.nonzero(info_state[18:])[0]
46 board = np.full(3 * 3, ".")
47 board[x_locations] = "X"
48 board[o_locations] = "0"
49 board = np.reshape(board, (3, 3))
50 return board

53def command_line_action(time_step):
54 """Gets a valid action from the user on the command line."""
55 current_player = time_step.observations["current_player"]
56 legal_actions = time_step.observations["legal_actions"][current_player]
57 action = -1
58 while action not in legal_actions:
59 print("Choose an action from {}:".format(legal_actions))
60 sys.stdout.flush()
61 action_str = input()
62 try:
63 action = int(action_str)
64 except ValueError:
65 continue
66 return action

69def eval_against_random_bots(env, trained_agents, random_agents, num_episodes):
70 """Evaluates `trained_agents` against `random_agents` for `num_episodes`."""
71 num_players = len(trained_agents)
72 sum_episode_rewards = np.zeros(num_players)
73 for player_pos in range(num_players):
74 cur_agents = random_agents[:]
75 cur_agents[player_pos] = trained_agents[player_pos]
76 for _ in range(num_episodes):
77 time_step = env.reset()
78 episode_rewards = 0
79 while not time_step.last():
80 player_id = time_step.observations["current_player"]
81 agent_output = cur_agents[player_id].step(time_step, is_evaluation=True)
82 action_list = [agent_output.action]
83 time_step = env.step(action_list)
84 episode_rewards += time_step.rewards[player_pos]
85 sum_episode_rewards[player_pos] += episode_rewards
86 return sum_episode_rewards / num_episodes

89def main(_):
90 game = "tic_tac_toe"
91 num_players = 2
92 env = rl_environment.Environment(game)
93 state_size = env.observation_spec()["info_state"][0]
94 num_actions = env.action_spec()["num_actions"]
95
96 hidden_layers_sizes = [32, 32]
97 replay_buffer_capacity = int(1e4)
98 train_episodes = FLAGS.num_episodes
99 loss_report_interval = 1000
100
101 with tf.Session() as sess:
102 dqn_agent = dqn.DQN(
103 sess,
104 player_id=0,
105 state_representation_size=state_size,
106 num_actions=num_actions,
107 hidden_layers_sizes=hidden_layers_sizes,
108 replay_buffer_capacity=replay_buffer_capacity)
109 tabular_q_agent = tabular_qlearner.QLearner(
110 player_id=1, num_actions=num_actions)
111 agents = [dqn_agent, tabular_q_agent]
112
113 sess.run(tf.global_variables_initializer())
114
115 # Train agent
116 for ep in range(train_episodes):
117 if ep and ep % loss_report_interval == 0:
118 logging.info("[%s/%s] DQN loss: %s", ep, train_episodes, agents[0].loss)
119 time_step = env.reset()
120 while not time_step.last():
121 player_id = time_step.observations["current_player"]
122 agent_output = agents[player_id].step(time_step)
123 action_list = [agent_output.action]
124 time_step = env.step(action_list)
125
126 # Episode is over, step all agents with final info state.
127 for agent in agents:
128 agent.step(time_step)
129
130 # Evaluate against random agent
131 random_agents = [
132 random_agent.RandomAgent(player_id=idx, num_actions=num_actions)
133 for idx in range(num_players)
134 ]
135 r_mean = eval_against_random_bots(env, agents, random_agents, 1000)
136 logging.info("Mean episode rewards: %s", r_mean)
137
138 if not FLAGS.interactive_play:
139 return
140
141 # Play from the command line against the trained DQN agent.
142 human_player = 1
143 while True:
144 logging.info("You are playing as %s", "X" if human_player else "0")
145 time_step = env.reset()
146 while not time_step.last():
147 player_id = time_step.observations["current_player"]
148 if player_id == human_player:
149 agent_out = agents[human_player].step(time_step, is_evaluation=True)
150 logging.info("\n%s", agent_out.probs.reshape((3, 3)))
151 logging.info("\n%s", pretty_board(time_step))
152 action = command_line_action(time_step)
153 else:
154 agent_out = agents[1 - human_player].step(
155 time_step, is_evaluation=True)
156 action = agent_out.action
157 time_step = env.step([action])
158
159 logging.info("\n%s", pretty_board(time_step))
160
161 logging.info("End of game!")
162 if time_step.rewards[human_player] > 0:
163 logging.info("You win")
164 elif time_step.rewards[human_player] < 0:
165 logging.info("You lose")
166 else:
167 logging.info("Draw")
168 # Switch order of players
169 human_player = 1 - human_player

172if __name__ == "__main__":
173 app.run(main)
