32def make_iterated_matrix_game(
33 game: str, iterations=5, batch_size=8
34) -> rl_environment.Environment:
35 matrix_game = pyspiel.load_matrix_game(game)
36 config = {'num_repetitions': iterations, 'batch_size': batch_size}
37 game = pyspiel.create_repeated_game(matrix_game, config)
38 env = rl_environment.Environment(game)
39 return env

42def make_agent_networks(
43 num_actions: int,
44) -> Tuple[hk.Transformed, hk.Transformed]:
45 def policy(obs):
46 logits = hk.nets.MLP(output_sizes=[8, 8, num_actions], with_bias=True)(obs)
47 logits = jnp.nan_to_num(logits)
48 return distrax.Categorical(logits=logits)
49
50 def value_fn(obs):
51 values = hk.nets.MLP(output_sizes=[8, 8, 1], with_bias=True)(obs)
52 return values
53
54 return hk.without_apply_rng(hk.transform(policy)), hk.without_apply_rng(
55 hk.transform(value_fn)
56 )

59def run_agents(
60 agents: typing.List[OpponentShapingAgent],
61 env: rl_environment.Environment,
62 num_steps=1000,
63):
64 time_step = env.reset()
65 for _ in range(num_steps):
66 actions = []
67 for agent in agents:
68 action, _ = agent.step(time_step)
69 if action is not None:
70 action = action.squeeze()
71 actions.append(action)
72 if time_step.last():
73 time_step = env.reset()
74 else:
75 time_step = env.step(actions)
76 time_step.observations['actions'] = np.array(actions)
