48template <class T>
49T* AllocBuf(const std::vector<T>& vec, int* size) {
50 *size = vec.size();
51 size_t num_bytes = *size * sizeof(T);
52 T* buf = static_cast<T*>(malloc(num_bytes));
53 memcpy(buf, vec.data(), num_bytes);
54 return buf;
55}
56
57char* AllocAndCopyString(const std::string& str) {
58 char* buf = static_cast<char*>(malloc(str.length() * sizeof(char)));
59 strncpy(buf, str.data(), str.length());
60 return buf;
61}

247double* StateChanceOutcomeProbs(const void* state_ptr, int* size) {
248 const State* state = reinterpret_cast<const State*>(state_ptr);
249 ActionsAndProbs chance_outcomes = state->ChanceOutcomes();
250 *size = chance_outcomes.size();
251 size_t num_bytes = *size * sizeof(double);
252 double* buf = static_cast<double*>(malloc(num_bytes));
253 for (int i = 0; i < chance_outcomes.size(); ++i) {
254 buf[i] = chance_outcomes[i].second;
255 }
256 return buf;
257}
