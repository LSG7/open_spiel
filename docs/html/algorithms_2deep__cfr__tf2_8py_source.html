15 """Implements Deep CFR Algorithm.
16 
17 See https://arxiv.org/abs/1811.00164.
18 
19 The algorithm defines an `advantage` and `strategy` networks that compute
20 advantages used to do regret matching across information sets and to approximate
21 the strategy profiles of the game. To train these networks a reservoir buffer
22 (other data structures may be used) memory is used to accumulate samples to
23 train the networks.
24 
25 This implementation uses skip connections as described in the paper if two
26 consecutive layers of the advantage or policy network have the same number
27 of units, except for the last connection. Before the last hidden layer
28 a layer normalization is applied.
29 """

49 class ReservoirBuffer(object):
50  """Allows uniform sampling over a stream of data.
51 
52  This class supports the storage of arbitrary elements, such as observation
53  tensors, integer actions, etc.
54 
55  See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
56  """
57 
58  def __init__(self, reservoir_buffer_capacity):
59  self._reservoir_buffer_capacity_reservoir_buffer_capacity = reservoir_buffer_capacity
60  self._data_data = []
61  self._add_calls_add_calls = 0
62 
63  def add(self, element):
64  """Potentially adds `element` to the reservoir buffer.
65 
66  Args:
67  element: data to be added to the reservoir buffer.
68  """
69  if len(self._data_data) < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
70  self._data_data.append(element)
71  else:
72  idx = np.random.randint(0, self._add_calls_add_calls + 1)
73  if idx < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
74  self._data_data[idx] = element
75  self._add_calls_add_calls += 1
76 
77  def sample(self, num_samples):
78  """Returns `num_samples` uniformly sampled from the buffer.
79 
80  Args:
81  num_samples: `int`, number of samples to draw.
82 
83  Returns:
84  An iterable over `num_samples` random elements of the buffer.
85 
86  Raises:
87  ValueError: If there are less than `num_samples` elements in the buffer
88  """
89  if len(self._data_data) < num_samples:
90  raise ValueError('{} elements could not be sampled from size {}'.format(
91  num_samples, len(self._data_data)))
92  return random.sample(self._data_data, num_samples)

94  def clear(self):
95  self._data_data = []
96  self._add_calls_add_calls = 0

99  return len(self._data_data)

102  return iter(self._data_data)

123 class PolicyNetwork(tf.keras.Model):
124  """Implements the policy network as an MLP.
125 
126  Implements the policy network as a MLP with skip connections in adjacent
127  layers with the same number of units, except for the last hidden connection
128  where a layer normalization is applied.
129  """
130 
131  def __init__(self,
132  input_size,
133  policy_network_layers,
134  num_actions,
135  activation='leakyrelu',
136  **kwargs):
137  super().__init__(**kwargs)
138  self._input_size_input_size = input_size
139  self._num_actions_num_actions = num_actions
140  if activation == 'leakyrelu':
141  self.activationactivation = tf.keras.layers.LeakyReLU(alpha=0.2)
142  elif activation == 'relu':
143  self.activationactivation = tf.keras.layers.ReLU()
144  else:
145  self.activationactivation = activation
146 
147  self.softmaxsoftmax = tf.keras.layers.Softmax()
148 
149  self.hiddenhidden = []
150  prevunits = 0
151  for units in policy_network_layers[:-1]:
152  if prevunits == units:
153  self.hiddenhidden.append(SkipDense(units))
154  else:
155  self.hiddenhidden.append(
156  tf.keras.layers.Dense(units, kernel_initializer='he_normal'))
157  prevunits = units
158  self.normalizationnormalization = tf.keras.layers.LayerNormalization()
159  self.lastlayerlastlayer = tf.keras.layers.Dense(
160  policy_network_layers[-1], kernel_initializer='he_normal')
161 
162  self.out_layerout_layer = tf.keras.layers.Dense(num_actions)
163 
164  @tf.function
165  def call(self, inputs):
166  """Applies Policy Network.
167 
168  Args:
169  inputs: Tuple representing (info_state, legal_action_mask)
170 
171  Returns:
172  Action probabilities
173  """
174  x, mask = inputs
175  for layer in self.hiddenhidden:
176  x = layer(x)
177  x = self.activationactivation(x)
178 
179  x = self.normalizationnormalization(x)
180  x = self.lastlayerlastlayer(x)
181  x = self.activationactivation(x)
182  x = self.out_layerout_layer(x)
183  x = tf.where(mask == 1, x, -10e20)
184  x = self.softmaxsoftmax(x)
185  return x

229  """Applies Policy Network.
230 
231  Args:
232  inputs: Tuple representing (info_state, legal_action_mask)
233 
234  Returns:

236  """

543  def _traverse_game_tree(self, state, player):
544  """Performs a traversal of the game tree using external sampling.
545 
546  Over a traversal the advantage and strategy memories are populated with
547  computed advantage values and matched regrets respectively.
548 
549  Args:
550  state: Current OpenSpiel game state.
551  player: (int) Player index for this traversal.
552 
553  Returns:
554  Recursively returns expected payoffs for each action.
555  """
556  if state.is_terminal():
557  # Terminal state get returns.
558  return state.returns()[player]
559  elif state.is_chance_node():
560  # If this is a chance node, sample an action
561  chance_outcome, chance_proba = zip(*state.chance_outcomes())
562  action = np.random.choice(chance_outcome, p=chance_proba)
563  return self._traverse_game_tree_traverse_game_tree(state.child(action), player)
564  elif state.current_player() == player:
565  # Update the policy over the info set & actions via regret matching.
