15// This is a simple example that is used to demonstrate how to use OpenSpiel
16// when it is built as a shared library. To use OpenSpiel as a library,
17// see: https://github.com/deepmind/open_spiel/blob/master/docs/library.md

26void PrintLegalActions(const open_spiel::State& state,
27 open_spiel::Player player,
28 const std::vector<open_spiel::Action>& movelist) {
29 std::cerr << "Legal moves for player " << player << ":" << std::endl;
30 for (open_spiel::Action action : movelist) {
31 std::cerr << " " << state.ActionToString(player, action) << std::endl;
32 }
33}

35int main(int argc, char** argv) {
36 if (argc < 2) {
37 std::cerr << "Usage shared_library_example <game string>" << std::endl;
38 return -1;
39 }
40
41 // Print out registered games.
42 std::cerr << "Registered games:" << std::endl;
43 std::vector<std::string> names = open_spiel::RegisteredGames();
44 for (const std::string& name : names) {
45 std::cerr << name << std::endl;
46 }
47
48 // Random number generator.
49 std::mt19937 rng(time(0));
50
51 // Load the game.
52 std::cerr << "Loading game..\n" << std::endl;
53 std::shared_ptr<const open_spiel::Game> game = open_spiel::LoadGame(argv[1]);
54
55 if (!game) {
56 std::cerr << "problem with loading game, exiting..." << std::endl;
57 return -1;
58 }
59
60 std::cerr << "Starting new game..." << std::endl;
61 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
62
63 std::cerr << "Initial state:" << std::endl;
64 std::cerr << "State:" << std::endl << state->ToString() << std::endl;
65
66 while (!state->IsTerminal()) {
67 std::cerr << "player " << state->CurrentPlayer() << std::endl;
68
69 if (state->IsChanceNode()) {
70 // Chance node; sample one according to underlying distribution.
71 std::vector<std::pair<open_spiel::Action, double>> outcomes =
72 state->ChanceOutcomes();
73 open_spiel::Action action = open_spiel::SampleAction(outcomes, rng).first;
74 std::cerr << "sampled outcome: "
75 << state->ActionToString(open_spiel::kChancePlayerId, action)
76 << std::endl;
77 state->ApplyAction(action);
78 } else if (state->IsSimultaneousNode()) {
79 // open_spiel::Players choose simultaneously?
80 std::vector<open_spiel::Action> joint_action;
81 std::vector<float> infostate(game->InformationStateTensorSize());
82
83 // Sample a action for each player
84 for (auto player = open_spiel::Player{0}; player < game->NumPlayers();
85 ++player) {
86 std::vector<open_spiel::Action> actions = state->LegalActions(player);
87 PrintLegalActions(*state, player, actions);
88
89 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
90 open_spiel::Action action = actions[dis(rng)];
91 joint_action.push_back(action);
92 std::cerr << "player " << player << " chose "
93 << state->ActionToString(player, action) << std::endl;
94 }
95
96 state->ApplyActions(joint_action);
97 } else {
98 // Decision node, sample one uniformly.
99 auto player = state->CurrentPlayer();
100 std::vector<open_spiel::Action> actions = state->LegalActions();
101 PrintLegalActions(*state, player, actions);
102
103 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
104 auto action = actions[dis(rng)];
105 std::cerr << "chose action: " << state->ActionToString(player, action)
106 << std::endl;
107 state->ApplyAction(action);
108 }
109
110 std::cerr << "State: " << std::endl << state->ToString() << std::endl;
111 }
112
113 auto returns = state->Returns();
114 for (auto p = open_spiel::Player{0}; p < game->NumPlayers(); p++) {
115 std::cerr << "Final return to player " << p << " is " << returns[p]
116 << std::endl;
117 }
118}
