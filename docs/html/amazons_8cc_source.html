85 std::vector<Action> AmazonsState::GetHorizontalMoves(Action cell) const {
86  std::vector<Action> horizontalMoves;
87 
88  unsigned char col = cell % kNumRows; // The column the cell is in
89  unsigned char left =
90  col; // The maximum amount of spaces to check left of given cell
91  unsigned char right =
92  kNumCols - col -
93  1; // The maximal amount of spaces to check right of given cell
94  Action focus;
95 
96  // <-----X
97  // Walk until we encounter a blocking piece or end of row
98  int count = 1;
99  while (count <= left) {
100  focus = cell - count;
101  if (board_[focus] == CellState::kEmpty) {
102  horizontalMoves.push_back(focus);
103  count++;
104  } else {
105  // We have encountered a blocking piece
106  break;
107  }
108  }
109 
110  // X---->
111  // Walk until we encounter a blocking piece or end of row
112  count = 1;
113  while (count <= right) {
114  focus = cell + count;
115  if (board_[focus] == CellState::kEmpty) {
116  horizontalMoves.push_back(focus);
117  count++;
118  } else {
119  // We have encountered a blocking piece
120  break;
121  }
122  }
123 
124  return horizontalMoves;
125 }

177 std::vector<Action> AmazonsState::GetDiagonalMoves(Action cell) const {
178  std::vector<Action> diagonalMoves;
179  unsigned char col = cell % kNumCols; // The column the cell is in
180  unsigned char row = cell / kNumRows; // The row the cell is in
181  unsigned char upLeft = std::min(
182  row,
183  col); // The maximum amount of spaces to check up and left of given cell
184  unsigned char upRight = std::min(
185  row,
186  (unsigned char)(kNumCols - col - 1)); // The maximum amount of spaces to
187  // check up and right of given cell
188  // The maximum amount of spaces to check
189  // down and left of given cell
190  unsigned char downLeft =
191  std::min(static_cast<unsigned char>(kNumRows - row - 1), col);
192  // The maximum amount of spaces to check down
193  // and right of given cell
194  unsigned char downRight =
195  std::min(static_cast<unsigned char>(kNumRows - row - 1),
196  static_cast<unsigned char>(kNumCols - col - 1));
197  Action focus;
198 
199  // Up and left
200  int count = 1;
201  focus = cell;
202  while (count <= upLeft) {
203  focus -= (kNumRows + 1);
204  if (board_[focus] == CellState::kEmpty) {
205  diagonalMoves.push_back(focus);
206  count++;
207  } else {
208  // We have encountered a blocking piece
209  break;
210  }
211  }
212 
213  // Up and right
214  count = 1;
215  focus = cell;
216  while (count <= upRight) {
217  focus -= (kNumRows - 1);
218  if (board_[focus] == CellState::kEmpty) {
219  diagonalMoves.push_back(focus);
220  count++;
221  } else {
222  // We have encountered a blocking piece
223  break;
224  }
225  }
226 
227  // Down and left
228  count = 1;
229  focus = cell;
230  while (count <= downLeft) {
231  focus += (kNumRows - 1);
232  if (board_[focus] == CellState::kEmpty) {
233  diagonalMoves.push_back(focus);
234  count++;
235  } else {
236  // We have encountered a blocking piece
237  break;
238  }
239  }
240 
241  // Down and right
242  count = 1;
243  focus = cell;
244  while (count <= downRight) {
245  focus += (kNumRows + 1);
246  if (board_[focus] == CellState::kEmpty) {
247  diagonalMoves.push_back(focus);
248  count++;
249  } else {
250  // We have encountered a blocking piece
251  break;
252  }
253  }
254 
255  return diagonalMoves;
256 }

258 std::vector<Action> AmazonsState::GetAllMoves(Action cell) const {
259  std::vector<Action> horizontals = GetHorizontalMoves(cell);
260  std::vector<Action> verticals = GetVerticalMoves(cell);
261  std::vector<Action> diagonals = GetDiagonalMoves(cell);
262 
263  std::vector<Action> acc = horizontals;
264 
265  acc.insert(acc.end(), verticals.begin(), verticals.end());
266  acc.insert(acc.end(), diagonals.begin(), diagonals.end());
267

271 void AmazonsState::DoApplyAction(Action action) {
272  switch (state_) {
273  case amazon_select: {
274  SPIEL_CHECK_EQ(board_[action], PlayerToState(CurrentPlayer()));
275  from_ = action;
276  board_[from_] = CellState::kEmpty;
277  state_ = destination_select;
278  }
279  break;
280 
281  case destination_select: {
282  SPIEL_CHECK_EQ(board_[action], CellState::kEmpty);
283  to_ = action;
284  board_[to_] = PlayerToState(CurrentPlayer());
285  state_ = shot_select;
286  break;
287  }
288 
289  case shot_select: {
290  SPIEL_CHECK_EQ(board_[action], CellState::kEmpty);
291  shoot_ = action;
292  board_[shoot_] = CellState::kBlock;
293  current_player_ = 1 - current_player_;
294  state_ = amazon_select;
295  // Check if game is over
296  if (LegalActions().empty()) {
297  // outcome = winner
298  outcome_ = 1 - current_player_;
299  }
300  }
301  break;
302  }
303  ++num_moves_;
304 }

306 void AmazonsState::UndoAction(Player player, Action move) {
307  switch (state_) {
308  case amazon_select: {
309  shoot_ = move;

311  current_player_ = player;
312  outcome_ = kInvalidPlayer;

316 
317  case destination_select: {
318  from_ = move;
319  board_[from_] = PlayerToState(player);
320  state_ = amazon_select;
321  }
322  break;
323 
324  case shot_select: {
325  to_ = move;
326  board_[to_] = CellState::kEmpty;
327  state_ = destination_select;
328  }
329  break;
330  }
331 
332  --num_moves_;
333  --move_number_;
334  history_.pop_back();

337 std::vector<Action> AmazonsState::LegalActions() const {
338  if (IsTerminal()) return {};
339 
340  std::vector<Action> actions;
341 
342  switch (state_) {
343  case amazon_select:
344  for (int i = 0; i < board_.size(); i++) {
345  if (board_[i] == PlayerToState(CurrentPlayer())) {
346  // check if the selected amazon has a possible move
347  if (GetAllMoves(i).empty()) continue;
348 
349  actions.push_back(i);
350  }
351  }
352 
353  break;
354 
355  case destination_select:
356  actions = GetAllMoves(from_);
357  break;
358 
359  case shot_select:
360  actions = GetAllMoves(to_);
361  break;
362  }
363 
364  sort(actions.begin(), actions.end());
365 
366  return actions;
367 }
