475 void StonesNGemsState::ObservationTensor(Player player,
476  absl::Span<float> values) const {
477  SPIEL_CHECK_GE(player, 0);
478  SPIEL_CHECK_LT(player, num_players_);
479 
480  // Treat `values` as a 3-d tensor.
481  TensorView<3> view(
482  values, {kNumVisibleCellType, grid_.num_rows, grid_.num_cols}, true);
483 
484  // No observations at chance nodes.
485  if (IsChanceNode()) {
486  std::fill(values.begin(), values.end(), 0);
487  return;
488  }
489 
490  int i = 0;
491  for (int row = 0; row < grid_.num_rows; ++row) {
492  for (int col = 0; col < grid_.num_cols; ++col) {
493  int channel = static_cast<int>(grid_.elements[i].visible_type);
494  view[{channel, row, col}] = obs_show_ids_ ? grid_.ids[i] : 1.0;
495  ++i;
496  }
497  }
498 }

555 int StonesNGemsState::IndexFromAction(int index, int action) const {

559  col += offsets.first;
560  row += offsets.second;
561  return (grid_.num_cols * row) + col;

849 void StonesNGemsState::UpdateAgent(int index, int action) {
850  if (IsType(index, kElEmpty, action) || IsType(index, kElDirt, action)) {
851  // Move if empty/dirt
852  MoveItem(index, action);
853  } else if (IsType(index, kElDiamond, action) ||
854  IsType(index, kElDiamondFalling, action)) {
855  // Collect gems
856  ++gems_collected_;
857  current_reward_ += kGemPoints.at(GetItem(index, action));
858  sum_reward_ += kGemPoints.at(GetItem(index, action));
859  MoveItem(index, action);
860  } else if (IsActionHorz(action) && (IsType(index, kElStone, action) ||
861  IsType(index, kElNut, action) ||
862  IsType(index, kElBomb, action))) {
863  // Push stone, nut, or bomb if action is horizontal
864  Push(index, GetItem(index, action), kElToFalling.at(GetItem(index, action)),
865  action);
866  } else if (IsKey(GetItem(index, action))) {
867  // Collecting key, set gate open
868  OpenGate(kKeyToGate.at(GetItem(index, action)));
869  MoveItem(index, action);
870  } else if (IsOpenGate(GetItem(index, action))) {
871  // Walking through an open gate, with traversable element on other side
872  int index_gate = IndexFromAction(index, action);
873  if (HasProperty(index_gate, ElementProperties::kTraversable, action)) {
874  // Correct for landing on traversable elements
875  if (IsType(index_gate, kElDiamond, action)) {
876  ++gems_collected_;
877  current_reward_ += kGemPoints.at(GetItem(index_gate, action));
878  sum_reward_ += kGemPoints.at(GetItem(index_gate, action));
879  } else if (IsKey(GetItem(index_gate, action))) {
880  OpenGate(kKeyToGate.at(GetItem(index_gate, action)));
881  }
882  SetItem(index_gate, kElAgent, grid_.ids[index], action);
883  SetItem(index, kElEmpty, ++id_counter_);
884  }
885  } else if (IsType(index, kElExitOpen, action)) {
886  // Walking into exit after collecting enough gems
887  MoveItem(index, action);
888  SetItem(index, kElAgentInExit, ++id_counter_, action);
889  current_reward_ += steps_remaining_;
890  sum_reward_ += steps_remaining_;
891  }

1033 void StonesNGemsState::DoApplyAction(Action move) {
1034  if (cur_player_ == kChancePlayerId) {
1035  // Check each cell and apply respective dynamics function
1036  for (int index = 0; index < grid_.num_cols * grid_.num_rows; ++index) {
1037  Element &e = grid_.elements[index];
1038  if (e.has_updated) {
1039  continue;
1040  } else if (e == kElStone) {
1041  UpdateStone(index);
1042  } else if (e == kElStoneFalling) {
1043  UpdateStoneFalling(index);
1044  } else if (e == kElDiamond) {
1045  UpdateDiamond(index);
1046  } else if (e == kElDiamondFalling) {
1047  UpdateDiamondFalling(index);
1048  } else if (e == kElNut) {
1049  UpdateNut(index);
1050  } else if (e == kElNutFalling) {
1051  UpdateNutFalling(index);
1052  } else if (e == kElBomb) {
1053  UpdateBomb(index);
1054  } else if (e == kElBombFalling) {
1055  UpdateBombFalling(index);
1056  } else if (e == kElExitClosed) {
1057  UpdateExit(index);
1058  } else if (IsButterfly(e)) {
1059  UpdateButterfly(index, kButterflyToDirection.at(e));
1060  } else if (IsFirefly(e)) {
1061  UpdateFirefly(index, kFireflyToDirection.at(e));
1062  } else if (IsOrange(e)) {
1063  UpdateOrange(index, kOrangeToDirection.at(e));
1064  } else if (IsMagicWall(e)) {
1065  UpdateMagicWall(index);
1066  } else if (e == kElBlob) {
1067  UpdateBlob(index);
1068  } else if (IsExplosion(e)) {
1069  UpdateExplosions(index);
1070  }
1071  }
1072  EndScan();
