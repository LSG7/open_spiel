43  class LossInfo {
44  public:
45  LossInfo() {}
46  LossInfo(double policy, double value, double l2) :
47  policy_(policy), value_(value), l2_(l2), batches_(1) {}
48 
49  // Merge another LossInfo into this one.
50  LossInfo& operator+=(const LossInfo& other) {
51  policy_ += other.policy_;
52  value_ += other.value_;
53  l2_ += other.l2_;
54  batches_ += other.batches_;
55  return *this;
56  }
57 
58  // Return the average losses over all merged into this one.
59  double Policy() const { return policy_ / batches_; }
60  double Value() const { return value_ / batches_; }
61  double L2() const { return l2_ / batches_; }
62  double Total() const { return Policy() + Value() + L2(); }
63 
64  private:
65  double policy_ = 0;
66  double value_ = 0;
67  double l2_ = 0;
68  int batches_ = 0;
