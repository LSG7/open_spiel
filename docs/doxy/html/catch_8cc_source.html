82std::vector<Action> CatchState::LegalActions() const {
83 if (IsTerminal()) return {};
84 if (initialized_) {
85 return {0, 1, 2}; // Left, stay, right.
86 }
87 std::vector<Action> moves;
88 moves.reserve(num_columns_);
89 for (int i = 0; i < num_columns_; i++) moves.push_back(i);
90 return moves;
91}

93ActionsAndProbs CatchState::ChanceOutcomes() const {
94 SPIEL_CHECK_TRUE(!initialized_);
95 ActionsAndProbs action_and_probs;
96 action_and_probs.reserve(num_columns_);
97 for (int c = 0; c < num_columns_; c++) {
98 action_and_probs.emplace_back(c, 1. / num_columns_);
99 }
100 return action_and_probs;
101}

103CellState CatchState::BoardAt(int row, int column) const {
104 if (row == num_rows_ - 1 && column == paddle_col_)
105 return CellState::kPaddle;
106 else if (row == ball_row_ && column == ball_col_)
107 return CellState::kBall;
108 return CellState::kEmpty;
109}

111std::string CatchState::ActionToString(Player player, Action action_id) const {

127std::string CatchState::ToString() const {
128 std::string str;

174void CatchState::DoApplyAction(Action move) {
175 if (!initialized_) {
176 initialized_ = true;
177 ball_col_ = move;
178 ball_row_ = 0;
179 paddle_col_ = num_columns_ / 2;
180 } else {
181 ball_row_++;
182 int direction = move - 1;
183 paddle_col_ =
184 std::min(std::max(paddle_col_ + direction, 0), num_columns_ - 1);
185 }
186}
