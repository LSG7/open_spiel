36def print_iteration(time_step, actions, player_id):
37 """Print TimeStep information."""
38 obs = time_step.observations
39 logging.info("Player: %s", player_id)
40 if time_step.step_type.first():
41 logging.info("Info state: %s, - - %s", obs["info_state"][player_id],

44 logging.info("Info state: %s, %s %s %s", obs["info_state"][player_id],

51def turn_based_example(unused_arg):
52 """Example usage of the RL environment for turn-based games."""
53 # `rl_main_loop.py` contains more details and simultaneous move examples.
54 logging.info("Registered games: %s", rl_environment.registered_games())
55 logging.info("Creating game %s", FLAGS.game)
56
57 env_configs = {"players": FLAGS.num_players} if FLAGS.num_players else {}
58 env = rl_environment.Environment(FLAGS.game, **env_configs)
59
60 logging.info("Env specs: %s", env.observation_spec())
61 logging.info("Action specs: %s", env.action_spec())
62
63 time_step = env.reset()
64
65 while not time_step.step_type.last():
66 pid = time_step.observations["current_player"]
67 actions = select_actions(time_step.observations, pid)
68 print_iteration(time_step, actions, pid)
69 time_step = env.step(actions)
70
71 # Print final state of end game.
72 for pid in range(env.num_players):
73 print_iteration(time_step, actions, pid)
