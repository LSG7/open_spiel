98open_spiel::Action GetAction(const open_spiel::State& state,
99 const absl::string_view action_str) {
100 for (open_spiel::Action action : state.LegalActions()) {
101 if (action_str == state.ActionToString(state.CurrentPlayer(), action))
102 return action;
103 }
104 open_spiel::SpielFatalError(absl::StrCat("Illegal action: ", action_str));
105}

107std::pair<std::unique_ptr<algorithms::SearchNode>, std::unique_ptr<State>>
108SearchTicTacToeState(const absl::string_view initial_actions) {
109 auto game = LoadGame("tic_tac_toe");
110 std::unique_ptr<State> state = game->NewInitialState();
111 for (const auto& action_str : absl::StrSplit(initial_actions, ' ')) {
112 state->ApplyAction(GetAction(*state, action_str));
113 }
114 auto evaluator = std::make_shared<RandomRolloutEvaluator>(20, 42);
115 algorithms::MCTSBot bot(*game, evaluator, UCT_C,
116 /*max_simulations=*/ 10000,
117 /*max_memory_mb=*/ 10,
118 /*solve=*/ true,
119 /*seed=*/ 42,
120 /*verbose=*/ false);
121 return {bot.MCTSearch(*state), std::move(state)};
122}

124void MCTSTest_SolveDraw() {
125 auto [root, state] = SearchTicTacToeState("x(1,1) o(0,0) x(2,2)");
126 SPIEL_CHECK_EQ(state->ToString(), "o..\n.x.\n..x");
127 SPIEL_CHECK_EQ(root->outcome[root->player], 0);
128 for (const algorithms::SearchNode& c : root->children)
129 SPIEL_CHECK_LE(c.outcome[c.player], 0); // No winning moves.
130 const algorithms::SearchNode& best = root->BestChild();
131 SPIEL_CHECK_EQ(best.outcome[best.player], 0);
132 std::string action_str = state->ActionToString(best.player, best.action);
133 if (action_str != "o(2,0)" && action_str != "o(0,2)") // All others lose.
134 SPIEL_CHECK_EQ(action_str, "o(2,0)"); // "o(0,2)" is also valid.
135}

137void MCTSTest_SolveLoss() {
138 auto [root, state] =
139 SearchTicTacToeState("x(1,1) o(0,0) x(2,2) o(0,1) x(0,2)");
140 SPIEL_CHECK_EQ(state->ToString(), "oox\n.x.\n..x");
141 SPIEL_CHECK_EQ(root->outcome[root->player], -1);
142 for (const algorithms::SearchNode& c : root->children)
143 SPIEL_CHECK_EQ(c.outcome[c.player], -1); // All losses.
144}

146void MCTSTest_SolveWin() {
147 auto [root, state] = SearchTicTacToeState("x(0,1) o(2,2)");
148 SPIEL_CHECK_EQ(state->ToString(), ".x.\n...\n..o");
149 SPIEL_CHECK_EQ(root->outcome[root->player], 1);
150 const algorithms::SearchNode& best = root->BestChild();
151 SPIEL_CHECK_EQ(best.outcome[best.player], 1);
152 SPIEL_CHECK_EQ(state->ActionToString(best.player, best.action), "x(0,2)");
153}

155void MCTSTest_GarbageCollect() {

157 std::unique_ptr<State> state = game->NewInitialState();
158 auto evaluator = std::make_shared<RandomRolloutEvaluator>(1, 42);
