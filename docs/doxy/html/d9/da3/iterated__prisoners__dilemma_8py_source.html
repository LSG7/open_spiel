90class IteratedPrisonersDilemmaState(pyspiel.State):
91 """Current state of the game."""
92
93 def __init__(self, game, termination_probability):
94 """Constructor; should only be called by Game.new_initial_state."""
95 super().__init__(game)
96 self._current_iteration = 1
97 self._termination_probability = termination_probability
98 self._is_chance = False
99 self._game_over = False
100 self._rewards = np.zeros(_NUM_PLAYERS)
101 self._returns = np.zeros(_NUM_PLAYERS)
102
103 # OpenSpiel (PySpiel) API functions are below. This is the standard set that
104 # should be implemented by every simultaneous-move game with chance.
105
106 def current_player(self):
107 """Returns id of the next player to move, or TERMINAL if game is over."""
108 if self._game_over:
109 return pyspiel.PlayerId.TERMINAL
110 elif self._is_chance:
111 return pyspiel.PlayerId.CHANCE
112 else:
113 return pyspiel.PlayerId.SIMULTANEOUS
114
115 def _legal_actions(self, player):
116 """Returns a list of legal actions, sorted in ascending order."""
117 assert player >= 0
118 return [Action.COOPERATE, Action.DEFECT]
119
120 def chance_outcomes(self):
121 """Returns the possible chance outcomes and their probabilities."""
122 assert self._is_chance
123 return [(Chance.CONTINUE, 1 - self._termination_probability),
124 (Chance.STOP, self._termination_probability)]
125
126 def _apply_action(self, action):
127 """Applies the specified action to the state."""
128 # This is not called at simultaneous-move states.
129 assert self._is_chance and not self._game_over
130 self._current_iteration += 1
131 self._is_chance = False
132 self._game_over = (action == Chance.STOP)
133 if self._current_iteration > self.get_game().max_game_length():
134 self._game_over = True
135
136 def _apply_actions(self, actions):
137 """Applies the specified actions (per player) to the state."""
138 assert not self._is_chance and not self._game_over
139 self._is_chance = True
140 self._rewards[0] = _PAYOFF[actions[0]][actions[1]]
141 self._rewards[1] = _PAYOFF[actions[1]][actions[0]]
142 self._returns += self._rewards
143
144 def _action_to_string(self, player, action):
145 """Action -> string."""
146 if player == pyspiel.PlayerId.CHANCE:
147 return Chance(action).name
148 else:
149 return Action(action).name
150
151 def is_terminal(self):
152 """Returns True if the game is over."""
153 return self._game_over
154
155 def rewards(self):
156 """Reward at the previous step."""
157 return self._rewards
158
159 def returns(self):
160 """Total reward for each player over the course of the game so far."""
161 return self._returns
162
163 def __str__(self):
164 """String for debug purposes. No particular semantics are required."""
165 return (f"p0:{self.action_history_string(0)} "
166 f"p1:{self.action_history_string(1)}")
167
168 def action_history_string(self, player):
169 return "".join(
170 self._action_to_string(pa.player, pa.action)[0]
171 for pa in self.full_history()
172 if pa.player == player)
