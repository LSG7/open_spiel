84 TEST(CordzFunctionsTest, ShouldProfileRate) {
85  static constexpr int kDesiredMeanInterval = 1000;
86  static constexpr int kSamples = 10000;
87  int32_t orig_sample_rate = get_cordz_mean_interval();
88 
89  set_cordz_mean_interval(kDesiredMeanInterval);
90 
91  int64_t sum_of_intervals = 0;
92  for (int i = 0; i < kSamples; i++) {
93  // Setting next_sample to 0 will force cordz_should_profile to generate a
94  // new value for next_sample each iteration.
95  cordz_set_next_sample_for_testing(0);
96  cordz_should_profile();
97  sum_of_intervals += cordz_next_sample;
98  }
99 
100  // The sum of independent exponential variables is an Erlang distribution,
101  // which is a gamma distribution where the shape parameter is equal to the
102  // number of summands. The distribution used for cordz_should_profile is
103  // actually floor(Exponential(1/mean)) which introduces bias. However, we can
104  // apply the squint-really-hard correction factor. That is, when mean is
105  // large, then if we squint really hard the shape of the distribution between
106  // N and N+1 looks like a uniform distribution. On average, each value for
107  // next_sample will be about 0.5 lower than we would expect from an
108  // exponential distribution. This squint-really-hard correction approach won't
109  // work when mean is smaller than about 10 but works fine when mean is 1000.
110  //
111  // We can use R to calculate a confidence interval. This
112  // shows how to generate a confidence interval with a false positive rate of
113  // one in a billion.
114  //
115  // $ R -q
116  // > mean = 1000
117  // > kSamples = 10000
118  // > errorRate = 1e-9
119  // > correction = -kSamples / 2
120  // > low = qgamma(errorRate/2, kSamples, 1/mean) + correction
121  // > high = qgamma(1 - errorRate/2, kSamples, 1/mean) + correction
122  // > low
123  // [1] 9396115
124  // > high
125  // [1] 10618100
126  EXPECT_THAT(sum_of_intervals, Ge(9396115));
127  EXPECT_THAT(sum_of_intervals, Le(10618100));
128 
129  set_cordz_mean_interval(orig_sample_rate);
130 }
