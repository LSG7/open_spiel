15"""Reinforcement learning loss functions.
16
17All the loss functions implemented here compute the loss for the policy (actor).
18The critic loss functions are typically regression loss are omitted for their
19simplicity.
20
21For the batch QPG, RM and RPG loss, please refer to the paper:
22https://papers.nips.cc/paper/7602-actor-critic-policy-optimization-in-partially-observable-multiagent-environments.pdf
23
24The BatchA2C loss uses code from the `TRFL` library:
25https://github.com/deepmind/trfl/blob/master/trfl/discrete_policy_gradient_ops.py
26"""

34def _assert_rank_and_shape_compatibility(tensors, rank):
35 if not tensors:
36 raise ValueError("List of tensors cannot be empty")
37
38 union_of_shapes = tf.TensorShape(None)
39 for tensor in tensors:
40 tensor_shape = tensor.get_shape()
41 tensor_shape.assert_has_rank(rank)
42 union_of_shapes = union_of_shapes.merge_with(tensor_shape)

45def compute_baseline(policy, action_values):
46 # V = pi * Q, backprop through pi but not Q.
47 return tf.reduce_sum(
48 tf.multiply(policy, tf.stop_gradient(action_values)), axis=1)
49
50
51def compute_regrets(policy_logits, action_values):
52 """Compute regrets using pi and Q."""
53 # Compute regret.
54 policy = tf.nn.softmax(policy_logits, axis=1)
55 # Avoid computing gradients for action_values.
56 action_values = tf.stop_gradient(action_values)
57
58 baseline = compute_baseline(policy, action_values)
59
60 regrets = tf.reduce_sum(
61 tf.nn.relu(action_values - tf.expand_dims(baseline, 1)), axis=1)
62
63 return regrets

66def compute_advantages(policy_logits, action_values, use_relu=False):
67 """Compute advantages using pi and Q."""
68 # Compute advantage.
69 policy = tf.nn.softmax(policy_logits, axis=1)
70 # Avoid computing gradients for action_values.
71 action_values = tf.stop_gradient(action_values)
72
73 baseline = compute_baseline(policy, action_values)
74
75 advantages = action_values - tf.expand_dims(baseline, 1)
76 if use_relu:
77 advantages = tf.nn.relu(advantages)
78
79 # Compute advantage weighted by policy.
80 policy_advantages = -tf.multiply(policy, tf.stop_gradient(advantages))
81 return tf.reduce_sum(policy_advantages, axis=1)

84def compute_a2c_loss(policy_logits, actions, advantages):
85 cross_entropy = tf.nn.sparse_softmax_cross_entropy_with_logits(
86 labels=actions, logits=policy_logits)
87 advantages = tf.stop_gradient(advantages)
88 advantages.get_shape().assert_is_compatible_with(cross_entropy.get_shape())
89 return tf.multiply(cross_entropy, advantages)

92def compute_entropy(policy_logits):
93 return tf.reduce_sum(
94 -tf.nn.softmax(policy_logits) * tf.nn.log_softmax(policy_logits), axis=-1)

97def compute_entropy_loss(policy_logits):
98 """Compute an entropy loss.
99
100 We want a value that we can minimize along with other losses, and where
101 minimizing means driving the policy towards a uniform distribution over
102 the actions. We thus scale it by negative one so that it can be simply
103 added to other losses (and so it can be considered a bonus for having
104 entropy).
105
106 Args:
107 policy_logits: the policy logits.
108
109 Returns:
110 entropy loss (negative entropy).
111 """
112 entropy = compute_entropy(policy_logits)
113 scale = tf.constant(-1.0, dtype=tf.float32)
114 entropy_loss = tf.multiply(scale, entropy, name="entropy_loss")
115 return entropy_loss

118class BatchQPGLoss(object):
119 """Defines the batch QPG loss op."""
120
121 def __init__(self, entropy_cost=None, name="batch_qpg_loss"):
122 self._entropy_cost = entropy_cost
123 self._name = name
124
125 def loss(self, policy_logits, action_values):
126 """Constructs a TF graph that computes the QPG loss for batches.
127
128 Args:
129 policy_logits: `B x A` tensor corresponding to policy logits.
130 action_values: `B x A` tensor corresponding to Q-values.
131
132 Returns:
133 loss: A 0-D `float` tensor corresponding the loss.
134 """
135 _assert_rank_and_shape_compatibility([policy_logits, action_values], 2)
136 advantages = compute_advantages(policy_logits, action_values)
137 _assert_rank_and_shape_compatibility([advantages], 1)
138 total_adv = tf.reduce_mean(advantages, axis=0)
139
140 total_loss = total_adv
141 if self._entropy_cost:
142 entropy_loss = tf.reduce_mean(compute_entropy_loss(policy_logits))
143 scaled_entropy_loss = tf.multiply(
144 float(self._entropy_cost), entropy_loss, name="scaled_entropy_loss")
145 total_loss = tf.add(
146 total_loss, scaled_entropy_loss, name="total_loss_with_entropy")
147
148 return total_loss
