49 def test_matching_pennies_3p(self):
50 # We don't expect Deep CFR to necessarily converge on 3-player games but
51 # it's nonetheless interesting to see this result.
52 game = pyspiel.load_game_as_turn_based('matching_pennies_3p')
53 with tf.Session() as sess:
54 deep_cfr_solver = deep_cfr.DeepCFRSolver(
55 sess,
56 game,
57 policy_network_layers=(16, 8),
58 advantage_network_layers=(32, 16),
59 num_iterations=2,
60 num_traversals=2,
61 learning_rate=1e-3,
62 batch_size_advantage=None,
63 batch_size_strategy=None,
64 memory_capacity=1e7)
65 sess.run(tf.global_variables_initializer())
66 deep_cfr_solver.solve()
67 conv = exploitability.nash_conv(
68 game,
69 policy.tabular_policy_from_callable(
70 game, deep_cfr_solver.action_probabilities))
71 print('Deep CFR in Matching Pennies 3p. NashConv: {}'.format(conv))
