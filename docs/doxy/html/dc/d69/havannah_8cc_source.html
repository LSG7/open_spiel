60// The board is represented as a flattened 2d array of the form:
61// 1 2 3
62// a 0 1 2 0 1 0 1
63// b 3 4 5 => 3 4 5 => 3 4 5
64// c 6 7 8 7 8 7 8
65//
66// Neighbors are laid out in this pattern:
67// 0 1
68// 5 X 2
69// 4 3
70
71// Direct neighbors of a cell, clockwise.
72constexpr std::array<Move, kMaxNeighbors> neighbor_offsets = {
73 Move(-1, -1, kMoveOffset), Move(0, -1, kMoveOffset),
74 Move(1, 0, kMoveOffset), Move(1, 1, kMoveOffset),
75 Move(0, 1, kMoveOffset), Move(-1, 0, kMoveOffset),
76};
77
78// Precomputed list of neighbors per board_size: [board_size][cell][direction]
79std::vector<NeighborList> neighbor_list;
80
81NeighborList gen_neighbors(int board_size) {
82 int diameter = board_size * 2 - 1;
83 NeighborList out;
84 out.resize(diameter * diameter);
85 for (int y = 0; y < diameter; y++) {
86 for (int x = 0; x < diameter; x++) {
87 int xy = x + y * diameter; // Don't use Move.xy so it works off-board.
88 for (int dir = 0; dir < neighbor_offsets.size(); dir++) {
89 Move offset = neighbor_offsets[dir];
90 out[xy][dir] = Move(x + offset.x, y + offset.y, board_size);
91 }
92 }
93 }
94 return out;
95}
96
97const NeighborList& get_neighbors(int board_size) {
98 if (board_size >= neighbor_list.size()) {
99 neighbor_list.resize(board_size + 1);
100 }
101 if (neighbor_list[board_size].empty()) {
102 neighbor_list[board_size] = gen_neighbors(board_size);
103 }
104 return neighbor_list[board_size];
105}

117int Move::Corner(int board_size) const {
118 if (!OnBoard()) return 0;
119
120 int m = board_size - 1;
121 int e = m * 2;
122
123 if (x == 0 && y == 0) return 1 << 0;
124 if (x == m && y == 0) return 1 << 1;
125 if (x == e && y == m) return 1 << 2;
126 if (x == e && y == e) return 1 << 3;
127 if (x == m && y == e) return 1 << 4;
128 if (x == 0 && y == m) return 1 << 5;
129
130 return 0;
131}

133int Move::Edge(int board_size) const {
134 if (!OnBoard()) return 0;
135
136 int m = board_size - 1;
137 int e = m * 2;
138
139 if (y == 0 && x != 0 && x != m) return 1 << 0;
140 if (x - y == m && x != m && x != e) return 1 << 1;
141 if (x == e && y != m && y != e) return 1 << 2;
142 if (y == e && x != e && x != m) return 1 << 3;
143 if (y - x == m && x != m && x != 0) return 1 << 4;
144 if (x == 0 && y != m && y != 0) return 1 << 5;
145
146 return 0;
147}

158HavannahState::HavannahState(std::shared_ptr<const Game> game, int board_size,
159 bool ansi_color_output, bool allow_swap)
160 : State(game),
161 board_size_(board_size),
162 board_diameter_(board_size * 2 - 1),
163 valid_cells_((board_size * 2 - 1) * (board_size * 2 - 1) -
164 board_size * (board_size - 1)), // diameter^2 - corners
165 neighbors_(get_neighbors(board_size)),
166 ansi_color_output_(ansi_color_output),
167 allow_swap_(allow_swap) {
168 board_.resize(board_diameter_ * board_diameter_);
169 for (int i = 0; i < board_.size(); i++) {
170 Move m = ActionToMove(i);
171 board_[i] = Cell((m.OnBoard() ? kPlayerNone : kPlayerInvalid), i,
172 m.Corner(board_size), m.Edge(board_size));
173 }
174}

181std::vector<Action> HavannahState::LegalActions() const {
182 // Can move in any empty cell.
183 std::vector<Action> moves;
184 if (IsTerminal()) return {};
185 moves.reserve(board_.size() - moves_made_);
186 for (int cell = 0; cell < board_.size(); ++cell) {
187 if (board_[cell].player == kPlayerNone) {
188 moves.push_back(cell);
189 }
190 }
191 if (AllowSwap()) { // The second move is allowed to replace the first one.
192 moves.push_back(last_move_.xy);
193 absl::c_sort(moves);
194 }
195 return moves;
196}

207std::string HavannahState::ToString() const {
208 // Generates something like:
209 // a b c d e
210 // 1 @ O . @ O f
211 // 2 . O O @ O @ g
212 // 3 . @ @ . . @ O h
213 // 4 . @ @ . . . O O i
214 // 5 @ . . O . @ @ O .
215 // 6 @ O . O O @ @[O]
216 // 7 . @ O . O O O
217 // 8 @ O @ O O O
218 // 9 @ O @ @ @
219
220 std::string white = "O";
221 std::string black = "@";
222 std::string empty = ".";
223 std::string coord = "";
224 std::string reset = "";
225 if (ansi_color_output_) {
226 std::string esc = "\033";
227 reset = esc + "[0m";
228 coord = esc + "[1;37m"; // bright white
229 empty = reset + ".";
230 white = esc + "[1;33m" + "@"; // bright yellow
231 black = esc + "[1;34m" + "@"; // bright blue
232 }
233
234 std::ostringstream out;
235
236 // Top x coords.
237 out << std::string(board_size_ + 3, ' ');
238 for (int x = 0; x < board_size_; x++) {
239 out << ' ' << coord << static_cast<char>('a' + x);
240 }
241 out << '\n';
242
243 for (int y = 0; y < board_diameter_; y++) {
244 out << std::string(abs(board_size_ - 1 - y) + 1 + ((y + 1) < 10), ' ');
245 out << coord << (y + 1); // Leading y coord.
246
247 bool found_last = false;
248 int start_x = (y < board_size_ ? 0 : y - board_size_ + 1);
249 int end_x = (y < board_size_ ? board_size_ + y : board_diameter_);
250 for (int x = start_x; x < end_x; x++) {
251 Move pos(x, y, board_size_);
252
253 // Spacing and last-move highlight.
254 if (found_last) {
255 out << coord << ']';
256 found_last = false;
257 } else if (last_move_ == pos) {
258 out << coord << '[';
259 found_last = true;
260 } else {
261 out << ' ';
262 }
263
264 // Actual piece.
265 Player p = board_[pos.xy].player;
266 if (p == kPlayerNone) out << empty;
267 if (p == kPlayer1) out << white;
268 if (p == kPlayer2) out << black;
269 }
270 if (found_last) {
271 out << coord << ']';
272 }
273 if (y < board_size_ - 1) { // Trailing x coord.
274 out << ' ' << coord << static_cast<char>('a' + board_size_ + y);
275 }
276 out << '\n';
277 }
278 out << reset;
279 return out.str();
280}

328void HavannahState::DoApplyAction(Action action) {
329 SPIEL_CHECK_EQ(outcome_, kPlayerNone);
330
331 Move move = ActionToMove(action);
332 SPIEL_CHECK_TRUE(move.OnBoard());
333
334 if (last_move_ == move) {
335 SPIEL_CHECK_TRUE(AllowSwap());
336 } else {
337 SPIEL_CHECK_EQ(board_[move.xy].player, kPlayerNone);
338 moves_made_++;
339 last_move_ = move;
340 }
341 board_[move.xy].player = current_player_;
342
343 bool alreadyjoined = false; // Useful for finding rings.
344 bool skip = false;
345 for (const Move& m : neighbors_[move.xy]) {
346 if (skip) {
347 skip = false;
348 } else if (m.OnBoard()) {
349 if (current_player_ == board_[m.xy].player) {
350 alreadyjoined |= JoinGroups(move.xy, m.xy);
351
352 // Skip the next one. If it is the same group, it is already connected
353 // and forms a sharp corner, which we can ignore.
354 skip = true;
355 }
356 }
357 }
358
359 const Cell& group = board_[FindGroupLeader(move.xy)];
360 if (group.NumEdges() >= 3 || group.NumCorners() >= 2 ||
361 (alreadyjoined && CheckRingDFS(move, 0, 3))) {
362 outcome_ = current_player_;
363 } else if (moves_made_ == valid_cells_) {
364 outcome_ = kPlayerDraw;
365 }
366
367 current_player_ = (current_player_ == kPlayer1 ? kPlayer2 : kPlayer1);
368}

370int HavannahState::FindGroupLeader(int cell) {
371 int parent = board_[cell].parent;
372 if (parent != cell) {
373 do { // Follow the parent chain up to the group leader.
374 parent = board_[parent].parent;
375 } while (parent != board_[parent].parent);
376 // Do path compression, but only the current one to avoid recursion.
377 board_[cell].parent = parent;
378 }
379 return parent;
380}

382bool HavannahState::JoinGroups(int cell_a, int cell_b) {
383 int leader_a = FindGroupLeader(cell_a);
384 int leader_b = FindGroupLeader(cell_b);
385
386 if (leader_a == leader_b) // Already the same group.
387 return true;
388
389 if (board_[leader_a].size < board_[leader_b].size) {
390 // Force group a's subtree to be bigger.
391 std::swap(leader_a, leader_b);
392 }
393
394 // Group b joins group a.
395 board_[leader_b].parent = leader_a;
396 board_[leader_a].size += board_[leader_b].size;
397 board_[leader_a].corner |= board_[leader_b].corner;
398 board_[leader_a].edge |= board_[leader_b].edge;
399
400 return false;
401}

403bool HavannahState::CheckRingDFS(const Move& move, int left, int right) {
404 if (!move.OnBoard()) return false;
405
406 Cell& c = board_[move.xy];
407 if (current_player_ != c.player) return false;
408 if (c.mark) return true; // Found a ring!
409
410 c.mark = true;
411 bool success = false;
412 for (int i = left; !success && i <= right; i++) {
413 int dir = (i + 6) % 6; // Normalize.
414 success = CheckRingDFS(neighbors_[move.xy][dir], dir - 1, dir + 1);
415 }
416 c.mark = false;
417 return success;
418}
