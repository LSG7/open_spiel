69 void BlottoState::DoApplyActions(const std::vector<Action>& actions) {
70  joint_action_ = actions;
71 
72  // Now determine returns.
73  returns_.resize(num_players_, 0);
74  std::vector<int> scores(num_players_, 0);
75  std::vector<std::vector<int>> player_actions;
76 
77  for (int f = 0; f < fields_; ++f) {
78  int num_winners = 0;
79  int winner = 0;
80  int max_value = -1;
81 
82  for (auto p = Player{0}; p < num_players_; ++p) {
83  // Get the expanded action if necessary.
84  if (p >= player_actions.size()) {
85  player_actions.push_back(action_map_->at(joint_action_[p]));
86  }
87 
88  if (player_actions[p][f] > max_value) {
89  num_winners = 1;
90  winner = p;
91  max_value = player_actions[p][f];
92  } else if (player_actions[p][f] == max_value) {
93  num_winners++;
94  }
95  }
96 
97  // Give the winner of this field one point. Draw if tied.
98  if (num_winners == 1) {
99  scores[winner]++;
100  }
101  }
102 
103  // Find the global winner(s).
104  std::set<int> winners;
105  int max_points = 0;
106  for (auto p = Player{0}; p < num_players_; ++p) {
107  if (scores[p] > max_points) {
108  max_points = scores[p];
109  winners = {p};
110  } else if (scores[p] == max_points) {
111  winners.insert(p);
112  }
113  }
114 
115  // Finally, assign returns. Each winner gets 1/num_winners, each loser gets
116  // -1 / num_losers.
117  for (auto p = Player{0}; p < num_players_; ++p) {
118  if (winners.size() == num_players_) {
119  // All players won same number of fields. Draw.
120  returns_[p] = 0;
121  } else if (winners.find(p) != winners.end()) {
122  SPIEL_CHECK_GE(winners.size(), 1);
123  returns_[p] = 1.0 / winners.size();
124  } else {
125  SPIEL_CHECK_GE(num_players_ - winners.size(), 1);
126  returns_[p] = -1.0 / (num_players_ - winners.size());
127  }
128  }
129 }

166 void BlottoGame::CreateActionMapRec(int* count, int coins_left,
167  const std::vector<int>& action) {
168  if (action.size() == fields_) {
169  if (coins_left == 0) {
170  // All coins used, valid move.
171  (*action_map_)[*count] = action;
172  (*count)++;
173  return;
174  } else {
175  // Not all coins used, invalid move.
176  return;
177  }
178  } else {
179  for (int num_coins = 0; num_coins <= coins_left; ++num_coins) {
180  std::vector<int> new_action = action;
181  new_action.push_back(num_coins);
182  CreateActionMapRec(count, coins_left - num_coins, new_action);
183  }
184  }
185 }
