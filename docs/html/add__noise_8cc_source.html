74 double AddNoiseGame::GetNoise(const AddNoiseState& state) {
75  std::string state_str = state.HistoryString();
76  auto it = noise_table_.find(state_str);
77  if (it != noise_table_.end()) {
78  return it->second;
79  }
80 
81  std::uniform_real_distribution<double> dist(-epsilon_, epsilon_);
82  double noise = dist(rng_);
83  noise_table_[state_str] = noise;
84  return noise;
85 }

99 std::vector<double> AddNoiseState::Returns() const {
100  std::vector<double> returns = state_->Returns();
101  SPIEL_CHECK_EQ(returns.size(), 2);
102 
103  if (state_->IsTerminal()) {
104  auto const_noise_game = down_cast<const AddNoiseGame*>(game_.get());
105  AddNoiseGame* noise_game = const_cast<AddNoiseGame*>(const_noise_game);
106  double noise = noise_game->GetNoise(*this);
107  returns[0] += noise;
108  returns[1] -= noise;
109  }
110 
111  return returns;
112 }
