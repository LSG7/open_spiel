20class PolicyBot(pyspiel.Bot):
21 """Samples an action from action probabilities based on a policy.
22
23 This bot plays actions as specified by the underlying Policy. Problems may
24 occur if the policy assigns non-zero probability to invalid actions, or if the
25 policy is not complete, or if probabilities don't sum to 1.
26 """

28 def __init__(self, player_id, rng, policy):
29 """Initializes a policy bot.
30
31 Args:
32 player_id: The integer id of the player for this bot, e.g. `0` if acting
33 as the first player.
34 rng: A random number generator supporting a `choice` method, e.g.
35 `np.random`
36 policy: A policy to get action distributions
37 """
38 pyspiel.Bot.__init__(self)
39 self._player_id = player_id

49 def step_with_policy(self, state):
50 """Returns the stochastic policy and selected action in the given state.
51
52 Args:
53 state: The current state of the game.
54
55 Returns:
56 A `(policy, action)` pair, where policy is a `list` of
57 `(action, probability)` pairs for each legal action, with
58 `probability` defined by the policy action probabilities.
59 The `action` is sampled from the distribution,
60 or `pyspiel.INVALID_ACTION` if there are no actions available.
61 """
62 policy = self._policy.action_probabilities(state, self._player_id)
63 action_list = list(policy.keys())
64 if not any(action_list):
65 return [], pyspiel.INVALID_ACTION
66
67 action = self._rng.choice(action_list, p=list(policy.values()))
68 return list(policy.items()), action
69
70 def step(self, state):
71 return self.step_with_policy(state)[1]
