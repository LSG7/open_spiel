77std::string GameParameter::Serialize(const std::string& delimiter) const {
78 std::string val;
79 switch (type_) {
80 case Type::kString:
81 val = absl::StrReplaceAll(ToString(), {{"\n", kSerializedNewline}});
82 break;
83 case Type::kGame:
84 val = SerializeGameParameters(game_value());
85 break;
86 default:
87 val = ToString();
88 }
89 return absl::StrCat(GameParameterTypeToString(type_), delimiter, val,
90 delimiter, is_mandatory() ? "true" : "false");

93GameParameter DeserializeGameParameter(const std::string& data,
94 const std::string& delimiter) {
95 std::vector<std::string> parts = absl::StrSplit(data, delimiter);
96 SPIEL_CHECK_EQ(parts.size(), 3);
97 bool mandatory = (parts[2] == "True" || parts[2] == "true");
98 if (parts[0] == "kUnset") {
99 return GameParameter(GameParameter::Type::kUnset, mandatory);
100 } else if (parts[0] == "kInt") {
101 int value;
102 SPIEL_CHECK_TRUE(absl::SimpleAtoi(parts[1], &value));
103 return GameParameter(value, mandatory);
104 } else if (parts[0] == "kDouble") {
105 double value;
106 SPIEL_CHECK_TRUE(absl::SimpleAtod(parts[1], &value));
107 return GameParameter(value, mandatory);
108 } else if (parts[0] == "kString") {
109 return GameParameter(
110 absl::StrReplaceAll(parts[1], {{kSerializedNewline, "\n"}}), mandatory);
111 } else if (parts[0] == "kBool") {
112 return GameParameter(parts[1] == "True" || parts[1] == "true", mandatory);
113 } else if (parts[0] == "kGame") {
114 return GameParameter(DeserializeGameParameters(parts[1]), mandatory);
115 } else {
116 SpielFatalError(absl::StrCat("Unrecognized type: ", parts[0]));
117 }
118}

120std::string SerializeGameParameters(const GameParameters& game_params,
121 const std::string& name_delimiter,
122 const std::string& parameter_delimeter) {
123 std::list<std::string> serialized_params;
124
125 for (const auto& key_val : game_params) {
126 std::string name = key_val.first;
127 GameParameter parameter = key_val.second;
128
129 serialized_params.push_back(
130 absl::StrCat(name, name_delimiter, parameter.Serialize()));
131 }
132
133 return absl::StrJoin(serialized_params, parameter_delimeter);
134}

136GameParameters DeserializeGameParameters(
137 const std::string& data, const std::string& name_delimiter,
138 const std::string& parameter_delimeter) {
139 GameParameters game_params;
140 std::vector<std::string> parts = absl::StrSplit(data, parameter_delimeter);
141
142 for (const auto& part : parts) {
143 if (!part.empty()) {
144 std::pair<std::string, std::string> pair =
145 absl::StrSplit(part, name_delimiter);
146 game_params.insert(std::pair<std::string, GameParameter>(
147 pair.first, DeserializeGameParameter(pair.second)));
148 }
149 }
150 return game_params;
151}

153std::string GameParametersToString(const GameParameters& game_params) {
154 std::string str;
155 if (game_params.empty()) return "";
156 if (game_params.count("name")) str = game_params.at("name").string_value();
157 str.push_back('(');
158 bool first = true;
159 for (const auto& key_val : game_params) {
160 if (key_val.first != "name") {
161 if (!first) str.push_back(',');
162 str.append(key_val.first);
163 str.append("=");
164 str.append(key_val.second.ToString());
165 first = false;
166 }
167 }
168 str.push_back(')');
169 return str;
170}

172GameParameter GameParameterFromString(const std::string& str) {
173 if (str == "True" || str == "true") {
174 return GameParameter(true);
175 } else if (str == "False" || str == "false") {
176 return GameParameter(false);
177 } else if (str.find_first_not_of("+-0123456789") == std::string::npos) {
178 int value;
179 bool success = absl::SimpleAtoi(str, &value);
180 SPIEL_CHECK_TRUE(success);
181 return GameParameter(value);
182 } else if (str.find_first_not_of("+-0123456789.") == std::string::npos) {
183 double value;
184 bool success = absl::SimpleAtod(str, &value);
185 SPIEL_CHECK_TRUE(success);
186 return GameParameter(value);
187 } else if (str.back() == ')') {
188 return GameParameter(GameParametersFromString(str));
189 } else {
190 return GameParameter(str);
191 }
192}

194GameParameters GameParametersFromString(const std::string& game_string) {
195 GameParameters params;
196 if (game_string.empty()) return params;
197 int first_paren = game_string.find('(');
198 if (first_paren == std::string::npos) {
199 params["name"] = GameParameter(game_string);
200 return params;
201 }
202 params["name"] = GameParameter(game_string.substr(0, first_paren));
203 int start = first_paren + 1;
204 int parens = 1;
205 int equals = -1;
206 for (int i = start; i < game_string.length(); ++i) {
207 if (game_string[i] == '(') {
208 ++parens;
209 } else if (game_string[i] == ')') {
210 --parens;
211 } else if (game_string[i] == '=' && parens == 1) {
212 equals = i;
213 }
214 if ((game_string[i] == ',' && parens == 1) ||
215 (game_string[i] == ')' && parens == 0 && i > start + 1)) {
216 params[game_string.substr(start, equals - start)] =
217 GameParameterFromString(
218 game_string.substr(equals + 1, i - equals - 1));
219 start = i + 1;
220 equals = -1;
221 }
222 }
223 if (parens > 0) SpielFatalError("Missing closing bracket ')'.");
224 return params;
225}
