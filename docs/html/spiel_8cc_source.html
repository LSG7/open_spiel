51 // Returns the available parameter keys, to be used as a utility function.
52 std::string ListValidParameters(
53  const GameParameters& param_spec) {
54  std::vector<std::string> available_keys;
55  available_keys.reserve(param_spec.size());
56  for (const auto& item : param_spec) {
57  available_keys.push_back(item.first);
58  }
59  std::sort(available_keys.begin(), available_keys.end());
60  return absl::StrJoin(available_keys, ", ");

63 // Check on supplied parameters for game creation.
64 // Issues a SpielFatalError if any are missing, of the wrong type, or
65 // unexpectedly present.
66 void ValidateParams(const GameParameters& params,
67  const GameParameters& param_spec) {
68  // Check all supplied parameters are supported and of the right type.
69  for (const auto& param : params) {
70  const auto it = param_spec.find(param.first);
71  if (it == param_spec.end()) {
72  SpielFatalError(absl::StrCat(
73  "Unknown parameter '", param.first,
74  "'. Available parameters are: ", ListValidParameters(param_spec)));
75  }
76  if (it->second.type() != param.second.type()) {
77  SpielFatalError(absl::StrCat(
78  "Wrong type for parameter ", param.first,
79  ". Expected type: ", GameParameterTypeToString(it->second.type()),
80  ", got ", GameParameterTypeToString(param.second.type()), " with ",
81  param.second.ToString()));
82  }
83  }
84  // Check we aren't missing any mandatory parameters.
85  for (const auto& param : param_spec) {
86  if (param.second.is_mandatory() && !params.count(param.first)) {
87  SpielFatalError(absl::StrCat("Missing parameter ", param.first));
88  }
89  }
90 }

192 std::shared_ptr<const Game> DeserializeGame(const std::string& serialized) {
193  std::pair<std::string, std::string> game_and_rng_state =
194  absl::StrSplit(serialized, kSerializeGameRNGStateSectionHeader);
195 
196  // Remove the trailing "\n" from the game section.
197  if (game_and_rng_state.first.length() > 0 &&
198  game_and_rng_state.first.back() == '\n') {
199  game_and_rng_state.first.pop_back();
200  }
201  std::shared_ptr<const Game> game = LoadGame(game_and_rng_state.first);
202 
203  if (game_and_rng_state.second.length() > 0) {
204  // Game is implicitly stochastic.
205  // Remove the trailing "\n" from the RNG state section.
206  if (game_and_rng_state.second.back() == '\n') {
207  game_and_rng_state.second.pop_back();
208  }
209  game->SetRNGState(game_and_rng_state.second);
210  }
211  return game;
212 }

251 void NormalizePolicy(ActionsAndProbs* policy) {
252  const double sum = absl::c_accumulate(
253  *policy, 0.0, [](double a, auto& b) { return a + b.second; });
254  absl::c_for_each(*policy, [sum](auto& o) { o.second /= sum; });
255 }

257 std::pair<Action, double> SampleAction(const ActionsAndProbs& outcomes,
258  absl::BitGenRef rng) {
259  return SampleAction(outcomes, absl::Uniform(rng, 0.0, 1.0));
260 }
261 std::pair<Action, double> SampleAction(const ActionsAndProbs& outcomes,
262  double z) {
263  SPIEL_CHECK_GE(z, 0);
264  SPIEL_CHECK_LT(z, 1);
265 
266  // Special case for one-item lists.
267  if (outcomes.size() == 1) {
268  SPIEL_CHECK_FLOAT_EQ(outcomes[0].second, 1.0);
269  return outcomes[0];
270  }
271 
272  // First do a check that this is indeed a proper discrete distribution.
273  double sum = 0;
274  for (const std::pair<Action, double>& outcome : outcomes) {
275  double prob = outcome.second;
276  SPIEL_CHECK_PROB(prob);
277  sum += prob;
278  }
279  SPIEL_CHECK_FLOAT_EQ(sum, 1.0);
280 
281  // Now sample an outcome.
282  sum = 0;
283  for (const std::pair<Action, double>& outcome : outcomes) {
284  double prob = outcome.second;
285  if (sum <= z && z < (sum + prob)) {
286  return outcome;
287  }
288  sum += prob;
289  }
290 
291  // If we get here, something has gone wrong
292  std::cerr << "Chance sampling failed; outcomes:" << std::endl;
293  for (const std::pair<Action, double>& outcome : outcomes) {
294  std::cerr << outcome.first << " " << outcome.second << std::endl;
295  }
296  SpielFatalError(
297  absl::StrCat("Internal error: failed to sample an outcome; z=", z));
298 }

300 std::string State::Serialize() const {
301  // This simple serialization doesn't work for the following games:
302  // - games with sampled chance nodes, since the history doesn't give us enough
303  // information to reconstruct the state.
304  // - Mean field games, since this base class does not store the history of
305  // state distributions passed in UpdateDistribution() (and it would be
306  // very expensive to do so for games with many possible states and a long
307  // time horizon).
308  // If you wish to serialize states in such games, you must implement custom
309  // serialization and deserialization for the state.
310  SPIEL_CHECK_NE(game_->GetType().chance_mode,
311  GameType::ChanceMode::kSampledStochastic);
312  SPIEL_CHECK_NE(game_->GetType().dynamics, GameType::Dynamics::kMeanField);
313  return absl::StrCat(absl::StrJoin(History(), "\n"), "\n");
314 }

349 void State::ApplyActions(const std::vector<Action>& actions) {
350  // history_ needs to be modified *after* DoApplyActions which could
351  // be using it.
352  DoApplyActions(actions);
353  history_.reserve(history_.size() + actions.size());
354  for (int player = 0; player < actions.size(); ++player) {
355  history_.push_back({player, actions[player]});
356  }
357  ++move_number_;
358 }

360 void State::ApplyActionsWithLegalityChecks(const std::vector<Action>& actions) {
361  for (Player player = 0; player < actions.size(); ++player) {
362  std::vector<Action> legal_actions = LegalActions(player);
363  if (!legal_actions.empty() &&
364  absl::c_find(legal_actions, actions[player]) == legal_actions.end()) {
365  SpielFatalError(
366  absl::StrCat("Player ", player, " calling ApplyAction ",
367  "with illegal action (", actions[player], "): ",
368  ActionToString(player, actions[player])));
369  }
370  }
371  ApplyActions(actions);

396 std::unique_ptr<State> Game::DeserializeState(const std::string& str) const {
397  // This does not work for games with sampled chance nodes and for mean field
398  // games. See comments in State::Serialize() for the explanation. If you wish
399  // to serialize states in such games, you must implement custom serialization
400  // and deserialization for the state.
401  SPIEL_CHECK_NE(game_type_.chance_mode,
402  GameType::ChanceMode::kSampledStochastic);
403  SPIEL_CHECK_NE(game_type_.dynamics,
404  GameType::Dynamics::kMeanField);
405 
406  std::unique_ptr<State> state = NewInitialState();
407  if (str.length() == 0) {
408  return state;
409  }
410  std::vector<std::string> lines = absl::StrSplit(str, '\n');
411  for (int i = 0; i < lines.size(); ++i) {
412  if (lines[i].empty()) continue;
413  if (state->IsSimultaneousNode()) {
414  std::vector<Action> actions;
415  for (int p = 0; p < state->NumPlayers(); ++p, ++i) {
416  SPIEL_CHECK_LT(i, lines.size());
417  Action action = static_cast<Action>(std::stol(lines[i]));
418  actions.push_back(action);
419  }
420  state->ApplyActions(actions);
421  // Must decrement i here, otherwise it is incremented too many times.
422  --i;
423  } else {
424  Action action = static_cast<Action>(std::stol(lines[i]));
425  state->ApplyAction(action);
426  }
427  }
428  return state;
429 }

453 std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>
454 DeserializeGameAndState(const std::string& serialized_state) {
455  std::vector<std::string> lines = absl::StrSplit(serialized_state, '\n');
456 
457  enum Section { kInvalid = -1, kMeta = 0, kGame = 1, kState = 2 };
458  std::vector<std::string> section_strings = {"", "", ""};
459  Section cur_section = kInvalid;
460 
461  for (int i = 0; i < lines.size(); ++i) {
462  if (lines[i].length() == 0 || lines[i].at(0) == '#') {
463  // Skip comments and blank lines.
464  } else if (lines[i] == kSerializeMetaSectionHeader) {
465  SPIEL_CHECK_EQ(cur_section, kInvalid);
466  cur_section = kMeta;
467  } else if (lines[i] == kSerializeGameSectionHeader) {
468  SPIEL_CHECK_EQ(cur_section, kMeta);
469  cur_section = kGame;
470  } else if (lines[i] == kSerializeStateSectionHeader) {
471  SPIEL_CHECK_EQ(cur_section, kGame);
472  cur_section = kState;
473  } else {
474  SPIEL_CHECK_NE(cur_section, kInvalid);
475  absl::StrAppend(&section_strings[cur_section], lines[i], "\n");
476  }
477  }
478 
479  // Remove the trailing "\n" from the game and state sections.
480  if (section_strings[kGame].length() > 0 &&
481  section_strings[kGame].back() == '\n') {
482  section_strings[kGame].pop_back();
483  }
484  if (section_strings[kState].length() > 0 &&
485  section_strings[kState].back() == '\n') {
486  section_strings[kState].pop_back();
487  }
488 
489  // We currently just ignore the meta section.
490  std::shared_ptr<const Game> game = DeserializeGame(section_strings[kGame]);
491  std::unique_ptr<State> state =
492  game->DeserializeState(section_strings[kState]);
493 
494  return std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>(
495  game, std::move(state));
496 }

837 std::pair<std::shared_ptr<const Game>,
838  std::unique_ptr<State>> BuildStateFromHistoryString(
839  const std::string& game_string,
840  const std::string& history,
841  int max_steps) {
842  std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>> game_and_state;
843  game_and_state.first = LoadGame(game_string);
844  game_and_state.second = game_and_state.first->NewInitialState();
845  std::string history_copy(absl::StripAsciiWhitespace(history));
846  if (history_copy[0] == '[') {
847  history_copy = history_copy.substr(1);
848  }
849  if (history_copy[history_copy.length() - 1] == ']') {
850  history_copy = history_copy.substr(0, history_copy.length() - 1);
851  }
852 
853  std::vector<Action> legal_actions;
854  State* state = game_and_state.second.get();
855  int steps = 0;
856  std::vector<std::string> parts = absl::StrSplit(history_copy, ',');
857  for (const std::string& part : parts) {
858  if (max_steps > 0 && steps >= max_steps) {
859  break;
860  }
861  Action action;
862  bool atoi_ret = absl::SimpleAtoi(absl::StripAsciiWhitespace(part), &action);
863  if (!atoi_ret) {
864  SpielFatalError(absl::StrCat("Problem parsing action: ", part));
865  }
866  legal_actions = state->LegalActions();
867  if (absl::c_find(legal_actions, action) == legal_actions.end()) {
868  SpielFatalError(absl::StrCat("Illegal move detected!\nState:\n",
869  state->ToString(), "\nAction: ", action,
870  " (", state->ActionToString(action), ")\n",
871  "History: ", state->HistoryString()));
872  }
873  state->ApplyAction(action);
874  steps++;
875  }
876 
877  return game_and_state;
878 }
