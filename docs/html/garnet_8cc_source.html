67 std::string StateToString(int x, int t, Action last_action, Player player_id,
68  bool is_chance_init) {
69  if (is_chance_init) {
70  return "initial";
71  } else if (player_id == 0) {
72  return absl::Substitute("($0, $1)", x, t);
73  } else if (player_id == kMeanFieldPlayerId) {
74  return absl::Substitute("($0, $1)_a", x, t);
75  } else if (player_id == kChancePlayerId) {
76  return absl::Substitute("($0, $1, $2)_a_mu", x, t, last_action);
77  } else {
78  SpielFatalError(
79  absl::Substitute("Unexpected state (player_id: $0, is_chance_init: $1)",
80  player_id, is_chance_init));
81  }
82 }

90 GarnetState::GarnetState(std::shared_ptr<const Game> game, int size,
91  int horizon, int seed, int num_action,
92  int num_chance_action, double sparsity_factor,
93  double eta)
94  : State(game),
95  size_(size),
96  horizon_(horizon),
97  seed_(seed),
98  num_action_(num_action),
99  num_chance_action_(num_chance_action),
100  sparsity_factor_(sparsity_factor),
101  eta_(eta),
102  distribution_(size_, 1. / size_) {
103  std::mt19937 rng(seed_);
104  double normalization;
105  double proba;
106  double cdf_proba;
107  for (int i = 0; i < size; ++i) {
108  for (int j = 0; j < num_action; ++j) {
109  double r_sparse = absl::Uniform<double>(rng, 0.0, 1.0);
110  if (r_sparse < sparsity_factor_) {
111  garnet_reward_.push_back(absl::Uniform<double>(rng, 0.0, 1.0));
112  } else {
113  garnet_reward_.push_back(0.0);
114  }
115 
116  normalization = 0;
117  std::vector<double> cdf;
118  cdf.push_back(0.0);
119  cdf.push_back(1.0);
120  for (int kk = 0; kk < num_chance_action - 1; ++kk) {
121  cdf_proba = absl::Uniform<double>(rng, 0.0, 1.0);
122  cdf.push_back(cdf_proba);
123  }
124  std::sort(cdf.begin(), cdf.end());
125  for (int k = 0; k < num_chance_action; ++k) {
126  proba = cdf[k+1]-cdf[k];
127  normalization += proba;
128  garnet_transition_proba_unnormalized_.push_back(proba);
129  garnet_transition_.push_back(absl::Uniform<int>(rng, 0, size));
130  }
131  garnet_transition_proba_normalization_.push_back(normalization);
132  }
133  }
134 }

197 std::vector<Action> GarnetState::LegalActions() const {
198  if (IsTerminal()) return {};
199  if (IsChanceNode()) return LegalChanceOutcomes();
200  if (IsMeanFieldNode()) return {};
201  SPIEL_CHECK_TRUE(IsPlayerNode());
202  std::vector<Action> outcomes;
203  outcomes.reserve(num_action_);
204  for (int i = 0; i < num_action_; ++i) {
205  outcomes.push_back(i);
206  }
207  return outcomes;

210 ActionsAndProbs GarnetState::ChanceOutcomes() const {
211  if (is_chance_init_) {
212  ActionsAndProbs outcomes;
213  for (int i = 0; i < size_; ++i) {
214  outcomes.push_back({i, 1. / size_});
215  }
216  return outcomes;
217  }
218  ActionsAndProbs outcomes;
219  double proba;
220  for (int i = 0; i < num_chance_action_; ++i) {
221  proba = GetTransitionProba(x_, last_action_, i);
222  outcomes.push_back({i, proba});
223  }
224  return outcomes;
225 }

227 void GarnetState::DoApplyAction(Action action) {
228  SPIEL_CHECK_NE(current_player_, kMeanFieldPlayerId);
229  return_value_ += Rewards()[0];
230  if (is_chance_init_) {
231  SPIEL_CHECK_GE(action, 0);
232  SPIEL_CHECK_LT(action, size_);
233  SPIEL_CHECK_EQ(current_player_, kChancePlayerId);
234  x_ = action;
235  is_chance_init_ = false;
236  current_player_ = 0;
237  } else if (current_player_ == kChancePlayerId) {
238  x_ = GetTransition(x_, last_action_, action);
239  ++t_;
240  current_player_ = kMeanFieldPlayerId;
241  } else {
242  SPIEL_CHECK_EQ(current_player_, 0);
243  last_action_ = action;
244  current_player_ = kChancePlayerId;
245  }
246 }

274 std::vector<double> GarnetState::Rewards() const {
275  if (current_player_ != 0) {
276  return {0.};
277  }
278  double r_x = GetReward(x_, last_action_);
279  double r_mu = -std::log(distribution_[x_] + kEpsilon);
280  return {r_x + eta_ * r_mu};

303 void GarnetState::ObservationTensor(Player player,
304  absl::Span<float> values) const {
305  SPIEL_CHECK_GE(player, 0);
306  SPIEL_CHECK_LT(player, num_players_);
307  SPIEL_CHECK_EQ(values.size(), size_ + horizon_ + 1);
308  SPIEL_CHECK_LT(x_, size_);
309  SPIEL_CHECK_GE(t_, 0);
310  // Allow t_ == horizon_.
311  SPIEL_CHECK_LE(t_, horizon_);
312  std::fill(values.begin(), values.end(), 0.);
313  if (x_ >= 0) {
314  values[x_] = 1.;
315  }
316  // x_ equals -1 for the initial (blank) state, don't set any
317  // position bit in that case.
318  values[size_ + t_] = 1.;
319 }

350 std::unique_ptr<State> GarnetGame::DeserializeState(
351  const std::string& str) const {
352  std::vector<std::string> lines = absl::StrSplit(str, '\n');
353  if (lines.size() != 2) {
354  SpielFatalError(absl::StrCat("Expected 2 lines in serialized state, got: ",
355  lines.size()));
356  }
357  Player current_player;
358  int is_chance_init;
359  int x;
360  int t;
361  int last_action;
362  double return_value;
363  std::vector<std::string> properties = absl::StrSplit(lines[0], ',');
364  if (properties.size() != 6) {
365  SpielFatalError(
366  absl::StrCat("Expected 6 properties for serialized state, got: ",
367  properties.size()));
368  }
369  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[0], &current_player));
370  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[1], &is_chance_init));
371  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[2], &x));
372  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[3], &t));
373  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[4], &last_action));
374  SPIEL_CHECK_TRUE(absl::SimpleAtod(properties[5], &return_value));
375  std::vector<std::string> serialized_distrib = absl::StrSplit(lines[1], ',');
376  std::vector<double> distribution;
377  distribution.reserve(serialized_distrib.size());
378  for (std::string& v : serialized_distrib) {
379  double parsed_weight;
380  SPIEL_CHECK_TRUE(absl::SimpleAtod(v, &parsed_weight));
381  distribution.push_back(parsed_weight);
382  }
383  return absl::make_unique<GarnetState>(
384  shared_from_this(), size_, horizon_, seed_, num_action_,
385  num_chance_action_, sparsity_factor_, eta_, current_player,
386  is_chance_init, x, t, last_action, return_value, distribution);
387 }
