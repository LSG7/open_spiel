78 // Modifies a given device string to one that can be accepted by the
79 // Torch library.
80 //
81 // The Torch library accepts 'cpu', 'cpu:0', 'cuda:0', 'cuda:1',
82 // 'cuda:2', 'cuda:3'..., but complains when there's a slash in front
83 // of the device name.
84 //
85 // Currently, this function only disregards a slash if it exists at the
86 // beginning of the device string, more functionality can be added if
87 // needed.
88 std::string TorchDeviceName(const std::string& device) {
89  if (device[0] == '/') {
90  return device.substr(1);
91  }
92  return device;
93 }

151 std::vector<VPNetModel::InferenceOutputs> VPNetModel::Inference(
152  const std::vector<InferenceInputs>& inputs) {
153  int inference_batch_size = inputs.size();
154 
155  // Torch tensors by default use a dense, row-aligned memory layout.
156  // - Their default data type is a 32-bit float
157  // - Use the byte data type for boolean
158 
159  torch::Tensor torch_inf_inputs =
160  torch::empty({inference_batch_size, flat_input_size_}, torch_device_);
161  torch::Tensor torch_inf_legal_mask = torch::full(
162  {inference_batch_size, num_actions_}, false,
163  torch::TensorOptions().dtype(torch::kByte).device(torch_device_));
164 
165  for (int batch = 0; batch < inference_batch_size; ++batch) {
166  // Copy legal mask(s) to a Torch tensor.
167  for (Action action : inputs[batch].legal_actions) {
168  torch_inf_legal_mask[batch][action] = true;
169  }
170 
171  // Copy the observation(s) to a Torch tensor.
172  for (int i = 0; i < inputs[batch].observations.size(); ++i) {
173  torch_inf_inputs[batch][i] = inputs[batch].observations[i];
174  }
175  }
176 
177  // Run the inference.
178  model_->eval();
179  std::vector<torch::Tensor> torch_outputs =
180  model_(torch_inf_inputs, torch_inf_legal_mask);
181 
182  torch::Tensor value_batch = torch_outputs[0];
183  torch::Tensor policy_batch = torch_outputs[1];
184 
185  // Copy the Torch tensor output to the appropriate structure.
186  std::vector<InferenceOutputs> output;
187  output.reserve(inference_batch_size);
188  for (int batch = 0; batch < inference_batch_size; ++batch) {
189  double value = value_batch[batch].item<double>();
190 
191  ActionsAndProbs state_policy;
192  state_policy.reserve(inputs[batch].legal_actions.size());
193  for (Action action : inputs[batch].legal_actions) {
194  state_policy.push_back(
195  {action, policy_batch[batch][action].item<float>()});
196  }
197 
198  output.push_back({value, state_policy});
199  }
200 
201  return output;
202 }

204 VPNetModel::LossInfo VPNetModel::Learn(const std::vector<TrainInputs>& inputs) {
205  int training_batch_size = inputs.size();
206 
207  // Torch tensors by default use a dense, row-aligned memory layout.
208  // - Their default data type is a 32-bit float
209  // - Use the byte data type for boolean
210 
211  torch::Tensor torch_train_inputs =
212  torch::empty({training_batch_size, flat_input_size_}, torch_device_);
213  torch::Tensor torch_train_legal_mask = torch::full(
214  {training_batch_size, num_actions_}, false,
215  torch::TensorOptions().dtype(torch::kByte).device(torch_device_));
216  torch::Tensor torch_policy_targets =
217  torch::zeros({training_batch_size, num_actions_}, torch_device_);
218  torch::Tensor torch_value_targets =
219  torch::empty({training_batch_size, 1}, torch_device_);
220 
221  for (int batch = 0; batch < training_batch_size; ++batch) {
222  // Copy the legal mask(s) to a Torch tensor.
223  for (Action action : inputs[batch].legal_actions) {
224  torch_train_legal_mask[batch][action] = true;
225  }
226 
227  // Copy the observation(s) to a Torch tensor.
228  for (int i = 0; i < inputs[batch].observations.size(); ++i) {
229  torch_train_inputs[batch][i] = inputs[batch].observations[i];
230  }
231 
232  // Copy the policy target(s) to a Torch tensor.
233  for (const auto& [action, probability] : inputs[batch].policy) {
234  torch_policy_targets[batch][action] = probability;
235  }
236 
237  // Copy the value target(s) to a Torch tensor.
238  torch_value_targets[batch][0] = inputs[batch].value;
239  }
240 
241  // Run a training step and get the losses.
242  model_->train();
243  model_->zero_grad();
244 
245  std::vector<torch::Tensor> torch_outputs =
246  model_->losses(torch_train_inputs, torch_train_legal_mask,
247  torch_policy_targets, torch_value_targets);
248 
249  torch::Tensor total_loss =
250  torch_outputs[0] + torch_outputs[1] + torch_outputs[2];
251 
252  total_loss.backward();
253 
254  model_optimizer_.step();
255 
256  return LossInfo(torch_outputs[0].item<float>(),
257  torch_outputs[1].item<float>(),
258  torch_outputs[2].item<float>());
259 }
