25def n_choose_k(n, k):
26 """Returns the combination choose k among n items."""
27 f = math.factorial
28 return int(f(n) / f(k) / f(n - k))

31def grid_simplex(step=.1, boundary=False):
32 """Generator for regular 'lattice' on the 2-simplex.
33
34 Args:
35 step: Defines spacing along one dimension.
36 boundary: Include points on the boundary/face of the simplex.
37
38 Yields:
39 Next point on the grid.
40 """
41 eps = 1e-8
42 start = 0. if boundary else step
43 stop = 1. + eps if boundary else 1. - step + eps
44 for a in np.arange(start, stop, step, dtype=np.double):
45 for b in np.arange(start, stop - a, step, dtype=np.double):
46 yield [a, b, 1. - a - b]

49def sample_from_simplex(n, dim=3, vmin=0.):
50 """Samples random points from a k-simplex.
51
52 See Donald B. Rubin (1981) "The Bayesian Bootstrap", page 131.
53
54 Args:
55 n: Number of points that are sampled.
56 dim: Dimension of the points to be sampled, e.g. dim=3 samples points from
57 the 2-simplex.
58 vmin: Minimum value of any coordinate of the resulting points, e.g. set
59 vmin>0. to exclude points on the faces of the simplex.
60
61 Returns:
62 `ndarray(shape=(k, dim))` of uniformly random points on the (num-1)-simplex.
63 """
64 assert vmin >= 0.
65 p = np.random.rand(n, dim - 1)
66 p = np.sort(p, axis=1)
67 p = np.hstack((np.zeros((n, 1)), p, np.ones((n, 1))))
68 return (p[:, 1:] - p[:, 0:-1]) * (1 - 2 * vmin) + vmin

71def game_payoffs_array(game):
72 """Returns a `numpy.ndarray` of utilities for a game.
73
74 NOTE: if the game is not a MatrixGame or a TensorGame then this may be costly.
75
76 Args:
77 game: A game.
78
79 Returns:
80 `numpy.ndarray` of dimension `num_players` + 1.
81 First dimension is the player, followed by the actions of all players, e.g.
82 a 3x3 game (2 players) has dimension [2,3,3].
83 """
84 if isinstance(game, pyspiel.MatrixGame):
85 return np.stack([game.row_utilities(), game.col_utilities()])
86
87 if not isinstance(game, pyspiel.TensorGame):
88 game = pyspiel.extensive_to_tensor_game(game)
89 return np.stack(
90 [game.player_utilities(player) for player in range(game.num_players())])

93def distribute(num_items, num_slots, normalize=False):
94 """Yields all ways of distributing `num_items` items over `num_slots` slots.
95
96 We assume that the ordering of the slots doesn't matter.
97
98 Args:
99 num_items: The number of items to distribute.
100 num_slots: The number of slots.
101 normalize: Normalizes the yielded tuple to contain floats in [0, 1] summing
102 to 1.
103
104 Yields:
105 A tuple T containing `num_slots` positive integers such that
106 `np.sum(T) == num_items` if `normalize == False` or `np.sum(T) == 1` if
107 `normalize == True'.
108 """
109 normalization = 1
110 if normalize:
111 normalization = num_items
112 # This is just the standard "bars and stars" problem.
113 # See https://stackoverflow.com/questions/28965734/general-bars-and-stars.
114 for c in itertools.combinations(
115 range(num_items + num_slots - 1), num_slots - 1):
116 # The combinations give you the indices of the internal bars.
117 # pylint: disable=g-complex-comprehension
118 yield tuple((b - a - 1) / normalization
119 for (a, b) in zip([
120 -1,
121 ] + list(c),
122 list(c) + [num_items + num_slots - 1]))

158def get_valid_next_profiles(num_strats_per_population, cur_profile):
159 """Generates monomorphic strategy profile transitions given cur_profile.
160
161 Given a current strategy profile, cur_profile, this generates all follow-up
162 profiles that involve only a single other population changing its current
163 monomorphic strategy to some other monomorphic strategy. Note that
164 self-transitions from cur_profile to cur_profile are not included here, as
165 they are a special case in our Markov chain.
166
167 Args:
168 num_strats_per_population: List of strategy sizes for each population.
169 cur_profile: Current strategy profile.
170
171 Yields:
172 The next valid strategy profile transition.
173 """
174 num_populations = len(num_strats_per_population)
175
176 for i_population_to_change in range(num_populations):
177 for new_strat in range(num_strats_per_population[i_population_to_change]):
178 # Ensure a transition will actually happen
179 if new_strat != cur_profile[i_population_to_change]:
180 next_profile = cur_profile.copy()
181 next_profile[i_population_to_change] = new_strat
182 yield i_population_to_change, next_profile

185def get_num_strats_per_population(payoff_tables, payoffs_are_hpt_format):
186 """Returns a [num_populations] array of the num.
187
188 of strategies per population.
189
190 E.g., for a 3 population game, this returns
191 [num_strats_population1, num_strats_population2, num_strats_population3]
192
193 Args:
194 payoff_tables: List of game payoff tables, one for each agent identity. Each
195 payoff_table may be either a 2D numpy array, or a _PayoffTableInterface
196 object.
197 payoffs_are_hpt_format: True indicates HPT format (i.e.
198 _PayoffTableInterface object, False indicates 2D numpy array.
199 """
200
201 if payoffs_are_hpt_format:
202 return np.asarray(
203 [payoff_table.num_strategies for payoff_table in payoff_tables])
204 else:
205 # Non-HPT payoffs are matrices, so can directly return the payoff size
206 return np.asarray(np.shape(payoff_tables[0]))

222def get_strat_profile_labels(payoff_tables, payoffs_are_hpt_format):
223 """Returns strategy labels corresponding to a payoff_table.
224
225 Namely, for games where strategies have no human-understandable labels
226 available, this function returns a labels object corresponding to the
227 strategy profiles.
228
229 Examples:
230 Generated labels for a single-population game with 3 strategies:
231 ['0','1','2'].
232 Generated labels for a 3-population game with 2 strategies per population:
233 {0: ['0','1'], 1: ['0','1'], 2: ['0','1']}
234
235 Args:
236 payoff_tables: List of game payoff tables, one for each agent identity. Each
237 payoff_table may be either a 2D numpy array, or a _PayoffTableInterface
238 object.
239 payoffs_are_hpt_format: Boolean indicating whether each payoff table in
240 payoff_tables is a 2D numpy array, or a _PayoffTableInterface object (AKA
241 Heuristic Payoff Table or HPT). True indicates HPT format, False indicates
242 2D numpy array.
243
244 Returns:
245 Strategy labels.
246 """
247
248 num_populations = len(payoff_tables)
249
250 if num_populations == 1:
251 num_strats_per_population = get_num_strats_per_population(
252 payoff_tables, payoffs_are_hpt_format)
253 labels = [str(x) for x in range(num_strats_per_population[0])]
254 else:
255 num_strats_per_population = get_num_strats_per_population(
256 payoff_tables, payoffs_are_hpt_format)
257 labels = dict()
258 label_text = []
259 # Construct a list of strategy labels for each population
260 for num_strats in num_strats_per_population:
261 label_text.append([str(i_strat) for i_strat in range(num_strats)])
262 population_ids = range(num_populations)
263 labels = dict(zip(population_ids, label_text))
264
265 return labels

268def get_strat_profile_from_id(num_strats_per_population, profile_id):
269 """Returns the strategy profile corresponding to a requested strategy ID.
270
271 This is the inverse of the function get_id_from_strat_profile(). See that
272 function for the indexing mechanism.
273
274 Args:
275 num_strats_per_population: List of strategy sizes for each population.
276 profile_id: Integer ID of desired strategy profile, in
277 {0,...,get_num_profiles-1}.
278
279 Returns:
280 The strategy profile whose ID was looked up.
281 """
282
283 num_populations = len(num_strats_per_population)
284 strat_profile = np.zeros(num_populations, dtype=np.int32)
285
286 for i_population in range(num_populations - 1, -1, -1):
287 strat_profile[i_population] = (
288 profile_id % num_strats_per_population[i_population])
289 profile_id = profile_id // num_strats_per_population[i_population]
290
291 return strat_profile

294def get_label_from_strat_profile(num_populations, strat_profile, strat_labels):
295 """Returns a human-readable label corresponding to the strategy profile.
296
297 E.g., for Rock-Paper-Scissors, strategies 0,1,2 have labels "R","P","S".
298 For strat_profile (1,2,0,1), this returns "(P,S,R,P)". If strat_profile is a
299 single strategy (e.g., 0) this returns just its label (e.g., "R").
300
301 Args:
302 num_populations: Number of populations.
303 strat_profile: Strategy profile of interest.
304 strat_labels: Strategy labels.
305
306 Returns:
307 Human-readable label string.
308 """
309 if num_populations == 1:
310 return strat_labels[strat_profile]
311 else:
312 label = "("
313 for i_population, i_strat in enumerate(strat_profile):
314 label += strat_labels[i_population][i_strat]
315 if i_population < len(strat_profile) - 1:
316 label += ","
317 label += ")"
318 return label

321def get_id_from_strat_profile(num_strats_per_population, strat_profile):
322 """Returns a unique integer ID representing the requested strategy profile.
323
324 Map any `strat_profile` (there are `np.prod(num_strats_per_population)` such
325 profiles) to {0,..., num_strat_profiles - 1}.
326
327 The mapping is done using a usual counting strategy: With
328 num_strats_per_population = [a1, ..., a_n]
329 strat_profile = [b1, ..., b_n]
330
331 we have
332
333 id = b_1 + a1 * (b2 + a_2 * (b3 + a_3 *...))
334
335
336 This is helpful for querying the element of our finite-population Markov
337 transition matrix that corresponds to a transition between a specific pair of
338 strategy profiles.
339
340 Args:
341 num_strats_per_population: List of strategy sizes for each population.
342 strat_profile: The strategy profile (list of integers corresponding to the
343 strategy of each agent) whose ID is requested.
344
345 Returns:
346 Unique ID of strat_profile.
347 """
348
349 if len(strat_profile) == 1:
350 return strat_profile[0]
351
352 return strat_profile[-1] + (num_strats_per_population[-1] *
353 get_id_from_strat_profile(
354 num_strats_per_population[:-1],
355 strat_profile[:-1]))

370def check_is_constant_sum(payoff_table, payoffs_are_hpt_format):
371 """Checks if single-population matrix game is constant-sum.
372
373 Args:
374 payoff_table: Either a 2D numpy array, or a _PayoffTableInterface object.
375 payoffs_are_hpt_format: Boolean indicating whether payoff table is a
376 _PayoffTableInterface object (AKA Heuristic Payoff Table or HPT), or a 2D
377 numpy array. True indicates HPT, and False indicates numpy array.
378
379 Returns:
380 is_constant_sum: Boolean, True if constant-sum game.
381 payoff_sum: Payoff sum if game is constant-sum, or None if not.
382 """
383
384 if payoffs_are_hpt_format:
385 payoff_sum_table = np.asarray(payoff_table._payoffs).sum(axis=1) # pylint: disable=protected-access
386 is_constant_sum = np.isclose(
387 payoff_sum_table, payoff_sum_table[0], atol=1e-14).all()
388 payoff_sum = payoff_sum_table[0] if is_constant_sum else None
389 else:
390 payoff_sum_table = payoff_table + payoff_table.T
391 is_constant_sum = np.isclose(
392 payoff_sum_table, payoff_sum_table[0, 0], atol=1e-14).all()
393 payoff_sum = payoff_sum_table[0, 0] if is_constant_sum else None
394 return is_constant_sum, payoff_sum

397def cluster_strats(pi, matching_decimals=4):
398 """Clusters strategies using stationary distribution (pi) masses.
399
400 Args:
401 pi: stationary distribution.
402 matching_decimals: the number of stationary distribution decimals that
403 should match for strategies to be considered in the same cluster.
404
405 Returns:
406 Dictionary that maps unique stationary distribution masses to strategies.
407 """
408
409 rounded_masses = pi.round(decimals=matching_decimals)
410 masses_to_strats = {}
411 for i in np.unique(rounded_masses):
412 masses_to_strats[i] = np.where(rounded_masses == i)[0]
413 return masses_to_strats

416def print_rankings_table(payoff_tables,
417 pi,
418 strat_labels,
419 num_top_strats_to_print=8):
420 """Prints nicely-formatted table of strategy rankings.
421
422 Args:
423 payoff_tables: List of game payoff tables, one for each agent identity. Each
424 payoff_table may be either a 2D numpy array, or a _PayoffTableInterface
425 object.
426 pi: Finite-population Markov chain stationary distribution.
427 strat_labels: Strategy labels.
428 num_top_strats_to_print: Number of top strategies to print.
429 """
430
431 num_populations = len(payoff_tables)
432 payoffs_are_hpt_format = check_payoffs_are_hpt(payoff_tables)
433 num_strats_per_population = get_num_strats_per_population(
434 payoff_tables, payoffs_are_hpt_format)
435
436 # More than total number of strats requested for printing, compute top and
437 # use an extra row to indicate additional strategies not shown.
438 row_for_lowrank_strats = True
439 if num_top_strats_to_print >= len(pi):
440 num_top_strats_to_print = len(pi)
441 row_for_lowrank_strats = False
442
443 # Cluster strategies according to stationary distr. (in case of tied ranks)
444 masses_to_strats = cluster_strats(pi)
445
446 def print_3col(col1, col2, col3):
447 print("%-12s %-12s %-12s" % (col1, col2, col3))
448
449 print_3col("Agent", "Rank", "Score")
450 print_3col("-----", "----", "-----")
451
452 rank = 1
453 num_strats_printed = 0
454 # Print a table of strategy rankings from highest to lowest mass
455 for _, strats in sorted(masses_to_strats.items(), reverse=True):
456 for strat in strats:
457 if num_strats_printed >= num_top_strats_to_print:
458 break
459 rounded_pi = np.round(pi[strat], decimals=2)
460 if num_populations == 1:
461 strat_profile = strat
462 else:
463 strat_profile = get_strat_profile_from_id(num_strats_per_population,
464 strat)
465 label = get_label_from_strat_profile(num_populations, strat_profile,
466 strat_labels)
467 print_3col(label, str(rank), str(np.abs(rounded_pi)))
468 num_strats_printed += 1
469 rank += 1
470 if num_strats_printed >= num_top_strats_to_print:
471 break
472
473 # Ellipses to signify additional low-rank strategies are not printed
474 if row_for_lowrank_strats:
475 print_3col("...", "...", "...")

478def is_symmetric_matrix_game(payoff_tables):
479 """Checks if payoff_tables corresponds to a symmetric matrix game."""
480 payoffs_are_hpt_format = check_payoffs_are_hpt(payoff_tables)
481
482 if len(payoff_tables) == 2:
483 if payoffs_are_hpt_format and np.array_equal(payoff_tables[0](),
484 payoff_tables[1]()):
485 return True, [payoff_tables[0]]
486 elif ~payoffs_are_hpt_format and np.array_equal(payoff_tables[0],
487 payoff_tables[1].T):
488 return True, [payoff_tables[0]]
489 return False, payoff_tables
