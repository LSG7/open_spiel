89 def get_chance_policy(self) -> Dict[int, float]:
90 """See base class."""
91 if self.is_terminalis_terminal():
92 return {}
93
94 if not self.state.is_chance_node():
95 return {0: 1}
96
97 chance_policy = {
98 action: prob for (action, prob) in self.state.chance_outcomes()
99 }
100
101 if self._perturbation:
102 probs = [random.random() for _ in self.state.chance_outcomes()]
103 chance_policy = {
104 action: probs[i] / sum(probs)
105 for i, (action, prob) in enumerate(self.state.chance_outcomes())
106 }
