33 itertools.product(("rpg", "qpg", "rm", "a2c"),

35 def test_run_game(self, loss_str, game_name):
36 env = rl_environment.Environment(game_name)
37 env.seed(SEED)
38 info_state_size = env.observation_spec()["info_state"][0]
39 num_actions = env.action_spec()["num_actions"]
40
41 agents = [
42 policy_gradient.PolicyGradient( # pylint: disable=g-complex-comprehension
43 player_id=player_id,
44 info_state_size=info_state_size,
45 num_actions=num_actions,
46 loss_str=loss_str,
47 hidden_layers_sizes=[32, 32],
48 lambda_=1.0,
49 entropy_cost=0.001,
50 critic_learning_rate=0.01,
51 pi_learning_rate=0.01,
52 num_critic_before_pi=4,
53 seed=SEED) for player_id in [0, 1]
54 ]
55
56 for _ in range(2):
57 time_step = env.reset()
58 while not time_step.last():
59 current_player = time_step.observations["current_player"]
60 current_agent = agents[current_player]
61 agent_output = current_agent.step(time_step)
62 time_step = env.step([agent_output.action])
63
64 for agent in agents:
65 agent.step(time_step)
