109class RestrictedNashResponseGame : public WrappedGame {
110 public:
111 explicit RestrictedNashResponseGame(
112 std::shared_ptr<const Game> game, Player fixed_player, double p,
113 std::shared_ptr<Policy> fixed_policy = nullptr);
114 std::shared_ptr<Observer> MakeObserver(
115 absl::optional<IIGObservationType> iig_obs_type,
116 const GameParameters& params) const;
117
118 std::unique_ptr<State> NewInitialState() const override {
119 return std::unique_ptr<State>(new RestrictedNashResponseState(
120 shared_from_this(), game_->NewInitialState(), false, fixed_player_,
121 true, p_, fixed_policy_));
122 }
123
124 int NumDistinctActions() const override {
125 return game_->NumDistinctActions();
126 }
127
128 int MaxChanceOutcomes() const override {
129 if (fixed_policy_) {
130 // If a fixed policy is set, then we have a real RNR game, which means
131 // there is at least one chance node with 2 outcomes. But also, the
132 // fixed player actions are also treated as chance nodes, so the number
133 // of distinct actions can also determine the maximum number of chance
134 // outcomes.
135 std::vector<int> candidates = {
136 game_->MaxChanceOutcomes(), 2, game_->NumDistinctActions()
137 };
138 return *std::max_element(candidates.begin(), candidates.end());
139 } else {
140 // Otherwise, it's the normal game.
141 return game_->MaxChanceOutcomes();
142 }
143 }
144
145 int NumPlayers() const override { return game_->NumPlayers(); }
146
147 double MinUtility() const override { return game_->MinUtility(); }
148
149 double MaxUtility() const override { return game_->MaxUtility(); }
150
151 absl::optional<double> UtilitySum() const override {
152 return game_->UtilitySum();
153 }
154
155 std::vector<int> InformationStateTensorShape() const override {
156 // Underlying game plus
157 return {2 + game_->InformationStateTensorSize()};
158 }
159
160 std::vector<int> ObservationTensorShape() const override {
161 // We flatten the representation of the underlying game and add one-hot
162 // indications of the to-play player and the observing player.
163 return {2 + game_->ObservationTensorSize()};
164 }
165
166 int MaxGameLength() const override { return game_->MaxGameLength() + 1; }
167
168 int MaxChanceNodesInHistory() const override {
169 if (fixed_policy_) {
170 // If a fixed policy is set, then we have a real RNR game, which has an
171 // extra chance node.
172 return game_->MaxChanceNodesInHistory() + 1;
173 } else {
174 // Otherwise, it's just the normal game.
175 return game_->MaxChanceNodesInHistory();
176 }
177 }
178 // old observation API
179 std::shared_ptr<RestrictedNashResponseObserver> default_observer_;
180 std::shared_ptr<RestrictedNashResponseObserver> info_state_observer_;
181
182 private:
183 // Fixed player and p constants to be passed to the initial state
184 const Player fixed_player_;
185 const double p_;
186 // Constants for the fixed strategy and if we use explicit fixed strategy
187 std::shared_ptr<Policy> fixed_policy_;
188};
