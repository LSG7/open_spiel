64class Move {
65 public:
66 Move(int move) : row_(move / kNumCols), col_(move % kNumCols) {
67 SPIEL_CHECK_GE(move, 0);
68 SPIEL_CHECK_LT(move, kNumCells);
69 }
70
71 Move(int row, int col) : row_(row), col_(col) {}
72
73 inline int GetRow() const { return row_; }
74 inline int GetColumn() const { return col_; }
75 inline int GetAction() const { return row_ * kNumCols + col_; }
76 inline bool OnBoard() const;
77
78 Move Next(Direction dir) const;
79 std::string ToString() const;
80
81 private:
82 int row_;
83 int col_;
84};

140// Game object.
141class OthelloGame : public Game {
142 public:
143 explicit OthelloGame(const GameParameters& params);
144 int NumDistinctActions() const override { return kNumCells + 1; } // can pass
145 std::unique_ptr<State> NewInitialState() const override {
146 return std::unique_ptr<State>(new OthelloState(shared_from_this()));
147 }
148 int NumPlayers() const override { return kNumPlayers; }
149 double MinUtility() const override { return -1; }
150 absl::optional<double> UtilitySum() const override { return 0; }
151 double MaxUtility() const override { return 1; }
152 std::vector<int> ObservationTensorShape() const override {
153 return {kCellStates, kNumRows, kNumCols};
154 }
155
156 // Conservative upper bound due to pass moves.
157 int MaxGameLength() const override { return 2*kNumCells; }
158};
