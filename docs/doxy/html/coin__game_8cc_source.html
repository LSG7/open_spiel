90enum struct SymbolType { kEmpty = 0, kCoin = 1, kPlayer = 2 };

93SymbolType GetSymbolType(char symbol) {
94 if (symbol == kEmptySymbol) {
95 return SymbolType::kEmpty;
96 } else if ('a' <= symbol && symbol <= 'z') {
97 return SymbolType::kCoin;
98 } else if ('0' <= symbol && symbol <= '9') {
99 return SymbolType::kPlayer;
100 }
101 SpielFatalError(absl::StrCat("Unexpected symbol: ", std::string(1, symbol)));
102}

170GamePhase CoinState::GetPhase() const {
171 if (cur_player_ != kChancePlayerId) {
172 return GamePhase::kPlay;
173 } else if (setup_.num_players_assigned_preference < num_players_) {
174 return GamePhase::kAssignPreferences;
175 } else if (setup_.num_players_on_field < num_players_) {
176 return GamePhase::kDeployPlayers;
177 } else if (setup_.num_coins_on_field < parent_game_.TotalCoins()) {
178 return GamePhase::kDeployCoins;
179 } else {
180 SpielFatalError("Inconsistent setup versus current_player state");
181 }
182}

184std::vector<Action> CoinState::LegalActions() const {
185 if (IsTerminal()) return {};
186 switch (GetPhase()) {
187 case GamePhase::kAssignPreferences:
188 return ActionRange(setup_.available_coin_colors_);
189 case GamePhase::kDeployPlayers:
190 return ActionRange(setup_.available_positions_);
191 case GamePhase::kDeployCoins:
192 return ActionRange(setup_.available_positions_);
193 case GamePhase::kPlay:
194 return Range(offsets.size());
195 default:
196 SpielFatalError("Unknown phase.");
197 }
198}

248void CoinState::ApplyAssignPreferenceAction(Action coin_color) {
249 SPIEL_CHECK_LT(coin_color, parent_game_.NumCoinColors());
250 player_preferences_[setup_.num_players_assigned_preference] = coin_color;
251 ++setup_.num_players_assigned_preference;
252 setup_.available_coin_colors_.erase(coin_color);
253}
254
255void CoinState::ApplyDeployPlayersAction(Action index) {
256 SPIEL_CHECK_LT(index, field_.size());
257 SPIEL_CHECK_TRUE(GetSymbolType(field_[index]) == SymbolType::kEmpty);
258 field_[index] = PlayerSymbol(setup_.num_players_on_field);
259 player_location_[setup_.num_players_on_field] = LocationFromIndex(index);
260 ++setup_.num_players_on_field;
261 setup_.available_positions_.erase(index);
262}
263
264void CoinState::ApplyDeployCoinsAction(Action index) {
265 SPIEL_CHECK_LT(index, field_.size());
266 SPIEL_CHECK_TRUE(GetSymbolType(field_[index]) == SymbolType::kEmpty);
267
268 int coin_color = setup_.num_coins_on_field / parent_game_.NumCoinsPerColor();
269 field_[index] = CoinSymbol(coin_color);
270 ++setup_.num_coins_on_field;
271 setup_.available_positions_.erase(index);
272
273 if (setup_.num_coins_on_field == parent_game_.TotalCoins()) {
274 // Switch to play phase.
275 setup_.available_positions_.clear(); // Release memory.
276 setup_.available_coin_colors_.clear(); // Release memory.
277 cur_player_ = 0;
278 }
279}

281void CoinState::ApplyPlayAction(Action move) {
282 ++total_moves_;
283
284 Location old_loc = player_location_[cur_player_];
285 SPIEL_CHECK_EQ(GetField(old_loc), PlayerSymbol(cur_player_));
286
287 Location new_loc = old_loc + offsets[move];
288 if (InBounds(new_loc)) {
289 char target = GetField(new_loc);
290 SymbolType target_type = GetSymbolType(target);
291 if (target_type == SymbolType::kCoin) {
292 IncPlayerCoinCount(cur_player_, CoinId(target));
293 }
294 if (target_type == SymbolType::kCoin || target_type == SymbolType::kEmpty) {
295 player_location_[cur_player_] = new_loc;
296 SetField(old_loc, kEmptySymbol);
297 SetField(new_loc, PlayerSymbol(cur_player_));
298 }
299 }
300 cur_player_ = (cur_player_ + 1) % num_players_;
301}

303void CoinState::DoApplyAction(Action action) {
304 switch (GetPhase()) {
305 case GamePhase::kAssignPreferences:
306 ApplyAssignPreferenceAction(action);
307 break;
308 case GamePhase::kDeployPlayers:
309 ApplyDeployPlayersAction(action);
310 break;
311 case GamePhase::kDeployCoins:
312 ApplyDeployCoinsAction(action);
313 break;
314 case GamePhase::kPlay:
315 ApplyPlayAction(action);
316 break;
317 }
318}

348void CoinState::PrintCoinsCollected(std::ostream& out) const {
349 // Prints table with players as rows and coin_colors as columns.
350 out << " ";
351 for (int coint_color = 0; coint_color < parent_game_.NumCoinColors();
352 coint_color++) {
353 out << CoinSymbol(coint_color) << " ";
354 }
355 out << "\n";
356 for (auto player = Player{0}; player < num_players_; player++) {
357 out << "player" << player << " ";
358 for (int coint_color = 0; coint_color < parent_game_.NumCoinColors();
359 coint_color++) {
360 out << GetPlayerCoinCount(player, coint_color) << " ";
361 }
362 out << "\n";
363 }

409std::vector<double> CoinState::Returns() const {
410 if (!IsTerminal()) {
411 return std::vector<double>(num_players_, 0.0);
412 }
413
414 int collected_coins = 0;
415 std::vector<int> coin_count(parent_game_.NumCoinColors());
416 for (int coin_color = 0; coin_color < parent_game_.NumCoinColors();
417 coin_color++) {
418 for (auto player = Player{0}; player < num_players_; player++) {
419 Player player_coins = GetPlayerCoinCount(player, coin_color);
420 coin_count[coin_color] += player_coins;
421 collected_coins += player_coins;
422 }
423 }
424 int good_coins = 0;
425 for (int preference : player_preferences_) {
426 good_coins += coin_count[preference];
427 }
428 const int bad_coins = collected_coins - good_coins;
429 std::vector<double> rewards(num_players_);
430 for (auto player = Player{0}; player < num_players_; player++) {
431 int self_coins = coin_count[player_preferences_[player]];
432 int other_coins = good_coins - self_coins;
433 rewards[player] = (std::pow(self_coins, 2) + std::pow(other_coins, 2) -
434 std::pow(bad_coins, 2));
435 }
436 return rewards;
437}
