32// Return all permutations of a vector.
33// This returns n! vectors, where n is the size of the vector.
34template <typename T>
35std::vector<std::vector<T>> Permutations(std::vector<T> v) {
36 std::vector<std::vector<T>> vs;
37 int perm_size = 1;
38 for (int i = 2; i <= v.size(); ++i) perm_size *= i;
39 vs.reserve(perm_size);
40 std::sort(v.begin(), v.end());
41 do {
42 vs.push_back(v);
43 } while (std::next_permutation(v.begin(), v.end()));
44 return vs;
45}

47// Return all subsets of size k from a vector of size n (all combinations).
48// This implements "n choose k" (or also known as binomial coefficient).
49// Returns (n k) = n! / ( k! * (n-k)! ) vectors.
50template <typename T>
51std::vector<std::vector<T>> SubsetsOfSize(const std::vector<T>& v, int k) {
52 SPIEL_CHECK_LE(k, v.size());
53 SPIEL_CHECK_GE(k, 0);
54 std::vector<bool> bitset(v.size() - k, 0);
55 bitset.resize(v.size(), 1);
56 std::vector<std::vector<T>> vs;
57
58 do {
59 std::vector<T> x;
60 x.reserve(k);
61 for (std::size_t i = 0; i != v.size(); ++i) {
62 if (bitset[i]) {
63 x.push_back(v[i]);
64 }
65 }
66 vs.push_back(x);
67 } while (std::next_permutation(bitset.begin(), bitset.end()));
68
69 return vs;
70}

74// Return the power set of a vector of size n.
75// Returns 2^n vectors.
76template <typename T>
77std::vector<std::vector<T>> PowerSet(const std::vector<T>& v) {
78 std::vector<bool> bitset(v.size());
79 std::vector<std::vector<T>> vs;
80 do {
81 std::vector<T> x;
82 for (std::size_t i = 0; i != v.size(); ++i) {
83 if (bitset[i]) {
84 x.push_back(v[i]);
85 }
86 }
87 vs.push_back(x);
88 } while (NextPowerSetMask(&bitset));
89 return vs;
90}

95// Return all k-variations without repetition of a vector with the size n.
96// Also known as k-permutations of n.
97// The input is assumed that it does not contain repetitions.
98// This returns n! / (n-k)! vectors.
99// TODO(author13): more efficient implementation with something like
100// NextVariationsWithoutRepetitionMask
101template <typename T>
102std::vector<std::vector<T>> VariationsWithoutRepetition(const std::vector<T>& v,
103 int k) {
104 SPIEL_CHECK_LE(k, v.size());
105 SPIEL_CHECK_GE(k, 0);
106
107 // Generate masks -- avoid copying of T, as that might
108 // be more expensive than juggling integers.
109 std::vector<int> current_permutation;
110 std::vector<int> rng(v.size());
111 std::iota(rng.begin(), rng.end(), 0);
112 auto masks = GenerateMasks(rng, k, current_permutation);
113
114 // Apply the masks.
115 std::vector<std::vector<T>> vs;
116 for (auto& mask : masks) {
117 std::vector<T> x;
118 for (auto& i : mask) {
119 x.push_back(v[i]);
120 }
121 vs.push_back(x);
122 }
123 return vs;
124}

128// Returns the k^th permutation of the elements in v. This algorithm skips over
129// ranges of digits by computing the number of permutations for each digit from
130// factorials over the suffixes in reading order.
131//
132// E.g. for v = {0, 1, 2, 3} and k = 19
133// - Skip over 6 (= 3!) permutations starting with 0.
134// - Skip over 6 (= 3!) permutations starting with 1.
135// - Skip over 6 (= 3!) permutations starting with 2.
136// - Skip over two permutations starting with (3, 0) ((3, 0, 1, 2) and
137// (3, 0, 2, 1)) to arrive at (3, 1, 0, 2).
138template <typename T>
139std::vector<T> UnrankPermutation(const std::vector<T>& v, int k) {
140 int n = v.size();
141 std::vector<bool> used(v.size(), false);
142 std::vector<T> perm(v.size());
143 for (int i = 1; i <= n; ++i) {
144 int divisor = Factorial(n - i);
145 int digit_idx = k / divisor;
146 int j = 0, l = 0;
147 for (; j < n; ++j) {
148 if (used[j]) {
149 continue;
150 }
151 if (l == digit_idx) {
152 break;
153 }
154 ++l;
155 }
156 perm[i - 1] = v[j];
157 used[j] = true;
158 k -= digit_idx * divisor;
159 }
160 return perm;
161}
