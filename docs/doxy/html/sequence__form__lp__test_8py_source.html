15"""Tests for LP solvers."""

17from absl.testing import absltest
18
19from open_spiel.python.algorithms import sequence_form_lp
20import pyspiel
21
22
23class SFLPTest(absltest.TestCase):
24
25 def test_rock_paper_scissors(self):
26 game = pyspiel.load_game_as_turn_based("matrix_rps")
27 val1, val2, _, _ = sequence_form_lp.solve_zero_sum_game(game)
28 self.assertAlmostEqual(val1, 0)
29 self.assertAlmostEqual(val2, 0)
30
31 def test_kuhn_poker(self):
32 game = pyspiel.load_game("kuhn_poker")
33 val1, val2, _, _ = sequence_form_lp.solve_zero_sum_game(game)
34 # value from Kuhn 1950 or https://en.wikipedia.org/wiki/Kuhn_poker
35 self.assertAlmostEqual(val1, -1 / 18)
36 self.assertAlmostEqual(val2, +1 / 18)
37
38 def test_kuhn_poker_efg(self):
39 game = pyspiel.load_efg_game(pyspiel.get_kuhn_poker_efg_data())
40 val1, val2, _, _ = sequence_form_lp.solve_zero_sum_game(game)
41 # value from Kuhn 1950 or https://en.wikipedia.org/wiki/Kuhn_poker
42 self.assertAlmostEqual(val1, -1 / 18)
43 self.assertAlmostEqual(val2, +1 / 18)
44
45 def test_leduc_poker(self):
46 game = pyspiel.load_game("leduc_poker")
47 val1, val2, _, _ = sequence_form_lp.solve_zero_sum_game(game)
48 # values obtained from Appendix E.2 of Lanctot et al. 2017, A Unified
49 # Game-Theoretic Approach to Multiagent Reinforcement Learning.
50 # https://arxiv.org/abs/1711.00832
51 self.assertAlmostEqual(val1, -0.085606424078, places=6)
52 self.assertAlmostEqual(val2, 0.085606424078, places=6)
53
54 def test_iigoofspiel4(self):
55 game = pyspiel.load_game_as_turn_based("goofspiel", {
56 "imp_info": True,
57 "num_cards": 4,
58 "points_order": "descending",
59 })
60 val1, val2, _, _ = sequence_form_lp.solve_zero_sum_game(game)
61 # symmetric game, should be 0
62 self.assertAlmostEqual(val1, 0)
63 self.assertAlmostEqual(val2, 0)
