29 def mip_nash(game, objective, solver='GLPK_MI'):
30  """Solves for the optimal Nash for two-player general-sum games.
31 
32  Using mixed-integer programming:
33  min f(x_0, x_1, p_mat)
34  s.t.
35  (u_0, u_1 are Nash payoffs variables of player 0 and 1)
36  p_mat[0] * x_1 <= u_0
37  x_0^T*p_mat[1] <= u_1
38  (if a pure strategy is in the support then its payoff is Nash payoff)
39  u_0 - p_mat[0] * x_1 <= u_max_0 * b_0
40  u_1 - x_0^T*p_mat[1] <= u_max_1 * b_1
41  (if a pure strategy is not in the support its probability mass is 0)
42  x_0 <= 1 - b_0
43  x_1 <= 1 - b_1
44  (probability constraints)
45  x_0 >= 0
46  1^T * x_0 = 1
47  x_1 >= 0
48  1^T * x_1 = 1
49  for all n, b_0[n] in {0, 1},
50  for all m, b_1[m] in {0, 1},
51  u_max_0, u_max_1 are the maximum payoff differences of player 0 and 1.
52  Note: this formulation is a basic one that may only work well
53  for simple objective function or low-dimensional inputs.
54  GLPK_MI solver only handles linear objective.
55  To handle nonlinear and high-dimensional cases,
56  it is recommended to use advance solvers such as GUROBI,
57  or use a piecewise linear approximation of the objective.
58  Args:
59  game: a pyspiel matrix game object
60  objective: a string representing the objective (e.g., MAX_SOCIAL_WELFARE)
61  solver: the mixed-integer solver used by cvxpy
62 
63  Returns:
64  optimal Nash (x_0, x_1)
65  """
66 
67  p_mat = game_payoffs_array(game)
68  if len(p_mat) != 2:
69  raise ValueError('MIP-Nash only works for two players.')
70 
71  assert len(p_mat) == 2
72  assert p_mat[0].shape == p_mat[1].shape
73 
74  (m_0, m_1) = p_mat[0].shape
75 
76  u_max_0 = np.max(p_mat[0]) - np.min(p_mat[0])
77  u_max_1 = np.max(p_mat[1]) - np.min(p_mat[1])
78 
79  x_0 = cp.Variable(m_0)
80  x_1 = cp.Variable(m_1)
81  u_0 = cp.Variable(1)
82  u_1 = cp.Variable(1)
83  b_0 = cp.Variable(m_0, boolean=True)
84  b_1 = cp.Variable(m_1, boolean=True)
85 
86  u_m = p_mat[0] @ x_1
87  u_n = x_0 @ p_mat[1]
88 
89  # probabilities constraints
90  constraints = [x_0 >= 0, x_1 >= 0, cp.sum(x_0) == 1, cp.sum(x_1) == 1]
91  # support constraints
92  constraints.extend([u_m <= u_0, u_0 - u_m <= u_max_0 * b_0, x_0 <= 1 - b_0])
93  constraints.extend([u_n <= u_1, u_1 - u_n <= u_max_1 * b_1, x_1 <= 1 - b_1])
94 
95  variables = {
96  'x_0': x_0,
97  'x_1': x_1,
98  'u_0': u_0,
99  'u_1': u_1,
100  'b_0': b_0,
101  'b_1': b_1,
102  'p_mat': p_mat,
103  }
104 
105  obj = TWO_PLAYER_OBJECTIVE[objective](variables)
106  prob = cp.Problem(obj, constraints)
107  prob.solve(solver=solver)
108 
109  return _simplex_projection(x_0.value.reshape(-1)), _simplex_projection(
110  x_1.value.reshape(-1)
111  )
