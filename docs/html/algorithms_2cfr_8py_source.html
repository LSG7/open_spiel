15 """Python implementation of the counterfactual regret minimization algorithm.
16 
17 One iteration of CFR consists of:
18 1) Compute current strategy from regrets (e.g. using Regret Matching).
19 2) Compute values using the current strategy
20 3) Compute regrets from these values
21 
22 The average policy is what converges to a Nash Equilibrium.
23 """

47 def _apply_regret_matching_plus_reset(info_state_nodes):
48  """Resets negative cumulative regrets to 0.
49 
50  Regret Matching+ corresponds to the following cumulative regrets update:
51  cumulative_regrets = max(cumulative_regrets + regrets, 0)
52 
53  This must be done at the level of the information set, and thus cannot be
54  done during the tree traversal (which is done on histories). It is thus
55  performed as an additional step.
56 
57  This function is a module level function to be reused by both CFRSolver and
58  CFRBRSolver.
59 
60  Args:
61  info_state_nodes: A dictionary {`info_state_str` -> `_InfoStateNode`}.
62  """
63  for info_state_node in info_state_nodes.values():
64  action_to_cum_regret = info_state_node.cumulative_regret
65  for action, cumulative_regret in action_to_cum_regret.items():
66  if cumulative_regret < 0:
67  action_to_cum_regret[action] = 0

89 def _update_average_policy(average_policy, info_state_nodes):
90  """Updates in place `average_policy` to the average of all policies iterated.
91 
92  This function is a module level function to be reused by both CFRSolver and
93  CFRBRSolver.
94 
95  Args:
96  average_policy: A `policy.TabularPolicy` to be updated in-place.
97  info_state_nodes: A dictionary {`info_state_str` -> `_InfoStateNode`}.
98  """
99  for info_state, info_state_node in info_state_nodes.items():
100  info_state_policies_sum = info_state_node.cumulative_policy
101  state_policy = average_policy.policy_for_key(info_state)
102  probabilities_sum = sum(info_state_policies_sum.values())
103  if probabilities_sum == 0:
104  num_actions = len(info_state_node.legal_actions)
105  for action in info_state_node.legal_actions:
106  state_policy[action] = 1 / num_actions
107  else:
108  for action, action_prob_sum in info_state_policies_sum.items():
109  state_policy[action] = action_prob_sum / probabilities_sum

130  def __init__(self, game, alternating_updates, linear_averaging,
131  regret_matching_plus):
132  # pyformat: disable
133  """Initializer.
134 
135  Args:
136  game: The `pyspiel.Game` to run on.
137  alternating_updates: If `True`, alternating updates are performed: for
138  each player, we compute and update the cumulative regrets and policies.
139  In that case, and when the policy is frozen during tree traversal, the
140  cache is reset after each update for one player.
141  Otherwise, the update is simultaneous.
142  linear_averaging: Whether to use linear averaging, i.e.
143  cumulative_policy[info_state][action] += (
144  iteration_number * reach_prob * action_prob)
145 
146  or not:
147 
148  cumulative_policy[info_state][action] += reach_prob * action_prob
149  regret_matching_plus: Whether to use Regret Matching+:
150  cumulative_regrets = max(cumulative_regrets + regrets, 0)
151  or simply regret matching:
152  cumulative_regrets = cumulative_regrets + regrets
153  """

155  assert game.get_type().dynamics == pyspiel.GameType.Dynamics.SEQUENTIAL, (
156  "CFR requires sequential games. If you're trying to run it " +
157  "on a simultaneous (or normal-form) game, please first transform it " +
158  "using turn_based_simultaneous_game.")
159 
160  self._game_game = game
161  self._num_players_num_players = game.num_players()

174  self._regret_matching_plus_regret_matching_plus = regret_matching_plus

239  def _compute_counterfactual_regret_for_player(self, state, policies,
240  reach_probabilities, player):
241  """Increments the cumulative regrets and policy for `player`.
242 
243  Args:
244  state: The initial game state to analyze from.
245  policies: A list of `num_players` callables taking as input an
246  `info_state_node` and returning a {action: prob} dictionary. For CFR,
247  this is simply returning the current policy, but this can be used in
248  the CFR-BR solver, to prevent code duplication. If None,
249  `_get_infostate_policy` is used.
250  reach_probabilities: The probability for each player of reaching `state`
251  as a numpy array [prob for player 0, for player 1,..., for chance].
252  `player_reach_probabilities[player]` will work in all cases.
253  player: The 0-indexed player to update the values for. If `None`, the
254  update for all players will be performed.
255 
256  Returns:
257  The utility of `state` for all players, assuming all players follow the
258  current policy defined by `self.Policy`.
259  """
260  if state.is_terminal():
261  return np.asarray(state.returns())
262 
263  if state.is_chance_node():
264  state_value = 0.0
265  for action, action_prob in state.chance_outcomes():
266  assert action_prob > 0
267  new_state = state.child(action)
268  new_reach_probabilities = reach_probabilities.copy()
269  new_reach_probabilities[-1] *= action_prob
270  state_value += action_prob * self._compute_counterfactual_regret_for_player_compute_counterfactual_regret_for_player(
271  new_state, policies, new_reach_probabilities, player)
272  return state_value
273 
274  current_player = state.current_player()
275  info_state = state.information_state_string(current_player)
276 
277  # No need to continue on this history branch as no update will be performed
278  # for any player.
279  # The value we return here is not used in practice. If the conditional
280  # statement is True, then the last taken action has probability 0 of
281  # occurring, so the returned value is not impacting the parent node value.
282  if all(reach_probabilities[:-1] == 0):
283  return np.zeros(self._num_players_num_players)
284 
285  state_value = np.zeros(self._num_players_num_players)
286 
287  # The utilities of the children states are computed recursively. As the
288  # regrets are added to the information state regrets for each state in that
289  # information state, the recursive call can only be made once per child
290  # state. Therefore, the utilities are cached.
291  children_utilities = {}
292 
293  info_state_node = self._info_state_nodes_info_state_nodes[info_state]
294  if policies is None:
295  info_state_policy = self._get_infostate_policy_get_infostate_policy(info_state)
296  else:
297  info_state_policy = policies[current_player](info_state)
298  for action in state.legal_actions():
299  action_prob = info_state_policy.get(action, 0.)
300  new_state = state.child(action)
301  new_reach_probabilities = reach_probabilities.copy()
302  new_reach_probabilities[current_player] *= action_prob
303  child_utility = self._compute_counterfactual_regret_for_player_compute_counterfactual_regret_for_player(
304  new_state,
305  policies=policies,
306  reach_probabilities=new_reach_probabilities,
307  player=player)
308 
309  state_value += action_prob * child_utility
310  children_utilities[action] = child_utility
311 
312  # If we are performing alternating updates, and the current player is not
313  # the current_player, we skip the cumulative values update.
314  # If we are performing simultaneous updates, we do update the cumulative
315  # values.
316  simulatenous_updates = player is None
317  if not simulatenous_updates and current_player != player:
318  return state_value
319 
320  reach_prob = reach_probabilities[current_player]
321  counterfactual_reach_prob = (
322  np.prod(reach_probabilities[:current_player]) *
323  np.prod(reach_probabilities[current_player + 1:]))
324  state_value_for_player = state_value[current_player]
325 
326  for action, action_prob in info_state_policy.items():
327  cfr_regret = counterfactual_reach_prob * (
328  children_utilities[action][current_player] - state_value_for_player)
329 
330  info_state_node.cumulative_regret[action] += cfr_regret
331  if self._linear_averaging_linear_averaging:
332  info_state_node.cumulative_policy[
333  action] += self._iteration_iteration * reach_prob * action_prob
334  else:
335  info_state_node.cumulative_policy[action] += reach_prob * action_prob
336 
337  return state_value

349 def _regret_matching(cumulative_regrets, legal_actions):
350  """Returns an info state policy by applying regret-matching.
351 
352  Args:
353  cumulative_regrets: A {action: cumulative_regret} dictionary.
354  legal_actions: the list of legal actions at this state.
355 
356  Returns:
357  A dict of action -> prob for all legal actions.
358  """
359  regrets = cumulative_regrets.values()
360  sum_positive_regrets = sum((regret for regret in regrets if regret > 0))
361 
362  info_state_policy = {}
363  if sum_positive_regrets > 0:
364  for action in legal_actions:
365  positive_action_regret = max(0.0, cumulative_regrets[action])
366  info_state_policy[action] = (
367  positive_action_regret / sum_positive_regrets)
368  else:
369  for action in legal_actions:
370  info_state_policy[action] = 1.0 / len(legal_actions)
371  return info_state_policy

374 class _CFRSolver(_CFRSolverBase):
375  r"""Implements the Counterfactual Regret Minimization (CFR) algorithm.
376 
377  The algorithm computes an approximate Nash policy for 2 player zero-sum games.
378 
379  CFR can be view as a policy iteration algorithm. Importantly, the policies
380  themselves do not converge to a Nash policy, but their average does.
381 
382  The main iteration loop is implemented in `evaluate_and_update_policy`:
383 
384  ```python
385  game = pyspiel.load_game("game_name")
386  initial_state = game.new_initial_state()
387 
388  cfr_solver = CFRSolver(game)
389 
390  for i in range(num_iterations):
391  cfr.evaluate_and_update_policy()
392  ```
393 
394  Once the policy has converged, the average policy (which converges to the Nash
395  policy) can be computed:
396  ```python
397  average_policy = cfr_solver.ComputeAveragePolicy()
398  ```
399 
400  # Policy and average policy
401 
402  policy(0) and average_policy(0) are not technically defined, but these
403  methods will return arbitrarily the uniform_policy.
404 
405  Then, we are expected to have:
406 
407  ```
408  for t in range(1, N):
409  cfr_solver.evaluate_and_update_policy()
410  policy(t) = RM or RM+ of cumulative regrets
411  avg_policy(t)(s, a) ~ \sum_{k=1}^t player_reach_prob(t)(s) * policy(k)(s, a)
412 
413  With Linear Averaging, the avg_policy is proportional to:
414  \sum_{k=1}^t k * player_reach_prob(t)(s) * policy(k)(s, a)
415  ```
416  """
417 
418  def evaluate_and_update_policy(self):
419  """Performs a single step of policy evaluation and policy improvement."""
420  self._iteration_iteration += 1
421  if self._alternating_updates_alternating_updates:
422  for player in range(self._game_game.num_players()):
423  self._compute_counterfactual_regret_for_player_compute_counterfactual_regret_for_player(
424  self._root_node_root_node,
425  policies=None,
426  reach_probabilities=np.ones(self._game_game.num_players() + 1),
427  player=player)
428  if self._regret_matching_plus_regret_matching_plus:
429  _apply_regret_matching_plus_reset(self._info_state_nodes_info_state_nodes)
430  _update_current_policy(self._current_policy_current_policy, self._info_state_nodes_info_state_nodes)
431  else:
432  self._compute_counterfactual_regret_for_player_compute_counterfactual_regret_for_player(
433  self._root_node_root_node,
434  policies=None,
435  reach_probabilities=np.ones(self._game_game.num_players() + 1),
436  player=None)
437  if self._regret_matching_plus_regret_matching_plus:
438  _apply_regret_matching_plus_reset(self._info_state_nodes_info_state_nodes)
439  _update_current_policy(self._current_policy_current_policy, self._info_state_nodes_info_state_nodes)
