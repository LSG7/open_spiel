40class EntropySchedule:
41 """An increasing list of steps where the regularisation network is updated.
42
43 Example
44 EntropySchedule([3, 5, 10], [2, 4, 1])
45 => [0, 3, 6, 11, 16, 21, 26, 10]
46 | 3 x2 | 5 x4 | 10 x1
47 """
48
49 def __init__(self, *, sizes: Sequence[int], repeats: Sequence[int]):
50 """Constructs a schedule of entropy iterations.
51
52 Args:
53 sizes: the list of iteration sizes.
54 repeats: the list, parallel to sizes, with the number of times for each
55 size from `sizes` to repeat.
56 """
57 try:
58 if len(repeats) != len(sizes):
59 raise ValueError("`repeats` must be parallel to `sizes`.")
60 if not sizes:
61 raise ValueError("`sizes` and `repeats` must not be empty.")
62 if any([(repeat <= 0) for repeat in repeats]):
63 raise ValueError("All repeat values must be strictly positive")
64 if repeats[-1] != 1:
65 raise ValueError("The last value in `repeats` must be equal to 1, "
66 "ince the last iteration size is repeated forever.")
67 except ValueError as e:
68 raise ValueError(
69 f"Entropy iteration schedule: repeats ({repeats}) and sizes"
70 f" ({sizes})."
71 ) from e
72
73 schedule = [0]
74 for size, repeat in zip(sizes, repeats):
75 schedule.extend([schedule[-1] + (i + 1) * size for i in range(repeat)])
76
77 self.schedule = np.array(schedule, dtype=np.int32)
78
79 def __call__(self, learner_step: int) -> Tuple[float, bool]:
80 """Entropy scheduling parameters for a given `learner_step`.
81
82 Args:
83 learner_step: The current learning step.
84
85 Returns:
86 alpha: The mixing weight (from [0, 1]) of the previous policy with
87 the one before for computing the intrinsic reward.
88 update_target_net: A boolean indicator for updating the target network
89 with the current network.
90 """
91
92 # The complexity below is because at some point we might go past
93 # the explicit schedule, and then we'd need to just use the last step
94 # in the schedule and apply the logic of
95 # ((learner_step - last_step) % last_iteration) == 0)
96
97 # The schedule might look like this:
98 # X----X-------X--X--X--X--------X
99 # learner_step | might be here ^ |
100 # or there ^ |
101 # or even past the schedule ^
102
103 # We need to deal with two cases below.
104 # Instead of going for the complicated conditional, let's just
105 # compute both and then do the A * s + B * (1 - s) with s being a bool
106 # selector between A and B.
107
108 # 1. assume learner_step is past the schedule,
109 # ie schedule[-1] <= learner_step.
110 last_size = self.schedule[-1] - self.schedule[-2]
111 last_start = self.schedule[-1] + (
112 learner_step - self.schedule[-1]) // last_size * last_size
113 # 2. assume learner_step is within the schedule.
114 start = jnp.amax(self.schedule * (self.schedule <= learner_step))
115 finish = jnp.amin(
116 self.schedule * (learner_step < self.schedule),
117 initial=self.schedule[-1],
118 where=(learner_step < self.schedule))
119 size = finish - start
120
121 # Now select between the two.
122 beyond = (self.schedule[-1] <= learner_step) # Are we past the schedule?
123 iteration_start = (last_start * beyond + start * (1 - beyond))
124 iteration_size = (last_size * beyond + size * (1 - beyond))
125
126 update_target_net = jnp.logical_and(
127 learner_step > 0, jnp.sum(learner_step == iteration_start))
128 alpha = jnp.minimum(
129 (2.0 * (learner_step - iteration_start)) / iteration_size, 1.0)
130
131 return alpha, update_target_net # pytype: disable=bad-return-type # jax-types

134@chex.dataclass(frozen=True)
135class FineTuning:
136 """Fine tuning options, aka policy post-processing.
137
138 Even when fully trained, the resulting softmax-based policy may put
139 a small probability mass on bad actions. This results in an agent
140 waiting for the opponent (itself in self-play) to commit an error.
141
142 To address that the policy is post-processed using:
143 - thresholding: any action with probability smaller than self.threshold
144 is simply removed from the policy.
145 - discretization: the probability values are rounded to the closest
146 multiple of 1/self.discretization.
147
148 The post-processing is used on the learner, and thus must be jit-friendly.
149 """
150 # The learner step after which the policy post processing (aka finetuning)
151 # will be enabled when learning. A strictly negative value is equivalent
152 # to infinity, ie disables finetuning completely.
153 from_learner_steps: int = -1
154 # All policy probabilities below `threshold` are zeroed out. Thresholding
155 # is disabled if this value is non-positive.
156 policy_threshold: float = 0.03
157 # Rounds the policy probabilities to the "closest"
158 # multiple of 1/`self.discretization`.
159 # Discretization is disabled for non-positive values.
160 policy_discretization: int = 32
161
162 def __call__(self, policy: chex.Array, mask: chex.Array,
163 learner_steps: int) -> chex.Array:
164 """A configurable fine tuning of a policy."""
165 chex.assert_equal_shape((policy, mask))
166 do_finetune = jnp.logical_and(self.from_learner_stepsfrom_learner_steps >= 0,
167 learner_steps > self.from_learner_stepsfrom_learner_steps)
168
169 return jnp.where(do_finetune, self.post_process_policy(policy, mask),
170 policy)
171
172 def post_process_policy(
173 self,
174 policy: chex.Array,
175 mask: chex.Array,
176 ) -> chex.Array:
177 """Unconditionally post process a given masked policy."""
178 chex.assert_equal_shape((policy, mask))
179 policy = self._threshold(policy, mask)
180 policy = self._discretize(policy)
181 return policy
182
183 def _threshold(self, policy: chex.Array, mask: chex.Array) -> chex.Array:
184 """Remove from the support the actions 'a' where policy(a) < threshold."""
185 chex.assert_equal_shape((policy, mask))
186 if self.policy_threshold <= 0:
187 return policy
188
189 mask = mask * (
190 # Values over the threshold.
191 (policy >= self.policy_threshold) +
192 # Degenerate case is when policy is less than threshold *everywhere*.
193 # In that case we just keep the policy as-is.
194 (jnp.max(policy, axis=-1, keepdims=True) < self.policy_threshold))
195 return mask * policy / jnp.sum(mask * policy, axis=-1, keepdims=True)
196
197 def _discretize(self, policy: chex.Array) -> chex.Array:
198 """Round all action probabilities to a multiple of 1/self.discretize."""
199 if self.policy_discretization <= 0:
200 return policy
201
202 # The unbatched/single policy case:
203 if len(policy.shape) == 1:
204 return self._discretize_single(policy)
205
206 # policy may be [B, A] or [T, B, A], etc. Thus add hk.BatchApply.
207 dims = len(policy.shape) - 1
208
209 # TODO(author18): avoid mixing vmap and BatchApply since the two could
210 # be folded into either a single BatchApply or a sequence of vmaps, but
211 # not the mix.
212 vmapped = jax.vmap(self._discretize_single)
213 policy = hk.BatchApply(vmapped, num_dims=dims)(policy)
214
215 return policy
216
217 def _discretize_single(self, mu: chex.Array) -> chex.Array:
218 """A version of self._discretize but for the unbatched data."""
219 # TODO(author18): try to merge _discretize and _discretize_single
220 # into one function that handles both batched and unbatched cases.
221 if len(mu.shape) == 2:
222 mu_ = jnp.squeeze(mu, axis=0)
223 else:
224 mu_ = mu
225 n_actions = mu_.shape[-1]
226 roundup = jnp.ceil(mu_ * self.policy_discretization).astype(jnp.int32)
227 result = jnp.zeros_like(mu_)
228 order = jnp.argsort(-mu_) # Indices of descending order.
229 weight_left = self.policy_discretization
230
231 def f_disc(i, order, roundup, weight_left, result):
232 x = jnp.minimum(roundup[order[i]], weight_left)
233 result = jax.numpy.where(weight_left >= 0, result.at[order[i]].add(x),
234 result)
235 weight_left -= x
236 return i + 1, order, roundup, weight_left, result
237
238 def f_scan_scan(carry, x):
239 i, order, roundup, weight_left, result = carry
240 i_next, order_next, roundup_next, weight_left_next, result_next = f_disc(
241 i, order, roundup, weight_left, result)
242 carry_next = (i_next, order_next, roundup_next, weight_left_next,
243 result_next)
244 return carry_next, x
245
246 (_, _, _, weight_left_next, result_next), _ = jax.lax.scan(
247 f_scan_scan,
248 init=(jnp.asarray(0), order, roundup, weight_left, result),
249 xs=None,
250 length=n_actions)
251
252 result_next = jnp.where(weight_left_next > 0,
253 result_next.at[order[0]].add(weight_left_next),
254 result_next)
255 if len(mu.shape) == 2:
256 result_next = jnp.expand_dims(result_next, axis=0)
257 return result_next / self.policy_discretization

260def _legal_policy(logits: chex.Array, legal_actions: chex.Array) -> chex.Array:
261 """A soft-max policy that respects legal_actions."""
262 chex.assert_equal_shape((logits, legal_actions))
263 # Fiddle a bit to make sure we don't generate NaNs or Inf in the middle.
264 l_min = logits.min(axis=-1, keepdims=True)
265 logits = jnp.where(legal_actions, logits, l_min)
266 logits -= logits.max(axis=-1, keepdims=True)
267 logits *= legal_actions
268 exp_logits = jnp.where(legal_actions, jnp.exp(logits),
269 0) # Illegal actions become 0.
270 exp_logits_sum = jnp.sum(exp_logits, axis=-1, keepdims=True)
271 return exp_logits / exp_logits_sum

274def legal_log_policy(logits: chex.Array,
275 legal_actions: chex.Array) -> chex.Array:
276 """Return the log of the policy on legal action, 0 on illegal action."""
277 chex.assert_equal_shape((logits, legal_actions))
278 # logits_masked has illegal actions set to -inf.
279 logits_masked = logits + jnp.log(legal_actions)
280 max_legal_logit = logits_masked.max(axis=-1, keepdims=True)
281 logits_masked = logits_masked - max_legal_logit
282 # exp_logits_masked is 0 for illegal actions.
283 exp_logits_masked = jnp.exp(logits_masked)
284
285 baseline = jnp.log(jnp.sum(exp_logits_masked, axis=-1, keepdims=True))
286 # Subtract baseline from logits. We do not simply return
287 # logits_masked - baseline
288 # because that has -inf for illegal actions, or
289 # legal_actions * (logits_masked - baseline)
290 # because that leads to 0 * -inf == nan for illegal actions.
291 log_policy = jnp.multiply(legal_actions,
292 (logits - max_legal_logit - baseline))
293 return log_policy

296def _player_others(player_ids: chex.Array, valid: chex.Array,
297 player: int) -> chex.Array:
298 """A vector of 1 for the current player and -1 for others.
299
300 Args:
301 player_ids: Tensor [...] containing player ids (0 <= player_id < N).
302 valid: Tensor [...] containing whether these states are valid.
303 player: The player id as int.
304
305 Returns:
306 player_other: is 1 for the current player and -1 for others [..., 1].
307 """
308 chex.assert_equal_shape((player_ids, valid))
309 current_player_tensor = (player_ids == player).astype(jnp.int32) # pytype: disable=attribute-error # numpy-scalars
310
311 res = 2 * current_player_tensor - 1
312 res = res * valid
313 return jnp.expand_dims(res, axis=-1)

316def _policy_ratio(pi: chex.Array, mu: chex.Array, actions_oh: chex.Array,
317 valid: chex.Array) -> chex.Array:
318 """Returns a ratio of policy pi/mu when selecting action a.
319
320 By convention, this ratio is 1 on non valid states
321 Args:
322 pi: the policy of shape [..., A].
323 mu: the sampling policy of shape [..., A].
324 actions_oh: a one-hot encoding of the current actions of shape [..., A].
325 valid: 0 if the state is not valid and else 1 of shape [...].
326
327 Returns:
328 pi/mu on valid states and 1 otherwise. The shape is the same
329 as pi, mu or actions_oh but without the last dimension A.
330 """
331 chex.assert_equal_shape((pi, mu, actions_oh))
332 chex.assert_shape((valid,), actions_oh.shape[:-1])
333
334 def _select_action_prob(pi):
335 return (jnp.sum(actions_oh * pi, axis=-1, keepdims=False) * valid +
336 (1 - valid))
337
338 pi_actions_prob = _select_action_prob(pi)
339 mu_actions_prob = _select_action_prob(mu)
340 return pi_actions_prob / mu_actions_prob

343def _where(pred: chex.Array, true_data: chex.ArrayTree,
344 false_data: chex.ArrayTree) -> chex.ArrayTree:
345 """Similar to jax.where but treats `pred` as a broadcastable prefix."""
346
347 def _where_one(t, f):
348 chex.assert_equal_rank((t, f))
349 # Expand the dimensions of pred if true_data and false_data are higher rank.
350 p = jnp.reshape(pred, pred.shape + (1,) * (len(t.shape) - len(pred.shape)))
351 return jnp.where(p, t, f)
352
353 return tree.tree_map(_where_one, true_data, false_data)

356def _has_played(valid: chex.Array, player_id: chex.Array,
357 player: int) -> chex.Array:
358 """Compute a mask of states which have a next state in the sequence."""
359 chex.assert_equal_shape((valid, player_id))
360
361 def _loop_has_played(carry, x):
362 valid, player_id = x
363 chex.assert_equal_shape((valid, player_id))
364
365 our_res = jnp.ones_like(player_id)
366 opp_res = carry
367 reset_res = jnp.zeros_like(carry)
368
369 our_carry = carry
370 opp_carry = carry
371 reset_carry = jnp.zeros_like(player_id)
372
373 # pyformat: disable
374 return _where(valid, _where((player_id == player),
375 (our_carry, our_res),
376 (opp_carry, opp_res)),
377 (reset_carry, reset_res))
378 # pyformat: enable
379
380 _, result = lax.scan(
381 f=_loop_has_played,
382 init=jnp.zeros_like(player_id[-1]),
383 xs=(valid, player_id),
384 reverse=True)
385 return result

388# V-Trace
389#
390# Custom implementation of VTrace to handle trajectories having a mix of
391# different player steps. The standard rlax.vtrace can't be applied here
392# out of the box because a trajectory could look like '121211221122'.
393
394
395def v_trace(
396 v: chex.Array,
397 valid: chex.Array,
398 player_id: chex.Array,
399 acting_policy: chex.Array,
400 merged_policy: chex.Array,
401 merged_log_policy: chex.Array,
402 player_others: chex.Array,
403 actions_oh: chex.Array,
404 reward: chex.Array,
405 player: int,
406 # Scalars below.
407 eta: float,
408 lambda_: float,
409 c: float,
410 rho: float,
411) -> Tuple[Any, Any, Any]:
412 """Custom VTrace for trajectories with a mix of different player steps."""
413 gamma = 1.0
414
415 has_played = _has_played(valid, player_id, player)
416
417 policy_ratio = _policy_ratio(merged_policy, acting_policy, actions_oh, valid)
418 inv_mu = _policy_ratio(
419 jnp.ones_like(merged_policy), acting_policy, actions_oh, valid)
420
421 eta_reg_entropy = (-eta *
422 jnp.sum(merged_policy * merged_log_policy, axis=-1) *
423 jnp.squeeze(player_others, axis=-1))
424 eta_log_policy = -eta * merged_log_policy * player_others
425
426 @chex.dataclass(frozen=True)
427 class LoopVTraceCarry:
428 """The carry of the v-trace scan loop."""
429 reward: chex.Array
430 # The cumulated reward until the end of the episode. Uncorrected (v-trace).
431 # Gamma discounted and includes eta_reg_entropy.
432 reward_uncorrected: chex.Array
433 next_value: chex.Array
434 next_v_target: chex.Array
435 importance_sampling: chex.Array
436
437 init_state_v_trace = LoopVTraceCarry(
438 reward=jnp.zeros_like(reward[-1]),
439 reward_uncorrected=jnp.zeros_like(reward[-1]),
440 next_value=jnp.zeros_like(v[-1]),
441 next_v_target=jnp.zeros_like(v[-1]),
442 importance_sampling=jnp.ones_like(policy_ratio[-1]))
443
444 def _loop_v_trace(carry: LoopVTraceCarry, x) -> Tuple[LoopVTraceCarry, Any]:
445 (cs, player_id, v, reward, eta_reg_entropy, valid, inv_mu, actions_oh,
446 eta_log_policy) = x
447
448 reward_uncorrected = (
449 reward + gamma * carry.reward_uncorrected + eta_reg_entropy)
450 discounted_reward = reward + gamma * carry.reward
451
452 # V-target:
453 our_v_target = (
454 v + jnp.expand_dims(
455 jnp.minimum(rho, cs * carry.importance_sampling), axis=-1) *
456 (jnp.expand_dims(reward_uncorrected, axis=-1) +
457 gamma * carry.next_value - v) + lambda_ * jnp.expand_dims(
458 jnp.minimum(c, cs * carry.importance_sampling), axis=-1) * gamma *
459 (carry.next_v_target - carry.next_value))
460
461 opp_v_target = jnp.zeros_like(our_v_target)
462 reset_v_target = jnp.zeros_like(our_v_target)
463
464 # Learning output:
465 our_learning_output = (
466 v + # value
467 eta_log_policy + # regularisation
468 actions_oh * jnp.expand_dims(inv_mu, axis=-1) *
469 (jnp.expand_dims(discounted_reward, axis=-1) + gamma * jnp.expand_dims(
470 carry.importance_sampling, axis=-1) * carry.next_v_target - v))
471
472 opp_learning_output = jnp.zeros_like(our_learning_output)
473 reset_learning_output = jnp.zeros_like(our_learning_output)
474
475 # State carry:
476 our_carry = LoopVTraceCarry(
477 reward=jnp.zeros_like(carry.reward),
478 next_value=v,
479 next_v_target=our_v_target,
480 reward_uncorrected=jnp.zeros_like(carry.reward_uncorrected),
481 importance_sampling=jnp.ones_like(carry.importance_sampling))
482 opp_carry = LoopVTraceCarry(
483 reward=eta_reg_entropy + cs * discounted_reward,
484 reward_uncorrected=reward_uncorrected,
485 next_value=gamma * carry.next_value,
486 next_v_target=gamma * carry.next_v_target,
487 importance_sampling=cs * carry.importance_sampling)
488 reset_carry = init_state_v_trace
489
490 # Invalid turn: init_state_v_trace and (zero target, learning_output)
491 # pyformat: disable
492 return _where(valid, # pytype: disable=bad-return-type # numpy-scalars
493 _where((player_id == player),
494 (our_carry, (our_v_target, our_learning_output)),
495 (opp_carry, (opp_v_target, opp_learning_output))),
496 (reset_carry, (reset_v_target, reset_learning_output)))
497 # pyformat: enable
498
499 _, (v_target, learning_output) = lax.scan(
500 f=_loop_v_trace,
501 init=init_state_v_trace,
502 xs=(policy_ratio, player_id, v, reward, eta_reg_entropy, valid, inv_mu,
503 actions_oh, eta_log_policy),
504 reverse=True)
505
506 return v_target, has_played, learning_output

509def get_loss_v(v_list: Sequence[chex.Array],
510 v_target_list: Sequence[chex.Array],
511 mask_list: Sequence[chex.Array]) -> chex.Array:
512 """Define the loss function for the critic."""
513 chex.assert_trees_all_equal_shapes(v_list, v_target_list)
514 # v_list and v_target_list come with a degenerate trailing dimension,
515 # which mask_list tensors do not have.
516 chex.assert_shape(mask_list, v_list[0].shape[:-1])
517 loss_v_list = []
518 for (v_n, v_target, mask) in zip(v_list, v_target_list, mask_list):
519 assert v_n.shape[0] == v_target.shape[0]
520
521 loss_v = jnp.expand_dims(
522 mask, axis=-1) * (v_n - lax.stop_gradient(v_target))**2
523 normalization = jnp.sum(mask)
524 loss_v = jnp.sum(loss_v) / (normalization + (normalization == 0.0))
525
526 loss_v_list.append(loss_v)
527 return sum(loss_v_list)

530def apply_force_with_threshold(decision_outputs: chex.Array, force: chex.Array,
531 threshold: float,
532 threshold_center: chex.Array) -> chex.Array:
533 """Apply the force with below a given threshold."""
534 chex.assert_equal_shape((decision_outputs, force, threshold_center))
535 can_decrease = decision_outputs - threshold_center > -threshold
536 can_increase = decision_outputs - threshold_center < threshold
537 force_negative = jnp.minimum(force, 0.0)
538 force_positive = jnp.maximum(force, 0.0)
539 clipped_force = can_decrease * force_negative + can_increase * force_positive
540 return decision_outputs * lax.stop_gradient(clipped_force)

543def renormalize(loss: chex.Array, mask: chex.Array) -> chex.Array:
544 """The `normalization` is the number of steps over which loss is computed."""
545 chex.assert_equal_shape((loss, mask))
546 loss = jnp.sum(loss * mask)
547 normalization = jnp.sum(mask)
548 return loss / (normalization + (normalization == 0.0))

551def get_loss_nerd(logit_list: Sequence[chex.Array],
552 policy_list: Sequence[chex.Array],
553 q_vr_list: Sequence[chex.Array],
554 valid: chex.Array,
555 player_ids: Sequence[chex.Array],
556 legal_actions: chex.Array,
557 importance_sampling_correction: Sequence[chex.Array],
558 clip: float = 100,
559 threshold: float = 2) -> chex.Array:
560 """Define the nerd loss."""
561 assert isinstance(importance_sampling_correction, list)
562 loss_pi_list = []
563 for k, (logit_pi, pi, q_vr, is_c) in enumerate(
564 zip(logit_list, policy_list, q_vr_list, importance_sampling_correction)):
565 assert logit_pi.shape[0] == q_vr.shape[0]
566 # loss policy
567 adv_pi = q_vr - jnp.sum(pi * q_vr, axis=-1, keepdims=True)
568 adv_pi = is_c * adv_pi # importance sampling correction
569 adv_pi = jnp.clip(adv_pi, a_min=-clip, a_max=clip)
570 adv_pi = lax.stop_gradient(adv_pi)
571
572 logits = logit_pi - jnp.mean(
573 logit_pi * legal_actions, axis=-1, keepdims=True)
574
575 threshold_center = jnp.zeros_like(logits)
576
577 nerd_loss = jnp.sum(
578 legal_actions *
579 apply_force_with_threshold(logits, adv_pi, threshold, threshold_center),
580 axis=-1)
581 nerd_loss = -renormalize(nerd_loss, valid * (player_ids == k))
582 loss_pi_list.append(nerd_loss)
583 return sum(loss_pi_list)

710class RNaDSolver(policy_lib.Policy):
711 """Implements a solver for the R-NaD Algorithm.
712
713 See https://arxiv.org/abs/2206.15378.
714
715 Define all networks. Derive losses & learning steps. Initialize the game
716 state and algorithmic variables.
717 """
718
719 def __init__(self, config: RNaDConfig):
720 self.config = config
721
722 # Learner and actor step counters.
723 self.learner_steps = 0
724 self.actor_steps = 0
725
726 self.init()

728 def init(self):
729 """Initialize the network and losses."""
730 # The random facilities for jax and numpy.
731 self._rngkey = jax.random.PRNGKey(self.config.seed)
732 self._np_rng = np.random.RandomState(self.config.seed)
733 # TODO(author16): serialize both above to get the fully deterministic behaviour.
734
735 # Create a game and an example of a state.
736 self._game = pyspiel.load_game(self.config.game_name)
737 self._ex_state = self._play_chance(self._game.new_initial_state())
738
739 # The network.
740 def network(
741 env_step: EnvStep
742 ) -> Tuple[chex.Array, chex.Array, chex.Array, chex.Array]:
743 mlp_torso = hk.nets.MLP(
744 self.config.policy_network_layers, activate_final=True
745 )
746 torso = mlp_torso(env_step.obs)
747
748 mlp_policy_head = hk.nets.MLP([self._game.num_distinct_actions()])
749 logit = mlp_policy_head(torso)
750
751 mlp_policy_value = hk.nets.MLP([1])
752 v = mlp_policy_value(torso)
753
754 pi = _legal_policy(logit, env_step.legal)
755 log_pi = legal_log_policy(logit, env_step.legal)
756 return pi, v, log_pi, logit
757
758 self.network = hk.without_apply_rng(hk.transform(network))
759
760 # The machinery related to updating parameters/learner.
761 self._entropy_schedule = EntropySchedule(
762 sizes=self.config.entropy_schedule_size,
763 repeats=self.config.entropy_schedule_repeats)
764 self._loss_and_grad = jax.value_and_grad(self.loss, has_aux=False)
765
766 # Create initial parameters.
767 env_step = self._state_as_env_step(self._ex_state)
768 key = self._next_rng_key() # Make sure to use the same key for all.
769 self.params = self.network.init(key, env_step)
770 self.params_target = self.network.init(key, env_step)
771 self.params_prev = self.network.init(key, env_step)
772 self.params_prev_ = self.network.init(key, env_step)
773
774 # Parameter optimizers.
775 self.optimizer = optax_optimizer(
776 self.params,
777 optax.chain(
778 optax.scale_by_adam(
779 eps_root=0.0,
780 **self.config.adam,
781 ), optax.scale(-self.config.learning_rate),
782 optax.clip(self.config.clip_gradient)))
783 self.optimizer_target = optax_optimizer(
784 self.params_target, optax.sgd(self.config.target_network_avg))

786 def loss(self, params: Params, params_target: Params, params_prev: Params,
787 params_prev_: Params, ts: TimeStep, alpha: float,
788 learner_steps: int) -> float:
789 rollout = jax.vmap(self.network.apply, (None, 0), 0)
790 pi, v, log_pi, logit = rollout(params, ts.env)
791
792 policy_pprocessed = self.config.finetune(pi, ts.env.legal, learner_steps)
793
794 _, v_target, _, _ = rollout(params_target, ts.env)
795 _, _, log_pi_prev, _ = rollout(params_prev, ts.env)
796 _, _, log_pi_prev_, _ = rollout(params_prev_, ts.env)
797 # This line creates the reward transform log(pi(a|x)/pi_reg(a|x)).
798 # For the stability reasons, reward changes smoothly between iterations.
799 # The mixing between old and new reward transform is a convex combination
800 # parametrised by alpha.
801 log_policy_reg = log_pi - (alpha * log_pi_prev + (1 - alpha) * log_pi_prev_)
802
803 v_target_list, has_played_list, v_trace_policy_target_list = [], [], []
804 for player in range(self._game.num_players()):
805 reward = ts.actor.rewards[:, :, player] # [T, B, Player]
806 v_target_, has_played, policy_target_ = v_trace(
807 v_target,
808 ts.env.valid,
809 ts.env.player_id,
810 ts.actor.policy,
811 policy_pprocessed,
812 log_policy_reg,
813 _player_others(ts.env.player_id, ts.env.valid, player),
814 ts.actor.action_oh,
815 reward,
816 player,
817 lambda_=1.0,
818 c=self.config.c_vtrace,
819 rho=np.inf,
820 eta=self.config.eta_reward_transform)
821 v_target_list.append(v_target_)
822 has_played_list.append(has_played)
823 v_trace_policy_target_list.append(policy_target_)
824 loss_v = get_loss_v([v] * self._game.num_players(), v_target_list,
825 has_played_list)
826
827 is_vector = jnp.expand_dims(jnp.ones_like(ts.env.valid), axis=-1)
828 importance_sampling_correction = [is_vector] * self._game.num_players()
829 # Uses v-trace to define q-values for Nerd
830 loss_nerd = get_loss_nerd(
831 [logit] * self._game.num_players(), [pi] * self._game.num_players(),
832 v_trace_policy_target_list,
833 ts.env.valid,
834 ts.env.player_id,
835 ts.env.legal,
836 importance_sampling_correction,
837 clip=self.config.nerd.clip,
838 threshold=self.config.nerd.beta)
839 return loss_v + loss_nerd # pytype: disable=bad-return-type # numpy-scalars

841 @functools.partial(jax.jit, static_argnums=(0,))
842 def update_parameters(
843 self,
844 params: Params,
845 params_target: Params,
846 params_prev: Params,
847 params_prev_: Params,
848 optimizer: Optimizer,
849 optimizer_target: Optimizer,
850 timestep: TimeStep,
851 alpha: float,
852 learner_steps: int,
853 update_target_net: bool):
854 """A jitted pure-functional part of the `step`."""
855 loss_val, grad = self._loss_and_grad(params, params_target, params_prev,
856 params_prev_, timestep, alpha,
857 learner_steps)
858 # Update `params`` using the computed gradient.
859 params = optimizer(params, grad)
860 # Update `params_target` towards `params`.
861 params_target = optimizer_target(
862 params_target, tree.tree_map(lambda a, b: a - b, params_target, params))
863
864 # Rolls forward the prev and prev_ params if update_target_net is 1.
865 # pyformat: disable
866 params_prev, params_prev_ = jax.lax.cond(
867 update_target_net,
868 lambda: (params_target, params_prev),
869 lambda: (params_prev, params_prev_))
870 # pyformat: enable
871
872 logs = {
873 "loss": loss_val,
874 }
875 return (params, params_target, params_prev, params_prev_, optimizer,
876 optimizer_target), logs

953 def _state_as_env_step(self, state: pyspiel.State) -> EnvStep:
954 # A terminal state must be communicated to players, however since
955 # it's a terminal state things like the state_representation or
956 # the set of legal actions are meaningless and only needed
957 # for the sake of creating well a defined trajectory tensor.
958 # Therefore the code below:
959 # - extracts the rewards
960 # - if the state is terminal, uses a dummy other state for other fields.
961 rewards = np.array(state.returns(), dtype=np.float64)
962
963 valid = not state.is_terminal()
964 if not valid:
965 state = self._ex_state
966
967 if self.config.state_representation == StateRepresentation.OBSERVATION:
968 obs = state.observation_tensor()
969 elif self.config.state_representation == StateRepresentation.INFO_SET:
970 obs = state.information_state_tensor()
971 else:
972 raise ValueError(
973 f"Invalid StateRepresentation: {self.config.state_representation}.")
974
975 # TODO(author16): clarify the story around rewards and valid.
976 return EnvStep(
977 obs=np.array(obs, dtype=np.float64),
978 legal=np.array(state.legal_actions_mask(), dtype=np.int8),
979 player_id=np.array(state.current_player(), dtype=np.float64),
980 valid=np.array(valid, dtype=np.float64),
981 rewards=rewards)

1021 def collect_batch_trajectory(self) -> TimeStep:
1022 states = [
1023 self._play_chance(self._game.new_initial_state())
1024 for _ in range(self.config.batch_size)
1025 ]
1026 timesteps = []
1027
1028 env_step = self._batch_of_states_as_env_step(states)
1029 for _ in range(self.config.trajectory_max):
1030 prev_env_step = env_step
1031 a, actor_step = self.actor_step(env_step)
1032
1033 states = self._batch_of_states_apply_action(states, a)
1034 env_step = self._batch_of_states_as_env_step(states)
1035 timesteps.append(
1036 TimeStep(
1037 env=prev_env_step,
1038 actor=ActorStep(
1039 action_oh=actor_step.action_oh,
1040 policy=actor_step.policy,
1041 rewards=env_step.rewards),
1042 ))
1043 # Concatenate all the timesteps together to form a single rollout [T, B, ..]
1044 return jax.tree_util.tree_map(lambda *xs: np.stack(xs, axis=0), *timesteps)

1051 def _batch_of_states_apply_action(
1052 self, states: Sequence[pyspiel.State],
1053 actions: chex.Array) -> Sequence[pyspiel.State]:
1054 """Apply a batch of `actions` to a parallel list of `states`."""
1055 for state, action in zip(states, list(actions)):
1056 if not state.is_terminal():
1057 self.actor_steps += 1
1058 state.apply_action(action)
1059 self._play_chance(state)

1062 def _play_chance(self, state: pyspiel.State) -> pyspiel.State:
1063 """Plays the chance nodes until we end up at another type of node.

1065 Args:
1066 state: to be updated until it does not correspond to a chance node.
1067 Returns:
1068 The same input state object, but updated. The state is returned
1069 only for convenience, to allow chaining function calls.
1070 """
1071 while state.is_chance_node():
1072 chance_outcome, chance_proba = zip(*state.chance_outcomes())
1073 action = self._np_rng.choice(chance_outcome, p=chance_proba)
1074 state.apply_action(action)
