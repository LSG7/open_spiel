24// This file contains data structures used in imperfect information games.
25// Specifically, we implement an infostate tree, a representation of a game
26// from the perspective of an acting player.
27//
28// The information-state tree [1] contains information states, which describe
29// where the player is a) acting, b) getting observations, or c) receiving
30// terminal utilities (when the game ends). See `InfostateNodeType` for more
31// details.
32//
33// The tree can be constructed with a depth limit, so we make a distinction
34// between leaf nodes and non-leaf nodes. All terminal nodes are leaf nodes.
35//
36// The identification of infostates is based on strings from an information
37// state observer, i.e. one that is constructed using `kInfoStateObsType`.
38//
39// As algorithms typically need to store information associated to specific
40// nodes of the tree, we provide following indexing mechanisms (see the classes
41// below for more details):
42//
43// - `DecisionId` refers to a decision infostate where the player acts.
44// - `SequenceId` refers to an observation infostate that follows the decision
45// infostate after some action.
46// - `LeafId` refers to an infostate node which is a leaf.
47//
48// All of these ids are very cheap (they are just typed `size_t`s).
49// They can be used to get a pointer to the corresponding infostate node.
50//
51// To enable some algorithmic optimizations we construct the trees "balanced".
52// We call a _balanced_ tree one which has all leaf nodes at the same depth.
53// To make the tree balanced, we may need to pad "dummy" observation nodes as
54// prefixes for the (previously too shallow) leafs. This is not too expensive,
55// as most games are balanced by default due to game rules.
56//
57// [1]: Rethinking Formal Models of Partially Observable Multiagent Decision
58// Making https://arxiv.org/abs/1906.11110
59

63// To categorize infostate nodes we use nomenclature from [2]:
64//
65// - In _decision nodes_, the acting player selects actions.
66// - In _observation nodes_ the acting player receives observations.
67// They can correspond to State that is a chance node, or opponent's node.
68// Importantly, they can correspond also to the acting player's node,
69// as the player may have discovered something as a result of its action
70// in the previous decision node. (This is especially important for the tree
71// construction in simultaneous-move games).
72// - Additionally, we introduce _terminal nodes_, which correspond to a single
73// terminal history.
74//
75// The terminal nodes store player's utility as well as cumulative chance reach
76// probability.
77//
78// [2]: Faster Game Solving via Predictive Blackwell Approachability:
79// Connecting Regret Matching and Mirror Descent
80// https://arxiv.org/abs/2007.14358
81enum InfostateNodeType {
82 kDecisionInfostateNode,
83 kObservationInfostateNode,
84 kTerminalInfostateNode
85};
86
87// Representing the game via infostates leads actually to a graph structure
88// of a forest (a collection of trees), as the player may be acting for the
89// first time in distinct situations. We trivially make it into a proper tree
90// by introducing a "dummy" root node, which we set as an observation node.
91// It can be interpreted as "the player observes the start of the game".
92// This node also corresponds to the empty sequence.
93// Following is the infostate string for this node.
94constexpr const char* kDummyRootNodeInfostate = "(root)";
95
96// Sometimes we need to create infostate nodes that do not have a corresponding
97// game State, and therefore we cannot retrieve their string representations.
98// This happens in simultaneous move games or if we rebalance game trees.
99constexpr const char* kFillerInfostate = "(fill)";

106// An implementation detail - Not to be used directly.
107//
108// We use various indexing schemes (SequenceId, DecisionId, LeafId) to access
109// specific nodes in the tree. Not all nodes can have an Id defined, for example
110// a DecisionId is not defined for decision nodes. In this case they will
111// default to the following value.
112constexpr size_t kUndefinedNodeId = -1; // This is a large number.
113
114// An implementation detail - Not to be used directly.
115//
116// Create an indexing of specific infostate nodes.
117//
118// In release-mode the implementation is as cheap as the underlying size_t
119// identifier. Therefore it is preferable to pass the Ids by copy and not
120// by pointers / references.
121// Most importantly in debug-mode we add checks to make sure that we are using
122// the ids on appropriate trees and we do not try to index any opponents' trees.
123//
124// We use CRTP as it allows us to reuse the implementation for derived classes.
125template <class Self>
126class NodeId {
127 size_t identifier_ = kUndefinedNodeId;
128#ifndef NDEBUG // Allow additional automatic debug-time checks.
129 const InfostateTree* tree_ = nullptr;
130
131 public:
132 NodeId(size_t id_value, const InfostateTree* tree_ptr)
133 : identifier_(id_value), tree_(tree_ptr) {}
134 NodeId& operator=(Self&& rhs) {
135 SPIEL_CHECK_TRUE(
136 tree_ == rhs.tree_ ||
137 // The NodeId may be uninitialized, so allow to copy the rhs tree.
138 tree_ == nullptr && rhs.tree_ != nullptr);
139 identifier_ = rhs.id();
140 tree_ == rhs.tree_;
141 return *this;
142 }
143 bool operator==(const Self& rhs) const {
144 SPIEL_CHECK_EQ(tree_, rhs.tree_);
145 return id() == rhs.id();
146 }
147 bool operator!=(const Self& rhs) const {
148 SPIEL_CHECK_EQ(tree_, rhs.tree_);
149 return id() != rhs.id();
150 }
151 bool BelongsToTree(const InfostateTree* other) const {
152 return tree_ == other;
153 }
154#else
155
156 public:
157 // Do not save the tree pointer, but expose the same interface
158 // so it's easy to use.
159 NodeId(size_t id_value, const InfostateTree*) : identifier_(id_value) {}
160 Self& operator=(Self&& rhs) {
161 identifier_ = rhs.id();
162 return this;
163 }
164 bool operator==(const Self& rhs) const { return id() == rhs.id(); }
165 bool operator!=(const Self& rhs) const { return id() != rhs.id(); }
166 // BelongsToTree is not implemented on purpose:
167 // It must not be called in release mode -- used only by DCHECK statements.
168#endif
169 constexpr NodeId() {}
170 size_t id() const {
171 SPIEL_CHECK_NE(identifier_, kUndefinedNodeId);
172 return identifier_;
173 }
174 bool is_undefined() const { return identifier_ == kUndefinedNodeId; }
175 void next() {
176 SPIEL_CHECK_NE(identifier_, kUndefinedNodeId);
177 ++identifier_;
178 }
179};

181} // namespace internal

183// `SequenceId` refers to an observation infostate that follows the decision
184// infostate after following some action. It indexes the decision space of
185// an agent, and its strategy can formulated in terms of values associated with
186// the agent's sequences. See `TreeplexVector` for more details.
187// The smallest sequence ids correspond to the deepest nodes and the highest
188// value corresponds to the empty sequence.
189class SequenceId final : public internal::NodeId<SequenceId> {
190 using NodeId<SequenceId>::NodeId;
191};
192// When the tree is still under construction and a node doesn't
193// have a final sequence id assigned yet, we use this value.
194constexpr SequenceId kUndefinedSequenceId = SequenceId();
195
196// `DecisionId` refers to an infostate node where the player acts,
197// i.e. an infostate node with the type `kDecisionInfostateNode`.
198class DecisionId final : public internal::NodeId<DecisionId> {
199 using NodeId<DecisionId>::NodeId;
200};
201// When a node isn't a decision infostate, we use this value instead.
202constexpr DecisionId kUndefinedDecisionId = DecisionId();
203
204// `LeafId` refers to an infostate node which is a leaf. Note that this can be
205// an arbitrary infostate node type. A kTerminalInfostateNode is always
206// a leaf node.
207// Note that leaf decision nodes do not have assigned any `DecisionId`, and
208// similarly leaf observation nodes do not have assigned any `SequenceId`.
209class LeafId final : public internal::NodeId<LeafId> {
210 using internal::NodeId<LeafId>::NodeId;
211};
212// When a node isn't a leaf, we use this value instead.
213constexpr LeafId kUndefinedLeafId = LeafId();

219template <typename T>
220class LeafVector;

222class DecisionVector;

225// A convenience iterator over a contiguous range of node ids.
226template <class Id>
227class RangeIterator {
228 size_t id_;
229 const InfostateTree* tree_;
230
231 public:
232 RangeIterator(size_t id, const InfostateTree* tree) : id_(id), tree_(tree) {}
233 RangeIterator& operator++() {
234 ++id_;
235 return *this;
236 }
237 bool operator!=(const RangeIterator& other) const {
238 return id_ != other.id_ || tree_ != other.tree_;
239 }
240 Id operator*() { return Id(id_, tree_); }
241};
242template <class Id>
243class Range {
244 const size_t start_;
245 const size_t end_;
246 const InfostateTree* tree_;
247
248 public:
249 Range(size_t start, size_t end, const InfostateTree* tree)
250 : start_(start), end_(end), tree_(tree) {
251 SPIEL_CHECK_LE(start_, end_);
252 }
253 RangeIterator<Id> begin() const { return RangeIterator<Id>(start_, tree_); }
254 RangeIterator<Id> end() const { return RangeIterator<Id>(end_, tree_); }
255};

445// Iterate over a vector of unique pointers, but expose only the raw pointers.
446template <class T>
447class VecWithUniquePtrsIterator {
448 int pos_;
449 const std::vector<std::unique_ptr<T>>& vec_;
450
451 public:
452 explicit VecWithUniquePtrsIterator(const std::vector<std::unique_ptr<T>>& vec,
453 int pos = 0)
454 : pos_(pos), vec_(vec) {}
455 VecWithUniquePtrsIterator& operator++() {
456 pos_++;
457 return *this;
458 }
459 bool operator==(VecWithUniquePtrsIterator other) const {
460 return pos_ == other.pos_;
461 }
462 bool operator!=(VecWithUniquePtrsIterator other) const {
463 return !(*this == other);
464 }
465 T* operator*() { return vec_[pos_].get(); }
466 VecWithUniquePtrsIterator begin() const { return *this; }
467 VecWithUniquePtrsIterator end() const {
468 return VecWithUniquePtrsIterator(vec_, vec_.size());
469 }
470};

472class InfostateNode final {
473 // Note that all of the following members are const or they should be const.
474 // However we can't make all of them const during the node construction
475 // because they might be computed only after the whole tree is built.
476 private:
477 // Reference to the tree this node belongs to. This reference has a valid
478 // lifetime, as it is allocated once on the heap and never moved.
479 const InfostateTree& tree_;
480 // Pointer to the parent node. Null for the root node.
481 // This is not const so that we can change it when we rebalance the tree.
482 /*const*/ InfostateNode* parent_;
483 // Position of this node in the parent's children, i.e. it holds that
484 // parent_->children_.at(incoming_index_).get() == this.
485 //
486 // For decision nodes this corresponds also to the
487 // State::LegalActions(player_).at(incoming_index_)
488 //
489 // This is not const so that we can change it when we rebalance the tree.
490 /*const*/ int incoming_index_;
491 // Type of the node.
492 const InfostateNodeType type_;
493 // Identifier of the infostate.
494 const std::string infostate_string_;
495 // Decision identifier of this node.
496 // This is not const as the ids are assigned after the tree is built.
497 /*const*/ DecisionId decision_id_ = kUndefinedDecisionId;
498 // Sequence identifier of this node.
499 // The first is the parent sequence of the infostate, while the last
500 // two sequence IDs represent the sequence id of the first and last action + 1
501 // at the infostate node. Because sequences assigned to an infostate
502 // are contiguous, we don't need to store all intermediate sequence IDs.
503 // We can thus use a Range iterable to make looping frictionless.
504 // This is not const as the ids can be assigned only after the tree is built.
505 /*const*/ SequenceId sequence_id_ = kUndefinedSequenceId;
506 /*const*/ SequenceId start_sequence_id_ = kUndefinedSequenceId;
507 /*const*/ SequenceId end_sequence_id_ = kUndefinedSequenceId;
508 // Sequence identifier of this node.
509 // This is not const as the ids are assigned after the tree is rebalanced.
510 /*const*/ LeafId leaf_id_ = kUndefinedLeafId;
511 // Utility of terminal state corresponding to the terminal infostate node.
512 // If the node is not terminal, the value is NaN.
513 const double terminal_utility_;
514 // Cumulative product of chance probabilities leading up to a terminal node.
515 // If the node is not terminal, the value is NaN.
516 const double terminal_chn_reach_prob_;
517 // Depth of the node, i.e. number of edges on the path from the root.
518 // Note that depth does not necessarily correspond to the MoveNumber()
519 // of corresponding states.
520 // This is not const because tree rebalancing can change this value.
521 /*const*/ size_t depth_;
522 // Children infostate nodes. Notice the node owns its children.
523 // This is not const so that we can add children.
524 /*const*/ std::vector<std::unique_ptr<InfostateNode>> children_;
525 // Store States that correspond to a leaf node.
526 // This is not const so that we can add corresponding states.
527 /*const*/ std::vector<std::unique_ptr<State>> corresponding_states_;
528 // Store chance reach probs for States that correspond to a leaf node.
529 // This is not const so that we can add corresponding reaches.
530 /*const*/ std::vector<double> corresponding_ch_reaches_;
531 // Stored only for decision nodes.
532 const std::vector<Action> legal_actions_;
533 // Stored only for terminal nodes.
534 const std::vector<Action> terminal_history_;

655// Create a common TreeVector container that can be indexed
656// with the respective NodeIds. This is later specialized for the individual
657// indexing of the trees.
658template <typename T, typename Id>
659class TreeVector {
660 const InfostateTree* tree_;
661 std::vector<T> vec_;
662
663 public:
664 explicit TreeVector(const InfostateTree* tree)
665 : tree_(tree), vec_(tree_->num_ids(Id(kUndefinedNodeId, tree))) {}
666 TreeVector(const InfostateTree* tree, std::vector<T> vec)
667 : tree_(tree), vec_(std::move(vec)) {
668 SPIEL_CHECK_EQ(tree_->num_ids(Id(kUndefinedNodeId, tree)), vec_.size());
669 }
670 T& operator[](const Id& id) {
671 SPIEL_DCHECK_TRUE(id.BelongsToTree(tree_));
672 SPIEL_DCHECK_LE(0, id.id());
673 SPIEL_DCHECK_LT(id.id(), vec_.size());
674 return vec_[id.id()];
675 }
676 const T& operator[](const Id& id) const {
677 SPIEL_DCHECK_TRUE(id.BelongsToTree(tree_));
678 SPIEL_DCHECK_LE(0, id.id());
679 SPIEL_DCHECK_LT(id.id(), vec_.size());
680 return vec_[id.id()];
681 }
682 std::ostream& operator<<(std::ostream& os) const {
683 return os << vec_ << " (for player " << tree_->acting_player() << ')';
684 }
685 size_t size() const { return vec_.size(); }
686 Range<Id> range() { return Range<Id>(0, vec_.size(), tree_); }
687 Range<Id> range(size_t from, size_t to) { return Range<Id>(from, to, tree_); }
688 const InfostateTree* tree() const { return tree_; }
689};
