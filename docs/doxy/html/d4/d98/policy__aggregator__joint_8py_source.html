15"""Policy aggregator.

17A joint policy is a list of `num_players` policies.
18This files enables to compute mixtures of such joint-policies to get a new

27def _aggregate_at_state(joint_policies, state, player):
28 """Returns {action: prob} for `player` in `state` for all joint policies.
29
30 Args:
31 joint_policies: List of joint policies.
32 state: Openspiel State
33 player: Current Player
34
35 Returns:
36 {action: prob} for `player` in `state` for all joint policies.
37 """
38 return [
39 joint_policy[player].action_probabilities(state, player_id=player)
40 for joint_policy in joint_policies
41 ]

121 def aggregate(self, pids, joint_policies, weights):
122 r"""Computes the weighted-mixture of the joint policies.
123
124 Let P of shape [num_players] be the joint policy, and W some weights.
125 Let N be the number of policies (i.e. len(policies)).
126 We return the policy P' such that for all state `s`:
127
128 P[s] ~ \sum_{i=0}^{N-1} (policies[i][player(s)](s) * weights[i] *
129 reach_prob(s, policies[i]))
130
131 Arguments:
132 pids: Spiel player ids of the players the strategies belong to.
133 joint_policies: List of list of policies (One list per joint strategy)
134 weights: List of weights to attach to each joint strategy.
135
136 Returns:
137 A _DictPolicy, a callable object representing the policy.
138 """
139 aggr_policies = []
140 self._joint_policies = joint_policies
141
142 # To do(pmuller): We should be able to do a single recursion.
143 for pid in pids:
144 aggr_policies.append(self._sub_aggregate(pid, weights))
145 return _DictPolicy(self._game, aggr_policies)

147 def _sub_aggregate(self, pid, weights):
148 """Aggregate the list of policies for one player.
149
150 Arguments:
151 pid: Spiel player id of the player the strategies belong to.
152 weights: List of weights to attach to each joint strategy.
153
154 Returns:
155 A _DictPolicy, a callable object representing the policy.
156 """
157
158 # string of state -> probs list
159 self._policy = {}
160
161 state = self._game.new_initial_state()
162 self._rec_aggregate(pid, state, copy.deepcopy(weights))
163
164 # Now normalize
165 for key in self._policy:
166 actions, probabilities = zip(*self._policy[key].items())
167 new_probs = [prob + self._epsilon for prob in probabilities]
168 denom = sum(new_probs)
169 for i in range(len(actions)):
170 self._policy[key][actions[i]] = new_probs[i] / denom
171 return self._policy

173 def _rec_aggregate(self, pid, state, my_reaches):
174 """Recursively traverse game tree to compute aggregate policy."""
175 if state.is_terminal():
176 return
177
178 if state.is_simultaneous_node():
179 policies = _aggregate_at_state(self._joint_policies, state, pid)
180 state_key = self._state_key(state, pid)
181
182 self._policy[state_key] = {}
183 used_moves = state.legal_actions(pid)
184
185 for uid in used_moves:
186 new_reaches = copy.deepcopy(my_reaches)
187 for i in range(len(policies)):
188 # compute the new reach for each policy for this action
189 new_reaches[i] *= policies[i].get(uid, 0)
190 # add reach * prob(a) for this policy to the computed policy
191 if uid in self._policy[state_key].keys():
192 self._policy[state_key][uid] += new_reaches[i]
193 else:
194 self._policy[state_key][uid] = new_reaches[i]
195
196 num_players = self._game.num_players()
197 all_other_used_moves = []
198 for player in range(num_players):
199 if player != pid:
200 all_other_used_moves.append(state.legal_actions(player))
201
202 other_joint_actions = itertools.product(*all_other_used_moves)
203
204 # enumerate every possible other-agent actions for next-state
205 for other_joint_action in other_joint_actions:
206 for uid in used_moves:
207 new_reaches = copy.deepcopy(my_reaches)
208 for i in range(len(policies)):
209 # compute the new reach for each policy for this action
210 new_reaches[i] *= policies[i].get(uid, 0)
211
212 joint_action = list(
213 other_joint_action[:pid] + (uid,) + other_joint_action[pid:]
214 )
215 new_state = state.clone()
216 new_state.apply_actions(joint_action)
217 self._rec_aggregate(pid, new_state, new_reaches)
218 return
219
220 if state.is_chance_node():
221 for action in state.legal_actions():
222 new_state = state.child(action)
223 self._rec_aggregate(pid, new_state, my_reaches)
224 return
225
226 current_player = state.current_player()
227
228 state_key = self._state_key(state, current_player)
229 action_probabilities_list = _aggregate_at_state(self._joint_policies, state,
230 current_player)
231 if pid == current_player:
232 # update the current node
233 # will need the observation to query the policies
234 if state_key not in self._policy:
235 self._policy[state_key] = {}
236
237 for action in state.legal_actions():
238 new_reaches = copy.deepcopy(my_reaches)
239 if pid == current_player:
240 for idx, state_action_probs in enumerate(action_probabilities_list):
241 # compute the new reach for each policy for this action
242 new_reaches[idx] *= state_action_probs.get(action, 0)
243 # add reach * prob(a) for this policy to the computed policy
244 if action in self._policy[state_key].keys():
245 self._policy[state_key][action] += new_reaches[idx]
246 else:
247 self._policy[state_key][action] = new_reaches[idx]
248
249 # recurse
250 self._rec_aggregate(pid, state.child(action), new_reaches)
