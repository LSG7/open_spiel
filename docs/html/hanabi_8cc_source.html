149 std::vector<Action> OpenSpielHanabiState::LegalActions() const {
150  if (IsTerminal()) {
151  return {};
152  } else if (IsChanceNode()) {
153  auto outcomes_and_probs = state_.ChanceOutcomes();
154  const int n = outcomes_and_probs.first.size();
155  std::vector<Action> chance_outcomes;
156  chance_outcomes.reserve(n);
157  for (int i = 0; i < n; ++i) {
158  chance_outcomes.emplace_back(
159  game_->HanabiGame().GetChanceOutcomeUid(outcomes_and_probs.first[i]));
160  }
161  return chance_outcomes;
162  } else {
163  auto moves = state_.LegalMoves(CurrentPlayer());
164  std::vector<Action> actions;
165  actions.reserve(moves.size());
166  for (auto m : moves) actions.push_back(game_->HanabiGame().GetMoveUid(m));
167  return actions;
168  }
169 }

187 void OpenSpielHanabiState::DoApplyAction(Action action) {
188  auto move = IsChanceNode() ? game_->HanabiGame().GetChanceOutcome(action)
189  : game_->HanabiGame().GetMove(action);
190  if (state_.MoveIsLegal(move)) {
191  if (!IsChanceNode()) prev_state_score_ = state_.Score();
192  state_.ApplyMove(move);
193  } else {
194  SpielFatalError(absl::StrCat("Invalid move ", move.ToString()));
195  }
