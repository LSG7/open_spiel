29 // Implements the recursive traversal using a general way to access the
30 // player's policies via a function that takes as arguments the player id and
31 // information state.
32 // We have a special case for the case where we can get a policy just from the
33 // InfostateString as that gives us a 2x speedup.
34 std::vector<double> ExpectedReturnsImpl(
35  const State& state,
36  const std::function<ActionsAndProbs(Player, const std::string&)>&
37  policy_func,
38  int depth_limit,
39  float prob_cut_threshold) {
40  if (state.IsTerminal() || depth_limit == 0) {
41  return state.Rewards();
42  }
43 
44  int num_players = state.NumPlayers();
45  std::vector<double> values(num_players, 0.0);
46  if (state.IsChanceNode()) {
47  ActionsAndProbs action_and_probs = state.ChanceOutcomes();
48  for (const auto& action_and_prob : action_and_probs) {
49  if (action_and_prob.second <= prob_cut_threshold) continue;
50  std::unique_ptr<State> child = state.Child(action_and_prob.first);
51  std::vector<double> child_values =
52  ExpectedReturnsImpl(
53  *child, policy_func, depth_limit - 1, prob_cut_threshold);
54  for (auto p = Player{0}; p < num_players; ++p) {
55  values[p] += action_and_prob.second * child_values[p];
56  }
57  }
58  } else if (state.IsSimultaneousNode()) {
59  // Walk over all the joint actions, and weight by the product of
60  // probabilities to choose them.
61  values = state.Rewards();
62  auto smstate = dynamic_cast<const SimMoveState*>(&state);
63  SPIEL_CHECK_TRUE(smstate != nullptr);
64  std::vector<ActionsAndProbs> state_policies(num_players);
65  for (auto p = Player{0}; p < num_players; ++p) {
66  state_policies[p] = policy_func(p, state.InformationStateString(p));
67  if (state_policies[p].empty()) {
68  SpielFatalError("Error in ExpectedReturnsImpl; infostate not found.");
69  }
70  }
71  for (const Action flat_action : smstate->LegalActions()) {
72  std::vector<Action> actions =
73  smstate->FlatJointActionToActions(flat_action);
74  double joint_action_prob = 1.0;
75  for (auto p = Player{0}; p < num_players; ++p) {
76  double player_action_prob = GetProb(state_policies[p], actions[p]);
77  SPIEL_CHECK_GE(player_action_prob, 0.0);
78  SPIEL_CHECK_LE(player_action_prob, 1.0);
79  joint_action_prob *= player_action_prob;
80  if (joint_action_prob <= prob_cut_threshold) {
81  break;
82  }
83  }
84 
85  if (joint_action_prob > prob_cut_threshold) {
86  std::unique_ptr<State> child = state.Clone();
87  child->ApplyActions(actions);
88  std::vector<double> child_values =
89  ExpectedReturnsImpl(
90  *child, policy_func, depth_limit - 1, prob_cut_threshold);
91  for (auto p = Player{0}; p < num_players; ++p) {
92  values[p] += joint_action_prob * child_values[p];
93  }
94  }
95  }
96  } else {
97  // Turn-based decision node.
98  Player player = state.CurrentPlayer();
99  ActionsAndProbs state_policy =
100  policy_func(player, state.InformationStateString());
101  if (state_policy.empty()) {
102  SpielFatalError("Error in ExpectedReturnsImpl; infostate not found.");
103  }
104  values = state.Rewards();
105  float total_prob = 0.0;
106  for (const Action action : state.LegalActions()) {
107  std::unique_ptr<State> child = state.Child(action);
108  // GetProb can return -1 for legal actions not in the policy. We treat
109  // these as having zero probability, but check that at least some actions
110  // have positive probability.
111  double action_prob = GetProb(state_policy, action);
112  SPIEL_CHECK_LE(action_prob, 1.0);
113  if (action_prob > prob_cut_threshold) {
114  SPIEL_CHECK_GE(action_prob, 0.0);
115  total_prob += action_prob;
116  std::vector<double> child_values =
117  ExpectedReturnsImpl(
118  *child, policy_func, depth_limit - 1, prob_cut_threshold);
119  for (auto p = Player{0}; p < num_players; ++p) {
120  values[p] += action_prob * child_values[p];
121  }
122  }
123  }
124  // Check that there is a least some positive mass on at least one action.
125  // Consider using: SPIEL_CHECK_FLOAT_EQ(total_prob, 1.0);
126  SPIEL_CHECK_GT(total_prob, 0.0);
127  }
128  SPIEL_CHECK_EQ(values.size(), state.NumPlayers());
129  return values;
130 }

206  SPIEL_CHECK_GE(action_prob, 0.0);

222 std::vector<double> ExpectedReturnsOfDeterministicPoliciesFromSeedsImpl(
223  const State& state,
224  const std::vector<int>& policy_seeds,
225  const std::vector<const Policy*>& policies) {
226  if (state.IsTerminal()) {
227  return state.Rewards();
228  }
229  const int num_players = state.NumPlayers();
230  std::vector<double> values(num_players, 0.0);
231  if (state.IsSimultaneousNode()) {
232  SpielFatalError("Simultaneous not implemented.");
233  } else if (state.IsChanceNode()) {
234  ActionsAndProbs actions_and_probs = state.ChanceOutcomes();
235  for (const auto& action_and_prob : actions_and_probs) {
236  if (action_and_prob.second <= 0.0) continue;
237  std::unique_ptr<const State> child = state.Child(action_and_prob.first);
238  const std::vector<double> child_values = (
239  ExpectedReturnsOfDeterministicPoliciesFromSeedsImpl(
240  *child, policy_seeds, policies));
241  for (auto p = Player{0}; p < num_players; ++p) {
242  values[p] += action_and_prob.second * child_values[p];
243  }
244  }
245  } else {
246  // Get information state string.
247  std::string info_state_string = state.InformationStateString();
248  const int player = state.CurrentPlayer();
249 
250  // Search for policy in policies.
251  ActionsAndProbs actions_and_probs = {};
252  for (const auto& policy : policies) {
253  actions_and_probs = policy->GetStatePolicy(state);
254  if (!actions_and_probs.empty()) {
255  break;
256  }
257  }
258  if (!actions_and_probs.empty()) {
259  for (const auto& action_and_prob : actions_and_probs) {
260  if (action_and_prob.second <= 0.0) continue;
261  std::unique_ptr<const State> child = state.Child(action_and_prob.first);
262  const std::vector<double> child_values = (
263  ExpectedReturnsOfDeterministicPoliciesFromSeedsImpl(
264  *child, policy_seeds, policies));
265  for (auto p = Player{0}; p < num_players; ++p) {
266  values[p] += action_and_prob.second * child_values[p];
267  }
268  }
269  return values;
270  }
271 
272  // Determine the state seed from the policy seed.
273  auto state_seed = std::hash<std::string>{}(info_state_string);
274  state_seed += policy_seeds[player];
275  state_seed += state.MoveNumber() * num_players;
276  state_seed += player;
277  std::mt19937 gen(state_seed);
278 
279  const auto legal_actions = state.LegalActions();
280  std::uniform_int_distribution<int> dist(0, legal_actions.size() - 1);
281  const int sampled_action_index = dist(gen);
282  const Action action = legal_actions[sampled_action_index];
283 
284  SPIEL_CHECK_GE(action, 0);
285  std::unique_ptr<const State> child = state.Child(action);
286  std::vector<double> child_values = (
287  ExpectedReturnsOfDeterministicPoliciesFromSeedsImpl(
288  *child, policy_seeds, policies));
289  for (auto p = Player{0}; p < num_players; ++p) {
290  values[p] += child_values[p];
291  }
292  }
293  SPIEL_CHECK_EQ(values.size(), state.NumPlayers());
294  return values;
295 }
