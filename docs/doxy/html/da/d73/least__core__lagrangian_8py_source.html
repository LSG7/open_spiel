49def compute_least_core_value(
50 cvc: coalitional_game.CoalitionalGame, alg_config
51) -> LeastCoreValue:
52 """Computes the least core value of a game."""
53 opt_primal = optax.adam(learning_rate=alg_config.init.lr_primal)
54 opt_dual = optax.adam(learning_rate=alg_config.init.lr_dual)
55 evaluation_iterations = alg_config.eval.evaluation_iterations
56 evaluate_every = 2 * alg_config.solve.n_iter # do not evaluate
57 cl = CoreLagrangian(cvc, opt_primal, opt_dual)
58 payoffs, epsilons, _, duration = cl.solve(
59 evaluation_iterations=evaluation_iterations,
60 evaluate_every=evaluate_every,
61 **alg_config.solve,
62 )
63 lcvs = np.full(payoffs.shape[0], np.inf)
64 payoff = payoffs[-1]
65 lcv = np.inf
66 for i in range(payoffs.shape[0]):
67 payoff = payoffs[i]
68 epsilon = epsilons[i]
69 max_violation = payoff_evaluation(
70 cvc, payoff, epsilon, evaluation_iterations)
71 lcv = epsilon + max_violation
72 lcvs[i] = lcv
73 meta = dict(payoffs=payoffs, epsilons=epsilons, lcvs=lcvs)
74 return LeastCoreValue(payoff, lcv, duration, meta)

77def payoff_evaluation(
78 cv_calc: coalitional_game.CoalitionalGame,
79 payoffs: np.ndarray,
80 epsilon: float,
81 batch_size: int,
82 max_exponent: int = 13,
83) -> float:
84 """Evaluate deficit over a set of random coalitions.
85
86 Args:
87 cv_calc: the game to work on
88 payoffs: the payoff vector to test
89 epsilon: desired approximation of the epsilon-core
90 batch_size: number of random coalitions to sample
91 max_exponent: examine at maxixum 2**max_exponent constraints in one batch
92 default 13, assume 2**13 ~ 10k coalitions is mem limit for single batch
93
94 Returns:
95 Expected loss, relu(deficit), over random batch of coalitions
96 """
97 max_batch = 2**max_exponent
98 num_players = cv_calc.num_players()
99 violation = 0.
100 if batch_size >= 2**num_players:
101 num_suffix_repeats = min(max_exponent, num_players)
102 num_prefix_repeats = max(0, num_players - num_suffix_repeats)
103 zo = [0, 1]
104 suffix = np.array(list(itertools.product(zo, repeat=num_suffix_repeats)))
105 prefixes = itertools.product(zo, repeat=num_prefix_repeats)
106 for prefix in prefixes:
107 if prefix:
108 prefix_rep = np.repeat([prefix], suffix.shape[0], axis=0)
109 coalitions = np.concatenate([prefix_rep, suffix], axis=1)
110 else:
111 coalitions = suffix
112 batch_contributions = cv_calc.coalition_values(coalitions)
113 batch_payouts = np.dot(coalitions, payoffs)
114 batch_deficits = batch_contributions - batch_payouts - epsilon
115 batch_deficits = np.clip(batch_deficits, 0., np.inf)
116 violation = max(violation, np.max(batch_deficits))
117 else:
118 q, r = divmod(batch_size, max_batch)
119 num_loops = q + (r > 0)
120 for _ in range(num_loops):
121 coalitions = np.random.randint(2, size=(max_batch, num_players))
122 batch_contributions = cv_calc.coalition_values(coalitions)
123 batch_payouts = np.dot(coalitions, payoffs)
124 batch_deficits = batch_contributions - batch_payouts - epsilon
125 batch_deficits = np.clip(batch_deficits, 0., np.inf)
126 violation = max(violation, np.max(batch_deficits))
127 return float(violation)

130class CoreSolver():
131 """Find an epsilon-core."""
132
133 def __init__(self,
134 cvc: coalitional_game.CoalitionalGame):
135 self.cvc = cvc
136 self.num_players = cvc.num_players()
137 # we assume grand_coalition is optimal coalition
138 grand_coalition = np.full(cvc.num_players(), 1, dtype=np.int32)
139 self.grand_coalition_value = cvc.coalition_value(grand_coalition)
140
141 self.payoffs = None
142 self.losses = None
143 self.max_deficits = None
144 self.evaluation_losses = None
145
146 def logits_to_payoff(self, logits):
147 logits_ext = jnp.append(logits, 0.)
148 payoff = jax.nn.softmax(logits_ext)
149 payoff *= self.grand_coalition_value
150 return payoff
151
152 def loss_deficit(self, current_payoff, coalitions, coalition_values, epsilon):
153 """Compute Mean Loss and Max Deficit."""
154 coalition_payment = jnp.dot(coalitions, current_payoff)
155 deficit = coalition_values - epsilon - coalition_payment
156 coalition_size = jnp.sum(coalitions, axis=1)
157 weight = 1.0 / jnp.clip(coalition_size, 1, self.num_players)
158 losses = 0.5 * weight * jax.nn.relu(deficit) ** 2.0
159 return jnp.mean(losses, axis=0), jnp.max(jax.nn.relu(deficit))

162class CoreOptimization(CoreSolver):
163 """Find an epsilon-core via optimization."""
164
165 def __init__(self,
166 cvc: coalitional_game.CoalitionalGame,
167 opt,
168 epsilon):
169 super().__init__(cvc)
170 self.opt = opt
171 self.epsilon = epsilon
172
173 @functools.partial(jax.jit, static_argnums=[0])
174 def loss(self, params, data):
175 """Compute Loss."""
176 current_payoff = params
177 coalitions, coalition_values = data
178
179 return self.loss_deficit(current_payoff, coalitions, coalition_values,
180 self.epsilon)
181
182 @functools.partial(jax.jit, static_argnums=[0])
183 def update_step(self, params, data, opt_state):
184 """GD update step."""
185
186 # data = (coalitions, coalition_values)
187
188 # Convert losses into pure functions.
189 loss_fn = lambda p: self.loss(p, data)[0]
190
191 # Compute saddle-point gradients.
192 grads_fn = jax.grad(loss_fn, argnums=0)
193 grads = grads_fn(params)
194
195 updates, opt_state = self.opt.update(grads, opt_state, params)
196
197 params = optax.apply_updates(params, updates)
198
199 params = jnp.clip(params, 0, np.inf)
200 scale = self.grand_coalition_value / jnp.sum(params)
201 params = params * scale
202
203 return params, opt_state

205 def solve(self, n_iter: int, batch_size: int = 100,
206 save_every: int = 2,
207 evaluate_every: int = 2, evaluation_iterations: int = 100,
208 seed: int = 0
209 ) -> Tuple[np.ndarray, np.ndarray, float]:
210 """Find a least-core via Lagrange multipliers.
211
212 Additional optimization metrics are stored as class variables:
213 self.payoffs
214 self.losses
215 self.max_deficits
216 self.evaluation_losses
217 self.duration
218
219 Args:
220 n_iter: number of iterations
221 batch_size: number of constraints to evaluate at each step
222 save_every: int, how often to record optimization variables
223 evaluate_every: int, how often to evaluate the max constraint violation
224 evaluation_iterations: int, how many constraints to measure violations
225 for, if number if less than number of coalitions a batch of constraints
226 is sampled randomly. otherwise, all constraints are evaluated
227 seed: int, for sampling minibatches of constraints
228
229 Returns:
230 payoffs over training
231 max deficit over training
232 runtime duration (sec)
233 """
234
235 qe, re = divmod(n_iter, evaluate_every)
236 num_eval = qe + (re > 0)
237 qs, rs = divmod(n_iter, save_every)
238 num_save = qs + (rs > 0)
239
240 max_violations = np.empty(num_eval, dtype=np.float32)
241 losses = np.empty(num_save, dtype=np.float32)
242 max_deficits = np.empty(num_save, dtype=np.float32)
243 payoffs = np.empty((num_save, self.num_playersnum_players), dtype=np.float32)
244
245 scale = self.grand_coalition_value / self.num_playersnum_players
246 grand_coalition = np.full(self.num_playersnum_players, 1, dtype=np.int32)
247 current_payoff = jnp.array(grand_coalition * scale)
248 params = current_payoff
249
250 opt_state = self.opt.init(params)
251
252 logging.debug('Uniform payoff %s', current_payoff)
253
254 rng = jax.random.PRNGKey(seed)
255
256 start = time.time()
257 for iter_id in range(n_iter):
258 if batch_size < 2**self.num_playersnum_players:
259 rng, key = jax.random.split(rng, 2)
260 coalitions = jax.random.randint(key,
261 shape=(batch_size, self.num_playersnum_players),
262 minval=0,
263 maxval=2,
264 dtype=jnp.int32)
265 else:
266 prod_space = itertools.product([0, 1], repeat=self.num_playersnum_players)
267 coalitions = np.stack(list(prod_space))
268 coalition_values = self.cvccvc.coalition_values(np.array(coalitions))
269
270 data = (coalitions, coalition_values)
271 loss, max_deficit = self.loss(params, data)
272 params, opt_state = self.update_step(params, data, opt_state)
273
274 # Done updating, save if needed
275 if iter_id % save_every == 0:
276 logging.debug('Saving...')
277 idx = iter_id // save_every
278 losses[idx] = loss
279 max_deficits[idx] = max_deficit
280 current_payoff = params
281 payoffs[idx] = current_payoff
282 logging.debug('Loss was %f, Max deficit was %f, New payoff %s',
283 loss, max_deficit, current_payoff)
284
285 # Done updating, evaluate if needed
286 if (evaluate_every < n_iter) and (iter_id % evaluate_every == 0):
287 logging.debug('Evaluating...')
288 estimated_loss = payoff_evaluation(
289 self.cvccvc,
290 current_payoff,
291 self.epsilon,
292 evaluation_iterations,
293 )
294 max_violations[iter_id // evaluate_every] = estimated_loss
295 logging.debug('Estimated loss %f', estimated_loss)
296 end = time.time()
297 duration = end - start
298
299 self.payoffspayoffs = np.array(payoffs)
300 self.losseslosses = np.array(losses)
301 self.max_deficitsmax_deficits = np.array(max_deficits)
302 self.max_violations = np.array(max_violations)
303 self.duration = duration
304
305 return (np.array(payoffs),
306 np.array(max_deficits),
307 duration)

453class CoreLagrangian(CoreSolver):
454 """Find a least-core via Lagrange multipliers."""
455
456 def __init__(self,
457 cvc: coalitional_game.CoalitionalGame,
458 opt_primal,
459 opt_dual):
460 super().__init__(cvc)
461 self.opt_primal = opt_primal
462 self.opt_dual = opt_dual
463
464 current_logits_keys = ['current_logits' for _ in range(self.num_playersnum_players)]
465 keys_primal = {'current_logits': current_logits_keys,
466 'epsilon': 'epsilon'}
467 keys_dual = {'mu': 'mu'}
468 self.keys = (keys_primal, keys_dual)
469 self.nonnegative_keys = ('epsilon', 'mu')
470
471 self.epsilons = None
472 self.mus = None
473 self.lagrangians = None
474
475 @functools.partial(jax.jit, static_argnums=[0])
476 def lagrangian(self, primal, dual, data):
477 """Compute Lagrangian."""
478 current_logits, epsilon = primal['current_logits'], primal['epsilon']
479 mu = dual['mu']
480 coalitions, coalition_values, gamma_adj = data
481
482 current_payoff = self.logits_to_payoff(current_logits)
483 mean_loss, max_deficit = self.loss_deficit(current_payoff,
484 coalitions,
485 coalition_values,
486 epsilon)
487 lagrangian = epsilon + mu * (mean_loss - gamma_adj)
488 lagrangian = jnp.sum(lagrangian) # just for converting (1,) array to scalar
489 return lagrangian, (mean_loss, max_deficit)
490
491 @functools.partial(jax.jit, static_argnums=[0])
492 def update_step(self, params, data, opt_state):
493 """SimGD update step."""
494
495 # data = (coalitions, coalition_values, gamma_adj)
496 params_primal, params_dual = params
497 opt_state_primal, opt_state_dual = opt_state
498
499 # Convert losses into pure functions.
500 loss_primal_fn = lambda p, d: self.lagrangian(p, d, data)[0]
501 loss_dual_fn = lambda p, d: -self.lagrangian(p, d, data)[0]
502
503 # Compute saddle-point gradients.
504 grads_primal_fn = jax.grad(loss_primal_fn, argnums=0)
505 grads_primal = grads_primal_fn(params_primal, params_dual)
506 grads_dual_fn = jax.grad(loss_dual_fn, argnums=1)
507 grads_dual = grads_dual_fn(params_primal, params_dual)
508
509 updates_primal, opt_state_primal = self.opt_primal.update(grads_primal,
510 opt_state_primal,
511 params_primal)
512 updates_dual, opt_state_dual = self.opt_dual.update(grads_dual,
513 opt_state_dual,
514 params_dual)
515
516 params_primal = optax.apply_updates(params_primal, updates_primal)
517 params_dual = optax.apply_updates(params_dual, updates_dual)
518
519 params = (params_primal, params_dual)
520 opt_state = (opt_state_primal, opt_state_dual)
521
522 clip = (
523 lambda x, k: jnp.clip(x, 0, np.inf) if k in self.nonnegative_keys else x
524 )
525 params = jax.tree_util.tree_map(clip, params, self.keys)
526
527 return params, opt_state

529 def solve(self, n_iter: int, batch_size: int = 100, gamma: float = 1e-2,
530 mu_init: float = 1000.,
531 save_every: int = 2,
532 evaluate_every: int = 2, evaluation_iterations: int = 100,
533 seed: int = 0,
534 ) -> Tuple[np.ndarray, np.ndarray, np.ndarray, float]:
535 """Find a least-core via Lagrange multipliers.
536
537 Additional optimization metrics are stored as class variables:
538 self.payoffs
539 self.epsilons
540 self.mus
541 self.lagrangians
542 self.losses
543 self.max_deficits
544 self.evaluation_losses
545 self.duration
546
547 Args:
548 n_iter: number of iterations
549 batch_size: number of constraints to evaluate at each step
550 gamma: float, slack allowed in core constraints
551 mu_init: float, initialize the lagrange multiplier to this value
552 save_every: int, how often to record optimization variables
553 evaluate_every: int, how often to evaluate the max constraint violation
554 evaluation_iterations: int, how many constraints to measure violations
555 for, if number if less than number of coalitions a batch of constraints
556 is sampled randomly. otherwise, all constraints are evaluated
557 seed: int, for sampling minibatches of constraints
558
559 Returns:
560 payoffs over training
561 epsilon over training
562 max deficit over training
563 runtime duration (sec)
564 """
565
566 qe, re = divmod(n_iter, evaluate_every)
567 num_eval = qe + (re > 0)
568 qs, rs = divmod(n_iter, save_every)
569 num_save = qs + (rs > 0)
570
571 max_violations = np.empty(num_eval, dtype=np.float32)
572 lagrangians = np.empty(num_save, dtype=np.float32)
573 losses = np.empty(num_save, dtype=np.float32)
574 max_deficits = np.empty(num_save, dtype=np.float32)
575 epsilons = np.empty(num_save, dtype=np.float32)
576 payoffs = np.empty((num_save, self.num_playersnum_players), dtype=np.float32)
577 mus = np.empty(num_save, dtype=np.float32)
578
579 current_logits = jnp.zeros(self.num_playersnum_players - 1, dtype=jnp.float32)
580 epsilon = self.grand_coalition_value * jnp.ones(1, dtype=jnp.float32)
581 mu = jnp.ones(1, dtype=jnp.float32) * mu_init
582
583 params_primal = {'current_logits': current_logits,
584 'epsilon': epsilon}
585 params_dual = {'mu': mu}
586 params = (params_primal, params_dual)
587
588 opt_state_primal = self.opt_primal.init(params_primal)
589 opt_state_dual = self.opt_dual.init(params_dual)
590 opt_state = (opt_state_primal, opt_state_dual)
591
592 current_payoff = np.asarray(self.logits_to_payoff(current_logits))
593 logging.debug('Uniform payoff %s', current_payoff)
594
595 if self.num_playersnum_players < 30:
596 gamma_adj = gamma**2.0 / (2**self.num_playersnum_players - 1)
597 else:
598 # Set arbitrary value if the above would result in a too tiny number.
599 gamma_adj = 1e-6
600
601 rng = jax.random.PRNGKey(seed)
602
603 start = time.time()
604 for iter_id in range(n_iter):
605 if batch_size < 2**self.num_playersnum_players:
606 rng, key = jax.random.split(rng, 2)
607 coalitions = jax.random.randint(key,
608 shape=(batch_size, self.num_playersnum_players),
609 minval=0,
610 maxval=2,
611 dtype=jnp.int32)
612 else:
613 prod_space = itertools.product([0, 1], repeat=self.num_playersnum_players)
614 coalitions = np.stack(list(prod_space))
615 coalition_values = self.cvccvc.coalition_values(np.array(coalitions))
616
617 data = (coalitions, coalition_values, gamma_adj)
618 lagrangian, (loss, max_deficit) = self.lagrangian(*params, data)
619 params, opt_state = self.update_step(params, data, opt_state)
620
621 params_primal, params_dual = params
622
623 # Done updating, save if needed
624 if iter_id % save_every == 0:
625 logging.debug('Saving...')
626 idx = iter_id // save_every
627 lagrangians[idx] = lagrangian
628 losses[idx] = loss
629 max_deficits[idx] = max_deficit
630 epsilons[idx] = params_primal['epsilon'].item()
631 mus[idx] = params_dual['mu'].item()
632 current_payoff = np.asarray(self.logits_to_payoff(
633 params_primal['current_logits']))
634 payoffs[idx] = current_payoff
