79 std::string NegotiationState::ActionToString(Player player,
80  Action move_id) const {
81  if (player == kChancePlayerId) {
82  return absl::StrCat("chance outcome ", move_id);
83  } else {
84  std::string action_string = "";
85  if (turn_type_ == TurnType::kProposal) {
86  if (move_id == parent_game_.NumDistinctProposals() - 1) {
87  absl::StrAppend(&action_string, "Proposal: Agreement reached!");
88  } else {
89  std::vector<int> proposal = DecodeProposal(move_id);
90  std::string prop_str = absl::StrJoin(proposal, ", ");
91  absl::StrAppend(&action_string, "Proposal: [", prop_str, "]");
92  }
93  } else {
94  std::vector<int> utterance = DecodeUtterance(move_id);
95  std::string utt_str = absl::StrJoin(utterance, ", ");
96  absl::StrAppend(&action_string, ", Utterance: [", utt_str, "]");
97  }
98  return action_string;
99  }

102 bool NegotiationState::IsTerminal() const {
103  // If utterances are enabled, force the agent to utter something even when
104  // they accept the proposal or run out of steps (i.e. on ther last turn).
105  bool utterance_check =
106  (enable_utterances_ ? utterances_.size() == proposals_.size() : true);
107  return (agreement_reached_ || proposals_.size() >= max_steps_) &&
108  utterance_check;
109 }

111 std::vector<double> NegotiationState::Returns() const {
112  if (!IsTerminal() || !agreement_reached_) {
113  return std::vector<double>(num_players_, 0.0);
114  }
115 
116  int proposing_player = proposals_.size() % 2 == 1 ? 0 : 1;
117  int other_player = 1 - proposing_player;
118  const std::vector<int>& final_proposal = proposals_.back();
119 
120  std::vector<double> returns(num_players_, 0.0);
121  for (int j = 0; j < num_items_; ++j) {
122  returns[proposing_player] +=
123  agent_utils_[proposing_player][j] * final_proposal[j];
124  returns[other_player] +=
125  agent_utils_[other_player][j] * (item_pool_[j] - final_proposal[j]);
126  }
127 
128  return returns;
129 }

164 // 1D vector with shape:
165 // - Current player: kNumPlayers bits
166 // - Current turn type: 2 bits
167 // - Terminal status: 2 bits: (Terminal? and Agreement reached?)
168 // - Context:
169 // - item pool (num_items * (max_quantity + 1) bits)
170 // - my utilities (num_items * (max_value + 1) bits)
171 // - Last proposal: (num_items * (max_quantity + 1) bits)
172 // If utterances are enabled, another:
173 // - Last utterance: (utterance_dim * num_symbols) bits)
174 std::vector<int> NegotiationGame::ObservationTensorShape() const {
175  return {kNumPlayers + 2 + 2 + (num_items_ * (kMaxQuantity + 1)) +
176  (num_items_ * (kMaxValue + 1)) + (num_items_ * (kMaxQuantity + 1)) +
177  (enable_utterances_ ? utterance_dim_ * num_symbols_ : 0)};
178 }

180 void NegotiationState::ObservationTensor(Player player,
181  absl::Span<float> values) const {
182  SPIEL_CHECK_GE(player, 0);
183  SPIEL_CHECK_LT(player, num_players_);
184 
185  SPIEL_CHECK_EQ(values.size(), parent_game_.ObservationTensorSize());
186  std::fill(values.begin(), values.end(), 0);
187 
188  // No observations at chance nodes.
189  if (IsChanceNode()) {
190  return;
191  }
192 
193  // 1D vector with shape:
194  // - Current player: 2 bits
195  // - Turn type: 2 bits
196  // - Terminal status: 2 bits: (Terminal? and Agreement reached?)
197  // - Context:
198  // - item pool (num_items * (max_quantity + 1) bits)
199  // - my utilities (num_items * (max_value + 1) bits)
200  // - Last proposal: (num_items * (max_quantity + 1) bits)
201  // If utterances are enabled, another:
202  // - Last utterance: (utterance_dim * num_symbols) bits)
203 
204  // Current player.
205  int offset = 0;
206  if (!IsTerminal()) {
207  values[offset + CurrentPlayer()] = 1;
208  }
209  offset += kNumPlayers;
210 
211  // Current turn type.
212  if (turn_type_ == TurnType::kProposal) {
213  values[offset] = 1;
214  } else {
215  values[offset + 1] = 1;
216  }
217  offset += 2;
218 
219  // Terminal status: 2 bits
220  values[offset] = IsTerminal() ? 1 : 0;
221  values[offset + 1] = agreement_reached_ ? 1 : 0;
222  offset += 2;
223 
224  // Item pool.
225  for (int item = 0; item < num_items_; ++item) {
226  values[offset + item_pool_[item]] = 1;
227  offset += kMaxQuantity + 1;
228  }
229 
230  // Utilities.
231  for (int item = 0; item < num_items_; ++item) {
232  values[offset + agent_utils_[player][item]] = 1;
233  offset += kMaxValue + 1;
234  }
235 
236  // Last proposal.
237  if (!proposals_.empty()) {
238  for (int item = 0; item < num_items_; ++item) {
239  values[offset + proposals_.back()[item]] = 1;
240  offset += kMaxQuantity + 1;
241  }
242  } else {
243  offset += num_items_ * (kMaxQuantity + 1);
244  }
245 
246  // Last utterance.
247  if (enable_utterances_) {
248  if (!utterances_.empty()) {
249  for (int dim = 0; dim < utterance_dim_; ++dim) {
250  values[offset + utterances_.back()[dim]] = 1;
251  offset += num_symbols_;
252  }
253  } else {
254  offset += utterance_dim_ * num_symbols_;
255  }
256  }
257 
258  SPIEL_CHECK_EQ(offset, values.size());
259 }

283 // From Sec 2.1 of the paper: "At each round (i) an item pool is sampled
284 // uniformly, instantiating a quantity (between 0 and 5) for each of the types
285 // and represented as a vector i \in {0...5}^3 and (ii) each agent j receives a
286 // utility function sampled uniformly, which specifies how rewarding one unit of
287 // each item is (with item rewards between 0 and 10, and with the constraint
288 // that there is at least one item with non-zero utility), represented as a
289 // vector u_j \in {0...10}^3".
290 void NegotiationState::DetermineItemPoolAndUtilities() {
291  // Generate max number of rounds (max number of steps for the episode): we
292  // sample N between 4 and 10 at the start of each episode, according to a
293  // truncated Poissondistribution with mean 7, as done in the Cao et al. '18
294  // paper.
295  max_steps_ = -1;
296  absl::poisson_distribution<int> steps_dist(7.0);
297  while (!(max_steps_ >= 4 && max_steps_ <= 10)) {
298  max_steps_ = steps_dist(*parent_game_.RNG());
299  }
300 
301  // Generate the pool of items.
302  absl::uniform_int_distribution<int> quantity_dist(0, kMaxQuantity);
303  for (int i = 0; i < num_items_; ++i) {
304  item_pool_.push_back(quantity_dist(*parent_game_.RNG()));
305  }
306 
307  // Generate agent utilities.
308  absl::uniform_int_distribution<int> util_dist(0, kMaxValue);
309  for (int i = 0; i < num_players_; ++i) {
310  agent_utils_.push_back({});
311  int sum_util = 0;
312  while (sum_util == 0) {
313  for (int j = 0; j < num_items_; ++j) {
314  agent_utils_[i].push_back(util_dist(*parent_game_.RNG()));
315  sum_util += agent_utils_[i].back();
316  }
317  }
318  }
319 }

326 void NegotiationState::DoApplyAction(Action move_id) {
327  if (IsChanceNode()) {
328  DetermineItemPoolAndUtilities();
329  InitializeEpisode();
330  } else {
331  if (turn_type_ == TurnType::kProposal) {
332  if (move_id == parent_game_.NumDistinctProposals() - 1) {
333  // Agreement!
334  agreement_reached_ = true;
335  } else {
336  std::vector<int> proposal = DecodeProposal(move_id);
337  proposals_.push_back(proposal);
338  }
339 
340  if (enable_utterances_) {
341  // Note: do not move to next player yet.
342  turn_type_ = TurnType::kUtterance;
343  } else {
344  // Keep it at kProposal, but move to next player.
345  cur_player_ = 1 - cur_player_;
346  }
347  } else {
348  SPIEL_CHECK_TRUE(enable_utterances_);
349  std::vector<int> utterance = DecodeUtterance(move_id);
350  utterances_.push_back(utterance);
351  turn_type_ = TurnType::kProposal;
352  cur_player_ = 1 - cur_player_;
353  }
354  }
355 }

357 bool NegotiationState::NextProposal(std::vector<int>* proposal) const {
358  // Starting from the right, move left trying to increase the value. When
359  // successful, increment the value and set all the right digits back to 0.
360  for (int i = num_items_ - 1; i >= 0; --i) {
361  if ((*proposal)[i] + 1 <= item_pool_[i]) {
362  // Success!
363  (*proposal)[i]++;
364  for (int j = i + 1; j < num_items_; ++j) {
365  (*proposal)[j] = 0;
366  }
367  return true;
368  }
369  }
370 
371  return false;
372 }

374 std::vector<int> NegotiationState::DecodeInteger(int encoded_value,
375  int dimensions,
376  int num_digit_values) const {
377  std::vector<int> decoded(dimensions, 0);
378  int i = dimensions - 1;
379  while (encoded_value > 0) {
380  SPIEL_CHECK_GE(i, 0);
381  SPIEL_CHECK_LT(i, dimensions);
382  decoded[i] = encoded_value % num_digit_values;
383  encoded_value /= num_digit_values;
384  i--;
385  }
386  return decoded;
387 }

389 int NegotiationState::EncodeInteger(const std::vector<int>& container,
390  int num_digit_values) const {
391  int encoded_value = 0;
392  for (int digit : container) {
393  encoded_value = encoded_value * num_digit_values + digit;
394  }
395  return encoded_value;
396 }

404 Action NegotiationState::EncodeUtterance(
405  const std::vector<int>& utterance) const {
406  SPIEL_CHECK_EQ(utterance.size(), utterance_dim_);
407  // Utterance ids are offset from zero (starting at NumDistinctProposals()).
408  return parent_game_.NumDistinctProposals() +

423 std::vector<Action> NegotiationState::LegalActions() const {
424  if (IsChanceNode()) {
425  return LegalChanceOutcomes();
426  } else if (IsTerminal()) {
427  return {};
428  } else if (turn_type_ == TurnType::kProposal) {
429  std::vector<Action> legal_actions;
430 
431  // Proposals are always enabled, so first contruct them.
432  std::vector<int> proposal(num_items_, 0);
433  legal_actions.push_back(EncodeProposal(proposal));
434 
435  while (NextProposal(&proposal)) {
436  legal_actions.push_back(EncodeProposal(proposal));
437  }
438 
439  if (!proposals_.empty()) {
440  // Add the agreement action only if there's been at least one proposal.
441  legal_actions.push_back(parent_game_.NumDistinctProposals() - 1);
442  }
443 
444  return legal_actions;
445  } else {
446  SPIEL_CHECK_TRUE(enable_utterances_);
447  SPIEL_CHECK_FALSE(parent_game_.LegalUtterances().empty());
448  return parent_game_.LegalUtterances();
449  }
450 }

453  const {
454  SPIEL_CHECK_TRUE(IsChanceNode());

568 std::unique_ptr<State> NegotiationGame::DeserializeState(
569  const std::string& str) const {
570  if (str == "chance") {
571  return NewInitialState();
572  } else {
573  std::vector<std::string> lines = absl::StrSplit(str, '\n');
574  std::unique_ptr<State> state = NewInitialState();
575  SPIEL_CHECK_EQ(lines.size(), 5);
576  NegotiationState& nstate = static_cast<NegotiationState&>(*state);
577  // Take the chance action, but then reset the quantities. Make sure game's
578  // RNG state is not advanced during deserialization so copy it beforehand
579  // in order to be able to restore after the chance action.
580  std::unique_ptr<std::mt19937> rng = std::make_unique<std::mt19937>(*rng_);
581  nstate.ApplyAction(0);
582  rng_ = std::move(rng);
583  nstate.ItemPool().clear();
584  nstate.AgentUtils().clear();
585  // Max steps
586  nstate.SetMaxSteps(std::stoi(lines[0]));
587  // Item pool.
588  std::vector<std::string> parts = absl::StrSplit(lines[1], ' ');
589  for (const auto& part : parts) {
590  nstate.ItemPool().push_back(std::stoi(part));
591  }
592  // Agent utilities.
593  for (Player player : {0, 1}) {
594  parts = absl::StrSplit(lines[2 + player], ' ');
595  nstate.AgentUtils().push_back({});
596  for (const auto& part : parts) {
597  nstate.AgentUtils()[player].push_back(std::stoi(part));
598  }
599  }
600  nstate.SetCurrentPlayer(0);
601  // Actions.
602  if (lines.size() == 5) {
603  parts = absl::StrSplit(lines[4], ' ');
604  // Skip the first one since it is the chance node.
605  for (int i = 1; i < parts.size(); ++i) {
606  Action action = static_cast<Action>(std::stoi(parts[i]));
607  nstate.ApplyAction(action);
608  }
609  }
610  return state;
611  }
612 }
