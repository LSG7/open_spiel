91def get_action_probs(
92 agent: OpponentShapingAgent, game: str
93) -> List[typing.Dict[str, typing.Any]]:
94 """Returns the probability of cooperation and a string repr for each state.
95
96 Args:
97 agent: The agent.
98 game: The name of the game.
99
100 Returns:
101 A list of dictionaries, each containing the probability of cooperation
102 and a string representation
103 """
104 actions = ['C', 'D'] if game == 'ipd' else ['H', 'T']
105 states = ['s0'] + [''.join(s) for s in itertools.product(actions, repeat=2)]
106 params = agent.train_state.policy_params[agent.player_id]
107 action_probs = []
108 for i, state_str in enumerate(states):
109 state = np.eye(len(states))[i]
110 prob = agent.policy_network.apply(params, state).prob(0)
111 action = actions[0]
112 action_probs.append(
113 {'prob': prob.item(), 'name': f'P({action}|{state_str})'}
114 )
115 return action_probs

118def log_epoch_data(epoch: int, agents: List[OpponentShapingAgent], eval_batch):
119 """Logs data to wandb and prints it to the console.
120
121 Args:
122 epoch: The current epoch.
123 agents: A list of agents.
124 eval_batch: A batch of episodes.
125 """
126 logs = {}
127 for agent in agents:
128 avg_step_reward = np.mean(
129 [ts.rewards[agent.player_id] for ts in eval_batch]
130 )
131 probs = get_action_probs(agent, game=FLAGS.game)
132 for info in probs:
133 logs[f'agent_{agent.player_id}/{info["name"]}'] = info['prob']
134 probs = ', '.join([f'{info["name"]}: {info["prob"]:.2f}' for info in probs])
135 metrics = agent.metrics()
136 logs.update({
137 f'agent_{agent.player_id}/avg_step_reward': avg_step_reward,
138 **{
139 f'agent_{agent.player_id}/{k}': v.item() for k, v in metrics.items()
140 },
141 })
142 print(
143 f'[epoch {epoch}] Agent {agent.player_id}: {avg_step_reward:.2f} |'
144 f' {probs}'
145 )
146 wandb.log(logs)

149def collect_batch(
150 env: Environment, agents: List[OpponentShapingAgent], eval_mode: bool
151) -> List[TimeStep]:
152 """Collects one episode.
153
154 Args:
155 env: The environment.
156 agents: A list of opponent shaping agents.
157 eval_mode: If true, the agents will be run in evaluation mode.
158
159 Returns:
160 A list of time steps.
161 """
162 episode = []
163 time_step = env.reset()
164 episode.append(time_step)
165 while not time_step.last():
166 actions = []
167 for agent in agents:
168 action, _ = agent.step(time_step, is_evaluation=eval_mode)
169 if action is not None:
170 action = action.squeeze()
171 actions.append(action)
172 time_step = env.step(np.stack(actions, axis=1))
173 time_step.observations['actions'] = actions
174 episode.append(time_step)
175
176 for agent in agents:
177 agent.step(time_step, is_evaluation=eval_mode)
178 return episode

181def make_agent(
182 key: jax.random.PRNGKey,
183 player_id: int,
184 env: Environment,
185 networks: Tuple[hk.Transformed, hk.Transformed],

187 """Creates an opponent shaping agent.

226def make_agent_networks(
227 num_states: int, num_actions: int
228) -> Tuple[hk.Transformed, hk.Transformed]:
229 """Creates action weights for each state-action pair and values for each state.
230
231 Args:
232 num_states: The number of distinct states.
233 num_actions: The number of distinct actions.
234
235 Returns:
236 A tuple of policy and critic networks transformed by hk.transform.
237 """
238
239 def policy(obs):
240 theta = hk.get_parameter(
241 'theta',
242 init=hk.initializers.Constant(0),
243 shape=(num_states, num_actions),
244 )
245 logits = jnp.select(obs, theta)
246 logits = jnp.nan_to_num(logits)
247 return distrax.Categorical(logits=logits)
248
249 def value_fn(obs):
250 w = hk.get_parameter(
251 'w', [num_states], init=jnp.zeros
252 ) # @pylint: disable=invalid-name
253 return w[jnp.argmax(obs, axis=-1)].reshape(*obs.shape[:-1], 1)
254
255 return hk.without_apply_rng(hk.transform(policy)), hk.without_apply_rng(
256 hk.transform(value_fn)
257 )

260def make_env(game: str, iterations: int, batch_size: int) -> Environment:
261 """Creates an environment.
262
263 The environment is either iterated prisoners dilemma or iterated matching
264 pennies.
265
266 Args:
267 game: The game to play. Either 'ipd' or 'imp'.
268 iterations: The number of iterations to play.
269 batch_size: The batch size.
270
271 Returns:
272 An environment instance.
273 """
274 if game == 'ipd':
275 env = IteratedPrisonersDilemma(iterations=iterations, batch_size=batch_size)
276 elif game == 'imp':
277 env = IteratedMatchingPennies(iterations=iterations, batch_size=batch_size)
278 else:
279 raise ValueError(f'Unknown game: {game}')
280 return env

283def setup_agents(
284 env: Environment, rng: hk.PRNGSequence
285) -> List[OpponentShapingAgent]:
286 """Creates an opponent shaping agent for each player in the environment.
287
288 Args:
289 env: The environment.
290 rng: A random seed key.
291
292 Returns:
293 A list of opponent shaping agents.
294 """
295 agents = []
296 num_actions = env.action_spec()['num_actions']
297 info_state_shape = env.observation_spec()['info_state']
298 for player_id in range(env.num_players):
299 networks = make_agent_networks(
300 num_states=info_state_shape[player_id][0],
301 num_actions=num_actions[player_id],
302 )
303 agent = make_agent(
304 key=next(rng), player_id=player_id, env=env, networks=networks
305 )
306 agents.append(agent)
307 return agents

311 """Updates the weights of the opponent models.

316 Returns:
317 None

319 agent: OpponentShapingAgent
320 for agent in agents:
321 for opp in [a for a in agents if a.player_id != agent.player_id]:

325def main(_):
326 """Main function. Runs the experiment."""
327 if FLAGS.exp_name is None:
328 FLAGS.exp_name = f'{FLAGS.game}_{FLAGS.seed}'
329 if not FLAGS.debug:
330 wandb.login(key=os.environ.get('WANDB_API_KEY', None))
331 wandb.init(
332 project='open-spiel-opponent-modelling',
333 group=FLAGS.exp_name,
334 config={
335 'game': FLAGS.game,
336 'seed': FLAGS.seed,
337 'epochs': FLAGS.epochs,
338 'batch_size': FLAGS.batch_size,
339 'critic_mini_batches': FLAGS.critic_mini_batches,
340 'game_iterations': FLAGS.game_iterations,
341 'policy_lr': FLAGS.policy_lr,
342 'opp_policy_lr': FLAGS.opp_policy_lr,
343 'critic_lr': FLAGS.critic_lr,
344 'correction_type': FLAGS.correction_type,
345 'n_lookaheads': FLAGS.n_lookaheads,
346 'correction_max_grad_norm': FLAGS.correction_max_grad_norm,
347 'discount': FLAGS.discount,
348 'policy_update_interval': FLAGS.policy_update_interval,
349 'use_opponent_modelling': FLAGS.use_opponent_modelling,
350 'opp_policy_mini_batches': FLAGS.opp_policy_mini_batches,
351 'opponent_model_learning_rate': FLAGS.opponent_model_learning_rate,
352 },
353 mode='disabled' if FLAGS.debug else 'online',
354 )
355
356 rng = hk.PRNGSequence(key_or_seed=FLAGS.seed)
357 env = make_env(
358 iterations=FLAGS.game_iterations,
359 batch_size=FLAGS.batch_size,
360 game=FLAGS.game,
361 )
362 agents = setup_agents(env=env, rng=rng)
363
364 if not FLAGS.use_opponent_modelling:
365 update_weights(agents)
366
367 batch = collect_batch(env=env, agents=agents, eval_mode=True)
368 log_epoch_data(epoch=0, agents=agents, eval_batch=batch)
369 for epoch in range(1, FLAGS.epochs + 1):
370 batch = collect_batch(env=env, agents=agents, eval_mode=False)
371 if not FLAGS.use_opponent_modelling:
372 update_weights(agents)
373 log_epoch_data(epoch=epoch, agents=agents, eval_batch=batch)
374 print('#' * 100)
375
376 wandb.finish()
