28def solve_game(state):
29 state_str = str(state)
30 if state_str in solved:
31 return solved[state_str].value
32 if state.is_terminal():
33 return state.returns()[0]
34
35 max_player = state.current_player() == 0
36 obs = state.observation_tensor()
37 act_mask = np.array(state.legal_actions_mask())
38 values = np.full(act_mask.shape, -2 if max_player else 2)
39 for action in state.legal_actions():
40 values[action] = solve_game(state.child(action))
41 value = values.max() if max_player else values.min()
42 best_actions = np.where((values == value) & act_mask)
43 policy = np.zeros_like(act_mask)
44 policy[best_actions[0][0]] = 1 # Choose the first for a deterministic policy.
45 solved[state_str] = model_lib.TrainInput(obs, act_mask, policy, value)
46 return value

57 @parameterized.parameters(model_lib.Model.valid_model_types)
58 def test_model_learns_simple(self, model_type):
59 game = pyspiel.load_game("tic_tac_toe")
60 model = build_model(game, model_type)
61 print("Num variables:", model.num_trainable_variables)
62 model.print_trainable_variables()
63
64 train_inputs = []
65 state = game.new_initial_state()
66 while not state.is_terminal():
67 obs = state.observation_tensor()
68 act_mask = state.legal_actions_mask()
69 action = state.legal_actions()[0]
70 policy = np.zeros(len(act_mask), dtype=float)
71 policy[action] = 1
72 train_inputs.append(model_lib.TrainInput(obs, act_mask, policy, value=1))
73 state.apply_action(action)
74 value, policy = model.inference([obs], [act_mask])
75 self.assertLen(policy, 1)
76 self.assertLen(value, 1)
77 self.assertLen(policy[0], game.num_distinct_actions())
78 self.assertLen(value[0], 1)
79
80 losses = []
81 policy_loss_goal = 0.05
82 value_loss_goal = 0.05
83 for i in range(200):
84 loss = model.update(train_inputs)
85 print(i, loss)
86 losses.append(loss)
87 if loss.policy < policy_loss_goal and loss.value < value_loss_goal:
88 break
89
90 self.assertGreater(losses[0].total, losses[-1].total)
91 self.assertGreater(losses[0].policy, losses[-1].policy)
92 self.assertGreater(losses[0].value, losses[-1].value)
93 self.assertLess(losses[-1].value, value_loss_goal)
94 self.assertLess(losses[-1].policy, policy_loss_goal)
