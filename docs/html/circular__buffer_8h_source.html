25 // A simple circular buffer of fixed size.
26 template <class T>
27 class CircularBuffer {
28  public:
29  explicit CircularBuffer(int max_size)
30  : max_size_(max_size), total_added_(0) {}
31 
32  // Add one element, replacing the oldest once it's full.
33  void Add(const T& value) {
34  if (data_.size() < max_size_) {
35  data_.push_back(value);
36  } else {
37  data_[total_added_ % max_size_] = value;
38  }
39  total_added_ += 1;
40  }
41 
42  // Return `num` elements without replacement.
43  std::vector<T> Sample(std::mt19937* rng, int num) {
44  std::vector<T> out;
45  out.reserve(num);
46  std::sample(data_.begin(), data_.end(), std::back_inserter(out), num, *rng);
47  return out;
48  }
49 
50  // Return the full buffer.
51  const std::vector<T>& Data() const { return data_; }
52 
53  // Access a single element from the buffer.
54  const T& operator[](int i) const { return data_[i]; }
55 
56  // How many elements are in the buffer.

58 
59  // Is the buffer empty?
60  bool Empty() const { return data_.empty(); }
61 
62  // How many elements have ever been added to the buffer.
63  int64_t TotalAdded() const { return total_added_; }
64 
65  protected:
