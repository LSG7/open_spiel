101 Move Move::Next(Direction dir) const {
102  switch (dir) {
103  case Direction::kUp:
104  return Move(row_ - 1, col_);
105  case Direction::kDown:
106  return Move(row_ + 1, col_);
107  case Direction::kLeft:
108  return Move(row_, col_ - 1);
109  case Direction::kRight:
110  return Move(row_, col_ + 1);
111  case Direction::kUpRight:
112  return Move(row_ - 1, col_ + 1);
113  case Direction::kUpLeft:
114  return Move(row_ - 1, col_ - 1);
115  case Direction::kDownRight:

117  case Direction::kDownLeft:
118  return Move(row_ + 1, col_ - 1);

132 int OthelloState::CountSteps(Player player, int action, Direction dir) const {
133  Move move = Move(action).Next(dir);
134 
135  int count = 0;
136  CellState cell = PlayerToState(player);
137  while (move.OnBoard()) {
138  if (BoardAt(move) == cell) {
139  return count;
140  } else if (BoardAt(move) == CellState::kEmpty) {
141  return 0;
142  }
143 
144  count++;
145  move = move.Next(dir);
146  }
147 
148  return 0;
149 }

163 void OthelloState::Capture(Player player, int action, Direction dir,
164  int steps) {
165  Move move = Move(action).Next(dir);
166 
167  CellState cell = PlayerToState(player);
168  for (int step = 0; step < steps; step++) {
169  if (BoardAt(move) == CellState::kEmpty || BoardAt(move) == cell) {
170  SpielFatalError(absl::StrCat("Cannot capture cell ", move.ToString()));
171  }
172 
173  board_[move.GetAction()] = cell;
174  move = move.Next(dir);
175  }
176 }

191 void OthelloState::DoApplyAction(Action action) {
192  if (action == kPassMove) { // pass
193  current_player_ = 1 - current_player_;
194  return;
195  }
196 
197  SPIEL_CHECK_TRUE(ValidAction(current_player_, action));
198 
199  CellState cell = PlayerToState(current_player_);
200  board_[action] = cell;
201 
202  for (auto direction : kDirections) {
203  int steps = CountSteps(current_player_, action, direction);
204  if (steps > 0) {
205  Capture(current_player_, action, direction, steps);
206  }
207  }
208 
209  if (NoValidActions()) { // check for end game state
210  int count_zero = DiskCount(Player(0));
211  int count_one = DiskCount(Player(1));
212  if (count_zero > count_one) {
213  outcome_ = Player(0);
214  } else if (count_zero < count_one) {
215  outcome_ = Player(1);
216  } else {
217  outcome_ = Player(kInvalidPlayer); // tie
218  }
219  current_player_ = Player(kTerminalPlayerId);
220  } else {
221  current_player_ = 1 - current_player_;
222  }
223 }

302 void OthelloState::ObservationTensor(Player player,
303  absl::Span<float> values) const {
304  SPIEL_CHECK_GE(player, 0);
305  SPIEL_CHECK_LT(player, num_players_);
306 
307  // Treat `values` as a 2-d tensor.
308  TensorView<2> view(values, {kCellStates, kNumCells}, true);
309 
310  for (int cell = 0; cell < kNumCells; ++cell) {
311  if (board_[cell] == CellState::kEmpty) {
312  view[{0, cell}] = 1;
313  } else if (board_[cell] == PlayerToState(player)) {
314  view[{1, cell}] = 1;
315  } else { // Opponent's piece
316  view[{2, cell}] = 1;
317  }
318  }
