71void TurnBasedSimultaneousState::DetermineWhoseTurn() {
72 if (state_->CurrentPlayer() == kSimultaneousPlayerId) {
73 // When the underlying game's node is at a simultaneous move node, they get
74 // rolled out as turn-based, starting with player 0.
75 current_player_ = -1;
76 rollout_mode_ = kStartRollout;
77 RolloutModeIncrementCurrentPlayer();
78 // If the rollout mode is used, then at least one player should have a valid
79 // action.
80 SPIEL_CHECK_LT(current_player_, num_players_);
81 } else {
82 // Otherwise, just execute it normally.
83 current_player_ = state_->CurrentPlayer();
84 rollout_mode_ = kNoRollout;
85 }
86}
87
88void TurnBasedSimultaneousState::RolloutModeIncrementCurrentPlayer() {
89 current_player_++;
90
91 // Make sure to skip over the players that do not have legal actions.
92 while (current_player_ < num_players_ &&
93 state_->LegalActions(current_player_).empty()) {
94 // Unnecessary to set an action here, but leads to a nicer ToString.
95 action_vector_[current_player_] = 0;
96 current_player_++;
97 }
98}

100void TurnBasedSimultaneousState::DoApplyAction(Action action_id) {
101 if (state_->IsChanceNode()) {
102 SPIEL_CHECK_FALSE(rollout_mode_);
103 state_->ApplyAction(action_id);
104 DetermineWhoseTurn();
105 } else {
106 if (rollout_mode_) {
107 // If we are currently rolling out a simultaneous move node, then simply
108 // buffer the action in the action vector.
109 rollout_mode_ = kMidRollout;
110 action_vector_[current_player_] = action_id;
111 RolloutModeIncrementCurrentPlayer();
112 // Check if we then need to apply it.
113 if (current_player_ == num_players_) {
114 state_->ApplyActions(action_vector_);
115 DetermineWhoseTurn();
116 }
117 } else {
118 SPIEL_CHECK_NE(state_->CurrentPlayer(), kSimultaneousPlayerId);
119 state_->ApplyAction(action_id);
120 DetermineWhoseTurn();
121 }
122 }
123}

165std::string TurnBasedSimultaneousState::InformationStateString(
166 Player player) const {
167 SPIEL_CHECK_GE(player, 0);
168 SPIEL_CHECK_LT(player, num_players_);
169
170 std::string extra_info = "";
171 extra_info = "Current player: ";
172 absl::StrAppend(&extra_info, current_player_);
173 extra_info.push_back('\n');
174 if (rollout_mode_) {
175 // Include the player's action if they have take one already.
176 if (player < current_player_) {
177 absl::StrAppend(&extra_info, "Observer's action this turn: ");
178 absl::StrAppend(&extra_info, action_vector_[player]);
179 extra_info.push_back('\n');
180 }
181 }
182 return extra_info + state_->InformationStateString(player);
183}

185void TurnBasedSimultaneousState::InformationStateTensor(
186 Player player, absl::Span<float> values) const {
187 SPIEL_CHECK_GE(player, 0);
188 SPIEL_CHECK_LT(player, num_players_);
189
190 SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorSize());
191 auto value_it = values.begin();
192
193 // First, get the 2 * num_players bits to encode whose turn it is and who
194 // the observer is.
195 for (auto p = Player{0}; p < num_players_; ++p) {
196 *value_it++ = (p == current_player_ ? 1 : 0);
197 }
198 for (auto p = Player{0}; p < num_players_; ++p) {
199 *value_it++ = (p == player ? 1 : 0);
200 }
201
202 // Then get the underlying observation
203 state_->InformationStateTensor(player,
204 absl::MakeSpan(value_it, values.end()));
205}

260GameType ConvertType(GameType type) {
261 type.dynamics = GameType::Dynamics::kSequential;
262 type.information = GameType::Information::kImperfectInformation;
263 type.short_name = kGameType.short_name;
264 type.long_name = "Turn-based " + type.long_name;
265 type.parameter_specification = kGameType.parameter_specification;
