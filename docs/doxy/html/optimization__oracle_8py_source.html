15"""Class of Optimization Oracles generating best response against opponents.
16
17Oracles are as defined in (Lanctot et Al., 2017,
18https://arxiv.org/pdf/1711.00832.pdf ), functions generating a best response
19against a probabilistic mixture of opponents. This class implements the abstract
20class of oracles, and a simple oracle using Evolutionary Strategy as
21optimization method.
22"""

27def strategy_sampler_fun(total_policies, probabilities_of_playing_policies):
28 """Samples strategies according to distribution over them.
29
30 Args:
31 total_policies: List of lists of policies for each player.
32 probabilities_of_playing_policies: List of numpy arrays representing the
33 probability of playing a strategy.
34
35 Returns:
36 One sampled joint strategy.
37 """
38 policies_selected = []
39 for k in range(len(total_policies)):
40 selected_opponent = np.random.choice(
41 total_policies[k],
42 1,
43 p=probabilities_of_playing_policies[k]).reshape(-1)[0]
44 policies_selected.append(selected_opponent)
45 return policies_selected

99 def evaluate_policy(self, game, pol, total_policies, current_player,
100 probabilities_of_playing_policies,
101 strategy_sampler=strategy_sampler_fun,
102 **oracle_specific_execution_kwargs):
103 """Evaluates a specific policy against a nash mixture of policies.
104
105 Args:
106 game: The game on which the optimization process takes place.
107 pol: The current policy, in policy.Policy, from which we wish to start
108 optimizing.
109 total_policies: A list of all policy.Policy strategies used for training,
110 including the one for the current player.
111 current_player: Integer representing the current player.
112 probabilities_of_playing_policies: A list of arrays representing, per
113 player, the probabilities of playing each policy in total_policies for
114 the same player.
115 strategy_sampler: callable sampling strategy.
116 **oracle_specific_execution_kwargs: Other set of arguments, for
117 compatibility purposes. Can for example represent whether to Rectify
118 Training or not.
119
120 Returns:
121 Average return for policy when played against policies_played_against.
122 """
123 del oracle_specific_execution_kwargs # Unused.

125 totals = 0
126 count = 0
127 for _ in range(self._number_policies_sampled):
128 policies_selected = strategy_sampler(total_policies,
129 probabilities_of_playing_policies)
130 policies_selected[current_player] = pol
131
132 new_return = self.sample_episode(
133 game,
134 policies_selected)[current_player]
135
136 totals += new_return
137 count += 1
138
139 # Avoid the 0 / 0 case.
140 return totals / max(1, count)
