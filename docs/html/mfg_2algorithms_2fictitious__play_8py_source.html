14 """Implementation of Fictitious Play from Perrin & al.
15 
16 Reference: https://arxiv.org/abs/2007.03458.
17 As presented, the Fictitious Play algorithm provides a robust approximation
18 scheme for Nash equilibrium by iteratively computing the best response
19 against the distribution induced by the average of the past best responses.
20 The provided formulation of Deep Fictitious Play mirrors this procedure,
21 but substitutes out the exact best reponse computation with an approximation
22 of best response values through a Reinforcement Learning approach (where
23 the RL method in question is a user-determined parameter for each iteration).
24 
25 Policy is initialized to uniform policy.
26 Each iteration:
27  1. Compute best response against policy
28  2. Update policy as weighted average of best response and current policy
29  (default learning rate is 1 / num_iterations + 1).
30 
31 To use fictitious play one should initialize it and run multiple iterations:
32 fp = FictitiousPlay(game)
33 for _ in range(num_iterations):
34  fp.iteration()
35 policy = fp.get_policy()
36 """

66  Args:
67  game: The game to analyze.
68  player_ids: list of player ids for which this policy applies; each should

70  policies: A `List[policy_std.Policy]` object.
71  distributions: A `List[distribution_std.Distribution]` object.
72  weights: A `List[float]` object. The elements should sum to 1.

75  self._policies_policies = policies
76  self._distributions_distributions = distributions
77  self._weights_weights = weights
78  assert len(policies) == len(distributions), (
79  f'Length mismatch {len(policies)} != {len(distributions)}')

81  f'Length mismatch {len(policies)} != {len(weights)}')

83  sum(weights),

86  def action_probabilities(self, state, player_id=None):
87  action_prob = []
88  legal = state.legal_actions()
89  num_legal = len(legal)
90  for a in legal:
91  merged_pi = 0.0
92  norm_merged_pi = 0.0
93  for p, d, w in zip(self._policies_policies, self._distributions_distributions, self._weights_weights):
94  try:
95  merged_pi += w * d(state) * p(state)[a]
96  norm_merged_pi += w * d(state)
97  except (KeyError, ValueError):
98  # This happens when the state was not observed in the merged
99  # distributions or policies.
100  pass
101  if norm_merged_pi > 0.0:
102  action_prob.append((a, merged_pi / norm_merged_pi))
103  else:
104  action_prob.append((a, 1.0 / num_legal))
105  return dict(action_prob)

108 class FictitiousPlay(object):
109  """Computes the value of a specified strategy."""
110 
111  def __init__(self,
112  game: pyspiel.Game,
113  lr: Optional[float] = None,
114  temperature: Optional[float] = None):
115  """Initializes the greedy policy.
116 
117  Args:
118  game: The game to analyze.
119  lr: The learning rate of mirror descent. If None, at iteration i it will
120  be set to 1/i.
121  temperature: If set, then instead of the greedy policy a softmax policy
122  with the specified temperature will be used to update the policy at each
123  iteration.
124  """
125  self._game_game = game
126  self._lr_lr = lr
127  self._temperature_temperature = temperature
128  self._policy_policy = policy_std.UniformRandomPolicy(self._game_game)
129  self._fp_step_fp_step = 0
130 
131  def get_policy(self):
132  return self._policy_policy
133 
134  def iteration(self, br_policy=None, learning_rate=None):
135  """Returns a new `TabularPolicy` equivalent to this policy.
136 
137  Args:
138  br_policy: Policy to compute the best response value for each iteration.
139  If none provided, the exact value is computed.
140  learning_rate: The learning rate.
141  """
142  self._fp_step_fp_step += 1
143 
144  distrib = distribution.DistributionPolicy(self._game_game, self._policy_policy)
145 
146  if br_policy:
147  br_value = policy_value.PolicyValue(self._game_game, distrib, br_policy)
148  else:
149  br_value = best_response_value.BestResponse(
150  self._game_game, distrib, value.TabularValueFunction(self._game_game))
151 
152  # Policy is either greedy or softmax with respect to the best response if
153  # temperature is specified.
154  player_ids = list(range(self._game_game.num_players()))
155  if self._temperature_temperature is None:
156  pi = greedy_policy.GreedyPolicy(self._game_game, player_ids, br_value)
157  else:
158  pi = softmax_policy.SoftmaxPolicy(self._game_game, player_ids,
159  self._temperature_temperature, br_value)
160  pi = pi.to_tabular()
161 
162  distrib_pi = distribution.DistributionPolicy(self._game_game, pi)
163 
164  if learning_rate:
165  weight = learning_rate
166  else:
167  weight = self._lr_lr if self._lr_lr else 1.0 / (self._fp_step_fp_step + 1)
168 
169  if math.isclose(weight, 1.0):
170  self._policy_policy = pi
171  else:
172  self._policy_policy = MergedPolicy(self._game_game, player_ids, [self._policy_policy, pi],
173  [distrib, distrib_pi],
174  [1.0 - weight, weight]).to_tabular()
