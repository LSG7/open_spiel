29// An basic implementation of Neller and Hnath 2011, "Approximating Optimal Dudo
30// Play with Fixed-Strategy Iteration Counterfactual Regret Minimization"
31// https://cupola.gettysburg.edu/csfac/2/.
32//
33// This implementation currently assumes the following:
34// - All chance events occur at the start of the game (before any decisions)
35// - There exists a perfect ranking between a player's chance event outcomes
36// and their outcome, encoded as a chance_id integer (one per player)
37//
38// This implementation was built for and only tested on Liar's dice. For a usage
39// example, see examples/fsicfr_liars_dice.cc.
40
41struct FSICFRNode {
42 // Maximum number of predecessor nodes (used for top sort order).
43 int max_predecessors = 0;
44
45 int id = -1;
46
47 // Chance id corresponding to the player to play at this node.
48 int chance_id = -1;
49
50 bool terminal = false;
51 double p0_utility = 0;
52
53 std::string string_key = "";
54 Player player = kInvalidPlayer;
55 int T = 0;
56 int visits = 0;
57 double v = 0;
58
59 // This is an (Action, other player chance id) -> node id map.
60 absl::flat_hash_map<std::pair<Action, int>, int> children;
61
62 std::vector<int> parent_ids;
63 std::vector<Action> legal_actions;
64
65 std::vector<double> ssum;
66 std::vector<double> psum;
67 std::vector<double> strategy;
68 std::vector<double> regrets;
69
70 FSICFRNode();
71 void AddChild(Action action, int chance_id, FSICFRNode* child);
72 std::string ToString();
73 void ApplyRegretMatching();
74};

76class FSICFRGraph {
77 public:

83 FSICFRNode* GetOrCreateTerminalNode(const std::string& terminal_string_key,
84 double p0_utility, int max_predecessors);
85 FSICFRNode* GetNode(int id) const {
86 if (id < 0 || id >= nodes_.size()) {
87 return nullptr;
88 } else {
89 FSICFRGraph* this_graph = const_cast<FSICFRGraph*>(this);
90 return &this_graph->nodes_[id];
91 }
92 }
93
94 int size() const { return nodes_.size(); }
95
96 // Topologically sort the graph (in order of non-decreasing max_predecessors).
97 void TopSort();
98
99 int ordered_node_id(int idx) const { return ordered_ids_[idx]; }
100
101 private:
102 // Infostate/terminal string key to node id map
103 absl::flat_hash_map<std::string, int> string_key_to_node_id_map_;
104
105 // Nodes. Ids correspond to indices.
106 std::vector<FSICFRNode> nodes_;
107
108 // Topologically sorted nodes ids. A more space-efficient implementation could
109 // remove this vector and simply build the node list in a such a way that
110 // nodes_ is already topologically-ordered.
111 std::vector<int> ordered_ids_;
112};
