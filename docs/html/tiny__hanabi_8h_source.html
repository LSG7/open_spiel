23 // This is the cooperative two-turn game defined in [1]
24 //
25 // Optimal score in this game is 10 (perfect cooperation).
26 // There is also a no-cooperation equilibrium scoring 8; some intermediate
27 // strategies are feasible also.
28 //
29 // Benchmark results:
30 // Bayesian Action Decoder 9.5 [1]
31 // Simplified Action Decoder 9.5 [2]
32 // Policy gradient / population-based training 9.0 [1]
33 // Independent Q learning 8.8 [2]
34 //
35 // Refs:
36 // [1] Bayesian Action Decoder, Foerster et al (2018)
37 // https://arxiv.org/abs/1811.01458
38 // [2] Simplified Action Decoder, under review (2019)

44 class TinyHanabiPayoffMatrix {
45  public:
46  int operator()(const std::vector<State::PlayerAction>& history) const {
47  SPIEL_CHECK_EQ(num_players_ * 2, history.size());
48  int idx = 0;
49  for (int i = 0; i < num_players_; ++i)
50  idx = (idx * num_chance_) + history[i].action;
51  for (int i = num_players_; i < 2 * num_players_; ++i)
52  idx = (idx * num_actions_) + history[i].action;
53  return payoff_[idx];
54  }
55  TinyHanabiPayoffMatrix(int num_players, int num_chance, int num_actions,
56  std::vector<int> payoff)
57  : num_players_(num_players),
58  num_chance_(num_chance),
59  num_actions_(num_actions),
60  payoff_(payoff) {
61  // Check payoff.size() == (num_chance * num_actions)**num_players
62  const int n = num_chance_ * num_actions_;
63  int expected_payoff_size = 1;
64  for (int i = 0; i < num_players; ++i) expected_payoff_size *= n;
65  SPIEL_CHECK_EQ(payoff_.size(), expected_payoff_size);
66  }
67 
68  int NumPlayers() const { return num_players_; }
69  int NumChance() const { return num_chance_; }
70  int NumActions() const { return num_actions_; }
71  int MinUtility() const { return *absl::c_min_element(payoff_); }
72  int MaxUtility() const { return *absl::c_max_element(payoff_); }
73 
74  private:
75  int num_players_;
76  int num_chance_;
77  int num_actions_;
78  std::vector<int> payoff_;
79 };
