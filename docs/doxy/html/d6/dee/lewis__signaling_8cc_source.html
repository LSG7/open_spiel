83std::vector<double> LewisSignalingState::Returns() const {
84 if (!IsTerminal()) {
85 return {0.0, 0.0};
86 } else {
87 // Find payoff from the payoff matrix based on state, action
88 int payoff_idx = num_states_ * state_ + action_;
89 return {payoffs_[payoff_idx], payoffs_[payoff_idx]};

93std::string LewisSignalingState::ObservationString(Player player) const {
94 SPIEL_CHECK_GE(player, 0);
95 SPIEL_CHECK_LT(player, num_players_);
96 if (IsChanceNode()) {
97 return "ChanceNode -- no observation";
98 }
99
100 std::string str = "";
101
102 // Whose turn is it?
103 absl::StrAppend(&str, "Current turn: ", cur_player_, "\n");
104
105 // Show state to the sender, message to the receiver
106 if (static_cast<Players>(player) == Players::kSender) {
107 absl::StrAppend(&str, "State: ", state_, "\n");
108 } else if (static_cast<Players>(player) == Players::kReceiver) {
109 absl::StrAppend(&str, "Message: ", message_, "\n");
110 } else {
111 SpielFatalError("Invalid player");
112 }
113
114 return str;
115}

117void LewisSignalingState::ObservationTensor(Player player,
118 absl::Span<float> values) const {
119 SPIEL_CHECK_GE(player, 0);
120 SPIEL_CHECK_LT(player, num_players_);
121
122 SPIEL_CHECK_EQ(values.size(), game_->ObservationTensorSize());
123 std::fill(values.begin(), values.end(), 0);
124
125 if (IsChanceNode()) {
126 // No observations at chance nodes.
127 return;
128 }
129
130 // 2 bits to indicate whose turn it is.
131 int offset = 0;
132 values[cur_player_] = 1;
133 offset += 2;
134
135 // 1 bit to indicate whether it's terminal
136 values[offset] = IsTerminal() ? 1 : 0;
137 offset += 1;
138
139 // one-hot vector for the state/message
140 if (static_cast<Players>(player) == Players::kSender) {
141 if (state_ != kUnassignedValue) {
142 values[offset + state_] = 1;
143 offset += num_states_;
144 }
145 } else if (static_cast<Players>(player) == Players::kReceiver) {
146 if (message_ != kUnassignedValue) {
147 values[offset + message_] = 1;
148 offset += num_messages_;
149 }
150 } else {
151 SpielFatalError("Invalid player");
152 }
153}

171void LewisSignalingState::DoApplyAction(Action action) {
172 if (IsChanceNode()) {
173 SPIEL_CHECK_LT(action, num_states_);
174 state_ = action;
175 cur_player_ = static_cast<int>(Players::kSender);
176 } else {
177 if (static_cast<Players>(cur_player_) == Players::kSender) {
178 SPIEL_CHECK_LT(action, num_messages_);
179 message_ = action;
180 cur_player_ = static_cast<int>(Players::kReceiver);
181 } else if (static_cast<Players>(cur_player_) == Players::kReceiver) {
182 action_ = action;
183 } else {
184 SpielFatalError("Invalid player");
185 }
186 }

189std::vector<Action> LewisSignalingState::LegalActions() const {
190 if (IsChanceNode()) {
191 return LegalChanceOutcomes();
192 } else if (IsTerminal()) {
193 return {};
194 } else if (static_cast<Players>(cur_player_) == Players::kSender) {
195 // Choose one of the messages if player is the sender
196 std::vector<Action> legal_actions;
197 legal_actions.reserve(num_messages_);
198 for (int i = 0; i < num_messages_; ++i) {
199 legal_actions.push_back(i);
200 }
201 return legal_actions;
202 } else if (static_cast<Players>(cur_player_) == Players::kReceiver) {
203 // Choose one of the actions if player is the receiver
204 std::vector<Action> legal_actions;
205 legal_actions.reserve(num_states_);
206 for (int i = 0; i < num_states_; ++i) {
207 legal_actions.push_back(i);
208 }
209 return legal_actions;
210 } else {
211 SpielFatalError("Invalid node");
212 }
213}

215std::vector<std::pair<Action, double>> LewisSignalingState::ChanceOutcomes()
216 const {
217 SPIEL_CHECK_TRUE(IsChanceNode());
218 std::vector<std::pair<Action, double>> outcomes;
219 outcomes.reserve(num_states_);
220 for (int i = 0; i < num_states_; ++i) {
221 outcomes.push_back({i, 1.0 / num_states_});
222 }
223 return outcomes;
224}

226std::string LewisSignalingState::ToString() const {
227 switch (history_.size()) {
228 case 0: // Before allocating state
229 return "Initial chance node";
230 break;
231
232 case 1: // After allocating state
233 return absl::StrCat("State ", state_);
234 break;
235
236 case 2: // After sending a message
237 return absl::StrCat("State ", state_, ", Message ", message_);
238 break;
239
240 case 3: // After taking an action
241 return absl::StrCat("State ", state_, ", Message ", message_, ", Action ",
242 action_);
243 break;
244
245 default:
246 SpielFatalError("Invalid state");
247 }
248}

254LewisSignalingGame::LewisSignalingGame(const GameParameters& params)
255 : Game(kGameType, params),
256 num_states_(ParameterValue<int>("num_states", kDefaultNumStates)),
257 num_messages_(ParameterValue<int>("num_messages", kDefaultNumMessages)) {
258 std::string payoffs_string =
259 ParameterValue<std::string>("payoffs", kDefaultPayoffs);
260 std::vector<std::string> parts = absl::StrSplit(payoffs_string, ',');
261 SPIEL_CHECK_EQ(parts.size(), num_states_ * num_states_);
262 payoffs_.resize(parts.size());
263 for (int i = 0; i < parts.size(); ++i) {
264 bool success = absl::SimpleAtod(parts[i], &payoffs_[i]);
265 SPIEL_CHECK_TRUE(success);
266 }
267 SPIEL_CHECK_LE(num_messages_, num_states_);
268}
