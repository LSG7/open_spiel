44 int STDCALL AnalysePlayBin(
45  deal dl,
46  playTraceBin play,
47  solvedPlay * solvedp,
48  int thrId)
49 {
50  if (! sysdep.ThreadOK(thrId))
51  return RETURN_THREAD_INDEX;
52 
53  ThreadData * thrp = memory.GetPtr(static_cast<unsigned>(thrId));
54 
55  moveType move;
56  futureTricks fut;
57 
58  int ret = SolveBoardInternal(thrp, dl, -1, 1, 1, &fut);
59  if (ret != RETURN_NO_FAULT)
60  return ret;
61 
62  const int numTricks = ((thrp->iniDepth + 3) >> 2) + 1;
63  const int numCardsPlayed = ((48 - thrp->iniDepth) % 4) + 1;
64 
65  int last_trick = (play.number + 3) / 4;
66  int last_card = ((play.number + 3) % 4) + 1;
67  if (last_trick >= numTricks)
68  {
69  last_trick = numTricks-1;
70  last_card = 4;
71  }
72  solvedp->number = 0;
73 
74  solvedp->tricks[0] = (numCardsPlayed % 2 == 1 ?
75  numTricks - fut.score[0] : fut.score[0]);
76  int hint = solvedp->tricks[0];
77  int hintDir;
78 
79  int running_remainder = numTricks;
80  int running_declarer = 0;
81  int running_player = dl.first;
82  int running_side = 1; /* defenders */
83  int start_side = running_player % 2;
84  int solved_declarer = solvedp->tricks[0];
85 #if DEBUG
86  int initial_par = solved_declarer;
87  fout.open("trace.txt", ofstream::out | ofstream::app);
88  fout << "Initial solve: " << initial_par << "\n";
89  fout << "no " << play.number << ", Last trick " << last_trick <<
90  ", last card " << last_card << "\n";
91  fout << setw(5) << "trick" << setw(6) << "card" <<
92  setw(6) << "rest" << setw(9) << "declarer" <<
93  setw(7) << "player" << setw(7) << "side" <<
94  setw(6) << "soln0" << setw(6) << "soln1" << setw(6) << "diff" << "\n";
95 #endif
96 
97  for (int trick = 1; trick <= last_trick; trick++)
98  {
99  int best_card = 0, best_suit = 0, best_player = 0, trump_played = 0;
100  int lc = (trick == last_trick ? last_card : 4);
101 
102  bool haveCurrent = (numCardsPlayed > 1 && trick == 1);
103  int offset = 4 * (trick - 1) - (numCardsPlayed - 1);
104 
105  for (int card = 1; card <= lc; card++)
106  {
107  int suit, rr;
108  bool usingCurrent = (haveCurrent && card < numCardsPlayed);
109  if (usingCurrent)
110  {
111  suit = dl.currentTrickSuit[card - 1];
112  rr = dl.currentTrickRank[card - 1];
113  }
114  else
115  {
116  suit = play.suit[offset + card - 1];
117  rr = play.rank[offset + card - 1];
118  }
119  unsigned hold = static_cast<unsigned>(bitMapRank[rr] << 2);
120 
121  move.suit = suit;
122  move.rank = rr;
123  move.sequence = rr;
124 
125  /* Keep track of the winner of the trick so far */
126  if (card == 1)
127  {
128  best_card = rr;
129  best_suit = suit;
130  best_player = dl.first;
131  trump_played = (suit == dl.trump ? 1 : 0);
132  }
133  else if (suit == dl.trump)
134  {
135  if (! trump_played || rr > best_card)
136  {
137  best_card = rr;
138  best_suit = suit;
139  best_player = running_player;
140  trump_played = 1;
141  }
142  }
143  else if (! trump_played && suit == best_suit && rr > best_card)
144  {
145  best_card = rr;
146  best_player = running_player;
147  }
148 
149  if ((dl.remainCards[running_player][suit] & hold) == 0)
150  {
151  if (! usingCurrent)
152  {
153 #if DEBUG
154  fout << "ERR Trick " << trick << " card " << card <<
155  " pl " << running_player << ": suit " << suit <<
156  " hold " << hold << "\n";
157  fout.close();
158 #endif
159  return RETURN_PLAY_FAULT;
160  }
161  }
162  else
163  dl.remainCards[running_player][suit] ^= hold;
164 
165 #if DEBUG
166  int resp_player = running_player;
167 #endif
168 
169  if (card == 4)
170  {
171  running_declarer += (best_player % 2 == start_side ? 0 : 1);
172  running_remainder--;
173 
174  if ((dl.first + best_player) % 2 == 0)
175  {
176  hintDir = 0; // Same side leads again; lower bound
177  hint = running_remainder - fut.score[0];
178  }
179  else
180  {
181  hintDir = 1; // Other ("our") side wins trick; upper bound
182  hint = fut.score[0] - 1;
183  }
184 
185  dl.first = best_player;
186  running_side = (dl.first % 2 == start_side ? 1 : 0);
187  running_player = dl.first;
188  }
189  else
190  {
191  running_player = (running_player + 1) % 4;
192  running_side = 1 - running_side;
193  hint = running_remainder - fut.score[0];
194  hintDir = 0;
195  }
196 
197  if (usingCurrent)
198  continue;
199 
200  if ((ret = AnalyseLaterBoard(thrp, dl.first, &move, hint,
201  hintDir, &fut))
202  != RETURN_NO_FAULT)
203  {
204 #if DEBUG
205  fout << "SolveBoard failed, ret " << ret << "\n";
206  fout.close();
207 #endif
208  return ret;
209  }
210 
211  int new_solved_decl = running_declarer + (running_side ?
212  running_remainder - fut.score[0] : fut.score[0]);
213 
214  solvedp->tricks[offset + card] = new_solved_decl;
215 
216 #if DEBUG
217  fout << setw(5) << trick <<
218  setw(6) << card <<
219  setw(6) << running_remainder <<
220  setw(9) << running_declarer <<
221  setw(7) << cardHand[resp_player] <<
222  setw(7) << running_side <<
223  setw(6) << solved_declarer <<
224  setw(6) << new_solved_decl <<
225  setw(6) << new_solved_decl - solved_declarer << "\n";
226 #endif
227 
228  solved_declarer = new_solved_decl;
229  }
230  }
231  solvedp->number = 4 * last_trick + last_card - 3 - (numCardsPlayed - 1);
232 
233 #if DEBUG
234  fout.close();
235 #endif
236  return RETURN_NO_FAULT;
237 }

240 int STDCALL AnalysePlayPBN(
241  dealPBN dlPBN,
242  playTracePBN playPBN,
243  solvedPlay * solvedp,
244  int thrId)
245 {
246  deal dl;
247  playTraceBin play;
248 
249  if (ConvertFromPBN(dlPBN.remainCards, dl.remainCards) !=
250  RETURN_NO_FAULT)
251  return RETURN_PBN_FAULT;
252 
253  dl.first = dlPBN.first;
254  dl.trump = dlPBN.trump;
255  for (int i = 0; i <= 2; i++)
256  {
257  dl.currentTrickSuit[i] = dlPBN.currentTrickSuit[i];
258  dl.currentTrickRank[i] = dlPBN.currentTrickRank[i];
259  }
260 
261  if (ConvertPlayFromPBN(playPBN, play) != RETURN_NO_FAULT)
262  return RETURN_PLAY_FAULT;
263 
264  return AnalysePlayBin(dl, play, solvedp, thrId);
