23 // Get all the information states. Note that there might contain duplicates.
24 void GetSubgameInformationStates(
25  State* state, std::vector<std::vector<std::string>>* all_info_states,
26  int depth_limit, int depth) {
27  if (state->IsTerminal()) {
28  return;
29  }
30 
31  if (depth_limit >= 0 && depth > depth_limit) {
32  return;
33  }
34 
35  for (auto action : state->LegalActions()) {
36  auto next_state = state->Clone();
37  next_state->ApplyAction(action);
38 
39  if (!state->IsChanceNode()) {
40  int player = state->CurrentPlayer();
41  SPIEL_CHECK_GE(player, 0);
42  SPIEL_CHECK_LT(player, state->NumPlayers());
43  (*all_info_states)[player].push_back(state->InformationStateString());
44  }
45 
46  GetSubgameInformationStates(next_state.get(), all_info_states, depth_limit,
47  depth + 1);
48  }
49 }

53 std::vector<std::vector<std::string>> GetAllInformationStates(const Game& game,
54  int depth_limit) {
55  // Get the root state.
56  std::unique_ptr<State> state = game.NewInitialState();
57  std::vector<std::vector<std::string>> all_infostates(game.NumPlayers());
58 
59  // Then, do a recursive tree walk to fill up the vector.
60  GetSubgameInformationStates(state.get(), &all_infostates, depth_limit, 0);
61 
62  // Remove duplicates by sorting the info states and calling std::unique.
63  for (Player p = 0; p < all_infostates.size(); ++p) {
64  absl::c_sort(all_infostates[p]);
65  auto last = std::unique(all_infostates[p].begin(), all_infostates[p].end());
66  all_infostates[p].erase(last, all_infostates[p].end());
67  }
68 
69  return all_infostates;
70 }
