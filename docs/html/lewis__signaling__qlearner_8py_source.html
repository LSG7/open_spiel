52 def run_experiment(num_players, env, payoffs, centralized):
53  """Run the experiments."""
54  num_states = FLAGS.num_states
55  num_messages = FLAGS.num_messages
56  num_actions = env.action_spec()["num_actions"]
57 
58  # Results to store
59  num_runs = FLAGS.num_runs
60  training_episodes = FLAGS.num_episodes
61  log_interval = FLAGS.log_interval
62  rewards = np.zeros((num_runs, training_episodes // log_interval))
63  opts = np.zeros((num_runs, training_episodes // log_interval))
64  converge_point = np.zeros((num_states, num_states))
65  percent_opt = 0
66 
67  # Repeat the experiment num_runs times
68  for i in range(num_runs):
69  eps_schedule = rl_tools.LinearSchedule(
70  FLAGS.eps_init, FLAGS.eps_final, FLAGS.eps_decay_steps *
71  2) # *2 since there are 2 agent steps per episode
72 
73  agents = [
74  # pylint: disable=g-complex-comprehension
75  tabular_qlearner.QLearner(
76  player_id=idx,
77  num_actions=num_actions,
78  step_size=FLAGS.step_size,
79  epsilon_schedule=eps_schedule,
80  centralized=centralized) for idx in range(num_players)
81  ]
82 
83  # 1. Train the agents
84  for cur_episode in range(training_episodes):
85  time_step = env.reset()
86  # Find cur_state for logging. See lewis_signaling.cc for info_state
87  # details.
88  cur_state = time_step.observations["info_state"][0][3:].index(1)
89  while not time_step.last():
90  player_id = time_step.observations["current_player"]
91  agent_output = agents[player_id].step(time_step)
92  time_step = env.step([agent_output.action])
93 
94  # Episode is over, step all agents with final info state.
95  for agent in agents:
96  agent.step(time_step)
97 
98  # Store rewards
99  reward = time_step.rewards[0]
100  max_reward = payoffs[cur_state].max()
101  cur_idx = (i, cur_episode // log_interval)
102  rewards[cur_idx] += reward / log_interval
103  opts[cur_idx] += np.isclose(reward, max_reward) / log_interval
104 
105  base_info_state0 = [1.0, 0.0, 0.0] + [0.0] * num_states
106  base_info_state1 = [0.0, 1.0, 0.0] + [0.0] * num_states
107  if centralized:
108  base_info_state0 = [base_info_state0, base_info_state0.copy()]
109  base_info_state1 = [base_info_state1, base_info_state1.copy()]
110 
111  for s in range(num_states):
112  info_state0 = copy.deepcopy(base_info_state0)
113  if centralized:
114  info_state0[0][3 + s] = 1.0
115  else:
116  info_state0[3 + s] = 1.0
117  # pylint: disable=protected-access
118  m, _ = agents[0]._epsilon_greedy(
119  str(info_state0), np.arange(num_messages), 0)
120  info_state1 = copy.deepcopy(base_info_state1)
121  if centralized:
122  info_state1[0][3 + s] = 1.0
123  info_state1[1][3 + m] = 1.0
124  else:
125  info_state1[3 + m] = 1.0
126  a, _ = agents[1]._epsilon_greedy(
127  str(info_state1), np.arange(num_states), 0)
128  converge_point[s, a] += 1
129  best_act = payoffs[s].argmax()
130  percent_opt += int(a == best_act) / num_runs / num_states
131  return rewards, opts, converge_point, percent_opt

134 def main(_):
135  game = "lewis_signaling"
136  num_players = 2
137 
138  num_states = FLAGS.num_states
139  num_messages = FLAGS.num_messages
140  if FLAGS.payoffs == "random":
141  payoffs = np.random.random((num_states, num_states))
142  payoffs_str = ",".join([str(x) for x in payoffs.flatten()])
143  elif FLAGS.payoffs == "climbing":
144  # This is a particular payoff matrix that is hard for decentralized
145  # algorithms. Introduced in C. Claus and C. Boutilier, "The dynamics of
146  # reinforcement learning in cooperative multiagent systems", 1998, for
147  # simultaneous action games, but it is difficult even in the case of
148  # signaling games.
149  payoffs = np.array([[11, -30, 0], [-30, 7, 6], [0, 0, 5]]) / 30
150  payoffs_str = ",".join([str(x) for x in payoffs.flatten()])
151  else:
152  payoffs_str = FLAGS.payoffs
153  try:
154  payoffs_list = [float(x) for x in payoffs_str.split(",")]
155  payoffs = np.array(payoffs_list).reshape((num_states, num_states))
156  except ValueError:
157  raise ValueError(
158  "There should be {} (states * actions) elements in payoff. Found {} elements"
159  .format(num_states * num_states, len(payoffs_list))) from None
160 
161  env_configs = {
162  "num_states": num_states,
163  "num_messages": num_messages,
164  "payoffs": payoffs_str
165  }
166 
167  env = rl_environment.Environment(game, **env_configs)
168 
169  if FLAGS.compare:
170  rewards_list = []
171  opts_list = []
172  converge_point_list = []
173  percent_opt_list = []
174  for centralized in [True, False]:
175  rewards, opts, converge_point, percent_opt = run_experiment(
176  num_players, env, payoffs, centralized)
177  rewards_list += [rewards]
178  opts_list += [opts]
179  converge_point_list += [converge_point]
180  percent_opt_list += [percent_opt]
181  else:
182  rewards, opts, converge_point, percent_opt = run_experiment(
183  num_players, env, payoffs, FLAGS.centralized)
184  rewards_list = [rewards]
185  opts_list = [opts]
186  converge_point_list = [converge_point]
187  percent_opt_list = [percent_opt]
188 
189  if FLAGS.plot:

209  def plot_scalars(scalars,
210  repetition_axis=0,
211  scalar_labels=None,
212  title=None,
213  ax_labels=None):
214  """Plots scalar on ax by filling 1 standard error.
215 
216  Args:
217  scalars: List of scalars to plot (mean taken over repetition
218  axis)
219  repetition_axis: Axis to take the mean over
220  scalar_labels: Labels for the scalars (for legend)
221  title: Figure title
222  ax_labels: Labels for x and y axis (list of 2 strings)
223  """
224  if not all([len(s.shape) == 2 for s in scalars]):
225  raise ValueError("Only 2D arrays supported for plotting")
226 
227  if scalar_labels is None:
228  scalar_labels = [None] * len(scalars)
229 
230  if len(scalars) != len(scalar_labels):
231  raise ValueError(
232  "Wrong number of scalar labels, expected {} but received {}".format(
233  len(scalars), len(scalar_labels)))
234 
235  _, plot_axis = init_fig()
236  for i, scalar in enumerate(scalars):
237  xs = np.arange(scalar.shape[1 - repetition_axis]) * FLAGS.log_interval
238  mean = scalar.mean(axis=repetition_axis)
239  sem = stats.sem(scalar, axis=repetition_axis)
240  plot_axis.plot(xs, mean, label=scalar_labels[i])
241  plot_axis.fill_between(xs, mean - sem, mean + sem, alpha=0.5)
242 
243  if title is not None:
244  plot_axis.set_title(title)
245  if ax_labels is not None:
246  plot_axis.set_xlabel(ax_labels[0])
247  plot_axis.set_ylabel(ax_labels[1])

249  def plot_confusion_matrix(cm, cmap=plt.cm.Blues, title=None):
250  """Plots the confusion matrix.
251 
252  Args:
253  cm (np.ndarray): Confusion matrix to plot
254  cmap: Color map to be used in matplotlib's imshow
255  title: Figure title
256 
257  Returns:
258  Figure and axis on which the confusion matrix is plotted
259  """
260  fig, ax = plt.subplots()
261  ax.imshow(cm, interpolation="nearest", cmap=cmap)
262  ax.set_xticks([])
263  ax.set_yticks([])
264  ax.set_xlabel("Receiver's action", fontsize=14)
265  ax.set_ylabel("Sender's state", fontsize=14)
266  # Loop over data dimensions and create text annotations.
267  fmt = "d"
268  thresh = cm.max() / 2.
269  for i in range(cm.shape[0]):
270  for j in range(cm.shape[1]):
271  ax.text(
272  j,
273  i,
274  format(cm[i, j], fmt),
275  ha="center",
276  va="center",
277  color="white" if cm[i, j] > thresh else "black")
278  fig.tight_layout()
279  if title is not None:
280  ax.set_title(title)
281  return fig, ax
282 
283  if FLAGS.compare:
284  labels = ["Centralized", "Decentralized"]
285  else:
286  labels = ["Centralized"] if FLAGS.centralized else ["Decentralized"]
287  plot_scalars(
288  rewards_list,
289  scalar_labels=labels,
290  title="Reward graph (Tabular Q-Learning)",
291  ax_labels=["Episodes", "Reward per episode"])
292  plt.legend()
293  plot_scalars(
294  opts_list,
295  scalar_labels=labels,
296  title="Percentage of optimal actions (Tabular Q-Learning)",
297  ax_labels=["Episodes", "% optimal actions"])
298  plt.legend()
299 
300  for i, cp in enumerate(converge_point_list):
301  plot_confusion_matrix(
302  cp.astype(int),
303  title="Final policy (Tabular {})".format(labels[i]))
304 
305  plt.show()
306 
307  return percent_opt_list
