94  @parameterized.parameters(*_DATA)
95  def test_solution_fixed_point(self, game, inverse_alpha, gambit_qre_sol):
96  # Check if a QRE solution is a fixed point of MMD
97  mmd = mmd_dilated.MMDDilatedEnt(game, 1. / inverse_alpha)
98  mmd.sequences = copy.deepcopy(gambit_qre_sol)
99  mmd.update_sequences()
100  np.testing.assert_allclose(
101  mmd.current_sequences()[0], gambit_qre_sol[0], rtol=1e-6)
102  np.testing.assert_allclose(
103  mmd.current_sequences()[1], gambit_qre_sol[1], rtol=1e-6)

111  @parameterized.parameters((0.), (0.5), (1.), (1.5))
112  def test_rps_update(self, alpha):
113  game = pyspiel.load_game_as_turn_based('matrix_rps')
114  start_sequences = [
115  np.array([1, 0.2, 0.2, 0.6]),
116  np.array([1, 0.5, 0.2, 0.3])
117  ]
118  mmd = mmd_dilated.MMDDilatedEnt(game, alpha)
119  mmd.sequences = copy.deepcopy(start_sequences)
120 
121  mmd.update_sequences()
122  updated_sequences = copy.deepcopy(start_sequences)
123  # manually perform update for p1
124  updated_sequences[0][1:] = updated_sequences[0][1:] * np.exp(
125  mmd.stepsize * -mmd.payoff_mat[1:, 1:] @ start_sequences[1][1:])
126  updated_sequences[0][1:] = updated_sequences[0][1:]**(
127  1. / (1 + mmd.stepsize * alpha))
128  updated_sequences[0][1:] = updated_sequences[0][1:] / np.sum(
129  updated_sequences[0][1:])
130  np.testing.assert_allclose(mmd.current_sequences()[0], updated_sequences[0])
131 
132  # manually perform update for p2
133  updated_sequences[1][1:] = updated_sequences[1][1:] * np.exp(
134  mmd.stepsize * mmd.payoff_mat[1:, 1:].T @ start_sequences[0][1:])
135  updated_sequences[1][1:] = updated_sequences[1][1:]**(
136  1. / (1 + mmd.stepsize * alpha))
137  updated_sequences[1][1:] = updated_sequences[1][1:] / np.sum(
138  updated_sequences[1][1:])
139  np.testing.assert_allclose(mmd.current_sequences()[1], updated_sequences[1])
