29  """Returns a Random Uniform distribution on policies.
30 
31  Args:
32  solver: GenPSROSolver instance.
33  return_joint: If true, only returns marginals. Otherwise marginals as well

35 
36  Returns:
37  uniform distribution on strategies.
38  """

83 def renormalize(probabilities):
84  """Replaces all negative entries with zeroes and normalizes the result.
85 
86  Args:
87  probabilities: probability vector to renormalize. Has to be one-dimensional.
88 
89  Returns:
90  Renormalized probabilities.
91  """
92  probabilities[probabilities < 0] = 0
93  probabilities = probabilities / np.sum(probabilities)
94  return probabilities

97 def get_joint_strategy_from_marginals(probabilities):
98  """Returns a joint strategy matrix from a list of marginals.
99 
100  Args:
101  probabilities: list of probabilities.
102 
103  Returns:
104  A joint strategy from a list of marginals.
105  """
106  probas = []
107  for i in range(len(probabilities)):
108  probas_shapes = [1] * len(probabilities)
109  probas_shapes[i] = -1
110  probas.append(probabilities[i].reshape(*probas_shapes))
111  result = np.prod(probas)
112  return result.reshape(-1)

115 def nash_strategy(solver, return_joint=False):
116  """Returns nash distribution on meta game matrix.
117 
118  This method only works for two player zero-sum games.
119 
120  Args:
121  solver: GenPSROSolver instance.
122  return_joint: If true, only returns marginals. Otherwise marginals as well
123  as joint probabilities.
124 
125  Returns:
126  Nash distribution on strategies.
127  """
128  meta_games = solver.get_meta_game()
129  if not isinstance(meta_games, list):
130  meta_games = [meta_games, -meta_games]
131  meta_games = [x.tolist() for x in meta_games]
132  if len(meta_games) != 2:
133  raise NotImplementedError(
134  "nash_strategy solver works only for 2p zero-sum"
135  "games, but was invoked for a {} player game".format(len(meta_games)))
136  nash_prob_1, nash_prob_2, _, _ = (
137  lp_solver.solve_zero_sum_matrix_game(
138  pyspiel.create_matrix_game(*meta_games)))
139  result = [
140  renormalize(np.array(nash_prob_1).reshape(-1)),
141  renormalize(np.array(nash_prob_2).reshape(-1))
142  ]
143 
144  if not return_joint:
145  return result
146  else:
147  joint_strategies = get_joint_strategy_from_marginals(result)
