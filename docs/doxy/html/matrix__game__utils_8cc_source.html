29std::shared_ptr<const MatrixGame> LoadMatrixGame(const std::string& name) {
30 std::shared_ptr<const Game> game = LoadGame(name);
31 // Make sure it is indeed a matrix game.
32 const MatrixGame* matrix_game = dynamic_cast<const MatrixGame*>(game.get());
33 if (matrix_game == nullptr) {
34 // If it is not already a matrix game, check if it is a 2-player NFG.
35 // If so, convert it.
36 const NormalFormGame* nfg = dynamic_cast<const NormalFormGame*>(game.get());
37 if (nfg != nullptr && nfg->NumPlayers() == 2) {
38 return AsMatrixGame(nfg);
39 } else {
40 SpielFatalError(absl::StrCat("Cannot load ", name, " as a matrix game."));
41 }
42 }
43 return std::static_pointer_cast<const MatrixGame>(game);
44}

52std::shared_ptr<const MatrixGame> AsMatrixGame(const NormalFormGame* game) {
53 SPIEL_CHECK_EQ(game->NumPlayers(), 2);
54 std::unique_ptr<State> initial_state = game->NewInitialState();
55 std::vector<std::vector<Action>> legal_actions = {
56 initial_state->LegalActions(0), initial_state->LegalActions(1)};
57
58 std::vector<std::string> row_names;
59 std::vector<std::string> col_names;
60 std::vector<double> row_utils;
61 std::vector<double> col_utils;
62 int num_rows = legal_actions[0].size();
63 int num_cols = legal_actions[1].size();
64
65 GameType type = game->GetType();
66 type.min_num_players = 2;
67 type.max_num_players = 2;
68
69 for (int r = 0; r < num_rows; ++r) {
70 Action row_action = legal_actions[0][r];
71 row_names.push_back(initial_state->ActionToString(0, row_action));
72
73 for (int c = 0; c < num_cols; ++c) {
74 Action col_action = legal_actions[1][c];
75 if (col_names.size() < num_cols) {
76 col_names.push_back(initial_state->ActionToString(1, col_action));
77 }
78
79 std::unique_ptr<State> clone = initial_state->Clone();
80 clone->ApplyActions({row_action, col_action});
81 SPIEL_CHECK_TRUE(clone->IsTerminal());
82 std::vector<double> returns = clone->Returns();
83 SPIEL_CHECK_EQ(returns.size(), 2);
84
85 row_utils.push_back(returns[0]);
86 col_utils.push_back(returns[1]);
87 }
88 }
89
90 return std::shared_ptr<MatrixGame>(
91 new MatrixGame(type, {}, row_names, col_names, row_utils, col_utils));
92}

94std::shared_ptr<const MatrixGame> ExtensiveToMatrixGame(const Game& game) {
95 SPIEL_CHECK_EQ(game.NumPlayers(), 2);
96
97 std::vector<std::string> row_names;
98 std::vector<std::string> col_names;
99 std::vector<std::vector<double>> row_player_utils;
100 std::vector<std::vector<double>> col_player_utils;
101
102 GameType type = game.GetType();
103
104 std::vector<DeterministicTabularPolicy> policies = {
105 DeterministicTabularPolicy(game, 0), DeterministicTabularPolicy(game, 1)};
106
107 bool first_row = true;
108 do {
109 policies[1].ResetDefaultPolicy();
110 row_names.push_back(policies[0].ToString(" --- "));
111 std::vector<double> row_utils;
112 std::vector<double> col_utils;
113 do {
114 if (first_row) {
115 col_names.push_back(policies[1].ToString(" --- "));
116 }
117 std::unique_ptr<State> state = game.NewInitialState();
118 std::vector<double> returns =
119 ExpectedReturns(*state, {&policies[0], &policies[1]}, -1);
120 row_utils.push_back(returns[0]);
121 col_utils.push_back(returns[1]);
122 } while (policies[1].NextPolicy());
123 first_row = false;
124 row_player_utils.push_back(row_utils);
125 col_player_utils.push_back(col_utils);
126 } while (policies[0].NextPolicy());
127
128 return matrix_game::CreateMatrixGame(type.short_name, type.long_name,
129 row_names, col_names, row_player_utils,
130 col_player_utils);
131}
