15"""Weighted Voting Games.
16
17A weighted voting game is a game where every player i has a weight w_i, and
18there is a fixed quota q, the characteristic function for coalition c is:
19
20 v(c) = 1 if sum_{i in c} w_i > q,
21 0 otherwise.
22
23For more detail, see Chapter 4 of "Computational Aspects of Cooperative
24Game Theory" text book by Georgios Chalkiadakis, Edith Elkind, and Michael
25Wooldridge.
26"""
27
28import numpy as np
29from open_spiel.python.coalitional_games import coalitional_game

31
32class WeightedVotingGame(coalitional_game.CoalitionalGame):
33 """Weighted Voting Game."""
34
35 def __init__(self, weights: np.ndarray, quota: float):
36 super().__init__(num_players=len(weights))
37 assert len(weights.shape) == 1
38 self._weights = weights
39 self._quota = quota
40
41 def coalition_value(self, coalition: np.ndarray) -> float:
42 assert len(coalition) == self._num_players
43 total_weight = np.inner(coalition, self._weights)
44 if total_weight > self._quota:
45 return 1.0
46 else:
47 return 0.0
