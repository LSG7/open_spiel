52 void SequenceFormLpSpecification::SpecifyReachProbsConstraints(
53  InfostateNode* player_node) {
54  node_spec_[player_node].var_reach_prob = solver_->MakeNumVar(
55  /*lb=*/0.0, /*ub=*/1., "");
56 
57  if (player_node->type() == kTerminalInfostateNode) return; // Nothing to do.
58  if (player_node->type() == kObservationInfostateNode) {
59  for (InfostateNode* player_child : player_node->child_iterator()) {
60  SpecifyReachProbsConstraints(player_child);
61 
62  // Equality constraint: parent = child
63  opres::MPConstraint* ct = node_spec_[player_child].ct_parent_reach_prob =
64  solver_->MakeRowConstraint(/*lb=*/0, /*ub=*/0, "");
65  ct->SetCoefficient(node_spec_[player_node].var_reach_prob, -1);
66  ct->SetCoefficient(node_spec_[player_child].var_reach_prob, 1);
67  }
68  return;
69  }
70  if (player_node->type() == kDecisionInfostateNode) {
71  // Equality constraint: parent = sum of children
72  opres::MPConstraint* ct = node_spec_[player_node].ct_child_reach_prob =
73  solver_->MakeRowConstraint(/*lb=*/0, /*ub=*/0, "");
74  ct->SetCoefficient(node_spec_[player_node].var_reach_prob, -1);
75  for (InfostateNode* player_child : player_node->child_iterator()) {
76  SpecifyReachProbsConstraints(player_child);
77  ct->SetCoefficient(node_spec_[player_child].var_reach_prob, 1);
78  }
79  return;
80  }
81 
82  SpielFatalError("Exhausted pattern match!");
83 }

85 void SequenceFormLpSpecification::SpecifyCfValuesConstraints(
86  InfostateNode* opponent_node) {
87  node_spec_[opponent_node].var_cf_value = solver_->MakeNumVar(
88  /*lb=*/-opres::MPSolver::infinity(),
89  /*ub=*/opres::MPSolver::infinity(), "");
90 
91  if (opponent_node->type() == kDecisionInfostateNode) {
92  for (InfostateNode* opponent_child : opponent_node->child_iterator()) {
93  SpecifyCfValuesConstraints(opponent_child);
94  opres::MPConstraint* ct = node_spec_[opponent_child].ct_parent_cf_value =
95  solver_->MakeRowConstraint();
96  ct->SetUB(0.);
97  ct->SetCoefficient(node_spec_[opponent_node].var_cf_value, -1);
98  ct->SetCoefficient(node_spec_[opponent_child].var_cf_value, 1);
99  }
100  return;
101  }
102 
103  opres::MPConstraint* ct = node_spec_[opponent_node].ct_child_cf_value =
104  solver_->MakeRowConstraint();
105  ct->SetUB(0.);
106  ct->SetCoefficient(node_spec_[opponent_node].var_cf_value, -1);
107 
108  if (opponent_node->type() == kTerminalInfostateNode) {
109  const std::map<const InfostateNode*, const InfostateNode*>& terminal_map =
110  terminal_bijection_.association(opponent_node->tree().acting_player());
111  const InfostateNode* player_node = terminal_map.at(opponent_node);
112  const double value = opponent_node->terminal_utility() *
113  opponent_node->terminal_chance_reach_prob();
114  // Terminal value constraint comes from the opponent.
115  ct->SetCoefficient(node_spec_[player_node].var_reach_prob, value);
116  return;
117  }
118  if (opponent_node->type() == kObservationInfostateNode) {
119  // Value constraint: sum of children = parent
120  ct->SetLB(0.);
121  for (InfostateNode* opponent_child : opponent_node->child_iterator()) {
122  SpecifyCfValuesConstraints(opponent_child);
123  ct->SetCoefficient(node_spec_[opponent_child].var_cf_value, 1);
124  }
125  return;
126  }
127 
128  SpielFatalError("Exhausted pattern match!");
129 }

186 TabularPolicy SequenceFormLpSpecification::OptimalPolicy(Player for_player) {
187  SPIEL_CHECK_TRUE(for_player == 0 || for_player == 1);
188  const InfostateTree* tree = trees_[for_player].get();
189  TabularPolicy policy;
190  for (DecisionId id : tree->AllDecisionIds()) {
191  const InfostateNode* node = tree->decision_infostate(id);
192  absl::Span<const Action> actions = node->legal_actions();
193  SPIEL_CHECK_EQ(actions.size(), node->num_children());
194  ActionsAndProbs state_policy;
195  state_policy.reserve(node->num_children());
196  double rp_sum = 0.;
197  for (int i = 0; i < actions.size(); ++i) {
198  rp_sum += node_spec_[node->child_at(i)].var_reach_prob->solution_value();
199  }
200  for (int i = 0; i < actions.size(); ++i) {
201  double prob;
202  if (rp_sum) {
203  prob = node_spec_[node->child_at(i)].var_reach_prob->solution_value() /
204  rp_sum;
205  } else {
206  // If the infostate is unreachable, the strategy is not defined.
207  // However some code in the library may require having the strategy,
208  // so we just put an uniform strategy here.
209  prob = 1. / actions.size();
210  }
211  state_policy.push_back({actions[i], prob});
212  }
213  policy.SetStatePolicy(node->infostate_string(), state_policy);
214  }
215  return policy;
216 }

229 BijectiveContainer<const InfostateNode*> ConnectTerminals(
230  const InfostateTree& tree_a, const InfostateTree& tree_b) {
231  BijectiveContainer<const InfostateNode*> out;
232 
233  using History = absl::Span<const Action>;
234  absl::btree_map<History, const InfostateNode*> history_map;
235  for (InfostateNode* node_b : tree_b.leaf_nodes()) {
236  history_map[node_b->TerminalHistory()] = node_b;
237  }
238 
239  for (InfostateNode* node_a : tree_a.leaf_nodes()) {
240  const InfostateNode* node_b = history_map[node_a->TerminalHistory()];
241  out.put({node_a, node_b});
242  }
243  return out;
244 }
