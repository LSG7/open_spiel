32// Dark chess - imperfect information version of chess:
33// https://en.wikipedia.org/wiki/Dark_chess
34//
35// Parameters:
36// "board_size" int Number of squares in each row and column (default: 8)
37// "fen" string String describing the chess board position in
38// Forsyth-Edwards Notation. The FEN has to match
39// the board size. Default values are available for

130 // RepetitionTable records how many times the given hash exists in the history
131 // stack (including the current board).
132 // We are already indexing by board hash, so there is no need to hash that
133 // hash again, so we use a custom passthrough hasher.
134 class PassthroughHash {
135 public:
136 std::size_t operator()(uint64_t x) const {
137 return static_cast<std::size_t>(x);
138 }
139 };
140 using RepetitionTable = absl::flat_hash_map<uint64_t, int, PassthroughHash>;
141 RepetitionTable repetitions_;
142 mutable absl::optional<std::vector<Action>> cached_legal_actions_;

145// Game object.
146class DarkChessGame : public Game {
147 public:
148 explicit DarkChessGame(const GameParameters& params);
149 int NumDistinctActions() const override {
150 return chess::NumDistinctActions();
151 }
152 std::unique_ptr<State> NewInitialState() const override {
153 return absl::make_unique<DarkChessState>(shared_from_this(), board_size_,
154 fen_);
155 }
156 int NumPlayers() const override { return chess::NumPlayers(); }
157 double MinUtility() const override { return LossUtility(); }
158 absl::optional<double> UtilitySum() const override { return DrawUtility(); }
159 double MaxUtility() const override { return WinUtility(); }
160 std::vector<int> ObservationTensorShape() const override {
161 std::vector<int> shape{
162 (13 + // public boards: piece types * colours + empty
163 14) // private boards: piece types * colours + empty + unknown
164 * board_size_ * board_size_ +
165 3 + // public: repetitions count, one-hot encoding
166 2 + // public: side to play
167 1 + // public: irreversible move counter -- a fraction of $n over 100
168 2 * 2 // private: left/right castling rights, one-hot encoded.
169 };
170 return shape;
171 }
172 int MaxGameLength() const override { return chess::MaxGameLength(); }
173 std::shared_ptr<Observer> MakeObserver(
174 absl::optional<IIGObservationType> iig_obs_type,
175 const GameParameters& params) const;
176
177 std::shared_ptr<DarkChessObserver> default_observer_;
178
179 private:
180 const int board_size_;
181 const std::string fen_;
182};
