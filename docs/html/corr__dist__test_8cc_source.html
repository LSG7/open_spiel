42 void TestGibson13MatrixGameExample() {
43  // Tests that the example from Sec 2.2 of Gibson 2013, Regret Minimization in
44  // Non-Zero-Sum Games with Applications to Building Champion Multiplayer
45  // Computer Poker Agents (https://arxiv.org/abs/1305.0034).
46  //
47  // a b
48  // A 1,0 0,0
49  // B 0,0 2,0
50  // C -1,0 1,0
51  //
52  std::shared_ptr<const matrix_game::MatrixGame> gibson_game =
53  matrix_game::CreateMatrixGame({{1, 0}, {0, 2}, {-1, 1}},
54  {{0, 0}, {0, 0}, {0, 0}});
55 
56  NormalFormCorrelationDevice mu = {
57  {0.5, {0, 0}}, // (A, a) = 0.5
58  {0.25, {1, 1}}, // (B, b) = 0.25
59  {0.25, {2, 1}} // (C, b) = 0.25
60  };
61 
62  // mu is a CCE.
63  SPIEL_CHECK_TRUE(Near(CCEDist(*gibson_game, mu), 0.0));
64 
65  // mu is not a CE, because first player gains 1 by deviating to B after
66  // receiving the third recommendation, which happens with prob 0.25
67  SPIEL_CHECK_TRUE(Near(CEDist(*gibson_game, mu), 0.25));
68 
69  // Repeat these tests with a turn-based simultaneous game.
70  SPIEL_CHECK_TRUE(Near(CCEDist(*ConvertToTurnBased(*gibson_game), mu), 0.0));
71  SPIEL_CHECK_TRUE(Near(CEDist(*ConvertToTurnBased(*gibson_game), mu), 0.25));
72 }

74 void TestShapleysGame() {
75  // Shapley's game is a general-sum version of Rock, Paper, Scissors.
76  // See Fig 7.6 of rown '09, http://www.masfoundations.org/mas.pdf:
77  //
78  // R P S
79  // R 0,0 0,1 1,0
80  // P 1,0 0,0 0,1
81  // S 0,1 1,0 0,0
82  std::shared_ptr<const Game> shapleys_game = LoadGame("matrix_shapleys_game");
83 
84  // There is a unique Nash eq at (1/3, 1/3, 1/3). So by Sec 3.4.5 of Shoham
85  // and Leyton-Brown there is a CE with 1/9 on all the entries.
86  NormalFormCorrelationDevice mu = {
87  {1.0 / 9.0, {0, 0}}, {1.0 / 9.0, {0, 1}}, {1.0 / 9.0, {0, 2}},
88  {1.0 / 9.0, {1, 0}}, {1.0 / 9.0, {1, 1}}, {1.0 / 9.0, {1, 2}},
89  {1.0 / 9.0, {2, 0}}, {1.0 / 9.0, {2, 1}}, {1.0 / 9.0, {2, 2}}};
90 
91  SPIEL_CHECK_TRUE(Near(CEDist(*shapleys_game, mu), 0.0));
92  std::vector<double> expected_values_full_support =
93  ExpectedValues(*shapleys_game, mu);
94  SPIEL_CHECK_TRUE(Near(expected_values_full_support[0], 1.0 / 3.0));
95  SPIEL_CHECK_TRUE(Near(expected_values_full_support[1], 1.0 / 3.0));
96 
97  // There is another CE with 1/6 on the off-diagonals.
98  mu = {{1.0 / 6.0, {0, 1}}, {1.0 / 6.0, {0, 2}}, {1.0 / 6.0, {1, 0}},
99  {1.0 / 6.0, {1, 2}}, {1.0 / 6.0, {2, 0}}, {1.0 / 6.0, {2, 1}}};
100 
101  SPIEL_CHECK_TRUE(Near(CEDist(*shapleys_game, mu), 0.0));
102  std::vector<double> expected_values_off_diagonals =
103  ExpectedValues(*shapleys_game, mu);
104  SPIEL_CHECK_TRUE(Near(expected_values_off_diagonals[0], 0.5));
105  SPIEL_CHECK_TRUE(Near(expected_values_off_diagonals[1], 0.5));
106 }

132 void TestSignalingExampleVonStengelForges2008() {
133  // Runs a test based on the signaling game example in Section 2.3 of von
134  // Stengel & Forges 2008, Extensive-Form Correlated Equilibrium:
135  // Definition and Computational Complexity.
136 
137  // First, check the CE of the normal-form version in Figure 2.
138  std::shared_ptr<const matrix_game::MatrixGame> signaling_game_nfg =
139  matrix_game::CreateMatrixGame(
140  {{5, 5, 0, 0}, {5, 2, 3, 0}, {5, 3, 2, 0}, {5, 0, 5, 0}},
141  {{5, 5, 6, 6}, {5, 8, 3, 6}, {5, 3, 8, 6}, {5, 6, 5, 6}});
142 
143  // Mix equally a'' = b'' = c'' == d'' = 1/4.
144  NormalFormCorrelationDevice mu_nfg = {
145  {0.25, {0, 3}}, {0.25, {1, 3}}, {0.25, {2, 3}}, {0.25, {3, 3}}};
146  SPIEL_CHECK_TRUE(Near(CEDist(*signaling_game_nfg, mu_nfg), 0.0));
147  std::vector<double> expected_values =
148  ExpectedValues(*signaling_game_nfg, mu_nfg);
149  SPIEL_CHECK_TRUE(Near(expected_values[0], 0.0));
150  SPIEL_CHECK_TRUE(Near(expected_values[1], 6.0));
151 
152  // Now do the extensive-form version. From von Stengel & Forges '08:
153  std::shared_ptr<const Game> efg_game =
154  efg_game::LoadEFGGame(efg_game::GetSignalingEFGData());
155  const efg_game::EFGGame* signaling_game =
156  dynamic_cast<const efg_game::EFGGame*>(efg_game.get());
157  SPIEL_CHECK_TRUE(signaling_game != nullptr);
158 
159  // "However, there is an EFCE with better payoff to both players compared to
160  // the outcome with payoff pair (0, 6): A signal X_G or Y_G is chosen with
161  // equal probability for type G, and player 2 is told to accept when receiving
162  // the chosen signal and to refuse when receiving the other signal (so X_G and
163  // lX rY are perfectly correlated, as well as Y_G and r_X l_Y ).
164  TabularPolicy XG_XB_policy = efg_game::EFGGameTabularPolicy(efg_game,
165  {{{0, "G"}, {{"X_G", 1.0}, {"Y_G", 0.0}}},
166  {{0, "B"}, {{"X_B", 1.0}, {"Y_B", 0.0}}},
167  {{1, "X"}, {{"l_X", 1.0}, {"r_X", 0.0}}},
168  {{1, "Y"}, {{"l_Y", 0.0}, {"r_Y", 1.0}}}});
169 
170  TabularPolicy YG_XB_policy = efg_game::EFGGameTabularPolicy(efg_game,
171  {{{0, "G"}, {{"X_G", 0.0}, {"Y_G", 1.0}}},
172  {{0, "B"}, {{"X_B", 1.0}, {"Y_B", 0.0}}},
173  {{1, "X"}, {{"l_X", 0.0}, {"r_X", 1.0}}},
174  {{1, "Y"}, {{"l_Y", 1.0}, {"r_Y", 0.0}}}});
175 
176  TabularPolicy XG_YB_policy = efg_game::EFGGameTabularPolicy(efg_game,
177  {{{0, "G"}, {{"X_G", 1.0}, {"Y_G", 0.0}}},
178  {{0, "B"}, {{"X_B", 0.0}, {"Y_B", 1.0}}},
179  {{1, "X"}, {{"l_X", 1.0}, {"r_X", 0.0}}},
180  {{1, "Y"}, {{"l_Y", 0.0}, {"r_Y", 1.0}}}});
181 
182  TabularPolicy YG_YB_policy = efg_game::EFGGameTabularPolicy(efg_game,
183  {{{0, "G"}, {{"X_G", 0.0}, {"Y_G", 1.0}}},
184  {{0, "B"}, {{"X_B", 0.0}, {"Y_B", 1.0}}},
185  {{1, "X"}, {{"l_X", 0.0}, {"r_X", 1.0}}},
186  {{1, "Y"}, {{"l_Y", 1.0}, {"r_Y", 0.0}}}});
187 
188  // Finally test to see if it's an EFCE.
189  CorrelationDevice mu = {
190  {0.25, XG_XB_policy},
191  {0.25, YG_XB_policy},
192  {0.25, XG_YB_policy},
193  {0.25, YG_YB_policy},
194  };
195  expected_values = ExpectedValues(*efg_game, mu);
196  SPIEL_CHECK_TRUE(Near(expected_values[0], 3.5));
197  SPIEL_CHECK_TRUE(Near(expected_values[1], 6.5));
198 
199  CorrDistConfig config;
200  SPIEL_CHECK_TRUE(Near(EFCEDist(*efg_game, config, mu), 0.0));
201 
202  // EFCEs are contained withing EFCCE (see Section 5 of
203  // https://arxiv.org/abs/1908.09893), so mu is also an EFCCE in this game.
204  SPIEL_CHECK_TRUE(Near(EFCCEDist(*efg_game, config, mu), 0.0));
205 }

207 void Test1PInOutGame() {
208  // Example game described in Section 2.4 of von Stengel & Forges,
209  // Extensive Form Correlated Equilibrium: Definition and Computational
210  // Complexity. CDAM Research Report LSE-CDAM-2006-04.
211  // http://www.cdam.lse.ac.uk/Reports/Files/cdam-2006-04.pdf
212  //
213  // This is a simple example that illustrates the difference between AFCE and
214  // EFCE.
215  const char* kInOutGameData = R"###(
216  EFG 2 R "InOutGame" { "P1" } ""
217 
218  p "ROOT" 1 1 "Root Infoset" { "In" "Out" } 0
219  p "In" 1 2 "In Infoset" { "In" "Out" } 0
220  t "In In" 1 "Outcome In In" { 1.0 }
221  t "In Out" 2 "Outcome In Out" { 0.0 }
222  p "Out" 1 3 "Out Infoset" { "In" "Out" } 0
223  t "Out In" 3 "Outcome Out In" { 0.0 }
224  t "Out Out" 4 "Outcome Out Out" { 0.0 }
225  )###";
226  std::shared_ptr<const Game> efg_game = efg_game::LoadEFGGame(kInOutGameData);
227 
228  TabularPolicy single_policy = efg_game::EFGGameTabularPolicy(
229  efg_game, {{{0, "Root Infoset"}, {{"In", 0.0}, {"Out", 1.0}}},
230  {{0, "In Infoset"}, {{"In", 0.0}, {"Out", 1.0}}},
231  {{0, "Out Infoset"}, {{"In", 0.0}, {"Out", 1.0}}}});
232 
233  CorrelationDevice mu = {{1.0, single_policy}};
234 
235  std::vector<double> expected_values = ExpectedValues(*efg_game, mu);
236  SPIEL_CHECK_TRUE(Near(expected_values[0], 0.0));
237 
238  CorrDistConfig config;
239  SPIEL_CHECK_TRUE(Near(AFCEDist(*efg_game, config, mu), 0.0));
240 
241  // Player has incentive to switch to In at the first decision and, once having
242  // deviated switch to In again, achieving a value of 1. This is 1 more than
243  // the correlation device's expected value of 0.
244  SPIEL_CHECK_FLOAT_NEAR(EFCEDist(*efg_game, config, mu), 1.0, kFloatTolerance);
245 }

257  {{{0, "Root infoset"}, {{"L", 1.0}, {"R", 0.0}}},

260  {{0, "Right P1 infoset"}, {{"l2", 1.0}, {"r2", 0.0}}}});
261 
262  TabularPolicy LBl1_policy = efg_game::EFGGameTabularPolicy(efg_game,
263  {{{0, "Root infoset"}, {{"L", 1.0}, {"R", 0.0}}},
264  {{1, "P2 infoset"}, {{"A", 0.0}, {"B", 1.0}}},
265  {{0, "Left P1 infoset"}, {{"l1", 1.0}, {"r1", 0.0}}},
266  {{0, "Right P1 infoset"}, {{"l2", 0.0}, {"r2", 1.0}}}});
267 
268  CorrelationDevice mu = {{0.5, LAl1_policy}, {0.5, LBl1_policy}};

270 
271  // This *is* an AFCE and AFCCE.

274  SPIEL_CHECK_FLOAT_NEAR(AFCCEDist(*efg_game, config, mu), 0.0,
275  kFloatTolerance);
276 
277  // However, *not* an EFCE nor EFCCE.
278  SPIEL_CHECK_GT(EFCEDist(*efg_game, config, mu), 0.0);
279  SPIEL_CHECK_GT(EFCCEDist(*efg_game, config, mu), 0.0);
280  }

344 void TestCCECEDistCFRGoofSpiel() {
345  std::shared_ptr<const Game> game = LoadGame(
346  "turn_based_simultaneous_game(game=goofspiel(num_cards=3,points_order="
347  "descending,returns_type=total_points))");
348  for (int num_iterations : {1, 10, 100}) {
349  std::vector<TabularPolicy> policies;
350  policies.reserve(num_iterations);
351  CFRSolverBase solver(*game,
352  /*alternating_updates=*/true,
353  /*linear_averaging=*/false,
354  /*regret_matching_plus=*/false,
355  /*random_initial_regrets*/ false);
356  for (int i = 0; i < num_iterations; i++) {
357  solver.EvaluateAndUpdatePolicy();
358  TabularPolicy current_policy =
359  static_cast<CFRCurrentPolicy*>(solver.CurrentPolicy().get())
360  ->AsTabular();
361  policies.push_back(current_policy);
362  }
363 
364  CorrelationDevice mu = UniformCorrelationDevice(policies);
365  CorrDistInfo cce_dist_info = CCEDist(*game, mu);
366  std::cout << "num_iterations: " << num_iterations
367  << ", cce_dist: " << cce_dist_info.dist_value << std::endl;
368 
369  // Disabled in test because it's really slow.
370  // double ce_dist = CEDist(*game, DeterminizeCorrDev(mu));
371  // std::cout << "num_iterations: " << num_iterations
372  // << ", approximate ce_dist: " << ce_dist << std::endl;
373  CorrDistInfo ce_dist_info =
374  CEDist(*game, SampledDeterminizeCorrDev(mu, 100));
