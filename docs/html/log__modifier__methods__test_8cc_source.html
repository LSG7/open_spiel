156  Timestamp(Eq(absl::UnixEpoch())),
157  ThreadID(Eq(absl::LogEntry::tid_t{456})),
158  TextMessage(Eq("forwarded: hello world")), Verbosity(Eq(7)),

160  EqualsProto(R"pb(value { literal: "forwarded: " }

162
