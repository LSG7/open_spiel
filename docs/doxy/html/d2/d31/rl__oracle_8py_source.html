26def update_episodes_per_oracles(episodes_per_oracle, played_policies_indexes):
27 """Updates the current episode count per policy.
28
29 Args:
30 episodes_per_oracle: List of list of number of episodes played per policy.
31 One list per player.
32 played_policies_indexes: List with structure (player_index, policy_index) of
33 played policies whose count needs updating.
34
35 Returns:
36 Updated count.
37 """
38 for player_index, policy_index in played_policies_indexes:
39 episodes_per_oracle[player_index][policy_index] += 1
40 return episodes_per_oracle

54def random_count_weighted_choice(count_weight):
55 """Returns a randomly sampled index i with P ~ 1 / (count_weight[i] + 1).
56
57 Allows random sampling to prioritize indexes that haven't been sampled as many
58 times as others.
59
60 Args:
61 count_weight: A list of counts to sample an index from.
62
63 Returns:
64 Randomly-sampled index.
65 """
66 indexes = list(range(len(count_weight)))
67 p = np.array([1 / (weight + 1) for weight in count_weight])
68 p /= np.sum(p)
69 chosen_index = np.random.choice(indexes, p=p)
70 return chosen_index

106 def sample_episode(self, unused_time_step, agents, is_evaluation=False):
107 time_step = self._env.reset()
108 cumulative_rewards = 0.0
109 while not time_step.last():
110 if time_step.is_simultaneous_move():
111 action_list = []
112 for agent in agents:
113 output = agent.step(time_step, is_evaluation=is_evaluation)
114 action_list.append(output.action)
115 time_step = self._env.step(action_list)
116 cumulative_rewards += np.array(time_step.rewards)
117 else:
118 player_id = time_step.observations["current_player"]
119
120 # is_evaluation is a boolean that, when False, lets policies train. The
121 # setting of PSRO requires that all policies be static aside from those
122 # being trained by the oracle. is_evaluation could be used to prevent
123 # policies from training, yet we have opted for adding frozen attributes
124 # that prevents policies from training, for all values of is_evaluation.
125 # Since all policies returned by the oracle are frozen before being
126 # returned, only currently-trained policies can effectively learn.
127 agent_output = agents[player_id].step(
128 time_step, is_evaluation=is_evaluation)
129 action_list = [agent_output.action]
130 time_step = self._env.step(action_list)
131 cumulative_rewards += np.array(time_step.rewards)
132
133 if not is_evaluation:
134 for agent in agents:
135 agent.step(time_step)
136
137 return cumulative_rewards

139 def _has_terminated(self, episodes_per_oracle):
140 # The oracle has terminated when all policies have at least trained for
141 # self._number_training_episodes. Given the stochastic nature of our
142 # training, some policies may have more training episodes than that value.
143 return np.all(
144 episodes_per_oracle.reshape(-1) > self._number_training_episodes)

146 def sample_policies_for_episode(self, new_policies, training_parameters,
147 episodes_per_oracle, strategy_sampler):
148 """Randomly samples a set of policies to run during the next episode.
149
150 Note : sampling is biased to select players & strategies that haven't
151 trained as much as the others.
152
153 Args:
154 new_policies: The currently training policies, list of list, one per
155 player.
156 training_parameters: List of list of training parameters dictionaries, one
157 list per player, one dictionary per training policy.
158 episodes_per_oracle: List of list of integers, computing the number of
159 episodes trained on by each policy. Used to weight the strategy
160 sampling.
161 strategy_sampler: Sampling function that samples a joint strategy given
162 probabilities.
163
164 Returns:
165 Sampled list of policies (One policy per player), index of currently
166 training policies in the list.
167 """
168 num_players = len(training_parameters)
169
170 # Prioritizing players that haven't had as much training as the others.
171 episodes_per_player = [sum(episodes) for episodes in episodes_per_oracle]
172 chosen_player = random_count_weighted_choice(episodes_per_player)
173
174 # Uniformly choose among the sampled player.
175 agent_chosen_ind = np.random.randint(
176 0, len(training_parameters[chosen_player]))
177 agent_chosen_dict = training_parameters[chosen_player][agent_chosen_ind]
178 new_policy = new_policies[chosen_player][agent_chosen_ind]
179
180 # Sample other players' policies.
181 total_policies = agent_chosen_dict["total_policies"]
182 probabilities_of_playing_policies = agent_chosen_dict[
183 "probabilities_of_playing_policies"]
184 episode_policies = strategy_sampler(total_policies,
185 probabilities_of_playing_policies)
186
187 live_agents_player_index = [(chosen_player, agent_chosen_ind)]
188
189 for player in range(num_players):
190 if player == chosen_player:
191 episode_policies[player] = new_policy
192 assert not new_policy.is_frozen()
193 else:
194 # Sample a bernoulli with parameter 'self_play_proportion' to determine
195 # whether we do self-play with 'player'.
196 if np.random.binomial(1, self._self_play_proportion):
197 # If we are indeed doing self-play on that round, sample among the
198 # trained strategies of current_player, with priority given to less-
199 # selected agents.
200 agent_index = random_count_weighted_choice(
201 episodes_per_oracle[player])
202 self_play_agent = new_policies[player][agent_index]
203 episode_policies[player] = self_play_agent
204 live_agents_player_index.append((player, agent_index))
205 else:
206 assert episode_policies[player].is_frozen()
207
208 return episode_policies, live_agents_player_index

213 def generate_new_policies(self, training_parameters):
214 """Generates new policies to be trained into best responses.
215
216 Args:
217 training_parameters: list of list of training parameter dictionaries, one
218 list per player.
219
220 Returns:
221 List of list of the new policies, following the same structure as
222 training_parameters.
223 """
224 new_policies = []
225 for player in range(len(training_parameters)):
226 player_parameters = training_parameters[player]
227 new_pols = []
228 for param in player_parameters:
229 current_pol = param["policy"]
230 if isinstance(current_pol, self._best_response_class):
231 new_pol = current_pol.copy_with_noise(self._kwargs.get("sigma", 0.0))
232 else:
233 new_pol = self._best_response_class(self._env, player,
234 **self._best_response_kwargs)
235 new_pol.unfreeze()
236 new_pols.append(new_pol)
237 new_policies.append(new_pols)
238 return new_policies

240 def __call__(self,
241 game,
242 training_parameters,
243 strategy_sampler=utils.sample_strategy,
244 **oracle_specific_execution_kwargs):
245 """Call method for oracle, returns best responses against a set of policies.
246
247 Args:
248 game: The game on which the optimization process takes place.
249 training_parameters: A list of list of dictionaries (One list per player),
250 each dictionary containing the following fields :
251 - policy : the policy from which to start training.
252 - total_policies: A list of all policy.Policy strategies used for
253 training, including the one for the current player.
254 - current_player: Integer representing the current player.
255 - probabilities_of_playing_policies: A list of arrays representing, per
256 player, the probabilities of playing each policy in total_policies for
257 the same player.
258 strategy_sampler: Callable that samples strategies from total_policies
259 using probabilities_of_playing_policies. It only samples one joint
260 set of policies for all players. Implemented to be able to take into
261 account joint probabilities of action (For Alpharank)
262 **oracle_specific_execution_kwargs: Other set of arguments, for
263 compatibility purposes. Can for example represent whether to Rectify
264 Training or not.
265
266 Returns:
267 A list of list, one for each member of training_parameters, of (epsilon)
268 best responses.
269 """
270 episodes_per_oracle = [[0
271 for _ in range(len(player_params))]
272 for player_params in training_parameters]
273 episodes_per_oracle = np.array(episodes_per_oracle)
274
275 new_policies = self.generate_new_policies(training_parameters)
276
277 # TODO(author4): Look into multithreading.
278 while not self._has_terminated(episodes_per_oracle):
279 agents, indexes = self.sample_policies_for_episode(
280 new_policies, training_parameters, episodes_per_oracle,
281 strategy_sampler)
282 self._rollout(game, agents, **oracle_specific_execution_kwargs)
283 episodes_per_oracle = update_episodes_per_oracles(episodes_per_oracle,
284 indexes)
285 # Freeze the new policies to keep their weights static. This allows us to
286 # later not have to make the distinction between static and training
287 # policies in training iterations.
288 freeze_all(new_policies)
289 return new_policies
