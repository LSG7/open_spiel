49// Used to convert actions represented as integers in mixed bases.
50Action RankActionMixedBase(const std::vector<int>& bases,
51 const std::vector<int>& digits) {
52 SPIEL_CHECK_EQ(bases.size(), digits.size());
53 SPIEL_CHECK_GT(digits.size(), 0);
54
55 Action action = 0;
56 int one_plus_max = 1;
57 for (int i = digits.size() - 1; i >= 0; --i) {
58 SPIEL_CHECK_GE(digits[i], 0);
59 SPIEL_CHECK_LT(digits[i], bases[i]);
60 SPIEL_CHECK_GT(bases[i], 1);
61 action += digits[i] * one_plus_max;
62 one_plus_max *= bases[i];
63 SPIEL_CHECK_LT(action, one_plus_max);
64 }
65
66 return action;
67}

69std::vector<int> UnrankActionMixedBase(Action action,
70 const std::vector<int>& bases) {
71 std::vector<int> digits(bases.size());
72 for (int i = digits.size() - 1; i >= 0; --i) {
73 SPIEL_CHECK_GT(bases[i], 1);
74 digits[i] = action % bases[i];
75 action /= bases[i];
76 }
77 SPIEL_CHECK_EQ(action, 0);
78 return digits;
79}

81absl::optional<std::string> FindFile(const std::string& filename, int levels) {

100std::string FormatDouble(double value) {
101 // We cannot use StrCat as that would default to exponential notation
102 // sometimes. For example, the default format of 10^-9 is the string
103 // "1e-9". For that reason, we use StrFormat with %f explicitly, and add
104 // the .0 if necessary (to clarify that it's a double value).
105 std::string double_str = absl::StrFormat("%.15f", value);
106 size_t idx = double_str.find('.');
107
108 if (double_str.find('.') == std::string::npos) { // NOLINT
109 absl::StrAppend(&double_str, ".0");
110 } else {
111 // Remove the extra trailing zeros, if there are any.
112 while (double_str.length() > idx + 2 && double_str.back() == '0') {
113 double_str.pop_back();
114 }
115 }
116 return double_str;
117}

143void Normalize(absl::Span<double> weights) {
144 SPIEL_CHECK_FALSE(weights.empty());
145 const double normalizer = absl::c_accumulate(weights, 0.);
146 SPIEL_CHECK_FALSE(std::isnan(normalizer));
147 const double uniform_prob = 1.0 / weights.size();
148 absl::c_for_each(weights, [&](double& w) {
149 w = (normalizer == 0.0 ? uniform_prob : w / normalizer);
150 });
151}

183int SamplerFromRng::operator()(absl::Span<const double> probs) {
184 const float cutoff = rng_();
185 float sum = 0.0f;
186 for (int i = 0; i < probs.size(); ++i) {
187 sum += probs[i];
188 if (cutoff < sum) {
189 return i;
190 }
191 }
192
193 // To be on the safe side, cover case cutoff == 1.0 and sum < 1
194 for (int i = probs.size() - 1; i >= 0; --i) {
195 if (probs[i] > 0.0) return i;
196 }
197
198 SpielFatalError("SamplerFromRng: not a probability distribution.");
199}
