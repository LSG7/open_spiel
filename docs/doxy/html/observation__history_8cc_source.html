32ActionObservationHistory::ActionObservationHistory(Player player,
33 const State& target)
34 : player_(player) {
35 SPIEL_CHECK_GE(player_, 0);
36 SPIEL_CHECK_LT(player_, target.NumPlayers());
37 SPIEL_CHECK_TRUE(target.GetGame()->GetType().provides_observation_string);
38
39 const std::vector<State::PlayerAction>& history = target.FullHistory();
40 history_.reserve(history.size());
41
42 std::unique_ptr<State> state = target.GetGame()->NewInitialState();
43 history_.push_back({absl::nullopt, state->ObservationString(player)});
44 for (int i = 0; i < history.size(); i++) {
45 const auto& [history_player, action] = history[i];
46 const bool is_acting = state->CurrentPlayer() == player;
47 state->ApplyAction(action);
48 history_.push_back({
49 is_acting ? action : static_cast<absl::optional<Action>>(absl::nullopt),
50 state->ObservationString(player)
51 });
52 }
53}

98bool ActionObservationHistory::IsPrefixOf(
99 const ActionObservationHistory& other) const {
100 if (player_ != other.player_) return false;
101
102 if (CorrespondsToInitialState()) return true;
103 if (other.CorrespondsToInitialState()) return false;
104
105 const auto& a = history_;
106 const auto& b = other.history_;
107 if (a.size() > b.size()) return false;
108 if (a.size() == b.size()) return a == b;
109 return std::equal(a.begin(), a.end(), b.begin());
110}

112bool ActionObservationHistory::IsPrefixOf(Player pl, const State& state) const {
113 const std::shared_ptr<const Game> game = state.GetGame();
114 SPIEL_CHECK_TRUE(game->GetType().provides_observation_string);

116 if (CorrespondsToInitialState()) return true;

128bool ActionObservationHistory::IsExtensionOf(Player pl,
129 const State& state) const {
130 const std::shared_ptr<const Game> game = state.GetGame();
131 SPIEL_CHECK_TRUE(game->GetType().provides_observation_string);
132
133 if (state.IsInitialState()) return true;
134 // Cannot be extension if state is later.
135 if (state.MoveNumber() > MoveNumber()) return false;
136
137 // Check the latest observation is identical -- most observations
138 // will differ only in the last items.
139 if (state.ObservationString(pl) != ObservationAt(state.MoveNumber()))
140 return false;
141
142 return CheckStateCorrespondenceInSimulation(pl, state, state.MoveNumber());
143}

155bool ActionObservationHistory::CheckStateCorrespondenceInSimulation(
156 Player pl, const State& state, int until_time) const {
157 const std::vector<State::PlayerAction>& state_history = state.FullHistory();
158 std::unique_ptr<State> simulation = state.GetGame()->NewInitialState();
159
160 int i = 0; // The index for state_history access.
161 int j = 1; // The index for history_ access.
162 while (simulation->MoveNumber() < until_time) {
163 SPIEL_CHECK_LT(i, state_history.size());
164 SPIEL_CHECK_LT(j, history_.size());
165 SPIEL_CHECK_FALSE(simulation->IsTerminal());
166
167 if (simulation->CurrentPlayer() == pl) {
168 if (history_[j].first != state_history[i].action) return false;
169 } else {
170 if (history_[j].first != absl::nullopt) return false;
171 }
172
173 simulation->ApplyAction(state_history[i].action);
174 i++;
175
176 if (history_[j].second != simulation->ObservationString(pl)) return false;
177 j++;
178 }
179 return true;
180}

184 for (int i = 0; i < history_.size(); i++) {

191 ", observation=\"", action_observation.second, "\")");

200PublicObservationHistory::PublicObservationHistory(const State& target)
201 : observer_(target.GetGame()->MakeObserver(
202 IIGObservationType{/*public_info*/true,
203 /*perfect_recall*/false,
204 /*private_info*/PrivateInfoType::kNone},
205 {})) {
206 history_.reserve(target.FullHistory().size());
207
208 std::unique_ptr<State> state = target.GetGame()->NewInitialState();
209 // Use FullHistory even though we don't need the player -- prevent
210 // doing a copy.
211 for (const auto& [_, action] : target.FullHistory()) {
212 history_.push_back(observer_->StringFrom(*state, kDefaultPlayerId));
213 state->ApplyAction(action);
214 }
215 history_.push_back(observer_->StringFrom(*state, kDefaultPlayerId));
216}

292bool PublicObservationHistory::CheckStateCorrespondenceInSimulation(
293 const State& state, int until_time) const {
294 const std::vector<State::PlayerAction>& state_history = state.FullHistory();
295 std::unique_ptr<State> simulation = state.GetGame()->NewInitialState();
296
297 int i = 0; // The index for state_history access.
298 int j = 1; // The index for history_ access.
299 while (simulation->MoveNumber() < until_time) {
300 SPIEL_CHECK_LT(i, state_history.size());
301 SPIEL_CHECK_LT(j, history_.size());
302 SPIEL_CHECK_FALSE(simulation->IsTerminal());
303
304 simulation->ApplyAction(state_history[i].action);
305 i++;
306
307 if (history_.at(j) != observer_->StringFrom(*simulation, kDefaultPlayerId))
308 return false;
309 j++;
310 }
311 return true;
312}
