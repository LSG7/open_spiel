47 template <typename K, typename V>
48 class LRUCache { // Least Recently Used Cache.
49  // TODO(author7): Consider the performance implications here. Some ideas:
50  // - Shard the cache to avoid lock contention. Can be done by the user.
51  // - Use shared pointers to avoid copying data out, and shorten the lock.
52  // - Use two generations to avoid order updates on hot items. The mature
53  // generation wouldn't be ordered or evicted so can use a reader/writer lock
54  // - Use atomics for hits/misses to shorten lock times.
55  // - Embed the list directly into the map value to avoid extra indirection.
56  public:
57  explicit LRUCache(int max_size) : hits_(0), misses_(0) {
58  SetMaxSize(max_size);
59  }
60 
61  // Move only, not copyable.
62  LRUCache(LRUCache&& other) = default;
63  LRUCache& operator=(LRUCache&& other) = default;
64  LRUCache(const LRUCache&) = delete;
65  LRUCache& operator=(const LRUCache&) = delete;
66 
67  void SetMaxSize(int max_size) { max_size_ = std::max(max_size, 4); }
68 
69  int Size() {
70  absl::MutexLock lock(&m_);
71  return map_.size();
72  }
73 
74  void Clear() {
75  absl::MutexLock lock(&m_);
76  order_.clear();
77  map_.clear();
78  hits_ = 0;
79  misses_ = 0;
80  }
81 
82  void Set(const K& key, const V& value) {
83  absl::MutexLock lock(&m_);
84  auto pos = map_.find(key);
85  if (pos == map_.end()) { // Not found, add it.
86  if (map_.size() >= max_size_) { // Make space if needed.
87  map_.erase(order_.back());
88  order_.pop_back();
89  }
90  order_.push_front(key);
91  map_[key] = Entry{value, order_.begin()};
92  } else { // Found, move it to the front.
93  order_.erase(pos->second.order_iterator);
94  order_.push_front(key);
95  pos->second.order_iterator = order_.begin();
96  }
97  }
98 
99  absl::optional<const V> Get(const K& key) {
100  absl::MutexLock lock(&m_);
101  auto pos = map_.find(key);
102  if (pos == map_.end()) { // Not found.
103  misses_ += 1;
104  return absl::nullopt;
105  } else { // Found, move it to the front, and return the value.
106  hits_ += 1;
107  order_.erase(pos->second.order_iterator);
108  order_.push_front(key);
109  pos->second.order_iterator = order_.begin();
110  return pos->second.value;
111  }
112  }
113 
114  LRUCacheInfo Info() {
115  absl::MutexLock lock(&m_);
116  return LRUCacheInfo{hits_, misses_, static_cast<int>(map_.size()),
117  max_size_};
118  }
119 
120  private:
121  struct Entry {
122  V value;
123  typename std::list<K>::iterator order_iterator;
124  };
125 
126  int64_t hits_;
127  int64_t misses_;
128  int max_size_;
129  std::list<K> order_;
130  absl::flat_hash_map<K, Entry> map_;
131  absl::Mutex m_;
132 };
