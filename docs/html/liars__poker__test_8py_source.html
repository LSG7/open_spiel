53  def test_draw_hands(self):

56  state = game.new_initial_state()
57  expected_hands = [[] for _ in range(game.num_players())]
58  for i in range(game.num_players() * game.hand_length):
59  # Verify we have chance nodes until all player hands are filled.
60  self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
61 
62  # Draw a digit.
63  outcomes_with_probs = state.chance_outcomes()
64  action_list, prob_list = zip(*outcomes_with_probs)
65  action = np.random.choice(action_list, p=prob_list)
66 
67  # Verify players' hands are filled correctly.
68  cur_player = i % game.num_players()
69  expected_hands[cur_player].append(action)
70  state.apply_action(action)
71  self.assertEqual(state.hands, expected_hands)
72  # Assert after all hands are filled, we have non-chance nodes.
73  cur_player = state.current_player()
74  self.assertNotEqual(cur_player, pyspiel.PlayerId.CHANCE)

161  def test_single_deterministic_round(self):
162  """Runs a single round where cards are dealt deterministically."""
163  game = liars_poker.LiarsPoker({"hand_length": 3, "num_digits": 3})
164  state = game.new_initial_state()
165 
166  # Deal player 0 all "1" cards and player 1 all "2" cards.
167  for i in range(game.num_players() * game.hand_length):
168  if i % 2 == 0:
169  # Deal card to player 0
170  state.apply_action(1)
171  else:
172  # Deal card to player 1
173  state._apply_action(2)
174 
175  # Have player 0 bid that there are four 1's.
176  state.apply_action(state.encode_bid(4, 1) + liars_poker.BID_ACTION_OFFSET)
177  # Player 1 challenges.
178  state.apply_action(liars_poker.CHALLENGE_ACTION)
179  # Player 0 accepts the challenge.
180  state.apply_action(liars_poker.CHALLENGE_ACTION)
181  # Verify game ends with player 0 losing.
182  self.assertTrue(state.is_terminal())
183  self.assertEqual(state.loser(), 0)
184  expected_returns = [1.0 for _ in range(game.num_players())]
185  expected_returns[state.loser()] = -1.0 * (game.num_players() - 1)
186  self.assertEqual(state.returns(), expected_returns)
