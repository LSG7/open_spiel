45void Memory::Resize(
46 const unsigned n,
47 const TTmemory flag,
48 const int memDefault_MB,
49 const int memMaximum_MB)
50{
51 if (nThreads == n)
52 return;
53
54 if (nThreads > n)
55 {
56 // Downsize.
57 for (unsigned i = n; i < nThreads; i++)
58 {
59 memory[i]->transTable->ReturnAllMemory();
60 delete memory[i]->transTable;
61 delete memory[i];
62 }
63 memory.resize(static_cast<unsigned>(n));
64 threadSizes.resize(static_cast<unsigned>(n));
65 }
66 else
67 {
68 // Upsize.
69 memory.resize(n);
70 threadSizes.resize(n);
71 for (unsigned i = nThreads; i < n; i++)
72 {
73 memory[i] = new ThreadData();
74 if (flag == DDS_TT_SMALL)
75 {
76 memory[i]->transTable = new TransTableS;
77 threadSizes[i] = "S";
78 }
79 else
80 {
81 memory[i]->transTable = new TransTableL;
82 threadSizes[i] = "L";
83 }
84
85 memory[i]->transTable->SetMemoryDefault(memDefault_MB);
86 memory[i]->transTable->SetMemoryMaximum(memMaximum_MB);
87
88 memory[i]->transTable->MakeTT();
89 }
90 }
91
92 nThreads = n;
93}
