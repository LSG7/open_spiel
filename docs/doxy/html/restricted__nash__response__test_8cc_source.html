29void SimulateGame(std::mt19937 *rng, const Game &game,
30 std::unique_ptr<State> normal_state,
31 std::unique_ptr<State> rnr_state, bool fixed,
32 Player fixed_player) {
33 // Now check that the states are identical via the ToString().
34 std::string state_prefix =
35 fixed ? "Rnr state string of state in fixed part with underlying state:\n"
36 : "Rnr state string of state in free part with underlying state:\n";
37 std::string infostate_prefix = fixed ? "[Rnr: fixed]" : "[Rnr: free]";
38 while (!normal_state->IsTerminal()) {
39 SPIEL_CHECK_EQ(state_prefix + normal_state->ToString(),
40 rnr_state->ToString());
41 if (game.GetType().provides_information_state_string) {
42 // Check the information states to each player are consistent.
43 for (auto p = Player{0}; p < game.NumPlayers(); p++) {
44 SPIEL_CHECK_EQ((p == fixed_player ? infostate_prefix : "") +
45 normal_state->InformationStateString(p),
46 rnr_state->InformationStateString(p));
47 }
48 }
49
50 if (normal_state->IsChanceNode()) {
51 SPIEL_CHECK_TRUE(rnr_state->IsChanceNode());
52
53 // Chance node; sample one according to underlying distribution
54 std::vector<std::pair<Action, double>> outcomes =
55 normal_state->ChanceOutcomes();
56 Action action =
57 open_spiel::SampleAction(
58 outcomes, std::uniform_real_distribution<double>(0.0, 1.0)(*rng))
59 .first;
60
61 normal_state->ApplyAction(action);
62 rnr_state->ApplyAction(action);
63 } else if (normal_state->CurrentPlayer() == kSimultaneousPlayerId) {
64 SPIEL_CHECK_EQ(rnr_state->CurrentPlayer(), kSimultaneousPlayerId);
65
66 // Players choose simultaneously.
67 std::vector<Action> joint_action;
68
69 // Sample an action for each player
70 for (auto p = Player{0}; p < game.NumPlayers(); p++) {
71 std::vector<Action> actions;
72 actions = normal_state->LegalActions(p);
73 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
74 Action action = actions[dis(*rng)];
75 joint_action.push_back(action);
76 }
77
78 normal_state->ApplyActions(joint_action);
79 rnr_state->ApplyActions(joint_action);
80 } else {
81 // Chance or player node
82 SPIEL_CHECK_EQ(normal_state->CurrentPlayer(), rnr_state->CurrentPlayer());
83
84 Player p = normal_state->CurrentPlayer();
85
86 std::vector<Action> actions;
87 actions = normal_state->LegalActions(p);
88 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
89 Action action = actions[dis(*rng)];
90
91 normal_state->ApplyAction(action);
92 rnr_state->ApplyAction(action);
93 }
94 }
95
96 SPIEL_CHECK_TRUE(rnr_state->IsTerminal());
97
98 auto sim_returns = normal_state->Returns();
99 auto turn_returns = rnr_state->Returns();
100
101 for (auto player = Player{0}; player < sim_returns.size(); player++) {
102 double utility = sim_returns[player];
103 SPIEL_CHECK_GE(utility, game.MinUtility());
104 SPIEL_CHECK_LE(utility, game.MaxUtility());
105
106 double other_utility = turn_returns[player];
107 SPIEL_CHECK_EQ(utility, other_utility);
108 }
109}

124void TestBasicCreation() {
125 std::mt19937 rng;
126
127 // Create different games for RNR and check the simulation
128 for (const std::string& name :
129 {"blotto", "goofspiel", "kuhn_poker", "tiny_hanabi", "phantom_ttt",
130 "matrix_rps", "leduc_poker"}) {
131 std::cout << "RestrictedNashResponse: Testing " << name << std::endl;
132 for (Player fixed_player = 0; fixed_player < 2; fixed_player++) {
133 for (int i = 0; i < 100; ++i) {
134 std::shared_ptr<const Game> normal_game_game = LoadGame(name);
135 std::shared_ptr<const Game> rnr_game =
136 ConvertToRNR(*LoadGame(name), fixed_player, 0.5);
137 auto normal_init_fixed = normal_game_game->NewInitialState();
138 auto rnr_init_fixed = rnr_game->NewInitialState();
139 rnr_init_fixed->ApplyAction(Action(kFixedAction));

141 std::move(rnr_init_fixed), true, fixed_player);

147 std::move(rnr_init_free), false, fixed_player);

153void TestMatchingPenniesCreation() {
154 // Check the creation of matching pennies game
155 Player fixed_player = 1;
156 std::shared_ptr<const Game> game = LoadGame("matrix_mp");
157 std::shared_ptr<const Game> rnr_game = ConvertToRNR(*game, fixed_player, 0.4);
158 SPIEL_CHECK_EQ(game->MaxGameLength() + 1, rnr_game->MaxGameLength());
159 SPIEL_CHECK_EQ(rnr_game->NumPlayers(), game->NumPlayers());
160 SPIEL_CHECK_EQ(rnr_game->MaxUtility(), game->MaxUtility());
161 SPIEL_CHECK_EQ(rnr_game->MinUtility(), game->MinUtility());
162 auto state = rnr_game->NewInitialState();
163 SPIEL_CHECK_EQ("Initial restricted Nash response state.", state->ToString());
164 SPIEL_CHECK_EQ(state->LegalActions().size(), 2);
165
166 auto chance_outcomes = state->ChanceOutcomes();
167 SPIEL_CHECK_EQ(chance_outcomes[0].second, 0.4);
168 SPIEL_CHECK_EQ(chance_outcomes[1].second, 0.6);
169
170 // Fixed part
171 auto fixed_child = state->Child(kFixedAction);
172 SPIEL_CHECK_EQ(fixed_child->CurrentPlayer(), kSimultaneousPlayerId);
173
174 // Free part
175 auto free_child = state->Child(kFreeAction);
176 SPIEL_CHECK_EQ(free_child->CurrentPlayer(), kSimultaneousPlayerId);
177
178 for (Action joint_action : free_child->LegalActions()) {
179 auto new_fixed_child = fixed_child->Child(joint_action);
180 auto new_free_child = free_child->Child(joint_action);
181 SPIEL_CHECK_EQ(new_fixed_child->Rewards(), new_free_child->Rewards());
182 SPIEL_CHECK_EQ(new_fixed_child->InformationStateString(1 - fixed_player),
183 new_free_child->InformationStateString(1 - fixed_player));
184 SPIEL_CHECK_NE(new_fixed_child->InformationStateString(fixed_player),
185 new_free_child->InformationStateString(fixed_player));
186 }
187}

189void TestFixedPolicyGame() {
190 // Check the RNR which automatically puts the strategy in the game as chance
191 // nodes Setup
192 Player fixed_player = 1;
193 std::shared_ptr<const Game> game = LoadGameAsTurnBased("matrix_mp");
194 std::shared_ptr<TabularPolicy> fixed_policy =
195 std::make_shared<TabularPolicy>(*game);
196 auto initial_state = game->NewInitialState();
197 initial_state->ApplyAction(0);
198 fixed_policy->SetStatePolicy(initial_state->InformationStateString(),
199 {{0, 1}, {1, 0}});
200 // P 0.6 case when the resulting strategy is pure
201 std::shared_ptr<const Game> rnr_game =
202 ConvertToRNR(*game, fixed_player, 0.6, fixed_policy);
203 algorithms::CFRPlusSolver solver(*rnr_game);
204 for (int i = 0; i < 1000; i++) {
205 solver.EvaluateAndUpdatePolicy();
206 }
207 const std::shared_ptr<Policy> average_policy = solver.AveragePolicy();
208 auto player_two_policy = average_policy->GetStatePolicy(
209 "[Rnr: free]Current player: 1\nObserving player: 1. Non-terminal");
210 for (int i = 0; i < player_two_policy.size(); i++) {
211 SPIEL_CHECK_FLOAT_NEAR(player_two_policy[i].second, i, 0.001);
212 }
213 auto player_one_policy = average_policy->GetStatePolicy(
214 "Current player: 0\nObserving player: 0. Non-terminal");
215 for (int i = 0; i < player_one_policy.size(); i++) {
216 SPIEL_CHECK_FLOAT_NEAR(player_one_policy[i].second, 1 - i, 0.001);
217 }
218 // P 0.6 case when the resulting strategy is pure
219 rnr_game = ConvertToRNR(*game, fixed_player, 0.4, fixed_policy);
220 algorithms::CFRPlusSolver solver_two(*rnr_game);
221 for (int i = 0; i < 1000; i++) {
222 solver_two.EvaluateAndUpdatePolicy();
223 }
224 const std::shared_ptr<Policy> average_policy_two = solver_two.AveragePolicy();
225 auto player_two_policy_two = average_policy_two->GetStatePolicy(
226 "[Rnr: free]Current player: 1\nObserving player: 1. Non-terminal");
227 double check_policy[] = {1. / 6, 5. / 6};
228 for (int i = 0; i < player_two_policy_two.size(); i++) {
229 SPIEL_CHECK_FLOAT_NEAR(player_two_policy_two[i].second, check_policy[i],
230 0.001);
231 }
232 auto player_one_policy_two = average_policy_two->GetStatePolicy(
233 "Current player: 0\nObserving player: 0. Non-terminal");
234 check_policy[0] = check_policy[1] = 0.5;
235 for (int i = 0; i < player_one_policy_two.size(); i++) {
236 SPIEL_CHECK_FLOAT_NEAR(player_one_policy_two[i].second, check_policy[i],
237 0.001);
238 }
239}
