52 std::string GamutGenerator::TmpFile() {
53  for (int retries = 0; retries < kNumTmpfileRetries; ++retries) {
54  // Try random files until we find one that does not exist.
55  absl::c_shuffle(rand_string_, rng_);
56  std::string candidate =
57  absl::StrCat(file::GetTmpDir(), "/gamut_tmpgame_",
58  rand_string_.substr(0, kNumRandomChars));
59  if (!file::Exists(candidate)) {
60  return candidate;
61  }
62  }
63 
64  SpielFatalError(absl::StrCat("Could not get a temporary file after ",
65  kNumTmpfileRetries, " tries."));
66 }

68 std::shared_ptr<const Game> GamutGenerator::GenerateGame(
69  const std::vector<std::string>& cmdline_args) {
70  // Check that there's no -f and no -output in the command-line args. The get
71  // added by this generator.
72  for (const auto& arg : cmdline_args) {
73  if (arg == "-f") {
74  SpielFatalError("Do not use -f in the command-line arguments.");
75  } else if (arg == "-output") {
76  SpielFatalError("Do not use -output in the command-line arguments.");
77  }
78  }
79 
80  std::vector<std::string> arguments = cmdline_args;
81  arguments.push_back("-output");
82  arguments.push_back("GambitOutput");
83 
84  // Lock here to prevent concurrently writing / removal.
85  std::shared_ptr<const Game> game;
86  {
87  absl::MutexLock lock(&generation_mutex_);
88  // Get a temporary file and add it to the arguments.
89  std::string tmp_filename = TmpFile();
90  arguments.push_back("-f");
91  arguments.push_back(tmp_filename);
92  std::string full_cmd = absl::StrCat(java_path_, " -jar ", jar_path_, " ",
93  absl::StrJoin(arguments, " "));
94  int ret_code = system(full_cmd.c_str());
95  SPIEL_CHECK_EQ(ret_code, 0);
96  SPIEL_CHECK_TRUE(file::Exists(tmp_filename));
97  game = LoadGame("nfg_game", {{"filename", GameParameter(tmp_filename)}});
98  file::Remove(tmp_filename);
99  }
100  return game;
101 }
