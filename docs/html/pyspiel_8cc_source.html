89 // This exception class is used to forward errors from Spiel to Python.
90 // Do not create exceptions of this type directly! Instead, call
91 // SpielFatalError, which will raise a Python exception when called from
92 // Python, and exit the process otherwise.
93 class SpielException : public std::exception {
94  public:
95  explicit SpielException(std::string message) : message_(message) {}
96  const char* what() const noexcept override { return message_.c_str(); }
97 
98  private:
99  std::string message_;
100 };
