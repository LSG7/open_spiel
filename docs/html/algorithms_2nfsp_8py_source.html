319 class ReservoirBuffer(object):
320  """Allows uniform sampling over a stream of data.
321 
322  This class supports the storage of arbitrary elements, such as observation
323  tensors, integer actions, etc.
324 
325  See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
326  """
327 
328  def __init__(self, reservoir_buffer_capacity):
329  self._reservoir_buffer_capacity_reservoir_buffer_capacity = reservoir_buffer_capacity
330  self._data_data = []
331  self._add_calls_add_calls = 0
332 
333  def add(self, element):
334  """Potentially adds `element` to the reservoir buffer.
335 
336  Args:
337  element: data to be added to the reservoir buffer.
338  """
339  if len(self._data_data) < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
340  self._data_data.append(element)
341  else:
342  idx = np.random.randint(0, self._add_calls_add_calls + 1)
343  if idx < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
344  self._data_data[idx] = element
345  self._add_calls_add_calls += 1
346 
347  def sample(self, num_samples):
348  """Returns `num_samples` uniformly sampled from the buffer.
349 
350  Args:
351  num_samples: `int`, number of samples to draw.
352 
353  Returns:
354  An iterable over `num_samples` random elements of the buffer.
355 
356  Raises:
357  ValueError: If there are less than `num_samples` elements in the buffer
358  """
359  if len(self._data_data) < num_samples:
360  raise ValueError("{} elements could not be sampled from size {}".format(
361  num_samples, len(self._data_data)))
362  return random.sample(self._data_data, num_samples)
363 
364  def clear(self):
365  self._data_data = []
366  self._add_calls_add_calls = 0
367 
368  def __len__(self):
369  return len(self._data_data)

371  def __iter__(self):
372  return iter(self._data_data)
