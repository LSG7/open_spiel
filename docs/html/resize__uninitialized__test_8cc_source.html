24 // A mock string class whose only purpose is to track how many times its
25 // resize()/append() methods have been called.
26 struct resizable_string {
27  using value_type = char;
28  size_t size() const { return 0; }
29  size_t capacity() const { return 0; }
30  char& operator[](size_t) {
31  static char c = '\0';
32  return c;
33  }
34  void resize(size_t) { resize_call_count += 1; }
35  void append(size_t, value_type) { append_call_count += 1; }
36  void reserve(size_t) {}

38 };

120 TEST(ResizeUninit, Amortized) {
