28void TestDataLogger() {
29 std::string val = std::to_string(std::rand()); // NOLINT
30 std::string tmp_dir = file::GetTmpDir();
31 std::string dir = tmp_dir + "/open_spiel-test-" + val;
32 std::string filename = dir + "/data-test.jsonl";
33
34 SPIEL_CHECK_TRUE(file::Exists(tmp_dir));
35 SPIEL_CHECK_TRUE(file::IsDirectory(tmp_dir));
36 SPIEL_CHECK_FALSE(file::Exists(dir));
37 SPIEL_CHECK_TRUE(file::Mkdir(dir));
38 SPIEL_CHECK_TRUE(file::Exists(dir));
39 SPIEL_CHECK_TRUE(file::IsDirectory(dir));
40
41 {
42 DataLoggerJsonLines logger(dir, "data-test");
43 logger.Write({{"step", 1}, {"avg", 1.5}});
44 logger.Write({{"step", 2}, {"avg", 2.5}});
45 }
46
47 {
48 file::File f(filename, "r");
49 std::vector<std::string> lines = absl::StrSplit(f.ReadContents(), '\n');
50 SPIEL_CHECK_EQ(lines.size(), 3);
51 SPIEL_CHECK_EQ(lines[2], "");
52
53 json::Object obj1 = json::FromString(lines[0])->GetObject();
54 SPIEL_CHECK_EQ(obj1["step"], 1);
55 SPIEL_CHECK_EQ(obj1["avg"], 1.5);
56 SPIEL_CHECK_TRUE(obj1["time_str"].IsString());
57 SPIEL_CHECK_TRUE(obj1["time_abs"].IsDouble());
58 SPIEL_CHECK_GT(obj1["time_abs"].GetDouble(), 1'500'000'000); // July 2017
59 SPIEL_CHECK_TRUE(obj1["time_rel"].IsDouble());
60 SPIEL_CHECK_GT(obj1["time_rel"].GetDouble(), 0);
61
62 json::Object obj2 = json::FromString(lines[1])->GetObject();
63 SPIEL_CHECK_EQ(obj2["step"], 2);
64 SPIEL_CHECK_EQ(obj2["avg"], 2.5);
65
66 SPIEL_CHECK_LT(obj1["time_abs"].GetDouble(), obj2["time_abs"].GetDouble());
67 SPIEL_CHECK_LT(obj1["time_rel"].GetDouble(), obj2["time_rel"].GetDouble());
68 }
69
70 SPIEL_CHECK_TRUE(file::Remove(filename));
71 SPIEL_CHECK_TRUE(file::Remove(dir));
72}
