175  def step(self, time_step, is_evaluation=False, add_transition_record=True):
176  """Returns the action to be taken and updates the Q-network if needed.
177 
178  Args:
179  time_step: an instance of rl_environment.TimeStep.
180  is_evaluation: bool, whether this is a training or evaluation call.
181  add_transition_record: Whether to add to the replay buffer on this step.
182

184  A `rl_agent.StepOutput` containing the action probs and chosen action.
185  """

188  if (not time_step.last()) and (
189  time_step.is_simultaneous_move() or
190  self.player_idplayer_id == time_step.current_player()):
191  info_state = time_step.observations["info_state"][self.player_idplayer_id]
192  legal_actions = time_step.observations["legal_actions"][self.player_idplayer_id]
193  epsilon = self._get_epsilon_get_epsilon(is_evaluation)
194  action, probs = self._epsilon_greedy_epsilon_greedy(info_state, legal_actions, epsilon)
195  else:
196  action = None
197  probs = []
198 
199  # Don't mess up with the state during evaluation.
200  if not is_evaluation:
201  self._step_counter_step_counter += 1
202 
203  if self._step_counter_step_counter % self._learn_every_learn_every == 0:
204  self._last_loss_value_last_loss_value = self.learnlearn()
205

207  self._session_session.run(self._update_target_network_update_target_network)
208 
209  if self._prev_timestep_prev_timestep and add_transition_record:
210  # We may omit record adding here if it's done elsewhere.
211  self.add_transitionadd_transition(self._prev_timestep_prev_timestep, self._prev_action_prev_action, time_step)
212 
213  if time_step.last(): # prepare for the next episode.
214  self._prev_timestep_prev_timestep = None
215  self._prev_action_prev_action = None

225

269  def _epsilon_greedy(self, info_state, legal_actions, epsilon):
270  """Returns a valid epsilon-greedy action and valid action probs.
271 
272  Action probabilities are given by a softmax over legal q-values.
273 
274  Args:
275  info_state: hashable representation of the information state.
276  legal_actions: list of legal actions at `info_state`.
277  epsilon: float, probability of taking an exploratory action.
278 
279  Returns:
280  A valid epsilon-greedy action and valid action probabilities.
281  """
282  probs = np.zeros(self._num_actions_num_actions)
283  if np.random.rand() < epsilon:
284  action = np.random.choice(legal_actions)
285  probs[legal_actions] = 1.0 / len(legal_actions)
286  else:
287  info_state = np.reshape(info_state, [1, -1])
288  q_values = self._session_session.run(
289  self._q_values_q_values, feed_dict={self._info_state_ph_info_state_ph: info_state})[0]
290  legal_q_values = q_values[legal_actions]
291  action = legal_actions[np.argmax(legal_q_values)]
292  probs[action] = 1.0
293  return action, probs

295  def _get_epsilon(self, is_evaluation, power=1.0):
296  """Returns the evaluation or decayed epsilon value."""
297  if is_evaluation:
298  return 0.0
299  decay_steps = min(self._step_counter_step_counter, self._epsilon_decay_duration_epsilon_decay_duration)
300  decayed_epsilon = (
301  self._epsilon_end_epsilon_end + (self._epsilon_start_epsilon_start - self._epsilon_end_epsilon_end) *
302  (1 - decay_steps / self._epsilon_decay_duration_epsilon_decay_duration)**power)
303  return decayed_epsilon

305  def learn(self):
306  """Compute the loss on sampled transitions and perform a Q-network update.
307 
308  If there are not enough elements in the buffer, no loss is computed and
309  `None` is returned instead.
310 
311  Returns:
312  The average loss obtained on this batch of transitions or `None`.
313  """
314 
315  if (len(self._replay_buffer_replay_buffer) < self._batch_size_batch_size or
316  len(self._replay_buffer_replay_buffer) < self._min_buffer_size_to_learn_min_buffer_size_to_learn):
317  return None
318 
319  transitions = self._replay_buffer_replay_buffer.sample(self._batch_size_batch_size)
320  info_states = [t.info_state for t in transitions]
321  actions = [t.action for t in transitions]
322  rewards = [t.reward for t in transitions]
323  next_info_states = [t.next_info_state for t in transitions]
324  are_final_steps = [t.is_final_step for t in transitions]
325  legal_actions_mask = [t.legal_actions_mask for t in transitions]
326  loss, _ = self._session_session.run(
327  [self._loss_loss, self._learn_step_learn_step],
328  feed_dict={
329  self._info_state_ph_info_state_ph: info_states,
330  self._action_ph_action_ph: actions,
331  self._reward_ph_reward_ph: rewards,
332  self._is_final_step_ph_is_final_step_ph: are_final_steps,
333  self._next_info_state_ph_next_info_state_ph: next_info_states,
334  self._legal_actions_mask_ph_legal_actions_mask_ph: legal_actions_mask,
335  })
336  return loss
