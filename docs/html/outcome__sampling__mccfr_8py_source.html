61  def _episode(self, state, update_player, my_reach, opp_reach, sample_reach):
62  """Runs an episode of outcome sampling.
63 
64  Args:
65  state: the open spiel state to run from (will be modified in-place).
66  update_player: the player to update regrets for (the other players
67  update average strategies)
68  my_reach: reach probability of the update player
69  opp_reach: reach probability of all the opponents (including chance)
70  sample_reach: reach probability of the sampling (behavior) policy
71 
72  Returns:
73  util is a real value representing the utility of the update player
74  """
75  if state.is_terminal():
76  return state.player_return(update_player)
77 
78  if state.is_chance_node():
79  outcomes, probs = zip(*state.chance_outcomes())
80  aidx = np.random.choice(range(len(outcomes)), p=probs)
81  state.apply_action(outcomes[aidx])
82  return self._episode_episode(state, update_player, my_reach,
83  probs[aidx] * opp_reach, probs[aidx] * sample_reach)
84 
85  cur_player = state.current_player()
86  info_state_key = state.information_state_string(cur_player)
87  legal_actions = state.legal_actions()
88  num_legal_actions = len(legal_actions)
89  infostate_info = self._lookup_infostate_info_lookup_infostate_info(info_state_key,
90  num_legal_actions)
91  policy = self._regret_matching_regret_matching(infostate_info[mccfr.REGRET_INDEX],
92  num_legal_actions)
93  if cur_player == update_player:
94  uniform_policy = (
95  np.ones(num_legal_actions, dtype=np.float64) / num_legal_actions)
96  sample_policy = self._expl_expl * uniform_policy + (1.0 - self._expl_expl) * policy
97  else:
98  sample_policy = policy
99  sampled_aidx = np.random.choice(range(num_legal_actions), p=sample_policy)
100  state.apply_action(legal_actions[sampled_aidx])
101  if cur_player == update_player:
102  new_my_reach = my_reach * policy[sampled_aidx]
103  new_opp_reach = opp_reach
104  else:
105  new_my_reach = my_reach
106  new_opp_reach = opp_reach * policy[sampled_aidx]
107  new_sample_reach = sample_reach * sample_policy[sampled_aidx]
108  child_value = self._episode_episode(state, update_player, new_my_reach,
109  new_opp_reach, new_sample_reach)
110 
111  # Compute each of the child estimated values.
112  child_values = np.zeros(num_legal_actions, dtype=np.float64)
113  for aidx in range(num_legal_actions):
114  child_values[aidx] = self._baseline_corrected_child_value_baseline_corrected_child_value(
115  state, infostate_info, sampled_aidx, aidx, child_value,
116  sample_policy[aidx])
117  value_estimate = 0
118  for aidx in range(num_legal_actions):
119  value_estimate += policy[aidx] * child_values[aidx]
120 
121  if cur_player == update_player:
122  # Now the regret and avg strategy updates.
123  policy = self._regret_matching_regret_matching(infostate_info[mccfr.REGRET_INDEX],
124  num_legal_actions)
125 
126  # Estimate for the counterfactual value of the policy.
127  cf_value = value_estimate * opp_reach / sample_reach
128 
129  # Update regrets.
130  #
131  # Note: different from Chapter 4 of Lanctot '13 thesis, the utilities
132  # coming back from the recursion are already multiplied by the players'
133  # tail reaches and divided by the sample tail reach. So when adding
134  # regrets to the table, we need only multiply by the opponent reach and
135  # divide by the sample reach to this point.
136  for aidx in range(num_legal_actions):
137  # Estimate for the counterfactual value of the policy replaced by always
138  # choosing sampled_aidx at this information state.
139  cf_action_value = child_values[aidx] * opp_reach / sample_reach
140  self._add_regret_add_regret(info_state_key, aidx, cf_action_value - cf_value)
141 
142  # update the average policy
143  for aidx in range(num_legal_actions):
144  increment = my_reach * policy[aidx] / sample_reach
145  self._add_avstrat_add_avstrat(info_state_key, aidx, increment)
146 
147  return value_estimate
