75 def grid_to_forbidden_states(grid):
76  """Converts a grid into string representation of forbidden states.
77 
78  Args:
79  grid: Rows of the grid. '#' character denotes a forbidden state. All rows
80  should have the same number of columns, i.e. cells.
81 
82  Returns:
83  String representation of forbidden states in the form of x (column) and y
84  (row) pairs, e.g. [1|1;0|2].
85  """
86  forbidden_states = []
87  num_cols = len(grid[0])
88  for y, row in enumerate(grid):
89  assert len(row) == num_cols, f"Number of columns should be {num_cols}."
90  for x, cell in enumerate(row):
91  if cell == "#":
92  forbidden_states.append(f"{x}|{y}")
93  return "[" + ";".join(forbidden_states) + "]"

371  """Returns the possible chance outcomes and their probabilities."""
