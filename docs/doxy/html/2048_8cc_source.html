62 return r >= 0 && r < kRows && c >= 0 && c < kColumns;

65// An array that dictates the order of traveral of row and column coordinated
66// by direction. E.g, kTraversals[direction][0] is an array of size four
67// refering to the row order, and kTraversals[direction][1] is an array of
68// size four refering to the column order.
69constexpr std::array<std::array<std::array<int, 4>, 2>, 4> kTraversals = {{
70 {{{0, 1, 2, 3}, {0, 1, 2, 3}}}, // Up
71 {{{0, 1, 2, 3}, {3, 2, 1, 0}}}, // Right
72 {{{3, 2, 1, 0}, {0, 1, 2, 3}}}, // Down
73 {{{0, 1, 2, 3}, {0, 1, 2, 3}}} // Left
74}};

109constexpr Coordinate GetVector(int direction) {
110 switch (direction) {
111 case kMoveUp:
112 return Coordinate(-1, 0);
113 case kMoveRight:
114 return Coordinate(0, 1);
115 case kMoveDown:
116 return Coordinate(1, 0);
117 case kMoveLeft:
118 return Coordinate(0, -1);
119 default:
120 SpielFatalError("Unrecognized direction");
121 }
122}

124std::array<Coordinate, 2> TwentyFortyEightState::FindFarthestPosition(
125 int r, int c, int direction) const {
126 // Progress towards the vector direction until an obstacle is found
127 Coordinate prev = Coordinate(r, c);
128 Coordinate direction_diff = GetVector(direction);
129 do {
130 prev = Coordinate(r, c);
131 r += direction_diff.row;
132 c += direction_diff.column;
133 } while (InBounds(r, c) && CellAvailable(r, c));
134 return std::array<Coordinate, 2>{prev, Coordinate(r, c)};
135}

137bool TwentyFortyEightState::TileMatchAvailable(int r, int c) const {
138 int tile = BoardAt(r, c).value;
139 if (tile > 0) {
140 for (int direction : kPlayerActions) {
141 Coordinate vector = GetVector(direction);
142 int other = GetCellContent(r + vector.row, c + vector.column);
143 if (other > 0 && other == tile) {
144 return true; // These two tiles can be merged
145 }
146 }
147 }
148 return false;
149}

176void TwentyFortyEightState::DoApplyAction(Action action) {
177 if (IsChanceNode()) {
178 // The original 2048 game starts with two random tiles. To achieve this,
179 // an extra move is given to the chance player during the beginning of the
180 // game.
181 if (!extra_chance_turn_) {
182 current_player_ = 0;
183 }
184 extra_chance_turn_ = false;
185
186 if (action == kNoCellAvailableAction) {
187 return;
188 }
189 ChanceAction chance_action = SpielActionToChanceAction(action);
190 SetBoard(
191 chance_action.row, chance_action.column,
192 Tile(chance_action.is_four ? kChanceTiles[1] : kChanceTiles[0], false));
193 return;
194 }
195 action_score_ = 0;
196 const std::array<std::array<int, 4>, 2>& traversals = kTraversals[action];
197 PrepareTiles();
198 for (int r : traversals[0]) {
199 for (int c : traversals[1]) {
200 int tile = GetCellContent(r, c);
201 if (tile > 0) {
202 bool moved = false;
203 std::array<Coordinate, 2> positions =
204 FindFarthestPosition(r, c, action);
205 Coordinate farthest_pos = positions[0];
206 Coordinate next_pos = positions[1];
207 int next_cell = GetCellContent(next_pos.row, next_pos.column);
208 if (next_cell > 0 && next_cell == tile &&
209 !BoardAt(next_pos).is_merged) {
210 int merged = tile * 2;
211 action_score_ += merged;
212 SetBoard(next_pos.row, next_pos.column, Tile(merged, true));
213 moved = true;
214 } else if (farthest_pos.row != r || farthest_pos.column != c) {
215 SetBoard(farthest_pos.row, farthest_pos.column, Tile(tile, false));
216 moved = true;
217 }
218 if (moved) {
219 SetBoard(r, c, Tile(0, false));
220 current_player_ = kChancePlayerId;
221 }
222 }
223 }
224 }
225 total_score_ += action_score_;
226 total_actions_++;
227}

229bool TwentyFortyEightState::DoesActionChangeBoard(Action action) const {
230 const std::array<std::array<int, 4>, 2>& traversals = kTraversals[action];
231 for (int r : traversals[0]) {
232 for (int c : traversals[1]) {
233 int tile = GetCellContent(r, c);
234 if (tile > 0) {
235 std::array<Coordinate, 2> positions =
236 FindFarthestPosition(r, c, action);
237 Coordinate farthest_pos = positions[0];
238 Coordinate next_pos = positions[1];
239 int next_cell = GetCellContent(next_pos.row, next_pos.column);
240 if (next_cell > 0 && next_cell == tile &&
241 !BoardAt(next_pos).is_merged) {
242 return true;
243 } else if (farthest_pos.row != r || farthest_pos.column != c) {
244 return true;
245 }
246 }
247 }
248 }
249 return false;
250}

252std::string TwentyFortyEightState::ActionToString(Player player,
253 Action action_id) const {
254 if (player == kChancePlayerId) {
255 if (action_id == kNoCellAvailableAction) {
256 return "No Cell Available";
257 }
258 ChanceAction chance_action = SpielActionToChanceAction(action_id);
259 return absl::StrCat(chance_action.is_four ? 4 : 2, " added to row ",
260 chance_action.row + 1, ", column ",
261 chance_action.column + 1);
262 }
263 switch (action_id) {
264 case kMoveUp:
265 return "Up";
266 case kMoveRight:
267 return "Right";
268 case kMoveDown:
269 return "Down";
270 case kMoveLeft:
271 return "Left";
272 default:
273 return "Invalid action";
274 }
275}

289ActionsAndProbs TwentyFortyEightState::ChanceOutcomes() const {
290 int count = AvailableCellCount();
291 if (count == 0) {
292 return {{kNoCellAvailableAction, 1.0}};
293 }
294 ActionsAndProbs action_and_probs;
295 action_and_probs.reserve(count * 2);
296 for (int r = 0; r < kRows; r++) {
297 for (int c = 0; c < kColumns; c++) {
298 if (BoardAt(r, c).value == 0) {
299 // 2 appearing randomly on the board should be 9 times as likely as a 4.
300 action_and_probs.emplace_back(
301 ChanceActionToSpielAction(ChanceAction(r, c, false)), .9 / count);
302 action_and_probs.emplace_back(
303 ChanceActionToSpielAction(ChanceAction(r, c, true)), .1 / count);
304 }
305 }
306 }
307 return action_and_probs;
308}

310std::vector<Action> TwentyFortyEightState::LegalActions() const {
311 if (IsTerminal()) {
312 return {};
313 }
314 if (IsChanceNode()) {
315 return LegalChanceOutcomes();
316 }
317
318 // Construct a vector from the array.
319 std::vector<Action> actions =
320 std::vector<Action>(kPlayerActions.begin(), kPlayerActions.end());
321
322 std::vector<Action> actions_allowed = {};
323 for (Action action : actions) {
324 if (DoesActionChangeBoard(action)) actions_allowed.push_back(action);
325 }
326 return actions_allowed;
327}

342bool TwentyFortyEightState::IsTerminal() const {
343 if (move_number_ >= parent_game_.MaxGameLength()) {
344 return true;
345 }
346
347 // Scan the board.
348 int count = 0;
349 int tile_matches_available = 0;
350 for (int r = 0; r < kRows; r++) {
351 for (int c = 0; c < kColumns; c++) {
352 // Check for 2048, if necessary,
353 if (BoardAt(r, c).value == parent_game_.max_tile()) {
354 return true;
355 }
356
357 // Check for increase of available cell count.
358 if (BoardAt(r, c).value == 0) {
359 count++;
360 }
361
362 // Check for tile matches.
363 if (TileMatchAvailable(r, c)) {
364 tile_matches_available++;
365 }
366 }
367 }
368
369 if (count == 0 && tile_matches_available == 0) {
370 return true;
371 } else {
372 return false;
373 }
374}
