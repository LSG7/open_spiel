76 class MFGLinearQuadraticGame(pyspiel.Game):
77  """A Mean-Field Linear Quadratic game.
78 
79  For now, only single-population setting is covered. A game starts by an
80  initial chance node that selects the initial state of the player in the MFG.
81  Then the game sequentially alternates between:
82  - An action selection node (where the player id is >= 0)
83  - A chance node (the player id is pyspiel.PlayerId.CHANCE)
84  - A Mean Field node (the player id is pyspiel.PlayerId.MEAN_FIELD)
85  """

310  def eta_t(self):
311  """Computes the theoretical policy's `eta_t` term."""
312  # pylint: disable=invalid-name
313  kappa = self.kappakappa
314  K = self.mean_revertmean_revert
315  q = self.cross_qcross_q
316  c = self.terminal_costterminal_cost
317  T = self.horizonhorizon
318  t = self.tt
319 
320  R = (K + q) ** 2 + (kappa - q**2)
321  deltap = -(K + q) + math.sqrt(R)
322  deltam = -(K + q) - math.sqrt(R)
323  numerator = -(kappa - q**2) * (
324  math.exp((deltap - deltam) * (T - t)) - 1
325  ) - c * (deltap * math.exp((deltap - deltam) * (T - t)) - deltam)
326  denominator = (
327  deltam * math.exp((deltap - deltam) * (T - t)) - deltap
328  ) - c * (math.exp((deltap - deltam) * (T - t)) - 1)
329  return numerator / denominator
