15"""Train a policy net on Hearts actions based given a dataset of trajectories.

17Trajectories from the Hearts bot Xinxin can be generated using

63def _trajectory(line: str):
64 """Returns parsed action trajectory."""
65 actions = [int(x) for x in line.split(' ')]
66 return tuple(actions)

69def make_dataset(file: str):
70 """Creates dataset as a generator of single examples."""
71 lines = [line for line in open(file)]
72 while True:
73 np.random.shuffle(lines)
74 for line in lines:
75 trajectory = _trajectory(line)
76 # skip pass_dir and deal actions
77 action_index = np.random.randint(NUM_CARDS + 1, len(trajectory))
78 state = GAME.new_initial_state()
79 for action in trajectory[:action_index]:
80 state.apply_action(action)
81 yield (state.information_state_tensor(), trajectory[action_index])

84def batch(dataset, batch_size: int):
85 """Creates a batched dataset from a one-at-a-time dataset."""
86 observations = np.zeros([batch_size] + GAME.information_state_tensor_shape(),
87 np.float32)
88 labels = np.zeros(batch_size, dtype=np.int32)
89 while True:
90 for batch_index in range(batch_size):
91 observations[batch_index], labels[batch_index] = next(dataset)
92 yield observations, labels

95def one_hot(x, k):
96 """Returns a one-hot encoding of `x` of size `k`."""
97 return jnp.array(x[..., jnp.newaxis] == jnp.arange(k), dtype=np.float32)

100def net_fn(x):
101 """Haiku module for our network."""
102 layers = []
103 for layer_size in FLAGS.hidden_layer_sizes:
104 layers.append(hk.Linear(int(layer_size)))
105 layers.append(jax.nn.relu)
106 layers.append(hk.Linear(NUM_ACTIONS))
107 layers.append(jax.nn.log_softmax)
108 net = hk.Sequential(layers)
109 return net(x)

112def main(argv):
113 if len(argv) > 1:
114 raise app.UsageError('Too many command-line arguments.')
115
116 if FLAGS.hidden_layer_sizes is None:
117 # Cannot pass default arguments as lists due to style requirements, so we
118 # override it here if they are not set.
119 FLAGS.hidden_layer_sizes = DEFAULT_LAYER_SIZES
120
121 # Make the network.
122 net = hk.without_apply_rng(hk.transform(net_fn))
123
124 # Make the optimiser.
125 opt = optax.adam(FLAGS.step_size)
126
127 @jax.jit
128 def loss(
129 params: Params,
130 inputs: np.ndarray,
131 targets: np.ndarray,
132 ) -> jax.Array:
133 """Cross-entropy loss."""
134 assert targets.dtype == np.int32
135 log_probs = net.apply(params, inputs)
136 return -jnp.mean(one_hot(targets, NUM_ACTIONS) * log_probs)
137
138 @jax.jit
139 def accuracy(
140 params: Params,
141 inputs: np.ndarray,
142 targets: np.ndarray,
143 ) -> jax.Array:
144 """Classification accuracy."""
145 predictions = net.apply(params, inputs)
146 return jnp.mean(jnp.argmax(predictions, axis=-1) == targets)
147
148 @jax.jit
149 def update(
150 params: Params,
151 opt_state: OptState,
152 inputs: np.ndarray,
153 targets: np.ndarray,
154 ) -> Tuple[Params, OptState]:
155 """Learning rule (stochastic gradient descent)."""
156 _, gradient = jax.value_and_grad(loss)(params, inputs, targets)
157 updates, opt_state = opt.update(gradient, opt_state)
158 new_params = optax.apply_updates(params, updates)
159 return new_params, opt_state

161 def output_samples(params: Params, max_samples: int):
162 """Output some cases where the policy disagrees with the dataset action."""
163 if max_samples == 0:
164 return
165 count = 0
166 with open(os.path.join(FLAGS.data_path, 'test.txt')) as f:
167 lines = list(f)
168 np.random.shuffle(lines)
169 for line in lines:
170 state = GAME.new_initial_state()
171 actions = _trajectory(line)
172 for action in actions:
173 if not state.is_chance_node():
174 observation = np.array(state.information_state_tensor(), np.float32)
175 policy = np.exp(net.apply(params, observation))
176 probs_actions = [(p, a) for a, p in enumerate(policy)]
177 pred = max(probs_actions)[1]
178 if pred != action:
179 print(state)
180 for p, a in reversed(sorted(probs_actions)[-TOP_K_ACTIONS:]):
181 print('{:7} {:.2f}'.format(state.action_to_string(a), p))
182 print('Ground truth {}\n'.format(state.action_to_string(action)))
183 count += 1
184 break
185 state.apply_action(action)
186 if count >= max_samples:
187 return
188
189 # Store what we need to rebuild the Haiku net.
190 if FLAGS.save_path:
191 filename = os.path.join(FLAGS.save_path, 'layers.txt')
192 with open(filename, 'w') as layer_def_file:
193 for s in FLAGS.hidden_layer_sizes:
194 layer_def_file.write(f'{s} ')
195 layer_def_file.write('\n')
196
197 # Make datasets.
198 if FLAGS.data_path is None:
199 raise app.UsageError(
200 'Please generate your own supervised training data and supply the local'
201 'location as --data_path')
202 train = batch(
203 make_dataset(os.path.join(FLAGS.data_path, 'train.txt')),
204 FLAGS.train_batch)
205 test = batch(
206 make_dataset(os.path.join(FLAGS.data_path, 'test.txt')), FLAGS.eval_batch)
207
208 # Initialize network and optimiser.
209 if FLAGS.checkpoint_file:
210 with open(FLAGS.checkpoint_file, 'rb') as pkl_file:
211 params, opt_state = pickle.load(pkl_file)
212 else:
213 rng = jax.random.PRNGKey(FLAGS.rng_seed) # seed used for network weights
214 inputs, unused_targets = next(train)
215 params = net.init(rng, inputs)
216 opt_state = opt.init(params)
217
218 # Train/eval loop.
219 for step in range(FLAGS.iterations):
220 # Do SGD on a batch of training examples.
221 inputs, targets = next(train)
222 params, opt_state = update(params, opt_state, inputs, targets)
223
224 # Periodically evaluate classification accuracy on the test set.
225 if (1 + step) % FLAGS.eval_every == 0:
226 inputs, targets = next(test)
227 test_accuracy = accuracy(params, inputs, targets)
228 print(f'After {1+step} steps, test accuracy: {test_accuracy}.')
229 if FLAGS.save_path:
230 filename = os.path.join(FLAGS.save_path, f'checkpoint-{1 + step}.pkl')
231 with open(filename, 'wb') as pkl_file:
232 pickle.dump((params, opt_state), pkl_file)
233 output_samples(params, FLAGS.num_examples)

237 app.run(main)
