36 struct TournamentSettings {
37  // All times are in miliseconds.
38  int timeout_ready = 200;
39  int timeout_start = 100;
40  int timeout_act = 100;
41  int timeout_ponder = 50;
42  int timeout_match_over = 100;
43  int time_tournament_over = 100;
44 
45  // Number of invalid responses of a bot that are tolerated within a match.
46  // Exceeding this number results in marking the match as corrupted,
47  // random actions are selected instead, and the bot is forced to restart.
48  // If this happens in too many matches, the bot will be disqualified.
49  int max_invalid_behaviors = 1;
50 
51  // If the bot corrupts more than this fraction of tournament matches,
52  // it is disqualified.
53  double disqualification_rate = 0.1;
54 };

72 struct TournamentResults {
73  const int num_bots;
74 
75  // Match result for each played match.
76  std::vector<MatchResult> matches;
77 
78  // Incremental computation of match statistics (mean, variance), per bot.
79  std::vector<double> returns_mean;
80  // For computation of variance, must be normalized first.
81  std::vector<double> returns_agg;
82  // Average length of a match.
83  double history_len_mean = 0.;
84 
85  // Summary statistics of how many corrupted matches occurred for each player,
86  // i.e. the player did not respond entirely correctly in some played match.
87  //
88  // A match is marked as corrupted if:
89  // 1) There was a protocol error.
90  // 2) The number of other errors (illegal_actions, ponder_error, time_over)
91  // exceeded the TournamentSettings::max_invalid_behaviors
92  std::vector<int> corrupted_matches;
93 
94  // Flag whether a given bot was disqualified.
95  // The disqualification criteria are following:
96  //
97  // 1) The bot could not be properly started.
98  // 2) The number of corrupted matches exceeds corruption_threshold,
99  // i.e. num_matches * TournamentSettings::disqualification_rate
100  std::vector<bool> disqualified;
101 
102  // Number of bot restarts. A restart is forced if a match is corrupted.
103  std::vector<int> restarts;
104 
105  TournamentResults(int num_bots);
106  int num_matches() const { return matches.size(); }
107  double returns_var(int pl) const { return returns_agg[pl] / (num_matches()); }
108  std::string ToString() const;
109  void PrintVerbose(std::ostream&) const;
110  void PrintCsv(std::ostream&, bool print_header = false) const;
111 };
