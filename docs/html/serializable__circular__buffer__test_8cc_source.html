53 void TestSerializableCircularBuffer() {
54  SerializableCircularBuffer<int> buffer(4);
55  std::mt19937 rng;
56  std::vector<int> sample;
57 
58  SPIEL_CHECK_TRUE(buffer.Empty());
59  SPIEL_CHECK_EQ(buffer.Size(), 0);
60 
61  buffer.Add(13);
62  SPIEL_CHECK_FALSE(buffer.Empty());
63  SPIEL_CHECK_EQ(buffer.Size(), 1);
64  SPIEL_CHECK_EQ(buffer.TotalAdded(), 1);
65  SPIEL_CHECK_EQ(buffer[0], 13);
66 
67  sample = buffer.Sample(&rng, 1);
68  SPIEL_CHECK_EQ(sample.size(), 1);
69  SPIEL_CHECK_EQ(sample[0], 13);
70 
71  buffer.Add(14);
72  buffer.Add(15);
73  buffer.Add(16);
74 
75  SPIEL_CHECK_EQ(buffer.Size(), 4);
76  SPIEL_CHECK_EQ(buffer.TotalAdded(), 4);
77 
78  sample = buffer.Sample(&rng, 2);
79  SPIEL_CHECK_EQ(sample.size(), 2);
80  SPIEL_CHECK_GE(sample[0], 13);
81  SPIEL_CHECK_LE(sample[0], 16);
82  SPIEL_CHECK_GE(sample[1], 13);
83  SPIEL_CHECK_LE(sample[1], 16);
84 
85  buffer.Add(17);
86  buffer.Add(18);
87 
88  SPIEL_CHECK_EQ(buffer.Size(), 4);
89  SPIEL_CHECK_EQ(buffer.TotalAdded(), 6);
90 
91  sample = buffer.Sample(&rng, 1);
92  SPIEL_CHECK_EQ(sample.size(), 1);
93  SPIEL_CHECK_GE(sample[0], 15);
94  SPIEL_CHECK_LE(sample[0], 18);
95 }
