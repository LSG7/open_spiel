41class UniformRandomBot : public Bot {
42 public:
43 UniformRandomBot(Player player_id, int seed)
44 : player_id_(player_id), rng_(seed) {}
45 ~UniformRandomBot() = default;
46
47 void RestartAt(const State&) override {}
48 Action Step(const State& state) override {
49 return StepWithPolicy(state).second;
50 }
51 bool ProvidesPolicy() override { return true; }
52 ActionsAndProbs GetPolicy(const State& state) override {
53 ActionsAndProbs policy;
54 auto legal_actions = state.LegalActions(player_id_);
55 const int num_legal_actions = legal_actions.size();
56 const double p = 1.0 / num_legal_actions;
57 for (auto action : legal_actions) policy.emplace_back(action, p);
58 return policy;
59 }
60
61 std::pair<ActionsAndProbs, Action> StepWithPolicy(
62 const State& state) override {
63 ActionsAndProbs policy = GetPolicy(state);
64 const int num_legal_actions = policy.size();
65
66 int selection =
67 absl::uniform_int_distribution<int>(0, num_legal_actions - 1)(rng_);
68 return std::make_pair(policy, policy[selection].first);
69 }
70
71 bool IsClonable() const override { return true; }
72 std::unique_ptr<Bot> Clone() override {
73 return std::make_unique<UniformRandomBot>(*this);
74 }
75 UniformRandomBot(const UniformRandomBot& other) = default;
76
77 private:
78 const Player player_id_;
79 std::mt19937 rng_;
80};

82// A UniformRandomBot that keeps a copy of the state up to date. This exists
83// primarily to verify that InformAction is called correctly by the run loop.
84class StatefulRandomBot : public UniformRandomBot {
85 public:
86 StatefulRandomBot(const Game& game, Player player_id, int seed)
87 : UniformRandomBot(player_id, seed), state_(game.NewInitialState()) {}
88
89 void Restart() override { state_ = state_->GetGame()->NewInitialState(); }
90 void RestartAt(const State& state) override { state_ = state.Clone(); }
91 void InformAction(const State& state, Player player_id,
92 Action action) override {
93 CheckStatesEqual(state, *state_);
94 state_->ApplyAction(action);
95 }
96 ActionsAndProbs GetPolicy(const State& state) override {
97 CheckStatesEqual(state, *state_);
98 return UniformRandomBot::GetPolicy(*state_);
99 }
100 std::pair<ActionsAndProbs, Action> StepWithPolicy(
101 const State& state) override {
102 std::pair<ActionsAndProbs, Action> ret =
103 UniformRandomBot::StepWithPolicy(*state_);
104 state_->ApplyAction(ret.second);
105 return ret;
106 }
107
108 std::unique_ptr<Bot> Clone() override {
109 return std::make_unique<StatefulRandomBot>(*this);
110 }
111 StatefulRandomBot(const StatefulRandomBot& other)
112 : UniformRandomBot(other), state_(other.state_->Clone()) {}
113
114 private:
115 void CheckStatesEqual(const State& state1, const State& state2) const {
116 SPIEL_CHECK_EQ(state1.History(), state2.History());
117 SPIEL_CHECK_EQ(state1.CurrentPlayer(), state2.CurrentPlayer());
118 SPIEL_CHECK_EQ(state1.LegalActions(), state2.LegalActions());
119 if (!state1.IsChanceNode()) {
120 SPIEL_CHECK_EQ(state1.ObservationTensor(), state2.ObservationTensor());
121 }
122 }
123 std::unique_ptr<State> state_;
124};

158class FixedActionPreferenceBot : public Bot {
159 public:
160 FixedActionPreferenceBot(Player player_id, const std::vector<Action>& actions)
161 : Bot(), player_id_(player_id), actions_(actions) {}
162 ~FixedActionPreferenceBot() = default;
163
164 void RestartAt(const State&) override {}
165 Action Step(const State& state) override {
166 return StepWithPolicy(state).second;
167 }
168 bool ProvidesPolicy() override { return true; }
169 ActionsAndProbs GetPolicy(const State& state) override {
170 std::vector<Action> legal_actions = state.LegalActions(player_id_);
171 std::unordered_set<Action> legal_actions_set =
172 std::unordered_set<Action>(legal_actions.begin(), legal_actions.end());
173 for (Action action : actions_) {
174 if (legal_actions_set.count(action) == 1) {
175 return {{action, 1.0}};
176 }
177 }
178 SpielFatalError("No legal actions in action list.");
179 }
180
181 std::pair<ActionsAndProbs, Action> StepWithPolicy(
182 const State& state) override {
183 ActionsAndProbs actions_and_probs = GetPolicy(state);
184 return {actions_and_probs, actions_and_probs[0].first};
185 }
186
187 bool IsClonable() const override { return true; }
188 std::unique_ptr<Bot> Clone() override {
189 return std::make_unique<FixedActionPreferenceBot>(*this);
190 }
191 FixedActionPreferenceBot(const FixedActionPreferenceBot& other) = default;
192
193 private:
194 const Player player_id_;
195 std::vector<Action> actions_;
196};

245std::vector<Action> ActionsFromStr(const absl::string_view& str,
246 const absl::string_view& delim) {
247 std::vector<Action> actions;
248 for (absl::string_view token : absl::StrSplit(str, delim)) {
249 int v;
250 SPIEL_CHECK_TRUE(absl::SimpleAtoi(token, &v));
251 actions.push_back(v);
252 }
253 return actions;
254}

288 const std::string& bot_name, std::shared_ptr<const Game> game,
289 Player player_id, const GameParameters& params) {
290 auto iter = factories().find(bot_name);
291 if (iter == factories().end()) {

293 "'. Available bots are:\n",
294 absl::StrJoin(RegisteredBots(), "\n")));

296 } else {
297 const std::unique_ptr<BotFactory>& factory = iter->second;
298 return factory->Create(std::move(game), player_id, params);

313std::vector<std::string> BotRegisterer::BotsThatCanPlayGame(const Game& game) {
314 std::vector<std::string> bot_names;
315 for (const auto& key_val : factories()) {
316 bool can_play_for_all = true;
317 for (int player_id = 0; player_id < game.NumPlayers(); ++player_id) {
318 if (!key_val.second->CanPlayGame(game, player_id)) {
319 can_play_for_all = false;
320 break;
321 }
322 }
323 if (can_play_for_all) bot_names.push_back(key_val.first);
324 }
325 return bot_names;
326}

351std::unique_ptr<Bot> LoadBot(const std::string& bot_name,
352 const std::shared_ptr<const Game>& game,
353 Player player_id) {
354 GameParameters params = GameParametersFromString(bot_name);
355
356 // We use the "name" parameter, as that is the "short_name", which is what we
357 // want. Otherwise, this will use the "long name", which includes the config.
358 // e.g. if the bot_name is "my_bot(parameter=value)", then we want the
359 // bot_name here to be "my_bot", not "my_bot(parameter=value)".
360 return LoadBot(params["name"].string_value(), game, player_id, params);
361}

366 std::unique_ptr<Bot> result =
