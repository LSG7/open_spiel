128std::unique_ptr<State> PyState::Clone() const {
129 // Create a new State of the right type.
130 auto rv = game_->NewInitialState();
131
132 // Copy the Python-side properties of the state.
133 py::function deepcopy = py::module::import("copy").attr("deepcopy");
134 py::object py_state = py::cast(*rv);
135 for (auto [k, v] : PyDict(*this)) {
136 py_state.attr(k) = deepcopy(v);
137 }
138
139 // Copy the C++-side properties of the state (all on the parent class).
140 // Since we started with a valid initial state, we only need to copy
141 // properties that change during the life of the state - hence num_players,
142 // num_distinct_actions are omitted.
143 PyState* state = open_spiel::down_cast<PyState*>(rv.get());
144 state->history_ = history_;
145 state->move_number_ = move_number_;
146
147 return rv;
148}

159// Register a Python game.
160void RegisterPyGame(const GameType& game_type, py::function creator) {
161 GameRegisterer::RegisterGame(
162 game_type, [game_type, creator](const GameParameters& game_parameters) {
163 py::dict params = py::cast(game_parameters);
164 for (const auto& [k, v] : game_type.parameter_specification) {
165 if (game_parameters.count(k) == 0) {
166 params[pybind11::str(k)] = v;
167 }
168 }
169 auto py_game = creator(params);
170 return py::cast<std::shared_ptr<Game>>(py_game);
171 });
172}

177// Wrapper for using a Python observer from C++.
178// This is not a 'trampoline' class, just a wrapper.
179class PyObserver : public Observer {
180 public:
181 PyObserver(py::object py_observer);
182 void WriteTensor(const State& state, int player,
183 Allocator* allocator) const override;
184 std::string StringFrom(const State& state, int player) const override;
185
186 private:
187 py::object py_observer_;
188 py::function set_from_;
189 py::function string_from_;
190};
191
192PyObserver::PyObserver(py::object py_observer)
193 : Observer(/*has_string=*/true, /*has_tensor=*/true),
194 py_observer_(py_observer),
195 set_from_(py_observer_.attr("set_from")),
196 string_from_(py_observer_.attr("string_from")) {
197 has_tensor_ = !py_observer_.attr("tensor").is_none();
198}
199
200void PyObserver::WriteTensor(const State& state, int player,
201 Allocator* allocator) const {
202 using Array = py::array_t<float, py::array::c_style | py::array::forcecast>;
203 const PyState& py_state = open_spiel::down_cast<const PyState&>(state);
204 set_from_(py_state, player);
205 py::dict dict = py_observer_.attr("dict");
206 for (auto [k, v] : dict) {
207 auto a = py::cast<Array>(v);
208 const int dims = a.ndim();
209 absl::InlinedVector<int, 4> shape(dims);
210 for (int i = 0; i < dims; ++i) shape[i] = a.shape(i);
211 SpanTensor out = allocator->Get(k.cast<std::string>(), shape);
212 std::copy(a.data(), a.data() + a.size(), out.data().begin());
213 }
214}

222 absl::optional<IIGObservationType> iig_obs_type,
223 const GameParameters& params) const {
224 py::object h = py::cast(this);

226 if (!f) SpielFatalError("make_py_observer not implemented");

310// Serialization form for the Python-side attributes is a b64-encoded pickled
311// Python dict (the __dict__ member of the Python object).
312
313py::dict decode_dict(const absl::string_view str) {
314 std::string bytes;
315 SPIEL_CHECK_TRUE(absl::Base64Unescape(str, &bytes));
316 py::function pickle_loads = py::module::import("pickle").attr("loads");
317 return pickle_loads(py::bytes(bytes));
318}
319
320std::string encode_dict(py::dict dict) {
321 py::function pickle_dumps = py::module::import("pickle").attr("dumps");
322 py::bytes bytes = pickle_dumps(dict);
323 return absl::Base64Escape(std::string(bytes));
324}

326inline constexpr const absl::string_view kTagHistory = "history=";
327inline constexpr const absl::string_view kTagMoveNumber = "move_number=";
328inline constexpr const absl::string_view kTagDict = "__dict__=";
329
330void PyState::Deserialize(const std::string& str) {
331 std::vector<absl::string_view> pieces =
332 absl::StrSplit(str, absl::MaxSplits('\n', 2));
333 SPIEL_CHECK_EQ(pieces.size(), 3);
334
335 SPIEL_CHECK_EQ(pieces[0].substr(0, kTagHistory.size()), kTagHistory);
336 auto history_str = pieces[0].substr(kTagHistory.size());
337 if (!history_str.empty()) {
338 for (auto& h : absl::StrSplit(history_str, ',')) {
339 std::vector<absl::string_view> p = absl::StrSplit(h, ':');
340 SPIEL_CHECK_EQ(p.size(), 2);
341 int player, action;
342 SPIEL_CHECK_TRUE(absl::SimpleAtoi(p[0], &player));
343 SPIEL_CHECK_TRUE(absl::SimpleAtoi(p[1], &action));
344 history_.push_back({player, action});
345 }
346 }
347
348 SPIEL_CHECK_EQ(pieces[1].substr(0, kTagMoveNumber.size()), kTagMoveNumber);
349 SPIEL_CHECK_TRUE(
350 absl::SimpleAtoi(pieces[1].substr(kTagMoveNumber.size()), &move_number_));
351
352 SPIEL_CHECK_EQ(pieces[2].substr(0, kTagDict.size()), kTagDict);
353 py::object py_state = py::cast(*this);
354 for (const auto& [k, v] : decode_dict(pieces[2].substr(kTagDict.size()))) {
355 py_state.attr(k) = v;
356 }
357}
358
359std::string PyState::Serialize() const {
360 return absl::StrCat(
361 // C++ Attributes
362 kTagHistory,
363 absl::StrJoin(history_, ",",
364 [](std::string* out, const PlayerAction& pa) {
365 absl::StrAppend(out, pa.player, ":", pa.action);
366 }),
367 "\n", kTagMoveNumber, move_number_, "\n",
368 // Python attributes
369 kTagDict, encode_dict(PyDict(*this)));
370}
