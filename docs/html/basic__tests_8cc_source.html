40 // Information about the simulation history. Used to track past states and
41 // actions for rolling back simulations via UndoAction, and check History.
42 // For simultaneous games, a simultaneous move will be stored as several items.
43 // The state will be nullptr and the player kInvalidHistoryPlayer for invalid
44 // transitions.
45 // The transition state_0 --[action]--> state_1 --[action2]--> ... is stored as:
46 // (state_0, state_0.CurrentPlayer(), action),
47 // (state_1, state_1.CurrentPlayer(), action2), ...
48 struct HistoryItem {
49  std::unique_ptr<State> state;
50  Player player;
51  Action action;
52  HistoryItem(std::unique_ptr<State> _state, Player _player, int _action)
53  : state(std::move(_state)), player(_player), action(_action) {}
54 };

84 void LegalActionsIsEmptyForOtherPlayers(const Game& game, State& state) {
85  if (game.GetType().dynamics == GameType::Dynamics::kSimultaneous) {
86  return;
87  }
88 
89  Player current_player = state.CurrentPlayer();
90  for (Player player = 0; player < game.NumPlayers(); ++player) {
91  if (state.IsChanceNode()) {
92  continue;
93  }
94  if (player != current_player) {
95  int size = state.LegalActions(player).size();
96  // We do not use SPIEL_CHECK_EQ because it does not print the values.
97  if (size != 0) {
98  std::string str = "";
99  absl::StrJoin(state.LegalActions(player), str);
100  SpielFatalError(absl::StrCat(
101  __FILE__, ":", __LINE__, " ", size, " should be 0 for player ",
102  player, "(current_player:", current_player, ")", str));
103  }
104  }
105  }

108 void LegalActionsMaskTest(const Game& game, const State& state, int player,
109  const std::vector<Action>& legal_actions) {
110  std::vector<int> legal_actions_mask = state.LegalActionsMask(player);
111  const int expected_length = state.IsChanceNode() ? game.MaxChanceOutcomes()
112  : game.NumDistinctActions();
113  SPIEL_CHECK_EQ(legal_actions_mask.size(), expected_length);
114  for (Action action : legal_actions) {
115  SPIEL_CHECK_GE(action, 0);
116  SPIEL_CHECK_LT(action, expected_length);
117  SPIEL_CHECK_EQ(legal_actions_mask[action], 1);
118  }
119 
120  int num_ones = 0;
121  for (int i = 0; i < expected_length; ++i) {
122  SPIEL_CHECK_TRUE(legal_actions_mask[i] == 0 || legal_actions_mask[i] == 1);
123  num_ones += legal_actions_mask[i];
124  }
125 
126  SPIEL_CHECK_EQ(num_ones, legal_actions.size());
127 }

187 void CheckReturnsSum(const Game& game, const State& state) {
188  std::vector<double> returns = state.Returns();
189  double rsum = std::accumulate(returns.begin(), returns.end(), 0.0);
190  absl::optional<double> utility_sum = game.UtilitySum();
191 
192  switch (game.GetType().utility) {
193  case GameType::Utility::kZeroSum: {
194  SPIEL_CHECK_EQ(utility_sum, 0.0);
195  SPIEL_CHECK_TRUE(Near(rsum, 0.0, kRewardEpsilon));
196  break;
197  }
198  case GameType::Utility::kConstantSum: {
199  SPIEL_CHECK_TRUE(utility_sum.has_value());
200  SPIEL_CHECK_FLOAT_NEAR(rsum, *utility_sum, kRewardEpsilon);
201  break;
202  }
203  case GameType::Utility::kIdentical: {
204  SPIEL_CHECK_FALSE(utility_sum.has_value());
205  for (int i = 1; i < returns.size(); ++i) {
206  SPIEL_CHECK_TRUE(Near(returns[i], returns[i - 1], kRewardEpsilon));
207  }
208  break;
209  }
210  case GameType::Utility::kGeneralSum: {
211  SPIEL_CHECK_FALSE(utility_sum.has_value());
212  }
213  }
214 }

216 // Tests all observation and information_state related methods which are
217 // supported by the game, for all players.
218 //
219 // The following functions should return valid outputs for valid player, even
220 // on terminal states:
221 // - std::string InformationStateString(Player player)
222 // - std::vector<float> InformationStateTensor(Player player)
223 // - std::string ObservationString(Player player)
224 // - std::vector<float> ObservationTensor(Player player)
225 //
226 // These functions should crash on invalid players: this is tested in
227 // api_test.py as it's simpler to catch the error from Python.
228 void CheckObservables(const Game& game,
229  const State& state,
230  Observation* observation // Can be nullptr
231  ) {
232  for (auto p = Player{0}; p < game.NumPlayers(); ++p) {
233  if (game.GetType().provides_information_state_tensor) {
234  std::vector<float> tensor = state.InformationStateTensor(p);
235  for (float val : tensor) SPIEL_CHECK_TRUE(std::isfinite(val));
236  SPIEL_CHECK_EQ(tensor.size(), game.InformationStateTensorSize());
237  }
238  if (game.GetType().provides_observation_tensor) {
239  std::vector<float> tensor = state.ObservationTensor(p);
240  for (float val : tensor) SPIEL_CHECK_TRUE(std::isfinite(val));
241  SPIEL_CHECK_EQ(tensor.size(), game.ObservationTensorSize());
242  }
243  if (game.GetType().provides_information_state_string) {
244  // Checking it does not raise errors.
245  state.InformationStateString(p);
246  }
247  if (game.GetType().provides_observation_string) {
248  // Checking it does not have errors.
249  state.ObservationString(p);
250  }
251 
252  if (observation != nullptr) {
253  if (observation->HasString()) observation->StringFrom(state, p);
254  if (observation->HasTensor()) observation->SetFrom(state, p);
255  }
256  }
257 }

284 // This is used for mean-field games.
285 std::vector<double> RandomDistribution(int num_states, std::mt19937* rng) {
286  std::uniform_real_distribution<double> rand(0, 1);
287  std::vector<double> distrib;
288  distrib.reserve(num_states);
289  for (int i = 0; i < num_states; ++i) {
290  distrib.push_back(rand(*rng));
291  }
292  double sum = std::accumulate(distrib.begin(), distrib.end(), 0.);
293  for (int i = 0; i < num_states; ++i) {
294  distrib[i] /= sum;
295  }
296  return distrib;
297 }

299 void RandomSimulation(std::mt19937* rng, const Game& game, bool undo,
300  bool serialize, bool verbose, bool mask_test,
301  std::shared_ptr<Observer> observer, // Can be nullptr
302  std::function<void(const State&)> state_checker_fn,
303  int mean_field_population = -1) {
304  std::unique_ptr<Observation> observation =
305  observer == nullptr ? nullptr
306  : std::make_unique<Observation>(game, observer);
307  std::vector<HistoryItem> history;
308  std::vector<double> episode_returns(game.NumPlayers(), 0);
309 
310  int infostate_vector_size = game.GetType().provides_information_state_tensor
311  ? game.InformationStateTensorSize()
312  : 0;
313  if (verbose) {
314  std::cout << "Information state vector size: " << infostate_vector_size
315  << std::endl;
316  }
317 
318  int observation_vector_size = game.GetType().provides_observation_tensor
319  ? game.ObservationTensorSize()
320  : 0;
321  if (verbose) {
322  std::cout << "Observation vector size: " << observation_vector_size
323  << std::endl;
324  }
325 
326  SPIEL_CHECK_TRUE(game.MinUtility() < game.MaxUtility());
327  if (verbose) {
328  std::cout << "Utility range: " << game.MinUtility() << " "
329  << game.MaxUtility() << std::endl;
330 
331  std::cout << "Starting new game.." << std::endl;
332  }
333  std::unique_ptr<open_spiel::State> state;
334  if (mean_field_population == -1) {
335  state = game.NewInitialState();
336  } else {
337  state = game.NewInitialStateForPopulation(mean_field_population);
338  }
339 
340  if (verbose) {
341  std::cout << "Initial state:" << std::endl;
342  std::cout << "State:" << std::endl << state->ToString() << std::endl;
343  }
344  int game_length = 0;
345  int num_moves = 0;
346 
347  while (!state->IsTerminal()) {
348  state_checker_fn(*state);
349 
350  if (verbose) {
351  std::cout << "player " << state->CurrentPlayer() << std::endl;
352  }
353 
354  LegalActionsIsEmptyForOtherPlayers(game, *state);
355  CheckLegalActionsAreSorted(game, *state);
356  CheckActionStringsAreUnique(game, *state);
357 
358  // Test cloning the state.
359  std::unique_ptr<open_spiel::State> state_copy = state->Clone();
360  SPIEL_CHECK_EQ(state->ToString(), state_copy->ToString());
361  SPIEL_CHECK_EQ(state->History(), state_copy->History());
362 
363  if (game.GetType().dynamics == GameType::Dynamics::kMeanField) {
364  SPIEL_CHECK_LT(state->MoveNumber(), game.MaxMoveNumber());
365  SPIEL_CHECK_EQ(state->MoveNumber(), num_moves);
366  }
367 
368  if (serialize && (history.size() < 10 || IsPowerOfTwo(history.size()))) {
369  TestSerializeDeserialize(game, state.get());
370  }
371 
372  if (state->IsChanceNode()) {
373  if (mask_test) LegalActionsMaskTest(game, *state, kChancePlayerId,
374  state->LegalActions());
375  // Chance node; sample one according to underlying distribution
376  std::vector<std::pair<Action, double>> outcomes = state->ChanceOutcomes();
377  auto [action, prob] = open_spiel::SampleAction(outcomes, *rng);
378 
379  if (verbose) {
380  std::cout << "sampled outcome: "
381  << state->ActionToString(kChancePlayerId, action)
382  << " with prob " << prob
383  << std::endl;
384  }
385  history.emplace_back(state->Clone(), kChancePlayerId, action);
386  state->ApplyAction(action);
387 
388  if (undo && (history.size() < 10 || IsPowerOfTwo(history.size()))) {
389  TestUndo(state->Clone(), history);
390  }
391  num_moves++;
392  } else if (state->CurrentPlayer() == open_spiel::kSimultaneousPlayerId) {
393  std::vector<double> rewards = state->Rewards();
394  std::vector<double> returns = state->Returns();
395  SPIEL_CHECK_EQ(rewards.size(), game.NumPlayers());
396  SPIEL_CHECK_EQ(returns.size(), game.NumPlayers());
397  for (auto p = Player{0}; p < game.NumPlayers(); ++p) {
398  episode_returns[p] += rewards[p];
399  }
400  if (verbose) {
401  std::cout << "Rewards: " << absl::StrJoin(rewards, " ") << std::endl;
402  std::cout << "Returns: " << absl::StrJoin(returns, " ") << std::endl;
403  std::cout << "Sum Rewards: " << absl::StrJoin(episode_returns, " ")
404  << std::endl;
405  }
406  for (auto p = Player{0}; p < game.NumPlayers(); ++p) {
407  SPIEL_CHECK_TRUE(Near(episode_returns[p], returns[p], kRewardEpsilon));
408  }
409 
410  // Players choose simultaneously.
411  std::vector<Action> joint_action;
412 
413  // Sample an action for each player
414  for (auto p = Player{0}; p < game.NumPlayers(); p++) {
415  std::vector<Action> actions = state->LegalActions(p);
416  Action action = 0;
417  if (!actions.empty()) {
418  if (mask_test) LegalActionsMaskTest(game, *state, p, actions);
419  std::uniform_int_distribution<int> dis(0, actions.size() - 1);
420  action = actions[dis(*rng)];
421  }
422  joint_action.push_back(action);
423  if (p == 0) {
424  history.emplace_back(state->Clone(), kInvalidHistoryPlayer, action);
425  } else {
426  history.emplace_back(nullptr, kInvalidHistoryPlayer, action);
427  }
428  if (verbose) {
429  std::cout << "player " << p << " chose "
430  << state->ActionToString(p, action) << std::endl;
431  }
432  CheckObservables(game, *state, observation.get());
433  }
434 
435  ApplyActionTestClone(game, state.get(), joint_action);
436  game_length++;
437  } else if (state->CurrentPlayer() == open_spiel::kMeanFieldPlayerId) {
438  auto support = state->DistributionSupport();
439  state->UpdateDistribution(RandomDistribution(support.size(), rng));
440  } else {
441  std::vector<double> rewards = state->Rewards();
442  std::vector<double> returns = state->Returns();
443  SPIEL_CHECK_EQ(rewards.size(), game.NumPlayers());
444  SPIEL_CHECK_EQ(returns.size(), game.NumPlayers());
445  for (auto p = Player{0}; p < game.NumPlayers(); ++p) {
446  episode_returns[p] += rewards[p];
447  }
448  if (verbose) {
449  std::cout << "Rewards: " << absl::StrJoin(rewards, " ") << std::endl;
450  std::cout << "Returns: " << absl::StrJoin(returns, " ") << std::endl;
451  std::cout << "Sum Rewards: " << absl::StrJoin(episode_returns, " ")
452  << std::endl;
453  }
454  for (auto p = Player{0}; p < game.NumPlayers(); ++p) {
455  SPIEL_CHECK_TRUE(Near(episode_returns[p], returns[p], kRewardEpsilon));
456  }
457 
458  // Decision node.
459  Player player = state->CurrentPlayer();
460 
461  CheckObservables(game, *state, observation.get());
462 
463  // Sample an action uniformly.
464  std::vector<Action> actions = state->LegalActions();
465  if (mask_test) LegalActionsMaskTest(game, *state, state->CurrentPlayer(),
466  actions);
467  if (state->IsTerminal())
468  SPIEL_CHECK_TRUE(actions.empty());
469  else
470  SPIEL_CHECK_FALSE(actions.empty());
471  std::uniform_int_distribution<int> dis(0, actions.size() - 1);
472  Action action = actions[dis(*rng)];
473 
474  if (verbose) {
475  std::cout << "chose action: " << action << " ("
476  << state->ActionToString(player, action) << ")" << std::endl;
477  }
478  history.emplace_back(state->Clone(), player, action);
479  ApplyActionTestClone(game, state.get(), action);
480  game_length++;
481  num_moves++;
482 
483  if (undo && (history.size() < 10 || IsPowerOfTwo(history.size()))) {
484  TestUndo(state->Clone(), history);
485  }
486  }
487 
488  if (verbose) {
489  std::cout << "State: " << std::endl << state->ToString() << std::endl;
490  }
491  }
492 
493  state_checker_fn(*state);
494  SPIEL_CHECK_LE(game_length, game.MaxGameLength());
495 
496  if (verbose) {
497  std::cout << "Reached a terminal state!" << std::endl;
498  }
499  SPIEL_CHECK_EQ(state->CurrentPlayer(), kTerminalPlayerId);
500  std::vector<double> rewards = state->Rewards();
501  if (verbose) {
502  std::cout << "Rewards: " << absl::StrJoin(rewards, " ") << std::endl;
503  }
504 
505  history.emplace_back(state->Clone(), kTerminalPlayerId,
506  kInvalidHistoryAction);
507  TestHistoryContainsActions(game, history);
508 
509  // Check the information state of the terminal, too. This is commonly needed,
510  // for example, as a final observation in an RL environment.
511  CheckObservables(game, *state, observation.get());
512 
513  // Check that the returns satisfy the constraints based on the game type.
514  CheckReturnsSum(game, *state);
515 
516  // Now, check each individual return is within bounds.
517  auto returns = state->Returns();
518  SPIEL_CHECK_EQ(returns.size(), game.NumPlayers());
519  for (Player player = 0; player < game.NumPlayers(); player++) {
520  double final_return = returns[player];
521  SPIEL_CHECK_FLOAT_EQ(final_return, state->PlayerReturn(player));
522  SPIEL_CHECK_GE(final_return, game.MinUtility());
523  SPIEL_CHECK_LE(final_return, game.MaxUtility());
524  if (verbose) {
525  std::cout << "Final return to player " << player << " is " << final_return
526  << std::endl;
527  }
528  episode_returns[player] += rewards[player];
529  SPIEL_CHECK_TRUE(Near(episode_returns[player], final_return));
530  }
531 }

591 // Check chance outcomes in a state and all child states.
592 // We check that:
593 // - That LegalActions(kChancePlayerId) (which often defaults to the actions in
594 // ChanceOutcomes) and LegalActions() return the same result.
595 // - All the chance outcome actions are legal actions
596 // - All the chance outcome actions are different from each other.
597 // - That the probabilities are within [0, 1] and sum to 1.
598 void CheckChanceOutcomes(const State& state) {
599  if (state.IsTerminal()) return;
600  if (state.IsChanceNode()) {
601  auto legal_actions = state.LegalActions(kChancePlayerId);
602  auto default_legal_actions = state.LegalActions();
603  if (legal_actions != default_legal_actions) {
604  SpielFatalError(absl::StrCat(
605  "Legalactions() and LegalActions(kChancePlayerId) do not give the "
606  "same result:",
607  "\nLegalActions(): ",
608  absl::StrJoin(default_legal_actions, ", "),
609  "\nLegalActions(kChancePlayerId): ",
610  absl::StrJoin(legal_actions, ", ")));
611  }
612  absl::btree_set<Action> legal_action_set(legal_actions.begin(),
613  legal_actions.end());
614  auto chance_outcomes = state.ChanceOutcomes();
615 
616  std::vector<Action> chance_outcome_actions;
617  double sum = 0;
618  for (const auto& [action, prob] : chance_outcomes) {
619  chance_outcome_actions.push_back(action);
620  if (legal_action_set.count(action) == 0) {
621  SpielFatalError(absl::StrCat("LegalActions()=[",
622  absl::StrJoin(legal_actions, ", "),
623  "] inconsistent with ChanceOutcomes()=",
624  ChanceOutcomeStr(chance_outcomes), "."));
625  }
626  if (prob <= 0. || prob > 1) {
627  SpielFatalError(absl::StrCat(
628  "Invalid probability for outcome: P(", action, ")=", prob,
629  "; all outcomes=", ChanceOutcomeStr(chance_outcomes)));
630  }
631  sum += prob;
632  }
633  absl::btree_set<Action> chance_outcome_actions_set(
634  chance_outcome_actions.begin(), chance_outcome_actions.end());
635  if (chance_outcome_actions.size() != chance_outcome_actions_set.size()) {
636  std::sort(chance_outcome_actions.begin(), chance_outcome_actions.end());
637  SpielFatalError(absl::StrCat(
638  "There are some duplicate actions in ChanceOutcomes\n. There are: ",
639  chance_outcome_actions_set.size(), " unique legal actions over ",
640  chance_outcome_actions.size(),
641  " chance outcome actions.\n Sorted legal actions:\n",
642  absl::StrJoin(chance_outcome_actions, ", ")));
643  }
644  constexpr double eps = 1e-5;
645  if (sum < 1 - eps || sum > 1 + eps) {
646  SpielFatalError(
647  absl::StrCat("Invalid probabilities; sum=", sum,
648  "; all outcomes=", ChanceOutcomeStr(chance_outcomes)));
649  }
650  }
651  // Handles chance nodes, player nodes, including simultaneous nodes if
652  // supported.
653  for (auto action : state.LegalActions()) {
654  auto next_state = state.Child(action);
655  CheckChanceOutcomes(*next_state);
656  }
657 }
