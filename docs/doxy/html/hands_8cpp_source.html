295void equals_to_string(int equals, char * res)
296{
297 int p = 0;
298 int m = equals >> 2;
299 for (int i = 15; i >= 2; i--)
300 {
301 if (m & static_cast<int>(dbitMapRank[i]))
302 res[p++] = static_cast<char>(dcardRank[i]);
303 }
304 res[p] = 0;
305}

323 if (fut->score [i] != cardsScores[handno][i]) return false;

493void PrintHand(char title[],
494 unsigned int remainCards[DDS_HANDS][DDS_SUITS])
495{
496 int c, h, s, r;
497 char text[DDS_HAND_LINES][DDS_FULL_LINE];
498
499 for (int l = 0; l < DDS_HAND_LINES; l++)
500 {
501 memset(text[l], ' ', DDS_FULL_LINE);
502 text[l][DDS_FULL_LINE - 1] = '\0';
503 }
504
505 for (h = 0; h < DDS_HANDS; h++)
506 {
507 int offset, line;
508 if (h == 0)
509 {
510 offset = DDS_HAND_OFFSET;
511 line = 0;
512 }
513 else if (h == 1)
514 {
515 offset = 2 * DDS_HAND_OFFSET;
516 line = 4;
517 }
518 else if (h == 2)
519 {
520 offset = DDS_HAND_OFFSET;
521 line = 8;
522 }
523 else
524 {
525 offset = 0;
526 line = 4;
527 }
528
529 for (s = 0; s < DDS_SUITS; s++)
530 {
531 c = offset;
532 for (r = 14; r >= 2; r--)
533 {
534 if ((remainCards[h][s] >> 2) & dbitMapRank[r])
535 text[line + s][c++] = static_cast<char>(dcardRank[r]);
536 }
537
538 if (c == offset)
539 text[line + s][c++] = '-';
540
541 if (h != 3)
542 text[line + s][c] = '\0';
543 }
544 }
545 printf("%s", title);
546 char dashes[80];
547 int l = static_cast<int>(strlen(title)) - 1;
548 for (int i = 0; i < l; i++)
549 dashes[i] = '-';
550 dashes[l] = '\0';
551 printf("%s\n", dashes);
552 for (int i = 0; i < DDS_HAND_LINES; i++)
553 printf("%s\n", text[i]);
554 printf("\n\n");
555}

566int ConvertPBN(char * dealBuff,
567 unsigned int remainCards[DDS_HANDS][DDS_SUITS])
568{
569 int bp = 0, firstl, card, hand, handRelFirst, suitInHand, h, s;
570
571 for (h = 0; h < DDS_HANDS; h++)
572 for (s = 0; s < DDS_SUITS; s++)
573 remainCards[h][s] = 0;
574
575 while (((dealBuff[bp] != 'W') && (dealBuff[bp] != 'N') &&
576 (dealBuff[bp] != 'E') && (dealBuff[bp] != 'S') &&
577 (dealBuff[bp] != 'w') && (dealBuff[bp] != 'n') &&
578 (dealBuff[bp] != 'e') && (dealBuff[bp] != 's')) && (bp < 3))
579 bp++;
580
581 if (bp >= 3)
582 return 0;
583
584 if ((dealBuff[bp] == 'N') || (dealBuff[bp] == 'n'))
585 firstl = 0;
586 else if ((dealBuff[bp] == 'E') || (dealBuff[bp] == 'e'))
587 firstl = 1;
588 else if ((dealBuff[bp] == 'S') || (dealBuff[bp] == 's'))
589 firstl = 2;
590 else
591 firstl = 3;
592
593 bp++;
594 bp++;
595
596 handRelFirst = 0;
597 suitInHand = 0;
598
599 while ((bp < 80) && (dealBuff[bp] != '\0'))
600 {
601 card = IsACard(dealBuff[bp]);
602 if (card)
603 {
604 switch (firstl)
605 {
606 case 0:
607 hand = handRelFirst;
608 break;
609 case 1:
610 if (handRelFirst == 0)
611 hand = 1;
612 else if (handRelFirst == 3)
613 hand = 0;
614 else
615 hand = handRelFirst + 1;
616 break;
617 case 2:
618 if (handRelFirst == 0)
619 hand = 2;
620 else if (handRelFirst == 1)
621 hand = 3;
622 else
623 hand = handRelFirst - 2;
624 break;
625 default:
626 if (handRelFirst == 0)
627 hand = 3;
628 else
629 hand = handRelFirst - 1;
630 }
631
632 remainCards[hand][suitInHand] |=
633 static_cast<unsigned>((dbitMapRank[card] << 2));
634
635 }
636 else if (dealBuff[bp] == '.')
637 suitInHand++;
638 else if (dealBuff[bp] == ' ')
639 {
640 handRelFirst++;
641 suitInHand = 0;
642 }
643 bp++;
644 }
