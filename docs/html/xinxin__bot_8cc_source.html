30 Action GetOpenSpielAction(::hearts::card card) {
31  // xinxin keeps ranks in the reverse order of open_spiel
32  int rank = kNumCardsPerSuit - ::hearts::Deck::getrank(card) - 1;
33  Suit suit = kXinxinSuits[::hearts::Deck::getsuit(card)];
34  return Card(suit, rank);
35 }

37 ::hearts::card GetXinxinAction(Action action) {
38  int rank = kNumCardsPerSuit - CardRank(action) - 1;
39  int suit =
40  std::find(kXinxinSuits, kXinxinSuits + kNumSuits, CardSuit(action)) -
41  kXinxinSuits;
42  return ::hearts::Deck::getcard(suit, rank);
43 }

60  int iimc_num_worlds, bool use_threads)
61  : uct_num_runs_(uct_num_runs),
62  uct_c_val_(uct_c_val),
63  iimc_num_worlds_(iimc_num_worlds),
64  use_threads_(use_threads),
65  initial_state_(nullptr) {
66  pass_dir_ = ::hearts::tPassDir::kHold;
67  num_cards_dealt_ = 0;
68  game_state_ = std::make_unique<::hearts::HeartsGameState>();
69  game_state_->setRules(rules);
70  game_state_->deletePlayers();
71  for (int i = 0; i < kNumPlayers; i++) {
72  initial_deal_.push_back(std::vector<::hearts::card>());
73  // The game state destructor deletes the players, so we do not manage their
74  // memory in this bot.
75  std::unique_ptr<::hearts::SafeSimpleHeartsPlayer> player = CreatePlayer();
76  ::hearts::SafeSimpleHeartsPlayer* released_player = player.release();
77  game_state_->addPlayer(released_player);
78  released_player->setGameState(game_state_.get());
79  }
80  SPIEL_CHECK_EQ(xinxin_uct_.size(), kNumPlayers);
81  SPIEL_CHECK_EQ(xinxin_mc_.size(), kNumPlayers);
82  SPIEL_CHECK_EQ(xinxin_playouts_.size(), kNumPlayers);
83 }

138 Action XinxinBot::Step(const State& state) {
139  // check that xinxin and open_spiel agree on legal actions
140  ::hearts::Move* all_moves = game_state_->getAllMoves();
141  std::vector<Action> xinxin_actions;
142  while (all_moves != nullptr) {
143  ::hearts::card card = static_cast<::hearts::CardMove*>(all_moves)->c;
144  xinxin_actions.push_back(GetOpenSpielAction(card));
145  all_moves = all_moves->next;
146  }
147  absl::c_sort(xinxin_actions);
148  std::vector<Action> legal_actions = state.LegalActions();
149  if (legal_actions != xinxin_actions) {
150  LogStateMismatchError(state,
151  "xinxin legal actions != OpenSpiel legal actions.");
152  }
153  // test passed!
154  ::hearts::CardMove* move =
155  static_cast<::hearts::CardMove*>(game_state_->getNextPlayer()->Play());
156  game_state_->ApplyMove(move);
157  Action act = GetOpenSpielAction(move->c);
158  game_state_->freeMove(move);
159  SPIEL_CHECK_TRUE(absl::c_binary_search(legal_actions, act));
160  return act;
161 }

163 void XinxinBot::InformAction(const State& state, Player player_id,
164  Action action) {
165  if (player_id == kChancePlayerId) {
166  if (state.ChanceOutcomes().size() == 4 && num_cards_dealt_ == 0) {
167  // this is guaranteed to be the pass dir selection action as long as
168  // that remains as the first optional chance node in the Hearts
169  // implementation
170  pass_dir_ = kXinxinPassDir[action];
171  } else {
172  ::hearts::card card = GetXinxinAction(action);
173  initial_deal_[num_cards_dealt_ % kNumPlayers].push_back(card);
174  if (++num_cards_dealt_ == kNumCards) {
175  NewDeal(&initial_deal_, pass_dir_, 0);
176  }
177  }
178  } else {
179  if (state.IsTerminal()) {
180  if (!game_state_->Done()) {
181  LogStateMismatchError(state, "xinxin state is not terminal.");
182  }
183  std::vector<double> returns = state.Returns();
184  for (Player p = 0; p < returns.size(); p++) {
185  // returns in open_spiel hearts are transformed from the score
186  // to reflect that getting the least number of total points is better
187  if (returns[p] != kTotalPositivePoints - game_state_->score(p)) {
188  LogStateMismatchError(state, "xinxin score != OpenSpiel score");
189  }
190  }
191  } else {
192  ::hearts::Move* move =
193  new ::hearts::CardMove(GetXinxinAction(action), player_id);
194  game_state_->ApplyMove(move);
195  game_state_->freeMove(move);
196  }
197  }
198 }

207 int XinxinBot::XinxinRules(GameParameters params) {
208  int rules = ::hearts::kQueenPenalty;
209  if (params["pass_cards"].bool_value()) rules |= ::hearts::kDoPassCards;
210  if (params["no_pts_on_first_trick"].bool_value())
211  rules |= ::hearts::kNoHeartsFirstTrick | ::hearts::kNoQueenFirstTrick;
212  if (params["can_lead_any_club"].bool_value()) {
213  rules |= ::hearts::kLeadClubs;
214  } else {
215  rules |= ::hearts::kLead2Clubs;
216  }
217  if (params["jd_bonus"].bool_value()) rules |= ::hearts::kJackBonus;
218  if (params["avoid_all_tricks_bonus"].bool_value())
219  rules |= ::hearts::kNoTrickBonus;
220  if (params["qs_breaks_hearts"].bool_value())
221  rules |= ::hearts::kQueenBreaksHearts;
222  if (params["must_break_hearts"].bool_value())
223  rules |= ::hearts::kMustBreakHearts;
224  if (params["can_lead_hearts_instead_of_qs"].bool_value()) {
225  SpielFatalError("Xinxin does not support leading hearts instead of qs");
226  }
227  return rules;
228 }
