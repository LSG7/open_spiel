30void PlayGame(const Game& game, algorithms::ISMCTSBot* bot, std::mt19937* rng) {
31 std::unique_ptr<State> state = game.NewInitialState();
32 while (!state->IsTerminal()) {
33 std::cout << "State:" << std::endl;
34 std::cout << state->ToString() << std::endl;
35
36 Action chosen_action = kInvalidAction;
37 if (state->IsChanceNode()) {
38 chosen_action =
39 SampleAction(state->ChanceOutcomes(), absl::Uniform(*rng, 0.0, 1.0))
40 .first;
41 } else {
42 chosen_action = bot->Step(*state);
43 }
44
45 std::cout << "Chosen action: " << state->ActionToString(chosen_action)
46 << std::endl;
47 state->ApplyAction(chosen_action);
48 }
49
50 std::cout << "Terminal state:" << std::endl;
51 std::cout << state->ToString() << std::endl;
52 std::cout << "Returns: " << absl::StrJoin(state->Returns(), " ") << std::endl;
53}

55void ISMCTSTest_PlayGame(const std::string& game_name) {
56 std::shared_ptr<const Game> game = LoadGame(game_name);
57 auto evaluator =
58 std::make_shared<algorithms::RandomRolloutEvaluator>(1, kSeed);
59
60 for (algorithms::ISMCTSFinalPolicyType type :
61 {algorithms::ISMCTSFinalPolicyType::kNormalizedVisitCount,
62 algorithms::ISMCTSFinalPolicyType::kMaxVisitCount,
63 algorithms::ISMCTSFinalPolicyType::kMaxValue}) {
64 auto bot1 = std::make_unique<algorithms::ISMCTSBot>(
65 kSeed, evaluator, 5.0, 1000, algorithms::kUnlimitedNumWorldSamples,
66 type, false, false);
67
68 std::mt19937 rng(kSeed);
69
70 std::cout << "Testing " << game_name << ", bot 1" << std::endl;
71 PlayGame(*game, bot1.get(), &rng);
72
73 auto bot2 = std::make_unique<algorithms::ISMCTSBot>(
74 kSeed, evaluator, 5.0, 1000, 10, type, false, false);
75 std::cout << "Testing " << game_name << ", bot 2" << std::endl;
76 PlayGame(*game, bot2.get(), &rng);
77 }
78}
