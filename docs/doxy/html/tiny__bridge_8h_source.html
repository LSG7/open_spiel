23// A very small version of bridge, with 8 cards in total, created by Edward
24// Lockhart, inspired by a research project at University of Alberta by Michael
25// Bowling, Kate Davison, and Nathan Sturtevant. For the mechanics of the full
26// game, see https://en.wikipedia.org/wiki/Contract_bridge.
27//
28// This smaller game has two suits (hearts and spades), each with
29// four cards (Jack, Queen, King, Ace). Each of the four players gets
30// two cards each.
31//
32// The game comprises a bidding phase, in which the players bid for the
33// right to choose the trump suit (or for there not to be a trump suit), and
34// perhaps also to bid a 'slam' contract which scores bonus points.
35//
36// The play phase is not very interesting with only two tricks being played.
37// For simplicity, we replace it with a perfect-information result, which is
38// computed using minimax on a two-player perfect-information game representing
39// the play phase.
40//
41// The game comes in two varieties - the full four-player version, and a
42// simplified two-player version in which one partnership does not make
43// any bids in the auction phase.
44//
45// Scoring is as follows, for the declaring partnership:
46// +10 for making 1H/S/NT (+10 extra if overtrick)
47// +30 for making 2H/S
48// +35 for making 2NT
49// -20 per undertrick
50// Doubling (only in the 4p game) multiplies all scores by 2. Redoubling by a
51// further factor of 2.
52//
53// An abstracted version of the game is supported, where the 28 possible hands
54// are grouped into 12 buckets, using the following abstractions:
55// - When holding only one card in a suit, we consider J/Q/K equivalent
56// - We consider KQ and KJ in a single suit equivalent
57// - We consider AK and AQ in a single suit equivalent (but not AJ)
58

62inline constexpr int kNumBids = 6; // 1H, 1S, 1NT, 2H, 2S, 2NT
63inline constexpr int kNumActions2p = 1 + kNumBids; // Plus Pass
64inline constexpr int kNumActions4p = 3 + kNumBids; // Pass, Double, Redouble
65enum Call { kPass = 0, k1H, k1S, k1NT, k2H, k2S, k2NT, kDouble, kRedouble };
66inline constexpr int kNumRanks = 4;
67inline constexpr int kNumSuits = 2;
68inline constexpr int kDeckSize = kNumRanks * kNumSuits;
69inline constexpr int kNumSeats = 4;
70inline constexpr int kNumTricks = kDeckSize / kNumSeats;
71inline constexpr int kNumAbstractHands = 12;
72
73// Number of possible private states (hands) for a single player.
74inline constexpr int kNumPrivates = (kDeckSize * (kDeckSize - 1)) / 2;
75inline constexpr std::array<const char*, kNumActions4p> kActionStr{
76 "Pass", "1H", "1S", "1NT", "2H", "2S", "2NT", "Dbl", "RDbl"};
77enum Seat { kInvalidSeat = -1, kWest = 0, kNorth = 1, kEast = 2, kSouth = 3 };

79// Two-player game. Only one partnership gets to bid, so this
80// is a purely-cooperative two-player game.
81class TinyBridgeGame2p : public Game {
82 public:
83 explicit TinyBridgeGame2p(const GameParameters& params);
84 int NumDistinctActions() const override { return kNumActions2p; }
85 std::unique_ptr<State> NewInitialState() const override;
86 int NumPlayers() const override { return 2; }
87 double MinUtility() const override { return -40; } // Bid 2NT, 0 tricks
88 double MaxUtility() const override { return 35; } // Bid 2NT, 2 tricks
89 int MaxGameLength() const override { return 8; }
90 // TODO: verify whether this bound is tight and/or tighten it.
91 int MaxChanceNodesInHistory() const override { return MaxGameLength(); }
92 int MaxChanceOutcomes() const override { return kNumPrivates; }
93 std::vector<int> InformationStateTensorShape() const override {
94 return {(is_abstracted_ ? kNumAbstractHands : kDeckSize) +
95 kNumActions2p * 2};
96 }
97 std::vector<int> ObservationTensorShape() const override {
98 return {(is_abstracted_ ? kNumAbstractHands : kDeckSize) + kNumActions2p};
99 }
100
101 private:
102 const bool is_abstracted_;
103};

105// Four-player game. This is a zero-sum game of two partnerships.

110 std::unique_ptr<State> NewInitialState() const override;
111 int NumPlayers() const override { return 4; }
112 double MinUtility() const override { return -160; }
113 absl::optional<double> UtilitySum() const override { return 0; }
114 double MaxUtility() const override { return 160; }
115 int MaxGameLength() const override { return 57; }
116 // TODO: verify whether this bound is tight and/or tighten it.
117 int MaxChanceNodesInHistory() const override { return MaxGameLength(); }
118 int MaxChanceOutcomes() const override { return kNumPrivates; }
119 std::vector<int> InformationStateTensorShape() const override {
120 return {kDeckSize + (kNumBids * 3 + 1) * NumPlayers()};
121 }
122 std::vector<int> ObservationTensorShape() const override {
123 return {kDeckSize + kNumBids + 4 * NumPlayers()};
124 }
125};
