18// The full game of contract bridge.
19// See https://en.wikipedia.org/wiki/Contract_bridge
20// This is played by four players in two partnerships; it consists of a bidding
21// phase followed by a play phase. The bidding phase determines the contract for
22// the play phase. The contract has four components:
23// - Which of the four players is the 'declarer'. (The first play is made by
24// the player on declarer's left. Declarer's partner (the 'dummy') then
25// places their cards face-up for everyone to see; their plays are chosen
26// by declarer.)
27// - The trump suit (or no-trumps).
28// - The level, i.e. the trick target for the declaring partnership.
29// - Whether the contract is doubled or redoubled (increasing the stakes).
30//
31// There is then a play phase, in which 13 tricks are allocated between the
32// two partnerships. The declaring side gets a positive score if they take
33// at least as many tricks as contracted for, otherwise their score is negative.
34//
35// We support an option to replace the play phase with a perfect-information
36// solution (the 'double dummy result' in bridge jargon).
37//
38// The action space is as follows:
39// 0..51 Cards, used for both dealing (chance events) and play;
40// 52+ Calls (Pass, Dbl, RDbl, and bids), used during the auction phase.
41//
42// During the play phase, the dummy's cards are played by the declarer (their
43// partner). There will thus be 26 turns for declarer, and 13 turns for each
44// of the defenders during the play.

54inline constexpr int kBiddingActionBase = kNumCards; // First bidding action.
55inline constexpr int kNumObservationTypes = 4; // Bid, lead, declare, defend
56// Because bids always increase, any individual bid can be made at most once.
57// Thus for each bid, we only need to track (a) who bid it (if anyone), (b) who
58// doubled it (if anyone), and (c) who redoubled it (if anyone).
59// We also report the number of passes before the first bid; we could
60// equivalently report which player made the first call.
61// This is much more compact than storing the auction call-by-call, which
62// requires 318 turns * 38 possible calls per turn = 12084 bits (although
63// in practice almost all auctions have fewer than 80 calls).
64inline constexpr int kAuctionTensorSize =
65 kNumPlayers * (1 // Did this player pass before the opening bid?
66 + kNumBids // Did this player make each bid?
67 + kNumBids // Did this player double each bid?
68 + kNumBids // Did this player redouble each bid?
69 ) +
70 kNumCards // Our hand
71 + kNumVulnerabilities * kNumPartnerships; // Vulnerability of each side
72inline constexpr int kPublicInfoTensorSize =
73 kAuctionTensorSize // The auction
74 - kNumCards // But not any player's cards
75 + kNumPlayers; // Plus trailing passes
76inline constexpr int kMaxAuctionLength =
77 kNumBids * (1 + kNumPlayers * 2) + kNumPlayers;
78inline constexpr Player kFirstPlayer = 0;

219 static int GetPlayTensorSize(int num_tricks) {
220 return kNumBidLevels // What the contract is
221 + kNumDenominations // What trumps are
222 + kNumOtherCalls // Undoubled / doubled / redoubled
223 + kNumPlayers // Who declarer is
224 + kNumVulnerabilities // Vulnerability of the declaring side
225 + kNumCards // Our remaining cards
226 + kNumCards // Dummy's remaining cards
227 + num_tricks * kNumPlayers * kNumCards // Number of played tricks
228 + kNumTricks // Number of tricks we have won
229 + kNumTricks; // Number of tricks they have won
230 }
