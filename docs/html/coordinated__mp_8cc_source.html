70  std::string StringFrom(const State &observed_state, int player) const {
71  const PenniesState &state =
72  open_spiel::down_cast<const PenniesState &>(observed_state);
73  SPIEL_CHECK_GE(player, 0);
74  SPIEL_CHECK_LT(player, state.num_players_);
75 
76  std::string str;
77  if (iig_obs_type_.perfect_recall) {
78  absl::StrAppend(&str, state.MoveNumber());
79  }
80 
81  if (iig_obs_type_.perfect_recall &&
82  (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers ||
83  (player == 0 &&
84  iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer))) {
85  if (state.actionA_ == kHeads) str.push_back('H');
86  if (state.actionA_ == kTails) str.push_back('T');
87  }
88 
89  if (iig_obs_type_.private_info != PrivateInfoType::kNone) {
90  // TODO(author13)
91  // This information appears to be private information for both players,
92  // but not public information. Is this a bug?
93  if (state.infoset_ == kTop) str.push_back('T');
94  if (state.infoset_ == kBottom) str.push_back('B');
95  }
96 
97  if (iig_obs_type_.perfect_recall &&
98  (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers ||
99  (player == 1 &&
100  iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer))) {
101  if (state.actionB_ == kHeads) str.push_back('H');
102  if (state.actionB_ == kTails) str.push_back('T');
103  }
104 
105  if (iig_obs_type_.public_info &&
106  iig_obs_type_.private_info == PrivateInfoType::kNone) {
107  if (state.IsInitialState())
108  absl::StrAppend(&str, "start game");
109  else
110  absl::StrAppend(&str, "clock tick");
111  }
112 
113  return str;
114  }
115 
116  private:
117  IIGObservationType iig_obs_type_;
118 };

122 int PenniesState::CurrentPlayer() const {
123  if (actionA_ == kNoAction) {
124  // When first player acts, these should not be set yet.
125  SPIEL_CHECK_EQ(infoset_, kNoInfoset);
126  SPIEL_CHECK_EQ(actionB_, kNoAction);
127  return Player(0);
128  }
129  if (infoset_ == kNoInfoset) {
130  // When chance player acts, second player shoud have no action.
131  SPIEL_CHECK_EQ(actionB_, kNoAction);
132  return kChancePlayerId;
133  }
134  if (actionB_ == kNoAction) {
135  return Player(1);
136  }
137 
138  SPIEL_CHECK_TRUE(IsTerminal());
139  return kTerminalPlayerId;
140 }

142 void PenniesState::DoApplyAction(Action move) {
143  switch (CurrentPlayer()) {
144  case Player(0):
145  actionA_ = static_cast<ActionType>(move);
146  break;
147  case Player(1):
148  actionB_ = static_cast<ActionType>(move);
149  break;
150  case kChancePlayerId:
151  infoset_ = static_cast<InfosetPosition>(move);
152  break;
153  default:
154  SpielFatalError("Should not match");
155  }
156 }

188 bool PenniesState::IsTerminal() const {

193 std::vector<double> PenniesState::Returns() const {
194  if (!IsTerminal()) return {0., 0.};
195  const double matching = actionA_ == actionB_ ? 1. : -1.;
196  return {matching * 1., matching * -1.};
