26 def _max_entropy_symmetric_nash(p_mat, eps=1e-9):
27  """Solves for the maxent symmetric nash for symmetric 2P zero-sum games.
28 
29  Using convex programming:
30  min p^Tlog(p)
31  s.t.
32  p_mat.dot(p) <= 0, since game value must be 0
33  p >= 0
34  1^T * p = 1
35 
36  Args:
37  p_mat: an N*N anti-symmetric payoff matrix for the row player
38  eps: minimum probability threshold
39 
40  Returns:
41  p*: a maxent symmetric nash
42  """
43  assert np.array_equal(p_mat, -p_mat.T) and eps >= 0 and eps <= 0.5
44  n = len(p_mat)
45  x = cp.Variable(shape=n)
46  obj = cp.Maximize(cp.sum(cp.entr(x)))
47  constraints = [p_mat @ x <= 0, x >= eps * np.ones(n)]
48  constraints.append(cp.sum(x) == 1)
49  prob = cp.Problem(obj, constraints)
50  prob.solve()
51  return x.value.reshape((-1, 1))

54 def _max_entropy_symmetric_nash_avt(p_mat, num_agents, num_tasks, eps=1e-9):
55  """Solves for the maxent symmetric nash for symmetric 2P zero-sum games.
56 
57  This covers the agent-vs-task cases.
58 
59  Using convex programming:
60  min x^Tlog(x) + y^Tlog(y)
61  s.t.
62  x >= 0
63  1^T * x = 1
64  y >= 0
65  1^T * y = 1
66  forall s, such that s has exactly one unit mass on an agent strategy
67  and one unit mass on a task strategy,
68  s^T*p_mat*z <= 0, where z = [x, y], since game-value is 0.
69 
70  Args:
71  p_mat: an N*N anti-symmetric payoff matrix for the row player
72  num_agents: number of agents
73  num_tasks: number of tasks
74  eps: minimum probability threshold
75 
76  Returns:
77  (x*, y*): a maxent symmetric nash
78  """
79  assert np.array_equal(p_mat, -p_mat.T) and eps >= 0 and eps <= 0.5
80  n = len(p_mat)
81  assert n == num_agents + num_tasks
82  x = cp.Variable(shape=num_agents)
83  y = cp.Variable(shape=num_tasks)
84  z = cp.hstack([x, y])
85  obj = cp.Maximize(cp.sum(cp.entr(z)))
86  constraints = [
87  x >= eps * np.ones(num_agents),
88  cp.sum(x) == 1,
89  y >= eps * np.ones(num_tasks),
90  cp.sum(y) == 1,
91  ]
92 
93  dev_payoffs = p_mat @ z
94  for a_idx in range(num_agents):
95  for t_idx in range(num_tasks):
96  pure_strategy = np.zeros(n)
97  pure_strategy[a_idx] = 1
98  pure_strategy[num_agents + t_idx] = 1
99  pure_strategy = pure_strategy.reshape((1, -1))
100  constraints.append(pure_strategy @ dev_payoffs <= 0)
101 
102  prob = cp.Problem(obj, constraints)
103  prob.solve()
104  return x.value.reshape((-1, 1)), y.value.reshape((-1, 1))

107 def nash_averaging_avt_matrix(s_mat, eps=0.0):
108  """Apply the agent-vs-task Nash Averaging from Appendix D, from a matrix.
109 
110  Args:
111  s_mat: The S matrix from the paper, representing m rows (agents) and n
112  columns (tasks), with scores for the agent on the task. Note that the
113  values need not be normalized, but will be normalized across tasks before
114  being processed.
115  eps: minimum probability threshold.
116 
117  Returns:
118  maxent_nash: nash mixture for row player and column player
119  nash_avg_score: the expected payoff under maxent_nash
120  """
121  m, n = s_mat.shape
122  min_payoffs = np.min(s_mat, axis=0)
123  max_payoffs = np.max(s_mat, axis=0)
124  std_p_mat = (s_mat - min_payoffs) / (max_payoffs - min_payoffs)
125  a_mat = np.block([
126  [np.zeros(shape=(m, m)), std_p_mat],
127  [-std_p_mat.T, np.zeros(shape=(n, n))],
128  ])
129  pa_sol, pe_sol = _max_entropy_symmetric_nash_avt(
130  a_mat, num_agents=m, num_tasks=n, eps=eps)
131  pa, pe = np.asarray(pa_sol), np.asarray(pe_sol)
132  return (pa, pe), (std_p_mat.dot(pe), -std_p_mat.T.dot(pa))

135 def nash_averaging(game, eps=0.0, a_v_a=True):
136  """Nash averaging, see https://arxiv.org/abs/1806.02643.
137 
138  Args:
139  game: a pyspiel game
140  eps: minimum probability mass for maxent nash
141  a_v_a: whether it is Agent-vs-Agent or Agent-vs-Task
142 
143  Returns:
144  maxent_nash: nash mixture for row player and column player
145  nash_avg_score: the expected payoff under maxent_nash
146  """
147 
148  p_mat = game_payoffs_array(game)
149  if len(p_mat) != 2:
150  raise ValueError("Nash Averaging works only for two players.")
151  if np.max(np.abs(p_mat[0] + p_mat[1])) > 0:
152  raise ValueError("Must be zero-sum")
153  if a_v_a:
154  if not np.array_equal(p_mat[0], -p_mat[0].T):
155  raise ValueError(
156  "AvA only works for symmetric two-player zero-sum games.")
157  maxent_nash = np.array(_max_entropy_symmetric_nash(p_mat[0], eps=eps))
158  return maxent_nash, p_mat[0].dot(maxent_nash)

160  # For AvT, see appendix D of the paper.
161  # Here assumes the row player represents agents and the column player
162  # represents tasks.
163  # game does not have to be symmetric
164  return nash_averaging_avt_matrix(p_mat[0], eps=eps)
