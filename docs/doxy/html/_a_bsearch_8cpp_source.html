62bool ABsearch(
63 pos * posPoint,
64 const int target,
65 const int depth,
66 ThreadData * thrp)
67{
68 /* posPoint points to the current look-ahead position,
69 target is number of tricks to take for the player,
70 depth is the remaining search length, must be positive,
71 the value of the subtree is returned.
72 This is a specialized AB function for handRelFirst == 0. */
73
74 int hand = posPoint->first[depth];
75 int tricks = depth >> 2;
76 bool success = (thrp->nodeTypeStore[hand] == MAXNODE ? true : false);
77 bool value = ! success;
78
79#ifdef DDS_TOP_LEVEL
80 thrp->nodes++;
81#endif
82
83 TIMER_START(TIMER_NO_MOVEGEN, depth);
84 for (int ss = 0; ss < DDS_SUITS; ss++)
85 thrp->lowestWin[depth][ss] = 0;
86
87 thrp->moves.MoveGen0(
88 tricks,
89 * posPoint,
90 thrp->bestMove[depth],
91 thrp->bestMoveTT[depth],
92 thrp->rel);
93 thrp->moves.Purge(tricks, 0, thrp->forbiddenMoves);
94
95 TIMER_END(TIMER_NO_MOVEGEN, depth);
96
97 for (int ss = 0; ss < DDS_SUITS; ss++)
98 posPoint->winRanks[depth][ss] = 0;
99
100 while (1)
101 {
102 TIMER_START(TIMER_NO_MAKE, depth);
103 moveType const * mply = thrp->moves.MakeNext(tricks, 0,
104 posPoint->winRanks[depth]);
105#ifdef DDS_AB_STATS
106 thrp->ABStats.IncrNode(depth);
107#endif
108 TIMER_END(TIMER_NO_MAKE, depth);
109
110 if (mply == NULL)
111 break;
112
113 Make0(posPoint, depth, mply);
114
115 TIMER_START(TIMER_NO_AB, depth - 1);
116 value = ABsearch1(posPoint, target, depth - 1, thrp);
117 TIMER_END(TIMER_NO_AB, depth - 1);
118
119 TIMER_START(TIMER_NO_UNDO, depth);
120 Undo1(posPoint, depth, * mply);
121 TIMER_END(TIMER_NO_UNDO, depth);
122
123 if (value == success) /* A cut-off? */
124 {
125 for (int ss = 0; ss < DDS_SUITS; ss++)
126 posPoint->winRanks[depth][ss] =
127 posPoint->winRanks[depth - 1][ss];
128
129 thrp->bestMove[depth] = * mply;
130#ifdef DDS_MOVES
131 thrp->moves.RegisterHit(tricks, 0);
132#endif
133 goto ABexit;
134 }
135 for (int ss = 0; ss < DDS_SUITS; ss++)
136 posPoint->winRanks[depth][ss] |=
137 posPoint->winRanks[depth - 1][ss];
138
139 TIMER_START(TIMER_NO_NEXTMOVE, depth);
140 TIMER_END(TIMER_NO_NEXTMOVE, depth);
141 }
142
143ABexit:
144
145 AB_COUNT(AB_MOVE_LOOP, value, depth);
146#ifdef DDS_AB_STATS
147 thrp->ABStats.PrintStats(thrp->fileABstats.GetStream());
148#endif
149
150 return value;
151}

171 thrp->nodes++;

174 for (int ss = 0; ss < DDS_SUITS; ss++)
175 posPoint->winRanks[depth][ss] = 0;
176
177 if (depth >= 20)
178 {
179 /* Find node that fits the suit lengths */
180 int limit;
181 if (thrp->nodeTypeStore[0] == MAXNODE)
182 limit = target - posPoint->tricksMAX - 1;
183 else
184 limit = tricks - (target - posPoint->tricksMAX - 1);
185
186 bool lowerFlag;
187 TIMER_START(TIMER_NO_LOOKUP, depth);

202 posPoint->winRanks[depth][ss] =
203 winRanks[ posPoint->aggr[ss] ]

209 thrp->bestMoveTT[depth].rank = cardsP->bestMoveRank;

226 {
227 AB_COUNT(AB_TARGET_REACHED, false, depth);
228 return false;
229 }

234 TIMER_END(TIMER_NO_EVALUATE, depth);

242 return value;
243 }

247 int qtricks = QuickTricks(* posPoint, hand, depth, target,
248 trump, res, * thrp);
249 TIMER_END(TIMER_NO_QT, depth);
250
251 if (thrp->nodeTypeStore[hand] == MAXNODE)
252 {
253 if (res)
254 {

257 }
258
259 TIMER_START(TIMER_NO_LT, depth);
260 res = LaterTricksMIN(* posPoint, hand, depth, target, trump, * thrp);
261 TIMER_END(TIMER_NO_LT, depth);
262
263 if (! res)
264 {
265 // Is 1 right here?!
266 AB_COUNT(AB_LATERTRICKS, true, depth);
267 return false;
268 }
269 }
270 else
271 {
272 if (res)
273 {
274 AB_COUNT(AB_QUICKTRICKS, false, depth);
275 return (qtricks == 0 ? true : false);
276 }
277
278 TIMER_START(TIMER_NO_LT, depth);
279 res = LaterTricksMAX(* posPoint, hand, depth, target, trump, * thrp);
280 TIMER_END(TIMER_NO_LT, depth);
281
282 if (res)
283 {
284 AB_COUNT(AB_LATERTRICKS, false, depth);
285 return true;
286 }
287 }
288
289 if (depth < 20)
290 {
291 /* Find node that fits the suit lengths */
292 int limit;
293 if (thrp->nodeTypeStore[0] == MAXNODE)
294 limit = target - posPoint->tricksMAX - 1;
295 else
296 limit = tricks - (target - posPoint->tricksMAX - 1);
297

300 nodeCardsType const * cardsP =
301 thrp->transTable->Lookup(
302 tricks, hand, posPoint->aggr, posPoint->handDist,
303 limit, lowerFlag);
304 TIMER_END(TIMER_NO_LOOKUP, depth);
305
306 if (cardsP)
307 {
308#ifdef DDS_AB_HITS
309 DumpRetrieved(thrp->fileRetrieved.GetStream(),
310 * posPoint, * cardsP, target, depth);
311#endif
312
313 for (int ss = 0; ss < DDS_SUITS; ss++)
314 posPoint->winRanks[depth][ss] =
315 winRanks[ posPoint->aggr[ss] ]
316 [ static_cast<int>(cardsP->leastWin[ss]) ];
317
318 if (cardsP->bestMoveRank != 0)
319 {
320 thrp->bestMoveTT[depth].suit = cardsP->bestMoveSuit;
321 thrp->bestMoveTT[depth].rank = cardsP->bestMoveRank;
322 }
323
324 bool scoreFlag =
325 (thrp->nodeTypeStore[0] == MAXNODE ? lowerFlag : ! lowerFlag);
326
327 AB_COUNT(AB_MAIN_LOOKUP, scoreFlag, depth);
328 return scoreFlag;
329 }
330 }
331
332 bool success = (thrp->nodeTypeStore[hand] == MAXNODE ? true : false);
333 bool value = ! success;
334
335 TIMER_START(TIMER_NO_MOVEGEN, depth);
336 for (int ss = 0; ss < DDS_SUITS; ss++)
337 thrp->lowestWin[depth][ss] = 0;
338
339 thrp->moves.MoveGen0(
340 tricks,
341 * posPoint,
342 thrp->bestMove[depth],
343 thrp->bestMoveTT[depth],
344 thrp->rel);
345
346 TIMER_END(TIMER_NO_MOVEGEN, depth);
347
348 for (int ss = 0; ss < DDS_SUITS; ss++)
349 posPoint->winRanks[depth][ss] = 0;
350

352 {
353 TIMER_START(TIMER_NO_MAKE, depth);
354 moveType const * mply = thrp->moves.MakeNext(tricks, 0,
355 posPoint->winRanks[depth]);
356#ifdef DDS_AB_STATS

365
366 TIMER_START(TIMER_NO_AB, depth - 1);
367 value = ABsearch1(posPoint, target, depth - 1, thrp);
368 TIMER_END(TIMER_NO_AB, depth - 1);
369
370 TIMER_START(TIMER_NO_UNDO, depth);
371 Undo1(posPoint, depth, * mply);
372 TIMER_END(TIMER_NO_UNDO, depth);
373
374 if (value == success) /* A cut-off? */
375 {
376 for (int ss = 0; ss < DDS_SUITS; ss++)
377 posPoint->winRanks[depth][ss] =
378 posPoint->winRanks[depth - 1][ss];
379

381#ifdef DDS_MOVES
382 thrp->moves.RegisterHit(tricks, 0);
383#endif

385 }
386 for (int ss = 0; ss < DDS_SUITS; ss++)
387 posPoint->winRanks[depth][ss] |=
388 posPoint->winRanks[depth - 1][ss];
389
390 TIMER_START(TIMER_NO_NEXTMOVE, depth);
391 TIMER_END(TIMER_NO_NEXTMOVE, depth);
392 }
393
394ABexit:
395 nodeCardsType first;
396 if (value)
397 {
398 if (thrp->nodeTypeStore[0] == MAXNODE)
399 {
400 first.ubound = static_cast<char>(tricks + 1);
401 first.lbound = static_cast<char>(target - posPoint->tricksMAX);
402 }
403 else
404 {
405 first.ubound = static_cast<char>
406 (tricks + 1 - target + posPoint->tricksMAX);
407 first.lbound = 0;
408 }
409 }
410 else
411 {
412 if (thrp->nodeTypeStore[0] == MAXNODE)
413 {
414 first.ubound = static_cast<char>
415 (target - posPoint->tricksMAX - 1);
416 first.lbound = 0;
417 }
418 else
419 {
420 first.ubound = static_cast<char>(tricks + 1);
421 first.lbound = static_cast<char>
422 (tricks + 1 - target + posPoint->tricksMAX + 1);
423 }
424 }
425
426 first.bestMoveSuit = static_cast<char>(thrp->bestMove[depth].suit);
427 first.bestMoveRank = static_cast<char>(thrp->bestMove[depth].rank);
428
429 bool flag =
430 ((thrp->nodeTypeStore[hand] == MAXNODE && value) ||
431 (thrp->nodeTypeStore[hand] == MINNODE && !value))
432 ? true : false;
433
434 TIMER_START(TIMER_NO_BUILD, depth);
435 thrp->transTable->Add(
436 tricks,
437 hand,
438 posPoint->aggr,
439 posPoint->winRanks[depth],
440 first,
441 flag);
442 TIMER_END(TIMER_NO_BUILD, depth);
443
444#ifdef DDS_AB_HITS
445 DumpStored(thrp->fileStored.GetStream(),
446 * posPoint, thrp->moves, first, target, depth);
447#endif
448
449 AB_COUNT(AB_MOVE_LOOP, value, depth);
450 return value;
451}

525#endif
526 goto ABexit;
527 }

530 posPoint->winRanks[depth][ss] |=
531 posPoint->winRanks[depth - 1][ss];
532
533 TIMER_START(TIMER_NO_NEXTMOVE, depth);
534 TIMER_END(TIMER_NO_NEXTMOVE, depth);
535 }

537ABexit:
538 AB_COUNT(AB_MOVE_LOOP, value, depth);
539 return value;

623bool ABsearch3(
624 pos * posPoint,
625 const int target,
626 const int depth,
627 ThreadData * thrp)
628{
629 /* This is a specialized AB function for handRelFirst == 3. */
630
631 unsigned short int makeWinRank[DDS_SUITS];
632
633 int hand = handId(posPoint->first[depth], 3);
634 bool success = (thrp->nodeTypeStore[hand] == MAXNODE ? true : false);
635 bool value = ! success;
636
637#ifdef DDS_TOP_LEVEL
638 thrp->nodes++;
639#endif
640
641 TIMER_START(TIMER_NO_MOVEGEN, depth);
642 for (int ss = 0; ss < DDS_SUITS; ss++)
643 thrp->lowestWin[depth][ss] = 0;
644 int tricks = (depth + 3) >> 2;
645
646 thrp->moves.MoveGen123(tricks, 3, * posPoint);
647 if (depth == thrp->iniDepth)
648 thrp->moves.Purge(tricks, 3, thrp->forbiddenMoves);
649
650 TIMER_END(TIMER_NO_MOVEGEN, depth);
651
652 for (int ss = 0; ss < DDS_SUITS; ss++)
653 posPoint->winRanks[depth][ss] = 0;
654
655 while (1)
656 {
657 TIMER_START(TIMER_NO_MAKE, depth);
658 moveType const * mply = thrp->moves.MakeNext(tricks, 3,
659 posPoint->winRanks[depth]);
660#ifdef DDS_AB_STATS
661 thrp->ABStats.IncrNode(depth);
662#endif
663 TIMER_END(TIMER_NO_MAKE, depth);
664
665 if (mply == NULL)
666 break;
667
668 Make3(posPoint, makeWinRank, depth, mply, thrp);
669
670 thrp->trickNodes++; // As handRelFirst == 0
671
672 if (thrp->nodeTypeStore[posPoint->first[depth - 1]] == MAXNODE)
673 posPoint->tricksMAX++;
674
675 TIMER_START(TIMER_NO_AB, depth - 1);
676 value = ABsearch0(posPoint, target, depth - 1, thrp);
677 TIMER_END(TIMER_NO_AB, depth - 1);
678
679 TIMER_START(TIMER_NO_UNDO, depth);
680 Undo0(posPoint, depth, * mply, thrp);
681
682 if (thrp->nodeTypeStore[posPoint->first[depth - 1]] == MAXNODE)
683 posPoint->tricksMAX--;
684
685 TIMER_END(TIMER_NO_UNDO, depth);
686
687 if (value == success) /* A cut-off? */
688 {
689 for (int ss = 0; ss < DDS_SUITS; ss++)
690 posPoint->winRanks[depth][ss] = static_cast<unsigned short>(
691 posPoint->winRanks[depth - 1][ss] | makeWinRank[ss]);
692
693 thrp->bestMove[depth] = * mply;
694#ifdef DDS_MOVES
695 thrp->moves.RegisterHit(tricks, 3);
696#endif
697 goto ABexit;
698 }
699 for (int ss = 0; ss < DDS_SUITS; ss++)
700 posPoint->winRanks[depth][ss] |=
701 posPoint->winRanks[depth - 1][ss] | makeWinRank[ss];
702
703 TIMER_START(TIMER_NO_NEXTMOVE, depth);
704 TIMER_END(TIMER_NO_NEXTMOVE, depth);
705 }
706
707ABexit:
708 AB_COUNT(AB_MOVE_LOOP, value, depth);
709 return value;
710}

713void Make0(
714 pos * posPoint,
715 const int depth,
716 moveType const * mply)
717{
718 /* First hand is not changed in next move */
719 int h = posPoint->first[depth];
720 int s = mply->suit;
721 int r = mply->rank;
722
723 posPoint->first[depth - 1] = h;
724 posPoint->move[depth] = * mply;
725
726 posPoint->rankInSuit[h][s] &= (~bitMapRank[r]);
727 posPoint->aggr[s] ^= bitMapRank[r];
728 posPoint->handDist[h] -= handDelta[s];
729 posPoint->length[h][s]--;
730}

774 pos * posPoint,
775 unsigned short trickCards[DDS_SUITS],
776 const int depth,
777 moveType const * mply,
778 ThreadData * thrp)
779{
780 int firstHand = posPoint->first[depth];
781
782 const trickDataType& data = thrp->moves.GetTrickData((depth + 3) >> 2);
783
784 posPoint->first[depth - 1] = handId(firstHand, data.relWinner);
785 /* Defines who is first in the next move */
786
787 int h = handId(firstHand, 3);
788 /* Hand pointed to by posPoint->first will lead the next trick */
789
790 for (int suit = 0; suit < DDS_SUITS; suit++)
791 trickCards[suit] = 0;
792
793 int ss = data.bestSuit;
794 if (data.playCount[ss] >= 2)
795 {
796 // Win by rank when some else played that suit, too.
797 int rr = data.bestRank;
798 trickCards[ss] = static_cast<unsigned short>
799 (bitMapRank[rr] | data.bestSequence);
800 }
801
802 int r = mply->rank;
803 int s = mply->suit;
804 posPoint->rankInSuit[h][s] &= (~bitMapRank[r]);
805 posPoint->aggr[s] ^= bitMapRank[r];
806 posPoint->handDist[h] -= handDelta[s];
807 posPoint->length[h][s]--;
808
809 // Changes that we may have to undo.
810 WinnersType * wp = &thrp->winners[ (depth + 3) >> 2];
811 wp->number = 0;
812
813 for (int st = 0; st < 4; st++)
814 {
815 if (data.playCount[st])
816 {
817 int n = wp->number;
818 wp->winner[n].suit = st;
819 wp->winner[n].winnerRank = posPoint->winner[st].rank;
820 wp->winner[n].winnerHand = posPoint->winner[st].hand;
821 wp->winner[n].secondRank = posPoint->secondBest[st].rank;
822 wp->winner[n].secondHand = posPoint->secondBest[st].hand;
823 wp->number++;
824
825 int aggr = posPoint->aggr[st];
826
827 posPoint->winner[st].rank = thrp->rel[aggr].absRank[1][st].rank;
828 posPoint->winner[st].hand = thrp->rel[aggr].absRank[1][st].hand;
829 posPoint->secondBest[st].rank = thrp->rel[aggr].absRank[2][st].rank;
830 posPoint->secondBest[st].hand = thrp->rel[aggr].absRank[2][st].hand;
831
832 }
833 }

963evalType Evaluate(
964 pos const * posPoint,
965 const int trump,
966 ThreadData const * thrp)
967{
968 int s, h, hmax = 0, count = 0, k = 0;
969 unsigned short rmax = 0;
970 evalType eval;
971
972 int firstHand = posPoint->first[0];
973 assert((firstHand >= 0) && (firstHand <= 3));
974
975 for (s = 0; s < DDS_SUITS; s++)
976 eval.winRanks[s] = 0;
977
978 /* Who wins the last trick? */
979 if (trump != DDS_NOTRUMP) /* Highest trump card wins */
980 {
981 for (h = 0; h < DDS_HANDS; h++)
982 {
983 if (posPoint->rankInSuit[h][trump] != 0)
984 count++;
985 if (posPoint->rankInSuit[h][trump] > rmax)
986 {
987 hmax = h;
988 rmax = posPoint->rankInSuit[h][trump];
989 }
990 }
991
992 if (rmax > 0) /* Trumpcard wins */
993 {
994 if (count >= 2)
995 eval.winRanks[trump] = rmax;
996
997 if (thrp->nodeTypeStore[hmax] == MAXNODE)
998 goto maxexit;
999 else
1000 goto minexit;
1001 }
1002 }
1003
1004 /* Who has the highest card in the suit played by 1st hand? */
1005
1006 k = 0;
1007 while (k <= 3) /* Find the card the 1st hand played */
1008 {
1009 if (posPoint->rankInSuit[firstHand][k] != 0) /* Is this the card? */
1010 break;
1011 k++;
1012 }
1013
1014 assert(k < 4);
1015
1016 for (h = 0; h < DDS_HANDS; h++)
1017 {
1018 if (posPoint->rankInSuit[h][k] != 0)
1019 count++;
1020 if (posPoint->rankInSuit[h][k] > rmax)
1021 {
1022 hmax = h;
1023 rmax = posPoint->rankInSuit[h][k];
1024 }
1025 }
1026
1027 if (count >= 2)
1028 eval.winRanks[k] = rmax;
1029
1030 if (thrp->nodeTypeStore[hmax] == MAXNODE)
1031 goto maxexit;
1032 else
1033 goto minexit;
1034
1035maxexit:
1036 eval.tricks = posPoint->tricksMAX + 1;
1037 return eval;
1038
1039minexit:
1040 eval.tricks = posPoint->tricksMAX;
1041 return eval;
1042}
