61 // Do not use std::get here, as it causes compilation problems on on older
62 // MacOS. For details, see:
63 // https://stackoverflow.com/questions/52521388/stdvariantget-does-not-compile-with-apple-llvm-10-0
64 template <class T> T get_val() const {
65 if (auto *val = std::get_if<T>(this)) {
66 return *val;
67 } else {
68 SpielFatalError(absl::StrCat(
69 "Value does not contain the specified type: ", typeid(T).name()));
70 }
71 }

73 template <class T> T &get_ref() {
74 if (auto *val = std::get_if<T>(this)) {
75 return *val;
76 } else {
77 SpielFatalError(absl::StrCat(
78 "Value does not contain the specified type: ", typeid(T).name()));
79 }

82 template <class T> const T &get_const_ref() const {
83 if (auto *val = std::get_if<T>(this)) {

124// Accept a std::vector of any of the types that are constructible to Value.
125// For example accept std::vector<int>.
126template <typename T>
127Array CastToArray(std::vector<T> vec) {
128 Array out;
129 out.reserve(vec.size());
130 for (const T& val : vec) {
131 out.emplace_back(val);
132 }
133 return out;
134}
135
136// Accept a std::vector of any type with a fn that converts to any type
137// constructible to Value.
138template <typename T, typename Fn>
139Array TransformToArray(std::vector<T> vec, Fn fn) {
140 Array out;
141 out.reserve(vec.size());
142 for (const T& val : vec) {
143 out.emplace_back(fn(val));
144 }
145 return out;
146}
