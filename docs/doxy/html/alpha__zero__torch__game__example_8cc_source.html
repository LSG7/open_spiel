63std::unique_ptr<open_spiel::Bot>
64InitBot(std::string type, const open_spiel::Game &game,
65 open_spiel::Player player,
66 std::shared_ptr<open_spiel::algorithms::Evaluator> evaluator,
67 std::shared_ptr<open_spiel::algorithms::torch_az::VPNetEvaluator>
68 az_evaluator) {
69 if (type == "az") {
70 return std::make_unique<open_spiel::algorithms::MCTSBot>(
71 game, std::move(az_evaluator), absl::GetFlag(FLAGS_uct_c),
72 absl::GetFlag(FLAGS_max_simulations),
73 absl::GetFlag(FLAGS_max_memory_mb), absl::GetFlag(FLAGS_solve), Seed(),
74 absl::GetFlag(FLAGS_verbose),
75 open_spiel::algorithms::ChildSelectionPolicy::PUCT, 0, 0,
76 /*dont_return_chance_node=*/true);
77 }
78 if (type == "human") {
79 return std::make_unique<open_spiel::HumanBot>();
80 }
81 if (type == "mcts") {
82 return std::make_unique<open_spiel::algorithms::MCTSBot>(
83 game, std::move(evaluator), absl::GetFlag(FLAGS_uct_c),
84 absl::GetFlag(FLAGS_max_simulations),
85 absl::GetFlag(FLAGS_max_memory_mb), absl::GetFlag(FLAGS_solve), Seed(),
86 absl::GetFlag(FLAGS_verbose));
87 }
88 if (type == "random") {
89 return open_spiel::MakeUniformRandomBot(player, Seed());
90 }
91
92 open_spiel::SpielFatalError(
93 "Bad player type. Known types: az, human, mcts, random");
94}

96open_spiel::Action GetAction(const open_spiel::State &state,
97 std::string action_str) {
98 for (open_spiel::Action action : state.LegalActions()) {
99 if (action_str == state.ActionToString(state.CurrentPlayer(), action))
100 return action;
101 }
102 return open_spiel::kInvalidAction;
103}

105std::pair<std::vector<double>, std::vector<std::string>>
106PlayGame(const open_spiel::Game &game,
107 std::vector<std::unique_ptr<open_spiel::Bot>> &bots, std::mt19937 &rng,
108 const std::vector<std::string> &initial_actions) {
109 bool quiet = absl::GetFlag(FLAGS_quiet);
110 std::unique_ptr<open_spiel::State> state = game.NewInitialState();
111 std::vector<std::string> history;
112
113 if (!quiet)
114 std::cerr << "Initial state:\n" << state << std::endl;
115
116 // Play the initial actions (if there are any).
117 for (const auto &action_str : initial_actions) {
118 open_spiel::Player current_player = state->CurrentPlayer();
119 open_spiel::Action action = GetAction(*state, action_str);
120
121 if (action == open_spiel::kInvalidAction)
122 open_spiel::SpielFatalError(absl::StrCat("Invalid action: ", action_str));
123
124 history.push_back(action_str);
125 state->ApplyAction(action);
126
127 if (!quiet) {
128 std::cerr << "Player " << current_player
129 << " forced action: " << action_str << std::endl;
130 std::cerr << "Next state:\n" << state->ToString() << std::endl;
131 }
132 }
133
134 while (!state->IsTerminal()) {
135 open_spiel::Player player = state->CurrentPlayer();
136
137 open_spiel::Action action;
138 if (state->IsChanceNode()) {
139 // Chance node; sample one according to underlying distribution.
140 open_spiel::ActionsAndProbs outcomes = state->ChanceOutcomes();
141 action = open_spiel::SampleAction(outcomes, rng).first;
142 } else {
143 // The state must be a decision node, ask the right bot to make its
144 // action.
145 action = bots[player]->Step(*state);
146 }
147 if (!quiet)
148 std::cerr << "Player " << player
149 << " chose action: " << state->ActionToString(player, action)
150 << std::endl;
151
152 // Inform the other bot of the action performed.
153 for (open_spiel::Player p = 0; p < bots.size(); ++p) {
154 if (p != player) {
155 bots[p]->InformAction(*state, player, action);
156 }
157 }
158
159 // Update history and get the next state.
160 history.push_back(state->ActionToString(player, action));
161 state->ApplyAction(action);
162
163 if (!quiet)
164 std::cerr << "Next state:\n" << state->ToString() << std::endl;
165 }
166
167 std::cerr << "Returns: " << absl::StrJoin(state->Returns(), ", ")
168 << std::endl;
169 std::cerr << "Game actions: " << absl::StrJoin(history, ", ") << std::endl;
170
171 return {state->Returns(), history};
172}

174int main(int argc, char **argv) {
175 std::vector<char *> positional_args = absl::ParseCommandLine(argc, argv);
176 std::mt19937 rng(Seed()); // Random number generator.
177
178 // Create the game.
179 std::string game_name = absl::GetFlag(FLAGS_game);
180 std::cerr << "Game: " << game_name << std::endl;
181 std::shared_ptr<const open_spiel::Game> game =
182 open_spiel::LoadGame(game_name);
183
184 // Ensure the game is AlphaZero-compatible and arguments are compatible.
185 open_spiel::GameType game_type = game->GetType();
186 if (game->NumPlayers() != 2)
187 open_spiel::SpielFatalError("AlphaZero can only handle 2-player games.");
188 if (game_type.reward_model != open_spiel::GameType::RewardModel::kTerminal)
189 open_spiel::SpielFatalError("Game must have terminal rewards.");
190 if (game_type.dynamics != open_spiel::GameType::Dynamics::kSequential)
191 open_spiel::SpielFatalError("Game must have sequential turns.");
192 if (absl::GetFlag(FLAGS_az_path).empty())
193 open_spiel::SpielFatalError("AlphaZero path must be specified.");
194 if (absl::GetFlag(FLAGS_player1) != "az" &&
195 absl::GetFlag(FLAGS_player2) != "az")
196 open_spiel::SpielFatalError("One of the players must be AlphaZero.");
197
198 open_spiel::algorithms::torch_az::DeviceManager device_manager;
199 device_manager.AddDevice(open_spiel::algorithms::torch_az::VPNetModel(
200 *game, absl::GetFlag(FLAGS_az_path), absl::GetFlag(FLAGS_az_graph_def),
201 "/cpu:0"));
202 device_manager.Get(0, 0)->LoadCheckpoint(absl::GetFlag(FLAGS_az_checkpoint));
203 auto az_evaluator =
204 std::make_shared<open_spiel::algorithms::torch_az::VPNetEvaluator>(
205 /*device_manager=*/&device_manager,
206 /*batch_size=*/absl::GetFlag(FLAGS_az_batch_size),
207 /*threads=*/absl::GetFlag(FLAGS_az_threads),
208 /*cache_size=*/absl::GetFlag(FLAGS_az_cache_size),
209 /*cache_shards=*/absl::GetFlag(FLAGS_az_cache_shards));
210 auto evaluator =
211 std::make_shared<open_spiel::algorithms::RandomRolloutEvaluator>(
212 absl::GetFlag(FLAGS_rollout_count), Seed());
213
214 std::vector<std::unique_ptr<open_spiel::Bot>> bots;
215 bots.push_back(
216 InitBot(absl::GetFlag(FLAGS_player1), *game, 0, evaluator, az_evaluator));
217 bots.push_back(
218 InitBot(absl::GetFlag(FLAGS_player2), *game, 1, evaluator, az_evaluator));
219
220 std::vector<std::string> initial_actions;
221 for (int i = 1; i < positional_args.size(); ++i) {
222 initial_actions.push_back(positional_args[i]);
223 }
224
225 std::map<std::string, int> histories;
226 std::vector<double> overall_returns(2, 0);
227 std::vector<int> overall_wins(2, 0);
228 int num_games = absl::GetFlag(FLAGS_num_games);
229 for (int game_num = 0; game_num < num_games; ++game_num) {
230 auto [returns, history] = PlayGame(*game, bots, rng, initial_actions);
231 histories[absl::StrJoin(history, " ")] += 1;
232 for (int i = 0; i < returns.size(); ++i) {
233 double v = returns[i];
234 overall_returns[i] += v;
235 if (v > 0) {
236 overall_wins[i] += 1;
237 }
238 }
239 }
240
241 std::cerr << "Number of games played: " << num_games << std::endl;
242 std::cerr << "Number of distinct games played: " << histories.size()
243 << std::endl;
244 std::cerr << "Players: " << absl::GetFlag(FLAGS_player1) << ", "
245 << absl::GetFlag(FLAGS_player2) << std::endl;
246 std::cerr << "Overall wins: " << absl::StrJoin(overall_wins, ", ")
247 << std::endl;
248 std::cerr << "Overall returns: " << absl::StrJoin(overall_returns, ", ")
249 << std::endl;
250
251 return 0;
252}
