55int GetPlayerHomePit(Player player) {
56 if (player == 0) {
57 return kTotalPits / 2;
58 }
59 return 0;
60}

62bool IsPlayerPit(Player player, int pit) {
63 if (player == 0) {
64 if (pit < kTotalPits / 2 && pit > 0) return true;
65 return false;
66 }
67 if (pit > kTotalPits / 2) return true;
68 return false;
69}
70
71int GetOppositePit(int pit) { return kTotalPits - pit; }
72
73int GetNextPit(Player player, int pit) {
74 int next_pit = (pit + 1) % kTotalPits;
75 if (next_pit == GetPlayerHomePit(1 - player)) next_pit++;
76 return next_pit;
77}

80void MancalaState::DoApplyAction(Action move) {
81 SPIEL_CHECK_GT(board_[move], 0);
82 int num_beans = board_[move];
83 board_[move] = 0;
84 int current_pit = move;
85 for (int i = 0; i < num_beans; ++i) {
86 current_pit = GetNextPit(current_player_, current_pit);
87 board_[current_pit]++;
88 }
89
90 // capturing logic
91 if (board_[current_pit] == 1 && IsPlayerPit(current_player_, current_pit) &&
92 board_[GetOppositePit(current_pit)] > 0) {
93 board_[GetPlayerHomePit(current_player_)] +=
94 (1 + board_[GetOppositePit(current_pit)]);
95 board_[current_pit] = 0;
96 board_[GetOppositePit(current_pit)] = 0;
97 }
98
99 if (current_pit != GetPlayerHomePit(current_player_))
100 current_player_ = 1 - current_player_;
101}

103std::vector<Action> MancalaState::LegalActions() const {
104 if (IsTerminal()) return {};
105 std::vector<Action> moves;
106 if (current_player_ == 0) {
107 for (int i = 0; i < kNumPits; ++i) {
108 if (board_[i + 1] > 0) {
109 moves.push_back(i + 1);
110 }
111 }
112 } else {
113 for (int i = 0; i < kNumPits; ++i) {
114 if (board_[board_.size() - 1 - i] > 0) {
115 moves.push_back(board_.size() - 1 - i);
116 }
117 }
118 }
119 std::sort(moves.begin(), moves.end());
120 return moves;
121}

167bool MancalaState::IsTerminal() const {
168 if (move_number_ > game_->MaxGameLength()) {
169 return true;
170 }
171
172 bool player_0_has_moves = false;
173 bool player_1_has_moves = false;
174 for (int i = 0; i < kNumPits; ++i) {
175 if (board_[board_.size() - 1 - i] > 0) {
176 player_1_has_moves = true;
177 break;
178 }
179 }
180 for (int i = 0; i < kNumPits; ++i) {
181 if (board_[i + 1] > 0) {
182 player_0_has_moves = true;
183 break;
184 }
185 }
186 return !player_0_has_moves || !player_1_has_moves;
187}

189std::vector<double> MancalaState::Returns() const {
190 if (IsTerminal()) {
191 int player_0_bean_sum = std::accumulate(
192 board_.begin() + 1, board_.begin() + kTotalPits / 2 + 1, 0);
193 int player_1_bean_sum =
194 std::accumulate(board_.begin() + kTotalPits / 2 + 1, board_.end(), 0) +
195 board_[0];
196 if (player_0_bean_sum > player_1_bean_sum) {
197 return {1.0, -1.0};
198 } else if (player_0_bean_sum < player_1_bean_sum) {
199 return {-1.0, 1.0};
200 } else {
201 // Reaches max game length or they have the same bean sum, it is a draw.
202 return {0.0, 0.0};
203 }
204 }
205 return {0.0, 0.0};
206}
