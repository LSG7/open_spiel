33  game = pyspiel.load_game("kuhn_poker")

35  br = best_response.BestResponsePolicy(game, policy=test_policy, player_id=0)

38  "1": 1, # Bet in case opponent folds when winning

41  # but we have computed best-response actions anyway.

44  "2pb": 1, # Call - we've won

120  def test_best_response_tic_tac_toe_value_is_consistent(self):
121  # This test was failing because of use of str(state) in the best response,
122  # which is imperfect recall. We now use state.history_str() throughout.
123 
124  # Chose a policy at random; not the uniform random policy.
125  game = pyspiel.load_game("tic_tac_toe")
126  pi = policy.TabularPolicy(game)
127  rng = np.random.RandomState(1234)
128  pi.action_probability_array[:] = rng.rand(*pi.legal_actions_mask.shape)
129  pi.action_probability_array *= pi.legal_actions_mask
130  pi.action_probability_array /= np.sum(
131  pi.action_probability_array, axis=1, keepdims=True)
132 
133  # Compute a best response and verify the best response value is consistent.
134  br = best_response.BestResponsePolicy(game, 1, pi)
135  self.assertAlmostEqual(
136  expected_game_score.policy_value(game.new_initial_state(), [pi, br])[1],
137  br.value(game.new_initial_state()))

158  def test_best_response_prisoner_dilemma_simultaneous_game(self):
159  """Test best response computation for simultaneous game."""
160  game = pyspiel.load_game(
161  "python_iterated_prisoners_dilemma(max_game_length=5)")
162  test_policy = policy.UniformRandomPolicy(game)
163  br = best_response.BestResponsePolicy(game, policy=test_policy, player_id=0)
164 
165  # Best policy is always to defect; we verify this for a handful of states
166  self.assertEqual(br.best_response_action("us:CCCC op:CCCC"), 1)
167  self.assertEqual(br.best_response_action("us:DDDD op:CCCC"), 1)
168  self.assertEqual(br.best_response_action("us:CDCD op:DCDC"), 1)
169  self.assertEqual(br.best_response_action("us:CCCC op:DDDD"), 1)
170 
171  # Expected value per turn = 5.5 (avg of 1 and 10)
172  # Expected game length = sum(0.875**i for i in range(5)) = 3.896728515625
173  # Game value = 5.5 * 3.896728515625 = 21.4320068359375
174  self.assertAlmostEqual(br.value(game.new_initial_state()), 21.4320068359375)
