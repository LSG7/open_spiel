218 string DumpTopHeader(
219  const ThreadData& thrd,
220  const int tricks,
221  const int lower,
222  const int upper,
223  const int printMode)
224 {
225  string stext;
226  if (printMode == 0)
227  {
228  // Trying just one target.
229  stext = "Single target " + to_string(tricks) + ", " + "achieved";
230  }
231  else if (printMode == 1)
232  {
233  // Looking for best score.
234  stext = "Loop target " + to_string(tricks) + ", " +
235  "bounds " + to_string(lower) + " .. " + to_string(upper) + ", " +
236  TopMove(thrd.val, thrd.bestMove[thrd.iniDepth]) + "";
237  }
238  else if (printMode == 2)
239  {
240  // Looking for other moves with best score.
241  stext = "Loop for cards with score " + to_string(tricks) + ", " +
242  TopMove(thrd.val, thrd.bestMove[thrd.iniDepth]);
243  }
244  return stext + "\n" + string(stext.size(), '-') + "\n";
245 }

266  const int errCode,

268  const int target,
269  const int solutions,
270  const int mode)
271 {

274 
275  fout << "Error code=" << errCode << "\n\n";
276  fout << "Deal data:\n";
277  fout << "trump=";
278 
279  if (dl.trump == DDS_NOTRUMP)
280  fout << "N\n";
281  else
282  fout << cardSuit[dl.trump] << "\n";
283  fout << "first=" << cardHand[dl.first] << "\n";
284 
285  unsigned short ranks[4][4];
286 
287  for (int k = 0; k <= 2; k++)
288  if (dl.currentTrickRank[k] != 0)
289  {
290  fout << "index=" << k <<
291  " currentTrickSuit=" << cardSuit[dl.currentTrickSuit[k]] <<
292  " currentTrickRank= " << cardRank[dl.currentTrickRank[k]] << "\n";
293  }
294 
295  for (int h = 0; h < DDS_HANDS; h++)
296  for (int s = 0; s < DDS_SUITS; s++)
297  {
298  fout << "index1=" << h << " index2=" << s <<
299  " remainCards=" << dl.remainCards[h][s] << "\n";
300  ranks[h][s] = static_cast<unsigned short>
301  (dl.remainCards[h][s] >> 2);
302  }
303 
304  fout << "\ntarget=" << target << "\n";
305  fout << "solutions=" << solutions << "\n";
306  fout << "mode=" << mode << "\n\n\n";
307  fout << PrintDeal(ranks, 8);
308 
309  fout.close();
310  return 0;
311 }
