31 Action GetOptimalAction(
32  absl::flat_hash_map<std::pair<std::string, Action>, double> q_values,
33  const std::unique_ptr<State>& state) {
34  std::vector<Action> legal_actions = state->LegalActions();
35  const auto state_str = state->ToString();
36 
37  Action optimal_action = open_spiel::kInvalidAction;
38  double value = -1;
39  for (const Action& action : legal_actions) {
40  double q_val = q_values[{state_str, action}];
41  if (q_val >= value) {
42  value = q_val;
43  optimal_action = action;
44  }
45  }
46  return optimal_action;
47 }

49 void SolveTicTacToe() {
50  std::shared_ptr<const Game> game = open_spiel::LoadGame("tic_tac_toe");
51  open_spiel::algorithms::TabularSarsaSolver tabular_sarsa_solver(game);
52 
53  int iter = 100000;
54  while (iter-- > 0) {
55  tabular_sarsa_solver.RunIteration();
56  }
57 
58  const absl::flat_hash_map<std::pair<std::string, Action>, double>& q_values =
59  tabular_sarsa_solver.GetQValueTable();
60  std::unique_ptr<State> state = game->NewInitialState();
61  while (!state->IsTerminal()) {
62  Action optimal_action = GetOptimalAction(q_values, state);
63  state->ApplyAction(optimal_action);
64  }
65 
66  // Tie.
67  SPIEL_CHECK_EQ(state->Rewards()[0], 0);
68  SPIEL_CHECK_EQ(state->Rewards()[1], 0);
69 }

71 void SolveTicTacToeEligibilityTraces() {

73  open_spiel::algorithms::TabularSarsaSolver tabular_sarsa_solver_lambda00(

77  /*learning_rate=*/0.01,
78  /*discount_factor=*/0.99,

80  open_spiel::algorithms::TabularSarsaSolver tabular_sarsa_solver_lambda03(
81  /*game=*/game,
82  /*depth_limit=*/-1.0,
83  /*epsilon=*/0.1,
84  /*learning_rate=*/0.01,
85  /*discount_factor=*/0.99,
86  /*lambda=*/0.3);
87 
88  const int runs = 1000;
89  int count_tie_games_lambda00 = 0;
90  int count_tie_games_lambda03 = 0;
91  for (int i = 0; i < runs; ++i) {
92  tabular_sarsa_solver_lambda00.RunIteration();
93 
94  const absl::flat_hash_map<std::pair<std::string, Action>, double>
95  &q_values_lambda00 = tabular_sarsa_solver_lambda00.GetQValueTable();
96  std::unique_ptr<State> state = game->NewInitialState();
97 
98  while (!state->IsTerminal()) {
99  state->ApplyAction(GetOptimalAction(q_values_lambda00, state));
100  }
101 
102  count_tie_games_lambda00 += state->Rewards()[0] == 0 ? 1 : 0;
103  }
104 
105  for (int i = 0; i < runs; ++i) {
106  tabular_sarsa_solver_lambda03.RunIteration();
107 
108  const absl::flat_hash_map<std::pair<std::string, Action>, double>
109  &q_values_lambda01 = tabular_sarsa_solver_lambda03.GetQValueTable();
110  std::unique_ptr<State> state = game->NewInitialState();
111 
112  while (!state->IsTerminal()) {
113  state->ApplyAction(GetOptimalAction(q_values_lambda01, state));
114  }
115 
116  count_tie_games_lambda03 += state->Rewards()[0] == 0 ? 1 : 0;
117  }
118 
119  // SARSA(0.3) gets equilibrium faster than SARSA(0.0). More ties in the same
