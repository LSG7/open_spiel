37class LRUCache(object):
38 """A Least Recently Used cache.
39
40 This is more general than functools.lru_cache since that one requires the
41 key to also be the input to the function to generate the value, which
42 isn't possible when the input is not hashable, eg a numpy.ndarray.
43 """
44
45 def __init__(self, max_size):
46 self._max_size = max_size
47 self._data = collections.OrderedDict()
48 self._hits = 0
49 self._misses = 0
50
51 def clear(self):
52 self._data.clear()
53 self._hits = 0
54 self._misses = 0
55
56 def make(self, key, fn):
57 """Return the value, either from cache, or make it and save it."""
58 try:
59 val = self._data.pop(key) # Take it out.
60 self._hits += 1
61 except KeyError:
62 self._misses += 1
63 val = fn()
64 if len(self._data) >= self._max_size:
65 self._data.popitem(False)
66 self._data[key] = val # Insert/reinsert it at the back.
67 return val
68
69 def get(self, key):
70 """Get the value and move it to the back, or return None on a miss."""
71 try:
72 val = self._data.pop(key) # Take it out.
73 self._data[key] = val # Reinsert it at the back.
74 self._hits += 1
75 return val
76 except KeyError:
77 self._misses += 1
78 return None
79
80 def set(self, key, val):
81 """Set the value."""
82 self._data.pop(key, None) # Take it out if it existed.
83 self._data[key] = val # Insert/reinsert it at the back.
84 if len(self._data) > self._max_size:
85 self._data.popitem(False)
86 return val
87
88 def info(self):
89 return CacheInfo(self._hits, self._misses, len(self._data), self._max_size)
90
91 def __len__(self):
92 return len(self._data)
