213 TEST(ScopedMockLogTest, LogFromMultipleThreads) {

217  // execute their log statements in different order.

221  log.StartCapturingLogs();

227  });

229  barrier.Block();
230  LOG(INFO) << "Thread 2";
231  });

233  thread1.join();
234  thread2.join();
