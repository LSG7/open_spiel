90void OshiZumoState::DoApplyActions(const std::vector<Action>& actions) {
91 SPIEL_CHECK_EQ(actions.size(), 2);
92 SPIEL_CHECK_TRUE(actions[0] >= 0);
93 SPIEL_CHECK_TRUE(actions[1] >= 0);
94 SPIEL_CHECK_TRUE(actions[0] <= coins_[0]);
95 SPIEL_CHECK_TRUE(actions[1] <= coins_[1]);
96
97 // Move the wrestler.
98 if (actions[0] > actions[1]) {
99 wrestler_pos_++;
100 } else if (actions[0] < actions[1]) {
101 wrestler_pos_--;
102 }
103
104 // Remove coins.
105 coins_[0] -= actions[0];
106 coins_[1] -= actions[1];
107
108 // Check winner.
109 if (wrestler_pos_ == 0) {
110 winner_ = 1;
111 } else if (wrestler_pos_ == (2 * size_ + 2)) {
112 winner_ = 0;
113 }
114
115 total_moves_++;
116}

118std::vector<Action> OshiZumoState::LegalActions(Player player) const {
119 if (IsTerminal()) return {};
120 if (player == kSimultaneousPlayerId) return LegalFlatJointActions();
121 SPIEL_CHECK_FALSE(IsChanceNode());
122 SPIEL_CHECK_TRUE(player == Player{0} || player == Player{1});
123
124 std::vector<Action> movelist;
125 for (int bet = min_bid_; bet <= coins_[player]; bet++) {
126 movelist.push_back(bet);
127 }
128
129 if (movelist.empty()) {
130 // Player does not have the minimum bid: force them to play what they have
131 // left.
132 movelist.push_back(coins_[player]);
133 }
134
135 return movelist;
136}

148std::string OshiZumoState::ToString() const {
149 std::string result = "Coins: ";
150
151 absl::StrAppend(&result, coins_[0]);
152 absl::StrAppend(&result, " ");
153 absl::StrAppend(&result, coins_[1]);
154 absl::StrAppend(&result, ", Field: ");
155
156 for (int p = 0; p <= 2 * size_ + 2; p++) {
157 if (p == wrestler_pos_) {
158 result += kWrestler;
159 } else if (p == 0 || p == (2 * size_ + 2)) {
160 result += kBoundaryPos;
161 } else {
162 result += kOpenPos;
163 }
164 }
165
166 absl::StrAppend(&result, "\n");
167 return result;
168}

170bool OshiZumoState::IsTerminal() const {
171 return (total_moves_ >= horizon_ || winner_ != kNoWinner ||
172 (coins_[0] == 0 && coins_[1] == 0));

175std::vector<double> OshiZumoState::Returns() const {
176 if (!IsTerminal()) {
177 return {0.0, 0.0};
178 }
179
180 if (winner_ == 0) {
181 return {1.0, -1.0};
182 } else if (winner_ == 1) {
183 return {-1.0, 1.0};
184 } else {
185 // Wrestler not off the edge.
186 if (alesia_) {
187 return {0.0, 0.0};
188 } else if (wrestler_pos_ > (size_ + 1)) {
189 return {1.0, -1.0};
190 } else if (wrestler_pos_ < (size_ + 1)) {
191 return {-1.0, 1.0};
192 } else {
193 return {0.0, 0.0};
194 }
195 }
196}

210void OshiZumoState::ObservationTensor(Player player,
211 absl::Span<float> values) const {
212 SPIEL_CHECK_GE(player, 0);
213 SPIEL_CHECK_LT(player, num_players_);
214
215 SPIEL_CHECK_EQ(values.size(), parent_game_.ObservationTensorShape()[0]);
216 std::fill(values.begin(), values.end(), 0.);
217
218 // 1 bit per coin value of player 1. { 0, 1, ... , starting_coins_ }
219 // 1 bit per coin value of player 2. { 0, 1, ... , starting_coins_ }
220 // 1 bit per position of the field. { 0, 1, ... , 2*size_+2 }
221
222 int offset = 0;
223 values[offset + coins_[0]] = 1;
224
225 offset += (starting_coins_ + 1);
226 values[offset + coins_[1]] = 1;
227
228 offset += (starting_coins_ + 1);
229 values[offset + wrestler_pos_] = 1;
230}

251std::vector<int> OshiZumoGame::ObservationTensorShape() const {
252 // 1 bit per coin value of player 1. { 0, 1, ..., starting_coins_ }
253 // 1 bit per coin value of player 2. { 0, 1, ..., starting_coins_ }
254 // 1 bit per position of the field. { 0, 1, ... , 2*size_+2 }
255 return {(2 * (starting_coins_ + 1)) + (2 * size_ + 3)};
