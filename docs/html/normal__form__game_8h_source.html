29 // This class describes an n-player normal-form game. A normal-form game is
30 // also known as a one-shot game or strategic-form game. Essentially, all
31 // players act simultaneously and the game ends after a single joint action
32 // taken by all players. E.g. a matrix game is an example of a 2-player normal
33 // form game.

108 class NormalFormGame : public SimMoveGame {
109  public:
110  // Game has one state.
111  std::vector<int> InformationStateTensorShape() const override {
112  return {1};
113  }
114  std::vector<int> ObservationTensorShape() const override { return {1}; }
115 
116  // Game lasts one turn.
117  int MaxGameLength() const override { return 1; }
118  // There aren't chance nodes in these games.
119  int MaxChanceNodesInHistory() const override { return 0; }
120 
121  // Direct access to utility. This is just a default implementation, which is
122  // overridden in subclasses for faster access.
123  virtual std::vector<double> GetUtilities(
124  const std::vector<Action>& joint_action) const {
125  std::unique_ptr<State> state = NewInitialState();
126  state->ApplyActions(joint_action);
127  return state->Returns();
128  }
129 
130  virtual double GetUtility(Player player,
131  const std::vector<Action>& joint_action) const {
132  return GetUtilities(joint_action)[player];
133  }
134 
135  absl::optional<double> UtilitySum() const override {
136  if (game_type_.utility == GameType::Utility::kZeroSum) {
137  return 0.0;
138  } else if (game_type_.utility == GameType::Utility::kConstantSum) {
139  std::vector<Action> joint_action(NumPlayers(), 0);
140  std::vector<double> utilities = GetUtilities(joint_action);
141  return std::accumulate(utilities.begin(), utilities.end(), 0.0);
142  } else {
143  return absl::nullopt;
144  }
145  }
146 
147  protected:
148  NormalFormGame(GameType game_type, GameParameters game_parameters)
149  : SimMoveGame(game_type, game_parameters) {}
150 };
