45 void TestCorrelatedEquilibrium() {
46  // Wikipedia example:
47  // https://en.wikipedia.org/wiki/Correlated_equilibrium#An_example
48  std::shared_ptr<const matrix_game::MatrixGame> chicken_dare =
49  matrix_game::CreateMatrixGame({{0, 7}, {2, 6}}, {{0, 2}, {7, 6}});
50  NormalFormCorrelationDevice mu =
51  ComputeCorrelatedEquilibrium(*chicken_dare, CorrEqObjType::kAny, 0.0);
52  for (const auto &item : mu) {
53  std::cout << item.probability << " " << absl::StrJoin(item.actions, " ")
54  << std::endl;
55  }
56  std::cout << std::endl;
57 
58  // There is a CE with 1/3 (C,C), 1/3 (D,C), and 1/3 (C,D).
59  mu = ComputeCorrelatedEquilibrium(*chicken_dare,
60  CorrEqObjType::kSocialWelfareAtLeast, 10.0);
61  for (const auto &item : mu) {
62  std::cout << item.probability << " " << absl::StrJoin(item.actions, " ")
63  << std::endl;
64  }
65  std::cout << std::endl;
66 
67  std::vector<double> expected_values = ExpectedValues(*chicken_dare, mu);
68  double social_welfare =
69  std::accumulate(expected_values.begin(), expected_values.end(), 0.0);
70  std::cout << social_welfare << std::endl;
71  SPIEL_CHECK_GE(social_welfare, 10.0 - kErrorTolerance);
72 
73  // There is a better one that gets 10.5: 1/4 (C,D), 1/4 (D,C), 1/2 (C, C)
74  mu = ComputeCorrelatedEquilibrium(*chicken_dare,
75  CorrEqObjType::kSocialWelfareMax, 0);
76  for (const auto &item : mu) {
77  std::cout << item.probability << " " << absl::StrJoin(item.actions, " ")
78  << std::endl;
79  }
80  std::cout << std::endl;
81 
82  expected_values = ExpectedValues(*chicken_dare, mu);
83  social_welfare =
84  std::accumulate(expected_values.begin(), expected_values.end(), 0.0);
85  std::cout << social_welfare << std::endl;
86  SPIEL_CHECK_FLOAT_NEAR(social_welfare, 10.5, kErrorTolerance);
87  for (const auto &item : mu) {
88  if (item.actions[0] + item.actions[1] == 1) {
89  SPIEL_CHECK_FLOAT_NEAR(item.probability, 1.0 / 4.0, kErrorTolerance);
90  } else if (item.actions[0] + item.actions[1] == 2) {
91  SPIEL_CHECK_FLOAT_NEAR(item.probability, 1.0 / 2.0, kErrorTolerance);
92  }
93  }
94 }
