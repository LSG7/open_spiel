138 def legal_actions_to_mask(legal_actions_list, num_actions):
139  """Converts a list of legal actions to a mask.
140 
141  The mask has size num actions with a 1 in a legal positions.
142 
143  Args:
144  legal_actions_list: the list of legal actions
145  num_actions: number of actions (width of mask)
146 
147  Returns:
148  legal actions mask.
149  """
150  legal_actions_mask = torch.zeros((len(legal_actions_list), num_actions),
151  dtype=torch.bool)
152  for i, legal_actions in enumerate(legal_actions_list):
153  legal_actions_mask[i, legal_actions] = 1
154  return legal_actions_mask

308  def learn(self, time_step):
309  next_obs = torch.Tensor(
310  np.array([
311  np.reshape(ts.observations["info_state"][self.player_idplayer_id],
312  self.input_shapeinput_shape) for ts in time_step
313  ])).to(self.devicedevice)
314 
315  # bootstrap value if not done
316  with torch.no_grad():
317  next_value = self.get_valueget_value(next_obs).reshape(1, -1)
318  if self.gaegae:
319  advantages = torch.zeros_like(self.rewardsrewards).to(self.devicedevice)
320  lastgaelam = 0
321  for t in reversed(range(self.steps_per_batchsteps_per_batch)):
322  nextvalues = next_value if t == self.steps_per_batchsteps_per_batch - 1 else self.valuesvalues[
323  t + 1]
324  nextnonterminal = 1.0 - self.donesdones[t]
325  delta = self.rewardsrewards[
326  t] + self.gammagamma * nextvalues * nextnonterminal - self.valuesvalues[t]
327  advantages[
328  t] = lastgaelam = delta + self.gammagamma * self.gae_lambdagae_lambda * nextnonterminal * lastgaelam
329  returns = advantages + self.valuesvalues
330  else:
331  returns = torch.zeros_like(self.rewardsrewards).to(self.devicedevice)
332  for t in reversed(range(self.steps_per_batchsteps_per_batch)):
333  next_return = next_value if t == self.steps_per_batchsteps_per_batch - 1 else returns[
334  t + 1]
335  nextnonterminal = 1.0 - self.donesdones[t]
336  returns[
337  t] = self.rewardsrewards[t] + self.gammagamma * nextnonterminal * next_return
338  advantages = returns - self.valuesvalues

368  old_approx_kl = (-logratio).mean()

370  clipfracs += [
371  ((ratio - 1.0).abs() > self.clip_coefclip_coef).float().mean().item()
372  ]
373 
374  mb_advantages = b_advantages[mb_inds]
375  if self.normalize_advantagesnormalize_advantages:

377  mb_advantages.std() + 1e-8)
378 
379  # Policy loss
380  pg_loss1 = -mb_advantages * ratio
381  pg_loss2 = -mb_advantages * torch.clamp(ratio, 1 - self.clip_coefclip_coef,
382  1 + self.clip_coefclip_coef)
383  pg_loss = torch.max(pg_loss1, pg_loss2).mean()
384 
385  # Value loss
386  newvalue = newvalue.view(-1)
387  if self.clip_vlossclip_vloss:
388  v_loss_unclipped = (newvalue - b_returns[mb_inds])**2

391  -self.clip_coefclip_coef,
392  self.clip_coefclip_coef,
393  )
394  v_loss_clipped = (v_clipped - b_returns[mb_inds])**2
395  v_loss_max = torch.max(v_loss_unclipped, v_loss_clipped)
396  v_loss = 0.5 * v_loss_max.mean()
