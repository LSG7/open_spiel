29// This class describes an n-player normal-form game. A normal-form game is
30// also known as a one-shot game or strategic-form game. Essentially, all
31// players act simultaneously and the game ends after a single joint action
32// taken by all players. E.g. a matrix game is an example of a 2-player normal
33// form game.
34

37class NFGState : public SimMoveState {
38 public:
39 NFGState(std::shared_ptr<const Game> game) : SimMoveState(game) {}
40
41 // There are no chance nodes in a normal-form game (there is only one state),
42 Player CurrentPlayer() const final {
43 return IsTerminal() ? kTerminalPlayerId : kSimultaneousPlayerId;
44 }
45
46 // Since there's only one state, we can implement the representations here.
47 std::string InformationStateString(Player player) const override {
48 SPIEL_CHECK_GE(player, 0);
49 SPIEL_CHECK_LT(player, num_players_);
50 std::string info_state = absl::StrCat("Observing player: ", player, ". ");
51 if (!IsTerminal()) {
52 absl::StrAppend(&info_state, "Non-terminal");
53 } else {
54 absl::StrAppend(&info_state,
55 "Terminal. History string: ", HistoryString());
56 }
57 return info_state;
58 }
59
60 std::string ObservationString(Player player) const override {
61 SPIEL_CHECK_GE(player, 0);
62 SPIEL_CHECK_LT(player, num_players_);
63 std::string obs_str;
64 if (!IsTerminal()) {
65 absl::StrAppend(&obs_str, "Non-terminal");
66 } else {
67 absl::StrAppend(&obs_str, "Terminal. History string: ", HistoryString());
68 }
69 return obs_str;
70 }
71
72 std::string ToString() const override {
73 std::string result = "Normal form game default NFGState::ToString. ";
74 if (IsTerminal()) {
75 absl::StrAppend(&result, "Terminal, history: ", HistoryString(),
76 ", returns: ", absl::StrJoin(Returns(), ","));
77 } else {
78 absl::StrAppend(&result, "Non-terminal");
79 }
80 return result;
81 }
82
83 void InformationStateTensor(Player player,
84 absl::Span<float> values) const override {
85 SPIEL_CHECK_GE(player, 0);
86 SPIEL_CHECK_LT(player, num_players_);
87 SPIEL_CHECK_EQ(values.size(), 1);
88 if (IsTerminal()) {
89 values[0] = 1;
90 } else {
91 values[0] = 0;
92 }
93 }
94
95 void ObservationTensor(Player player,
96 absl::Span<float> values) const override {
97 SPIEL_CHECK_GE(player, 0);
98 SPIEL_CHECK_LT(player, num_players_);
99 SPIEL_CHECK_EQ(values.size(), 1);
100 if (IsTerminal()) {
101 values[0] = 1;
102 } else {
103 values[0] = 0;
104 }
105 }
106};

108class NormalFormGame : public SimMoveGame {
109 public:
110 // Game has one state.
111 std::vector<int> InformationStateTensorShape() const override {
112 return {1};
113 }
114 std::vector<int> ObservationTensorShape() const override { return {1}; }
115
116 // Game lasts one turn.
117 int MaxGameLength() const override { return 1; }
118 // There aren't chance nodes in these games.
119 int MaxChanceNodesInHistory() const override { return 0; }
120
121 // Direct access to utility. This is just a default implementation, which is
122 // overridden in subclasses for faster access.
123 virtual std::vector<double> GetUtilities(
124 const std::vector<Action>& joint_action) const {
125 std::unique_ptr<State> state = NewInitialState();
126 state->ApplyActions(joint_action);
127 return state->Returns();
128 }
129
130 virtual double GetUtility(Player player,
131 const std::vector<Action>& joint_action) const {
132 return GetUtilities(joint_action)[player];
133 }
134
135 absl::optional<double> UtilitySum() const override {
136 if (game_type_.utility == GameType::Utility::kZeroSum) {
137 return 0.0;
138 } else if (game_type_.utility == GameType::Utility::kConstantSum) {
139 std::vector<Action> joint_action(NumPlayers(), 0);
140 std::vector<double> utilities = GetUtilities(joint_action);
141 return std::accumulate(utilities.begin(), utilities.end(), 0.0);
142 } else {
143 return absl::nullopt;
144 }
145 }
146
147 protected:
148 NormalFormGame(GameType game_type, GameParameters game_parameters)
149 : SimMoveGame(game_type, game_parameters) {}
150};
