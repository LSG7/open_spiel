80// The Observer class is responsible for creating representations of the game
81// state for use in learning algorithms. It handles both string and tensor
82// representations, and any combination of public information and private
83// information (none, observing player only, or all players).
84//
85// If a perfect recall observation is requested, it must be possible to deduce
86// all previous observations for the same information type from the current
87// observation.
88
89class LeducObserver : public Observer {
90 public:
91 LeducObserver(IIGObservationType iig_obs_type)
92 : Observer(/*has_string=*/true, /*has_tensor=*/true),
93 iig_obs_type_(iig_obs_type) {}
94
95 //
96 // These helper methods each write a piece of the tensor observation.
97 //
98
99 // Identity of the observing player. One-hot vector of size num_players.
100 static void WriteObservingPlayer(const LeducState& state, int player,
101 Allocator* allocator) {
102 auto out = allocator->Get("player", {state.num_players_});
103 out.at(player) = 1;
104 }
105
106 // Private card of the observing player. One-hot vector of size num_cards.
107 static void WriteSinglePlayerCard(const LeducState& state, int player,
108 Allocator* allocator) {
109 auto out = allocator->Get("private_card", {state.NumObservableCards()});
110 int card = state.private_cards_[player];
111 if (card != kInvalidCard) out.at(card) = 1;
112 }
113
114 // Private cards of all players. Tensor of shape [num_players, num_cards].
115 static void WriteAllPlayerCards(const LeducState& state,
116 Allocator* allocator) {
117 auto out = allocator->Get("private_cards",
118 {state.num_players_, state.NumObservableCards()});
119 for (int p = 0; p < state.num_players_; ++p) {
120 int card = state.private_cards_[p];
121 if (card != kInvalidCard) out.at(p, state.private_cards_[p]) = 1;
122 }
123 }
124
125 // Community card (if any). One-hot vector of size num_cards.
126 static void WriteCommunityCard(const LeducState& state,
127 Allocator* allocator) {
128 auto out = allocator->Get("community_card", {state.NumObservableCards()});
129 if (state.public_card_ != kInvalidCard) {
130 out.at(state.public_card_) = 1;
131 }
132 }
133
134 // Betting sequence; shape [num_rounds, bets_per_round, num_actions].
135 static void WriteBettingSequence(const LeducState& state,
136 Allocator* allocator) {
137 const int kNumRounds = 2;
138 const int kBitsPerAction = 2;
139 const int max_bets_per_round = state.MaxBetsPerRound();
140 auto out = allocator->Get("betting",
141 {kNumRounds, max_bets_per_round, kBitsPerAction});
142 for (int round : {0, 1}) {
143 const auto& bets =
144 (round == 0) ? state.round1_sequence_ : state.round2_sequence_;
145 for (int i = 0; i < bets.size(); ++i) {
146 if (bets[i] == ActionType::kCall) {
147 out.at(round, i, 0) = 1; // Encode call as 10.
148 } else if (bets[i] == ActionType::kRaise) {
149 out.at(round, i, 1) = 1; // Encode raise as 01.
150 }
151 }
152 }
153 }
154
155 // Pot contribution per player (integer per player).
156 static void WritePotContribution(const LeducState& state,
157 Allocator* allocator) {
158 auto out = allocator->Get("pot_contribution", {state.num_players_});
159 for (auto p = Player{0}; p < state.num_players_; p++) {
160 out.at(p) = state.ante_[p];
161 }
162 }
163
164 // Writes the complete observation in tensor form.
165 // The supplied allocator is responsible for providing memory to write the
166 // observation into.
167 void WriteTensor(const State& observed_state, int player,
168 Allocator* allocator) const override {
169 auto& state = open_spiel::down_cast<const LeducState&>(observed_state);
170 SPIEL_CHECK_GE(player, 0);
171 SPIEL_CHECK_LT(player, state.num_players_);
172
173 // Observing player.
174 WriteObservingPlayer(state, player, allocator);
175
176 // Private card(s).
177 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
178 WriteSinglePlayerCard(state, player, allocator);
179 } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
180 WriteAllPlayerCards(state, allocator);
181 }
182
183 // Public information.
184 if (iig_obs_type_.public_info) {
185 WriteCommunityCard(state, allocator);
186 iig_obs_type_.perfect_recall ? WriteBettingSequence(state, allocator)
187 : WritePotContribution(state, allocator);
188 }
189 }
190
191 // Writes an observation in string form. It would be possible just to
192 // turn the tensor observation into a string, but we prefer something
193 // somewhat human-readable.
194
195 std::string StringFrom(const State& observed_state,
196 int player) const override {
197 auto& state = open_spiel::down_cast<const LeducState&>(observed_state);
198 SPIEL_CHECK_GE(player, 0);
199 SPIEL_CHECK_LT(player, state.num_players_);
200 std::string result;
201
202 // Private card(s).
203 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
204 absl::StrAppend(&result, "[Observer: ", player, "]");
205 absl::StrAppend(&result, "[Private: ", state.private_cards_[player], "]");
206 } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
207 absl::StrAppend(
208 &result, "[Privates: ", absl::StrJoin(state.private_cards_, ""), "]");
209 }
210
211 // Public info. Not all of this is strictly necessary, but it makes the
212 // string easier to understand.
213 if (iig_obs_type_.public_info) {
214 absl::StrAppend(&result, "[Round ", state.round_, "]");
215 absl::StrAppend(&result, "[Player: ", state.cur_player_, "]");
216 absl::StrAppend(&result, "[Pot: ", state.pot_, "]");
217 absl::StrAppend(&result, "[Money: ", absl::StrJoin(state.money_, " "),
218 "]");
219 if (state.public_card_ != kInvalidCard) {
220 absl::StrAppend(&result, "[Public: ", state.public_card_, "]");
221 }
222 if (iig_obs_type_.perfect_recall) {
223 // Betting Sequence (for the perfect recall case)
224 absl::StrAppend(
225 &result, "[Round1: ", absl::StrJoin(state.round1_sequence_, " "),
226 "][Round2: ", absl::StrJoin(state.round2_sequence_, " "), "]");
227 } else {
228 // Pot contributions (imperfect recall)
229 absl::StrAppend(&result, "[Ante: ", absl::StrJoin(state.ante_, " "),
230 "]");
231 }
232 }
233
234 // Done.
235 return result;
236 }
237
238 private:
239 IIGObservationType iig_obs_type_;
240};

242LeducState::LeducState(std::shared_ptr<const Game> game, bool action_mapping,
243 bool suit_isomorphism)
244 : State(game),
245 cur_player_(kChancePlayerId),
246 num_calls_(0),
247 num_raises_(0),
248 round_(1), // Round number (1 or 2).
249 stakes_(1), // The current 'level' of the bet.
250 num_winners_(-1),
251 pot_(kAnte * game->NumPlayers()), // Number of chips in the pot.
252 public_card_(kInvalidCard),
253 // Number of cards remaining; not equal deck_.size()!
254 deck_size_((game->NumPlayers() + 1) * kNumSuits),
255 private_cards_dealt_(0),
256 remaining_players_(game->NumPlayers()),
257 // Is this player a winner? Indexed by pid.
258 winner_(game->NumPlayers(), false),
259 // Each player's single private card. Indexed by pid.
260 private_cards_(game->NumPlayers(), kInvalidCard),
261 // How much money each player has, indexed by pid.
262 money_(game->NumPlayers(), kStartingMoney - kAnte),
263 // How much each player has contributed to the pot, indexed by pid.
264 ante_(game->NumPlayers(), kAnte),
265 // Flag for whether the player has folded, indexed by pid.
266 folded_(game->NumPlayers(), false),
267 // Sequence of actions for each round. Needed to report information
268 // state.
269 round1_sequence_(),
270 round2_sequence_(),
271 // Always regard all actions as legal, and internally map otherwise
272 // illegal actions to check/call.
273 action_mapping_(action_mapping),
274 // Players cannot distinguish between cards of different suits with the
275 // same rank.
276 suit_isomorphism_(suit_isomorphism) {
277 // Cards by value (0-6 for standard 2-player game, kInvalidCard if no longer
278 // in the deck.)
279 deck_.resize(deck_size_);
280 std::iota(deck_.begin(), deck_.end(), 0);
281}

291// In a chance node, `move` should be the card to deal to the current
292// underlying player.
293// On a player node, it should be ActionType::{kFold, kCall, kRaise}
294void LeducState::DoApplyAction(Action move) {
295 if (IsChanceNode()) {
296 SPIEL_CHECK_GE(move, 0);
297 SPIEL_CHECK_LT(move, deck_.size());
298 if (suit_isomorphism_) {
299 // One of the two identical cards must be left in the deck.
300 SPIEL_CHECK_TRUE(deck_[move * 2] != kInvalidCard ||
301 deck_[move * 2 + 1] != kInvalidCard);
302 } else {
303 SPIEL_CHECK_NE(deck_[move], kInvalidCard);
304 }
305
306 if (private_cards_dealt_ < num_players_) {
307 SetPrivate(private_cards_dealt_, move);
308 } else {
309 // Round 2: A single public card.
310 if (suit_isomorphism_) {
311 public_card_ = move;
312 if (deck_[move * 2] != kInvalidCard) {
313 deck_[move * 2] = kInvalidCard;
314 } else if (deck_[move * 2 + 1] != kInvalidCard) {
315 deck_[move * 2 + 1] = kInvalidCard;
316 } else {
317 SpielFatalError("Suit isomorphism error.");
318 }
319 deck_size_--;
320 } else {
321 public_card_ = deck_[move];
322 deck_[move] = kInvalidCard;
323 deck_size_--;
324 }
325
326 // We have finished the public card, let's bet!
327 cur_player_ = NextPlayer();
328 }
329 } else {
330 // Player node.
331 if (action_mapping_) {
332 // Map otherwise illegal actions to kCall.
333 if (move == ActionType::kFold) {
334 if (stakes_ <= ante_[cur_player_]) {
335 move = ActionType::kCall;
336 }
337 } else if (move == ActionType::kRaise) {
338 if (num_raises_ >= 2) {
339 move = ActionType::kCall;
340 }
341 }
342 }
343
344 if (move == ActionType::kFold) {
345 SPIEL_CHECK_NE(cur_player_, kChancePlayerId);
346 SequenceAppendMove(ActionType::kFold);
347
348 // Player is now out.
349 folded_[cur_player_] = true;
350 remaining_players_--;
351
352 if (IsTerminal()) {
353 ResolveWinner();
354 } else if (ReadyForNextRound()) {
355 NewRound();
356 } else {
357 cur_player_ = NextPlayer();
358 }
359 } else if (move == ActionType::kCall) {
360 SPIEL_CHECK_NE(cur_player_, kChancePlayerId);
361
362 // Current player puts in an amount of money equal to the current level
363 // (stakes) minus what they have contributed to level their contribution
364 // off. Note: this action also acts as a 'check' where the stakes are
365 // equal to each player's ante.
366 SPIEL_CHECK_GE(stakes_, ante_[cur_player_]);
367 int amount = stakes_ - ante_[cur_player_];
368 Ante(cur_player_, amount);
369 num_calls_++;
370 SequenceAppendMove(ActionType::kCall);
371
372 if (IsTerminal()) {
373 ResolveWinner();
374 } else if (ReadyForNextRound()) {
375 NewRound();
376 } else {
377 cur_player_ = NextPlayer();
378 }
379 } else if (move == ActionType::kRaise) {
380 SPIEL_CHECK_NE(cur_player_, kChancePlayerId);
381
382 // This player matches the current stakes and then brings the stakes up.
383 SPIEL_CHECK_LT(num_raises_, kMaxRaises);
384 int call_amount = stakes_ - ante_[cur_player_];
385
386 // First, match the current stakes if necessary
387 SPIEL_CHECK_GE(call_amount, 0);
388 if (call_amount > 0) {
389 Ante(cur_player_, call_amount);
390 }
391
392 // Now, raise the stakes.
393 int raise_amount = (round_ == 1 ? kFirstRaiseAmount : kSecondRaiseAmount);
394 stakes_ += raise_amount;
395 Ante(cur_player_, raise_amount);
396 num_raises_++;
397 num_calls_ = 0;
398 SequenceAppendMove(ActionType::kRaise);
399
400 if (IsTerminal()) {
401 ResolveWinner();
402 } else {
403 cur_player_ = NextPlayer();
404 }
405 } else {
406 SpielFatalError(absl::StrCat("Move ", move, " is invalid. ChanceNode?",
407 IsChanceNode()));
408 }
409 }
410}

412std::vector<Action> LeducState::LegalActions() const {
413 if (IsTerminal()) return {};
414 std::vector<Action> movelist;
415 if (IsChanceNode()) {
416 if (suit_isomorphism_) {
417 // Consecutive cards are identical under suit isomorphism.
418 for (int card = 0; card < deck_.size() / 2; card++) {
419 if (deck_[card * 2] != kInvalidCard ||
420 deck_[card * 2 + 1] != kInvalidCard) {
421 movelist.push_back(card);
422 }
423 }
424 } else {
425 for (int card = 0; card < deck_.size(); card++) {
426 if (deck_[card] != kInvalidCard) movelist.push_back(card);
427 }
428 }
429 return movelist;
430 }
431
432 if (action_mapping_) {
433 // All actions are regarded as legal
434 movelist.push_back(ActionType::kFold);
435 movelist.push_back(ActionType::kCall);
436 movelist.push_back(ActionType::kRaise);
437 return movelist;
438 }
439
440 // Can't just randomly fold; only allow fold when under pressure.
441 if (stakes_ > ante_[cur_player_]) {
442 movelist.push_back(ActionType::kFold);
443 }
444
445 // Can always call/check
446 movelist.push_back(ActionType::kCall);
447
448 if (num_raises_ < 2) {
449 movelist.push_back(ActionType::kRaise);
450 }
451
452 return movelist;
453}

459std::string LeducState::ToString() const {
460 std::string result;
461
462 absl::StrAppend(&result, "Round: ", round_, "\nPlayer: ", cur_player_,
463 "\nPot: ", pot_, "\nMoney (p1 p2 ...):");
464 for (auto p = Player{0}; p < num_players_; p++) {
465 absl::StrAppend(&result, " ", money_[p]);
466 }
467 absl::StrAppend(&result, "\nCards (public p1 p2 ...): ", public_card_, " ");
468 for (Player player_index = 0; player_index < num_players_; player_index++) {
469 absl::StrAppend(&result, private_cards_[player_index], " ");
470 }
471
472 absl::StrAppend(&result, "\nRound 1 sequence: ");
473 for (int i = 0; i < round1_sequence_.size(); ++i) {
474 Action action = round1_sequence_[i];
475 if (i > 0) absl::StrAppend(&result, ", ");
476 absl::StrAppend(&result, StatelessActionToString(action));
477 }
478 absl::StrAppend(&result, "\nRound 2 sequence: ");
479 for (int i = 0; i < round2_sequence_.size(); ++i) {
480 Action action = round2_sequence_[i];
481 if (i > 0) absl::StrAppend(&result, ", ");
482 absl::StrAppend(&result, StatelessActionToString(action));
483 }
484 absl::StrAppend(&result, "\n");
485
486 return result;
487}

537std::vector<std::pair<Action, double>> LeducState::ChanceOutcomes() const {
538 SPIEL_CHECK_TRUE(IsChanceNode());
539 std::vector<std::pair<Action, double>> outcomes;
540
541 if (suit_isomorphism_) {
542 const double p = 1.0 / deck_size_;
543 // Consecutive cards in deck are viewed identically.
544 for (int card = 0; card < deck_.size() / 2; card++) {
545 if (deck_[card * 2] != kInvalidCard &&
546 deck_[card * 2 + 1] != kInvalidCard) {
547 outcomes.push_back({card, p * 2});
548 } else if (deck_[card * 2] != kInvalidCard ||
549 deck_[card * 2 + 1] != kInvalidCard) {
550 outcomes.push_back({card, p});
551 }
552 }
553 return outcomes;
554 }
555
556 const double p = 1.0 / deck_size_;
557 for (int card = 0; card < deck_.size(); card++) {
558 // This card is still in the deck, prob is 1/decksize.
559 if (deck_[card] != kInvalidCard) outcomes.push_back({card, p});
560 }
561 return outcomes;
562}

564int LeducState::NextPlayer() const {
565 // If we are on a chance node, it is the first player to play
566 int current_real_player;
567 if (cur_player_ == kChancePlayerId) {
568 current_real_player = -1;
569 } else {
570 current_real_player = cur_player_;
571 }
572 // Go to the next player who's still in.
573 for (int i = 1; i < num_players_; ++i) {
574 Player player = (current_real_player + i) % num_players_;
575
576 SPIEL_CHECK_TRUE(player >= 0);
577 SPIEL_CHECK_TRUE(player < num_players_);
578 if (!folded_[player]) {
579 return player;
580 }
581 }
582
583 SpielFatalError("Error in LeducState::NextPlayer(), should not get here.");
584}

586int LeducState::RankHand(Player player) const {
587 int hand[] = {public_card_, private_cards_[player]};
588 // Put the lower card in slot 0, the higher in slot 1.
589 if (hand[0] > hand[1]) {
590 std::swap(hand[0], hand[1]);
591 }
592
593 if (suit_isomorphism_) {
594 int num_cards = deck_.size() / 2;
595 if (hand[0] == hand[1]) {
596 // Pair! Offset by deck_size_^2 to put higher than every singles combo.
597 return (num_cards * num_cards + hand[0]);
598 } else {
599 // Otherwise card value dominates. Suit isomorphism has already removed
600 // the distinction between suits, so we can compare the ranks directly.
601 // This could lead to ties/draws and/or multiple winners.
602 return hand[1] * num_cards + hand[0];
603 }
604 }
605
606 // E.g. rank for two players:
607 // 0 J1, 1 J2, 2 Q1, 3 Q2, 4 K1, 5 K2.
608 int num_cards = deck_.size();
609
610 if (hand[0] % 2 == 0 && hand[1] == hand[0] + 1) {
611 // Pair! Offset by deck_size_^2 to put higher than every singles combo.
612 return (num_cards * num_cards + hand[0]);
613 } else {
614 // Otherwise card value dominates. No high/low suit: only two suits, and
615 // given ordering above, dividing by gets the value (integer division
616 // intended.) This could lead to ties/draws and/or multiple winners.
617 return (hand[1] / 2) * num_cards + (hand[0] / 2);
618 }
619}

621void LeducState::ResolveWinner() {
622 num_winners_ = kInvalidPlayer;
623
624 if (remaining_players_ == 1) {
625 // Only one left in? They get the pot!
626 for (Player player_index = 0; player_index < num_players_; player_index++) {
627 if (!folded_[player_index]) {
628 num_winners_ = 1;
629 winner_[player_index] = true;
630 money_[player_index] += pot_;
631 pot_ = 0;
632 return;
633 }
634 }
635
636 } else {
637 // Otherwise, showdown!
638 // Find the best hand among those still in.
639 SPIEL_CHECK_NE(public_card_, kInvalidCard);
640 int best_hand_rank = -1;
641 num_winners_ = 0;
642 std::fill(winner_.begin(), winner_.end(), false);
643
644 for (Player player_index = 0; player_index < num_players_; player_index++) {
645 if (!folded_[player_index]) {
646 int rank = RankHand(player_index);
647 if (rank > best_hand_rank) {
648 // Beat the current best hand! Clear the winners list, then add.
649 best_hand_rank = rank;
650 std::fill(winner_.begin(), winner_.end(), false);
651 winner_[player_index] = true;
652 num_winners_ = 1;
653 } else if (rank == best_hand_rank) {
654 // Tied with best hand rank, so this player is a winner as well.
655 winner_[player_index] = true;
656 num_winners_++;
657 }
658 }
659 }
660
661 // Split the pot among the winners (possibly only one).
662 SPIEL_CHECK_TRUE(1 <= num_winners_ && num_winners_ <= num_players_);
663 for (Player player_index = 0; player_index < num_players_; player_index++) {
664 if (winner_[player_index]) {
665 // Give this player their share.
666 money_[player_index] += static_cast<double>(pot_) / num_winners_;
667 }
668 }
669 pot_ = 0;
670 }
671}

701std::vector<int> LeducState::padded_betting_sequence() const {
702 std::vector<int> history = round1_sequence_;
703
704 // We pad the history to the end of the first round with kPaddingAction.
705 history.resize(game_->MaxGameLength() / 2, kInvalidAction);
706
707 // We insert the actions that happened in the second round, and fill to
708 // MaxGameLength.
709 history.insert(history.end(), round2_sequence_.begin(),
710 round2_sequence_.end());
711 history.resize(game_->MaxGameLength(), kInvalidAction);
712 return history;
713}

715void LeducState::SetPrivate(Player player, Action move) {
716 // Round 1. `move` refers to the card value to deal to the current
717 // underlying player (given by `private_cards_dealt_`).
718 if (suit_isomorphism_) {
719 // Consecutive cards are identical under suit isomorphism.
720 private_cards_[player] = move;
721 if (deck_[move * 2] != kInvalidCard) {
722 deck_[move * 2] = kInvalidCard;
723 } else if (deck_[move * 2 + 1] != kInvalidCard) {
724 deck_[move * 2 + 1] = kInvalidCard;
725 } else {
726 SpielFatalError("Suit isomorphism error.");
727 }
728 } else {
729 private_cards_[player] = deck_[move];
730 deck_[move] = kInvalidCard;
731 }
732 --deck_size_;
733 ++private_cards_dealt_;
734
735 // When all private cards are dealt, move to player 0.
736 if (private_cards_dealt_ == num_players_) cur_player_ = 0;
737}

739std::unique_ptr<State> LeducState::ResampleFromInfostate(
740 int player_id, std::function<double()> rng) const {
741 std::unique_ptr<State> clone = game_->NewInitialState();
742
743 // First, deal out cards:
744 Action player_chance = history_.at(player_id).action;
745 for (int p = 0; p < GetGame()->NumPlayers(); ++p) {
746 if (p == player_id) {
747 clone->ApplyAction(history_.at(p).action);
748 } else {
749 Action chosen_action = player_chance;
750 while (chosen_action == player_chance || chosen_action == public_card_) {
751 chosen_action = SampleAction(clone->ChanceOutcomes(), rng()).first;
752 }
753 clone->ApplyAction(chosen_action);
754 }
755 }
756 for (int action : round1_sequence_) clone->ApplyAction(action);
757 if (public_card_ != kInvalidCard) {
758 clone->ApplyAction(public_card_);
759 for (int action : round2_sequence_) clone->ApplyAction(action);
760 }
761 return clone;
762}

823double LeducGame::MaxUtility() const {
824 // In poker, the utility is defined as the money a player has at the end of
825 // the game minus then money the player had before starting the game.
826 // The most a player can win *per opponent* is the most each player can put
827 // into the pot, which is the raise amounts on each round times the maximum
828 // number raises, plus the original chip they put in to play.
829 return (num_players_ - 1) * (kTotalRaisesPerRound * kFirstRaiseAmount +
830 kTotalRaisesPerRound * kSecondRaiseAmount + 1);
831}
832
833double LeducGame::MinUtility() const {
834 // In poker, the utility is defined as the money a player has at the end of
835 // the game minus then money the player had before starting the game.
836 // The most any single player can lose is the maximum number of raises per
837 // round times the amounts of each of the raises, plus the original chip
838 // they put in to play.
839 return -1 * (kTotalRaisesPerRound * kFirstRaiseAmount +
840 kTotalRaisesPerRound * kSecondRaiseAmount + 1);
841}
