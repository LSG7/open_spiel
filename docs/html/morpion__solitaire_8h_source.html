45 struct Point {
46  int x{}, y{};
47  Point() = default;
48  Point(int a, int b) {
49  this->x = a;
50  this->y = b;
51  }
52 
53  bool operator==(const Point& other_point) const {
54  return (x == other_point.x) && (y == other_point.y);
55  }
56 
57  bool operator<(const Point& other_point) const {
58  if (x < other_point.x) {
59  return true;
60  } else if (x == other_point.x) {
61  if (y < other_point.y) {
62  return true;
63  }
64  }
65  return false;
66  }
67 
68  std::string ToString() const { return absl::StrCat("[", x, ",", y, "]"); }
69 };

134 // Game object.
135 class MorpionGame : public Game {
136  public:
137  explicit MorpionGame(const GameParameters& params);
138 
139  std::unique_ptr<State> NewInitialState() const override {
140  return std::unique_ptr<State>(new MorpionState(shared_from_this()));
141  }
142 
143  // Number of distinct actions equals all possible lines drawn on the board.
144  // Given 13x13 grid (see above), 4 points per line (4D),
145  // For line directions [0, 1], [1, 0]: 10 possible lines (13 - 3) per row and
146  // column. For line directions [1, -1], [1, 1]:
147  // - 10 lines (13 - 3) down center diagonal
148  // - 2 x (9 + 8 + .. 1) for other diagonals
149  // In total (10 * 13 * 2) + 2 * (10 + (2 * (9 + 8 +... 1))) = 460
150  int NumDistinctActions() const override { return 460; }
151 
152  // 4D fully solved by enumeration in 2008, with max 35 moves.
153  // http://www.morpionsolitaire.com/English/Enumeration.htm
154  // http://oeis.org/A204109
155  int MaxGameLength() const override { return 35; }
156 
157  int NumPlayers() const override { return 1; }
158  double MinUtility() const override { return 0; }
159  double MaxUtility() const override { return MaxGameLength(); }
160 };
