150// Game object.
151class DotsAndBoxesGame : public Game {
152 public:
153 explicit DotsAndBoxesGame(const GameParameters& params);
154 int NumDistinctActions() const override {
155 return (num_rows_ + 1) * num_cols_ + num_rows_ * (num_cols_ + 1);
156 }
157 std::unique_ptr<State> NewInitialState() const override {
158 return std::unique_ptr<State>(new DotsAndBoxesState(
159 shared_from_this(), num_rows_, num_cols_, utility_margin_));
160 }
161 std::unique_ptr<State> NewInitialState(
162 const std::string& str) const override {
163 return std::make_unique<DotsAndBoxesState>(shared_from_this(), num_rows_,
164 num_cols_, utility_margin_, str);
165 }
166 int NumPlayers() const override { return kNumPlayers; }
167 double MinUtility() const override;
168 absl::optional<double> UtilitySum() const override;
169 double MaxUtility() const override;
170 std::vector<int> ObservationTensorShape() const override {
171 return {3, num_cells_, 3};
172 }
173 int MaxGameLength() const override {
174 return (num_rows_ + 1) * num_cols_ + num_cols_ * (num_rows_ + 1);
175 }
176
177 private:
178 const int num_rows_;
179 const int num_cols_;
180 const int num_cells_;
181 const bool utility_margin_;
182};
