74void TwixTState::SetPegAndLinksOnTensor(absl::Span<float> values,
75 const Cell &cell, int offset, bool turn,
76 Position position) const {
77 TensorView<3> view(values, {kNumPlanes, board_.size(), board_.size() - 2},
78 false);
79 Position tensorPosition = board_.GetTensorPosition(position, turn);
80
81 if (cell.HasLinks()) {
82 for (int dir = 0; dir < 4; dir++) {
83 if (cell.HasLink(dir)) {
84 // peg has link in direction dir: set 1.0 on plane 1..4 / 7..10
85 view[{offset + 1 + dir, tensorPosition.x, tensorPosition.y}] = 1.0;
86 }
87 }
88 } else {
89 // peg has no links: set 1.0 on plane 0 / 6
90 view[{offset + 0, tensorPosition.x, tensorPosition.y}] = 1.0;
91 }
92
93 // peg has blocked neighbors: set 1.0 on plane 5 / 11
94 if (cell.HasBlockedNeighborsEast()) {
95 view[{offset + 5, tensorPosition.x, tensorPosition.y}] = 1.0;
96 }
97}

99void TwixTState::ObservationTensor(open_spiel::Player player,
100 absl::Span<float> values) const {
101 SPIEL_CHECK_GE(player, 0);
102 SPIEL_CHECK_LT(player, kNumPlayers);
103
104 const int kPlaneOffset[2] = {0, kNumPlanes / 2};
105 int size = board_.size();
106
107 // 2 x 6 planes of size boardSize x (boardSize-2):
108 // each plane excludes the endlines of the opponent
109 // plane 0/6 is for the pegs
110 // plane 1..4 / 7..10 is for the links NNE, ENE, ESE, SSE, resp.
111 // plane 5/11 is pegs that have blocked neighbors
112
113 TensorView<3> view(values, {kNumPlanes, board_.size(), board_.size() - 2},
114 true);
115
116 for (int c = 0; c < size; c++) {
117 for (int r = 0; r < size; r++) {
118 Position position = {c, r};
119 const Cell &cell = board_.GetConstCell(position);
120 int color = cell.color();
121 if (color == kRedColor) {
122 // no turn
123 SetPegAndLinksOnTensor(values, cell, kPlaneOffset[0], false, position);
124 } else if (color == kBlueColor) {
125 // 90 degr turn
126 SetPegAndLinksOnTensor(values, cell, kPlaneOffset[1], true, position);
127 }
128 }
129 }
130}
