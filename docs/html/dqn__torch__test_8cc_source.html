36 void TestSimpleGame() {
37  std::shared_ptr<const Game> game = efg_game::LoadEFGGame(
38  efg_game::GetSimpleForkEFGData());
39  SPIEL_CHECK_TRUE(game != nullptr);
40  DQNSettings settings = {
41  /*seed*/ kSeed,
42  /*use_observation*/ game->GetType().provides_observation_tensor,
43  /*player_id*/ 0,
44  /*state_representation_size*/ game->InformationStateTensorSize(),
45  /*num_actions*/ game->NumDistinctActions(),
46  /*hidden_layers_sizes*/ {16},
47  /*replay_buffer_capacity*/ 100,
48  /*batch_size*/ 5,
49  /*learning_rate*/ 0.01,
50  /*update_target_network_every*/ 20,
51  /*learn_every*/ 5,
52  /*discount_factor*/ 1.0,
53  /*min_buffer_size_to_learn*/ 5,
54  /*epsilon_start*/ 0.02,
55  /*epsilon_end*/ 0.01};
56  DQN dqn(settings);
57  int total_reward = 0;
58  std::unique_ptr<State> state;
59  for (int i = 0; i < 150; i++) {
60  state = game->NewInitialState();
61  while (!state->IsTerminal()) {
62  open_spiel::Action action = dqn.Step(*state);
63  state->ApplyAction(action);
64  total_reward += state->PlayerReward(0);
65  }
66  dqn.Step(*state);
67  }
68 
69  SPIEL_CHECK_GE(total_reward, 120);
70 }
