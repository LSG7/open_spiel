16"""Tic tac toe (noughts and crosses), implemented in Python.
17
18This is a demonstration of implementing a deterministic perfect-information
19game in Python.
20
21Python games are significantly slower than C++, but it may still be suitable
22for prototyping or for small games.
23
24It is possible to run C++ algorithms on Python-implemented games. This is likely
25to have good performance if the algorithm simply extracts a game tree and then
26works with that (e.g. CFR algorithms). It is likely to be poor if the algorithm
27relies on processing and updating states as it goes, e.g., MCTS.
28"""

64class TicTacToeGame(pyspiel.Game):
65 """A Python version of the Tic-Tac-Toe game."""
66
67 def __init__(self, params=None):
68 super().__init__(_GAME_TYPE, _GAME_INFO, params or dict())

70 def new_initial_state(self):
71 """Returns a state corresponding to the start of a game."""
72 return TicTacToeState(self)

83class TicTacToeState(pyspiel.State):
84 """A python version of the Tic-Tac-Toe state."""
85
86 def __init__(self, game):
87 """Constructor; should only be called by Game.new_initial_state."""
88 super().__init__(game)
89 self._cur_player = 0
90 self._player0_score = 0.0
91 self._is_terminal = False
92 self.board = np.full((_NUM_ROWS, _NUM_COLS), ".")
93
94 # OpenSpiel (PySpiel) API functions are below. This is the standard set that
95 # should be implemented by every perfect-information sequential-move game.
96
97 def current_player(self):
98 """Returns id of the next player to move, or TERMINAL if game is over."""
99 return pyspiel.PlayerId.TERMINAL if self._is_terminal else self._cur_player
100
101 def _legal_actions(self, player):
102 """Returns a list of legal actions, sorted in ascending order."""
103 return [a for a in range(_NUM_CELLS) if self.board[_coord(a)] == "."]
104
105 def _apply_action(self, action):
106 """Applies the specified action to the state."""
107 self.board[_coord(action)] = "x" if self._cur_player == 0 else "o"
108 if _line_exists(self.board):
109 self._is_terminal = True
110 self._player0_score = 1.0 if self._cur_player == 0 else -1.0
111 elif all(self.board.ravel() != "."):
112 self._is_terminal = True
113 else:
114 self._cur_player = 1 - self._cur_player
115
116 def _action_to_string(self, player, action):
117 """Action -> string."""
118 row, col = _coord(action)
119 return "{}({},{})".format("x" if player == 0 else "o", row, col)
120
121 def is_terminal(self):
122 """Returns True if the game is over."""
123 return self._is_terminal
124
125 def returns(self):
126 """Total reward for each player over the course of the game so far."""
127 return [self._player0_score, -self._player0_score]
128
129 def __str__(self):
130 """String for debug purposes. No particular semantics are required."""
131 return _board_to_string(self.board)

134class BoardObserver:
135 """Observer, conforming to the PyObserver interface (see observation.py)."""
136
137 def __init__(self, params):
138 """Initializes an empty observation tensor."""
139 if params:
140 raise ValueError(f"Observation parameters not supported; passed {params}")
141 # The observation should contain a 1-D tensor in `self.tensor` and a
142 # dictionary of views onto the tensor, which may be of any shape.
143 # Here the observation is indexed `(cell state, row, column)`.
144 shape = (1 + _NUM_PLAYERS, _NUM_ROWS, _NUM_COLS)
145 self.tensor = np.zeros(np.prod(shape), np.float32)
146 self.dict = {"observation": np.reshape(self.tensor, shape)}
147
148 def set_from(self, state, player):
149 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
150 del player
151 # We update the observation via the shaped tensor since indexing is more
152 # convenient than with the 1-D tensor. Both are views onto the same memory.
153 obs = self.dict["observation"]
154 obs.fill(0)
155 for row in range(_NUM_ROWS):
156 for col in range(_NUM_COLS):
157 cell_state = ".ox".index(state.board[row, col])
158 obs[cell_state, row, col] = 1
159
160 def string_from(self, state, player):
161 """Observation of `state` from the PoV of `player`, as a string."""
162 del player
163 return _board_to_string(state.board)

166# Helper functions for game details.

169def _line_value(line):
170 """Checks a possible line, returning the winning symbol if any."""
171 if all(line == "x") or all(line == "o"):
172 return line[0]
173
174
175def _line_exists(board):
176 """Checks if a line exists, returns "x" or "o" if so, and None otherwise."""
177 return (_line_value(board[0]) or _line_value(board[1]) or
178 _line_value(board[2]) or _line_value(board[:, 0]) or
179 _line_value(board[:, 1]) or _line_value(board[:, 2]) or
180 _line_value(board.diagonal()) or
181 _line_value(np.fliplr(board).diagonal()))
182
183
184def _coord(move):
185 """Returns (row, col) from an action id."""
186 return (move // _NUM_COLS, move % _NUM_COLS)
187
188
189def _board_to_string(board):
190 """Returns a string representation of the board."""
191 return "\n".join("".join(row) for row in board)
