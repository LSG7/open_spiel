33void MCCFR_2PGameTest(const std::string& game_name, std::mt19937* rng,
34 int iterations, double nashconv_upperbound) {
35 std::shared_ptr<const Game> game = LoadGame(game_name);
36 OutcomeSamplingMCCFRSolver solver(*game);
37 for (int i = 0; i < iterations; i++) {
38 solver.RunIteration(rng);
39 }
40 const std::shared_ptr<Policy> average_policy = solver.AveragePolicy();
41 double nash_conv = NashConv(*game, *average_policy, true);
42 std::cout << "Game: " << game_name << ", iters = " << iterations
43 << ", NashConv: " << nash_conv << std::endl;
44 SPIEL_CHECK_LE(nash_conv, nashconv_upperbound);
45}

47void MCCFR_SerializationTest() {
48 auto game = LoadGame("kuhn_poker");
49 OutcomeSamplingMCCFRSolver solver = OutcomeSamplingMCCFRSolver(*game);
50 double exploitability0 = Exploitability(*game, *solver.AveragePolicy());
51
52 for (int i = 0; i < 500; i++) {
53 solver.RunIteration();
54 }
55 double exploitability1 = Exploitability(*game, *solver.AveragePolicy());
56 SPIEL_CHECK_GT(exploitability0, exploitability1);
57
58 std::string serialized = solver.Serialize();
59 std::unique_ptr<OutcomeSamplingMCCFRSolver> deserialized_solver =
60 DeserializeOutcomeSamplingMCCFRSolver(serialized);
61 SPIEL_CHECK_EQ(solver.InfoStateValuesTable().size(),
62 deserialized_solver->InfoStateValuesTable().size());
63 double exploitability2 =
64 Exploitability(*game, *deserialized_solver->AveragePolicy());
65 SPIEL_CHECK_FLOAT_NEAR(exploitability1, exploitability2, 1e-15);
66
67 for (int i = 0; i < 500; i++) {
68 deserialized_solver->RunIteration();
69 }
70 double exploitability3 =
71 Exploitability(*game, *deserialized_solver->AveragePolicy());
72 SPIEL_CHECK_GT(exploitability2, exploitability3);
73}
