65 def _rnn_forwards(lstm_hidden_sizes: List[int], mlp_hidden_sizes: List[int],
66  batch_size: int) -> hk.Transformed:
67  """Returns a haiku transformation of the RNN model to be used in optimizer.
68 
69  Args:
70  lstm_hidden_sizes: List containing size of lstm layers.
71  mlp_hidden_sizes: List containing size of linear layers.
72  batch_size: Batch size.
73 
74  Returns:
75  Haiku transformation of the RNN network.
76  """
77  def forward_fn(inputs):
78  rnn = _make_rnn_network(lstm_hidden_sizes, mlp_hidden_sizes)
79  initial_state = rnn.initial_state(batch_size=batch_size)
80  outputs, _ = hk.dynamic_unroll(rnn, inputs, initial_state, time_major=False)
81  return outputs

96  def __call__(self, inputs, prev_state):
97  x = inputs
98  curr_state = [None] * len(prev_state)
99  for k, layer in enumerate(self._layers_layers):
100  if isinstance(layer, hk.RNNCore):
101  x, curr_state[k] = layer(x, prev_state[k])
102  else:
103  x = layer(x)
104  return x, tuple(curr_state)

107  layerwise_init_state = []
108  for layer in self._layers_layers:
