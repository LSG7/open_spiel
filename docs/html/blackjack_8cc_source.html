83 std::vector<double> BlackjackState::Returns() const {
84  if (!IsTerminal()) {
85  return {0};
86  }
87 
88  int player_total = GetBestPlayerTotal(kPlayerId);
89  int dealer_total = GetBestPlayerTotal(DealerId());
90  if (player_total > kApproachScore) {
91  // Bust.
92  return {-1};
93  } else if (dealer_total > kApproachScore) {
94  // Bust.
95  return {+1};
96  } else if (player_total > dealer_total) {
97  return {+1};
98  } else if (player_total < dealer_total) {
99  return {-1};
100  } else {
101  // Tie.
102  return {0};
103  }
104 }

112 void BlackjackState::ObservationTensor(Player player,
113  absl::Span<float> values) const {
114  std::fill(values.begin(), values.end(), 0);
115  int offset = 0;
116 
117  // Whose turn is it?
118  if (cur_player_ + 1 >= 0) { // do not support kTerminalPlayerId
119  values[cur_player_ + 1] = 1; // to support kChancePlayerId (equals to -1)
120  }
121  offset += game_->NumPlayers() + 1;
122 
123  // Terminal?
124  values[offset] = IsTerminal();
125  offset += 1;
126 
127  // Number of aces for each player (incl. dealer)
128  for (std::size_t player_id = 0; player_id < cards_.size(); player_id++) {
129  values[offset + num_aces_[player_id]] = 1.0;
130  offset += (kNumSuits + 1);
131  }
132 
133  // Cards used by each player (incl. dealer)
134  for (std::size_t player_id = 0; player_id < cards_.size(); player_id++) {
135  for (const int& card : cards_[player_id]) {
136  values[offset + card] = 1;
137  }
138  offset += kDeckSize;
139  }
140 
141  SPIEL_CHECK_EQ(offset, values.size());
142 }

161 void BlackjackState::DealCardToPlayer(int player, int card) {
162  // Remove card from deck.
163  auto new_end = std::remove(deck_.begin(), deck_.end(), card);
164  if (new_end == deck_.end()) SpielFatalError("Card not present in deck");
165  deck_.erase(new_end, deck_.end());
166 
167  cards_[player].push_back(card);
168  const int value = CardValue(card);
169  if (value == kAceValue) {
170  num_aces_[player]++;
171  } else {
172  non_ace_total_[player] += value;
173  }
174 }

193 int BlackjackState::GetBestPlayerTotal(int player) const {
194  // Return the max possible total <= kApproachScore, depending on hard or soft
195  // aces. 'Best' refers to the max non-bust score possible for the player.
196  // If it is not possible, some value > kApproachScore is returned.
197  int total = non_ace_total_[player] + num_aces_[player];
198  for (int i = 1; i <= num_aces_[player]; i++) {
199  int soft_total =
200  non_ace_total_[player] + i * 11 + (num_aces_[player] - i) * 1;
201  if (soft_total <= kApproachScore) {
202  total = std::max(total, soft_total);
203  }
204  }
205  return total;
206 }

223 void BlackjackState::DoApplyAction(Action move) {
224  SPIEL_CHECK_EQ(IsTerminal(), false);
225 
226  if (!InitialCardsDealt(DealerId())) {
227  // Still in the initial dealing phase. Deal the 'move' card to turn_player_.
228  SPIEL_CHECK_EQ(IsChanceNode(), true);
229 
230  DealCardToPlayer(turn_player_, move);
231  cur_player_ = kChancePlayerId;
232  if (InitialCardsDealt(turn_player_)) {
233  // Next player.
234  turn_player_++;
235  if (InitialCardsDealt(DealerId())) {
236  // Hit/stand part of the game commences.
237  turn_player_ = kPlayerId;
238  cur_player_ = kPlayerId;
239  }
240  }
241  return;
242  }
243 
244  if (IsChanceNode()) {
245  // Deal the 'move' card to turn_player_.
246  DealCardToPlayer(turn_player_, move);
247  cur_player_ = turn_player_;
248  if (GetBestPlayerTotal(turn_player_) > kApproachScore) {
249  if (turn_player_ != DealerId()) --live_players_;
250  EndPlayerTurn(turn_player_);
251  }
252  MaybeApplyDealerAction();
253  return;
254  }
255 
256  total_moves_++;
257  if (move == kHit) {
258  cur_player_ = kChancePlayerId;
259  } else if (move == kStand) {
260  EndPlayerTurn(turn_player_);
261  MaybeApplyDealerAction();
262  }
263 }

265 void BlackjackState::MaybeApplyDealerAction() {
266  // If there are no players still live, dealer doesn't play.
267  if (live_players_ == 0) {
268  EndPlayerTurn(DealerId());
269  }
270 
271  // Otherwise, hits 16 or less, stands on 17 or more.
272  if (cur_player_ == DealerId()) {
273  if (GetBestPlayerTotal(DealerId()) <= 16) {
274  cur_player_ = kChancePlayerId;
275  } else {
276  EndPlayerTurn(cur_player_);
277  }
278  }
279 }
