15 """Implements an Ephemeral Value Adjustment Agent.
16 
17 See https://arxiv.org/abs/1810.08163.
18 
19 The algorithm queries trajectories from a replay buffer based on similarities
20 to embedding representations and uses a parametric model to compute values for
21 counterfactual state-action pairs when integrating across those trajectories.
22 Finally, a weighted average between the parametric (DQN in this case) and the
23 non-parametric model is used to compute the policy.
24 """

48 class QueryableFixedSizeRingBuffer(dqn.ReplayBuffer):
49  """ReplayBuffer of fixed size with a FIFO replacement policy.
50 
51  Stored transitions can be sampled uniformly. This extends the DQN replay
52  buffer by allowing the contents to be fetched by L2 proximity to a query
53  value.
54 
55  The underlying datastructure is a ring buffer, allowing 0(1) adding and
56  sampling.
57  """

59  def knn(self, key, key_name, k, trajectory_len=1):
60  """Computes top-k neighbours based on L2 distance.
61 
62  Args:
63  key: (np.array) key value to query memory.
64  key_name: (str) attribute name of key in memory elements.
65  k: (int) number of neighbours to fetch.
66  trajectory_len: (int) length of trajectory to fetch from replay buffer.
67 
68  Returns:
69  List of tuples (L2 negative distance, BufferElement) sorted in increasing
70  order by the negative L2 distqances from the key.
71  """
72  distances = [(np.linalg.norm(getattr(sample, key_name) - key, 2,
73  axis=0), sample) for sample in self._data_data]
74  return sorted(distances, key=lambda v: -v[0])[:k]

322  def _trajectory_centric_planning(self, trajectories):
323  """Performs trajectory centric planning.
324 
325  Uses trajectories from the replay buffer to update the non-parametric values
326  while supplying counter-factual values with the parametric model.
327 
328  Args:
329  trajectories: Current OpenSpiel game state.
330  """
331  # Calculate non-parametric values over the trajectories.
332  # Iterate backward through trajectories
333  for t in range(len(trajectories) - 1, 0, -1):
334  elem = trajectories[t][1]
335  s_tp1 = tuple(elem.next_info_state)
336  s_t = tuple(elem.info_state)
337  a_t = elem.action
338  r_t = elem.reward
339  legal_actions = elem.legal_actions_mask
340  if t < len(trajectories) - 1:
341  for action in range(len(legal_actions)):
342  if not legal_actions[action]:
343  continue
344  if action == elem.action:
345  self._q_np_q_np[s_t][a_t] = (r_t + self._discount_discount * self._v_np_v_np[s_tp1])
346  else:
347  q_values_parametric = self._session_session.run(
348  self._agent_agent.q_values,
349  feed_dict={
350  self._agent_agent.info_state_ph:
351  np.expand_dims(elem.info_state, axis=0)
352  })
353  self._q_np_q_np[s_t][a_t] = q_values_parametric[0][action]
354 
355  # Set V(s_t)
356  if t == len(trajectories) - 1:
357  # Sample from the parametric model.
358  q_values_parametric = self._session_session.run(
359  self._agent_agent.q_values,
360  feed_dict={
361  self._agent_agent.info_state_ph:
362  np.expand_dims(elem.info_state, axis=0)
363  })
364  self._v_np_v_np[s_t] = np.max(q_values_parametric)
365  else:
366  self._v_np_v_np[s_t] = max(self._q_np_q_np[s_t])

368  def _epsilon_greedy(self, q_values, legal_actions, epsilon):
369  """Returns a valid epsilon-greedy action and valid action probs.
370 
371  Action probabilities are given by a softmax over legal q-values.
372 
373  Args:
374  q_values: list of Q-values by action.
375  legal_actions: list of legal actions at `info_state`.
376  epsilon: float, probability of taking an exploratory action.
377 
378  Returns:
379  A valid epsilon-greedy action and valid action probabilities.
380  """
381  probs = np.zeros(self._num_actions_num_actions)
382  q_values = np.array(q_values)
383  if np.random.rand() < epsilon:
384  action = np.random.choice(legal_actions)
385  probs[legal_actions] = 1.0 / len(legal_actions)
386  else:
387  legal_q_values = q_values[legal_actions]
388  action = legal_actions[np.argmax(legal_q_values)]
389  # Reduce max_q for numerical stability. Result is the same.
390  max_q = np.max(legal_q_values)
391  e_x = np.exp(legal_q_values - max_q)
392  probs[legal_actions] = e_x / e_x.sum(axis=0)
393  return action, probs

397  if is_evaluation:
398  return 0.0
