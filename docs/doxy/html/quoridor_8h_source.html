54struct Offset {
55 int x, y;
56
57 Offset(int x_, int y_) : x(x_), y(y_) {}
58
59 Offset operator+(const Offset& o) const { return Offset(x + o.x, y + o.y); }
60 Offset operator-(const Offset& o) const { return Offset(x - o.x, y - o.y); }
61 Offset operator*(const int i) const { return Offset(x * i, y * i); }
62 Offset rotate_left() const { return Offset(-y, x); }
63 Offset rotate_right() const { return Offset(y, -x); }
64};

66struct Move {
67 int x, y;
68 int xy; // Precomputed x + y * size.
69 int size;
70
71 Move() : x(0), y(0), xy(-1), size(-1) {}
72 Move(int x_, int y_, int size_)
73 : x(x_), y(y_), xy(x_ + (y_ * size_)), size(size_) {}
74
75 std::string ToString() const;
76
77 bool IsValid() const { return x >= 0 && y >= 0 && x < size && y < size; }
78 bool IsWall() const { return x & 1 || y & 1; }
79 bool IsHorizontalWall() const { return y & 1; }
80 bool IsVerticalWall() const { return x & 1; }
81
82 bool operator==(const Move& b) const { return xy == b.xy; }
83 bool operator!=(const Move& b) const { return xy != b.xy; }
84 bool operator<(const Move& b) const { return xy < b.xy; }
85
86 Move operator+(const Offset& o) const { return Move(x + o.x, y + o.y, size); }
87 Move operator-(const Offset& o) const { return Move(x - o.x, y - o.y, size); }
88 Offset operator-(const Move& o) const { return Offset(x - o.x, y - o.y); }
89};

162// Game object.
163class QuoridorGame : public Game {
164 public:
165 explicit QuoridorGame(const GameParameters& params);
166
167 int NumDistinctActions() const override { return Diameter() * Diameter(); }
168 std::unique_ptr<State> NewInitialState() const override {
169 return std::unique_ptr<State>(new QuoridorState(
170 shared_from_this(), board_size_, wall_count_, ansi_color_output_));
171 }
172 int NumPlayers() const override { return num_players_; }
173 int NumCellStates() const { return num_players_ + 1; }
174 double MinUtility() const override { return -1; }
175 absl::optional<double> UtilitySum() const override { return 0; }
176 double MaxUtility() const override { return 1; }
177 std::vector<int> ObservationTensorShape() const override {
178 return {NumCellStates() + num_players_, Diameter(), Diameter()};
179 }
180 int MaxGameLength() const override {
181 // There's no anti-repetition rule, so this could be infinite, but no sane
182 // agent would take more moves than placing all the walls and visiting
183 // all squares.
184 return kMaxGameLengthFactor * board_size_ * board_size_;
185 }
186
187 private:
188 int Diameter() const { return board_size_ * 2 - 1; }
189 const int board_size_;
190 const int wall_count_;
191 const bool ansi_color_output_ = false;
192 const int num_players_;
193};
