25 // Keeps track of a bunch of VPNet models, intended to be one per device, and
26 // gives them out based on usage. When you request a device you specify how much
27 // work you're going to give it, which is assumed done once the loan is
28 // returned.

61  // Gives the device with the fewest outstanding requests.
62  DeviceLoan Get(int requests, int device_id = -1) {
63  absl::MutexLock lock(&m_);
64  if (device_id < 0) {
65  device_id = 0;
66  for (int i = 1; i < devices.size(); ++i) {
67  if (devices[i].requests < devices[device_id].requests) {
68  device_id = i;
69  }
70  }
71  }
72  devices[device_id].requests += requests;
73  return DeviceLoan(this, &devices[device_id].model, device_id, requests);
74  }
75 
76  int Count() const { return devices.size(); }
77 
78  private:
79  void Return(int device_id, int requests) {
80  absl::MutexLock lock(&m_);
81  devices[device_id].requests -= requests;
82  }
