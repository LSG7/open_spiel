22 def _get_future_states(possibilities, state, reach=1.0):
23  """Does a lookahead over chance nodes to all next states after (s,a).
24 
25  Also works if there are no chance nodes (i.e. base case).
26 
27  Arguments:
28  possibilities: an empty list, that will be filled with (str(next_state),
29  transition probability) pairs for all possible next states
30  state: the state following some s.apply_action(a), can be a chance node
31  reach: chance reach probability of getting to this point from (s,a)
32  Returns: nothing.
33  """
34  if not state.is_chance_node() or state.is_terminal():
35  # Base case
36  possibilities.append((str(state), reach))
37  else:
38  assert state.is_chance_node()
39  for outcome, prob in state.chance_outcomes():
40  next_state = state.child(outcome)
41  _get_future_states(possibilities, next_state, reach * prob)

73 def value_iteration(game, depth_limit, threshold, cyclic_game=False):
74  """Solves for the optimal value function of a game.
75 
76  For small games only! Solves the game using value iteration,
77  with the maximum error for the value function less than threshold.
78  This algorithm works for sequential 1-player games or 2-player zero-sum
79  games, with or without chance nodes.
80 
81  Arguments:
82  game: The game to analyze, as returned by `load_game`.
83  depth_limit: How deeply to analyze the game tree. Negative means no limit, 0
84  means root-only, etc.
85  threshold: Maximum error for state values..
86  cyclic_game: set to True if the game has cycles (from state A we can get to
87  state B, and from state B we can get back to state A).
88 
89  Returns:
90  A `dict` with string keys and float values, mapping string encoding of
91  states to the values of those states.
92  """
93  assert game.num_players() in (1,
94  2), ("Game must be a 1-player or 2-player game")
95  if game.num_players() == 2:
96  assert game.get_type().utility == pyspiel.GameType.Utility.ZERO_SUM, (
97  "2-player games must be zero sum games")
98 
99  # Must be perfect information or one-shot (not imperfect information).
100  assert (game.get_type().information == pyspiel.GameType.Information.ONE_SHOT
101  or game.get_type().information ==
102  pyspiel.GameType.Information.PERFECT_INFORMATION)
103 
104  # We expect Value Iteration to be used with perfect information games, in
105  # which `str` is assumed to display the state of the game.
106  states = get_all_states.get_all_states(
107  game,
108  depth_limit,
109  True,
110  False,
111  to_string=str,
112  stop_if_encountered=cyclic_game)
113  values = {}
114  transitions = {}
115 
116  _initialize_maps(states, values, transitions)
117  error = threshold + 1 # A value larger than threshold
118  min_utility = game.min_utility()
119  while error > threshold:
120  error = 0
121  for key, state in states.items():
122  if state.is_terminal():
123  continue
124  elif state.is_simultaneous_node():
125  # Simultaneous node. Assemble a matrix game from the child utilities.
126  # and solve it using a matrix game solver.
127  p0_utils = [] # row player
128  p1_utils = [] # col player
129  row = 0
130  for p0action in state.legal_actions(0):
131  # new row
132  p0_utils.append([])
133  p1_utils.append([])
134  for p1action in state.legal_actions(1):
135  # loop from left-to-right of columns
136  next_states = transitions[(key, p0action, p1action)]
137  joint_q_value = sum(
138  p * values[next_state] for next_state, p in next_states)
139  p0_utils[row].append(joint_q_value)
140  p1_utils[row].append(-joint_q_value)
141  row += 1
142  stage_game = pyspiel.create_matrix_game(p0_utils, p1_utils)
143  solution = lp_solver.solve_zero_sum_matrix_game(stage_game)
144  value = solution[2]
145  else:
146  # Regular decision node
147  player = state.current_player()
148  value = min_utility if player == 0 else -min_utility
149  for action in state.legal_actions():
150  next_states = transitions[(key, action)]
151  q_value = sum(p * values[next_state] for next_state, p in next_states)
152  if player == 0:
153  value = max(value, q_value)
154  else:
155  value = min(value, q_value)
156  error = max(abs(values[key] - value), error)
157  values[key] = value
158 
159  return values
