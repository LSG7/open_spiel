25// This wrapper turns any n-player simultaneous move game into an equivalent
26// turn-based game where simultaneous move nodes are encoded as n turns.
27//
28// The underlying game must provide InformationStateString and
29// InformationStateTensor for the wrapped functions to work.
30//
31// TODO:
32// - implement UndoAction for these games.
33// - generalize to use Observation as well as Information state

102 std::vector<int> InformationStateTensorShape() const override {
103 // We flatten the representation of the underlying game and add one-hot
104 // indications of the to-play player and the observing player.
105 return {2 * NumPlayers() + game_->InformationStateTensorSize()};
106 }
107 std::vector<int> ObservationTensorShape() const override {
108 // We flatten the representation of the underlying game and add one-hot
109 // indications of the to-play player and the observing player.
110 return {2 * NumPlayers() + game_->ObservationTensorSize()};
111 }
112 int MaxGameLength() const override {
113 return game_->MaxGameLength() * NumPlayers();
114 }
115 int MaxChanceNodesInHistory() const override {
116 return game_->MaxChanceNodesInHistory();
117 }

119 private:
120 std::shared_ptr<const Game> game_;
