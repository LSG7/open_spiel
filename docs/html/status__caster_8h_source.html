26 // Convert NoThrowStatus by dispatching to a caster for StatusType with the
27 // argument throw_exception = false. StatusType should be an absl::Status
28 // (rvalue, lvalue, reference, or pointer), or an absl::StatusOr value.
29 // Only return values trigger exceptions, so NoThrowStatus has no meaning for
30 // input values. Therefore only C++->Python casting is supported.
31 template <typename StatusType>
32 struct type_caster<google::NoThrowStatus<StatusType>> {
33  using InputType = google::NoThrowStatus<StatusType>;
34  using StatusCaster = make_caster<StatusType>;
35  static constexpr auto name = NoThrowStatusType<StatusType>::name;
36 
37  // Convert C++->Python.
38  static handle cast(const InputType& src, return_value_policy policy,
39  handle parent) {
40  // pybind11::cast applies a const qualifier, so this takes a const reference
41  // argument. The qualifiers we care about are in StatusType, and we will
42  // forward those, but to do so, we must strip the const off the InputType.
43  return StatusCaster::cast(
44  std::forward<StatusType>(const_cast<InputType&>(src).status), policy,
45  parent, false);
46  }

74  handle parent, bool throw_exception) {
75  google::internal::CheckStatusModuleImported();
76  if (!throw_exception) {
77  // Use the built-in/standard pybind11 caster.
78  return type_caster_base<absl::Status>::cast(std::forward<CType>(src),
79  policy, parent);
80  } else if (!src.ok()) {
81  // Convert a non-ok status into an exception.
