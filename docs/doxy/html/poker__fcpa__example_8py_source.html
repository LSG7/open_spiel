34def LoadAgent(agent_type, game, player_id, rng):
35 """Return a bot based on the agent type."""
36 if agent_type == "random":
37 return uniform_random.UniformRandomBot(player_id, rng)
38 elif agent_type == "human":
39 return human.HumanBot()
40 elif agent_type == "check_call":
41 policy = pyspiel.PreferredActionPolicy([1, 0])
42 return pyspiel.make_policy_bot(game, player_id, FLAGS.seed, policy)
43 elif agent_type == "fold":
44 policy = pyspiel.PreferredActionPolicy([0, 1])
45 return pyspiel.make_policy_bot(game, player_id, FLAGS.seed, policy)
46 else:
47 raise RuntimeError("Unrecognized agent type: {}".format(agent_type))

50def main(_):
51 rng = np.random.RandomState(FLAGS.seed)
52
53 # Make sure poker is compiled into the library, as it requires an optional
54 # dependency: the ACPC poker code. To ensure it is compiled in, prepend both
55 # the install.sh and build commands with OPEN_SPIEL_BUILD_WITH_ACPC=ON.
56 # See here:
57 # https://github.com/deepmind/open_spiel/blob/master/docs/install.md#configuration-conditional-dependencies
58 # for more details on optional dependencies.
59 games_list = pyspiel.registered_names()
60 assert "universal_poker" in games_list
61
62 fcpa_game_string = pyspiel.hunl_game_string("fcpa")
63 print("Creating game: {}".format(fcpa_game_string))
64 game = pyspiel.load_game(fcpa_game_string)
65
66 agents = [
67 LoadAgent(FLAGS.player0, game, 0, rng),
68 LoadAgent(FLAGS.player1, game, 1, rng)
69 ]
70
71 state = game.new_initial_state()
72
73 # Print the initial state
74 print("INITIAL STATE")
75 print(str(state))
76
77 while not state.is_terminal():
78 # The state can be three different types: chance node,
79 # simultaneous node, or decision node
80 current_player = state.current_player()
81 if state.is_chance_node():
82 # Chance node: sample an outcome
83 outcomes = state.chance_outcomes()
84 num_actions = len(outcomes)
85 print("Chance node with " + str(num_actions) + " outcomes")
86 action_list, prob_list = zip(*outcomes)
87 action = rng.choice(action_list, p=prob_list)
88 print("Sampled outcome: ",
89 state.action_to_string(state.current_player(), action))
90 state.apply_action(action)
91 else:
92 # Decision node: sample action for the single current player
93 legal_actions = state.legal_actions()
94 for action in legal_actions:
95 print("Legal action: {} ({})".format(
96 state.action_to_string(current_player, action), action))
97 action = agents[current_player].step(state)
98 action_string = state.action_to_string(current_player, action)
99 print("Player ", current_player, ", chose action: ",
100 action_string)
101 state.apply_action(action)
102
103 print("")
104 print("NEXT STATE:")
105 print(str(state))
106
107 # Game is now done. Print utilities for each player
108 returns = state.returns()
109 for pid in range(game.num_players()):
110 print("Utility for player {} is {}".format(pid, returns[pid]))
