41 ActionsAndProbs SimpleGinRummyBot::GetPolicy(const State& state) {
42  ActionsAndProbs policy;
43  auto legal_actions = state.LegalActions(player_id_);
44  auto chosen_action = Step(state);
45  for (auto action : legal_actions)
46  policy.emplace_back(action, action == chosen_action ? 1.0 : 0.0);

60 Action SimpleGinRummyBot::Step(const State& state) {
61  std::vector<float> observation;
62  state.ObservationTensor(player_id_, &observation);
63 
64  std::vector<int> hand;
65  std::vector<int> layed_melds;
66  std::vector<int> discard_pile;
67  absl::optional<int> upcard = absl::nullopt;
68  int knock_card = 0;
69  int stock_size = 0;
70 
71  // Decode observation tensor.
72  int offset = 0;
73  SPIEL_CHECK_TRUE(observation[player_id_] == 1);
74  offset += kNumPlayers;
75  // Player hand.
76  if (player_id_ == 1) offset += kDefaultNumCards;
77  for (int i = 0; i < kDefaultNumCards; ++i) {
78  if (observation[offset + i] == 1) hand.push_back(i);
79  }
80  offset += kDefaultNumCards;
81  if (player_id_ == 0) offset += kDefaultNumCards;
82  // Current player.
83  SPIEL_CHECK_EQ(observation[offset + player_id_], 1);
84  offset += kNumPlayers;
85  // Knock card.
86  for (int i = 0; i < kDefaultKnockCard; ++i) {
87  if (observation[offset + i] == 1) knock_card += 1;
88  }
89  offset += kDefaultKnockCard;
90  // Upcard.
91  for (int i = 0; i < kDefaultNumCards; ++i) {
92  if (observation[offset + i] == 1) upcard = i;
93  }
94  offset += kDefaultNumCards;
95  // Discard pile.
96  for (int i = 0; i < kDefaultNumCards; ++i) {
97  if (observation[offset + i] == 1) discard_pile.push_back(i);
98  }
99  offset += kDefaultNumCards;
100  // Stock size.
101  for (int i = 0; i < kDefaultNumCards; ++i) {
102  if (observation[offset + i] == 1) stock_size += 1;
103  }
104  offset += kDefaultNumCards;
105  // Layed melds. Player 0 looks at player 1's layed melds and vice versa.
106  if (player_id_ == 0) offset += kNumMeldActions;
107  for (int i = 0; i < kNumMeldActions; ++i) {
108  if (observation[offset + i] == 1) {
109  layed_melds.push_back(i);
110  knocked_ = true;
111  }
112  } // Completed decoding observation.
113 
114  auto legal_actions = state.LegalActions(player_id_);
115  // Next actions must be legal, in order from back to front.
116  if (!next_actions_.empty()) {
117  Action action = next_actions_.back();
118  if (std::find(legal_actions.begin(), legal_actions.end(), action) ==
119  legal_actions.end()) {
120  std::cerr << "Game state:" << std::endl;
121  std::cerr << state.ToString() << std::endl;
122  std::cerr << "Legal actions: " << legal_actions << std::endl;
123  std::cerr << "Bot next actions: " << next_actions_ << std::endl;
124  SpielFatalError("Previously determined next action is illegal.");
125  }
126  next_actions_.pop_back();
127  return action;
128  }
129 
130  // When knocking, bot decides how to lay the hand all at once and saves the
131  // corresponding meld actions in next_actions_.
132  if (knocked_) {
133  if (!layed_melds.empty()) {
134  // Opponent knocked.
135  next_actions_.push_back(kPassAction); // Bot never lays off.
136  for (int meld_id : GetMelds(hand)) {
137  next_actions_.push_back(kMeldActionBase + meld_id);
138  }
139  next_actions_.push_back(kPassAction);
140  } else {
141  next_actions_.push_back(kPassAction);
142  std::vector<int> melds_to_lay = GetMelds(hand);
143  for (int meld_id : melds_to_lay) {
144  next_actions_.push_back(kMeldActionBase + meld_id);
145  }
146  int best_discard = GetDiscard(hand);
147  next_actions_.push_back(best_discard);
148  }
149  Action action = next_actions_.back();
150  SPIEL_CHECK_TRUE(std::find(legal_actions.begin(),
151  legal_actions.end(), action) != legal_actions.end());
152  next_actions_.pop_back();
153  return action;
154  } else if (!upcard.has_value()) {
155  // MoveType kDiscard
156  if (hand.size() != hand_size_ + 1) {
157  std::cerr << "Game state:" << std::endl;
158  std::cerr << state.ToString() << std::endl;
159  std::cerr << "Bot hand:" << std::endl;
160  std::cerr << utils_.HandToString(hand);
161  SpielFatalError("Discarding with an insufficient number of cards.");
162  }
163  int deadwood = utils_.MinDeadwood(hand);
164  if (deadwood <= knock_card && !knocked_) {
165  knocked_ = true;
166  return kKnockAction;
167  } else {
168  int best_discard = GetDiscard(hand);
169  if (best_discard >= 0) {
170  return best_discard;
171  } else {
172  return legal_actions[0];
173  }
174  }
175  } else {
176  // MoveType kDraw
177  if (stock_size == kWallStockSize) {
178  // Special rules apply when we've reached the wall.
179  if (legal_actions.back() == kKnockAction) {
180  knocked_ = true;
181  return kKnockAction;
182  } else {
183  return kPassAction;
184  }
185  } else if (utils_.MinDeadwood(hand, upcard) <= knock_card ||
186  !absl::c_linear_search(GetBestDeadwood(hand, upcard), upcard)) {
187  // Draw upcard if doing so permits a knock, or if the upcard would not be
188  // in the "best" deadwood (=> upcard would be in a "best" meld).
189  return kDrawUpcardAction;
190  } else {
191  return legal_actions.back(); // Draw from stock or pass.
192  }
193  }
194 }

196 // Returns the "best" deadwood, i.e. the cards that do not belong to one of the
197 // "best" melds. Here "best" means any meld group that achieves the lowest
198 // possible deadwood count for the given hand. In general this is non-unique.
199 std::vector<int> SimpleGinRummyBot::GetBestDeadwood(std::vector<int> hand,
200  const absl::optional<int> card) const {
201  if (card.has_value()) hand.push_back(card.value());
202  for (const auto& meld : utils_.BestMeldGroup(hand)) {
203  for (auto card : meld) {
204  hand.erase(remove(hand.begin(), hand.end(), card), hand.end());
205  }
206  }
207  return hand;
208 }

210 int SimpleGinRummyBot::GetDiscard(const std::vector<int> &hand) const {
211  std::vector<int> deadwood = GetBestDeadwood(hand);
212  if (!deadwood.empty()) {
213  std::sort(deadwood.begin(), deadwood.end(),
214  RankComparator(kDefaultNumRanks));
215  return deadwood.back();
216  } else {
217  // 11 card gin. All cards are melded so there is no deadwood to throw from.
218  // Must be careful to throw a card from a meld that does not break up that
219  // meld. E.g. consider an 11 card gin containing the meld As2s3s4s. With a
220  // knock card of 10, all of these cards are legal discards following a
221  // knock, but only the As and 4s preserve gin.
222  for (int i = 0; i < hand.size(); ++i) {
223  std::vector<int> hand_copy = hand;
224  hand_copy.erase(hand_copy.begin() + i);
225  if (utils_.MinDeadwood(hand_copy) == 0)
226  return hand[i];
227  }
228  SpielFatalError("11 card gin error.");
229  }
230 }

232 std::vector<int> SimpleGinRummyBot::GetMelds(std::vector<int> hand) const {
233  if (hand.size() == hand_size_ + 1 && utils_.MinDeadwood(hand) == 0) {
234  // 11 card gin. Must select discard that preserves gin. See GetDiscard().
235  hand.erase(remove(hand.begin(), hand.end(), GetDiscard(hand)), hand.end());
236  }
237  std::vector<int> rv;
238  for (const auto& meld : utils_.BestMeldGroup(hand)) {
239  rv.push_back(utils_.meld_to_int.at(meld));
240  }
241  return rv;
242 }
