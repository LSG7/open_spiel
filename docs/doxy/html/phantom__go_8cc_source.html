29const GameType kGameType{
30 /*short_name=*/"phantom_go",
31 /*long_name=*/"Phantom Go",
32 GameType::Dynamics::kSequential,
33 GameType::ChanceMode::kDeterministic,
34 GameType::Information::kImperfectInformation,
35 GameType::Utility::kZeroSum,
36 GameType::RewardModel::kTerminal,

40 /*provides_information_state_tensor=*/false,

63std::vector<VirtualPoint> HandicapStones(int num_handicap) {
64 if (num_handicap < 2 || num_handicap > 9) return {};
65
66 static std::array<VirtualPoint, 9> placement = {
67 {MakePoint("d4"), MakePoint("q16"), MakePoint("d16"), MakePoint("q4"),
68 MakePoint("d10"), MakePoint("q10"), MakePoint("k4"), MakePoint("k16"),
69 MakePoint("k10")}};
70 static VirtualPoint center = MakePoint("k10");
71
72 std::vector points(placement.begin(), placement.begin() + num_handicap);
73
74 if (num_handicap >= 5 && num_handicap % 2 == 1) {
75 points[num_handicap - 1] = center;
76 }
77
78 return points;
79}

89 void WriteTensor(const State &observed_state, int player,
90 Allocator *allocator) const override {
91 const PhantomGoState &state =
92 open_spiel::down_cast<const PhantomGoState &>(observed_state);
93
94 const int totalBoardPoints =
95 state.board().board_size() * state.board().board_size();
96
97 {
98 auto out = allocator->Get("stone-counts", {2});
99 auto stoneCount = state.GetStoneCount();
100 out.at(0) = stoneCount[0];
101 out.at(1) = stoneCount[1];
102 }
103
104 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
105 {
106 auto observation = state.board().GetObservationByID(player);
107
108 auto out_empty =
109 allocator->Get("player_observation_empty", {totalBoardPoints});
110 auto out_white =
111 allocator->Get("player_observation_white", {totalBoardPoints});
112 auto out_black =
113 allocator->Get("player_observation_black", {totalBoardPoints});
114 auto out_komi = allocator->Get("komi", {totalBoardPoints});
115
116 for (int i = 0; i < totalBoardPoints; i++) {
117 switch (observation[i]) {
118 case GoColor::kBlack:
119 out_black.at(i) = true;
120 out_white.at(i) = false;
121 out_empty.at(i) = false;
122 break;
123
124 case GoColor::kWhite:
125 out_black.at(i) = false;
126 out_white.at(i) = true;
127 out_empty.at(i) = false;
128 break;
129
130 case GoColor::kEmpty:
131 out_black.at(i) = false;
132 out_white.at(i) = false;
133 out_empty.at(i) = true;
134 break;
135
136 default:
137 SpielFatalError(absl::StrCat("Unhandled case: ", observation[i]));
138 }
139 if (state.CurrentPlayer() == (uint8_t)GoColor::kWhite) {
140 out_komi.at(i) = 1;
141 } else {
142 out_komi.at(i) = 0;
143 }
144 }
145 }
146 }
147 }

187std::vector<Action> PhantomGoState::LegalActions() const {
188 std::vector<Action> actions{};
189 if (IsTerminal()) return actions;
190 for (VirtualPoint p : BoardPoints(board_.board_size())) {
191 if (board_.IsLegalMove(p, to_play_)) {
192 actions.push_back(board_.VirtualActionToAction(p));
193 }
194 }
195 actions.push_back(board_.pass_action());
196 return actions;
197}

231bool PhantomGoState::IsTerminal() const {
232 if (history_.size() < 2) return false;
233 return (history_.size() >= max_game_length_) || superko_ ||
234 (history_[history_.size() - 1].action == board_.pass_action() &&
235 history_[history_.size() - 2].action == board_.pass_action());
236}

238std::vector<double> PhantomGoState::Returns() const {
239 if (!IsTerminal()) return {0.0, 0.0};
240
241 if (superko_) {
242 // Superko rules (https://senseis.xmp.net/?Superko) are complex and vary
243 // between rulesets.
244 // For simplicity and because superkos are very rare, we just treat them as
245 // a draw.
246 return {DrawUtility(), DrawUtility()};
247 }
248
249 // Score with Tromp-Taylor.
250 float black_score = TrompTaylorScore(board_, komi_, handicap_);
251
252 std::vector<double> returns(phantom_go::NumPlayers());
253 if (black_score > 0) {
254 returns[ColorToPlayer(GoColor::kBlack)] = WinUtility();
255 returns[ColorToPlayer(GoColor::kWhite)] = LossUtility();
256 } else if (black_score < 0) {
257 returns[ColorToPlayer(GoColor::kBlack)] = LossUtility();
258 returns[ColorToPlayer(GoColor::kWhite)] = WinUtility();
259 } else {
260 returns[ColorToPlayer(GoColor::kBlack)] = DrawUtility();
261 returns[ColorToPlayer(GoColor::kWhite)] = DrawUtility();
262 }
263 return returns;
264}

270void PhantomGoState::UndoAction(Player player, Action action) {
271 // We don't have direct undo functionality, but copying the board and
272 // replaying all actions is still pretty fast (> 1 million undos/second).
273 history_.pop_back();
274 --move_number_;
275 ResetBoard();
276 for (auto [_, action] : history_) {
277 DoApplyAction(action);
278 }
279}

281void PhantomGoState::DoApplyAction(Action action) {
282 if (board_.PlayMove(board_.ActionToVirtualAction(action), to_play_)) {
283 to_play_ = OppColor(to_play_);
284 bool was_inserted = repetitions_.insert(board_.HashValue()).second;
285 if (!was_inserted && action != board_.pass_action()) {
286 // We have encountered this position before.
287 superko_ = true;
288 }
289 }
290}

292void PhantomGoState::ResetBoard() {
293 board_.Clear();
294 if (handicap_ < 2) {
295 to_play_ = GoColor::kBlack;
296 } else {
297 for (VirtualPoint p : HandicapStones(handicap_)) {
298 board_.PlayMove(p, GoColor::kBlack);
299 }
300 to_play_ = GoColor::kWhite;
301 }
302
303 repetitions_.clear();
304 repetitions_.insert(board_.HashValue());
305 superko_ = false;
306}

310bool PhantomGoState::equalMetaposition(const PhantomGoState &state1,
311 const PhantomGoState &state2,
312 int playerID) {
313 if (state1.board_.board_size() != state2.board_.board_size()) {
314 return false;
315 }
316
317 std::array<int, 2> stoneCount1 = state1.board_.GetStoneCount();
318 std::array<int, 2> stoneCount2 = state2.board_.GetStoneCount();
319
320 if (stoneCount1[0] != stoneCount2[0] || stoneCount1[1] != stoneCount2[1]) {
321 return false;
322 }
323
324 int boardSize = state1.board_.board_size();
325
326 auto observation1 = state1.board_.GetObservationByID(playerID);
327 auto observation2 = state2.board_.GetObservationByID(playerID);
328
329 for (int i = 0; i < boardSize * boardSize; i++) {
330 if (observation1[i] != observation2[i]) {
331 return false;
332 }
333 }
334
335 if (state1.to_play_ != state2.to_play_) {
336 return false;
337 }
338
339 return true;
340}
