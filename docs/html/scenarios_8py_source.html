63 def play_bot_in_scenarios(game, bots, scenarios=None):
64  """Plays a bot against a number of scenarios.
65 
66  Args:
67  game: The game the bot is playing.
68  bots: A list of length game.num_players() of pyspiel.Bots (or equivalent).
69  Must implement the apply_action and step methods.
70  scenarios: The scenarios we evaluate the bot in. A List[Scenario].
71 
72  Returns:
73  A dict mapping scenarios to their scores (with an additional "mean_score"
74  field containing the mean score across all scenarios).
75  The average score across all scenarios.
76  """
77  if scenarios is None:
78  scenarios = get_default_scenarios(game.get_type().short_name)
79 
80  results = []
81  total_score = 0
82  for scenario in scenarios:
83  state = game.new_initial_state()
84  bot = bots[scenario.player_id]
85  bot.restart()
86  for action_str in scenario.init_actions:
87  action = state.string_to_action(action_str)
88  if state.current_player() == scenario.player_id:
89  bot.force_action(state, action)
90  state.apply_action(action)
91  actions_and_probs, _ = bot.step(state)
92  expected_action = state.string_to_action(scenario.expected_action_str)
93  for action, prob in actions_and_probs:
94  if action == expected_action:
95  actual_prob = prob
96  break
97  score = 1 - abs(actual_prob - scenario.expected_prob)
98  results.append((scenario.name, score, scenario.expected_action_str,
99  scenario.expected_prob, actual_prob))
100  total_score += score
101 
102  if scenarios:
103  total_score /= len(scenarios)
104  logging.info("Average score across all scenarios: %.4f.", total_score)
105  results_dict = {}
106  for name, score, expected_action, expected_prob, actual_prob in results:
107  logging.info("************************************************************")
108  logging.info("Scenario: '%s'. Score: %.4f.", name, score)
109  logging.info("Expected action %s with probability %.4f but assigned %.4f.",
110  expected_action, expected_prob, actual_prob)
111  logging.info("***************************")
112  results_dict["scenario_score: " + name] = score
113  results_dict["mean_score"] = total_score
114  return results_dict
