31void CalcSingleCommon(
32 const int thrId,
33 const int bno)
34{
35 // Solves a single deal and strain for all four declarers.
36
37 futureTricks fut;
38 cparam.bop->deals[bno].first = 0;
39
40 START_THREAD_TIMER(thrId);
41 int res = SolveBoard(
42 cparam.bop->deals[bno],
43 cparam.bop->target[bno],
44 cparam.bop->solutions[bno],
45 cparam.bop->mode[bno],
46 &fut,
47 thrId);
48
49 // SH: I'm making a terrible use of the fut structure here.
50
51 if (res == 1)
52 cparam.solvedp->solvedBoard[bno].score[0] = fut.score[0];
53 else
54 cparam.error = res;
55
56 ThreadData * thrp = memory.GetPtr(static_cast<unsigned>(thrId));
57 for (int k = 1; k < DDS_HANDS; k++)
58 {
59 int hint = (k == 2 ? fut.score[0] : 13 - fut.score[0]);
60
61 cparam.bop->deals[bno].first = k; // Next declarer
62
63 res = SolveSameBoard(thrp, cparam.bop->deals[bno], &fut, hint);
64
65 if (res == 1)
66 cparam.solvedp->solvedBoard[bno].score[k] = fut.score[0];
67 else
68 cparam.error = res;
69 }
70 END_THREAD_TIMER(thrId);
71}

75{
76 for (unsigned i = 0; i < crossrefs.size(); i++)
77 {
78 if (crossrefs[i] == -1)
79 continue;

83 cparam.solvedp->solvedBoard[i].score[k] =

85 END_THREAD_TIMER(thrId);
86 }

90void CalcChunkCommon(
91 const int thrId)
92{
93 // Solves each deal and strain for all four declarers.
94 vector<futureTricks> fut;
95 fut.resize(static_cast<unsigned>(cparam.noOfBoards));
96
97 int index;
98 schedType st;
99
100 while (1)
101 {
102 st = scheduler.GetNumber(thrId);
103 index = st.number;
104 if (index == -1)
105 break;
106
107 if (st.repeatOf != -1)
108 {
109 START_THREAD_TIMER(thrId);
110 for (int k = 0; k < DDS_HANDS; k++)
111 {
112 cparam.bop->deals[index].first = k;
113
114 cparam.solvedp->solvedBoard[index].score[k] =
115 cparam.solvedp->solvedBoard[ st.repeatOf ].score[k];
116 }
117 END_THREAD_TIMER(thrId);
118 continue;
119 }
120
121 CalcSingleCommon(thrId, index);
122 }
123}

126int CalcAllBoardsN(
127 boards * bop,
128 solvedBoards * solvedp)
129{
130 cparam.error = 0;
131
132 if (bop->noOfBoards > MAXNOOFBOARDS)
133 return RETURN_TOO_MANY_BOARDS;
134
135 cparam.bop = bop;
136 cparam.solvedp = solvedp;
137 cparam.noOfBoards = bop->noOfBoards;
138
139 scheduler.RegisterRun(DDS_RUN_CALC, * bop);
140 sysdep.RegisterRun(DDS_RUN_CALC, * bop);
141
142 for (int k = 0; k < MAXNOOFBOARDS; k++)
143 solvedp->solvedBoard[k].cards = 0;
144
145 START_BLOCK_TIMER;
146 int retRun = sysdep.RunThreads();
147 END_BLOCK_TIMER;
148
149 if (retRun != RETURN_NO_FAULT)
150 return retRun;
151
152 solvedp->noOfBoards = cparam.noOfBoards;
153
154#ifdef DDS_SCHEDULER
155 scheduler.PrintTiming();
156#endif
157
158 if (cparam.error == 0)
159 return RETURN_NO_FAULT;
160 else
161 return cparam.error;
162}

166int STDCALL CalcDDtable(
167 ddTableDeal tableDeal,
168 ddTableResults * tablep)
169{
170 deal dl;
171 boards bo;
172 solvedBoards solved;
173
174 for (int h = 0; h < DDS_HANDS; h++)
175 for (int s = 0; s < DDS_SUITS; s++)
176 dl.remainCards[h][s] = tableDeal.cards[h][s];
177
178 for (int k = 0; k <= 2; k++)
179 {
180 dl.currentTrickRank[k] = 0;
181 dl.currentTrickSuit[k] = 0;
182 }
183
184 int ind = 0;
185 bo.noOfBoards = DDS_STRAINS;
186
187 for (int tr = DDS_STRAINS-1; tr >= 0; tr--)
188 {
189 dl.trump = tr;
190 bo.deals[ind] = dl;
191 bo.target[ind] = -1;
192 bo.solutions[ind] = 1;
193 bo.mode[ind] = 1;
194 ind++;
195 }
196
197 int res = CalcAllBoardsN(&bo, &solved);
198 if (res != 1)
199 return res;
200
201 for (int index = 0; index < DDS_STRAINS; index++)
202 {
203 int strain = bo.deals[index].trump;
204
205 // SH: I'm making a terrible use of the fut structure here.
206
207 for (int first = 0; first < DDS_HANDS; first++)
208 {
209 tablep->resTable[strain][ rho[first] ] =
210 13 - solved.solvedBoard[index].score[first];
211 }
212 }
213 return RETURN_NO_FAULT;
214}

217int STDCALL CalcAllTables(
218 ddTableDeals * dealsp,
219 int mode,
220 int trumpFilter[5],
221 ddTablesRes * resp,
222 allParResults * presp)
223{
224 /* mode = 0: par calculation, vulnerability None
225 mode = 1: par calculation, vulnerability All
226 mode = 2: par calculation, vulnerability NS
227 mode = 3: par calculation, vulnerability EW
228 mode = -1: no par calculation */
229
230 boards bo;
231 solvedBoards solved;
232 int count = 0;
233 bool okey = false;
234
235 for (int k = 0; k < DDS_STRAINS; k++)
236 {
237 if (!trumpFilter[k])
238 {
239 okey = true;
240 count++;
241 }
242 }
243
244 if (!okey)
245 return RETURN_NO_SUIT;
246
247 if (count * dealsp->noOfTables > MAXNOOFTABLES * DDS_STRAINS)
248 return RETURN_TOO_MANY_TABLES;
249
250 int ind = 0;
251 int lastIndex = 0;
252 resp->noOfBoards = 0;
253
254 for (int m = 0; m < dealsp->noOfTables; m++)
255 {
256 for (int tr = DDS_STRAINS-1; tr >= 0; tr--)
257 {
258 if (trumpFilter[tr])
259 continue;
260
261 for (int h = 0; h < DDS_HANDS; h++)
262 for (int s = 0; s < DDS_SUITS; s++)
263 bo.deals[ind].remainCards[h][s] =
264 dealsp->deals[m].cards[h][s];
265
266 bo.deals[ind].trump = tr;
267
268 for (int k = 0; k <= 2; k++)
269 {
270 bo.deals[ind].currentTrickRank[k] = 0;
271 bo.deals[ind].currentTrickSuit[k] = 0;
272 }
273
274 bo.target[ind] = -1;
275 bo.solutions[ind] = 1;
276 bo.mode[ind] = 1;
277 lastIndex = ind;
278 ind++;
279 }
280 }
281
282 bo.noOfBoards = lastIndex + 1;
283
284 int res = CalcAllBoardsN(&bo, &solved);
285 if (res != 1)
286 return res;
287
288 resp->noOfBoards += 4 * solved.noOfBoards;
289
290 for (int m = 0; m < dealsp->noOfTables; m++)
291 {
292 for (int strainIndex = 0; strainIndex < count; strainIndex++)
293 {
294 int index = m * count + strainIndex;
295 int strain = bo.deals[index].trump;
296
297 // SH: I'm making a terrible use of the fut structure here.
298
299 for (int first = 0; first < DDS_HANDS; first++)
300 {
301 resp->results[m].resTable[strain][ rho[first] ] =
302 13 - solved.solvedBoard[index].score[first];
303 }
304 }
305 }
306
307 if ((mode > -1) && (mode < 4) && (count == 5))
308 {
309 /* Calculate par */
310 for (int k = 0; k < dealsp->noOfTables; k++)
311 {
312 res = Par(&(resp->results[k]), &(presp->presults[k]), mode);
313 /* vulnerable 0: None 1: Both 2: NS 3: EW */
314 if (res != 1)
315 return res;
316 }
317 }
318 return RETURN_NO_FAULT;
319}

322int STDCALL CalcAllTablesPBN(
323 ddTableDealsPBN * dealsp,
324 int mode,
325 int trumpFilter[5],
326 ddTablesRes * resp,
327 allParResults * presp)
328{
329 ddTableDeals dls;
330 for (int k = 0; k < dealsp->noOfTables; k++)
331 if (ConvertFromPBN(dealsp->deals[k].cards, dls.deals[k].cards) != 1)
332 return RETURN_PBN_FAULT;
333
334 dls.noOfTables = dealsp->noOfTables;
335
336 int res = CalcAllTables(&dls, mode, trumpFilter, resp, presp);
337 return res;
338}
