220 void CoopBoxPushingState::MoveForward(Player player) {
221  SPIEL_CHECK_GE(player, 0);
222  SPIEL_CHECK_LE(player, 1);
223 
224  OrientationType dir = player_orient_[player];
225  auto next = NextCoord(player_coords_[player], dir);
226 
227  if (!InBounds(next)) {
228  // Bump.. out of bounds!
229  AddReward(kBumpPenalty);
230  } else if (field(next) == '.') {
231  // Uninterrupted move.
232  SetField(player_coords_[player], '.');
233  SetPlayer(next, player);
234  } else if (field(next) == 'b') {
235  auto next_next = NextCoord(next, dir);
236  if (!InBounds(next_next)) {
237  // Bump, can't push box out of bounds!
238  AddReward(kBumpPenalty);
239  } else if (field(next_next) == '.') {
240  // Move the small box.
241  SetField(next_next, 'b');
242  SetField(player_coords_[player], '.');
243  SetPlayer(next, player);
244 
245  // Check for reward.
246  if (next_next.first == 0 && next.first != 0) {
247  AddReward(kSmallBoxReward);
248  }
249  } else {
250  // Trying to move box into something else.. bump!
251  AddReward(kBumpPenalty);
252  }
253  } else {
254  // Also bump!
255  AddReward(kBumpPenalty);
256  }
257 }

264 void CoopBoxPushingState::ResolveMoves() {
265  // Check for successful move of the big box.
266  if (moves_[0] == ActionType::kMoveForward &&
267  moves_[1] == ActionType::kMoveForward &&
268  action_status_[0] == ActionStatusType::kSuccess &&
269  action_status_[1] == ActionStatusType::kSuccess) {
270  std::array<std::pair<int, int>, 2> next_coords;
271  std::array<std::pair<int, int>, 2> next_next_coords;
272 
273  next_coords[0] = NextCoord(player_coords_[0], player_orient_[0]);
274  next_coords[1] = NextCoord(player_coords_[1], player_orient_[1]);
275  next_next_coords[0] = NextCoord(next_coords[0], player_orient_[0]);
276  next_next_coords[1] = NextCoord(next_coords[1], player_orient_[1]);
277 
278  if (InBounds(next_coords[0]) && InBounds(next_coords[1]) &&
279  InBounds(next_next_coords[0]) && InBounds(next_next_coords[1]) &&
280  field(next_coords[0]) == 'B' && field(next_coords[1]) == 'B' &&
281  field(next_next_coords[0]) == '.' &&
282  field(next_next_coords[1]) == '.') {
283  SetField(next_next_coords[0], 'B');
284  SetField(next_next_coords[1], 'B');
285  SetField(player_coords_[0], '.');
286  SetField(player_coords_[1], '.');
287  SetPlayer(next_coords[0], 0);
288  SetPlayer(next_coords[1], 1);
289 
290  if (next_next_coords[0].first == 0 && next_coords[0].first != 0) {
291  AddReward(kBigBoxReward);
292  win_ = true;
293  return;
294  }
295  }
296  }
297 
298  // Otherwise, just resolve them independently.
299  for (int i = 0; i < 2; i++) {
300  // Player order depends on initiative.
301  int p = (i + initiative_) % 2;
302 
303  SPIEL_CHECK_GE(p, 0);
304  SPIEL_CHECK_LT(p, 2);
305  SPIEL_CHECK_TRUE(action_status_[p] != ActionStatusType::kUnresolved);
306 
307  ActionType move = moves_[p];
308 
309  // Action failed or deliberate stay => nothing happens to this agent.
310  if (action_status_[p] == ActionStatusType::kFail ||
311  move == ActionType::kStay) {
312  continue;
313  }
314 
315  if (move == ActionType::kTurnLeft || move == ActionType::kTurnRight) {
316  SetPlayer(player_coords_[p], p, Rotate(player_orient_[p], move));
317  } else if (move == ActionType::kMoveForward) {
318  MoveForward(p);
319  }
320  }
321 
322  // Reset the action statuses and current player.
323  cur_player_ = kSimultaneousPlayerId;
324  action_status_[0] = ActionStatusType::kUnresolved;
325  action_status_[1] = ActionStatusType::kUnresolved;
326 
327  AddReward(kDelayPenalty);
328  total_moves_++;
329 }

331 void CoopBoxPushingState::DoApplyAction(Action action) {
332  reward_ = 0;
333  if (IsSimultaneousNode()) return ApplyFlatJointAction(action);
334 
335  if (action == kChanceSuccess) {
336  // Success.
337  if (action_status_[0] == ActionStatusType::kUnresolved) {
338  action_status_[0] = ActionStatusType::kSuccess;
339  } else if (action_status_[1] == ActionStatusType::kUnresolved) {
340  action_status_[1] = ActionStatusType::kSuccess;
341  } else {
342  SpielFatalError(absl::StrCat("Invalid chance move case: ", action));
343  }
344  } else if (action == kChanceFail) {
345  // Fail!
346  if (action_status_[0] == ActionStatusType::kUnresolved) {
347  action_status_[0] = ActionStatusType::kFail;
348  } else if (action_status_[1] == ActionStatusType::kUnresolved) {
349  action_status_[1] = ActionStatusType::kFail;
350  } else {
351  SpielFatalError(absl::StrCat("Invalid chance move case: ", action));
352  }
353  } else if (action == kChanceInit1) {
354  // Player 1 moves first.
355  initiative_ = 0;
356  ResolveMoves();
357  } else {
358  // Player 2 moves first.
359  initiative_ = 1;
360  ResolveMoves();
361  }
362 }

364 std::vector<Action> CoopBoxPushingState::LegalActions(Player player) const {
365  if (player == kSimultaneousPlayerId) {
366  return LegalFlatJointActions();
367  } else if (IsTerminal()) {
368  return {};
369  } else if (IsChanceNode()) {
370  if (action_status_[0] == ActionStatusType::kUnresolved ||
371  action_status_[1] == ActionStatusType::kUnresolved) {
372  return {0, 1};
373  } else {
374  return {2, 3};
375  }
376  }
377  // All the actions are legal at every state.
378  return {0, 1, 2, 3};
379 }

381 ActionsAndProbs CoopBoxPushingState::ChanceOutcomes() const {
382  SPIEL_CHECK_TRUE(IsChanceNode());
383 
384  if (action_status_[0] == ActionStatusType::kUnresolved ||
385  action_status_[1] == ActionStatusType::kUnresolved) {
386  // Determine success (0) or failure (1) of a player's action.
387  return {std::pair<Action, double>(0, 0.9),
388  std::pair<Action, double>(1, 0.1)};
389  } else {
390  // Determine initiative outcomes (2 and 3)
391  return {std::pair<Action, double>(2, 0.5),
392  std::pair<Action, double>(3, 0.5)};

413 ObservationType CoopBoxPushingState::PartialObservation(Player player) const {
414  std::pair<int, int> adj_coord = {
415  player_coords_[player].first + row_offsets[player_orient_[player]],
416  player_coords_[player].second + col_offsets[player_orient_[player]]};
417 
418  if (!InBounds(adj_coord)) {
419  return kWallObs;
420  } else {
421  switch (field(adj_coord)) {
422  case kField:
423  return kEmptyFieldObs;
424  case kLeft:
425  case kRight:
426  case kUp:
427  case kDown:
428  return kOtherAgentObs;
429  case kSmallBox:
430  return kSmallBoxObs;
431  case kBigBox:
432  return kBigBoxObs;
433  default:
434  SpielFatalError("Unrecognized field char: " +
435  std::to_string(field(adj_coord)));
436  }
437  }

483 int CoopBoxPushingState::ObservationPlane(std::pair<int, int> coord,
484  Player player) const {
485  int plane = 0;
486  switch (field(coord)) {
487  case kField:
488  plane = 0;
489  break;
490  case kSmallBox:
491  plane = 1;
492  break;
493  case kBigBox:
494  plane = 2;
495  break;
496  case kLeft:
497  plane = (SameAsPlayer(coord, player)) ? 3 : 4;
498  break;
499  case kRight:
500  plane = (SameAsPlayer(coord, player)) ? 5 : 6;
501  break;
502  case kUp:
503  plane = (SameAsPlayer(coord, player)) ? 7 : 8;
504  break;
505  case kDown:
506  plane = (SameAsPlayer(coord, player)) ? 9 : 10;
507  break;
508  default:
509  std::cerr << "Invalid character on field: " << field(coord) << std::endl;
510  std::cerr << ToString() << std::endl;
511  plane = -1;
512  break;
513  }
514 
515  return plane;
516 }

518 void CoopBoxPushingState::ObservationTensor(Player player,
519  absl::Span<float> values) const {
520  SPIEL_CHECK_GE(player, 0);
521  SPIEL_CHECK_LT(player, num_players_);
522  if (fully_observable_) {
523  TensorView<3> view(values, {kCellStates, kRows, kCols}, true);
524 
525  for (int r = 0; r < kRows; r++) {
526  for (int c = 0; c < kCols; c++) {
527  int plane = ObservationPlane({r, c}, player);
528  SPIEL_CHECK_TRUE(plane >= 0 && plane < kCellStates);
529  view[{plane, r, c}] = 1.0;
530  }
531  }
532  } else {
533  SPIEL_CHECK_EQ(values.size(), kNumObservations);
534  std::fill(values.begin(), values.end(), 0);
535  ObservationType obs = PartialObservation(player);
536  values[obs] = 1;
537  }
