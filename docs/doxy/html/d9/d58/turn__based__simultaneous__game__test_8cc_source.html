36// An n-player version of (repeated) matching pennies with n rounds. On round
37// j, the j^th player has no legal moves, and every other player can play heads
38// or tails. On the j^th round: a matching pennies games is being played between
39// players j+1 (mod n) and j+2 (mod n), i.e. cumulative utilities are adjusted
40// based on the matching pennies matrix betwen those players, and the other
41// players' actions have no effect on the return.
42//
43// This game was specifically designed to test the case of some players not
44// having any legal actions at simultaneous nodes, which are then skipped over
45// when converted to turn-based games.
46const GameType kGameType{/*short_name=*/"mprmp",
47 /*long_name=*/"Missing Player Repeated MP",
48 GameType::Dynamics::kSimultaneous,
49 GameType::ChanceMode::kDeterministic,
50 GameType::Information::kPerfectInformation,
51 GameType::Utility::kZeroSum,
52 GameType::RewardModel::kTerminal,
53 /*max_num_players=*/2,
54 /*min_num_players=*/2,
55 /*provides_information_state_string=*/true,
56 /*provides_information_state_tensor=*/false,
57 /*provides_observation_string=*/false,
58 /*provides_observation_tensor=*/false,
59 /*parameter_specification=*/
60 {{"num_players", GameParameter(4)}}};

62class MissingPlayerRepeatedMatchingPenniesState : public SimMoveState {
63 public:
64 MissingPlayerRepeatedMatchingPenniesState(std::shared_ptr<const Game> game)
65 : SimMoveState(game), turns_(0), returns_(game->NumPlayers(), 0.0) {}
66
67 Player CurrentPlayer() const override {
68 return IsTerminal() ? kTerminalPlayerId : kSimultaneousPlayerId;
69 }
70
71 std::string ActionToString(Player player, Action action_id) const override {
72 return absl::StrCat("Player ", player, " action: ", action_id);
73 }
74
75 std::string ToString() const override { return HistoryString(); }
76 bool IsTerminal() const override { return turns_ == num_players_; };
77 std::vector<double> Returns() const override {
78 return IsTerminal() ? returns_ : std::vector<double>(num_players_, 0.);
79 }
80 std::string InformationStateString(Player player) const override {
81 return absl::StrCat(HistoryString(), " P:", player);
82 }
83 std::unique_ptr<State> Clone() const override {
84 return std::make_unique<MissingPlayerRepeatedMatchingPenniesState>(*this);
85 }
86 std::vector<Action> LegalActions(Player player) const override {
87 if (player == turns_) {
88 return {};
89 } else {
90 return {0, 1};
91 }
92 }
93
94 protected:
95 void DoApplyActions(const std::vector<Action>& actions) override {
96 int missing_pid = turns_;
97 int row_pid = (missing_pid + 1) % num_players_;
98 int col_pid = (row_pid + 1) % num_players_;
99 SPIEL_CHECK_NE(actions[row_pid], kInvalidAction);
100 SPIEL_CHECK_NE(actions[col_pid], kInvalidAction);
101 if (actions[row_pid] == actions[col_pid]) {
102 // Match. Win for row player.
103 returns_[row_pid] += 1.0;
104 returns_[col_pid] -= 1.0;
105 } else {
106 // No match. Win for col player.
107 returns_[row_pid] -= 1.0;
108 returns_[col_pid] += 1.0;
109 }
110 turns_++;
111 }
112
113 private:
114 int turns_;
115 std::vector<double> returns_;
116};

148void SimulateGames(std::mt19937* rng, const Game& game, State* sim_state,
149 State* turn_based_state) {
150 while (!sim_state->IsTerminal()) {
151 const State* wrapped_sim_state =
152 dynamic_cast<TurnBasedSimultaneousState*>(turn_based_state)
153 ->SimultaneousGameState();
154
155 // Now check that the states are identical via the ToString().
156 SPIEL_CHECK_EQ(sim_state->ToString(), wrapped_sim_state->ToString());
157
158 if (sim_state->IsChanceNode()) {
159 SPIEL_CHECK_TRUE(turn_based_state->IsChanceNode());
160
161 // Chance node; sample one according to underlying distribution
162 std::vector<std::pair<Action, double>> outcomes =
163 sim_state->ChanceOutcomes();
164 Action action =
165 open_spiel::SampleAction(
166 outcomes, std::uniform_real_distribution<double>(0.0, 1.0)(*rng))
167 .first;
168
169 std::cout << "sampled outcome: %s\n"
170 << sim_state->ActionToString(kChancePlayerId, action)
171 << std::endl;
172
173 sim_state->ApplyAction(action);
174 turn_based_state->ApplyAction(action);
175 } else if (sim_state->CurrentPlayer() == kSimultaneousPlayerId) {
176 SPIEL_CHECK_EQ(wrapped_sim_state->CurrentPlayer(), kSimultaneousPlayerId);
177
178 // Players choose simultaneously.
179 std::vector<Action> joint_action;
180
181 // Sample an action for each player
182 for (auto p = Player{0}; p < game.NumPlayers(); p++) {
183 if (game.GetType().provides_information_state_string) {
184 // Check the information states to each player are consistent.
185 SPIEL_CHECK_EQ(sim_state->InformationStateString(p),
186 wrapped_sim_state->InformationStateString(p));
187 }
188
189 std::vector<Action> actions;
190 actions = sim_state->LegalActions(p);
191 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
192 Action action = actions[dis(*rng)];
193 joint_action.push_back(action);
194 std::cout << "player " << p << " chose "
195 << sim_state->ActionToString(p, action) << std::endl;
196 SPIEL_CHECK_EQ(p, turn_based_state->CurrentPlayer());
197 turn_based_state->ApplyAction(action);
198 }
199
200 sim_state->ApplyActions(joint_action);
201 } else {
202 SPIEL_CHECK_EQ(sim_state->CurrentPlayer(),
203 wrapped_sim_state->CurrentPlayer());
204 SPIEL_CHECK_EQ(sim_state->CurrentPlayer(),
205 turn_based_state->CurrentPlayer());
206
207 Player p = sim_state->CurrentPlayer();
208
209 if (game.GetType().provides_information_state_string) {
210 // Check the information states to each player are consistent.
211 SPIEL_CHECK_EQ(sim_state->InformationStateString(p),
212 wrapped_sim_state->InformationStateString(p));
213 }
214
215 std::vector<Action> actions;
216 actions = sim_state->LegalActions(p);
217 absl::uniform_int_distribution<> dis(0, actions.size() - 1);
218 Action action = actions[dis(*rng)];
219
220 std::cout << "player " << p << " chose "
221 << sim_state->ActionToString(p, action) << std::endl;
222
223 turn_based_state->ApplyAction(action);
224 sim_state->ApplyAction(action);
225 }
226
227 std::cout << "State: " << std::endl << sim_state->ToString() << std::endl;
228 }
229
230 SPIEL_CHECK_TRUE(turn_based_state->IsTerminal());
231
232 auto sim_returns = sim_state->Returns();
233 auto turn_returns = turn_based_state->Returns();
234
235 for (auto player = Player{0}; player < game.NumPlayers(); player++) {
236 double utility = sim_returns[player];
237 SPIEL_CHECK_GE(utility, game.MinUtility());
238 SPIEL_CHECK_LE(utility, game.MaxUtility());
239 std::cout << "Utility to player " << player << " is " << utility
240 << std::endl;
241
242 double other_utility = turn_returns[player];
243 SPIEL_CHECK_EQ(utility, other_utility);
244 }
245}

247void BasicTurnBasedSimultaneousTests() {
248 std::mt19937 rng;
249
250 for (const GameType& type : RegisteredGameTypes()) {
251 if (!type.ContainsRequiredParameters() && type.default_loadable) {
252 std::string name = type.short_name;
253 if (type.dynamics == GameType::Dynamics::kSimultaneous) {
254 std::cout << "TurnBasedSimultaneous: Testing " << name << std::endl;
255 for (int i = 0; i < 100; ++i) {
256 std::shared_ptr<const Game> sim_game = LoadGame(name);
257 std::shared_ptr<const Game> turn_based_game =
258 ConvertToTurnBased(*LoadGame(name));
259 auto sim_state = sim_game->NewInitialState();
260 auto turn_based_state = turn_based_game->NewInitialState();
261 SimulateGames(&rng, *sim_game, sim_state.get(),
262 turn_based_state.get());
263 }
264 }
265 }
266 }

269void SomePlayersHaveNoLegalActionsTests() {
270 std::shared_ptr<const Game> game(
271 new MissingPlayerRepeatedMatchingPenniesGame({}));
272 testing::RandomSimTest(*game, 10);
273
274 std::shared_ptr<const Game> turn_based_game = ConvertToTurnBased(*game);
275 testing::RandomSimTest(*turn_based_game, 10);
276
277 // Hey, while we're here, why not try CFR?
278 algorithms::CFRSolverBase solver(*turn_based_game,
279 /*alternating_updates*/true,
280 /*linear_averaging*/false,
281 /*regret_matching_plus*/false,
282 /*random_initial_regrets*/true,
283 /*seed*/78846817);
284 for (int i = 0; i < 5; i++) {
285 solver.EvaluateAndUpdatePolicy();
286 const std::shared_ptr<Policy> average_policy = solver.AveragePolicy();
287 std::vector<double> expected_returns = algorithms::ExpectedReturns(
288 *turn_based_game->NewInitialState(), *average_policy,
289 /*depth_limit*/-1);
290 std::cout << "Iter " << i << ", expected returns:";
291 for (double val : expected_returns) {
292 std::cout << " " << val;
293 }
294 std::cout << std::endl;
295 }
296}
