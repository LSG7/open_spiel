40 def init_vars(self, num_strats, num_players):
41 """Initialize solver parameters."""
42 self.num_players = num_players
43 if self.rnd_init:
44 init_dist = self.random.rand(num_strats)
45 else:
46 init_dist = np.ones(num_strats)
47 init_dist /= init_dist.sum()
48 init_regret = np.zeros(num_strats)
49 return (init_dist, init_regret)

82 def update(self, params, grads, t):
83 """Update cumulative regret and strategy (dist).
84
85 Args:
86 params: tuple of variables to be updated (dist, regret)
87 grads: tuple of variable gradients (grad_dist, grad_regret)
88 t: int, solver iteration (not used)
89 Returns:
90 new_params: tuple of update params (new_dist, new_regret)
91 """
92 dist, regret = params
93 regret_delta = grads[1]
94 if self.discount:
95 gamma = t / float(t + 1)
96 else:
97 gamma = 1
98 new_regret = gamma * regret + regret_delta
99 new_clipped_regrets = np.clip(new_regret + self.optimism * regret_delta,
100 0.,
101 np.inf)
102 if np.sum(new_clipped_regrets) > 0:
103 new_dist = new_clipped_regrets / new_clipped_regrets.sum()
104 else:
105 new_dist = np.ones_like(dist) / dist.size
106 new_params = (new_dist, new_regret)
107 return new_params

110def gradients(dist, regret, payoff_matrices):
111 """Computes regret delta to be added to regret in update.
112
113 Args:
114 dist: 1-d np.array, current estimate of nash distribution
115 regret: 1-d np.array (same shape as dist), current estimate of regrets
116 payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
117 Returns:
118 deltas w.r.t. (dist, regret) as tuple
119 unregularized exploitability (stochastic estimate)
120 solver exploitability (stochastic estimate) - NaN
121 """
122 del regret
123
124 nabla = payoff_matrices[0].dot(dist)
125 utility = nabla.dot(dist)
126
127 grad_dist = np.NaN * np.ones_like(dist)
128 grad_regret = nabla - utility
129
130 unreg_exp = np.max(nabla) - nabla.dot(dist)
131
132 return (grad_dist, grad_regret), unreg_exp, np.NaN
