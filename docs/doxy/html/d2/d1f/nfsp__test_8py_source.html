28 def test_run_kuhn(self):
29 env = rl_environment.Environment("kuhn_poker")
30 state_size = env.observation_spec()["info_state"][0]
31 num_actions = env.action_spec()["num_actions"]
32
33 with self.session() as sess:
34 agents = [
35 nfsp.NFSP( # pylint: disable=g-complex-comprehension
36 sess,
37 player_id,
38 state_representation_size=state_size,
39 num_actions=num_actions,
40 hidden_layers_sizes=[16],
41 reservoir_buffer_capacity=10,
42 anticipatory_param=0.1) for player_id in [0, 1]
43 ]
44 sess.run(tf.global_variables_initializer())
45
46 for unused_ep in range(10):
47 time_step = env.reset()
48 while not time_step.last():
49 current_player = time_step.observations["current_player"]
50 current_agent = agents[current_player]
51 agent_output = current_agent.step(time_step)
52 time_step = env.step([agent_output.action])
53
54 for agent in agents:
55 agent.step(time_step)
