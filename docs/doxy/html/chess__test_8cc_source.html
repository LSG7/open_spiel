34uint64_t Perft(const ChessBoard& board, int depth) {
35 std::vector<Move> legal_moves;
36 board.GenerateLegalMoves([&legal_moves](const Move& move) -> bool {
37 legal_moves.push_back(move);
38 return true;
39 });
40 if (depth == 1) {
41 return legal_moves.size();
42 } else {
43 uint64_t ret = 0;
44 for (const auto& move : legal_moves) {
45 ChessBoard board_copy = board;
46 board_copy.ApplyMove(move);
47 ret += Perft(board_copy, depth - 1);
48 }
49 return ret;
50 }
51}

53uint64_t Perft(const char* fen, int depth) {

57void CheckUndo(const char* fen, const char* move_san, const char* fen_after) {
58 std::shared_ptr<const Game> game = LoadGame("chess");
59 ChessState state(game, fen);
60 Player player = state.CurrentPlayer();
61 absl::optional<Move> maybe_move = state.Board().ParseSANMove(move_san);
62 SPIEL_CHECK_TRUE(maybe_move);
63 Action action = MoveToAction(*maybe_move, state.BoardSize());
64 state.ApplyAction(action);
65 SPIEL_CHECK_EQ(state.Board().ToFEN(), fen_after);
66 state.UndoAction(player, action);
67 SPIEL_CHECK_EQ(state.Board().ToFEN(), fen);
68}

70void ApplySANMove(const char* move_san, ChessState* state) {
71 absl::optional<Move> maybe_move = state->Board().ParseSANMove(move_san);
72 SPIEL_CHECK_TRUE(maybe_move);
73 state->ApplyAction(MoveToAction(*maybe_move, state->BoardSize()));

169void ObservationTensorTests() {
170 std::shared_ptr<const Game> game = LoadGame("chess");
171 ChessState initial_state(game);
172 auto shape = game->ObservationTensorShape();
173 std::vector<float> v(game->ObservationTensorSize());
174 initial_state.ObservationTensor(initial_state.CurrentPlayer(),
175 absl::MakeSpan(v));
176
177 // For each piece type, check one square that's supposed to be occupied, and
178 // one that isn't.
179 // Kings.
180 SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "e1"), 1.0);
181 SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "d1"), 0.0);
182 SPIEL_CHECK_EQ(ValueAt(v, shape, 1, "e8"), 1.0);
183 SPIEL_CHECK_EQ(ValueAt(v, shape, 1, "e1"), 0.0);
184
185 // Queens.
186 SPIEL_CHECK_EQ(ValueAt(v, shape, 2, "d1"), 1.0);
187 SPIEL_CHECK_EQ(ValueAt(v, shape, 2, "e1"), 0.0);
188 SPIEL_CHECK_EQ(ValueAt(v, shape, 3, "d8"), 1.0);
189 SPIEL_CHECK_EQ(ValueAt(v, shape, 3, "d1"), 0.0);
190
191 // Rooks.
192 SPIEL_CHECK_EQ(ValueAt(v, shape, 4, "a1"), 1.0);
193 SPIEL_CHECK_EQ(ValueAt(v, shape, 4, "e8"), 0.0);
194 SPIEL_CHECK_EQ(ValueAt(v, shape, 5, "h8"), 1.0);
195 SPIEL_CHECK_EQ(ValueAt(v, shape, 5, "c5"), 0.0);
196
197 // Bishops.
198 SPIEL_CHECK_EQ(ValueAt(v, shape, 6, "c1"), 1.0);
199 SPIEL_CHECK_EQ(ValueAt(v, shape, 6, "b1"), 0.0);
200 SPIEL_CHECK_EQ(ValueAt(v, shape, 7, "f8"), 1.0);
201 SPIEL_CHECK_EQ(ValueAt(v, shape, 7, "f7"), 0.0);
202
203 // Knights.
204 SPIEL_CHECK_EQ(ValueAt(v, shape, 8, "b1"), 1.0);
205 SPIEL_CHECK_EQ(ValueAt(v, shape, 8, "c3"), 0.0);
206 SPIEL_CHECK_EQ(ValueAt(v, shape, 9, "g8"), 1.0);
207 SPIEL_CHECK_EQ(ValueAt(v, shape, 9, "g7"), 0.0);
208
209 // Pawns.
210 SPIEL_CHECK_EQ(ValueAt(v, shape, 10, "a2"), 1.0);
211 SPIEL_CHECK_EQ(ValueAt(v, shape, 10, "a3"), 0.0);
212 SPIEL_CHECK_EQ(ValueAt(v, shape, 11, "e7"), 1.0);
213 SPIEL_CHECK_EQ(ValueAt(v, shape, 11, "e6"), 0.0);
214
215 // Empty.
216 SPIEL_CHECK_EQ(ValueAt(v, shape, 12, "e4"), 1.0);
217 SPIEL_CHECK_EQ(ValueAt(v, shape, 12, "e2"), 0.0);
218
219 // Repetition count.
220 SPIEL_CHECK_EQ(ValueAt(v, shape, 13, 0, 0), 0.0);
221
222 // Side to move.
223 SPIEL_CHECK_EQ(ValueAt(v, shape, 14, 0, 0), 1.0);
224
225 // Irreversible move counter.
226 SPIEL_CHECK_EQ(ValueAt(v, shape, 15, 0, 0), 0.0);
227
228 // Castling rights.
229 SPIEL_CHECK_EQ(ValueAt(v, shape, 16, 0, 0), 1.0);
230 SPIEL_CHECK_EQ(ValueAt(v, shape, 17, 1, 1), 1.0);
231 SPIEL_CHECK_EQ(ValueAt(v, shape, 18, 2, 2), 1.0);
232 SPIEL_CHECK_EQ(ValueAt(v, shape, 19, 3, 3), 1.0);
233
234 ApplySANMove("e4", &initial_state);
235 ApplySANMove("e5", &initial_state);
236 ApplySANMove("Ke2", &initial_state);
237
238 initial_state.ObservationTensor(initial_state.CurrentPlayer(),
239 absl::MakeSpan(v));
240 SPIEL_CHECK_EQ(v.size(), game->ObservationTensorSize());
241
242 // Now it's black to move.
243 SPIEL_CHECK_EQ(ValueAt(v, shape, 14, 0, 0), 0.0);
244
245 // White king is now on e2.
246 SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "e1"), 0.0);
247 SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "e2"), 1.0);
248
249 // Irreversible move counter incremented to 1 (king moving and losing castling
250 // rights is in fact irreversible in this case, but it doesn't reset the
251 // counter according to FIDE rules).
252 SPIEL_CHECK_FLOAT_EQ(ValueAt(v, shape, 15, 0, 0), 1.0 / 101.0);
253
254 // And white no longer has castling rights.
255 SPIEL_CHECK_EQ(ValueAt(v, shape, 16, 0, 0), 0.0);
256 SPIEL_CHECK_EQ(ValueAt(v, shape, 17, 1, 1), 0.0);
257 SPIEL_CHECK_EQ(ValueAt(v, shape, 18, 2, 2), 1.0);
258 SPIEL_CHECK_EQ(ValueAt(v, shape, 19, 3, 3), 1.0);
259}

261void MoveConversionTests() {
262 auto game = LoadGame("chess");
263 std::mt19937 rng(23);
264 for (int i = 0; i < 100; ++i) {
265 std::unique_ptr<State> state = game->NewInitialState();
266 while (!state->IsTerminal()) {
267 const ChessState* chess_state =
268 dynamic_cast<const ChessState*>(state.get());
269 std::vector<Action> legal_actions = state->LegalActions();
270 absl::uniform_int_distribution<int> dist(0, legal_actions.size() - 1);
271 int action_index = dist(rng);
272 Action action = legal_actions[action_index];
273 Move move = ActionToMove(action, chess_state->Board());
274 Action action_from_move = MoveToAction(move, chess_state->BoardSize());
275 SPIEL_CHECK_EQ(action, action_from_move);
276 const ChessBoard& board = chess_state->Board();
277 ChessBoard fresh_board = chess_state->StartBoard();
278 for (Move move : chess_state->MovesHistory()) {
279 fresh_board.ApplyMove(move);
280 }
281 SPIEL_CHECK_EQ(board.ToFEN(), fresh_board.ToFEN());
282 Action action_from_lan =
283 MoveToAction(*board.ParseLANMove(move.ToLAN()), board.BoardSize());
284 SPIEL_CHECK_EQ(action, action_from_lan);
285 state->ApplyAction(action);
286 }
287 }
288}
