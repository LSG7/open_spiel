15 r"""Python implementation of the magnetic mirror descent (MMD) algorithm.
16 
17 The algorithm operated over the sequence-from with dilated entropy.
18 
19 See https://arxiv.org/abs/2206.05825.
20 
21 One iteration of MMD consists of:
22 1) Compute gradients of dilated entropy
23  and payoffs for current sequence form policies.
24 2) Compute behavioural form policy starting from the bottom
25  of the tree and updating gradients of parent nodes along the way.
26 3) Convert behavioural form policy to equivalent sequence form policy.
27 
28 The last sequence form policy converges linearly (exponentially fast)
29 to a \alpha-reduced normal-form QRE.
30 """

58 def divergence(x, y, psi_x, psi_y, grad_psi_y):

74 def dilated_dgf_divergence(mmd_1, mmd_2):
75  """Bregman divergence between two MMDDilatedEnt objects.
76 
77  The value is equivalent to a sum of two Bregman divergences
78  over the sequence form, one for each player.
79 
80  Args:
81  mmd_1: MMDDilatedEnt Object
82  mmd_2: MMDDilatedEnt Object
83 
84  Returns:
85  Scalar.
86  """
87 
88  dgf_values = [mmd_1.dgf_eval(), mmd_2.dgf_eval()]
89  dgf_grads = mmd_2.dgf_grads()
90  div = 0
91  for player in range(2):
92  div += divergence(mmd_1.sequences[player], mmd_2.sequences[player],
93  dgf_values[0][player], dgf_values[1][player],
94  dgf_grads[player])
95  return div

98 class MMDDilatedEnt(object):
99  r"""Implements Magnetic Mirror Descent (MMD) with Dilated Entropy.
100 
101  The implementation uses the sequence form representation.
102 
103  The policies converge to a \alpha-reduced normal form QRE of a
104  two-player zero-sum extensive-form game. If \alpha is set
105  to zero then the method is equivalent to mirror descent ascent
106  over the sequence form with dilated entropy and the policies
107  will converge on average to a nash equilibrium with
108  the appropriate stepsize schedule (or approximate equilirbrium
109  for fixed stepsize).
110 
111  The main iteration loop is implemented in `update_sequences`:
112 
113  ```python
114  game = pyspiel.load_game("game_name")
115  mmd = MMDDilatedEnt(game, alpha=0.1)
116  for i in range(num_iterations):
117  mmd.update_sequences()
118  ```
119  The gap in the regularized game (i.e. 2x exploitability) converges
120  to zero and can be computed:
121 
122  ```python
123  gap = mmd.get_gap()
124  ```
125  The policy (i.e. behavioural form policy) can be retrieved:
126  ```python
127  policies = mmd.get_policies()
128  ```
129 
130  The average sequences and policies can be retrieved:
131 
132  ```python
133  avg_sequences = mmd.get_avg_sequences()
134  avg_policies = mmd.get_avg_policies()
135  ```
136 
137  """

233  def dgf_grads(self):
234  """Computes gradients of dilated entropy for each player and current seqs.
235 
236  Returns:
237  A list of numpy arrays.
238  """
239  grads = [np.zeros(len(self.sequencessequences[0])), np.zeros(len(self.sequencessequences[1]))]
240  for player in range(2):
241  for infostate in self.infosets[player]:
242 
243  # infostates contain empty sequence for root variable
244  if is_root(infostate):
245  continue
246 
247  parent_seq = self.get_parent_seqget_parent_seq(player, infostate)
248  if parent_seq > 0:
249 
250  for isa_key in self.infoset_action_maps[player][infostate]:
251  # compute infostate term
252  seq_idx = self.infoset_actions_to_seq[player][isa_key]
253  seq = self.sequencessequences[player][seq_idx]
254  grads[player][seq_idx] += np.log(seq / parent_seq) + 1
255 
256  # compute terms from children if there are any
257  num_children = len(self.infoset_actions_children[player].get(
258  isa_key, []))
259  grads[player][seq_idx] -= num_children
260  return grads

304

306  _get_action_from_key(child_isa_key)
307  for child_isa_key in child_isa_keys

309  policy_child = pol.policy_for_key(child)[:]
310  policy_child = np.array([policy_child[a] for a in actions_child])
311  g[isa_idx] += np.dot(g_child, policy_child)
312  g[isa_idx] += neg_entropy(policy_child)

316  return

319  g_infostate = np.array([g[idx] for idx in seq_idx])
320  actions = [_get_action_from_key(isa_key) for isa_key in isa_keys]
321  new_state_policy = softmax(-g_infostate)

325  def get_gap(self):
326  """Computes saddle point gap of the regularized game.
327 
328  The gap measures convergence to the alpha-QRE.
329 
330  Returns:
331  Scalar.
332  """
333  assert self.alphaalpha > 0, "gap cannot be computed for alpha = 0"
334  grads = [(self.payoff_mat @ self.sequencessequences[1]) / (self.alphaalpha),
335  (-self.payoff_mat.T @ self.sequencessequences[0]) / (self.alphaalpha)]
336  dgf_values = self.dgf_evaldgf_eval()
337 
338  br_policy = policy.TabularPolicy(self.gamegame)
339  for player in range(2):
340  self._update_state_sequences_update_state_sequences(self.empty_infoset_keysempty_infoset_keys[player],
341  grads[player], player, br_policy)
342 
343  br_sequences = policy_to_sequence(self.gamegame, br_policy,
344  self.infoset_actions_to_seq)
345  curr_sequences = copy.deepcopy(self.sequencessequences)
346  self.sequencessequences = br_sequences
347  br_dgf_values = self.dgf_evaldgf_eval()
348  self.sequencessequences = curr_sequences
349 
350  # gap of sequences (x,y)
351  # d(x) + max_y' x.T A y'-d(y') + d(y) - min_x' d(x') + x'.T Ay
352 
353  gap = 0
354  gap += curr_sequences[0].T @ self.payoff_mat @ br_sequences[1]
355  gap += self.alphaalpha * (dgf_values[1] - br_dgf_values[1])
356  gap += self.alphaalpha * (dgf_values[0] - br_dgf_values[0])
357  gap += -br_sequences[0].T @ self.payoff_mat @ curr_sequences[1]
358  return gap
