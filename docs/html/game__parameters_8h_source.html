146  bool operator==(const GameParameter& rhs) const {
147  switch (type_) {
148  case Type::kInt:
149  return rhs.has_int_value() && int_value_ == rhs.int_value();
150  case Type::kDouble:
151  return rhs.has_double_value() && double_value_ == rhs.double_value();
152  case Type::kString:
153  return rhs.has_string_value() && string_value_ == rhs.string_value();
154  case Type::kBool:
155  return rhs.has_bool_value() && bool_value_ == rhs.bool_value();
156  case Type::kGame:
157  return rhs.has_game_value() && game_value_ == rhs.game_value();
158  case Type::kUnset:
159  return rhs.type_ == Type::kUnset;
160  }
161  std::cerr << "Unrecognized parameter type in operator=="
162  << ", returning false." << std::endl;
163  return false;
164  }
165  bool operator!=(const GameParameter& rhs) const { return !(*this == rhs); }

170  // Default initializations are required here. This is because some games mark
171  // parameters as not mandatory and also do not specify default values when
172  // registering the game type.. instead, setting the documented defaults upon
173  // game creation (often due to missing information at registration time).
174  // This causes a problem when inspecting the game types themselves, even after
175  // the game is created via Game::GetType(), which returns the type as it was
176  // when it was registered. These initial values are used for those cases.
177  int int_value_ = 0;
178  double double_value_ = 0.0;
179  std::string string_value_ = "";

207 template <typename T>
208 T ParameterValue(const GameParameters& params, const std::string& key,
209  absl::optional<T> default_value = absl::nullopt) {
210  auto iter = params.find(key);
211  if (iter == params.end()) {
212  if (!default_value.has_value()) {
213  SpielFatalError(absl::StrCat("Cannot find parameter and no default "
214  "value passed for key: ", key));
215  }
216 
217  return *default_value;
218  } else {
219  return iter->second.value<T>();
220  }
221 }
