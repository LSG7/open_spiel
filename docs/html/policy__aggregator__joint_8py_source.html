121  def aggregate(self, pids, joint_policies, weights):
122  r"""Computes the weighted-mixture of the joint policies.
123 
124  Let P of shape [num_players] be the joint policy, and W some weights.
125  Let N be the number of policies (i.e. len(policies)).
126  We return the policy P' such that for all state `s`:
127 
128  P[s] ~ \sum_{i=0}^{N-1} (policies[i][player(s)](s) * weights[i] *
129  reach_prob(s, policies[i]))
130 
131  Arguments:
132  pids: Spiel player ids of the players the strategies belong to.
133  joint_policies: List of list of policies (One list per joint strategy)
134  weights: List of weights to attach to each joint strategy.
135 
136  Returns:
137  A _DictPolicy, a callable object representing the policy.
138  """
139  aggr_policies = []
140  self._joint_policies_joint_policies = joint_policies
141 
142  # To do(pmuller): We should be able to do a single recursion.
143  for pid in pids:
144  aggr_policies.append(self._sub_aggregate_sub_aggregate(pid, weights))
145  return _DictPolicy(self._game_game, aggr_policies)

173  def _rec_aggregate(self, pid, state, my_reaches):
174  """Recursively traverse game tree to compute aggregate policy."""
175  if state.is_terminal():
176  return
177 
178  if state.is_simultaneous_node():
179  policies = _aggregate_at_state(self._joint_policies_joint_policies, state, pid)
180  state_key = self._state_key_state_key(state, pid)
181 
182  self._policy_policy[state_key] = {}
183  used_moves = state.legal_actions(pid)
184 
185  for uid in used_moves:
186  new_reaches = copy.deepcopy(my_reaches)
187  for i in range(len(policies)):
188  # compute the new reach for each policy for this action
189  new_reaches[i] *= policies[i].get(uid, 0)
190  # add reach * prob(a) for this policy to the computed policy
191  if uid in self._policy_policy[state_key].keys():
192  self._policy_policy[state_key][uid] += new_reaches[i]
193  else:
194  self._policy_policy[state_key][uid] = new_reaches[i]
195 
196  num_players = self._game_game.num_players()
197  all_other_used_moves = []
198  for player in range(num_players):
199  if player != pid:
200  all_other_used_moves.append(state.legal_actions(player))
201 
202  other_joint_actions = itertools.product(*all_other_used_moves)
203 
204  # enumerate every possible other-agent actions for next-state
205  for other_joint_action in other_joint_actions:
206  for uid in used_moves:
207  new_reaches = copy.deepcopy(my_reaches)
208  for i in range(len(policies)):
209  # compute the new reach for each policy for this action
210  new_reaches[i] *= policies[i].get(uid, 0)
211

214  )
215  new_state = state.clone()
216  new_state.apply_actions(joint_action)
217  self._rec_aggregate_rec_aggregate(pid, new_state, new_reaches)
218  return

221  for action in state.legal_actions():
222  new_state = state.child(action)
223  self._rec_aggregate_rec_aggregate(pid, new_state, my_reaches)
224  return
