20class ReplayBuffer(object):
21 """ReplayBuffer of fixed size with a FIFO replacement policy.
22
23 Stored transitions can be sampled uniformly.
24
25 The underlying datastructure is a ring buffer, allowing 0(1) adding and
26 sampling.
27 """
28
29 def __init__(self, replay_buffer_capacity):
30 self._replay_buffer_capacity = replay_buffer_capacity
31 self._data = []
32 self._next_entry_index = 0
33
34 def add(self, element):
35 """Adds `element` to the buffer.
36
37 If the buffer is full, the oldest element will be replaced.
38
39 Args:
40 element: data to be added to the buffer.
41 """
42 if len(self._data) < self._replay_buffer_capacity:
43 self._data.append(element)
44 else:
45 self._data[self._next_entry_index] = element
46 self._next_entry_index += 1
47 self._next_entry_index %= self._replay_buffer_capacity
48
49 def sample(self, num_samples):
50 """Returns `num_samples` uniformly sampled from the buffer.
51
52 Args:
53 num_samples: `int`, number of samples to draw.
54
55 Returns:
56 An iterable over `num_samples` random elements of the buffer.
57
58 Raises:
59 ValueError: If there are less than `num_samples` elements in the buffer
60 """
61 if len(self._data) < num_samples:
62 raise ValueError("{} elements could not be sampled from size {}".format(
63 num_samples, len(self._data)))
64 return random.sample(self._data, num_samples)
65
66 def reset(self):
67 """Resets the contents of the replay buffer."""
68 self._data = []
69 self._next_entry_index = 0
70
71 def __len__(self):
72 return len(self._data)
73
74 def __iter__(self):
75 return iter(self._data)
