276bool SkatState::IsTrump(int card) const {
277 // Nothing is trump in Null games. Otherwise Jacks are always trump. In a Suit
278 // game all cards of that suits are trump as well as Jacks.
279 if (game_type_ == kNullGame) return false;
280 if (CardRank(card) == kJack) return true;
281 switch (game_type_) {
282 case kDiamondsTrump:
283 return CardSuit(card) == kDiamonds;
284 case kHeartsTrump:
285 return CardSuit(card) == kHearts;
286 case kSpadesTrump:
287 return CardSuit(card) == kSpades;
288 case kClubsTrump:
289 return CardSuit(card) == kClubs;
290 default:
291 return false;
292 }
293}

295int SkatState::CardOrder(int card, int first_card) const {
296 if (IsTrump(card)) {
297 return 7 + TrumpOrder(card);
298 } else if (CardSuit(card) == CardSuit(first_card)) { // Following suit.
299 if (game_type_ == kNullGame) {
300 return NullOrder(CardRank(card));
301 } else {
302 return static_cast<int>(CardRank(card));
303 }
304 } else {
305 return -1;
306 }
307}
308
309int SkatState::TrumpOrder(int card) const {
310 if (!IsTrump(card)) {
311 return -1;
312 } else if (CardRank(card) == kJack) {
313 return static_cast<int>(CardSuit(card)) + static_cast<int>(kJack);
314 } else {
315 return static_cast<int>(CardRank(card));
316 }
317}

342int SkatState::WinsTrick() const {
343 std::vector<int> cards = PreviousTrick().GetCards();
344 if (cards.empty()) return -1;
345 int winning_position = 0;
346 for (int i = 1; i < cards.size(); i++) {
347 if (CardOrder(cards[i], cards[0]) >
348 CardOrder(cards[winning_position], cards[0])) {
349 winning_position = i;
350 }
351 }
352 return PreviousTrick().PlayerAtPosition(winning_position);
353}

355void SkatState::DoApplyAction(Action action) {
356 switch (phase_) {
357 case kDeal:
358 return ApplyDealAction(action);
359 case kBidding:
360 return ApplyBiddingAction(action - kBiddingActionBase);
361 case kDiscardCards:
362 return ApplyDiscardCardsAction(action);
363 case kPlay:
364 return ApplyPlayAction(action);
365 case kGameOver:
366 SpielFatalError("Cannot act in terminal states");
367 }
368}

370void SkatState::ApplyDealAction(int card) {
371 SPIEL_CHECK_EQ(card_locations_[card], kDeck);
372 int deal_round = history_.size();
373 // Cards 0-2, 11-14, 23-25 to player 1.
374 // Cards 3-5, 15-18, 26-28 to player 2.
375 // Cards 6-8, 19-22, 29-31 to player 3.
376 // Cards 9-10 into the Skat.
377 // While this might seem a bit weird, this is the official order Skat cards
378 // are dealt.
379 if ((deal_round >= 0 && deal_round <= 2) ||
380 (deal_round >= 11 && deal_round <= 14) ||
381 (deal_round >= 23 && deal_round <= 25)) {
382 card_locations_[card] = kHand0;
383 } else if ((deal_round >= 3 && deal_round <= 5) ||
384 (deal_round >= 15 && deal_round <= 18) ||
385 (deal_round >= 26 && deal_round <= 28)) {
386 card_locations_[card] = kHand1;
387 } else if ((deal_round >= 6 && deal_round <= 8) ||
388 (deal_round >= 19 && deal_round <= 22) ||
389 (deal_round >= 29 && deal_round <= 31)) {
390 card_locations_[card] = kHand2;
391 } else if (deal_round == 9 || deal_round == 10) {
392 card_locations_[card] = kSkat;
393 }
394 if (deal_round == kNumCards - 1) {
395 current_player_ = 0;
396 phase_ = kBidding;
397 }
398}

400void SkatState::ApplyBiddingAction(int game_type) {
401 // Simplified bidding as first come first serve. Players can say if they want
402 // to play or not on a first come first serve basis. Currently, the solo
403 // player is not able to touch the Skat.
404 player_bids_[current_player_] = game_type;
405 if (game_type == kPass) {
406 if (current_player_ < 2) {
407 current_player_ = NextPlayer();
408 } else { // No one wants to play.
409 phase_ = kGameOver;
410 }
411 } else {
412 EndBidding(current_player_, SkatGameType(game_type));
413 }
414}
415
416void SkatState::EndBidding(Player winner, SkatGameType game_type) {
417 solo_player_ = winner;
418 current_player_ = winner;
419 game_type_ = game_type;
420 // Winner takes up Skat cards.
421 for (int card = 0; card < kNumCards; card++) {
422 if (card_locations_[card] == kSkat) {
423 card_locations_[card] = PlayerToLocation(winner);
424 }
425 }
426 phase_ = kDiscardCards;
427}

429int SkatState::CardsInSkat() const {
430 int cards_in_skat = 0;
431 for (int card = 0; card < kNumCards; card++) {
432 if (card_locations_[card] == kSkat) cards_in_skat++;
433 }
434 return cards_in_skat;
435}

437void SkatState::ApplyDiscardCardsAction(int card) {
438 SPIEL_CHECK_LT(CardsInSkat(), 2);
439 SPIEL_CHECK_TRUE(current_player_ == solo_player_);
440 SPIEL_CHECK_TRUE(card_locations_[card] == PlayerToLocation(solo_player_));
441 card_locations_[card] = kSkat;
442
443 if (CardsInSkat() == 2) {
444 phase_ = kPlay;
445 current_player_ = 0;
446 }
447}

449void SkatState::ApplyPlayAction(int card) {
450 SPIEL_CHECK_TRUE(card_locations_[card] == PlayerToLocation(current_player_));
451 card_locations_[card] = kTrick;
452 if (num_cards_played_ == 0) {
453 CurrentTrick() = Trick(current_player_);
454 }
455 CurrentTrick().PlayCard(card);
456 num_cards_played_++;
457 if (num_cards_played_ % kNumPlayers == 0) {
458 last_trick_winner_ = WinsTrick();
459 current_player_ = last_trick_winner_;
460 // When num_cards_played_ == kNumCards + kNumCardsInSkat CurrentTrick() is
461 // the same as PreviousTrick() and we don't want to overwrite it.
462 if (num_cards_played_ < kNumCards - kNumCardsInSkat) {
463 CurrentTrick() = Trick(current_player_);
464 }
465 // Winner plays next card.
466 if (last_trick_winner_ == solo_player_) {
467 points_solo_ += PreviousTrick().Points();
468 if (game_type_ == kNullGame) {
469 // The solo player loses a Null game if they win any trick. The trick
470 // they win could be without points so we add one to make sure ScoreUp
471 // knows that the solo_player has won a trick.
472 points_solo_++;
473 phase_ = kGameOver;
474 ScoreUp();
475 }
476 } else {
477 points_team_ += PreviousTrick().Points();
478 }
479 } else {
480 current_player_ = NextPlayer();
481 }
482
483 if (num_cards_played_ == kNumCards - kNumCardsInSkat) {
484 phase_ = kGameOver;
485 ScoreUp();
486 }
487}
488
489void SkatState::ScoreUp() {
490 if (game_type_ == kNullGame) {
491 // Since we're using points as a reward we need to come up with some special
492 // rule for Null.
493 if (points_solo_ > 0) {
494 points_solo_ = 30;
495 points_team_ = 90;
496 } else {
497 points_solo_ = 90;
498 points_team_ = 30;
499 }
500 } else {
501 // Solo player gets the cards in the Skat (unless Null is played).
502 for (int card = 0; card < kNumCards; card++) {
503 if (card_locations_[card] == kSkat) {
504 points_solo_ += CardValue(card);
505 }
506 }
507 }
508 for (int pl = 0; pl < kNumPlayers; ++pl) {
509 if (solo_player_ == pl) {
510 returns_[pl] = (points_solo_ - 60) / 120.0;
511 } else {
512 returns_[pl] = (points_team_ - 60) / 240.0;
513 }
514 }
515}

584std::vector<Action> SkatState::DiscardCardsLegalActions() const {
585 std::vector<Action> legal_actions;
586 for (int card = 0; card < kNumCards; ++card) {

594std::vector<Action> SkatState::PlayLegalActions() const {
595 std::vector<Action> legal_actions;
596 legal_actions.reserve(kNumTricks - num_cards_played_ / kNumPlayers);
597 if (num_cards_played_ % kNumPlayers != 0) {
598 // Check if we can follow suit.
599 int first_card = CurrentTrick().FirstCard();
600 int suit = CardSuit(first_card);
601 if (game_type_ == kNullGame) {
602 for (int rank = 0; rank < kNumRanks; ++rank) {
603 int card = static_cast<int>(suit) * kNumRanks + rank;
604 if (card_locations_[card] == PlayerToLocation(current_player_)) {
605 legal_actions.push_back(card);
606 }
607 }
608 } else {
609 // This is a bid fidely but it makes sure the legal actions are sorted
610 // (which is required), which the special status of jacks makes hard
611 // otherwise.
612 for (int card = 0; card < kNumCards; ++card) {
613 if ((IsTrump(first_card) && IsTrump(card)) ||
614 (suit == CardSuit(card) &&
615 CardRank(card) != kJack &&
616 CardRank(first_card) != kJack)) {
617 if (card_locations_[card] == PlayerToLocation(current_player_)) {
618 legal_actions.push_back(card);
619 }
620 }
621 }
622 }
623 }
624
625 if (!legal_actions.empty()) {
626 return legal_actions;
627 }
628
629 // Otherwise, we can play any of our cards.
630 for (int card = 0; card < kNumCards; ++card) {
631 if (card_locations_[card] == current_player_ + 1) {
632 legal_actions.push_back(card);
633 }
634 }
635 return legal_actions;
636}

638std::vector<std::pair<Action, double>> SkatState::ChanceOutcomes() const {
639 std::vector<std::pair<Action, double>> outcomes;
640 int num_cards_remaining = kNumCards - history_.size();
641 outcomes.reserve(num_cards_remaining);
642 const double p_card = 1.0 / static_cast<double>(num_cards_remaining);
643 for (int card = 0; card < kNumCards; ++card) {
644 if (card_locations_[card] == kDeck) outcomes.emplace_back(card, p_card);
645 }
646 return outcomes;
647}

649void SkatState::ObservationTensor(Player player,
650 absl::Span<float> values) const {
651 SPIEL_CHECK_GE(player, 0);
652 SPIEL_CHECK_LT(player, num_players_);
653
654 std::fill(values.begin(), values.end(), 0.0);
655 SPIEL_CHECK_EQ(values.size(), game_->ObservationTensorSize());
656 if (phase_ == Phase::kDeal) return;
657 auto ptr = values.begin();
658 // Position:
659 ptr[player] = 1;
660 ptr += kNumPlayers;
661 // Phase
662 if (phase_ >= kBidding && phase_ <= kPlay) ptr[phase_ - kBidding] = 1;
663 ptr += 3;
664 // Players Cards
665 for (int i = 0; i < kNumCards; ++i)
666 if (card_locations_[i] == PlayerToLocation(player)) ptr[i] = 1;
667 ptr += kNumCards;
668 // All player bids.
669 for (int i = 0; i < kNumPlayers; i++) {
670 ptr[player_bids_[i]] = 1;
671 ptr += kNumGameTypes;
672 }
673 // Who is the solo player.
674 if (solo_player_ >= 0) ptr[solo_player_] = 1;
675 ptr += kNumPlayers;
676 // Information about the Skat only for the solo_player_.
677 if (player == solo_player_) {
678 for (int i = 0; i < kNumCards; ++i)
679 if (card_locations_[i] == kSkat) ptr[i] = 1;
680 }
681 ptr += kNumCards;
682 // Game type
683 ptr[game_type_] = 1;
684 ptr += kNumGameTypes;
685 // Current trick
686 if (phase_ == kPlay) {
687 ptr[CurrentTrick().Leader()] = 1;
688 ptr += kNumPlayers;
689 const auto& cards = CurrentTrick().GetCards();
690 for (int i = 0; i < kNumPlayers; i++) {
691 if (cards.size() > i) ptr[cards[i]] = 1;
692 ptr += kNumCards;
693 }
694 } else {
695 ptr += kNumPlayers + kNumPlayers * kNumCards;
696 }
697 // Previous Trick
698 if (CurrentTrickIndex() > 0) {
699 ptr[PreviousTrick().Leader()] = 1;
700 ptr += kNumPlayers;
701 const auto& cards = PreviousTrick().GetCards();
702 for (int i = 0; i < kNumPlayers; i++) {
703 if (cards.size() > i) ptr[cards[i]] = 1;
704 ptr += kNumCards;
705 }
706 } else {
707 ptr += kNumPlayers + kNumPlayers * kNumCards;
708 }
709}

728std::string SkatState::ObservationString(Player player) const {
729 SPIEL_CHECK_GE(player, 0);
730 SPIEL_CHECK_LT(player, num_players_);
731 // We construct the ObservationString from the ObservationTensor to give
732 // some indication that the tensor representation is correct & complete.
733 if (phase_ == Phase::kDeal) {
734 return "No Observation";
735 }
736 std::vector<float> tensor(game_->ObservationTensorSize());
737 ObservationTensor(player, absl::MakeSpan(tensor));
738 std::string rv;
739 auto ptr = tensor.begin();
740 int player_pos = GetIntFromOneHot(ptr, kNumPlayers);
741 absl::StrAppend(&rv, "PlPos:", player_pos);
742 ptr += kNumPlayers;
743 Phase phase = kDeal;
744 if (ptr[0]) phase = kBidding;
745 else if (ptr[1]) phase = kDiscardCards;
746 else if (ptr[2]) phase = kPlay;
747 else
748 phase = kGameOver;
749 absl::StrAppend(&rv, "|Phase:", PhaseToString(phase));
750 ptr += 3;
751 std::vector<int> player_cards = GetCardsFromMultiHot(ptr);
752 absl::StrAppend(&rv, "|Hand:", CardsToString(player_cards));
753 ptr += kNumCards;
754 absl::StrAppend(&rv, "|Bids:");
755 for (int i = 0; i < kNumPlayers; i++) {
756 int player_bid = GetIntFromOneHot(ptr, kNumGameTypes);
757 absl::StrAppend(
758 &rv, SkatGameTypeToString(static_cast<SkatGameType>(player_bid)), " ");
759 ptr += kNumGameTypes;
760 }
761 Player solo_player = GetIntFromOneHot(ptr, kNumPlayers);
762 absl::StrAppend(&rv, "|SoloPl:", solo_player);
763 ptr += kNumPlayers;
764 std::vector<int> skat_cards = GetCardsFromMultiHot(ptr);
765 absl::StrAppend(&rv, "|Skat:", CardsToString(skat_cards));
766 ptr += kNumCards;
767 SkatGameType game_type = SkatGameType(GetIntFromOneHot(ptr, kNumGameTypes));
768 absl::StrAppend(&rv, "|Game:", SkatGameTypeToString(game_type));
769 ptr += kNumGameTypes;
770 Player current_trick_leader = GetIntFromOneHot(ptr, kNumPlayers);
771 absl::StrAppend(&rv, "|CurrTrick(Leader:", current_trick_leader, "):");
772 ptr += kNumPlayers;
773 for (int i = 0; i < kNumPlayers; i++) {
774 int card = GetIntFromOneHot(ptr, kNumCards);
775 if (card >= 0) absl::StrAppend(&rv, ToCardSymbol(card), " ");
776 ptr += kNumCards;
777 }
778 Player previous_trick_leader = GetIntFromOneHot(ptr, kNumPlayers);
779 if (previous_trick_leader >= 0) {
780 absl::StrAppend(&rv, "|PrevTrick(Leader:", previous_trick_leader, "):");
781 ptr += kNumPlayers;
782 for (int i = 0; i < kNumPlayers; i++) {
783 int card = GetIntFromOneHot(ptr, kNumCards);
784 if (card >= 0) absl::StrAppend(&rv, ToCardSymbol(card), " ");
785 ptr += kNumCards;
786 }
787 }
788 return rv;
789}
