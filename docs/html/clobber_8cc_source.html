153 ClobberState::ClobberState(std::shared_ptr<const Game> game, int rows,
154  int columns)
155  : State(game), rows_(rows), columns_(columns) {
156  SPIEL_CHECK_GE(rows_, 1);
157  SPIEL_CHECK_GE(columns_, 1);
158  SPIEL_CHECK_LE(rows_, 99); // Only supports 1 and 2 digit row numbers.
159  SPIEL_CHECK_LE(columns_, 26); // Only 26 letters to represent columns.
160 
161  board_ = std::vector<CellState>(rows_ * columns_, CellState::kEmpty);
162 
163  // Put the pieces on the board (checkerboard pattern) starting with
164  // the first player (White, or 'o') in the bottom left corner.
165  for (int row = rows_ - 1; row >= 0; row--) {
166  for (int column = 0; column < columns_; column++) {
167  if ((IsEven(row + (rows_ - 1)) && IsEven(column)) ||
168  (!IsEven(row + (rows_ - 1)) && !IsEven(column))) {
169  SetBoard(row, column, CellState::kWhite);
170  } else {
171  SetBoard(row, column, CellState::kBlack);
172  }
173  }
174  }
175 }

177 ClobberState::ClobberState(std::shared_ptr<const Game> game, int rows,
178  int columns, const std::string& board_string)
179  : State(game), rows_(rows), columns_(columns) {
180  SPIEL_CHECK_GE(rows_, 1);
181  SPIEL_CHECK_GE(columns_, 1);
182  SPIEL_CHECK_LE(rows_, 99); // Only supports 1 and 2 digit row numbers.
183  SPIEL_CHECK_LE(columns_, 26); // Only 26 letters to represent columns.
184  SPIEL_CHECK_GE(board_string[0], '0');
185  SPIEL_CHECK_LE(board_string[0], '1');
186  SPIEL_CHECK_EQ(rows_ * columns_, board_string.length() - 1);
187 
188  board_ = std::vector<CellState>(rows_ * columns_, CellState::kEmpty);
189  current_player_ = board_string[0] - '0';
190 
191  // Create the board from the board string. The character 'o' is White
192  // (first player), 'x' is Black (second player), and the character '.'
193  // is an Empty cell. Population goes from top left to bottom right.
194  for (int row = 0; row < rows_; row++) {
195  for (int column = 0; column < columns_; column++) {
196  char state_character = board_string[1 + row * columns_ + column];
197  CellState state = StringToState(std::string(1, state_character));
198  SetBoard(row, column, state);
199  }
200  }
201 
202  // If the given state is terminal, the current player
203  // cannot play. Therefore, the other player wins.
204  if (!MovesRemaining()) {
205  outcome_ = 1 - current_player_;
206  }
207 }

209 void ClobberState::DoApplyAction(Action action) {
210  std::vector<int> values =
211  UnrankActionMixedBase(action, {rows_, columns_, kNumDirections});
212 
213  const int start_row = values[0];
214  const int start_column = values[1];
215  const int direction = values[2];
216  const int end_row = start_row + kDirRowOffsets[direction];
217  const int end_column = start_column + kDirColumnOffsets[direction];
218 
219  SPIEL_CHECK_TRUE(InBounds(start_row, start_column));
220  SPIEL_CHECK_TRUE(InBounds(end_row, end_column));
221  SPIEL_CHECK_EQ(BoardAt(start_row, start_column),
222  OpponentState(BoardAt(end_row, end_column)));
223 
224  SetBoard(end_row, end_column, BoardAt(start_row, start_column));
225  SetBoard(start_row, start_column, CellState::kEmpty);
226 
227  // Does the other player have any moves left?
228  if (!MovesRemaining()) {
229  outcome_ = current_player_;
230  }
231 
232  current_player_ = 1 - current_player_;
233  num_moves_++;
234 }

254 std::vector<Action> ClobberState::LegalActions() const {
255  std::vector<Action> move_list;
256 
257  if (IsTerminal()) {
258  return move_list;
259  }
260 
261  CellState current_player_state = PlayerToState(CurrentPlayer());
262  std::vector<int> action_bases = {rows_, columns_, kNumDirections};
263  std::vector<int> action_values = {0, 0, 0};
264 
265  for (int row = 0; row < rows_; row++) {
266  for (int column = 0; column < columns_; column++) {
267  if (BoardAt(row, column) == current_player_state) {
268  for (int direction = 0; direction < kNumDirections; direction++) {
269  int adjacent_row = row + kDirRowOffsets[direction];
270  int adjacent_column = column + kDirColumnOffsets[direction];
271 
272  if (InBounds(adjacent_row, adjacent_column)) {
273  CellState adjacent_state = BoardAt(adjacent_row, adjacent_column);
274  CellState opponent_state = OpponentState(current_player_state);
275 
276  if (adjacent_state == opponent_state) {
277  // The adjacent cell is in bounds and contains the opponent
278  // player, therefore playing to this adjacent cell would be
279  // a valid move.
280  action_values[0] = row;
281  action_values[1] = column;
282  action_values[2] = direction;
283 
284  move_list.push_back(
285  RankActionMixedBase(action_bases, action_values));
286  }
287  }
288  }
289  }
290  }
291  }
292 
293  return move_list;
294 }

338 bool ClobberState::MovesRemaining() const {
339  for (int row = 0; row < rows_; row++) {
340  for (int column = 0; column < columns_; column++) {
341  CellState current_cell_state = BoardAt(row, column);
342 
343  if (current_cell_state == CellState::kEmpty) {
344  continue;
345  }
346 
347  for (int direction = 0; direction < kNumDirections; direction++) {
348  int adjacent_row = row + kDirRowOffsets[direction];
349  int adjacent_column = column + kDirColumnOffsets[direction];
350 
351  if (InBounds(adjacent_row, adjacent_column)) {
352  CellState adjacent_state = BoardAt(adjacent_row, adjacent_column);
353  CellState opponent_state = OpponentState(current_cell_state);
354 
355  if (adjacent_state == opponent_state) {
356  return true;
357  }
358  }
359  }
360  }
361  }
362 
363  return false;
364 }

390 void ClobberState::ObservationTensor(Player player,
391  absl::Span<float> values) const {
392  SPIEL_CHECK_GE(player, 0);
393  SPIEL_CHECK_LT(player, num_players_);
394 
395  TensorView<3> view(values, {kCellStates, rows_, columns_},
396  true);
397 
398  // Observation Tensor Representation:
399  // Plane 0: 1's where the current player's pieces are, 0's elsewhere.
400  // Plane 1: 1's where the oppponent's pieces are, 0's elsewhere.
401  // Plane 2: 1's where the empty cells are, 0's elsewhere.
402  for (int row = 0; row < rows_; row++) {
403  for (int column = 0; column < columns_; column++) {
404  int plane = ObservationPlane(BoardAt(row, column), player);
405  view[{plane, row, column}] = 1.0;
406  }
407  }

432  history_.pop_back();
