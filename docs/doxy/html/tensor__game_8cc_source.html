31// Check the utilities to see if the game is constant-sum or identical
32// (cooperative).
33GameType::Utility GetUtilityType(
34 const std::vector<std::vector<double>>& utils) {
35 double util_sum = 0;
36 // Assume both are true until proven otherwise.
37 bool constant_sum = true;
38 bool identical = true;
39 for (int i = 0; i < utils[0].size(); ++i) {
40 double util_sum_i = 0;
41 for (int player = 0; player < utils.size(); ++player) {
42 util_sum_i += utils[player][i];
43 }
44
45 if (i == 0) {
46 util_sum = util_sum_i;
47 } else {
48 if (constant_sum && !Near(util_sum_i, util_sum)) {
49 constant_sum = false;
50 }
51 }
52
53 if (identical) {
54 for (int player = 1; player < utils.size(); ++player) {
55 if (utils[0][i] != utils[player][i]) {
56 identical = false;
57 break;
58 }
59 }
60 }
61 }
62
63 if (constant_sum && Near(util_sum, 0.0)) {
64 return GameType::Utility::kZeroSum;
65 } else if (constant_sum) {
66 return GameType::Utility::kConstantSum;
67 } else if (identical) {
68 return GameType::Utility::kIdentical;
69 } else {
70 return GameType::Utility::kGeneralSum;
71 }
72}

98 for (Player player = 0; player < shape.size(); ++player) {

106// Create a matrix game with the specified utilities and row/column names.
107// Utilities must be in row-major form.
108
109std::shared_ptr<const TensorGame> CreateTensorGame(
110 const std::string& short_name, const std::string& long_name,
111 const std::vector<std::vector<std::string>>& action_names,
112 const std::vector<std::vector<double>>& utils) {
113 const int size =
114 std::accumulate(action_names.begin(), action_names.end(), 1,
115 [](const int s, auto names) { return s * names.size(); });
116 SPIEL_CHECK_TRUE(
117 std::all_of(utils.begin(), utils.end(), [size](const auto& player_utils) {
118 return player_utils.size() == size;
119 }));
120
121 // Detect the utility type from the utilities.
122 const GameType::Utility utility = GetUtilityType(utils);
123
124 const GameType game_type{
125 /*short_name=*/short_name,
126 /*long_name=*/long_name,
127 GameType::Dynamics::kSimultaneous,
128 GameType::ChanceMode::kDeterministic,
129 GameType::Information::kOneShot,
130 utility,
131 GameType::RewardModel::kTerminal,
132 /*max_num_players=*/static_cast<int>(utils.size()),
133 /*min_num_players=*/static_cast<int>(utils.size()),
134 /*provides_information_state_string=*/true,
135 /*provides_information_state_tensor=*/true,
136 /*provides_observation_string=*/false,
137 /*provides_observation_tensor=*/false,
138 /*parameter_specification=*/{} // no parameters
139 };
140
141 return std::shared_ptr<const TensorGame>(
142 new TensorGame(game_type, {}, action_names, utils));
143}
