30 void TestMoveActionId() {
31  std::vector<SuitType> suit_order = {SuitType::kSpades, SuitType::kHearts,
32  SuitType::kClubs, SuitType::kDiamonds};
33  std::vector<RankType> rank_order = {
34  RankType::k2, RankType::k3, RankType::k4, RankType::k5,
35  RankType::k6, RankType::k7, RankType::k8, RankType::k9,
36  RankType::kT, RankType::kJ, RankType::kQ,
37  };
38  std::vector<LocationType> location_order = {LocationType::kFoundation,
39  LocationType::kTableau};
40 
41  std::vector<Move> valid_moves = {};
42 
43  // Adds cards for normal moves (excludes ace and king targets)
44  for (const auto &suit : suit_order) {
45  for (const auto &rank : rank_order) {
46  for (const auto &location : location_order) {
47  auto target_card = Card(false, suit, rank, location);
48  for (const auto &child : target_card.LegalChildren()) {
49  valid_moves.emplace_back(target_card, child);
50  }
51  }
52  }
53  }
54 
55  // Adds ace-to-empty-foundation moves
56  for (const auto &suit : suit_order) {
57  valid_moves.emplace_back(RankType::kNone, suit, RankType::kA, suit);
58  }
59 
60  // Adds king-to-empty-tableau moves
61  for (const auto &suit : suit_order) {
62  valid_moves.emplace_back(RankType::kNone, SuitType::kNone, RankType::kK,
63  suit);
64  }
65 
66  // Adds 2-to-ace moves
67  for (const auto &suit : suit_order) {
68  valid_moves.emplace_back(RankType::kA, suit, RankType::k2, suit);
69  }
70 
71  // Adds queen-to-king moves
72  for (const auto &suit : suit_order) {
73  auto target_card = Card(false, suit, RankType::kK, LocationType::kTableau);
74  for (const auto &child : target_card.LegalChildren()) {
75  valid_moves.emplace_back(target_card, child);
76  }
77  }
78 
79  // Checks that the action id of a move can be converted back into the original
80  // move
81  for (const auto &move : valid_moves) {
82  std::cout << move.ToString() << " == " << Move(move.ActionId()).ToString()
83  << std::endl;
84  SPIEL_CHECK_EQ(move.ToString(), Move(move.ActionId()).ToString());
85  }
86 }
