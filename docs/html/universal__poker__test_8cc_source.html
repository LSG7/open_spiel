158  "50,firstPlayer=2 1 1 "

160  "400)");
161  std::unique_ptr<State> state = game->NewInitialState();
162  while (state->IsChanceNode()) {
163  state->ApplyAction(state->LegalActions()[0]);
164  }
165  std::cout << state->InformationStateString() << std::endl;
166  // Pot bet: call 50, and raise by 200.
167  state->ApplyAction(universal_poker::kBet);
168 
169  // Now, the minimum bet size is larger than the pot, so player 0 can only
170  // fold, call, or go all-in.
171  std::vector<Action> actions = state->LegalActions();
172  absl::c_sort(actions);
173

176  SPIEL_CHECK_EQ(actions[1], universal_poker::kCall);
177  SPIEL_CHECK_EQ(actions[2], universal_poker::kAllIn);

256 // Checks min raising functionality.
257 void FullNLBettingTest1() {
258  std::shared_ptr<const Game> game = LoadGame(
259  "universal_poker(betting=nolimit,"
260  "numPlayers=2,"
261  "numRounds=4,"
262  "blind=2 1,"
263  "firstPlayer=2 1 1 1,"
264  "numSuits=4,"
265  "numRanks=13,"
266  "numHoleCards=2,"
267  "numBoardCards=0 3 1 1,"
268  "stack=20 20,"
269  "bettingAbstraction=fullgame)");
270  std::unique_ptr<State> state = game->NewInitialState();
271  SPIEL_CHECK_EQ(game->NumDistinctActions(), 21);
272  while (state->IsChanceNode())
273  state->ApplyAction(state->LegalActions()[0]); // deal hole cards
274  // check valid raise actions, smallest valid raise is double the big blind
275  SPIEL_CHECK_FALSE(absl::c_binary_search(state->LegalActions(), 3));
276  for (int i = 4; i <= 20; ++i)
277  SPIEL_CHECK_TRUE(absl::c_binary_search(state->LegalActions(), i));
278  SPIEL_CHECK_FALSE(absl::c_binary_search(state->LegalActions(), 21));
279  state->ApplyAction(1); // call big blind
280  state->ApplyAction(1); // check big blind
281  for (int i = 0; i < 3; ++i)
282  state->ApplyAction(state->LegalActions()[0]); // deal flop
283  // check valid raise actions, smallest valid raise is double the big blind
284  SPIEL_CHECK_FALSE(absl::c_binary_search(state->LegalActions(), 3));
285  for (int i = 4; i <= 20; ++i)
286  SPIEL_CHECK_TRUE(absl::c_binary_search(state->LegalActions(), i));
287  SPIEL_CHECK_FALSE(absl::c_binary_search(state->LegalActions(), 21));
288  // each player keeps min raising until one is all in
289  for (int i = 4; i <= 20; i += 2) state->ApplyAction(i);
290  state->ApplyAction(1); // call last raise
291  state->ApplyAction(state->LegalActions()[0]); // deal turn
292  state->ApplyAction(state->LegalActions()[0]); // deal river
293  SPIEL_CHECK_EQ(state->Returns()[0], state->Returns()[1]); // hand is a draw
294  SPIEL_CHECK_TRUE(
295  absl::StrContains(state->ToString(),
296  "ACPC State: STATE:0:cc/r4r6r8r10r12r14r16r18r20c//"
297  ":2c2d|2h2s/3c3d3h/3s/4c"));
298 }

300 // Checks that raises must double previous bet within the same round but
301 // each new round resets betting with the min bet size equal to the big blind.
302 void FullNLBettingTest2() {
303  std::shared_ptr<const Game> game = LoadGame(
304  "universal_poker(betting=nolimit,"
305  "numPlayers=2,"
306  "numRounds=4,"
307  "blind=100 50,"
308  "firstPlayer=2 1 1 1,"
309  "numSuits=4,"
310  "numRanks=13,"
311  "numHoleCards=2,"
312  "numBoardCards=0 3 1 1,"
313  "stack=10000 10000,"
314  "bettingAbstraction=fullgame)");
315  std::unique_ptr<State> state = game->NewInitialState();
316  SPIEL_CHECK_EQ(game->NumDistinctActions(), 10001);
317  while (state->IsChanceNode())
318  state->ApplyAction(state->LegalActions()[0]); // deal hole cards
319  // check valid raise actions
320  std::vector<Action> legal_actions = state->LegalActions();
321  SPIEL_CHECK_FALSE(absl::c_binary_search(legal_actions, 199));
322  for (int i = 200; i <= 10000; ++i)
323  SPIEL_CHECK_TRUE(absl::c_binary_search(legal_actions, i));
324  SPIEL_CHECK_FALSE(absl::c_binary_search(legal_actions, 10001));
325  state->ApplyAction(5100); // bet just over half stack
326  // raise must double the size of the bet
327  // only legal actions now are fold, call, raise all-in
328  SPIEL_CHECK_EQ(state->LegalActions().size(), 3);
329  SPIEL_CHECK_EQ(state->LegalActions().back(), 10000);
330  state->ApplyAction(1); // call
331  for (int i = 0; i < 3; ++i)
332  state->ApplyAction(state->LegalActions()[0]); // deal flop
333  // new round of betting so we can bet as small as the big blind
334  legal_actions = state->LegalActions();
335  SPIEL_CHECK_FALSE(absl::c_binary_search(legal_actions, 5199));
336  for (int i = 5200; i <= 10000; ++i)
337  SPIEL_CHECK_TRUE(absl::c_binary_search(legal_actions, i));
338  state->ApplyAction(5200); // min bet
339  // now we can raise as small as the big blind or as big as an all-in
340  legal_actions = state->LegalActions();
341  for (int i = 5300; i <= 10000; ++i)
342  SPIEL_CHECK_TRUE(absl::c_binary_search(legal_actions, i));
343  state->ApplyAction(1); // opt just to call
344  state->ApplyAction(state->LegalActions()[0]); // deal turn
345  state->ApplyAction(5400); // bet 2 big blinds
346  state->ApplyAction(5600); // raise to 4 big blinds
347  state->ApplyAction(5900); // reraise to 7 big blinds
348  // now a reraise must increase by at least 3 more big blinds
349  legal_actions = state->LegalActions();
350  SPIEL_CHECK_FALSE(absl::c_binary_search(legal_actions, 6199));
351  for (int i = 6200; i <= 10000; ++i)
352  SPIEL_CHECK_TRUE(absl::c_binary_search(legal_actions, i));
353  state->ApplyAction(1); // opt to just call
354  state->ApplyAction(state->LegalActions()[0]); // deal river
355  // new round of betting so we can bet as small as the big blind
356  legal_actions = state->LegalActions();
357  SPIEL_CHECK_FALSE(absl::c_binary_search(legal_actions, 5999));
358  for (int i = 6000; i <= 10000; ++i)
359  SPIEL_CHECK_TRUE(absl::c_binary_search(legal_actions, i));
360  state->ApplyAction(10000); // all-in!
361  state->ApplyAction(0); // fold
362  SPIEL_CHECK_EQ(state->Returns()[0], 5900);
363  SPIEL_CHECK_EQ(state->Returns()[1], -5900);
364  SPIEL_CHECK_TRUE(absl::StrContains(state->ToString(),
365  "ACPC State: STATE:0:r5100c/r5200c/r5400r5600r5900c/r10000f"
366  ":2c2d|2h2s/3c3d3h/3s/4c"));
367 }

472  information_state_tensor_set.insert(tensor);
473  state->ApplyAction(1); // call

475  // check the infostate tensor and add to set
476  tensor = state->InformationStateTensor();
477  SPIEL_CHECK_FALSE(information_state_tensor_set.count(tensor));
478  information_state_tensor_set.insert(tensor);
479  state->ApplyAction(1); // check
480  // check the infostate tensor and add to set
481  tensor = state->InformationStateTensor();

484  for (int i=300; i < 2000; i+=100) {
