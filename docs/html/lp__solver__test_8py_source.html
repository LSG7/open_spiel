60  def test_asymmetric_pure_nonzero_val(self):
61  # c0 c1 c2
62  # r0 | 2, -2 | 1, -1 | 5, -5
63  # r1 |-3, 3 | -4, 4 | -2, 2
64  #
65  # Pure eq (r0,c1) for a value of (1, -1)
66  # 2nd row is dominated, and then second player chooses 2nd col.

182  def test_iterated_dominance_auction(self):
183  # find a strategy through iterated dominance that's not strictly dominant
184  auction = pyspiel.extensive_to_matrix_game(
185  pyspiel.load_game("first_sealed_auction(max_value=3)"))
186  auction_dom, auction_live = self._checked_iterated_dominance_checked_iterated_dominance(
187  auction, lp_solver.DOMINANCE_STRICT)
188  # there's just one non-dominated action
189  self.assertEqual(auction_dom.num_rows(), 1)
190  self.assertEqual(auction_dom.num_cols(), 1)
191  best_action = [
192  auction.row_action_name(row) for row in range(auction.num_rows())
193  ].index(auction_dom.row_action_name(0))
194  self.assertTrue(auction_live[0][best_action])
195  # other actions are all weakly but not all strictly dominated
196  self.assertNotIn(False, [
197  lp_solver.is_dominated(action, auction, 0, lp_solver.DOMINANCE_WEAK)
198  for action in range(6)
199  if action != best_action
200  ])
201  self.assertIn(False, [
202  lp_solver.is_dominated(action, auction, 0, lp_solver.DOMINANCE_STRICT)
203  for action in range(6)
204  if action != best_action
