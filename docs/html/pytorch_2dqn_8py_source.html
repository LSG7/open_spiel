36 class SonnetLinear(nn.Module):
37  """A Sonnet linear module.
38 
39  Always includes biases and only supports ReLU activations.
40  """
41 
42  def __init__(self, in_size, out_size, activate_relu=True):
43  """Creates a Sonnet linear layer.
44 
45  Args:
46  in_size: (int) number of inputs
47  out_size: (int) number of outputs
48  activate_relu: (bool) whether to include a ReLU activation layer
49  """
50  super(SonnetLinear, self).__init__()
51  self._activate_relu_activate_relu = activate_relu
52  stddev = 1.0 / math.sqrt(in_size)
53  mean = 0
54  lower = (-2 * stddev - mean) / stddev
55  upper = (2 * stddev - mean) / stddev
56  # Weight initialization inspired by Sonnet's Linear layer,
57  # which cites https://arxiv.org/abs/1502.03167v3
58  # pytorch default: initialized from
59  # uniform(-sqrt(1/in_features), sqrt(1/in_features))
60  self._weight_weight = nn.Parameter(
61  torch.Tensor(
62  stats.truncnorm.rvs(
63  lower, upper, loc=mean, scale=stddev, size=[out_size,
64  in_size])))
65  self._bias_bias = nn.Parameter(torch.zeros([out_size]))
66 
67  def forward(self, tensor):
68  y = F.linear(tensor, self._weight_weight, self._bias_bias)
69  return F.relu(y) if self._activate_relu_activate_relu else y

81 
82  Args:
83  input_size: (int) number of inputs

85  output_size: (int) number of outputs
86  activate_final: (bool) should final layer should include a ReLU
87  """
88 
89  super(MLP, self).__init__()

91  # Hidden layers
92  for size in hidden_sizes:
93  self._layers_layers.append(SonnetLinear(in_size=input_size, out_size=size))
94  input_size = size
95  # Output layer
96  self._layers_layers.append(
97  SonnetLinear(
98  in_size=input_size,

191  def step(self, time_step, is_evaluation=False, add_transition_record=True):
192  """Returns the action to be taken and updates the Q-network if needed.
193 
194  Args:
195  time_step: an instance of rl_environment.TimeStep.
196  is_evaluation: bool, whether this is a training or evaluation call.
197  add_transition_record: Whether to add to the replay buffer on this step.
198 
199  Returns:
200  A `rl_agent.StepOutput` containing the action probs and chosen action.
201  """
202 
203  # Act step: don't act at terminal info states or if its not our turn.
204  if (not time_step.last()) and (
205  time_step.is_simultaneous_move() or
206  self.player_idplayer_id == time_step.current_player()):
207  info_state = time_step.observations["info_state"][self.player_idplayer_id]
208  legal_actions = time_step.observations["legal_actions"][self.player_idplayer_id]
209  epsilon = self._get_epsilon_get_epsilon(is_evaluation)
210  action, probs = self._epsilon_greedy_epsilon_greedy(info_state, legal_actions, epsilon)
211  else:
212  action = None
213  probs = []
214 
215  # Don't mess up with the state during evaluation.
216  if not is_evaluation:
217  self._step_counter_step_counter += 1
218

220  self._last_loss_value_last_loss_value = self.learnlearn()
221

223  # state_dict method returns a dictionary containing a whole state of the
224  # module.
225  self._target_q_network_target_q_network.load_state_dict(self._q_network_q_network.state_dict())
226 
227  if self._prev_timestep_prev_timestep and add_transition_record:
228  # We may omit record adding here if it's done elsewhere.
229  self.add_transitionadd_transition(self._prev_timestep_prev_timestep, self._prev_action_prev_action, time_step)
230 
231  if time_step.last(): # prepare for the next episode.
232  self._prev_timestep_prev_timestep = None
233  self._prev_action_prev_action = None

241  def add_transition(self, prev_time_step, prev_action, time_step):
242  """Adds the new transition using `time_step` to the replay buffer.
243 
244  Adds the transition from `self._prev_timestep` to `time_step` by
245  `self._prev_action`.
246 
247  Args:
248  prev_time_step: prev ts, an instance of rl_environment.TimeStep.
249  prev_action: int, action taken at `prev_time_step`.
250  time_step: current ts, an instance of rl_environment.TimeStep.
251  """
252  assert prev_time_step is not None
253  legal_actions = (time_step.observations["legal_actions"][self.player_idplayer_id])
254  legal_actions_mask = np.zeros(self._num_actions_num_actions)
255  legal_actions_mask[legal_actions] = 1.0
256  transition = Transition(
257  info_state=(
258  prev_time_step.observations["info_state"][self.player_idplayer_id][:]),
259  action=prev_action,
260  reward=time_step.rewards[self.player_idplayer_id],
261  next_info_state=time_step.observations["info_state"][self.player_idplayer_id][:],

266  def _epsilon_greedy(self, info_state, legal_actions, epsilon):
267  """Returns a valid epsilon-greedy action and valid action probs.
268 
269  Action probabilities are given by a softmax over legal q-values.
270 
271  Args:
272  info_state: hashable representation of the information state.
273  legal_actions: list of legal actions at `info_state`.
274  epsilon: float, probability of taking an exploratory action.
275 
276  Returns:
277  A valid epsilon-greedy action and valid action probabilities.
278  """
279  probs = np.zeros(self._num_actions_num_actions)
280  if np.random.rand() < epsilon:
281  action = np.random.choice(legal_actions)
282  probs[legal_actions] = 1.0 / len(legal_actions)
283  else:
284  info_state = torch.Tensor(np.reshape(info_state, [1, -1]))
285  q_values = self._q_network_q_network(info_state).detach()[0]
286  legal_q_values = q_values[legal_actions]
287  action = legal_actions[torch.argmax(legal_q_values)]
288  probs[action] = 1.0
289  return action, probs

291  def _get_epsilon(self, is_evaluation, power=1.0):
292  """Returns the evaluation or decayed epsilon value."""
293  if is_evaluation:
294  return 0.0
295  decay_steps = min(self._step_counter_step_counter, self._epsilon_decay_duration_epsilon_decay_duration)
296  decayed_epsilon = (
297  self._epsilon_end_epsilon_end + (self._epsilon_start_epsilon_start - self._epsilon_end_epsilon_end) *
298  (1 - decay_steps / self._epsilon_decay_duration_epsilon_decay_duration)**power)
299  return decayed_epsilon

301  def learn(self):
302  """Compute the loss on sampled transitions and perform a Q-network update.
303 
304  If there are not enough elements in the buffer, no loss is computed and
305  `None` is returned instead.
306 
307  Returns:
308  The average loss obtained on this batch of transitions or `None`.
309  """
310 
311  if (len(self._replay_buffer_replay_buffer) < self._batch_size_batch_size or
312  len(self._replay_buffer_replay_buffer) < self._min_buffer_size_to_learn_min_buffer_size_to_learn):
313  return None
314 
315  transitions = self._replay_buffer_replay_buffer.sample(self._batch_size_batch_size)
316  info_states = torch.Tensor([t.info_state for t in transitions])
317  actions = torch.LongTensor([t.action for t in transitions])
318  rewards = torch.Tensor([t.reward for t in transitions])
319  next_info_states = torch.Tensor([t.next_info_state for t in transitions])
320  are_final_steps = torch.Tensor([t.is_final_step for t in transitions])
321  legal_actions_mask = torch.Tensor(
322  np.array([t.legal_actions_mask for t in transitions]))
323 
324  self._q_values_q_values = self._q_network_q_network(info_states)
325  self._target_q_values_target_q_values = self._target_q_network_target_q_network(next_info_states).detach()
326 
327  illegal_actions_mask = 1 - legal_actions_mask
328  legal_target_q_values = self._target_q_values_target_q_values.masked_fill(
329  illegal_actions_mask, ILLEGAL_ACTION_LOGITS_PENALTY)
330  max_next_q = torch.max(legal_target_q_values, dim=1)[0]
331 
332  target = (
333  rewards + (1 - are_final_steps) * self._discount_factor_discount_factor * max_next_q)
334  action_indices = torch.stack([
335  torch.arange(self._q_values_q_values.shape[0], dtype=torch.long), actions
336  ],
337  dim=0)
338  predictions = self._q_values_q_values[list(action_indices)]
339 
340  loss = self.loss_classloss_class(predictions, target)
341 
342  self._optimizer_optimizer.zero_grad()
343  loss.backward()
344  self._optimizer_optimizer.step()
345 
346  return loss

377  def copy_with_noise(self, sigma=0.0, copy_weights=True):
378  """Copies the object and perturbates it with noise.
379 
380  Args:

382  (1+sigma*epsilon), epsilon standard gaussian variable, multiplies each
383  model weight. sigma=0 means no perturbation.

385  just model hyperparameters.
386 
387  Returns:
388  Perturbated copy of the model.

393  q_network = getattr(copied_object, "_q_network")
394  target_q_network = getattr(copied_object, "_target_q_network")
395 
396  if copy_weights:
397  with torch.no_grad():
398  for q_model in q_network.model:
399  q_model.weight *= (1 + sigma * torch.randn(q_model.weight.shape))
400  for tq_model in target_q_network.model:
401  tq_model.weight *= (1 + sigma * torch.randn(tq_model.weight.shape))
402  return copied_object
