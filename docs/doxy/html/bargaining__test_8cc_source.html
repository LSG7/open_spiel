50void BargainingMaxTurnsTest() {
51 std::shared_ptr<const Game> game = LoadGame("bargaining(max_turns=200)");
52 std::unique_ptr<State> state = game->NewInitialState();
53 int num_turns = 200;
54 while (num_turns > 0) {
55 if (state->IsChanceNode()) {
56 ActionsAndProbs chance_outcomes = state->ChanceOutcomes();
57 state->ApplyAction(chance_outcomes[0].first);
58 } else {
59 SPIEL_CHECK_TRUE(!state->IsTerminal());
60 num_turns--;
61 std::vector<Action> legal_actions = state->LegalActions();
62 state->ApplyAction(legal_actions[0]);
63 }
64 }
65 SPIEL_CHECK_TRUE(state->IsTerminal());
66}

68void BargainingDiscountTest() {
69 std::shared_ptr<const Game> game = LoadGame("bargaining(discount=0.9)");
70 std::unique_ptr<State> state = game->NewInitialState();
71 BargainingState* bargaining_state =
72 static_cast<BargainingState*>(state.get());
73 ActionsAndProbs chance_outcomes = state->ChanceOutcomes();
74 state->ApplyAction(chance_outcomes[0].first);
75 std::vector<Action> legal_actions = state->LegalActions();
76 state->ApplyAction(legal_actions[0]);
77 state->ApplyAction(legal_actions[0]);
78 state->ApplyAction(legal_actions[0]);
79 state->ApplyAction(legal_actions[0]);
80 state->ApplyAction(bargaining_state->AgreeAction());
81 // P0 offers [0,0,0] then P1, then P0, then P1, then P0 agrees.
82 // P0 would get 10, but it's discounted by 0.9 three times
83 SPIEL_CHECK_FLOAT_EQ(state->PlayerReturn(0), 0.9 * 0.9 * 0.9 * 10);
84 SPIEL_CHECK_FLOAT_EQ(state->PlayerReturn(1), 0.0);
85}

87void BargainingProbEndContinueTest() {
88 std::shared_ptr<const Game> game = LoadGame("bargaining(prob_end=0.1)");
89 std::unique_ptr<State> state = game->NewInitialState();
90 state->ApplyAction(state->ChanceOutcomes()[0].first);
91 std::vector<Action> legal_actions = state->LegalActions();
92 state->ApplyAction(legal_actions[0]);
93 state->ApplyAction(legal_actions[0]);
94 for (int i = 0; i < (bargaining::kDefaultMaxTurns - 2); ++i) {
95 SPIEL_CHECK_TRUE(state->IsChanceNode());
96 state->ApplyAction(state->ChanceOutcomes()[0].first);
97 SPIEL_CHECK_TRUE(!state->IsChanceNode());
98 legal_actions = state->LegalActions();
99 state->ApplyAction(legal_actions[0]);
100 }
101 SPIEL_CHECK_TRUE(state->IsTerminal());
102}

104void BargainingProbEndEndTest() {
105 std::shared_ptr<const Game> game = LoadGame("bargaining(prob_end=0.1)");
106 std::unique_ptr<State> state = game->NewInitialState();
107 state->ApplyAction(state->ChanceOutcomes()[0].first);
108 std::vector<Action> legal_actions = state->LegalActions();
109 state->ApplyAction(legal_actions[0]);
110 state->ApplyAction(legal_actions[0]);

112 SPIEL_CHECK_TRUE(state->IsChanceNode());
113 state->ApplyAction(state->ChanceOutcomes()[0].first);
114 SPIEL_CHECK_TRUE(!state->IsChanceNode());
115 legal_actions = state->LegalActions();
116 state->ApplyAction(legal_actions[0]);
117 }
118 SPIEL_CHECK_TRUE(state->IsChanceNode());
119 SPIEL_CHECK_TRUE(!state->IsTerminal());
120 state->ApplyAction(state->ChanceOutcomes()[1].first);
121 SPIEL_CHECK_TRUE(state->IsTerminal());
122 SPIEL_CHECK_FLOAT_EQ(state->PlayerReturn(0), 0.0);
123 SPIEL_CHECK_FLOAT_EQ(state->PlayerReturn(1), 0.0);
124}
