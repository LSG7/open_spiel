38std::string GenerateBoard(std::mt19937* rng) {
39 bool valid_board = false;
40 std::string board_string;
41
42 while (!valid_board) {
43 Board board;
44 // Generate the player's chips.
45 int width = kNumChipsUpperBound - kNumChipsLowerBound + 1;
46 for (int p = 0; p < board.num_players; ++p) {
47 // First their number of chips.
48 board.num_chips[p] =
49 kNumChipsLowerBound + absl::Uniform<int>(*rng, 0, width);
50 // Then, their chips
51 for (int i = 0; i < board.num_chips[p]; ++i) {
52 int chip = absl::Uniform<int>(*rng, 0, board.num_colors);
53 board.chips[p][chip]++;
54 }
55 }
56
57 // Now, the board.
58 for (int r = 0; r < board.size; ++r) {
59 for (int c = 0; c < board.size; ++c) {
60 int idx = r * board.size + c;
61 board.board[idx] = absl::Uniform<int>(*rng, 0, board.num_colors);
62 }
63 }
64
65 // Now the player positions.
66 // The flag position is the last one, hence positions.size() here
67 for (int p = 0; p < board.positions.size(); ++p) {
68 int candidate = -1;
69 while (absl::c_find(board.positions, candidate) !=
70 board.positions.end()) {
71 candidate = absl::Uniform<int>(*rng, 0, board.size * board.size);
72 }
73 board.positions[p] = candidate;
74 }
75
76 // Check the board.
77 valid_board = CheckBoard(board);
78 board_string = board.ToString();
79 }
80
81 return board_string;
82}

84void GenerateBoards(int num) {
85 std::string filename = absl::GetFlag(FLAGS_filename);
86 int seed = absl::GetFlag(FLAGS_seed);
87 std::mt19937 rng(seed);
88
89 std::cout << "Starting." << std::endl;
90 TradeInfo trade_info;
91 InitTradeInfo(&trade_info, kDefaultNumColors);
92 std::cout << "Num combos: " << trade_info.chip_combinations.size()
93 << ", possible trades " << trade_info.possible_trades.size()
94 << std::endl;
95
96 std::cout << "Opening file: " << filename << std::endl;
97 open_spiel::file::File outfile(filename, "w");
98 for (int i = 0; i < num; ++i) {
99 std::cout << "Generating board " << i << std::endl;
100 std::string line = GenerateBoard(&rng);
101 line.push_back('\n');
102 std::cout << line;
103 outfile.Write(line);
104 }
105 std::cout << "Wrote to file: " << filename << std::endl;
106}
