84 Action DQN::Step(const State& state, bool is_evaluation) {
85  // Chance nodes should be handled externally to the agent.
86  SPIEL_CHECK_TRUE(!state.IsChanceNode());
87 
88  Action action;
89  if (!state.IsTerminal() &&
90  (state.CurrentPlayer() == player_id_ || state.IsSimultaneousNode())) {
91  std::vector<float> info_state = GetInfoState(state,
92  player_id_,
93  use_observation_);
94  std::vector<Action> legal_actions = state.LegalActions(player_id_);
95  double epsilon = GetEpsilon(is_evaluation);
96  action = EpsilonGreedy(info_state, legal_actions, epsilon);
97  } else {
98  action = 0;
99  }
100 
101  if (!is_evaluation) {
102  step_counter_++;
103 
104  if (step_counter_ % learn_every_ == 0) {
105  Learn();
106  }
107  if (step_counter_ % update_target_network_every_ == 0) {
108  std::stringstream stream;
109  torch::save(q_network_, stream);
110  torch::load(target_q_network_, stream);
111  }
112  if (exists_prev_) {
113  AddTransition(*prev_state_, prev_action_, state);
114  }
115  }
116 
117  if (state.IsTerminal()) {
118  exists_prev_ = false;
119  prev_action_ = 0;
120  prev_state_ = nullptr;
121  return kInvalidAction;
122  } else {
123  exists_prev_ = true;
124  prev_state_ = state.Clone();
125  prev_action_ = action;
126  }
127 
128  return action;
129 }

131 void DQN::AddTransition(const State& prev_state,
132  Action prev_action,
133  const State& state) {
134  Transition transition = {
135  /*info_state=*/GetInfoState(prev_state, player_id_, use_observation_),
136  /*action=*/prev_action_,
137  /*reward=*/state.PlayerReward(player_id_),
138  /*next_info_state=*/GetInfoState(state, player_id_, use_observation_),
139  /*is_final_step=*/state.IsTerminal(),
140  /*legal_actions_mask=*/state.LegalActionsMask()};
141  replay_buffer_.Add(transition);
142 }

144 Action DQN::EpsilonGreedy(std::vector<float> info_state,
145  std::vector<Action> legal_actions,
146  double epsilon) {
147  Action action;
148  if (legal_actions.empty()) {
149  // In some simultaneous games, some players can have no legal actions.
150  return 0;
151  } else if (legal_actions.size() == 1) {
152  return legal_actions[0];
153  }
154 
155  if (absl::Uniform(rng_, 0.0, 1.0) < epsilon) {
156  ActionsAndProbs actions_probs;
157  std::vector<double> probs(legal_actions.size(), 1.0 / legal_actions.size());
158  for (int i = 0; i < legal_actions.size(); i++) {
159  actions_probs.push_back({legal_actions[i], probs[i]});
160  }
161  action = SampleAction(actions_probs, rng_).first;
162  } else {
163  torch::Tensor info_state_tensor =
164  torch::from_blob(info_state.data(), {info_state.size()},
165  torch::dtype(torch::kFloat32))
166  .view({1, -1});
167  q_network_->eval();
168  torch::Tensor q_values = q_network_->forward(info_state_tensor).detach();
169  torch::Tensor illegal_actions_mask =
170  torch::full({num_actions_}, true, torch::dtype(torch::kBool));
171  for (const auto& action : legal_actions) {
172  illegal_actions_mask[action] = false;
173  }
174  torch::Tensor legal_q_values = torch::masked_fill(
175  q_values, illegal_actions_mask, kIllegalActionLogitsPenalty);
176  action = legal_q_values.argmax(1).item().toInt();
177  }
178  return action;
179 }

181 double DQN::GetEpsilon(bool is_evaluation, int power) {
182  if (is_evaluation) {
183  return 0.0;
184  }
185 
186  double decay_steps = std::min(
187  static_cast<double>(step_counter_), epsilon_decay_duration_);
188  double decayed_epsilon = (
189  epsilon_end_ + (epsilon_start_ - epsilon_end_) *
190  std::pow((1 - decay_steps / epsilon_decay_duration_), power));
191  return decayed_epsilon;
192 }

194 void DQN::Learn() {
195  if (replay_buffer_.Size() < batch_size_
196  || replay_buffer_.Size() < min_buffer_size_to_learn_) return;
197  std::vector<Transition> transition = replay_buffer_.Sample(&rng_,
198  batch_size_);
199  std::vector<torch::Tensor> info_states;
200  std::vector<torch::Tensor> next_info_states;
201  std::vector<torch::Tensor> legal_actions_mask;
202  std::vector<Action> actions;
203  std::vector<float> rewards;
204  std::vector<int> are_final_steps;
205  for (auto t : transition) {
206  info_states.push_back(
207  torch::from_blob(
208  t.info_state.data(),
209  {1, t.info_state.size()},
210  torch::TensorOptions().dtype(torch::kFloat32)).clone());
211  next_info_states.push_back(
212  torch::from_blob(
213  t.next_info_state.data(),
214  {1, t.next_info_state.size()},
215  torch::TensorOptions().dtype(torch::kFloat32)).clone());
216  legal_actions_mask.push_back(
217  torch::from_blob(t.legal_actions_mask.data(),
218  {1, t.legal_actions_mask.size()},
219  torch::TensorOptions().dtype(torch::kInt32))
220  .clone());
221  actions.push_back(t.action);
222  rewards.push_back(t.reward);
223  are_final_steps.push_back(t.is_final_step);
224  }
225  torch::Tensor info_states_tensor = torch::stack(info_states, 0);
226  torch::Tensor next_info_states_tensor = torch::stack(next_info_states, 0);
227  q_network_->train();
228  torch::Tensor q_values = q_network_->forward(info_states_tensor);
229  target_q_network_->eval();
230  torch::Tensor target_q_values = target_q_network_->forward(
231  next_info_states_tensor).detach();
232 
233  torch::Tensor illegal_action_masks_tensor =
234  1.0 - torch::stack(legal_actions_mask, 0);
235  torch::Tensor legal_q_values =
236  torch::masked_fill(target_q_values, illegal_action_masks_tensor,
237  kIllegalActionLogitsPenalty);
238 
239  torch::Tensor max_next_q = std::get<0>(legal_q_values.max(2));
240 
241  torch::Tensor are_final_steps_tensor = torch::from_blob(
242  are_final_steps.data(),
243  {batch_size_},
244  torch::TensorOptions().dtype(torch::kInt32)).to(torch::kFloat32);
245  torch::Tensor rewards_tensor = torch::from_blob(
246  rewards.data(),
247  {batch_size_},
248  torch::TensorOptions().dtype(torch::kFloat32));
249  torch::Tensor target = rewards_tensor + (
250  1.0 - are_final_steps_tensor) * max_next_q.squeeze(1) * discount_factor_;
251  torch::Tensor actions_tensor = torch::from_blob(
252  actions.data(),
253  {batch_size_},
254  torch::TensorOptions().dtype(torch::kInt64));
255  torch::Tensor predictions = q_values.index(
256  {torch::arange(q_values.size(0)),
257  torch::indexing::Slice(),
258  actions_tensor});
259 
260  optimizer_.zero_grad();
261  torch::Tensor value_loss;
262  if (loss_str_ == "mse") {
263  torch::nn::MSELoss mse_loss;
264  value_loss = mse_loss(predictions.squeeze(1), target);
265  } else if (loss_str_ == "huber") {
266  torch::nn::SmoothL1Loss l1_loss;
267  value_loss = l1_loss(predictions.squeeze(1), target);
268  } else {
269  SpielFatalError("Not implemented, choose from 'mse', 'huber'.");
270  }
271  value_loss.backward();
272  optimizer_.step();
273 }

292 std::vector<double> RunEpisodes(std::mt19937* rng, const Game& game,
293  const std::vector<Agent*>& agents,
294  int num_episodes, bool is_evaluation) {
295  SPIEL_CHECK_GE(num_episodes, 1);
296  SPIEL_CHECK_EQ(agents.size(), game.NumPlayers());
297  std::vector<double> total_returns(game.NumPlayers(), 0.0);
298  for (int i = 0; i < num_episodes; i++) {
299  std::unique_ptr<open_spiel::State> state = game.NewInitialState();
300  while (!state->IsTerminal()) {
301  Player player = state->CurrentPlayer();
302  open_spiel::Action action;
303  if (state->IsChanceNode()) {
304  action = open_spiel::SampleAction(state->ChanceOutcomes(),
305  absl::Uniform(*rng, 0.0, 1.0))
306  .first;
307  state->ApplyAction(action);
308  } else if (state->IsSimultaneousNode()) {
309  std::vector<Action> joint_action(game.NumPlayers());
310  for (Player p = 0; p < game.NumPlayers(); ++p) {
311  joint_action[p] = agents[p]->Step(*state, is_evaluation);
312  }
313  state->ApplyActions(joint_action);
314  } else {
315  action = agents[player]->Step(*state, is_evaluation);
316  state->ApplyAction(action);
317  }
318  }
319  std::vector<double> episode_returns = state->Returns();
320  for (Player p = 0; p < game.NumPlayers(); ++p) {
321  agents[p]->Step(*state, is_evaluation);
322  total_returns[p] += episode_returns[p];
323  }
324  }
325 
326  for (Player p = 0; p < game.NumPlayers(); ++p) {
327  total_returns[p] /= num_episodes;
328  }
329 
330  return total_returns;
331 }
