16"""Export game trees in gambit format.
17
18An exporter for the .efg format used by Gambit:
19http://www.gambit-project.org/gambit14/formats.html
20
21See `examples/gambit_example.py` for an example of usage.
22
23"""
24
25import collections
26import functools
27
28
29def quote(x):
30 return f"\"{x}\""
31
32
33def export_gambit(game):
34 """Builds gambit representation of the game tree.
35
36 Args:
37 game: A `pyspiel.Game` object.
38
39 Returns:
40 string: Gambit tree
41 """
42 players = " ".join([f"\"Pl{i}\"" for i in range(game.num_players())])
43 ret = f"EFG 2 R {quote(game)} {{ {players} }} \n"
44
45 terminal_idx = 1
46 chance_idx = 1
47
48 # We will keep separate infoset idx per each player.
49 # Note that gambit infosets start at 1, but we start them here at 0 because
50 # they get incremented when accessed from infoset_tables below.
51 infoset_idx = [0] * game.num_players()
52
53 def infoset_next_id(player):
54 nonlocal infoset_idx
55 infoset_idx[player] += 1
56 return infoset_idx[player]
57
58 infoset_tables = [
59 collections.defaultdict(functools.partial(infoset_next_id, player))
60 for player in range(game.num_players())
61 ]
62
63 def build_tree(state, depth):
64 nonlocal ret, terminal_idx, chance_idx, infoset_tables
65
66 ret += " " * depth # add nice spacing
67 state_str = str(state)
68 if len(state_str) > 10:
69 state_str = ""
70
71 if state.is_terminal():
72 utils = " ".join(map(str, state.returns()))
73 ret += f"t {quote(state_str)} {terminal_idx} \"\" {{ {utils} }}\n"
74 terminal_idx += 1
75 return
76
77 if state.is_chance_node():
78 ret += f"c {quote(state_str)} {chance_idx} \"\" {{ "
79 for action, prob in state.chance_outcomes():
80 action_str = state.action_to_string(state.current_player(), action)
81 ret += f"{quote(action_str)} {prob:.16f} "
82 ret += " } 0\n"
83 chance_idx += 1
84
85 else: # player node
86 player = state.current_player()
87 gambit_player = player + 1 # cannot be indexed from 0
88 infoset = state.information_state_string()
89 infoset_idx = infoset_tables[player][infoset]
90
91 ret += f"p {quote(state_str)} {gambit_player} {infoset_idx} \"\" {{ "
92 for action in state.legal_actions():
93 action_str = state.action_to_string(state.current_player(), action)
94 ret += f"{quote(action_str)} "
95 ret += " } 0\n"
96
97 for action in state.legal_actions():
98 child = state.child(action)
99 build_tree(child, depth + 1)
100
101 build_tree(game.new_initial_state(), 0)
102 return ret
