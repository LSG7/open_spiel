83 policy_logits = torch.FloatTensor([[1., 1., 1.], [1., 1., 4.]])
84 baseline = torch.FloatTensor([1. / 3, 0.5])
85 actions = torch.LongTensor([1, 2])
86 returns = torch.FloatTensor([0., 1.])

88 # Compute expected quantities.
89 # advantages = returns - baseline = [-1./3, 0.5]

91 # = [1.0986, 0.09492]
92 # policy_loss = cross_entropy * advantages = [-0.3662, 0.04746]
93 expected_policy_entropy = (1.0986 + 0.3665) / 2
94 expected_policy_loss = (-0.3662 + 0.04746) / 2
95 expected_total_loss = (
