28 def test_external_sampling_leduc_2p_simple(self):
29 np.random.seed(SEED)
30 game = pyspiel.load_game("leduc_poker")
31 es_solver = external_sampling_mccfr.ExternalSamplingSolver(
32 game, external_sampling_mccfr.AverageType.SIMPLE)
33 for _ in range(10):
34 es_solver.iteration()
35 conv = exploitability.nash_conv(game, es_solver.average_policy())
36 print("Leduc2P, conv = {}".format(conv))
37 self.assertLess(conv, 5)
38 # ensure that to_tabular() works on the returned policy and
39 # the tabular policy is equivalent
40 tabular_policy = es_solver.average_policy().to_tabular()
41 conv2 = exploitability.nash_conv(game, tabular_policy)
42 self.assertEqual(conv, conv2)
