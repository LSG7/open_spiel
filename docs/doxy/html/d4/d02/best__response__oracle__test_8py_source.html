15"""Tests for open_spiel.python.algorithms.psro_v2.best_response_oracle."""

17from absl.testing import absltest
18from absl.testing import parameterized
19
20from open_spiel.python import policy
21from open_spiel.python.algorithms import best_response
22from open_spiel.python.algorithms.psro_v2 import best_response_oracle
23import pyspiel
24
25
26class BestResponseOracleTest(parameterized.TestCase, absltest.TestCase):
27
28 @parameterized.parameters(("kuhn_poker", 2), ("kuhn_poker", 3),
29 ("leduc_poker", 2))
30 def test_cpp_python_best_response_oracle(self, game_name, num_players):
31 # Tests that these best responses interface well with Best Response Oracle
32 game = pyspiel.load_game(game_name, {"players": num_players})
33 all_states, _ = best_response.compute_states_and_info_states_if_none(
34 game, all_states=None, state_to_information_state=None)
35
36 current_best = [
37 [policy.TabularPolicy(game).__copy__()] for _ in range(num_players)
38 ]
39 probabilities_of_playing_policies = [[1.] for _ in range(num_players)]
40
41 # Construct the python oracle
42 py_oracle = best_response_oracle.BestResponseOracle(
43 best_response_backend="py")
44
45 # Construct the cpp oracle. Note that in this regime, BestResponseOracle
46 # uses base_policy to construct and cache TabularBestResponse internally.
47 cpp_oracle = best_response_oracle.BestResponseOracle(
48 game=game, best_response_backend="cpp")
49
50 # Prepare the computation of the best responses with each backend
51 # pylint:disable=g-complex-comprehension
52 training_params = [[{
53 "total_policies": current_best,
54 "current_player": i,
55 "probabilities_of_playing_policies": probabilities_of_playing_policies
56 }] for i in range(num_players)]
57 # pylint:enable=g-complex-comprehension
58
59 py_best_rep = py_oracle(game, training_params)
60
61 cpp_best_rep = cpp_oracle(game, training_params)
62
63 # Compare the policies
64 for state in all_states.values():
65 i_player = state.current_player()
66 py_dict = py_best_rep[i_player][0].action_probabilities(state)
67 cpp_dict = cpp_best_rep[i_player][0].action_probabilities(state)
68
69 for action in py_dict.keys():
70 self.assertEqual(py_dict.get(action, 0.0), cpp_dict.get(action, 0.0))
71 for action in cpp_dict.keys():
72 self.assertEqual(py_dict.get(action, 0.0), cpp_dict.get(action, 0.0))
73
74
75if __name__ == "__main__":
76 absltest.main()
