40def compress_internal_weights(nus, regrets):
41 """Compress internal weights.
42
43 Via optimization, identify which regret timesteps are useful and which aren't
44 for internal regret.
45
46 Args:
47 nus: Distribution per timestep.
48 regrets: Regret value per timestep and action.
49
50 Returns:
51 Weights over nus which can be used to average the no-regret distribution.
52 """
53
54 def get_c(nus):
55 return np.concatenate((np.array([1.0]), np.zeros(nus.shape[0])))
56
57 def get_max_constraint(regrets):
58 regrets = np.transpose(np.array(regrets), axes=[0, 2, 1])
59 regrets = regrets.reshape(-1, regrets.shape[-1])
60 A = np.zeros((regrets.shape[0], 1 + regrets.shape[1]))
61 A[:, 1:] = regrets
62 A[:, 0] = -1.0
63
64 b = np.zeros(A.shape[0])
65 return A, b
66
67 def get_a_ub(nus, regrets):
68 Amax, bmax = get_max_constraint(regrets)
69 Apos, bpos = get_proba_constraints_positivity(nus)
70 return np.concatenate((Amax, Apos), axis=0), np.concatenate(
71 (bmax, bpos), axis=0
72 )
73
74 c = get_c(nus)
75
76 A_ub, b_ub = get_a_ub(nus, regrets)
77 A_eq, b_eq = get_proba_constraint_sum_eq(nus)
78
79 res = scipy.optimize.linprog(
80 c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq, options={"tol": 1e-10}
81 )
82 new_weights = res.x
83 return new_weights[1:]

86def compress_external_weights(nus, regrets, lbd=0.0):
87 """Compress internal weights.
88
89 Via optimization, identify which regret timesteps are useful and which aren't
90 for external regret.
91
92 Args:
93 nus: Distribution per timestep.
94 regrets: Regret value per timestep and action.
95 lbd: Sparsity penalty.
96
97 Returns:
98 Weights over nus which can be used to average the no-regret distribution.
99 """
100
101 def get_c(nus):
102 return np.concatenate((np.array([1.0]), np.zeros(nus.shape[0])))
103
104 def get_max_constraints(nus, regrets, lbd):
105 A = np.zeros((regrets.shape[1], 1 + nus.shape[0]))
106 A[:, 0] = -1.0
107 A[:, 1:] = np.transpose(
108 regrets
109 - np.sum(regrets * nus, axis=1).reshape(-1, 1)
110 - lbd * np.abs(regrets)
111 )
112 return A, np.zeros(A.shape[0])
113
114 def get_a_ub(nus, regrets, lbd):
115 Amax, bmax = get_max_constraints(nus, regrets, lbd)
116 Apos, bpos = get_proba_constraints_positivity(nus)
117 return np.concatenate((Amax, Apos), axis=0), np.concatenate(
118 (bmax, bpos), axis=0
119 )
120
121 c = get_c(nus)
122
123 A_ub, b_ub = get_a_ub(nus, regrets, lbd)
124 A_eq, b_eq = get_proba_constraint_sum_eq(nus)
125
126 res = scipy.optimize.linprog(
127 c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq, options={"tol": 1e-10}
128 )
129 new_weights = res.x
130 return new_weights[1:]

133# Faster than using scipy.linalg.eig.
134def power_method(w_nus):
135 """Quick implementation of the power method.
136
137 Args:
138 w_nus:
139
140 Returns:
141 Highest eigenvalue of the system.
142
143 Raises:
144 ValueError: when the power method did not converge after 10.000 trials.
145 """
146 p = np.ones(len(w_nus))
147 pprime = np.dot(p, w_nus)
148 n_trials = 10000
149 i = 0
150 while np.sum(np.abs(pprime - p)) > 1e-8 and i < n_trials:
151 p = pprime
152 pprime = np.dot(p, w_nus)
153 pprime[pprime < 0] = 0.0
154 pprime /= np.sum(pprime)
155 i += 1
156
157 if np.sum(np.abs(pprime - p)) > 1e-8 and i >= n_trials:
158 raise ValueError(
159 "Power method did not converge after {} trials.".format(n_trials)
160 )
161
162 p[p < 0] = 0.0
163 return p / np.sum(p)
