55// Example code for using MCCFR on a univeral_poker game loaded from an ACPC
56// gamedef (via the wrapper function).
57int main(int argc, char** argv) {
58 absl::ParseCommandLine(argc, argv);
59 std::cout << "Input ACPC gamedef (raw): " << absl::GetFlag(FLAGS_acpc_gamedef)
60 << std::endl;
61
62 std::shared_ptr<const open_spiel::Game> game =
63 open_spiel::universal_poker::LoadUniversalPokerGameFromACPCGamedef(
64 absl::GetFlag(FLAGS_acpc_gamedef));
65
66 // Downcasting to UniversalPokerGame so we can call GetACPCGame(), which isn't
67 // on the higher level open_spiel::Game.
68 const open_spiel::universal_poker::UniversalPokerGame& game_down_cast =
69 open_spiel::down_cast<
70 const open_spiel::universal_poker::UniversalPokerGame&>(*game);
71 std::cout << "Resulting ACPC gamedef used for universal_poker:\n"
72 << game_down_cast.GetACPCGame()->ToString() << std::endl;
73
74 open_spiel::algorithms::ExternalSamplingMCCFRSolver solver(*game);
75 std::cerr << "Starting MCCFR on " << game->GetType().short_name << "..."
76 << std::endl;
77
78 for (int i = 0; i < absl::GetFlag(FLAGS_num_iters); ++i) {
79 solver.RunIteration();
80 if (i % absl::GetFlag(FLAGS_report_every) == 0 ||
81 i == absl::GetFlag(FLAGS_num_iters) - 1) {
82 double exploitability = open_spiel::algorithms::Exploitability(
83 *game, *solver.AveragePolicy());
84 std::cerr << "Iteration " << i << " exploitability=" << exploitability
85 << std::endl;
86 }
87 }
88}
