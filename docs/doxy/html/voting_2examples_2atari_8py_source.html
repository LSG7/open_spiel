41def main(_):
42 print("Loading dataset(s)...")
43 dataset_filename = (_DATASET_PATH_PREFIX.value + "/" +
44 atari_datasets.RAINBOW_TABLE5)
45 dataset = atari_datasets.parse_atari_table(dataset_filename)
46
47 # If you load others, you can merge some columns from them like this:

51 print(dataset.agent_names)
52 print(dataset.game_names)
53 print(f"Num agents: {len(dataset.agent_names)}")
54 print(f"Num games: {len(dataset.game_names)}")
55
56 # Alts for rainbow table 5:
57 # dqn a3c ddqn prior-ddqn dueling-ddqn distrib-dqn noisy-dqn rainbow
58
59 game_names = []
60 profile = base.PreferenceProfile(alternatives=dataset.agent_names)
61 for game_name, scores in dataset.table_data.items():
62 profile.add_vote_from_values(scores)
63 game_names.append(game_name)
64
65 # Group up the profile and then print it to show that every vote is unique.
66 profile.group()
67 print(profile)
68
69 print("Margin matrix:")
70 margin_matrix = profile.margin_matrix()
71 print(margin_matrix)
72 print(
73 "Weak Condorcet winners? "
74 + f"{profile.condorcet_winner(False, margin_matrix)}"
75 )
76 print(
77 "Strong Condorcet winner? "
78 + f"{profile.condorcet_winner(True, margin_matrix)}"
79 )
