62// History is represented by a string of numbers separated by kActionSeparator.
63std::vector<Action> HistoryFromString(const std::string& str) {
64 std::vector<Action> history;
65 if (str.empty()) return history; // Identity transformation.
66
67 std::vector<absl::string_view> str_actions =
68 absl::StrSplit(str, kActionSeparator);
69 for (const auto& str_action : str_actions) {
70 Action a;
71 if (!absl::SimpleAtoi(str_action, &a)) {
72 SpielFatalError(
73 absl::StrCat("Error when parsing the action: ", str_action));
74 }
75 history.push_back(a);
76 }
77 return history;
78}

80std::unique_ptr<State> StateFromHistory(std::shared_ptr<const Game> game,
81 const std::vector<Action>& history) {
82 std::unique_ptr<State> state = game->NewInitialState();
83 for (const Action& a : history) state->ApplyAction(a);
84 return state;
85}
