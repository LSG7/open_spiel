42void CountStates2p() {
43 std::shared_ptr<const Game> game = LoadGame("tiny_bridge_2p");
44 auto states =
45 open_spiel::algorithms::GetAllStates(*game, /*depth_limit=*/-1,
46 /*include_terminals=*/true,
47 /*include_chance_states=*/false);
48 // Chance nodes are not counted.
49 // For each of 420 deals:
50 // 64 combinations of bids
51 // *2 for initial pass
52 // *2 for terminal pass
53 // -1 for double-counting the auction with a single 'Pass'
54 // => 420 * (64 * 4 - 1) = 107100 states
55 SPIEL_CHECK_EQ(states.size(), 107100);
56}
