62 def compute_regrets(payoff_batch, strategy_x, strategy_y):
63  values_y = -jnp.matmul(strategy_x, payoff_batch)
64  values_x = jnp.transpose(
65  jnp.matmul(payoff_batch, jnp.transpose(strategy_y, [0, 2, 1])), [0, 2, 1])
66  value_x = jnp.matmul(
67  jnp.matmul(strategy_x, payoff_batch),
68  jnp.transpose(strategy_y, [0, 2, 1]))
69  value_y = -value_x
