102 void PhantomTTTState::DoApplyAction(Action move) {
103  // Current player's view.
104  Player cur_player = CurrentPlayer();
105  auto& cur_view = cur_player == 0 ? x_view_ : o_view_;
106 
107  // Two cases: either there is a mark already there, or not.
108  if (state_.BoardAt(move) == CellState::kEmpty) {
109  // No mark on board, so play this normally.
110  state_.ApplyAction(move);
111  }
112 
113  // Update current player's view, and action sequence.
114  SPIEL_CHECK_EQ(cur_view[move], CellState::kEmpty);
115  cur_view[move] = state_.BoardAt(move);
116  action_sequence_.push_back(std::pair<int, Action>(cur_player, move));
117 
118  // Note: do not modify player's turn here, it will have been done above
119  // if necessary.
120 }

151 std::string PhantomTTTState::ActionSequenceToString(Player player) const {
152  SPIEL_CHECK_GE(player, 0);
153  SPIEL_CHECK_LT(player, num_players_);
154 
155  std::string str;
156  for (const auto& player_with_action : action_sequence_) {
157  if (player_with_action.first == player) {
158  // Always include the observing player's actions.
159  absl::StrAppend(&str, player_with_action.first, ",");
160  absl::StrAppend(&str, player_with_action.second, " ");
161  } else if (obs_type_ == ObservationType::kRevealNumTurns) {
162  // If the number of turns are revealed, then each of the other player's
163  // actions will show up as unknowns.
164  absl::StrAppend(&str, player_with_action.first, ",? ");
165  } else {
166  // Do not reveal anything about the number of actions taken by opponent.
167  SPIEL_CHECK_EQ(obs_type_, ObservationType::kRevealNothing);
168  }
169  }
170  return str;
171 }

185 void PhantomTTTState::InformationStateTensor(Player player,
186  absl::Span<float> values) const {
187  SPIEL_CHECK_GE(player, 0);
188  SPIEL_CHECK_LT(player, num_players_);
189 
190  // First 27 bits encodes the player's view in the same way as TicTacToe.
191  // Then the action sequence follows (one-hot encoded, per action).
192  // Encoded in the same way as InformationStateAsString, so full sequences
193  // which may contain action value 10 to represent "I don't know."
194  const auto& player_view = player == 0 ? x_view_ : o_view_;
195  SPIEL_CHECK_EQ(values.size(), kNumCells * kCellStates +
196  kLongestSequence * (1 + kBitsPerAction));
197  std::fill(values.begin(), values.end(), 0.);
198  for (int cell = 0; cell < kNumCells; ++cell) {
199  values[kNumCells * static_cast<int>(player_view[cell]) + cell] = 1.0;
200  }
201 
202  // Now encode the sequence. Each (player, action) pair uses 11 bits:
203  // - first bit is the player taking the action (0 or 1)
204  // - next 10 bits is the one-hot encoded action (10 = "I don't know")
205  int offset = kNumCells * kCellStates;
206  for (const auto& player_with_action : action_sequence_) {
207  if (player_with_action.first == player) {
208  // Always include the observing player's actions.
209  values[offset] = player_with_action.first; // Player 0 or 1
210  values[offset + 1 + player_with_action.second] = 1.0;
211  } else if (obs_type_ == ObservationType::kRevealNumTurns) {
212  // If the number of turns are revealed, then each of the other player's
213  // actions will show up as unknowns.
214  values[offset] = player_with_action.first;
215  values[offset + 1 + 10] = 1.0; // I don't know.
216  } else {
217  // Do not reveal anything about the number of actions taken by opponent.
218  SPIEL_CHECK_EQ(obs_type_, ObservationType::kRevealNothing);
219  }
220 
221  offset += (1 + kBitsPerAction);
222  }
223 }

235 void PhantomTTTState::ObservationTensor(Player player,
236  absl::Span<float> values) const {
237  SPIEL_CHECK_GE(player, 0);
238  SPIEL_CHECK_LT(player, num_players_);
239  SPIEL_CHECK_EQ(values.size(), game_->ObservationTensorSize());
240  std::fill(values.begin(), values.end(), 0.);
241 
242  // First 27 bits encodes the player's view in the same way as TicTacToe.
243  const auto& player_view = player == 0 ? x_view_ : o_view_;
244  for (int cell = 0; cell < kNumCells; ++cell) {
245  values[kNumCells * static_cast<int>(player_view[cell]) + cell] = 1.0;
246  }
247 
248  // Then a one-hot to represent total number of turns.
249  if (obs_type_ == ObservationType::kRevealNumTurns) {
250  values[kNumCells * kCellStates + action_sequence_.size()] = 1.0;

258 void PhantomTTTState::UndoAction(Player player, Action move) {
259  Action last_move = action_sequence_.back().second;
260  SPIEL_CHECK_EQ(last_move, move);
261 
262  if (state_.BoardAt(move) == PlayerToState(player)) {
263  // If the board has a mark that is the undoing player, then this was
264  // a successful move. Undo as normal.
265  state_.UndoAction(player, move);
266  }
267 
268  // Undo the action from that player's view, and pop from the action seq
269  auto& player_view = player == 0 ? x_view_ : o_view_;
270  player_view[move] = CellState::kEmpty;
271  action_sequence_.pop_back();
272 
273  history_.pop_back();
274  --move_number_;
275  // Note, do not change the player.. this will already have been done above
276  // if necessary.
277 }
