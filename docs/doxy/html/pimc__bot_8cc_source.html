58ActionsAndProbs PIMCBot::PolicyFromBestAction(const State& state,
59 Action best_action) const {
60 ActionsAndProbs actions_and_probs;
61 for (Action action : state.LegalActions()) {
62 if (action == best_action) {
63 actions_and_probs.push_back({action, 1.0});
64 } else {
65 actions_and_probs.push_back({action, 0.0});
66 }
67 }
68 return actions_and_probs;

71std::pair<std::vector<int>, Action> PIMCBot::Search(const State& root_state) {
72 int num_determinizations = num_determinizations_;
73
74 GameType type = root_state.GetGame()->GetType();
75 if (type.information == GameType::Information::kPerfectInformation) {
76 num_determinizations = 1;
77 // TODO(author5): drop down to expectimax or alpha-beta if 2-player
78 }
79
80 Player player = root_state.CurrentPlayer();
81 std::vector<Action> legal_actions = root_state.LegalActions();
82 const int num_legal_actions = legal_actions.size();
83 std::vector<int> counts(num_legal_actions, 0);
84 absl::flat_hash_map<Action, int> action_counts;
85 for (Action action : legal_actions) {
86 action_counts[action] = 0;
87 }
88
89 auto rng_func = [this]() {
90 return absl::Uniform<double>(this->rng_, 0.0, 1.0);
91 };
92
93 for (int i = 0; i < num_determinizations; ++i) {
94 std::unique_ptr<State> state = nullptr;
95
96 if (num_determinizations == 1) {
97 state = root_state.Clone();
98 } else {
99 state = root_state.ResampleFromInfostate(player, rng_func);
100 }
101
102 if (type.utility == GameType::Utility::kZeroSum &&
103 type.chance_mode == GameType::ChanceMode::kDeterministic &&
104 root_state.NumPlayers() == 2) {
105 // Special case for two-player zero-sum deterministic games: use
106 // alpha-beta.
107 std::pair<double, Action> search_result = algorithms::AlphaBetaSearch(
108 *state->GetGame(), state.get(),
109 [this, player](const State& state) {
110 return this->value_function_(state, player);
111 },
112 depth_limit_, player, /*use_undo*/ false);
113 action_counts[search_result.second] += 1;
114 } else {
115 std::pair<std::vector<double>, Action> search_result =
116 algorithms::MaxNSearch(*state->GetGame(), state.get(),
117 value_function_, depth_limit_);
118 action_counts[search_result.second] += 1;
119 }
120 }
121
122 Action best_action = kInvalidAction;
123 int highest_count = -1;
124 for (int aidx = 0; aidx < num_legal_actions; ++aidx) {
125 Action action = legal_actions[aidx];
126 counts[aidx] = action_counts[action];
127 if (counts[aidx] > highest_count) {
128 highest_count = counts[aidx];
129 best_action = action;
130 }
131 }
132
133 return {counts, best_action};
134}
