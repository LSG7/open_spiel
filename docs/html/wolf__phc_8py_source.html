31 class WoLFSchedule(rl_tools.ValueSchedule):
32  """Schedule rules described in the WoLF paper.
33 
34  at step t the step size is (t0 / (t + t1))
35  """
36 
37  def __init__(self, t0, t1):
38  super(WoLFSchedule, self).__init__()
39  self._t0_t0 = t0
40  self._t1_t1 = t1
41  self._step_taken_step_taken = 0
42 
43  def step(self):
44  value = (self._t0_t0 / (self._step_taken_step_taken + self._t1_t1))
45  self._step_taken_step_taken += 1
46  return value
47 
48  @property
49  def value(self):
50  return self._t0_t0 / (self._step_taken_step_taken + self._t1_t1)

87  def _hill_climbing(self, info_state, legal_actions):
88  """Does the hill-climbing update.
89 
90  Args:
91  info_state: hashable representation of the information state.
92  legal_actions: list of actions at `info_state`.
93  """
94 
95  greedy_q = max(
96  [self._q_values_q_values[info_state][action] for action in legal_actions])
97  greedy_actions = [
98  action for action in legal_actions
99  if self._q_values_q_values[info_state][action] == greedy_q
100  ]
101  if len(greedy_actions) == len(legal_actions):
102  return
103 
104  deltas = { # pylint: disable=g-complex-comprehension
105  action:
106  min(self._cur_policy_cur_policy[info_state][action],
107  self._cur_delta_value_cur_delta_value / (len(legal_actions) - len(greedy_actions)))
108  for action in legal_actions
109  }
110 
111  delta_greedy = sum([
112  deltas[action]
113  for action in legal_actions
114  if action not in greedy_actions
115  ]) / len(greedy_actions)
116 
117  deltas = {
118  action:
119  -deltas[action] if action not in greedy_actions else delta_greedy
120  for action in legal_actions
121  }
122  new_policy = np.array([
123  self._cur_policy_cur_policy[info_state][action] + deltas[action]
124  for action in legal_actions
125  ])
126  new_policy = _simplex_projection(new_policy)
127  for i in range(len(legal_actions)):
128  self._cur_policy_cur_policy[info_state][legal_actions[i]] = new_policy[i]

130  def _get_action_probs(self, info_state, legal_actions, epsilon):
131  """Returns a selected action and the probabilities of legal actions.
132 
133  To be overwritten by subclasses that implement other action selection
134  methods.
135  Args:
136  info_state: hashable representation of the information state.
137  legal_actions: list of actions at `info_state`.
138  epsilon: float: current value of the epsilon schedule or 0 in case
139  evaluation. QLearner uses it as the exploration parameter in
140  epsilon-greedy, but subclasses are free to interpret in different ways
141  (e.g. as temperature in softmax).
142  """
143  if info_state not in self._cur_policy_cur_policy:
144  for action in legal_actions:
145  self._cur_policy_cur_policy[info_state][action] = 1. / len(legal_actions)
146  self._avg_policy_avg_policy[info_state][action] = 1. / len(legal_actions)
147 
148  probs = np.zeros(self._num_actions_num_actions)
149  for action in legal_actions:
150  probs[action] = ((1-epsilon) * self._cur_policy_cur_policy[info_state][action] +
151  epsilon * 1.0 / len(legal_actions))
152  action = np.random.choice(range(self._num_actions_num_actions), p=probs)
153  return action, probs

155  def step(self, time_step, is_evaluation=False):
156  """Returns the action to be taken and updates the Q-values if needed.
157 
158  Args:
159  time_step: an instance of rl_environment.TimeStep.
160  is_evaluation: bool, whether this is a training or evaluation call.
161 
162  Returns:
163  A `rl_agent.StepOutput` containing the action probs and chosen action.
164  """
165 
166  info_state = str(time_step.observations["info_state"][self._player_id_player_id])
167  legal_actions = time_step.observations["legal_actions"][self._player_id_player_id]
168 
169  # Prevent undefined errors if this agent never plays until terminal step
170  action, probs = None, None
171 
172  # Act step: don't act at terminal states.
173  if not time_step.last():
174  epsilon = 0.0 if is_evaluation else self._epsilon_epsilon
175  action, probs = self._get_action_probs_get_action_probs(info_state, legal_actions, epsilon)
176 
177  # Learn step: don't learn during evaluation or at first agent steps.
178  if self._prev_info_state_prev_info_state and not is_evaluation:
179  target = time_step.rewards[self._player_id_player_id]
180  if not time_step.last(): # Q values are zero for terminal.
181  target += self._discount_factor_discount_factor * max(
182  [self._q_values_q_values[info_state][a] for a in legal_actions])
183 
184  prev_q_value = self._q_values_q_values[self._prev_info_state_prev_info_state][self._prev_action_prev_action]
185  self._last_loss_value_last_loss_value = target - prev_q_value
186  self._q_values_q_values[self._prev_info_state_prev_info_state][self._prev_action_prev_action] += (
187  self._step_size_step_size.value * self._last_loss_value_last_loss_value)
188 
189  self._state_counters_state_counters[info_state] += 1
190  for action_ in legal_actions:
191  self._avg_policy_avg_policy[info_state][action_] = (
192  self._avg_policy_avg_policy[info_state][action_] +
193  1 / self._state_counters_state_counters[info_state] * (
194  self._cur_policy_cur_policy[info_state][action_] -
195  self._avg_policy_avg_policy[info_state][action_]))
196 
197  assert self._delta_l_delta_l.value > self._delta_w_delta_w.value
198  cur_policy_value = sum([
199  self._cur_policy_cur_policy[info_state][action] *
200  self._q_values_q_values[info_state][action] for action in legal_actions
201  ])
202  avg_policy_value = sum([
203  self._avg_policy_avg_policy[info_state][action] *
204  self._q_values_q_values[info_state][action] for action in legal_actions
205  ])
206  if cur_policy_value > avg_policy_value:
207  self._cur_delta_value_cur_delta_value = self._delta_w_delta_w.value
208  else:
209  self._cur_delta_value_cur_delta_value = self._delta_l_delta_l.value
210 
211  if not time_step.last():
212  self._hill_climbing_hill_climbing(info_state, legal_actions)
213 
214  # Decay epsilon, if necessary.
215  self._epsilon_epsilon = self._epsilon_schedule_epsilon_schedule.step()
216  self._delta_l_delta_l.step()
217  self._delta_w_delta_w.step()
218  self._step_size_step_size.step()
219  else: # prepare for the next episode.
220  self._prev_info_state_prev_info_state = None
221  return
222 
223  # Don't mess up with the state during evaluation.
224  if not is_evaluation:
225  self._prev_info_state_prev_info_state = info_state
226  self._prev_action_prev_action = action
227  return rl_agent.StepOutput(action=action, probs=probs)
