26// A simple game that includes chance and imperfect information
27// https://en.wikipedia.org/wiki/Liar%27s_dice
28//
29// Currently only supports a single round and two players.
30// The highest face (`dice_sides`) is wild.
31//
32// Parameters:
33// "bidding_rule" string bidding variants ("reset-face" or
34// ("reset-quantity") (def. "reset-face")
35// "dice_sides" int number of sides on each die (def. = 6)
36// "numdice" int number of dice per player (def. = 1)
37// "numdiceX" int overridden number of dice for player X (def. = 1)
38// "players" int number of players (def. = 2)
39
40namespace open_spiel {
41namespace liars_dice {

43enum BiddingRule {
44 // The player may bid a higher quantity of any particular face, or the same
45 // quantity of a higher face (allowing a player to "re-assert" a face value
46 // they believe prevalent if another player increased the face value on their
47 // bid).
48 kResetFace = 1,
49
50 // The player may bid a higher quantity of the same face, or any particular
51 // quantity of a higher face (allowing a player to "reset" the quantity).
52 kResetQuantity = 2
53};
