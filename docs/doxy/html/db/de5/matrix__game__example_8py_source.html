26def _manually_create_game():
27 """Creates the game manually from the spiel building blocks."""
28 game_type = pyspiel.GameType(
29 "matching_pennies",
30 "Matching Pennies",
31 pyspiel.GameType.Dynamics.SIMULTANEOUS,
32 pyspiel.GameType.ChanceMode.DETERMINISTIC,
33 pyspiel.GameType.Information.ONE_SHOT,
34 pyspiel.GameType.Utility.ZERO_SUM,
35 pyspiel.GameType.RewardModel.TERMINAL,
36 2, # max num players
37 2, # min_num_players
38 True, # provides_information_state
39 True, # provides_information_state_tensor
40 False, # provides_observation
41 False, # provides_observation_tensor
42 dict() # parameter_specification
43 )
44 game = pyspiel.MatrixGame(
45 game_type,
46 {}, # game_parameters
47 ["Heads", "Tails"], # row_action_names
48 ["Heads", "Tails"], # col_action_names
49 [[-1, 1], [1, -1]], # row player utilities
50 [[1, -1], [-1, 1]] # col player utilities
51 )
52 return game

67def _import_data_create_game():
68 """Creates a game via imported payoff data."""
69 payoff_file = file_utils.find_file(
70 "open_spiel/data/paper_data/response_graph_ucb/soccer.txt", 2)
71 payoffs = np.loadtxt(payoff_file)*2-1
72 return pyspiel.create_matrix_game(payoffs, payoffs.T)

75def main(_):
76 games_list = pyspiel.registered_games()
77 print("Registered games:")
78 print(games_list)
79
80 # Load a two-player normal-form game as a two-player matrix game.
81 blotto_matrix_game = pyspiel.load_matrix_game("blotto")
82 print("Number of rows in 2-player Blotto with default settings is {}".format(
83 blotto_matrix_game.num_rows()))
84
85 # Several ways to load/create the same game of matching pennies.
86 print("Creating matrix game...")
87 game = pyspiel.load_matrix_game("matrix_mp")
88 game = _manually_create_game()
89 game = _import_data_create_game()
90 game = _easy_create_game()
91 game = _even_easier_create_game()
92
93 # Quick test: inspect top-left utility values:
94 print("Values for joint action ({},{}) is {},{}".format(
95 game.row_action_name(0), game.col_action_name(0),
96 game.player_utility(0, 0, 0), game.player_utility(1, 0, 0)))
97
98 state = game.new_initial_state()
99
100 # Print the initial state
101 print("State:")
102 print(str(state))
103
104 assert state.is_simultaneous_node()
105
106 # Simultaneous node: sample actions for all players.
107 chosen_actions = [
108 random.choice(state.legal_actions(pid))
109 for pid in range(game.num_players())
110 ]
111 print("Chosen actions: ", [
112 state.action_to_string(pid, action)
113 for pid, action in enumerate(chosen_actions)
114 ])
115 state.apply_actions(chosen_actions)
116
117 assert state.is_terminal()
118
119 # Game is now done. Print utilities for each player
120 returns = state.returns()
121 for pid in range(game.num_players()):
122 print("Utility for player {} is {}".format(pid, returns[pid]))
