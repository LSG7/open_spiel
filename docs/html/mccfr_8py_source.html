33  def action_probabilities(self, state, player_id=None):
34  """Returns the MCCFR average policy for a player in a state.
35 
36  If the policy is not defined for the provided state, a uniform
37  random policy is returned.
38 
39  Args:
40  state: A `pyspiel.State` object.
41  player_id: Optional, the player id for which we want an action. Optional
42  unless this is a simultaneous state at which multiple players can act.
43 
44  Returns:
45  A `dict` of `{action: probability}` for the specified player in the
46  supplied state. If the policy is defined for the state, this
47  will contain the average MCCFR strategy defined for that state.
48  Otherwise, it will contain all legal actions, each with the same
49  probability, equal to 1 / num_legal_actions.
50  """
51  if player_id is None:
52  player_id = state.current_player()
53  legal_actions = state.legal_actions()
54  info_state_key = state.information_state_string(player_id)
55  retrieved_infostate = self._infostates_infostates.get(info_state_key, None)
56  if retrieved_infostate is None:
57  return {a: 1 / len(legal_actions) for a in legal_actions}
58  avstrat = (
59  retrieved_infostate[AVG_POLICY_INDEX] /
60  retrieved_infostate[AVG_POLICY_INDEX].sum())
61  return {legal_actions[i]: avstrat[i] for i in range(len(legal_actions))}

72  def _lookup_infostate_info(self, info_state_key, num_legal_actions):
73  """Looks up an information set table for the given key.
74 
75  Args:
76  info_state_key: information state key (string identifier).
77  num_legal_actions: number of legal actions at this information state.
78 
79  Returns:
80  A list of:
81  - the average regrets as a numpy array of shape [num_legal_actions]
82  - the average strategy as a numpy array of shape
83  [num_legal_actions].
84  The average is weighted using `my_reach`
85  """
86  retrieved_infostate = self._infostates_infostates.get(info_state_key, None)
87  if retrieved_infostate is not None:
88  return retrieved_infostate
89 
90  # Start with a small amount of regret and total accumulation, to give a
91  # uniform policy: this will get erased fast.
92  self._infostates_infostates[info_state_key] = [
93  np.ones(num_legal_actions, dtype=np.float64) / 1e6,
94  np.ones(num_legal_actions, dtype=np.float64) / 1e6,
95  ]
96  return self._infostates_infostates[info_state_key]

114  def _regret_matching(self, regrets, num_legal_actions):
115  """Applies regret matching to get a policy.
116 
117  Args:
118  regrets: numpy array of regrets for each action.
119  num_legal_actions: number of legal actions at this state.
120 
121  Returns:
122  numpy array of the policy indexed by the index of legal action in the
123  list.
124  """
125  positive_regrets = np.maximum(regrets,
126  np.zeros(num_legal_actions, dtype=np.float64))
127  sum_pos_regret = positive_regrets.sum()
128  if sum_pos_regret <= 0:
129  return np.ones(num_legal_actions, dtype=np.float64) / num_legal_actions
130  else:
131  return positive_regrets / sum_pos_regret
