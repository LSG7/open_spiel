138 std::unique_ptr<TabularPolicy> DeserializeTabularPolicy(
139  const std::string& serialized, std::string delimiter) {
140  // Class’s identity is the very first line, see Policy::Serialize
141  // for more info.
142  std::pair<std::string, absl::string_view> cls_and_content =
143  absl::StrSplit(serialized, absl::MaxSplits(':', 1));
144  SPIEL_CHECK_EQ(cls_and_content.first, "TabularPolicy");
145 
146  std::unique_ptr<TabularPolicy> res = std::make_unique<TabularPolicy>();
147  if (cls_and_content.second.empty()) return res;
148 
149  std::vector<absl::string_view> splits =
150  absl::StrSplit(cls_and_content.second, delimiter);
151 
152  // Insert the actual values
153  Action action;
154  double prob;
155  for (int i = 0; i < splits.size(); i += 2) {
156  std::vector<absl::string_view> policy_values =
157  absl::StrSplit(splits.at(i + 1), ',');
158  ActionsAndProbs res_policy;
159  res_policy.reserve(policy_values.size());
160 
161  for (absl::string_view policy_value : policy_values) {
162  std::pair<absl::string_view, absl::string_view> action_and_prob =
163  absl::StrSplit(policy_value, '=');
164  SPIEL_CHECK_TRUE(absl::SimpleAtoi(action_and_prob.first, &action));
165  absl::from_chars(
166  action_and_prob.second.data(),
167  action_and_prob.second.data() + action_and_prob.second.size(), prob);
168  res_policy.push_back({action, prob});
169  }
170  res->SetStatePolicy(std::string(splits.at(i)), res_policy);
171  }
172  return res;
173 }

252 TabularPolicy GetEmptyTabularPolicy(const Game& game,
253  bool initialize_to_uniform,
254  Player player) {
255  std::unordered_map<std::string, ActionsAndProbs> policy;
256  if (game.GetType().dynamics != GameType::Dynamics::kSequential) {
257  SpielFatalError("Game is not sequential.");
258  return TabularPolicy(policy);
259  }
260  std::list<std::unique_ptr<State>> to_visit;
261  to_visit.push_back(game.NewInitialState());
262  while (!to_visit.empty()) {
263  std::unique_ptr<State> state = std::move(to_visit.back());
264  to_visit.pop_back();
265  if (state->IsTerminal()) {
266  continue;
267  }
268  if (state->IsChanceNode()) {
269  for (const auto& outcome_and_prob : state->ChanceOutcomes()) {
270  to_visit.emplace_back(state->Child(outcome_and_prob.first));
271  }
272  } else {
273  ActionsAndProbs infostate_policy;
274  std::vector<Action> legal_actions = state->LegalActions();
275  const int num_legal_actions = legal_actions.size();
276  SPIEL_CHECK_GT(num_legal_actions, 0.);
277  for (Action action : legal_actions) {
278  to_visit.push_back(state->Child(action));
279  }
280  if (player < 0 || state->IsPlayerActing(player)) {
281  double action_probability = 1.;
282  if (initialize_to_uniform) {
283  action_probability = 1. / num_legal_actions;
284  }
285  ActionsAndProbs infostate_policy;
286  infostate_policy.reserve(num_legal_actions);
287  for (Action action : legal_actions) {
288  infostate_policy.push_back({action, action_probability});
289  }
290  if (infostate_policy.empty()) {
291  SpielFatalError("State has zero legal actions.");
292  }
293  policy.insert({state->InformationStateString(), infostate_policy});
294  }
295  }
296  }
297  return TabularPolicy(policy);
298 }

304 template <typename RandomNumberDistribution>
305 TabularPolicy SamplePolicy(
306  const Game& game, int seed, RandomNumberDistribution& dist, Player player) {
307  std::mt19937 gen(seed);
308  TabularPolicy policy = GetEmptyTabularPolicy(game, false, player);
309  std::unordered_map<std::string, ActionsAndProbs>& policy_table =
310  policy.PolicyTable();
311  for (auto& kv : policy_table) {
312  ActionsAndProbs state_policy;
313  if (kv.second.empty()) {
314  SpielFatalError("State has zero legal actions.");
315  }
316  state_policy.reserve(kv.second.size());
317  double sum = 0;
318  double prob;
319  for (const auto& action_and_prob : kv.second) {
320  // We multiply the original probability by a random number greater than
321  // 0. We then normalize. This has the effect of randomly permuting the
322  // policy but all illegal actions still have zero probability.
323  prob = dist(gen) * action_and_prob.second;
324  sum += prob;
325  state_policy.push_back({action_and_prob.first, prob});
326  }
327  // We normalize the policy to ensure it sums to 1.
328  for (auto& action_and_prob : state_policy) {
329  action_and_prob.second /= sum;
330  }
331  // This is included as a sanity check.
332  double normalized_sum = 0;
333  for (auto& action_and_prob : state_policy) {
334  normalized_sum += action_and_prob.second;
335  }
336  SPIEL_CHECK_FLOAT_EQ(normalized_sum, 1.0);
337  kv.second = state_policy;
338  }
339  return policy;
340 }

353 TabularPolicy GetRandomDeterministicPolicy(
354  const Game& game, int seed, Player player) {
355  std::mt19937 gen(seed);
356  absl::node_hash_map<int, std::uniform_int_distribution<int>> dists;
357  std::unordered_map<std::string, ActionsAndProbs> policy;
358  if (game.GetType().dynamics != GameType::Dynamics::kSequential) {
359  SpielFatalError("Game is not sequential.");
360  return TabularPolicy(policy);
361  }
362  const GameType::Information information = game.GetType().information;
363  std::list<std::unique_ptr<State>> to_visit;
364  to_visit.push_back(game.NewInitialState());
365  while (!to_visit.empty()) {
366  std::unique_ptr<State> state = std::move(to_visit.back());
367  to_visit.pop_back();
368  if (state->IsTerminal()) {
369  continue;
370  } else if (state->IsChanceNode()) {
371  for (const auto& outcome_and_prob : state->ChanceOutcomes()) {
372  to_visit.emplace_back(state->Child(outcome_and_prob.first));
373  }
374  } else if (player < 0 || state->IsPlayerActing(player)) {
375  std::vector<Action> legal_actions = state->LegalActions();
376  const int num_legal_actions = legal_actions.size();
377  SPIEL_CHECK_GT(num_legal_actions, 0.);
378  if (dists.count(num_legal_actions) == 0) {
379  std::uniform_int_distribution<int> dist(0, num_legal_actions - 1);
380  dists.insert({num_legal_actions, std::move(dist)});
381  }
382  const int legal_action_index = dists[num_legal_actions](gen);
383  SPIEL_CHECK_GE(legal_action_index, 0);
384  SPIEL_CHECK_LT(legal_action_index, num_legal_actions);
385  const int action = legal_actions[legal_action_index];
386  ActionsAndProbs infostate_policy;
387  infostate_policy.reserve(1);
388  infostate_policy.push_back({action, 1.0});
389  policy.insert({state->InformationStateString(), infostate_policy});
390  if (information == GameType::Information::kPerfectInformation) {
391  to_visit.push_back(state->Child(action));
392  } else {
393  for (Action action : legal_actions) {
394  to_visit.push_back(state->Child(action));
395  }
396  }
397  } else {
398  std::vector<Action> legal_actions = state->LegalActions();
399  const int num_legal_actions = legal_actions.size();
400  SPIEL_CHECK_GT(num_legal_actions, 0.);
401  for (Action action : legal_actions) {
402  to_visit.push_back(state->Child(action));
403  }
404  }
405  }
406  return TabularPolicy(policy);
407 }

465 TabularPolicy ToTabularPolicy(const Game& game, const Policy* policy) {
466  TabularPolicy tabular_policy;
467  std::vector<std::unique_ptr<State>> to_visit;
468  to_visit.push_back(game.NewInitialState());
469  for (int idx = 0; idx < to_visit.size(); ++idx) {
470  const State* state = to_visit[idx].get();
471  if (state->IsTerminal()) {
472  continue;
473  }
474 
475  if (!state->IsChanceNode()) {
476  std::vector<Player> players(game.NumPlayers());
477  if (state->IsSimultaneousNode()) {
478  absl::c_iota(players, 0);
479  } else {
480  players = {state->CurrentPlayer()};
481  }
482 
483  for (Player player : players) {
484  ActionsAndProbs state_policy = policy->GetStatePolicy(*state);
485  tabular_policy.SetStatePolicy(state->InformationStateString(player),
486  state_policy);
487  }
488  }
489 
490  for (Action action : state->LegalActions()) {
491  to_visit.push_back(state->Child(action));
492  }
493  }
494  return tabular_policy;
495 }
