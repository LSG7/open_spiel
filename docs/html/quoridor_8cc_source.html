67 class QuoridorState::SearchState {
68  using DistanceAndMove = std::pair<int, Move>;
69 
70  class SearchQueue
71  : public std::priority_queue<DistanceAndMove,
72  std::vector<DistanceAndMove>,
73  std::greater<DistanceAndMove>> {
74  public:
75  void clear() { c.clear(); }
76  void reserve(int capacity) { c.reserve(capacity); }
77  };
78 
79  public:
80  explicit SearchState(int board_diameter) {
81  int size = board_diameter * board_diameter;
82  mark_.resize(size, false);
83  on_shortest_path_.resize(size, false);
84  distance_.resize(size, UndefinedDistance());
85  queue_.reserve(size);
86  }
87 
88  bool IsEmpty() const { return queue_.empty(); }
89 
90  void ClearSearchQueue() { queue_.clear(); }
91 
92  bool Push(int dist, Move move) {
93  if (mark_[move.xy] == false) {
94  mark_[move.xy] = true;
95  queue_.emplace(dist, move);
96  return true;
97  } else {
98  return false;
99  }
100  }
101 
102  Move Pop() {
103  Move move = queue_.top().second;
104  queue_.pop();
105  return move;
106  }
107 
108  void ResetSearchQueue() {
109  std::fill(mark_.begin(), mark_.end(), false);
110  queue_.clear();
111  }
112 
113  void ResetDists() {
114  std::fill(distance_.begin(), distance_.end(), UndefinedDistance());
115  }
116 
117  void SetDist(Move move, int dist) { distance_[move.xy] = dist; }
118  int GetDist(Move move) const { return distance_[move.xy]; }
119  void SetOnShortestPath(Move move) { on_shortest_path_[move.xy] = true; }
120  bool IsOnShortestPath(Move move) const { return on_shortest_path_[move.xy]; }
121 
122  static constexpr int UndefinedDistance() { return -1; }
123 
124  private:
125  SearchQueue queue_;
126  std::vector<bool> mark_; // Whether this position has been pushed before.
127  std::vector<int> distance_; // Distance from player.
128  std::vector<bool> on_shortest_path_; // Is this position on a shortest path?
129 };

144 QuoridorState::QuoridorState(std::shared_ptr<const Game> game, int board_size,
145  int wall_count, bool ansi_color_output)
146  : State(game),
147  board_size_(board_size),
148  board_diameter_(board_size * 2 - 1),
149  ansi_color_output_(ansi_color_output) {
150  board_.resize(board_diameter_ * board_diameter_, kPlayerNone);
151  players_.resize(num_players_);
152  // Account for order of turns (order of play is clockwise)
153  if (num_players_ == 2) {
154  players_[0] = kPlayer1;
155  players_[1] = kPlayer2;
156  } else if (num_players_ == 3) {
157  players_[0] = kPlayer1;
158  players_[1] = kPlayer3;
159  players_[2] = kPlayer2;
160  } else if (num_players_ == 4) {
161  players_[0] = kPlayer1;
162  players_[1] = kPlayer3;
163  players_[2] = kPlayer2;
164  players_[3] = kPlayer4;
165  }
166  wall_count_.resize(num_players_);
167  player_loc_.resize(num_players_);
168  end_zone_.resize(num_players_);
169  for (int i = 0; i < num_players_; ++i) {
170  wall_count_[players_[i]] = wall_count;
171  InitializePlayer(players_[i]);
172  }
173 }

175 void QuoridorState::InitializePlayer(QuoridorPlayer p) {
176  int center_field = board_size_ - (board_size_ % 2);
177  if (p == kPlayer1) {
178  player_loc_[p] = GetMove(center_field, board_diameter_ - 1);
179  SetPlayer(player_loc_[p], p, kPlayerNone);
180  end_zone_[p] = 0;
181  return;
182  }
183  if (p == kPlayer2) {
184  player_loc_[p] = GetMove(center_field, 0);
185  SetPlayer(player_loc_[p], kPlayer2, kPlayerNone);
186  end_zone_[p] = board_diameter_ - 1;
187  return;
188  }
189  if (p == kPlayer3) {
190  player_loc_[p] = GetMove(0, center_field);
191  SetPlayer(player_loc_[p], p, kPlayerNone);
192  end_zone_[p] = board_diameter_ - 1;
193  return;
194  }
195  if (p == kPlayer4) {
196  player_loc_[p] = GetMove(board_diameter_ - 1, center_field);
197  SetPlayer(player_loc_[p], p, kPlayerNone);
198  end_zone_[p] = 0;
199  return;
200  }
201 }

207 std::vector<Action> QuoridorState::LegalActions() const {
208  std::vector<Action> moves;
209  if (IsTerminal()) return moves;
210  int max_moves = 5; // Max pawn moves, including jumps.
211  if (wall_count_[current_player_] > 0) {
212  max_moves += 2 * (board_size_ - 1) * (board_size_ - 1); // Max wall moves.
213  }
214  moves.reserve(max_moves);
215 
216  // Pawn moves.
217  Move cur = player_loc_[current_player_];
218  AddActions(cur, Offset(1, 0), &moves);
219  AddActions(cur, Offset(0, 1), &moves);
220  AddActions(cur, Offset(-1, 0), &moves);
221  AddActions(cur, Offset(0, -1), &moves);
222 
223  // Wall placements.
224  if (wall_count_[current_player_] > 0) {
225  SearchState search_state(board_diameter_);
226  for (int i = 0; i < num_players_; ++i) {
227  SearchShortestPath(players_[i], &search_state);
228  }
229  for (int y = 0; y < board_diameter_ - 2; y += 2) {
230  for (int x = 0; x < board_diameter_ - 2; x += 2) {
231  Move h = GetMove(x, y + 1);
232  if (IsValidWall(h, &search_state)) {
233  moves.push_back(h.xy);
234  }
235  Move v = GetMove(x + 1, y);
236  if (IsValidWall(v, &search_state)) {
237  moves.push_back(v.xy);
238  }
239  }
240  }
241  }
242 
243  // If no action is possible add 'pass' action to list of moves
244  if (moves.empty()) {
245  moves.push_back(cur.xy);
246  }
247 
248  std::sort(moves.begin(), moves.end());
249  return moves;
250 }

252 void QuoridorState::AddActions(Move cur, Offset offset,
253  std::vector<Action>* moves) const {
254  SPIEL_CHECK_FALSE(cur.IsWall());
255 
256  if (IsWall(cur + offset)) {
257  // Hit a wall or edge in this direction.
258  return;
259  }
260 
261  Move forward = cur + offset * 2;
262  if (GetPlayer(forward) == kPlayerNone) {
263  // Normal single step in this direction.
264  moves->push_back(forward.xy);
265  return;
266  }
267 
268  // Other player, so which jumps are valid?
269 
270  if (!IsWall(cur + offset * 3)) {
271  // In two-players: A normal jump is allowed. We know that spot is empty.
272  // In >2 players, must check.
273  if (GetPlayer(cur + offset * 4) == kPlayerNone) {
274  moves->push_back((cur + offset * 4).xy);
275  return;
276  } else {
277  return;
278  }
279  }
280  // We are jumping over the other player against a wall, which side jumps are
281  // valid?
282 
283  Offset left = offset.rotate_left();
284  if (!IsWall(forward + left)) {
285  if (GetPlayer(forward + left * 2) == kPlayerNone) {
286  moves->push_back((forward + left * 2).xy);
287  }
288  }
289  Offset right = offset.rotate_right();
290  if (!IsWall(forward + right)) {
291  if (GetPlayer(forward + right * 2) == kPlayerNone) {
292  moves->push_back((forward + right * 2).xy);
293  }
294  }
295 }

297 bool QuoridorState::IsValidWall(Move m, SearchState* search_state) const {
298  Offset offset = (m.IsHorizontalWall() ? Offset(1, 0) : Offset(0, 1));
299 
300  if (IsWall(m + offset * 0) || IsWall(m + offset * 1) ||
301  IsWall(m + offset * 2)) {
302  // Already blocked by a wall.
303  return false;
304  }
305 
306  // Any wall that doesn't intersect with a shortest path is clearly legal.
307  // Walls that do intersect might still be legal because there's another way
308  // around, but that's more expensive to check.
309  if (!search_state->IsOnShortestPath(m) &&
310  !search_state->IsOnShortestPath(m + offset * 2)) {
311  return true;
312  }
313 
314  // If this wall doesn't connect two existing walls/edges, then it can't cut
315  // any paths. Even connecting to a node where 3 other walls meet, but without
316  // connecting them to anything else, can't cut any paths.
317  int count = (
318  // The 3 walls near the close end.
319  (IsWall(m - offset * 2) || IsWall(m - offset + offset.rotate_left()) ||
320  IsWall(m - offset + offset.rotate_right())) +
321  // The 3 walls near the far end.
322  (IsWall(m + offset * 4) ||
323  IsWall(m + offset * 3 + offset.rotate_left()) ||
324  IsWall(m + offset * 3 + offset.rotate_right())) +
325  // The 2 walls in the middle.
326  (IsWall(m + offset + offset.rotate_left()) ||
327  IsWall(m + offset + offset.rotate_right())));
328  if (count <= 1) return true;
329 
330  // Do a full search to verify both players can get to their respective goals.
331  bool pathExists = true;
332  for (int i = 0; i < num_players_; ++i) {
333  pathExists = pathExists &&
334  SearchEndZone(players_[i], m, m + offset * 2, search_state);
335  }
336  return pathExists;
337 }

339 bool QuoridorState::SearchEndZone(QuoridorPlayer p, Move wall1, Move wall2,
340  SearchState* search_state) const {
341  search_state->ResetSearchQueue();
342  Offset dir(1, 0); // Direction is arbitrary. Queue will make it fast.
343  int goal = end_zone_[p];
344  int goal_dir = (goal == 0 ? -1 : 1); // Sort for shortest dist in a min-heap.
345  search_state->Push(0, player_loc_[p]);
346  while (!search_state->IsEmpty()) {
347  Move c = search_state->Pop();
348  for (int i = 0; i < 4; ++i) {
349  Move wall = c + dir;
350  if (!IsWall(wall) && wall != wall1 && wall != wall2) {
351  Move move = c + dir * 2;
352  int moveCoord;
353  if (p == kPlayer1 || p == kPlayer2) {
354  moveCoord = move.y;
355  } else if (p == kPlayer3 || p == kPlayer4) {
356  moveCoord = move.x;
357  } else {
358  SpielFatalError("Case not handled for player in SearchEndZone.");
359  }
360  if (moveCoord == goal) {
361  return true;
362  }
363  search_state->Push(goal_dir * (goal - move.y), move);
364  }
365  dir = dir.rotate_left();
366  }
367  }
368 
369  return false;
370 }

372 void QuoridorState::SearchShortestPath(QuoridorPlayer p,
373  SearchState* search_state) const {
374  search_state->ResetSearchQueue();
375  search_state->ResetDists();
376  Offset dir(1, 0); // Direction is arbitrary. Queue will make it fast.
377  int goal = end_zone_[p];
378  int goal_dir = (goal == 0 ? -1 : 1); // Sort for shortest dist in a min-heap.
379  search_state->Push(0, player_loc_[p]);
380  search_state->SetDist(player_loc_[p], 0);
381  Move goal_found = GetMove(-1, -1); // invalid
382 
383  // A* search for the end-zone, keeping distances to each cell.
384  while (!search_state->IsEmpty()) {
385  Move c = search_state->Pop();
386  int dist = search_state->GetDist(c);
387  for (int i = 0; i < 4; ++i) {
388  Move wall = c + dir;
389  if (!IsWall(wall)) {
390  Move move = c + dir * 2;
391  int moveCoord;
392  if (p == kPlayer1 || p == kPlayer2) {
393  moveCoord = move.y;
394  } else if (p == kPlayer3 || p == kPlayer4) {
395  moveCoord = move.x;
396  } else {
397  SpielFatalError("Case not handled for player in SearchShortestPath");
398  }
399  if (moveCoord == goal) {
400  search_state->SetDist(move, dist + 1);
401  search_state->ClearSearchQueue(); // Break out of the search.
402  goal_found = move;
403  break;
404  }
405  if (search_state->Push(dist + 1 + goal_dir * (goal - moveCoord),
406  move)) {
407  search_state->SetDist(move, dist + 1);
408  }
409  }
410  dir = dir.rotate_left();
411  }
412  }
413 
414  // Trace the way back, setting them to be on a shortest path.
415  Move current = goal_found;
416  int dist = search_state->GetDist(current);
417  while (current != player_loc_[p]) {
418  for (int i = 0; i < 4; ++i) {
419  Move wall = current + dir;
420  if (!IsWall(wall)) {
421  Move move = current + dir * 2;
422  int dist2 = search_state->GetDist(move);
423  if (dist2 != search_state->UndefinedDistance() && dist2 + 1 == dist) {
424  search_state->SetOnShortestPath(wall);
425  current = move;
426  dist = dist2;
427  break;
428  }
429  }
430  dir = dir.rotate_left();
431  }
432  }
433 }

440 std::string QuoridorState::ToString() const {
441  // Generates something like:
442  // Board size: 5, walls: 0, 0
443  // a b c d e
444  // 1 . | . . . . 1
445  // + ---+---
446  // 2 . | . | . . . 2
447  // +
448  // 3 . . | O @ . 3
449  // ---+---
450  // 4 . | . . . . 4
451  // + ---+---
452  // 5 . | . . . . 5
453  // a b c d e
454 
455  std::string reset;
456  std::array<std::string, 4> colors, coords;
457  if (ansi_color_output_) {
458  std::string esc = "\033";
459  reset = esc + "[0m";
460  coords[0] = esc + "[1;33m";
461  coords[1] = esc + "[1;34m";
462  coords[2] = esc + "[1;35m";
463  coords[3] = esc + "[1;36m";
464  colors[0] = esc + "[1;33m" + " O " + reset;
465  colors[1] = esc + "[1;34m" + " @ " + reset;
466  colors[2] = esc + "[1;35m" + " # " + reset;
467  colors[3] = esc + "[1;36m" + " % " + reset;
468  } else {
469  std::string reset = "";
470  coords[0] = "";
471  coords[1] = "";
472  coords[2] = "";
473  coords[3] = "";
474  colors[0] = " 0 ";
475  colors[1] = " @ ";
476  colors[2] = " # ";
477  colors[3] = " % ";
478  }
479 
480  std::ostringstream out;
481  out << "Board size: " << board_size_ << ", walls: ";
482  for (int i = 0; i < num_players_; ++i) {
483  out << wall_count_[players_[i]];
484  if (i < num_players_ - 1) out << ", ";
485  }
486  out << "\n";
487 
488  // Top x coords.
489  for (int x = 0; x < board_size_; ++x) {
490  out << " " << coords[1] << static_cast<char>('a' + x);
491  }
492  out << reset << '\n';
493 
494  for (int y = 0; y < board_diameter_; ++y) {
495  if (y % 2 == 0) {
496  if (y / 2 + 1 < 10) out << " ";
497  out << coords[2] << (y / 2 + 1) << reset; // Leading y coord.
498  } else {
499  out << " "; // Wall lines.
500  }
501 
502  for (int x = 0; x < board_diameter_; ++x) {
503  QuoridorPlayer p = GetPlayer(GetMove(x, y));
504  if (x % 2 == 0 && y % 2 == 0) {
505  bool playerFound = false;
506  for (int i = 0; i < num_players_; ++i) {
507  if (p == players_[i]) {
508  out << colors[players_[i]];
509  playerFound = true;
510  }
511  }
512  if (!playerFound) {
513  out << " . ";
514  }
515  } else if (x % 2 == 1 && y % 2 == 1) {
516  out << (p == kPlayerWall ? "+" : " ");
517  } else if (x % 2 == 1) {
518  out << (p == kPlayerWall ? "|" : " ");
519  } else if (y % 2 == 1) {
520  out << (p == kPlayerWall ? "---" : " ");
521  }
522  }
523  if (y % 2 == 0) {
524  if (y / 2 + 1 < 10) out << " ";
525  out << coords[3] << (y / 2 + 1) << reset; // y coord on the right.
526  } else {
527  out << " "; // Wall lines.
528  }
529  out << '\n';
530  }
531  // Bottom x coords.
532  for (int x = 0; x < board_size_; ++x) {
533  out << " " << coords[0] << static_cast<char>('a' + x);
534  }
535  out << reset << '\n';
536  return out.str();
537 }

539 std::vector<double> QuoridorState::Returns() const {
540  std::vector<double> res(num_players_, 0.0);
541  for (int i = 0; i < num_players_; ++i) {
542  if (outcome_ == players_[i]) {
543  // If someone as won, set their reward to +1 and all the others to
544  // -1 / (num_players - 1).
545  std::fill(res.begin(), res.end(), -1.0 / (num_players_ - 1));
546  res[i] = 1.0;
547  break;
548  }
549  }
550  return res;
551 }

584 void QuoridorState::DoApplyAction(Action action) {
585  // If players is forced to pass it is valid to stay in place, on a field where
586  // there is already a player
587  if (board_[action] != current_player_) {
588  SPIEL_CHECK_EQ(board_[action], kPlayerNone);
589  }
590  SPIEL_CHECK_EQ(outcome_, kPlayerNone);
591 
592  Move move = ActionToMove(action);
593  SPIEL_CHECK_TRUE(move.IsValid());
594 
595  if (move.IsWall()) {
596  Offset offset = (move.IsHorizontalWall() ? Offset(1, 0) : Offset(0, 1));
597  SetPlayer(move + offset * 0, kPlayerWall, kPlayerNone);
598  SetPlayer(move + offset * 1, kPlayerWall, kPlayerNone);
599  SetPlayer(move + offset * 2, kPlayerWall, kPlayerNone);
600  wall_count_[current_player_] -= 1;
601  } else {
602  SetPlayer(player_loc_[current_player_], kPlayerNone, current_player_);
603  SetPlayer(move, current_player_, kPlayerNone);
604  player_loc_[current_player_] = move;
605 
606  int end_zone_coord;
607  if (current_player_ == kPlayer1 || current_player_ == kPlayer2) {
608  end_zone_coord = move.y;
609  } else {
610  end_zone_coord = move.x;
611  }
612 
613  outcome_ = kPlayerNone;
614  if (end_zone_coord == end_zone_[current_player_]) {
615  outcome_ = current_player_;
616  }
617  }
618 
619  ++moves_made_;
620  if (moves_made_ >= kMaxGameLengthFactor * board_size_ * board_size_) {
621  outcome_ = kPlayerDraw;
622  }
623 
624  current_player_index_ += 1;
625  if (current_player_index_ == num_players_) current_player_index_ = 0;
626  current_player_ = players_[current_player_index_];
627 }
