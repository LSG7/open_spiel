81 def ai_action(state, net, params):
82  observation = np.array(state.observation_tensor(), np.float32)
83  policy = np.exp(net.apply(params, observation))
84  probs_actions = [(p, a + MIN_ACTION) for a, p in enumerate(policy)]
85  pred = max(probs_actions)[1]
86  return pred

89 def _run_once(state, bots, net, params):
90  """Plays bots with each other, returns terminal utility for each player."""
91  for bot in bots:
92  bot.restart()
93  while not state.is_terminal():
94  if state.is_chance_node():
95  outcomes, probs = zip(*state.chance_outcomes())
96  state.apply_action(np.random.choice(outcomes, p=probs))
97  else:
98  if FLAGS.sleep:
99  time.sleep(FLAGS.sleep) # wait for the human to see how it goes
100  if state.current_player() % 2 == 1:
101  # Have simplest play for now
102  action = state.legal_actions()[0]
103  if action > 51:
104  # TODO(ed2k) extend beyond just bidding
105  action = ai_action(state, net, params)
106  state.apply_action(action)
107  else:
108  result = bots[state.current_player() // 2].step(state)
109  state.apply_action(result)
110  return state
