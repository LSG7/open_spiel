28SpanTensor ContiguousAllocator::Get(absl::string_view name,
29 const absl::InlinedVector<int, 4>& shape) {
30 const int size = absl::c_accumulate(shape, 1, std::multiplies<int>());
31 SPIEL_DCHECK_LE(offset_, data_.size());
32 auto buffer = data_.subspan(offset_, size);
33 offset_ += size;
34 return SpanTensor(SpanTensorInfo(name, shape), buffer);
35}

137std::shared_ptr<Observer> Game::MakeBuiltInObserver(
138 absl::optional<IIGObservationType> iig_obs_type) const {
139 if (!iig_obs_type) {
140 if (game_type_.provides_observation()) {
141 return absl::make_unique<DefaultObserver>(*this);
142 } else {
143 return nullptr;
144 }
145 }
146
147 const bool perfect_info_game =
148 game_type_.information == GameType::Information::kPerfectInformation;
149
150 // Perfect information games can provide public information regardless
151 // of requested PrivateInfoType (as they have no private information).
152 if (perfect_info_game) {
153 // Handle the dummy case, where we do not use any public information.
154 // The game will just have empty private observations.
155 if (!iig_obs_type->public_info)
156 return absl::make_unique<NoPrivateObserver>(*this);
157 if (game_type_.provides_information_state() && iig_obs_type->perfect_recall)
158 return absl::make_unique<InformationStateObserver>(*this);
159 if (game_type_.provides_observation() && !iig_obs_type->perfect_recall)
160 return absl::make_unique<DefaultObserver>(*this);
161 }
162
163 // TODO(author11) Reinstate this check
164 // SPIEL_CHECK_EQ(GetType().information,
165 // GameType::Information::kImperfectInformation);
166 if (iig_obs_type.value() == kDefaultObsType) {
167 if (game_type_.provides_observation())
168 return absl::make_unique<DefaultObserver>(*this);
169 }
170 if (iig_obs_type.value() == kInfoStateObsType) {
171 if (game_type_.provides_information_state())
172 return absl::make_unique<InformationStateObserver>(*this);
173 }
174 return nullptr;
175}

177std::shared_ptr<Observer> Game::MakeObserver(
178 absl::optional<IIGObservationType> iig_obs_type,
179 const ObservationParams& params) const {
180 // This implementation falls back to the original information state and
181 // observation methods in case of empty parameters and otherwise creates
182 // a registered observer based on its name.
183 // New games can register observers which can be selected by name, or override
184 // MakeObserver to return a game-specific observer.
185 if (params.empty()) {
186 return MakeBuiltInObserver(iig_obs_type);
187 } else {
188 return MakeRegisteredObserver(iig_obs_type, params);
189 }
190}

192SpanTensor TrackingVectorAllocator::Get(
193 absl::string_view name, const absl::InlinedVector<int, 4>& shape) {
194 SPIEL_DCHECK_TRUE(IsNameUnique(name));
195
196 SpanTensorInfo info(name, shape);
197 tensors_info_.push_back(info);
198
199 const int offset = data_.size();
200 const int size = info.size();
201 data_.resize(offset + size);
202 return SpanTensor(std::move(info),
203 absl::MakeSpan(data_).subspan(offset, size));
204}

206bool TrackingVectorAllocator::IsNameUnique(absl::string_view name) const {
207 for (const SpanTensorInfo& info : tensors_info_) {
208 if (info.name() == name) {
209 return false;
210 }
211 }
212 return true;
213}

232std::vector<SpanTensor> Observation::tensors() {
233 std::vector<SpanTensor> result;
234 result.reserve(tensors_info_.size());
235 int offset = 0;
236 for (const SpanTensorInfo& info : tensors_info_) {
237 const int size = info.size();
238 result.emplace_back(info, absl::MakeSpan(buffer_).subspan(offset, size));
239 offset += size;
240 }
241 return result;
242}

244// We may in the future support multiple compression schemes.
245// The Compress() method should select the most effective scheme adaptively.
246enum CompressionScheme : char {
247 kCompressionNone, // We weren't able to compress the data.
248 kCompressionBinary // Data is binary (all elements zero or one).
249};
250constexpr int kNumHeaderBytes = 1;
251
252// Binary compression.
253struct BinaryCompress {
254 static constexpr int kBitsPerByte = 8;
255
256 static std::string Compress(absl::Span<const float> buffer) {
257 const int num_bytes = (buffer.size() + kBitsPerByte - 1) / kBitsPerByte;
258 std::string str(num_bytes + kNumHeaderBytes, '\0');
259 str[0] = kCompressionBinary;
260
261 for (int i = 0; i < buffer.size(); ++i) {
262 if (buffer[i]) {
263 const int byte = i / kBitsPerByte;
264 const int bit = i % kBitsPerByte;
265 str[kNumHeaderBytes + byte] += (1 << bit);
266 }
267 }
268 return str;
269 }
270
271 static void Decompress(absl::string_view compressed,
272 absl::Span<float> buffer) {
273 const int num_bytes = (buffer.size() + kBitsPerByte - 1) / kBitsPerByte;
274 absl::c_fill(buffer, 0);
275 SPIEL_CHECK_EQ(compressed.size(), num_bytes + kNumHeaderBytes);
276 for (int byte = 0; byte < num_bytes; ++byte) {
277 for (int bit = 0; bit < kBitsPerByte; ++bit) {
278 if (compressed[kNumHeaderBytes + byte] & (1 << bit)) {
279 buffer[byte * kBitsPerByte + bit] = 1;
280 }
281 }
282 }
283 }
284};
285
286// No compression.
287struct NoCompress {
288 static std::string Compress(absl::Span<const float> buffer) {
289 const int num_bytes = sizeof(float) * buffer.size();
290 std::string str(num_bytes + 1, '\0');
291 str[0] = kCompressionNone;
292 memcpy(&str[kNumHeaderBytes], &buffer[0], num_bytes);
293 return str;
294 }
295
296 static void Decompress(absl::string_view compressed,
297 absl::Span<float> buffer) {
298 const int num_bytes = sizeof(float) * buffer.size();
299 SPIEL_CHECK_EQ(compressed.size(), num_bytes + kNumHeaderBytes);
300 memcpy(&buffer[0], &compressed[kNumHeaderBytes], num_bytes);
301 }
302};
303
304std::string Observation::Compress() const {
305 const bool data_is_binary =
306 absl::c_all_of(buffer_, [](float x) { return x == 0 || x == 1; });
307 return data_is_binary ? BinaryCompress::Compress(buffer_)
308 : NoCompress::Compress(buffer_);
309}
310
311void Observation::Decompress(absl::string_view compressed) {
312 SPIEL_CHECK_GT(compressed.size(), 0);
313 switch (compressed[0]) {
314 case kCompressionBinary:
315 return BinaryCompress::Decompress(compressed, absl::MakeSpan(buffer_));
316 case kCompressionNone:
317 return NoCompress::Decompress(compressed, absl::MakeSpan(buffer_));
318 default:
319 SpielFatalError(absl::StrCat("Unrecognized compression scheme in '",
320 compressed, "'"));
321 }
322}
