21// Do the tree traversal to fill the map. This function does a depth-first
22// search of all the subtrees to fill the map for all the information states.
23void FillMap(const State& state,
24 std::unordered_map<std::string, std::vector<Action>>* map,
25 int depth_limit, int depth, Player player) {
26 if (state.IsTerminal()) {
27 return;
28 }
29
30 if (depth_limit >= 0 && depth > depth_limit) {
31 return;
32 }
33
34 if (state.IsChanceNode()) {
35 // Do nothing at chance nodes (no information states).
36 } else if (state.IsSimultaneousNode()) {
37 // Many players can play at this node.
38 for (auto p = Player{0}; p < state.NumPlayers(); ++p) {
39 if (player == kInvalidPlayer || p == player) {
40 std::string info_state = state.InformationStateString(p);
41 if (map->find(info_state) == map->end()) {
42 // Only add it if we don't already have it.
43 std::vector<Action> legal_actions = state.LegalActions(p);
44 (*map)[info_state] = legal_actions;
45 }
46 }
47 }
48 } else {
49 // Regular decision node.
50 if (player == kInvalidPlayer || state.CurrentPlayer() == player) {
51 std::string info_state = state.InformationStateString();
52 if (map->find(info_state) == map->end()) {
53 // Only add it if we don't already have it.
54 std::vector<Action> legal_actions = state.LegalActions();
55 (*map)[info_state] = legal_actions;
56 }
57 }
58 }
59
60 // Recursively fill the map for each subtree below.
61 for (auto action : state.LegalActions()) {
62 std::unique_ptr<State> next_state = state.Child(action);
63 FillMap(*next_state, map, depth_limit, depth + 1, player);
64 }
65}

69std::unordered_map<std::string, std::vector<Action>> GetLegalActionsMap(
70 const Game& game, int depth_limit, Player player) {
71 std::unordered_map<std::string, std::vector<Action>> map;
72 std::unique_ptr<State> initial_state = game.NewInitialState();
73 FillMap(*initial_state, &map, depth_limit, 0, player);
74 return map;
75}
