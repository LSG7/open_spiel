21 bool IsOverlappingPlacement(const std::vector<ShipPlacement>& placement) {
22  for (int index = 1; index < placement.size(); ++index) {
23  for (int other = 0; other < index; ++other) {
24  if (placement.at(index).OverlapsWith(placement.at(other))) {
25  return true;
26  }
27  }
28  }
29 
30  return false;
31 }

34 bool ExistsFeasiblePlacement(const BattleshipConfiguration& conf,
35  std::vector<ShipPlacement>* partial_placement) {
36  // Debug-time check of preconditions.
37  SPIEL_DCHECK_LE(partial_placement->size(), conf.ships.size());
38  SPIEL_DCHECK_FALSE(IsOverlappingPlacement(*partial_placement));
39  for (int index = 0; index < partial_placement->size(); ++index) {
40  const ShipPlacement& placement = partial_placement->at(index);
41 
42  SPIEL_CHECK_EQ(placement.ship.id, conf.ships.at(index).id);
43  SPIEL_CHECK_TRUE(
44  placement.IsWithinBounds(conf.board_width, conf.board_height));
45  }
46 
47  if (partial_placement->size() == conf.ships.size()) {
48  // All ships have been placed. The placement is valid because of the
49  // precondition.
50  return true;
51  } else {
52  // We try to place the next ship in the board. We start by trying to place
53  // the ship horizontally.
54  //
55  // Because of the precondition, partial_placement is a placement of a prefix
56  // of the ships in conf.ships. Hence, the next ship that needs to be placed
57  // is simply:
58  const Ship& ship = conf.ships.at(partial_placement->size());
59 
60  // -- Horizontal placement.
61  for (int row = 0; row < conf.board_height; ++row) {
62  for (int col = 0; col < conf.board_width - ship.length + 1; ++col) {
63  // First, we append the placement of the next ship to the partial
64  // placement vector.
65  partial_placement->push_back({ShipPlacement::Direction::Horizontal,
66  /* ship = */ ship,
67  /* tl_corner = */ Cell{row, col}});
68  if (!IsOverlappingPlacement(*partial_placement) &&
69  ExistsFeasiblePlacement(conf, partial_placement)) {
70  // The new partial placement led to a solution. We honor the
71  // postcondition and early-return sucess.
72  partial_placement->pop_back();
73  return true;
74  } else {
75  // The new partial placement does not lead to a solution. We remove
76  // the placement and continue with the next placement.
77  partial_placement->pop_back();
78  }
79  }
80  }
81 
82  // -- Vertical placement.
83  for (int row = 0; row < conf.board_height - ship.length + 1; ++row) {
84  for (int col = 0; col < conf.board_width; ++col) {
85  // First, we append the placement of the next ship to the partial
86  // placement vector.
87  partial_placement->push_back({ShipPlacement::Direction::Vertical,
88  /* ship = */ ship,
89  /* tl_corner = */ Cell{row, col}});
90  if (!IsOverlappingPlacement(*partial_placement) &&
91  ExistsFeasiblePlacement(conf, partial_placement)) {
92  // The new partial placement led to a solution. We honor the
93  // postcondition and early-return sucess.
94  partial_placement->pop_back();
95  return true;
96  } else {
97  // The new partial placement does not lead to a solution. We remove
98  // the placement and continue with the next placement.
99  partial_placement->pop_back();
100  }
101  }
102  }
103  }
104 
105  return false;
106 }

121 bool ShipPlacement::CoversCell(const Cell& cell) const {
122  if (direction == Direction::Horizontal) {
123  return cell.row == tl_corner_.row && cell.col >= tl_corner_.col &&
124  cell.col < tl_corner_.col + ship.length;
125  } else {
126  SPIEL_CHECK_EQ(direction, Direction::Vertical);
127  return cell.col == tl_corner_.col && cell.row >= tl_corner_.row &&
128  cell.row < tl_corner_.row + ship.length;
129  }
130 }

132 Cell ShipPlacement::BottomRightCorner() const {

141 bool ShipPlacement::OverlapsWith(const ShipPlacement& other) const {
142  if (other.BottomRightCorner().row < TopLeftCorner().row) {
143  // `other` is completely above `this`.
144  return false;
145  } else if (other.TopLeftCorner().row > BottomRightCorner().row) {
146  // `other` is completely below `this`.
147  return false;
148  } else if (other.BottomRightCorner().col < TopLeftCorner().col) {
149  // `other` is completely to the left of `this`.
150  return false;
151  } else if (other.TopLeftCorner().col > BottomRightCorner().col) {
152  // `other` is completely to the right of `this`.
153  return false;
154  }
155  return true;
156 }

158 bool ShipPlacement::IsWithinBounds(const int board_width,
159  const int board_height) const {
160  const Cell tl_corner = TopLeftCorner();
161  const Cell br_corner = BottomRightCorner();
162 
163  return (tl_corner.row >= 0 && tl_corner.row < board_height) &&
164  (br_corner.row >= 0 && br_corner.row < board_height) &&
165  (tl_corner.col >= 0 && tl_corner.col < board_width) &&
166  (br_corner.col >= 0 && br_corner.col < board_width);
167 }
