27def solve_stackelberg(game, is_first_leader=True):
28 """Solves the optimal mixed strategty to commit to for the leader.
29
30 Args:
31 game: a pyspiel game,
32 is_first_leader: if true, then player 0 is the leader, o.w. player 1 is
33 the leader.
34
35 Returns:
36 (player0 strategy, player1 strategy, player0 payoff, player1 payoff) at an
37 SSE.
38 """
39 p_mat = game_payoffs_array(game)
40 assert len(p_mat) == 2
41 if is_first_leader:
42 leader_payoff, follower_payoff = p_mat[0], p_mat[1]
43 else:
44 leader_payoff, follower_payoff = p_mat[1].T, p_mat[0].T
45
46 num_leader_strategies, num_follower_strategies = leader_payoff.shape
47
48 leader_eq_value = -float("inf")
49 follower_eq_value = None
50 leader_eq_strategy = None
51 follower_eq_strategy = None
52
53 for t in range(num_follower_strategies):
54 p_s = cp.Variable(num_leader_strategies, nonneg=True)
55 constraints = [p_s <= 1, cp.sum(p_s) == 1]
56 for t_ in range(num_follower_strategies):
57 if t_ == t:
58 continue
59 constraints.append(
60 p_s @ follower_payoff[:, t_] <= p_s @ follower_payoff[:, t]
61 )
62 prob = cp.Problem(cp.Maximize(p_s @ leader_payoff[:, t]), constraints)
63 prob.solve()
64 p_s_value = p_s.value
65 if p_s_value is None:
66 continue
67 leader_strategy = _simplex_projection(p_s.value.reshape(-1)).reshape(-1, 1)
68 leader_value = leader_strategy.T.dot(leader_payoff)[0, t]
69 if leader_value > leader_eq_value:
70 leader_eq_strategy = leader_strategy
71 follower_eq_strategy = t
72 leader_eq_value = leader_value
73 follower_eq_value = leader_strategy.T.dot(follower_payoff)[0, t]
74
75 assert leader_eq_strategy is not None, p_mat
76 if is_first_leader:
77 return (
78 leader_eq_strategy.reshape(-1),
79 np.identity(num_follower_strategies)[follower_eq_strategy],
80 leader_eq_value,
81 follower_eq_value,
82 )
83 else:
84 return (np.identity(num_follower_strategies)[follower_eq_strategy],
85 leader_eq_strategy.reshape(-1), follower_eq_value, leader_eq_value)
