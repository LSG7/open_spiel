149std::vector<Action> OpenSpielHanabiState::LegalActions() const {
150 if (IsTerminal()) {
151 return {};
152 } else if (IsChanceNode()) {
153 auto outcomes_and_probs = state_.ChanceOutcomes();
154 const int n = outcomes_and_probs.first.size();
155 std::vector<Action> chance_outcomes;
156 chance_outcomes.reserve(n);
157 for (int i = 0; i < n; ++i) {
158 chance_outcomes.emplace_back(
159 game_->HanabiGame().GetChanceOutcomeUid(outcomes_and_probs.first[i]));
160 }
161 return chance_outcomes;
162 } else {
163 auto moves = state_.LegalMoves(CurrentPlayer());
164 std::vector<Action> actions;
165 actions.reserve(moves.size());
166 for (auto m : moves) actions.push_back(game_->HanabiGame().GetMoveUid(m));
167 return actions;
168 }
169}

187void OpenSpielHanabiState::DoApplyAction(Action action) {
188 auto move = IsChanceNode() ? game_->HanabiGame().GetChanceOutcome(action)
189 : game_->HanabiGame().GetMove(action);
190 if (state_.MoveIsLegal(move)) {
191 if (!IsChanceNode()) prev_state_score_ = state_.Score();
192 state_.ApplyMove(move);
193 } else {
194 SpielFatalError(absl::StrCat("Invalid move ", move.ToString()));
195 }
196}

205void OpenSpielHanabiState::ObservationTensor(Player player,
206 absl::Span<float> values) const {
207 SPIEL_CHECK_GE(player, 0);
208 SPIEL_CHECK_LT(player, num_players_);
209
210 auto obs = game_->Encoder().Encode(
211 hanabi_learning_env::HanabiObservation(state_, player));
212 SPIEL_CHECK_EQ(values.size(), obs.size());
213 for (int i = 0; i < obs.size(); ++i) values.at(i) = obs[i];

220ActionsAndProbs OpenSpielHanabiState::ChanceOutcomes() const {
221 auto outcomes_and_probs = state_.ChanceOutcomes();
222 const int n = outcomes_and_probs.first.size();
223 ActionsAndProbs chance_outcomes;
224 chance_outcomes.reserve(n);
225 for (int i = 0; i < n; ++i) {
226 chance_outcomes.emplace_back(
227 game_->HanabiGame().GetChanceOutcomeUid(outcomes_and_probs.first[i]),
228 outcomes_and_probs.second[i]);
229 }
230 return chance_outcomes;
