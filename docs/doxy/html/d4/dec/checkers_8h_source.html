18// Implementation of the board game Checkers.
19// https://en.wikipedia.org/wiki/Checkers
20//
21// Some notes about this implementation:
22// - Capturing:
23// When capturing an opponent's piece is possible, capturing is mandatory
24// in this implementation.
25// - Drawing:
26// Game is drawn if no pieces have been removed in 40 moves
27// http://www.flyordie.com/games/help/checkers/en/games_rules_checkers.html
28// - Custom board dimensions:
29// Dimensions of the board can be customised by calling the
30// CheckersState(rows, columns) constructer with the desired
31// number of rows and columns

83// This is a small helper to track historical turn info not stored in the moves.

152// Game object.
153class CheckersGame : public Game {
154 public:
155 explicit CheckersGame(const GameParameters& params);
156 int NumDistinctActions() const override;

158 return absl::make_unique<CheckersState>(shared_from_this(), rows_,
159 columns_);
160 }
161 int NumPlayers() const override { return kNumPlayers; }
162 double MinUtility() const override { return -1; }
163 absl::optional<double> UtilitySum() const override { return 0; }
164 double MaxUtility() const override { return 1; }
165 std::vector<int> ObservationTensorShape() const override {
166 return {kCellStates, rows_, columns_};
167 }
168 // There is arbitrarily chosen number to ensure the game is finite.
169 int MaxGameLength() const override { return 1000; }
170
171 private:
172 int rows_;
173 int columns_;
174};
