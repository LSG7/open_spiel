84void MatchingPennies3pState::DoApplyActions(
85 const std::vector<Action>& actions) {
86 SPIEL_CHECK_EQ(actions.size(), NumPlayers());
87
88 // Player 1 gets a +1 if they match P2, -1 otherwise
89 returns_[0] = (actions[0] == actions[1] ? 1.0 : -1.0);
90
91 // Player 2 gets a +1 if they match P3, -1 otherwise
92 returns_[1] = (actions[1] == actions[2] ? 1.0 : -1.0);
93
94 // Player 3 gets a +1 if they don't match P1, -1 otherwise
95 returns_[2] = (actions[2] != actions[0] ? 1.0 : -1.0);
96
97 terminal_ = true;
98}
