27 void RandomBotMainLoop() {
28  std::mt19937 rng;
29 
30  // Read the current setup.
31  std::string game_name;
32  int play_as;
33  std::cin >> game_name >> play_as;
34 
35  std::cerr << game_name << ' ' << play_as
36  << std::endl; // For debugging purposes.
37 
38  // Load the provided game.
39  std::shared_ptr<const Game> game = LoadGame(game_name);
40 
41  // Observations will be received later from the referee.
42  // The referee factors the observation into public (common knowledge across
43  // all players) and private parts.
44  std::shared_ptr<Observer> public_observer =
45  game->MakeObserver(kPublicObsType, {});
46  std::shared_ptr<Observer> private_observer =
47  game->MakeObserver(kPrivateObsType, {});
48  Observation public_observation(*game, public_observer);
49  Observation private_observation(*game, private_observer);
50 
51  // Now there is 5 secs warm-up time that could be used for loading relevant
52  // supplementary data. All data can be read/written from persistent /data
53  // directory mounted from an external storage.
54  std::cout << "ready" << std::endl;
55 
56  // Loop per match. This loop will end when referee instructs the player to do
57  // so.
58  while (true) {
59  // Acknowledge the match started.
60  std::cout << "start" << std::endl;
61 
62  // This is just a placeholder for other implementations -- we do not use
63  // state in random agent, as it receives list of actions it can pick from.
64  std::unique_ptr<State> state = game->NewInitialState();
65 
66  std::string message;
67  while (true) { // Loop per state in match.
68  std::getline(std::cin, message); // Read message from the referee.
69  if (message.empty()) continue;
70  std::cerr << message << std::endl; // For debugging purposes.
71 
72  if (message == "tournament over") {
73  // The tournament is now over: there is 60 sec shutdown time
74  // available for processing tournament results by the agent,
75  // for example to update supplementary data.
76  std::cout << "tournament over" << std::endl;
77  std::exit(0);
78  }
79 
80  if (message.rfind("match over", 0) == 0) {
81  // The full message has format "game over 123"
82  // where 123 is the final float reward received by this bot.
83  //
84  // Note that this message does not necessarily mean the match
85  // reached a terminal state: if opponent crashed / violated
86  // rules, the match will be over as well.
87  std::cout << "match over" << std::endl;
88  break;
89  }
90 
91  // Regular message: a public and private observation followed by
92  // a list of legal actions (if the bot should be acting).
93  std::vector<absl::string_view> xs = absl::StrSplit(message, ' ');
94  SPIEL_CHECK_GE(xs.size(), 2);
95  std::vector<Action> legal_actions;
96  for (int i = 0; i < xs.size(); ++i) {
97  absl::string_view x = xs[i];
98  if (i <= 1) { // Observations.
99  std::string decoded;
100  absl::Base64Unescape(x, &decoded);
101  if (i == 0)
102  public_observation.Decompress(decoded);
103  else if (i == 1)
104  private_observation.Decompress(decoded);
105  } else { // Legal actions.
106  Action a;
107  auto [p, ec] = std::from_chars(x.begin(), x.end(), a);
108  SPIEL_CHECK_TRUE(p == x.end());
109  legal_actions.push_back(a);
110  }
111  }
112 
113  const bool should_act = !legal_actions.empty();
114  if (should_act) {
115  std::uniform_int_distribution<int> dist(0, legal_actions.size() - 1);
116  std::cout << legal_actions[dist(rng)] << std::endl;
117  } else {
118  // Pondering phase, i.e. thinking when the bot is not acting.
119  // The time limit is always at least 0.2s, but can be longer,
120  // up to 5s, depending on how long the opponent thinks.
121  std::cout << "ponder" << std::endl; // This bot does not ponder.
122  }
123  }
124 
125  SPIEL_CHECK_EQ(message.rfind("match over", 0), 0);
126  int score = 0;
127  std::from_chars(message.data() + 11, message.data() + message.size(),
128  score);
129  std::cerr << "score: " << score << std::endl;
130  }
131 }
