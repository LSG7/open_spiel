37def add_all_constraints(
38 game: coalitional_game.CoalitionalGame,
39 x: cp.Variable,
40 e: cp.Variable,
41 constraints: List[Any]):
42 # \sum x_i + e >= v(S), for all subsets S \subseteq N
43 for c in itertools.product([0, 1], repeat=game.num_players()):
44 coalition = np.asarray(c)
45 val_coalition = game.coalition_value(coalition)
46 constraints.append(x @ coalition + e >= val_coalition)

50 num: int,

52 """Simple uniform constraint sampler (with replacement)."""

54 def func(game: coalitional_game.CoalitionalGame,
55 x: cp.Variable, e: cp.Variable, constraints: List[Any]):
56 for _ in range(num):
57 coalition = np.random.randint(2, size=game.num_players())
58 val_coalition = game.coalition_value(coalition)
59 constraints.append(x @ coalition + e >= val_coalition)

63def solve_least_core_lp(
64 game: coalitional_game.CoalitionalGame,
65 constraint_function: ConstraintsSamplingFuncType,
66) -> Tuple[np.ndarray, float]:
67 """Solve the LP described in Yan & Procaccia, equation (1).
68
69 This LP enumerates all (exponentially many!) possible coalitions, with one
70 constraint per coalition. Will not scale to games with too many players.
71
72 Args:
73 game: the game the LP solves.
74 constraint_function: function that adds the constraints
75
76 Returns:
77 solution: an array with num_players entries,
78 epsilon: the lowest epsilon.
79 """
80 # TODO(author5): handle errors gracefully. E.g. if solving the LP fails.
81
82 num_players = game.num_players()
83 val_gc = game.coalition_value(np.ones(game.num_players()))
84
85 # min e
86 # indices 0 - n-1 correspond to x_i, index n corresponds to e
87 x = cp.Variable(num_players, nonneg=True)
88 e = cp.Variable() # note: epsilon can be negative when the core is non-empty!
89
90 objective = cp.Minimize(e)
91 constraints = []
92
93 # \sum_{i in N} x_i = v(N)
94 constraints.append(x @ np.ones(num_players) == val_gc)
95
96 # Add the constraints
97 constraint_function(game, x, e, constraints)
98
99 prob = cp.Problem(objective, constraints)
100 _ = prob.solve()
101 # The optimal value for x is stored in `x.value`.
102
103 return x.value, e.value
