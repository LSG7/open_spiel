78 FixedActions ActionView::fixed_action(Player player, int action_index) const {
79  SPIEL_CHECK_EQ(current_player, kSimultaneousPlayerId);
80  int prod_after = 1;
81  for (int pl = player + 1; pl < legal_actions.size(); pl++) {
82  const std::vector<Action>& actions = legal_actions[pl];
83  if (!actions.empty()) prod_after *= actions.size();
84  }
85  int prod_before = 1;
86  for (int pl = 0; pl < player; pl++) {
87  const std::vector<Action>& actions = legal_actions[pl];
88  if (!actions.empty()) prod_before *= actions.size();
89  }
90  int num_actions = legal_actions[player].size();
91  return FixedActions{action_index, num_actions, prod_before, prod_after};
92 }

105 // This essentially imitates a generator that uses a nested for loop:
106 //
107 // for i in range(prod_after):
108 // for j in range(prod_before):
109 // yield prod_before * (fixed_action + i * num_actions) + j
