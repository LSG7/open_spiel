26class ProjectedPolicyMunchausen(mirror_descent.ProjectedPolicy):
27 """Project values on the policy simplex."""
28
29 def __init__(
30 self,
31 game: pyspiel.Game,
32 player_ids: List[int],
33 state_value: value.ValueFunction,
34 learning_rate: float,
35 policy: policy_lib.Policy,
36 ):
37 """Initializes the projected policy.
38
39 Args:
40 game: The game to analyze.
41 player_ids: list of player ids for which this policy applies; each should
42 be in the range 0..game.num_players()-1.
43 state_value: The state value to project.
44 learning_rate: The learning rate.
45 policy: The policy to project.
46 """
47 super().__init__(game, player_ids, state_value)
48 self._learning_rate = learning_rate
49 self._policy = policy
50
51 def action_probabilities(self,
52 state: pyspiel.State,
53 player_id: Optional[int] = None) -> Dict[int, float]:
54 del player_id
55 action_logit = [
56 (a, self._learning_rate * self.value(state, action=a) + np.log(p))
57 for a, p in self._policy.action_probabilities(state).items()
58 ]
59 action, logit = zip(*action_logit)
60 return dict(zip(action, mirror_descent.softmax_projection(logit)))

63class MunchausenMirrorDescent(mirror_descent.MirrorDescent):
64 """Munchausen Online Mirror Descent algorithm.
65
66 This algorithm is equivalent to the online mirror descent algorithm but
67 instead of summing value functions, it directly computes the cumulative
68 Q-function using a penalty with respect to the previous policy.
69 """
70
71 def eval_state(self, state: pyspiel.State, learning_rate: float):
72 """Evaluate the value of a state."""
73 state_str = state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)
74 # Return the already calculated value if present.
75 if self._state_value_state_value.has(state_str):
76 return self._state_value_state_value(state_str)
77 # Otherwise, calculate the value of the state.
78 v = self.get_state_value(state, learning_rate)
79 self._state_value_state_value.set_value(state_str, v)
80 return v
81
82 def get_projected_policy(self) -> policy_lib.Policy:
83 """Returns the projected policy."""
84 return ProjectedPolicyMunchausen(self._game_game,
85 list(range(self._game_game.num_players())),
86 self._state_value_state_value, self._lr_lr, self._policy_policy)
