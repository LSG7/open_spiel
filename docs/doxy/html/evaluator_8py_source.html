27 def __init__(self, game, model, cache_size=2**16):
28 """An AlphaZero MCTS Evaluator."""
29 if game.num_players() != 2:
30 raise ValueError("Game must be for two players.")
31 game_type = game.get_type()
32 if game_type.reward_model != pyspiel.GameType.RewardModel.TERMINAL:
33 raise ValueError("Game must have terminal rewards.")
34 if game_type.dynamics != pyspiel.GameType.Dynamics.SEQUENTIAL:
35 raise ValueError("Game must have sequential turns.")
36
37 self._model = model
38 self._cache = lru_cache.LRUCache(cache_size)
39
40 def cache_info(self):
41 return self._cache.info()
42
43 def clear_cache(self):
44 self._cache.clear()
45
46 def _inference(self, state):
47 # Make a singleton batch
48 obs = np.expand_dims(state.observation_tensor(), 0)
49 mask = np.expand_dims(state.legal_actions_mask(), 0)
50
51 # ndarray isn't hashable
52 cache_key = obs.tobytes() + mask.tobytes()
53
54 value, policy = self._cache.make(
55 cache_key, lambda: self._model.inference(obs, mask))
56
57 return value[0, 0], policy[0] # Unpack batch
58
59 def evaluate(self, state):
60 """Returns a value for the given state."""
61 value, _ = self._inference(state)
62 return np.array([value, -value])
63
64 def prior(self, state):
65 if state.is_chance_node():
66 return state.chance_outcomes()
67 else:
68 # Returns the probabilities for all actions.
69 _, policy = self._inference(state)
70 return [(action, policy[action]) for action in state.legal_actions()]
