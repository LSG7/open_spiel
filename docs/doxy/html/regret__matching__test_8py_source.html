56 def test_rps(self):
57 game = pyspiel.load_game("matrix_rps")
58 payoffs_array = game_payoffs_array(game)
59 strategies = regret_matching.regret_matching(
60 [payoffs_array[0], payoffs_array[1]],
61 initial_strategies=[
62 np.array([0.1, 0.4, 0.5]),
63 np.array([0.9, 0.1, 0.01])
64 ],
65 iterations=50000,
66 gamma=1e-6)
67 self.assertLen(strategies, 2, "Wrong strategy length.")
68 # places=1 corresponds to an absolute difference of < 0.001
69 self.assertAlmostEqual(strategies[0][0], 1 / 3., places=2)
70 self.assertAlmostEqual(strategies[0][1], 1 / 3., places=2)
71 self.assertAlmostEqual(strategies[0][2], 1 / 3., places=2)
