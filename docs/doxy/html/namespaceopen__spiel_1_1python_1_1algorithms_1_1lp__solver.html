248def solve_zero_sum_matrix_game(game):
249 """Solves a matrix game by using linear programming.
250
251 Args:
252 game: a pyspiel MatrixGame
253
254 Returns:
255 A 4-tuple containing:
256 - p0_sol (array-like): probability distribution over row actions
257 - p1_sol (array-like): probability distribution over column actions
258 - p0_sol_value, expected value to the first player
259 - p1_sol_value, expected value to the second player
260 """
261
262 # Solving a game for player i (e.g. row player) requires finding a mixed
263 # policy over player i's pure strategies (actions) such that a value of the
264 # mixed strategy against every opponent pure strategy is maximized.
265 #
266 # For more detail, please refer to Sec 4.1 of Shoham & Leyton-Brown, 2009:
267 # Multiagent Systems: Algorithmic, Game-Theoretic, and Logical Foundations
268 # http://www.masfoundations.org/mas.pdf
269 #
270 # For the row player the LP looks like:
271 # max V
272 # st. sigma_a1 \dot col_0 >= V
273 # sigma_a2 \dot col_1 >= V
274 # .
275 # .
276 # sigma_am \cot col_n >= V
277 # for all i, sigma_ai >= 0
278 # sigma \dot 1 = 1
279 assert isinstance(game, pyspiel.MatrixGame)
280 assert game.get_type().information == pyspiel.GameType.Information.ONE_SHOT
281 assert game.get_type().utility == pyspiel.GameType.Utility.ZERO_SUM
282 num_rows = game.num_rows()
283 num_cols = game.num_cols()
284 cvxopt.solvers.options["show_progress"] = False
285
286 # First, do the row player (player 0).
287 lp0 = LinearProgram(OBJ_MAX)
288 for r in range(num_rows): # one var per action / pure strategy
289 lp0.add_or_reuse_variable(r, lb=0)
290 lp0.add_or_reuse_variable(num_rows) # V
291 lp0.set_obj_coeff(num_rows, 1.0) # max V
292 for c in range(num_cols):
293 lp0.add_or_reuse_constraint(c, CONS_TYPE_GEQ)
294 for r in range(num_rows):
295 lp0.set_cons_coeff(c, r, game.player_utility(0, r, c))
296 lp0.set_cons_coeff(c, num_rows, -1.0) # -V >= 0
297 lp0.add_or_reuse_constraint(num_cols + 1, CONS_TYPE_EQ)
298 lp0.set_cons_rhs(num_cols + 1, 1.0)
299 for r in range(num_rows):
300 lp0.set_cons_coeff(num_cols + 1, r, 1.0)
301 sol = lp0.solve()
302 p0_sol = sol[:-1]
303 p0_sol_val = sol[-1]
304
305 # Now, the column player (player 1).
306 lp1 = LinearProgram(OBJ_MAX)
307 for c in range(num_cols): # one var per action / pure strategy
308 lp1.add_or_reuse_variable(c, lb=0)
309 lp1.add_or_reuse_variable(num_cols) # V
310 lp1.set_obj_coeff(num_cols, 1) # max V
311 for r in range(num_rows):
312 lp1.add_or_reuse_constraint(r, CONS_TYPE_GEQ)
313 for c in range(num_cols):
314 lp1.set_cons_coeff(r, c, game.player_utility(1, r, c))
315 lp1.set_cons_coeff(r, num_cols, -1.0) # -V >= 0
316 lp1.add_or_reuse_constraint(num_rows + 1, CONS_TYPE_EQ)
317 lp1.set_cons_rhs(num_rows + 1, 1.0)
318 for c in range(num_cols):
319 lp1.set_cons_coeff(num_rows + 1, c, 1.0)
320 sol = lp1.solve()
321 p1_sol = sol[:-1]
322 p1_sol_val = sol[-1]
323
324 return p0_sol, p1_sol, p0_sol_val, p1_sol_val
