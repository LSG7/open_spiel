96class LiarsPokerState(pyspiel.State):
97 """A python version of the Liars Poker state."""
98
99 def __init__(self, game):
100 """Constructor; should only be called by Game.new_initial_state."""
101 super().__init__(game)
102 # Game attributes
103 self._num_players = game.num_players()
104 self._hand_length = game.hand_length
105 self._num_digits = game.num_digits
106 self._deck = game.deck
107 self.hands = [[] for _ in range(self._num_players)]
108
109 # Action dynamics
110 self.total_possible_bids = (
111 game.hand_length * game.num_digits * self._num_players
112 )
113 self.bid_history = np.zeros((self.total_possible_bids, self._num_players))
114 self.challenge_history = np.zeros(
115 (self.total_possible_bids, self._num_players)
116 )
117 # self._current_player is only the valid current_player when cards have
118 # been dealt. Otherwise it's chance.
119 self._current_player = 0
120 self._max_bid = self._hand_length * self._num_digits * self._num_players
121 self._bid_originator = -1
122 self._current_action = -1
123 self._num_challenges = 0
124 self.is_rebid = False
125
126 # Game over dynamics
127 self._winner = -1
128 self._loser = -1
129
130 def current_player(self):
131 """Returns id of the current player to act.
132
133 The id is:
134 - TERMINAL if game is over.
135 - CHANCE if a player is drawing a number to fill out their hand.
136 - a number otherwise.
137 """
138 if self.is_terminal():
139 return pyspiel.PlayerId.TERMINAL
140 elif len(self.hands[self._num_players - 1]) < self._hand_length:
141 return pyspiel.PlayerId.CHANCE
142 else:
143 return self._current_player
144
145 def winner(self):
146 """Returns the id of the winner if the bid originator has won.
147
148 -1 otherwise.
149 """
150 return self._winner
151
152 def loser(self):
153 """Returns the id of the loser if the bid originator has lost.
154
155 -1 otherwise.
156 """
157 return self._loser
158
159 def _is_challenge_possible(self):
160 """A challenge is possible once the first bid is made."""
161 return self._current_action != -1
162
163 def _is_rebid_possible(self):
164 """A rebid is only possible when all players have challenged the original bid."""
165 return not self.is_rebid and self._num_challenges == self._num_players - 1
166
167 def _legal_actions(self, player):
168 """Returns a list of legal actions, sorted in ascending order."""
169 assert player >= 0
170 actions = []
171
172 if self._is_challenge_possible():
173 actions.append(CHALLENGE_ACTION)
174
175 if player != self._bid_originator or self._is_rebid_possible():
176 # Any move higher than the current bid is allowed.
177 # Bids start at BID_ACTION_OFFSET (1) as 0 represents the challenge
178 # action.
179 for bid in range(
180 max(BID_ACTION_OFFSET, self._current_action + 1), self._max_bid + 1
181 ):
182 actions.append(bid)
183
184 return actions
185
186 def chance_outcomes(self):
187 """Returns the possible chance outcomes and their probabilities."""
188 assert self.is_chance_node()
189 probability = 1.0 / self._num_digits
190 return [(digit, probability) for digit in self._deck]
191
192 def _decode_bid(self, bid):
193 """Turns a bid ID to a (count, number) tuple.
194
195 For example, take 2 players each with 2 numbers from the deck of 1, 2, and
196 3.
197 - A bid of two 1's would correspond to a bid id 1.
198 - Explanation: 1 is the lowest number, and the only lower bid would be
199 zero 1's.
200 - A bid of three 3's would correspond to a bid id 10.
201 - Explanation: 1-4 1's take bid ids 0-3. 1-4 2's take bid ids 4-7. 1 and
202 2 3's take bid ids 8 and 9.
203
204 Args:
205 bid: Bid ID in the range 0 to self._max_bid (non-inclusive).
206
207 Returns:
208 A tuple of (count, number). For example, (1, 2) represents one 2's.
209 """
210 number = bid % self._num_digits + 1
211 count = bid // self._num_digits + 1
212 return (count, number)
213
214 def encode_bid(self, count, number):
215 """Turns a count and number into a bid ID.
216
217 Bid ID is in the range 0 to self._max_bid (non-inclusive).
218
219 For example, take 2 players each with 2 numbers from the deck of 1, 2, and
220 3.
221 - A count of 2 and number of 1 would be a bid of two one's and a bid id 1.
222 - Explanation: 1 is the lowest number, and the only lower bid would be
223 zero 1's
224 corresponding to bid id 0.
225
226 Args:
227 count: The count of the bid.
228 number: The number of the bid.
229
230 Returns:
231 A single bid ID.
232 """
233 return (count - 1) * self._num_digits + number - 1
234
235 def _counts(self):
236 """Determines if the bid originator wins or loses."""
237 bid_count, bid_number = self._decode_bid(
238 self._current_action - BID_ACTION_OFFSET
239 )
240
241 # Count the number of bid_numbers from all players.
242 matches = 0
243 for player_id in range(self._num_players):
244 for digit in self.hands[player_id]:
245 if digit == bid_number:
246 matches += 1
247
248 # If the number of matches are at least the bid_count bid, then the bidder
249 # wins. Otherwise everyone else wins.
250 if matches >= bid_count:
251 self._winner = self._bid_originator
252 else:
253 self._loser = self._bid_originator
254
255 def _update_bid_history(self, bid, player):
256 """Writes a player's bid into memory."""
257 self.bid_history[bid][player] = 1
258
259 def _update_challenge_history(self, bid, player):
260 """Write a player's challenge for a bid into memory."""
261 self.challenge_history[bid][player] = 1
262
263 def _apply_action(self, action):
264 """Applies an action and updates the state."""
265 if self.is_chance_node():
266 # If we are still populating hands, draw a number for the current player.
267 self.hands[self._current_player].append(action)
268 elif action == CHALLENGE_ACTION:
269 assert self._is_challenge_possible()
270 self._update_challenge_history(
271 self._current_action - BID_ACTION_OFFSET, self._current_player
272 )
273 self._num_challenges += 1
274 # If there is no ongoing rebid, check if all players challenge before
275 # counting. If there is an ongoing rebid, count once all the players
276 # except the bidder challenges.
277 if (not self.is_rebid and self._num_challenges == self._num_players) or (
278 self.is_rebid and self._num_challenges == self._num_players - 1
279 ):
280 self._counts()
281 else:
282 # Set the current bid to the action.
283 self._current_action = action
284 if self._current_player == self._bid_originator:
285 # If the bid originator is bidding again, we have a rebid.
286 self.is_rebid = True
287 else:
288 # Otherwise, we have a regular bid.
289 self.is_rebid = False
290 # Set the bid originator to the current player.
291 self._bid_originator = self._current_player
292 self._update_bid_history(
293 self._current_action - BID_ACTION_OFFSET, self._current_player
294 )
295 self._num_challenges = 0
296 self._current_player = (self._current_player + 1) % self._num_players
297
298 def _action_to_string(self, player, action):
299 """Action -> string."""
300 if player == pyspiel.PlayerId.CHANCE:
301 return f"Deal: {action}"
302 elif action == CHALLENGE_ACTION:
303 return "Challenge"
304 else:
305 count, number = self._decode_bid(action - BID_ACTION_OFFSET)
306 return f"Bid: {count} of {number}"
307
308 def is_terminal(self):
309 """Returns True if the game is over."""
310 return self._winner >= 0 or self._loser >= 0
311
312 def returns(self):
313 """Total reward for each player over the course of the game so far."""
314 if self._winner != -1:
315 bidder_reward = self._num_players - 1
316 others_reward = -1.0
317 elif self._loser != -1:
318 bidder_reward = -1 * (self._num_players - 1)
319 others_reward = 1.0
320 else:
321 # Game is not over.
322 bidder_reward = 0.0
323 others_reward = 0.0
324 return [
325 others_reward if player_id != self._bid_originator else bidder_reward
326 for player_id in range(self._num_players)
327 ]
328
329 def __str__(self):
330 """String for debug purposes. No particular semantics are required."""
331 if self._current_action != -1:
332 count, number = self._decode_bid(self._current_action - BID_ACTION_OFFSET)
333 else:
334 count, number = "None", "None"
335 return (
336 "Hands: {}, Bidder: {}, Current Player: {}, Current Bid: {} of {},"
337 " Rebid: {}".format(
338 self.hands,
339 self._bid_originator,
340 self.current_player(),
341 count,
342 number,
343 self.is_rebid,
344 )
345 )

348class LiarsPokerObserver:
349 """Observer, conforming to the PyObserver interface (see observation.py).
350
351 An observation will consist of the following:
352 - One hot encoding of the current player number: [0 0 0 1 0 0 0]
353 - A vector of length hand_length containing the digits in a player's hand.
354 - Two matrices each of size (hand_length * num_digits * num_players,
355 num_players)
356 will store bids and challenges respectively. Each row in the matrix
357 corresponds
358 to a particular bid (e.g. one 1, two 5s, or eight 3s). 0 will represent no
359 action. 1 will represent a player's bid or a player's challenge.
360 - One bit for whether we are rebidding: [1] rebid occuring, [0] otherwise
361 - One bit for whether we are counting: [1] COUNTS called, [0] otherwise
362 """
363
364 def __init__(
365 self, iig_obs_type, num_players, hand_length, num_digits, params=None
366 ):
367 """Initiliazes an empty observation tensor."""
368 del params
369 self.num_players = num_players
370 self.hand_length = hand_length
371
372 # Determine which observation pieces we want to include.
373 # Pieces is a list of tuples containing observation pieces.
374 # Pieces are described by their (name, number of elements, and shape).
375 pieces = [(
376 "player",
377 num_players,
378 (num_players,),
379 )] # One-hot encoding for the player id.
380 if iig_obs_type.private_info == pyspiel.PrivateInfoType.SINGLE_PLAYER:
381 # Vector containing the digits in a player's hand
382 pieces.append(("private_hand", hand_length, (hand_length,)))
383 if iig_obs_type.public_info:
384 pieces.append(("rebid_state", 1, (1,)))
385 pieces.append(("counts_state", 1, (1,)))
386 if iig_obs_type.perfect_recall:
387 # One-hot encodings for players' moves at every round.
388 total_possible_rounds = hand_length * num_digits * num_players
389 pieces.append((
390 "bid_history",
391 total_possible_rounds * num_players,
392 (total_possible_rounds, num_players),
393 ))
394 pieces.append((
395 "challenge_history",
396 total_possible_rounds * num_players,
397 (total_possible_rounds, num_players),
398 ))
399
400 # Build the single flat tensor.
401 total_size = sum(size for name, size, shape in pieces)
402 self.tensor = np.zeros(total_size, np.float32)
403
404 # Build the named & reshaped views of the bits of the flat tensor.
405 self.dict = {}
406 index = 0
407 for name, size, shape in pieces:
408 self.dict[name] = self.tensor[index : index + size].reshape(shape)
409 index += size
410
411 def set_from(self, state, player):
412 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
413 self.tensor.fill(0)
414 if "player" in self.dict:
415 self.dict["player"][player] = 1
416 if (
417 "private_hand" in self.dict
418 and len(state.hands[player]) == self.hand_length
419 ):
420 self.dict["private_hand"] = np.asarray(state.hands[player])
421 if "rebid_state" in self.dict:
422 self.dict["rebid_state"][0] = int(state.is_rebid)
423 if "counts_state" in self.dict:
424 self.dict["counts_state"][0] = int(state.is_terminal())
425 if "bid_history" in self.dict:
426 self.dict["bid_history"] = state.bid_history
427 if "challenge_history" in self.dict:
428 self.dict["challenge_history"] = state.challenge_history
429
430 def string_from(self, state, player):
431 """Observation of `state` from the PoV of `player`, as a string."""
432 pieces = []
433 if "player" in self.dict:
434 pieces.append(f"p{player}")
435 if (
436 "private_hand" in self.dict
437 and len(state.hands[player]) == self.hand_length
438 ):
439 pieces.append(f"hand:{state.hands[player]}")
440 if "rebid_state" in self.dict:
441 pieces.append(f"rebid:{[int(state.is_rebid)]}")
442 if "counts_state" in self.dict:
443 pieces.append(f"counts:{[int(state.is_terminal())]}")
444 if "bid_history" in self.dict:
445 for bid in range(len(state.bid_history)):
446 if np.any(state.bid_history[bid] == 1):
447 pieces.append("b:{}.".format(bid))
448 if "challenge_history" in self.dict:
449 for bid in range(len(state.challenge_history)):
450 if np.any(state.challenge_history[bid] == 1):
451 pieces.append("c:{}.".format(bid))
452 return " ".join(str(p) for p in pieces)
