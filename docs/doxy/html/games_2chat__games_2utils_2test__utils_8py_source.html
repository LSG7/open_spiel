50class MockClient():
51 """Mock LLM client."""
52
53 def __init__(self):
54 # for cycling through mock response options
55 self._idxs = {'names': 0,
56 'tones': 0,
57 'examples': 0}
58
59 def sample(self, model: str, length: int, seed: int, prompt: str
60 ) -> MockResponse:
61 """Returns string responses according to fixed prompt styles."""
62 del model, length, seed
63 prompt_lower = prompt.lower()
64 if 'names' in prompt_lower:
65 dummy_names = ['Suzy', 'Bob', 'Alice', 'Doug', 'Arun', 'Maria', 'Zhang']
66 dummy_name = dummy_names[self._idxs['names']]
67 self._idxs['names'] = (self._idxs['names'] + 1) % len(dummy_names)
68 return MockResponse(dummy_name + '\n')
69 elif 'tones' in prompt_lower:
70 dummy_tones = ['Happy', 'Sad', 'Angry']
71 dummy_tone = dummy_tones[self._idxs['tones']]
72 self._idxs['tones'] = (self._idxs['tones'] + 1) % len(dummy_tones)
73 return MockResponse(dummy_tone + '\n')
74 elif 'list of items' in prompt_lower:
75 num_examples = 10
76 dummy_examples = [f'Example-{i}' for i in range(num_examples)]
77 dummy_example = dummy_examples[self._idxs['examples']]
78 self._idxs['examples'] = (self._idxs['examples'] + 1) % num_examples
79 return MockResponse(dummy_example + '\n')
80 elif 'score' in prompt_lower or 'value' in prompt_lower:
81 return MockResponse('5\n')
82 elif 'summary' in prompt_lower:
83 return MockResponse('This is a summary of the dialogue. We are happy.\n')
84 elif emails.BLOCK_OPT in prompt:
85 return MockResponse('\nThat all sounds good to me.\n')
86 else:
87 raise ValueError('Prompt not recognized!\n\n' + prompt)
