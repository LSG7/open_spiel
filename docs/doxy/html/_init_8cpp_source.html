89void STDCALL SetResources(
90 int maxMemoryMB,
91 int maxThreadsIn)
92{
93 // Figure out system resources.
94 int ncores;
95 unsigned long long kilobytesFree;
96 sysdep.GetHardware(ncores, kilobytesFree);
97
98 // Memory usage will be limited to the lower of:
99 // - maxMemoryMB + 30% (if given; statistically this works out)
100 // - 70% of free memory
101 // - 1800 MB if we're on a 32-bit system.
102
103 const int memMaxGivenMB = (maxMemoryMB == 0 ? 1000000 :
104 static_cast<int>(1.3 * maxMemoryMB));
105 const int memMaxFreeMB = static_cast<int>(0.70 * kilobytesFree / 1024);
106 const int memMax32bMB = (sizeof(void *) == 4 ? 1800 : 1000000);
107
108 int memMaxMB = min(memMaxGivenMB, memMaxFreeMB);
109 memMaxMB = min(memMaxMB, memMax32bMB);
110
111 // The number of threads will be limited by:
112 // - If threading set as single-threaded or compiled only
113 // single-threaded: 1
114 // - If threading set as one of the IMPL variants: ncores
115 // whatever the user says (as we currently don't have control)
116 // - Otherwise the lower of maxThreads and ncores
117
118 int thrMax;
119 if (sysdep.IsSingleThreaded())
120 thrMax = 1;
121 else if (sysdep.IsIMPL() || maxThreadsIn <= 0)
122 thrMax = ncores;
123 else
124 thrMax = min(maxThreadsIn, ncores);
125 thrMax = max(1, thrMax);
126
127 // For simplicity we won't vary the amount of memory per thread
128 // in the small and large versions.
129
130 int noOfThreads, noOfLargeThreads, noOfSmallThreads;
131 if (thrMax * THREADMEM_LARGE_MAX_MB <= memMaxMB)
132 {
133 // We have enough memory for the maximum number of large threads.
134 noOfThreads = thrMax;
135 noOfLargeThreads = thrMax;
136 noOfSmallThreads = 0;
137 }
138 else if (thrMax * THREADMEM_SMALL_MAX_MB > memMaxMB)
139 {
140 // We don't even have enough memory for only small threads.
141 // We'll limit the number of threads.
142 noOfThreads = max(1, static_cast<int>(memMaxMB /
143 static_cast<double>(THREADMEM_SMALL_MAX_MB)));
144 noOfLargeThreads = 0;
145 noOfSmallThreads = noOfThreads;
146 }
147 else
148 {
149 // We'll have a mixture with as many large threads as possible.
150 const double d = static_cast<double>(
151 THREADMEM_LARGE_MAX_MB - THREADMEM_SMALL_MAX_MB);
152
153 noOfThreads = thrMax;
154 noOfLargeThreads = static_cast<int>(
155 (memMaxMB - thrMax * THREADMEM_SMALL_MAX_MB) / d);
156 noOfSmallThreads = thrMax - noOfLargeThreads;
157 }
158
159 sysdep.RegisterParams(noOfThreads, memMaxMB);
160
161 scheduler.RegisterThreads(noOfThreads);
162
163 // Clear the thread memory and fill it up again.
164 memory.Resize(0, DDS_TT_SMALL, 0, 0);
165 if (noOfLargeThreads > 0)
166 memory.Resize(static_cast<unsigned>(noOfLargeThreads),
167 DDS_TT_LARGE, THREADMEM_LARGE_DEF_MB, THREADMEM_LARGE_MAX_MB);
168 if (noOfSmallThreads > 0)
169 memory.Resize(static_cast<unsigned>(noOfThreads),
170 DDS_TT_SMALL, THREADMEM_SMALL_DEF_MB, THREADMEM_SMALL_MAX_MB);
171
172 if (memory.NumThreads() < 1)
173 {
174 cout << "Critical Error Init.cpp. 0 threads available." << endl;
175 exit(1);
176 }
177 threadMgr.Reset(noOfThreads);
178
179 InitDebugFiles();
180
181 if (! _initialized)
182 {
183 _initialized = 1;
184 InitConstants();
185 }
186}

196void InitConstants()
197{
198 // highestRank[aggr] is the highest absolute rank in the
199 // suit represented by aggr. The absolute rank is 2 .. 14.
200 // Similarly for lowestRank.
201 highestRank[0] = 0;
202 lowestRank [0] = 0;
203 for (int aggr = 1; aggr < 8192; aggr++)
204 {
205 for (int r = 14; r >= 2; r--)
206 {
207 if (aggr & bitMapRank[r])
208 {
209 highestRank[aggr] = r;
210 break;
211 }
212 }
213 for (int r = 2; r <= 14; r++)
214 {
215 if (aggr & bitMapRank[r])
216 {
217 lowestRank[aggr] = r;
218 break;
219 }
220 }
221 }
222
223 // The use of the counttable to give the number of bits set to
224 // one in an integer follows an implementation by Thomas Andrews.
225
226 // counttable[aggr] is the number of '1' bits (binary weight)
227 // in aggr.
228 for (int aggr = 0; aggr < 8192; aggr++)
229 {
230 counttable[aggr] = 0;
231 for (int r = 0; r < 13; r++)
232 {
233 if (aggr & (1 << r))
234 {
235 counttable[aggr]++;
236 }
237 }
238 }
239
240 // relRank[aggr][absolute rank] is the relative rank of
241 // that absolute rank in the suit represented by aggr.
242 // The relative rank is 2 .. 14.
243 memset(relRank[0], 0, 15);
244 for (int aggr = 1; aggr < 8192; aggr++)
245 {
246 char ord = 0;
247 for (int r = 14; r >= 2; r--)
248 {
249 if (aggr & bitMapRank[r])
250 {
251 ord++;
252 relRank[aggr][r] = ord;
253 }
254 }
255 }
256
257 // winRanks[aggr][leastWin] is the absolute suit represented
258 // by aggr, but limited to its top "leastWin" bits.
259 for (int aggr = 0; aggr < 8192; aggr++)
260 {
261 winRanks[aggr][0] = 0;
262 for (int leastWin = 1; leastWin < 14; leastWin++)
263 {
264 int res = 0;
265 int nextBitNo = 1;
266 for (int r = 14; r >= 2; r--)
267 {
268 if (aggr & bitMapRank[r])
269 {
270 if (nextBitNo <= leastWin)
271 {
272 res |= bitMapRank[r];
273 nextBitNo++;
274 }
275 else
276 break;
277 }
278 }
279 winRanks[aggr][leastWin] = static_cast<unsigned short>(res);
280 }
281 }
282
283 // groupData[ris] is a representation of the suit (ris is
284 // "rank in suit") in terms of runs of adjacent bits.
285 // 1 1100 1101 0110
286 // has 4 runs, so lastGroup is 3, and the entries are
287 // 0: 4 and 0x0002, gap 0x0000 (lowest gap unused, though)
288 // 1: 6 and 0x0000, gap 0x0008
289 // 2: 9 and 0x0040, gap 0x0020
290 // 3: 14 and 0x0c00, gap 0x0300
291
292 int topside[15] =
293 {
294 0x0000, 0x0000, 0x0000, 0x0001, // 2, 3,
295 0x0003, 0x0007, 0x000f, 0x001f, // 4, 5, 6, 7,
296 0x003f, 0x007f, 0x00ff, 0x01ff, // 8, 9, T, J,
297 0x03ff, 0x07ff, 0x0fff // Q, K, A
298 };
299
300 int botside[15] =
301 {
302 0xffff, 0xffff, 0x1ffe, 0x1ffc, // 2, 3,
303 0x1ff8, 0x1ff0, 0x1fe0, 0x1fc0, // 4, 5, 6, 7,
304 0x1f80, 0x1f00, 0x1e00, 0x1c00, // 8, 9, T, J,
305 0x1800, 0x1000, 0x0000 // Q, K, A
306 };
307
308 // So the bit vector in the gap between a top card of K
309 // and a bottom card of T is
310 // topside[K] = 0x07ff &
311 // botside[T] = 0x1e00
312 // which is 0x0600, the binary code for QJ.
313
314 groupData[0].lastGroup = -1;
315
316 groupData[1].lastGroup = 0;
317 groupData[1].rank[0] = 2;
318 groupData[1].sequence[0] = 0;
319 groupData[1].fullseq[0] = 1;
320 groupData[1].gap[0] = 0;
321
322 int topBitRank = 1;
323 int nextBitRank = 0;
324 int topBitNo = 2;
325 int g;
326
327 for (int ris = 2; ris < 8192; ris++)
328 {
329 if (ris >= (topBitRank << 1))
330 {
331 // Next top bit
332 nextBitRank = topBitRank;
333 topBitRank <<= 1;
334 topBitNo++;
335 }
336
337 groupData[ris] = groupData[ris ^ topBitRank];
338
339 if (ris & nextBitRank) // 11... Extend group
340 {
341 g = groupData[ris].lastGroup;
342 groupData[ris].rank[g]++;
343 groupData[ris].sequence[g] |= nextBitRank;
344 groupData[ris].fullseq[g] |= topBitRank;
345 }
346 else // 10... New group
347 {
348 g = ++groupData[ris].lastGroup;
349 groupData[ris].rank[g] = topBitNo;
350 groupData[ris].sequence[g] = 0;
351 groupData[ris].fullseq[g] = topBitRank;
352 groupData[ris].gap[g] =
353 topside[topBitNo] & botside[ groupData[ris].rank[g - 1] ];
354 }
355 }
356}

434void SetDeal(
435 ThreadData * thrp)
436{
437 /* Initialization of the rel structure is inspired by
438 a solution given by Thomas Andrews */
439
440 for (int s = 0; s < DDS_SUITS; s++)
441 {
442 thrp->lookAheadPos.aggr[s] = 0;
443 for (int h = 0; h < DDS_HANDS; h++)
444 {
445 thrp->lookAheadPos.rankInSuit[h][s] = thrp->suit[h][s];
446 thrp->lookAheadPos.aggr[s] |= thrp->suit[h][s];
447 }
448 }
449
450 for (int s = 0; s < DDS_SUITS; s++)
451 {
452 for (int h = 0; h < DDS_HANDS; h++)
453 thrp->lookAheadPos.length[h][s] = static_cast<unsigned char>(
454 counttable[thrp->lookAheadPos.rankInSuit[h][s]]);
455 }
456
457 // Clubs are implicit, for a given trick number.
458 for (int h = 0; h < DDS_HANDS; h++)
459 {
460 thrp->lookAheadPos.handDist[h] =
461 static_cast<long long>(
462 (thrp->lookAheadPos.length[h][0] << 8) |
463 (thrp->lookAheadPos.length[h][1] << 4) |
464 (thrp->lookAheadPos.length[h][2] ));
465 }
466}

469void SetDealTables(
470 ThreadData * thrp)
471{
472 unsigned int topBitRank = 1;
473 unsigned int topBitNo = 2;
474
475 // Initialization of the rel structure is inspired by
476 // a solution given by Thomas Andrews.
477
478 // rel[aggr].absRank[absolute rank][suit].hand is the hand
479 // (N = 0, E = 1 etc.) which holds the absolute rank in
480 // the suit characterized by aggr.
481 // rel[aggr].absRank[absolute rank][suit].rank is the
482 // relative rank of that card.
483
484 for (int s = 0; s < DDS_SUITS; s++)
485 {
486 for (int ord = 1; ord <= 13; ord++)
487 {
488 thrp->rel[0].absRank[ord][s].hand = -1;
489 thrp->rel[0].absRank[ord][s].rank = 0;
490 }
491 }
492
493 // handLookup[suit][absolute rank] is the hand (N = 0 etc.)
494 // holding the absolute rank in suit.
495
496 int handLookup[DDS_SUITS][15];
497 for (int s = 0; s < DDS_SUITS; s++)
498 {
499 for (int r = 14; r >= 2; r--)
500 {
501 handLookup[s][r] = 0;
502 for (int h = 0; h < DDS_HANDS; h++)
503 {
504 if (thrp->suit[h][s] & bitMapRank[r])
505 {
506 handLookup[s][r] = h;
507 break;
508 }
509 }
510 }
511 }
512
513 thrp->transTable->Init(handLookup);
514
515 relRanksType * relp;
516 for (unsigned int aggr = 1; aggr < 8192; aggr++)
517 {
518 if (aggr >= (topBitRank << 1))
519 {
520 /* Next top bit */
521 topBitRank <<= 1;
522 topBitNo++;
523 }
524
525 thrp->rel[aggr] = thrp->rel[aggr ^ topBitRank];
526 relp = &thrp->rel[aggr];
527
528 int weight = counttable[aggr];
529 for (int c = weight; c >= 2; c--)
530 {
531 for (int s = 0; s < DDS_SUITS; s++)
532 {
533 relp->absRank[c][s].hand = relp->absRank[c - 1][s].hand;
534 relp->absRank[c][s].rank = relp->absRank[c - 1][s].rank;
535 }
536 }
537 for (int s = 0; s < DDS_SUITS; s++)
538 {
539 relp->absRank[1][s].hand =
540 static_cast<char>(handLookup[s][topBitNo]);
541 relp->absRank[1][s].rank = static_cast<char>(topBitNo);
542 }
543 }
544}

547void InitWinners(
548 const deal& dl,
549 pos& posPoint,
550 ThreadData const * thrp)
551{
552 int hand, suit, rank;
553 unsigned short int startMovesBitMap[DDS_HANDS][DDS_SUITS];
554
555 for (int h = 0; h < DDS_HANDS; h++)
556 for (int s = 0; s < DDS_SUITS; s++)
557 startMovesBitMap[h][s] = 0;
558
559 for (int k = 0; k < posPoint.handRelFirst; k++)
560 {
561 hand = handId(dl.first, k);
562 suit = dl.currentTrickSuit[k];
563 rank = dl.currentTrickRank[k];
564 startMovesBitMap[hand][suit] |= bitMapRank[rank];
565 }
566
567 int aggr;
568 for (int s = 0; s < DDS_SUITS; s++)
569 {
570 aggr = 0;
571 for (int h = 0; h < DDS_HANDS; h++)
572 aggr |= startMovesBitMap[h][s] | thrp->suit[h][s];
573
574 posPoint.winner[s].rank = thrp->rel[aggr].absRank[1][s].rank;
575 posPoint.winner[s].hand = thrp->rel[aggr].absRank[1][s].hand;
576 posPoint.secondBest[s].rank = thrp->rel[aggr].absRank[2][s].rank;
577 posPoint.secondBest[s].hand = thrp->rel[aggr].absRank[2][s].hand;
578 }
579}
