15"""Example use of the RCFR algorithm on Kuhn Poker."""
16
17from absl import app
18from absl import flags
19import tensorflow.compat.v1 as tf
20
21from open_spiel.python.algorithms import rcfr
22import pyspiel
23
24tf.enable_eager_execution()
25
26FLAGS = flags.FLAGS
27
28flags.DEFINE_integer("iterations", 100, "Number of iterations")
29flags.DEFINE_string("game", "kuhn_poker", "Name of the game")
30flags.DEFINE_integer("players", 2, "Number of players")
31flags.DEFINE_integer("print_freq", 10, "How often to print the exploitability")
32flags.DEFINE_boolean("bootstrap", False,
33 "Whether or not to use bootstrap targets")
34flags.DEFINE_boolean("truncate_negative", False,
35 "Whether or not to truncate negative targets to zero")
36flags.DEFINE_integer(
37 "buffer_size", -1,
38 "A reservoir buffer size. A non-positive size implies an effectively "
39 "infinite buffer.")
40flags.DEFINE_integer("num_hidden_layers", 1,
41 "The number of hidden layers in the regret model.")
42flags.DEFINE_integer("num_hidden_units", 13,
43 "The number of hidden layers in the regret model.")
44flags.DEFINE_integer(
45 "num_hidden_factors", 8,
46 "The number of factors in each hidden layer in the regret model.")
47flags.DEFINE_boolean(
48 "use_skip_connections", True,
49 "Whether or not to use skip connections in the regret model.")
50flags.DEFINE_integer(
51 "num_epochs", 200,
52 "The number of epochs to run between each iterations to update the regret "
53 "models.")
54flags.DEFINE_integer("batch_size", 100, "The regret model training batch size.")
55flags.DEFINE_float("step_size", 0.01, "The ADAM (AMSGrad) optimizer step size.")
56
57
58def main(_):
59 game = pyspiel.load_game(FLAGS.game, {"players": FLAGS.players})
60
61 models = []
62 for _ in range(game.num_players()):
63 models.append(
64 rcfr.DeepRcfrModel(
65 game,
66 num_hidden_layers=FLAGS.num_hidden_layers,
67 num_hidden_units=FLAGS.num_hidden_units,
68 num_hidden_factors=FLAGS.num_hidden_factors,
69 use_skip_connections=FLAGS.use_skip_connections))
70
71 if FLAGS.buffer_size > 0:
72 solver = rcfr.ReservoirRcfrSolver(
73 game,
74 models,
75 FLAGS.buffer_size,
76 truncate_negative=FLAGS.truncate_negative)
77 else:
78 solver = rcfr.RcfrSolver(
79 game,
80 models,
81 truncate_negative=FLAGS.truncate_negative,
82 bootstrap=FLAGS.bootstrap)
83
84 def _train_fn(model, data):
85 """Train `model` on `data`."""
86 data = data.shuffle(FLAGS.batch_size * 10)
87 data = data.batch(FLAGS.batch_size)
88 data = data.repeat(FLAGS.num_epochs)
89
90 optimizer = tf.keras.optimizers.Adam(lr=FLAGS.step_size, amsgrad=True)
91
92 @tf.function
93 def _train():
94 for x, y in data:
95 optimizer.minimize(
96 lambda: tf.losses.huber_loss(y, model(x), delta=0.01), # pylint: disable=cell-var-from-loop
97 model.trainable_variables)
98
99 _train()
100
101 # End of _train_fn
102
103 for i in range(FLAGS.iterations):
104 solver.evaluate_and_update_policy(_train_fn)
105 if i % FLAGS.print_freq == 0:
106 conv = pyspiel.exploitability(game, solver.average_policy())
107 print("Iteration {} exploitability {}".format(i, conv))
108
109
110if __name__ == "__main__":
111 app.run(main)
