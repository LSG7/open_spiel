15# Lint as python3
16"""Play a game, selecting random moves, and save what we see.
17
18This can be used to check by hand the behaviour of a game, and also
19as the basis for test cases.
20
21Example usage:
22```
23playthrough --game kuhn_poker --params players=3
24```
25"""
26
27from absl import app
28from absl import flags
29from absl import logging
30
31from open_spiel.python.algorithms import generate_playthrough
32
33FLAGS = flags.FLAGS
34
35flags.DEFINE_string(
36 "game", "kuhn_poker", "Name of the game, with optional parameters, e.g. "
37 "'kuhn_poker' or 'go(komi=4.5,board_size=19)'.")
38flags.DEFINE_string("output_file", None, "Where to write the data to.")
39flags.DEFINE_list("actions", None,
40 "A (possibly partial) list of action choices to make.")
41
42flags.DEFINE_string("update_path", None,
43 "If set, regenerates all playthroughs in the path.")
44flags.DEFINE_bool(
45 "alsologtostdout", False,
46 "If True, the trace will be written to std-out while it "
47 "is being constructed (in addition to the usual behavior).")
48flags.DEFINE_integer("shard", 0, "The shard to update.")
49flags.DEFINE_integer("num_shards", 1, "How many shards to use for updates.")
50
51
52def main(unused_argv):
53 if FLAGS.update_path:
54 generate_playthrough.update_path(FLAGS.update_path, FLAGS.shard,
55 FLAGS.num_shards)
56 else:
57 if not FLAGS.game:
58 raise ValueError("Must specify game")
59 actions = FLAGS.actions
60 if actions is not None:
61 actions = [int(x) for x in actions]
62 text = generate_playthrough.playthrough(
63 FLAGS.game, actions, alsologtostdout=FLAGS.alsologtostdout)
64 if FLAGS.output_file:
65 with open(FLAGS.output_file, "w") as f:
66 f.write(text)
67 else:
68 logging.info(text)
69
70
71if __name__ == "__main__":
72 app.run(main)
