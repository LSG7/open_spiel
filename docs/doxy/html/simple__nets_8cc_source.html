29SimpleLinearImpl::SimpleLinearImpl(int input_size, int output_size,
30 bool activate_relu)
31 : simple_linear_(torch::nn::LinearOptions(/*in_features*/ input_size,
32 /*out_features*/ output_size)),
33 activate_relu_(activate_relu) {
34 double stddev = 1.0 / std::sqrt(input_size);
35 double lower = -2.0 * stddev;
36 double upper = 2.0 * stddev;
37 for (auto& named_parameter : simple_linear_->named_parameters()) {
38 if (named_parameter.key().find("weight") != std::string::npos) {
39 torch::Tensor uniform_param = torch::nn::init::uniform_(
40 named_parameter.value()).to(torch::kFloat64);
41 double clip_lower = 0.5 * (1.0 + std::erf(lower / kSqrt2));
42 double clip_upper = 0.5 * (1.0 + std::erf(upper / kSqrt2));
43 torch::Tensor new_param = kSqrt2 * torch::erfinv(
44 2.0 * ((clip_upper - clip_lower) * uniform_param + clip_lower) - 1.0);
45 named_parameter.value().data() = new_param;
46 }
47 if (named_parameter.key().find("bias") != std::string::npos) {
48 named_parameter.value().data() = torch::zeros({output_size});
49 }
50 }
51 register_module("simple_linear_", simple_linear_);
52}

54torch::Tensor SimpleLinearImpl::forward(torch::Tensor x) {
55 if (activate_relu_) {
56 return torch::relu(simple_linear_->forward(x));
57 } else {
58 return simple_linear_->forward(x);;

62MLPImpl::MLPImpl(int input_size, const std::vector<int>& hidden_layers_sizes,
63 int output_size, bool activate_final)
64 : input_size_(input_size),
65 hidden_layers_sizes_(hidden_layers_sizes),
66 output_size_(output_size),
67 activate_final_(activate_final) {
68 int layer_size = input_size_;
69 for (auto h_size : hidden_layers_sizes_) {
70 layers_->push_back(SimpleLinear(/*input_size*/layer_size,
71 /*output_size*/h_size));
72 layer_size = h_size;
73 }
74 layers_->push_back(SimpleLinear(/*input_size*/ layer_size,
75 /*output_size*/ output_size_,
76 /*activate_final*/ activate_final_));
77 register_module("layers_", layers_);
78}

80torch::Tensor MLPImpl::forward(torch::Tensor x) {
81 for (int i=0; i < hidden_layers_sizes_.size() + 1; i++) {
