152 PieceType promotion_type = PieceType::kEmpty, bool is_castling = false)
153 : from(from),

162 std::string ToLAN() const;

164 // Converts to standard algebraic notation, as required by portable game
165 // notation (PGN). It is a chess move notation that is designed to be
166 // human-readable and concise.
167 //
168 // Unlike the LAN format, generating a SAN string requires the board the move
169 // is generated from.
170 //
171 // There are 3 types of SAN moves -
172 // 1. O-O (short castle)
173 // 2. O-O-O (long castle)
174 // 3. [piece type][from file][from rank][x][to square][=Promo][annotations]
175 //
176 // [piece type] is omitted for pawns
177 // [from file] is only included if 1) move is a pawn capture, or 2) it's
178 // required for disambiguation (see below).
179 // [from rank] is only included if it's required for disambiguation.
180 // [x] is only included for captures
181 // [to square] is always included
182 // [=Promo] is only included for promotions ("=N", "=B", "=R", "=Q" depending
183 // on type promoting to).
184 // [annotations] are a list of 0 or more characters added with different
185 // meanings. The ones we care about are '+' for check, and '#' for
186 // checkmate. All others are optional.
187 //
188 // Disambiguation:
189 // If a move is not uniquely-identified otherwise, file and/or rank of the
190 // from square is inserted to disambiguate. When either one will disambiguate,
191 // file should be used. If file is unique, file is used. Otherwise if rank is
192 // unique, rank is used. If neither is unique (this happens rarely, usually
193 // after under-promoting to a minor piece with both original pieces still
194 // intact, or double queen promotions with original queen still intact), both
195 // are used.
196 //
197 // Examples:
198 // * e4 (pawn to e4)
199 // * exd5 (pawn on file e capture the piece on d5)
200 // * Nf3 (knight to f3)
201 // * Nxd5 (knight captures piece on d5)
202 // * Bed5 (bishop on file e to d5)
203 // * B5xc3 (bishop on rank 5 capture piece on c3)
204 // * Ne5f7 (knight on e5 to f7, when there are 3 knights on the board, one on
205 // e file, and one on 5th rank)
206 // * exd8=N#!! (pawn on e file capture piece on d8 and promote to knight
207 // resulting in checkmate in a surprisingly good move)
208 // * O-O-O!!N+/- (a surprisingly good long castle that is a theoretical
209 // novelty that gives white a clear but not winning advantage)
210 std::string ToSAN(const ChessBoard& board) const;
211
212 bool operator==(const Move& other) const {
213 return from == other.from && to == other.to && piece == other.piece &&
214 promotion_type == other.promotion_type &&
215 is_castling == other.is_castling;
216 }
217};

236// Specifies policy for pseudo legal moves generation.
237enum PseudoLegalMoveSettings {
238 // Standard legal moves (do not allow to move past enemy pieces).
239 kAcknowledgeEnemyPieces,
240 // Pseudo-legal moves, where a piece can move anywhere (according to the rules
241 // for that piece), except if it was blocked from doing so by other player's
242 // pieces. This is used in games, where the player may not know the position
243 // of an enemy piece (like Kriegspiel or RBC) and it can try to move past the
244 // enemy (for example a rook can try to move the other side of the board, even
245 // if it is in fact blocked by an unseen opponent's pawn).
246 kBreachEnemyPieces,
247};

342 // Does either side have sufficient material to mate?
343 // FIDE rules say it must be impossible to mate even with "most unskilled"
344 // counterplay. This would technically include things like pawns blocking
345 // either side from making progress.
346 // Eg. "8/4k3/8/p1p1p1p1/P1P1P1P1/8/4K3/8 w - -".
347 // However, detecting all such positions will require solving chess... so
348 // we detect a more generally-accepted subset of positions - those with the
349 // following material combinations:
350 // 1. K vs K
