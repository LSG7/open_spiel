32class NetA2C(hk.Module):
33 """A simple network with a policy head and a baseline value head."""
34
35 def __init__(self, num_actions, hidden_layers_sizes):
36 super().__init__()
37 self._num_actions = num_actions
38 self._hidden_layers_sizes = hidden_layers_sizes
39
40 def __call__(self, info_state):
41 """Process a batch of observations."""
42 torso = hk.nets.MLP(self._hidden_layers_sizes, activate_final=True)
43 hidden = torso(info_state)
44 policy_logits = hk.Linear(self._num_actions)(hidden)
45 baseline = hk.Linear(1)(hidden)
46 return policy_logits, baseline

66def generate_a2c_pi_loss(net_apply, loss_class, entropy_cost, l2_actor_weight,
67 lambda_):
68 """A function generator generates loss function."""
69
70 def _a2c_pi_loss(net_params, batch):
71 info_states, actions, rewards, discounts = batch["info_states"], batch[
72 "actions"], batch["rewards"], batch["discounts"]
73 policy_logits, baselines = net_apply(net_params, info_states)
74 policy_logits = policy_logits[:-1]
75
76 baselines = jnp.squeeze(baselines, axis=1)
77 baselines = jnp.concatenate([baselines[:-1], jnp.zeros(1)])
78 td_returns = rlax.lambda_returns(
79 rewards,
80 discounts,
81 baselines[1:],
82 lambda_=lambda_,
83 stop_target_gradients=True)
84 advantages = td_returns - baselines[:-1]
85 chex.assert_equal_shape([td_returns, actions, advantages])
86 pi_loss = loss_class(
87 logits_t=policy_logits,
88 a_t=actions,
89 adv_t=advantages,
90 w_t=jnp.ones(td_returns.shape))
91 ent_loss = rlax.entropy_loss(
92 logits_t=policy_logits, w_t=jnp.ones(td_returns.shape))
93 l2_loss = jnp.sum(jnp.square(jax.flatten_util.ravel_pytree(net_params)[0]))
94 return pi_loss + entropy_cost * ent_loss + l2_actor_weight * l2_loss
95
96 return _a2c_pi_loss

99def generate_a2c_critic_loss(net_apply, l2_critic_weight, lambda_):
100 """A function generator generates loss function."""
101
102 def _a2c_critic_loss(net_params, batch):
103 info_states, rewards, discounts = batch["info_states"], batch[
104 "rewards"], batch["discounts"]
105 _, baselines = net_apply(net_params, info_states)
106 baselines = jnp.squeeze(baselines, axis=1)
107 baselines = jnp.concatenate([baselines[:-1], jnp.zeros(1)])
108
109 td_lambda = rlax.td_lambda(
110 v_tm1=baselines[:-1],
111 r_t=rewards,
112 discount_t=discounts,
113 v_t=baselines[1:],
114 lambda_=lambda_,
115 stop_target_gradients=True)
116 l2_loss = jnp.sum(jnp.square(jax.flatten_util.ravel_pytree(net_params)[0]))
117 return jnp.mean(jnp.square(td_lambda)) + l2_critic_weight * l2_loss
118
119 return _a2c_critic_loss

122def generate_pg_pi_loss(net_apply, loss_class, entropy_cost, l2_actor_weight):
123 """A function generator generates loss function."""
124
125 def _pg_loss(net_params, batch):
126 info_states = batch["info_states"]
127 policy_logits, q_values = net_apply(net_params, info_states)
128 chex.assert_equal_shape([policy_logits, q_values])
129 pi_loss = loss_class(logits_t=policy_logits, q_t=q_values)
130 ent_loss = rlax.entropy_loss(
131 logits_t=policy_logits, w_t=jnp.ones(policy_logits.shape[:1]))
132 l2_loss = jnp.sum(jnp.square(jax.flatten_util.ravel_pytree(net_params)[0]))
133 return pi_loss + entropy_cost * ent_loss + l2_actor_weight * l2_loss
134
135 return _pg_loss

138def generate_pg_critic_loss(net_apply, l2_critic_weight, lambda_):
139 """A function generator generates loss function."""
140
141 def _critic_loss(net_params, batch):
142 info_states, actions, rewards, discounts = batch["info_states"], batch[
143 "actions"], batch["rewards"], batch["discounts"]
144 _, q_values = net_apply(net_params, info_states)
145 q_values = q_values[:-1]
146 q_values = jnp.concatenate(
147 [q_values, jnp.zeros(q_values[-1].reshape(1, -1).shape)])
148
149 actions = jnp.concatenate([actions, jnp.zeros(1, dtype=int)])
150 sarsa_lambda = rlax.sarsa_lambda(
151 q_tm1=q_values[:-1],
152 a_tm1=actions[:-1],
153 r_t=rewards,
154 discount_t=discounts,
155 q_t=q_values[1:],
156 a_t=actions[1:],
157 lambda_=lambda_,
158 stop_target_gradients=True)
159 l2_loss = jnp.sum(jnp.square(jax.flatten_util.ravel_pytree(net_params)[0]))
160 return jnp.mean(jnp.square(sarsa_lambda)) + l2_critic_weight * l2_loss
161
162 return _critic_loss

165def generate_act_func(net_apply):
166 """A function generator generates act function."""
167
168 def _act(net_params, info_state, action_mask, rng):
169 info_state = jnp.reshape(info_state, [1, -1])
170 policy_logits, _ = net_apply(net_params, info_state)
171 policy_probs = jax.nn.softmax(policy_logits, axis=1)
172
173 # Remove illegal actions, re-normalize probs
174 probs = policy_probs[0] * action_mask
175
176 probs /= jnp.sum(probs)
177 action = jax.random.choice(rng, len(probs), p=probs)
178 return action, probs
179
180 return _act

183class PolicyGradient(rl_agent.AbstractAgent):
184 """Policy Gradient Agent implementation in JAX."""
185
186 def __init__(self,
187 player_id,
188 info_state_size,
189 num_actions,
190 loss_str="a2c",
191 loss_class=None,
192 hidden_layers_sizes=(128,),
193 lambda_=1.0,
194 critic_learning_rate=0.01,
195 pi_learning_rate=0.001,
196 entropy_cost=0.01,
197 l2_weight_actor=0.0,
198 l2_weight_critic=0.0,
199 num_critic_before_pi=8,
200 additional_discount_factor=1.0,
201 max_global_gradient_norm=None,
202 optimizer_str="sgd",
203 seed=42):
204 """Initialize the PolicyGradient agent.
205
206 Args:
207 player_id: int, player identifier. Usually its position in the game.
208 info_state_size: int, info_state vector size.
209 num_actions: int, number of actions per info state.
210 loss_str: string or None. If string, must be one of ["rpg", "qpg", "rm",
211 "a2c"] and defined in `_get_loss_class`. If None, a loss class must be
212 passed through `loss_class`. Defaults to "a2c".
213 loss_class: Class or None. If Class, it must define the policy gradient
214 loss. If None a loss class in a string format must be passed through
215 `loss_str`. Defaults to None.
216 hidden_layers_sizes: iterable, defines the neural network layers. Defaults
217 to (128,), which produces a NN: [INPUT] -> [128] -> ReLU -> [OUTPUT].
218 lambda_: float, lambda in TD(lambda) or SARSA(lambda). Defaults to 1.0.
219 critic_learning_rate: float, learning rate used for Critic (Q or V).
220 Defaults to 0.001.
221 pi_learning_rate: float, learning rate used for Pi. Defaults to 0.001.
222 entropy_cost: float, entropy cost used to multiply the entropy loss. Can
223 be set to None to skip entropy computation. Defaults to 0.001.
224 l2_weight_actor: l2 penaly weight for actor network. Defaults to 0.0.
225 l2_weight_critic: l2 penalty weight for critic network. Defaults to
226 0.0.
227 num_critic_before_pi: int, number of Critic (Q or V) updates before each
228 Pi update. Defaults to 8 (every 8th critic learning step, Pi also
229 learns).
230 additional_discount_factor: float, additional discount to compute returns.
231 Defaults to 1.0, in which case, no extra discount is applied. None that
232 users must provide *only one of* `loss_str` or `loss_class`.
233 max_global_gradient_norm: float or None, maximum global norm of a gradient
234 to which the gradient is shrunk if its value is larger.
235 optimizer_str: String defining which optimizer to use. Supported values
236 are {sgd, adam}
237 seed: random seed
238 """
239 assert bool(loss_str) ^ bool(loss_class), "Please provide only one option."
240 self._kwargs = locals()
241 loss_class = loss_class if loss_class else self._get_loss_class(loss_str)
242
243 self.player_id = player_id
244 self._num_actions = num_actions
245 self._extra_discount = additional_discount_factor
246 self._num_critic_before_pi = num_critic_before_pi
247
248 self._episode_data = []
249 self._dataset = collections.defaultdict(list)
250 self._prev_time_step = None
251 self._prev_action = None
252
253 # Step counters
254 self._step_counter = 0
255 self._episode_counter = 0
256 self._num_learn_steps = 0
257
258 # Keep track of the last training loss achieved in an update step.
259 self._last_loss_value = None
260
261 self._loss_str = loss_str
262
263 # Network
264 # activate final as we plug logit and qvalue heads afterwards.
265 net_class = NetA2C if loss_str == "a2c" else NetPG
266
267 def net_func(info_input):
268 net = net_class(num_actions, hidden_layers_sizes)
269 return net(info_input)
270
271 hk_net = hk.without_apply_rng(hk.transform(net_func))
272
273 hk_net_apply = hk_net.apply
274 self.rng = jax.random.PRNGKey(seed)
275 init_inputs = jnp.ones((1, info_state_size))
276 self.hk_net_params = hk_net.init(self.rng, init_inputs)
277
278 self._act = jax.jit(generate_act_func(hk_net_apply))
279
280 if optimizer_str == "adam":
281 critic_optimizer = optax.adam(critic_learning_rate)
282 pi_optimizer = optax.adam(pi_learning_rate)
283
284 elif optimizer_str == "sgd":
285 critic_optimizer = optax.sgd(critic_learning_rate)
286 pi_optimizer = optax.sgd(pi_learning_rate)
287
288 else:
289 raise ValueError("Not implemented, choose from 'adam' and 'sgd'.")
290
291 if max_global_gradient_norm:
292 pi_optimizer = optax.chain(
293 pi_optimizer, optax.clip_by_global_norm(max_global_gradient_norm))
294 critic_optimizer = optax.chain(
295 critic_optimizer, optax.clip_by_global_norm(max_global_gradient_norm))
296
297 pi_opt_init, pi_opt_update = pi_optimizer.init, pi_optimizer.update
298 critic_opt_init, critic_opt_update = critic_optimizer.init, critic_optimizer.update
299
300 self._pi_opt_state = pi_opt_init(self.hk_net_params)
301
302 if loss_str == "a2c":
303 pi_loss_and_grad = jax.value_and_grad(
304 generate_a2c_pi_loss(hk_net_apply, loss_class, entropy_cost,
305 l2_weight_actor, lambda_))
306 critic_loss_and_grad = jax.value_and_grad(
307 generate_a2c_critic_loss(hk_net_apply, l2_weight_critic, lambda_))
308 self._critic_opt_state = critic_opt_init(self.hk_net_params)
309 else:
310 pi_loss_and_grad = jax.value_and_grad(
311 generate_pg_pi_loss(hk_net_apply, loss_class, entropy_cost,
312 l2_weight_actor))
313 critic_loss_and_grad = jax.value_and_grad(
314 generate_pg_critic_loss(hk_net_apply, l2_weight_critic, lambda_))
315 self._critic_opt_state = critic_opt_init(self.hk_net_params)
316
317 self._jit_pi_update = jax.jit(
318 self._get_update(pi_opt_update, pi_loss_and_grad))
319 self._jit_critic_update = jax.jit(
320 self._get_update(critic_opt_update, critic_loss_and_grad))
321
322 def _get_loss_class(self, loss_str):
323 if loss_str == "rpg":
324 return rlax.rpg_loss
325 elif loss_str == "qpg":
326 return rlax.qpg_loss
327 elif loss_str == "rm":
328 return rlax.rm_loss
329 elif loss_str == "a2c":
330 return rlax.policy_gradient_loss
331
332 def _get_update(self, opt_update, loss_fn):
333
334 def update(net_params, opt_state, batch):
335 loss_val, grad_val = loss_fn(net_params, batch)
336 updates, new_opt_state = opt_update(grad_val, opt_state)
337 new_net_params = optax.apply_updates(net_params, updates)
338 return new_net_params, new_opt_state, loss_val
339
340 return update

342 def step(self, time_step, is_evaluation=False):
343 """Returns the action to be taken and updates the network if needed.
344
345 Args:
346 time_step: an instance of rl_environment.TimeStep.
347 is_evaluation: bool, whether this is a training or evaluation call.
348
349 Returns:
350 A `rl_agent.StepOutput` containing the action probs and chosen action.
351 """
352 # Act step: don't act at terminal info states or if its not our turn.
353 if (not time_step.last()) and (time_step.is_simultaneous_move() or
354 self.player_id
355 == time_step.current_player()):
356 info_state = time_step.observations["info_state"][self.player_id]
357 legal_actions = time_step.observations["legal_actions"][self.player_id]
358 action_mask = np.zeros(self._num_actions)
359 action_mask[legal_actions] = 1
360 self.rng, _ = jax.random.split(self.rng)
361 action, probs = self._act(self.hk_net_params, np.asarray(info_state),
362 action_mask, self.rng)
363 else:
364 action = None
365 probs = []
366
367 if not is_evaluation:
368 self._step_counter += 1
369
370 # Add data points to current episode buffer.
371 if self._prev_time_step:
372 self._add_transition(time_step)
373
374 # Episode done, add to dataset and maybe learn.
375
376 if time_step.last():
377 self._episode_counter += 1
378
379 self._critic_update()
380 self._num_learn_steps += 1
381 if self._num_learn_steps % self._num_critic_before_pi == 0:
382 self._pi_update()
383 self._episode_data = []
384
385 self._prev_time_step = None
386 self._prev_action = None
387 return
388 else:
389 self._prev_time_step = time_step
390 self._prev_action = action
391
392 return rl_agent.StepOutput(action=action, probs=probs)

398 def _add_transition(self, time_step):
399 """Adds intra-episode transition to the `_episode_data` buffer.
400
401 Adds the transition from `self._prev_time_step` to `time_step`.
402 Args:
403 time_step: an instance of rl_environment.TimeStep.
404 """
405 assert self._prev_time_step is not None
406 legal_actions = (
407 self._prev_time_step.observations["legal_actions"][self.player_id])
408 legal_actions_mask = np.zeros(self._num_actions)
409 legal_actions_mask[legal_actions] = 1.0
410 transition = Transition(
411 info_state=(
412 self._prev_time_step.observations["info_state"][self.player_id][:]),
413 action=self._prev_action,
414 reward=time_step.rewards[self.player_id],
415 discount=time_step.discounts[self.player_id],
416 legal_actions_mask=legal_actions_mask,
417 next_info_state=(
418 time_step.observations["info_state"][self.player_id][:]))
419
420 self._episode_data.append(transition)

422 def _critic_update(self):
423 """Compute the Critic loss on sampled transitions & perform a critic update.
424
425 Returns:
426 The average Critic loss obtained on this batch.
427 """
428 batch = {}
429 batch["info_states"] = jnp.asarray(
430 [transition.info_state for transition in self._episode_data] +
431 [self._episode_data[-1].next_info_state])
432 batch["rewards"] = jnp.asarray(
433 [transition.reward for transition in self._episode_data])
434 batch["discounts"] = jnp.asarray(
435 [transition.discount for transition in self._episode_data])
436 if self._loss_str != "a2c":
437 batch["actions"] = jnp.asarray(
438 [transition.action for transition in self._episode_data])
439
440 self.hk_net_params, self._critic_opt_state, self._last_critic_loss_value = self._jit_critic_update(
441 self.hk_net_params, self._critic_opt_state, batch)
442 return self._last_critic_loss_value
443
444 def _pi_update(self):
445 """Compute the Pi loss on sampled transitions and perform a Pi update.
446
447 Returns:
448 The average Pi loss obtained on this batch.
449 """
450 batch = {}
451 batch["info_states"] = jnp.asarray(
452 [transition.info_state for transition in self._episode_data] +
453 [self._episode_data[-1].next_info_state])
454
455 if self._loss_str == "a2c":
456 batch["discounts"] = jnp.asarray(
457 [transition.discount for transition in self._episode_data])
458 batch["actions"] = jnp.asarray(
459 [transition.action for transition in self._episode_data])
460 batch["rewards"] = jnp.asarray(
461 [transition.reward for transition in self._episode_data])
462 self.hk_net_params, self._pi_opt_state, self._last_pi_loss_value = self._jit_pi_update(
463 self.hk_net_params, self._pi_opt_state, batch)
464 return self._last_pi_loss_value
