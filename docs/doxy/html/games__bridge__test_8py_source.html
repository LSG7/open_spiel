198 def test_benchmark_observation(self):
199 game = pyspiel.load_game('bridge(use_double_dummy_result=false)')
200
201 def make_state():
202 state = game.new_initial_state()
203 for _ in range(60):
204 a = random.choice(state.legal_actions())
205 state.apply_action(a)
206 if state.is_terminal(): break
207 return state
208
209 batch_size = 16
210 obs_shape = [batch_size] + game.observation_tensor_shape()
211 states = [make_state() for _ in range(batch_size)]
212
213 def make_obs_copy():
214 inputs = np.zeros(obs_shape)
215 for i in range(batch_size):
216 inputs[i, :] = states[i].observation_tensor()
217 return inputs
218
219 def make_obs_inplace():
220 inputs = np.zeros(obs_shape, np.float32)
221 for i in range(batch_size):
222 states[i].write_observation_tensor(inputs[i])
223 return inputs
224
225 repeat = 2
226 number = 2
227 times = np.array(timeit.repeat(make_obs_copy, number=number, repeat=repeat))
228 print(f'OpenSpiel {times.mean():.4}s, min {times.min():.4}s')
229 times = np.array(
230 timeit.repeat(make_obs_inplace, number=number, repeat=repeat))
231 print(f'In-place {times.mean():.4}s, min {times.min():.4}s')
232
233 np.testing.assert_array_equal(make_obs_copy(), make_obs_inplace())
