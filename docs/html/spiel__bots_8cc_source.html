35 class UniformRandomBot : public Bot {
36  public:
37  UniformRandomBot(Player player_id, int seed)
38  : player_id_(player_id), rng_(seed) {}
39  ~UniformRandomBot() = default;
40 
41  void RestartAt(const State&) override {}
42  Action Step(const State& state) override {
43  return StepWithPolicy(state).second;
44  }
45  bool ProvidesPolicy() override { return true; }
46  ActionsAndProbs GetPolicy(const State& state) override {
47  ActionsAndProbs policy;
48  auto legal_actions = state.LegalActions(player_id_);
49  const int num_legal_actions = legal_actions.size();
50  const double p = 1.0 / num_legal_actions;
51  for (auto action : legal_actions) policy.emplace_back(action, p);
52  return policy;
53  }
54 
55  std::pair<ActionsAndProbs, Action> StepWithPolicy(
56  const State& state) override {
57  ActionsAndProbs policy = GetPolicy(state);
58  const int num_legal_actions = policy.size();
59 
60  int selection =
61  absl::uniform_int_distribution<int>(0, num_legal_actions - 1)(rng_);
62  return std::make_pair(policy, policy[selection].first);
63  }
64 
65  bool IsClonable() const override { return true; }
66  std::unique_ptr<Bot> Clone() override {
67  return std::make_unique<UniformRandomBot>(*this);
68  }
69  UniformRandomBot(const UniformRandomBot& other) = default;
70 
71  private:
72  const Player player_id_;
73  std::mt19937 rng_;
74 };

76 // A UniformRandomBot that keeps a copy of the state up to date. This exists
77 // primarily to verify that InformAction is called correctly by the run loop.
78 class StatefulRandomBot : public UniformRandomBot {
79  public:
80  StatefulRandomBot(const Game& game, Player player_id, int seed)
81  : UniformRandomBot(player_id, seed), state_(game.NewInitialState()) {}
82 
83  void Restart() override { state_ = state_->GetGame()->NewInitialState(); }
84  void RestartAt(const State& state) override { state_ = state.Clone(); }
85  void InformAction(const State& state, Player player_id,
86  Action action) override {
87  CheckStatesEqual(state, *state_);
88  state_->ApplyAction(action);
89  }
90  ActionsAndProbs GetPolicy(const State& state) override {
91  CheckStatesEqual(state, *state_);
92  return UniformRandomBot::GetPolicy(*state_);
93  }
94  std::pair<ActionsAndProbs, Action> StepWithPolicy(
95  const State& state) override {
96  std::pair<ActionsAndProbs, Action> ret =
97  UniformRandomBot::StepWithPolicy(*state_);
98  state_->ApplyAction(ret.second);
99  return ret;
100  }
101 
102  std::unique_ptr<Bot> Clone() override {
103  return std::make_unique<StatefulRandomBot>(*this);
104  }
105  StatefulRandomBot(const StatefulRandomBot& other)
106  : UniformRandomBot(other), state_(other.state_->Clone()) {}
107 
108  private:
109  void CheckStatesEqual(const State& state1, const State& state2) const {
110  SPIEL_CHECK_EQ(state1.History(), state2.History());
111  SPIEL_CHECK_EQ(state1.CurrentPlayer(), state2.CurrentPlayer());
112  SPIEL_CHECK_EQ(state1.LegalActions(), state2.LegalActions());
113  if (!state1.IsChanceNode()) {
114  SPIEL_CHECK_EQ(state1.ObservationTensor(), state2.ObservationTensor());
115  }
116  }
117  std::unique_ptr<State> state_;
118 };

152 class FixedActionPreferenceBot : public Bot {
153  public:
154  FixedActionPreferenceBot(Player player_id, const std::vector<Action>& actions)
155  : Bot(), player_id_(player_id), actions_(actions) {}
156  ~FixedActionPreferenceBot() = default;
157 
158  void RestartAt(const State&) override {}
159  Action Step(const State& state) override {
160  return StepWithPolicy(state).second;
161  }
162  bool ProvidesPolicy() override { return true; }
163  ActionsAndProbs GetPolicy(const State& state) override {
164  std::vector<Action> legal_actions = state.LegalActions(player_id_);
165  std::unordered_set<Action> legal_actions_set =
166  std::unordered_set<Action>(legal_actions.begin(), legal_actions.end());
167  for (Action action : actions_) {
168  if (legal_actions_set.count(action) == 1) {
169  return {{action, 1.0}};
170  }
171  }
172  SpielFatalError("No legal actions in action list.");
173  }
174 
175  std::pair<ActionsAndProbs, Action> StepWithPolicy(
176  const State& state) override {
177  ActionsAndProbs actions_and_probs = GetPolicy(state);
178  return {actions_and_probs, actions_and_probs[0].first};
179  }
180 
181  bool IsClonable() const override { return true; }
182  std::unique_ptr<Bot> Clone() override {
183  return std::make_unique<FixedActionPreferenceBot>(*this);
184  }
185  FixedActionPreferenceBot(const FixedActionPreferenceBot& other) = default;
186 
187  private:
188  const Player player_id_;
189  std::vector<Action> actions_;
190 };

239 std::vector<Action> ActionsFromStr(const absl::string_view& str,
240  const absl::string_view& delim) {
241  std::vector<Action> actions;
242  for (absl::string_view token : absl::StrSplit(str, delim)) {
243  int v;
244  SPIEL_CHECK_TRUE(absl::SimpleAtoi(token, &v));
245  actions.push_back(v);
246  }
247  return actions;

307 std::vector<std::string> BotRegisterer::BotsThatCanPlayGame(const Game& game) {
308  std::vector<std::string> bot_names;
309  for (const auto& key_val : factories()) {
310  bool can_play_for_all = true;
311  for (int player_id = 0; player_id < game.NumPlayers(); ++player_id) {
312  if (!key_val.second->CanPlayGame(game, player_id)) {
313  can_play_for_all = false;
314  break;
315  }
316  }
317  if (can_play_for_all) bot_names.push_back(key_val.first);
318  }
319  return bot_names;
320 }

345 std::unique_ptr<Bot> LoadBot(const std::string& bot_name,
346  const std::shared_ptr<const Game>& game,
347  Player player_id) {
348  GameParameters params = GameParametersFromString(bot_name);
349 
350  // We use the "name" parameter, as that is the "short_name", which is what we
351  // want. Otherwise, this will use the "long name", which includes the config.
352  // e.g. if the bot_name is "my_bot(parameter=value)", then we want the

354  return LoadBot(params["name"].string_value(), game, player_id, params);
