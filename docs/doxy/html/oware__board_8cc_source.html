45size_t OwareBoard::HashValue() const {
46 // Hashing similar to boost::hash_combine.
47 size_t hash = current_player;
48 for (int player_score : score) {
49 hash ^= (size_t)player_score + 0x9e3779b9 + (hash << 6) + (hash >> 2);
50 }
51 for (int house_seeds : seeds) {
52 hash ^= (size_t)house_seeds + 0x9e3779b9 + (hash << 6) + (hash >> 2);
53 }
54 return hash;
55}
56int OwareBoard::TotalSeeds() const {

63 }
64 return total;
