62def gradients(dist, payoff_matrices, num_players, proj_grad=True):
63 """Computes exploitablity gradient.
64
65 Args:
66 dist: list of 1-d np.arrays, current estimate of nash distribution
67 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
68 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
69 are sorted and arrays should be indexed in the same order
70 num_players: int, number of players, in case payoff_matrices is abbreviated
71 proj_grad: bool, if True, projects dist gradient onto simplex
72 Returns:
73 gradient of payoff w.r.t. (dist) as tuple
74 unregularized exploitability (stochastic estimate)
75 unregularized exploitability (stochastic estimate) *duplicate
76 """
77 # first compute best responses and payoff gradients
78 grad_dist = []
79 unreg_exp = []
80 for i in range(num_players):
81
82 nabla_i = np.zeros_like(dist[i])
83 # TODO(imgemp): decide if averaging over nablas provides best comparison
84 for j in range(num_players):
85 if j == i:
86 continue
87 if i < j:
88 hess_i_ij = payoff_matrices[(i, j)][0]
89 else:
90 hess_i_ij = payoff_matrices[(j, i)][1].T
91
92 nabla_ij = hess_i_ij.dot(dist[j])
93 nabla_i += nabla_ij / float(num_players - 1)
94
95 grad_dist_i = -nabla_i
96 if proj_grad:
97 grad_dist_i = simplex.project_grad(grad_dist_i)
98 grad_dist.append(nabla_i)
99
100 unreg_exp.append(np.max(nabla_i) - nabla_i.dot(dist[i]))
101
102 return (grad_dist,), np.mean(unreg_exp), np.mean(unreg_exp)
