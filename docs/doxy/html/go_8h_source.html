136 return std::unique_ptr<State>(
137 new GoState(shared_from_this(), board_size_, komi_, handicap_));

139
140 std::vector<int> ObservationTensorShape() const override {
141 // Planes: black, white, empty, and a bias plane indicating komi (whether
142 // white is to play).
143 return {CellStates() + 1, board_size_, board_size_};
144 }
145
146 TensorLayout ObservationTensorLayout() const override {
147 return TensorLayout::kCHW;

151

154 return LossUtility() + WinUtility();
155 }
156 double MaxUtility() const override { return WinUtility(); }

158 int MaxGameLength() const override { return max_game_length_; }
159
160 private:
