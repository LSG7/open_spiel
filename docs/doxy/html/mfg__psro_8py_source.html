80def main(unused_argv):
81 logging.info("Loading %s", FLAGS.game_name)
82 mfg_game = pyspiel.load_game(
83 FLAGS.game_name, GAME_SETTINGS.get(FLAGS.game_name, {})
84 )
85
86 eta = FLAGS.eta
87 regret_steps_per_step = FLAGS.regret_steps_per_step
88
89 best_responder = FLAGS.best_responder
90 compute_ce_gap = FLAGS.compute_ce_gap
91 compute_internal_regret = FLAGS.compute_internal_regret
92
93 if FLAGS.value_estimator == "sampled":
94 value_estimator = utils.sample_value
95 elif FLAGS.value_estimator == "exact":
96 value_estimator = utils.get_exact_value
97 else:
98 raise NameError(
99 "Unknown value estimator {}. Valid names are `sampled`, `exact`."
100 .format(FLAGS.value_estimator)
101 )
102
103 if FLAGS.regret_minimizer == "hedge":
104 regret_minimizer = hedge.Hedge(
105 mfg_game,
106 [],
107 eta,
108 regret_steps_per_step,
109 compress_nus=True,
110 compress_every=FLAGS.compress_every,
111 compress_lbd=FLAGS.compress_lbd,
112 value_estimator=value_estimator,
113 value_estimation_n=FLAGS.value_estimation_n,
114 compute_internal_regret=compute_internal_regret,
115 )
116 elif FLAGS.regret_minimizer == "rm":
117 regret_minimizer = regret_matching.RegretMatching(
118 mfg_game,
119 [],
120 eta,
121 regret_steps_per_step,
122 compress_nus=True,
123 compress_every=FLAGS.compress_every,
124 compress_lbd=FLAGS.compress_lbd,
125 value_estimator=value_estimator,
126 value_estimation_n=FLAGS.value_estimation_n,
127 compute_internal_regret=compute_internal_regret,
128 )
129 elif FLAGS.regret_minimizer == "poly":
130 regret_minimizer = polynomial_weights.PolynomialWeightAlgorithm(
131 mfg_game,
132 [],
133 eta,
134 regret_steps_per_step,
135 compress_nus=True,
136 compress_every=FLAGS.compress_every,
137 compress_lbd=FLAGS.compress_lbd,
138 value_estimator=value_estimator,
139 value_estimation_n=FLAGS.value_estimation_n,
140 compute_internal_regret=compute_internal_regret,
141 )
142 else:
143 raise NameError(
144 "Unknown regret minimizer {}.".format(FLAGS.regret_minimizer)
145 )
146
147 if best_responder == "cce":
148 best_responder = correlated_equilibrium.cce_br
149 elif best_responder == "ce":
150 best_responder = correlated_equilibrium.ce_br
151 elif best_responder == "ce_partial":
152 best_responder = correlated_equilibrium.partial_ce_br
153 else:
154 raise NameError(
155 "Unknown best responder {}. Valid names are `cce` and `ce`.".format(
156 FLAGS.best_responder
157 )
158 )
159
160 mfpsro = mf_psro.MeanFieldPSRO(
161 mfg_game,
162 regret_minimizer,
163 regret_steps_per_step,
164 best_responder=best_responder,
165 )
166
167 for j in range(FLAGS.n_iter):
168 logging.info("Iteration {} of MF-PSRO".format(j)) # pylint: disable=logging-format-interpolation

178 )
179

181 cce_gap_value = correlated_equilibrium.cce_gap(
182 mfg_game, policies, rhos, mus, nus, compute_true_rewards=True
183 )
184 if compute_ce_gap:
185 print("CE Gap Computation")
186 ce_gap_value = correlated_equilibrium.ce_gap(
187 mfg_game, policies, rhos, mus, nus, compute_true_rewards=True
188 )
189 else:
190 ce_gap_value = 0.0
191
192 print("CCE Gap value : {}".format(cce_gap_value))
193 print("CE Gap value : {}".format(ce_gap_value))
194 print("Average welfare : {}".format(average_welfare))
