30 // Action-Observation histories partition the game tree in the same way
31 // as information states, but they contain more structured information.
32 // Some algorithms use this structured information for targeted traversal
33 // of the imperfect information tree.
34 //
35 // Note that in the FOG paper, Action-Observation history $s$ for player $i$
36 // at world history $h$ is defined precisely as
37 //
38 // s_i(h) := (O_i^0, a_i^0, O_i^1, a_i^1, ... O_i^{t-1}, a_i^{t-1}, O_i^t)
39 //
40 // and this can be interpreted for simultaneous-move games as
41 //
42 // Initial Observation + List of Pair((Action, Observation))
43 //
44 // However in OpenSpiel the player is not always acting, as in sequential games,
45 // but we'd like to support those as well. So we make a compromise to just have
46 // AOH as a vector of *optional* action and observation.
47 class ActionObservationHistory {
48  private:
49  // Player to which this Action-Observation history belongs.
50  const Player player_;
51 
52  // Actual Action-Observation history.
53  std::vector<std::pair<absl::optional<Action>, std::string>> history_;
54 
55  public:
56  // Constructs an Action-Observation history for a given player at the target
57  // state. This method can be called only if the game provides an
58  // implementation of ObservationString().
59  //
60  // Note that this constructor makes a traversal of the state's history
61  // to collect player's observations and this can be expensive.
62  ActionObservationHistory(Player player, const State& target);
63 
64  // Constructs an Action-Observation history for the current player
65  // at the target state.
66  ActionObservationHistory(const State& target);
67 
68  // Constructs an Action-Observation history "manually" from history vector.
69  ActionObservationHistory(
70  Player player,
71  std::vector<std::pair<absl::optional<Action>, std::string>> history);
72 
73  ActionObservationHistory(const ActionObservationHistory&) = default;
74  ~ActionObservationHistory() = default;
75 
76  const std::vector<std::pair<absl::optional<Action>, std::string>>& History()
77  const { return history_; }
78  Player GetPlayer() const { return player_; }
79 
80  // Gets the current move number (time on the clock) - this allows to relate
81  // the "depth"of Action-Observation history to the "depth" of State,
82  // as it should correspond to the State::MoveNumber().
83  int MoveNumber() const;
84 
85  // Returns the player's observation (i.e. public+private observation)
86  // at the given time. Root node has time 0.
87  const std::string& ObservationAt(int time) const;
88 
89  // Returns the action at the given time.
90  // If player was not acting at requested time, returns a nullopt.
91  absl::optional<Action> ActionAt(int time) const;
92 
93  // Does the Action-Observation history correspond to the initial state?
94  bool CorrespondsToInitialState() const { return MoveNumber() == 0; }
95 
96  // Does the Action-Observation history correspond to the other
97  // Action-Observation history? This is just like an equality operator.
98  bool CorrespondsTo(const ActionObservationHistory& other) const;
99 
100  // Does the Action-Observation history correspond to the requested state?
101  //
102  // In other words, if we constructed Action-Observation history for the state,
103  // would that correspond to this Action-Observation history?
104  //
105  // As in the following:
106  //
107  // CorrespondsTo(pl, state) == CorrespondsTo(
108  // ActionObservationHistory(pl, state))
109  //
110  // This method is provided so that you do not need to construct
111  // Action-Observation History explicitly and is more efficient.
112  // There can be state1 != state2 but AOH(state1) == AOH(state2).
113  bool CorrespondsTo(Player pl, const State& state) const;
114 
115  // Is the current Action-Observation history prefix (or equal) of the other?
116  bool IsPrefixOf(const ActionObservationHistory& other) const;
117 
118  // Is the current Action-Observation history prefix (or equal) of the
119  // Action-Observation history that we could construct from the State?
120  bool IsPrefixOf(Player pl, const State& state) const;
121 
122  // Is the current Action-Observation history extension (or equal)
123  // of the other one?
124  bool IsExtensionOf(const ActionObservationHistory& other) const;
125 
126  // Is the current Action-Observation history extension (or equal)
127  // of the Action-Observation history that we could construct from the State?
128  bool IsExtensionOf(Player pl, const State& state) const;
129 
130  void Extend(const absl::optional<Action> action,
131  const std::string& observation_string);
132 
133  void RemoveLast();
134 
135  std::string ToString() const;
136 
137  bool operator==(const ActionObservationHistory& other) const {
138  return CorrespondsTo(other);
139  }
140 
141  private:
142  bool CheckStateCorrespondenceInSimulation(Player pl, const State& state,
143  int until_time) const;
144 };

146 // Public-observation histories partition the game tree according to available
147 // public information into a corresponding public tree. Public observation
148 // history identifies the current public state (a node in the public tree),
149 // and is useful for integration with public state API -- you can construct
150 // a PublicState by using the public observation history.
151 //
152 // Some algorithms use this structured information for targeted traversal
153 // of the (im)perfect information tree.
154 class PublicObservationHistory {
155  private:
156  std::vector<std::string> history_;
157 
158  public:
159  // Construct a history of public observations.
160  // This method can be called only if the game provides factored observations
161  // strings, mainly State::PublicObservationString() -- private observations
162  // are not used.
163  //
164  // Note that this constructor makes a traversal of the state's history
165  // to collect public observations and this can be expensive.
166  PublicObservationHistory(const State& target);
167 
168  // Constructs Public-Observation history "manually".
169  PublicObservationHistory(std::vector<std::string> history);

178  // as it should correspond to the State::MoveNumber().
179  int MoveNumber() const;

182  const std::string& ObservationAt(int time) const;
183
