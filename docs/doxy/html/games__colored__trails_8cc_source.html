39 m.attr("NUM_COLORS") = py::int_(kDefaultNumColors);
40 m.attr("NUM_CHIPS_LOWER_BOUND") = py::int_(kNumChipsLowerBound);

57 .def_readonly("num_players", &Board::num_players)
58 // one-dimensional list in row-major form, contains colors of each cell
59 .def_readonly("board", &Board::board)
60 // list integers, one per player, for the number of chips they have
61 .def_readonly("num_chips", &Board::num_chips)
62 // list of lists, one per player, of the actual chips that player has
63 .def_readonly("chips", &Board::chips)
64 // list if positions of the players and the flag (the last element)
65 .def_readonly("positions", &Board::positions)
66 // arguments: (player: List[int], trade: trade)
67 .def("apply_trade", &Board::ApplyTrade)
68 // no arguments; returns a clone of this board
69 .def("clone", &Board::Clone)
70 // in_bounds(row, col); returns true/false
71 .def("in_bounds", &Board::InBounds)
72 // return a string description of the board, as in the instances file
73 .def("to_string", &Board::ToString)

75 .def("pretty_board_string", &Board::PrettyBoardString);
