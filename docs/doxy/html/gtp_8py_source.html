27class GTPBot(pyspiel.Bot):
28 """A bot that uses an external agent over GTP to get the action to play.
29
30 The Go Text Protocol, GTP, is a text based protocol for communication with
31 computer Go programs (https://www.lysator.liu.se/~gunnar/gtp/). It has also
32 been adopted by agents in other games including Hex and Havannah. If you need
33 to configure your agent in some specific way (eg time/resource limits), you
34 can use `gtp_cmd` to send raw commands to it.
35 """

37 def __init__(self, game, exec_path, player_colors=("b", "w"),
38 suppress_stderr=True):
39 """Create a Bot that runs an external binary using GTP.
40
41 Args:
42 game: A Game object to pull the configuration (boardsize)
43 exec_path: A string or list to be passed to popen to launch the binary.
44 player_colors: A list or tuple of names to be passed to gtp's `play`
45 command to tell it which player made the move.
46 suppress_stderr: Whether to suppress stderr from the binary.
47 """
48 pyspiel.Bot.__init__(self)
49 self._process = subprocess.Popen(
50 exec_path, bufsize=0, stdin=subprocess.PIPE, stdout=subprocess.PIPE,
51 stderr=(subprocess.DEVNULL if suppress_stderr else None))
52
53 self._game = game
54 params = game.get_parameters()
55 if "board_size" in params:
56 self.gtp_cmd("boardsize", str(params["board_size"]))
57
58 if len(player_colors) != game.num_players():
59 raise ValueError(
60 ("player_colors has the wrong number of players for this game. Got "
61 "{}, expected {}").format(len(player_colors), game.num_players()))
62 self._player_colors = player_colors

67 def close(self):
68 """Tell the game to quit and wait for it to do so, killing eventually."""
69 # We support closing an already closed instance, as __del__ will be called
70 # when the object is deleted, thus closing a potentially already closed obj.
71 # The hasattr is in case Popen raises and self._process doesn't exist.
72 if hasattr(self, "_process") and self._process is not None:
73 if self.runningrunning:
74 try:
75 self.gtp_cmd("quit")
76 except (CommandError, IOError):
77 pass
78 self._process.stdin.close()
79 self._process.stdout.close()
80 _shutdown_proc(self._process, 3)
81 self._process = None

83 def gtp_cmd(self, *args):
84 """Send commands directly to the game, and get the response as a string."""
85 cmd = " ".join([str(a) for a in args]).encode()
86 self._process.stdin.write(cmd + b"\n")
87 response = ""
88 while True:
89 line = self._process.stdout.readline().decode()
90 if not line:
91 raise IOError("Engine closed the connection.")
92 if line == "\n":
93 if response:
94 break # A blank line signifies end of response.
95 else:
96 continue # Ignore leading newlines, possibly left from prev response.
97 response += line
98 if response.startswith("="):
99 return response[1:].strip().lower()
100 else:
101 raise CommandError(response[1:].strip())

103 def inform_action(self, state, player_id, action):
104 """Let the bot know of the other agent's actions."""

108 def step(self, state):
109 """Returns the selected action and steps the internal state forward."""
110 return state.string_to_action(self.gtp_cmd(

113 def restart(self):

116 def restart_at(self, state):
117 self.restart()
118 new_state = self._game.new_initial_state()
119 for action in state.history():
120 self.inform_action(new_state, new_state.current_player(),
121 new_state.action_to_string(action))
122 new_state.apply_action(action)

146def _shutdown_proc(p, timeout):
147 """Waits for a proc to shut down; terminates or kills it after `timeout`."""
148 freq = 10 # how often to check per second
149 for _ in range(1 + timeout * freq):
150 p.terminate()
151 ret = p.poll()
152 if ret is not None:
153 return ret
154 time.sleep(1 / freq)
155 p.kill()
156 return p.wait()
