27const std::string GameToNFGString(const Game& game) {
28 // NFG 1 R "Selten (IJGT, 75), Figure 2, normal form"
29 // { "Player 1" "Player 2" } { 3 2 }
30 // 1 1 0 2 0 2 1 1 0 3 2 0
31 const auto* nfg = dynamic_cast<const NormalFormGame*>(&game);
32 if (nfg == nullptr) {
33 SpielFatalError("Must be a normal-form game");
34 }
35
36 int num_players = nfg->NumPlayers();
37 std::vector<std::vector<Action>> legal_actions(num_players);
38 std::unique_ptr<State> initial_state = nfg->NewInitialState();
39 for (Player player = 0; player < num_players; ++player) {
40 legal_actions[player] = initial_state->LegalActions(player);
41 }
42
43 // Line 1.
44 std::string nfg_text =
45 absl::StrCat("NFG 1 R \"OpenSpiel export of ", nfg->ToString(), "\"\n");
46
47 // Line 2.
48 absl::StrAppend(&nfg_text, "{");
49 for (Player p = 0; p < num_players; ++p) {
50 absl::StrAppend(&nfg_text, " \"Player ", p, "\"");
51 }
52 absl::StrAppend(&nfg_text, " } {");
53 for (Player p = 0; p < num_players; ++p) {
54 absl::StrAppend(&nfg_text, " ", legal_actions[p].size());
55 }
56 absl::StrAppend(&nfg_text, " }\n\n");
57
58 // Now the payoffs.
59 for (auto flat_joint_action : initial_state->LegalActions()) {
60 std::vector<double> returns =
61 initial_state->Child(flat_joint_action)->Returns();
62 for (Player p = 0; p < returns.size(); ++p) {
63 absl::StrAppendFormat(&nfg_text, "%.15g ", returns[p]);
64 }
65 absl::StripAsciiWhitespace(&nfg_text);
66 absl::StrAppend(&nfg_text, "\n");
67 }
68
69 return nfg_text;
70}
