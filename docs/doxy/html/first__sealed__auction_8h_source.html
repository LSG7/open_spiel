26// First-Price Sealed-Bid Auction:
27// https://en.wikipedia.org/wiki/First-price_sealed-bid_auction
28//
29// Each player has a valuation of the target object from 1 to K, according to a
30// uniform distribution, and places bids from 0 to (valuation - 1). The highest
31// bidder gets reward (valuation - bid); the others get 0. In the case of a
32// tie, the winner is randomly determined amongst the highest bidders.
33//
34// Parameters:
35// "max_value" int maximum valuation (default = 10)
36// "players" int number of players (default = 2)
37

77// Game object.
78class FPSBAGame : public Game {
79 public:
80 explicit FPSBAGame(const GameParameters& params);
81 int NumDistinctActions() const override { return max_value_; }
82 std::unique_ptr<State> NewInitialState() const override {
83 return std::unique_ptr<State>(new FPSBAState(shared_from_this()));
84 }
85 int MaxChanceOutcomes() const override {
86 return std::max(max_value_ + 1, num_players_);
87 }
88 int NumPlayers() const override { return num_players_; }
89 double MinUtility() const override { return 0; }
90 double MaxUtility() const override { return max_value_; }
91 int MaxGameLength() const override { return num_players_; }
92 // There is an additional chance node after all the bids to determine a winner
93 // in the case of a tie.
94 int MaxChanceNodesInHistory() const override { return num_players_ + 1; }
95 std::vector<int> InformationStateTensorShape() const override {
96 return {max_value_ * 2 + num_players_};
97 };
98 std::vector<int> ObservationTensorShape() const override {
99 return {max_value_};
100 };
101
102 private:
103 // Number of players.
104 const int num_players_;
105 // Maximum valuation, which is one more than maximum bid.
106 const int max_value_;
107};
