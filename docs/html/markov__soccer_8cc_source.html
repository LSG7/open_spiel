93 std::string MarkovSoccerState::ActionToString(Player player,
94  Action action_id) const {
95  if (player == kSimultaneousPlayerId)
96  return FlatJointActionToString(action_id);
97  SPIEL_CHECK_GE(action_id, 0);
98 
99  std::string result = "";
100  if (player == kChancePlayerId) {
101  SPIEL_CHECK_LT(action_id, game_->MaxChanceOutcomes());
102 
103  // Chance moves.
104  if (action_id == kChanceInit0Action) {
105  result = "(A's action first)";
106  } else if (action_id == kChanceInit1Action) {
107  result = "(B's action first)";
108  } else {
109  int ball_loc = action_id - kNumInitiativeChanceOutcomes;
110  return absl::StrCat("(ball at ", grid_.ball_start_points[ball_loc].first,
111  ",", grid_.ball_start_points[ball_loc].second, ")");
112  }
113  } else {
114  SPIEL_CHECK_LT(action_id, game_->NumDistinctActions());
115 
116  // Regular move actions.
117  if (action_id == kUp) {
118  result = "up";
119  } else if (action_id == kDown) {
120  result = "down";
121  } else if (action_id == kLeft) {
122  result = "left";
123  } else if (action_id == kRight) {
124  result = "right";
125  } else if (action_id == kStand) {
126  result = "stand";
127  }
128  }
129  return result;
130 }

179 void MarkovSoccerState::ResolveMove(Player player, int move) {
180  int old_row = player_row_[player - 1];
181  int old_col = player_col_[player - 1];
182  int new_row = old_row + row_offsets[move];
183  int new_col = old_col + col_offsets[move];
184 
185  char from_piece = field(old_row, old_col);
186 
187  if (!InBounds(new_row, new_col)) {
188  // Check, this is a goal? If so, set the winner.
189  if (from_piece == 'A' && (new_row == 1 || new_row == 2) &&
190  (new_col == grid_.num_cols)) {
191  SetField(old_row, old_col, '.');
192  winner_ = 0;
193  } else if (from_piece == 'B' && (new_row == 1 || new_row == 2) &&
194  (new_col == -1)) {
195  SetField(old_row, old_col, '.');
196  winner_ = 1;
197  }
198 
199  // Otherwise, nothing happens.
200  return;
201  }
202 
203  // The move was in bounds!
204  char to_piece = field(new_row, new_col);
205 
206  // Stand?
207  if (old_row == new_row && old_col == new_col) {
208  return;
209  }
210 
211  // Otherwise: something interesting.
212  if (to_piece == '.') {
213  // open field, move'em!
214  SetField(new_row, new_col, field(old_row, old_col));
215  SetField(old_row, old_col, '.');
216  } else if (to_piece == 'O') {
217  // Nice! .. got the ball, way to go; a -> A or b -> B.
218  SPIEL_CHECK_TRUE(from_piece == 'a' || from_piece == 'b');
219 
220  if (from_piece == 'a') {
221  SetField(old_row, old_col, '.');
222  SetField(new_row, new_col, 'A');
223  } else if (from_piece == 'b') {
224  SetField(old_row, old_col, '.');
225  SetField(new_row, new_col, 'B');
226  }
227  } else if (from_piece == 'A' && to_piece == 'b') {
228  // Lost of possession to defender.
229  SetField(old_row, old_col, 'a');
230  SetField(new_row, new_col, 'B');
231  } else if (from_piece == 'B' && to_piece == 'a') {
232  // Lost of possession to defender.
233  SetField(old_row, old_col, 'b');
234  SetField(new_row, new_col, 'A');
235  }
236 }

238 void MarkovSoccerState::DoApplyAction(Action action_id) {
239  if (IsSimultaneousNode()) {
240  ApplyFlatJointAction(action_id);
241  return;
242  }
243  SPIEL_CHECK_TRUE(IsChanceNode());
244  SPIEL_CHECK_GE(action_id, 0);
245  SPIEL_CHECK_LT(action_id, game_->MaxChanceOutcomes());
246 
247  if (action_id == kChanceInit0Action) {
248  ResolveMove(1, moves_[0]);
249  ResolveMove(2, moves_[1]);
250  } else if (action_id == kChanceInit1Action) {
251  ResolveMove(2, moves_[1]);
252  ResolveMove(1, moves_[0]);
253  } else {
254  int ball_loc = action_id - kNumInitiativeChanceOutcomes;
255  SetField(grid_.ball_start_points[ball_loc].first,
256  grid_.ball_start_points[ball_loc].second, 'O');
257  }
258 
259  cur_player_ = kSimultaneousPlayerId;
260  total_moves_++;
261 }

263 std::vector<Action> MarkovSoccerState::LegalActions(Player player) const {
264  if (IsTerminal()) return {};
265  if (IsChanceNode()) {
266  if (total_moves_ == 0) {
267  std::vector<Action> outcomes(grid_.ball_start_points.size(),
268  kInvalidAction);
269  for (int i = 0; i < grid_.ball_start_points.size(); ++i) {
270  outcomes[i] = kNumInitiativeChanceOutcomes + i;
271  }
272  return outcomes;
273  } else {
274  return {ToAction(ChanceOutcome::kChanceInit0),
275  ToAction(ChanceOutcome::kChanceInit1)};
276  }
277  } else {
278  return {kUp, kDown, kLeft, kRight, kStand};
279  }
280 }

282 std::vector<std::pair<Action, double>> MarkovSoccerState::ChanceOutcomes()
283  const {
284  SPIEL_CHECK_TRUE(IsChanceNode());
285  if (total_moves_ == 0) {
286  std::vector<std::pair<Action, double>> outcomes(
287  grid_.ball_start_points.size(), {kInvalidAction, -1.0});
288  const double unif_prob = 1.0 / outcomes.size();
289  for (int i = 0; i < grid_.ball_start_points.size(); ++i) {
290  outcomes[i] = {kNumInitiativeChanceOutcomes + i, unif_prob};
291  }
292  return outcomes;
293  } else {
294  return {
295  std::pair<Action, double>(ToAction(ChanceOutcome::kChanceInit0), 0.5),
296  std::pair<Action, double>(ToAction(ChanceOutcome::kChanceInit1), 0.5)};
297  }

331 int MarkovSoccerState::observation_plane(int r, int c) const {
332  int plane = -1;
333  switch (field(r, c)) {
334  case 'a':
335  plane = 0;
336  break;
337  case 'A':
338  plane = 1;
339  break;
340  case 'b':
341  plane = 2;
342  break;
343  case 'B':
344  plane = 3;
345  break;
346  case 'O':
347  plane = 4;
348  break;
349  case '.':
350  plane = 5;
351  break;
352  default:
353  std::cerr << "Invalid character on field: " << field(r, c) << std::endl;
354  plane = -1;
355  break;
356  }
357 
358  return plane;
359 }

402 Grid ParseGrid(const std::string& grid_string) {
403  Grid grid{/*num_rows=*/0, /*num_cols=*/0};
404  int row = 0;
405  int col = 0;
406  int count_empty_cells = 0;
407  bool a_set = false;
408  bool b_set = false;
409  for (auto c : grid_string) {
410  if (c == '\n') {
411  row += 1;
412  col = 0;
413  } else {
414  if (row >= grid.num_rows) grid.num_rows = row + 1;
415  if (col >= grid.num_cols) grid.num_cols = col + 1;
416  if (c == 'O') {
417  grid.ball_start_points.emplace_back(row, col);
418  } else if (c == 'A') {
419  if (a_set == true) {
420  SpielFatalError("Can only have one A in grid.");
421  }
422  grid.a_start = {row, col};
423  a_set = true;
424  } else if (c == 'B') {
425  if (b_set == true) {
426  SpielFatalError("Can only have one B in grid.");
427  }
428  grid.b_start = {row, col};
429  b_set = true;
430  } else if (c == '.') {
431  ++count_empty_cells;
432  } else {
433  SpielFatalError(absl::StrCat("Invalid char '", std::string(1, c),
434  "' at grid (", row, ",", col, ")"));
435  }
436  col += 1;
437  }
438  }
439  // Must have at least one ball starting location.
440  SPIEL_CHECK_GE(grid.ball_start_points.size(), 0);
441  SPIEL_CHECK_EQ(grid.num_rows * grid.num_cols,
442  count_empty_cells + grid.ball_start_points.size() + 2);
443  return grid;
444 }
