15"""Double Oracle algorithm.
16
17Solves two-player zero-sum games, for more information see:
18McMahan et al. (2003). Planning in the presence of cost functions controlled by
19 an adversary. In Proceedings of the 20th International Conference on Machine
20 Learning (ICML-03) (pp. 536-543).
21"""

31 """Returns the sizes of lists in a list."""

35def solve_subgame(subgame_payoffs):
36 """Solves the subgame using OpenSpiel's LP solver."""
37 p0_sol, p1_sol, _, _ = lp_solver.solve_zero_sum_matrix_game(
38 pyspiel.create_matrix_game(*subgame_payoffs))
39 p0_sol, p1_sol = np.asarray(p0_sol), np.asarray(p1_sol)
40 return [p0_sol / p0_sol.sum(), p1_sol / p1_sol.sum()]

43class DoubleOracleSolver(object):
44 """Double Oracle solver."""
45
46 def __init__(self, game, enforce_symmetry=False):
47 """Initializes the Double Oracle solver.
48
49 Args:
50 game: pyspiel.MatrixGame (zero-sum).
51 enforce_symmetry: If True, enforces symmetry in the strategies appended by
52 each player, by using the first player's best response for the second
53 player as well; also asserts the game is symmetric and that players are
54 seeded with identical initial_strategies, default: False.
55 """
56 assert isinstance(game, pyspiel.MatrixGame)
57 assert game.get_type().utility == pyspiel.GameType.Utility.ZERO_SUM
58 # convert matrix game to numpy.ndarray of shape [2,rows,columns]
59 self.payoffs = utils.game_payoffs_array(game)
60 self.subgame_strategies = [[], []]
61 self.enforce_symmetry = enforce_symmetry
62 if self.enforce_symmetry:
63 assert utils.is_symmetric_matrix_game(self.payoffs), (
64 "enforce_symmetry is True, but payoffs are asymmetric!")
65
66 def subgame_payoffs(self):
67 # Select payoffs from the full game according to the subgame strategies.
68 assert all(lens(self.subgame_strategies)), "Need > 0 strategies per player."
69 subgame_payoffs = np.copy(self.payoffs)
70 for player, indices in enumerate(self.subgame_strategies):
71 subgame_payoffs = np.take(subgame_payoffs, indices, axis=player + 1)
72 return subgame_payoffs
73
74 def oracle(self, subgame_solution):
75 """Computes the best responses.
76
77 Args:
78 subgame_solution: List of subgame solution policies.
79
80 Returns:
81 best_response: For both players from the original set of pure strategies.
82 best_response_utility: Corresponding utility for both players.
83 """
84 assert lens(subgame_solution) == lens(self.subgame_strategies), (
85 f"{lens(subgame_solution)} != {lens(self.subgame_strategies)}")
86 best_response = [None, None]
87 best_response_utility = [None, None]
88 n_best_responders = 1 if self.enforce_symmetry else 2
89 for player in range(n_best_responders):
90 opponent = 1 - player
91 # collect relevant payoff entries
92 payoffs = np.take(
93 self.payoffs[player],
94 self.subgame_strategies[opponent],
95 axis=opponent)
96 # transpose to move player to leading dimension
97 payoffs = np.transpose(payoffs, [player, opponent])
98 avg_payoffs = (payoffs @ subgame_solution[opponent]).squeeze()
99 best_response[player] = np.argmax(avg_payoffs)
100 best_response_utility[player] = avg_payoffs[best_response[player]]
101
102 if self.enforce_symmetry:
103 best_response[1] = best_response[0]
104 best_response_utility[1] = best_response_utility[0]
105
106 return best_response, best_response_utility

108 def step(self):
109 """Performs one iteration."""
110 subgame_payoffs = self.subgame_payoffs()
111 subgame_solution = solve_subgame(subgame_payoffs)
112 best_response, best_response_utility = self.oracle(subgame_solution)
113
114 # Add best responses to the subgame strategies (if not included yet).
115 self.subgame_strategies = [
116 sorted(set(strategies + [br]))
117 for strategies, br in zip(self.subgame_strategies, best_response)
118 ]
119 return best_response, best_response_utility

121 def solve_yield(self,
122 initial_strategies,
123 max_steps,
124 tolerance,
125 verbose,
126 yield_subgame=False):
127 """Solves game using Double Oracle, yielding intermediate results.
128
129 Args:
130 initial_strategies: List of pure strategies for both players, optional.
131 max_steps: Maximum number of iterations, default: 20.
132 tolerance: Stop if the estimated value of the game is below the tolerance.
133 verbose: If False, no warning is shown, default: True.
134 yield_subgame: If True, yields the subgame on each iteration. Otherwise,
135 yields the final results only, default: False.
136
137 Yields:
138 solution: Policies for both players.
139 iteration: The number of iterations performed.
140 value: Estimated value of the game.
141 """
142 if self.enforce_symmetry and initial_strategies:
143 assert np.array_equal(initial_strategies[0], initial_strategies[1]), (
144 f"Players must use same initial_strategies as symmetry is enforced."
145 f"\ninitial_strategies[0]: {initial_strategies[0]}, "
146 f"\ninitial_strategies[1]: {initial_strategies[1]}")
147
148 self.subgame_strategies = (initial_strategies if initial_strategies
149 else [[0], [0]])
150 iteration = 0
151 while iteration < max_steps:
152 if yield_subgame:
153 yield None, iteration, None, self.subgame_payoffs()
154 iteration += 1
155 last_subgame_size = lens(self.subgame_strategies)
156 _, best_response_utility = self.step()
157 value = sum(best_response_utility)
158 if abs(value) < tolerance:
159 if verbose:
160 print("Last iteration={}; value below tolerance {} < {}."
161 .format(iteration, value, tolerance))
162 break
163 if lens(self.subgame_strategies) == last_subgame_size:
164 if verbose:
165 print(
166 "Last iteration={}; no strategies added, increase tolerance={} or check subgame solver."
167 .format(iteration, tolerance))
168 break
169
170 # Compute subgame solution and return solution in original strategy space.
171 subgame_solution = solve_subgame(self.subgame_payoffs())
172 solution = [np.zeros(k) for k in self.payoffs.shape[1:]]
173 for p in range(2):
174 solution[p][self.subgame_strategies[p]] = subgame_solution[p].squeeze()
175
176 yield solution, iteration, value, self.subgame_payoffs()
177
178 def solve(self,
179 initial_strategies=None,
180 max_steps=20,
181 tolerance=5e-5,
182 verbose=True):
183 """Solves the game using Double Oracle, returning the final solution."""
184 solution, iteration, value = None, None, None
185 generator = self.solve_yield(initial_strategies, max_steps, tolerance,
186 verbose, yield_subgame=False)
187 for solution, iteration, value, _ in generator:
188 pass
189 return solution, iteration, value
