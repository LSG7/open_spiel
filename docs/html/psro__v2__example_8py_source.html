118 def init_pg_responder(sess, env):
119  """Initializes the Policy Gradient-based responder and agents."""
120  info_state_size = env.observation_spec()["info_state"][0]
121  num_actions = env.action_spec()["num_actions"]
122 
123  agent_class = rl_policy.PGPolicy
124 
125  agent_kwargs = {
126  "session": sess,
127  "info_state_size": info_state_size,
128  "num_actions": num_actions,
129  "loss_str": FLAGS.loss_str,
130  "loss_class": False,
131  "hidden_layers_sizes": [FLAGS.hidden_layer_size] * FLAGS.n_hidden_layers,
132  "batch_size": FLAGS.batch_size,
133  "entropy_cost": FLAGS.entropy_cost,
134  "critic_learning_rate": FLAGS.critic_learning_rate,
135  "pi_learning_rate": FLAGS.pi_learning_rate,
136  "num_critic_before_pi": FLAGS.num_q_before_pi,
137  "optimizer_str": FLAGS.optimizer_str
138  }
139  oracle = rl_oracle.RLOracle(
140  env,
141  agent_class,
142  agent_kwargs,
143  number_training_episodes=FLAGS.number_training_episodes,
144  self_play_proportion=FLAGS.self_play_proportion,
145  sigma=FLAGS.sigma)
146 
147  agents = [
148  agent_class( # pylint: disable=g-complex-comprehension
149  env,
150  player_id,
151  **agent_kwargs)
152  for player_id in range(FLAGS.n_players)
153  ]
154  for agent in agents:
155  agent.freeze()
156  return oracle, agents

205 def print_policy_analysis(policies, game, verbose=False):
206  """Function printing policy diversity within game's known policies.
207 
208  Warning : only works with deterministic policies.
209  Args:
210  policies: List of list of policies (One list per game player)
211  game: OpenSpiel game object.
212  verbose: Whether to print policy diversity information. (True : print)
213 
214  Returns:
215  List of list of unique policies (One list per player)
216  """
217  states_dict = get_all_states.get_all_states(game, np.infty, False, False)
218  unique_policies = []
219  for player in range(len(policies)):
220  cur_policies = policies[player]
221  cur_set = set()
222  for pol in cur_policies:
223  cur_str = ""
224  for state_str in states_dict:
225  if states_dict[state_str].current_player() == player:
226  pol_action_dict = pol(states_dict[state_str])
227  max_prob = max(list(pol_action_dict.values()))
228  max_prob_actions = [
229  a for a in pol_action_dict if pol_action_dict[a] == max_prob
230  ]
231  cur_str += "__" + state_str
232  for a in max_prob_actions:
233  cur_str += "-" + str(a)
234  cur_set.add(cur_str)
235  unique_policies.append(cur_set)
236  if verbose:
237  print("\n=====================================\nPolicy Diversity :")
238  for player, cur_set in enumerate(unique_policies):
239  print("Player {} : {} unique policies.".format(player, len(cur_set)))
240  print("")
241  return unique_policies

244 def gpsro_looper(env, oracle, agents):
245  """Initializes and executes the GPSRO training loop."""
246  sample_from_marginals = True # TODO(somidshafiei) set False for alpharank
247  training_strategy_selector = FLAGS.training_strategy_selector or strategy_selectors.probabilistic
248 
249  g_psro_solver = psro_v2.PSROSolver(
250  env.game,
251  oracle,
252  initial_policies=agents,
253  training_strategy_selector=training_strategy_selector,
254  rectifier=FLAGS.rectifier,
255  sims_per_entry=FLAGS.sims_per_entry,
256  number_policies_selected=FLAGS.number_policies_selected,
257  meta_strategy_method=FLAGS.meta_strategy_method,
258  prd_iterations=50000,
259  prd_gamma=1e-10,
260  sample_from_marginals=sample_from_marginals,
261  symmetric_game=FLAGS.symmetric_game)
262 
263  start_time = time.time()
264  for gpsro_iteration in range(FLAGS.gpsro_iterations):
265  if FLAGS.verbose:
266  print("Iteration : {}".format(gpsro_iteration))
267  print("Time so far: {}".format(time.time() - start_time))
268  g_psro_solver.iteration()
269  meta_game = g_psro_solver.get_meta_game()
270  meta_probabilities = g_psro_solver.get_meta_strategies()
271  policies = g_psro_solver.get_policies()
272 
273  if FLAGS.verbose:
274  print("Meta game : {}".format(meta_game))
275  print("Probabilities : {}".format(meta_probabilities))
276 
277  # The following lines only work for sequential games for the moment.
278  if env.game.get_type().dynamics == pyspiel.GameType.Dynamics.SEQUENTIAL:
279  aggregator = policy_aggregator.PolicyAggregator(env.game)
280  aggr_policies = aggregator.aggregate(
281  range(FLAGS.n_players), policies, meta_probabilities)
282 
283  exploitabilities, expl_per_player = exploitability.nash_conv(
284  env.game, aggr_policies, return_only_nash_conv=False)
285 
286  _ = print_policy_analysis(policies, env.game, FLAGS.verbose)
287  if FLAGS.verbose:
288  print("Exploitabilities : {}".format(exploitabilities))
289  print("Exploitabilities per player : {}".format(expl_per_player))

292 def main(argv):
293  if len(argv) > 1:
294  raise app.UsageError("Too many command-line arguments.")
295 
296  np.random.seed(FLAGS.seed)
297 
298  game = pyspiel.load_game_as_turn_based(FLAGS.game_name,
299  {"players": FLAGS.n_players})
300  env = rl_environment.Environment(game)
301 
302  # Initialize oracle and agents
303  with tf.Session() as sess:
304  if FLAGS.oracle_type == "DQN":
305  oracle, agents = init_dqn_responder(sess, env)
306  elif FLAGS.oracle_type == "PG":
307  oracle, agents = init_pg_responder(sess, env)
308  elif FLAGS.oracle_type == "BR":
309  oracle, agents = init_br_responder(env)
310  sess.run(tf.global_variables_initializer())
311  gpsro_looper(env, oracle, agents)
312 
313 if __name__ == "__main__":
314  app.run(main)
