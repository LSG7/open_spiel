24class SoftmaxPolicy(policy.Policy):
25 """Computes the softmax policy of a value function."""
26
27 def __init__(self,
28 game,
29 player_ids,
30 temperature: float,
31 state_action_value: value.ValueFunction,
32 prior_policy: Optional[policy.Policy] = None):
33 """Initializes the softmax policy.
34
35 Args:
36 game: The game to analyze.
37 player_ids: list of player ids for which this policy applies; each
38 should be in the range 0..game.num_players()-1.
39 temperature: float to scale the values (multiplied by 1/temperature).
40 state_action_value: A state-action value function.
41 prior_policy: Optional argument. Prior policy to scale the softmax
42 policy.
43 """
44 super(SoftmaxPolicy, self).__init__(game, player_ids)
45 self._state_action_value = state_action_value
46 self._prior_policy = prior_policy
47 self._temperature = temperature
48
49 def action_probabilities(self, state, player_id=None):
50 legal_actions = state.legal_actions()
51 max_q = np.max(
52 [self._state_action_value(state, action) for action in legal_actions])
53 exp_q = [
54 np.exp((self._state_action_value(state, action) - max_q) /
55 self._temperature) for action in legal_actions
56 ]
57 if self._prior_policy is not None:
58 prior_probs = self._prior_policy.action_probabilities(state)
59 exp_q = [
60 prior_probs.get(action, 0) * exp_q[i]
61 for i, action in enumerate(legal_actions)
62 ]
63 denom = sum(exp_q)
64 smax_q = exp_q if denom == 0 else exp_q / denom
65 return dict(zip(legal_actions, smax_q))
