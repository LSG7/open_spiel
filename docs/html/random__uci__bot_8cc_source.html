33 void RandomUciBot() {
34  int seed = absl::GetFlag(FLAGS_seed);
35  std::mt19937 rng(seed);
36  std::unique_ptr<State> state;
37  chess::ChessState* chess_state = nullptr;
38  std::shared_ptr<const Game> game = LoadGame("chess");
39 
40  for (std::string line; std::getline(std::cin, line);) {
41  if (line == "uci") {
42  std::cout << "uciok" << std::endl;
43  } else if (line == "isready") {
44  std::cout << "readyok" << std::endl;
45  } else if (line == "ucinewgame") {
46  state = game->NewInitialState();
47  chess_state = down_cast<chess::ChessState*>(state.get());
48  } else if (absl::StartsWith(line, "position fen ")) {
49  // This command has following syntax:
50  // position fen <FEN> moves <MOVES>
51  std::vector<std::string> tokens = absl::StrSplit(line, ' ');
52  // Build up the <FEN> which can contain spaces.
53  std::stringstream fen;
54  int pos = 2;
55  bool has_moves = false;
56  while (pos < tokens.size()) {
57  if (tokens[pos] == "moves") {
58  has_moves = true;
59  break;
60  }
61  if (pos > 2) fen << ' ';
62  fen << tokens[pos];
63  ++pos;
64  }
65 
66  state = game->NewInitialState(fen.str());
67  chess_state = down_cast<chess::ChessState*>(state.get());
68 
69  if (has_moves) {
70  while (pos < tokens.size()) {
71  Action action = chess_state->ParseMoveToAction(tokens[pos]);
72  state->ApplyAction(action);
73  ++pos;
74  }
75  }
76  } else if (absl::StartsWith(line, "go movetime ")) {
77  std::vector<Action> legal_actions = state->LegalActions();
78  int index = absl::Uniform<int>(rng, 0, legal_actions.size());
79  Action action = legal_actions[index];
80  chess::Move move = ActionToMove(action, chess_state->Board());
81  std::cout << "bestmove " << move.ToLAN() << std::endl;
82  } else if (line == "quit") {
83  return;
84  } else {
85  std::cout << "Unrecognized command: " << line << std::endl;
86  }
87  }
88 }
