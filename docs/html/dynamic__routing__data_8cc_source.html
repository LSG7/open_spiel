29 std::unique_ptr<DynamicRoutingData> DynamicRoutingData::Create(
30  DynamicRoutingDataName name) {
31  std::unique_ptr<DynamicRoutingData> data =
32  absl::make_unique<DynamicRoutingData>();
33  switch (name) {
34  case DynamicRoutingDataName::kLine: {
35  absl::flat_hash_map<std::string, std::vector<std::string>>
36  adjacency_list = {{"bef_O", {"O"}},
37  {"O", {"A"}},
38  {"A", {"D"}},
39  {"D", {"aft_D"}},
40  {"aft_D", {}}};
41  data->network_ = Network::Create(adjacency_list);
42  data->od_demand_ =
43  absl::make_unique<std::vector<OriginDestinationDemand>>(std::vector{
44  OriginDestinationDemand("bef_O->O", "D->aft_D", 0, 100)});
45  return data;
46  }
47  case DynamicRoutingDataName::kBraess: {
48  const int kBraessNumPlayer = 5;
49  absl::flat_hash_map<std::string, std::vector<std::string>>
50  adjacency_list = {{"O", {"A"}}, {"A", {"B", "C"}}, {"B", {"C", "D"}},
51  {"C", {"D"}}, {"D", {"E"}}, {"E", {}}};
52  absl::flat_hash_map<std::string, std::pair<float, float>> node_position =
53  {{"O", {0, 0}}, {"A", {1, 0}}, {"B", {2, 1}},
54  {"C", {2, -1}}, {"D", {3, 0}}, {"E", {4, 0}}};
55  absl::flat_hash_map<std::string, float> bpr_a_coefficient = {
56  {"O->A", 0}, {"A->B", 1.0}, {"A->C", 0}, {"B->C", 0},
57  {"B->D", 0}, {"C->D", 1.0}, {"D->E", 0}};
58  absl::flat_hash_map<std::string, float> bpr_b_coefficient = {
59  {"O->A", 1.0}, {"A->B", 1.0}, {"A->C", 1.0}, {"B->C", 1.0},
60  {"B->D", 1.0}, {"C->D", 1.0}, {"D->E", 1.0}};
61  absl::flat_hash_map<std::string, float> capacity = {
62  {"O->A", kBraessNumPlayer}, {"A->B", kBraessNumPlayer},
63  {"A->C", kBraessNumPlayer}, {"B->C", kBraessNumPlayer},
64  {"B->D", kBraessNumPlayer}, {"C->D", kBraessNumPlayer},
65  {"D->E", kBraessNumPlayer}};
66  absl::flat_hash_map<std::string, float> free_flow_travel_time = {
67  {"O->A", 0}, {"A->B", 1.0}, {"A->C", 2.0}, {"B->C", 0.25},
68  {"B->D", 2.0}, {"C->D", 1.0}, {"D->E", 0}};
69  data->network_ =
70  Network::Create(adjacency_list, node_position, bpr_a_coefficient,
71  bpr_b_coefficient, capacity, free_flow_travel_time);
72  data->od_demand_ =
73  absl::make_unique<std::vector<OriginDestinationDemand>>(std::vector{
74  OriginDestinationDemand("O->A", "D->E", 0, kBraessNumPlayer)});
75  return data;
76  }
77  default:
78  open_spiel::SpielFatalError(
79  absl::StrCat("Unknown Dynamic Routing Data Name: ", name));
80  }
81  return data;
82 }
