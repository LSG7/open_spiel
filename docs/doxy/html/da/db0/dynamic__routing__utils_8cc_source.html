37template <typename KeyType, typename ValueType>
38absl::flat_hash_set<KeyType> GetKeySet(
39 const absl::flat_hash_map<KeyType, ValueType>& m) {
40 absl::flat_hash_set<std::string> keys;
41 for (const auto& pair : m) {
42 keys.emplace(pair.first);
43 }
44 return keys;
45}

47absl::flat_hash_map<std::string, float> AssignExistingOrDefaultValues(
48 absl::flat_hash_map<std::string, float> dict_object,
49 absl::flat_hash_set<std::string> road_sections, float default_value) {
50 if (!dict_object.empty()) {
51 SPIEL_CHECK_TRUE((GetKeySet<std::string, float>(dict_object)) ==
52 road_sections);
53 return dict_object;
54 }
55 absl::flat_hash_map<std::string, float> dict_object_returned;
56 for (const auto& key : road_sections) {
57 dict_object_returned.emplace(key, default_value);
58 }
59 return dict_object_returned;
60}

86Network::Network(
87 absl::flat_hash_map<std::string, std::vector<std::string>> adjacency_list,
88 absl::flat_hash_map<std::string, std::pair<float, float>> node_position,
89 absl::flat_hash_map<std::string, float> bpr_a_coefficient,
90 absl::flat_hash_map<std::string, float> bpr_b_coefficient,
91 absl::flat_hash_map<std::string, float> capacity,
92 absl::flat_hash_map<std::string, float> free_flow_travel_time) {
93 adjacency_list_ = adjacency_list;
94 // Sort the adjacency list to make the action id unique.
95 absl::btree_map<std::string, std::vector<std::string>> sorted_adjacency_list;
96 sorted_adjacency_list.insert(adjacency_list.begin(), adjacency_list.end());
97 action_by_road_section_.clear();
98 road_section_by_action.clear();
99 road_section_by_action.emplace_back(""); // Dummy road section at index 0.
100 int action_number = kNoPossibleAction + 1;
101 for (auto& [origin, successors] : sorted_adjacency_list) {
102 std::sort(successors.begin(), successors.end());
103 for (const auto& destination : successors) {
104 std::string road_section = RoadSectionFromNodes(origin, destination);
105 SPIEL_CHECK_FALSE(action_by_road_section_.contains(road_section));
106 action_by_road_section_.emplace(road_section, action_number);
107 road_section_by_action.emplace_back(road_section);
108 // Adds road_section with no successors to sink_road_sections_;
109 if (sorted_adjacency_list.at(destination).empty()) {
110 sink_road_sections_.emplace(road_section);
111 }
112 action_number++;
113 }
114 }
115 node_position_ = node_position;
116 absl::flat_hash_set<std::string> road_sections =
117 GetKeySet<std::string, int>(action_by_road_section_);
118 bpr_a_coefficient_ =
119 AssignExistingOrDefaultValues(bpr_a_coefficient, road_sections, 0);
120 bpr_b_coefficient_ =
121 AssignExistingOrDefaultValues(bpr_b_coefficient, road_sections, 1);
122 capacity_ = AssignExistingOrDefaultValues(capacity, road_sections, 1);
123 free_flow_travel_time_ =
124 AssignExistingOrDefaultValues(free_flow_travel_time, road_sections, 1);
125}

127float Network::GetTravelTime(absl::string_view road_section,
128 float volume) const {
129 SPIEL_CHECK_TRUE(free_flow_travel_time_.contains(road_section));
130 SPIEL_CHECK_TRUE(bpr_a_coefficient_.contains(road_section));
131 SPIEL_CHECK_TRUE(bpr_b_coefficient_.contains(road_section));
132 SPIEL_CHECK_TRUE(capacity_.contains(road_section));
133
134 float free_flow_travel_time = free_flow_travel_time_.at(road_section);
135 float a = bpr_a_coefficient_.at(road_section);
136 float b = bpr_b_coefficient_.at(road_section);
137 float capacity = capacity_.at(road_section);
138 return free_flow_travel_time * (1.0 + a * pow(volume / capacity, b));
139}

165int Network::GetRoadSectionAsInt(std::string section) const {
166 if (section.empty()) {
167 return 0;
168 }
169 std::vector<std::string> nodes = NodesFromRoadSection(section);
170 std::string start_node = nodes[0];
171 std::string end_node = nodes[1];
172 return GetActionIdFromMovement(start_node, end_node);
173}

175void Network::AssertValidAction(int action, std::string road_section) const {
176 SPIEL_CHECK_GE(action, 1);
177 SPIEL_CHECK_LT(action, num_actions());
178 if (!road_section.empty()) {
179 std::string new_road_section = GetRoadSectionFromActionId(action);
180 std::vector<std::string> nodes = NodesFromRoadSection(new_road_section);
181 std::string origin_new_section = nodes[0];
182 std::string end_new_section = nodes[1];
183 std::string end_section_node = NodesFromRoadSection(road_section)[1];
184 SPIEL_CHECK_EQ(end_section_node, origin_new_section);
185 std::vector<std::string> successors = GetSuccessors(origin_new_section);
186 SPIEL_CHECK_TRUE(std::find(successors.begin(), successors.end(),
187 end_new_section) != successors.end());
188 }
189}
