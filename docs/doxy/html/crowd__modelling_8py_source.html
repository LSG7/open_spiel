55class MFGCrowdModellingGame(pyspiel.Game):
56 """A Mean Field Crowd Modelling game.
57
58
59 A game starts by an initial chance node that select the initial state
60 of the MFG.
61 Then the game sequentially alternates between:
62 - An action selection node (Where the player Id >= 0)
63 - A chance node (the player id is pyspiel.PlayerId.CHANCE)
64 - A Mean Field node (the player id is pyspiel.PlayerId.MEAN_FIELD)
65 """
66
67 # pylint:disable=dangerous-default-value
68 def __init__(self, params: Mapping[str, Any] = _DEFAULT_PARAMS):
69 game_info = pyspiel.GameInfo(
70 num_distinct_actions=_NUM_ACTIONS,
71 max_chance_outcomes=max(params["size"], _NUM_CHANCE),
72 num_players=_NUM_PLAYERS,
73 min_utility=-np.inf,
74 max_utility=+np.inf,
75 utility_sum=None,
76 max_game_length=params["horizon"])
77 super().__init__(_GAME_TYPE, game_info, params)
78 self.size = params["size"]
79 self.horizon = params["horizon"]
80
81 def new_initial_state(self):
82 """Returns a state corresponding to the start of a game."""
83 return MFGCrowdModellingState(self)
84
85 def make_py_observer(self, iig_obs_type=None, params=None):
86 """Returns an object used for observing game state."""
87 if ((iig_obs_type is None) or
88 (iig_obs_type.public_info and not iig_obs_type.perfect_recall)):
89 return Observer(params, self)
90 return observation.IIGObserverForPublicInfoGame(iig_obs_type, params)
91
92 def max_chance_nodes_in_history(self):
93 """Maximun chance nodes in game history."""
94 return self.horizon + 1

97class MFGCrowdModellingState(pyspiel.State):
98 """A Mean Field Crowd Modelling state."""
99
100 # Maps legal actions to the corresponding move along the 1-D axis of the game.
101 _ACTION_TO_MOVE = {0: -1, 1: 0, 2: 1}
102 # Action that corresponds to no displacement.
103 _NEUTRAL_ACTION = 1
104
105 def __init__(self, game):
106 """Constructor; should only be called by Game.new_initial_state."""
107 super().__init__(game)
108 self._is_chance_init = True # is true for the first state of the game.
109 self._player_id = pyspiel.PlayerId.CHANCE
110 self._x = None
111 self._t = 0
112 # We initialize last_action to the neutral action. This makes sure
113 # that the first reward does not include any displacement penalty.
114 self._last_action = self._NEUTRAL_ACTION
115 self.size = game.size
116 self.horizon = game.horizon
117 self.return_value = 0.0
118
119 # Represents the current probability distribution over game states.
120 # Initialized with a uniform distribution.
121 self._distribution = [1. / self.size for _ in range(self.size)]
122
123 @property
124 def x(self):
125 return self._x
126
127 @property
128 def t(self):
129 return self._t
130
131 def state_to_str(self, x, t, player_id=pyspiel.PlayerId.DEFAULT_PLAYER_ID):
132 """A string that uniquely identify a triplet x, t, player_id."""
133 if self._is_chance_init:
134 return "initial"
135 if player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
136 return str((x, t))
137 if player_id == pyspiel.PlayerId.MEAN_FIELD:
138 return str((x, t)) + "_a"
139 if player_id == pyspiel.PlayerId.CHANCE:
140 return str((x, t)) + "_a_mu"
141 raise ValueError(
142 "player_id is not mean field, chance or default player id.")
143
144 # OpenSpiel (PySpiel) API functions are below. This is the standard set that
145 # should be implemented by every perfect-information sequential-move game.
146
147 def _legal_actions(self, player):
148 """Returns a list of legal actions for player and MFG nodes."""
149 if player == pyspiel.PlayerId.MEAN_FIELD:
150 return []
151 if (player == pyspiel.PlayerId.DEFAULT_PLAYER_ID
152 and player == self.current_player()):
153 return [0, 1, 2]
154 raise ValueError(f"Unexpected player {player}. "
155 "Expected a mean field or current player 0.")
156
157 def chance_outcomes(self):
158 """Returns the possible chance outcomes and their probabilities."""
159 if self._is_chance_init:
160 return list(enumerate(self._distribution))
161 return [(0, 1. / 3.), (1, 1. / 3.), (2, 1. / 3.)]
162
163 def _apply_action(self, action):
164 """Applies the specified action to the state."""
165 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
166 raise ValueError(
167 "_apply_action should not be called at a MEAN_FIELD state.")
168 self.return_value += self._rewards()
169 if self._is_chance_init:
170 # Here the action is between 0 and self.size - 1
171 if action < 0 or action >= self.size:
172 raise ValueError(
173 "The action is between 0 and self.size - 1 at an init chance node")
174 self._x = action
175 self._is_chance_init = False
176 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
177 elif self._player_id == pyspiel.PlayerId.CHANCE:
178 # Here the action is between 0 and 2
179 if action < 0 or action > 2:
180 raise ValueError(
181 "The action is between 0 and 2 at any chance node")
182 self._x = (self.xx + self._ACTION_TO_MOVE[action]) % self.size
183 self._t += 1
184 self._player_id = pyspiel.PlayerId.MEAN_FIELD
185 elif self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
186 # Here the action is between 0 and 2
187 if action < 0 or action > 2:
188 raise ValueError(
189 "The action is between 0 and 2 at any chance node")
190 self._x = (self.xx + self._ACTION_TO_MOVE[action]) % self.size
191 self._last_action = action
192 self._player_id = pyspiel.PlayerId.CHANCE
193
194 def _action_to_string(self, player, action):
195 """Action -> string."""
196 del player
197 if self.is_chance_node() and self._is_chance_init:
198 return f"init_state={action}"
199 return str(self._ACTION_TO_MOVE[action])
200
201 def distribution_support(self):
202 """return a list of state string."""
203 return [
204 self.state_to_str(
205 i, self.tt, player_id=pyspiel.PlayerId.MEAN_FIELD)
206 for i in range(self.size)
207 ]
208
209 def update_distribution(self, distribution):
210 """This function is central and specific to the logic of the MFG.
211
212 Args:
213 distribution: a distribution to register.
214
215 - function should be called when the node is in MEAN_FIELD state.
216 - distribution are probabilities that correspond to each game state
217 given by distribution_support.
218
219 """
220 if self._player_id != pyspiel.PlayerId.MEAN_FIELD:
221 raise ValueError(
222 "update_distribution should only be called at a MEAN_FIELD state.")
223 self._distribution = distribution.copy()
224 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
225
226 def is_terminal(self):
227 """Returns True if the game is over."""
228 return self.tt >= self.horizon
229
230 def current_player(self):
231 """Returns id of the next player to move, or TERMINAL if game is over."""
232 if self.is_terminal():
233 return pyspiel.PlayerId.TERMINAL
234 return self._player_id
235
236 def _rewards(self):
237 """Reward for the player for this state."""
238 if self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
239 r_x = 1 - (1.0 * np.abs(self.xx - self.size // 2)) / (self.size // 2)
240 r_a = -(1.0 * np.abs(self._ACTION_TO_MOVE[self._last_action])) / self.size
241 r_mu = - np.log(self._distribution[self.xx] + _EPSILON)
242 return r_x + r_a + r_mu
243 return 0.0
244
245 def rewards(self) -> List[float]:
246 """Rewards for all players."""
247 # For now, only single-population (single-player) mean field games
248 # are supported.
249 return [self._rewards()]
250
251 def _returns(self):
252 """Returns is the sum of all payoffs collected so far."""
253 return self.return_value + self._rewards()
254
255 def returns(self) -> List[float]:
256 """Returns for all players."""
257 # For now, only single-population (single-player) mean field games
258 # are supported.
259 return [self._returns()]
260
261 def __str__(self):
262 """A string that uniquely identify the current state."""
263 return self.state_to_str(self.xx, self.tt, player_id=self._player_id)

266class Observer:
267 """Observer, conforming to the PyObserver interface (see observation.py)."""
268
269 def __init__(self, params, game):
270 """Initializes an empty observation tensor."""
271 del params
272
273 self.size = game.size
274 self.horizon = game.horizon
275 # +1 to allow t == horizon.
276 self.tensor = np.zeros(self.size + self.horizon + 1, np.float32)
277 self.dict = {"x": self.tensor[:self.size], "t": self.tensor[self.size:]}
278
279 def set_from(self, state: MFGCrowdModellingState, player: int):
280 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
281 del player
282 # We update the observation via the shaped tensor since indexing is more
283 # convenient than with the 1-D tensor. Both are views onto the same memory.
284 self.tensor.fill(0)
285 # state.x is None for the initial (blank) state, don't set any
286 # position bit in that case.
287 if state.x is not None:
288 if not 0 <= state.x < self.size:
289 raise ValueError(
290 f"Expected {state} x position to be in [0, {self.size})")
291 self.dict["x"][state.x] = 1
292 if not 0 <= state.t <= self.horizon:
293 raise ValueError(f"Expected {state} time to be in [0, {self.horizon}]")
294 self.dict["t"][state.t] = 1
295
296 def string_from(self, state, player):
297 """Observation of `state` from the PoV of `player`, as a string."""
298 del player
299 return str(state)
