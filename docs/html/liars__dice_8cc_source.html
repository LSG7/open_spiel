158 void LiarsDiceState::ResolveWinner() {
159  const std::pair<int, int> bid = UnrankBid(current_bid_);
160  int quantity = bid.first, face = bid.second;
161  int matches = 0;
162 
163  // Count all the matches among all dice from all the players
164  // dice_sides_ (e.g. 6) is wild, so it always matches.
165  for (auto p = Player{0}; p < num_players_; p++) {
166  for (int d = 0; d < num_dice_[p]; d++) {
167  if (dice_outcomes_[p][d] == face ||
168  dice_outcomes_[p][d] == dice_sides()) {
169  matches++;
170  }
171  }
172  }
173 
174  // If the number of matches are at least the quantity bid, then the bidder
175  // wins. Otherwise, the caller wins.
176  if (matches >= quantity) {
177  winner_ = bidding_player_;
178  loser_ = calling_player_;
179  } else {
180  winner_ = calling_player_;
181  loser_ = bidding_player_;
182  }
183 }

193 void LiarsDiceState::DoApplyAction(Action action) {
194  if (IsChanceNode()) {
195  // Fill the next die roll for the current roller.
196  SPIEL_CHECK_GE(cur_roller_, 0);
197  SPIEL_CHECK_LT(cur_roller_, num_players_);
198 
199  SPIEL_CHECK_LT(num_dice_rolled_[cur_roller_], num_dice_[cur_roller_]);
200  int slot = num_dice_rolled_[cur_roller_];
201 
202  // Assign the roll.
203  dice_outcomes_[cur_roller_][slot] = action + 1;
204  num_dice_rolled_[cur_roller_]++;
205 
206  // Check to see if we must change the roller.
207  if (num_dice_rolled_[cur_roller_] == num_dice_[cur_roller_]) {
208  cur_roller_++;
209  if (cur_roller_ >= num_players_) {
210  // Time to start playing!
211  cur_player_ = 0;
212  // Sort all players' rolls
213  for (auto p = Player{0}; p < num_players_; p++) {
214  std::sort(dice_outcomes_[p].begin(), dice_outcomes_[p].end());
215  }
216  }
217  }
218  } else {
219  // Check for legal actions.
220  if (!bidseq_.empty() && action <= bidseq_.back()) {
221  SpielFatalError(absl::StrCat("Illegal action. ", action,
222  " should be strictly higher than ",
223  bidseq_.back()));
224  }
225  if (action == total_num_dice_ * dice_sides()) {
226  // This was the calling bid, game is over.
227  bidseq_.push_back(action);
228  calling_player_ = cur_player_;
229  ResolveWinner();
230  } else {
231  // Up the bid and move to the next player.
232  bidseq_.push_back(action);
233  current_bid_ = action;
234  bidding_player_ = cur_player_;
235  cur_player_ = NextPlayerRoundRobin(cur_player_, num_players_);
236  }
237 
238  total_moves_++;
239  }
240 }

242 std::vector<Action> LiarsDiceState::LegalActions() const {
243  if (IsTerminal()) return {};
244  // A chance node is a single die roll.
245  if (IsChanceNode()) {
246  return LegalChanceOutcomes();
247  }
248 
249  std::vector<Action> actions;
250 
251  // Any move higher than the current bid is allowed. (Bids start at 0)
252  for (int b = current_bid_ + 1; b < total_num_dice_ * dice_sides(); b++) {
253  actions.push_back(b);
254  }
255 
256  // Calling Liar is only available if at least one move has been made.
257  if (total_moves_ > 0) {
258  actions.push_back(total_num_dice_ * dice_sides());
259  }
260 
261  return actions;
262 }

264 std::vector<std::pair<Action, double>> LiarsDiceState::ChanceOutcomes() const {
265  SPIEL_CHECK_TRUE(IsChanceNode());
266 
267  std::vector<std::pair<Action, double>> outcomes;
268 
269  // A chance node is a single die roll.
270  outcomes.reserve(dice_sides());
271  for (int i = 0; i < dice_sides(); i++) {
272  outcomes.emplace_back(i, 1.0 / dice_sides());
273  }
274 
275  return outcomes;
276 }

336 void LiarsDiceState::InformationStateTensor(Player player,
337  absl::Span<float> values) const {
338  SPIEL_CHECK_GE(player, 0);
339  SPIEL_CHECK_LT(player, num_players_);
340 
341  // One-hot encoding for player number.
342  // One-hot encoding for each die (max_dice_per_player_ * sides).
343  // One slot(bit) for each legal bid.
344  // One slot(bit) for calling liar. (Necessary because observations and
345  // information states need to be defined at terminals)
346  int offset = 0;
347  std::fill(values.begin(), values.end(), 0.);
348  SPIEL_CHECK_EQ(values.size(), num_players_ +
349  (max_dice_per_player_ * dice_sides()) +
350  (total_num_dice_ * dice_sides()) + 1);
351  values[player] = 1;
352  offset += num_players_;
353 
354  int my_num_dice = num_dice_[player];
355 
356  for (int d = 0; d < my_num_dice; d++) {
357  int outcome = dice_outcomes_[player][d];
358  if (outcome != kInvalidOutcome) {
359  SPIEL_CHECK_GE(outcome, 1);
360  SPIEL_CHECK_LE(outcome, dice_sides());
361  values[offset + (outcome - 1)] = 1;
362  }
363  offset += dice_sides();
364  }
365 
366  // Skip to bidding part. If current player has fewer dice than the other
367  // players, all the remaining entries are 0 for those dice.
368  offset = num_players_ + max_dice_per_player_ * dice_sides();
369 
370  for (int b = 0; b < bidseq_.size(); b++) {
371  SPIEL_CHECK_GE(bidseq_[b], 0);
372  SPIEL_CHECK_LE(bidseq_[b], total_num_dice_ * dice_sides());
373  values[offset + bidseq_[b]] = 1;
374  }
375 }

377 void LiarsDiceState::ObservationTensor(Player player,
378  absl::Span<float> values) const {
379  SPIEL_CHECK_GE(player, 0);
380  SPIEL_CHECK_LT(player, num_players_);
381 
382  // One-hot encoding for player number.
383  // One-hot encoding for each die (max_dice_per_player_ * sides).
384  // One slot(bit) for the two last legal bid.
385  // One slot(bit) for calling liar. (Necessary because observations and
386  // information states need to be defined at terminals)
387  int offset = 0;
388  std::fill(values.begin(), values.end(), 0.);
389  SPIEL_CHECK_EQ(values.size(), num_players_ +
390  (max_dice_per_player_ * dice_sides()) +
391  (total_num_dice_ * dice_sides()) + 1);
392  values[player] = 1;
393  offset += num_players_;
394 
395  int my_num_dice = num_dice_[player];
396 
397  for (int d = 0; d < my_num_dice; d++) {
398  int outcome = dice_outcomes_[player][d];
399  if (outcome != kInvalidOutcome) {
400  SPIEL_CHECK_GE(outcome, 1);
401  SPIEL_CHECK_LE(outcome, dice_sides());
402  values[offset + (outcome - 1)] = 1;
403  }
404  offset += dice_sides();
405  }
406 
407  // Skip to bidding part. If current player has fewer dice than the other
408  // players, all the remaining entries are 0 for those dice.
409  offset = num_players_ + max_dice_per_player_ * dice_sides();
410 
411  // We only show the num_players_ last bids
412  int size_bid = bidseq_.size();
413  int bid_offset = std::max(0, size_bid - num_players_);
414  for (int b = bid_offset; b < size_bid; b++) {
415  SPIEL_CHECK_GE(bidseq_[b], 0);
416  SPIEL_CHECK_LE(bidseq_[b], total_num_dice_ * dice_sides());
417  values[offset + bidseq_[b]] = 1;
418  }
419 }

425 std::pair<int, int> LiarsDiceState::UnrankBid(int bidnum) const {
426  std::pair<int, int> bid;
427  SPIEL_CHECK_NE(bidnum, kInvalidBid);
428  SPIEL_CHECK_GE(bidnum, 0);
429  SPIEL_CHECK_LT(bidnum, dice_sides() * total_num_dice_);
430 
431  if (bidding_rule() == BiddingRule::kResetFace) {
432  // Bids have the form <quantity>-<face>
433  //
434  // So, in a two-player game where each die has 6 faces, we have
435  //
436  // Bid ID Quantity Face
437  // 0 1 1
438  // 1 1 2
439  // ...
440  // 5 1 6
441  // 6 2 1
442  // ...
443  // 11 2 6
444  //
445  // Bid ID #dice * #num faces encodes the special "liar" action.
446 
447  // The quantity occupies the higher bits, so it can be extracted using an
448  // integer division operation.
449  bid.first = bidnum / dice_sides() + 1;
450  // The face occupies the lower bits, so it can be extraced using a modulo
451  // operation.
452  bid.second = 1 + (bidnum % dice_sides());
453  } else {
454  SPIEL_CHECK_EQ(bidding_rule(), BiddingRule::kResetQuantity);
455  // Bids have the form <face>-<quantity>
456  //
457  // So, in a two-player game where each die has 6 faces, we have
458  //
459  // Bid ID Quantity Face
460  // 0 1 1
461  // 1 2 1
462  // 2 1 2
463  // 3 2 2
464  // ...
465  // 9 2 5
466  // 10 1 6
467  // 11 2 6
468  //
469  // Bid ID #dice * #num faces encodes the special "liar" action.
470  //
471  // This particular encoding scheme allows for very cheap comparison of bids:
472  // a bid is stronger if it is encoded to a higher ID.
473 
474  // The quantity occupies the lower bits, so it can be extracted using a
475  // modulo operation.
476  bid.first = 1 + (bidnum % total_num_dice_);
477  // The face occupies the higher bits, so it can be extracted using an
478  // integer division.
479  bid.second = bidnum / total_num_dice_ + 1;
480  }
481 
482  SPIEL_CHECK_GE(bid.first, 1);
483  // It doesn't make sense to bid more dice than the number of dice in the game.
484  SPIEL_CHECK_LE(bid.first, total_num_dice_);
485 
486  SPIEL_CHECK_GE(bid.second, 1);
487  // It doesn't make sense to bid a face that does not exist.
488  SPIEL_CHECK_LE(bid.second, dice_sides());
489 
490  return bid;
491 }

493 LiarsDiceGame::LiarsDiceGame(const GameParameters& params, GameType game_type)
494  : Game(game_type, params),
495  num_players_(ParameterValue<int>("players")),
496  dice_sides_(ParameterValue<int>("dice_sides")),
497  bidding_rule_(ParseBiddingRule(
498  ParameterValue<std::string>("bidding_rule"))) {
499  SPIEL_CHECK_GE(num_players_, kGameType.min_num_players);
500  SPIEL_CHECK_LE(num_players_, kGameType.max_num_players);
501  SPIEL_CHECK_GE(dice_sides_, 1);
502 
503  int def_num_dice = ParameterValue<int>("numdice");
504 
505  // Compute the number of dice for each player based on parameters,
506  // and set default outcomes of unknown face values (-1).
507  total_num_dice_ = 0;
508  num_dice_.resize(num_players_, 0);
509 
510  for (auto p = Player{0}; p < num_players_; p++) {
511  std::string key = absl::StrCat("numdice", p);
512 
513  int my_num_dice = def_num_dice;
514  if (IsParameterSpecified(game_parameters_, key)) {
515  my_num_dice = ParameterValue<int>(key);
516  }
517 
518  num_dice_[p] = my_num_dice;
519  total_num_dice_ += my_num_dice;
520  }
521 
522  // Compute max dice per player (used for observations.)
523  max_dice_per_player_ = -1;
524  for (int nd : num_dice_) {
525  if (nd > max_dice_per_player_) {
526  max_dice_per_player_ = nd;
527  }
528  }
529 }

585 std::string ImperfectRecallLiarsDiceState::InformationStateString(
586  Player player) const {
587  SPIEL_CHECK_GE(player, 0);
588  SPIEL_CHECK_LT(player, num_players_);
589  const auto* parent_game = down_cast<const ImperfectRecallLiarsDiceGame*>(
590  game_.get());
591 
592  std::string result =
593  absl::StrCat("P", player, " ", absl::StrJoin(dice_outcomes_[player], ""));
594 
595  // Imperfect recall. Show only the last recall_length bids.
596  int start_index = std::max<int>(bidseq_.size() - parent_game->recall_length(),
597  0);
598  for (int b = start_index; b < bidseq_.size(); b++) {
599  if (bidseq_[b] == parent_game->total_num_dice() * dice_sides()) {
600  absl::StrAppend(&result, " Liar");
601  } else {
602  const std::pair<int, int> bid = UnrankBid(bidseq_[b]);
603  absl::StrAppend(&result, " ", bid.first, "-", bid.second);
604  }
605  }
606  return result;
607 }
