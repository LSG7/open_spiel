25 int64_t NumDeterministicPolicies(const Game& game, Player player) {
26  int64_t num_policies = 1;
27  std::unordered_map<std::string, std::vector<Action>> legal_actions_map =
28  GetLegalActionsMap(game, -1, player);
29  for (const auto& infostate_str_actions : legal_actions_map) {
30  int64_t num_actions = infostate_str_actions.second.size();
31  SPIEL_CHECK_GT(num_actions, 0);
32 
33  // Check for integer overflow.
34  if (num_policies > INT64_MAX / num_actions) {
35  return -1;
36  }
37 
38  num_policies *= num_actions;
39  }
40  return num_policies;
41 }

93 bool DeterministicTabularPolicy::NextPolicy() {
94  // Treat the current indices as digits in a mixed base. Starting at the
95  // beginning of the table, add 1. If can't, continue trying. If we reach the
96  // end without being able to add 1, then this is the end of the order.
97  // Otherwise, increment the digit we land on by 1, and reset all the ones
98  // we skipped over earlier in the order.
99  for (auto iter = table_.begin(); iter != table_.end(); ++iter) {
100  if (iter->second.TryIncIndex()) {
101  for (auto iter2 = table_.begin(); iter2 != iter; ++iter2) {
102  iter2->second.index = 0;
103  }
104  return true;
105  }
106  }
107  return false;
108 }
