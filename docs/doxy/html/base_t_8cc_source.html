431 cell_observed += "E";
432 }

434 if (s.cells_v[crd.z][crd.y][crd.x].being_observed_by[p]) // p 에 의해 관찰되는 중

437 {
438 cell_observed += std::to_string(s.cells_v[crd.z][crd.y][crd.x].occupying_player);
439 cell_observed += std::to_string(s.cells_v[crd.z][crd.y][crd.x].occupying_unit_id);
440 }
441 }
442 else // 관찰 안되는 중
443 {
444 cell_observed += "?";
445 }
446
447 // 빈자리 채우기
448 if (max_units < 10) // GPI = 3문자

450 cell_observed.insert(cell_observed.cend(), (3 - cell_observed.length()), ' ');

453 {
454 cell_observed.insert(cell_observed.cend(), (4 - cell_observed.length()), ' ');
455 }
456

492 {

802
803 // Op player's public info
