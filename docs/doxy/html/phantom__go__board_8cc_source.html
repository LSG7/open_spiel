30// 8 adjacent directions.
31//
32// 405
33// 1 2
34// 637
35//
36// The order is important because it is used to index 3x3 patterns!
37//
38inline constexpr std::array<int, 9> Dir8 = {{
39 kVirtualBoardSize, // new line
40 -1, // new line
41 +1, // new line
42 -static_cast<int>(kVirtualBoardSize),
43 +static_cast<int>(kVirtualBoardSize) - 1,
44 +static_cast<int>(kVirtualBoardSize) + 1,
45 -static_cast<int>(kVirtualBoardSize) - 1,
46 -static_cast<int>(kVirtualBoardSize) + 1,
47 0 // Dummy element.
48}};

50// Calls f for all 4 direct neighbours of p.
51// f should have type void f(VirtualPoint n), but is passed as a template so we
52// can elide the function call overhead.
53template <typename F>
54void Neighbours(VirtualPoint p, const F &f) {
55 f(p + kVirtualBoardSize);
56 f(p + 1);
57 f(p - 1);
58 f(p - kVirtualBoardSize);
59}

61std::vector<VirtualPoint> MakeBoardPoints(int board_size) {
62 std::vector<VirtualPoint> points;
63 points.reserve(board_size * board_size);
64 for (int row = 0; row < board_size; ++row) {
65 for (int col = 0; col < board_size; ++col) {
66 points.push_back(VirtualPointFrom2DPoint({row, col}));
67 }
68 }
69 return points;
70}

94std::string MoveAsAscii(VirtualPoint p, GoColor c) {
95 static std::string code = "0123456789abcdefghijklmnopqrstuvwxyz";
96 static int mask = 31;
97 // 1 bit for color, 9 bits for the point.
98 uint16_t value = static_cast<int>(c) | (p << 1);
99 // Encode in 2 characters of 5 bit each.
100 std::string encoded;
101 encoded.push_back(code[(value >> 5) & mask]);
102 encoded.push_back(code[value & mask]);
103 return encoded;
104}

133std::pair<int, int> VirtualPointTo2DPoint(VirtualPoint p) {
134 if (p == kInvalidPoint || p == kVirtualPass) return std::make_pair(-1, -1);
135
136 const int row = static_cast<int>(p) / kVirtualBoardSize;
137 const int col = static_cast<int>(p) % kVirtualBoardSize;
138 return std::make_pair(row - 1, col - 1);
139}

146// Internally, the board is *always* 21*21 with a border of guard stones around
147// all sides of the board. Thus we need to map a coordinate in that space
148// to a coordinate in the normal board.
149Action VirtualActionToAction(int virtual_action, int board_size) {
150 if (virtual_action == kVirtualPass) return board_size * board_size;
151 const int virtual_row = static_cast<int>(virtual_action) / kVirtualBoardSize;
152 const int virtual_col = static_cast<int>(virtual_action) % kVirtualBoardSize;
153 return board_size * (virtual_row - 1) + (virtual_col - 1);
154}
155
156int ActionToVirtualAction(Action action, int board_size) {
157 if (action == board_size * board_size) return kVirtualPass;
158 int row = action / board_size;
159 int column = action % board_size;
160 return (row + 1) * kVirtualBoardSize + (column + 1);
161}

250VirtualPoint MakePoint(std::string s) {
251 std::transform(s.begin(), s.end(), s.begin(), ::tolower);
252
253 if (s == "pass") return kVirtualPass;
254 if (s.size() < 2 || s.size() > 3) return kInvalidPoint;
255
256 int col = s[0] < 'i' ? s[0] - 'a' : s[0] - 'a' - 1;
257 int row = s[1] - '0';
258 if (s.size() == 3) {
259 row *= 10;
260 row += s[2] - '0';
261 }
262 return VirtualPointFrom2DPoint({row - 1, col});
263}

276void PhantomGoBoard::Clear() {
277 zobrist_hash_ = 0;
278
279 for (int i = 0; i < board_size_ * board_size_; i++) {
280 observations_[(uint8_t)GoColor::kBlack][i] = GoColor::kEmpty;
281 observations_[(uint8_t)GoColor::kWhite][i] = GoColor::kEmpty;
282 }
283
284 stone_count_ = {0, 0};
285
286 last_move_valid = true;
287 last_move_pass = false;
288 last_move_captured = 0;
289
290 for (int i = 0; i < board_.size(); ++i) {
291 Vertex &v = board_[i];
292 v.color = GoColor::kGuard;
293 v.chain_head = static_cast<VirtualPoint>(i);
294 v.chain_next = static_cast<VirtualPoint>(i);
295 chains_[i].reset_border();
296 }
297
298 for (VirtualPoint p : BoardPoints(board_size_)) {
299 board_[p].color = GoColor::kEmpty;
300 chains_[p].reset();
301 }
302
303 for (VirtualPoint p : BoardPoints(board_size_)) {
304 Neighbours(p, [this, p](VirtualPoint n) {
305 if (IsEmpty(n)) chain(p).add_liberty(n);
306 });
307 }
308
309 for (int i = 0; i < last_captures_.size(); ++i) {
310 last_captures_[i] = kInvalidPoint;
311 }
312
313 last_ko_point_ = kInvalidPoint;
314}

316bool PhantomGoBoard::PlayMove(VirtualPoint p, GoColor c) {
317 if (p == kVirtualPass) {
318 last_ko_point_ = kInvalidPoint;
319 last_move_captured = 0;
320 last_move_pass = true;
321 last_move_valid = true;
322 return true;
323 } else {
324 last_move_pass = false;
325 }
326
327 observations_[(uint8_t)c][VirtualPointToBoardPoint(p, board_size_)] =
328 board_[p].color;
329
330 // playing illegal moves will occur during phantom go, it is even desired
331 if (!IsLegalMoveObserver(p, c)) {
332 last_move_captured = 0;
333 last_move_valid = false; // was a observational move
334 return false;
335 }
336
337 last_move_valid = true;
338
339 stone_count_[(uint8_t)c]++;
340
341 // Preparation for ko checking.
342 bool played_in_enemy_eye = true;
343 Neighbours(p, [this, c, &played_in_enemy_eye](VirtualPoint n) {
344 GoColor s = PointColor(n);
345 if (s == c || s == GoColor::kEmpty) {
346 played_in_enemy_eye = false;
347 }
348 });
349
350 JoinChainsAround(p, c);
351 SetStone(p, c);
352 RemoveLibertyFromNeighbouringChains(p);
353 int stones_captured = CaptureDeadChains(p, c);
354
355 stone_count_[(uint8_t)OppColor(c)] -= stones_captured;
356 last_move_captured = stones_captured;
357
358 observations_[(uint8_t)c][VirtualPointToBoardPoint(p, board_size_)] = c;
359
360 if (played_in_enemy_eye && stones_captured == 1) {
361 last_ko_point_ = last_captures_[0];
362 } else {
363 last_ko_point_ = kInvalidPoint;
364 }
365
366 if (stones_captured != 0) {
367 for (int point = 0; point < board_size_ * board_size_; point++) {
368 VirtualPoint vpoint = VirtualPointFromBoardPoint(point, board_size_);
369
370 if (observations_[(uint8_t)OppColor(c)][point] == OppColor(c) &&
371 board_[vpoint].color == GoColor::kEmpty) {
372 observations_[(uint8_t)GoColor::kBlack][point] = GoColor::kEmpty;
373 observations_[(uint8_t)GoColor::kWhite][point] = GoColor::kEmpty;
374 }
375 }
376 }
377
378 SPIEL_CHECK_GT(chain(p).num_pseudo_liberties, 0);
379
380 return true;
381}

383VirtualPoint PhantomGoBoard::SingleLiberty(VirtualPoint p) const {
384 VirtualPoint head = ChainHead(p);
385 VirtualPoint liberty = chain(p).single_liberty();
386
387 // Check it is really a liberty.
388 SPIEL_CHECK_TRUE(IsInBoardArea(liberty));
389 SPIEL_CHECK_TRUE(IsEmpty(liberty));
390
391 // Make sure the liberty actually borders the group.
392 for (auto n = Neighbours4(liberty); n; ++n) {
393 if (ChainHead(*n) == head) return liberty;
394 }
395
396 SpielFatalError(
397 absl::StrCat("liberty", liberty, " does not actually border group ", p));
398}

459// Combines the groups around the newly placed stone at vertex. If no groups
460// are available for joining, the new stone is placed as a new group.
461void PhantomGoBoard::JoinChainsAround(VirtualPoint p, GoColor c) {
462 VirtualPoint largest_chain_head = kInvalidPoint;
463 int largest_chain_size = 0;
464 Neighbours(
465 p, [this, c, &largest_chain_head, &largest_chain_size](VirtualPoint n) {
466 if (PointColor(n) == c) {
467 Chain &c = chain(n);
468 if (c.num_stones > largest_chain_size) {
469 largest_chain_size = c.num_stones;
470 largest_chain_head = ChainHead(n);
471 }
472 }
473 });
474 if (largest_chain_size == 0) {
475 InitNewChain(p);
476 return;
477 }
478
479 Neighbours(p, [this, c, &largest_chain_head](VirtualPoint n) {
480 if (PointColor(n) == c) {
481 VirtualPoint chain_head = ChainHead(n);
482 if (chain_head != largest_chain_head) {
483 chain(largest_chain_head).merge(chain(n));
484
485 // Set all stones in the smaller string to be part of the larger
486 // chain.
487 VirtualPoint cur = n;
488 do {
489 board_[cur].chain_head = largest_chain_head;
490 cur = board_[cur].chain_next;
491 } while (cur != n);
492
493 // Connect the 2 linked lists representing the stones in the two
494 // chains.
495 std::swap(board_[largest_chain_head].chain_next, board_[n].chain_next);
496 }
497 }
498 });
499
500 board_[p].chain_next = board_[largest_chain_head].chain_next;
501 board_[largest_chain_head].chain_next = p;
502 board_[p].chain_head = largest_chain_head;
503 chain(largest_chain_head).num_stones += 1;
504
505 Neighbours(p, [this, largest_chain_head](VirtualPoint n) {
506 if (IsEmpty(n)) {
507 chain(largest_chain_head).add_liberty(n);
508 }
509 });
510}

516int PhantomGoBoard::CaptureDeadChains(VirtualPoint p, GoColor c) {
517 int stones_captured = 0;
518 int capture_index = 0;
519 Neighbours(p, [this, c, &capture_index, &stones_captured](VirtualPoint n) {
520 if (PointColor(n) == OppColor(c) && chain(n).num_pseudo_liberties == 0) {
521 last_captures_[capture_index++] = ChainHead(n);
522 stones_captured += chain(n).num_stones;
523 RemoveChain(n);
524 }
525 });
526
527 for (; capture_index < last_captures_.size(); ++capture_index) {
528 last_captures_[capture_index] = kInvalidPoint;
529 }
530
531 return stones_captured;
532}

534void PhantomGoBoard::RemoveChain(VirtualPoint p) {
535 VirtualPoint this_chain_head = ChainHead(p);
536 VirtualPoint cur = p;
537 do {
538 VirtualPoint next = board_[cur].chain_next;
539
540 SetStone(cur, GoColor::kEmpty);
541 InitNewChain(cur);
542
543 Neighbours(cur, [this, this_chain_head, cur](VirtualPoint n) {
544 if (ChainHead(n) != this_chain_head || IsEmpty(n)) {
545 chain(n).add_liberty(cur);
546 }
547 });
548
549 cur = next;
550 } while (cur != p);
551}

553void PhantomGoBoard::InitNewChain(VirtualPoint p) {
554 board_[p].chain_head = p;
555 board_[p].chain_next = p;
556
557 Chain &c = chain(p);
558 c.reset();
559 c.num_stones += 1;
560
561 Neighbours(p, [this, &c](VirtualPoint n) {
562 if (IsEmpty(n)) {
563 c.add_liberty(n);
564 }
565 });
566}

574bool PhantomGoBoard::IsLegalMoveObserver(VirtualPoint p, GoColor c) const {
575 if (p == kVirtualPass) return true;
576 if (!IsInBoardArea(p)) return false;
577 if (!IsEmpty(p) || p == LastKoPoint()) return false;
578 if (chain(p).num_pseudo_liberties > 0) return true;
579
580 // For all checks below, the newly placed stone is completely surrounded by
581 // enemy and friendly stones.
582
583 // Allow to play if the placed stones connects to a group that still has at
584 // least one other liberty after connecting.
585 bool has_liberty = false;
586 Neighbours(p, [this, c, &has_liberty](VirtualPoint n) {
587 has_liberty |= (PointColor(n) == c && !chain(n).in_atari());
588 });
589 if (has_liberty) return true;
590
591 // Allow to play if the placed stone will kill at least one group.
592 bool kills_group = false;
593 Neighbours(p, [this, c, &kills_group](VirtualPoint n) {
594 kills_group |= (PointColor(n) == OppColor(c) && chain(n).in_atari());
595 });
596 if (kills_group) return true;
597
598 return false;
599}

604 GoColor::kEmpty) {
605 return true;
606 }

611 num_stones = 0;
612 // Need to have values big enough that they can never go below 0 even if
613 // all liberties are removed.
614 num_pseudo_liberties = 4;
615 liberty_vertex_sum = 32768;
616 liberty_vertex_sum_squared = 2147483648;

626void PhantomGoBoard::Chain::merge(const Chain &other) {
627 num_stones += other.num_stones;
628 num_pseudo_liberties += other.num_pseudo_liberties;
629 liberty_vertex_sum += other.liberty_vertex_sum;
630 liberty_vertex_sum_squared += other.liberty_vertex_sum_squared;
631}
632
633void PhantomGoBoard::Chain::add_liberty(VirtualPoint p) {
634 num_pseudo_liberties += 1;
635 liberty_vertex_sum += p;
636 liberty_vertex_sum_squared +=
637 static_cast<uint32_t>(p) * static_cast<uint32_t>(p);
638}
639
640void PhantomGoBoard::Chain::remove_liberty(VirtualPoint p) {
641 num_pseudo_liberties -= 1;
642 liberty_vertex_sum -= p;
643 liberty_vertex_sum_squared -=
644 static_cast<uint32_t>(p) * static_cast<uint32_t>(p);
645}

647VirtualPoint PhantomGoBoard::Chain::single_liberty() const {
648 SPIEL_CHECK_TRUE(in_atari());
649 // A point is in Atari if it has only a single liberty, i.e. all pseudo
650 // liberties are for the same point.
651 // This is true exactly when
652 // liberty_vertex_sum**2 == liberty_vertex_sum_squared * num_pseudo_liberties
653 // Since all pseudo liberties are for the same point, this is equivalent to
654 // (taking n = num_pseudo_liberties):
655 // (n * p)**2 = (n * p**2) * n
656 // Thus to obtain p, we simple need to divide out the number of pseudo
657 // liberties.
658 SPIEL_CHECK_EQ(liberty_vertex_sum % num_pseudo_liberties, 0);
659 return static_cast<VirtualPoint>(liberty_vertex_sum / num_pseudo_liberties);
660}

686std::ostream &operator<<(std::ostream &os, const PhantomGoBoard &board) {
687 os << "\n";
688 for (int row = board.board_size() - 1; row >= 0; --row) {
689 os << std::setw(2) << std::setfill(' ') << (row + 1) << " ";
690 for (int col = 0; col < board.board_size(); ++col) {
691 os << GoColorToChar(
692 board.PointColor(VirtualPointFrom2DPoint({row, col})));
693 }
694 os << std::endl;
695 }
696
697 std::string columns = "ABCDEFGHJKLMNOPQRST";
698 os << " " << columns.substr(0, board.board_size()) << std::endl;
699
700 // Encode the stones and print a URL that can be used to view the board.
701 std::string encoded;

716 --lib_i_;

718 Neighbours(chain_cur_, [this](VirtualPoint n) {

730// Returns the number of points surrounded entirely by one color.
731// Aborts early and returns 0 if the area borders both black and white stones.
732int NumSurroundedPoints(const PhantomGoBoard &board, const VirtualPoint p,
733 std::array<bool, kVirtualBoardPoints> *marked,
734 bool *reached_black, bool *reached_white) {
735 if ((*marked)[p]) return 0;
736 (*marked)[p] = true;
737
738 int num_points = 1;
739 Neighbours(p, [&board, &num_points, marked, reached_black,
740 reached_white](VirtualPoint n) {
741 switch (board.PointColor(n)) {
742 case GoColor::kBlack:
743 *reached_black = true;
744 break;
745 case GoColor::kWhite:
746 *reached_white = true;
747 break;
748 case GoColor::kEmpty:
749 num_points +=
750 NumSurroundedPoints(board, n, marked, reached_black, reached_white);
751 break;
752 case GoColor::kGuard:
753 // Ignore the border.
754 break;
755 }
756 });
757
758 return num_points;
759}

761float TrompTaylorScore(const PhantomGoBoard &board, float komi, int handicap) {
762 // The delta of how many points on the board black and white have occupied,
763 // from black's point of view, i.e. Black points - White points.
764 int occupied_delta = 0;
765
766 // We need to keep track of which empty points we've already counted as part
767 // of a larger territory.
768 std::array<bool, kVirtualBoardPoints> marked;
769 marked.fill(false);
770
771 for (VirtualPoint p : BoardPoints(board.board_size())) {
772 switch (board.PointColor(p)) {
773 case GoColor::kBlack:
774 ++occupied_delta;
775 break;
776 case GoColor::kWhite:
777 --occupied_delta;
778 break;
779 case GoColor::kEmpty: {
780 if (marked[p]) continue;
781 // If some empty points are surrounded entirely by one player, they
782 // count as that player's territory.
783 bool reached_black = false, reached_white = false;
784 int n = NumSurroundedPoints(board, p, &marked, &reached_black,
785 &reached_white);
786 if (reached_black && !reached_white) {
787 occupied_delta += n;
788 } else if (!reached_black && reached_white) {
789 occupied_delta -= n;
790 }
791 break;
792 }
793 case GoColor::kGuard:
794 SpielFatalError("unexpected color");
795 }
796 }
797
798 float score = occupied_delta - komi;
799 if (handicap >= 2) {
800 score -= handicap;
801 }
802 return score;
803}

805PhantomGoBoard CreateBoard(const std::string &initial_stones) {
806 PhantomGoBoard board(9);
807
808 int row = 0;
809 for (const auto &line : absl::StrSplit(initial_stones, '\n')) {
810 int col = 0;
811 bool stones_started = false;
812 for (const auto &c : line) {
813 if (c == ' ') {
814 if (stones_started) {
815 SpielFatalError(
816 "Whitespace is only allowed at the start of "
817 "the line. To represent empty intersections, "
818 "use +");
819 }
820 continue;
821 } else if (c == 'X') {
822 stones_started = true;
823 SPIEL_CHECK_TRUE(board.PlayMove(VirtualPointFrom2DPoint({row, col}),
824 GoColor::kBlack));
825 } else if (c == 'O') {
826 stones_started = true;
827 SPIEL_CHECK_TRUE(board.PlayMove(VirtualPointFrom2DPoint({row, col}),
828 GoColor::kWhite));
829 }
830 col++;
831 }
832 row++;
833 }
834
835 return board;
836}
