105 // Doesn't include construction or copy overhead in the loop.
106 template <typename Function, typename Callable, typename... Args>
107 void CallFunctionBenchmark(benchmark::State& state, const Callable& c,
108  Args... args) {
109  Function f = c;
110  for (auto _ : state) {
111  benchmark::DoNotOptimize(&f);
112  f(args...);
113  }
114 }
