15"""Perturbates any policy with tabular-saved, fixed noise.
16
17The policy's probabilities P' on each state s are computed as
18
19P'(s) = alpha * epsilon + (1-alpha) * P(s),
20
21with P the former policy's probabilities, and epsilon ~ Softmax(beta *
22Uniform)
23"""
24
25import numpy as np
26from open_spiel.python import policy as openspiel_policy
27
28
29class NoisyPolicy(openspiel_policy.Policy):
30 """Pyspiel Best Response with added noise.
31
32 This policy's probabilities P' on each `player_id` state s is computed as
33 P'(s) = alpha * epsilon + (1-alpha) * P(s),
34
35 with P the former policy's probabilities, and epsilon ~ Softmax(beta *
36 Uniform)
37 """
38
39 def __init__(self, policy, player_id=None, alpha=0.1, beta=1.0):
40 """Initializes the noisy policy.
41
42 Note that this noise only affects `player_id`.
43
44 Args:
45 policy: Any OpenSpiel `policy.Policy` object.
46 player_id: The player id, the policy of whom will be made noisy. If `None`
47 noise will be added to the policies for all players.
48 alpha: Mixing noise factor.
49 beta: Softmax inverse temperature factor.
50 """
51 self._policy = policy
52 self.game = policy.game
53 self.game_type = self.game.get_type()
54
55 self.player_id = player_id
56
57 self._noise_dict = {}
58 self._alpha = alpha
59 self._beta = beta
60
61 def _state_key(self, state, player):
62 """Returns the key to use to look up this (state, player) pair."""
63 if self.game_type.provides_information_state_string:
64 if player is None:
65 return state.information_state_string()
66 else:
67 return state.information_state_string(player)
68 elif self.game_type.provides_observation_string:
69 if player is None:
70 return state.observation_string()
71 else:
72 return state.observation_string(player)
73 else:
74 return str(state)
75
76 def get_or_create_noise(self, state, player_id=None):
77 """Get noisy policy or create it and return it.
78
79 Args:
80 state: the state to which the policy will be applied.
81 player_id: the player id that will apply the noisy policy. Default to
82 current_player. Should be defined in the case of simultaneous games.
83
84 Returns:
85 noise_action_probs: The noisy probability distribution on the set of legal
86 actions.
87 """
88 if player_id is None:
89 player_id = state.current_player()
90 info_state = self._state_key(state, player_id)
91 if info_state not in self._noise_dict:
92 action_ids = state.legal_actions(player_id)
93 noise = self._beta * np.random.normal(size=len(action_ids))
94 noise = np.exp(noise - noise.max())
95 noise /= np.sum(noise)
96
97 self._noise_dict[info_state] = {
98 action_ids[i]: noise[i] for i in range(len(noise))
99 }
100 return self._noise_dict[info_state]
101
102 def mix_probs(self, probs, noise_probs):
103 return {
104 i: (1 - self._alpha) * probs[i] + self._alpha * noise_probs[i]
105 for i in probs
106 }
107
108 @property
109 def policy(self):
110 return self._policy
111
112 def action_probabilities(self, state, player_id=None):
113 """Returns the policy for a player in a state.
114
115 Args:
116 state: A `pyspiel.State` object.
117 player_id: Optional, the player id for whom we want an action. Optional
118 unless this is a simultabeous state at which multiple players can act.
119
120 Returns:
121 A `dict` of `{action: probability}` for the specified player in the
122 supplied state.
123 """
124
125 # If self._player_id is None, or if self.player_id == current_player, add
126 # noise.
127 if ((self.player_id is None) or
128 (state.current_player() == self.player_id) or
129 (player_id == self.player_id)):
130 noise_probs = self.get_or_create_noise(state, player_id)
131 probs = self._policy.action_probabilities(state, player_id)
132 probs = self.mix_probs(probs, noise_probs)
133 return probs
134
135 # Send the default probabilities for all other players
136 return self._policy.action_probabilities(state, player_id)
