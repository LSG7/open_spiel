220 # The acting player in 1b is player 1 and they have not acted before, so
221 # the probability this player plays to this information state is 1, and
222 # the sequence probability of any action is just the probability of that
223 # action given the information state. On the first iteration, this
224 # probability is 0.5 for both actions. On the second iteration, the
225 # current policy is [0, 1], so the average cumulants should be
226 # [0.5, 1.5]. Normalizing this gives the average policy.
227 normalization = 0.5 + 0.5 + 1
228 np.testing.assert_allclose(cfr_solver.average_policy().policy_for_key("1b"),
229 [0.5 / normalization, (0.5 + 1) / normalization])
