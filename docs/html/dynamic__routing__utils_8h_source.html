83 // Network implementation.
84 //
85 // A network is a directed graph with a volume delay function on each
86 // of its edges. Each vertex is referred to as a string (for example "A") and
87 // each edge as a string f"{node1}->{node2}" (for example "A->B"). The network
88 // is created from an adjacency list. Each road section is mapped to an action
89 // index (positive integer) in road_section_to_action_, and vice versa in
90 // action_to_road_section_. The volume delay function on each road section rs
91 // is given by free_flow_travel_time_[rs]*(1+ a_[rs]*(v/capacity_[rs])**b_[rs])
92 // where v is the volume on the road section rs, according to the U.S. Bureau
93 // of Public Road (BPR). Such functions are called fundamental diagram of
94 // traffic flow.
95 class Network {

102  node_position = {},
103  const absl::flat_hash_map<std::string, float>& bpr_a_coefficient = {},
104  const absl::flat_hash_map<std::string, float>& bpr_b_coefficient = {},
105  const absl::flat_hash_map<std::string, float>& capacity = {},
106  const absl::flat_hash_map<std::string, float>& free_flow_travel_time =
107  {});
108

172  absl::flat_hash_map<std::string, std::pair<float, float>> node_position_;
173  // flat_hash_map that maps road section string representation to action.
174  absl::flat_hash_map<std::string, int> action_by_road_section_;
175  // vector that maps action to road section string representation.
176  std::vector<std::string> road_section_by_action;
177  // flat_hash_set that contains sink locations.

179 };
