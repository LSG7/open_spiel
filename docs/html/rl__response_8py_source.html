68 def eval_against_fixed_bots(env, trained_agents, fixed_agents, num_episodes):
69  """Evaluates `trained_agents` against `random_agents` for `num_episodes`."""
70  num_players = len(fixed_agents)
71  sum_episode_rewards = np.zeros(num_players)
72  for player_pos in range(num_players):
73  cur_agents = fixed_agents[:]
74  cur_agents[player_pos] = trained_agents[player_pos]
75  for _ in range(num_episodes):
76  time_step = env.reset()
77  episode_rewards = 0
78  turn_num = 0
79  while not time_step.last():
80  turn_num += 1
81  player_id = time_step.observations["current_player"]
82  if env.is_turn_based:
83  agent_output = cur_agents[player_id].step(
84  time_step, is_evaluation=True)
85  action_list = [agent_output.action]
86  else:
87  agents_output = [
88  agent.step(time_step, is_evaluation=True) for agent in cur_agents
89  ]
90  action_list = [agent_output.action for agent_output in agents_output]
91  time_step = env.step(action_list)
92  episode_rewards += time_step.rewards[player_pos]
93  sum_episode_rewards[player_pos] += episode_rewards
94  return sum_episode_rewards / num_episodes

131 class FirstActionAgent(rl_agent.AbstractAgent):
132  """An example agent class."""
133 
134  def __init__(self, player_id, num_actions, name="first_action_agent"):
135  assert num_actions > 0
136  self._player_id_player_id = player_id
137  self._num_actions_num_actions = num_actions
138 
139  def step(self, time_step, is_evaluation=False):
140  # If it is the end of the episode, don't select an action.
141  if time_step.last():
142  return
143 
144  # Pick the first legal action.
145  cur_legal_actions = time_step.observations["legal_actions"][self._player_id_player_id]
146  action = cur_legal_actions[0]
147  probs = np.zeros(self._num_actions_num_actions)
148  probs[action] = 1.0
149 
150  return rl_agent.StepOutput(action=action, probs=probs)

153 class RollingAverage(object):
154  """Class to store a rolling average."""
155 
156  def __init__(self, size=100):
157  self._size_size = size
158  self._values_values = np.array([0] * self._size_size, dtype=np.float64)
159  self._index_index = 0
160  self._total_additions_total_additions = 0
161 
162  def add(self, value):
163  self._values_values[self._index_index] = value
164  self._total_additions_total_additions += 1
165  self._index_index = (self._index_index + 1) % self._size_size
166 
167  def mean(self):
168  n = min(self._size_size, self._total_additions_total_additions)
169  if n == 0:
170  return 0
171  return self._values_values.sum() / n

198  rolling_averager = RollingAverage(FLAGS.window_size)
199  rolling_averager_p0 = RollingAverage(FLAGS.window_size)
200  rolling_averager_p1 = RollingAverage(FLAGS.window_size)
201  rolling_value = 0
202  total_value = 0
203  total_value_n = 0
204 
205  with tf.Session() as sess:
206  hidden_layers_sizes = [int(l) for l in FLAGS.hidden_layers_sizes]
207  # pylint: disable=g-complex-comprehension
208  learning_agents = create_training_agents(num_players, sess, num_actions,
209  info_state_size,
210  hidden_layers_sizes)
211  sess.run(tf.global_variables_initializer())
212 
213  print("Starting...")
214 
215  for ep in range(FLAGS.num_train_episodes):
216  if (ep + 1) % FLAGS.eval_every == 0:
217  r_mean = eval_against_fixed_bots(env, learning_agents, exploitee_agents,
218  FLAGS.eval_episodes)
219  value = r_mean[0] + r_mean[1]
220  rolling_averager.add(value)
221  rolling_averager_p0.add(r_mean[0])
222  rolling_averager_p1.add(r_mean[1])
223  rolling_value = rolling_averager.mean()
224  rolling_value_p0 = rolling_averager_p0.mean()
225  rolling_value_p1 = rolling_averager_p1.mean()
226  total_value += value
227  total_value_n += 1
228  avg_value = total_value / total_value_n
229  print(("[{}] Mean episode rewards {}, value: {}, " +
230  "rval: {} (p0/p1: {} / {}), aval: {}").format(
231  ep + 1, r_mean, value, rolling_value, rolling_value_p0,
232  rolling_value_p1, avg_value))
233 
234  agents_round1 = [learning_agents[0], exploitee_agents[1]]
235  agents_round2 = [exploitee_agents[0], learning_agents[1]]
236 
237  for agents in [agents_round1, agents_round2]:
238  time_step = env.reset()
239  while not time_step.last():
240  player_id = time_step.observations["current_player"]
241  if env.is_turn_based:
242  agent_output = agents[player_id].step(time_step)
243  action_list = [agent_output.action]
244  else:
245  agents_output = [agent.step(time_step) for agent in agents]
246  action_list = [
247  agent_output.action for agent_output in agents_output
248  ]
249  time_step = env.step(action_list)
250 
251  # Episode is over, step all agents with final info state.
252  for agent in agents:
253  agent.step(time_step)
