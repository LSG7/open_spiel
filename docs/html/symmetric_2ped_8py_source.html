49 def gradients(dist, payoff_matrices, num_players, proj_grad=True):
50  """Computes exploitablity gradient.
51 
52  Args:
53  dist: 1-d np.array, current estimate of nash distribution
54  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
55  num_players: int, number of players, in case payoff_matrices is abbreviated
56  proj_grad: bool, if True, projects dist gradient onto simplex
57  Returns:
58  gradient of exploitability w.r.t. (dist) as tuple
59  unregularized exploitability (stochastic estimate)
60  unregularized exploitability (stochastic estimate) *duplicate
61  """
62  nabla = payoff_matrices[0].dot(dist)
63 
64  power = np.inf
65  s = np.linalg.norm(nabla, ord=power)
66  br = np.zeros_like(dist)
67  maxima = (nabla == s)
68  br[maxima] = 1. / maxima.sum()
69 
70  unreg_exp = np.max(nabla) - nabla.dot(dist)
71 
72  grad_dist = -(nabla) + (num_players - 1) * payoff_matrices[1].dot(br - dist)
73  if proj_grad:
74  grad_dist = simplex.project_grad(grad_dist)
75 
76  return (grad_dist,), unreg_exp, unreg_exp
