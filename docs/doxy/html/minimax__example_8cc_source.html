32int BlackPieceAdvantage(const State& state) {
33 const auto& bstate = down_cast<const breakthrough::BreakthroughState&>(state);
34 return bstate.pieces(breakthrough::kBlackPlayerId) -
35 bstate.pieces(breakthrough::kWhitePlayerId);
36}

38void PlayBreakthrough() {
39 std::shared_ptr<const Game> game =
40 LoadGame("breakthrough", {{"rows", GameParameter(6)},
41 {"columns", GameParameter(6)}});
42 std::unique_ptr<State> state = game->NewInitialState();
43 while (!state->IsTerminal()) {
44 std::cout << std::endl << state->ToString() << std::endl;
45
46 Player player = state->CurrentPlayer();
47 std::pair<double, Action> value_action = algorithms::AlphaBetaSearch(
48 *game, state.get(), [player](const State& state) {
49 return (player == breakthrough::kBlackPlayerId ?
50 BlackPieceAdvantage(state) :
51 -BlackPieceAdvantage(state));
52 },
53 kSearchDepth, player);
54
55 std::cout << std::endl << "Player " << player << " choosing action "
56 << state->ActionToString(player, value_action.second)
57 << " with heuristic value (to black) " << value_action.first
58 << std::endl;
59
60 state->ApplyAction(value_action.second);
61 }
62
63 std::cout << "Terminal state: " << std::endl;
64 std::cout << state->ToString() << std::endl;
65}

67int FirstPlayerAdvantage(const State& state) {
68 const auto& pstate = down_cast<const pig::PigState&>(state);
69 return pstate.score(0) - pstate.score(1);

72void PlayPig(std::mt19937& rng) {
73 std::shared_ptr<const Game> game =
74 LoadGame("pig", {{"winscore", GameParameter(kWinscorePig)},
75 {"diceoutcomes", GameParameter(kDiceoutcomesPig)}});
76 std::unique_ptr<State> state = game->NewInitialState();
77 while (!state->IsTerminal()) {
78 std::cout << std::endl << state->ToString() << std::endl;
79
80 Player player = state->CurrentPlayer();
81 if (state->IsChanceNode()) {
82 // Chance node; sample one according to underlying distribution.
83 ActionsAndProbs outcomes = state->ChanceOutcomes();
84 Action action = open_spiel::SampleAction(outcomes, rng).first;
85 std::cerr << "Sampled action: " << state->ActionToString(player, action)
86 << std::endl;
87 state->ApplyAction(action);
88 } else {
89 std::pair<double, Action> value_action = algorithms::ExpectiminimaxSearch(
90 *game, state.get(),
91 [player](const State& state) {
92 return (player == Player{0} ? FirstPlayerAdvantage(state)
93 : -FirstPlayerAdvantage(state));
94 },
95 kSearchDepthPig, player);
96
97 std::cout << std::endl
98 << "Player " << player << " choosing action "
99 << state->ActionToString(player, value_action.second)
100 << " with heuristic value " << value_action.first << std::endl;
101
102 state->ApplyAction(value_action.second);
103 }
104 }
105
106 std::cout << "Terminal state: " << std::endl;
107 std::cout << state->ToString() << std::endl;
108}
