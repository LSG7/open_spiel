72 GridSpec ParseGrid(const std::string& grid_string, int max_num_players) {
73  GridSpec grid{/*num_rows=*/0, /*num_cols=*/0};
74  int row = 0;
75  int col = 0;
76  int count_empty_cells = 0;
77  absl::flat_hash_map<Player, std::pair<int, int>> starting_positions_map;
78  absl::flat_hash_map<Player, std::pair<int, int>> destinations_map;
79 
80  for (auto c : grid_string) {
81  if (c == '\n') {
82  row += 1;
83  col = 0;
84  } else {
85  if (row >= grid.num_rows) grid.num_rows = row + 1;
86  if (col >= grid.num_cols) grid.num_cols = col + 1;
87  if (c == '*') {
88  grid.obstacles.emplace_back(row, col);
89  } else if (islower(c)) {
90  // 97 is the ASCII code for 'a'.
91  Player player = static_cast<int>(c) - 97;
92  SPIEL_CHECK_GE(player, 0);
93  SPIEL_CHECK_LT(player, max_num_players);
94  starting_positions_map[player] = {row, col};
95  } else if (isupper(c)) {
96  // 65 is the ASCII code for 'A'.
97  Player player = static_cast<int>(c) - 65;
98  SPIEL_CHECK_GE(player, 0);
99  SPIEL_CHECK_LT(player, max_num_players);
100  destinations_map[player] = {row, col};
101  } else if (c == '.') {
102  ++count_empty_cells;
103  } else {
104  SpielFatalError(absl::StrCat("Invalid char '", std::string(1, c),
105  "' at grid (", row, ",", col, ")"));
106  }
107  col += 1;
108  }
109  }
110 
111  grid.num_players = starting_positions_map.size();
112  SPIEL_CHECK_EQ(starting_positions_map.size(), destinations_map.size());
113  SPIEL_CHECK_GE(grid.num_players, 1);
114  SPIEL_CHECK_LE(grid.num_players, max_num_players);
115 
116  // Move map entries to vectors.
117  grid.starting_positions.resize(grid.num_players);
118  grid.destinations.resize(grid.num_players);
119  for (Player p = 0; p < grid.num_players; ++p) {
120  // Check that we found a starting position, and move it to the vector.
121  const auto iter1 = starting_positions_map.find(p);
122  SPIEL_CHECK_TRUE(iter1 != starting_positions_map.end());
123  grid.starting_positions[p] = iter1->second;
124  // Check that we found a destination, and move it to the vector.
125  const auto iter2 = destinations_map.find(p);
126  SPIEL_CHECK_TRUE(iter2 != destinations_map.end());
127  grid.destinations[p] = iter2->second;
128  }
129  return grid;
130 }

134 PathfindingState::PathfindingState(std::shared_ptr<const Game> game,
135  const GridSpec& grid_spec, int horizon)
136  : SimMoveState(game),
137  parent_game_(down_cast<const PathfindingGame&>(*game)),
138  grid_spec_(grid_spec),
139  cur_player_(kSimultaneousPlayerId),
140  total_moves_(0),
141  horizon_(horizon),
142  player_positions_(num_players_),
143  actions_(num_players_, kInvalidAction),
144  rewards_(num_players_, 0.0),
145  returns_(num_players_, 0.0),
146  contested_players_(num_players_, 0),
147  reached_destinations_(num_players_, 0) {
148  grid_.reserve(grid_spec_.num_rows);
149  for (int r = 0; r < grid_spec_.num_rows; ++r) {
150  grid_.push_back(std::vector<int>(grid_spec_.num_cols, kEmpty));
151  }
152 
153  for (const std::pair<int, int>& c : grid_spec_.obstacles) {
154  grid_[c.first][c.second] = kWall;
155  }
156 
157  SPIEL_CHECK_EQ(grid_spec_.starting_positions.size(), num_players_);
158  for (Player p = 0; p < num_players_; ++p) {
159  const std::pair<int, int>& c = grid_spec_.starting_positions[p];
160  SPIEL_CHECK_EQ(grid_[c.first][c.second], kEmpty);
161  grid_[c.first][c.second] = p;
162  player_positions_[p] = c;
163  }
164 }

171 void PathfindingState::DoApplyActions(const std::vector<Action>& moves) {
172  SPIEL_CHECK_EQ(moves.size(), num_players_);
173  SPIEL_CHECK_EQ(cur_player_, kSimultaneousPlayerId);
174 
175  std::fill(rewards_.begin(), rewards_.end(), 0.0);
176  std::fill(contested_players_.begin(), contested_players_.end(), 0);
177 
178  actions_ = moves;
179  if (num_players_ == 1) {
180  ResolvePlayerAction(0);
181  } else {
182  ResolveActions();
183  }
184 
185  if (cur_player_ == kSimultaneousPlayerId) {
186  // Only increment total moves if actions fully resolved.
187  total_moves_++;
188  }
189 
190  // If all players are at their destinations.
191  if (AllPlayersOnDestinations()) {
192  // Terminal state reached, all players get a bonus.
193  for (Player p = 0; p < num_players_; ++p) {
194  rewards_[p] += parent_game_.group_reward();
195  returns_[p] += parent_game_.group_reward();
196  }
197  }
198 }

205 std::pair<int, int> PathfindingState::GetNextCoord(Player p) const {
206  int row = player_positions_[p].first + kRowOffsets[actions_[p]];
207  int col = player_positions_[p].second + kColOffsets[actions_[p]];
208  if (!InBounds(row, col) || grid_[row][col] == kWall) {
209  // Can't run out of bounds or into a wall.
210  return player_positions_[p];
211  }
212  return {row, col};
213 }

215 void PathfindingState::ResolvePlayerAction(Player p) {
216  const std::pair<int, int>& cur_coord = player_positions_[p];
217  std::pair<int, int> next_coord = GetNextCoord(p);
218 
219  // Check if there is a player there. If so, change next_coord to cur_coord.
220  Player other_player = PlayerAt(next_coord);
221  if (other_player != kInvalidPlayer && other_player != p) {
222  next_coord = cur_coord;
223  }
224 
225  // Distribute rewards.
226  if (next_coord != cur_coord && reached_destinations_[p] == 0 &&
227  next_coord == grid_spec_.destinations[p]) {
228  // Player is just getting to the destination for the first time!
229  rewards_[p] += parent_game_.solve_reward();
230  returns_[p] += parent_game_.solve_reward();
231  reached_destinations_[p] = 1;
232  } else if (next_coord == grid_spec_.destinations[p]) {
233  // Player getting to destination again, or staying there: no penalty.
234  } else {
235  rewards_[p] += parent_game_.step_reward();
236  returns_[p] += parent_game_.step_reward();
237  }
238 
239  grid_[cur_coord.first][cur_coord.second] = kEmpty;
240  grid_[next_coord.first][next_coord.second] = p;
241  player_positions_[p] = next_coord;
242 }

253 int PathfindingState::TryResolveContested() {
254  int num_resolutions = 0;
255  for (Player p = 0; p < num_players_; ++p) {
256  if (contested_players_[p] == 1) {
257  std::pair<int, int> next_coord = GetNextCoord(p);
258  // A contested player can be resolved iff:
259  // - There is no other player on the next coord, and
260  // - No other (contested) player is planning to go there.
261  Player other_player = PlayerAt(next_coord);
262  if (other_player == kInvalidPlayer) {
263  bool conflict = false;
264  for (Player op = 0; op < num_players_; ++op) {
265  if (p == op) {
266  continue;
267  }
268  if (contested_players_[op] == 1) {
269  std::pair<int, int> op_next_coord = GetNextCoord(op);
270  if (next_coord == op_next_coord) {
271  conflict = true;
272  break;
273  }
274  }
275  }
276 
277  if (!conflict) {
278  contested_players_[p] = 0;
279  num_resolutions++;
280  ResolvePlayerAction(p);
281  }
282  }
283  }
284  }
285 
286  return num_resolutions;
287 }

289 void PathfindingState::ResolveActions() {
290  // Get the next coords, and check for potentially conflicting actions.
291  std::vector<std::pair<int, int>> next_coords;
292  next_coords.reserve(num_players_);
293  for (Player p = 0; p < num_players_; ++p) {
294  std::pair<int, int> next_coord = GetNextCoord(p);
295  // If there is a different player there, mark as potentially contested.
296  // If another player is going there, mark both players as contested.
297  Player other_player = PlayerAt(next_coord);
298  if (other_player != kInvalidPlayer && other_player != p) {
299  // Different player already there. Potentially contested (other player
300  // may move out).
301  contested_players_[p] = 1;
302  } else if (actions_[p] == kStay) {
303  // Stay action is never contested.
304  } else {
305  // Check if another player planning to go there.
306  auto iter = std::find(next_coords.begin(), next_coords.end(), next_coord);
307  if (iter != next_coords.end()) {
308  Player other_player = iter - next_coords.begin();
309  contested_players_[p] = 1;
310  contested_players_[other_player] = 1;
311  }
312  }
313 
314  next_coords.push_back(next_coord);
315  }
316 
317  // Check for head-on collisions. These should not be marked as contested,
318  // because they result in a no-op.
319  for (Player p = 0; p < num_players_; ++p) {
320  if (contested_players_[p] == 1) {
321  int op = PlayerAt(next_coords[p]);
322  if (op != kInvalidPlayer && p != op) {
323  Player opp = PlayerAt(next_coords[op]);
324  if (opp != kInvalidPlayer && opp == p) {
325  contested_players_[p] = 0;
326  contested_players_[op] = 0;
327  continue;
328  }
329  }
330  }
331  }
332 
333  // Move the uncontested, and repeatedly check the contested players to see if
334  // moving resolves the contestations. If so, move them and mark as
335  // uncontested. Stop when there is a pass with no moves.
336  int num_contested = 0;
337  for (Player p = 0; p < num_players_; ++p) {
338  if (contested_players_[p] == 1) {
339  num_contested++;
340  } else {
341  ResolvePlayerAction(p);
342  }
343  }
344 
345  int num_resolved = 0;
346  do {
347  num_resolved = TryResolveContested();
348  num_contested -= num_resolved;
349  } while (num_resolved > 0);
350 
351  // If there remain contestations, must resolve them via a chance node, which
352  // will determine order of resolution.
353  if (num_contested > 0) {
354  cur_player_ = kChancePlayerId;
355  }
356 }

358 void PathfindingState::DoApplyAction(Action action_id) {
359  if (IsSimultaneousNode()) {
360  ApplyFlatJointAction(action_id);
361  return;
362  } else {
363  SPIEL_CHECK_TRUE(IsChanceNode());
364  int num_contested_players =
365  std::count_if(contested_players_.begin(), contested_players_.end(),
366  [](int i) { return i == 1; });
367  std::vector<Player> contested_player_ids;
368  contested_player_ids.reserve(num_contested_players);
369  for (Player p = 0; p < contested_players_.size(); ++p) {
370  if (contested_players_[p] == 1) {
371  contested_player_ids.push_back(p);
372  }
373  }
374  SPIEL_CHECK_EQ(contested_player_ids.size(), num_contested_players);
375  std::vector<int> indices(num_contested_players);
376  std::iota(indices.begin(), indices.end(), 0);
377  std::vector<int> resolution_order = UnrankPermutation(indices, action_id);
378  for (int idx : resolution_order) {
379  ResolvePlayerAction(contested_player_ids[idx]);
380  }
381  std::fill(contested_players_.begin(), contested_players_.end(), 0);
382  cur_player_ = kSimultaneousPlayerId;
383  total_moves_++;
384  }
385 }

396 std::vector<std::pair<Action, double>> PathfindingState::ChanceOutcomes()
397  const {
398  SPIEL_CHECK_TRUE(IsChanceNode());
399  int num_contested_players =
400  std::count_if(contested_players_.begin(), contested_players_.end(),
401  [](int i) { return i == 1; });
402  int num_permutations = Factorial(num_contested_players);
403  double prob = 1.0 / num_permutations;
404  ActionsAndProbs outcomes;
405  outcomes.reserve(num_permutations);
406  for (int i = 0; i < num_permutations; ++i) {
407  outcomes.push_back({i, prob});
408  }
409  return outcomes;
410 }

453 // Note: currently, the observations are current non-Markovian because the time
454 // step is not included and the horizon is finite.
455 void PathfindingState::ObservationTensor(int player,
456  absl::Span<float> values) const {
457  SPIEL_CHECK_GE(player, 0);
458  SPIEL_CHECK_LT(player, num_players_);
459  std::fill(values.begin(), values.end(), 0.0);
460 
461  TensorView<3> view(values,
462  {parent_game_.NumObservationPlanes(), grid_spec_.num_rows,
463  grid_spec_.num_cols},
464  true);
465 
466  // Let n be the number of players.
467  // - First n planes refer to player
468  // - Second n planes refer to player's starting position
469  // - Third n planes refer to player's destination position
470  // - 1 plane for wall
471  // - 1 plane for empty
472  //
473  // The first three sets of n planes corresponding to the players are each
474  // ordered ego-centrically:
475  // - the first plane is the observing player's plane, followed by the next
476  // player, followed by the next etc. so in a 4-player game, if player 2
477  // is the observing player, the planes would be ordered by player 2, 3, 0,
478  // 1.
479  for (int r = 0; r < grid_spec_.num_rows; ++r) {
480  for (int c = 0; c < grid_spec_.num_cols; ++c) {
481  // Player on the position.
482  if (grid_[r][c] >= 0 && grid_[r][c] < num_players_) {
483  view[{PlayerPlaneIndex(player, grid_[r][c]), r, c}] = 1.0;
484  }
485 
486  // Wall
487  if (grid_[r][c] == kWall) {
488  view[{3 * num_players_, r, c}] = 1.0;
489  }
490 
491  // Empty
492  if (grid_[r][c] == kEmpty) {
493  view[{3 * num_players_ + 1, r, c}] = 1.0;
494  }
495  }
496  }
497 
498  for (Player p = 0; p < num_players_; ++p) {
499  const std::pair<int, int>& start_pos = grid_spec_.starting_positions[p];
500  const std::pair<int, int>& dest_pos = grid_spec_.destinations[p];
501  int pidx = PlayerPlaneIndex(player, p);
502  view[{num_players_ + pidx, start_pos.first, start_pos.second}] = 1.0;
503  view[{2 * num_players_ + pidx, dest_pos.first, dest_pos.second}] = 1.0;
504  }
505 }

511  return false;
