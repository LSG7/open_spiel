44void CheckCorrelationDeviceProbDist(const CorrelationDevice& mu) {
45 double prob_sum = 0.0;
46 for (const std::pair<double, TabularPolicy>& item : mu) {
47 SPIEL_CHECK_PROB(item.first);
48 prob_sum += item.first;
49 }
50 SPIEL_CHECK_FLOAT_EQ(prob_sum, 1.0);
51}

53ActionsAndProbs CreateDeterministicPolicy(Action chosen_action,
54 int num_actions) {
55 ActionsAndProbs actions_and_probs;
56 actions_and_probs.reserve(num_actions);
57 int num_ones = 0;
58 int num_zeros = 0;
59 for (Action action = 0; action < num_actions; ++action) {
60 if (action == chosen_action) {
61 num_ones++;
62 actions_and_probs.push_back({action, 1.0});
63 } else {
64 num_zeros++;
65 actions_and_probs.push_back({action, 0.0});
66 }
67 }
68 SPIEL_CHECK_EQ(num_ones, 1);
69 SPIEL_CHECK_EQ(num_ones + num_zeros, num_actions);
70 return actions_and_probs;
71}

73CorrelationDevice ConvertCorrelationDevice(
74 const Game& turn_based_nfg, const NormalFormCorrelationDevice& mu) {
75 // First get all the infostate strings.
76 std::unique_ptr<State> state = turn_based_nfg.NewInitialState();
77 std::vector<std::string> infostate_strings;
78 infostate_strings.reserve(turn_based_nfg.NumPlayers());
79 for (Player p = 0; p < turn_based_nfg.NumPlayers(); ++p) {
80 infostate_strings.push_back(state->InformationStateString());
81 state->ApplyAction(0);
82 }
83 SPIEL_CHECK_TRUE(state->IsTerminal());
84
85 int num_actions = turn_based_nfg.NumDistinctActions();
86 CorrelationDevice new_mu;
87 new_mu.reserve(mu.size());
88
89 // Next, convert to tabular policies.
90 for (const NormalFormJointPolicyWithProb& jpp : mu) {
91 TabularPolicy policy;
92 SPIEL_CHECK_EQ(jpp.actions.size(), turn_based_nfg.NumPlayers());
93 for (Player p = 0; p < turn_based_nfg.NumPlayers(); p++) {
94 policy.SetStatePolicy(
95 infostate_strings[p],
96 CreateDeterministicPolicy(jpp.actions[p], num_actions));
97 }
98 new_mu.push_back({jpp.probability, policy});
99 }
100
101 return new_mu;
102}

105// Helper function to return a correlation device that is a uniform distribution
106// over the vector of tabular policies.
107CorrelationDevice UniformCorrelationDevice(
108 std::vector<TabularPolicy>& policies) {
109 CorrelationDevice mu;
110 mu.reserve(policies.size());
111 for (const TabularPolicy& policy : policies) {
112 mu.push_back({1.0 / policies.size(), policy});
113 }
114 return mu;
115}

117// Return a string representation of the correlation device.
118std::string ToString(const CorrelationDevice& corr_dev) {
119 std::string corr_dev_str;
120 for (const auto& prob_and_policy : corr_dev) {
121 absl::StrAppend(&corr_dev_str, "Prob: ", prob_and_policy.first, "\n");
122 absl::StrAppend(&corr_dev_str, prob_and_policy.second.ToStringSorted(),
123 "\n");
124 }
125 return corr_dev_str;
126}

128std::vector<double> ExpectedValues(const Game& game,
129 const CorrelationDevice& mu) {
130 CheckCorrelationDeviceProbDist(mu);
131 std::vector<double> values(game.NumPlayers(), 0);
132 for (const std::pair<double, TabularPolicy>& item : mu) {
133 std::vector<double> item_values =
134 ExpectedReturns(*game.NewInitialState(), item.second, -1, false);
135 for (Player p = 0; p < game.NumPlayers(); ++p) {
136 values[p] += item.first * item_values[p];
137 }
138 }
139 return values;
140}

142std::vector<double> ExpectedValues(const Game& game,
143 const NormalFormCorrelationDevice& mu) {
144 if (game.GetType().information == GameType::Information::kOneShot) {
145 std::shared_ptr<const Game> actual_game = ConvertToTurnBased(game);
146 CorrelationDevice converted_mu = ConvertCorrelationDevice(*actual_game, mu);
147 return ExpectedValues(*actual_game, converted_mu);
148 } else {
149 SPIEL_CHECK_EQ(game.GetType().dynamics, GameType::Dynamics::kSequential);
150 CorrelationDevice converted_mu = ConvertCorrelationDevice(game, mu);
151 return ExpectedValues(game, converted_mu);
152 }

155double EFCEDist(const Game& game, CorrDistConfig config,
156 const CorrelationDevice& mu) {
157 // Check that the config matches what is supported.
158 SPIEL_CHECK_TRUE(config.deterministic);
159
160 // Check for proper probability distribution.
161 CheckCorrelationDeviceProbDist(mu);
162
163 auto efce_game =
164 std::make_shared<EFCEGame>(game.shared_from_this(), config, mu);
165
166 // Note that the policies are already inside the game via the correlation
167 // device, mu. So this is a simple wrapper policy that simply follows the
168 // recommendations.
169 EFCETabularPolicy policy(config);
170 return NashConv(*efce_game, policy, true);
171}

257CorrDistInfo CCEDist(const Game& game, const CorrelationDevice& mu, int player,
258 const float prob_cut_threshold,
259 const float action_value_tolerance) {
260 // Check for proper probability distribution.
261 CheckCorrelationDeviceProbDist(mu);
262 CorrDistConfig config;
263 auto cce_game =
264 std::make_shared<CCEGame>(game.shared_from_this(), config, mu);
265
266 CorrDistInfo dist_info{
267 0.0,
268 std::vector<double>(1, std::numeric_limits<double>::quiet_NaN()),
269 std::vector<double>(1, 0),
270 std::vector<double>(1, 0),
271 std::vector<TabularPolicy>(1),
272 {}};
273
274 CCETabularPolicy policy;
275 std::unique_ptr<State> root = cce_game->NewInitialState();
276 TabularBestResponse best_response(*cce_game, player, &policy,
277 prob_cut_threshold, action_value_tolerance);
278 // Do not populate on policy values to save unnecessary computation.
279 // dist_info.on_policy_values[0] = ExpectedReturns(
280 // *root, policy, -1, false)[player];
281 dist_info.best_response_values[0] = best_response.Value(*root);
282 dist_info.best_response_policies[0] = best_response.GetBestResponsePolicy();
283 dist_info.deviation_incentives[0] = std::max(
284 0.0, dist_info.best_response_values[0] - dist_info.on_policy_values[0]);
285 dist_info.dist_value += dist_info.deviation_incentives[0];
286
287 return dist_info;
288}

290CorrDistInfo CCEDist(const Game& game, const CorrelationDevice& mu,
291 const float prob_cut_threshold,
292 const float action_value_tolerance) {
293 // Check for proper probability distribution.
294 CheckCorrelationDeviceProbDist(mu);
295 CorrDistConfig config;
296 auto cce_game =
297 std::make_shared<CCEGame>(game.shared_from_this(), config, mu);
298
299 CorrDistInfo dist_info{0.0,
300 std::vector<double>(game.NumPlayers(), 0),
301 std::vector<double>(game.NumPlayers(), 0),
302 std::vector<double>(game.NumPlayers(), 0),
303 std::vector<TabularPolicy>(game.NumPlayers()),
304 {}};
305
306 // Note: cannot simply call NashConv here as in the other examples. Because
307 // this auxiliary game does not have the "follow" action, it is possible that
308 // a best response against the correlated distribution is *negative* (i.e.
309 // the best deterministic policy is not as good as simply following the
310 // correlated recommendations), but the NashConv function has a check that the
311 // incentive is >= zero, so it would fail.
312
313 CCETabularPolicy policy;
314
315 std::unique_ptr<State> root = cce_game->NewInitialState();
316 for (auto p = Player{0}; p < cce_game->NumPlayers(); ++p) {
317 TabularBestResponse best_response(*cce_game, p, &policy, prob_cut_threshold,
318 action_value_tolerance);
319 dist_info.best_response_values[p] = best_response.Value(*root);
320 dist_info.best_response_policies[p] = best_response.GetBestResponsePolicy();
321 }
322 dist_info.on_policy_values = ExpectedReturns(*root, policy, -1, false);
323 SPIEL_CHECK_EQ(dist_info.best_response_values.size(),
324 dist_info.on_policy_values.size());
325 for (auto p = Player{0}; p < cce_game->NumPlayers(); ++p) {
326 // For reasons indicated in comment at the top of this funciton, we have
327 // max(0, ...) here.
328 dist_info.deviation_incentives[p] = std::max(
329 0.0, dist_info.best_response_values[p] - dist_info.on_policy_values[p]);
330 dist_info.dist_value += dist_info.deviation_incentives[p];
331 }
332 return dist_info;
333}

335CorrDistInfo CEDist(const Game& game, const CorrelationDevice& mu,
336 const float action_value_tolerance) {
337 // Check for proper probability distribution.
338 CheckCorrelationDeviceProbDist(mu);
339 CorrDistConfig config;
340 auto ce_game = std::make_shared<CEGame>(game.shared_from_this(), config, mu);
341
342 CorrDistInfo dist_info{
343 0.0,
344 std::vector<double>(game.NumPlayers(), 0),
345 std::vector<double>(game.NumPlayers(), 0),
346 std::vector<double>(game.NumPlayers(), 0),
347 {},
348 std::vector<std::vector<TabularPolicy>>(game.NumPlayers())};
349
350 CETabularPolicy policy(config);
351
352 // For similar reasons as in CCEDist, we must manually do NashConv.
353
354 std::unique_ptr<State> root = ce_game->NewInitialState();
355 for (auto p = Player{0}; p < ce_game->NumPlayers(); ++p) {
356 TabularBestResponse best_response(*ce_game, p, &policy, -1.0,
357 action_value_tolerance);
358 dist_info.best_response_values[p] = best_response.Value(*root);
359
360 // This policy has all of the conditional ones built in. We have to extract
361 // one per signal by mapping back the info states.
362 TabularPolicy big_br_policy = best_response.GetBestResponsePolicy();
363
364 absl::flat_hash_map<int, TabularPolicy> extracted_policies;
365
366 for (const auto& infostate_and_probs : big_br_policy.PolicyTable()) {
367 std::string full_info_state = infostate_and_probs.first;
368 const size_t idx = full_info_state.find(config.recommendation_delimiter);
369 SPIEL_CHECK_NE(idx, std::string::npos);
370 std::vector<std::string> parts =
371 absl::StrSplit(full_info_state, config.recommendation_delimiter);
372 SPIEL_CHECK_EQ(parts.size(), 2);
373 int signal = -1;
374 SPIEL_CHECK_TRUE(absl::SimpleAtoi(parts[1], &signal));
375 SPIEL_CHECK_GE(signal, 0);
376 extracted_policies[signal].SetStatePolicy(parts[0],
377 infostate_and_probs.second);
378 }
379
380 for (const auto& signal_and_policy : extracted_policies) {
381 dist_info.conditional_best_response_policies[p].push_back(
382 signal_and_policy.second);
383 }
384 }
385
386 dist_info.on_policy_values = ExpectedReturns(*root, policy, -1, false);
387 SPIEL_CHECK_EQ(dist_info.best_response_values.size(),
388 dist_info.on_policy_values.size());
389 for (auto p = Player{0}; p < ce_game->NumPlayers(); ++p) {
390 // For reasons indicated in comment at the top of this funciton, we have
391 // max(0, ...) here.
392 dist_info.deviation_incentives[p] = std::max(
393 0.0, dist_info.best_response_values[p] - dist_info.on_policy_values[p]);
394 dist_info.dist_value += dist_info.deviation_incentives[p];
395 }
396
397 return dist_info;
398}
