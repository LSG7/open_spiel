28// This is an equivalent wrapper to acpc evalHandTables.Cardset.
29// It stores the cards for each color over 16 * 4 bits. The use of a Union
30// allows to access only a specific color (16 bits) using bySuit[color].
31// A uint8_t card is defined by the integer <rank> * MAX_SUITS + <suit>
32class CardSet {
33 public:
34 union CardSetUnion {
35 CardSetUnion() : cards(0) {}
36 uint16_t bySuit[kMaxSuits];
37 uint64_t cards;
38 } cs;
39
40 public:
41 CardSet() : cs() {}
42 CardSet(std::string cardString);
43 CardSet(std::vector<int> cards);
44 // Returns a set containing num_ranks cards per suit for num_suits.
45 CardSet(uint16_t num_suits, uint16_t num_ranks);
46
47 std::string ToString() const;
48 // Returns the cards present in this set in ascending order.
49 std::vector<uint8_t> ToCardArray() const;
50
51 // Add a card, as MAX_RANKS * <suite> + <rank> to the CardSet.
52 void AddCard(uint8_t card);
53 // Toogle (does not remove) the bit associated to `card`.
54 void RemoveCard(uint8_t card);
55 bool ContainsCards(uint8_t card) const;
56
57 int NumCards() const;
58 // Returns the ranking value of this set of cards as evaluated by ACPC.
59 int RankCards() const;
60
61 // Returns all the possible nbCards-subsets of this CardSet.
62 std::vector<CardSet> SampleCards(int nbCards);
63};
