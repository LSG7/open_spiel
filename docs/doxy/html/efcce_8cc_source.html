57std::vector<Action> EFCCEState::LegalActions() const {
58 SPIEL_CHECK_FALSE(IsSimultaneousNode());
59
60 if (IsTerminal()) {
61 return {};
62 } else if (IsChanceNode()) {
63 return LegalChanceOutcomes();
64 }
65
66 if (!HasDefected(CurrentPlayer())) {
67 // If the player has not defected then they have exactly two choices:
68 // follow or defect.
69 return {follow_action_, defect_action_};
70 } else {
71 // Player has defected.. they are on their own.
72 return state_->LegalActions();
73 }
74}

76std::string EFCCEState::InformationStateString(Player player) const {
77 // should look like <infoset string> <delimiter> <defected? true | false>
78 // <recommendation sequence, (excluding current recommendation)>
79 std::string rec_str = absl::StrJoin(recommendation_seq_[player], ",");
80 std::string infoset_str = state_->InformationStateString(player);

82 std::string::npos);

84 HasDefected(player) ? "true " : "false ", rec_str);

112void EFCCEState::DoApplyAction(Action action_id) {
113 if (rec_index_ < 0) {
114 // Pick the joint policy which will provide recommendations.
115 rec_index_ = action_id;
116 SPIEL_CHECK_LT(rec_index_, mu_.size());
117 } else if (state_->IsChanceNode()) {
118 // Regular chance node
119 state_->ApplyAction(action_id);
120 } else {
121 Player cur_player = CurrentPlayer();
122 SPIEL_CHECK_GE(cur_player, 0);
123 SPIEL_CHECK_LT(cur_player, game_->NumPlayers());
124
125 if (!HasDefected(cur_player)) {
126 // Can only submit these two actions.
127 SPIEL_CHECK_TRUE(action_id == follow_action_ ||
128 action_id == defect_action_);
129
130 // Check for defection at this point. This is because the
131 // recommendations
132 Action recommendation = CurRecommendation();
133
134 if (action_id == follow_action_) {
135 // Follow recommendation.
136 std::vector<Action> legal_actions = state_->LegalActions();
137 SPIEL_CHECK_TRUE(absl::c_find(legal_actions, recommendation) !=
138 legal_actions.end());
139 state_->ApplyAction(recommendation);
140 recommendation_seq_[cur_player].push_back(recommendation);
141 } else {
142 // Defect.
143 defected_[cur_player] = 1;
144 }
145
146 } else {
147 // Regular game from here on.
148 state_->ApplyAction(action_id);
149 }
150 }
151}

153ActionsAndProbs EFCCETabularPolicy::GetStatePolicy(const State& state) const {
154 // The best response code has to have a policy defined everywhere when it
155 // builds its initial tree. For the fixed policies, the players will not
156 // defect, so we define a uniform policy in the regions where players have
157 // defected (which will not affect the best responding player, since the
158 // opponents will never reach these regions).
159 const auto* efcce_state = dynamic_cast<const EFCCEState*>(&state);
160 SPIEL_CHECK_TRUE(efcce_state != nullptr);
161 if (efcce_state->HasDefected(state.CurrentPlayer())) {
162 return UniformStatePolicy(state);
163 }
164
165 // Simply returns a fixed policy with prob 1 on the follow action
166 return {{follow_action_, 1.0}, {defect_action_, 0.0}};
167}
