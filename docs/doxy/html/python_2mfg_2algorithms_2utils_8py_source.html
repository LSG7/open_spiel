28class MergedPolicy(policy_std.Policy):
29 """Merge several policies."""
30
31 def __init__(
32 self,
33 game,
34 player_ids,
35 policies: List[policy_std.Policy],
36 weights: List[float],
37 distributions: Optional[List[distribution_std.Distribution]] = None,
38 ):
39 """Initializes the merged policy.
40
41 Args:
42 game: The game to analyze.
43 player_ids: list of player ids for which this policy applies; each should
44 be in the range 0..game.num_players()-1.
45 policies: A `List[policy_std.Policy]` object.
46 weights: A `List[float]` object. They should sum to 1.
47 distributions: A `List[distribution_std.Distribution]` object.
48 """
49 super(MergedPolicy, self).__init__(game, player_ids)
50 self._policies = policies
51 self._distributions = distributions
52 self._weights = weights
53 if distributions is None:
54 distributions = [
55 distribution.DistributionPolicy(game, policy) for policy in policies
56 ]
57 else:
58 assert len(policies) == len(
59 distributions
60 ), f'Length mismatch {len(policies)} != {len(distributions)}'
61 assert len(policies) == len(
62 weights
63 ), f'Length mismatch {len(policies)} != {len(weights)}'
64
65 def action_probabilities(self, state, player_id=None):
66 action_prob = []
67 legal = state.legal_actions()
68 num_legal = len(legal)
69 for a in legal:
70 merged_pi = 0.0
71 norm_merged_pi = 0.0
72 for p, d, w in zip(self._policies, self._distributions, self._weights):
73 merged_pi += w * d(state) * p(state)[a]
74 norm_merged_pi += w * d(state)
75 if norm_merged_pi > 0.0:
76 action_prob.append((a, merged_pi / norm_merged_pi))
77 else:
78 action_prob.append((a, 1.0 / num_legal))
79 return dict(action_prob)

82class MixedDistribution:
83 """Mixes a list of distributions wrt. a list of weights.
84
85 The mixed distribution remains a probability distribution over states.
86
87 Attributes:
88 mus: The state distributions being mixed.
89 weights: The list of weights of each `mus` member.
90 _mus: The state distributions being mixed, post-pruning.
91 _weights: The list of weights of each `mus` member, post-pruning.
92 _tol: Tolerance (`mus` members with weights below tolerance are ignored)
93 _value_str_cache: Cache for value_str calls.
94 """
95
96 def __init__(self, mus, weights, tol=1e-4):
97 """Mixes the distribution.
98
99 Args:
100 mus: List of distributions to mix.
101 weights: List of weights to mix `mus` over.
102 tol: Tolerance (`mus` members with weights below tolerance are ignored)
103 """
104 self.mus = mus
105 self.weights = weights
106 self._tol = tol
107 self._prune()
108 self._value_str_cache = {}
109
110 def _prune(self):
111 self._mus = [mu for w, mu in zip(self.weights, self.mus) if w > self._tol]
112 self._weights = [w for w in self.weights if w > self._tol]
113 self._weights = [w / sum(self._weights) for w in self._weights]
114
115 def value(self, state):
116 """Returns the probability of the distribution on the state.
117
118 Args:
119 state: A `pyspiel.State` object.
120
121 Returns:
122 A `float`.
123 """
124 return sum([
125 weight * mu.value(state) for weight, mu in zip(self._weights, self._mus)
126 ])
127
128 def value_str(self, state_str, default_value=None):
129 """Returns the probability of the distribution on the given state string.
130
131 Args:
132 state_str: A string.
133 default_value: If not None, return this value if the state is not in the
134 support of the distribution.
135
136 Returns:
137 A `float`.
138 """
139 if state_str not in self._value_str_cache:
140 self._value_str_cache[state_str] = sum([
141 weight * mu.value_str(state_str, default_value)
142 for weight, mu in zip(self._weights, self._mus)
143 ])
144 return self._value_str_cache[state_str]
145
146 def __call__(self, state):
147 """Turns the distribution into a callable.
148
149 Args:
150 state: The current state of the game.
151
152 Returns:
153 Float: probability.
154 """
155 return self.value(state)

158def get_exact_value(
159 pi: policy_std.Policy, mu: distribution_std.Distribution, game
160):
161 """Computes the exact value of playing `pi` against distribution `mu`.
162
163 Args:

175def sample_value(
176 pi: policy_std.Policy, mu: distribution_std.Distribution, game
177):
178 """Samples the value of playing `pi` against distribution `mu`.
179
180 Args:
181 pi: A policy object whose value is evaluated against `mu`.
182 mu: A distribution object against which `pi` is evaluated.
183 game: A pyspiel.Game object, the evaluation game.
184
185 Returns:
186 Sampled value of `pi` in `game` against `mu`.
187 """
188 mfg_state = game.new_initial_states()[0]
189 total_reward = 0.0
190 while not mfg_state.is_terminal():
191 if mfg_state.current_player() == pyspiel.PlayerId.CHANCE:
192 action_list, prob_list = zip(*mfg_state.chance_outcomes())
193 action = np.random.choice(action_list, p=prob_list)
194 mfg_state.apply_action(action)
195 elif mfg_state.current_player() == pyspiel.PlayerId.MEAN_FIELD:
196 dist_to_register = mfg_state.distribution_support()
197 dist = [mu.value_str(str_state, 0.0) for str_state in dist_to_register]
198 mfg_state.update_distribution(dist)
199 else:
200 total_reward += mfg_state.rewards()[0]
201 action_prob = pi(mfg_state)
202 action = np.random.choice(
203 list(action_prob.keys()), p=list(action_prob.values())
204 )
205 mfg_state.apply_action(action)
206
207 return total_reward

210def get_nu_values(policies, nu, game):
211 rewards = np.zeros(len(policies))
212 mu = distribution.DistributionPolicy(
