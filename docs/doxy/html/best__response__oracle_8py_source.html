28class BestResponseOracle(optimization_oracle.AbstractOracle):
29 """Oracle using exact best responses to compute BR to policies."""
30
31 def __init__(self,
32 best_response_backend='cpp',
33 game=None,
34 all_states=None,
35 state_to_information_state=None,
36 prob_cut_threshold=-1.0,
37 action_value_tolerance=-1.0,
38 **kwargs):
39 """Init function for the RLOracle.
40
41 Args:
42 best_response_backend: A string (either 'cpp' or 'py'), specifying the
43 best response backend to use (C++ or python, respectively). The cpp
44 backend should be preferred, generally, as it is significantly faster.
45 game: The game on which the optimization process takes place.
46 all_states: The result of calling get_all_states.get_all_states. Cached
47 for improved performance.
48 state_to_information_state: A dict mapping str(state) to
49 state.information_state for every state in the game. Cached for improved
50 performance.
51 prob_cut_threshold: For cpp backend, a partially computed best-response
52 can be computed when using a prob_cut_threshold >= 0.
53 action_value_tolerance: For cpp backend, the max-entropy best-response
54 policy is computed if a non-negative `action_value_tolerance` is used.
55 **kwargs: kwargs
56 """
57 super(BestResponseOracle, self).__init__(**kwargs)
58 self.best_response_backend = best_response_backend
59 if self.best_response_backend == 'cpp':
60 # Should compute all_states and state_to_information_state only once in
61 # the program, as caching them speeds up TabularBestResponse tremendously.
62 self.all_states, self.state_to_information_state = (
63 utils.compute_states_and_info_states_if_none(
64 game, all_states, state_to_information_state))
65
66 policy = openspiel_policy.UniformRandomPolicy(game)
67
68 policy_to_dict = policy_utils.policy_to_dict(
69 policy, game, self.all_states, self.state_to_information_state)
70
71 # pylint: disable=g-complex-comprehension
72 # Cache TabularBestResponse for players, due to their costly construction
73 # TODO(b/140426861): Use a single best-responder once the code supports
74 # multiple player ids.
75 self.best_response_processors = [
76 pyspiel.TabularBestResponse(game, best_responder_id, policy_to_dict,
77 prob_cut_threshold,
78 action_value_tolerance)
79 for best_responder_id in range(game.num_players())
80 ]
81 self.best_responders = [
82 best_response.CPPBestResponsePolicy(
83 game, i_player, policy, self.all_states,
84 self.state_to_information_state,
85 self.best_response_processors[i_player]
86 )
87 for i_player in range(game.num_players())
88 ]
89 # pylint: enable=g-complex-comprehension

91 def __call__(self,
92 game,
93 training_parameters,
94 strategy_sampler=utils.sample_strategy,
95 using_joint_strategies=False,
96 **oracle_specific_execution_kwargs):
97 """Call method for oracle, returns best responses for training_parameters.
98
99 Args:
100 game: The game on which the optimization process takes place.
101 training_parameters: List of list of dicts: one list per player, one dict
102 per selected agent in the pool for each player,
103 each dictionary containing the following fields:
104 - policy: the policy from which to start training.
105 - total_policies: A list of all policy.Policy strategies used for
106 training, including the one for the current player. Either
107 marginalized or joint strategies are accepted.
108 - current_player: Integer representing the current player.
109 - probabilities_of_playing_policies: A list of arrays representing, per
110 player, the probabilities of playing each policy in total_policies for
111 the same player.
112 strategy_sampler: Callable that samples strategies from `total_policies`
113 using `probabilities_of_playing_policies`. It only samples one joint
114 "action" for all players. Implemented to be able to take into account
115 joint probabilities of action.
116 using_joint_strategies: Whether the meta-strategies sent are joint (True)
117 or marginalized.
118 **oracle_specific_execution_kwargs: Other set of arguments, for
119 compatibility purposes. Can for example represent whether to Rectify
120 Training or not.
121
122 Returns:
123 A list of list of OpenSpiel Policy objects representing the expected
124 best response, following the same structure as training_parameters.
125 """
126 new_policies = []
127 for player_parameters in training_parameters:
128 player_policies = []
129 for params in player_parameters:
130 current_player = params['current_player']
131 total_policies = params['total_policies']
132 probabilities_of_playing_policies = params[
133 'probabilities_of_playing_policies']
134 if using_joint_strategies:
135 aggr_policy = utils.aggregate_joint_policies(
136 game, utils.marginal_to_joint(total_policies),
137 probabilities_of_playing_policies.reshape(-1))
138 else:
139 aggr_policy = utils.aggregate_policies(
140 game, total_policies, probabilities_of_playing_policies)
141
142 # This takes as input an aggregate policy, and computes a best response
143 # for current_player at the applicable information states by recursing
144 # through the game tree. At information states involving other players
145 # or chance, the aggr_policy is used to compute the expected value, such
146 # that a best response for current_player can be computed.
147 if self.best_response_backend == 'py':
148 best_resp = best_response.BestResponsePolicy(game, current_player,
149 aggr_policy)
150 else:
151 self.best_response_processors[current_player].set_policy(
152 policy_utils.policy_to_dict(aggr_policy, game, self.all_states,
153 self.state_to_information_state))
154
155 self.best_responders[current_player] = (
156 best_response.CPPBestResponsePolicy(
157 game, current_player, aggr_policy, self.all_states,
158 self.state_to_information_state,
159 self.best_response_processors[current_player]))
160 best_resp = self.best_responders[current_player]
161 player_policies.append(best_resp)
162 new_policies.append(player_policies)
163 return new_policies
