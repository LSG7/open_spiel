24class HistoryTreeNode:
25 """Tree node to build game tree in cfr and do DFS traverse on game tree.
26
27 Attributes:
28 world_state: Current world state representation.
29 reach_probs: Reach probability of tree node for each player. We consider
30 reach probability for chance player, player 1 and player 2.
31 action_probs: Probability of actions taken by each player. We consider
32 actions taken by chance player, player 1 and player 2. Keys of this
33 dictionary are tuples of (action_chance, action_player_1,
34 action_player_2).
35 children: A dictionary from a taken action from this node to the
36 HistoryTreeNode of the child we derive in the game tree by taking an
37 action.
38 """
39
40 def __init__(self, world_state: openspiel_api.WorldState):
41 self.world_state = world_state
42 self.reach_probs = [1.0, 1.0, 1.0]
43 self.action_probs = {}
44 self._value_p1 = 0
45 self.children = {}
46
47 def add_child(self, child_world_state: 'HistoryTreeNode',
48 actions: Tuple[int, int, int]) -> None:
49 """Adds the child world state to dictionary of children of this node."""
50 self.children[actions] = child_world_state
51
52 def get_child(self, actions: Tuple[int, int, int]) -> 'HistoryTreeNode':
53 """Returns a child world state that can be derived from an action."""
54 return self.children[actions]

57class InfoState:
58 """Information state class.
59
60 Attributes:
61 history_nodes: History of game as players play.
62 player: Index of current player.
63 infostate_string: String representation of current informantion state.
64 world_state: Current game world state.
65 children: Children nodes of information states. The keys are actions, and
66 values are dictionary from information state string to information state
67 node.
68 counterfactual_reach_prob: Counterfactural values of reach probability for
69 the current information state.
70 player_reach_prob: Reach probability of information state for the acting
71 player.
72 counterfactual_action_values: Counterfactual values for each action in this
73 information state. This is a dictionary from action to counterfactual
74 value of this action in this information state.
75 counterfactual_value: Counterfactual value of this information state.
76 regret: Regret of each action for all player's actions in this information
77 state.
78 policy: Policy of player in this information state.
79 average_policy: Average policy for all player's actions in this information
80 state.
81 average_policy_weight_sum: Sum of weighted average policy. This is used to
82 normalize average policy and derive policy in this information state.
83 """
84
85 def __init__(self, world_state: openspiel_api.WorldState, player: int,
86 infostate_string: Text):
87 self.history_nodes = []
88 self.player = player
89 self.infostate_string = infostate_string
90 self.world_state = world_state
91 self._actions = world_state.get_actions()
92 self.children = {a: {} for a in self._actions[player]}
93 self.counterfactual_reach_prob = 0.
94 self.player_reach_prob = 0.
95 self.counterfactual_action_values = {}
96 self.counterfactual_value = 0
97 self.regret = {a: 0. for a in self._actions[player]}
98
99 actions_count = len(self._actions[player])
100 self.policy = {
101 a: 1.0 / actions_count for a in world_state.get_actions()[player]
102 }
103
104 self.average_policy = {a: 0. for a in self._actions[player]}
105 self.average_policy_weight_sum = 0.
106
107 def add_history_node(self, history_node: HistoryTreeNode) -> None:
108 """Updates history nodes with a given(last) history node."""
109 self.history_nodes.append(history_node)
110
111 def add_child_infostate(self, action: int,
112 infostate_child: Any) -> None:
113 """Adds child infostate derived from taking an action to self.children."""
114 self.children[action][infostate_child.infostate_string] = infostate_child
115
116 def get_actions(self) -> List[int]:
117 """Returns legal actions in current information state for current player."""
118 return self.history_nodes[0].world_state.get_actions()[self.player]
119
120 def is_terminal(self) -> bool:
121 """Returns True if information state is terminal, False otherwise."""
122 return self.history_nodes[0].world_state.is_terminal()

125class GameTree:
126 """Game tree class to build for CFR-based algorithms.
127
128 Attributes:
129 first_history_node: Root node of game tree.
130 infostate_nodes: List of information state nodes for each player (including

132 all_infostates_map: List of dictionaries (mapping from information state
133 string representation to information state object) for each players
134 (including chance player).
135 """

138 infostate_nodes: List[InfoState],
139 all_infostates_map: List[Dict[str, InfoState]]):

145def build_tree_dfs(
146 world_state: openspiel_api.WorldState,
147 all_infostates_map: List[Dict[str, InfoState]]
148) -> Tuple[HistoryTreeNode, List[InfoState]]:
149 """Builds the game tree by DFS traversal.
150
151 Args:
152 world_state: An openspiel game world state representation that will be the
153 root of game tree.
154 all_infostates_map: List of dictionaries (mapping from information state
155 string representation to information state object) for each players
156 (including chance player). This list will be empty when this function is
157 called and it'll be population during DFS tree traversal.
158
159 Returns:
160 tree_node: Root of the game tree built in DFS traversal.
161 infostate_nodes: List of information state (root) tree node for each player
162 (including chance player).
163 """
164 tree_node = HistoryTreeNode(world_state)
165
166 infostate_nodes = [
167 InfoState(world_state, 1, world_state.get_infostate_string(1)),
168 InfoState(world_state, 1, world_state.get_infostate_string(1)),
169 InfoState(world_state, 2, world_state.get_infostate_string(2))
170 ]
171 for p in [cfr.Players.PLAYER_1, cfr.Players.PLAYER_2]:
172 infostate_string = world_state.get_infostate_string(p)
173 if infostate_string not in all_infostates_map[p]:
174 all_infostates_map[p][infostate_string] = InfoState(
175 world_state, p, infostate_string)
176
177 infostate = all_infostates_map[p][infostate_string]
178 infostate.add_history_node(tree_node)
179
180 infostate_nodes[p] = infostate
181 actions = world_state.get_actions()
182 actions_chance, actions_p1, actions_p2 = actions
183
184 for action_chance in actions_chance:
185 for action_p1 in actions_p1:
186 for action_p2 in actions_p2:
187 child_state = copy.deepcopy(world_state)
188 child_state.apply_actions((action_chance, action_p1, action_p2))
189 child_tree_node, child_infostates = build_tree_dfs(
190 child_state, all_infostates_map)
191
192 tree_node.add_child(child_tree_node,
193 (action_chance, action_p1, action_p2))
194 infostate_nodes[1].add_child_infostate(action_p1, child_infostates[1])
195 infostate_nodes[2].add_child_infostate(action_p2, child_infostates[2])
196
197 return tree_node, infostate_nodes

200def build_game_tree(world_state: openspiel_api.WorldState) -> GameTree:
201 """Builds game tree for CFR-based algorithms.
202
203 Args:
204 world_state: An openspiel game world state representation that will be the
205 root of game tree.
206
207 Returns:
208 Calls GameTree function which returns the following:
209 tree_node: Root of the game tree built in DFS traversal.
210 infostate_nodes: List of information state (root) tree node for each player
211 (including chance player).
212 """
213 all_infostates_map = [{}, {}, {}]
214 first_history_node, infostate_nodes = build_tree_dfs(world_state,
215 all_infostates_map)
216 return GameTree(first_history_node, infostate_nodes, all_infostates_map)
