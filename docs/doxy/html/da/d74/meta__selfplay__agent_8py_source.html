36def _mlp_forwards(mlp_hidden_sizes: List[int]) -> hk.Transformed:
37 """Returns a haiku transformation of the MLP model to be used in optimizer.
38
39 Args:
40 mlp_hidden_sizes: List containing size of linear layers.
41
42 Returns:
43 Haiku transformation of the RNN network.
44 """
45 def forward_fn(inputs):
46 mlp = hk.nets.MLP(mlp_hidden_sizes, activation=jax.nn.relu, name="mlp")
47 return mlp(inputs)
48 return hk.transform(forward_fn)

65 schedule_fn = optax.polynomial_schedule(

69 def get_optimizer_model(self):
70 schedule_fn = self.lr_scheduler(self.learning_rate)
71 opt_init, self.opt_update = optax.chain(
72 optax.scale_by_adam(), optax.scale_by_schedule(schedule_fn),
73 optax.scale(-self.learning_rate))
74 rng = jax.random.PRNGKey(10)

76 loc=0, scale=10., size=(FLAGS.batch_size, 1, FLAGS.num_actions))
77 self.net_params = self._net_init(rng, dummy_input)

93 def train(self):

96
97 def initial_policy(self):
98 x = self.net_apply(self.net_params, None, self.regret_sum)
99 self.last_policy = jax.nn.softmax(x)
100 self.step += 1
101 return self.last_policy
102
103 def next_policy(self, last_values):
104 value = jnp.matmul(self.last_policy, last_values)
105 curren_regret = jnp.transpose(last_values, [0, 2, 1]) - value
106 self.regret_sum += curren_regret
107
108 x = self.net_apply(self.net_params, None, self.regret_sum / (self.step + 1))
109 self.last_policy = jax.nn.softmax(x)
110 self.step += 1
111 return self.last_policy
