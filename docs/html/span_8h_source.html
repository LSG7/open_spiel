17 // span.h

19 //
20 // This header file defines a `Span<T>` type for holding a reference to existing
21 // array data. The `Span` object, much like the `absl::string_view` object,
22 // does not own such data itself, and the data being referenced by the span must
23 // outlive the span itself. Unlike `view` type references, a span can hold a
24 // reference to mutable data (and can mutate it for underlying types of
25 // non-const T.) A span provides a lightweight way to pass a reference to such
26 // data.
27 //
28 // Additionally, this header file defines `MakeSpan()` and `MakeConstSpan()`
29 // factory functions, for clearly creating spans of type `Span<T>` or read-only
30 // `Span<const T>` when such types may be difficult to identify due to issues
31 // with implicit conversion.
32 //
33 // The C++20 draft standard includes a `std::span` type. As of June 2020, the
34 // differences between `absl::Span` and `std::span` are:
35 // * `absl::Span` has `operator==` (which is likely a design bug,
36 // per https://abseil.io/blog/20180531-regular-types)
37 // * `absl::Span` has the factory functions `MakeSpan()` and
38 // `MakeConstSpan()`
39 // * bounds-checked access to `absl::Span` is accomplished with `at()`
40 // * `absl::Span` has compiler-provided move and copy constructors and
41 // assignment. This is due to them being specified as `constexpr`, but that
42 // implies const in C++11.
43 // * A read-only `absl::Span<const T>` can be implicitly constructed from an
44 // initializer list.
45 // * `absl::Span` has no `bytes()`, `size_bytes()`, `as_bytes()`, or
46 // `as_mutable_bytes()` methods
47 // * `absl::Span` has no static extent template parameter, nor constructors
48 // which exist only because of the static extent parameter.
49 // * `absl::Span` has an explicit mutable-reference constructor
50 //
51 // For more information, see the class comments below.

74 //------------------------------------------------------------------------------
75 // Span
76 //------------------------------------------------------------------------------
77 //
78 // A `Span` is an "array reference" type for holding a reference of contiguous
79 // array data; the `Span` object does not and cannot own such data itself. A
80 // span provides an easy way to provide overloads for anything operating on
81 // contiguous sequences without needing to manage pointers and array lengths
82 // manually.
83 
84 // A span is conceptually a pointer (ptr) and a length (size) into an already
85 // existing array of contiguous memory; the array it represents references the
86 // elements "ptr[0] .. ptr[size-1]". Passing a properly-constructed `Span`
87 // instead of raw pointers avoids many issues related to index out of bounds
88 // errors.
89 //
90 // Spans may also be constructed from containers holding contiguous sequences.
91 // Such containers must supply `data()` and `size() const` methods (e.g
92 // `std::vector<T>`, `absl::InlinedVector<T, N>`). All implicit conversions to
93 // `absl::Span` from such containers will create spans of type `const T`;
94 // spans which can mutate their values (of type `T`) must use explicit
95 // constructors.
96 //
97 // A `Span<T>` is somewhat analogous to an `absl::string_view`, but for an array
98 // of elements of type `T`, and unlike an `absl::string_view`, a span can hold a
99 // reference to mutable data. A user of `Span` must ensure that the data being
100 // pointed to outlives the `Span` itself.
101 //
102 // You can construct a `Span<T>` in several ways:
103 //
104 // * Explicitly from a reference to a container type
105 // * Explicitly from a pointer and size
106 // * Implicitly from a container type (but only for spans of type `const T`)
107 // * Using the `MakeSpan()` or `MakeConstSpan()` factory functions.
108 //
109 // Examples:
110 //
111 // // Construct a Span explicitly from a container:
112 // std::vector<int> v = {1, 2, 3, 4, 5};
113 // auto span = absl::Span<const int>(v);
114 //
115 // // Construct a Span explicitly from a C-style array:
116 // int a[5] = {1, 2, 3, 4, 5};
117 // auto span = absl::Span<const int>(a);
118 //
119 // // Construct a Span implicitly from a container
120 // void MyRoutine(absl::Span<const int> a) {
121 // ...
122 // }
123 // std::vector v = {1,2,3,4,5};
124 // MyRoutine(v) // convert to Span<const T>
125 //
126 // Note that `Span` objects, in addition to requiring that the memory they
127 // point to remains alive, must also ensure that such memory does not get
128 // reallocated. Therefore, to avoid undefined behavior, containers with
129 // associated spans should not invoke operations that may reallocate memory
130 // (such as resizing) or invalidate iterators into the container.
131 //
132 // One common use for a `Span` is when passing arguments to a routine that can
133 // accept a variety of array types (e.g. a `std::vector`, `absl::InlinedVector`,
134 // a C-style array, etc.). Instead of creating overloads for each case, you
135 // can simply specify a `Span` as the argument to such a routine.
136 //
137 // Example:
138 //
139 // void MyRoutine(absl::Span<const int> a) {
140 // ...
141 // }
142 //
143 // std::vector v = {1,2,3,4,5};
144 // MyRoutine(v);
145 //
146 // absl::InlinedVector<int, 4> my_inline_vector;
147 // MyRoutine(my_inline_vector);
148 //
149 // // Explicit constructor from pointer,size
150 // int* my_array = new int[10];
151 // MyRoutine(absl::Span<const int>(my_array, 10));
152 template <typename T>
153 class Span {
154  private:
155  // Used to determine whether a Span can be constructed from a container of
156  // type C.
157  template <typename C>
158  using EnableIfConvertibleFrom =
159  typename std::enable_if<span_internal::HasData<T, C>::value &&
160  span_internal::HasSize<C>::value>::type;

162  // Used to SFINAE-enable a function when the slice elements are const.
163  template <typename U>
164  using EnableIfValueIsConst =
165  typename std::enable_if<std::is_const<T>::value, U>::type;

168  template <typename U>
169  using EnableIfValueIsMutable =
170  typename std::enable_if<!std::is_const<T>::value, U>::type;

172  public:
173  using element_type = T;
174  using value_type = absl::remove_cv_t<T>;
175  using pointer = T*;
176  using const_pointer = const T*;
177  using reference = T&;
178  using const_reference = const T&;
179  using iterator = pointer;
180  using const_iterator = const_pointer;
181  using reverse_iterator = std::reverse_iterator<iterator>;
182  using const_reverse_iterator = std::reverse_iterator<const_iterator>;
183  using size_type = size_t;

197  // Explicit reference constructor for a mutable `Span<T>` type. Can be
198  // replaced with MakeSpan() to infer the type parameter.
199  template <typename V, typename = EnableIfConvertibleFrom<V>,
200  typename = EnableIfValueIsMutable<V>,
201  typename = span_internal::EnableIfNotIsView<V>>
202  explicit Span(
203  V& v
204  ABSL_ATTRIBUTE_LIFETIME_BOUND) noexcept // NOLINT(runtime/references)
205  : Span(span_internal::GetData(v), v.size()) {}
206 
207  // Implicit reference constructor for a read-only `Span<const T>` type
208  template <typename V, typename = EnableIfConvertibleFrom<V>,
209  typename = EnableIfValueIsConst<V>,
210  typename = span_internal::EnableIfNotIsView<V>>
211  constexpr Span(
212  const V& v
213  ABSL_ATTRIBUTE_LIFETIME_BOUND) noexcept // NOLINT(runtime/explicit)
214  : Span(span_internal::GetData(v), v.size()) {}

216  // Overloads of the above two functions that are only enabled for view types.
217  // This is so we can drop the ABSL_ATTRIBUTE_LIFETIME_BOUND annotation. These
218  // overloads must be made unique by using a different template parameter list
219  // (hence the = 0 for the IsView enabler).
220  template <typename V, typename = EnableIfConvertibleFrom<V>,
221  typename = EnableIfValueIsMutable<V>,
222  span_internal::EnableIfIsView<V> = 0>
223  explicit Span(V& v) noexcept // NOLINT(runtime/references)
224  : Span(span_internal::GetData(v), v.size()) {}
225  template <typename V, typename = EnableIfConvertibleFrom<V>,
226  typename = EnableIfValueIsConst<V>,
227  span_internal::EnableIfIsView<V> = 0>
228  constexpr Span(const V& v) noexcept // NOLINT(runtime/explicit)
229  : Span(span_internal::GetData(v), v.size()) {}

231  // Implicit constructor from an initializer list, making it possible to pass a
232  // brace-enclosed initializer list to a function expecting a `Span`. Such
233  // spans constructed from an initializer list must be of type `Span<const T>`.
234  //
235  // void Process(absl::Span<const int> x);
236  // Process({1, 2, 3});
237  //
238  // Note that as always the array referenced by the span must outlive the span.
239  // Since an initializer list constructor acts as if it is fed a temporary
240  // array (cf. C++ standard [dcl.init.list]/5), it's safe to use this
241  // constructor only when the `std::initializer_list` itself outlives the span.
242  // In order to meet this requirement it's sufficient to ensure that neither
243  // the span nor a copy of it is used outside of the expression in which it's
244  // created:
245  //
246  // // Assume that this function uses the array directly, not retaining any
247  // // copy of the span or pointer to any of its elements.
248  // void Process(absl::Span<const int> ints);
249  //
250  // // Okay: the std::initializer_list<int> will reference a temporary array
251  // // that isn't destroyed until after the call to Process returns.
252  // Process({ 17, 19 });
253  //
254  // // Not okay: the storage used by the std::initializer_list<int> is not
255  // // allowed to be referenced after the first line.
256  // absl::Span<const int> ints = { 17, 19 };
257  // Process(ints);
258  //
259  // // Not okay for the same reason as above: even when the elements of the
260  // // initializer list expression are not temporaries the underlying array
261  // // is, so the initializer list must still outlive the span.
262  // const int foo = 17;
263  // absl::Span<const int> ints = { foo };
264  // Process(ints);
265  //
266  template <typename LazyT = T,
267  typename = EnableIfValueIsConst<LazyT>>
268  Span(std::initializer_list<value_type> v
269  ABSL_ATTRIBUTE_LIFETIME_BOUND) noexcept // NOLINT(runtime/explicit)
270  : Span(v.begin(), v.size()) {}

304  //
305  // Returns a reference to the i'th element of this span.
306  constexpr reference at(size_type i) const {
307  return ABSL_PREDICT_TRUE(i < size()) //
308  ? *(data() + i)
309  : (base_internal::ThrowStdOutOfRange(
310  "Span::at failed bounds check"),

315  //
316  // Returns a reference to the first element of this span. The span must not

319  return ABSL_HARDENING_ASSERT(size() > 0), *data();

327  return ABSL_HARDENING_ASSERT(size() > 0), *(data() + size() - 1);

388  // Span::remove_prefix()
389  //
390  // Removes the first `n` elements from the span.
391  void remove_prefix(size_type n) noexcept {
392  ABSL_HARDENING_ASSERT(size() >= n);
393  ptr_ += n;
394  len_ -= n;

397  // Span::remove_suffix()
398  //
399  // Removes the last `n` elements from the span.

401  ABSL_HARDENING_ASSERT(size() >= n);
402  len_ -= n;

405  // Span::subspan()
406  //
407  // Returns a `Span` starting at element `pos` and of length `len`. Both `pos`
408  // and `len` are of type `size_type` and thus non-negative. Parameter `pos`
409  // must be <= size(). Any `len` value that points past the end of the span
410  // will be trimmed to at most size() - `pos`. A default `len` value of `npos`
411  // ensures the returned subspan continues until the end of the span.
412  //
413  // Examples:
414  //
415  // std::vector<int> vec = {10, 11, 12, 13};
416  // absl::MakeSpan(vec).subspan(1, 2); // {11, 12}
417  // absl::MakeSpan(vec).subspan(2, 8); // {12, 13}
418  // absl::MakeSpan(vec).subspan(1); // {11, 12, 13}
419  // absl::MakeSpan(vec).subspan(4); // {}
420  // absl::MakeSpan(vec).subspan(5); // throws std::out_of_range
421  constexpr Span subspan(size_type pos = 0, size_type len = npos) const {
422  return (pos <= size())
423  ? Span(data() + pos, (std::min)(size() - pos, len))
424  : (base_internal::ThrowStdOutOfRange("pos > size()"), Span());
425  }
426 
427  // Span::first()
428  //
429  // Returns a `Span` containing first `len` elements. Parameter `len` is of
430  // type `size_type` and thus non-negative. `len` value must be <= size().
431  //
432  // Examples:
433  //
434  // std::vector<int> vec = {10, 11, 12, 13};
435  // absl::MakeSpan(vec).first(1); // {10}
436  // absl::MakeSpan(vec).first(3); // {10, 11, 12}
437  // absl::MakeSpan(vec).first(5); // throws std::out_of_range
438  constexpr Span first(size_type len) const {
439  return (len <= size())
440  ? Span(data(), len)
441  : (base_internal::ThrowStdOutOfRange("len > size()"), Span());
442  }
443 
444  // Span::last()
445  //
446  // Returns a `Span` containing last `len` elements. Parameter `len` is of
447  // type `size_type` and thus non-negative. `len` value must be <= size().
448  //
449  // Examples:
450  //
451  // std::vector<int> vec = {10, 11, 12, 13};
452  // absl::MakeSpan(vec).last(1); // {13}
453  // absl::MakeSpan(vec).last(3); // {11, 12, 13}
454  // absl::MakeSpan(vec).last(5); // throws std::out_of_range
455  constexpr Span last(size_type len) const {
456  return (len <= size())
457  ? Span(size() - len + data(), len)
458  : (base_internal::ThrowStdOutOfRange("len > size()"), Span());
459  }

468  private:
469  pointer ptr_;
470  size_type len_;

646 // MakeSpan()
647 //
648 // Constructs a mutable `Span<T>`, deducing `T` automatically from either a
649 // container or pointer+size.
650 //
651 // Because a read-only `Span<const T>` is implicitly constructed from container
652 // types regardless of whether the container itself is a const container,
653 // constructing mutable spans of type `Span<T>` from containers requires
654 // explicit constructors. The container-accepting version of `MakeSpan()`
655 // deduces the type of `T` by the constness of the pointer received from the
656 // container's `data()` member. Similarly, the pointer-accepting version returns
657 // a `Span<const T>` if `T` is `const`, and a `Span<T>` otherwise.
658 //
659 // Examples:
660 //
661 // void MyRoutine(absl::Span<MyComplicatedType> a) {
662 // ...
663 // };
664 // // my_vector is a container of non-const types
665 // std::vector<MyComplicatedType> my_vector;
666 //
667 // // Constructing a Span implicitly attempts to create a Span of type
668 // // `Span<const T>`
669 // MyRoutine(my_vector); // error, type mismatch
670 //
671 // // Explicitly constructing the Span is verbose
672 // MyRoutine(absl::Span<MyComplicatedType>(my_vector));
673 //
674 // // Use MakeSpan() to make an absl::Span<T>
675 // MyRoutine(absl::MakeSpan(my_vector));
676 //
677 // // Construct a span from an array ptr+size
678 // absl::Span<T> my_span() {
679 // return absl::MakeSpan(&array[0], num_elements_);
680 // }
681 //
682 template <int&... ExplicitArgumentBarrier, typename T>
683 constexpr Span<T> MakeSpan(T* ptr, size_t size) noexcept {
684  return Span<T>(ptr, size);
685 }
686 
687 template <int&... ExplicitArgumentBarrier, typename T>
688 Span<T> MakeSpan(T* begin, T* end) noexcept {
689  return ABSL_HARDENING_ASSERT(begin <= end),
690  Span<T>(begin, static_cast<size_t>(end - begin));
691 }

704 // MakeConstSpan()
705 //
706 // Constructs a `Span<const T>` as with `MakeSpan`, deducing `T` automatically,
707 // but always returning a `Span<const T>`.
708 //
709 // Examples:
710 //
711 // void ProcessInts(absl::Span<const int> some_ints);
712 //
713 // // Call with a pointer and size.
714 // int array[3] = { 0, 0, 0 };
715 // ProcessInts(absl::MakeConstSpan(&array[0], 3));
716 //
717 // // Call with a [begin, end) pair.
718 // ProcessInts(absl::MakeConstSpan(&array[0], &array[3]));
719 //
720 // // Call directly with an array.
721 // ProcessInts(absl::MakeConstSpan(array));
722 //
723 // // Call with a contiguous container.
724 // std::vector<int> some_ints = ...;
725 // ProcessInts(absl::MakeConstSpan(some_ints));
726 // ProcessInts(absl::MakeConstSpan(std::vector<int>{ 0, 0, 0 }));
727 //
728 template <int&... ExplicitArgumentBarrier, typename T>
729 constexpr Span<const T> MakeConstSpan(T* ptr, size_t size) noexcept {
730  return Span<const T>(ptr, size);
731 }
