29def _uniform_policy(state):
30 legal_actions = state.legal_actions()
31 return [(action, 1.0 / len(legal_actions)) for action in legal_actions]

34def _callable_tabular_policy(tabular_policy):
35 """Turns a tabular policy into a callable.
36
37 Args:
38 tabular_policy: A dictionary mapping information state key to a dictionary
39 of action probabilities (action -> prob).
40
41 Returns:
42 A function `state` -> list of (action, prob)
43 """
44
45 def wrap(state):
46 infostate_key = state.information_state_string(state.current_player())
47 assert infostate_key in tabular_policy
48 ap_list = []
49 for action in state.legal_actions():
50 assert action in tabular_policy[infostate_key]
51 ap_list.append((action, tabular_policy[infostate_key][action]))
52 return ap_list
53
54 return wrap

75def _full_best_response_policy(br_infoset_dict):
76 """Turns a dictionary of best response action selections into a full policy.
77
78 Args:
79 br_infoset_dict: A dictionary mapping information state to a best response
80 action.
81
82 Returns:
83 A function `state` -> list of (action, prob)
84 """
85
86 def wrap(state):
87 infostate_key = state.information_state_string(state.current_player())
88 br_action = br_infoset_dict[infostate_key]
89 ap_list = []
90 for action in state.legal_actions():
91 ap_list.append((action, 1.0 if action == br_action else 0.0))
92 return ap_list
93
94 return wrap

97def _policy_dict_at_state(callable_policy, state):
98 """Turns a policy function into a dictionary at a specific state.
99
100 Args:
101 callable_policy: A function from `state` -> lis of (action, prob),
102 state: the specific state to extract the policy from.
103
104 Returns:
105 A dictionary of action -> prob at this state.
106 """
107
108 infostate_policy_list = callable_policy(state)
109 infostate_policy = {}
110 for ap in infostate_policy_list:
111 infostate_policy[ap[0]] = ap[1]
112 return infostate_policy

115class XFPSolver(object):
116 """An implementation of extensive-form fictitious play (XFP).
117
118 XFP is Algorithm 1 in (Heinrich, Lanctot, and Silver, 2015, "Fictitious
119 Self-Play in Extensive-Form Games"). Refer to the paper for details:
120 http://mlanctot.info/files/papers/icml15-fsp.pdf.
121 """
122
123 def __init__(self, game, save_oracles=False):
124 """Initialize the XFP solver.
125
126 Arguments:
127 game: the open_spiel game object.
128 save_oracles: a boolean, indicating whether or not to save all the BR
129 policies along the way (including the initial uniform policy). This
130 could take up some space, and is only used when generating the meta-game
131 for analysis.
132 """
133
134 self._game = game
135 self._num_players = self._game.num_players()
136
137 # A set of callables that take in a state and return a list of
138 # (action, probability) tuples.
139 self._oracles = [] if save_oracles else None
140
141 # A set of callables that take in a state and return a list of
142 # (action, probability) tuples.
143 self._policies = []
144 for _ in range(self._num_players):
145 self._policies.append(_uniform_policy)
146 if save_oracles:
147 self._oracles.append([_uniform_policy])
148
149 self._best_responses = [None] * self._num_players
150 self._iterations = 0
151 self._delta_tolerance = 1e-10
152 self._average_policy_tables = []
153
154 def average_policy_tables(self):
155 """Returns a dictionary of information state -> dict of action -> prob.
156
157 This is a joint policy (policy for all players).
158 """
159 return self._average_policy_tables
160
161 def average_policy(self):
162 """Returns the current average joint policy (policy for all players)."""
163 return JointPolicy(self._game, self._policies)
164
165 def iteration(self):
166 self._iterations += 1
167 self.compute_best_responses()
168 self.update_average_policies()
169
170 def compute_best_responses(self):
171 """Updates self._oracles to hold best responses for each player."""
172 for i in range(self._num_players):
173 # Compute a best response policy to pi_{-i}.
174 # First, construct pi_{-i}.
175 joint_policy = self.average_policy()
176 br_info = exploitability.best_response(self._game,
177 joint_policy.to_tabular(), i)
178 full_br_policy = _full_best_response_policy(
179 br_info["best_response_action"])
180 self._best_responses[i] = full_br_policy
181 if self._oracles is not None:
182 self._oracles[i].append(full_br_policy)
183
184 def update_average_policies(self):
185 """Update the average policies given the newly computed best response."""
186
187 br_reach_probs = np.ones(self._num_players)
188 avg_reach_probs = np.ones(self._num_players)
189 self._average_policy_tables = [{} for _ in range(self._num_players)]
190 self._recursively_update_average_policies(self._game.new_initial_state(),
191 avg_reach_probs, br_reach_probs)
192 for i in range(self._num_players):
193 self._policies[i] = _callable_tabular_policy(
194 self._average_policy_tables[i])
195
196 def _recursively_update_average_policies(self, state, avg_reach_probs,
197 br_reach_probs):
198 """Recursive implementation of the average strategy update."""
199
200 if state.is_terminal():
201 return
202 elif state.is_chance_node():
203 for action, _ in state.chance_outcomes():
204 new_state = state.clone()
205 new_state.apply_action(action)
206 self._recursively_update_average_policies(new_state, avg_reach_probs,
207 br_reach_probs)
208 else:
209 player = state.current_player()
210 avg_policy = _policy_dict_at_state(self._policies[player], state)
211 br_policy = _policy_dict_at_state(self._best_responses[player], state)
212 legal_actions = state.legal_actions()
213 infostate_key = state.information_state_string(player)
214 # First traverse the subtrees.
215 for action in legal_actions:
216 assert action in br_policy
217 assert action in avg_policy
218 new_state = state.clone()
219 new_state.apply_action(action)
220 new_avg_reach = np.copy(avg_reach_probs)
221 new_avg_reach[player] *= avg_policy[action]
222 new_br_reach = np.copy(br_reach_probs)
223 new_br_reach[player] *= br_policy[action]
224 self._recursively_update_average_policies(new_state, new_avg_reach,
225 new_br_reach)
226 # Now, do the updates.
227 if infostate_key not in self._average_policy_tables[player]:
228 alpha = 1 / (self._iterations + 1)
229 self._average_policy_tables[player][infostate_key] = {}
230 pr_sum = 0.0
231 for action in legal_actions:
232 pr = (
233 avg_policy[action] + (alpha * br_reach_probs[player] *
234 (br_policy[action] - avg_policy[action])) /
235 ((1.0 - alpha) * avg_reach_probs[player] +
236 alpha * br_reach_probs[player]))
237 self._average_policy_tables[player][infostate_key][action] = pr
238 pr_sum += pr
239 assert (1.0 - self._delta_tolerance <= pr_sum <=
240 1.0 + self._delta_tolerance)

242 def sample_episode(self, state, policies):
243 """Samples an episode according to the policies, starting from state.
244
245 Args:
246 state: Pyspiel state representing the current state.
247 policies: List of policy representing the policy executed by each player.
248
249 Returns:
250 The result of the call to returns() of the final state in the episode.
251 Meant to be a win/loss integer.
252 """
253
254 if state.is_terminal():
255 return np.array(state.returns(), dtype=np.float32)
256 elif state.is_chance_node():
257 outcomes = []
258 probs = []
259 for action, prob in state.chance_outcomes():
260 outcomes.append(action)
261 probs.append(prob)
262 outcome = np.random.choice(outcomes, p=probs)
263 state.apply_action(outcome)
264 return self.sample_episode(state, policies)
265 else:
266 player = state.current_player()
267 state_policy = _policy_dict_at_state(policies[player], state)
268 actions = []
269 probs = []
270 for action in state_policy:
271 actions.append(action)
272 probs.append(state_policy[action])
273 action = np.random.choice(actions, p=probs)
274 state.apply_action(action)
275 return self.sample_episode(state, policies)

277 def sample_episodes(self, policies, num):
278 """Samples episodes and averages their returns.
279
280 Args:
281 policies: A list of policies representing the policies executed by each
282 player.
283 num: Number of episodes to execute to estimate average return of policies.
284
285 Returns:
286 Average episode return over num episodes.
287 """
288
289 totals = np.zeros(self._num_players)
290 for _ in range(num):
291 totals += self.sample_episode(self._game.new_initial_state(), policies)
292 return totals / num

294 def get_empirical_metagame(self, sims_per_entry, seed=None):
295 """Gets a meta-game tensor of utilities from episode samples.
296
297 The tensor is a cross-table of all the saved oracles and initial uniform
298 policy.
299
300 Args:
301 sims_per_entry: number of simulations (episodes) to perform per entry in
302 the tables, i.e. each is a crude Monte Carlo estimate
303 seed: the seed to set for random sampling, for reproducibility
304
305 Returns:
306 the K^n (KxKx...K, with dimension n) meta-game tensor where n is the
307 number of players and K is the number of strategies (one more than the
308 number of iterations of fictitious play since the initial uniform
309 policy is included).
310 """
311
312 if seed is not None:
313 np.random.seed(seed=seed)
314 assert self._oracles is not None
315 num_strategies = len(self._oracles[0])
316 # Each metagame will be (num_strategies)^self._num_players.
317 # There are self._num_player metagames, one per player.
318 meta_games = []
319 for _ in range(self._num_players):
320 shape = [num_strategies] * self._num_players
321 meta_game = np.ndarray(shape=shape, dtype=np.float32)
322 meta_games.append(meta_game)
323 for coord in itertools.product(
324 range(num_strategies), repeat=self._num_players):
325 policies = []
326 for i in range(self._num_players):
327 iteration = coord[i]
328 policies.append(self._oracles[i][iteration])
329 utility_estimates = self.sample_episodes(policies, sims_per_entry)
330 for i in range(self._num_players):
331 meta_games[i][coord] = utility_estimates[i]
332 return meta_games
