133 std::string InformationStateTensorToString(Player player,
134  const std::vector<float>& tensor) {
135  PassDir pass_dir;
136  std::array<absl::optional<Player>, kNumCards> dealt_hand;
137  std::array<absl::optional<Player>, kNumCards> current_hand;
138  std::vector<int> passed_cards;
139  std::vector<int> received_cards;
140  std::vector<int> points;
141  std::array<Trick, kNumTricks> tricks;
142 
143  auto ptr = tensor.begin();
144  // Pass dir
145  for (int i = 0; i < kNumPlayers; ++i) {
146  if (ptr[i] == 1) {
147  pass_dir = static_cast<PassDir>(i);
148  break;
149  }
150  }
151  ptr += kNumPlayers;
152  // Dealt hand
153  for (int i = 0; i < kNumCards; ++i) {
154  if (ptr[i] == 1) dealt_hand[i] = player;
155  }
156  ptr += kNumCards;
157  // Passed cards
158  for (int i = 0; i < kNumCards; ++i) {
159  if (ptr[i] == 1) passed_cards.push_back(i);
160  }
161  ptr += kNumCards;
162  // Received cards
163  for (int i = 0; i < kNumCards; ++i) {
164  if (ptr[i] == 1) received_cards.push_back(i);
165  }
166  ptr += kNumCards;
167  // Current hand
168  for (int i = 0; i < kNumCards; ++i) {
169  if (ptr[i] == 1) current_hand[i] = player;
170  }
171  ptr += kNumCards;
172  // Points
173  for (int i = 0; i < kNumPlayers; ++i) {
174  int player_score = kPointsForJD;
175  for (int j = 0; j < kMaxScore; ++j) {
176  if (ptr[j] == 1) ++player_score;
177  }
178  points.push_back(player_score);
179  ptr += kMaxScore;
180  }
181  // Trick history
182  Player leader;
183  int num_cards_played = 0;
184  for (int trick = 0; trick < kNumTricks; ++trick) {
185  leader = kInvalidPlayer;
186  for (int i = 0; i < kNumPlayers * kNumCards; ++i) {
187  if (ptr[i] == 1) {
188  leader = i / kNumCards;
189  // jd_bonus is not relevant for our purposes, set to false.
190  tricks[trick] = Trick(leader, i % kNumCards, false);
191  ++num_cards_played;
192  break;
193  }
194  }
195  if (leader != kInvalidPlayer) {
196  ptr += (leader + 1) * kNumCards;
197  for (int i = 0; i < kNumPlayers - 1; ++i) {
198  for (int j = 0; j < kNumCards; ++j) {
199  if (ptr[j] == 1) {
200  tricks[trick].Play((leader + i + 1) % kNumPlayers, j);
201  ++num_cards_played;
202  }
203  }
204  ptr += kNumCards;
205  }
206  ptr += (kNumPlayers - std::max(leader, 0) - 1) * kNumCards;
207  } else {
208  ptr += kTrickTensorSize;
209  break;
210  }
211  }
212  // Now build InformationStateString.
213  std::string rv = "Pass Direction: ";
214  absl::StrAppend(&rv, pass_dir_str[static_cast<int>(pass_dir)], "\n\n");
215  absl::StrAppend(&rv, "Hand: \n");
216  std::array<std::string, kNumSuits> cards;
217  for (int suit = 0; suit < kNumSuits; ++suit) {
218  cards[suit].push_back(kSuitChar[suit]);
219  cards[suit].push_back(' ');
220  bool is_void = true;
221  for (int rank = kNumCardsPerSuit - 1; rank >= 0; --rank) {
222  if (player == current_hand[Card(Suit(suit), rank)]) {
223  cards[suit].push_back(kRankChar[rank]);
224  is_void = false;
225  }
226  }
227  if (is_void) absl::StrAppend(&cards[suit], "none");
228  }
229  for (int suit = kNumSuits - 1; suit >= 0; --suit)
230  absl::StrAppend(&rv, cards[suit], "\n");
231 
232  if (!passed_cards.empty()) {
233  absl::StrAppend(&rv, "\nPassed Cards: ");
234  for (int card : passed_cards) {
235  absl::StrAppend(&rv, CardString(card), " ");
236  }
237  absl::StrAppend(&rv, "\n");
238  }
239  if (!received_cards.empty()) {
240  absl::StrAppend(&rv, "\nReceived Cards: ");
241  for (int card : received_cards) {
242  absl::StrAppend(&rv, CardString(card), " ");
243  }
244  absl::StrAppend(&rv, "\n");
245  }
246  if (num_cards_played > 0) {
247  absl::StrAppend(&rv, "\nTricks:");
248  absl::StrAppend(&rv, "\nN E S W N E S");
249  for (int i = 0; i <= (num_cards_played - 1) / kNumPlayers; ++i) {
250  absl::StrAppend(&rv, "\n", std::string(3 * tricks[i].Leader(), ' '));
251  for (auto card : tricks[i].Cards()) {
252  absl::StrAppend(&rv, CardString(card), " ");
253  }
254  }
255  absl::StrAppend(&rv, "\n\nPoints:");
256  for (int i = 0; i < kNumPlayers; ++i) {
257  absl::StrAppend(&rv, "\n", DirString(i), ": ", points[i]);
258  }
259  }
260  return rv;
261 }

263 // Build InformationStateString from InformationStateTensor and check that it
264 // is equal to state->InformationStateString(player).
265 void InformationStateTensorTest(int num_games = 100) {
266  std::mt19937 rng(time(0));
267  GameParameters params;
268  params["jd_bonus"] = GameParameter(true);
269  std::shared_ptr<const open_spiel::Game> game =
270  open_spiel::LoadGame("hearts", params);
271  for (int i = 0; i < num_games; ++i) {
272  std::unique_ptr<open_spiel::State> state = game->NewInitialState();
273  while (!state->IsTerminal()) {
274  if (state->IsChanceNode()) {
275  std::vector<std::pair<open_spiel::Action, double>> outcomes =
276  state->ChanceOutcomes();
277  open_spiel::Action action =
278  open_spiel::SampleAction(outcomes, rng).first;
279  state->ApplyAction(action);
280  } else {
281  auto player = state->CurrentPlayer();
282  auto infostate = state->InformationStateTensor(player);
283 
284  std::string infostate_string = state->InformationStateString(player);
285  std::string rebuilt_infostate_string =
286  InformationStateTensorToString(player, infostate);
287  SPIEL_CHECK_EQ(infostate_string, rebuilt_infostate_string);
288 
289  std::vector<open_spiel::Action> actions = state->LegalActions();
290  std::uniform_int_distribution<> dis(0, actions.size() - 1);
291  auto action = actions[dis(rng)];
292  state->ApplyAction(action);
293  }
294  }
295  }
296 }
