57std::string DouDizhuState::ActionToString(Player player, Action action) const {
58 if (player == kChancePlayerId) {
59 if (action < kDealingActionBase) {
60 return absl::StrCat("Decide first card up position ", action);
61 } else if (action < kDealingActionBase + kNumCards) {
62 return absl::StrCat("Deal ", CardString(action - kDealingActionBase));
63 } else {
64 SpielFatalError(
65 absl::StrFormat("Non valid ID %d for chance player", action));
66 }
67 }
68
69 if (action == kPass) {
70 return "Pass";
71 } else if (action > kPass && action < kPlayActionBase) {
72 return absl::StrCat("Bid ", action - kBiddingActionBase);
73 } else if (action >= kPlayActionBase && action <= kRocketActionBase) {
74 // For aiplane combinations, need special treatment to resolve ambiguity
75 if (action >= kAirplaneWithSoloActionBase && action < kBombActionBase) {
76 return FormatAirplaneCombHand(action);
77 }
78 return FormatSingleHand(ActionToHand(action));
79 } else {
80 SpielFatalError("Non valid action ID!");
81 }
82}

130std::array<std::string, kNumRanks> FormatHand(
131 int player, bool mark_voids,
132 const std::array<std::array<int, kNumRanks>, kNumPlayers>& deal) {
133 std::array<std::string, kNumRanks> cards{};
134 for (int rank = 0; rank < kNumRanks - 2; ++rank) {
135 bool is_void = true;
136 for (int i = 0; i < deal[player][rank]; ++i) {
137 cards[rank].push_back(kRankChar[rank]);
138 is_void = false;
139 }
140 if (is_void && mark_voids) absl::StrAppend(&cards[rank], "none");
141 }
142 if (deal[player][kNumRanks - 2])
143 absl::StrAppend(&cards[kNumRanks - 2], "(BWJ)");
144 else if (mark_voids)
145 absl::StrAppend(&cards[kNumRanks - 2], "none");
146
147 if (deal[player][kNumRanks - 1])
148 absl::StrAppend(&cards[kNumRanks - 1], "(CJ)");
149 else if (mark_voids)
150 absl::StrAppend(&cards[kNumRanks - 1], "none");
151
152 return cards;
153}

156DouDizhuState::OriginalDeal() const {
157 SPIEL_CHECK_GE(history_.size(), kNumCards + 1);
158 std::array<std::array<int, kNumRanks>, kNumPlayers> deal{};
159 for (int i = 1; i < kNumCards - kNumCardsLeftOver + 1; ++i)
160 deal[((i - 1 + first_player_) % kNumPlayers)]
161 [CardToRank(history_[i].action - kDealingActionBase)]++;
162
163 for (int i = 0; i < kNumCardsLeftOver; ++i)
164 deal[dizhu_][cards_left_over_[i]]++;
165 return deal;

168std::string DouDizhuState::FormatDeal() const {
169 std::array<std::array<std::string, kNumRanks>, kNumPlayers> cards{};
170 if (IsTerminal()) {
171 // Include all cards in the terminal state to make reviewing the deal easier
172 auto deal = OriginalDeal();
173 for (int player = 0; player < kNumPlayers; ++player) {
174 cards[player] = FormatHand(player, /*mark_voids=*/false, deal);
175 }
176 } else {
177 for (int player = 0; player < kNumPlayers; ++player) {
178 cards[player] = FormatHand(player, /*mark_voids=*/false, holds_);
179 }
180 }
181 constexpr int kColumnWidth = 8;
182 std::string padding(kColumnWidth, ' ');
183 std::string rv;
184 for (int rank = 0; rank < kNumRanks; ++rank)
185 absl::StrAppend(&rv, absl::StrFormat("%-8s", cards[1][rank]), padding,
186 cards[2][rank], "\n");
187 for (int rank = 0; rank < kNumRanks; ++rank)
188 absl::StrAppend(&rv, padding, cards[0][rank], "\n");
189 return rv;
190}

214void DouDizhuState::WriteObservationTensor(Player player,
215 absl::Span<float> values) const {
216 SPIEL_CHECK_GE(player, 0);
217 SPIEL_CHECK_LT(player, num_players_);
218
219 absl::c_fill(values, 0.);
220 if (phase_ == Phase::kDeal) return;
221 auto values_iterator = values.begin();
222 const int played_deck_base = (kNumRanks - 2) * (kNumSuits + 1) + 2 * 2;
223 for (int i = 0; i < kNumRanks; ++i) {
224 values_iterator[i * (kNumSuits + 1) + holds_[player][i]] = 1;
225 values_iterator[played_deck_base + i * (kNumSuits + 1) + played_deck_[i]] =
226 1;
227 }
228
229 if (dizhu_ != kInvalidPlayer) {
230 const int from_dizhu_base = 2 * played_deck_base;
231 const int from_dizhu = (player - dizhu_ + kNumPlayers) % kNumPlayers;
232 values_iterator[from_dizhu_base + from_dizhu] = 1;
233 }
234
235 if (first_player_ != kInvalidPlayer) {
236 const int start_player_base = 2 * played_deck_base + kNumPlayers;
237 values_iterator[start_player_base + first_player_] = 1;
238 values_iterator[start_player_base + kNumPlayers + card_rank_face_up_] = 1;
239 }
240}

255std::vector<Action> DouDizhuState::DealLegalActions() const {
256 std::vector<Action> legal_actions;
257 legal_actions.reserve(kNumCards - history_.size() + 1);
258
259 if (card_face_up_position_ == -1) {
260 for (int i = 0; i < kDealingActionBase; ++i) legal_actions.push_back(i);
261 } else {
262 for (int i = 0; i < kNumCards; ++i) {
263 if (dealer_deck_[i]) legal_actions.push_back(i + kDealingActionBase);
264 }
265 }
266
267 return legal_actions;
268}

270std::vector<Action> DouDizhuState::BiddingLegalActions() const {
271 std::vector<Action> legal_actions = {kPass};
272 legal_actions.reserve(kNumBids + 1);
273
274 for (int bid = winning_bid_ + 1; bid <= kNumBids; ++bid) {
275 legal_actions.push_back(kBiddingActionBase + bid);
276 }
277 return legal_actions;
278}

280std::vector<Action> DouDizhuState::PlayLegalActions() const {
281 std::vector<Action> legal_actions;
282 // the leader of a trick must play./ an action and cannot pass
283 if (!new_trick_begin_) legal_actions.push_back(kPass);
284
285 std::array<int, kNumRanks> hand = holds_[current_player_];
286 const int prev_action = CurrentTrick().WinningAction();
287 SearchForLegalActions(&legal_actions, hand, prev_action);
288
289 absl::c_sort(legal_actions);
290 return legal_actions;
291}

293std::vector<std::pair<Action, double>> DouDizhuState::ChanceOutcomes() const {
294 std::vector<std::pair<Action, double>> outcomes;
295 int num_cards_remaining = 0;
296 for (int i = 0; i < kNumCards; ++i) num_cards_remaining += dealer_deck_[i];
297 outcomes.reserve(num_cards_remaining);
298
299 if (card_face_up_position_ == -1) {
300 for (int i = 0; i < kDealingActionBase; ++i)
301 outcomes.emplace_back(i, 1.0 / static_cast<double>(kDealingActionBase));
302 } else {
303 for (int card = 0; card < kNumCards; ++card)
304 if (dealer_deck_[card])
305 outcomes.emplace_back(card + kDealingActionBase,
306 1.0 / static_cast<double>(num_cards_remaining));
307 }
308
309 return outcomes;
310}

312void DouDizhuState::DoApplyAction(Action action) {
313 switch (phase_) {
314 case Phase::kDeal:
315 return ApplyDealAction(action);
316 case Phase::kAuction:
317 return ApplyBiddingAction(action);
318 case Phase::kPlay:
319 return ApplyPlayAction(action);
320 case Phase::kGameOver:
321 SpielFatalError("Cannot act in terminal states");
322 }

325void DouDizhuState::ApplyDealAction(int action) {
326 // First decide the face up card
327 if (card_face_up_position_ == -1) {
328 card_face_up_position_ = action;
329 return;
330 }
331
332 const int dealing_round = static_cast<int>(history_.size()) - 1;
333 // if the current player is dealt the face up card, make it the first one to
334 // bid
335 if (dealing_round == history_[0].action) {
336 first_player_ = dealing_round % kNumPlayers;
337 card_rank_face_up_ = CardToRank(action - kDealingActionBase);
338 }
339 const int dealt_player_idx = ((history_.size() - 1) % kNumPlayers);
340 const int dealt_rank = CardToRank(action - kDealingActionBase);
341 holds_[dealt_player_idx][dealt_rank]++;
342 dealer_deck_[action - kDealingActionBase]--;
343 if (history_.size() == kNumCards - kNumCardsLeftOver) {
344 phase_ = Phase::kAuction;
345 current_player_ = first_player_;
346 SPIEL_CHECK_GE(current_player_, 0);
347 SPIEL_CHECK_LE(current_player_, num_players_);
348 for (int card = 0; card < kNumCards; ++card)
349 if (dealer_deck_[card]) {
350 cards_left_over_.push_back(CardToRank(card));
351 }
352 }
353}

355void DouDizhuState::ApplyBiddingAction(int action) {
356 // Track the number of consecutive passes since the last bid (if any).
357 if (action == kPass) {
358 ++num_passes_;
359 } else {
360 num_passes_ = 0;
361 }
362
363 bool has_winner = false;
364
365 if (action == kPass) {
366 if (num_passes_ == kNumPlayers)
367 phase_ = Phase::kGameOver;
368 else if (num_passes_ == kNumPlayers - 1 && winning_bid_ > 0)
369 has_winner = true;
370 } else {
371 dizhu_ = current_player_;
372 winning_bid_ = action - kBiddingActionBase;
373 if (winning_bid_ == kNumBids) has_winner = true;
374 }
375 if (has_winner) {
376 for (int i = 0; i < kNumCardsLeftOver; ++i)
377 holds_[dizhu_][cards_left_over_[i]]++;
378 phase_ = Phase::kPlay;
379 current_player_ = dizhu_;
380 new_trick_begin_ = true;
381 tricks_.push_back(Trick(dizhu_, kInvalidAction));
382 num_passes_ = 0;
383 } else {
384 current_player_ = (current_player_ + 1) % kNumPlayers;
385 }
386}

388bool DouDizhuState::AfterPlayHand(int player, int action) {
389 std::array<int, kNumRanks> used_hand = ActionToHand(action);
390 bool flag = true;
391 for (int rank = 0; rank < kNumRanks; ++rank) {
392 SPIEL_CHECK_GE(holds_[player][rank], used_hand[rank]);
393 holds_[player][rank] -= used_hand[rank];
394 flag &= !holds_[player][rank];
395 played_deck_[rank] += used_hand[rank];
396 }
397 return flag;
398}

400void DouDizhuState::ApplyPlayAction(int action) {
401 num_played_++;
402
403 if (action == kPass) {
404 ++num_passes_;
405 } else {
406 num_passes_ = 0;
407 }
408
409 if (action == kPass) {
410 if (num_passes_ == kNumPlayers - 1) {
411 current_player_ = CurrentTrick().Winner();
412 trick_played_++;
413 num_passes_ = 0;
414 tricks_.push_back(Trick());
415 new_trick_begin_ = true;
416 return;
417 }
418 } else {
419 if (action >= kBombActionBase) bombs_played_++;
420 players_hands_played[current_player_]++;
421
422 if (new_trick_begin_) new_trick_begin_ = false;
423
424 CurrentTrick().Play(current_player_, action);
425
426 bool all_played = AfterPlayHand(current_player_, action);
427 if (all_played) {
428 final_winner_ = current_player_;
429 ScoreUp();
430 phase_ = Phase::kGameOver;
431 return;
432 }
433 }
434 current_player_ = (current_player_ + 1) % kNumPlayers;
435}

447void DouDizhuState::ScoreUp() {
448 // If no one bids, 0 for everyone
449 if (dizhu_ == kInvalidPlayer) return;
450
451 // if none of the farmers played, or the dizhu only played once
452 // then it is spring!
453 bool is_spring = false;
454 is_spring |= (players_hands_played[dizhu_] == 1);
455 is_spring |= ((!players_hands_played[(dizhu_ + 1) % 3]) &&
456 (!players_hands_played[(dizhu_ + 2) % 3]));
457
458 int paying = winning_bid_;
459 for (int i = 0; i < is_spring + bombs_played_; ++i) paying *= 2;
460 const int dizhu_sign = (final_winner_ == dizhu_) ? 1 : -1;
461
462 returns_[dizhu_] = dizhu_sign * 2 * paying;
463 returns_[(dizhu_ + 1) % 3] = -dizhu_sign * paying;
464 returns_[(dizhu_ + 2) % 3] = -dizhu_sign * paying;
465}
