15"""Implements Deep CFR Algorithm.
16
17See https://arxiv.org/abs/1811.00164.
18
19The algorithm defines an `advantage` and `strategy` networks that compute
20advantages used to do regret matching across information sets and to approximate
21the strategy profiles of the game. To train these networks a reservoir buffer
22(other data structures may be used) memory is used to accumulate samples to
23train the networks.
24"""

46class ReservoirBuffer(object):
47 """Allows uniform sampling over a stream of data.
48
49 This class supports the storage of arbitrary elements, such as observation
50 tensors, integer actions, etc.
51
52 See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
53 """
54
55 def __init__(self, reservoir_buffer_capacity):
56 self._reservoir_buffer_capacity = reservoir_buffer_capacity
57 self._data = []
58 self._add_calls = 0
59
60 def add(self, element):
61 """Potentially adds `element` to the reservoir buffer.
62
63 Args:
64 element: data to be added to the reservoir buffer.
65 """
66 if len(self._data) < self._reservoir_buffer_capacity:
67 self._data.append(element)
68 else:
69 idx = np.random.randint(0, self._add_calls + 1)
70 if idx < self._reservoir_buffer_capacity:
71 self._data[idx] = element
72 self._add_calls += 1
73
74 def sample(self, num_samples):
75 """Returns `num_samples` uniformly sampled from the buffer.
76
77 Args:
78 num_samples: `int`, number of samples to draw.
79
80 Returns:
81 An iterable over `num_samples` random elements of the buffer.
82
83 Raises:
84 ValueError: If there are less than `num_samples` elements in the buffer
85 """
86 if len(self._data) < num_samples:
87 raise ValueError("{} elements could not be sampled from size {}".format(
88 num_samples, len(self._data)))
89 return random.sample(self._data, num_samples)
90
91 def clear(self):
92 self._data = []
93 self._add_calls = 0
94
95 def __len__(self):
96 return len(self._data)
97
98 def __iter__(self):
99 return iter(self._data)

102class DeepCFRSolver(policy.Policy):
103 """Implements a solver for the Deep CFR Algorithm.
104
105 See https://arxiv.org/abs/1811.00164.
106
107 Define all networks and sampling buffers/memories. Derive losses & learning
108 steps. Initialize the game state and algorithmic variables.
109
110 Note: batch sizes default to `None` implying that training over the full
111 dataset in memory is done by default. To sample from the memories you
112 may set these values to something less than the full capacity of the
113 memory.
114 """
115
116 def __init__(self,
117 session,
118 game,
119 policy_network_layers=(256, 256),
120 advantage_network_layers=(128, 128),
121 num_iterations: int = 100,
122 num_traversals: int = 20,
123 learning_rate: float = 1e-4,
124 batch_size_advantage=None,
125 batch_size_strategy=None,
126 memory_capacity: int = int(1e6),
127 policy_network_train_steps: int = 1,
128 advantage_network_train_steps: int = 1,
129 reinitialize_advantage_networks: bool = True):
130 """Initialize the Deep CFR algorithm.
131
132 Args:
133 session: (tf.Session) TensorFlow session.
134 game: Open Spiel game.
135 policy_network_layers: (list[int]) Layer sizes of strategy net MLP.
136 advantage_network_layers: (list[int]) Layer sizes of advantage net MLP.
137 num_iterations: Number of iterations.
138 num_traversals: Number of traversals per iteration.
139 learning_rate: Learning rate.
140 batch_size_advantage: (int or None) Batch size to sample from advantage
141 memories.
142 batch_size_strategy: (int or None) Batch size to sample from strategy
143 memories.
144 memory_capacity: Number of samples that can be stored in memory.
145 policy_network_train_steps: Number of policy network training steps (per
146 iteration).
147 advantage_network_train_steps: Number of advantage network training steps
148 (per iteration).
149 reinitialize_advantage_networks: Whether to re-initialize the
150 advantage network before training on each iteration.
151 """
152 all_players = list(range(game.num_players()))
153 super(DeepCFRSolver, self).__init__(game, all_players)
154 self._game = game
155 if game.get_type().dynamics == pyspiel.GameType.Dynamics.SIMULTANEOUS:
156 # `_traverse_game_tree` does not take into account this option.
157 raise ValueError("Simulatenous games are not supported.")
158 self._session = session
159 self._batch_size_advantage = batch_size_advantage
160 self._batch_size_strategy = batch_size_strategy
161 self._policy_network_train_steps = policy_network_train_steps
162 self._advantage_network_train_steps = advantage_network_train_steps
163 self._num_players = game.num_players()
164 self._root_node = self._game.new_initial_state()
165 # TODO(author6) Allow embedding size (and network) to be specified.
166 self._embedding_size = len(self._root_node.information_state_tensor(0))
167 self._num_iterations = num_iterations
168 self._num_traversals = num_traversals
169 self._reinitialize_advantage_networks = reinitialize_advantage_networks
170 self._num_actions = game.num_distinct_actions()
171 self._iteration = 1
172 self._environment_steps = 0
173
174 # Create required TensorFlow placeholders to perform the Q-network updates.
175 self._info_state_ph = tf.placeholder(
176 shape=[None, self._embedding_size],
177 dtype=tf.float32,
178 name="info_state_ph")
179 self._info_state_action_ph = tf.placeholder(
180 shape=[None, self._embedding_size + 1],
181 dtype=tf.float32,
182 name="info_state_action_ph")
183 self._action_probs_ph = tf.placeholder(
184 shape=[None, self._num_actions],
185 dtype=tf.float32,
186 name="action_probs_ph")
187 self._iter_ph = tf.placeholder(
188 shape=[None, 1], dtype=tf.float32, name="iter_ph")
189 self._advantage_ph = []
190 for p in range(self._num_players):
191 self._advantage_ph.append(
192 tf.placeholder(
193 shape=[None, self._num_actions],
194 dtype=tf.float32,
195 name="advantage_ph_" + str(p)))
196
197 # Define strategy network, loss & memory.
198 self._strategy_memories = ReservoirBuffer(memory_capacity)
199 self._policy_network = simple_nets.MLP(self._embedding_size,
200 list(policy_network_layers),
201 self._num_actions)
202 action_logits = self._policy_network(self._info_state_ph)
203 # Illegal actions are handled in the traversal code where expected payoff
204 # and sampled regret is computed from the advantage networks.
205 self._action_probs = tf.nn.softmax(action_logits)
206 self._loss_policy = tf.reduce_mean(
207 tf.losses.mean_squared_error(
208 labels=tf.math.sqrt(self._iter_ph) * self._action_probs_ph,
209 predictions=tf.math.sqrt(self._iter_ph) * self._action_probs))
210 self._optimizer_policy = tf.train.AdamOptimizer(learning_rate=learning_rate)
211 self._learn_step_policy = self._optimizer_policy.minimize(self._loss_policy)
212
213 # Define advantage network, loss & memory. (One per player)
214 self._advantage_memories = [
215 ReservoirBuffer(memory_capacity) for _ in range(self._num_players)
216 ]
217 self._advantage_networks = [
218 simple_nets.MLP(self._embedding_size, list(advantage_network_layers),
219 self._num_actions) for _ in range(self._num_players)
220 ]
221 self._advantage_outputs = [
222 self._advantage_networks[i](self._info_state_ph)
223 for i in range(self._num_players)
224 ]
225 self._loss_advantages = []
226 self._optimizer_advantages = []
227 self._learn_step_advantages = []
228 for p in range(self._num_players):
229 self._loss_advantages.append(
230 tf.reduce_mean(
231 tf.losses.mean_squared_error(
232 labels=tf.math.sqrt(self._iter_ph) * self._advantage_ph[p],
233 predictions=tf.math.sqrt(self._iter_ph) *
234 self._advantage_outputs[p])))
235 self._optimizer_advantages.append(
236 tf.train.AdamOptimizer(learning_rate=learning_rate))
237 self._learn_step_advantages.append(self._optimizer_advantages[p].minimize(
238 self._loss_advantages[p]))

263 def solve(self):
264 """Solution logic for Deep CFR."""
265 advantage_losses = collections.defaultdict(list)
266 for _ in range(self._num_iterations):
267 for p in range(self._num_players):
268 for _ in range(self._num_traversals):
269 self._traverse_game_tree(self._root_node, p)
270 if self._reinitialize_advantage_networks:
271 # Re-initialize advantage network for player and train from scratch.
272 self.reinitialize_advantage_network(p)
273 advantage_losses[p].append(self._learn_advantage_network(p))
274 self._iteration += 1
275 # Train policy network.
276 policy_loss = self._learn_strategy_network()
277 return self._policy_network, advantage_losses, policy_loss

280 return self._environment_steps

282 def _traverse_game_tree(self, state, player):
283 """Performs a traversal of the game tree.
284
285 Over a traversal the advantage and strategy memories are populated with
286 computed advantage values and matched regrets respectively.
287 Args:
288 state: Current OpenSpiel game state.
289 player: (int) Player index for this traversal.
290 Returns:
291 Recursively returns expected payoffs for each action.
292 """
293 self._environment_steps += 1
294 expected_payoff = collections.defaultdict(float)
295 if state.is_terminal():
296 # Terminal state get returns.
297 return state.returns()[player]
298 elif state.is_chance_node():
299 # If this is a chance node, sample an action
300 chance_outcome, chance_proba = zip(*state.chance_outcomes())
301 action = np.random.choice(chance_outcome, p=chance_proba)
302 return self._traverse_game_tree(state.child(action), player)
303 elif state.current_player() == player:
304 sampled_regret = collections.defaultdict(float)
305 # Update the policy over the info set & actions via regret matching.
306 _, strategy = self._sample_action_from_advantage(state, player)
307 for action in state.legal_actions():
308 expected_payoff[action] = self._traverse_game_tree(
309 state.child(action), player)
310 cfv = 0
311 for a_ in state.legal_actions():
312 cfv += strategy[a_] * expected_payoff[a_]
313 for action in state.legal_actions():
314 sampled_regret[action] = expected_payoff[action]
315 sampled_regret[action] -= cfv
316 sampled_regret_arr = [0] * self._num_actions
317 for action in sampled_regret:
318 sampled_regret_arr[action] = sampled_regret[action]
319 self._advantage_memories[player].add(
320 AdvantageMemory(state.information_state_tensor(), self._iteration,
321 sampled_regret_arr, action))
322 return cfv
323 else:
324 other_player = state.current_player()
325 _, strategy = self._sample_action_from_advantage(state, other_player)
326 # Recompute distribution dor numerical errors.
327 probs = np.array(strategy)
328 probs /= probs.sum()
329 sampled_action = np.random.choice(range(self._num_actions), p=probs)
330 self._strategy_memories.add(
331 StrategyMemory(
332 state.information_state_tensor(other_player), self._iteration,
333 strategy))
334 return self._traverse_game_tree(state.child(sampled_action), player)

336 def _sample_action_from_advantage(self, state, player):
337 """Returns an info state policy by applying regret-matching.
338
339 Args:
340 state: Current OpenSpiel game state.
341 player: (int) Player index over which to compute regrets.
342 Returns:
343 1. (list) Advantage values for info state actions indexed by action.
344 2. (list) Matched regrets, prob for actions indexed by action.
345 """
346 info_state = state.information_state_tensor(player)
347 legal_actions = state.legal_actions(player)
348 advantages_full = self._session.run(
349 self._advantage_outputs[player],
350 feed_dict={self._info_state_ph: np.expand_dims(info_state, axis=0)})[0]
351 advantages = [max(0., advantage) for advantage in advantages_full]
352 cumulative_regret = np.sum([advantages[action] for action in legal_actions])
353 matched_regrets = np.array([0.] * self._num_actions)
354
355 if cumulative_regret > 0.:
356 for action in legal_actions:
357 matched_regrets[action] = advantages[action] / cumulative_regret
358 else:
359 matched_regrets[max(legal_actions, key=lambda a: advantages_full[a])] = 1
360
361 return advantages, matched_regrets

363 def action_probabilities(self, state):
364 """Returns action probabilities dict for a single batch."""
365 cur_player = state.current_player()
366 legal_actions = state.legal_actions(cur_player)
367 info_state_vector = np.array(state.information_state_tensor())
368 if len(info_state_vector.shape) == 1:
369 info_state_vector = np.expand_dims(info_state_vector, axis=0)
370 probs = self._session.run(
371 self._action_probs, feed_dict={self._info_state_ph: info_state_vector})
372 return {action: probs[0][action] for action in legal_actions}

374 def _learn_advantage_network(self, player):
375 """Compute the loss on sampled transitions and perform a Q-network update.
376
377 If there are not enough elements in the buffer, no loss is computed and
378 `None` is returned instead.
379
380 Args:
381 player: (int) player index.
382 Returns:
383 The average loss over the advantage network.
384 """
385 for _ in range(self._advantage_network_train_steps):
386 if self._batch_size_advantage:
387 if self._batch_size_advantage > len(self._advantage_memories[player]):
388
389 return None
390 samples = self._advantage_memories[player].sample(
391 self._batch_size_advantage)
392 else:
393 samples = self._advantage_memories[player]
394 info_states = []
395 advantages = []
396 iterations = []
397 for s in samples:
398 info_states.append(s.info_state)
399 advantages.append(s.advantage)
400 iterations.append([s.iteration])
401 # Ensure some samples have been gathered.
402 if not info_states:
403 return None
404
405 loss_advantages, _ = self._session.run(
406 [self._loss_advantages[player], self._learn_step_advantages[player]],
407 feed_dict={
408 self._info_state_ph: np.array(info_states),
409 self._advantage_ph[player]: np.array(advantages),
410 self._iter_ph: np.array(iterations),
411 })
412 return loss_advantages
413
414 def _learn_strategy_network(self):
415 """Compute the loss over the strategy network.
416
417 Returns:
418 The average loss obtained on this batch of transitions or `None`.
419 """
420 for _ in range(self._policy_network_train_steps):
421 if self._batch_size_strategy:
422 if self._batch_size_strategy > len(self._strategy_memories):
423
424 return None
425 samples = self._strategy_memories.sample(self._batch_size_strategy)
426 else:
427 samples = self._strategy_memories
428 info_states = []
429 action_probs = []
430 iterations = []
431 for s in samples:
432 info_states.append(s.info_state)
433 action_probs.append(s.strategy_action_probs)
434 iterations.append([s.iteration])
435
436 loss_strategy, _ = self._session.run(
437 [self._loss_policy, self._learn_step_policy],
438 feed_dict={
439 self._info_state_ph: np.array(info_states),
440 self._action_probs_ph: np.array(np.squeeze(action_probs)),
441 self._iter_ph: np.array(iterations),
442 })
443 return loss_strategy
