49  def action_probabilities(self, state, player_id=None):
50  legal_actions = state.legal_actions()
51  max_q = np.max(
52  [self._state_action_value_state_action_value(state, action) for action in legal_actions])
53  exp_q = [
54  np.exp((self._state_action_value_state_action_value(state, action) - max_q) /

59  exp_q = [

61  for i, action in enumerate(legal_actions)
62  ]
63  denom = sum(exp_q)
64  smax_q = exp_q if denom == 0 else exp_q / denom
