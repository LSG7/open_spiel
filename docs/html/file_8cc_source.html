113 bool Mkdirs(const std::string& path, int mode) {
114  struct stat info;
115  size_t pos = 0;
116  while (pos != std::string::npos) {
117  pos = path.find_first_of("\\/", pos + 1);
118  std::string sub_path = path.substr(0, pos);
119  if (stat(sub_path.c_str(), &info) == 0) {
120  if (info.st_mode & S_IFDIR) {
121  continue; // directory already exists
122  } else {
123  return false; // is a file?
124  }
125  }
126  if (!Mkdir(sub_path, mode)) {
127  return false; // permission error?
128  }
129  }
130  return true;
131 }
