27 // Convert absl::StatusOr<T>.
28 // It isn't possible to specify separate return value policies for the container
29 // (StatusOr) and the payload. Since StatusOr is processed and not ever actually
30 // represented in python, the return value policy applies to the payload. Eg, if
31 // you return a StatusOr<MyObject*> (note the * is inside the StatusOr) with a
32 // take_ownership return val policy and the status is ok (ie, it has a payload),
33 // python will take ownership of that payload and free it when it is garbage
34 // collected.
35 // However, if you return a StatusOr<MyObject>* (note the * is outside the
36 // StatusOr rather than inside it now) with a take_ownership return val policy,
37 // python does not take ownership of the StatusOr and will not free it (because
38 // again, that policy applies to MyObject, not StatusOr).
39 template <typename PayloadType>
40 struct type_caster<absl::StatusOr<PayloadType>> {
41  public:
42  using PayloadCaster = make_caster<PayloadType>;
43  using StatusCaster = make_caster<absl::Status>;
