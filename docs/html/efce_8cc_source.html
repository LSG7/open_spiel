65 std::string EFCEState::InformationStateString(Player player) const {
66  // should look like <infoset string> <delimiter> <recommendation sequence,
67  // (including current recommendation)>
68  SPIEL_CHECK_FALSE(IsChanceNode());
69  std::string rec_str = absl::StrJoin(recommendation_seq_[player], ",");

106 void EFCEState::DoApplyAction(Action action_id) {
107  if (rec_index_ < 0) {
108  // Pick the joint policy which will provide recommendations.
109  rec_index_ = action_id;
110  SPIEL_CHECK_GE(rec_index_, 0);
111  SPIEL_CHECK_LT(rec_index_, mu_.size());
112  } else if (state_->IsChanceNode()) {
113  // Regular chance node
114  state_->ApplyAction(action_id);
115  } else {
116  // Check for defection at this point.
117  const Action recommendation = CurRecommendation();
118 
119  Player cur_player = CurrentPlayer();
120  SPIEL_CHECK_GE(cur_player, 0);
121  SPIEL_CHECK_LT(cur_player, game_->NumPlayers());
122 
123  // If they have defected, don't add to the sequence
124  if (!HasDefected(cur_player)) {
125  recommendation_seq_[cur_player].push_back(recommendation);
126 
127  // If they chose an action other than the recommendation, they have now
128  // defected.
129  if (action_id != recommendation) {
130  defected_[cur_player] = 1;
131  }
132  }
133 
134  state_->ApplyAction(action_id);
135  }
136 }

138 ActionsAndProbs EFCETabularPolicy::GetStatePolicy(const State& state) const {
139  // The best response code has to have a policy defined everywhere when it
140  // builds its initial tree. For the fixed policies, the players will not
141  // defect, so we define a uniform policy in the regions where players have
142  // defected (which will not affect the best responding player, since the
143  // opponents will never reach these regions).
144  const auto* efce_state = dynamic_cast<const EFCEState*>(&state);
145  SPIEL_CHECK_TRUE(efce_state != nullptr);
146  if (efce_state->HasDefected(state.CurrentPlayer())) {
147  return UniformStatePolicy(state);
148  }
149 
150  // Otherwise, simply returns a fixed policy with prob 1 on the recommended
151  // action (extrapolated from the information state string) and 0 on the
152  // others.
153  std::string info_state = state.InformationStateString();
154  const size_t idx = info_state.find(config_.recommendation_delimiter);
155  SPIEL_CHECK_NE(idx, std::string::npos);
156  std::vector<std::string> rec_seq = absl::StrSplit(
157  info_state.substr(idx + config_.recommendation_delimiter.length()), ',');
158  SPIEL_CHECK_GE(rec_seq.size(), 1);
159  Action rec_action;
160  ActionsAndProbs state_policy;
161  std::vector<Action> legal_actions = state.LegalActions();
162  state_policy.reserve(legal_actions.size());
163  SPIEL_CHECK_TRUE(absl::SimpleAtoi(rec_seq.back(), &rec_action));
164  for (Action action : legal_actions) {
165  state_policy.push_back({action, action == rec_action ? 1.0 : 0.0});
166  }
167  return state_policy;
168 }
