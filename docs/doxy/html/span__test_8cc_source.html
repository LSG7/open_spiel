38MATCHER_P(DataIs, data,
39 absl::StrCat("data() ", negation ? "isn't " : "is ",
40 testing::PrintToString(data))) {
41 return arg.data() == data;
42}

44template <typename T>
45auto SpanIs(T data, size_t size)
46 -> decltype(testing::AllOf(DataIs(data), testing::SizeIs(size))) {
47 return testing::AllOf(DataIs(data), testing::SizeIs(size));
48}

50template <typename Container>
51auto SpanIs(const Container& c) -> decltype(SpanIs(c.data(), c.size())) {

463TEST(IntSpan, Equality) {
464 const int arr1[] = {1, 2, 3, 4, 5};
465 int arr2[] = {1, 2, 3, 4, 5};
466 std::vector<int> vec1(std::begin(arr1), std::end(arr1));
467 std::vector<int> vec2 = vec1;
468 std::vector<int> other_vec = {2, 4, 6, 8, 10};
469 // These two slices are from different vectors, but have the same size and
470 // have the same elements (right now). They should compare equal. Test both
471 // == and !=.
472 const absl::Span<const int> from1 = vec1;
473 const absl::Span<const int> from2 = vec2;
474 EXPECT_EQ(from1, from1);
475 EXPECT_FALSE(from1 != from1);
476 EXPECT_EQ(from1, from2);
477 EXPECT_FALSE(from1 != from2);
478
479 // These two slices have different underlying vector values. They should be
480 // considered not equal. Test both == and !=.
481 const absl::Span<const int> from_other = other_vec;
482 EXPECT_NE(from1, from_other);
483 EXPECT_FALSE(from1 == from_other);
484
485 // Comparison between a vector and its slice should be equal. And vice-versa.
486 // This ensures implicit conversion to Span works on both sides of ==.
487 EXPECT_EQ(vec1, from1);
488 EXPECT_FALSE(vec1 != from1);
489 EXPECT_EQ(from1, vec1);
490 EXPECT_FALSE(from1 != vec1);
491
492 // This verifies that absl::Span<T> can be compared freely with
493 // absl::Span<const T>.
494 const absl::Span<int> mutable_from1(vec1);
495 const absl::Span<int> mutable_from2(vec2);
496 EXPECT_EQ(from1, mutable_from1);
497 EXPECT_EQ(mutable_from1, from1);
498 EXPECT_EQ(mutable_from1, mutable_from2);
499 EXPECT_EQ(mutable_from2, mutable_from1);
500
501 // Comparison between a vector and its slice should be equal for mutable
502 // Spans as well.
503 EXPECT_EQ(vec1, mutable_from1);
504 EXPECT_FALSE(vec1 != mutable_from1);
505 EXPECT_EQ(mutable_from1, vec1);
506 EXPECT_FALSE(mutable_from1 != vec1);
507
508 // Comparison between convertible-to-Span-of-const and Span-of-mutable. Arrays
509 // are used because they're the only value type which converts to a
510 // Span-of-mutable. EXPECT_TRUE is used instead of EXPECT_EQ to avoid
511 // array-to-pointer decay.
512 EXPECT_TRUE(arr1 == mutable_from1);
513 EXPECT_FALSE(arr1 != mutable_from1);
514 EXPECT_TRUE(mutable_from1 == arr1);
515 EXPECT_FALSE(mutable_from1 != arr1);
516
517 // Comparison between convertible-to-Span-of-mutable and Span-of-const
518 EXPECT_TRUE(arr2 == from1);
519 EXPECT_FALSE(arr2 != from1);
520 EXPECT_TRUE(from1 == arr2);
521 EXPECT_FALSE(from1 != arr2);
522
523 // With a different size, the array slices should not be equal.
524 EXPECT_NE(from1, absl::Span<const int>(from1).subspan(0, from1.size() - 1));
525
526 // With different contents, the array slices should not be equal.
527 ++vec2.back();
528 EXPECT_NE(from1, from2);
529}
