57 """Returns a valid epsilon-greedy action and valid action probs.
58
59 If the agent has not been to `info_state`, a valid random action is chosen.
60
61 Args:
62 info_state: hashable representation of the information state.
63 legal_actions: list of actions at `info_state`.
64 epsilon: float, prob of taking an exploratory action.
65
66 Returns:
67 A valid epsilon-greedy action and valid action probabilities.
68 """
69 probs = np.zeros(self._num_actions)
70 greedy_q = max([self._q_values[info_state][a] for a in legal_actions])
71 greedy_actions = [
72 a for a in legal_actions if self._q_values[info_state][a] == greedy_q
73 ]
74 probs[legal_actions] = epsilon / len(legal_actions)
75 probs[greedy_actions] += (1 - epsilon) / len(greedy_actions)
76 action = np.random.choice(range(self._num_actions), p=probs)
77 return action, probs
