25 void BotTest_RandomVsRandom() {

28  auto bot1 = MakeStatefulRandomBot(*game, 1, /*seed=*/4321);

30  std::vector<double> average_results(num_players);
31  constexpr int num_iters = 100000;

33  auto this_results =
34  EvaluateBots(game->NewInitialState().get(), {bot0.get(), bot1.get()},
35  /*seed=*/iteration);

37  average_results[p] += this_results[p];
38  }
