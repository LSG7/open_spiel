35 // All CFR solvers support serialization. CFRSolver, CFRPlusSolver and
36 // CFRBRSolver all rely on CFRSolverBase::Serialize() to provide the
37 // functionality (each subtype only implements the SerializeThisType() method
38 // which returns the name of the serialized subclass).
39 // ExternalSamplingMCCFRSolver and OutcomeSamplingMCCFRSolver implement
40 // their own versions of Serialize().
41 //
42 // During the serialization we store multiple different sections which are
43 // described below:
44 // - [SolverType] is the name of the serialized class.
45 // - [SolverSpecificState] is a section to which different solvers write state
46 // that is specific to their own type. Note that serialization and
47 // deserialization of this section is left entirely to each of the
48 // subtypes.
49 // - [SolverValuesTable] is the section to which CFRInfoStateValuesTable is
50 // written.
51 //
52 // During deserialization all solvers rely on the
53 // PartiallyDeserializeCFRSolver() method which deserializes common properties
54 // but leaves deserialization of [SolverSpecificState] section to the caller.
55 //
56 // Note that there are some specifics that need to be taken into account when
57 // reading the de/serialization code:
58 // - Each solver's Serialize() method has two parameters which are not exposed
59 // in Python, i.e. int double_precision and std::string delimiter.
60 // - We try to avoid copying/moving of CFRInfoStateValuesTable where possible
61 // due to its potentially large size (mainly due to memory concerns). For
62 // that reason the PartiallyDeserializeCFRSolver() method also returns a
63 // string_view of the table and leaves deserialization to the callers
64 // which then in turn call the efficient
65 // DeserializeCFRInfoStateValuesTable().

103 void CFRAveragePolicy::GetStatePolicyFromInformationStateValues(
104  const CFRInfoStateValues& is_vals,
105  ActionsAndProbs* actions_and_probs) const {
106  double sum_prob = 0.0;
107  for (int aidx = 0; aidx < is_vals.num_actions(); ++aidx) {
108  sum_prob += is_vals.cumulative_policy[aidx];
109  }
110 
111  if (sum_prob == 0.0) {
112  // Return a uniform policy at this node
113  double prob = 1. / is_vals.num_actions();
114  for (Action action : is_vals.legal_actions) {
115  actions_and_probs->push_back({action, prob});
116  }
117  return;
118  }
119 
120  for (int aidx = 0; aidx < is_vals.num_actions(); ++aidx) {
121  actions_and_probs->push_back({is_vals.legal_actions[aidx],
122  is_vals.cumulative_policy[aidx] / sum_prob});
123  }
124 }

233 void CFRSolverBase::InitializeInfostateNodes(const State& state) {
234  if (state.IsTerminal()) {
235  return;
236  }
237  if (state.IsChanceNode()) {
238  for (const auto& action_prob : state.ChanceOutcomes()) {
239  InitializeInfostateNodes(*state.Child(action_prob.first));
240  }
241  return;
242  }
243 
244  int current_player = state.CurrentPlayer();
245  std::string info_state = state.InformationStateString(current_player);
246  std::vector<Action> legal_actions = state.LegalActions();
247 
248  if (random_initial_regrets_) {
249  CFRInfoStateValues is_vals(legal_actions, &rng_,
250  kRandomInitialRegretsMagnitude);
251  info_states_[info_state] = is_vals;
252  } else {
253  CFRInfoStateValues is_vals(legal_actions);
254  info_states_[info_state] = is_vals;
255  }
256 
257  for (const Action& action : legal_actions) {
258  InitializeInfostateNodes(*state.Child(action));
259  }
260 }

262 void CFRSolverBase::EvaluateAndUpdatePolicy() {
263  ++iteration_;
264  if (alternating_updates_) {
265  for (int player = 0; player < game_->NumPlayers(); player++) {
266  ComputeCounterFactualRegret(*root_state_, player, root_reach_probs_,
267  nullptr);
268  if (regret_matching_plus_) {
269  ApplyRegretMatchingPlusReset();
270  }
271  ApplyRegretMatching();
272  }
273  } else {
274  ComputeCounterFactualRegret(*root_state_, absl::nullopt, root_reach_probs_,
275  nullptr);
276  if (regret_matching_plus_) {
277  ApplyRegretMatchingPlusReset();
278  }
279  ApplyRegretMatching();
280  }
281 }

308 static double CounterFactualReachProb(
309  const std::vector<double>& reach_probabilities, const int player) {
310  double cfr_reach_prob = 1.0;
311  for (int i = 0; i < reach_probabilities.size(); i++) {
312  if (i != player) {
313  cfr_reach_prob *= reach_probabilities[i];
314  }
315  }
316  return cfr_reach_prob;
317 }
318 
319 // Compute counterfactual regrets. Alternates recursively with
320 // ComputeCounterFactualRegretForActionProbs.
321 //
322 // Args:
323 // - state: The state to start the recursion.
324 // - alternating_player: Optionally only update this player.
325 // - reach_probabilities: The reach probabilities of this state for each
326 // player, ending with the chance player.
327 //
328 // Returns:
329 // The value of the state for each player (excluding the chance player).
330 std::vector<double> CFRSolverBase::ComputeCounterFactualRegret(
331  const State& state, const absl::optional<int>& alternating_player,
332  const std::vector<double>& reach_probabilities,
333  const std::vector<const Policy*>* policy_overrides) {
334  if (state.IsTerminal()) {
335  return state.Returns();
336  }
337  if (state.IsChanceNode()) {
338  ActionsAndProbs actions_and_probs = state.ChanceOutcomes();
339  std::vector<double> dist(actions_and_probs.size(), 0);
340  std::vector<Action> outcomes(actions_and_probs.size(), 0);
341  for (int oidx = 0; oidx < actions_and_probs.size(); ++oidx) {
342  outcomes[oidx] = actions_and_probs[oidx].first;
343  dist[oidx] = actions_and_probs[oidx].second;
344  }
345  return ComputeCounterFactualRegretForActionProbs(
346  state, alternating_player, reach_probabilities, chance_player_, dist,
347  outcomes, nullptr, policy_overrides);
348  }
349  if (AllPlayersHaveZeroReachProb(reach_probabilities)) {
350  // The value returned is not used: if the reach probability for all players
351  // is 0, then the last taken action has probability 0, so the
352  // returned value is not impacting the parent node value.
353  return std::vector<double>(game_->NumPlayers(), 0.0);
354  }
355 
356  int current_player = state.CurrentPlayer();
357  std::string info_state = state.InformationStateString();
358  std::vector<Action> legal_actions = state.LegalActions(current_player);
359 
360  // Load current policy.
361  std::vector<double> info_state_policy;
362  if (policy_overrides && policy_overrides->at(current_player)) {
363  GetInfoStatePolicyFromPolicy(&info_state_policy, legal_actions,
364  policy_overrides->at(current_player),
365  info_state);
366  } else {
367  info_state_policy = GetPolicy(info_state, legal_actions);
368  }
369 
370  std::vector<double> child_utilities;
371  child_utilities.reserve(legal_actions.size());
372  const std::vector<double> state_value =
373  ComputeCounterFactualRegretForActionProbs(
374  state, alternating_player, reach_probabilities, current_player,
375  info_state_policy, legal_actions, &child_utilities, policy_overrides);
376 
377  // Perform regret and average strategy updates.
378  if (!alternating_player || *alternating_player == current_player) {
379  CFRInfoStateValues is_vals = info_states_[info_state];
380  SPIEL_CHECK_FALSE(is_vals.empty());
381 
382  const double self_reach_prob = reach_probabilities[current_player];
383  const double cfr_reach_prob =
384  CounterFactualReachProb(reach_probabilities, current_player);
385 
386  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
387  // Update regrets.
388  double cfr_regret = cfr_reach_prob *
389  (child_utilities[aidx] - state_value[current_player]);
390 
391  is_vals.cumulative_regrets[aidx] += cfr_regret;
392 
393  // Update average policy.
394  if (linear_averaging_) {
395  is_vals.cumulative_policy[aidx] +=
396  iteration_ * self_reach_prob * info_state_policy[aidx];
397  } else {
398  is_vals.cumulative_policy[aidx] +=
399  self_reach_prob * info_state_policy[aidx];
400  }
401  }
402 
403  info_states_[info_state] = is_vals;
404  }
405 
406  return state_value;
407 }

409 void CFRSolverBase::GetInfoStatePolicyFromPolicy(
410  std::vector<double>* info_state_policy,
411  const std::vector<Action>& legal_actions, const Policy* policy,
412  const std::string& info_state) const {
413  ActionsAndProbs actions_and_probs = policy->GetStatePolicy(info_state);
414  info_state_policy->reserve(legal_actions.size());
415 
416  // The policy may have extra ones not at this infostate
417  for (Action action : legal_actions) {
418  const auto& iter =
419  std::find_if(actions_and_probs.begin(), actions_and_probs.end(),
420  [action](const std::pair<Action, double>& ap) {
421  return ap.first == action;
422  });
423  info_state_policy->push_back(iter->second);
424  }
425 
426  SPIEL_CHECK_EQ(info_state_policy->size(), legal_actions.size());
427 }

429 // Compute counterfactual regrets given certain action probabilities.
430 // Alternates recursively with ComputeCounterFactualRegret.
431 //
432 // Args:
433 // - state: The state to start the recursion.
434 // - alternating_player: Optionally only update this player.
435 // - reach_probabilities: The reach probabilities of this state.
436 // - current_player: Either a player or chance_player_.
437 // - action_probs: The action probabilities to use for this state.
438 // - child_values_out: optional output parameter which is filled with the child
439 // utilities for each action, for current_player.
440 // Returns:
441 // The value of the state for each player (excluding the chance player).
442 std::vector<double> CFRSolverBase::ComputeCounterFactualRegretForActionProbs(
443  const State& state, const absl::optional<int>& alternating_player,
444  const std::vector<double>& reach_probabilities, const int current_player,
445  const std::vector<double>& info_state_policy,
446  const std::vector<Action>& legal_actions,
447  std::vector<double>* child_values_out,
448  const std::vector<const Policy*>* policy_overrides) {
449  std::vector<double> state_value(game_->NumPlayers());
450 
451  for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
452  const Action action = legal_actions[aidx];
453  const double prob = info_state_policy[aidx];
454  const std::unique_ptr<State> new_state = state.Child(action);
455  std::vector<double> new_reach_probabilities(reach_probabilities);
456  new_reach_probabilities[current_player] *= prob;
457  std::vector<double> child_value =
458  ComputeCounterFactualRegret(*new_state, alternating_player,
459  new_reach_probabilities, policy_overrides);
460  for (int i = 0; i < state_value.size(); ++i) {
461  state_value[i] += prob * child_value[i];
462  }
463  if (child_values_out != nullptr) {
464  child_values_out->push_back(child_value[current_player]);
465  }
466  }
467  return state_value;
468 }

533 CFRInfoStateValues DeserializeCFRInfoStateValues(absl::string_view serialized) {
534  CFRInfoStateValues res = CFRInfoStateValues();
535  if (serialized.empty()) return res;
536 
537  std::vector<std::vector<absl::string_view>> str_values;
538  str_values.reserve(4);
539  for (absl::string_view sv : absl::StrSplit(serialized, ';')) {
540  str_values.push_back(absl::StrSplit(sv, ','));
541  }
542 
543  int num_elements = str_values.at(0).size();
544  res.legal_actions.reserve(num_elements);
545  res.cumulative_regrets.reserve(num_elements);
546  res.cumulative_policy.reserve(num_elements);
547  res.current_policy.reserve(num_elements);
548 
549  // Insert the actual values
550  int la_value;
551  double cumu_regret_value, cumu_policy_value, curr_policy_value;
552  for (int i = 0; i < num_elements; i++) {
553  SPIEL_CHECK_TRUE(absl::SimpleAtoi(str_values.at(0).at(i), &la_value));
554  absl::from_chars(
555  str_values.at(1).at(i).data(),
556  str_values.at(1).at(i).data() + str_values.at(1).at(i).size(),
557  cumu_regret_value);
558  absl::from_chars(
559  str_values.at(2).at(i).data(),
560  str_values.at(2).at(i).data() + str_values.at(2).at(i).size(),
561  cumu_policy_value);
562  absl::from_chars(
563  str_values.at(3).at(i).data(),
564  str_values.at(3).at(i).data() + str_values.at(3).at(i).size(),
565  curr_policy_value);
566 
567  res.legal_actions.push_back(la_value);
568  res.cumulative_regrets.push_back(cumu_regret_value);
569  res.cumulative_policy.push_back(cumu_policy_value);
570  res.current_policy.push_back(curr_policy_value);
571  }
572  return res;
573 }

584 void CFRInfoStateValues::ApplyRegretMatchingAllPositive(double delta) {
585  SPIEL_CHECK_GT(delta, 0);

595 void CFRInfoStateValues::ApplyRegretMatching() {
596  double sum_positive_regrets = 0.0;
597 
598  for (int aidx = 0; aidx < num_actions(); ++aidx) {
599  if (cumulative_regrets[aidx] > 0) {
600  sum_positive_regrets += cumulative_regrets[aidx];
601  }
602  }
603 
604  for (int aidx = 0; aidx < num_actions(); ++aidx) {
605  if (sum_positive_regrets > 0) {
606  current_policy[aidx] =
607  cumulative_regrets[aidx] > 0
608  ? cumulative_regrets[aidx] / sum_positive_regrets
609  : 0;
610  } else {
611  current_policy[aidx] = 1.0 / legal_actions.size();
612  }
613  }

616 int CFRInfoStateValues::SampleActionIndex(double epsilon, double z) {
617  double sum = 0;
618  for (int aidx = 0; aidx < current_policy.size(); ++aidx) {
619  double prob = epsilon * 1.0 / current_policy.size() +
620  (1.0 - epsilon) * current_policy[aidx];
621  if (z >= sum && z < sum + prob) {
622  return aidx;
623  }
624  sum += prob;
625  }
626  SpielFatalError(absl::StrCat("SampleActionIndex: sum of probs is ", sum));
627 }

722 PartiallyDeserializedCFRSolver PartiallyDeserializeCFRSolver(
723  const std::string& serialized) {
724  // We don't copy the CFR values table section due to potential large size.
725  enum Section {
726  kInvalid = -1,
727  kMeta = 0,
728  kGame = 1,
729  kSolverType = 2,
730  kSolverSpecificState = 3
731  };
732  std::array<std::string, 4> section_strings = {"", "", "", ""};
733  Section current_section = kInvalid;
734 
735  std::vector<absl::string_view> lines = absl::StrSplit(serialized, '\n');
736  for (int i = 0; i < lines.size(); i++) {
737  if (lines[i].length() == 0 || lines[i].at(0) == '#') {
738  // Skip comments and blank lines
739  } else if (lines[i] == kSerializeMetaSectionHeader) {
740  SPIEL_CHECK_EQ(current_section, kInvalid);
741  current_section = kMeta;
742  } else if (lines[i] == kSerializeGameSectionHeader) {
743  SPIEL_CHECK_EQ(current_section, kMeta);
744  current_section = kGame;
745  } else if (lines[i] == kSerializeSolverTypeSectionHeader) {
746  SPIEL_CHECK_EQ(current_section, kGame);
747  current_section = kSolverType;
748  } else if (lines[i] == kSerializeSolverSpecificStateSectionHeader) {
749  SPIEL_CHECK_EQ(current_section, kSolverType);
750  current_section = kSolverSpecificState;
751  } else if (lines[i] == kSerializeSolverValuesTableSectionHeader) {
752  SPIEL_CHECK_EQ(current_section, kSolverSpecificState);
753  break;
754  } else {
755  SPIEL_CHECK_NE(current_section, kInvalid);
756  if (current_section == kSolverSpecificState) {
757  absl::StrAppend(&section_strings[current_section], lines[i], "\n");
758  } else {
759  absl::StrAppend(&section_strings[current_section], lines[i]);
760  }
761  }
762  }
763 
764  // We currently just ignore the meta section.
765  // In order to avod copying the CFR values table data we rather split it again
766  // and obtain a single string_view that can be deserialized later using the
767  // DeserializeCFRInfoStateValuesTable method.
768  std::pair<absl::string_view, absl::string_view> other_and_values_table_data =
769  absl::StrSplit(
770  serialized,
771  absl::StrCat(kSerializeSolverValuesTableSectionHeader, "\n"));
772  return PartiallyDeserializedCFRSolver(DeserializeGame(section_strings[kGame]),
773  section_strings[kSolverType],
774  section_strings[kSolverSpecificState],
775  other_and_values_table_data.second);
776 }
