28  def test_xfp(self):
29  game = pyspiel.load_game("kuhn_poker")
30  xfp_solver = fictitious_play.XFPSolver(game)
31  for _ in range(100):
32  xfp_solver.iteration()
33  average_policies = xfp_solver.average_policy_tables()
34  tabular_policy = policy.TabularPolicy(game)
35  for player_id in range(2):
36  for info_state, state_policy in average_policies[player_id].items():
37  policy_to_update = tabular_policy.policy_for_key(info_state)
38  for action, probability in state_policy.items():
39  policy_to_update[action] = probability
40  average_policy_values = expected_game_score.policy_value(
41  game.new_initial_state(), [tabular_policy, tabular_policy])
42  print("Kuhn 2P average values after 10 iterations")
43  print("P0: {}".format(average_policy_values[0]))
44  print("P1: {}".format(average_policy_values[1]))
45  self.assertIsNotNone(average_policy_values)
46  self.assertTrue(
47  np.allclose(average_policy_values, [-1 / 18, 1 / 18], atol=1e-3))
