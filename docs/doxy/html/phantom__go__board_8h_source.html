36// For simplicity and speed, we store the board in terms of a "virtual board",
37// with a border of guard stones around all sides of the board.
38// This allows us to skip bounds checking.
39// In Virtual mode, an action (row, col) is row * 21 + col, and pass is 21*21+1.
40// All functions in this file (except stated otherwise) use these virtual
41// coordinates.
42//
43// However, in the OpenSpiel API (in go.{h, cc}), the actions are still exposed
44// as actions within 0, board_size*boardsize) (with pass = board_size **2.
45//
46// Normal go is standardly played on board of size 19, for Phantom Go, standard
47// is size 9
48inline constexpr int kMaxBoardSize = 19;
49inline constexpr int kVirtualBoardSize = kMaxBoardSize + 2;
50inline constexpr int kVirtualBoardPoints =
51 kVirtualBoardSize * kVirtualBoardSize;
52

148 // Count of pseudo-liberties, i.e. each liberty is counted between 1 and 4
149 // times, once for each stone of the group that borders it.
150 // This is much faster than realLiberty(), so prefer it if possible.
151 inline int PseudoLiberty(VirtualPoint p) const {
152 return chain(p).num_pseudo_liberties == 0
153 ? 0
154 : (chain(p).in_atari() ? 1 : chain(p).num_pseudo_liberties);
155 }

162 // Actual liberty count, i.e. each liberty is counted exactly once.
163 // This is computed on the fly by actually walking the group and checking the
164 // neighbouring stones.
165 inline int RealLiberty(VirtualPoint p) const {
166 int num_lib = 0;
167 for (auto it = LibIter(p); it; ++it) {
168 ++num_lib;
169 }

171 }
