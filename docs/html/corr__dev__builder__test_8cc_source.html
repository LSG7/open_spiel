38 TabularPolicy MergeIndependentPolicies(const TabularPolicy& policy1,
39  const TabularPolicy& policy2) {
40  TabularPolicy merged_policy;
41  for (const auto& infostate_and_state_policy : policy1.PolicyTable()) {
42  merged_policy.SetStatePolicy(infostate_and_state_policy.first,
43  infostate_and_state_policy.second);
44  }
45  for (const auto& infostate_and_state_policy : policy2.PolicyTable()) {
46  merged_policy.SetStatePolicy(infostate_and_state_policy.first,
47  infostate_and_state_policy.second);
48  }
49  return merged_policy;
50 }

52 void BasicCorrDevBuilderTest() {
53  // Build a uniform correlation device for Kuhn poker.
54  std::shared_ptr<const Game> game = LoadGame("kuhn_poker");
55  CorrDevBuilder full_cd_builder;
56 
57  DeterministicTabularPolicy p1_policy(*game, 0);
58  DeterministicTabularPolicy p2_policy(*game, 1);
59  do {
60  do {
61  full_cd_builder.AddDeterminsticJointPolicy(MergeIndependentPolicies(
62  p1_policy.GetTabularPolicy(), p2_policy.GetTabularPolicy()));
63  } while (p2_policy.NextPolicy());
64  p2_policy.ResetDefaultPolicy();
65  } while (p1_policy.NextPolicy());
66 
67  CorrelationDevice mu = full_cd_builder.GetCorrelationDevice();
68  SPIEL_CHECK_EQ(mu.size(), 64 * 64);
69  for (const auto& prob_and_policy : mu) {
70  SPIEL_CHECK_FLOAT_NEAR(prob_and_policy.first, 1.0 / (64 * 64), 1e-10);
71  }
72 
73  std::vector<double> uniform_returns =
74  ExpectedReturns(*game->NewInitialState(), GetUniformPolicy(*game), -1);
75  std::vector<double> corr_dev_uniform_returns = ExpectedValues(*game, mu);
76  for (Player p = 0; p < game->NumPlayers(); ++p) {
77  SPIEL_CHECK_FLOAT_NEAR(uniform_returns[p], corr_dev_uniform_returns[p],
78  1e-10);
79  }
80 }
