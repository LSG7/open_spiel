32void KuhnTest() {
33 std::shared_ptr<const open_spiel::Game> game =
34 open_spiel::LoadGame("kuhn_poker");
35
36 LegalActionsMap map_p0 =
37 algorithms::GetLegalActionsMap(*game,
38 /*depth_limit=*/-1, open_spiel::Player{0});
39 SPIEL_CHECK_EQ(map_p0.size(), kuhn_poker::kNumInfoStatesP0);
40
41 LegalActionsMap map_p1 =
42 algorithms::GetLegalActionsMap(*game,
43 /*depth_limit=*/-1, open_spiel::Player{1});
44 SPIEL_CHECK_EQ(map_p1.size(), kuhn_poker::kNumInfoStatesP1);
45
46 LegalActionsMap map_both = algorithms::GetLegalActionsMap(
47 *game, /*depth_limit=*/-1, open_spiel::kInvalidPlayer);
48 SPIEL_CHECK_EQ(map_both.size(),
49 kuhn_poker::kNumInfoStatesP0 + kuhn_poker::kNumInfoStatesP1);
50 // They should all have two legal actions: pass and bet.
51 for (const auto& legal_actions : map_both) {
52 SPIEL_CHECK_EQ(legal_actions.second.size(), 2);
53 }
54}
