71 double MDP::Solve(double tolerance, TabularPolicy *br_policy) {
72  double delta = 0;
73 
74  do {
75  delta = 0.0;
76  for (auto &key_and_node : node_map_) {
77  MDPNode *node = key_and_node.second.get();
78 
79  if (node->terminal()) {
80  continue;
81  }
82 
83  double max_value = -std::numeric_limits<double>::infinity();
84  Action max_action = kInvalidAction;
85  double node_weight = node->total_weight();
86  SPIEL_CHECK_GE(node_weight, 0.0);
87 
88  // Compute Bellman value from children.
89  for (const auto &action_and_child : node->children()) {
90  double action_value = 0.0;
91  Action action = action_and_child.first;
92 
93  for (auto &child_value : node->children()[action]) {
94  MDPNode *child = child_value.first;
95  double transition_weight = child_value.second;
96  SPIEL_CHECK_TRUE(child != nullptr);
97  double prob = transition_weight / node_weight;
98  if (std::isnan(prob)) {
99  // When transition_weight = node_weight = 0, set to 0
100  prob = 0.0;
101  }
102 
103  SPIEL_CHECK_PROB(prob);
104  action_value += prob * child->value();
105  }
106 
107  if (action_value > max_value) {
108  max_value = action_value;
109  max_action = action;
110  }
111  }
112 
113  SPIEL_CHECK_NE(max_action, kInvalidAction);
114  delta += std::abs(node->value() - max_value);
115  node->set_value(max_value);
116 
117  // Set the best response to the maximum-value action, if it's non-null.
118  if (node->node_key() != kRootKey) {
119  ActionsAndProbs br_state_policy;
120  for (const auto &[action, child] : node->children()) {
121  SetProb(&br_state_policy, action, action == max_action ? 1.0 : 0.0);
122  }
123  br_policy->SetStatePolicy(node->node_key(), br_state_policy);
124  }
125  }
126  } while (delta > tolerance);
127 
128  return RootNode()->value();
129 }

143 void TabularBestResponseMDP::BuildMDPs(
144  const State &state, const std::vector<double>& reach_probs,
145  const std::vector<MDPNode*>& parent_nodes,
146  const std::vector<Action>& parent_actions, Player only_for_player) {
147  if (state.IsTerminal()) {
148  std::vector<double> terminal_values = state.Returns();
149  for (Player p = 0; p < game_.NumPlayers(); ++p) {
150  if (only_for_player == kInvalidPlayer || only_for_player == p) {
151  std::string node_key = state.ToString();
152  MDPNode *node = mdps_.at(p)->CreateTerminalNode(node_key);
153  node->set_value(terminal_values[p]);
154  double opponent_reach = OpponentReach(reach_probs, p);
155  SPIEL_CHECK_GE(opponent_reach, 0.0);
156  SPIEL_CHECK_LE(opponent_reach, 1.0);
157  // Following line is not actually necessary because the weight of a leaf
158  // is never in a denominator for a transition probability, but we
159  // include it to keep the semantics of the values consistent across the
160  // ISMDP.
161  node->add_weight(opponent_reach);
162  MDPNode *parent_node = parent_nodes[p];
163  SPIEL_CHECK_TRUE(parent_node != nullptr);
164  parent_node->IncTransitionWeight(parent_actions[p], node,
165  opponent_reach);
166  }
167  }
168  } else if (state.IsChanceNode()) {
169  ActionsAndProbs outcomes_and_probs = state.ChanceOutcomes();
170  for (const auto &[outcome, prob] : outcomes_and_probs) {
171  std::unique_ptr<State> state_copy = state.Clone();
172  state_copy->ApplyAction(outcome);
173  std::vector<double> new_reach_probs = reach_probs;
174  // Chance prob is at the end of the vector.
175  new_reach_probs[game_.NumPlayers()] *= prob;
176  BuildMDPs(*state_copy, new_reach_probs, parent_nodes, parent_actions,
177  only_for_player);
178  }
179  } else if (state.IsSimultaneousNode()) {
180  // Several nodes are created: one for each player as the maximizer.
181  std::vector<std::string> node_keys(num_players_);
182  std::vector<MDPNode*> nodes(num_players_, nullptr);
183  std::vector<double> opponent_reaches(num_players_, 1.0);
184  std::vector<ActionsAndProbs> fixed_state_policies(num_players_);
185 
186  for (Player player = 0; player < num_players_; ++player) {
187  if (only_for_player == kInvalidPlayer || only_for_player == player) {
188  node_keys[player] = GetNodeKey(state, player);
189  nodes[player] = mdps_.at(player)->LookupOrCreateNode(node_keys[player]);
190  opponent_reaches[player] = OpponentReach(reach_probs, player);
191 
192  SPIEL_CHECK_GE(opponent_reaches[player], 0.0);
193  SPIEL_CHECK_LE(opponent_reaches[player], 1.0);
194  nodes[player]->add_weight(opponent_reaches[player]);
195 
196  MDPNode* parent_node = parent_nodes[player];
197  SPIEL_CHECK_TRUE(parent_node != nullptr);
198  parent_node->IncTransitionWeight(parent_actions[player], nodes[player],
199  opponent_reaches[player]);
200  }
201 
202  if (only_for_player == kInvalidPlayer || only_for_player != player) {
203  fixed_state_policies[player] =
204  fixed_policy_.GetStatePolicy(state, player);
205  }
206  }
207 
208  // Traverse over the list of joint actions. For each one, first deconstruct
209  // the actions, and then recurse once for each player as the maximizer with
210  // the others as the fixed policies.
211  const auto& sim_move_state = down_cast<const SimMoveState&>(state);
212  for (Action joint_action : state.LegalActions()) {
213  std::vector<Action> actions =
214  sim_move_state.FlatJointActionToActions(joint_action);
215 
216  std::unique_ptr<State> state_copy = state.Clone();
217  state_copy->ApplyAction(joint_action);
218 
219  std::vector<double> new_reach_probs = reach_probs;
220  std::vector<MDPNode*> new_parent_nodes = parent_nodes;
221  std::vector<Action> new_parent_actions = parent_actions;
222  for (Player player = 0; player < num_players_; ++player) {
223  if (only_for_player == kInvalidPlayer || only_for_player != player) {
224  double action_prob = GetProb(fixed_state_policies[player],
225  actions[player]);
226  SPIEL_CHECK_PROB(action_prob);
227  new_reach_probs[player] *= action_prob;
228  }
229 
230  if (only_for_player == kInvalidPlayer || only_for_player == player) {
231  new_parent_nodes[player] = nodes[player];
232  }
233 
234  new_parent_actions[player] = actions[player];
235  }
236 
237  BuildMDPs(*state_copy, new_reach_probs, new_parent_nodes,
238  new_parent_actions, only_for_player);
239  }
240  } else {
241  // Normal decisions node.
242  std::vector<Action> legal_actions = state.LegalActions();
243  Player player = state.CurrentPlayer();
244  ActionsAndProbs state_policy; // Fixed joint policy we're responding to.
245  MDPNode* node = nullptr;
246 
247  // Check to see if we need to build this node.
248  if (only_for_player == kInvalidPlayer || only_for_player == player) {
249  std::string node_key = GetNodeKey(state, player);
250 
251  node = mdps_.at(player)->LookupOrCreateNode(node_key);
252  double opponent_reach = OpponentReach(reach_probs, player);
253 
254  SPIEL_CHECK_GE(opponent_reach, 0.0);
255  SPIEL_CHECK_LE(opponent_reach, 1.0);
256  node->add_weight(opponent_reach);
257  MDPNode *parent_node = parent_nodes[player];
258  SPIEL_CHECK_TRUE(parent_node != nullptr);
259  parent_node->IncTransitionWeight(parent_actions[player], node,
260  opponent_reach);
261  }
262 
263  // Get the fixed policy all the time if building all MDPs, or only at
264  // opponent nodes otherwise
265  if (only_for_player == kInvalidPlayer || only_for_player != player) {
266  state_policy = fixed_policy_.GetStatePolicy(state);
267  }
268 
269  for (Action action : legal_actions) {
270  std::unique_ptr<State> state_copy = state.Clone();
271  state_copy->ApplyAction(action);
272 
273  std::vector<double> new_reach_probs = reach_probs;
274  std::vector<MDPNode*> new_parent_nodes = parent_nodes;
275 
276  // If building all MDPs at once, modify reach probs in all cases.
277  // Otherwise, only at opponent nodes.
278  if (only_for_player == kInvalidPlayer || only_for_player != player) {
279  double action_prob = GetProb(state_policy, action);
280  SPIEL_CHECK_PROB(action_prob);
281  new_reach_probs[player] *= action_prob;
282  }
283 
284  // If building all MDPs at once, modify parent nodes for that MDP.
285  // Otherwise, only do it for the player we're building the MDP for.
286  if (only_for_player == kInvalidPlayer || only_for_player == player) {
287  new_parent_nodes[player] = node;
288  }
289 
290  std::vector<Action> new_parent_actions = parent_actions;
291  new_parent_actions[player] = action;
292 
293  BuildMDPs(*state_copy, new_reach_probs, new_parent_nodes,
294  new_parent_actions, only_for_player);
295  }
296  }
297 }

333 TabularBestResponseMDPInfo TabularBestResponseMDP::ComputeBestResponses() {
334  TabularBestResponseMDPInfo br_info(num_players_);
335 
336  // Initialize IS-MDPs for each player, if necessary.
337  if (mdps_.empty()) {
338  for (Player p = 0; p < num_players_; p++) {
339  mdps_.push_back(absl::make_unique<MDP>());
340  }
341  }
342 
343  std::vector<MDPNode*> parent_nodes;
344  parent_nodes.reserve(num_players_);
345  for (Player p = 0; p < num_players_; p++) {
346  parent_nodes.push_back(mdps_[p]->RootNode());
347  }
348  std::vector<double> reach_probs(num_players_ + 1, 1.0); // include chance.
349  std::vector<Action> parent_actions(num_players_, 0);
350 
351  std::unique_ptr<State> initial_state = game_.NewInitialState();
352  BuildMDPs(*initial_state, reach_probs, parent_nodes, parent_actions);
353 
354  for (Player p = 0; p < num_players_; p++) {
355  br_info.br_values[p] =
356  mdps_[p]->Solve(kSolveTolerance, &br_info.br_policies[p]);
357  }
358 
359  return br_info;
360 }

377  BuildMDPs(*initial_state, reach_probs, parent_nodes, parent_actions,
378  max_player);
379 
380  br_info.br_values[max_player] =
381  mdps_[max_player]->Solve(kSolveTolerance,
382  &br_info.br_policies[max_player]);
383  return br_info;

386 TabularBestResponseMDPInfo TabularBestResponseMDP::NashConv() {
387  TabularBestResponseMDPInfo br_info = ComputeBestResponses();
388  std::unique_ptr<State> state = game_.NewInitialState();
389  br_info.on_policy_values =
390  ExpectedReturns(*state, fixed_policy_,
391  /*depth_limit*/ -1, /*use_infostate_get_policy*/ false);
392  for (Player p = 0; p < num_players_; ++p) {
393  br_info.deviation_incentives[p] =
394  br_info.br_values[p] - br_info.on_policy_values[p];
395  br_info.nash_conv += br_info.deviation_incentives[p];
396  }
397  return br_info;
398 }

400 TabularBestResponseMDPInfo TabularBestResponseMDP::Exploitability() {
401  SPIEL_CHECK_TRUE(game_.GetType().utility == GameType::Utility::kZeroSum ||
402  game_.GetType().utility == GameType::Utility::kConstantSum);
403  TabularBestResponseMDPInfo br_info = ComputeBestResponses();
404  br_info.nash_conv = absl::c_accumulate(br_info.br_values, 0.0);
405  br_info.exploitability =
406  (br_info.nash_conv - *game_.UtilitySum()) / num_players_;
407  return br_info;
