56  def _epsilon_greedy(self, info_state, legal_actions, epsilon):
57  """Returns a valid epsilon-greedy action and valid action probs.
58 
59  If the agent has not been to `info_state`, a valid random action is chosen.
60 
61  Args:
62  info_state: hashable representation of the information state.
63  legal_actions: list of actions at `info_state`.
64  epsilon: float, prob of taking an exploratory action.
65 
66  Returns:
67  A valid epsilon-greedy action and valid action probabilities.
68  """
69  probs = np.zeros(self._num_actions_num_actions)
70  greedy_q = max([self._q_values_q_values[info_state][a] for a in legal_actions])
71  greedy_actions = [
72  a for a in legal_actions if self._q_values_q_values[info_state][a] == greedy_q
73  ]
74  probs[legal_actions] = epsilon / len(legal_actions)
75  probs[greedy_actions] += (1 - epsilon) / len(greedy_actions)
76  action = np.random.choice(range(self._num_actions_num_actions), p=probs)
77  return action, probs

79  def _get_action_probs(self, info_state, legal_actions, epsilon):
80  """Returns a selected action and the probabilities of legal actions.
81 
82  To be overwritten by subclasses that implement other action selection
83  methods.
84 
85  Args:
86  info_state: hashable representation of the information state.
87  legal_actions: list of actions at `info_state`.
88  epsilon: float: current value of the epsilon schedule or 0 in case
89  evaluation. QLearner uses it as the exploration parameter in
90  epsilon-greedy, but subclasses are free to interpret in different ways
91  (e.g. as temperature in softmax).
92  """
93  return self._epsilon_greedy_epsilon_greedy(info_state, legal_actions, epsilon)

95  def step(self, time_step, is_evaluation=False):
96  """Returns the action to be taken and updates the Q-values if needed.
97 
98  Args:
99  time_step: an instance of rl_environment.TimeStep.
100  is_evaluation: bool, whether this is a training or evaluation call.
101 
102  Returns:
103  A `rl_agent.StepOutput` containing the action probs and chosen action.
104  """
105  if self._centralized_centralized:
106  info_state = str(time_step.observations["info_state"])
107  else:
108  info_state = str(time_step.observations["info_state"][self._player_id_player_id])
109  legal_actions = time_step.observations["legal_actions"][self._player_id_player_id]
110 
111  # Prevent undefined errors if this agent never plays until terminal step
112  action, probs = None, None
113 
114  # Act step: don't act at terminal states.
115  if not time_step.last():
116  epsilon = 0.0 if is_evaluation else self._epsilon_epsilon
117  action, probs = self._get_action_probs_get_action_probs(info_state, legal_actions, epsilon)
118 
119  # Learn step: don't learn during evaluation or at first agent steps.
120  if self._prev_info_state_prev_info_state and not is_evaluation:
121  target = time_step.rewards[self._player_id_player_id]
122  if not time_step.last(): # Q values are zero for terminal.
123  target += self._discount_factor_discount_factor * max(
124  [self._q_values_q_values[info_state][a] for a in legal_actions])
125 
126  prev_q_value = self._q_values_q_values[self._prev_info_state_prev_info_state][self._prev_action_prev_action]
127  self._last_loss_value_last_loss_value = target - prev_q_value
128  self._q_values_q_values[self._prev_info_state_prev_info_state][self._prev_action_prev_action] += (
129  self._step_size_step_size * self._last_loss_value_last_loss_value)
130 
131  # Decay epsilon, if necessary.
132  self._epsilon_epsilon = self._epsilon_schedule_epsilon_schedule.step()
133 
134  if time_step.last(): # prepare for the next episode.
135  self._prev_info_state_prev_info_state = None
136  return
137 
138  # Don't mess up with the state during evaluation.
139  if not is_evaluation:
140  self._prev_info_state_prev_info_state = info_state
141  self._prev_action_prev_action = action
142  return rl_agent.StepOutput(action=action, probs=probs)
