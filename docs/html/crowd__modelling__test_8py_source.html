101  def test_compare_py_cpp(self):
102  """Compares py and cpp implementations of this game."""
103  py_game = pyspiel.load_game("python_mfg_crowd_modelling")
104  cpp_game = pyspiel.load_game("mfg_crowd_modelling")
105  np.random.seed(7)
106  py_state = py_game.new_initial_state()
107  cpp_state = cpp_game.new_initial_state()
108  t = 0
109  while not cpp_state.is_terminal():
110  self.assertFalse(py_state.is_terminal())
111  self.assertEqual(str(cpp_state), str(py_state))
112  self.assertAlmostEqual(cpp_state.returns()[0], py_state.returns()[0])
113  if cpp_state.current_player() == pyspiel.PlayerId.CHANCE:
114  actions, probs = zip(*cpp_state.chance_outcomes())
115  action = np.random.choice(actions, p=probs)
116  self.assertEqual(
117  cpp_state.action_to_string(action),
118  py_state.action_to_string(action))
119  cpp_state.apply_action(action)
120  py_state.apply_action(action)
121  elif cpp_state.current_player() == pyspiel.PlayerId.MEAN_FIELD:
122  num_cpp_states = len(cpp_state.distribution_support())
123  distribution = [1 / num_cpp_states] * num_cpp_states
124  cpp_state.update_distribution(distribution)
125  py_state.update_distribution(distribution)
126  else:
127  self.assertEqual(cpp_state.current_player(), 0)
128  legal_actions = cpp_state.legal_actions()
129  action = np.random.choice(legal_actions)
130  self.assertEqual(
131  cpp_state.action_to_string(action),
132  py_state.action_to_string(action))
133  cpp_state.apply_action(action)
134  py_state.apply_action(action)
135  t += 1
