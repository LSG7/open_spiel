127 def main(argv: Sequence[str]) -> None:
128  if len(argv) > 1:
129  raise app.UsageError('Too many command-line arguments.')

131  game = factory.create_game_with_setting(_GAME_NAME.value, _ENV_SETTING.value)
132  num_players = game.num_players()
133 
134  # Create the environments with uniform initial policy.
135  uniform_policy = policy_std.UniformRandomPolicy(game)
136  uniform_dist = distribution.DistributionPolicy(game, uniform_policy)
137 
138  envs = [
139  rl_environment.Environment(
140  game, mfg_distribution=uniform_dist, mfg_population=p)
141  for p in range(num_players)
142  ]
143 
144  env = envs[0]
145  info_state_size = env.observation_spec()['info_state'][0]
146  num_actions = env.action_spec()['num_actions']
147 
148  # Best response policy agents.
149  kwargs_dqn = {
150  'batch_size': _BATCH_SIZE.value,
151  'discount_factor': _DISCOUNT_FACTOR.value,
152  'epsilon_decay_duration': _EPSILON_DECAY_DURATION.value,
153  'epsilon_end': _EPSILON_END.value,
154  'epsilon_start': _EPSILON_START.value,
155  'gradient_clipping': _GRADIENT_CLIPPING.value,
156  'hidden_layers_sizes': [int(l) for l in _HIDDEN_LAYERS_SIZES.value],
157  'huber_loss_parameter': _HUBER_LOSS_PARAMETER.value,
158  'learn_every': _LEARN_EVERY.value,
159  'learning_rate': _LEARNING_RATE.value,
160  'loss_str': _LOSS.value,
161  'min_buffer_size_to_learn': _MIN_BUFFER_SIZE_TO_LEARN.value,
162  'optimizer_str': _OPTIMIZER.value,
163  'replay_buffer_capacity': _REPLAY_BUFFER_CAPACITY.value,
164  'seed': _SEED.value,
165  'update_target_network_every': _UPDATE_TARGET_NETWORK_EVERY.value,
166  }
167  br_rl_agents = [
168  dqn.DQN(p, info_state_size, num_actions, **kwargs_dqn)
169  for p in range(num_players)
170  ]
171 
172  num_training_steps_per_iteration = (
173  _AVG_POL_NUM_TRAINING_STEPS_PER_ITERATION.value)
174 
175  # Metrics writer will also log the metrics to stderr.
176  just_logging = _LOGDIR.value is None or jax.host_id() > 0
177  writer = metrics.create_default_writer(
178  _LOGDIR.value, just_logging=just_logging)
179 
180  def logging_fn(it, step, vals):
181  writer.write_scalars(it * num_training_steps_per_iteration + step, vals)
182 
183  # Average policy agents.
184  kwargs_avg = {
185  'batch_size': _AVG_POL_BATCH_SIZE.value,
186  'hidden_layers_sizes': [
187  int(l) for l in _AVG_POL_HIDDEN_LAYERS_SIZES.value
188  ],
189  'reservoir_buffer_capacity': _AVG_POL_RESERVOIR_BUFFER_CAPACITY.value,
190  'learning_rate': _AVG_POL_LEARNING_RATE.value,
191  'min_buffer_size_to_learn': _AVG_POL_MIN_BUFFER_SIZE_TO_LEARN.value,
192  'optimizer_str': _AVG_POL_OPTIMIZER.value,
193  'gradient_clipping': _AVG_GRADIENT_CLIPPING.value,
194  'seed': _SEED.value,
195  'tau': _AVG_POL_TAU.value
196  }
197  fp = average_network_fictitious_play.AverageNetworkFictitiousPlay(
198  game,
199  envs,
200  br_rl_agents,
201  _AVG_POL_NUM_EPISODES_PER_ITERATION.value,
202  num_training_steps_per_iteration,
203  eval_every=_EVAL_EVERY.value,
204  logging_fn=logging_fn,
205  **kwargs_avg)
206 
207  def log_metrics(it):
208  """Logs the training metrics for each iteration."""
209  initial_states = game.new_initial_states()
210  distrib = distribution.DistributionPolicy(game, fp.policy)
211  pi_value = policy_value.PolicyValue(game, distrib, fp.policy)
212  m = {
213  f'best_response/{state}': pi_value.eval_state(state)
214  for state in initial_states
215  }
216  m.update({
217  f'br_agent{i}/loss': agent.loss for i, agent in enumerate(br_rl_agents)
218  })
219  nash_conv_fp = nash_conv.NashConv(game, fp.policy)
220  m['nash_conv_fp'] = nash_conv_fp.nash_conv()
221  logging_fn(it, 0, m)
222 
223  # Also save the distribution.
224  if _LOG_DISTRIBUTION.value and not just_logging:
225  filename = os.path.join(_LOGDIR.value, f'distribution_{it}.pkl')
226  utils.save_parametric_distribution(nash_conv_fp.distribution, filename)
227 
228  for it in range(_NUM_ITERATIONS.value):
229  # Train the RL agent to learn a best response.
230  training.run_episodes(
231  envs,
232  br_rl_agents,
233  num_episodes=_NUM_DQN_EPISODES_PER_ITERATION.value,
234  is_evaluation=False)
235 
236  # Run an iteration of average-network fictitious play and log the metrics.
237  fp.iteration()
238  log_metrics(it + 1)
239 
240  # Make sure all values were written.
241  writer.flush()
242

244 if __name__ == '__main__':
245  jax.config.parse_flags_with_absl()
246  app.run(main)
