27// A threadsafe-queue.
28template <class T>
29class ThreadedQueue {
30 public:
31 explicit ThreadedQueue(int max_size) : max_size_(max_size) {}
32
33 // Add an element to the queue.
34 bool Push(const T& value) { return Push(value, absl::InfiniteDuration()); }
35 bool Push(const T& value, absl::Duration wait) {
36 return Push(value, absl::Now() + wait);
37 }
38 bool Push(const T& value, absl::Time deadline) {
39 absl::MutexLock lock(&m_);
40 if (block_new_values_) {
41 return false;
42 }
43 while (q_.size() >= max_size_) {
44 if (absl::Now() > deadline || block_new_values_) {
45 return false;
46 }
47 cv_.WaitWithDeadline(&m_, deadline);
48 }
49 q_.push(value);
50 cv_.Signal();
51 return true;
52 }
53
54 absl::optional<T> Pop() { return Pop(absl::InfiniteDuration()); }
55 absl::optional<T> Pop(absl::Duration wait) { return Pop(absl::Now() + wait); }
56 absl::optional<T> Pop(absl::Time deadline) {
57 absl::MutexLock lock(&m_);
58 while (q_.empty()) {
59 if (absl::Now() > deadline || block_new_values_) {
60 return absl::nullopt;
61 }
62 cv_.WaitWithDeadline(&m_, deadline);
63 }
64 T val = q_.front();
65 q_.pop();
66 cv_.Signal();
67 return val;
68 }
69
70 bool Empty() {
71 absl::MutexLock lock(&m_);
72 return q_.empty();
73 }
74
75 void Clear() {
76 absl::MutexLock lock(&m_);
77 while (!q_.empty()) {
78 q_.pop();
79 }
80 }
81
82 int Size() {
83 absl::MutexLock lock(&m_);
84 return q_.size();
85 }
86
87 // Causes pushing new values to fail. Useful for shutting down the queue.
88 void BlockNewValues() {
89 absl::MutexLock lock(&m_);
90 block_new_values_ = true;
91 cv_.SignalAll();
92 }
93
94 private:
95 bool block_new_values_ = false;
96 int max_size_;
97 std::queue<T> q_;
98 absl::Mutex m_;
99 absl::CondVar cv_;
100};
