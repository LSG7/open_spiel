26// Checks that the expected value of the policy is close to the Nash values.
27// Assumes two-player zero-sum games.
28void CheckNashValues(const Game& game, const Policy& policy,
29 double first_player_nash_value, double tolerance) {
30 const std::vector<double> game_value =
31 ExpectedReturns(*game.NewInitialState(), policy, -1);
32 SPIEL_CHECK_EQ(2, game_value.size());
33 SPIEL_CHECK_FLOAT_NEAR(game_value[0], first_player_nash_value, tolerance);
34 SPIEL_CHECK_FLOAT_NEAR(game_value[1], -first_player_nash_value, tolerance);
35}

37void CFRBRTest_KuhnPoker() {
38 std::shared_ptr<const Game> game = LoadGame("kuhn_poker");
39 CFRBRSolver solver(*game);
40 for (int i = 0; i < 300; i++) {
41 solver.EvaluateAndUpdatePolicy();
42 }
43 const std::shared_ptr<Policy> average_policy = solver.AveragePolicy();
44 // 1/18 is the Nash value. See https://en.wikipedia.org/wiki/Kuhn_poker
45 CheckNashValues(*game, *average_policy, -1.0 / 18, 0.001);
46 SPIEL_CHECK_LE(Exploitability(*game, *average_policy), 0.05);
47}

62void CFRBRTest_CFRBRSolverSerialization() {
63 auto game = LoadGame("kuhn_poker");
64 CFRBRSolver solver = CFRBRSolver(*game);
65 double exploitability0 = Exploitability(*game, *solver.AveragePolicy());
66
67 for (int i = 0; i < 50; i++) {
68 solver.EvaluateAndUpdatePolicy();
69 }
70 double exploitability1 = Exploitability(*game, *solver.AveragePolicy());
71 SPIEL_CHECK_GT(exploitability0, exploitability1);
72
73 std::string serialized = solver.Serialize();
74 std::unique_ptr<CFRBRSolver> deserialized_solver =
75 DeserializeCFRBRSolver(serialized);
76 SPIEL_CHECK_EQ(solver.InfoStateValuesTable().size(),
77 deserialized_solver->InfoStateValuesTable().size());
78 double exploitability2 =
79 Exploitability(*game, *deserialized_solver->AveragePolicy());
80 SPIEL_CHECK_FLOAT_NEAR(exploitability1, exploitability2, 1e-4);
81
82 for (int i = 0; i < 50; i++) {
83 deserialized_solver->EvaluateAndUpdatePolicy();
84 }
85 double exploitability3 =
86 Exploitability(*game, *deserialized_solver->AveragePolicy());
87 SPIEL_CHECK_GT(exploitability2, exploitability3);
88}
