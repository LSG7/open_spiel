26// This is the cooperative box-pushing domain presented by Seuken & Zilberstein
27// in their paper "Improved Memory-Bounded Dynamic Programming for Dec-POMDPs"
28// http://rbr.cs.umass.edu/papers/SZuai07.pdf
29//
30// Parameters:
31// "fully_observable" bool agents see everything, or only partial view as
32// described in the original paper (def: false)
33// "horizon" int length of horizon (default = 100)
34

38// To indicate the status of each agent's action.
39enum class ActionStatusType {
40 kUnresolved,
41 kSuccess,
42 kFail,
43};
44
45// Direction each agent can be facing.
46enum OrientationType {
47 kNorth = 0,
48 kEast = 1,
49 kSouth = 2,
50 kWest = 3,
51 kInvalid = 4
52};

54// When not fully-observable, the number of observations (taken from Seuken &
55// Zilberstein '12): empty field, wall, other agent, small box, large box.
56enum ObservationType {
57 kEmptyFieldObs,
58 kWallObs,
59 kOtherAgentObs,
60 kSmallBoxObs,
61 kBigBoxObs
62};
63constexpr int kNumObservations = 5;

65// Different actions used by the agent.
66enum class ActionType { kTurnLeft, kTurnRight, kMoveForward, kStay };
