24// This is the generic superclass for simultaneous move games. A simultaneous
25// move game (AKA Markov Game) is one where all agents submit actions on each
26// time step, and the state transists to a new state as a function of the joint
27// action.
28//
29// Note that this implementation also supports mixed turn-based and simultaneous
30// steps at which it is only a single player submits an action, when not at
31// simultaneous nodes and not at chance nodes.
32//
33// For normal-form or matrix games, see normal_form_game.h or
34// matrix.h.
35

40class SimMoveState : public State {
41 public:
42 SimMoveState(std::shared_ptr<const Game> game) : State(game) {}
43 SimMoveState(const SimMoveState&) = default;
44
45 // Subclasses must implement a per-player LegalActions function.
46 std::vector<Action> LegalActions(Player player) const override = 0;
47
48 // LegalActions() returns either the chance outcomes (at a chance node), a
49 // flattened form of the joint legal actions (at simultaneous move nodes) -
50 // see discussion below, or the actions for the current player (at nodes
51 // where only a single player is making a decision).
52 std::vector<Action> LegalActions() const override {
53 if (IsSimultaneousNode()) {
54 return LegalFlatJointActions();
55 } else if (IsTerminal()) {
56 return {};
57 } else if (IsChanceNode()) {
58 return LegalChanceOutcomes();
59 } else {
60 return LegalActions(CurrentPlayer());
61 }
62 }
63
64 // We override this rather than DoApplyAction() since we want to prevent
65 // saving the flat action in the history.
66 void ApplyAction(Action action) override {
67 if (IsSimultaneousNode()) {
68 ApplyFlatJointAction(action);
69 } else {
70 const Player player = CurrentPlayer();
71 DoApplyAction(action);
72 history_.push_back({player, action});
73 }
74 }
75
76 // Convert a flat joint action to a list of actions.
77 std::vector<Action> FlatJointActionToActions(Action flat_action) const;
78
79 protected:
80 // To make the implementation of algorithms which traverse the whole game
81 // tree easier, we support the mapping of joint actions (one per player)
82 // to a single flat action taken by the player kSimultaneousPlayerId.
83
84 // If we have three players with legal sets (a, b), (x, y, z), (p, q)
85 // respectively, then their 12 possible joint actions will be numbered as
86 // follows:
87 // 0 - (a, x, p)
88 // 1 - (b, x, p)
89 // 2 - (a, y, p)
90 // ...
91 // 10 - (b, y, q)
92 // 11 - (a, z, q)
93 // 12 - (b, z, q)
94
95 // Implementors of simultaneous move games don't have to worry about this
96 // mapping, but simply check for player == kSimultaneousPlayerId and forward
97 // method calls as follows:
98 // ActionToString --> FlatJointActionToString
99 // LegalActions --> LegalFlatJointActions
100 // ApplyAction --> ApplyFlatJointAction
101
102 // Since we repeatedly index into the list of legal actions, it is necessary
103 // that LegalActions returns the same list (in the same order) when called
104 // twice on the same state.
105
106 // if the number of legal actions overflows int64, this will of course not
107 // work correctly.
108
109 // Map a flat joint action for the simultaneous player to a string, in the
110 // form "[action1, action2, ...]".
111 std::string FlatJointActionToString(Action flat_action) const;
112
113 // Return a list of legal flat joint actions. See above for details.
114 std::vector<Action> LegalFlatJointActions() const;
115
116 // Apply a flat joint action, updating the state.
117 void ApplyFlatJointAction(Action flat_action);
118
119 void DoApplyActions(const std::vector<Action>& actions) override = 0;
120};
