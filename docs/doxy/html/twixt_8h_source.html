45 bool IsTerminal() const override {
46 int result = board_.result();
47 return (result == kRedWin || result == kBlueWin || result == kDraw);
48 };
49
50 std::vector<double> Returns() const override {
51 double reward;
52 int result = board_.result();
53 if (result == kOpen || result == kDraw) {
54 return {0.0, 0.0};
55 } else {
56 reward = 1.0;
57 if (result == kRedWin) {
58 return {reward, -reward};
59 } else {
60 return {-reward, reward};
61 }
62 }
63 };

86 std::vector<Action> LegalActions() const override {
87 if (IsTerminal()) return {};
88 return board_.GetLegalActions(current_player_);

90
91 protected:
92 void DoApplyAction(Action action) override {
93 const std::vector<Action> &v = LegalActions();
94 if (std::find(v.begin(), v.end(), action) == v.end()) {
95 SpielFatalError("Not a legal action: " + std::to_string(action));
96 }
97 board_.ApplyAction(CurrentPlayer(), action);
98 if (board_.result() == kOpen) {
99 set_current_player(1 - CurrentPlayer());
100 } else {
101 set_current_player(kTerminalPlayerId);
102 }
103 };
