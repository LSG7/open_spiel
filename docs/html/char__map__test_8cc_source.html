76 std::string ClosedRangeString(unsigned char lo, unsigned char hi) {
77  // Don't depend on lo<hi. Just increment until lo==hi.
78  std::string s;
79  while (true) {
80  s.push_back(lo);
81  if (lo == hi) break;
82  ++lo;
83  }
84  return s;
85 }
