70class NoopResetEnv(gym.Wrapper):
71 """Sample initial states by taking random number of no-ops on reset.
72
73 No-op is assumed to be action 0. :param env: the environment to wrap :param
74 noop_max: the maximum value of no-ops to run
75 """
76
77 def __init__(self, env: gym.Env, noop_max: int = 30):
78 gym.Wrapper.__init__(self, env)
79 self.noop_max = noop_max
80 self.override_num_noops = None
81 self.noop_action = 0
82 assert env.unwrapped.get_action_meanings()[0] == 'NOOP'
83
84 def reset(self, **kwargs) -> np.ndarray:
85 self.env.reset(**kwargs)
86 if self.override_num_noops is not None:
87 noops = self.override_num_noops
88 else:
89
90 noops = self.unwrapped.np_random.integers(1, self.noop_max + 1)
91
92 assert noops > 0
93 obs = np.zeros(0)
94 for _ in range(noops):
95 obs, _, done, _ = self.env.step(self.noop_action)
96 if done:
97 obs = self.env.reset(**kwargs)
98 return obs

101class AtariGame(pyspiel.Game):
102 """An OpenSpiel wrapper for the OpenAI Gym Atari games."""
103
104 def __init__(self, params=None):
105 super().__init__(_GAME_TYPE, _GAME_INFO, params or dict())
106 self.gym_id = params.get('gym_id', 'BreakoutNoFrameskip-v4')
107 self.seed = params.get('seed', 1)
108 self.idx = params.get('idx', 0)
109 self.capture_video = params.get('capture_video', False)
110 self.run_name = params.get('run_name', 'default')
111 self.use_episodic_life_env = params.get('use_episodic_life_env', True)
112
113 env = gym.make(self.gym_id)
114 env = gym.wrappers.RecordEpisodeStatistics(env)
115 if self.capture_video and self.idx == 0:
116 env = gym.wrappers.RecordVideo(env, f'videos/{self.run_name}')
117
118 # Apply the standard set of wrappers from CleanRL's PPO implementation.
119 # These wrappers have been tested on Breakout; different games may
120 # benefit from different wrappers (e.g., Space Invaders might benefit
121 # from frameskip=3 instead of 4; see https://arxiv.org/abs/1312.5602).
122 env = NoopResetEnv(env, noop_max=30)
123 env = MaxAndSkipEnv(env, skip=4)
124 if self.use_episodic_life_env:
125 env = EpisodicLifeEnv(env)
126 if 'FIRE' in env.unwrapped.get_action_meanings():
127 env = FireResetEnv(env)
128 env = ClipRewardEnv(env)
129 env = gym.wrappers.ResizeObservation(env, (84, 84))
130 env = gym.wrappers.GrayScaleObservation(env)
131 env = gym.wrappers.FrameStack(env, 4)
132 env.seed(self.seed)
133 env.action_space.seed(self.seed)
134 env.observation_space.seed(self.seed)
135 self.observation_shape = env.reset().shape
136 self.env = env
137
138 def observation_tensor_shape(self):
139 return self.observation_shape
140
141 def new_initial_state(self):
142 """Returns a state corresponding to the start of a game."""
143 return AtariState(self)
144
145 def make_py_observer(self, iig_obs_type=None, params=None):
146 """Returns an object used for observing game state."""
147 if params is None:
148 params = dict()
149
150 params['observation_shape'] = self.observation_shape
151 return AtariObserver(
152 iig_obs_type or pyspiel.IIGObservationType(perfect_recall=False),
153 params)

156class AtariState(pyspiel.State):
157 """A python version of the Atari Game state."""
158
159 def __init__(self, game):
160 """Constructor; should only be called by Game.new_initial_state."""
161 super().__init__(game)
162 self._is_terminal = False
163 self.tracked_rewards = 0
164 self.env = game.env
165 self.observation = self.env.reset()
166 self.last_reward = None
167 self.last_info = dict()
168
169 def current_player(self):
170 """Returns id of the next player to move, or TERMINAL if game is over."""
171 return pyspiel.PlayerId.TERMINAL if self._is_terminal else 0
172
173 def _legal_actions(self, player):
174 """Returns a list of legal actions, sorted in ascending order."""
175 return list(range(self.env.action_space.n))
176
177 def _apply_action(self, action):
178 """Applies the specified action to the state."""
179 observation, reward, done, info = self.env.step(action)
180 self.last_info = info
181 self.last_reward = reward
182 self.tracked_rewards += reward
183 if done:
184 self._is_terminal = True
185 self.observation = observation # Store this for later
186
187 def _action_to_string(self, player, action):
188 return self.env.get_action_meanings()[action]
189
190 def is_terminal(self):
191 """Returns True if the game is over."""
192 return self._is_terminal
193
194 def rewards(self):
195 return [self.last_reward]
196
197 def returns(self):
198 """Total reward for each player over the course of the game so far."""
199 return [self.tracked_rewards]
200
201 def __str__(self):
202 """String for debug purposes. No particular semantics are required."""
203 return 'DEBUG'

206class AtariObserver:
207 """Observer, conforming to the PyObserver interface (see observation.py)."""
208
209 # pylint: disable=unused-argument
210 def __init__(self, iig_obs_type, params):
211 """Initializes an empty observation tensor."""
212 # Determine which observation pieces we want to include.
213 pieces = []
214 pieces.append(('observation', prod(params['observation_shape']),
215 params['observation_shape']))
216
217 # Build the single flat tensor.
218 total_size = sum(size for name, size, shape in pieces)
219 self.tensor = np.zeros((total_size), np.float32)
220
221 # Build the named & reshaped views of the bits of the flat tensor.
222 self.dict = {}
223 index = 0
224 for name, size, shape in pieces:
225 self.dict[name] = self.tensor[index:index + size].reshape(shape)
226 index += size
227
228 def set_from(self, state, player):
229 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
230 self.tensor.fill(0)
231 if 'observation' in self.dict:
232 self.dict['observation'][:] = state.observation
233
234 def string_from(self, state, player):
235 """Observation of `state` from the PoV of `player`, as a string."""
236 pieces = []
237 return ' '.join(str(p) for p in pieces)
