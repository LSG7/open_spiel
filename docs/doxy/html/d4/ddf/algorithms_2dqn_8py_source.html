43 def __init__(self,
44 session,
45 player_id,
46 state_representation_size,
47 num_actions,
48 hidden_layers_sizes=128,
49 replay_buffer_capacity=10000,
50 batch_size=128,
51 replay_buffer_class=ReplayBuffer,
52 learning_rate=0.01,
53 update_target_network_every=1000,
54 learn_every=10,
55 discount_factor=1.0,
56 min_buffer_size_to_learn=1000,
57 epsilon_start=1.0,
58 epsilon_end=0.1,
59 epsilon_decay_duration=int(1e6),
60 optimizer_str="sgd",
61 loss_str="mse"):
62 """Initialize the DQN agent."""
63
64 # This call to locals() is used to store every argument used to initialize
65 # the class instance, so it can be copied with no hyperparameter change.
66 self._kwargs = locals()
67
68 self.player_id = player_id
69 self._session = session
70 self._num_actions = num_actions
71 if isinstance(hidden_layers_sizes, int):
72 hidden_layers_sizes = [hidden_layers_sizes]
73 self._layer_sizes = hidden_layers_sizes
74 self._batch_size = batch_size
75 self._update_target_network_every = update_target_network_every
76 self._learn_every = learn_every
77 self._min_buffer_size_to_learn = min_buffer_size_to_learn
78 self._discount_factor = discount_factor
79
80 self._epsilon_start = epsilon_start
81 self._epsilon_end = epsilon_end
82 self._epsilon_decay_duration = epsilon_decay_duration
83
84 # TODO(author6) Allow for optional replay buffer config.
85 if not isinstance(replay_buffer_capacity, int):
86 raise ValueError("Replay buffer capacity not an integer.")
87 self._replay_buffer = replay_buffer_class(replay_buffer_capacity)
88 self._prev_timestep = None
89 self._prev_action = None
90
91 # Step counter to keep track of learning, eps decay and target network.
92 self._step_counter = 0
93
94 # Keep track of the last training loss achieved in an update step.
95 self._last_loss_value = None
96
97 # Create required TensorFlow placeholders to perform the Q-network updates.
98 self._info_state_ph = tf.placeholder(
99 shape=[None, state_representation_size],
100 dtype=tf.float32,
101 name="info_state_ph")
102 self._action_ph = tf.placeholder(
103 shape=[None], dtype=tf.int32, name="action_ph")
104 self._reward_ph = tf.placeholder(
105 shape=[None], dtype=tf.float32, name="reward_ph")
106 self._is_final_step_ph = tf.placeholder(
107 shape=[None], dtype=tf.float32, name="is_final_step_ph")
108 self._next_info_state_ph = tf.placeholder(
109 shape=[None, state_representation_size],
110 dtype=tf.float32,
111 name="next_info_state_ph")
112 self._legal_actions_mask_ph = tf.placeholder(
113 shape=[None, num_actions],
114 dtype=tf.float32,
115 name="legal_actions_mask_ph")
116
117 self._q_network = simple_nets.MLP(state_representation_size,
118 self._layer_sizes, num_actions)
119 self._q_values = self._q_network(self._info_state_ph)
120
121 self._target_q_network = simple_nets.MLP(state_representation_size,
122 self._layer_sizes, num_actions)
123 self._target_q_values = self._target_q_network(self._next_info_state_ph)
124
125 # Stop gradient to prevent updates to the target network while learning
126 self._target_q_values = tf.stop_gradient(self._target_q_values)
127
128 self._update_target_network = self._create_target_network_update_op(
129 self._q_network, self._target_q_network)
130
131 # Create the loss operations.
132 # Sum a large negative constant to illegal action logits before taking the
133 # max. This prevents illegal action values from being considered as target.
134 illegal_actions = 1 - self._legal_actions_mask_ph
135 illegal_logits = illegal_actions * ILLEGAL_ACTION_LOGITS_PENALTY
136 max_next_q = tf.reduce_max(
137 tf.math.add(tf.stop_gradient(self._target_q_values), illegal_logits),
138 axis=-1)
139 target = (
140 self._reward_ph +
141 (1 - self._is_final_step_ph) * self._discount_factor * max_next_q)
142
143 action_indices = tf.stack(
144 [tf.range(tf.shape(self._q_values)[0]), self._action_ph], axis=-1)
145 predictions = tf.gather_nd(self._q_values, action_indices)
146
147 self._savers = [("q_network", tf.train.Saver(self._q_network.variables)),
148 ("target_q_network",
149 tf.train.Saver(self._target_q_network.variables))]
150
151 if loss_str == "mse":
152 loss_class = tf.losses.mean_squared_error
153 elif loss_str == "huber":
154 loss_class = tf.losses.huber_loss
155 else:
156 raise ValueError("Not implemented, choose from 'mse', 'huber'.")
157
158 self._loss = tf.reduce_mean(
159 loss_class(labels=target, predictions=predictions))
160
161 if optimizer_str == "adam":
162 self._optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate)
163 elif optimizer_str == "sgd":
164 self._optimizer = tf.train.GradientDescentOptimizer(
165 learning_rate=learning_rate)
166 else:
167 raise ValueError("Not implemented, choose from 'adam' and 'sgd'.")
168
169 self._learn_step = self._optimizer.minimize(self._loss)
170 self._initialize()

172 def get_step_counter(self):
173 return self._step_counter

175 def step(self, time_step, is_evaluation=False, add_transition_record=True):
176 """Returns the action to be taken and updates the Q-network if needed.
177
178 Args:
179 time_step: an instance of rl_environment.TimeStep.
180 is_evaluation: bool, whether this is a training or evaluation call.
181 add_transition_record: Whether to add to the replay buffer on this step.
182
183 Returns:
184 A `rl_agent.StepOutput` containing the action probs and chosen action.
185 """
186
187 # Act step: don't act at terminal info states or if its not our turn.
188 if (not time_step.last()) and (
189 time_step.is_simultaneous_move() or
190 self.player_id == time_step.current_player()):
191 info_state = time_step.observations["info_state"][self.player_id]
192 legal_actions = time_step.observations["legal_actions"][self.player_id]
193 epsilon = self._get_epsilon(is_evaluation)
194 action, probs = self._epsilon_greedy(info_state, legal_actions, epsilon)
195 else:
196 action = None
197 probs = []
198
199 # Don't mess up with the state during evaluation.
200 if not is_evaluation:
201 self._step_counter += 1
202
203 if self._step_counter % self._learn_every == 0:
204 self._last_loss_value = self.learn()
205
206 if self._step_counter % self._update_target_network_every == 0:
207 self._session.run(self._update_target_network)
208
209 if self._prev_timestep and add_transition_record:
210 # We may omit record adding here if it's done elsewhere.
211 self.add_transition(self._prev_timestep, self._prev_action, time_step)
212
213 if time_step.last(): # prepare for the next episode.
214 self._prev_timestep = None
215 self._prev_action = None
216 return
217 else:
218 self._prev_timestep = time_step
219 self._prev_action = action
220
221 return rl_agent.StepOutput(action=action, probs=probs)

223 def add_transition(self, prev_time_step, prev_action, time_step):
224 """Adds the new transition using `time_step` to the replay buffer.
225
226 Adds the transition from `self._prev_timestep` to `time_step` by
227 `self._prev_action`.
228
229 Args:
230 prev_time_step: prev ts, an instance of rl_environment.TimeStep.
231 prev_action: int, action taken at `prev_time_step`.
232 time_step: current ts, an instance of rl_environment.TimeStep.
233 """
234 assert prev_time_step is not None
235 legal_actions = (time_step.observations["legal_actions"][self.player_id])
236 legal_actions_mask = np.zeros(self._num_actions)
237 legal_actions_mask[legal_actions] = 1.0
238 transition = Transition(
239 info_state=(
240 prev_time_step.observations["info_state"][self.player_id][:]),
241 action=prev_action,
242 reward=time_step.rewards[self.player_id],
243 next_info_state=time_step.observations["info_state"][self.player_id][:],
244 is_final_step=float(time_step.last()),
245 legal_actions_mask=legal_actions_mask)
246 self._replay_buffer.add(transition)

248 def _create_target_network_update_op(self, q_network, target_q_network):
249 """Create TF ops copying the params of the Q-network to the target network.
250
251 Args:
252 q_network: A q-network object that implements provides the `variables`
253 property representing the TF variable list.
254 target_q_network: A target q-net object that provides the `variables`
255 property representing the TF variable list.
256
257 Returns:
258 A `tf.Operation` that updates the variables of the target.
259 """
260 self._variables = q_network.variables[:]
261 self._target_variables = target_q_network.variables[:]
262 assert self._variables
263 assert len(self._variables) == len(self._target_variables)
264 return tf.group([
265 tf.assign(target_v, v)
266 for (target_v, v) in zip(self._target_variables, self._variables)
267 ])

269 def _epsilon_greedy(self, info_state, legal_actions, epsilon):
270 """Returns a valid epsilon-greedy action and valid action probs.
271
272 Action probabilities are given by a softmax over legal q-values.
273
274 Args:
275 info_state: hashable representation of the information state.
276 legal_actions: list of legal actions at `info_state`.
277 epsilon: float, probability of taking an exploratory action.
278
279 Returns:
280 A valid epsilon-greedy action and valid action probabilities.
281 """
282 probs = np.zeros(self._num_actions)
283 if np.random.rand() < epsilon:
284 action = np.random.choice(legal_actions)
285 probs[legal_actions] = 1.0 / len(legal_actions)
286 else:
287 info_state = np.reshape(info_state, [1, -1])
288 q_values = self._session.run(
289 self._q_values, feed_dict={self._info_state_ph: info_state})[0]
290 legal_q_values = q_values[legal_actions]
291 action = legal_actions[np.argmax(legal_q_values)]
292 probs[action] = 1.0
293 return action, probs

295 def _get_epsilon(self, is_evaluation, power=1.0):
296 """Returns the evaluation or decayed epsilon value."""
297 if is_evaluation:
298 return 0.0
299 decay_steps = min(self._step_counter, self._epsilon_decay_duration)
300 decayed_epsilon = (
301 self._epsilon_end + (self._epsilon_start - self._epsilon_end) *
302 (1 - decay_steps / self._epsilon_decay_duration)**power)
303 return decayed_epsilon

305 def learn(self):
306 """Compute the loss on sampled transitions and perform a Q-network update.
307
308 If there are not enough elements in the buffer, no loss is computed and
309 `None` is returned instead.
310
311 Returns:
312 The average loss obtained on this batch of transitions or `None`.
313 """
314
315 if (len(self._replay_buffer) < self._batch_size or
316 len(self._replay_buffer) < self._min_buffer_size_to_learn):
317 return None
318
319 transitions = self._replay_buffer.sample(self._batch_size)
320 info_states = [t.info_state for t in transitions]
321 actions = [t.action for t in transitions]
322 rewards = [t.reward for t in transitions]
323 next_info_states = [t.next_info_state for t in transitions]
324 are_final_steps = [t.is_final_step for t in transitions]
325 legal_actions_mask = [t.legal_actions_mask for t in transitions]
326 loss, _ = self._session.run(
327 [self._loss, self._learn_step],
328 feed_dict={
329 self._info_state_ph: info_states,
330 self._action_ph: actions,
331 self._reward_ph: rewards,
332 self._is_final_step_ph: are_final_steps,
333 self._next_info_state_ph: next_info_states,
334 self._legal_actions_mask_ph: legal_actions_mask,
335 })
336 return loss

346 def save(self, checkpoint_dir):
347 """Saves the q network and the target q-network.
348
349 Note that this does not save the experience replay buffers and should
350 only be used to restore the agent's policy, not resume training.
351
352 Args:
353 checkpoint_dir: directory where checkpoints will be saved.
354 """
355 for name, saver in self._savers:
356 path = saver.save(
357 self._session,
358 self._full_checkpoint_name(checkpoint_dir, name),
359 latest_filename=self._latest_checkpoint_filename(name))
360 logging.info("Saved to path: %s", path)

362 def has_checkpoint(self, checkpoint_dir):
363 for name, _ in self._savers:
364 if tf.train.latest_checkpoint(
365 self._full_checkpoint_name(checkpoint_dir, name),
366 os.path.join(checkpoint_dir,
367 self._latest_checkpoint_filename(name))) is None:
368 return False
369 return True

413 def _initialize(self):
414 initialization_weights = tf.group(
415 *[var.initializer for var in self._variables])
416 initialization_target_weights = tf.group(
417 *[var.initializer for var in self._target_variables])
418 initialization_opt = tf.group(
419 *[var.initializer for var in self._optimizer.variables()])
420
421 self._session.run(
422 tf.group(*[
423 initialization_weights, initialization_target_weights,
424 initialization_opt,
425 ]))

428 variables = [self._session.run(self._q_network.variables)]

432 def copy_with_noise(self, sigma=0.0, copy_weights=True):
433 """Copies the object and perturbates it with noise.
434
435 Args:
436 sigma: gaussian dropout variance term : Multiplicative noise following
437 (1+sigma*epsilon), epsilon standard gaussian variable, multiplies each
438 model weight. sigma=0 means no perturbation.
439 copy_weights: Boolean determining whether to copy model weights (True) or
440 just model hyperparameters.
441
442 Returns:
443 Perturbated copy of the model.
444 """
445 _ = self._kwargs.pop("self", None)
446 copied_object = DQN(**self._kwargs)
447
448 q_network = getattr(copied_object, "_q_network")
449 target_q_network = getattr(copied_object, "_target_q_network")
450
451 if copy_weights:
452 copy_weights = tf.group(*[
453 va.assign(vb * (1 + sigma * tf.random.normal(vb.shape)))
454 for va, vb in zip(q_network.variables, self._q_network.variables)
455 ])
456 self._session.run(copy_weights)
457
458 copy_target_weights = tf.group(*[
459 va.assign(vb * (1 + sigma * tf.random.normal(vb.shape)))
460 for va, vb in zip(target_q_network.variables,
461 self._target_q_network.variables)
462 ])
463 self._session.run(copy_target_weights)
464 return copied_object
