94std::unique_ptr<State> TarokGame::DeserializeState(
95 const std::string& str) const {
96 std::unique_ptr<TarokState> state = NewInitialTarokState();
97 if (str.empty()) return state;
98
99 std::vector<std::string> lines = absl::StrSplit(str, '\n');
100 for (int i = 0; i < lines.size(); i++) {
101 if (i == 0) {
102 // chance node where we presisted the card dealing seed, see
103 // TarokState::DoApplyActionInCardDealing for more info
104 std::tie(state->talon_, state->players_cards_) =
105 DealCards(num_players_, std::stoi(lines.at(i)));
106 state->current_game_phase_ = GamePhase::kBidding;
107 state->current_player_ = 1;
108 state->AddPrivateCardsToInfoStates();
109 } else {
110 state->ApplyAction(std::stol(lines.at(i)));
111 }
112 }
113 return state;
114}

175std::vector<std::vector<Action>> TarokState::TalonSets() const {
176 if (current_game_phase_ != GamePhase::kTalonExchange) return {};
177
178 int num_talon_sets = talon_.size() / selected_contract_->num_talon_exchanges;
179 std::vector<std::vector<Action>> talon_sets;
180 talon_sets.reserve(num_talon_sets);
181
182 auto begin = talon_.begin();
183 for (int i = 0; i < num_talon_sets; i++) {
184 talon_sets.push_back(std::vector<Action>(
185 begin, begin + selected_contract_->num_talon_exchanges));
186 std::advance(begin, selected_contract_->num_talon_exchanges);
187 }
188 return talon_sets;

193std::vector<Action> TarokState::LegalActions() const {
194 // all card actions are encoded as 0, 1, ..., 52, 53 and correspond to card
195 // indices wrt. tarok_parent_game_->card_deck_, card actions are returned:
196 // - in the king calling phase
197 // - by LegalActionsInTalonExchange() after the talon set is selected (i.e.
198 // when discarding the cards)
199 // - by LegalActionsInTricksPlaying()
200 switch (current_game_phase_) {
201 case GamePhase::kCardDealing:
202 // return a dummy action due to implicit stochasticity
203 return {0};
204 case GamePhase::kBidding:
205 return LegalActionsInBidding();
206 case GamePhase::kKingCalling:
207 return {kKingOfHeartsAction, kKingOfDiamondsAction, kKingOfSpadesAction,
208 kKingOfClubsAction};
209 case GamePhase::kTalonExchange:
210 return LegalActionsInTalonExchange();
211 case GamePhase::kTricksPlaying:
212 return LegalActionsInTricksPlaying();
213 case GamePhase::kFinished:
214 return {};
215 }
216}

218std::vector<Action> TarokState::LegalActionsInBidding() const {
219 // actions 1 - 12 correspond to contracts in tarok_parent_game_->contracts_
220 // respectively, action 0 means pass
221 auto it = std::max_element(players_bids_.begin(), players_bids_.end());
222 int max_bid = *it;
223 int max_bid_player = it - players_bids_.begin();
224
225 std::vector<Action> actions;
226 if (current_player_ == 0 &&
227 players_bids_.at(current_player_) == kInvalidBidAction &&
228 AllButCurrentPlayerPassedBidding()) {
229 // no bidding has happened before so forehand can
230 // bid any contract but can't pass
231 actions.insert(actions.end(), {kBidKlopAction, kBidThreeAction});
232 } else if (!AllButCurrentPlayerPassedBidding()) {
233 // other players still playing
234 actions.push_back(kBidPassAction);
235 }
236
237 for (int action = 3; action <= 12; action++) {
238 if (num_players_ == 3 && action >= kBidSoloThreeAction &&
239 action <= kBidSoloOneAction) {
240 // skip solo contracts for three players
241 continue;
242 }
243 if (action < max_bid) {
244 continue;
245 }
246 if ((action > max_bid) ||
247 (action == max_bid && current_player_ <= max_bid_player)) {
248 actions.push_back(action);
249 }
250 }
251 return actions;
252}

254std::vector<Action> TarokState::LegalActionsInTalonExchange() const {
255 if (talon_.size() == 6) {
256 // choosing one of the talon card sets where actions are encoded as
257 // 0, 1, 2, etc. from left to right, i.e. 0 is the leftmost talon set
258 // as returned by TalonSets()
259 std::vector<Action> actions(6 / selected_contract_->num_talon_exchanges);
260 std::iota(actions.begin(), actions.end(), 0);
261 return actions;
262 }
263 // prevent discarding of taroks and kings
264 std::vector<Action> actions;
265 for (auto const& action : players_cards_.at(current_player_)) {
266 const Card& card = ActionToCard(action);
267 if (card.suit != CardSuit::kTaroks && card.points != 5)
268 actions.push_back(action);
269 }
270 // allow discarding of taroks (except of trula) if player has no other choice
271 if (actions.empty()) {
272 for (auto const& action : players_cards_.at(current_player_)) {
273 if (ActionToCard(action).points != 5) actions.push_back(action);
274 }
275 }
276 return actions;
277}

279std::vector<Action> TarokState::LegalActionsInTricksPlaying() const {
280 if (trick_cards_.empty()) {
281 // trick opening, i.e. the current player is choosing
282 // the first card for this trick
283 if (selected_contract_->is_negative)
284 return RemovePagatIfNeeded(players_cards_.at(current_player_));
285 return players_cards_.at(current_player_);
286 } else {
287 // trick following
288 return LegalActionsInTricksPlayingFollowing();
289 }

292std::vector<Action> TarokState::LegalActionsInTricksPlayingFollowing() const {
293 auto [can_follow_suit, cant_follow_suit_but_has_tarok] =
294 CanFollowSuitOrCantButHasTarok();
295
296 CardSuit take_suit;
297 if (can_follow_suit) {
298 take_suit = ActionToCard(trick_cards_.front()).suit;
299 } else if (cant_follow_suit_but_has_tarok) {
300 take_suit = CardSuit::kTaroks;
301 } else {
302 // can't follow suit and doesn't have taroks so any card can be played
303 return players_cards_.at(current_player_);
304 }
305
306 if (selected_contract_->is_negative)
307 return TakeSuitFromPlayerCardsInNegativeContracts(take_suit);
308 else
309 return TakeSuitFromPlayerCardsInPositiveContracts(take_suit);
310}

312std::tuple<bool, bool> TarokState::CanFollowSuitOrCantButHasTarok() const {
313 const Card& opening_card = ActionToCard(trick_cards_.front());
314 bool has_taroks = false;
315 for (auto const& action : players_cards_.at(current_player_)) {
316 const Card& current_card = ActionToCard(action);
317 if (current_card.suit == opening_card.suit) {
318 // note that the second return value is irrelevant in this case
319 return {true, false};
320 }
321 if (current_card.suit == CardSuit::kTaroks) {
322 has_taroks = true;
323 }
324 }
325 return {false, has_taroks};
326}

328std::vector<Action> TarokState::TakeSuitFromPlayerCardsInNegativeContracts(
329 CardSuit suit) const {
330 bool player_has_pagat =
331 ActionInActions(kPagatAction, players_cards_.at(current_player_));
332 if (player_has_pagat && ActionInActions(kMondAction, trick_cards_) &&
333 ActionInActions(kSkisAction, trick_cards_)) {
334 // the emperor trick, i.e. pagat has to be played as it is the only card
335 // that will win the trick
336 return {kPagatAction};
337 }
338
339 absl::optional<Action> action_to_beat = ActionToBeatInNegativeContracts(suit);
340 std::vector<Action> actions;
341
342 if (action_to_beat) {
343 const Card& card_to_beat = ActionToCard(*action_to_beat);
344 auto const& player_cards = players_cards_.at(current_player_);
345 // a higher card only has to be played when the player actually has a higher
346 // card otherwise any card of the suit can be played
347 bool has_higher_card = false;
348 for (auto const& action : player_cards) {
349 const Card& current_card = ActionToCard(action);
350 if (current_card.suit == suit && current_card.rank > card_to_beat.rank) {
351 has_higher_card = true;
352 break;
353 }
354 }
355 // collect the actual cards
356 for (auto const& action : player_cards) {
357 const Card& current_card = ActionToCard(action);
358 if (current_card.suit == suit &&
359 (!has_higher_card || current_card.rank > card_to_beat.rank)) {
360 actions.push_back(action);
361 }
362 }
363 } else {
364 // no need to beat any card so simply return all cards of the correct suit
365 actions = TakeSuitFromPlayerCardsInPositiveContracts(suit);
366 }
367
368 if (player_has_pagat)
369 return RemovePagatIfNeeded(actions);
370 else
371 return actions;
372}

374absl::optional<Action> TarokState::ActionToBeatInNegativeContracts(
375 CardSuit suit) const {
376 // there are two cases where no card has to be beaten; the player is following
377 // a colour suit and there is already at least one tarok in trick_cards_ or
378 // the player is forced to play a tarok and there are no taroks in
379 // trick_cards_
380 bool tarok_in_trick_cards = false;
381 for (auto const& action : trick_cards_) {
382 if (ActionToCard(action).suit == CardSuit::kTaroks) {
383 tarok_in_trick_cards = true;
384 break;
385 }
386 }
387 if ((suit != CardSuit::kTaroks && tarok_in_trick_cards) ||
388 (suit == CardSuit::kTaroks && !tarok_in_trick_cards)) {
389 return {};
390 }
391 // the specified suit should be present in trick_cards_ from here on because
392 // it is either a suit of the opening card or CardSuit::kTaroks with existing
393 // taroks in trick_cards_
394 Action action_to_beat = trick_cards_.front();
395 for (int i = 1; i < trick_cards_.size(); i++) {
396 const Card& card_to_beat = ActionToCard(action_to_beat);
397 const Card& current_card = ActionToCard(trick_cards_.at(i));
398 if (current_card.suit == suit && current_card.rank > card_to_beat.rank)
399 action_to_beat = trick_cards_.at(i);
400 }
401 return action_to_beat;
402}

404std::vector<Action> TarokState::RemovePagatIfNeeded(
405 const std::vector<Action>& actions) const {
406 if (actions.size() > 1) {
407 // mustn't play pagat unless it's the only card, note that actions
408 // can be all player's cards or a subset already filtered by the caller
409 std::vector<Action> actions_no_pagat;
410 for (auto const& action : actions) {
411 if (action != kPagatAction) actions_no_pagat.push_back(action);
412 }
413 return actions_no_pagat;
414 }
415 return actions;
416}

418std::vector<Action> TarokState::TakeSuitFromPlayerCardsInPositiveContracts(
419 CardSuit suit) const {
420 std::vector<Action> actions;
421 for (auto const& action : players_cards_.at(current_player_)) {
422 if (ActionToCard(action).suit == suit) actions.push_back(action);
423 }
424 return actions;

427std::string TarokState::ActionToString(Player player, Action action_id) const {
428 switch (current_game_phase_) {
429 case GamePhase::kCardDealing:
430 // return a dummy action due to implicit stochasticity
431 return "Deal";
432 case GamePhase::kBidding:
433 if (action_id == 0) return "Pass";
434 return ContractNameToString(
435 tarok_parent_game_->contracts_.at(action_id - 1).name);
436 case GamePhase::kKingCalling:
437 case GamePhase::kTricksPlaying:
438 return CardActionToString(action_id);
439 case GamePhase::kTalonExchange:
440 if (talon_.size() == 6) return absl::StrCat("Talon set ", action_id + 1);
441 return CardActionToString(action_id);
442 case GamePhase::kFinished:
443 return "";
444 }
445}

459void TarokState::DoApplyAction(Action action_id) {
460 if (!ActionInActions(action_id, LegalActions())) {
461 SpielFatalError(absl::StrCat("Action ", action_id,
462 " is not valid in the current state."));
463 }
464 switch (current_game_phase_) {
465 case GamePhase::kCardDealing:
466 DoApplyActionInCardDealing();
467 break;
468 case GamePhase::kBidding:
469 DoApplyActionInBidding(action_id);
470 break;
471 case GamePhase::kKingCalling:
472 DoApplyActionInKingCalling(action_id);
473 break;
474 case GamePhase::kTalonExchange:
475 DoApplyActionInTalonExchange(action_id);
476 break;
477 case GamePhase::kTricksPlaying:
478 DoApplyActionInTricksPlaying(action_id);
479 break;
480 case GamePhase::kFinished:
481 SpielFatalError("Calling DoApplyAction in a terminal state.");
482 }
483}

485void TarokState::DoApplyActionInCardDealing() {
486 // do the actual sampling here due to implicit stochasticity
487 do {
488 // seed is persisted for serialization purposes
489 card_dealing_seed_ = tarok_parent_game_->RNG();
490 // hands without taroks are illegal
491 std::tie(talon_, players_cards_) =
492 DealCards(num_players_, card_dealing_seed_);
493 } while (AnyPlayerWithoutTaroks());
494 current_game_phase_ = GamePhase::kBidding;
495 // lower player indices correspond to higher bidding priority,
496 // i.e. 0 is the forehand, num_players - 1 is the dealer
497 current_player_ = 1;
498 AddPrivateCardsToInfoStates();
499}

501bool TarokState::AnyPlayerWithoutTaroks() const {
502 // actions are sorted, i.e. taroks are always at the beginning
503 for (int i = 0; i < num_players_; i++) {
504 if (ActionToCard(players_cards_.at(i).front()).suit != CardSuit::kTaroks) {
505 return true;
506 }
507 }
508 return false;

518void TarokState::DoApplyActionInBidding(Action action_id) {
519 players_bids_.at(current_player_) = action_id;
520 AppendToAllInformationStates(std::to_string(action_id));
521 if (AllButCurrentPlayerPassedBidding()) {
522 FinishBiddingPhase(action_id);
523 AppendToAllInformationStates(";");
524 } else {
525 do {
526 NextPlayer();
527 } while (players_bids_.at(current_player_) == kBidPassAction);
528 AppendToAllInformationStates(",");
529 }

532bool TarokState::AllButCurrentPlayerPassedBidding() const {
533 for (int i = 0; i < num_players_; i++) {
534 if (i == current_player_) continue;
535 if (players_bids_.at(i) != kBidPassAction) return false;
536 }
537 return true;
538}

540void TarokState::FinishBiddingPhase(Action action_id) {
541 declarer_ = current_player_;
542 selected_contract_ = &tarok_parent_game_->contracts_.at(action_id - 1);
543 if (num_players_ == 4 && selected_contract_->needs_king_calling)
544 current_game_phase_ = GamePhase::kKingCalling;
545 else if (selected_contract_->NeedsTalonExchange())
546 current_game_phase_ = GamePhase::kTalonExchange;
547 else
548 StartTricksPlayingPhase();
549}

551void TarokState::DoApplyActionInKingCalling(Action action_id) {
552 called_king_ = action_id;
553 if (ActionInActions(action_id, talon_)) {
554 called_king_in_talon_ = true;
555 } else {
556 for (int i = 0; i < num_players_; i++) {
557 if (i == current_player_) {
558 continue;
559 } else if (ActionInActions(action_id, players_cards_.at(i))) {
560 declarer_partner_ = i;
561 break;
562 }
563 }
564 }
565 current_game_phase_ = GamePhase::kTalonExchange;
566 AppendToAllInformationStates(absl::StrCat(action_id, ";"));
567}

569void TarokState::DoApplyActionInTalonExchange(Action action_id) {
570 auto& player_cards = players_cards_.at(current_player_);
571
572 if (talon_.size() == 6) {
573 // add all talon cards to info states
574 AppendToAllInformationStates(absl::StrCat(absl::StrJoin(talon_, ","), ";"));
575
576 // choosing one of the talon card sets
577 int set_begin = action_id * selected_contract_->num_talon_exchanges;
578 int set_end = set_begin + selected_contract_->num_talon_exchanges;
579
580 bool mond_in_talon = ActionInActions(kMondAction, talon_);
581 bool mond_in_selected_talon_set = false;
582 for (int i = set_begin; i < set_end; i++) {
583 player_cards.push_back(talon_.at(i));
584 if (talon_.at(i) == kMondAction) mond_in_selected_talon_set = true;
585 }
586 if (mond_in_talon && !mond_in_selected_talon_set) {
587 // the captured mond penalty applies if mond is in talon and not part of
588 // the selected set
589 captured_mond_player_ = current_player_;
590 }
591
592 // add the selected talon set to info states
593 AppendToAllInformationStates(absl::StrCat(action_id, ";"));
594
595 std::sort(player_cards.begin(), player_cards.end());
596 talon_.erase(talon_.begin() + set_begin, talon_.begin() + set_end);
597 } else {
598 // discarding the cards
599 MoveActionFromTo(action_id, &player_cards,
600 &players_collected_cards_.at(current_player_));
601
602 bool talon_exchange_finished = player_cards.size() == 48 / num_players_;
603 std::string info_state_delimiter = talon_exchange_finished ? ";" : ",";
604
605 // note that all players see discarded tarok cards but only the discarder
606 // knows about discarded non-taroks
607 if (ActionToCard(action_id).suit == CardSuit::kTaroks) {
608 AppendToAllInformationStates(
609 absl::StrCat(action_id, info_state_delimiter));
610 } else {
611 AppendToInformationState(current_player_,
612 absl::StrCat(action_id, info_state_delimiter));
613 for (Player p = 0; p < num_players_; p++) {
614 if (p == current_player_) continue;
615 AppendToInformationState(p, absl::StrCat("d", info_state_delimiter));
616 }
617 }
618
619 if (talon_exchange_finished) StartTricksPlayingPhase();
620 }
621}

624 current_game_phase_ = GamePhase::kTricksPlaying;

626 current_player_ = declarer_;

631void TarokState::DoApplyActionInTricksPlaying(Action action_id) {
632 MoveActionFromTo(action_id, &players_cards_.at(current_player_),
633 &trick_cards_);
634 AppendToAllInformationStates(std::to_string(action_id));
635 if (trick_cards_.size() == num_players_) {
636 ResolveTrick();
637 if (players_cards_.at(current_player_).empty() ||
638 ((selected_contract_->name == ContractName::kBeggar ||
639 selected_contract_->name == ContractName::kOpenBeggar) &&
640 current_player_ == declarer_) ||
641 ((selected_contract_->name == ContractName::kColourValatWithout ||
642 selected_contract_->name == ContractName::kValatWithout) &&
643 current_player_ != declarer_)) {
644 current_game_phase_ = GamePhase::kFinished;
645 } else {
646 AppendToAllInformationStates(";");
647 }
648 } else {
649 NextPlayer();
650 AppendToAllInformationStates(",");
651 }
652}

654void TarokState::ResolveTrick() {
655 auto [trick_winner, winning_action] = ResolveTrickWinnerAndWinningAction();
656 auto& trick_winner_collected_cards =
657 players_collected_cards_.at(trick_winner);
658
659 for (auto const& action : trick_cards_) {
660 trick_winner_collected_cards.push_back(action);
661 }
662
663 if (selected_contract_->name == ContractName::kKlop && !talon_.empty()) {
664 // add the "gift" talon card in klop
665 trick_winner_collected_cards.push_back(talon_.front());
666 AppendToAllInformationStates(absl::StrCat(",", talon_.front()));
667 talon_.erase(talon_.begin());
668 } else if (winning_action == called_king_ && called_king_in_talon_) {
669 // declearer won the trick with the called king that was in talon so all
670 // of the talon cards belong to the declearer (note that this is only
671 // possible when talon exchange actually happened in the past)
672 bool mond_in_talon = false;
673 for (auto const& action : talon_) {
674 trick_winner_collected_cards.push_back(action);
675 if (action == kMondAction) mond_in_talon = true;
676 }
677 if (mond_in_talon) {
678 // the called king and mond were in different parts of the talon and
679 // declearer selected the set with the king plus won the mond as
680 // part of the obtained talon remainder, negating the captured mond
681 // penalty obtained during DoApplyActionInTalonExchange()
682 captured_mond_player_ = kInvalidPlayer;
683 }
684 talon_.clear();
685 } else if ((selected_contract_->NeedsTalonExchange() ||
686 selected_contract_->name == ContractName::kSoloWithout) &&
687 (winning_action == kSkisAction ||
688 winning_action == kPagatAction)) {
689 // check if mond is captured by skis or pagat (emperor's trick) and
690 // penalise the player of the mond in certain contracts
691 for (int i = 0; i < trick_cards_.size(); i++) {
692 if (trick_cards_.at(i) == kMondAction) {
693 captured_mond_player_ = TrickCardsIndexToPlayer(i);
694 }
695 }
696 }
697
698 trick_cards_.clear();
699 current_player_ = trick_winner;
700}
701
702TrickWinnerAndAction TarokState::ResolveTrickWinnerAndWinningAction() const {
703 // compute the winning action index within trick_cards_
704 int winning_action_i;
705 if ((ActionInActions(kPagatAction, trick_cards_) &&
706 ActionInActions(kMondAction, trick_cards_) &&
707 ActionInActions(kSkisAction, trick_cards_)) &&
708 (selected_contract_->name != ContractName::kColourValatWithout ||
709 ActionToCard(trick_cards_.front()).suit == CardSuit::kTaroks)) {
710 // the emperor trick, i.e. pagat wins over mond and skis in all cases but
711 // not in Contract::kColourValatWithout when a non-trump is led
712 winning_action_i =
713 std::find(trick_cards_.begin(), trick_cards_.end(), kPagatAction) -
714 trick_cards_.begin();
715 } else {
716 winning_action_i = 0;
717 for (int i = 1; i < trick_cards_.size(); i++) {
718 const Card& winning_card =
719 ActionToCard(trick_cards_.at(winning_action_i));
720 const Card& current_card = ActionToCard(trick_cards_.at(i));
721
722 if (((current_card.suit == CardSuit::kTaroks &&
723 selected_contract_->name != ContractName::kColourValatWithout) ||
724 current_card.suit == winning_card.suit) &&
725 current_card.rank > winning_card.rank) {
726 winning_action_i = i;
727 }
728 }
729 }
730 return {TrickCardsIndexToPlayer(winning_action_i),
731 trick_cards_.at(winning_action_i)};
732}

734Player TarokState::TrickCardsIndexToPlayer(int index) const {
735 Player player = current_player_;
736 for (int i = 0; i < trick_cards_.size() - 1 - index; i++) {
737 player -= 1;
738 if (player == -1) player = num_players_ - 1;
739 }
740 return player;
741}

774std::vector<int> TarokState::ScoresInKlop() const {
775 std::vector<int> scores;
776 scores.reserve(num_players_);
777
778 bool any_player_won_or_lost = false;
779 for (int i = 0; i < num_players_; i++) {
780 int points = CardPoints(players_collected_cards_.at(i),
781 tarok_parent_game_->card_deck_);
782 if (points > 35) {
783 any_player_won_or_lost = true;
784 scores.push_back(-70);
785 } else if (points == 0) {
786 any_player_won_or_lost = true;
787 scores.push_back(70);
788 } else {
789 scores.push_back(-points);
790 }
791 }
792 if (any_player_won_or_lost) {
793 // only the winners and losers score
794 for (int i = 0; i < num_players_; i++) {
795 if (std::abs(scores.at(i)) != 70) scores.at(i) = 0;
796 }
797 }
798 return scores;
799}

801std::vector<int> TarokState::ScoresInNormalContracts() const {
802 auto [collected_cards, opposite_collected_cards] =
803 SplitCollectedCardsPerTeams();
804
805 int score;
806 if (collected_cards.size() == 48) {
807 // valat won
808 score = 250;
809 } else if (opposite_collected_cards.size() == 48) {
810 // valat lost
811 score = -250;
812 } else {
813 int card_points =
814 CardPoints(collected_cards, tarok_parent_game_->card_deck_);
815 score = card_points - 35;
816
817 if (card_points > 35)
818 score += selected_contract_->score;
819 else
820 score -= selected_contract_->score;
821
822 // bonuses could be positive, negative or 0
823 int bonuses = NonValatBonuses(collected_cards, opposite_collected_cards);
824 score += bonuses;
825 }
826
827 std::vector<int> scores(num_players_, 0);
828 scores.at(declarer_) = score;
829 if (declarer_partner_ != kInvalidPlayer) scores.at(declarer_partner_) = score;
830 return scores;
831}

833CollectedCardsPerTeam TarokState::SplitCollectedCardsPerTeams() const {
834 std::vector<Action> collected_cards = players_collected_cards_.at(declarer_);
835 std::vector<Action> opposite_collected_cards;
836 for (Player p = 0; p < num_players_; p++) {
837 if (p != declarer_ && p != declarer_partner_) {
838 opposite_collected_cards.insert(opposite_collected_cards.end(),
839 players_collected_cards_.at(p).begin(),
840 players_collected_cards_.at(p).end());
841 } else if (p == declarer_partner_) {
842 collected_cards.insert(collected_cards.end(),
843 players_collected_cards_.at(p).begin(),
844 players_collected_cards_.at(p).end());
845 }
846 }
847 return {collected_cards, opposite_collected_cards};
848}
849
850int TarokState::NonValatBonuses(
851 const std::vector<Action>& collected_cards,
852 const std::vector<Action>& opposite_collected_cards) const {
853 int bonuses = 0;
854
855 // last trick winner is the current player
856 auto const& last_trick_winner_cards =
857 players_collected_cards_.at(current_player_);
858 // king ultimo and pagat ultimo
859 int ultimo_bonus = 0;
860 if (std::find(last_trick_winner_cards.end() - num_players_,
861 last_trick_winner_cards.end(),
862 called_king_) != last_trick_winner_cards.end()) {
863 // king ultimo
864 ultimo_bonus = 10;
865 } else if (std::find(last_trick_winner_cards.end() - num_players_,
866 last_trick_winner_cards.end(),
867 0) != last_trick_winner_cards.end()) {
868 // pagat ultimo
869 ultimo_bonus = 25;
870 }
871
872 if (ultimo_bonus > 0 &&
873 (current_player_ == declarer_ || current_player_ == declarer_partner_)) {
874 bonuses = ultimo_bonus;
875 } else if (ultimo_bonus > 0) {
876 bonuses = -ultimo_bonus;
877 }
878
879 // collected kings or trula
880 auto [collected_kings, collected_trula] =
881 CollectedKingsAndOrTrula(collected_cards);
882 auto [opposite_collected_kings, opposite_collected_trula] =
883 CollectedKingsAndOrTrula(opposite_collected_cards);
884
885 if (collected_kings)
886 bonuses += 10;
887 else if (opposite_collected_kings)
888 bonuses -= 10;
889 if (collected_trula)
890 bonuses += 10;
891 else if (opposite_collected_trula)
892 bonuses -= 10;
893 return bonuses;
894}
895
896std::tuple<bool, bool> TarokState::CollectedKingsAndOrTrula(
897 const std::vector<Action>& collected_cards) const {
898 int num_kings = 0, num_trula = 0;
899 for (auto const& action : collected_cards) {
900 if (action == kKingOfHeartsAction || kKingOfDiamondsAction == 37 ||
901 action == kKingOfSpadesAction || kKingOfClubsAction == 53) {
902 num_kings += 1;
903 } else if (action == kPagatAction || action == kMondAction ||
904 action == kSkisAction) {
905 num_trula += 1;
906 }
907 }
908 return {num_kings == 4, num_trula == 3};
909}

911std::vector<int> TarokState::ScoresInHigherContracts() const {
912 bool declarer_won;
913 if (selected_contract_->name == ContractName::kBeggar ||
914 selected_contract_->name == ContractName::kOpenBeggar) {
915 declarer_won = players_collected_cards_.at(declarer_).empty();
916 } else if (selected_contract_->name == ContractName::kColourValatWithout ||
917 selected_contract_->name == ContractName::kValatWithout) {
918 declarer_won = players_collected_cards_.at(declarer_).size() == 48;
919 } else {
920 // solo without
921 declarer_won = CardPoints(players_collected_cards_.at(declarer_),
922 tarok_parent_game_->card_deck_) > 35;
923 }
924
925 std::vector<int> scores(num_players_, 0);
926 if (declarer_won)
927 scores.at(declarer_) = selected_contract_->score;
928 else
929 scores.at(declarer_) = -selected_contract_->score;
930 return scores;

966 "\n");

968 return str;
