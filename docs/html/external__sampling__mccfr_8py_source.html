31  def __init__(self, game, average_type=AverageType.SIMPLE):
32  super().__init__(game)
33  # How to average the strategy. The 'simple' type does the averaging for
34  # player i + 1 mod num_players on player i's regret update pass; in two
35  # players this corresponds to the standard implementation (updating the
36  # average policy at opponent nodes). In n>2 players, this can be a problem
37  # for several reasons: first, it does not compute the estimate as described
38  # by the (unbiased) stochastically-weighted averaging in chapter 4 of
39  # Lanctot 2013 commonly used in MCCFR because the denominator (important
40  # sampling correction) should include all the other sampled players as well
41  # so the sample reach no longer cancels with reach of the player updating
42  # their average policy. Second, if one player assigns zero probability to an
43  # action (leading to a subtree), the average policy of a different player in
44  # that subtree is no longer updated. Hence, the full averaging does not
45  # update the average policy in the regret passes but does a separate pass to
46  # update the average policy. Nevertheless, we set the simple type as the
47  # default because it is faster, seems to work better empirically, and it
48  # matches what was done in Pluribus (Brown and Sandholm. Superhuman AI for
49  # multiplayer poker. Science, 11, 2019).
50  self._average_type_average_type = average_type

110  def _update_regrets(self, state, player):
111  """Runs an episode of external sampling.
112 
113  Args:
114  state: the open spiel state to run from
115  player: the player to update regrets for
116 
117  Returns:
118  value: is the value of the state in the game
119  obtained as the weighted average of the values
120  of the children
121  """
122  if state.is_terminal():
123  return state.player_return(player)
124 
125  if state.is_chance_node():
126  outcomes, probs = zip(*state.chance_outcomes())
127  outcome = np.random.choice(outcomes, p=probs)
128  return self._update_regrets_update_regrets(state.child(outcome), player)
129 
130  cur_player = state.current_player()
131  info_state_key = state.information_state_string(cur_player)
132  legal_actions = state.legal_actions()
133  num_legal_actions = len(legal_actions)
134 
135  infostate_info = self._lookup_infostate_info_lookup_infostate_info(info_state_key,
136  num_legal_actions)
137  policy = self._regret_matching_regret_matching(infostate_info[mccfr.REGRET_INDEX],
138  num_legal_actions)
139 
140  value = 0
141  child_values = np.zeros(num_legal_actions, dtype=np.float64)
142  if cur_player != player:
143  # Sample at opponent node
144  action_idx = np.random.choice(np.arange(num_legal_actions), p=policy)
145  value = self._update_regrets_update_regrets(
146  state.child(legal_actions[action_idx]), player)
147  else:
148  # Walk over all actions at my node
149  for action_idx in range(num_legal_actions):
150  child_values[action_idx] = self._update_regrets_update_regrets(
151  state.child(legal_actions[action_idx]), player)
152  value += policy[action_idx] * child_values[action_idx]
153 
154  if cur_player == player:
155  # Update regrets.
156  for action_idx in range(num_legal_actions):
157  self._add_regret_add_regret(info_state_key, action_idx,
158  child_values[action_idx] - value)
159  # Simple average does averaging on the opponent node. To do this in a game
160  # with more than two players, we only update the player + 1 mod num_players,
161  # which reduces to the standard rule in 2 players.
162  if self._average_type_average_type == AverageType.SIMPLE and cur_player == (
163  player + 1) % self._num_players_num_players:
164  for action_idx in range(num_legal_actions):
165  self._add_avstrat_add_avstrat(info_state_key, action_idx, policy[action_idx])
166
