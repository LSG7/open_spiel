36 @parameterized.named_parameters(("python", "python_mfg_crowd_modelling"),
37 ("cpp", "mfg_crowd_modelling"))
38 def test_run(self, name: str):
39 """Checks if fictitious play works."""
40 game = pyspiel.load_game(name)
41 fp = fictitious_play.FictitiousPlay(game)
42 for _ in range(10):
43 fp.iteration()
44 fp_policy = fp.get_policy()
45 nash_conv_fp = nash_conv.NashConv(game, fp_policy)

73 @parameterized.named_parameters(("python", "python_mfg_crowd_modelling"),
74 ("cpp", "mfg_crowd_modelling"))
75 def test_dqn(self, name):
76 """Checks if fictitious play with DQN-based value function works."""
77 game = pyspiel.load_game(name)
78 dfp = fictitious_play.FictitiousPlay(game)
79
80 uniform_policy = policy.UniformRandomPolicy(game)
81 dist = distribution.DistributionPolicy(game, uniform_policy)
82 envs = [
83 rl_environment.Environment(
84 game, mfg_distribution=dist, mfg_population=p)
85 for p in range(game.num_players())
86 ]
87 dqn_agent = dqn.DQN(
88 0,
89 state_representation_size=envs[0].observation_spec()["info_state"][0],
90 num_actions=envs[0].action_spec()["num_actions"],
91 hidden_layers_sizes=[256, 128, 64],
92 replay_buffer_capacity=100,
93 batch_size=5,
94 epsilon_start=0.02,
95 epsilon_end=0.01)
96
97 br_policy = rl_agent_policy.RLAgentPolicy(
98 game, dqn_agent, 0, use_observation=True)
99 for _ in range(10):
100 dfp.iteration(br_policy=br_policy)
101
102 dfp_policy = dfp.get_policy()
103 nash_conv_dfp = nash_conv.NashConv(game, dfp_policy)
104
105 self.assertAlmostEqual(nash_conv_dfp.nash_conv(), 1.056, places=3)

107 def test_average(self):
108 """Test the average of policies.
109
110 Here we test that the average of values is the value of the average policy.
111 """
112 game = crowd_modelling.MFGCrowdModellingGame()
113 uniform_policy = policy.UniformRandomPolicy(game)
114 mfg_dist = distribution.DistributionPolicy(game, uniform_policy)
115 br_value = best_response_value.BestResponse(
116 game, mfg_dist, value.TabularValueFunction(game))
117 py_value = policy_value.PolicyValue(game, mfg_dist, uniform_policy,
118 value.TabularValueFunction(game))
119 greedy_pi = greedy_policy.GreedyPolicy(game, None, br_value)
120 greedy_pi = greedy_pi.to_tabular()
121 merged_pi = fictitious_play.MergedPolicy(
122 game, list(range(game.num_players())), [uniform_policy, greedy_pi],
123 [mfg_dist, distribution.DistributionPolicy(game, greedy_pi)],
124 [0.5, 0.5])
125 merged_pi_value = policy_value.PolicyValue(game, mfg_dist, merged_pi,
126 value.TabularValueFunction(game))
127
128 self.assertAlmostEqual(
129 merged_pi_value(game.new_initial_state()),
130 (br_value(game.new_initial_state()) +
131 py_value(game.new_initial_state())) / 2)
