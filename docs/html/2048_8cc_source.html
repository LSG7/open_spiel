126 std::array<Coordinate, 2> TwentyFortyEightState::FindFarthestPosition(
127  int r, int c, int direction) const {
128  // Progress towards the vector direction until an obstacle is found
129  Coordinate prev = Coordinate(r, c);
130  Coordinate direction_diff = GetVector(direction);
131  do {
132  prev = Coordinate(r, c);
133  r += direction_diff.row;
134  c += direction_diff.column;
135  } while (InBounds(r, c) && CellAvailable(r, c));
136  return std::array<Coordinate, 2>{prev, Coordinate(r, c)};
137 }

139 bool TwentyFortyEightState::TileMatchAvailable(int r, int c) const {
140  int tile = BoardAt(r, c).value;
141  if (tile > 0) {
142  for (int direction : kPlayerActions) {
143  Coordinate vector = GetVector(direction);
144  int other = GetCellContent(r + vector.row, c + vector.column);
145  if (other > 0 && other == tile) {
146  return true; // These two tiles can be merged
147  }
148  }
149  }
150  return false;
151 }

178 void TwentyFortyEightState::DoApplyAction(Action action) {
179  if (IsChanceNode()) {
180  // The original 2048 game starts with two random tiles. To achieve this,
181  // an extra move is given to the chance player during the beginning of the
182  // game.
183  if (!extra_chance_turn_) {
184  current_player_ = 0;
185  }
186  extra_chance_turn_ = false;
187 
188  if (action == kNoCellAvailableAction) {
189  return;
190  }
191  ChanceAction chance_action = SpielActionToChanceAction(action);
192  SetBoard(
193  chance_action.row, chance_action.column,
194  Tile(chance_action.is_four ? kChanceTiles[1] : kChanceTiles[0], false));
195  return;
196  }
197  action_score_ = 0;
198  const std::array<std::array<int, 4>, 2>& traversals = kTraversals[action];
199  PrepareTiles();
200  for (int r : traversals[0]) {
201  for (int c : traversals[1]) {
202  int tile = GetCellContent(r, c);
203  if (tile > 0) {
204  bool moved = false;
205  std::array<Coordinate, 2> positions =
206  FindFarthestPosition(r, c, action);
207  Coordinate farthest_pos = positions[0];
208  Coordinate next_pos = positions[1];
209  int next_cell = GetCellContent(next_pos.row, next_pos.column);
210  if (next_cell > 0 && next_cell == tile &&
211  !BoardAt(next_pos).is_merged) {
212  int merged = tile * 2;
213  action_score_ += merged;
214  SetBoard(next_pos.row, next_pos.column, Tile(merged, true));
215  moved = true;
216  } else if (farthest_pos.row != r || farthest_pos.column != c) {
217  SetBoard(farthest_pos.row, farthest_pos.column, Tile(tile, false));
218  moved = true;
219  }
220  if (moved) {
221  SetBoard(r, c, Tile(0, false));
222  current_player_ = kChancePlayerId;
223  }
224  }
225  }
226  }
227  total_score_ += action_score_;
228  total_actions_++;
229 }

231 std::string TwentyFortyEightState::ActionToString(Player player,
232  Action action_id) const {
233  if (player == kChancePlayerId) {
234  if (action_id == kNoCellAvailableAction) {
235  return "No Cell Available";
236  }
237  ChanceAction chance_action = SpielActionToChanceAction(action_id);
238  return absl::StrCat(chance_action.is_four ? 4 : 2, " added to row ",
239  chance_action.row + 1, ", column ",
240  chance_action.column + 1);
241  }
242  switch (action_id) {
243  case kMoveUp:
244  return "Up";
245  case kMoveRight:
246  return "Right";
247  case kMoveDown:
248  return "Down";
249  case kMoveLeft:
250  return "Left";
251  default:
252  return "Invalid action";
253  }

268 ActionsAndProbs TwentyFortyEightState::ChanceOutcomes() const {
269  int count = AvailableCellCount();
270  if (count == 0) {
271  return {{kNoCellAvailableAction, 1.0}};
272  }
273  ActionsAndProbs action_and_probs;
274  action_and_probs.reserve(count * 2);
275  for (int r = 0; r < kRows; r++) {
276  for (int c = 0; c < kColumns; c++) {
277  if (BoardAt(r, c).value == 0) {
278  // 2 appearing randomly on the board should be 9 times as likely as a 4.
279  action_and_probs.emplace_back(
280  ChanceActionToSpielAction(ChanceAction(r, c, false)), .9 / count);
281  action_and_probs.emplace_back(
282  ChanceActionToSpielAction(ChanceAction(r, c, true)), .1 / count);
283  }
284  }
285  }
286  return action_and_probs;
287 }

314 bool TwentyFortyEightState::IsTerminal() const {
315  if (move_number_ >= parent_game_.MaxGameLength()) {
316  return true;
317  }
318 
319  // Scan the board.
320  int count = 0;
321  int tile_matches_available = 0;
322  for (int r = 0; r < kRows; r++) {
323  for (int c = 0; c < kColumns; c++) {
324  // Check for 2048, if necessary,
325  if (BoardAt(r, c).value == parent_game_.max_tile()) {
326  return true;
327  }
328 
329  // Check for increase of available cell count.
330  if (BoardAt(r, c).value == 0) {
331  count++;
332  }
333 
334  // Check for tile matches.
335  if (TileMatchAvailable(r, c)) {
336  tile_matches_available++;
337  }
338  }
339  }
340 
341  if (count == 0 && tile_matches_available == 0) {
342  return true;
343  } else {
344  return false;
345  }
346 }
