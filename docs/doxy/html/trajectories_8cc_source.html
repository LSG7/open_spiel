31std::string StateKey(const Game& game, const State& state,
32 Player player = kInvalidPlayer) {
33 if (game.GetType().provides_information_state_string) {
34 if (player == kInvalidPlayer) return state.InformationStateString();
35 return state.InformationStateString(player);
36 } else if (game.GetType().provides_observation_string) {
37 if (player == kInvalidPlayer) return state.ObservationString();
38 return state.ObservationString(player);
39 }
40 return state.ToString();
41}

57// Moves a trajectory of size [1, T] into the current trajectory at index.
58void BatchedTrajectory::MoveTrajectory(int index,
59 BatchedTrajectory* trajectory) {
60 // The passed trajectory must have a batch size of 1.
61 SPIEL_CHECK_EQ(trajectory->batch_size, 1);
62 max_trajectory_length =
63 std::max(max_trajectory_length, trajectory->max_trajectory_length);
64 observations[index] = std::move(trajectory->observations[0]);
65 state_indices[index] = std::move(trajectory->state_indices[0]);
66 legal_actions[index] = std::move(trajectory->legal_actions[0]);
67 actions[index] = std::move(trajectory->actions[0]);
68 player_policies[index] = std::move(trajectory->player_policies[0]);
69 player_ids[index] = std::move(trajectory->player_ids[0]);
70 rewards[index] = trajectory->rewards[0];
71 next_is_terminal[index] = std::move(trajectory->next_is_terminal[0]);
72 valid[index] = std::move(trajectory->valid[0]);
73}

75// Pads fields to make sure that they're all the same shape, i.e. [B, T, N],
76// where N = the size of each field.
77void BatchedTrajectory::ResizeFields(int length) {
78 if (length > 0) {
79 SPIEL_CHECK_GE(length, max_trajectory_length);
80 // We adjust max_trajectory_length as it's no longer correct.
81 max_trajectory_length = length;
82 }
83 // Only works for batches with at least one trajectory as otherwise we can't
84 // infer the field size.
85 SPIEL_CHECK_GT(batch_size, 0);
86 // TODO(author1): Replace this with a multi-threaded version.
87 for (int i = 0; i < batch_size; ++i) {
88 // Each field has shape [B, T, field_size], where N is a parameter that is
89 // fixed for each (game, field) pair. We thus have to get the size of N from
90 // the existing vectors.
91 if (!observations[0].empty()) {
92 observations[i].resize(max_trajectory_length,
93 std::vector<float>(observations[0][0].size(), 0));
94 }
95 state_indices[i].resize(max_trajectory_length, 0);
96 legal_actions[i].resize(max_trajectory_length,
97 std::vector<int>(legal_actions[0][0].size(), 1));
98
99 // Actions has shape [B, T, 1]
100 actions[i].resize(max_trajectory_length, 0);
101
102 // legal_actions has shape [B, T, num_distinct_actions], while
103 // player_policies[0][0].size() <= num_distinct_actions.
104 player_policies[i].resize(
105 max_trajectory_length,
106 std::vector<double>(legal_actions[0][0].size(), 1));
107 player_ids[i].resize(max_trajectory_length, 0);
108 next_is_terminal[i].resize(max_trajectory_length, false);
109 valid[i].resize(max_trajectory_length, false);
110 }
111}

113BatchedTrajectory RecordBatchedTrajectory(
114 const Game& game, const std::vector<TabularPolicy>& policies,
115 const State& initial_state,
116 const std::unordered_map<std::string, int>& state_to_index, int batch_size,
117 bool include_full_observations, std::mt19937* rng_ptr,
118 int max_unroll_length) {
119 SPIEL_CHECK_GT(batch_size, 0);
120 if (state_to_index.empty()) SPIEL_CHECK_TRUE(include_full_observations);
121 BatchedTrajectory batched_trajectory(batch_size);
122 // TODO(author1): Replace this with a multi-threaded version.
123 for (int i = 0; i < batch_size; ++i) {
124 BatchedTrajectory trajectory =
125 RecordTrajectory(game, policies, initial_state, state_to_index,
126 include_full_observations, rng_ptr);
127 SPIEL_CHECK_FALSE(trajectory.rewards[0].empty());
128 batched_trajectory.MoveTrajectory(i, &trajectory);
129 }
130 batched_trajectory.ResizeFields(max_unroll_length);
131 return batched_trajectory;
132}

134BatchedTrajectory RecordTrajectory(
135 const Game& game, const std::vector<TabularPolicy>& policies,
136 const State& initial_state,
137 const std::unordered_map<std::string, int>& state_to_index,
138 bool include_full_observations, std::mt19937* rng) {
139 if (state_to_index.empty()) SPIEL_CHECK_TRUE(include_full_observations);
140 BatchedTrajectory trajectory(/*batch_size=*/1);
141 std::unique_ptr<open_spiel::State> state = initial_state.Clone();
142 bool find_index = !state_to_index.empty();
143 while (!state->IsTerminal()) {
144 Action action = kInvalidAction;
145 if (state->IsChanceNode()) {
146 action = open_spiel::SampleAction(
147 state->ChanceOutcomes(),
148 std::uniform_real_distribution<double>(0.0, 1.0)(*rng))
149 .first;
150 } else if (state->IsSimultaneousNode()) {
151 open_spiel::SpielFatalError(
152 "We do not support games with simultaneous actions.");
153 } else {
154 // Then we're at a decision node.
155 trajectory.legal_actions[0].push_back(state->LegalActionsMask());
156 if (find_index) {
157 auto it = state_to_index.find(StateKey(game, *state));
158 SPIEL_CHECK_TRUE(it != state_to_index.end());
159 trajectory.state_indices[0].push_back(it->second);
160 } else {
161 trajectory.observations[0].push_back(state->InformationStateTensor());
162 }
163 ActionsAndProbs policy =
164 policies.at(state->CurrentPlayer())
165 .GetStatePolicy(state->InformationStateString());
166 if (policy.size() > state->LegalActions().size()) {
167 std::string policy_str = "";
168 for (const auto& item : policy) {
169 absl::StrAppend(&policy_str, "(", item.first, ",", item.second, ") ");
170 }
171 SpielFatalError(absl::StrCat(
172 "There are more actions than legal actions from ",
173 typeid(policies.at(state->CurrentPlayer())).name(),
174 "\n Legal actions are: ", absl::StrJoin(state->LegalActions(), " "),
175 " \n Available probabilities were:", policy_str));
176 }
177 std::vector<double> probs(game.NumDistinctActions(), 0.);
178 for (const std::pair<Action, double>& pair : policy) {
179 probs[pair.first] = pair.second;
180 }
181 trajectory.player_policies[0].push_back(probs);
182 trajectory.player_ids[0].push_back(state->CurrentPlayer());
183 action = SampleAction(policy, *rng).first;
184 trajectory.actions[0].push_back(action);
185 }
186 SPIEL_CHECK_NE(action, kInvalidAction);
187 state->ApplyAction(action);
188 }
189 trajectory.valid[0] = std::vector<int>(trajectory.actions[0].size(), true);
190 trajectory.rewards[0] = state->Returns();
191 trajectory.next_is_terminal[0].resize(trajectory.actions[0].size(), false);
192 trajectory.next_is_terminal[0][trajectory.next_is_terminal[0].size() - 1] =
193 true;
194
195 // We arbitrarily set max_trajectory_length based on the actions field. All
196 // the fields should have the same length.
197 trajectory.max_trajectory_length = trajectory.actions[0].size();
198 return trajectory;
199}
