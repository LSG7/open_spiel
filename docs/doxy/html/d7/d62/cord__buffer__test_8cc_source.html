137// Returns the maximum capacity for a given block_size and requested size.
138size_t MaxCapacityFor(size_t block_size, size_t requested) {
139 requested = (std::min)(requested, cord_internal::kMaxLargeFlatSize);
140 // Maximum returned size is always capped at block_size - kFlatOverhead.
141 return block_size - kFlatOverhead;
142}

282TEST(CordLargeBufferTest, CreateWithSmallLimit) {

284 ASSERT_THAT(buffer.capacity(), Ge(512 - kFlatOverhead));
285 EXPECT_THAT(buffer.capacity(), Le(MaxCapacityFor(512, 1024)));

287 // Ask for precise block size, should return size - kOverhead
288 buffer = CordBuffer::CreateWithCustomLimit(512, 512);
289 ASSERT_THAT(buffer.capacity(), Ge(512 - kFlatOverhead));
290 EXPECT_THAT(buffer.capacity(), Le(MaxCapacityFor(512, 512)));

292 // Corner case: 511 < block_size, but 511 + kOverhead is above
293 buffer = CordBuffer::CreateWithCustomLimit(512, 511);
294 ASSERT_THAT(buffer.capacity(), Ge(512 - kFlatOverhead));
295 EXPECT_THAT(buffer.capacity(), Le(MaxCapacityFor(512, 511)));

297 // Corner case: 498 + kOverhead < block_size
298 buffer = CordBuffer::CreateWithCustomLimit(512, 498);
299 ASSERT_THAT(buffer.capacity(), Ge(512 - kFlatOverhead));
