48def parse_battles_dataset(filter_ties=False):
49 """Parse the data set from the raw CSV."""
50 dataset = []
51 model_names = {}
52 with gfile.Open(DATASET_FILE, "r") as f:
53 lines = f.readlines()
54 for line in lines:
55 if line.startswith("#"):
56 continue
57 # ,question_id,model_a,model_b,winner,judge,conversation_a,conversation_b,turn,anony,language,tstamp,openai_moderation,toxic_chat_tag
58 parts = line.split(",")
59 model_a, model_b, winner = (
60 parts[2].strip(),
61 parts[3].strip(),
62 parts[4].strip(),
63 )
64 if filter_ties and winner.startswith("tie"):
65 continue
66 else:
67 model_names[model_a] = True
68 model_names[model_b] = True
69 if winner == "model_a":
70 dataset.append((model_a, model_b, -1))
71 elif winner == "model_b":
72 dataset.append((model_a, model_b, 1))
73 else:
74 assert winner.startswith("tie")
75 dataset.append((model_a, model_b, 0))
76 return list(model_names.keys()), dataset

120def ranked_pairs_viz(model_names, dataset):
121 """Produce the ranked pairs visualization."""
122
123 alternatives = model_names[:]
124 profile = base.PreferenceProfile(alternatives=alternatives)
125 num_alternatives = len(alternatives)
126 alt_dict = profile.alternatives_dict
127 for datapoint in dataset:
128 alt_a, alt_b, outcome = datapoint
129 if outcome == 0:
130 pass
131 elif outcome == -1:
132 profile.add_vote([alt_a, alt_b])
133 elif outcome == 1:
134 profile.add_vote([alt_b, alt_a])
135 margin_matrix = profile.margin_matrix()
136 method = ranked_pairs.RankedPairsVoting()
137 outcome = method.run_election(profile)
138 graph_mat = outcome.graph
139 # Visualize only over the top 8:
140 keep_alternatives = [
141 "gpt-4",
142 "claude-v1",
143 "claude-instant-v1",
144 "guanaco-33b",
145 "gpt-3.5-turbo",
146 "wizardlm-13b",
147 "palm-2",
148 "vicuna-13b",
149 ]
150 keep_alternatives.sort()
151 for j in range(num_alternatives):
152 idx = num_alternatives - j - 1
153 alt = alternatives[idx]
154 if alt not in keep_alternatives:
155 graph_mat = np.delete(graph_mat, (idx), axis=0)
156 graph_mat = np.delete(graph_mat, (idx), axis=1)
157 orig_alternatives = model_names[:]
158 alternatives = keep_alternatives
159 m = len(alternatives)
160 graph = pgv.AGraph(directed=True, strict=True)
161 for alternative in alternatives:
162 graph.add_node(alternative)
163 for i in range(m):
164 for j in range(m):
165 if graph_mat[i, j] == 1:
166 graph.add_edge(alternatives[i], alternatives[j])
167 idx_i = alt_dict[alternatives[i]]
168 idx_j = alt_dict[alternatives[j]]
169 edge = graph.get_edge(
170 orig_alternatives[idx_i], orig_alternatives[idx_j]
171 )
172 edge.attr["label"] = margin_matrix[idx_i, idx_j]
173 graph.write("/tmp/chatbot_arena_rps.dot") # write to simple.dot
174 graph.draw(
175 "/tmp/chatbot_arena_rps.png",
176 # args='-Gdpi=100',
177 prog="dot",
178 ) # , args="-n2") # draw
179 print("Wrote to /tmp/chatbot_arena_rps.png")
