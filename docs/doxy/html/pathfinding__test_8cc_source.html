58void ChainMovementTests() {
59 const char* kGrid =
60 "ABCDEF....\n"
61 "..........\n"
62 "..a.......\n"
63 "..bcd.....\n"
64 "....e.....\n"
65 "....f.....\n";
66 std::shared_ptr<const Game> game = LoadGame(
67 "pathfinding",
68 {{"grid", GameParameter(kGrid)}, {"horizon", GameParameter(100)}});
69
70 std::unique_ptr<State> state = game->NewInitialState();
71 auto* pf_state = static_cast<PathfindingState*>(state.get());
72
73 // All of them should move in lock-step. No conflict.
74 state->ApplyActions({kRight, kUp, kLeft, kLeft, kUp, kUp});
75 SPIEL_CHECK_FALSE(state->IsChanceNode());
76
77 // 01234
78 // 0..........
79 // 1..........
80 // 2..10......
81 // 3..234.....
82 // 4....5.....
83 // 5..........
84 SPIEL_CHECK_EQ(pf_state->PlayerPos(0), std::make_pair(2, 3));
85 SPIEL_CHECK_EQ(pf_state->PlayerPos(1), std::make_pair(2, 2));
86 SPIEL_CHECK_EQ(pf_state->PlayerPos(2), std::make_pair(3, 2));
87 SPIEL_CHECK_EQ(pf_state->PlayerPos(3), std::make_pair(3, 3));
88 SPIEL_CHECK_EQ(pf_state->PlayerPos(4), std::make_pair(3, 4));
89 SPIEL_CHECK_EQ(pf_state->PlayerPos(5), std::make_pair(4, 4));
90}

92void BasicHeadOnCollisionTest() {
93 const char* kGrid =
94 "ABCD......\n"
95 "..........\n"
96 "..a.....d.\n"
97 "..........\n"
98 "..b.....c.\n"
99 "..........\n";
100 std::shared_ptr<const Game> game = LoadGame(
101 "pathfinding",
102 {{"grid", GameParameter(kGrid)}, {"horizon", GameParameter(100)}});
103
104 std::unique_ptr<State> state = game->NewInitialState();
105
106 // Collision between 0 and 1
107 state->ApplyActions({kDown, kUp, kRight, kUp});
108 SPIEL_CHECK_TRUE(state->IsChanceNode());
109
110 // Should be two possible outcomes
111 std::vector<Action> legal_actions = state->LegalActions();
112 SPIEL_CHECK_EQ(legal_actions.size(), 2);
113
114 // 1st possibility (child1): {0, 1}: a makes it, b stays.
115 // 2nd possibility (child2): {1, 0}: b makes it, a stays.
116 std::unique_ptr<State> child1 = state->Child(legal_actions[0]);
117 std::unique_ptr<State> child2 = state->Child(legal_actions[1]);
118 auto* pf_child1 = static_cast<PathfindingState*>(child1.get());
119 auto* pf_child2 = static_cast<PathfindingState*>(child2.get());
120
121 // 1st
122 SPIEL_CHECK_EQ(pf_child1->PlayerPos(0), std::make_pair(3, 2));
123 SPIEL_CHECK_EQ(pf_child1->PlayerPos(1), std::make_pair(4, 2));
124 // 2nd
125 SPIEL_CHECK_EQ(pf_child2->PlayerPos(0), std::make_pair(2, 2));
126 SPIEL_CHECK_EQ(pf_child2->PlayerPos(1), std::make_pair(3, 2));
127
128 // Start over.
129 state = game->NewInitialState();
130 state->ApplyActions({kDown, kUp, kUp, kDown});
131 SPIEL_CHECK_TRUE(state->IsChanceNode());
132
133 // Factorial outcomes since these situtations are not factorized.
134 legal_actions = state->LegalActions();
135 SPIEL_CHECK_EQ(legal_actions.size(), 24);
136}

138void HeadOnCollision3pTest() {
139 const char* kGrid =
140 "ABC.......\n"
141 "..........\n"
142 "..a.......\n"
143 ".c........\n"
144 "..b.......\n"
145 "..........\n";
146 std::shared_ptr<const Game> game = LoadGame(
147 "pathfinding",
148 {{"grid", GameParameter(kGrid)}, {"horizon", GameParameter(100)}});
149
150 std::unique_ptr<State> state = game->NewInitialState();
151
152 state->ApplyActions({kDown, kUp, kRight});
153 SPIEL_CHECK_TRUE(state->IsChanceNode());
154
155 // Should be 3! = 6 possible outcomes.
156 std::vector<Action> legal_actions = state->LegalActions();
157 SPIEL_CHECK_EQ(legal_actions.size(), 6);
158
159 // Go through all resolutions. Make sure the agent that gets to 3,2 is equally
160 // distributed, and that only one of them makes it (and the other two don't
161 // move).
162 std::vector<std::pair<int, int>> positions = {{2, 2}, {4, 2}, {3, 1}};
163 std::vector<Player> counts = {0, 0, 0};
164 for (int idx = 0; idx < 6; ++idx) {
165 std::unique_ptr<State> child = state->Child(legal_actions[idx]);
166 SPIEL_CHECK_FALSE(child->IsChanceNode());
167 auto* pf_child = static_cast<PathfindingState*>(child.get());
168 Player player = pf_child->PlayerAtPos({3, 2});
169 SPIEL_CHECK_NE(player, kInvalidPlayer);
170 counts[player]++;
171 for (Player p = 0; p < 3; ++p) {
172 if (p != player) {
173 SPIEL_CHECK_EQ(pf_child->PlayerPos(p), positions[p]);
174 }
175 }
176 }
177
178 SPIEL_CHECK_EQ(counts[0], 2);
179 SPIEL_CHECK_EQ(counts[1], 2);
180 SPIEL_CHECK_EQ(counts[2], 2);
181}
