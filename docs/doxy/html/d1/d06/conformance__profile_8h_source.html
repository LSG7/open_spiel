18//
19// This file contains templates for representing "Regularity Profiles" and
20// concisely-named versions of commonly used Regularity Profiles.
21//
22// A Regularity Profile is a compile-time description of the types of operations
23// that a given type supports, along with properties of those operations when
24// they do exist. For instance, a Regularity Profile may describe a type that
25// has a move-constructor that is noexcept and a copy constructor that is not
26// noexcept. This description can then be examined and passed around to other
27// templates for the purposes of asserting expectations on user-defined types
28// via a series trait checks, or for determining what kinds of run-time tests
29// are able to be performed.
30//
31// Regularity Profiles are also used when creating "archetypes," which are
32// minimum-conforming types that meet all of the requirements of a given
33// Regularity Profile. For more information regarding archetypes, see
34// "conformance_archetypes.h".
35

70// ConformanceErrors

72class ConformanceErrors {
73 public:
74 // Setup the error reporting mechanism by seeding it with the name of the type
75 // that is being tested.
76 explicit ConformanceErrors(std::string type_name)
77 : assertion_result_(false), type_name_(std::move(type_name)) {
78 assertion_result_ << "\n\n"
79 "Assuming the following type alias:\n"
80 "\n"
81 " using _T = "
82 << type_name_ << ";\n\n";
83 outputDivider();
84 }
85
86 // Adds the test name to the list of successfully run tests iff it was not
87 // previously reported as failing. This behavior is useful for tests that
88 // have multiple parts, where failures and successes are reported individually
89 // with the same test name.
90 void addTestSuccess(absl::string_view test_name) {
91 auto normalized_test_name = absl::AsciiStrToLower(test_name);
92
93 // If the test is already reported as failing, do not add it to the list of
94 // successes.
95 if (test_failures_.find(normalized_test_name) == test_failures_.end()) {
96 test_successes_.insert(std::move(normalized_test_name));
97 }
98 }
99
100 // Streams a single error description into the internal buffer (a visual
101 // divider is automatically inserted after the error so that multiple errors
102 // are visibly distinct).
103 //
104 // This function increases the error count by 1.
105 //
106 // TODO(calabrese) Determine desired behavior when if this function throws.
107 template <class... P>
108 void addTestFailure(absl::string_view test_name, const P&... args) {
109 // Output a message related to the test failure.
110 assertion_result_ << "\n\n"
111 "Failed test: "
112 << test_name << "\n\n";
113 addTestFailureImpl(args...);
114 assertion_result_ << "\n\n";
115 outputDivider();
116
117 auto normalized_test_name = absl::AsciiStrToLower(test_name);
118
119 // If previous parts of this test succeeded, remove it from that set.
120 test_successes_.erase(normalized_test_name);
121
122 // Add the test name to the list of failed tests.
123 test_failures_.insert(std::move(normalized_test_name));
124
125 has_error_ = true;
126 }
127
128 // Convert this object into a testing::AssertionResult instance such that it
129 // can be used with gtest.
130 ::testing::AssertionResult assertionResult() const {
131 return has_error_ ? assertion_result_ : ::testing::AssertionSuccess();
132 }
133
134 // Convert this object into a testing::AssertionResult instance such that it
135 // can be used with gtest. This overload expects errors, using the specified
136 // matcher.
137 ::testing::AssertionResult expectFailedTests(
138 const std::set<std::string>& test_names) const {
139 // Since we are expecting nonconformance, output an error message when the
140 // type actually conformed to the specified profile.
141 if (!has_error_) {
142 return ::testing::AssertionFailure()
143 << "Unexpected conformance of type:\n"
144 " "
145 << type_name_ << "\n\n";
146 }
147
148 // Get a list of all expected failures that did not actually fail
149 // (or that were not run).
150 std::vector<std::string> nonfailing_tests;
151 absl::c_set_difference(test_names, test_failures_,
152 std::back_inserter(nonfailing_tests));
153
154 // Get a list of all "expected failures" that were never actually run.
155 std::vector<std::string> unrun_tests;
156 absl::c_set_difference(nonfailing_tests, test_successes_,
157 std::back_inserter(unrun_tests));
158
159 // Report when the user specified tests that were not run.
160 if (!unrun_tests.empty()) {
161 const bool tests_were_run =
162 !(test_failures_.empty() && test_successes_.empty());
163
164 // Prepare an assertion result used in the case that tests pass that were
165 // expected to fail.
166 ::testing::AssertionResult result = ::testing::AssertionFailure();
167 result << "When testing type:\n " << type_name_
168 << "\n\nThe following tests were expected to fail but were not "
169 "run";
170
171 if (tests_were_run) result << " (was the test name spelled correctly?)";
172
173 result << ":\n\n";
174
175 // List all of the tests that unexpectedly passed.
176 for (const auto& test_name : unrun_tests) {
177 result << " " << test_name << "\n";
178 }
179
180 if (!tests_were_run) result << "\nNo tests were run.";
181
182 if (!test_failures_.empty()) {
183 // List test failures
184 result << "\nThe tests that were run and failed are:\n\n";
185 for (const auto& test_name : test_failures_) {
186 result << " " << test_name << "\n";
187 }
188 }
189
190 if (!test_successes_.empty()) {
191 // List test successes
192 result << "\nThe tests that were run and succeeded are:\n\n";
193 for (const auto& test_name : test_successes_) {
194 result << " " << test_name << "\n";
195 }
196 }
197
198 return result;
199 }
200
201 // If some tests passed when they were expected to fail, alert the caller.
202 if (nonfailing_tests.empty()) return ::testing::AssertionSuccess();
203
204 // Prepare an assertion result used in the case that tests pass that were
205 // expected to fail.
206 ::testing::AssertionResult unexpected_successes =
207 ::testing::AssertionFailure();
208 unexpected_successes << "When testing type:\n " << type_name_
209 << "\n\nThe following tests passed when they were "
210 "expected to fail:\n\n";
211
212 // List all of the tests that unexpectedly passed.
213 for (const auto& test_name : nonfailing_tests) {
214 unexpected_successes << " " << test_name << "\n";
215 }
216
217 return unexpected_successes;
218 }
219
220 private:
221 void outputDivider() {
222 assertion_result_ << "========================================";
223 }
224
225 void addTestFailureImpl() {}
226
227 template <class H, class... T>
228 void addTestFailureImpl(const H& head, const T&... tail) {
229 assertion_result_ << head;
230 addTestFailureImpl(tail...);
231 }
232
233 ::testing::AssertionResult assertion_result_;
234 std::set<std::string> test_failures_;
235 std::set<std::string> test_successes_;
236 std::string type_name_;
237 bool has_error_ = false;
238};

263// An enum that describes the various expectations on an operations existence.
264enum class function_support { maybe, yes, nothrow, trivial };
265
266constexpr const char* PessimisticPropertyDescription(function_support v) {
267 return v == function_support::maybe
268 ? "no"
269 : v == function_support::yes
270 ? "yes, potentially throwing"
271 : v == function_support::nothrow ? "yes, nothrow"
272 : "yes, trivial";
273}

275// Return a string that describes the kind of property support that was
276// expected.
277inline std::string ExpectedFunctionKindList(function_support min,
278 function_support max) {
279 if (min == max) {
280 std::string result =
281 absl::StrCat("Expected:\n ",
282 PessimisticPropertyDescription(
283 static_cast<function_support>(UnderlyingValue(min))),
284 "\n");
285 return result;
286 }
287
288 std::string result = "Expected one of:\n";
289 for (auto curr_support = UnderlyingValue(min);
290 curr_support <= UnderlyingValue(max); ++curr_support) {
291 absl::StrAppend(&result, " ",
292 PessimisticPropertyDescription(
293 static_cast<function_support>(curr_support)),
294 "\n");
295 }
296
297 return result;
298}

300template <class Enum>
301void ExpectModelOfImpl(ConformanceErrors* errors, Enum min_support,
302 Enum max_support, Enum kind) {
303 const auto kind_value = UnderlyingValue(kind);
304 const auto min_support_value = UnderlyingValue(min_support);
305 const auto max_support_value = UnderlyingValue(max_support);
306
307 if (!(kind_value >= min_support_value && kind_value <= max_support_value)) {
308 errors->addTestFailure(
309 PropertyName(kind), "**Failed property expectation**\n\n",
310 ExpectedFunctionKindList(
311 static_cast<function_support>(min_support_value),
312 static_cast<function_support>(max_support_value)),
313 '\n', "Actual:\n ",
314 PessimisticPropertyDescription(
315 static_cast<function_support>(kind_value)));
316 } else {
317 errors->addTestSuccess(PropertyName(kind));
318 }
319}
