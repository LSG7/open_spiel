25// Transform for creating a repeated game from a normal-form game.
26// https://en.wikipedia.org/wiki/Repeated_game.
27//
28// Parameters:
29// "enable_infostate" bool Enable the sequence of round outcomes as the
30// information state tensor and string (default:
31// false).
32// "stage_game" game The game that will be repeated.
33// "num_repetitions" int Number of times that the game is repeated.
34// "recall" int Number of previous steps that defines the
35// observations when enable_infostate is false
36// (default: 1).

96 absl::optional<double> UtilitySum() const override {

99 ? per_stage_utility_sum
100 : per_stage_utility_sum.value() * num_repetitions_;
101 }

103 std::vector<int> ObservationTensorShape() const override;
104
105 const Game* StageGame() const { return stage_game_.get(); }
106
107 private:

109 const int num_repetitions_;
110 const int recall_;
111};
