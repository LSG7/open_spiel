154bool CheckMap(const absl::flat_hash_map<int, int>& map,
155 const std::vector<std::pair<int, int>>& keys_and_values) {
156 for (const auto& kvp : keys_and_values) {
157 auto found = map.find(kvp.first);
158 if (found == map.end()) {
159 return false;
160 }
161 if (found->second != kvp.second) {
162 return false;
163 }
164 }
165 return true;
166}

201// Span
202bool CheckSpan(absl::Span<const int> span, const std::vector<int>& values) {
203 if (span.size() != values.size()) return false;
204 for (size_t i = 0; i < span.size(); ++i) {
205 if (span[i] != values[i]) return false;
206 }
207 return true;
208}

248struct A {

250};
251struct B {
252 int b;
253};
254typedef absl::variant<A, B> AOrB;
255
256int VariantToInt(AOrB value) {
257 if (absl::holds_alternative<A>(value)) {
258 return absl::get<A>(value).a;
259 } else if (absl::holds_alternative<B>(value)) {
260 return absl::get<B>(value).b;
261 } else {
262 throw std::exception();
263 }
264}

266std::vector<AOrB> IdentityWithCopy(const std::vector<AOrB>& value) {
267 return value;
268}
