45void open_spiel::init_pyspiel_games_chess(py::module& m) {
46 py::module_ chess = m.def_submodule("chess");
47
48 py::enum_<Color>(chess, "Color")
49 .value("BLACK", Color::kBlack)
50 .value("WHITE", Color::kWhite)
51 .value("EMPTY", Color::kEmpty)
52 .export_values();
53
54 py::enum_<PieceType>(chess, "PieceType")
55 .value("EMPTY", PieceType::kEmpty)
56 .value("KING", PieceType::kKing)
57 .value("QUEEN", PieceType::kQueen)
58 .value("ROOK", PieceType::kRook)
59 .value("BISHOP", PieceType::kBishop)
60 .value("KNIGHT", PieceType::kKnight)
61 .value("PAWN", PieceType::kPawn)
62 .export_values();
63
64 py::class_<Piece>(chess, "Piece")
65 .def(py::init<>())
66 .def_readonly("color", &Piece::color)
67 .def_readonly("type", &Piece::type);
68
69 py::class_<Square>(chess, "Square")
70 .def(py::init<>())
71 .def_readonly("x", &Square::x)
72 .def_readonly("y", &Square::y);
73
74 py::class_<Move>(chess, "Move")
75 .def(py::init<>())
76 .def_readonly("from_square", &Move::from) // "from" is a python keyword
77 .def_readonly("to_square", &Move::to)
78 .def_readonly("piece", &Move::piece)
79 .def_readonly("promotion_type", &Move::promotion_type)
80 .def_readonly("is_castling", &Move::is_castling)
81 .def("to_string", &Move::ToString)
82 .def("to_san", &Move::ToSAN)
83 .def("to_lan", &Move::ToLAN);
84
85 py::classh<ChessBoard>(chess, "ChessBoard")
86 .def("has_legal_moves", &ChessBoard::HasLegalMoves)
87 .def("debug_string", &ChessBoard::DebugString)
88 .def("to_fen", &ChessBoard::ToFEN)
89 .def("to_unicode_string", &ChessBoard::ToUnicodeString);
90
91 py::classh<ChessState, State>(m, "ChessState")
92 .def("board", py::overload_cast<>(&ChessState::Board))
93 .def("debug_string", &ChessState::DebugString)
94 .def("parse_move_to_action", &ChessState::ParseMoveToAction)
95 .def("moves_history", py::overload_cast<>(&ChessState::MovesHistory))
96 // Pickle support
97 .def(py::pickle(
98 [](const ChessState& state) { // __getstate__
99 return SerializeGameAndState(*state.GetGame(), state);
100 },
101 [](const std::string& data) { // __setstate__
102 std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>
103 game_and_state = DeserializeGameAndState(data);
104 return dynamic_cast<ChessState*>(game_and_state.second.release());
105 }));
106
107 // action_to_move(action: int, board: ChessBoard)
108 chess.def("action_to_move", &chess::ActionToMove);
109
110 // move_to_action(move: Move, board_size: int = default_size)
111 chess.def("move_to_action", &chess::MoveToAction,
112 py::arg("move"), py::arg("board_size") = chess::kDefaultBoardSize);
113}
