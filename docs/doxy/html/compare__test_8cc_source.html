24// This is necessary to avoid a bunch of lint warnings suggesting that we use
25// EXPECT_EQ/etc., which doesn't work in this case because they convert the `0`
26// to an int, which can't be converted to the unspecified zero type.
27bool Identity(bool b) { return b; }

29TEST(Compare, WeakEquality) {
30 EXPECT_TRUE(Identity(weak_equality::equivalent == 0));
31 EXPECT_TRUE(Identity(0 == weak_equality::equivalent));
32 EXPECT_TRUE(Identity(weak_equality::nonequivalent != 0));
33 EXPECT_TRUE(Identity(0 != weak_equality::nonequivalent));
34 const weak_equality values[] = {weak_equality::equivalent,
35 weak_equality::nonequivalent};
36 for (const auto& lhs : values) {
37 for (const auto& rhs : values) {
38 const bool are_equal = &lhs == &rhs;

41 }
42 }

241struct WeakOrderingLess {
242 template <typename T>
243 absl::weak_ordering operator()(const T& a, const T& b) const {
244 return a < b ? absl::weak_ordering::less
245 : a == b ? absl::weak_ordering::equivalent
246 : absl::weak_ordering::greater;
247 }
248};
