62// Adds a plane to the information state vector corresponding to the presence
63// and absence of the given piece type and colour at each square.
64void AddPieceTypePlane(Color color, PieceType piece_type,
65 const ChessBoard& board,
66 absl::Span<float>::iterator& value_it) {
67 for (int8_t y = 0; y < kMaxBoardSize; ++y) {
68 for (int8_t x = 0; x < kMaxBoardSize; ++x) {
69 Piece piece_on_board = board.at(Square{x, y});
70 *value_it++ =
71 (piece_on_board.color == color && piece_on_board.type == piece_type
72 ? 1.0
73 : 0.0);
74 }
75 }
76}
77
78// Adds a uniform scalar plane scaled with min and max.
79template <typename T>
80void AddScalarPlane(T val, T min, T max,
81 absl::Span<float>::iterator& value_it) {
82 double normalized_val = static_cast<double>(val - min) / (max - min);
83 for (int i = 0; i < k2dMaxBoardSize; ++i) *value_it++ = normalized_val;
84}
85
86// Adds a binary scalar plane.
87void AddBinaryPlane(bool val, absl::Span<float>::iterator& value_it) {
88 AddScalarPlane<int>(val ? 1 : 0, 0, 1, value_it);
89}

157Action MoveToAction(const Move& move, int board_size) {
158 // Special-case for pass move.
159 if (move == kPassMove) return kPassAction;
160
161 Color color = move.piece.color;
162 // We rotate the move to be from player p's perspective.
163 Move player_move(move);
164
165 // Rotate move to be from player p's perspective.
166 player_move.from.y = ReflectRank(color, board_size, player_move.from.y);
167 player_move.to.y = ReflectRank(color, board_size, player_move.to.y);
168
169 // For each starting square, we enumerate 73 actions:
170 // - 9 possible underpromotions
171 // - 56 queen moves
172 // - 8 knight moves
173 // In total, this results in 64 * 73 = 4672 indices.
174 // This is the union of all possible moves, by reducing this to the number of
175 // moves actually available from each starting square this could still be
176 // reduced a little to 1816 indices.
177 int starting_index =
178 EncodeMove(player_move.from, 0, kMaxBoardSize, kNumActionDestinations);
179 int8_t x_diff = player_move.to.x - player_move.from.x;
180 int8_t y_diff = player_move.to.y - player_move.from.y;
181 Offset offset{x_diff, y_diff};
182 bool is_under_promotion = move.promotion_type != PieceType::kEmpty &&
183 move.promotion_type != PieceType::kQueen;
184 if (is_under_promotion) {
185 // We have to indicate underpromotions as special moves, because in terms of
186 // from/to they are identical to queen promotions.
187 // For a given starting square, an underpromotion can have 3 possible
188 // destination squares (straight, left diagonal, right diagonal) and 3
189 // possible piece types.
190 SPIEL_CHECK_EQ(move.piece.type, PieceType::kPawn);
191 SPIEL_CHECK_TRUE((move.piece.color == color &&
192 player_move.from.y == board_size - 2 &&
193 player_move.to.y == board_size - 1) ||
194 (move.piece.color == OppColor(color) &&
195 player_move.from.y == 1 && player_move.to.y == 0));
196
197 int promotion_index;
198 {
199 auto itr = absl::c_find(kUnderPromotionIndexToType, move.promotion_type);
200 SPIEL_CHECK_TRUE(itr != kUnderPromotionIndexToType.end());
201 promotion_index = std::distance(kUnderPromotionIndexToType.begin(), itr);
202 }
203
204 int direction_index;
205 {
206 auto itr = absl::c_find_if(
207 kUnderPromotionDirectionToOffset,
208 [offset](Offset o) { return o.x_offset == offset.x_offset; });
209 SPIEL_CHECK_TRUE(itr != kUnderPromotionDirectionToOffset.end());
210 direction_index =
211 std::distance(kUnderPromotionDirectionToOffset.begin(), itr);
212 }
213 return starting_index +
214 kUnderPromotionDirectionToOffset.size() * promotion_index +
215 direction_index;
216 } else {
217 // For the normal moves, we simply encode starting and destination square.
218 int destination_index =
219 OffsetToDestinationIndex(offset, kKnightOffsets, kMaxBoardSize);
220 SPIEL_CHECK_TRUE(destination_index >= 0 && destination_index < 64);
221 return starting_index + kNumUnderPromotions + destination_index;
222 }
223}

225std::pair<Square, int> ActionToDestination(int action, int board_size,
226 int num_actions_destinations) {
227 const int xy = action / num_actions_destinations;
228 SPIEL_CHECK_GE(xy, 0);
229 SPIEL_CHECK_LT(xy, board_size * board_size);
230 const int8_t x = xy / board_size;
231 const int8_t y = xy % board_size;
232 const int destination_index = action % num_actions_destinations;
233 SPIEL_CHECK_GE(destination_index, 0);
234 SPIEL_CHECK_LT(destination_index, num_actions_destinations);
235 return {Square{x, y}, destination_index};
236}

238Move ActionToMove(const Action& action, const ChessBoard& board) {
239 SPIEL_CHECK_GE(action, 0);
240 SPIEL_CHECK_LT(action, NumDistinctActions());
241
242 // Some chess variants (e.g. RBC) allow pass moves.
243 if (board.AllowPassMove() && action == kPassAction) {
244 return kPassMove;
245 }
246
247 // The encoded action represents an action encoded from color's perspective.
248 Color color = board.ToPlay();
249 int board_size = board.BoardSize();
250 PieceType promotion_type = PieceType::kEmpty;
251 bool is_castling = false;
252 auto [from_square, destination_index] =
253 ActionToDestination(action, kMaxBoardSize, kNumActionDestinations);
254 SPIEL_CHECK_LT(destination_index, kNumActionDestinations);
255
256 bool is_under_promotion = destination_index < kNumUnderPromotions;
257 Offset offset;
258 if (is_under_promotion) {
259 int promotion_index = destination_index / 3;
260 int direction_index = destination_index % 3;
261 promotion_type = kUnderPromotionIndexToType[promotion_index];
262 offset = kUnderPromotionDirectionToOffset[direction_index];
263 } else {
264 destination_index -= kNumUnderPromotions;
265 offset = DestinationIndexToOffset(destination_index, kKnightOffsets,
266 kMaxBoardSize);
267 }
268 Square to_square = from_square + offset;
269
270 from_square.y = ReflectRank(color, board_size, from_square.y);
271 to_square.y = ReflectRank(color, board_size, to_square.y);
272
273 // This uses the current state to infer the piece type.
274 Piece piece = {board.ToPlay(), board.at(from_square).type};
275
276 // Check for queen promotion.
277 if (!is_under_promotion && piece.type == PieceType::kPawn &&
278 ReflectRank(color, board_size, from_square.y) == board_size - 2 &&
279 ReflectRank(color, board_size, to_square.y) == board_size - 1) {
280 promotion_type = PieceType::kQueen;
281 }
282
283 // Check for castling which is defined here just as king moves horizontally
284 // by 2 spaces.
285 // TODO(b/149092677): Chess no longer supports chess960. Distinguish between
286 // left/right castle.
287 if (piece.type == PieceType::kKing && std::abs(offset.x_offset) == 2) {
288 is_castling = true;
289 }
290 Move move(from_square, to_square, piece, promotion_type, is_castling);
291 return move;
292}

322void ChessState::ObservationTensor(Player player,
323 absl::Span<float> values) const {
324 SPIEL_CHECK_GE(player, 0);
325 SPIEL_CHECK_LT(player, num_players_);
326
327 auto value_it = values.begin();
328
329 // Piece configuration.
330 for (const auto& piece_type : kPieceTypes) {
331 AddPieceTypePlane(Color::kWhite, piece_type, Board(), value_it);
332 AddPieceTypePlane(Color::kBlack, piece_type, Board(), value_it);
333 }
334
335 AddPieceTypePlane(Color::kEmpty, PieceType::kEmpty, Board(), value_it);
336
337 const auto entry = repetitions_.find(Board().HashValue());
338 SPIEL_CHECK_FALSE(entry == repetitions_.end());
339 int repetitions = entry->second;
340
341 // Num repetitions for the current board.
342 AddScalarPlane(repetitions, 1, 3, value_it);
343
344 // Side to play.
345 AddScalarPlane(ColorToPlayer(Board().ToPlay()), 0, 1, value_it);
346
347 // Irreversible move counter.
348 AddScalarPlane(Board().IrreversibleMoveCounter(), 0, 101, value_it);
349
350 // Castling rights.
351 AddBinaryPlane(Board().CastlingRight(Color::kWhite, CastlingDirection::kLeft),
352 value_it);
353
354 AddBinaryPlane(
355 Board().CastlingRight(Color::kWhite, CastlingDirection::kRight),
356 value_it);
357
358 AddBinaryPlane(Board().CastlingRight(Color::kBlack, CastlingDirection::kLeft),
359 value_it);
360
361 AddBinaryPlane(
362 Board().CastlingRight(Color::kBlack, CastlingDirection::kRight),
363 value_it);
364
365 SPIEL_CHECK_EQ(value_it, values.end());
366}

391absl::optional<std::vector<double>> ChessState::MaybeFinalReturns() const {
392 if (!Board().HasSufficientMaterial()) {
393 return std::vector<double>{DrawUtility(), DrawUtility()};
394 }
395
396 if (IsRepetitionDraw()) {
397 return std::vector<double>{DrawUtility(), DrawUtility()};
398 }
399 // Compute and cache the legal actions.
400 MaybeGenerateLegalActions();
401 SPIEL_CHECK_TRUE(cached_legal_actions_);
402 bool have_legal_moves = !cached_legal_actions_->empty();
403
404 // If we don't have legal moves we are either stalemated or checkmated,
405 // depending on whether we are in check or not.
406 if (!have_legal_moves) {
407 if (!Board().InCheck()) {
408 return std::vector<double>{DrawUtility(), DrawUtility()};
409 } else {
410 std::vector<double> returns(NumPlayers());
411 auto next_to_play = ColorToPlayer(Board().ToPlay());
412 returns[next_to_play] = LossUtility();
413 returns[OtherPlayer(next_to_play)] = WinUtility();
414 return returns;
415 }
416 }
417
418 if (Board().IrreversibleMoveCounter() >= kNumReversibleMovesToDraw) {
419 // This is theoretically a draw that needs to be claimed, but we implement
420 // it as a forced draw for now.
421 return std::vector<double>{DrawUtility(), DrawUtility()};
422 }
423
424 return absl::nullopt;
425}

436std::unique_ptr<State> ChessGame::DeserializeState(
437 const std::string& str) const {
438 const std::string prefix("FEN: ");
439 if (!absl::StartsWith(str, prefix)) {
440 // Backward compatibility.
441 return Game::DeserializeState(str);
442 }
443 std::vector<std::string> lines = absl::StrSplit(str, '\n');
444 // Create initial state from FEN (first line of serialized state).
445 std::unique_ptr<State> state = NewInitialState(
446 lines[0].substr(prefix.length()));
447 for (int i = 1; i < lines.size(); ++i) {
448 if (lines[i].empty()) {
449 break;
450 }
451 Action action = static_cast<Action>(std::stol(lines[i]));
452 state->ApplyAction(action);
453 }
454 return state;
455}
