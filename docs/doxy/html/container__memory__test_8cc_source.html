61template <typename T>
62struct TypeCountingAllocator {
63 TypeCountingAllocator() = default;
64 template <typename U>
65 TypeCountingAllocator(const TypeCountingAllocator<U>&) {} // NOLINT
66
67 using value_type = T;
68
69 T* allocate(size_t n, const void* = nullptr) {
70 AllocationMap()[typeid(T)] += n;
71 return std::allocator<T>().allocate(n);
72 }
73 void deallocate(T* p, std::size_t n) {
74 AllocationMap()[typeid(T)] -= n;
75 return std::allocator<T>().deallocate(p, n);
76 }
77};

79TEST(Memory, AllocateDeallocateMatchType) {
