15 """An implementation of sequence-form linear programming.
16 
17 This is a classic algorithm for solving two-player zero-sum games with imperfect
18 information. For a general introduction to the concepts, see Sec 5.2.3 of
19 Shoham & Leyton-Brown '09, Multiagent Systems: Algorithmic, Game-Theoretic, and
20 Logical Foundations http://www.masfoundations.org/mas.pdf.
21 
22 In this implementation, we follow closely the construction in Koller, Megiddo,
23 and von Stengel, Fast Algorithms for Finding Randomized Strategies in Game Trees
24 http://theory.stanford.edu/~megiddo/pdf/stoc94.pdf. Specifically, we construct
25 and solve equations (8) and (9) from this paper.
26 """

39 def _construct_lps(state, infosets, infoset_actions, infoset_action_maps,
40  chance_reach, lps, parent_is_keys, parent_isa_keys):
41  """Build the linear programs recursively from this state.
42 
43  Args:
44  state: an open spiel state (root of the game tree)
45  infosets: a list of dicts, one per player, that maps infostate to an id. The
46  dicts are filled by this function and should initially only contain root
47  values.
48  infoset_actions: a list of dicts, one per player, that maps a string of
49  (infostate, action) pair to an id. The dicts are filled by this function
50  and should inirially only contain the root values
51  infoset_action_maps: a list of dicts, one per player, that maps each
52  info_state to a list of (infostate, action) string
53  chance_reach: the contribution of chance's reach probability (should start
54  at 1).
55  lps: a list of linear programs, one per player. The first one will be
56  constructred as in Eq (8) of Koller, Megiddo and von Stengel. The second
57  lp is Eq (9). Initially these should contain only the root-level
58  constraints and variables.
59  parent_is_keys: a list of parent information state keys for this state
60  parent_isa_keys: a list of parent (infostate, action) keys
61  """
62  if state.is_terminal():
63  returns = state.returns()
64  # Left-most term of: -Ay + E^t p >= 0
65  lps[0].add_or_reuse_constraint(parent_isa_keys[0], lp_solver.CONS_TYPE_GEQ)
66  lps[0].add_to_cons_coeff(parent_isa_keys[0], parent_isa_keys[1],
67  -1.0 * returns[0] * chance_reach)
68  # Right-most term of: -Ay + E^t p >= 0
69  lps[0].set_cons_coeff(parent_isa_keys[0], parent_is_keys[0], 1.0)
70  # Left-most term of: x^t (-A) - q^t F <= 0
71  lps[1].add_or_reuse_constraint(parent_isa_keys[1], lp_solver.CONS_TYPE_LEQ)
72  lps[1].add_to_cons_coeff(parent_isa_keys[1], parent_isa_keys[0],
73  -1.0 * returns[0] * chance_reach)
74  # Right-most term of: x^t (-A) - q^t F <= 0
75  lps[1].set_cons_coeff(parent_isa_keys[1], parent_is_keys[1], -1.0)
76  return
77 
78  if state.is_chance_node():
79  for action, prob in state.chance_outcomes():
80  new_state = state.child(action)
81  _construct_lps(new_state, infosets, infoset_actions, infoset_action_maps,
82  prob * chance_reach, lps, parent_is_keys, parent_isa_keys)
83  return
84 
85  player = state.current_player()
86  info_state = state.information_state_string(player)
87  legal_actions = state.legal_actions(player)
88 
89  # p and q variables, inequality constraints, and part of equality constraints
90  if player == 0:
91  # p
92  lps[0].add_or_reuse_variable(info_state)
93  # -Ay + E^t p >= 0
94  lps[0].add_or_reuse_constraint(parent_isa_keys[0], lp_solver.CONS_TYPE_GEQ)
95  lps[0].set_cons_coeff(parent_isa_keys[0], parent_is_keys[0], 1.0)
96  lps[0].set_cons_coeff(parent_isa_keys[0], info_state, -1.0)
97  # x^t E^t = e^t
98  lps[1].add_or_reuse_constraint(info_state, lp_solver.CONS_TYPE_EQ)
99  lps[1].set_cons_coeff(info_state, parent_isa_keys[0], -1.0)
100  else:
101  # q
102  lps[1].add_or_reuse_variable(info_state)
103  # x^t (-A) - q^t F <= 0
104  lps[1].add_or_reuse_constraint(parent_isa_keys[1], lp_solver.CONS_TYPE_LEQ)
105  lps[1].set_cons_coeff(parent_isa_keys[1], parent_is_keys[1], -1.0)
106  lps[1].set_cons_coeff(parent_isa_keys[1], info_state, 1.0)
107  # -Fy = -f
108  lps[0].add_or_reuse_constraint(info_state, lp_solver.CONS_TYPE_EQ)
109  lps[0].set_cons_coeff(info_state, parent_isa_keys[1], -1.0)
110 
111  # Add to the infostate maps
112  if info_state not in infosets[player]:
113  infosets[player][info_state] = len(infosets[player])
114  if info_state not in infoset_action_maps[player]:
115  infoset_action_maps[player][info_state] = []
116 
117  new_parent_is_keys = parent_is_keys[:]
118  new_parent_is_keys[player] = info_state
119 
120  for action in legal_actions:
121  isa_key = info_state + _DELIMITER + str(action)
122  if isa_key not in infoset_actions[player]:
123  infoset_actions[player][isa_key] = len(infoset_actions[player])
124  if isa_key not in infoset_action_maps[player][info_state]:
125  infoset_action_maps[player][info_state].append(isa_key)
126 
127  # x and y variables, and finish equality constraints coeff
128  if player == 0:
129  lps[1].add_or_reuse_variable(isa_key, lb=0) # x
130  lps[1].set_cons_coeff(info_state, isa_key, 1.0) # x^t E^t = e^t
131  else:
132  lps[0].add_or_reuse_variable(isa_key, lb=0) # y
133  lps[0].set_cons_coeff(info_state, isa_key, 1.0) # -Fy = -f
134 
135  new_parent_isa_keys = parent_isa_keys[:]
136  new_parent_isa_keys[player] = isa_key
137 
138  new_state = state.child(action)
139  _construct_lps(new_state, infosets, infoset_actions, infoset_action_maps,
140  chance_reach, lps, new_parent_is_keys, new_parent_isa_keys)

143 def solve_zero_sum_game(game, solver=None):
144  """Solve the two-player zero-sum game using sequence-form LPs.
145 
146  Args:
147  game: the spiel game tp solve (must be zero-sum, sequential, and have chance
148  mode of deterministic or explicit stochastic).
149  solver: a specific solver to use, sent to cvxopt (i.e. 'lapack', 'blas',
150  'glpk'). A value of None uses cvxopt's default solver.
151 
152  Returns:
153  A 4-tuple containing:
154  - player 0 value
155  - player 1 value
156  - player 0 policy: a policy.TabularPolicy for player 0
157  - player 1 policy: a policy.TabularPolicy for player 1
158  """
159  assert game.num_players() == 2
160  assert game.get_type().utility == pyspiel.GameType.Utility.ZERO_SUM
161  assert game.get_type().dynamics == pyspiel.GameType.Dynamics.SEQUENTIAL
162  assert (
163  game.get_type().chance_mode == pyspiel.GameType.ChanceMode.DETERMINISTIC
164  or game.get_type().chance_mode ==
165  pyspiel.GameType.ChanceMode.EXPLICIT_STOCHASTIC)
166  # There are several import matrices and vectors that form the LPs that
167  # are built by this function:
168  #
169  # A is expected payoff to p1 of each (infoset0,action0) + (infoset1,action1)
170  # belong to p1 and p2 respectively, which lead to a terminal state. It has
171  # dimensions (infoset-actions0) x (infoset-actions1)
172  # E,F are p1 / p2's strategy matrices (infosets) x (infoset-actions)
173  # e,f are infosets+ x 1 column vector of (1 0 0 ... 0)
174  # p,q are unconstrained variables each with infosets x 1.
175  # x,y are realization plans of size infoset-actions
176  #
177  # In each of the computations above there is a special "root infoset" and
178  # "root infoset-action" denote \emptyset. So the values are actually equal to
179  # number of infosets + 1 and infoset-actions + 1.
180  #
181  # Equation (8) is min_{y,p} e^T p
182  #
183  # s.t. -Ay + E^t p >= 0
184  # -Fy = -f
185  # y >= 0
186  #
187  # Equation (9) is max_{x,q} -q^T f
188  #
189  # s.t. x^t(-A) - q^t F <= 0
190  # x^t E^t = e^t
191  # x >= 0
192  #
193  # So, the first LP has:
194  # - |y| + |p| variables (infoset-actions1 + infosets0)
195  # - infoset-actions0 inequality constraints (other than var lower-bounds)
196  # - infosets1 equality constraints
197  #
198  # And the second LP has:
199  # - |x| + |q| variables (infoset-actions0 + infosets1)
200  # - infoset-actions1 inequality constraints (other than var lower-bounds)
201  # - infosets0 equality constraints
202  infosets = [{_EMPTY_INFOSET_KEYS[0]: 0}, {_EMPTY_INFOSET_KEYS[1]: 0}]
203  infoset_actions = [{
204  _EMPTY_INFOSET_ACTION_KEYS[0]: 0
205  }, {
206  _EMPTY_INFOSET_ACTION_KEYS[1]: 0
207  }]
208  infoset_action_maps = [{}, {}]
209  lps = [
210  lp_solver.LinearProgram(lp_solver.OBJ_MIN), # Eq. (8)
211  lp_solver.LinearProgram(lp_solver.OBJ_MAX) # Eq. (9)
212  ]
213  # Root-level variables and constraints.
214  lps[0].add_or_reuse_variable(_EMPTY_INFOSET_ACTION_KEYS[1], lb=0) # y root
215  lps[0].add_or_reuse_variable(_EMPTY_INFOSET_KEYS[0]) # p root
216  lps[1].add_or_reuse_variable(_EMPTY_INFOSET_ACTION_KEYS[0], lb=0) # x root
217  lps[1].add_or_reuse_variable(_EMPTY_INFOSET_KEYS[1]) # q root
218  # objective coefficients
219  lps[0].set_obj_coeff(_EMPTY_INFOSET_KEYS[0], 1.0) # e^t p
220  lps[1].set_obj_coeff(_EMPTY_INFOSET_KEYS[1], -1.0) # -q^t f
221  # y_root = 1 (-Fy = -f)
222  lps[0].add_or_reuse_constraint(_EMPTY_INFOSET_KEYS[1], lp_solver.CONS_TYPE_EQ)
223  lps[0].set_cons_coeff(_EMPTY_INFOSET_KEYS[1], _EMPTY_INFOSET_ACTION_KEYS[1],
224  -1.0)
225  lps[0].set_cons_rhs(_EMPTY_INFOSET_KEYS[1], -1.0)
226  # x_root = 1 (x^t E^t = e^t)
227  lps[1].add_or_reuse_constraint(_EMPTY_INFOSET_KEYS[0], lp_solver.CONS_TYPE_EQ)
228  lps[1].set_cons_coeff(_EMPTY_INFOSET_KEYS[0], _EMPTY_INFOSET_ACTION_KEYS[0],
229  1.0)
230  lps[1].set_cons_rhs(_EMPTY_INFOSET_KEYS[0], 1.0)
231  _construct_lps(game.new_initial_state(), infosets, infoset_actions,
232  infoset_action_maps, 1.0, lps, _EMPTY_INFOSET_KEYS[:],
233  _EMPTY_INFOSET_ACTION_KEYS[:])
234  # Solve the programs.
235  solutions = [lps[0].solve(solver=solver), lps[1].solve(solver=solver)]
236  # Extract the policies (convert from realization plan to behavioral form).
237  policies = [policy.TabularPolicy(game), policy.TabularPolicy(game)]
238  for i in range(2):
239  for info_state in infoset_action_maps[i]:
240  total_weight = 0
241  num_actions = 0
242  for isa_key in infoset_action_maps[i][info_state]:
243  total_weight += solutions[1 - i][lps[1 - i].get_var_id(isa_key)]
244  num_actions += 1
245  unif_pr = 1.0 / num_actions
246  state_policy = policies[i].policy_for_key(info_state)
247  for isa_key in infoset_action_maps[i][info_state]:
248  # The 1 - i here is due to Eq (8) yielding a solution for player 1 and
249  # Eq (9) a solution for player 0.
250  rel_weight = solutions[1 - i][lps[1 - i].get_var_id(isa_key)]
251  _, action_str = isa_key.split(_DELIMITER)
252  action = int(action_str)
253  pr_action = rel_weight / total_weight if total_weight > 0 else unif_pr
254  state_policy[action] = pr_action
255  return (solutions[0][lps[0].get_var_id(_EMPTY_INFOSET_KEYS[0])],
256  solutions[1][lps[1].get_var_id(_EMPTY_INFOSET_KEYS[1])], policies[0],
257  policies[1])
