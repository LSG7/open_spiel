25def _print_columns(strings):
26 """Prints a list of strings in columns."""
27 padding = 2
28 longest = max(len(s) for s in strings)
29 max_columns = math.floor((_MAX_WIDTH - 1) / (longest + 2 * padding))
30 rows = math.ceil(len(strings) / max_columns)
31 columns = math.ceil(len(strings) / rows) # Might not fill all max_columns.
32 for r in range(rows):
33 for c in range(columns):
34 i = r + c * rows
35 if i < len(strings):
36 print(" " * padding + strings[i].ljust(longest + padding), end="")
37 print()

43 def step_with_policy(self, state):
44 """Returns the stochastic policy and selected action in the given state."""
45 legal_actions = state.legal_actions(state.current_player())
46 if not legal_actions:
47 return [], pyspiel.INVALID_ACTION
48 p = 1 / len(legal_actions)
49 policy = [(action, p) for action in legal_actions]
50
51 action_map = {
52 state.action_to_string(state.current_player(), action): action
53 for action in legal_actions
54 }
55
56 while True:
57 action_str = input("Choose an action (empty to print legal actions): ")
58
59 if not action_str:
60 print("Legal actions(s):")
61 longest_num = max(len(str(action)) for action in legal_actions)
62 _print_columns([
63 "{}: {}".format(str(action).rjust(longest_num), action_str)
64 for action_str, action in sorted(action_map.items())
65 ])
66 continue
67
68 if action_str in action_map:
69 return policy, action_map[action_str]
70
71 try:
72 action = int(action_str)
73 except ValueError:
74 print("Could not parse the action:", action_str)
75 continue
76
77 if action in legal_actions:
78 return policy, action
79
80 print("Illegal action selected:", action_str)
