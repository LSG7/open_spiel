38class VPNetModel {
39 public:
40 // A class to handle the network's loss.
41 class LossInfo {
42 public:
43 LossInfo() {}
44 LossInfo(double policy, double value, double l2)
45 : policy_(policy), value_(value), l2_(l2), batches_(1) {}
46
47 // Merge another LossInfo into this one.
48 LossInfo& operator+=(const LossInfo& other) {
49 policy_ += other.policy_;
50 value_ += other.value_;
51 l2_ += other.l2_;
52 batches_ += other.batches_;
53 return *this;
54 }
55
56 // Return the average losses over all merged into this one.
57 double Policy() const { return policy_ / batches_; }
58 double Value() const { return value_ / batches_; }
59 double L2() const { return l2_ / batches_; }
60 double Total() const { return Policy() + Value() + L2(); }
61
62 private:
63 double policy_ = 0;
64 double value_ = 0;
65 double l2_ = 0;
66 int batches_ = 0;
67 };
