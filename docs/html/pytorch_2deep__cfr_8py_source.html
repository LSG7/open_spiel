15 """Implements Deep CFR Algorithm.

20 advantages used to do regret matching across information sets and to approximate

141 class ReservoirBuffer(object):

338  def _traverse_game_tree(self, state, player):
339  """Performs a traversal of the game tree.
340 
341  Over a traversal the advantage and strategy memories are populated with
342  computed advantage values and matched regrets respectively.
343 
344  Args:
345  state: Current OpenSpiel game state.
346  player: (int) Player index for this traversal.
347 
348  Returns:
349  (float) Recursively returns expected payoffs for each action.
350  """
351  expected_payoff = collections.defaultdict(float)
352  if state.is_terminal():
353  # Terminal state get returns.
354  return state.returns()[player]
355  elif state.is_chance_node():
356  # If this is a chance node, sample an action
357  chance_outcome, chance_proba = zip(*state.chance_outcomes())
358  action = np.random.choice(chance_outcome, p=chance_proba)
359  return self._traverse_game_tree_traverse_game_tree(state.child(action), player)
360  elif state.current_player() == player:
361  sampled_regret = collections.defaultdict(float)
362  # Update the policy over the info set & actions via regret matching.
363  _, strategy = self._sample_action_from_advantage_sample_action_from_advantage(state, player)
364  for action in state.legal_actions():
365  expected_payoff[action] = self._traverse_game_tree_traverse_game_tree(
366  state.child(action), player)
367  cfv = 0
368  for a_ in state.legal_actions():
369  cfv += strategy[a_] * expected_payoff[a_]
370  for action in state.legal_actions():
371  sampled_regret[action] = expected_payoff[action]
372  sampled_regret[action] -= cfv
373  sampled_regret_arr = [0] * self._num_actions_num_actions
374  for action in sampled_regret:
375  sampled_regret_arr[action] = sampled_regret[action]
376  self._advantage_memories_advantage_memories[player].add(
377  AdvantageMemory(state.information_state_tensor(), self._iteration_iteration,
378  sampled_regret_arr, action))
379  return cfv
380  else:
381  other_player = state.current_player()
382  _, strategy = self._sample_action_from_advantage_sample_action_from_advantage(state, other_player)
383  # Recompute distribution for numerical errors.
384  probs = np.array(strategy)
385  probs /= probs.sum()
386  sampled_action = np.random.choice(range(self._num_actions_num_actions), p=probs)
387  self._strategy_memories_strategy_memories.add(
388  StrategyMemory(
389  state.information_state_tensor(other_player), self._iteration_iteration,
390  strategy))
391  return self._traverse_game_tree_traverse_game_tree(state.child(sampled_action), player)

393  def _sample_action_from_advantage(self, state, player):
394  """Returns an info state policy by applying regret-matching.
395 
396  Args:
397  state: Current OpenSpiel game state.
398  player: (int) Player index over which to compute regrets.
399 
400  Returns:
401  1. (list) Advantage values for info state actions indexed by action.
402  2. (list) Matched regrets, prob for actions indexed by action.
403  """
404  info_state = state.information_state_tensor(player)
405  legal_actions = state.legal_actions(player)
406  with torch.no_grad():
407  state_tensor = torch.FloatTensor(np.expand_dims(info_state, axis=0))
408  raw_advantages = self._advantage_networks_advantage_networks[player](state_tensor)[0].numpy()
409  advantages = [max(0., advantage) for advantage in raw_advantages]
410  cumulative_regret = np.sum([advantages[action] for action in legal_actions])
411  matched_regrets = np.array([0.] * self._num_actions_num_actions)
412  if cumulative_regret > 0.:
413  for action in legal_actions:
414  matched_regrets[action] = advantages[action] / cumulative_regret
415  else:
416  matched_regrets[max(legal_actions, key=lambda a: raw_advantages[a])] = 1
417  return advantages, matched_regrets
