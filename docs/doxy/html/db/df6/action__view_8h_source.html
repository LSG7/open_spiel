69// Provides a number of iterators that are useful for dealing
70// with simultaneous move nodes.
71struct ActionView {
72 const Player current_player;
73 const std::vector<std::vector<Action>> legal_actions;
74 // Collects legal actions at the specified state.
75 explicit ActionView(const State& state);
76 // Construct a custom action view.
77 ActionView(const Player current_player,
78 const std::vector<std::vector<Action>> legal_actions);
79
80 int num_players() const { return legal_actions.size(); }
81 int num_actions(Player pl) const { return legal_actions.at(pl).size(); }
82
83 // Provides an iterator over all flattened joint actions.
84 //
85 // It computes the number of possible joint actions = \prod #actions(i)
86 // over all the players with any legal actions available.
87 // The possible joint actions are just numbered 0, 1, 2, .... and can be
88 // decomposed into the individual actions of the players.
89 //
90 // As this is an iterator, it does not allocate memory for the whole cartesian
91 // product of the actions.
92 FlatJointActions flat_joint_actions() const;
93
94 // Provides an iterator over flattened actions, while we fix one action
95 // for the specified player.
96 FixedActions fixed_action(Player player, int action_index) const;
97};
