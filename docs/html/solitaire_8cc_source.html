167 std::vector<SuitType> GetOppositeSuits(const SuitType& suit) {
168  /* Just returns a vector of the suits of opposite color. For red suits
169  * (SuitType::kHearts and SuitType::kDiamonds), this returns the black suits
170  * (SuitType::kSpades and SuitType::kClubs). For a black suit_, this returns
171  * the red suits. The last `SuitType` would be `SuitType::kNone` which should
172  * only occur with empty tableau cards or hidden cards. Empty tableau
173  * cards should accept any suit, but hidden cards are the opposite; they
174  * shouldn't accept any. There isn't really a use case for calling this
175  * function with the suit of a hidden card though. */
176 
177  switch (suit) {
178  case SuitType::kSpades: {
179  return {SuitType::kHearts, SuitType::kDiamonds};
180  }
181  case SuitType::kHearts: {
182  return {SuitType::kSpades, SuitType::kClubs};
183  }
184  case SuitType::kClubs: {
185  return {SuitType::kHearts, SuitType::kDiamonds};
186  }
187  case SuitType::kDiamonds: {
188  return {SuitType::kSpades, SuitType::kClubs};
189  }
190  case SuitType::kNone: {
191  return {SuitType::kSpades, SuitType::kHearts, SuitType::kClubs,
192  SuitType::kDiamonds};
193  }
194  default: {
195  SpielFatalError("suit is not in (s, h, c, d)");
196  }
197  }
198 }

200 int GetCardIndex(RankType rank, SuitType suit) {
201  /* Using a given rank and/or suit, gets an integer representing the index
202  * of the card. */
203 
204  if (rank == RankType::kHidden || suit == SuitType::kHidden) {
205  // Handles hidden_ cards_
206  return kHiddenCard;
207  } else if (rank == RankType::kNone) {
208  // Handles special cards_
209  if (suit == SuitType::kNone) {
210  // Handles empty tableau cards_
211  return kEmptyTableauCard;
212  } else {
213  // Handles empty foundation cards
214  switch (suit) {
215  case SuitType::kSpades: {
216  return kEmptySpadeCard;
217  }
218  case SuitType::kHearts: {
219  return kEmptyHeartCard;
220  }
221  case SuitType::kClubs: {
222  return kEmptyClubCard;
223  }
224  case SuitType::kDiamonds: {
225  return kEmptyDiamondCard;
226  }
227  default: {
228  SpielFatalError("Failed to get card index_");
229  }
230  }
231  }
232  } else {
233  // Handles ordinary cards (e.g. 0-13 -> spades, 14-26 -> hearts, etc.)
234  return (static_cast<int>(suit) - 1) * kNumRanks + static_cast<int>(rank);
235  }
236 }

238 int GetMaxSize(LocationType location) {
239  if (location >= LocationType::kDeck && location <= LocationType::kWaste) {
240  // Cards can only be removed from the waste_& there are 24 cards_ in it
241  // at the start of the game
242  return kMaxSizeWaste;
243  } else if (location == LocationType::kFoundation) {
244  // There are 13 cards_ in a suit_
245  return kMaxSizeFoundation;
246  } else if (location == LocationType::kTableau) {
247  // There are a maximum of 6 hidden cards and 13 non-hidden cards in a
248  // tableau (1 for each rank)
249  return kMaxSizeTableau;
250  } else {
251  return 0;
252  }
253 }

262 Card::Card(int index, bool hidden, LocationType location)
263  : location_(location), hidden_(hidden), index_(index) {
264  if (!hidden_) {
265  switch (index_) {
266  case kHiddenCard: {
267  rank_ = RankType::kHidden;
268  suit_ = SuitType::kHidden;
269  break;
270  }
271  case kEmptyTableauCard: {
272  rank_ = RankType::kNone;
273  suit_ = SuitType::kNone;
274  break;
275  }
276  case kEmptySpadeCard: {
277  rank_ = RankType::kNone;
278  suit_ = SuitType::kSpades;
279  break;
280  }
281  case kEmptyHeartCard: {
282  rank_ = RankType::kNone;
283  suit_ = SuitType::kHearts;
284  break;
285  }
286  case kEmptyClubCard: {
287  rank_ = RankType::kNone;
288  suit_ = SuitType::kClubs;
289  break;
290  }
291  case kEmptyDiamondCard: {
292  rank_ = RankType::kNone;
293  suit_ = SuitType::kDiamonds;
294  break;
295  }
296  default: {
297  // Converts an index back into a rank and suit for ordinary cards
298  rank_ = static_cast<RankType>(1 + ((index_ - 1) % kNumRanks));
299  suit_ = static_cast<SuitType>(
300  static_cast<int>(1 + floor((index_ - 1) / 13.0)));
301  }
302  }
303  }
304 }

335 std::string Card::ToString(bool colored) const {
336  std::string result;
337 
338  // Determine color of string
339  if (colored && !hidden_) {
340  if (suit_ == SuitType::kSpades || suit_ == SuitType::kClubs) {
341  absl::StrAppend(&result, kBlack);
342  } else if (suit_ == SuitType::kHearts || suit_ == SuitType::kDiamonds) {
343  absl::StrAppend(&result, kRed);
344  }
345  }
346 
347  // Determine contents of string
348  if (rank_ == RankType::kHidden || suit_ == SuitType::kHidden) {

358  // Reset color if applicable
359  absl::StrAppend(&result, kReset);

365 std::vector<Card> Card::LegalChildren() const {
366  if (hidden_) {
367  return {};
368  } else {
369  RankType child_rank;
370  std::vector<SuitType> child_suits;
371 
372  // A card can have a maximum of 4 children
373  // (specifically, an empty tableau card can accept a king of any suit)
374  child_suits.reserve(4);
375 
376  switch (location_) {
377  case LocationType::kTableau: {
378  if (rank_ == RankType::kNone) {
379  if (suit_ == SuitType::kNone) {
380  // Empty tableaus can accept a king of any suit
381  child_rank = RankType::kK;
382  child_suits = kSuits;
383  break;
384  } else {
385  return {};
386  }
387  } else if (rank_ >= RankType::k2 && rank_ <= RankType::kK) {
388  // Ordinary cards (except aces) can accept cards of an opposite
389  // suit that is one rank lower
390  child_rank = static_cast<RankType>(static_cast<int>(rank_) - 1);
391  child_suits = GetOppositeSuits(suit_);
392  break;
393  } else {
394  // This will catch RankType::kA and RankType::kHidden
395  return {};
396  }
397  break;
398  }
399  case LocationType::kFoundation: {
400  if (rank_ == RankType::kNone) {
401  if (suit_ != SuitType::kNone) {
402  child_rank = static_cast<RankType>(static_cast<int>(rank_) + 1);
403  child_suits = {suit_};
404  } else {
405  return {};
406  }
407  } else if (rank_ >= RankType::kA && rank_ <= RankType::kQ) {
408  // Cards (except kings) can accept a card of the same suit that is
409  // one rank higher
410  child_rank = static_cast<RankType>(static_cast<int>(rank_) + 1);
411  child_suits = {suit_};
412  } else {
413  // This could catch RankType::kK and RankType::kHidden
414  return {};
415  }
416  break;
417  }
418  default: {
419  // This catches all cards_ that aren't located in a tableau or
420  // foundation
421  return {};
422  }
423  }
424 
425  std::vector<Card> legal_children;
426  legal_children.reserve(4);
427 
428  if (child_suits.empty()) {
429  SpielFatalError("child_suits should not be empty");
430  }
431 
432  for (const auto& child_suit : child_suits) {
433  auto child = Card(false, child_suit, child_rank);
434  legal_children.push_back(child);
435  }
436 
437  return legal_children;
438  }
439 }

792 Move::Move(Action action) {
793  // `base` refers to the starting point that indices start from (e.g. if it's
794  // 7, and there's 3 cards in its group, their action ids will be 8, 9, 10).
795  // `residual` is just the difference between the id and the base.
796 
797  int residual;
798  int target_rank;
799  int source_rank;
800  int target_suit;
801  int source_suit;
802 
803  std::vector<SuitType> opposite_suits;
804  action -= kActionOffset;
805 
806  // The numbers used in the cases below are just used to divide action ids into
807  // groups (e.g. 1-132 are regular moves, 133-136 are the action ids of moves
808  // that move an ace to an empty foundation, etc.)
809 
810  if (action >= 1 && action <= 132) {
811  // Handles ordinary moves
812  target_rank = ((action - 1) / 3) % 11 + 2;
813  target_suit = ((action - 1) / 33) + 1;
814  residual = ((action - 1) % 3);
815  if (residual == 0) {
816  source_rank = target_rank + 1;
817  source_suit = target_suit;
818  } else {
819  opposite_suits = GetOppositeSuits(static_cast<SuitType>(target_suit));
820  source_rank = target_rank - 1;
821  source_suit = static_cast<int>(opposite_suits[residual - 1]);
822  }
823  } else if (action >= 133 && action <= 136) {
824  // Handles ace to empty foundation moves
825  target_rank = 0;
826  target_suit = action - 132;
827  source_rank = 1;
828  source_suit = target_suit;
829  } else if (action >= 137 && action <= 140) {
830  // Handles king to empty tableau moves
831  target_rank = 0;
832  target_suit = 0;
833  source_rank = 13;
834  source_suit = action - 136;
835  } else if (action >= 141 && action <= 144) {
836  // Handles moves with ace targets
837  target_rank = 1;
838  target_suit = action - 140;
839  source_rank = 2;
840  source_suit = target_suit;
841  } else if (action >= 145 && action <= 152) {
842  // Handles moves with king targets
843  target_rank = 13;
844  target_suit = (action - 143) / 2;
845 
846  residual = (action - 143) % 2;
847  opposite_suits = GetOppositeSuits(static_cast<SuitType>(target_suit));
848 
849  source_rank = 12;
850  source_suit = static_cast<int>(opposite_suits[residual]);
851  } else {
852  SpielFatalError("action provided does not correspond with a move");
853  }
854 
855  target_ = Card(false, static_cast<SuitType>(target_suit),
856  static_cast<RankType>(target_rank));
857  source_ = Card(false, static_cast<SuitType>(source_suit),
858  static_cast<RankType>(source_rank));
859 }

869 Action Move::ActionId() const {
870  int target_rank = static_cast<int>(target_.GetRank());
871  int source_rank = static_cast<int>(source_.GetRank());
872  int target_suit = static_cast<int>(target_.GetSuit());
873  int source_suit = static_cast<int>(source_.GetSuit());
874 
875  int base;
876  int residual;
877 
878  // `base` refers to the starting point that indices start from (e.g. if it's
879  // 7, and there's 3 cards in its group, their action ids will be 8, 9, 10).
880  // `residual` is just the difference between the id and the base.
881 
882  switch (target_rank) {
883  case static_cast<int>(RankType::kNone): {
884  switch (source_rank) {
885  case static_cast<int>(RankType::kA): {
886  base = 132;
887  break;
888  }
889  case static_cast<int>(RankType::kK): {
890  base = 136;
891  break;
892  }
893  default: {
894  base = -999;
895  break;
896  // SpielFatalError("source_.rank_ has an incorrect value");
897  }
898  }
899  return base + source_suit + kActionOffset;
900  }
901  case static_cast<int>(RankType::kA): {
902  base = 140;
903  return base + source_suit + kActionOffset;
904  }
905  case static_cast<int>(RankType::kK): {
906  base = 144;
907  if (source_suit <= 2) {
908  residual = -1;
909  } else {
910  residual = 0;
911  }
912  return base + (2 * target_suit) + residual + kActionOffset;
913  }
914  default: {
915  base = (target_suit - 1) * 33 + (target_rank - 2) * 3;
916  if (target_suit == source_suit) {
917  residual = 1;
918  } else if (source_suit <= 2) {
919  residual = 2;
920  } else {
921  residual = 3;
922  }
923  return base + residual + kActionOffset;
924  }
925  }
926 }
