57inline int CalcXY(int x, int y, int board_size) {
58 int diameter = board_size * 2 - 1;
59 if (x >= 0 && y >= 0 && x < diameter && y < diameter &&
60 (y - x < board_size) && (x - y < board_size)) {
61 return x + y * diameter;
62 } else {
63 return kMoveUnknown;
64 }
65}

67struct Move {
68 int8_t x, y; // The x,y coordinates
69 int16_t xy; // precomputed x + y * board_diameter as an index into the array.
70
71 inline constexpr Move(MoveSpecial m = kMoveUnknown) : x(-1), y(-1), xy(m) {}
72 inline constexpr Move(int x_, int y_, MoveSpecial m) : x(x_), y(y_), xy(m) {}
73 Move(int x_, int y_, int board_size)
74 : x(x_), y(y_), xy(CalcXY(x_, y_, board_size)) {}
75
76 std::string ToString() const;
77
78 bool operator==(const Move& b) const { return xy == b.xy; }
79 bool operator!=(const Move& b) const { return xy != b.xy; }
80 bool operator==(const MoveSpecial& b) const { return xy == b; }
81 bool operator!=(const MoveSpecial& b) const { return xy != b; }
82
83 // Whether the move is valid and on the board. May be invalid because it is
84 // a MoveSpecial, in the cut-off corners, or otherwise off the board.
85 bool OnBoard() const { return xy >= 0; }
86
87 // Flags for which edge/corner this move is part of.
88 int Edge(int board_size) const;
89 int Corner(int board_size) const;
90};

95// State of an in-play game.
96class HavannahState : public State {
97 // Represents a single cell on the board, as well as the structures needed for
98 // groups of cells. Groups of cells are defined by a union-find structure
99 // embedded in the array of cells. Following the `parent` indices will lead to
100 // the group leader which has the up to date size, corner and edge
101 // connectivity of that group. Size, corner and edge are not valid for any
102 // cell that is not a group leader.
103 struct Cell {
104 // Who controls this cell.
105 HavannahPlayer player;
106
107 // Whether this cell is marked/visited in a ring search. Should always be
108 // false except while running CheckRingDFS.
109 bool mark;
110
111 // A parent index to allow finding the group leader. It is the leader of the
112 // group if it points to itself. Allows path compression to shorten the path
113 // from a direct parent to the leader.
114 uint16_t parent;
115
116 // These three are only defined for the group leader's cell.
117 uint16_t size; // Size of this group of cells.
118 uint8_t corner; // A bitset of which corners this group is connected to.
119 uint8_t edge; // A bitset of which edges this group is connected to.
120
121 Cell() {}
122 Cell(HavannahPlayer player_, int parent_, int corner_, int edge_)
123 : player(player_),
124 mark(false),
125 parent(parent_),
126 size(1),
127 corner(corner_),
128 edge(edge_) {}
129
130 // How many corner or edges this group of cell is connected to. Only defined
131 // if called on the group leader.
132 int NumCorners() const;
133 int NumEdges() const;
134 };

202 int NumDistinctActions() const override {
203 // Really diameter^2 - size*(size-1), but that's harder to represent, so
204 // the extra actions in the corners are never legal.

214 absl::optional<double> UtilitySum() const override { return 0; }
215 double MaxUtility() const override { return 1; }
216 std::vector<int> ObservationTensorShape() const override {
217 return {kCellStates, Diameter(), Diameter()};
218 }
219 int MaxGameLength() const override {
220 // The true number of playable cells on the board.
221 // No stones are removed, and it is possible to draw by filling the board.
222 return Diameter() * Diameter() - board_size_ * (board_size_ - 1) +
223 allow_swap_;
224 }
225
226 private:
227 int Diameter() const { return board_size_ * 2 - 1; }
228 const int board_size_;
229 const bool ansi_color_output_ = false;
230 const bool allow_swap_ = false;
