141 def test_policy_on_game(self, game, policy_object, player=-1):
142  """Checks the policy conforms to the conventions.
143 
144  Checks the Policy.action_probabilities contains only legal actions (but not
145  necessarily all).
146  Checks that the probabilities are positive and sum to 1.
147 
148  Args:
149  self: The Test class. This methid targets as being used as a utility
150  function to test policies.
151  game: A `pyspiel.Game`, same as the one used in the policy.
152  policy_object: A `policy.Policy` object on `game`. to test.
153  player: Restrict testing policy to a player.
154  """
155 
156  all_states = get_all_states.get_all_states(
157  game,
158  depth_limit=-1,
159  include_terminals=False,
160  include_chance_states=False,
161  to_string=lambda s: s.information_state_string())
162 
163  for state in all_states.values():
164  legal_actions = set(state.legal_actions())
165  action_probabilities = policy_object.action_probabilities(state)
166 
167  for action in action_probabilities.keys():
168  # We want a clearer error message to be able to debug.
169  actions_missing = set(legal_actions) - set(action_probabilities.keys())
170  illegal_actions = set(action_probabilities.keys()) - set(legal_actions)
171  self.assertIn(
172  action,
173  legal_actions,
174  msg="The action {} is present in the policy but is not a legal "
175  "actions (these are {})\n"
176  "Legal actions missing from policy: {}\n"
177  "Illegal actions present in policy: {}".format(
178  action, legal_actions, actions_missing, illegal_actions))
179 
180  sum_ = 0
181  for prob in action_probabilities.values():
182  sum_ += prob
183  self.assertGreaterEqual(prob, 0)
184  if player < 0 or state.current_player() == player:
185  self.assertAlmostEqual(1, sum_)
186  else:
187  self.assertAlmostEqual(0, sum_)
