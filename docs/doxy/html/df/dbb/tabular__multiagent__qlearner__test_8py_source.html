116 def test_rps_run(self):
117 env = rl_environment.Environment("matrix_rps")
118 nashqlearner0 = MultiagentQLearner(0, 2,
119 [env.game.num_distinct_actions()] * 2,
120 TwoPlayerNashSolver())
121
122 nashqlearner1 = MultiagentQLearner(1, 2,
123 [env.game.num_distinct_actions()] * 2,
124 TwoPlayerNashSolver())
125
126 for _ in range(1000):
127 time_step = env.reset()
128 actions = [None, None]
129 actions = [
130 nashqlearner0.step(time_step, actions).action,
131 nashqlearner1.step(time_step, actions).action
132 ]
133 time_step = env.step(actions)
134 nashqlearner0.step(time_step, actions)
135 nashqlearner1.step(time_step, actions)
136
137 with self.subTest("correct_rps_strategy"):
138 time_step = env.reset()
139 actions = [None, None]
140 learner0_strategy, learner1_strategy = nashqlearner0.step(
141 time_step, actions).probs, nashqlearner1.step(time_step,
142 actions).probs
143 np.testing.assert_array_almost_equal(
144 np.asarray([1 / 3, 1 / 3, 1 / 3]),
145 learner0_strategy.reshape(-1),
146 decimal=4)
147 np.testing.assert_array_almost_equal(
148 np.asarray([1 / 3, 1 / 3, 1 / 3]),
149 learner1_strategy.reshape(-1),
150 decimal=4)
151
152 with self.subTest("correct_rps_value"):
153 time_step = env.reset()
154 ground_truth_values = game_payoffs_array(
155 pyspiel.load_matrix_game("matrix_rps"))
156 info_state = str(time_step.observations["info_state"])
157 learner0_values, learner1_values = nashqlearner0._get_payoffs_array(
158 info_state), nashqlearner1._get_payoffs_array(info_state)
159 np.testing.assert_array_almost_equal(
160 ground_truth_values, learner0_values, decimal=4)
161 np.testing.assert_array_almost_equal(
162 ground_truth_values, learner1_values, decimal=4)
