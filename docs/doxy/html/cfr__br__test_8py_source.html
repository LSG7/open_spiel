62 def test_policy_and_average_policy(self):
63 game = pyspiel.load_game("kuhn_poker")
64 cfrbr_solver = cfr_br.CFRBRSolver(game)
65 for _ in range(300):
66 cfrbr_solver.evaluate_and_update_policy()
67 average_policy = cfrbr_solver.average_policy()
68 average_policy_values = expected_game_score.policy_value(
69 game.new_initial_state(), [average_policy] * 2)
70 # 1/18 is the Nash value. See https://en.wikipedia.org/wiki/Kuhn_poker
71 np.testing.assert_allclose(
72 average_policy_values, [-1 / 18, 1 / 18], atol=1e-3)

74 cfrbr_solver.current_policy()

83 def test_cpp_and_python_cfr_br(self, game, solver_cls,
84 expected_exploitability):
85 solver = solver_cls(game)
86 for step in range(5):
87 solver.evaluate_and_update_policy()
88
89 # We do not compare the policy directly as we do not have an easy way to
90 # convert one to the other, so we use the exploitability as a proxy.
91 avg_policy = solver.average_policy()
92 if solver_cls == pyspiel.CFRBRSolver:
93 exploitability_ = pyspiel.nash_conv(game, avg_policy)
94 else:
95 exploitability_ = exploitability.nash_conv(game, avg_policy)

97 self.assertAlmostEqual(
98 expected_exploitability[step], exploitability_, places=10
