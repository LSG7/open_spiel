54 def one_hot(self, x, n):
55 return np.eye(n)[x]

57 @property

61 def observation_spec(self):
62 info_state_spec, legal_actions_spec = [], []
63 for i in range(self._num_players_num_players):
64 num_actions = np.prod(self._payoff_matrix.shape[:-1]) + 1
65 if self._include_remaining_iterations:
66 num_actions += 1
67 info_state_spec.append([num_actions])
68 legal_actions_spec.append(self._payoff_matrix.shape[i])
69 return {

71 'legal_actions': tuple(legal_actions_spec),
72 'current_player': (),

75 def action_spec(self):
76 num_actions, mins, maxs = [], [], []
77 for i in range(self._num_players_num_players):
78 num_actions.append(self._payoff_matrix.shape[i])
79 mins.append(0)
80 maxs.append(self._payoff_matrix.shape[i] - 1)
81
82 return {
83 'num_actions': tuple(num_actions),
84 'min': tuple(mins),
85 'max': tuple(maxs),
86 'dtype': int,

89 def step(self, actions: np.ndarray):
90 if actions.ndim == 1:
91 actions = actions[None, :]
92 payoffs = self._payoff_matrix[tuple(actions.T)]
93 s1 = self.one_hot(
94 self._actions[tuple(actions.T)] + 1, n=np.max(self._actions) + 2
95 )
96 s2 = self.one_hot(
97 self._actions[tuple(actions[..., ::-1].T)] + 1,
98 n=np.max(self._actions) + 2,
99 )
100 rewards = [
101 np.squeeze(p)
102 for p in np.split(
103 payoffs, indices_or_sections=self._num_players_num_players, axis=1
104 )
105 ]
106 discounts = [np.ones_like(r) for r in rewards]
107 if self._t == self._iterations - 1:
108 step_type = StepType.LAST
109 else:
110 step_type = StepType.MID
111 self._t += 1
112 remaining_iters = float((self._iterations - self._t)) / self._iterations
113
114 info_state = [s1, s2]
115 if self._include_remaining_iterations:
116 info_state = np.concatenate(
117 [
118 info_state,
119 np.full((self._batch_size, 1), fill_value=remaining_iters),
120 ],
121 axis=-1,
122 )
123
124 legal_actions = self._get_legal_actions()
125 return TimeStep(
126 observations={
127 'info_state': info_state,
128 'legal_actions': legal_actions,
129 'batch_size': actions.shape[0],
130 'current_player': PlayerId.SIMULTANEOUS,
131 },
132 rewards=rewards,
133 discounts=discounts,
134 step_type=step_type,
135 )
