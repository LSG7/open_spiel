34void GetAllStatesTest() {
35 auto catch_game = LoadGame("catch");
36 auto states = algorithms::GetAllStates(*catch_game,
37 /*depth_limit=*/-1,
38 /*include_terminals=*/true,
39 /*include_chance_states=*/false);
40 SPIEL_CHECK_EQ(
41 states.size(),
42 kDefaultRows * kDefaultColumns * kDefaultColumns - 6 * kDefaultColumns);
43
44 // Verify number of states that lead to win and loss.
45 int num_wins = 0;
46 int num_losses = 0;
47 for (const auto& pair : states) {
48 const auto& state = pair.second;
49 if (state->IsTerminal()) {
50 if (state->PlayerReturn(0) == 1)
51 num_wins++;
52 else if (state->PlayerReturn(0) == -1)
53 num_losses++;
54 else
55 SpielFatalError("Unexpected return");
56 }
57 }
58 SPIEL_CHECK_EQ(num_wins, 5);
59 SPIEL_CHECK_EQ(num_losses, 20);
60
61 // Verify normalized observation matches string represtation.
62 for (const auto& pair : states) {
63 std::vector<float> obs(catch_game->ObservationTensorSize());
64 pair.second->ObservationTensor(0, absl::MakeSpan(obs));
65 const std::string& str = pair.first;
66 SPIEL_CHECK_EQ(obs.size(), str.size() - kDefaultRows);
67 for (int i = 0; i < obs.size(); i++) {
68 SPIEL_CHECK_EQ(obs[i] == 1, str[i + i / kDefaultColumns] != '.');
69 }
70 }
71}

73void PlayAndWinTest() {
74 auto game = LoadGame("catch");
75 auto state = game->NewInitialState();
76 SPIEL_CHECK_EQ(state->CurrentPlayer(), kChancePlayerId);
77 state->ApplyAction(3);
78 SPIEL_CHECK_EQ(state->CurrentPlayer(), 0);
79 state->ApplyAction(2); // Right.
80 for (int i = 0; i < kDefaultRows - 2; i++) {
81 SPIEL_CHECK_FALSE(state->IsTerminal());
82 state->ApplyAction(1); // Stay.
83 }
84 SPIEL_CHECK_TRUE(state->IsTerminal());
85 SPIEL_CHECK_EQ(state->PlayerReturn(0), 1);
86}
