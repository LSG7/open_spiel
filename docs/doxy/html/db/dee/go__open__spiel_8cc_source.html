149void StateFillChanceOutcomes(const void* state_ptr, void* action_ptr,
150 void* proba_ptr) {
151 const State* state = reinterpret_cast<const State*>(state_ptr);
152 int* action_array = reinterpret_cast<int*>(action_ptr);
153 double* proba_array = reinterpret_cast<double*>(proba_ptr);
154 std::vector<std::pair<open_spiel::Action, double>> outcomes =
155 state->ChanceOutcomes();
156 std::pair<open_spiel::Action, double> outcome;
157 for (int i = 0; i < outcomes.size(); ++i) {
158 outcome = outcomes[i];
159 action_array[i] = outcome.first;
160 proba_array[i] = outcome.second;
161 }
162}
