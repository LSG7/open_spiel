74 Player SheriffState::CurrentPlayer() const {
75  if (!num_illegal_items_) {
76  // The smuggler still hasn't decided the number of illegal items to
77  // place in the cargo. The game has just begun.
78  return kSmuggler;
79  } else if (bribes_.size() == inspection_feedback_.size()) {
80  // The smuggler has received feedback for all the bribes.
81  //
82  // If the number of bribes is equal to the number of bribing turns in the
83  // game, the game is over.
84  if (bribes_.size() == sheriff_game_->conf.num_rounds) {
85  return kTerminalPlayerId;
86  } else {
87  // Otherwise, a new bribing round begins.
88  return kSmuggler;
89  }
90  } else {
91  // The smuggles has made a bribe, but no feedback has been given out yet.
92  return kSheriff;
93  }
94 }

96 std::vector<Action> SheriffState::LegalActions() const {
97  const SheriffGame::SheriffGameConfiguration& conf = sheriff_game_->conf;
98 
99  std::vector<Action> action_ids;
100  if (IsTerminal()) {
101  return {};
102  } else if (!num_illegal_items_) {
103  // This is the beginning of the game. The smuggles must decide how many
104  // illegal items to place in the cargo, which must be an integer in the
105  // range [0, conf.max_items]. The action id will correspond to the number
106  // of illegal items placed in the cargo.
107 
108  action_ids.reserve(conf.max_items + 1);
109  for (uint32_t num_illegal_items = 0; num_illegal_items <= conf.max_items;
110  ++num_illegal_items) {
111  action_ids.push_back(
112  sheriff_game_->SerializeItemPlacementAction(num_illegal_items));
113  }
114  } else {
115  // If we are here, we are inside of a bribing round. There are two cases:
116  // - it is the *smuggler's* turn. This means that the bribing round has
117  // just started. The actions that the player can use correspond to the
118  // set of valid bribes, which is the range [0, conf.max_bribe].
119  //
120  // The action id corresponds to the bribe amount.
121  // - it is the *sheriff's* turn. The sheriff can decide to say they will
122  // _not_ inspect (action id: 0), or that they _will_ inspect the cargo
123  // (action id: 1).
124  const Player player = CurrentPlayer();
125 
126  if (player == kSmuggler) {
127  action_ids.reserve(conf.max_bribe + 1);
128  for (uint32_t bribe = 0; bribe <= conf.max_bribe; ++bribe) {
129  action_ids.push_back(sheriff_game_->SerializeBribe(bribe));
130  }
131  } else {
132  action_ids = {sheriff_game_->SerializeInspectionFeedback(false),
133  sheriff_game_->SerializeInspectionFeedback(true)};
134  }
135  }
136 
137  return action_ids;
138 }

167 std::vector<double> SheriffState::Returns() const {
168  if (!IsTerminal()) {
169  return {0.0, 0.0};
170  } else {
171  SPIEL_CHECK_EQ(inspection_feedback_.size(), bribes_.size());
172  SPIEL_CHECK_GT(inspection_feedback_.size(), 0);
173  SPIEL_CHECK_TRUE(num_illegal_items_);
174 
175  const uint32_t num_illegal_items = *num_illegal_items_;
176  const uint32_t bribe = bribes_.back();
177  const bool sheriff_inspects = inspection_feedback_.back();
178  const SheriffGame::SheriffGameConfiguration& conf = sheriff_game_->conf;
179 
180  if (sheriff_inspects) {
181  if (num_illegal_items > 0) {
182  // The smuggler was caught red-handed.
183  return {-static_cast<double>(num_illegal_items) * conf.item_penalty,
184  static_cast<double>(num_illegal_items) * conf.item_penalty};
185  } else {
186  // The sheriff must pay up for inspecting a legal cargo.
187  return {conf.sheriff_penalty, -conf.sheriff_penalty};
188  }
189  } else {
190  return {static_cast<double>(num_illegal_items) * conf.item_value - bribe,
191  static_cast<double>(bribe)};
192  }
193  }
194 }

200 std::string SheriffState::InformationStateString(Player player) const {
201  SPIEL_CHECK_TRUE(player >= 0 && player < NumPlayers());
202 
203  std::string infostring = absl::StrCat("T=", MoveNumber(), " ");
204  if (player == kSmuggler) {
205  absl::StrAppend(&infostring, "num_illegal_items:");
206  if (num_illegal_items_) {
207  absl::StrAppend(&infostring, *num_illegal_items_);
208  } else {
209  absl::StrAppend(&infostring, "none");

211  }
212 
213  SPIEL_CHECK_GE(inspection_feedback_.size() + 1, bribes_.size());
214  SPIEL_CHECK_LE(inspection_feedback_.size(), bribes_.size());
215  for (size_t index = 0; index < bribes_.size(); ++index) {
216  absl::StrAppend(&infostring, "/bribe:", bribes_.at(index));
217 
218  if (index < inspection_feedback_.size()) {
219  absl::StrAppend(&infostring,
220  "/feedback:", inspection_feedback_.at(index));
221  }
222  }
223 
224  return infostring;

227 std::vector<int> SheriffGame::InformationStateTensorShape() const {
228  return {
229  2 + // Whose turn?
230  2 + // Who is observing?
231  static_cast<int>(conf.num_rounds) + 1 + // Move number (0 to rounds)
232  static_cast<int>(conf.max_items) + 1 + // Number of items (0 to max)
233  // Each round, a bribe in { 0, 1, ..., max_bribe } plus one bit for yes/no
234  static_cast<int>(conf.num_rounds) *
235  (static_cast<int>(conf.max_bribe) + 1 + 1)
236  };
237 }

239 void SheriffState::InformationStateTensor(
240  Player player, absl::Span<float> values) const {
241  SPIEL_CHECK_TRUE(player >= 0 && player < NumPlayers());
242 
243  SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorSize());
244  std::fill(values.begin(), values.end(), 0);
245 
246  // Two-player game.
247  SPIEL_CHECK_TRUE(player == 0 || player == 1);
248 
249  int offset = 0;
250  const int num_players = game_->NumPlayers();
251  const Player cur_player = CurrentPlayer();
252  const auto* parent_game = down_cast<const SheriffGame*>(game_.get());
253 
254  // Set a bit to indicate whose turn it is.
255  if (cur_player != kTerminalPlayerId) {
256  values[cur_player] = 1;
257  }
258  offset += num_players;
259 
260  // Set a bit to indicate whose is observing
261  values[offset + player] = 1;
262  offset += num_players;
263 
264  // Move number
265  values[offset + MoveNumber()] = 1;
266  offset += parent_game->num_rounds() + 1;
267 
268  // Number of items chosen by the smuggler
269  if (player == kSmuggler) {
270  int index = (num_illegal_items_ ? *num_illegal_items_ : 0);
271  values[offset + index] = 1;
272  }
273  offset += parent_game->max_items() + 1;
274 
275  SPIEL_CHECK_GE(inspection_feedback_.size() + 1, bribes_.size());
276  SPIEL_CHECK_LE(inspection_feedback_.size(), bribes_.size());
277  for (size_t index = 0; index < bribes_.size(); ++index) {
278  int inner_offset = index * (parent_game->max_bribe() + 2);
279  values[offset + inner_offset + bribes_.at(index)] = 1;
280 
281  if (index < inspection_feedback_.size()) {
282  int bool_bit = inspection_feedback_.at(index) ? 0 : 1;
283  values[offset + inner_offset + parent_game->max_bribe() + 1] = bool_bit;
284  }
285  }
286  offset += parent_game->num_rounds() * (parent_game->max_bribe() + 2);
287 
288  SPIEL_CHECK_EQ(offset, values.size());
289 }

291 void SheriffState::UndoAction(Player player, Action action_id) {
292  SPIEL_CHECK_TRUE(!history_.empty() &&
293  (history_.back() == PlayerAction{player, action_id}));
294  history_.pop_back();
295  --move_number_;
296 
297  if (!bribes_.empty()) {
298  if (bribes_.size() == inspection_feedback_.size()) {
299  // The last action must have been for the sheriff to return feedback about
300  // whether or not the sheriff would inspect the cargo.
301  inspection_feedback_.pop_back();
302  } else {
303  // The last action must have been for the smuggler to offer a new bribe.
304  bribes_.pop_back();
305  }
306  } else {
307  // If there are no bribes yet, then the only possibility is that the game
308  // has just started and the only action so far was for the smuggler to
309  // select the number of illegal items to place into the smuggler's cargo.
310  SPIEL_CHECK_TRUE(num_illegal_items_);
311  num_illegal_items_ = absl::nullopt;
312  }
313 }

315 void SheriffState::DoApplyAction(Action action_id) {
316  SPIEL_CHECK_FALSE(IsTerminal());
317 
318  if (!num_illegal_items_) {
319  // The action must represent the selection of the number of illegal items in
320  // the cargo.
321 
322  SPIEL_CHECK_EQ(CurrentPlayer(), kSmuggler);
323  num_illegal_items_ =
324  sheriff_game_->DeserializeItemPlacementAction(action_id);
325  } else if (bribes_.size() == inspection_feedback_.size()) {
326  // The action must represent a new bribe made by the smuggler.
327  SPIEL_CHECK_EQ(CurrentPlayer(), kSmuggler);
328  bribes_.push_back(sheriff_game_->DeserializeBribe(action_id));
329  } else {
330  // The action must represent the inspection feedback returned by the
331  // sheriff.
332  SPIEL_CHECK_EQ(CurrentPlayer(), kSheriff);
333  inspection_feedback_.push_back(
334  sheriff_game_->DeserializeInspectionFeedback(action_id));
335  }
336 }

383 std::string SheriffGame::ActionToString(Player player, Action action_id) const {
384  std::string action_string;
385 
386  if (action_id < 2) {
387  SPIEL_CHECK_EQ(player, kSheriff);
388  const bool feedback = DeserializeInspectionFeedback(action_id);

401  } else {
402  SPIEL_CHECK_EQ(player, kSmuggler);
403 
404  const uint32_t bribe = DeserializeBribe(action_id);
405  absl::StrAppend(&action_string, "Bribe(amount=", bribe, ")");
