32void init_pyspiel_games_bridge(py::module& m) {
33 py::classh<BridgeState, State>(m, "BridgeState")
34 .def("contract_index", &BridgeState::ContractIndex)
35 .def("possible_contracts", &BridgeState::PossibleContracts)
36 .def("score_by_contract", &BridgeState::ScoreByContract)
37 .def("score_for_contracts", &BridgeState::ScoreForContracts)
38 .def("current_phase", &BridgeState::CurrentPhase)
39 .def("write_observation_tensor",
40 [](const BridgeState& state,
41 py::array_t<float, py::array::c_style> array) {
42 py::buffer_info buf = array.request();
43 SPIEL_CHECK_EQ(buf.ndim, 1);
44 SPIEL_CHECK_EQ(buf.strides.front(), buf.itemsize);
45 state.WriteObservationTensor(
46 state.CurrentPlayer(),
47 absl::MakeSpan(static_cast<float*>(buf.ptr),
48 buf.shape.front()));
49 })
50 .def("private_observation_tensor", &BridgeState::PrivateObservationTensor)
51 .def("public_observation_tensor", &BridgeState::PublicObservationTensor)
52 // Pickle support
53 .def(py::pickle(
54 [](const BridgeState& state) { // __getstate__
55 return SerializeGameAndState(*state.GetGame(), state);
56 },
57 [](const std::string& data) { // __setstate__
58 std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>
59 game_and_state = DeserializeGameAndState(data);
60 return dynamic_cast<BridgeState*>(game_and_state.second.release());
61 }));
62
63 py::classh<BridgeGame, Game>(m, "BridgeGame")
64 .def("num_possible_contracts", &BridgeGame::NumPossibleContracts)
65 .def("contract_string", &BridgeGame::ContractString)
66 .def("private_observation_tensor_size",
67 &BridgeGame::PrivateObservationTensorSize)
68 .def("public_observation_tensor_size",
69 &BridgeGame::PublicObservationTensorSize)
70 // Pickle support
71 .def(py::pickle(
72 [](std::shared_ptr<const BridgeGame> game) { // __getstate__
73 return game->ToString();
74 },
75 [](const std::string& data) { // __setstate__
76 return std::dynamic_pointer_cast<BridgeGame>(
77 std::const_pointer_cast<Game>(LoadGame(data)));
78 }));
79}
