79  std::string rv;
80  constexpr int kHandStringSize = 174;
81  rv.reserve(kHandStringSize);
82  // Top border
83  absl::StrAppend(&rv, "+");
84  for (int i = 0; i < num_ranks; ++i)
85  absl::StrAppend(&rv, "--");
86  absl::StrAppend(&rv, "+\n");
87  // One row for each suit

89  absl::StrAppend(&rv, "|");

92  absl::StrAppend(&rv, CardString((i * num_ranks) + j));
93  } else {
94  absl::StrAppend(&rv, " ");
95  }
96  }
97  absl::StrAppend(&rv, "|\n");
98  }
99  // Bottom border
100  absl::StrAppend(&rv, "+");
101  for (int i = 0; i < num_ranks; ++i)
102  absl::StrAppend(&rv, "--");
103  absl::StrAppend(&rv, "+\n");
104  return rv;

152 bool GinRummyUtils::IsSuitMeld(const VecInt &cards) const {
153  if (cards.size() < 3) {
154  return false;
155  }
156  // Check all of the same suit.
157  for (int i = 1; i < cards.size(); ++i) {
158  if (CardSuit(cards[0]) != CardSuit(cards[i])) {
159  return false;
160  }
161  }
162  // Check ranks are consecutive.
163  VecInt ranks;
164  for (int i = 0; i < cards.size(); ++i) {
165  ranks.push_back(CardRank(cards[i]));
166  }
167  absl::c_sort(ranks);
168  return IsConsecutive(ranks);
169 }

171 // Returns all possible rank melds that can be formed from cards.
172 VecVecInt GinRummyUtils::RankMelds(VecInt cards) const {
173  VecVecInt melds;
174  if (cards.size() < 3) {
175  return melds;
176  }
177  absl::c_sort(cards, rank_comp);
178  // First do a sweep for 4 card melds.
179  for (int i = 0; i < cards.size() - 3; ++i) {
180  // Found 4 card meld - implies there are four 3 card melds as well.
181  // We only add two of the 3 card melds here, the other two get added
182  // during the 3 card meld sweep.
183  if (CardRank(cards[i]) == CardRank(cards[i + 3])) {
184  melds.emplace_back(VecInt(cards.begin() + i, cards.begin() + i + 4));
185  melds.emplace_back(VecInt{cards[i], cards[i + 1], cards[i + 3]});
186  melds.emplace_back(VecInt{cards[i], cards[i + 2], cards[i + 3]});
187  }
188  }
189  // Sweep for 3 card melds.
190  for (int i = 0; i < cards.size() - 2; ++i) {
191  if (CardRank(cards[i]) == CardRank(cards[i + 2])) {
192  melds.emplace_back(VecInt(cards.begin() + i, cards.begin() + i + 3));
193  }
194  }
195  return melds;
196 }

198 // Returns all possible suit melds that can be formed from cards.
199 VecVecInt GinRummyUtils::SuitMelds(VecInt cards) const {
200  VecVecInt melds;
201  if (cards.size() < 3) {
202  return melds;
203  }
204  absl::c_sort(cards, suit_comp);
205  // Find all suit melds of length 5.
206  if (cards.size() >= 5) {
207  for (int i = 0; i < cards.size() - 4; ++i) {
208  if (cards[i] == cards[i + 4] - 4 &&
209  CardSuit(cards[i]) == CardSuit(cards[i + 4])) {
210  melds.emplace_back(VecInt(cards.begin() + i, cards.begin() + i + 5));
211  }
212  }
213  }
214  // Find all suit melds of length 4.
215  if (cards.size() >= 4) {
216  for (int i = 0; i < cards.size() - 3; ++i) {
217  if (cards[i] == cards[i + 3] - 3 &&
218  CardSuit(cards[i]) == CardSuit(cards[i + 3])) {
219  melds.emplace_back(VecInt(cards.begin() + i, cards.begin() + i + 4));
220  }
221  }
222  }
223  // Find all suit melds of length 3.
224  for (int i = 0; i < cards.size() - 2; ++i) {
225  if (cards[i] == cards[i + 2] - 2 &&
226  CardSuit(cards[i]) == CardSuit(cards[i + 2])) {
227  melds.emplace_back(VecInt(cards.begin() + i, cards.begin() + i + 3));
228  }
229  }
230  return melds;
231 }

242 bool GinRummyUtils::VectorsIntersect(VecInt *v1, VecInt *v2) const {
243  absl::c_sort(*v1);
244  absl::c_sort(*v2);
245  VecInt::iterator first1 = v1->begin();
246  VecInt::iterator last1 = v1->end();
247  VecInt::iterator first2 = v2->begin();
248  VecInt::iterator last2 = v2->end();
249 
250  while (first1 != last1 && first2 != last2) {
251  if (*first1 < *first2) {
252  ++first1;
253  } else if (*first2 < *first1) {
254  ++first2;
255  } else {
256  return true;
257  }
258  }
259  return false;
260 }

274 // Depth first search used by AllMeldGroups.
275 void GinRummyUtils::AllPaths(VecInt *meld, VecVecInt *all_melds,
276  VecVecInt *path, VecVecVecInt *all_paths) const {
277  path->push_back(*meld);
278  VecVecInt child_melds = NonOverlappingMelds(meld, all_melds);
279  if (child_melds.empty()) {
280  all_paths->push_back(*path);
281  } else {
282  for (auto child_meld : child_melds) {
283  AllPaths(&child_meld, &child_melds, path, all_paths);
284  }
285  }
286  path->pop_back();
287 }

289 // A meld group is an arrangement of cards into distinct melds.
290 // Accordingly, no two melds in a meld group can share the same card.
291 VecVecVecInt GinRummyUtils::AllMeldGroups(const VecInt &cards) const {
292  VecVecInt all_melds = AllMelds(cards);
293  VecVecVecInt all_meld_groups;
294  for (VecInt meld : all_melds) {
295  VecVecInt path;
296  AllPaths(&meld, &all_melds, &path, &all_meld_groups);
297  }
298  return all_meld_groups;
299 }

301 // "Best" means any meld group that achieves the lowest possible deadwood
302 // count for the given cards. In general this is non-unique.
303 VecVecInt GinRummyUtils::BestMeldGroup(const VecInt &cards) const {
304  int best_meld_group_total_value = 0;
305  VecVecInt best_meld_group;
306  VecVecVecInt all_meld_groups = AllMeldGroups(cards);
307  for (const auto &meld_group : all_meld_groups) {
308  int meld_group_total_value = TotalCardValue(meld_group);
309  if (meld_group_total_value > best_meld_group_total_value) {
310  best_meld_group_total_value = meld_group_total_value;
311  best_meld_group = meld_group;
312  }
313  }
314  return best_meld_group;
315 }

323 // Minimum deadwood count over all meld groups.
324 int GinRummyUtils::MinDeadwood(const VecInt &hand) const {
325  VecInt deadwood = hand;
326  VecVecInt best_melds = BestMeldGroup(hand);
327 
328  for (const auto &meld : best_melds) {
329  for (auto card : meld) {
330  deadwood.erase(std::remove(deadwood.begin(), deadwood.end(), card),
331  deadwood.end());
332  }
333  }
334  // If we have just drawn a card, we can discard the one worth the most points.
335  if (hand.size() == hand_size + 1 && !deadwood.empty()) {
336  absl::c_sort(deadwood, rank_comp);
337  deadwood.pop_back();
338  }
339  int deadwood_total = 0;
340  for (int card : deadwood) deadwood_total += CardValue(card);
341  return deadwood_total;
342 }

344 // Returns the unique card that can be layed off on a given 3-card rank meld.
345 int GinRummyUtils::RankMeldLayoff(const VecInt &meld) const {
346  SPIEL_CHECK_EQ(meld.size(), 3);
347  SPIEL_CHECK_TRUE(IsRankMeld(meld));
348  VecInt suits = {0, 1, 2, 3};
349  for (int card : meld) {
350  suits.erase(std::remove(suits.begin(), suits.end(), CardSuit(card)),
351  suits.end());
352  }
353  return CardRank(meld[0]) + suits[0] * num_ranks;
354 }

356 // Suit melds have two layoffs, except if the meld ends in an ace or king.
357 VecInt GinRummyUtils::SuitMeldLayoffs(const VecInt &meld) const {
358  VecInt layoffs;
359  int min_card_index = *std::min_element(meld.begin(), meld.end());
360  if (CardRank(min_card_index) > 0) {
361  layoffs.push_back(min_card_index - 1);
362  }
363  int max_card_index = *std::max_element(meld.begin(), meld.end());
364  if (CardRank(max_card_index) < num_ranks - 1) {
365  layoffs.push_back(max_card_index + 1);
366  }
367  return layoffs;
368 }

370 // Finds melds which can be layed legally given a knock card.
371 // Consider 6s7s8s, 6c7c8c, 8s8c8d. Laying 8s8c8d prevents us from using
372 // the 6's and 7's in melds, leaving us with 26 points. Laying the two suit
373 // melds leaves only the 8d for 8 points.
374 // Returns vector of meld_ids (see MeldToInt).
375 VecInt GinRummyUtils::LegalMelds(const VecInt &hand, int knock_card) const {
376  int total_hand_value = TotalCardValue(hand);
377  std::set<int> meld_set;
378  VecInt hand_(hand);
379  VecVecVecInt all_meld_groups = AllMeldGroups(hand_);
380  for (const auto &meld_group : all_meld_groups) {
381  int meld_group_total_value = TotalCardValue(meld_group);
382  if (total_hand_value - meld_group_total_value <= knock_card) {
383  for (const auto &meld : meld_group) {
384  meld_set.insert(meld_to_int.at(meld));
385  }
386  }
387  }
388  return VecInt(meld_set.begin(), meld_set.end());
389 }

391 // Returns the legal discards when a player has knocked. Normally a player can
392 // discard any card in their hand. When a player knocks, however, they must
393 // discard a card that preseves the ability to arrange the hand so that the
394 // total deadwood is less than the knock card.
395 VecInt GinRummyUtils::LegalDiscards(const VecInt &hand, int knock_card) const {
396  std::set<int> legal_discards;
397  for (int i = 0; i < hand.size(); ++i) {
398  VecInt hand_(hand);
399  hand_.erase(hand_.begin() + i);
400  int deadwood = MinDeadwood(hand_);
401  if (deadwood <= knock_card) {
402  legal_discards.insert(hand[i]);
403  }
404  }
405  return VecInt(legal_discards.begin(), legal_discards.end());
406 }

408 VecInt GinRummyUtils::AllLayoffs(const VecInt &layed_melds,
409  const VecInt &previous_layoffs) const {
410  std::set<int> layoffs;
411  for (int meld_id : layed_melds) {
412  VecInt meld = int_to_meld.at(meld_id);
413  if (IsRankMeld(meld) && meld.size() == 3) {
414  layoffs.insert(RankMeldLayoff(meld));
415  } else if (IsSuitMeld(meld)) {
416  VecInt suit_layoffs = SuitMeldLayoffs(meld);
417  for (int card : previous_layoffs) {
418  if (absl::c_linear_search(suit_layoffs, card)) {
419  meld.push_back(card);
420  }
421  }
422  suit_layoffs = SuitMeldLayoffs(meld);
423  for (int card : suit_layoffs) {
424  layoffs.insert(card);
425  }
426  }
427  }
428  return VecInt(layoffs.begin(), layoffs.end());
429 }

465 // Defines a mapping from melds to ints.
466 // There are 185 distinct melds in total, 65 rank melds and 120 suit melds.
467 // Rank melds are ordered by ascending rank. For each rank, there are 5 melds.
468 // The four melds of size 3 are ordered by the suit of the card missing from
469 // the meld (i.e. 2c2d2h precedes 2s2h2d because the 2s, missing from the first
470 // meld, precedes the 2c, missing from the second).
471 // The fifth rank meld is the unique meld containing all four cards of a
472 // given rank.
473 // Suit melds are ordered first by size, then by suit (scdh), then by rank.
474 int GinRummyUtils::MeldToInt(VecInt meld) const {
475  const int kNumRanks = 13;
476  const int kNumSuits = 4;
477  if (IsRankMeld(meld)) {
478  if (meld.size() == 3) {
479  VecInt suits;
480  for (int i = 0; i < kNumSuits; ++i) suits.push_back(i);
481  for (int card : meld) {
482  suits.erase(std::remove(suits.begin(), suits.end(), CardSuit(card)),
483  suits.end());
484  }
485  return (CardRank(meld[0]) * 5) + suits[0];
486  } else if (meld.size() == 4) {
487  return (CardRank(meld[0]) * 5) + 4;
488  }
489  SpielFatalError("Impossible meld size");
490  } else if (IsSuitMeld(meld)) {
491  absl::c_sort(meld, rank_comp);
492  int offset = 65; // 65 rank melds
493  if (meld.size() == 3) {
494  return offset + (CardSuit(meld[0]) * (kNumRanks - 2)) +
495  CardRank(meld[0]);
496  }
497  offset += 44; // 44 suit melds of size three
498  if (meld.size() == 4) {
499  return offset + (CardSuit(meld[0]) * (kNumRanks - 3)) +
500  CardRank(meld[0]);
501  }
502  offset += 40; // 40 suit melds of size four
503  if (meld.size() == 5) {
504  return offset + (CardSuit(meld[0]) * (kNumRanks - 4)) +
505  CardRank(meld[0]);
506  }
507  SpielFatalError("Impossible meld size");
508  } else {
509  SpielFatalError("Not a meld");
510  }
511 }
