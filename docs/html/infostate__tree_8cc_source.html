78 std::string InfostateNode::MakeCertificate() const {
79  if (type_ == kTerminalInfostateNode) return "{}";
80 
81  std::vector<std::string> certificates;
82  for (InfostateNode* child : child_iterator()) {
83  certificates.push_back(child->MakeCertificate());
84  }
85  std::sort(certificates.begin(), certificates.end());
86 
87  std::string open, close;
88  if (type_ == kDecisionInfostateNode) {
89  open = "[";
90  close = "]";
91  } else if (type_ == kObservationInfostateNode) {
92  open = "(";
93  close = ")";
94  }
95 
96  return absl::StrCat(
97  open, absl::StrJoin(certificates.begin(), certificates.end(), ""), close);
98 }

100 void InfostateNode::RebalanceSubtree(int target_depth, int current_depth) {
101  SPIEL_DCHECK_LE(current_depth, target_depth);
102  depth_ = current_depth;
103 
104  if (is_leaf_node() && target_depth != current_depth) {
105  // Prepare the chain of dummy observations.
106  depth_ = target_depth;
107  std::unique_ptr<InfostateNode> node = Release();
108  InfostateNode* node_parent = node->parent();
109  int position_in_leaf_parent = node->incoming_index();
110  std::unique_ptr<InfostateNode> chain_head =
111  std::unique_ptr<InfostateNode>(new InfostateNode(
112  /*tree=*/tree_, /*parent=*/nullptr,
113  /*incoming_index=*/position_in_leaf_parent,
114  kObservationInfostateNode,
115  /*infostate_string=*/kFillerInfostate,
116  /*terminal_utility=*/NAN, /*terminal_ch_reach_prob=*/NAN,
117  current_depth, /*legal_actions=*/{}, /*terminal_history=*/{}));
118  InfostateNode* chain_tail = chain_head.get();
119  for (int i = 1; i < target_depth - current_depth; ++i) {
120  chain_tail =
121  chain_tail->AddChild(std::unique_ptr<InfostateNode>(new InfostateNode(
122  /*tree=*/tree_, /*parent=*/chain_tail,
123  /*incoming_index=*/0, kObservationInfostateNode,
124  /*infostate_string=*/kFillerInfostate,
125  /*terminal_utility=*/NAN, /*terminal_ch_reach_prob=*/NAN,
126  current_depth + i, /*legal_actions=*/{},
127  /*terminal_history=*/{})));
128  }
129  chain_tail->children_.push_back(nullptr);
130 
131  // First put the node to the chain. If we did it in reverse order,
132  // i.e chain to parent and then node to the chain, the node would
133  // become freed.
134  auto* node_ptr = node.get();
135  node_ptr->SwapParent(std::move(node), /*target=*/chain_tail, 0);
136  auto* chain_head_ptr = chain_head.get();
137  chain_head_ptr->SwapParent(std::move(chain_head), /*target=*/node_parent,
138  position_in_leaf_parent);
139  }
140 
141  for (std::unique_ptr<InfostateNode>& child : children_) {
142  child->RebalanceSubtree(target_depth, current_depth + 1);
143  }
144 }

152 void InfostateNode::SwapParent(std::unique_ptr<InfostateNode> self,
153  InfostateNode* target, int at_index) {
154  // This node is still who it thinks it is :)
155  SPIEL_DCHECK_TRUE(self.get() == this);
156  target->children_.at(at_index) = std::move(self);
157  this->parent_ = target;
158  this->incoming_index_ = at_index;

161 InfostateTree::InfostateTree(const std::vector<const State*>& start_states,
162  const std::vector<double>& chance_reach_probs,
163  std::shared_ptr<Observer> infostate_observer,
164  Player acting_player, int max_move_ahead_limit)
165  : acting_player_(acting_player),
166  infostate_observer_(std::move(infostate_observer)),
167  root_(MakeRootNode()) {
168  SPIEL_CHECK_FALSE(start_states.empty());
169  SPIEL_CHECK_EQ(start_states.size(), chance_reach_probs.size());
170  SPIEL_CHECK_GE(acting_player_, 0);
171  SPIEL_CHECK_LT(acting_player_, start_states[0]->GetGame()->NumPlayers());
172  SPIEL_CHECK_TRUE(infostate_observer_->HasString());
173 
174  int start_max_move_number = 0;
175  for (const State* start_state : start_states) {
176  start_max_move_number =
177  std::max(start_max_move_number, start_state->MoveNumber());
178  }
179 
180  for (int i = 0; i < start_states.size(); ++i) {
181  RecursivelyBuildTree(root_.get(), /*depth=*/1, *start_states[i],
182  start_max_move_number + max_move_ahead_limit,
183  chance_reach_probs[i]);
184  }
185 
186  // Operations to make after building the tree.
187  RebalanceTree();
188  nodes_at_depths_.resize(tree_height() + 1);
189  CollectNodesAtDepth(mutable_root(), 0);
190  LabelNodesWithIds();
191 }

197 void InfostateTree::CollectNodesAtDepth(InfostateNode* node, size_t depth) {
198  nodes_at_depths_[depth].push_back(node);
199  for (InfostateNode* child : node->child_iterator())
200  CollectNodesAtDepth(child, depth + 1);

252 void InfostateTree::RecursivelyBuildTree(InfostateNode* parent, size_t depth,
253  const State& state, int move_limit,
254  double chance_reach_prob) {
255  if (state.IsTerminal())
256  return BuildTerminalNode(parent, depth, state, chance_reach_prob);
257  else if (state.IsPlayerActing(acting_player_))
258  return BuildDecisionNode(parent, depth, state, move_limit,
259  chance_reach_prob);
260  else
261  return BuildObservationNode(parent, depth, state, move_limit,
262  chance_reach_prob);
263 }

265 void InfostateTree::BuildTerminalNode(InfostateNode* parent, size_t depth,
266  const State& state,
267  double chance_reach_prob) {
268  const double terminal_utility = state.Returns()[acting_player_];
269  InfostateNode* terminal_node = parent->AddChild(
270  MakeNode(parent, kTerminalInfostateNode,
271  infostate_observer_->StringFrom(state, acting_player_),
272  terminal_utility, chance_reach_prob, depth, &state));
273  UpdateLeafNode(terminal_node, state, depth, chance_reach_prob);
274 }

276 void InfostateTree::BuildDecisionNode(InfostateNode* parent, size_t depth,
277  const State& state, int move_limit,
278  double chance_reach_prob) {
279  SPIEL_DCHECK_EQ(parent->type(), kObservationInfostateNode);
280  std::string info_state =
281  infostate_observer_->StringFrom(state, acting_player_);
282  InfostateNode* decision_node = parent->GetChild(info_state);
283  const bool is_leaf_node = state.MoveNumber() >= move_limit;
284 
285  if (decision_node) {
286  // The decision node has been already constructed along with children
287  // for each action: these are observation nodes.
288  // Fetches the observation child and goes deeper recursively.
289  SPIEL_DCHECK_EQ(decision_node->type(), kDecisionInfostateNode);
290 
291  if (is_leaf_node) { // Do not build deeper.
292  return UpdateLeafNode(decision_node, state, depth, chance_reach_prob);
293  }
294 
295  if (state.IsSimultaneousNode()) {
296  const ActionView action_view(state);
297  for (int i = 0; i < action_view.legal_actions[acting_player_].size();
298  ++i) {
299  InfostateNode* observation_node = decision_node->child_at(i);
300  SPIEL_DCHECK_EQ(observation_node->type(), kObservationInfostateNode);
301 
302  for (Action flat_actions :
303  action_view.fixed_action(acting_player_, i)) {
304  std::unique_ptr<State> child = state.Child(flat_actions);
305  RecursivelyBuildTree(observation_node, depth + 2, *child, move_limit,
306  chance_reach_prob);
307  }
308  }
309  } else {
310  std::vector<Action> legal_actions = state.LegalActions(acting_player_);
311  for (int i = 0; i < legal_actions.size(); ++i) {
312  InfostateNode* observation_node = decision_node->child_at(i);
313  SPIEL_DCHECK_EQ(observation_node->type(), kObservationInfostateNode);
314  std::unique_ptr<State> child = state.Child(legal_actions.at(i));
315  RecursivelyBuildTree(observation_node, depth + 2, *child, move_limit,
316  chance_reach_prob);
317  }
318  }
319  } else { // The decision node was not found yet.
320  decision_node = parent->AddChild(MakeNode(
321  parent, kDecisionInfostateNode, info_state,
322  /*terminal_utility=*/NAN, /*chance_reach_prob=*/NAN, depth, &state));
323 
324  if (is_leaf_node) { // Do not build deeper.
325  return UpdateLeafNode(decision_node, state, depth, chance_reach_prob);
326  }
327 
328  // Build observation nodes right away after the decision node.
329  // This is because the player might be acting multiple times in a row:
330  // each time it might get some observations that branch the infostate
331  // tree.
332 
333  if (state.IsSimultaneousNode()) {
334  ActionView action_view(state);
335  for (int i = 0; i < action_view.legal_actions[acting_player_].size();
336  ++i) {
337  // We build a dummy observation node.
338  // We can't ask for a proper infostate string or an originating state,
339  // because such a thing is not properly defined after only a partial
340  // application of actions for the sim move state
341  // (We need to supply all the actions).
342  InfostateNode* observation_node = decision_node->AddChild(
343  MakeNode(decision_node, kObservationInfostateNode,
344  /*infostate_string=*/kFillerInfostate,
345  /*terminal_utility=*/NAN, /*chance_reach_prob=*/NAN, depth,
346  /*originating_state=*/nullptr));
347 
348  for (Action flat_actions :
349  action_view.fixed_action(acting_player_, i)) {
350  // Only now we can advance the state, when we have all actions.
351  std::unique_ptr<State> child = state.Child(flat_actions);
352  RecursivelyBuildTree(observation_node, depth + 2, *child, move_limit,
353  chance_reach_prob);
354  }
355  }
356  } else { // Not a sim move node.
357  for (Action a : state.LegalActions()) {
358  std::unique_ptr<State> child = state.Child(a);
359  InfostateNode* observation_node = decision_node->AddChild(
360  MakeNode(decision_node, kObservationInfostateNode,
361  infostate_observer_->StringFrom(*child, acting_player_),
362  /*terminal_utility=*/NAN, /*chance_reach_prob=*/NAN, depth,
363  child.get()));
364  RecursivelyBuildTree(observation_node, depth + 2, *child, move_limit,
365  chance_reach_prob);
366  }
367  }
368  }
369 }

371 void InfostateTree::BuildObservationNode(InfostateNode* parent, size_t depth,
372  const State& state, int move_limit,
373  double chance_reach_prob) {
374  SPIEL_DCHECK_TRUE(state.IsChanceNode() ||
375  !state.IsPlayerActing(acting_player_));
376  const bool is_leaf_node = state.MoveNumber() >= move_limit;
377  const std::string info_state =
378  infostate_observer_->StringFrom(state, acting_player_);
379 
380  InfostateNode* observation_node = parent->GetChild(info_state);
381  if (!observation_node) {
382  observation_node = parent->AddChild(MakeNode(
383  parent, kObservationInfostateNode, info_state,
384  /*terminal_utility=*/NAN, /*chance_reach_prob=*/NAN, depth, &state));
385  }
386  SPIEL_DCHECK_EQ(observation_node->type(), kObservationInfostateNode);
387 
388  if (is_leaf_node) { // Do not build deeper.
389  return UpdateLeafNode(observation_node, state, depth, chance_reach_prob);
390  }
391 
392  if (state.IsChanceNode()) {
393  for (std::pair<Action, double> action_prob : state.ChanceOutcomes()) {
394  std::unique_ptr<State> child = state.Child(action_prob.first);
395  RecursivelyBuildTree(observation_node, depth + 1, *child, move_limit,
396  chance_reach_prob * action_prob.second);
397  }
398  } else {
399  for (Action a : state.LegalActions()) {
400  std::unique_ptr<State> child = state.Child(a);
401  RecursivelyBuildTree(observation_node, depth + 1, *child, move_limit,
402  chance_reach_prob);
403  }
404  }

427 std::shared_ptr<InfostateTree> MakeInfostateTree(
428  const std::vector<const InfostateNode*>& start_nodes,
429  int max_move_ahead_limit) {
430  SPIEL_CHECK_FALSE(start_nodes.empty());
431  const InfostateNode* some_node = start_nodes[0];
432  const InfostateTree& originating_tree = some_node->tree();
433  SPIEL_DCHECK_TRUE([&]() {
434  for (const InfostateNode* node : start_nodes) {
435  if (!node) return false;
436  if (!node->is_leaf_node()) return false;
437  if (node->depth() != some_node->depth()) return false;
438  if (&node->tree() != &originating_tree) return false;
439  }
440  return true;
441  }());
442 
443  // We reserve a larger number of states, as infostate nodes typically contain
444  // a large number of States. (8 is an arbitrary choice though).
445  std::vector<const State*> start_states;
446  start_states.reserve(start_nodes.size() * 8);
447  std::vector<double> chance_reach_probs;
448  chance_reach_probs.reserve(start_nodes.size() * 8);
449 
450  for (const InfostateNode* node : start_nodes) {
451  for (int i = 0; i < node->corresponding_states_size(); ++i) {
452  start_states.push_back(node->corresponding_states()[i].get());
453  chance_reach_probs.push_back(node->corresponding_chance_reach_probs()[i]);
454  }
455  }
456 
457  // Uses new instead of make_shared, because shared_ptr is not a friend and
458  // can't call private constructors.
459  return std::shared_ptr<InfostateTree>(new InfostateTree(
460  start_states, chance_reach_probs, originating_tree.infostate_observer_,
461  originating_tree.acting_player_, max_move_ahead_limit));
462 }

504 std::vector<DecisionId> InfostateTree::DecisionIdsWithParentSeq(
505  const SequenceId& sequence_id) const {
506  std::vector<DecisionId> out;
507  const InfostateNode* observation_node = sequences_.at(sequence_id.id());
508  std::stack<const InfostateNode*> open_set;
509  for (const InfostateNode* child : observation_node->child_iterator()) {
510  open_set.push(child);
511  }
512  while (!open_set.empty()) {
513  const InfostateNode* node = open_set.top();
514  open_set.pop();
515  if (node->type() == kDecisionInfostateNode &&
516  node->sequence_id() == sequence_id) {
517  out.push_back(node->decision_id());
518  } else {
519  for (const InfostateNode* child : node->child_iterator()) {
520  open_set.push(child);
521  }
522  }
523  }
524  return out;
525 }

527 void InfostateTree::LabelNodesWithIds() {
528  // Idea of labeling: label the leaf sequences first, and continue up the tree.
529  size_t sequence_index = 0;
530  size_t decision_index = 0;
531 
532  // Do not label leaf nodes with sequences.
533  const int start_depth = nodes_at_depths_.size() - 2;
534 
535  for (int depth = start_depth; depth >= 0; --depth) {
536  for (InfostateNode* node : nodes_at_depths_[depth]) {
537  if (node->type() != kDecisionInfostateNode) continue;
538  decision_infostates_.push_back(node);
539  node->decision_id_ = DecisionId(decision_index++, this);
540 
541  for (InfostateNode* child : node->child_iterator()) {
542  sequences_.push_back(child);
543  child->sequence_id_ = SequenceId(sequence_index++, this);
544  }
545  // We could use sequence_index to set start and end sequences for
546  // the decision infostate right away here, however we'd like to make
547  // sure to label correctly all nodes in the tree.
548  }
549  }
550  // Finally label the last sequence (an empty sequence) in the root node.
551  sequences_.push_back(mutable_root());
552  mutable_root()->sequence_id_ = SequenceId(sequence_index, this);
553 
554  CollectStartEndSequenceIds(mutable_root(), mutable_root()->sequence_id());
555 }

557 // Make a recursive call to assign the parent's sequences appropriately.
558 // Collect pairs of (start, end) sequence ids from children and propagate
559 // them up the tree. In case that deep nodes (close to the leaves) do not
560 // have any child decision nodes, set the (start, end) to the parent sequence.
561 // In this way the range iterator will be empty (start==end) and well defined.
562 std::pair<size_t, size_t> InfostateTree::CollectStartEndSequenceIds(
563  InfostateNode* node, const SequenceId parent_sequence) {
564  size_t min_index = kUndefinedNodeId; // This is a large number.
565  size_t max_index = 0;
566  const SequenceId propagate_sequence_id =
567  node->sequence_id_.is_undefined()
568  ? parent_sequence
569  : node->sequence_id(); // This becomes the parent for next nodes.
570 
571  for (InfostateNode* child : node->child_iterator()) {
572  auto [min_child, max_child] =
573  CollectStartEndSequenceIds(child, propagate_sequence_id);
574  min_index = std::min(min_child, min_index);
575  max_index = std::max(max_child, max_index);
576  }
577 
578  if (min_index != kUndefinedNodeId) {
579  SPIEL_CHECK_LE(min_index, max_index);
580  node->start_sequence_id_ = SequenceId(min_index, this);
581  node->end_sequence_id_ = SequenceId(max_index + 1, this);
582  } else {
583  node->start_sequence_id_ = propagate_sequence_id;
584  node->end_sequence_id_ = propagate_sequence_id;
585  }
586 
587  if (node->sequence_id_.is_undefined()) {
588  // Propagate children limits.
589  node->sequence_id_ = parent_sequence;
590  return {min_index, max_index};
591  } else {
592  // We have hit a defined sequence id, propagate it up.
593  return {node->sequence_id_.id(), node->sequence_id_.id()};
594  }
595 }

597 std::pair<double, SfStrategy> InfostateTree::BestResponse(
598  TreeplexVector<double>&& gradient) const {
599  SPIEL_CHECK_EQ(this, gradient.tree());
600  SPIEL_CHECK_EQ(num_sequences(), gradient.size());
601  SfStrategy response(this);
602 
603  // 1. Compute counterfactual best response
604  // (i.e. in all infostates, even unreachable ones)
605  SequenceId current(0, this);
606  const double init_value = -std::numeric_limits<double>::infinity();
607  while (current.id() <= empty_sequence().id()) {
608  double max_value = init_value;
609  SequenceId max_id = current;
610  const InfostateNode* node = observation_infostate(current);
611  for (current = node->start_sequence_id();
612  current != node->end_sequence_id(); current.next()) {
613  if (gradient[current] > max_value) {
614  max_value = gradient[current];
615  max_id = current;
616  }
617  }
618  if (init_value != max_value) {
619  gradient[node->sequence_id()] += max_value;
620  response[max_id] = 1.;
621  }
622  current.next();
623  }
624  SPIEL_CHECK_EQ(current.id(), empty_sequence().id() + 1);
625 
626  // 2. Prune away unreachable subtrees.
627  //
628  // This can be done with a more costly recursion.
629  // Instead we make a more cache-friendly double pass through the response
630  // vector: we increment the visited path by 1, resulting in a value of 2.
631  // Then we zero-out all values but 2.
632  current = empty_sequence();
633  response[current] = 2.;
634  while (!IsLeafSequence(current)) {
635  for (SequenceId seq : observation_infostate(current)->AllSequenceIds()) {
636  if (response[seq] == 1.) {
637  current = seq;
638  response[seq] += 1.;
639  break;
640  }
641  }
642  }
643  for (SequenceId seq : response.range()) {
644  response[seq] = response[seq] == 2. ? 1. : 0.;
645  }
646  SPIEL_DCHECK_TRUE(IsValidSfStrategy(response));
647  return {gradient[empty_sequence()], response};
648 }

650 double InfostateTree::BestResponseValue(LeafVector<double>&& gradient) const {
651  // Loop over all heights.
652  for (int d = tree_height_ - 1; d >= 0; d--) {
653  int left_offset = 0;
654  // Loop over all parents of current nodes.
655  for (int parent_idx = 0; parent_idx < nodes_at_depths_[d].size();
656  parent_idx++) {
657  const InfostateNode* node = nodes_at_depths_[d][parent_idx];
658  const int num_children = node->num_children();
659  const Range<LeafId> children_range =
660  gradient.range(left_offset, left_offset + num_children);
661  const LeafId parent_id(parent_idx, this);
662 
663  if (node->type() == kDecisionInfostateNode) {
664  double max_value = std::numeric_limits<double>::min();
665  for (LeafId id : children_range) {
666  max_value = std::fmax(max_value, gradient[id]);
667  }
668  gradient[parent_id] = max_value;
669  } else {
670  SPIEL_DCHECK_EQ(node->type(), kObservationInfostateNode);
671  double sum_value = 0.;
672  for (LeafId id : children_range) {
673  sum_value += gradient[id];
674  }
675  gradient[parent_id] = sum_value;
676  }
677  left_offset += num_children;
678  }
679  // Check that we passed over all of the children.
680  SPIEL_DCHECK_EQ(left_offset, nodes_at_depths_[d + 1].size());
681  }
682  const LeafId root_id(0, this);
683  return gradient[root_id];
684 }

695 bool CheckSum(const SfStrategy& strategy, SequenceId id, double expected_sum) {
696  if (fabs(strategy[id] - expected_sum) > 1e-13) {
697  return false;
698  }
699 
700  const InfostateTree* tree = strategy.tree();
701  if (tree->IsLeafSequence(id)) {
702  return true;
703  }
704 
705  double actual_sum = 0.;
706  const InfostateNode* node = tree->observation_infostate(id);
707  for (SequenceId sub_seq : node->AllSequenceIds()) {
708  actual_sum += strategy[sub_seq];
709  }
710  if (fabs(actual_sum - expected_sum) > 1e-13) {
711  return false;
712  }
713 
714  for (SequenceId sub_seq : node->AllSequenceIds()) {
715  if (!CheckSum(strategy, sub_seq, strategy[sub_seq])) {
716  return false;
717  }
718  }
719  return true;
720 }
