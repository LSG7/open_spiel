66std::string EFCEState::InformationStateString(Player player) const {
67 // should look like <infoset string> <delimiter> <recommendation sequence,
68 // (including current recommendation)>
69 SPIEL_CHECK_FALSE(IsChanceNode());
70 std::string rec_str = absl::StrJoin(recommendation_seq_[player], ",");
71 if (!HasDefected(player)) {
72 absl::StrAppend(&rec_str, ",", CurRecommendation());
73 }
74 std::string infoset_str = state_->InformationStateString(player);
75 SPIEL_CHECK_EQ(infoset_str.find(config_.recommendation_delimiter),
76 std::string::npos);
77 return absl::StrCat(infoset_str, config_.recommendation_delimiter, rec_str);
78}

97Action EFCEState::CurRecommendation() const {
98 SPIEL_CHECK_GE(rec_index_, 0);
99 SPIEL_CHECK_LT(rec_index_, mu_.size());
100 ActionsAndProbs actions_and_probs =
101 mu_[rec_index_].second.GetStatePolicy(state_->InformationStateString());
102 Action rec_action = GetAction(actions_and_probs);
103 SPIEL_CHECK_TRUE(rec_action != kInvalidAction);
104 return rec_action;
105}

107void EFCEState::DoApplyAction(Action action_id) {
108 if (rec_index_ < 0) {
109 // Pick the joint policy which will provide recommendations.
110 rec_index_ = action_id;
111 SPIEL_CHECK_GE(rec_index_, 0);
112 SPIEL_CHECK_LT(rec_index_, mu_.size());
113 } else if (state_->IsChanceNode()) {
114 // Regular chance node
115 state_->ApplyAction(action_id);
116 } else {
117 // Check for defection at this point.
118 const Action recommendation = CurRecommendation();
119
120 Player cur_player = CurrentPlayer();
121 SPIEL_CHECK_GE(cur_player, 0);
122 SPIEL_CHECK_LT(cur_player, game_->NumPlayers());
123
124 // If they have defected, don't add to the sequence
125 if (!HasDefected(cur_player)) {
126 recommendation_seq_[cur_player].push_back(recommendation);
127
128 // If they chose an action other than the recommendation, they have now
129 // defected.
130 if (action_id != recommendation) {
131 defected_[cur_player] = 1;
132 }
133 }
134
135 state_->ApplyAction(action_id);
136 }
137}

139ActionsAndProbs EFCETabularPolicy::GetStatePolicy(const State& state) const {
140 // The best response code has to have a policy defined everywhere when it
141 // builds its initial tree. For the fixed policies, the players will not
142 // defect, so we define a uniform policy in the regions where players have
143 // defected (which will not affect the best responding player, since the
144 // opponents will never reach these regions).
145 const auto* efce_state = dynamic_cast<const EFCEState*>(&state);
146 SPIEL_CHECK_TRUE(efce_state != nullptr);
147 if (efce_state->HasDefected(state.CurrentPlayer())) {
148 return UniformStatePolicy(state);
149 }
150
151 // Otherwise, simply returns a fixed policy with prob 1 on the recommended
152 // action (extrapolated from the information state string) and 0 on the
153 // others.
154 std::string info_state = state.InformationStateString();
155 const size_t idx = info_state.find(config_.recommendation_delimiter);
156 SPIEL_CHECK_NE(idx, std::string::npos);
157 std::vector<std::string> rec_seq = absl::StrSplit(
158 info_state.substr(idx + config_.recommendation_delimiter.length()), ',');
159 SPIEL_CHECK_GE(rec_seq.size(), 1);
160 Action rec_action;
161 ActionsAndProbs state_policy;
162 std::vector<Action> legal_actions = state.LegalActions();
163 state_policy.reserve(legal_actions.size());
164 SPIEL_CHECK_TRUE(absl::SimpleAtoi(rec_seq.back(), &rec_action));
165 for (Action action : legal_actions) {
166 state_policy.push_back({action, action == rec_action ? 1.0 : 0.0});
167 }
168 return state_policy;
169}
