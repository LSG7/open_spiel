15"""Regression counterfactual regret minimization (RCFR) [Waugh et al., 2015; Morrill, 2016].
16
17In contrast to (tabular) counterfactual regret minimization (CFR)
18[Zinkevich et al., 2007], RCFR replaces the table of regrets that generate the
19current policy profile with a profile of regression models. The average
20policy is still tracked exactly with a full game-size table. The exploitability
21of the average policy in zero-sum games decreases as the model accuracy and
22the number of iterations increase [Waugh et al., 2015; Morrill, 2016]. As long
23as the regression model errors decrease across iterations, the average policy
24converges toward a Nash equilibrium in zero-sum games.
25
26# References
27
28Dustin Morrill. Using Regret Estimation to Solve Games Compactly.
29 M.Sc. thesis, Computing Science Department, University of Alberta,
30 Apr 1, 2016, Edmonton Alberta, Canada.
31Kevin Waugh, Dustin Morrill, J. Andrew Bagnell, and Michael Bowling.
32 Solving Games with Functional Regret Estimation. At the Twenty-Ninth AAAI
33 Conference on Artificial Intelligence, January 25-29, 2015, Austin Texas,
34 USA. Pages 2138-2145.
35Martin Zinkevich, Michael Johanson, Michael Bowling, and Carmelo Piccione.
36 Regret Minimization in Games with Incomplete Information.
37 At Advances in Neural Information Processing Systems 20 (NeurIPS). 2007.
38"""

57def tensor_to_matrix(tensor):
58 """Converts `tensor` to a matrix (a rank-2 tensor) or raises an exception.
59
60 Args:
61 tensor: The tensor to convert.
62
63 Returns:
64 A TensorFlow matrix (rank-2 `tf.Tensor`).
65 Raises:
66 ValueError: If `tensor` cannot be trivially converted to a matrix, i.e.
67 `tensor` has a rank > 2.
68 """
69 tensor = tf.convert_to_tensor(tensor)
70 rank = tensor.shape.rank
71 if rank > 2:
72 raise ValueError(
73 ("Tensor {} cannot be converted into a matrix as it is rank "
74 "{} > 2.").format(tensor, rank))
75 elif rank < 2:
76 num_columns = 1 if rank == 0 else tensor.shape[0].value
77 tensor = tf.reshape(tensor, [1, num_columns])
78 return tensor

81def with_one_hot_action_features(state_features, legal_actions,
82 num_distinct_actions):
83 """Constructs features for each sequence by extending state features.
84
85 Sequences features are constructed by concatenating one-hot features
86 indicating each action to the information state features and stacking them.
87
88 Args:
89 state_features: The features for the information state alone. Must be a
90 `tf.Tensor` with a rank less than or equal to (if batched) 2.
91 legal_actions: The list of legal actions in this state. Determines the
92 number of rows in the returned feature matrix.
93 num_distinct_actions: The number of globally distinct actions in the game.
94 Determines the length of the action feature vector concatenated onto the
95 state features.
96
97 Returns:
98 A `tf.Tensor` feature matrix with one row for each sequence and # state
99 features plus `num_distinct_actions`-columns.
100
101 Raises:
102 ValueError: If `state_features` has a rank > 2.
103 """
104 state_features = tensor_to_matrix(state_features)
105 with_action_features = []
106 for action in legal_actions:
107 action_features = tf.one_hot([action], num_distinct_actions)
108 action_features = tf.tile(action_features, [tf.shape(state_features)[0], 1])
109 all_features = tf.concat([state_features, action_features], axis=1)
110 with_action_features.append(all_features)
111 return tf.concat(with_action_features, axis=0)

143class RootStateWrapper(object):
144 """Analyzes the subgame at a given root state.
145
146 It enumerates features for each player sequence, creates a mapping between
147 information states to sequence index offsets, and caches terminal values
148 in a dictionary with history string keys.
149
150 Properties:
151 root: An OpenSpiel `State`.
152 sequence_features: A `list` of sequence feature matrices, one for each
153 player. This list uses depth-first, information state-major ordering, so
154 sequences are grouped by information state. I.e. the first legal action
155 in the first state has index 0, the second action in the same information
156 state has index 1, the third action will have index 3, and so on.
157 Sequences in the next information state descendant of the first action
158 will begin indexing its sequences at the number of legal actions in the
159 ancestor information state.
160 num_player_sequences: The number of sequences for each player.
161 info_state_to_sequence_idx: A `dict` mapping each information state string
162 to the `sequence_features` index of the first sequence in the
163 corresponding information state.
164 terminal_values: A `dict` mapping history strings to terminal values for
165 each player.
166 """
167
168 def __init__(self, state):
169 self.root = state
170 self._num_distinct_actions = len(state.legal_actions_mask(0))
171
172 self.sequence_features = [[] for _ in range(state.num_players())]
173 self.num_player_sequences = [0] * state.num_players()
174 self.info_state_to_sequence_idx = {}
175 self.terminal_values = {}
176 self._walk_descendants(state)
177 self.sequence_features = [
178 tf.concat(rows, axis=0) for rows in self.sequence_features
179 ]
180
181 def _walk_descendants(self, state):
182 """Records information about `state` and its descendants."""
183 if state.is_terminal():
184 self.terminal_values[state.history_str()] = np.array(state.returns())
185 return
186
187 elif state.is_chance_node():
188 for action, _ in state.chance_outcomes():
189 self._walk_descendants(state.child(action))
190 return
191
192 player = state.current_player()
193 info_state = state.information_state_string(player)
194 actions = state.legal_actions()
195
196 if info_state not in self.info_state_to_sequence_idx:
197 n = self.num_player_sequences[player]
198 self.info_state_to_sequence_idx[info_state] = n
199 self.sequence_features[player].append(
200 sequence_features(state, self._num_distinct_actions))
201 self.num_player_sequences[player] += len(actions)
202
203 for action in actions:
204 self._walk_descendants(state.child(action))

206 def sequence_weights_to_policy(self, sequence_weights, state):
207 """Returns a behavioral policy at `state` from sequence weights.
208
209 Args:
210 sequence_weights: An array of non-negative weights, one for each of
211 `state.current_player()`'s sequences in `state`'s game.
212 state: An OpenSpiel `State` that represents an information state in an
213 alternating-move game.
214
215 Returns:
216 A `np.array<double>` probability distribution representing the policy in
217 `state` encoded by `sequence_weights`. Weights corresponding to actions
218 in `state` are normalized by their sum.
219
220 Raises:
221 ValueError: If there are too few sequence weights at `state`.
222 """
223 info_state = state.information_state_string()
224 sequence_offset = self.info_state_to_sequence_idx[info_state]
225 actions = state.legal_actions()
226
227 sequence_idx_end = sequence_offset + len(actions)
228 weights = sequence_weights[sequence_offset:sequence_idx_end]
229
230 if len(weights) < len(actions):
231 raise ValueError(
232 ("Invalid policy: Policy {player} at sequence offset "
233 "{sequence_offset} has only {policy_len} elements but there "
234 "are {num_actions} legal actions.").format(
235 player=state.current_player(),
236 sequence_offset=sequence_offset,
237 policy_len=len(weights),
238 num_actions=len(actions)))
239 return normalized_by_sum(weights)
240
241 def sequence_weights_to_policy_fn(self, player_sequence_weights):
242 """Returns a policy function based on sequence weights for each player.
243
244 Args:
245 player_sequence_weights: A list of weight arrays, one for each player.
246 Each array should have a weight for each of that player's sequences in
247 `state`'s game.
248
249 Returns:
250 A `State` -> `np.array<double>` function. The output of this function is
251 a probability distribution that represents the policy at the given
252 `State` encoded by `player_sequence_weights` according to
253 `sequence_weights_to_policy`.
254 """
255
256 def policy_fn(state):
257 player = state.current_player()
258 return self.sequence_weights_to_policy(player_sequence_weights[player],
259 state)
260
261 return policy_fn

268 def counterfactual_regrets_and_reach_weights(self, regret_player,
269 reach_weight_player,
270 *sequence_weights):
271 """Returns counterfactual regrets and reach weights as a tuple.
272
273 Args:
274 regret_player: The player for whom counterfactual regrets are computed.
275 reach_weight_player: The player for whom reach weights are computed.
276 *sequence_weights: A list of non-negative sequence weights for each player
277 determining the policy profile. Behavioral policies are generated by
278 normalizing sequence weights corresponding to actions in each
279 information state by their sum.
280
281 Returns:
282 The counterfactual regrets and reach weights as an `np.array`-`np.array`
283 tuple.
284
285 Raises:
286 ValueError: If there are too few sequence weights at any information state
287 for any player.
288 """
289 num_players = len(sequence_weights)
290 regrets = np.zeros(self.num_player_sequences[regret_player])
291 reach_weights = np.zeros(self.num_player_sequences[reach_weight_player])
292
293 def _walk_descendants(state, reach_probabilities, chance_reach_probability):
294 """Compute `state`'s counterfactual regrets and reach weights.
295
296 Args:
297 state: An OpenSpiel `State`.
298 reach_probabilities: The probability that each player plays to reach
299 `state`'s history.
300 chance_reach_probability: The probability that all chance outcomes in
301 `state`'s history occur.
302
303 Returns:
304 The counterfactual value of `state`'s history.
305 Raises:
306 ValueError if there are too few sequence weights at any information
307 state for any player.
308 """
309
310 if state.is_terminal():
311 player_reach = (
312 np.prod(reach_probabilities[:regret_player]) *
313 np.prod(reach_probabilities[regret_player + 1:]))
314
315 counterfactual_reach_prob = player_reach * chance_reach_probability
316 u = self.terminal_values[state.history_str()]
317 return u[regret_player] * counterfactual_reach_prob
318
319 elif state.is_chance_node():
320 v = 0.0
321 for action, action_prob in state.chance_outcomes():
322 v += _walk_descendants(
323 state.child(action), reach_probabilities,
324 chance_reach_probability * action_prob)
325 return v
326
327 player = state.current_player()
328 info_state = state.information_state_string(player)
329 sequence_idx_offset = self.info_state_to_sequence_idx[info_state]
330 actions = state.legal_actions(player)
331
332 sequence_idx_end = sequence_idx_offset + len(actions)
333 my_sequence_weights = sequence_weights[player][
334 sequence_idx_offset:sequence_idx_end]
335
336 if len(my_sequence_weights) < len(actions):
337 raise ValueError(
338 ("Invalid policy: Policy {player} at sequence offset "
339 "{sequence_idx_offset} has only {policy_len} elements but there "
340 "are {num_actions} legal actions.").format(
341 player=player,
342 sequence_idx_offset=sequence_idx_offset,
343 policy_len=len(my_sequence_weights),
344 num_actions=len(actions)))
345
346 policy = normalized_by_sum(my_sequence_weights)
347 action_values = np.zeros(len(actions))
348 state_value = 0.0
349
350 is_reach_weight_player_node = player == reach_weight_player
351 is_regret_player_node = player == regret_player
352
353 reach_prob = reach_probabilities[player]
354 for action_idx, action in enumerate(actions):
355 action_prob = policy[action_idx]
356 next_reach_prob = reach_prob * action_prob
357
358 if is_reach_weight_player_node:
359 reach_weight_player_plays_down_this_line = next_reach_prob > 0
360 if not reach_weight_player_plays_down_this_line:
361 continue
362 sequence_idx = sequence_idx_offset + action_idx
363 reach_weights[sequence_idx] += next_reach_prob
364
365 reach_probabilities[player] = next_reach_prob
366
367 action_value = _walk_descendants(
368 state.child(action), reach_probabilities, chance_reach_probability)
369
370 if is_regret_player_node:
371 state_value = state_value + action_prob * action_value
372 else:
373 state_value = state_value + action_value
374 action_values[action_idx] = action_value
375
376 reach_probabilities[player] = reach_prob
377
378 if is_regret_player_node:
379 regrets[sequence_idx_offset:sequence_idx_end] += (
380 action_values - state_value)
381 return state_value
382
383 # End of _walk_descendants
384
385 _walk_descendants(self.root, np.ones(num_players), 1.0)
386 return regrets, reach_weights

389def normalized_by_sum(v, axis=0, mutate=False):
390 """Divides each element of `v` along `axis` by the sum of `v` along `axis`.
391
392 Assumes `v` is non-negative. Sets of `v` elements along `axis` that sum to
393 zero are normalized to `1 / v.shape[axis]` (a uniform distribution).
394
395 Args:
396 v: Non-negative array of values.
397 axis: An integer axis.
398 mutate: Whether or not to store the result in `v`.
399
400 Returns:
401 The normalized array.
402 """
403 v = np.asarray(v)
404 denominator = v.sum(axis=axis, keepdims=True)
405 denominator_is_zero = denominator == 0
406
407 # Every element of `denominator_is_zero` that is true corresponds to a
408 # set of elements in `v` along `axis` that are all zero. By setting these
409 # denominators to `v.shape[axis]` and adding 1 to each of the corresponding
410 # elements in `v`, these elements are normalized to `1 / v.shape[axis]`
411 # (a uniform distribution).
412 denominator += v.shape[axis] * denominator_is_zero
413 if mutate:
414 v += denominator_is_zero
415 v /= denominator
416 else:
417 v = (v + denominator_is_zero) / denominator
418 return v

421def relu(v):
422 """Returns the element-wise maximum between `v` and 0."""
423 return np.maximum(v, 0)

426def _descendant_states(state, depth_limit, depth, include_terminals,
427 include_chance_states):
428 """Recursive descendant state generator.
429
430 Decision states are always yielded.
431
432 Args:
433 state: The current state.
434 depth_limit: The descendant depth limit. Zero will ensure only
435 `initial_state` is generated and negative numbers specify the absence of a
436 limit.
437 depth: The current descendant depth.
438 include_terminals: Whether or not to include terminal states.
439 include_chance_states: Whether or not to include chance states.
440
441 Yields:
442 `State`, a state that is `initial_state` or one of its descendants.
443 """
444 if state.is_terminal():
445 if include_terminals:
446 yield state
447 return
448
449 if depth > depth_limit >= 0:
450 return
451
452 if not state.is_chance_node() or include_chance_states:
453 yield state
454
455 for action in state.legal_actions():
456 state_for_search = state.child(action)
457 for substate in _descendant_states(state_for_search, depth_limit, depth + 1,
458 include_terminals,
459 include_chance_states):
460 yield substate

463def all_states(initial_state,
464 depth_limit=-1,
465 include_terminals=False,
466 include_chance_states=False):
467 """Generates states from `initial_state`.
468
469 Generates the set of states that includes only the `initial_state` and its
470 descendants that satisfy the inclusion criteria specified by the remaining
471 parameters. Decision states are always included.
472
473 Args:
474 initial_state: The initial state from which to generate states.
475 depth_limit: The descendant depth limit. Zero will ensure only
476 `initial_state` is generated and negative numbers specify the absence of a
477 limit. Defaults to no limit.
478 include_terminals: Whether or not to include terminal states. Defaults to
479 `False`.
480 include_chance_states: Whether or not to include chance states. Defaults to
481 `False`.
482
483 Returns:
484 A generator that yields the `initial_state` and its descendants that
485 satisfy the inclusion criteria specified by the remaining parameters.
486 """
487 return _descendant_states(
488 state=initial_state,
489 depth_limit=depth_limit,
490 depth=0,
491 include_terminals=include_terminals,
492 include_chance_states=include_chance_states)

495def sequence_weights_to_tabular_profile(root, policy_fn):
496 """Returns the `dict` of `list`s of action-prob pairs-form of `policy_fn`."""
497 tabular_policy = {}
498 players = range(root.num_players())
499 for state in all_states(root):
500 for player in players:
501 legal_actions = state.legal_actions(player)
502 if len(legal_actions) < 1:
503 continue
504 info_state = state.information_state_string(player)
505 if info_state in tabular_policy:
506 continue
507 my_policy = policy_fn(state)
508 tabular_policy[info_state] = list(zip(legal_actions, my_policy))
509 return tabular_policy

512@tf.function
513def feedforward_evaluate(layers,
514 x,
515 use_skip_connections=False,
516 hidden_are_factored=False):
517 """Evaluates `layers` as a feedforward neural network on `x`.
518
519 Args:
520 layers: The neural network layers (`tf.Tensor` -> `tf.Tensor` callables).
521 x: The array-like input to evaluate. Must be trivially convertible to a
522 matrix (tensor rank <= 2).
523 use_skip_connections: Whether or not to use skip connections between layers.
524 If the layer input has too few features to be added to the layer output,
525 then the end of input is padded with zeros. If it has too many features,
526 then the input is truncated.
527 hidden_are_factored: Whether or not hidden logical layers are factored into
528 two separate linear transformations stored as adjacent elements of
529 `layers`.
530
531 Returns:
532 The `tf.Tensor` evaluation result.
533
534 Raises:
535 ValueError: If `x` has a rank greater than 2.
536 """
537 x = tensor_to_matrix(x)
538 i = 0
539 while i < len(layers) - 1:
540 y = layers[i](x)
541 i += 1
542 if hidden_are_factored:
543 y = layers[i](y)
544 i += 1
545 if use_skip_connections:
546 my_num_features = x.shape[1].value
547 padding = y.shape[1].value - my_num_features
548 if padding > 0:
549 zeros = tf.zeros([tf.shape(x)[0], padding])
550 x = tf.concat([x, zeros], axis=1)
551 elif padding < 0:
552 x = tf.strided_slice(x, [0, 0], [tf.shape(x)[0], y.shape[1].value])
553 y = x + y
554 x = y
555 return layers[-1](x)

558class DeepRcfrModel(object):
559 """A flexible deep feedforward RCFR model class.
560
561 Properties:
562 layers: The `tf.keras.Layer` layers describing this model.
563 trainable_variables: The trainable `tf.Variable`s in this model's `layers`.
564 losses: This model's layer specific losses (e.g. regularizers).
565 """
566
567 def __init__(self,
568 game,
569 num_hidden_units,
570 num_hidden_layers=1,
571 num_hidden_factors=0,
572 hidden_activation=tf.nn.relu,
573 use_skip_connections=False,
574 regularizer=None):
575 """Creates a new `DeepRcfrModel.
576
577 Args:
578 game: The OpenSpiel game being solved.
579 num_hidden_units: The number of units in each hidden layer.
580 num_hidden_layers: The number of hidden layers. Defaults to 1.
581 num_hidden_factors: The number of hidden factors or the matrix rank of the
582 layer. If greater than zero, hidden layers will be split into two
583 separate linear transformations, the first with
584 `num_hidden_factors`-columns and the second with
585 `num_hidden_units`-columns. The result is that the logical hidden layer
586 is a rank-`num_hidden_units` matrix instead of a rank-`num_hidden_units`
587 matrix. When `num_hidden_units < num_hidden_units`, this is effectively
588 implements weight sharing. Defaults to 0.
589 hidden_activation: The activation function to apply over hidden layers.
590 Defaults to `tf.nn.relu`.
591 use_skip_connections: Whether or not to apply skip connections (layer
592 output = layer(x) + x) on hidden layers. Zero padding or truncation is
593 used to match the number of columns on layer inputs and outputs.
594 regularizer: A regularizer to apply to each layer. Defaults to `None`.
595 """
596 self._use_skip_connections = use_skip_connections
597 self._hidden_are_factored = num_hidden_factors > 0
598
599 self.layers = []
600 for _ in range(num_hidden_layers):
601 if self._hidden_are_factored:
602 self.layers.append(
603 tf.keras.layers.Dense(
604 num_hidden_factors,
605 use_bias=True,
606 kernel_regularizer=regularizer))
607
608 self.layers.append(
609 tf.keras.layers.Dense(
610 num_hidden_units,
611 use_bias=True,
612 activation=hidden_activation,
613 kernel_regularizer=regularizer))
614
615 self.layers.append(
616 tf.keras.layers.Dense(1, use_bias=True, kernel_regularizer=regularizer))
617
618 # Construct variables for all layers by exercising the network.
619 x = tf.zeros([1, num_features(game)])
620 for layer in self.layers:
621 x = layer(x)
622
623 self.trainable_variables = sum(
624 [layer.trainable_variables for layer in self.layers], [])
625 self.losses = sum([layer.losses for layer in self.layers], [])
626
627 def __call__(self, x):
628 """Evaluates this model on `x`."""
629 return feedforward_evaluate(
630 layers=self.layers,
631 x=x,
632 use_skip_connections=self._use_skip_connections,
633 hidden_are_factored=self._hidden_are_factored)

636class _RcfrSolver(object):
637 """An abstract RCFR solver class.
638
639 Requires that subclasses implement `evaluate_and_update_policy`.
640 """
641
642 def __init__(self, game, models, truncate_negative=False, session=None):
643 """Creates a new `_RcfrSolver`.
644
645 Args:
646 game: An OpenSpiel `Game`.
647 models: Current policy models (optimizable array-like -> `tf.Tensor`
648 callables) for both players.
649 truncate_negative: Whether or not to truncate negative (approximate)
650 cumulative regrets to zero to implement RCFR+. Defaults to `False`.
651 session: A TensorFlow `Session` to convert sequence weights from
652 `tf.Tensor`s produced by `models` to `np.array`s. If `None`, it is
653 assumed that eager mode is enabled. Defaults to `None`.
654 """
655 self._game = game
656 self._models = models
657 self._truncate_negative = truncate_negative
658 self._root_wrapper = RootStateWrapper(game.new_initial_state())
659 self._session = session
660
661 self._cumulative_seq_probs = [
662 np.zeros(n) for n in self._root_wrapper.num_player_sequences
663 ]
664
665 def _sequence_weights(self, player=None):
666 """Returns regret-like weights for each sequence as an `np.array`.
667
668 Negative weights are truncated to zero.
669
670 Args:
671 player: The player to compute weights for, or both if `player` is `None`.
672 Defaults to `None`.
673 """
674 if player is None:
675 return [
676 self._sequence_weights(player)
677 for player in range(self._game.num_players())
678 ]
679 else:
680 tensor = tf.nn.relu(
681 tf.squeeze(self._models[player](
682 self._root_wrapper.sequence_features[player])))
683 return tensor.numpy() if self._session is None else self._session(tensor)

700
701 Returns:
702 A `dict<info state, list<Action, probability>>` that maps info state
703 strings to `Action`-probability pairs describing each player's policy.
704 """
705 return self._root_wrapper.sequence_weights_to_tabular_profile(
706 self._sequence_weights())

708 def average_policy(self):
709 """Returns the average of all policies iterated.
710
711 This average policy converges toward a Nash policy as the number of
712 iterations increases as long as the regret prediction error decreases
713 continually [Morrill, 2016].
714
715 The policy is computed using the accumulated policy probabilities computed
716 using `evaluate_and_update_policy`.
717
718 Returns:
719 A `dict<info state, list<Action, probability>>` that maps info state
720 strings to (Action, probability) pairs describing each player's policy.
721 """
722 return self._root_wrapper.sequence_weights_to_tabular_profile(
723 self._cumulative_seq_probs)

725 def _previous_player(self, player):
726 """The previous player in the turn ordering."""
727 return player - 1 if player > 0 else self._game.num_players() - 1
728
729 def _average_policy_update_player(self, regret_player):
730 """The player for whom the average policy should be updated."""
731 return self._previous_player(regret_player)

734class RcfrSolver(_RcfrSolver):
735 """RCFR with an effectively infinite regret data buffer.
736
737 Exact or bootstrapped cumulative regrets are stored as if an infinitely
738 large data buffer. The average strategy is updated and stored in a full
739 game-size table. Reproduces the RCFR versions used in experiments by
740 Waugh et al. [2015] and Morrill [2016] except that this class does not
741 restrict the user to regression tree models.
742 """
743
744 def __init__(self,
745 game,
746 models,
747 bootstrap=None,
748 truncate_negative=False,
749 session=None):
750 self._bootstrap = bootstrap
751 super(RcfrSolver, self).__init__(
752 game, models, truncate_negative=truncate_negative, session=session)
753
754 self._regret_targets = [
755 np.zeros(n) for n in self._root_wrapper.num_player_sequences
756 ]
757
758 def evaluate_and_update_policy(self, train_fn):
759 """Performs a single step of policy evaluation and policy improvement.
760
761 Args:
762 train_fn: A (model, `tf.data.Dataset`) function that trains the given
763 regression model to accurately reproduce the x to y mapping given x-y
764 data.
765 """
766 sequence_weights = self._sequence_weights()
767 player_seq_features = self._root_wrapper.sequence_features
768 for regret_player in range(self._game.num_players()):
769 seq_prob_player = self._average_policy_update_player(regret_player)
770
771 regrets, seq_probs = (
772 self._root_wrapper.counterfactual_regrets_and_reach_weights(
773 regret_player, seq_prob_player, *sequence_weights))
774
775 if self._bootstrap:
776 self._regret_targets[regret_player][:] = sequence_weights[regret_player]
777 if self._truncate_negative_truncate_negative:
778 regrets = np.maximum(-relu(self._regret_targets[regret_player]),
779 regrets)
780
781 self._regret_targets[regret_player] += regrets
782 self._cumulative_seq_probs[seq_prob_player] += seq_probs
783
784 targets = tf.expand_dims(self._regret_targets[regret_player], axis=1)
785 data = tf.data.Dataset.from_tensor_slices(
786 (player_seq_features[regret_player], targets))
787
788 regret_player_model = self._models[regret_player]
789 train_fn(regret_player_model, data)
790 sequence_weights[regret_player] = self._sequence_weights(regret_player)

793class ReservoirBuffer(object):
794 """A generic reservoir buffer data structure.
795
796 After every insertion, its contents represents a `size`-size uniform
797 random sample from the stream of candidates that have been encountered.
798 """
799
800 def __init__(self, size):
801 self.size = size
802 self.num_elements = 0
803 self._buffer = np.full([size], None, dtype=object)
804 self._num_candidates = 0
805
806 @property
807 def buffer(self):
808 return self._buffer[:self.num_elements]
809
810 def insert(self, candidate):
811 """Consider this `candidate` for inclusion in this sampling buffer."""
812 self._num_candidates += 1
813 if self.num_elements < self.size:
814 self._buffer[self.num_elements] = candidate
815 self.num_elements += 1
816 return
817 idx = np.random.choice(self._num_candidates)
818 if idx < self.size:
819 self._buffer[idx] = candidate
820
821 def insert_all(self, candidates):
822 """Consider all `candidates` for inclusion in this sampling buffer."""
823 for candidate in candidates:
824 self.insert(candidate)
825
826 def num_available_spaces(self):
827 """The number of freely available spaces in this buffer."""
828 return self.size - self.num_elements
