91std::string MaednState::ActionToString(Player player, Action move_id) const {
92 if (player == kChancePlayerId) {
93 // Normal chance roll.
94 return absl::StrCat("chance outcome ", move_id,
95 " (roll: ", kChanceOutcomeValues[move_id], ")");
96 } else {
97 // Assemble a human-readable string representation of the move.
98 if (move_id == kBringInAction) {
99 return absl::StrCat(move_id, " - brings in new piece");
100 } else if (move_id == kPassAction) {
101 return absl::StrCat(move_id, " - passes");
102 } else {
103 return absl::StrCat(move_id, " - moves piece on field ",
104 move_id - kFieldActionsOffset);
105 }

115void MaednState::ObservationTensor(Player player,
116 absl::Span<float> values) const {
117 SPIEL_CHECK_GE(player, 0);
118 SPIEL_CHECK_LT(player, num_players_);
119 SPIEL_CHECK_EQ(values.size(), kStateEncodingSize);
120 auto value_it = values.begin();
121
122 // Tensor should contain state from the player's PoV, so relative
123 // positions are used and converted to absolute positions.
124 int position = PlayerToPosition(player);
125 for (int i = 0; i < kNumCommonFields; i++) {
126 int abs_pos = RelPosToAbsPos(i, position);
127 int piece = board_[abs_pos];
128 *value_it++ = ((piece == 1) ? 1 : 0);
129 *value_it++ = ((piece == 2) ? 1 : 0);
130 *value_it++ = ((piece == 3) ? 1 : 0);
131 *value_it++ = ((piece == 4) ? 1 : 0);
132 }
133
134 // Rotated goal fields to one hot encoded tensor.
135 for (int p = 0; p < kMaxNumPlayers; p++) {
136 int ply_position = PlayerToPosition((player + p) % kMaxNumPlayers);
137 for (int i = 0; i < kNumGoalFieldsPerPlayer; i++) {
138 int abs_pos = RelPosToAbsPos(kNumCommonFields + i, ply_position);
139 int piece = board_[abs_pos];
140 *value_it++ = ((piece == 1) ? 1 : 0);
141 *value_it++ = ((piece == 2) ? 1 : 0);
142 *value_it++ = ((piece == 3) ? 1 : 0);
143 *value_it++ = ((piece == 4) ? 1 : 0);
144 }
145 }
146
147 // Rotated number of pieces outside of field per player.
148 for (int p = 0; p < kMaxNumPlayers; p++) {
149 *value_it++ = (out_[(player + p) % kMaxNumPlayers]);
150 }
151
152 if (cur_player_ == kChancePlayerId) {
153 // Encode chance player with all zeros.
154 for (int i = 0; i < kMaxNumPlayers; i++) {
155 *value_it++ = 0;
156 }
157 } else {
158 int rotated_current_player =
159 (num_players_ + cur_player_ - player) % num_players_;
160 // Rotated current player id to one hot encoded tensor.
161 for (int i = 0; i < kMaxNumPlayers; i++) {
162 *value_it++ = (rotated_current_player == i) ? 1 : 0;
163 }
164 }
165
166 *value_it++ = ((dice_ == 1) ? 1 : 0);
167 *value_it++ = ((dice_ == 2) ? 1 : 0);
168 *value_it++ = ((dice_ == 3) ? 1 : 0);
169 *value_it++ = ((dice_ == 4) ? 1 : 0);
170 *value_it++ = ((dice_ == 5) ? 1 : 0);
171 *value_it++ = ((dice_ == 6) ? 1 : 0);
172
173 SPIEL_CHECK_EQ(value_it, values.end());
174}

176void MaednState::FromObservationTensor(Player player, absl::Span<float> values,
177 Player prev_player, int prev_dice) {
178 SPIEL_CHECK_GE(player, 0);
179 SPIEL_CHECK_LT(player, num_players_);
180 SPIEL_CHECK_EQ(values.size(), kStateEncodingSize);
181
182 prev_player_ = prev_player;
183 prev_dice_ = prev_dice;
184
185 auto value_it = values.begin();
186
187 // Tensor should contain state from the player's PoV, so relative
188 // positions are used and converted to absolute positions.
189 int position = PlayerToPosition(player);
190 for (int i = 0; i < kNumCommonFields; i++) {
191 int abs_pos = RelPosToAbsPos(i, position);
192 int one = *value_it++;
193 int two = *value_it++;
194 int three = *value_it++;
195 int four = *value_it++;
196 int piece = one ? 1 : (two ? 2 : (three ? 3 : (four ? 4 : 0)));
197 board_[abs_pos] = piece;
198 }
199
200 // rotated goal fields to one hot encoded tensor
201 for (int p = 0; p < kMaxNumPlayers; p++) {
202 int ply_position = PlayerToPosition((player + p) % kMaxNumPlayers);
203 for (int i = 0; i < kNumGoalFieldsPerPlayer; i++) {
204 int abs_pos = RelPosToAbsPos(kNumCommonFields + i, ply_position);
205 int one = *value_it++;
206 int two = *value_it++;
207 int three = *value_it++;
208 int four = *value_it++;
209 int piece = one ? 1 : (two ? 2 : (three ? 3 : (four ? 4 : 0)));
210 board_[abs_pos] = piece;
211 }
212 }
213
214 // rotated number of pieces outside of field per player
215 for (int p = 0; p < kMaxNumPlayers; p++) {
216 out_[(player + p) % kMaxNumPlayers] = *value_it++;
217 }
218
219 int zero = *value_it++;
220 int one = *value_it++;
221 int two = *value_it++;
222 int three = *value_it++;
223
224 if (zero + one + two + three == 0) {
225 cur_player_ = kChancePlayerId;
226 } else {
227 int rotated_current_player = zero ? 0 : (one ? 1 : (two ? 2 : 3));
228
229 cur_player_ = (rotated_current_player + player) % num_players_;
230 }
231
232 int dice_1 = *value_it++;
233 int dice_2 = *value_it++;
234 int dice_3 = *value_it++;
235 int dice_4 = *value_it++;
236 int dice_5 = *value_it++;
237 int dice_6 = *value_it++;
238
239 dice_ = dice_1 ? 1
240 : (dice_2 ? 2
241 : (dice_3 ? 3
242 : dice_4 ? 4
243 : (dice_5 ? 5 : (dice_6 ? 6 : 0))));
244
245 SPIEL_CHECK_EQ(value_it, values.end());
246}

272void MaednState::DoApplyAction(Action move) {
273 if (IsChanceNode()) {
274 // Chance action.
275 turn_history_info_.push_back(TurnHistoryInfo(kChancePlayerId, prev_player_,
276 dice_, prev_dice_, move, 0));
277
278 SPIEL_CHECK_TRUE(dice_ == 0);
279 dice_ = kChanceOutcomeValues[move];
280 if (prev_dice_ == 6) {
281 // if last dice roll was a 6, same player moves again
282 cur_player_ = prev_player_;
283 } else {
284 // next player
285 cur_player_ = (prev_player_ + 1) % num_players_;
286 turns_++;
287 }
288 return;
289 }
290
291 // Normal move action.
292 int thrown_out_player = -1;
293
294 if (move != kPassAction) {
295 if (move == kBringInAction) {
296 // Bring in new piece.
297 int players_first_field = GetPlayersFirstField(cur_player_);
298
299 thrown_out_player = board_[players_first_field] - 1;
300 board_[players_first_field] = cur_player_ + 1;
301 out_[cur_player_]--;
302 } else {
303 // Normal piece move.
304 std::pair<int, int> fields =
305 GetFieldsFromAction(move, cur_player_, dice_);
306
307 board_[fields.first] = 0;
308 thrown_out_player = board_[fields.second] - 1;
309 board_[fields.second] = cur_player_ + 1;
310 }
311
312 if (thrown_out_player >= 0) {
313 out_[thrown_out_player]++;
314 }
315 }
316
317 turn_history_info_.push_back(TurnHistoryInfo(
318 cur_player_, prev_player_, dice_, prev_dice_, move, thrown_out_player));
319
320 prev_player_ = cur_player_;
321 prev_dice_ = dice_;
322
323 cur_player_ = kChancePlayerId;
324 dice_ = 0;
325}

327void MaednState::UndoAction(Player player, Action action) {
328 {
329 const TurnHistoryInfo& thi = turn_history_info_.back();
330 SPIEL_CHECK_EQ(thi.player, player);
331 SPIEL_CHECK_EQ(action, thi.action);
332 cur_player_ = thi.player;
333 prev_player_ = thi.prev_player;
334 dice_ = thi.dice;
335 prev_dice_ = thi.prev_dice;
336 if (player != kChancePlayerId && action != kPassAction) {
337 // Undo move.
338 // Code basically is the inverse of DoApplyAction(Action move).
339 if (action == kBringInAction) {
340 // Un-bring in new piece.
341 int players_first_field = GetPlayersFirstField(cur_player_);
342
343 board_[players_first_field] = thi.thrown_out_player + 1;
344 out_[cur_player_]++;
345 } else {
346 // Normal piece move.
347 std::pair<int, int> fields =
348 GetFieldsFromAction(action, cur_player_, dice_);
349
350 board_[fields.first] = cur_player_ + 1;
351 board_[fields.second] = thi.thrown_out_player + 1;
352 }
353
354 if (thi.thrown_out_player >= 0) {
355 out_[thi.thrown_out_player]--;
356 }
357 }
358 }
359 turn_history_info_.pop_back();
360 history_.pop_back();
361 --move_number_;
362}

364std::pair<int, int> MaednState::GetFieldsFromAction(Action action,
365 Player player,
366 int dice) const {
367 int position = PlayerToPosition(player);
368 int relative_source_field = action - kFieldActionsOffset;
369 int relative_target_field = relative_source_field + dice;
370
371 return {RelPosToAbsPos(relative_source_field, position),
372 RelPosToAbsPos(relative_target_field, position)};

375int MaednState::RelPosToAbsPos(int relative_position, int position) const {
376 if (relative_position < kNumCommonFields) {
377 int players_first_field = (kNumCommonFields / kMaxNumPlayers) * position;
378 return (relative_position + players_first_field) % kNumCommonFields;
379 } else {
380 return kNumGoalFieldsPerPlayer * position + relative_position;
381 }
382}
383
384int MaednState::AbsPosToRelPos(int absolute_position, int position) const {
385 if (absolute_position < kNumCommonFields) {
386 int playersFirstField = (kNumCommonFields / kMaxNumPlayers) * position;
387 return (kNumCommonFields + absolute_position - playersFirstField) %
388 kNumCommonFields;
389 } else {
390 return absolute_position - kNumGoalFieldsPerPlayer * position;
391 }
392}

394int MaednState::GetPlayersFirstField(Player player) const {
395 int position = PlayerToPosition(player);
396 return (kNumCommonFields / kMaxNumPlayers) * position;
397}

404std::vector<Action> MaednState::LegalActions() const {
405 if (IsChanceNode()) return LegalChanceOutcomes();
406 if (IsTerminal()) return {};
407
408 std::vector<Action> legal_actions;
409
410 // Follows these rules in this exact order:
411 // - If a player's own piece is standing on the start field
412 // and player has at least one piece off the board, player
413 // MUST move the piece on the start field away unless it is
414 // blocked by another own piece. If that is the case,
415 // player is free to move any own piece.
416 // - If player rolls a 6 and has at least one piece off the
417 // board, player MUST bring in a new piece.
418 // - If player has no (moveable) piece on the board, player
419 // must pass.
420 // - In any other case, player is free to move any own piece
421 // on the board.
422 int players_first_field = GetPlayersFirstField(cur_player_);
423 if (out_[cur_player_] > 0) {
424 if (board_[players_first_field] == cur_player_ + 1) {
425 // Is piece on start field moveable by dice roll?
426 // (playersFirstField + dice) cannot overflow, simple
427 // addition is suitable.
428 if (board_[players_first_field + dice_] != cur_player_ + 1) {
429 legal_actions.push_back(kFieldActionsOffset);
430 return legal_actions;
431 }
432 }
433
434 if (dice_ == 6) {
435 // Player MUST bring in a new piece if possible.
436 // Check whether start field is bloked.
437 if (board_[players_first_field] != cur_player_ + 1) {
438 legal_actions.push_back(kBringInAction);
439 return legal_actions;
440 }
441 // Start field is blocked and this piece itself is
442 // blocked due (has already been checked).
443 }
444 }
445
446 // Look for pieces of current player on board if there is at least one:
447 if (out_[cur_player_] < 4) {
448 int position = PlayerToPosition(cur_player_);
449 const int max_field = kNumCommonFields + kNumGoalFieldsPerPlayer - dice_;
450 for (int relative_source_field = 0; relative_source_field < max_field;
451 relative_source_field++) {
452 int relative_target_field = relative_source_field + dice_;
453
454 int absolute_source_field =
455 RelPosToAbsPos(relative_source_field, position);
456 int absolute_target_field =
457 RelPosToAbsPos(relative_target_field, position);
458
459 if (board_[absolute_source_field] == cur_player_ + 1) {
460 if (board_[absolute_target_field] != cur_player_ + 1) {
461 legal_actions.push_back(relative_source_field + kFieldActionsOffset);
462 }
463 }
464 }
465 }
466
467 // If nothing is possible, player must pass.
468 if (legal_actions.empty()) {
469 legal_actions.push_back(kPassAction);
470 }
471
472 return legal_actions;
473}

475std::string MaednState::ToString() const {
476 std::vector<std::string> board_array = {
477 ". . o-o-S . .", ". . o . o . .", " o . o ",
478 " o . o ", "S-o-o-o-o . o-o-o-o-o", "o . . . . . . . . o",
479 "o-o-o-o-o . o-o-o-o-S", " o . o ", " o . o ",
480 ". . o . o . .", ". . S-o-o . .",
481 };
482
483 // Fill the board.
484 for (int pos = 0; pos < kNumFields; pos++) {
485 if (board_[pos] > 0) {
486 Coords coords = kFieldToBoardString[pos];
487 board_array[coords.y][coords.x] = 48 + board_[pos];
488 }
489 }
490 // Pieces off the board.
491 for (int ply = 0; ply < kMaxNumPlayers; ply++) {
492 int out = out_[ply];
493 int position = PlayerToPosition(ply);
494 int offset = kNumFields + kNumGoalFieldsPerPlayer * position;
495 for (int i = 0; i < out; i++) {
496 Coords coords = kFieldToBoardString[offset + i];
497 board_array[coords.y][coords.x] = 49 + ply;
498 }
499 }
500
501 std::string board_str = absl::StrJoin(board_array, "\n") + "\n";
502
503 // Extra info like whose turn it is etc.
504 absl::StrAppend(&board_str, "Turn: ");
505 absl::StrAppend(&board_str, CurPlayerToString(cur_player_));
506 absl::StrAppend(&board_str, "\n");
507 absl::StrAppend(&board_str, "Dice: ");
508 absl::StrAppend(&board_str, dice_ != 0 ? std::to_string(dice_) : "");
509 absl::StrAppend(&board_str, "\n");
510
511 return board_str;
512}

514bool MaednState::AllInGoal(Player player) const {
515 int position = PlayerToPosition(player);
516 int offset = kNumCommonFields + position * kNumGoalFieldsPerPlayer;
517 return board_[offset] != 0 && board_[offset + 1] != 0 &&
518 board_[offset + 2] != 0 && board_[offset + 3] != 0;
519}

521bool MaednState::IsTerminal() const {
522 for (int ply = 0; ply < num_players_; ply++) {
523 if (AllInGoal(ply)) {
