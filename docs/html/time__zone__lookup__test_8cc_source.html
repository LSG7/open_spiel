669 int VersionCmp(time_zone tz, const std::string& target) {

671  if (version.empty() && !target.empty()) return 1; // unknown > known
672  return version.compare(target);

678 TEST(TimeZones, LoadZonesConcurrently) {

680  std::shared_future<void> ready_future(ready_promise.get_future());

682  std::set<std::string>* failures) {

686  std::string zone = *np;

699  for (std::size_t i = 0; i != n_threads; ++i) {
700  std::promise<void> started;

1047  //
1048  // We limit this test case to environments where:
1049  // 1) we know how to change the time zone used by localtime()/mktime(),

1051  // 3) we have some idea about how mktime() behaves during transitions.

1054  std::string tz_name = (ep != nullptr) ? ep : "";
1055  for (const char* const* np = kTimeZoneNames; *np != nullptr; ++np) {
1056  ASSERT_EQ(0, setenv("TZ", *np, 1)); // change what "localtime" means
1057  const auto zi = local_time_zone();
1058  const auto lc = LoadZone("libc:localtime");
1059  time_zone::civil_transition transition;
1060  for (auto tp = zi.lookup(civil_second()).trans;
1061  zi.next_transition(tp, &transition);
1062  tp = zi.lookup(transition.to).trans) {
1063  const auto fcl = zi.lookup(transition.from);
1064  const auto tcl = zi.lookup(transition.to);
1065  civil_second cs; // compare cs in zi and lc
1066  if (fcl.kind == time_zone::civil_lookup::UNIQUE) {
1067  if (tcl.kind == time_zone::civil_lookup::UNIQUE) {
1068  // Both unique; must be an is_dst or abbr change.
1069  ASSERT_EQ(transition.from, transition.to);

1073  const auto pal = zi.lookup(tprev);

1075  ASSERT_STRNE(pal.abbr, tal.abbr);

1083  ASSERT_EQ(time_zone::civil_lookup::SKIPPED, fcl.kind);
