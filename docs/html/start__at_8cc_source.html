61 // History is represented by a string of numbers separated by kActionSeparator.
62 std::vector<Action> HistoryFromString(const std::string& str) {
63  std::vector<Action> history;
64  if (str.empty()) return history; // Identity transformation.
65 
66  std::vector<absl::string_view> str_actions =
67  absl::StrSplit(str, kActionSeparator);
68  for (const auto& str_action : str_actions) {
69  Action a;
70  if (!absl::SimpleAtoi(str_action, &a)) {
71  SpielFatalError(
72  absl::StrCat("Error when parsing the action: ", str_action));
73  }
74  history.push_back(a);
75  }
76  return history;
77 }
