40 void CheckUndo(const char* fen, const char* move_san, const char* fen_after) {
41  std::shared_ptr<const Game> game = LoadGame("chess");
42  ChessState state(game, fen);
43  Player player = state.CurrentPlayer();
44  absl::optional<Move> maybe_move = state.Board().ParseSANMove(move_san);
45  SPIEL_CHECK_TRUE(maybe_move);
46  Action action = MoveToAction(*maybe_move, state.BoardSize());
47  state.ApplyAction(action);
48  SPIEL_CHECK_EQ(state.Board().ToFEN(), fen_after);
49  state.UndoAction(player, action);
50  SPIEL_CHECK_EQ(state.Board().ToFEN(), fen);
51 }

126 void ObservationTensorTests() {
127  std::shared_ptr<const Game> game = LoadGame("chess");
128  ChessState initial_state(game);
129  auto shape = game->ObservationTensorShape();
130  std::vector<float> v(game->ObservationTensorSize());
131  initial_state.ObservationTensor(initial_state.CurrentPlayer(),
132  absl::MakeSpan(v));
133 
134  // For each piece type, check one square that's supposed to be occupied, and
135  // one that isn't.
136  // Kings.
137  SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "e1"), 1.0);
138  SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "d1"), 0.0);
139  SPIEL_CHECK_EQ(ValueAt(v, shape, 1, "e8"), 1.0);
140  SPIEL_CHECK_EQ(ValueAt(v, shape, 1, "e1"), 0.0);
141 
142  // Queens.
143  SPIEL_CHECK_EQ(ValueAt(v, shape, 2, "d1"), 1.0);
144  SPIEL_CHECK_EQ(ValueAt(v, shape, 2, "e1"), 0.0);
145  SPIEL_CHECK_EQ(ValueAt(v, shape, 3, "d8"), 1.0);
146  SPIEL_CHECK_EQ(ValueAt(v, shape, 3, "d1"), 0.0);
147 
148  // Rooks.
149  SPIEL_CHECK_EQ(ValueAt(v, shape, 4, "a1"), 1.0);
150  SPIEL_CHECK_EQ(ValueAt(v, shape, 4, "e8"), 0.0);
151  SPIEL_CHECK_EQ(ValueAt(v, shape, 5, "h8"), 1.0);
152  SPIEL_CHECK_EQ(ValueAt(v, shape, 5, "c5"), 0.0);
153 
154  // Bishops.
155  SPIEL_CHECK_EQ(ValueAt(v, shape, 6, "c1"), 1.0);
156  SPIEL_CHECK_EQ(ValueAt(v, shape, 6, "b1"), 0.0);
157  SPIEL_CHECK_EQ(ValueAt(v, shape, 7, "f8"), 1.0);
158  SPIEL_CHECK_EQ(ValueAt(v, shape, 7, "f7"), 0.0);
159 
160  // Knights.
161  SPIEL_CHECK_EQ(ValueAt(v, shape, 8, "b1"), 1.0);
162  SPIEL_CHECK_EQ(ValueAt(v, shape, 8, "c3"), 0.0);
163  SPIEL_CHECK_EQ(ValueAt(v, shape, 9, "g8"), 1.0);
164  SPIEL_CHECK_EQ(ValueAt(v, shape, 9, "g7"), 0.0);
165 
166  // Pawns.
167  SPIEL_CHECK_EQ(ValueAt(v, shape, 10, "a2"), 1.0);
168  SPIEL_CHECK_EQ(ValueAt(v, shape, 10, "a3"), 0.0);
169  SPIEL_CHECK_EQ(ValueAt(v, shape, 11, "e7"), 1.0);
170  SPIEL_CHECK_EQ(ValueAt(v, shape, 11, "e6"), 0.0);
171 
172  // Empty.
173  SPIEL_CHECK_EQ(ValueAt(v, shape, 12, "e4"), 1.0);
174  SPIEL_CHECK_EQ(ValueAt(v, shape, 12, "e2"), 0.0);
175 
176  // Repetition count.
177  SPIEL_CHECK_EQ(ValueAt(v, shape, 13, 0, 0), 0.0);
178 
179  // Side to move.
180  SPIEL_CHECK_EQ(ValueAt(v, shape, 14, 0, 0), 1.0);
181 
182  // Irreversible move counter.
183  SPIEL_CHECK_EQ(ValueAt(v, shape, 15, 0, 0), 0.0);
184 
185  // Castling rights.
186  SPIEL_CHECK_EQ(ValueAt(v, shape, 16, 0, 0), 1.0);
187  SPIEL_CHECK_EQ(ValueAt(v, shape, 17, 1, 1), 1.0);
188  SPIEL_CHECK_EQ(ValueAt(v, shape, 18, 2, 2), 1.0);
189  SPIEL_CHECK_EQ(ValueAt(v, shape, 19, 3, 3), 1.0);
190 
191  ApplySANMove("e4", &initial_state);
192  ApplySANMove("e5", &initial_state);
193  ApplySANMove("Ke2", &initial_state);
194 
195  initial_state.ObservationTensor(initial_state.CurrentPlayer(),
196  absl::MakeSpan(v));
197  SPIEL_CHECK_EQ(v.size(), game->ObservationTensorSize());
198 
199  // Now it's black to move.
200  SPIEL_CHECK_EQ(ValueAt(v, shape, 14, 0, 0), 0.0);
201 
202  // White king is now on e2.
203  SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "e1"), 0.0);
204  SPIEL_CHECK_EQ(ValueAt(v, shape, 0, "e2"), 1.0);
205 
206  // Irreversible move counter incremented to 1 (king moving and losing castling
207  // rights is in fact irreversible in this case, but it doesn't reset the
208  // counter according to FIDE rules).
209  SPIEL_CHECK_FLOAT_EQ(ValueAt(v, shape, 15, 0, 0), 1.0 / 101.0);
210 
211  // And white no longer has castling rights.
212  SPIEL_CHECK_EQ(ValueAt(v, shape, 16, 0, 0), 0.0);
213  SPIEL_CHECK_EQ(ValueAt(v, shape, 17, 1, 1), 0.0);
214  SPIEL_CHECK_EQ(ValueAt(v, shape, 18, 2, 2), 1.0);
215  SPIEL_CHECK_EQ(ValueAt(v, shape, 19, 3, 3), 1.0);
216 }

218 void MoveConversionTests() {
219  auto game = LoadGame("chess");
220  std::mt19937 rng(23);
221  for (int i = 0; i < 100; ++i) {
222  std::unique_ptr<State> state = game->NewInitialState();
223  while (!state->IsTerminal()) {
224  const ChessState* chess_state =
225  dynamic_cast<const ChessState*>(state.get());
226  std::vector<Action> legal_actions = state->LegalActions();
227  absl::uniform_int_distribution<int> dist(0, legal_actions.size() - 1);
228  int action_index = dist(rng);
229  Action action = legal_actions[action_index];
230  Move move = ActionToMove(action, chess_state->Board());
231  Action action_from_move = MoveToAction(move, chess_state->BoardSize());
232  SPIEL_CHECK_EQ(action, action_from_move);
233  const ChessBoard& board = chess_state->Board();
234  ChessBoard fresh_board = chess_state->StartBoard();
235  for (Move move : chess_state->MovesHistory()) {
236  fresh_board.ApplyMove(move);
237  }
238  SPIEL_CHECK_EQ(board.ToFEN(), fresh_board.ToFEN());
239  Action action_from_lan =
240  MoveToAction(*board.ParseLANMove(move.ToLAN()), board.BoardSize());
241  SPIEL_CHECK_EQ(action, action_from_lan);
242  state->ApplyAction(action);
243  }
244  }
245 }
