95void ConnectFourState::DoApplyAction(Action move) {
96 SPIEL_CHECK_EQ(CellAt(kRows - 1, move), CellState::kEmpty);
97 int row = 0;
98 while (CellAt(row, move) != CellState::kEmpty) ++row;
99 CellAt(row, move) = PlayerToState(CurrentPlayer());
100
101 if (HasLine(current_player_)) {
102 outcome_ = static_cast<Outcome>(current_player_);
103 } else if (IsFull()) {
104 outcome_ = Outcome::kDraw;
105 }
106
107 current_player_ = 1 - current_player_;
108}

110std::vector<Action> ConnectFourState::LegalActions() const {
111 // Can move in any non-full column.
112 std::vector<Action> moves;
113 if (IsTerminal()) return moves;
114 for (int col = 0; col < kCols; ++col) {
115 if (CellAt(kRows - 1, col) == CellState::kEmpty) moves.push_back(col);
116 }
117 return moves;
118}

132bool ConnectFourState::HasLineFromInDirection(Player player, int row, int col,
133 int drow, int dcol) const {
134 if (row + 3 * drow >= kRows || col + 3 * dcol >= kCols ||
135 row + 3 * drow < 0 || col + 3 * dcol < 0)
136 return false;
137 CellState c = PlayerToState(player);
138 for (int i = 0; i < 4; ++i) {
139 if (CellAt(row, col) != c) return false;
140 row += drow;
141 col += dcol;
142 }
143 return true;
144}

146bool ConnectFourState::HasLine(Player player) const {
147 CellState c = PlayerToState(player);
148 for (int col = 0; col < kCols; ++col) {
149 for (int row = 0; row < kRows; ++row) {
150 if (CellAt(row, col) == c && HasLineFrom(player, row, col)) return true;
151 }
152 }
153 return false;
154}

156bool ConnectFourState::IsFull() const {
157 for (int col = 0; col < kCols; ++col) {
158 if (CellAt(kRows - 1, col) == CellState::kEmpty) return false;
159 }
160 return true;
161}

200int PlayerRelative(CellState state, Player current) {
201 switch (state) {
202 case CellState::kNought:
203 return current == 0 ? 0 : 1;
204 case CellState::kCross:
205 return current == 1 ? 0 : 1;
206 case CellState::kEmpty:
207 return 2;

209 SpielFatalError("Unknown player type.");

232ConnectFourState::ConnectFourState(std::shared_ptr<const Game> game,
233 const std::string& str)
234 : State(game) {
235 int xs = 0;
236 int os = 0;
237 int r = 5;
238 int c = 0;
239 for (const char ch : str) {
240 switch (ch) {
241 case '.':
242 CellAt(r, c) = CellState::kEmpty;
243 break;
244 case 'x':
245 ++xs;
246 CellAt(r, c) = CellState::kCross;
247 break;
248 case 'o':
249 ++os;
250 CellAt(r, c) = CellState::kNought;
251 break;
252 }
253 if (ch == '.' || ch == 'x' || ch == 'o') {
254 ++c;
255 if (c >= kCols) {
256 r--;
257 c = 0;
258 }
259 }
260 }
261 SPIEL_CHECK_TRUE(xs == os || xs == (os + 1));
262 SPIEL_CHECK_TRUE(r == -1 && ("Problem parsing state (incorrect rows)."));
263 SPIEL_CHECK_TRUE(c == 0 &&
264 ("Problem parsing state (column value should be 0)"));
265 current_player_ = (xs == os) ? 0 : 1;
266
267 if (HasLine(0)) {
268 outcome_ = Outcome::kPlayer1;
269 } else if (HasLine(1)) {
270 outcome_ = Outcome::kPlayer2;
271 } else if (IsFull()) {
272 outcome_ = Outcome::kDraw;
273 }
274}
