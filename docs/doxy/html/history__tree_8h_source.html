79// History here refers to the fact that we're using histories- i.e.
80// representations of all players private information in addition to the public
81// information- as the underlying abstraction. Other trees are possible, such as
82// PublicTrees, which use public information as the base abstraction, and
83// InformationStateTrees, which use all of the information available to one
84// player as the base abstraction.
85class HistoryTree {
86 public:
87 // Builds a tree of histories. player_id is needed here as we view all chance
88 // and terminal nodes from the viewpoint of player_id. Decision nodes are
89 // viewed from the perspective of the player making the decision.
90 HistoryTree(std::unique_ptr<State> state, Player player_id);
91
92 HistoryNode* Root() { return root_.get(); }
93
94 HistoryNode* GetByHistory(const std::string& history);
95 HistoryNode* GetByHistory(const State& state) {
96 return GetByHistory(state.HistoryString());
97 }
98
99 // For test use only.
100 std::vector<std::string> GetHistories();
101
102 Action NumHistories() { return state_to_node_.size(); }
103
104 private:
105 std::unique_ptr<HistoryNode> root_;
106
107 // Maps histories to HistoryNodes.
108 absl::flat_hash_map<std::string, HistoryNode*> state_to_node_;
109};

111// Returns a map of infostate strings to a vector of history nodes with
112// corresponding counter-factual probabilities, where counter-factual
113// probabilities are calculatd using the passed policy for the opponent's
114// actions, a probability of 1 for all of the best_responder's actions, and the
115// natural chance probabilty for all change actions. We return all infosets
116// (i.e. all sets of history nodes grouped by infostate) for the sub-game rooted
117// at state, from the perspective of the player with id best_responder.
118absl::flat_hash_map<std::string, std::vector<std::pair<HistoryNode*, double>>>
119GetAllInfoSets(std::unique_ptr<State> state, Player best_responder,
120 const Policy* policy, HistoryTree* tree);
121
