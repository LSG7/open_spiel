37// x corresponds to file (column / letter)
38// y corresponds to rank (row / number).
39struct Square {
40 Square& operator+=(const Offset& offset) {
41 x += offset.x_offset;
42 y += offset.y_offset;
43 return *this;
44 }
45
46 bool operator==(const Square& other) const {
47 return x == other.x && y == other.y;
48 }
49
50 bool operator!=(const Square& other) const { return !(*this == other); }
51
52 // Required by std::set.
53 bool operator<(const Square& other) const {
54 if (x != other.x) {
55 return x < other.x;
56 } else {
57 return y < other.y;
58 }
59 }
60
61 std::string ToString() const {
62 std::string s;
63 s.push_back('a' + x);
64 s.push_back('1' + y);
65 return s;
66 }
67
68 int8_t x;
69 int8_t y;
70};

123// n-dimensional array of uniform random numbers.
124// Example:
125// ZobristTable<int, 3, 4, 5> table;
126//
127// table[a][b][c] is a random int where a < 3, b < 4, c < 5
128//
129template <typename T, std::size_t InnerDim, std::size_t... OtherDims>
130class ZobristTable {
131 public:
132 using Generator = std::mt19937_64;
133 using NestedTable = ZobristTable<T, OtherDims...>;
134
135 ZobristTable(Generator::result_type seed) {
136 Generator generator(seed);
137 absl::uniform_int_distribution<Generator::result_type> dist;
138 data_.reserve(InnerDim);
139 for (std::size_t i = 0; i < InnerDim; ++i) {
140 data_.emplace_back(dist(generator));
141 }
142 }
143
144 const NestedTable& operator[](std::size_t inner_index) const {
145 return data_[inner_index];
146 }
147
148 private:
149 std::vector<NestedTable> data_;
150};
151
152// 1-dimensional array of uniform random numbers.
153template <typename T, std::size_t InnerDim>
154class ZobristTable<T, InnerDim> {
155 public:
156 using Generator = std::mt19937_64;
157
158 ZobristTable(Generator::result_type seed) : data_(InnerDim) {
159 Generator generator(seed);
160 absl::uniform_int_distribution<T> dist;
161 for (auto& field : data_) {
162 field = dist(generator);
163 }
164 }
165
166 T operator[](std::size_t index) const { return data_[index]; }
167
168 private:
169 std::vector<T> data_;
170};
