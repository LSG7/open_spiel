51def main(_):
52 game = "lewis_signaling"
53 num_players = 2
54
55 num_states = FLAGS.num_states
56 num_messages = FLAGS.num_messages
57 if FLAGS.payoffs == "random":
58 payoffs = np.random.random((num_states, num_states))
59 payoffs_str = ",".join([str(x) for x in payoffs.flatten()])
60 elif FLAGS.payoffs == "climbing":
61 # This is a particular payoff matrix that is hard for decentralized
62 # algorithms. Introduced in C. Claus and C. Boutilier, "The dynamics of
63 # reinforcement learning in cooperative multiagent systems", 1998, for
64 # simultaneous action games, but it is difficult even in the case of
65 # signaling games.
66 payoffs = np.array([[11, -30, 0], [-30, 7, 6], [0, 0, 5]]) / 30
67 payoffs_str = ",".join([str(x) for x in payoffs.flatten()])
68 else:
69 payoffs_str = FLAGS.payoffs
70 try:
71 payoffs_list = [float(x) for x in payoffs_str.split(",")]
72 payoffs = np.array(payoffs_list).reshape((num_states, num_states))
73 except ValueError:
74 raise ValueError(
75 "There should be {} (states * actions) elements in payoff. Found {} elements"
76 .format(num_states * num_states, len(payoffs_list))) from None
77
78 env_configs = {
79 "num_states": num_states,
80 "num_messages": num_messages,
81 "payoffs": payoffs_str
82 }
83
84 env = rl_environment.Environment(game, **env_configs)
85 state_size = env.observation_spec()["info_state"][0]
86 num_actions = env.action_spec()["num_actions"]
87 replay_buffer_capacity = FLAGS.replay_buffer_capacity
88
89 # Results to store
90 num_runs = FLAGS.num_runs
91 training_episodes = FLAGS.num_episodes
92 log_interval = FLAGS.log_interval
93 rewards = np.zeros((num_runs, training_episodes // log_interval))
94 opts = np.zeros((num_runs, training_episodes // log_interval))
95 converge_point = np.zeros((num_states, num_states))
96 percent_opt = 0
97
98 # Repeat the experiment num_runs times
99 for i in range(num_runs):
100 with tf.Session() as sess:
101 # pylint: disable=g-complex-comprehension
102 agents = [
103 dqn.DQN(
104 sess,
105 player_id=idx,
106 state_representation_size=state_size,
107 num_actions=num_actions,
108 learning_rate=FLAGS.step_size,
109 replay_buffer_capacity=replay_buffer_capacity,
110 epsilon_start=FLAGS.eps_init,
111 epsilon_end=FLAGS.eps_final,
112 epsilon_decay_duration=FLAGS.eps_decay_steps * 2)
113 for idx in range(num_players)
114 ]
115
116 # 1. Train the agents
117 for cur_episode in range(training_episodes):
118 time_step = env.reset()
119 # Find cur_state for logging. See lewis_signaling.cc for info_state
120 # details
121 cur_state = time_step.observations["info_state"][0][3:].index(1)
122 while not time_step.last():
123 player_id = time_step.observations["current_player"]
124 agent_output = agents[player_id].step(time_step)
125 time_step = env.step([agent_output.action])
126
127 # Episode is over, step all agents with final info state.
128 for agent in agents:
129 agent.step(time_step)
130
131 # Store rewards
132 reward = time_step.rewards[0]
133 max_reward = payoffs[cur_state].max()
134 cur_idx = (i, cur_episode // log_interval)
135 rewards[cur_idx] += reward / log_interval
136 opts[cur_idx] += np.isclose(reward, max_reward) / log_interval
137
138 base_info_state0 = [1.0, 0.0, 0.0] + [0.0] * num_states
139 base_info_state1 = [0.0, 1.0, 0.0] + [0.0] * num_states
140
141 for s in range(num_states):
142 info_state0 = copy.deepcopy(base_info_state0)
143 info_state0[3 + s] = 1.0
144 # pylint: disable=protected-access
145 m, _ = agents[0]._epsilon_greedy(info_state0, np.arange(num_messages),
146 0)
147 info_state1 = copy.deepcopy(base_info_state1)
148 info_state1[3 + m] = 1.0
149 a, _ = agents[1]._epsilon_greedy(info_state1, np.arange(num_states), 0)
150 converge_point[s, a] += 1
151 best_act = payoffs[s].argmax()
152 percent_opt += int(a == best_act) / num_runs / num_states

174 def plot_scalars(scalars,
175 repetition_axis=0,
176 scalar_labels=None,
177 title=None,
178 ax_labels=None):
179 """Plots scalar on ax by filling 1 standard error.
180
181 Args:
182 scalars: List of scalars to plot (mean taken over repetition
183 axis)
184 repetition_axis: Axis to take the mean over
185 scalar_labels: Labels for the scalars (for legend)
186 title: Figure title
187 ax_labels: Labels for x and y axis (list of 2 strings)
188 """
189 if not all([len(s.shape) == 2 for s in scalars]):
190 raise ValueError("Only 2D arrays supported for plotting")
191
192 if scalar_labels is None:
193 scalar_labels = [None] * len(scalars)
194
195 if len(scalars) != len(scalar_labels):
196 raise ValueError(
197 "Wrong number of scalar labels, expected {} but received {}".format(
198 len(scalars), len(scalar_labels)))
199
200 _, plot_axis = init_fig()
201 for i, scalar in enumerate(scalars):
202 xs = np.arange(scalar.shape[1 - repetition_axis]) * FLAGS.log_interval
203 mean = scalar.mean(axis=repetition_axis)
204 sem = stats.sem(scalar, axis=repetition_axis)
205 plot_axis.plot(xs, mean, label=scalar_labels[i])
206 plot_axis.fill_between(xs, mean - sem, mean + sem, alpha=0.5)
207
208 if title is not None:
209 plot_axis.set_title(title)
210 if ax_labels is not None:
211 plot_axis.set_xlabel(ax_labels[0])
212 plot_axis.set_ylabel(ax_labels[1])

214 def plot_confusion_matrix(cm, cmap=plt.cm.Blues, title=None):
215 """Plot the confusion matrix.
216
217 Args:
218 cm (np.ndarray): Confusion matrix to plot
219 cmap: Color map to be used in matplotlib's imshow
220 title: Figure title
221
222 Returns:
223 Figure and axis on which the confusion matrix is plotted.
224 """
225 fig, ax = plt.subplots()
226 ax.imshow(cm, interpolation="nearest", cmap=cmap)
227 ax.set_xticks([])
228 ax.set_yticks([])
229 ax.set_xlabel("Receiver's action", fontsize=14)
230 ax.set_ylabel("Sender's state", fontsize=14)
231 # Loop over data dimensions and create text annotations.
232 fmt = "d"
233 thresh = cm.max() / 2.
234 for i in range(cm.shape[0]):
235 for j in range(cm.shape[1]):
236 ax.text(
237 j,
238 i,
239 format(cm[i, j], fmt),
240 ha="center",
241 va="center",
242 color="white" if cm[i, j] > thresh else "black")
243 fig.tight_layout()
244 if title is not None:
245 ax.set_title(title)
246 return fig, ax
247
248 plot_scalars([rewards],
249 title="Reward graph (DQN)",
250 ax_labels=["Episodes", "Reward per episode"])
251 plot_scalars([opts],
252 title="Percentage of optimal actions (DQN)",
253 ax_labels=["Episodes", "% optimal actions"])
254
255 plot_confusion_matrix(
256 converge_point.astype(int), title="Final policy (DQN)")
257
258 plt.show()
259
260 return percent_opt
