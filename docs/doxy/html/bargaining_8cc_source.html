92std::vector<double> BargainingState::Returns() const {
93 if (agreement_reached_) {
94 int proposing_player = (offers_.size() + 1) % kNumPlayers;
95 int other_player = 1 - proposing_player;
96 std::vector<double> returns(kNumPlayers, 0);
97 for (int i = 0; i < kNumItemTypes; ++i) {
98 returns[proposing_player] +=
99 instance_.values[proposing_player][i] * offers_.back().quantities[i];
100 returns[other_player] +=
101 instance_.values[other_player][i] *
102 (instance_.pool[i] - offers_.back().quantities[i]);
103 }
104 // Apply discount.
105 if (discount_ < 1.0) {
106 for (Player p = 0; p < num_players_; ++p) {
107 returns[p] *= discount_;
108 }
109 }
110 return returns;
111 } else {
112 return std::vector<double>(kNumPlayers, 0);
113 }

159std::unique_ptr<State> BargainingState::ResampleFromInfostate(
160 int player_id, std::function<double()> rng) const {
161 std::vector<int> valid_indices;
162 const int num_instances = parent_game_->AllInstances().size();
163 for (int i = 0; i < num_instances; ++i) {
164 const Instance& instance = parent_game_->GetInstance(i);
165 if (instance_.pool == instance.pool &&
166 instance_.values[player_id] == instance.values[player_id]) {
167 valid_indices.push_back(i);
168 }
169 }
170
171 SPIEL_CHECK_FALSE(valid_indices.empty());
172 int idx = static_cast<int>(rng() * valid_indices.size());
173 SPIEL_CHECK_GE(idx, 0);
174 SPIEL_CHECK_LT(idx, valid_indices.size());
175
176 int instance_idx = valid_indices[idx];
177 std::unique_ptr<State> state = parent_game_->NewInitialState();
178 for (Action action : History()) {
179 if (state->IsChanceNode()) {
180 state->ApplyAction(instance_idx);
181 } else {
182 state->ApplyAction(action);
183 }
184 }
185 return state;
186}

188void BargainingState::ObservationTensor(Player player,
189 absl::Span<float> values) const {
190 SPIEL_CHECK_GE(player, 0);
191 SPIEL_CHECK_LT(player, num_players_);
192 SPIEL_CHECK_EQ(values.size(), game_->ObservationTensorSize());
193 std::fill(values.begin(), values.end(), 0);
194
195 if (IsChanceNode()) {
196 // No observations at chance nodes.
197 return;
198 }
199
200 int offset = 0;
201
202 // Agreement reached?
203 if (agreement_reached_) {
204 values[offset] = 1;
205 }
206 offset += 1;
207
208 // How many trade offers have happened?
209 values[offers_.size()] = 1;
210 offset += parent_game_->max_turns() + 1;
211
212 // Pool
213 for (int i = 0; i < kNumItemTypes; ++i) {
214 for (int j = 0; j <= instance_.pool[i]; ++j) {
215 values[offset + j] = 1;
216 }
217 offset += kPoolMaxNumItems + 1;
218 }
219
220 // My values
221 for (int i = 0; i < kNumItemTypes; ++i) {
222 for (int j = 0; j <= instance_.values[player][i]; ++j) {
223 values[offset + j] = 1;
224 }
225 offset += kTotalValueAllItems + 1;
226 }
227
228 // Just the last offer
229 if (!offers_.empty()) {
230 for (int i = 0; i < kNumItemTypes; ++i) {
231 for (int j = 0; j <= offers_.back().quantities[i]; ++j) {
232 values[offset + j] = 1;
233 }
234 offset += kPoolMaxNumItems + 1;
235 }
236 } else {
237 offset += (kPoolMaxNumItems + 1) * kNumItemTypes;
238 }
239
240 SPIEL_CHECK_EQ(offset, values.size());
241}

331void BargainingState::DoApplyAction(Action action) {
332 if (IsChanceNode()) {
333 if (move_number_ == 0) {
334 instance_ = parent_game_->GetInstance(action);
335 cur_player_ = 0;
336 } else {
337 if (action == parent_game_->ContinueOutcome()) {
338 cur_player_ = next_player_;
339 } else {
340 SPIEL_CHECK_EQ(action, parent_game_->EndOutcome());
341 game_ended_ = true;
342 cur_player_ = kTerminalPlayerId;
343 }
344 }
345 } else {
346 // Check to apply discount.
347 if (move_number_ >= 3 && parent_game_->discount() < 1.0) {
348 discount_ *= parent_game_->discount();
349 }
350
351 const std::vector<Offer>& all_offers = parent_game_->AllOffers();
352 if (action != AgreeAction()) {
353 offers_.push_back(all_offers.at(action));
354
355 if (move_number_ >= 2 && parent_game_->prob_end() > 0.0) {
356 next_player_ = 1 - cur_player_;
357 cur_player_ = kChancePlayerId;
358 } else {
359 cur_player_ = 1 - cur_player_;
360 }
361 } else {
362 // Agree action.
363 SPIEL_CHECK_EQ(action, AgreeAction());
364 agreement_reached_ = true;
365 }
366 }

369bool BargainingState::IsLegalOffer(const Offer& offer) const {
370 // An offer is legal if it's a proper subset of the current pool.
371 for (int i = 0; i < kNumItemTypes; ++i) {
372 if (offer.quantities[i] > instance_.pool[i]) {
373 return false;
374 }
375 }
376 return true;
377}

379std::vector<Action> BargainingState::LegalActions() const {
380 if (IsChanceNode()) {
381 return LegalChanceOutcomes();
382 } else if (IsTerminal()) {
383 return {};
384 } else {
385 const std::vector<Offer>& all_offers = parent_game_->AllOffers();
386 std::vector<Action> legal_actions;
387 for (int i = 0; i < all_offers.size(); ++i) {
388 if (IsLegalOffer(all_offers.at(i))) {
389 legal_actions.push_back(i);
390 }
391 }
392 // Add the agree action if there's at least one offer on the table.
393 if (!offers_.empty()) {
394 legal_actions.push_back(all_offers.size());
395 }
396 return legal_actions;
397 }

400std::vector<std::pair<Action, double>> BargainingState::ChanceOutcomes() const {
401 SPIEL_CHECK_TRUE(IsChanceNode());
402 std::vector<std::pair<Action, double>> outcomes;
403 const int num_boards = parent_game_->AllInstances().size();
404
405 if (move_number_ == 0) {
406 // First chance move of the game. This is for determining the instance.
407 outcomes.reserve(num_boards);
408 double uniform_prob = 1.0 / num_boards;
409 for (int i = 0; i < num_boards; ++i) {
410 outcomes.push_back({i, uniform_prob});
411 }
412 } else {
413 const double prob_end = parent_game_->prob_end();
414 SPIEL_CHECK_TRUE(move_number_ >= 3);
415 outcomes = {{parent_game_->ContinueOutcome(), 1.0 - prob_end},
416 {parent_game_->EndOutcome(), prob_end}};
417 }
418 return outcomes;
419}

446void BargainingGame::ParseInstancesString(const std::string& instances_string) {
447 std::vector<std::string> lines = absl::StrSplit(instances_string, '\n');
448 SPIEL_CHECK_GT(lines.size(), 1);
449 for (const std::string& line : lines) {
450 if (!line.empty()) {
451 std::vector<std::string> parts = absl::StrSplit(line, ' ');
452 SPIEL_CHECK_EQ(parts.size(), kNumItemTypes);
453 Instance instance;
454 // pool
455 std::vector<std::string> pool_parts = absl::StrSplit(parts[0], ',');
456 for (int i = 0; i < kNumItemTypes; ++i) {
457 SPIEL_CHECK_TRUE(absl::SimpleAtoi(pool_parts[i], &instance.pool[i]));
458 }
459 // p1 values
460 std::vector<std::string> p1values_parts = absl::StrSplit(parts[1], ',');
461 for (int i = 0; i < kNumItemTypes; ++i) {
462 SPIEL_CHECK_TRUE(
463 absl::SimpleAtoi(p1values_parts[i], &instance.values[0][i]));
464 }
465 // p2 values
466 std::vector<std::string> p2values_parts = absl::StrSplit(parts[2], ',');
467 for (int i = 0; i < kNumItemTypes; ++i) {
468 SPIEL_CHECK_TRUE(
469 absl::SimpleAtoi(p2values_parts[i], &instance.values[1][i]));
470 }
471 all_instances_.push_back(instance);
472 }
473 }

476void BargainingGame::CreateOffers() {
477 std::vector<int> cur_offer(kNumItemTypes, 0);
478 bool done = false;
479 do {
480 if (std::accumulate(cur_offer.begin(), cur_offer.end(), 0) <=
481 kPoolMaxNumItems) {
482 all_offers_.push_back(Offer(cur_offer));
483 }
484
485 // Try adding a digit to the left-most, keep going until you can. Then
486 // set everything to the left of it to 0.
487 done = true;
488 for (int i = 0; i < kNumItemTypes; ++i) {
489 if (cur_offer[i] < kPoolMaxNumItems) {
490 done = false;
491 cur_offer[i]++;
492 for (int j = i - 1; j >= 0; j--) {
493 cur_offer[j] = 0;
494 }
495 break;
496 }
497 }
498 } while (!done);
499}
