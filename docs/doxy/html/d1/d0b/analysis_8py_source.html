53def print_columns(strings, max_width=MAX_WIDTH):
54 """Prints a list of strings in columns."""
55 padding = 2
56 shortest = min(len(s) for s in strings)
57 max_columns = max(1, math.floor((max_width - 1) / (shortest + 2 * padding)))
58 for cols in range(max_columns, 0, -1):
59 rows = math.ceil(len(strings) / cols)
60 chunks = [strings[i:i + rows] for i in range(0, len(strings), rows)]
61 col_widths = [max(len(s) for s in chunk) for chunk in chunks]
62 if sum(col_widths) + 2 * padding * len(col_widths) <= max_width:
63 break
64 for r in range(rows):
65 for c in range(cols):
66 i = r + c * rows
67 if i < len(strings):
68 print(" " * padding + strings[i].ljust(col_widths[c] + padding), end="")
69 print()

77def sub_sample(data, count):
78 return data[::(max(1, len(data) // count))]

81def smooth(data, count):
82 for k in data.keys():
83 if not isinstance(k, str) or not k.startswith("time_"):
84 data[k] = data[k].rolling(max(1, len(data) // count)).mean()
85 return data

98def prepare(data, cols):
99 """Given the dataset and a list of columns return a small pandas dataframe."""
100 for col in ["step", "total_states", "total_trajectories", "time_rel"]:
101 cols[col] = [col]
102 subdata = {key: select(data, col) for key, col in cols.items()}
103 # subdata = list(zip(*subdata)) # transpose
104 df = pd.DataFrame(subdata)
105 df = smooth(df, SMOOTHING_RATE)
106 df = sub_sample(df, SUBSAMPLING_MAX)
107 df["time_rel_h"] = df["time_rel"] / 3600
108 df["zero"] = 0
109 return df

118def plot_avg_stddev(ax, x, data, data_col):
119 """Plot stats produced by open_spiel::BasicStats::ToJson."""
120 cols = ["avg", "std_dev", "min", "max"]
121 df = prepare(data, {v: data_col + [v] for v in cols})
122 df.plot(ax=ax, x=x, y="avg", color="b")
123 plt.fill_between(
124 x=df[x], color="b", alpha=0.2, label="std dev",
125 y1=np.nanmax([df["min"], df["avg"] - df["std_dev"]], 0),
126 y2=np.nanmin([df["max"], df["avg"] + df["std_dev"]], 0))
127 plt.fill_between(
128 x=df[x], color="b", alpha=0.2, label="min/max",
129 y1=df["min"], y2=df["max"])
130 plot_zero(df, ax, x)

133def plot_histogram_numbered(ax, x, data, data_col):
134 """Plot stats produced by open_spiel::HistogramNumbered::ToJson."""
135 x_min, x_max = 0, data[-1][x]
136 y_min, y_max = 0, len(subselect(data, [0] + data_col))
137 z_min, z_max = 0, 1
138 z = np.array([subselect(row, data_col) for row in data], dtype=float)
139 z = np.concatenate((z, np.zeros((x_max, 1))), axis=1) # Don't cut off the top
140 # TODO(author7): smoothing
141 z = sub_sample(z, SUBSAMPLING_MAX).transpose()
142 p = np.percentile(z, 99)
143 if p > 0:
144 z /= p
145 z[z > 1] = 1
146 ax.grid(False)
147 ax.imshow(z, cmap="Reds", vmin=z_min, vmax=z_max,
148 extent=[x_min, x_max, y_min, y_max + 1],
149 interpolation="nearest", origin="lower", aspect="auto")

152def plot_histogram_named(ax, x, data, data_col, normalized=True):
153 """Plot stats produced by open_spiel::HistogramNamed::ToJson."""
154 names = subselect(data, [0] + data_col + ["names"])
155 df = prepare(data, {name: data_col + ["counts", i]
156 for i, name in enumerate(names)})
157 if normalized:
158 total = sum(df[n] for n in names)
159 for n in names:
160 df[n] /= total
161 df.plot.area(ax=ax, x=x, y=names)
