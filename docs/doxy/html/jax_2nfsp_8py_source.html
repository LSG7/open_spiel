50 def __init__(self,
51 player_id,
52 state_representation_size,
53 num_actions,
54 hidden_layers_sizes,
55 reservoir_buffer_capacity,
56 anticipatory_param,
57 batch_size=128,
58 rl_learning_rate=0.01,
59 sl_learning_rate=0.01,
60 min_buffer_size_to_learn=1000,
61 learn_every=64,
62 optimizer_str="sgd",
63 **kwargs):
64 """Initialize the `NFSP` agent."""
65 self.player_id = player_id
66 self._num_actions = num_actions
67 self._layer_sizes = hidden_layers_sizes
68 self._batch_size = batch_size
69 self._learn_every = learn_every
70 self._anticipatory_param = anticipatory_param
71 self._min_buffer_size_to_learn = min_buffer_size_to_learn
72
73 self._reservoir_buffer = ReservoirBuffer(reservoir_buffer_capacity)
74 self._prev_timestep = None
75 self._prev_action = None
76
77 # Step counter to keep track of learning.
78 self._step_counter = 0
79
80 # Inner RL agent
81 kwargs.update({
82 "batch_size": batch_size,
83 "learning_rate": rl_learning_rate,
84 "learn_every": learn_every,
85 "min_buffer_size_to_learn": min_buffer_size_to_learn,
86 "optimizer_str": optimizer_str,
87 })
88 self._rl_agent = dqn.DQN(player_id, state_representation_size,
89 num_actions, hidden_layers_sizes, **kwargs)
90
91 # Keep track of the last training loss achieved in an update step.
92 self._last_rl_loss_value = lambda: self._rl_agent.loss
93 self._last_sl_loss_value = None
94
95 # Average policy network.
96 def network(x):
97 mlp = hk.nets.MLP(self._layer_sizes + [num_actions])
98 return mlp(x)
99
100 self.hk_avg_network = hk.without_apply_rng(hk.transform(network))
101
102 def avg_network_policy(param, info_state):
103 action_values = self.hk_avg_network.apply(param, info_state)
104 action_probs = jax.nn.softmax(action_values, axis=1)
105 return action_values, action_probs
106
107 self._avg_network_policy = jax.jit(avg_network_policy)
108
109 rng = jax.random.PRNGKey(42)
110 x = jnp.ones([1, state_representation_size])
111 self.params_avg_network = self.hk_avg_network.init(rng, x)
112 self.params_avg_network = jax.device_put(self.params_avg_network)
113
114 self._savers = [
115 ("q_network", self._rl_agent.params_q_network),
116 ("avg_network", self.params_avg_network)
117 ]
118
119 if optimizer_str == "adam":
120 opt_init, opt_update = optax.chain(
121 optax.scale_by_adam(b1=0.9, b2=0.999, eps=1e-8),
122 optax.scale(sl_learning_rate))
123 elif optimizer_str == "sgd":
124 opt_init, opt_update = optax.sgd(sl_learning_rate)
125 else:
126 raise ValueError("Not implemented. Choose from ['adam', 'sgd'].")
127 self._opt_update_fn = self._get_update_func(opt_update)
128 self._opt_state = opt_init(self.params_avg_network)
129 self._loss_and_grad = jax.value_and_grad(self._loss_avg, has_aux=False)
130
131 self._sample_episode_policy()
132 self._jit_update = jax.jit(self.get_update())

134 def _get_update_func(self, opt_update):
135
136 def update(params, opt_state, gradient):
137 """Learning rule (stochastic gradient descent)."""
138 updates, opt_state = opt_update(gradient, opt_state)
139 new_params = optax.apply_updates(params, updates)
140 return new_params, opt_state
141
142 return update

145 return self._step_counter

147 @contextlib.contextmanager
148 def temp_mode_as(self, mode):
149 """Context manager to temporarily overwrite the mode."""
150 previous_mode = self._mode
151 self._mode = mode
152 yield
153 self._mode = previous_mode

155 def _sample_episode_policy(self):
156 if np.random.rand() < self._anticipatory_param:
157 self._mode = MODE.best_response
158 else:
159 self._mode = MODE.average_policy
160
161 def _act(self, info_state, legal_actions):
162 info_state = np.reshape(info_state, [1, -1])
163 action_values, action_probs = self._avg_network_policy(
164 self.params_avg_network, info_state
165 )
166
167 self._last_action_values = action_values[0]
168 # Remove illegal actions, normalize probs
169 probs = np.zeros(self._num_actions)
170 action_probs = np.asarray(action_probs)
171 probs[legal_actions] = action_probs[0][legal_actions]
172 probs /= sum(probs)
173 action = np.random.choice(len(probs), p=probs)
174 return action, probs
175
176 @property
177 def mode(self):
178 return self._mode
179
180 @property
181 def loss(self):
182 return (self._last_sl_loss_value, self._last_rl_loss_value())

184 def step(self, time_step, is_evaluation=False):
185 """Returns the action to be taken and updates the Q-networks if needed.
186
187 Args:
188 time_step: an instance of rl_environment.TimeStep.
189 is_evaluation: bool, whether this is a training or evaluation call.
190
191 Returns:
192 A `rl_agent.StepOutput` containing the action probs and chosen action.
193 """
194 if self._mode == MODE.best_response:
195 agent_output = self._rl_agent.step(time_step, is_evaluation)
196 if not is_evaluation and not time_step.last():
197 self._add_transition(time_step, agent_output)
198
199 elif self._mode == MODE.average_policy:
200 # Act step: don't act at terminal info states.
201 if not time_step.last():
202 info_state = time_step.observations["info_state"][self.player_id]
203 legal_actions = time_step.observations["legal_actions"][self.player_id]
204 action, probs = self._act(info_state, legal_actions)
205 agent_output = rl_agent.StepOutput(action=action, probs=probs)
206
207 if self._prev_timestep and not is_evaluation:
208 self._rl_agent.add_transition(self._prev_timestep, self._prev_action,
209 time_step)
210 else:
211 raise ValueError("Invalid mode ({})".format(self._mode))
212
213 if not is_evaluation:
214 self._step_counter += 1
215
216 if self._step_counter % self._learn_every == 0:
217 self._last_sl_loss_value = self._learn()
218 # If learn step not triggered by rl policy, learn.
219 if self._mode == MODE.average_policy:
220 self._rl_agent.learn()
221
222 # Prepare for the next episode.
223 if time_step.last():
224 self._sample_episode_policy()
225 self._prev_timestep = None
226 self._prev_action = None
227 return
228 else:
229 self._prev_timestep = time_step
230 self._prev_action = agent_output.action
231 return agent_output

233 def _add_transition(self, time_step, agent_output):
234 """Adds the new transition using `time_step` to the reservoir buffer.
235
236 Transitions are in the form (time_step, agent_output.probs, legal_mask).
237
238 Args:
239 time_step: an instance of rl_environment.TimeStep.
240 agent_output: an instance of rl_agent.StepOutput.
241 """
242 legal_actions = time_step.observations["legal_actions"][self.player_id]
243 legal_actions_mask = np.zeros(self._num_actions)
244 legal_actions_mask[legal_actions] = 1.0
245 transition = Transition(
246 info_state=(time_step.observations["info_state"][self.player_id][:]),
247 action_probs=agent_output.probs,
248 legal_actions_mask=legal_actions_mask)
249 self._reservoir_buffer.add(transition)
250
251 def _loss_avg(self, param_avg, info_states, action_probs):
252 avg_logit = self.hk_avg_network.apply(param_avg, info_states)
253 loss_value = -jnp.sum(
254 action_probs * jax.nn.log_softmax(avg_logit)) / avg_logit.shape[0]
255 return loss_value
256
257 def get_update(self):
258 def update(param_avg, opt_state_avg, info_states, action_probs):
259 loss_val, grad_val = self._loss_and_grad(param_avg, info_states,
260 action_probs)
261 new_param_avg, new_opt_state_avg = self._opt_update_fn(
262 param_avg, opt_state_avg, grad_val)
263 return new_param_avg, new_opt_state_avg, loss_val
264 return update

266 def _learn(self):
267 """Compute the loss on sampled transitions and perform a avg-network update.
268
269 If there are not enough elements in the buffer, no loss is computed and
270 `None` is returned instead.
271
272 Returns:
273 The average loss obtained on this batch of transitions or `None`.
274 """
275 if (len(self._reservoir_buffer) < self._batch_size or
276 len(self._reservoir_buffer) < self._min_buffer_size_to_learn):
277 return None
278
279 transitions = self._reservoir_buffer.sample(self._batch_size)
280 info_states = np.asarray([t.info_state for t in transitions])
281 action_probs = np.asarray([t.action_probs for t in transitions])
282
283 self.params_avg_network, self._opt_state, loss_val_avg = self._jit_update(
284 self.params_avg_network, self._opt_state, info_states, action_probs)
285 return loss_val_avg
