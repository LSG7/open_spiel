48  """Returns a function filtering players' strategies wrt.
49 
50  'filter_function'.
51 
52  This function is used to select which strategy to start training from. As
53  such, and in the Rectified Nash Response logic, filter_function expects a
54  certain set of arguments:
55  - player_policies: The list of policies for the current player.
56  - player: The current player id.
57  - effective_number_selected: The effective number of policies to select.
58  - solver: In case the above arguments weren't enough, the solver instance so
59  the filter_function can have more complex behavior.
60  And returns the selected policies and policy indexes for the current player.
61 
62  Args:
63  filter_function: A filter function following the specifications above, used
64  to filter which strategy to start training from for each player.
65 
66  Returns:
67  A filter function on all players.
68  """

70  def filter_policies(solver, number_policies_selected=1):
71  """Filters each player's policies according to 'filter_function'.
72 
73  Args:
74  solver: The PSRO solver.
75  number_policies_selected: The expected number of policies to select. If
76  there are fewer policies than 'number_policies_selected', behavior will
77  saturate at num_policies.
78 
79  Returns:
80  used_policies : List of length 'num_players' of lists of length
81  min('number_policies_selected', num_policies') containing selected
82  policies.
83  used_policies_indexes: List of lists of the same shape as used_policies,
84  containing the list indexes of selected policies.
85 
86  """
87  policies = solver.get_policies()
88  num_players = len(policies)
89  meta_strategy_probabilities = solver.get_meta_strategies()
90 
91  used_policies = []
92  used_policy_indexes = []
93  for player in range(num_players):
94  player_policies = policies[player]
95  current_selection_probabilities = meta_strategy_probabilities[player]
96  effective_number = min(number_policies_selected, len(player_policies))
97 
98  used_policy, used_policy_index = filter_function(
99  player_policies, current_selection_probabilities, player,
100  effective_number, solver)
101  used_policies.append(used_policy)
102  used_policy_indexes.append(used_policy_index)
103  return used_policies, used_policy_indexes
104 
105  # Return the created function.
106  return filter_policies

109 def rectified_filter(player_policies, selection_probabilities, player,
110  effective_number_to_select, solver):
111  """Returns every strategy with nonzero selection probability.
112 
113  Args:
114  player_policies: A list of policies for the current player.
115  selection_probabilities: Selection probabilities for 'player_policies'.
116  player: Player id.
117  effective_number_to_select: Effective number of policies to select.
118  solver: PSRO solver instance if kwargs needed.
119 
120  Returns:
121  selected_policies : List of size 'effective_number_to_select'

123  selected_indexes: List of the same shape as selected_policies,
124  containing the list indexes of selected policies.
125  """
126  del effective_number_to_select, solver, player

269 def get_current_and_average_payoffs(ps2ro_trainer, current_player,
270  current_strategy):
271  """Returns the current player's and average players' payoffs.
272 
273  These payoffs are returned when current_player's strategy's index is
274  'current_strategy'.
275 
276  Args:
277  ps2ro_trainer: A ps2ro object.
278  current_player: Integer, current player index.
279  current_strategy: Integer, current player's strategy index.
280 
281  Returns:
282  Payoff tensor for current player, Average payoff tensor over all players.
283  """
284  # Get the vector of payoffs associated with current_player's strategy ind
285  meta_games = ps2ro_trainer.meta_games
286  current_payoff = meta_games[current_player]
287  current_payoff = np.take(
288  current_payoff, current_strategy, axis=current_player)
289 
290  # Get average per-player payoff matrix.
291  average_payoffs = np.mean(meta_games, axis=0)
292  average_payoffs = np.take(
293  average_payoffs, current_strategy, axis=current_player)
294  return current_payoff, average_payoffs

297 def rectified_selector(ps2ro_trainer, current_player, current_strategy):
298  current_payoff, average_payoffs = get_current_and_average_payoffs(
299  ps2ro_trainer, current_player, current_strategy)
300 
301  # Rectified Nash condition : select those strategies where we do better
302  # than others.
303  res = current_payoff >= average_payoffs
304  return np.expand_dims(res, axis=current_player)
