21def parse_preflib_data(string_data: str) -> base.PreferenceProfile:
22 """Parses the contents of a PrefLib data file.
23
24 Currently only supports SOC and SOI. See https://www.preflib.org/format.
25
26 Args:
27 string_data: the name of the file to parse.
28
29 Returns:
30 A preference profile.
31 """
32 lines = string_data.split("\n")
33 alternatives = []
34 num_alternatives = None
35 num_votes = None
36 profile = base.PreferenceProfile()
37 for raw_line in lines:
38 line = raw_line.strip()
39 if not line: continue
40 if line.startswith("#"):
41 parts = line.split(" ")
42 if line.startswith("# DATA TYPE: "):
43 assert(parts[3] == "soc" or parts[3] == "soi")
44 elif line.startswith("# NUMBER ALTERNATIVES:"):
45 num_alternatives = int(parts[3])
46 alternatives = [None] * num_alternatives
47 elif line.startswith("# NUMBER VOTERS:"):
48 num_votes = int(parts[3])
49 elif line.startswith("# ALTERNATIVE NAME "):
50 num = int(parts[3].split(":")[0])
51 index_of_colon = line.index(":")
52 assert 1 <= num <= num_alternatives
53 alternatives[num-1] = line[index_of_colon+2:]
54 else:
55 if profile.num_alternatives() == 0:
56 profile = base.PreferenceProfile(alternatives=alternatives)
57 index_of_colon = line.index(":")
58 weight = int(line[:index_of_colon])
59 vote_parts = line[index_of_colon+2:].split(",")
60 vote = [alternatives[int(part) - 1] for part in vote_parts]
61 if weight > 0:
62 profile.add_vote(vote, weight)
63 assert num_votes == profile.num_votes()
64 return profile

67def parse_preflib_datafile(filename: str) -> base.PreferenceProfile:
68 """Parses a Preflib data file.
69
70 Currently only supports SOC and SOI. See https://www.preflib.org/format.
71
72 Args:
73 filename: the name of the file to parse.
74
75 Returns:
76 A preference profile.
77 """
