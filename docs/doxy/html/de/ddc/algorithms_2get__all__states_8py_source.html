15"""Example algorithm to get all states from a game.
16
17The algorithm does not support mean field games where the game evolution depends
18on the mean field distribution.
19"""
20
21import itertools
22
23from open_spiel.python import games # pylint:disable=unused-import
24import pyspiel

27def _get_subgames_states(state, all_states, depth_limit, depth,
28 include_terminals, include_chance_states,
29 include_mean_field_states, to_string,
30 stop_if_encountered):
31 """Extract non-chance states for a subgame into the all_states dict."""
32 if state.is_terminal():
33 if include_terminals:
34 # Include if not already present and then terminate recursion.
35 state_str = to_string(state)
36 if state_str not in all_states:
37 all_states[state_str] = state.clone()
38 return
39
40 if depth > depth_limit >= 0:
41 return
42 is_mean_field = state.current_player() == pyspiel.PlayerId.MEAN_FIELD
43 if (state.is_chance_node() and
44 include_chance_states) or (is_mean_field and
45 include_mean_field_states) or not (
46 state.is_chance_node() or is_mean_field):
47 # Add only if not already present
48 state_str = to_string(state)
49 if state_str not in all_states:
50 all_states[state_str] = state.clone()
51 else:
52 # We already saw this one. Stop the recursion if the flag is set
53 if stop_if_encountered:
54 return
55
56 if is_mean_field:
57 support = state.distribution_support()
58 state_for_search = state.clone()
59 support_length = len(support)
60 # update with a dummy distribution
61 state_for_search.update_distribution(
62 [1.0 / support_length for _ in range(support_length)])
63 _get_subgames_states(state_for_search, all_states, depth_limit, depth + 1,
64 include_terminals, include_chance_states,
65 include_mean_field_states, to_string,
66 stop_if_encountered)
67 elif state.is_simultaneous_node():
68 joint_legal_actions = [
69 state.legal_actions(player)
70 for player in range(state.get_game().num_players())
71 ]
72 for joint_actions in itertools.product(*joint_legal_actions):
73 state_for_search = state.clone()
74 state_for_search.apply_actions(list(joint_actions))
75 _get_subgames_states(state_for_search, all_states, depth_limit, depth + 1,
76 include_terminals, include_chance_states,
77 include_mean_field_states, to_string,
78 stop_if_encountered)
79 else:
80 for action in state.legal_actions():
81 state_for_search = state.child(action)
82 _get_subgames_states(state_for_search, all_states, depth_limit, depth + 1,
83 include_terminals, include_chance_states,
84 include_mean_field_states, to_string,
85 stop_if_encountered)

88def get_all_states(game,
89 depth_limit=-1,
90 include_terminals=True,
91 include_chance_states=False,
92 include_mean_field_states=False,
93 to_string=lambda s: s.history_str(),
94 stop_if_encountered=True):
95 """Gets all states in the game, indexed by their string representation.
96
97 For small games only! Useful for methods that solve the games explicitly,
98 i.e. value iteration. Use this default implementation with caution as it does
99 a recursive tree walk of the game and could easily fill up memory for larger
100 games or games with long horizons.
101
102 Currently only works for sequential games.
103
104 Arguments:
105 game: The game to analyze, as returned by `load_game`.
106 depth_limit: How deeply to analyze the game tree. Negative means no limit, 0
107 means root-only, etc.
108 include_terminals: If True, include terminal states.
109 include_chance_states: If True, include chance node states.
110 include_mean_field_states: If True, include mean field node states.
111 to_string: The serialization function. We expect this to be
112 `lambda s: s.history_str()` as this enforces perfect recall, but for
113 historical reasons, using `str` is also supported, but the goal is to
114 remove this argument.
115 stop_if_encountered: if this is set, do not keep recursively adding states
116 if this state is already in the list. This allows support for games that
117 have cycles.
118
119 Returns:
120 A `dict` with `to_string(state)` keys and `pyspiel.State` values containing
121 all states encountered traversing the game tree up to the specified depth.
122 """
123 root_states = game.new_initial_states()
124 all_states = dict()
125
126 for root in root_states:
127 # Then, do a recursive tree walk to fill up the map.
128 _get_subgames_states(
129 state=root,
130 all_states=all_states,
131 depth_limit=depth_limit,
132 depth=0,
133 include_terminals=include_terminals,
134 include_chance_states=include_chance_states,
135 include_mean_field_states=include_mean_field_states,
136 to_string=to_string,
137 stop_if_encountered=stop_if_encountered)
138
139 if not all_states:
140 raise ValueError("GetSubgameStates returned 0 states!")
141
142 return all_states
