31def _make_network(lstm_hidden_sizes: List[int],
32 mlp_hidden_sizes: List[int],
33 output_dim: int) -> hk.RNNCore:
34 """set up the network."""
35
36 layers = []
37 for k, hidden_size in enumerate(lstm_hidden_sizes):
38 layers += [hk.LSTM(hidden_size, name=f'lstm_layer_{k}'), jax.nn.relu]
39 layers += [hk.nets.MLP(mlp_hidden_sizes + [output_dim], name='mlp')]
40 return RNNModel(layers)

43def _make_forwards(lstm_hidden_sizes: List[int], mlp_hidden_sizes: List[int],
44 output_dim: int, batch_size: int) -> hk.Transformed:
45
46 """Forward pass."""
47
48 def forward_fn(inputs):
49 rnn = _make_network(lstm_hidden_sizes, mlp_hidden_sizes, output_dim)
50 initial_state = rnn.initial_state(batch_size=batch_size)
51 outputs, _ = hk.dynamic_unroll(rnn, inputs, initial_state, time_major=False)
52 return outputs
53
54 network = hk.transform(forward_fn)
55 return network

58def meta_loss(opt_params, net_apply, payoff, steps, rng):
59 """Meta loss function."""
60
61 regret_sum_x = np.zeros(shape=[FLAGS.batch_size, 1, FLAGS.num_actions])
62 regret_sum_y = np.zeros(shape=[FLAGS.batch_size, 1, FLAGS.num_actions])
63 total_loss = 0
64
65 @jax.jit
66 def body_fun(s, total_loss):
67 nonlocal regret_sum_x
68 nonlocal regret_sum_y
69 x = net_apply(opt_params, rng, regret_sum_x / (s + 1))
70 y = net_apply(opt_params, rng, regret_sum_y / (s + 1))
71
72 strategy_x = jax.nn.softmax(x)
73 strategy_y = jnp.transpose(jax.nn.softmax(y), [0, 2, 1])
74
75 values_x = jnp.matmul(payoff, strategy_y)
76 values_y = -jnp.matmul(strategy_x, payoff)
77
78 value_x = jnp.matmul(jnp.matmul(strategy_x, payoff), strategy_y)
79 value_y = -value_x
80
81 curren_regret_x = values_x - value_x
82 curren_regret_y = values_y - value_y
83 curren_regret_x = jnp.transpose(curren_regret_x, [0, 2, 1])
84
85 regret_sum_x += curren_regret_x
86 regret_sum_y += curren_regret_y
87
88 current_loss = jnp.max(
89 jax.numpy.concatenate([curren_regret_x, curren_regret_y], axis=2),
90 axis=[1, 2])
91 total_loss += current_loss
92 return total_loss
93 def fori_loop(lower, steps, body_fun, total_loss):
94 val = total_loss
95 for i in range(lower, steps):
96 val = body_fun(i, total_loss)
97 return val
98 total_loss = fori_loop(0, steps, body_fun, total_loss)
99 return jnp.mean(total_loss)

102class OptimizerModel:
103 """Optimizer model."""
104
105 def __init__(self, learning_rate):
106 self.learning_rate = learning_rate
107 self.model = _make_forwards(
108 lstm_hidden_sizes=[20],
109 mlp_hidden_sizes=[],
110 output_dim=3,
111 batch_size=FLAGS.batch_size)
112 self.net_apply = self.model.apply
113 self.net_init = self.model.init
114 self.opt_update, self.net_params, self.opt_state = None, None, None
115
116 def lr_scheduler(self, init_value):
117 schedule_fn = optax.polynomial_schedule(
118 init_value=init_value, end_value=0.05, power=1., transition_steps=50)
119 return schedule_fn
120
121 def get_optimizer_model(self):
122 schedule_fn = self.lr_scheduler(self.learning_rate)
123 opt_init, self.opt_update = optax.chain(
124 optax.scale_by_adam(), optax.scale_by_schedule(schedule_fn),
125 optax.scale(-self.learning_rate))
126 rng = jax.random.PRNGKey(10)
127 dummy_input = np.random.normal(
128 loc=0, scale=10., size=(FLAGS.batch_size, 1, FLAGS.num_actions))
129 self.net_params = self.net_init(rng, dummy_input)
130 self.opt_state = opt_init(self.net_params)

133class MetaSelfplayAgent:
134 """Meta player agent."""
135
136 def __init__(self, repeats, training_epochs, data_loader):
137 self.repeats = repeats
138 self.training_epochs = training_epochs
139 self.net_apply = None
140 self.net_params = None
141 self.regret_sum = None
142 self.step = 0
143 self.data_loader = data_loader
144 self._rng = hk.PRNGSequence(10)
145
146 def train(self):
147 self.training_optimizer()
148 self.regret_sum = jnp.zeros(shape=[FLAGS.batch_size, 1, FLAGS.num_actions])
149
150 def initial_policy(self):
151 x = self.net_apply(self.net_params, next(self._rng), self.regret_sum)
152 self.last_policy = jax.nn.softmax(x)
153 self.step += 1
154 return self.last_policy
155
156 def next_policy(self, last_values):
157 value = jnp.matmul(self.last_policy, last_values)
158 curren_regret = jnp.transpose(last_values, [0, 2, 1]) - value
159 self.regret_sum += curren_regret
160
161 x = self.net_apply(self.net_params, next(self._rng),
162 self.regret_sum / (self.step + 1))
163 self.last_policy = jax.nn.softmax(x)
164 self.step += 1
165 return self.last_policy
166
167 def training_optimizer(self):
168 """Train optimizer."""
169
170 optimizer = OptimizerModel(0.01)
171 optimizer.get_optimizer_model()
172 for _ in range(FLAGS.num_batches):
173 batch_payoff = next(self.data_loader)
174 for _ in range(self.repeats):
175 grads = jax.grad(
176 meta_loss, has_aux=False)(optimizer.net_params, optimizer.net_apply,
177 batch_payoff, self.training_epochs,
178 next(self._rng))
179
180 updates, optimizer.opt_state = optimizer.opt_update(
181 grads, optimizer.opt_state)
182 optimizer.net_params = optax.apply_updates(optimizer.net_params,
183 updates)
184 self.net_apply = optimizer.net_apply
185 self.net_params = optimizer.net_params
