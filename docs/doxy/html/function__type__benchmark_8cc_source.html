41template <typename Function, typename... Args>
42void ABSL_ATTRIBUTE_NOINLINE CallFunction(Function f, Args&&... args) {
43 f(std::forward<Args>(args)...);
44}
45
46template <typename Function, typename Callable, typename... Args>
47void ConstructAndCallFunctionBenchmark(benchmark::State& state,
48 const Callable& c, Args&&... args) {
49 for (auto _ : state) {
50 CallFunction<Function>(c, std::forward<Args>(args)...);
51 }
52}

105// Doesn't include construction or copy overhead in the loop.
106template <typename Function, typename Callable, typename... Args>
107void CallFunctionBenchmark(benchmark::State& state, const Callable& c,
108 Args... args) {
109 Function f = c;
110 for (auto _ : state) {
111 benchmark::DoNotOptimize(&f);
112 f(args...);
113 }
114}
