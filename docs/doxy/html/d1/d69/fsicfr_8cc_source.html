42void FSICFRNode::ApplyRegretMatching() {
43 double pos_rsum = 0.0;
44 for (int a = 0; a < legal_actions.size(); ++a) {
45 pos_rsum += std::max(0.0, regrets[a]);
46 }
47 for (int a = 0; a < legal_actions.size(); ++a) {
48 strategy[a] = pos_rsum > 0 ? std::max(0.0, regrets[a]) / pos_rsum
49 : 1.0 / legal_actions.size();
50 SPIEL_CHECK_PROB(strategy[a]);
51 }
52}

54FSICFRNode* FSICFRGraph::GetOrCreateDecisionNode(
55 const std::vector<Action>& legal_actions,
56 const std::string& info_state_string, Player player, int max_predecessors,
57 int chance_id) {
58 auto iter = string_key_to_node_id_map_.find(info_state_string);
59 if (iter != string_key_to_node_id_map_.end()) {
60 return &nodes_[iter->second];
61 } else {
62 FSICFRNode node;
63 node.terminal = false;
64 node.string_key = info_state_string;
65 node.player = player;
66 node.chance_id = chance_id;
67 node.max_predecessors = max_predecessors;
68 node.legal_actions = legal_actions;
69 node.psum = {0.0, 0.0};
70 node.strategy =
71 std::vector<double>(legal_actions.size(), 1.0 / legal_actions.size());
72 node.regrets =
73 std::vector<double>(legal_actions.size(), 1.0 / legal_actions.size());
74 node.ssum = std::vector<double>(legal_actions.size(), 0.0);
75 node.id = nodes_.size();
76 string_key_to_node_id_map_[info_state_string] = node.id;
77 nodes_.push_back(node);
78 return &nodes_[node.id];
79 }
80}

101void FSICFRGraph::TopSort() {
102 int max_value = -1;
103 int cur_value = 0;
104 bool done = false;
105 int num_nodes = 0;
106
107 while (!done) {
108 num_nodes = 0;
109 for (int i = 0; i < nodes_.size(); ++i) {
110 max_value = std::max(max_value, nodes_[i].max_predecessors);
111 if (nodes_[i].max_predecessors == cur_value) {
112 // std::cout << nodes_[i].max_predecessors << " "
113 // << nodes_[i].string_key << std::endl;
114 ordered_ids_.push_back(i);
115 num_nodes++;
116 }
117 }
118
119 cur_value++;
120 if (cur_value > max_value) {
121 done = true;
122 }
123 }
124
125 SPIEL_CHECK_EQ(nodes_.size(), ordered_ids_.size());
126}

138void FSICFRSolver::RunIteration() {
139 // Predetermine chance outcomes (one per player).
140 for (int i = 0; i < sampled_chance_outcomes_.size(); ++i) {
141 sampled_chance_outcomes_[i] =
142 absl::Uniform<int>(rng_, 0, chance_outcome_ranges_[i]);
143 SPIEL_CHECK_GE(sampled_chance_outcomes_[i], 0);
144 SPIEL_CHECK_LT(sampled_chance_outcomes_[i], chance_outcome_ranges_[i]);
145 }
146 ForwardPass();
147 BackwardPass();
148 total_iterations_++;
149}

151void FSICFRSolver::RunIterations(int n) {
152 for (int i = 0; i < n; ++i) {
153 RunIteration();

157void FSICFRSolver::ForwardPass() {
158 bool done_first = false;
159 for (int idx = 0; idx < graph_->size(); ++idx) {
160 int node_id = graph_->ordered_node_id(idx);
161 FSICFRNode* node = graph_->GetNode(node_id);
162 if (!node->terminal &&
163 node->chance_id == sampled_chance_outcomes_[node->player]) {
164 if (!done_first) {
165 node->visits = 1;
166 node->psum = {1.0, 1.0};
167 done_first = true;
168 }
169 node->ApplyRegretMatching();
170 double my_reach = node->psum[node->player];
171 int opp_chance_id = sampled_chance_outcomes_[1 - node->player];
172 for (int a = 0; a < node->legal_actions.size(); ++a) {
173 node->ssum[a] += my_reach * node->strategy[a];
174 Action action = node->legal_actions[a];
175 auto iter = node->children.find({action, opp_chance_id});
176 SPIEL_CHECK_TRUE(iter != node->children.end());
177 int child_id = iter->second;
178 FSICFRNode* child = graph_->GetNode(child_id);
179 if (!child->terminal) {
180 child->visits += node->visits;
181 SPIEL_CHECK_GT(child->visits, 0);
182 for (int p : {0, 1}) {
183 child->psum[p] +=
184 node->psum[p] * (node->player == p ? node->strategy[a] : 1.0);
185 SPIEL_CHECK_GE(child->psum[p], 0);
186 }
187 }
188 }
189 }
190 }
191}

193void FSICFRSolver::BackwardPass() {
194 for (int idx = graph_->size() - 1; idx >= 0; --idx) {
195 int node_id = graph_->ordered_node_id(idx);
196 FSICFRNode* node = graph_->GetNode(node_id);
197 if (!node->terminal &&
198 node->chance_id == sampled_chance_outcomes_[node->player]) {
199 node->v = 0;
200 int opp_chance_id = sampled_chance_outcomes_[1 - node->player];
201 std::vector<double> values(node->legal_actions.size(), 0);
202 double opp_reach = node->psum[1 - node->player];
203 for (int a = 0; a < node->legal_actions.size(); ++a) {
204 Action action = node->legal_actions[a];
205 auto iter = node->children.find({action, opp_chance_id});
206 SPIEL_CHECK_TRUE(iter != node->children.end());
207 int child_id = iter->second;
208 FSICFRNode* child = graph_->GetNode(child_id);
209 if (child->terminal) {
210 SPIEL_CHECK_TRUE(child->p0_utility == -1 || child->p0_utility == 1);
211 values[a] =
212 node->player == 0 ? child->p0_utility : -child->p0_utility;
213 } else {
214 values[a] = node->player == child->player ? child->v : -child->v;
215 }
216 node->v += node->strategy[a] * values[a];
217 }
218 for (int a = 0; a < node->legal_actions.size(); ++a) {
219 node->regrets[a] = (node->T * node->regrets[a] +
220 node->visits * opp_reach * (values[a] - node->v)) /
221 (node->T + node->visits);
222 }
223 node->T += node->visits;
224 node->visits = 0;
225 node->psum[0] = 0;
226 node->psum[1] = 0;
227 }
228 }
229}

231TabularPolicy FSICFRSolver::GetAveragePolicy() const {
232 TabularPolicy policy;
233 for (int idx = 0; idx < graph_->size(); ++idx) {
234 FSICFRNode* node = graph_->GetNode(idx);
235 if (!node->terminal) {
236 ActionsAndProbs state_policy;
237 double denom = std::accumulate(node->ssum.begin(), node->ssum.end(), 0.0);
238 SPIEL_CHECK_GE(denom, 0.0);
239 for (int a = 0; a < node->legal_actions.size(); ++a) {
240 Action action = node->legal_actions[a];
241 double prob = denom > 0 ? node->ssum[a] / denom
242 : 1.0 / node->legal_actions.size();
243 SPIEL_CHECK_PROB(prob);
244 state_policy.push_back({action, prob});
245 }
246 policy.SetStatePolicy(node->string_key, state_policy);
247 }
248 }
249 return policy;
250}
