20def policy_to_dict(player_policy,
21 game,
22 all_states=None,
23 state_to_information_state=None):
24 """Converts a Policy instance into a tabular policy represented as a dict.
25
26 This is compatible with the C++ TabularExploitability code (i.e.
27 pyspiel.exploitability, pyspiel.TabularBestResponse, etc.).
28
29 While you do not have to pass the all_states and state_to_information_state
30 arguments, creating them outside of this funciton will speed your code up
31 dramatically.
32
33 Args:
34 player_policy: The policy you want to convert to a dict.
35 game: The game the policy is for.
36 all_states: The result of calling get_all_states.get_all_states. Can be
37 cached for improved performance.
38 state_to_information_state: A dict mapping str(state) to
39 state.information_state for every state in the game. Can be cached for
40 improved performance.
41
42 Returns:
43 A dictionary version of player_policy that can be passed to the C++
44 TabularBestResponse, Exploitability, and BestResponse functions/classes.
45 """
46 if all_states is None:
47 all_states = get_all_states.get_all_states(
48 game,
49 depth_limit=-1,
50 include_terminals=False,
51 include_chance_states=False)
52 state_to_information_state = {
53 state: all_states[state].information_state_string()
54 for state in all_states
55 }
56 tabular_policy = dict()
57 for state in all_states:
58 information_state = state_to_information_state[state]
59 tabular_policy[information_state] = list(
60 player_policy.action_probabilities(all_states[state]).items())
61 return tabular_policy

64def get_best_response_actions_as_string(best_response_actions):
65 """Turns a dict<bytes, int> into a bytestring compatible with C++.
66
67 i.e. the bytestring can be copy-pasted as the brace initialization for a
68 {std::unordered_,std::,absl::flat_hash_}map<std::string, int>.
69
70 Args:
71 best_response_actions: A dict mapping bytes to ints.
72
73 Returns:
74 A bytestring that can be copy-pasted to brace-initialize a C++
75 std::map<std::string, T>.
76 """
77 best_response_keys = sorted(best_response_actions.keys())
78 best_response_strings = [
79 "%s: %i" % (k, best_response_actions[k]) for k in best_response_keys
80 ]
81 return "{%s}" % (", ".join(best_response_strings))

84def tabular_policy_to_cpp_map(policy):
85 """Turns a policy into a C++ compatible bytestring for brace-initializing.
86
87 Args:
88 policy: A dict representing a tabular policy. The keys are infostate
89 bytestrings.
90
91 Returns:
92 A bytestring that can be copy-pasted to brace-initialize a C++
93 std::map<std::string, open_spiel::ActionsAndProbs>.
94 """
95 cpp_entries = []
96 policy_keys = sorted(policy.keys())
97 for key in policy_keys:
98 tuple_strs = ["{%i, %s}" % (p[0], p[1].astype(str)) for p in policy[key]]
99 value = "{" + ", ".join(tuple_strs) + "}"
100 cpp_entries.append('{"%s", %s}' % (key, value))
101 return "{%s}" % (",\n".join(cpp_entries))
