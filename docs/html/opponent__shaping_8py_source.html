87 def get_minibatches(
88  batch: TransitionBatch, num_minibatches: int
89 ) -> typing.Iterator[TransitionBatch]:
90  """Yields an iterator over minibatches of the given batch.
91 
92  Args:
93  batch: A transition batch.
94  num_minibatches: The number of minibatches to return.
95 
96  Yields:
97  An iterator over minibatches of the given batch.
98  """
99 
100  def get_minibatch(x, start, end):
101  return x[:, start:end] if len(x.shape) > 2 else x
102 
103  for i in range(num_minibatches):
104  start, end = i * (batch.reward.shape[1] // num_minibatches), (i + 1) * (
105  batch.reward.shape[1] // num_minibatches
106  )
107  mini_batch = jax.tree_util.tree_map(
108  partial(get_minibatch, start=start, end=end), batch
109  )
110  yield mini_batch

113 def get_critic_update_fn(
114  agent_id: int,
115  critic_network: hk.Transformed,
116  optimizer: optax.TransformUpdateFn,
117  num_minibatches: int = 8,
118  gamma: float = 0.99,
119 ) -> UpdateFn:
120  """Returns the update function for the critic parameters.
121 
122  Args:
123  agent_id: The id of the agent that will be updated.
124  critic_network: A transformed haiku function.
125  optimizer: Optimizer update function.
126  num_minibatches: the number of minibatches.
127  gamma: the discount factor.
128 
129  Returns:
130  An update function that takes the current train state together with a
131  transition batch and returns the new train state and a dictionary of
132  metrics.
133  """
134 
135  def loss_fn(params, batch: TransitionBatch):
136  info_states, rewards = batch.info_state[agent_id], batch.reward[agent_id]
137  discounts = jnp.ones_like(rewards) * gamma
138  values = critic_network.apply(params, info_states).squeeze()
139  v_t = values[:, :-1].reshape(-1)
140  v_tp1 = values[:, 1:].reshape(-1)
141  r_t = rewards[:, :-1].reshape(-1)
142  d_t = discounts[:, 1:].reshape(-1)
143  td_error = jax.lax.stop_gradient(r_t + d_t * v_tp1) - v_t
144  return jnp.mean(td_error**2)
145 
146  def update(train_state: TrainState, batch: TransitionBatch):
147  """The critic update function.
148 
149  Updates the critic parameters of the train state with the given
150  transition batch.
151 
152  Args:
153  train_state: The current train state.
154  batch: A transition batch.
155 
156  Returns:
157  The updated train state with the new critic params and a dictionary
158  with the critic loss
159  """
160  losses = []
161  critic_params = train_state.critic_params[agent_id]
162  opt_state = train_state.critic_opt_states[agent_id]
163  for mini_batch in get_minibatches(batch, num_minibatches):
164  loss, grads = jax.value_and_grad(loss_fn)(critic_params, mini_batch)
165  updates, opt_state = optimizer(grads, opt_state)
166  critic_params = optax.apply_updates(critic_params, updates)
167  losses.append(loss)
168  train_state = deepcopy(train_state)
169  state = TrainState(
170  policy_params=train_state.policy_params,
171  policy_opt_states=train_state.policy_opt_states,
172  critic_params={**train_state.critic_params, agent_id: critic_params},
173  critic_opt_states={
174  **train_state.critic_opt_states,
175  agent_id: opt_state,
176  },
177  )
178  return state, {'loss': jnp.mean(jnp.array(losses))}
179 
180  return update

183 def get_dice_update_fn(
184  agent_id: int,
185  rng: hk.PRNGSequence,
186  policy_network: hk.Transformed,
187  critic_network: hk.Transformed,
188  optimizer: optax.TransformUpdateFn,
189  opp_pi_lr: float,
190  env: rl_environment.Environment,
191  n_lookaheads: int = 1,
192  gamma: float = 0.99,
193 ):
194  """Get the DiCE update function."""
195  def magic_box(x):
196  return jnp.exp(x - jax.lax.stop_gradient(x))
197 
198  @jax.jit
199  @partial(jax.vmap, in_axes=(None, 0, 0))
200  def get_action(params, s, rng_key):
201  pi = policy_network.apply(params, s)
202  action = pi.sample(seed=rng_key)
203  return action
204 
205  def rollout(params, other_params):
206  states, rewards, actions = [], [], []
207  step = env.reset()
208  batch_size = (
209  step.observations['batch_size']
210  if 'batch_size' in step.observations
211  else 1
212  )
213  while not step.last():
214  obs = step.observations
215  s_1, s_2 = jnp.array(obs['info_state'][0]), jnp.array(
216  obs['info_state'][1]
217  )
218  if batch_size == 1:
219  s_1, s_2 = s_1[None, :], s_2[None, :]
220  a_1 = get_action(params, s_1, jax.random.split(next(rng), num=batch_size))
221  a_2 = get_action(
222  other_params, s_2, jax.random.split(next(rng), num=batch_size)
223  )
224  a = jnp.stack([a_1, a_2], axis=1)
225  step = env.step(a.squeeze())
226  r_1, r_2 = jnp.array(step.rewards[0]), jnp.array(step.rewards[1])
227  if batch_size == 1:
228  r_1, r_2 = r_1[None], r_2[None]
229  actions.append(a.T)
230  states.append(jnp.stack([s_1, s_2], axis=0))
231  rewards.append(jnp.stack([r_1, r_2], axis=0))
232  return {
233  'states': jnp.stack(states, axis=2),
234  'rewards': jnp.stack(rewards, axis=2),
235  'actions': jnp.stack(actions, axis=2),
236  }

238  def dice_correction(train_state: TrainState):
239  """Computes the dice update for the given train state.
240 
241  Args:
242  train_state: The current train state.
243 
244  Returns:
245  The updated train state with the new policy params and metrics dict.
246  """
247 
248  @jax.jit
249  def dice_objective(params, other_params, states, actions, rewards, values):
250  self_logprobs = vmap(
251  vmap(lambda s, a: policy_network.apply(params, s).log_prob(a))
252  )(states[0], actions[0])
253  other_logprobs = vmap(
254  vmap(lambda s, a: policy_network.apply(other_params, s).log_prob(a))
255  )(states[1], actions[1])
256  # apply discount:
257  cum_discount = jnp.cumprod(gamma * jnp.ones_like(rewards), axis=1) / gamma
258  discounted_rewards = rewards * cum_discount
259  discounted_values = values.squeeze() * cum_discount
260 
261  # stochastics nodes involved in rewards dependencies:
262  dependencies = jnp.cumsum(self_logprobs + other_logprobs, axis=1)
263  # logprob of each stochastic nodes:
264  stochastic_nodes = self_logprobs + other_logprobs
265  # dice objective:
266  dice_objective = jnp.mean(
267  jnp.sum(magic_box(dependencies) * discounted_rewards, axis=1)
268  )
269  baseline_term = jnp.mean(
270  jnp.sum((1 - magic_box(stochastic_nodes)) * discounted_values, axis=1)
271  )
272  dice_objective = dice_objective + baseline_term
273  return -dice_objective # want to minimize -objective
274 
275  def outer_update(params, opp_params, agent_id, opp_id):
276  other_theta = opp_params
277  for _ in range(n_lookaheads):
278  trajectories = rollout(other_theta, params)
279  other_grad = jax.grad(dice_objective)(
280  other_theta,
281  other_params=params,
282  states=trajectories['states'],
283  actions=trajectories['actions'],
284  rewards=trajectories['rewards'][0],
285  values=critic_network.apply(
286  train_state.critic_params[opp_id], trajectories['states'][0]
287  ),
288  )
289  # Update the other player's policy:
290  other_theta = jax.tree_util.tree_map(
291  lambda param, grad: param - opp_pi_lr * grad,
292  other_theta,
293  other_grad,
294  )
295 
296  trajectories = rollout(params, other_theta)
297  values = critic_network.apply(
298  train_state.critic_params[agent_id], trajectories['states'][0]
299  )
300  loss = dice_objective(
301  params=params,
302  other_params=other_theta,
303  states=trajectories['states'],
304  actions=trajectories['actions'],
305  rewards=trajectories['rewards'][0],
306  values=values,
307  )
308  return loss, {'loss': loss}
309 
310  opp = 1 - agent_id
311  grads, metrics = grad(outer_update, has_aux=True)(
312  train_state.policy_params[agent_id],
313  opp_params=train_state.policy_params[opp],
314  agent_id=agent_id,
315  opp_id=opp,
316  )
317  return grads, metrics

381  def flat_params(
382  params,
383  ) -> typing.Tuple[
384  typing.Dict[str, jnp.ndarray], typing.Dict[typing.Any, typing.Callable]
385  ]:
386  """Flattens the policy parameters.
387 
388  Flattens the parameters of the policy network into a single vector and
389  returns the unravel function.
390 
391  Args:
392  params: The policy parameters.
393 
394  Returns:
395  A tuple (flat_params, unravel_fn)
396  """
397  flat_param_dict = {
398  agent_id: jax.flatten_util.ravel_pytree(p)
399  for agent_id, p in params.items()
400  }
401 
402  params = dict((k, flat_param_dict[k][0]) for k in flat_param_dict)
403  unravel_fns = dict((k, flat_param_dict[k][1]) for k in flat_param_dict)
404  return params, unravel_fns

406  def lola_correction(
407  train_state: TrainState, batch: TransitionBatch
408  ) -> hk.Params:
409  """Computes the LOLA correction term.
410 
411  Args:
412  train_state: The agent's current train state.
413  batch: A transition batch.
414 
415  Returns:
416  The LOLA correction term.
417  """
418  a_t, o_t, r_t, values = (
419  batch.action,
420  batch.info_state,
421  batch.reward,
422  batch.values,
423  )
424  params, unravel_fns = flat_params(train_state.policy_params)
425 
426  compute_returns = partial(rlax.lambda_returns, lambda_=0.0)
427  g_t = vmap(vmap(compute_returns))(
428  r_t=r_t, v_t=values, discount_t=jnp.full_like(r_t, gamma)
429  )
430  g_t = (g_t - g_t.mean()) / (g_t.std() + 1e-8)
431 
432  def log_pi(params, i, a_t, o_t):
433  return policy_network.apply(unravel_fns[i](params), o_t).log_prob(a_t)
434 
435  opp_id = 1 - agent_id
436 
437  def cross_term(a_t, o_t, r_t):
438  """Computes the second order correction term of the LOLA update.
439 
440  Args:
441  a_t: actions of both players
442  o_t: observations of both players
443  r_t: rewards of both players
444 
445  Returns:
446  The second order correction term.
447  """
448  grad_log_pi = vmap(jax.value_and_grad(log_pi), in_axes=(None, None, 0, 0))
449  log_probs, grads = grad_log_pi(
450  params[agent_id], agent_id, a_t[agent_id], o_t[agent_id]
451  )
452  opp_logrpobs, opp_grads = grad_log_pi(
453  params[opp_id], opp_id, a_t[opp_id], o_t[opp_id]
454  )
455  grads = grads.cumsum(axis=0)
456  opp_grads = opp_grads.cumsum(axis=0)
457  log_probs = log_probs.cumsum(axis=0)
458  opp_logrpobs = opp_logrpobs.cumsum(axis=0)
459  cross_term = 0.0
460  for t in range(0, len(a_t[agent_id])):
461  discounted_reward = r_t[opp_id, t] * jnp.power(gamma, t)
462  cross_term += (
463  discounted_reward
464  * jnp.outer(grads[t], opp_grads[t])
465  * jnp.exp(log_probs[t] + opp_logrpobs[t])
466  )
467  return cross_term # * jnp.exp(log_probs.sum() + opp_logrpobs.sum())
468 
469  def policy_gradient(a_t, o_t, g_t):
470  grad_log_pi = vmap(grad(log_pi), in_axes=(None, None, 0, 0))
471  opp_grads = grad_log_pi(params[opp_id], opp_id, a_t[opp_id], o_t[opp_id])
472  pg = g_t[agent_id] @ opp_grads
473  return pg
474 
475  cross = vmap(cross_term, in_axes=(1, 1, 1))(a_t, o_t, r_t).mean(axis=0)
476  pg = vmap(policy_gradient, in_axes=(1, 1, 1))(a_t, o_t, g_t).mean(axis=0)
477  correction = -pi_lr * (pg @ cross)
478  return unravel_fns[agent_id](correction)

480  def policy_loss(params, agent_id, batch):
481  """Computes the policy gradient loss.
482 
483  Args:
484  params: The policy parameters.
485  agent_id: The agent's id.
486  batch: A transition batch.
487 
488  Returns:
489  The policy gradient loss.
490  """
491  a_t, o_t, r_t, values = (
492  batch.action[agent_id],
493  batch.info_state[agent_id],
494  batch.reward[agent_id],
495  batch.values[agent_id],
496  )
497  logits_t = vmap(vmap(lambda s: policy_network.apply(params, s).logits))(o_t)
498  discount = jnp.full(r_t.shape, gamma)
499  returns = vmap(rlax.lambda_returns)(
500  r_t=r_t,
501  v_t=values,
502  discount_t=discount,
503  lambda_=jnp.ones_like(discount),
504  )
505  adv_t = returns - values
506  loss = vmap(rlax.policy_gradient_loss)(
507  logits_t=logits_t, a_t=a_t, adv_t=adv_t, w_t=jnp.ones_like(adv_t)
508  )
509  return loss.mean()

511  def update(
512  train_state: TrainState, batch: TransitionBatch
513  ) -> typing.Tuple[TrainState, typing.Dict]:
514  """Updates the policy parameters in train_state.
515 
516  If lola_weight > 0, the correction term by Foerster et al. will be applied.
517 
518  Args:
519  train_state: the agent's train state.
520  batch: a transition batch
521 
522  Returns:
523  A tuple (new_train_state, metrics)
524  """
525  loss, policy_grads = jax.value_and_grad(policy_loss)(
526  train_state.policy_params[agent_id], agent_id, batch
527  )
528  correction = lola_correction(train_state, batch)
529  policy_grads = jax.tree_util.tree_map(
530  lambda grad, corr: grad - lola_weight * corr, policy_grads, correction
531  )
532  updates, opt_state = optimizer(
533  policy_grads, train_state.policy_opt_states[agent_id]
534  )
535  policy_params = optax.apply_updates(
536  train_state.policy_params[agent_id], updates
537  )
538  train_state = TrainState(
539  policy_params={**train_state.policy_params, agent_id: policy_params},
540  policy_opt_states={
541  **train_state.policy_opt_states,
542  agent_id: opt_state,
543  },
544  critic_params=deepcopy(train_state.critic_params),
545  critic_opt_states=deepcopy(train_state.critic_opt_states),
546  )
547  return train_state, {'loss': loss}
548 
549  return update

552 def get_opponent_update_fn(
553  agent_id: int,
554  policy_network: hk.Transformed,
555  optimizer: optax.TransformUpdateFn,
556  num_minibatches: int = 1,
557 ) -> UpdateFn:
558  """Get the opponent update function."""
559  def loss_fn(params, batch: TransitionBatch):
560  def loss(p, states, actions):
561  log_prob = policy_network.apply(p, states).log_prob(actions)
562  return log_prob
563 
564  log_probs = vmap(vmap(loss, in_axes=(None, 0, 0)), in_axes=(None, 0, 0))(
565  params, batch.info_state[agent_id], batch.action[agent_id]
566  )
567  return -log_probs.sum(axis=-1).mean()

571  ) -> typing.Tuple[TrainState, typing.Dict]:

576  loss, policy_grads = jax.value_and_grad(loss_fn)(
577  policy_params, mini_batch
578  )
579  updates, opt_state = optimizer(policy_grads, opt_state)
580  policy_params = optax.apply_updates(
581  train_state.policy_params[agent_id], updates
582  )
583 
584  train_state = TrainState(
585  policy_params={**train_state.policy_params, agent_id: policy_params},

595  return update
