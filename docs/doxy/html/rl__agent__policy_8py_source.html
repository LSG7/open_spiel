23class JointRLAgentPolicy(policy.Policy):
24 """Joint policy denoted by the RL agents of a game.
25
26 Given a list of RL agents of players for a game, this class can be used derive
27 the corresponding (joint) policy. In particular, the distribution over
28 possible actions will be those that are returned by the step() method of
29 the RL agents given the state.
30 """
31
32 def __init__(self, game, agents: Dict[int, rl_agent.AbstractAgent],
33 use_observation: bool):
34 """Initializes the joint RL agent policy.
35
36 Args:
37 game: The game.
38 agents: Dictionary of agents keyed by the player IDs.
39 use_observation: If true then observation tensor will be used as the
40 `info_state` in the step() calls; otherwise, information state tensor
41 will be used. See `use_observation` property of
42 rl_environment.Environment.
43 """
44 player_ids = list(sorted(agents.keys()))
45 super().__init__(game, player_ids)
46 self._agents = agents
47 self._obs = {
48 "info_state": [None] * game.num_players(),
49 "legal_actions": [None] * game.num_players()
50 }
51 self._use_observation = use_observation
52
53 def action_probabilities(self, state, player_id=None):
54 if state.is_simultaneous_node():
55 assert player_id is not None, "Player ID should be specified."
56 else:
57 if player_id is None:
58 player_id = state.current_player()
59 else:
60 assert player_id == state.current_player()
61
62 # Make sure that player_id is an integer and not an enum as it is used to
63 # index lists.
64 player_id = int(player_id)
65
66 legal_actions = state.legal_actions(player_id)
67
68 self._obs["current_player"] = player_id
69 self._obs["info_state"][player_id] = (
70 state.observation_tensor(player_id)
71 if self._use_observation else state.information_state_tensor(player_id))
72 self._obs["legal_actions"][player_id] = legal_actions
73
74 info_state = rl_environment.TimeStep(
75 observations=self._obs, rewards=None, discounts=None, step_type=None)
76
77 p = self._agents[player_id].step(info_state, is_evaluation=True).probs
78 prob_dict = {action: p[action] for action in legal_actions}
79 return prob_dict

82class RLAgentPolicy(JointRLAgentPolicy):
83 """A policy for a specific agent trained in an RL environment."""
84
85 def __init__(self, game, agent: rl_agent.AbstractAgent, player_id: int,
86 use_observation: bool):
87 """Initializes the RL agent policy.
88
89 Args:
90 game: The game.
91 agent: RL agent.
92 player_id: ID of the player.
93 use_observation: See JointRLAgentPolicy above.
94 """
95 self._player_id = player_id
96 super().__init__(game, {player_id: agent}, use_observation)
97
98 def action_probabilities(self, state, player_id=None):
99 return super().action_probabilities(
100 state, self._player_id if player_id is None else player_id)
