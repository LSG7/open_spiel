28class GameStats:

44def traverse_game_tree(game: pyspiel.Game,
45 state: pyspiel.State,
46 game_stats: GameStats):
47 """Traverses the game tree, collecting information about the game."""
48
49 if state.is_terminal():
50 game_stats.num_terminals += 1
51 elif state.is_chance_node():
52 game_stats.num_chance_nodes += 1
53 for outcome in state.legal_actions():
54 child = state.child(outcome)
55 traverse_game_tree(game, child, game_stats)
56 elif state.is_simultaneous_node():
57 game_stats.num_simultaneous_nodes += 1
58 # Using joint actions for convenience. Can use legal_actions(player) to
59 # and state.apply_actions when walking over individual players
60 for joint_action in state.legal_actions():
61 child = state.child(joint_action)
62 traverse_game_tree(game, child, game_stats)
63 else:
64 game_stats.num_decision_nodes += 1
65 legal_actions = state.legal_actions()
66 if game.get_type().provides_information_state_string:
67 game_stats.info_state_dict[
68 state.information_state_string()] = legal_actions
69 for action in state.legal_actions():
70 # print(f"Decision node: \n {state}")
71 # print(f"Taking action {action} ({state.action_to_string(action)}")
72 child = state.child(action)
73 traverse_game_tree(game, child, game_stats)
