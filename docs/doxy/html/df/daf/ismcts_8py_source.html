109 def run_search(self, state):
110 self.reset()
111 assert state.get_game().get_type(
112 ).dynamics == pyspiel.GameType.Dynamics.SEQUENTIAL
113 assert state.get_game().get_type(
114 ).information == pyspiel.GameType.Information.IMPERFECT_INFORMATION
115
116 legal_actions = state.legal_actions()
117 if len(legal_actions) == 1:
118 return [(legal_actions[0], 1.0)]
119
120 self._root_node = self.create_new_node(state)
121
122 assert self._root_node
123
124 root_infostate_key = self.get_state_key(state)
125
126 for _ in range(self._max_simulations):
127 # how to sample a pyspiel.state from another pyspiel.state?
128 sampled_root_state = self.sample_root_state(state)
129 assert root_infostate_key == self.get_state_key(sampled_root_state)
130 assert sampled_root_state
131 self.run_simulation(sampled_root_state)
132
133 if self._allow_inconsistent_action_sets: # when this happens?
134 legal_actions = state.legal_actions()
135 temp_node = self.filter_illegals(self._root_node, legal_actions)
136 assert temp_node.total_visits > 0
137 return self.get_final_policy(state, temp_node)
138 else:
139 return self.get_final_policy(state, self._root_node)

154 def get_final_policy(self, state, node):
155 assert node
156 if self._final_policy_type == ISMCTSFinalPolicyType.NORMALIZED_VISITED_COUNT:
157 assert node.total_visits > 0
158 total_visits = node.total_visits
159 policy = [(action, child.visits / total_visits)
160 for action, child in node.child_info.items()]
161 elif self._final_policy_type == ISMCTSFinalPolicyType.MAX_VISIT_COUNT:
162 assert node.total_visits > 0
163 max_visits = -float('inf')
164 count = 0
165 for action, child in node.child_info.items():
166 if child.visits == max_visits:
167 count += 1
168 elif child.visits > max_visits:
169 max_visits = child.visits
170 count = 1
171 policy = [(action, 1. / count if child.visits == max_visits else 0.0)
172 for action, child in node.child_info.items()]
173 elif self._final_policy_type == ISMCTSFinalPolicyType.MAX_VALUE:
174 assert node.total_visits > 0
175 max_value = -float('inf')
176 count = 0
177 for action, child in node.child_info.items():
178 if child.value() == max_value:
179 count += 1
180 elif child.value() > max_value:
181 max_value = child.value()
182 count = 1
183 policy = [(action, 1. / count if child.value() == max_value else 0.0)
184 for action, child in node.child_info.items()]
185
186 policy_size = len(policy)
187 legal_actions = state.legal_actions()
188 if policy_size < len(legal_actions): # do we really need this step?
189 for action in legal_actions:
190 if action not in node.child_info:
191 policy.append((action, 0.0))
192 return policy

194 def sample_root_state(self, state):
195 if self._max_world_samples == UNLIMITED_NUM_WORLD_SAMPLES:
196 return self.resample_from_infostate(state)
197 elif len(self._root_samples) < self._max_world_samples:
198 self._root_samples.append(self.resample_from_infostate(state))
199 return self._root_samples[-1].clone()
200 elif len(self._root_samples) == self._max_world_samples:
201 idx = self._random_state.randint(len(self._root_samples))
202 return self._root_samples[idx].clone()
203 else:
204 raise pyspiel.SpielError(
205 'Case not handled (badly set max_world_samples..?)')

236 def filter_illegals(self, node, legal_actions):
237 new_node = copy.deepcopy(node)
238 for action, child in node.child_info.items():
239 if action not in legal_actions:
240 new_node.total_visits -= child.visits

244 def expand_if_necessary(self, node, action):

248 def select_action_tree_policy(self, node, legal_actions):
249 if self._allow_inconsistent_action_sets:
250 temp_node = self.filter_illegals(node, legal_actions)
251 if temp_node.total_visits == 0:
252 action = legal_actions[self._random_state.randint(
253 len(legal_actions))] # prior?
254 self.expand_if_necessary(node, action)
255 return action
256 else:
257 return self.select_action(temp_node)
258 else:
259 return self.select_action(node)

261 def select_action(self, node):
262 candidates = []
263 max_value = -float('inf')
264 for action, child in node.child_info.items():
265 assert child.visits > 0
266
267 action_value = child.value()
268 if self._child_selection_policy == ChildSelectionPolicy.UCT:
269 action_value += (self._uct_c *
270 np.sqrt(np.log(node.total_visits)/child.visits))
271 elif self._child_selection_policy == ChildSelectionPolicy.PUCT:
272 action_value += (self._uct_c * child.prior *
273 np.sqrt(node.total_visits)/(1 + child.visits))
274 else:
275 raise pyspiel.SpielError('Child selection policy unrecognized.')
276 if action_value > max_value + TIE_TOLERANCE:
277 candidates = [action]
278 max_value = action_value
279 elif (action_value > max_value - TIE_TOLERANCE and
280 action_value < max_value + TIE_TOLERANCE):
281 candidates.append(action)
282 max_value = action_value
283
284 assert len(candidates) >= 1
285 return candidates[self._random_state.randint(len(candidates))]

287 def check_expand(self, node, legal_actions):
288 if not self._allow_inconsistent_action_sets and len(
289 node.child_info) == len(legal_actions):
290 return pyspiel.INVALID_ACTION
291 legal_actions_copy = copy.deepcopy(legal_actions)
292 self._random_state.shuffle(legal_actions_copy)
293 for action in legal_actions_copy:
294 if action not in node.child_info:
295 return action
296 return pyspiel.INVALID_ACTION

298 def run_simulation(self, state):
299 if state.is_terminal():
300 return state.returns()
301 elif state.is_chance_node():
302 action_list, prob_list = zip(*state.chance_outcomes())
303 chance_action = self._random_state.choice(action_list, p=prob_list)
304 state.apply_action(chance_action)
305 return self.run_simulation(state)
306 legal_actions = state.legal_actions()
307 cur_player = state.current_player()
308 node = self.lookup_or_create_node(state)
309
310 assert node
311
312 if node.total_visits == UNEXPANDED_VISIT_COUNT:
313 node.total_visits = 0
314 for action, prob in self._evaluator.prior(state):
315 node.prior_map[action] = prob
316 return self._evaluator.evaluate(state)
317 else:
318 chosen_action = self.check_expand(
319 node, legal_actions) # add one children at a time?
320 if chosen_action != pyspiel.INVALID_ACTION:
321 # check if all actions have been expanded, if not, select one?
322 # if yes, ucb?
323 self.expand_if_necessary(node, chosen_action)
324 else:
325 chosen_action = self.select_action_tree_policy(node, legal_actions)
326
327 assert chosen_action != pyspiel.INVALID_ACTION
328
329 node.total_visits += 1
330 node.child_info[chosen_action].visits += 1
331 state.apply_action(chosen_action)
332 returns = self.run_simulation(state)
333 node.child_info[chosen_action].return_sum += returns[cur_player]
334 return returns
