137 // Returns the maximum capacity for a given block_size and requested size.
138 size_t MaxCapacityFor(size_t block_size, size_t requested) {
139  requested = (std::min)(requested, cord_internal::kMaxLargeFlatSize);
140  // Maximum returned size is always capped at block_size - kFlatOverhead.
141  return block_size - kFlatOverhead;
142 }
