24 // Transforms a co-operative game into a 1-player environment, in which
25 // actions build a policy in the underlying game.
26 //
27 // We make very strong assumptions about the underlying game:
28 // - The initial num_players actions must be chance actions, one per player.
29 // These are assumed to map 1:1 to the private state for that player.
30 // - The public state of the game is determined solely by the last non-chance
31 // action.
32 //
33 // These assumptions hold for tiny_hanabi and tiny_bridge_2p, but are unlikely
34 // to hold in other games.

39 // Information we have about each player's private state.
40 struct PlayerPrivate {
41  // Each private state may either - have a valid action assigned, be waiting
42  // for an action assignment, or have been ruled out by prior play.
43  static inline constexpr Action kImpossible = -100;
44  static inline constexpr Action kUnassigned = -99;
45  std::vector<Action> assignments;
46  int next_unassigned; // index into assignments
47 
48  // Name of each private state.
49  std::vector<std::string> names;
50 
51  PlayerPrivate(int num_privates)
52  : assignments(num_privates, kUnassigned),
53  next_unassigned(0),
54  names(num_privates) {}
55 
56  // Assign the next unassigned private.
57  void Assign(Action action) {
58  assignments[next_unassigned++] = action;
59  while (next_unassigned < assignments.size() &&
60  assignments[next_unassigned] != kUnassigned)
61  ++next_unassigned;
62  }
63 
64  // Have all assignments been made?
65  bool AssignmentsComplete() const {
66  return next_unassigned == assignments.size();
67  }
68 
69  // Reset assignments for the next action choice.
70  void Reset(Action action) {
71  next_unassigned = assignments.size();
72  for (int i = 0; i < assignments.size(); ++i) {
73  if (assignments[i] == action) {
74  if (next_unassigned == assignments.size()) next_unassigned = i;
75  assignments[i] = kUnassigned;
76  } else {
77  assignments[i] = kImpossible;
78  }
79  }
80  }
81 };
