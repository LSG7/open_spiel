37 def thresholded(logits, regrets, threshold=2.0):
38  """Zeros out `regrets` where `logits` are too negative or too large."""
39  can_decrease = tf.cast(tf.greater(logits, -threshold), tf.float32)
40  can_increase = tf.cast(tf.less(logits, threshold), tf.float32)
41  regrets_negative = tf.minimum(regrets, 0.0)
42  regrets_positive = tf.maximum(regrets, 0.0)
43  return can_decrease * regrets_negative + can_increase * regrets_positive

46 @tf.function
47 def train(model,
48  data,
49  batch_size,
50  step_size=1.0,
51  threshold=2.0,
52  random_shuffle_size=None,
53  autoencoder_loss=None):
54  """Train NeuRD `model` on `data`."""
55  if random_shuffle_size is None:
56  random_shuffle_size = 10 * batch_size
57  data = data.shuffle(random_shuffle_size)
58  data = data.batch(batch_size)
59  data = data.repeat(1)
60 
61  for x, regrets in data:
62  with tf.GradientTape() as tape:
63  output = model(x, training=True)
64  logits = output[:, :1]
65  logits = logits - tf.reduce_mean(logits, keepdims=True)
66 
67  regrets = tf.stop_gradient(
68  thresholded(logits, regrets, threshold=threshold))
69  utility = tf.reduce_mean(logits * regrets)
70 
71  if autoencoder_loss is not None:
72  utility = utility - autoencoder_loss(x, output[:, 1:])
73 
74  grad = tape.gradient(utility, model.trainable_variables)
75 
76  for i, var in enumerate(model.trainable_variables):
77  var.assign_add(step_size * grad[i])

114  use_skip_connections: Whether or not to apply skip connections (layer
115  output = layer(x) + x) on hidden layers. Zero padding or truncation is
116  used to match the number of columns on layer inputs and outputs.

119  being called. Defaults to `False`.

123  self._use_skip_connections_use_skip_connections = use_skip_connections
124  self._hidden_are_factored_hidden_are_factored = num_hidden_factors > 0
