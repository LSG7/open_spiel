26// An implementation of the classic: https://en.wikipedia.org/wiki/Backgammon
27// using rule set from
28// http://usbgf.org/learn-backgammon/backgammon-rules-and-terms/rules-of-backgammon/
29// where red -> 'x' (player 0) and white -> 'o' (player 1).
30//
31// Currently does not support the doubling cube nor "matches" (multiple games
32// where outcomes are scored and tallied to 21).
33//
34// Parameters:
35// "hyper_backgammon" bool Use Hyper-backgammon variant [1] (def: false)
36// "scoring_type" string Type of scoring for the game: "winloss_scoring"
37// (default), "enable_gammons", or "full_scoring"
38//
39// [1] https://bkgm.com/variants/HyperBackgammon.html. Hyper-backgammon is a
40// simplified backgammon start setup which is small enough to solve. Note that
41// it is not the full Hyper-backgammon sinc do not have cube is not implemented.
42

64// The action encoding stores a number in { 0, 1, ..., 1351 }. If the high
65// roll is to move first, then the number is encoded as a 2-digit number in
66// base 26 ({0, 1, .., 23, kBarPos, Pass}) (=> first 676 numbers). Otherwise,
67// the low die is to move first and, 676 is subtracted and then again the
68// number is encoded as a 2-digit number in base 26.

78// Game scoring type, whether to score gammons/backgammons specially.
79enum class ScoringType {
80 kWinLossScoring, // "winloss_scoring": Score only 1 point per player win.
81 kEnableGammons, // "enable_gammons": Score 2 points for a "gammon".
82 kFullScoring, // "full_scoring": Score gammons as well as 3 points for a
83 // "backgammon".
84};

86struct CheckerMove {
87 // Pass is encoded as (pos, num, hit) = (-1, -1, false).
88 int pos; // 0-24 (0-23 for locations on the board and kBarPos)
89 int num; // 1-6
90 bool hit;
91 CheckerMove(int _pos, int _num, bool _hit)
92 : pos(_pos), num(_num), hit(_hit) {}
93 bool operator<(const CheckerMove& rhs) const {
94 return (pos * 6 + (num - 1)) < (rhs.pos * 6 + rhs.num - 1);
95 }
96};

271 }
272

274 // the doubles).

276

278 int MaxGameLength() const override { return 1000; }
279

281 // determining starting player.
282 int MaxChanceNodesInHistory() const override { return MaxGameLength() + 1; }
283
284 int NumPlayers() const override { return 2; }
285 double MinUtility() const override { return -MaxUtility(); }
286 absl::optional<double> UtilitySum() const override { return 0; }
287 double MaxUtility() const override;
288
289 std::vector<int> ObservationTensorShape() const override {
290 // Encode each point on the board as four doubles:
291 // - One double for whether there is one checker or not (1 or 0).
292 // - One double for whether there are two checkers or not (1 or 0).
293 // - One double for whether there are three checkers or not (1 or 0).
294 // - One double if there are more than 3 checkers, the number of checkers.
295 // more than three that are on that point.
296 //
297 // Return a vector encoding:
298 // Every point listed for the current player.
299 // Every point listed for the opponent.
300 // One double for the number of checkers on the bar for the current player.
301 // One double for the number of checkers scored for the current player.
302 // One double for whether it's the current player's turn (1 or 0).
303 // One double for the number of checkers on the bar for the opponent.
304 // One double for the number of checkers scored for the opponent.
305 // One double for whether it's the opponent's turn (1 or 0).
306 // One double for the first dice's value.
307 // One double for the second dice's value.
308
309 return {kStateEncodingSize};
310 }
311
312 int NumCheckersPerPlayer() const;
313
314 private:
315 ScoringType scoring_type_; // Which rules apply when scoring the game.
316 bool hyper_backgammon_; // Is hyper-backgammon variant enabled?
317};
