51 def _recursive_check(self, g, h):
52 self.assertEqual(g.current_player(), h.current_player())
53 self.assertEqual(g.is_chance_node(), h.is_chance_node())
54 self.assertEqual(g.is_terminal(), h.is_terminal())
55 if g.is_terminal():
56 self.assertEqual(g.returns(), h.returns())
57 return
58
59 if g.is_chance_node():
60 self.assertEqual(g.chance_outcomes(), h.chance_outcomes())
61 else:
62 self.assertEqual(g.legal_actions(), h.legal_actions())
63 self._infoset_table_orig[g.information_state_string()].append(g.history())
64 self._infoset_table_efg[h.information_state_string()].append(h.history())
65
66 for a, b in zip(g.legal_actions(), h.legal_actions()):
67 self._recursive_check(g.child(a), h.child(b))

69 def _check_infoset_isomorphism(self, a, b):
70 a_prime = []
