15"""This compares the speed/results of Jax CFR to of the original impl of CFR.
16
17The results slightly differ due to different rounding of regrets between
18original implmentation and CFR. When setting clamping of regrets to 1e-8 the
19results are exactly the same.
20"""

26from open_spiel.python.algorithms.best_response import BestResponsePolicy

32def compare_cfr_with_jax_cfr(game):
33 """Do the comparison."""
34
35 start = time.time()
36 jax_cfr = JaxCFR(game)
37 print(time.time() - start)
38 jax_cfr.multiple_steps(10000)
39 print(time.time() - start)
40
41 start = time.time()
42 print(time.time() - start)
43 cfr = CFRPlusSolver(game)
44 for _ in range(1000):
45 cfr.evaluate_and_update_policy()
46
47 print(time.time() - start)
48
49 jax_strat = jax_cfr.average_policy()
50 jax_br1 = BestResponsePolicy(jax_cfr.game, 1, jax_strat)
51 jax_br2 = BestResponsePolicy(jax_cfr.game, 0, jax_strat)
52
53 cfr_strat = jax_cfr.average_policy()
54 cfr_br1 = BestResponsePolicy(jax_cfr.game, 1, cfr_strat)
55 cfr_br2 = BestResponsePolicy(jax_cfr.game, 0, cfr_strat)
56
57 print("Jax P1: ", jax_br1.value(jax_cfr.game.new_initial_state()))
58 print("CFR P1: ", cfr_br1.value(jax_cfr.game.new_initial_state()))
59 print("Jax P2: ", jax_br2.value(jax_cfr.game.new_initial_state()))
60 print("CFR P2: ", cfr_br2.value(jax_cfr.game.new_initial_state()))
