33void XinxinBot_BasicPlayGame() {
34 int num_games = 5;
35 std::mt19937 rng(Seed());
36 auto game = open_spiel::LoadGame("hearts");
37 std::vector<std::unique_ptr<open_spiel::Bot>> bots;
38 std::vector<Bot *> bot_ptrs;
39
40 for (int i = 0; i < hearts::kNumPlayers; i++) {
41 bots.push_back(open_spiel::hearts::MakeXinxinBot(game->GetParameters()));
42 bot_ptrs.push_back(bots.back().get());
43 }
44
45 for (int i = 0; i < num_games; i++) {
46 std::unique_ptr<open_spiel::State> state = game->NewInitialState();
47 EvaluateBots(state.get(), bot_ptrs,
48 absl::Uniform<int>(rng, 0, std::numeric_limits<int>::max()));
49 // call xinxinbot with terminal state so that xinxin's internal state's
50 // returns can be checked against the OpenSpiel returns
51 for (auto bot : bot_ptrs)
52 bot->InformAction(*state, kTerminalPlayerId, kInvalidAction);
53 }
54}
