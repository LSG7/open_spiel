15"""Visualizing game trees with graphviz.
16
17GameTree builds a `pygraphviz.AGraph` reprensentation of the game tree. The
18resulting tree can be directly visualized in Jupyter notebooks or Google Colab
19via SVG plotting - or written to a file by calling `draw(filename, prog="dot")`.
20
21See `examples/treeviz_example.py` for a more detailed example.
22
23This module relies on external dependencies, which need to be installed before
24use. On a debian system follow these steps:
25```
26sudo apt-get install graphviz libgraphviz-dev
27pip install pygraphviz
28```
29"""

44_PLAYER_SHAPES = {0: "square", 1: "ellipse"}
45_PLAYER_COLORS = {-1: "black", 0: "blue", 1: "red"}
46_FONTSIZE = 8
47_WIDTH = _HEIGHT = 0.25
48_ARROWSIZE = .5
49_MARGIN = 0.01

52def default_node_decorator(state):
53 """Decorates a state-node of the game tree.
54
55 This method can be called by a custom decorator to prepopulate the attributes
56 dictionary. Then only relevant attributes need to be changed, or added.
57
58 Args:
59 state: The state.
60
61 Returns:
62 `dict` with graphviz node style attributes.
63 """
64 player = state.current_player()
65 attrs = {
66 "label": "",
67 "fontsize": _FONTSIZE,
68 "width": _WIDTH,
69 "height": _HEIGHT,
70 "margin": _MARGIN
71 }
72 if state.is_terminal():
73 attrs["label"] = ", ".join(map(str, state.returns()))
74 attrs["shape"] = "diamond"
75 elif state.is_chance_node():
76 attrs["shape"] = "point"
77 attrs["width"] = _WIDTH / 2.
78 attrs["height"] = _HEIGHT / 2.
79 else:
80 attrs["label"] = str(state.information_state_string())
81 attrs["shape"] = _PLAYER_SHAPES.get(player, "ellipse")
82 attrs["color"] = _PLAYER_COLORS.get(player, "black")
83 return attrs

86def default_edge_decorator(parent, unused_child, action):
87 """Decorates a state-node of the game tree.
88
89 This method can be called by a custom decorator to prepopulate the attributes
90 dictionary. Then only relevant attributes need to be changed, or added.
91
92 Args:
93 parent: The parent state.
94 unused_child: The child state, not used in the default decorator.
95 action: `int` the selected action in the parent state.
96
97 Returns:
98 `dict` with graphviz node style attributes.
99 """
100 player = parent.current_player()
101 attrs = {
102 "label": " " + parent.action_to_string(player, action),
103 "fontsize": _FONTSIZE,
104 "arrowsize": _ARROWSIZE
105 }
106 attrs["color"] = _PLAYER_COLORS.get(player, "black")
107 return attrs

110class GameTree(pygraphviz.AGraph):
111 """Builds `pygraphviz.AGraph` of the game tree.
112
113 Attributes:
114 game: A `pyspiel.Game` object.
115 depth_limit: Maximum depth of the tree. Optional, default=-1 (no limit).
116 node_decorator: Decorator function for nodes (states). Optional, default=
117 `treeviz.default_node_decorator`.
118 edge_decorator: Decorator function for edges (actions). Optional, default=
119 `treeviz.default_edge_decorator`.
120 group_terminal: Whether to display all terminal states at same level,
121 default=False.
122 group_infosets: Whether to group infosets together, default=False.
123 group_pubsets: Whether to group public sets together, default=False.
124 target_pubset: Whether to group all public sets "*" or a specific one.
125 infoset_attrs: Attributes to style infoset grouping.
126 pubset_attrs: Attributes to style public set grouping.
127 kwargs: Keyword arguments passed on to `pygraphviz.AGraph.__init__`.
128 """
129
130 def __init__(self,
131 game=None,
132 depth_limit=-1,
133 node_decorator=default_node_decorator,
134 edge_decorator=default_edge_decorator,
135 group_terminal=False,
136 group_infosets=False,
137 group_pubsets=False,
138 target_pubset="*",
139 infoset_attrs=None,
140 pubset_attrs=None,
141 **kwargs):
142
143 kwargs["directed"] = kwargs.get("directed", True)
144 super(GameTree, self).__init__(**kwargs)
145
146 # We use pygraphviz.AGraph.add_subgraph to cluster nodes, and it requires a
147 # default constructor. Thus game needs to be optional.
148 if game is None:
149 return
150
151 self.game = game
152 self._node_decorator = node_decorator
153 self._edge_decorator = edge_decorator
154
155 self._group_infosets = group_infosets
156 self._group_pubsets = group_pubsets
157 if self._group_infosets:
158 if not self.game.get_type().provides_information_state_string:
159 raise RuntimeError(
160 "Grouping of infosets requested, but the game does not "
161 "provide information state string.")
162 if self._group_pubsets:
163 if not self.game.get_type().provides_factored_observation_string:
164 raise RuntimeError(
165 "Grouping of public sets requested, but the game does not "
166 "provide factored observations strings.")
167
168 self._infosets = collections.defaultdict(lambda: [])
169 self._pubsets = collections.defaultdict(lambda: [])
170 self._terminal_nodes = []
171
172 root = game.new_initial_state()
173 self.add_node(self.state_to_str(root), **self._node_decorator(root))
174 self._build_tree(root, 0, depth_limit)
175
176 for (player, info_state), sibblings in self._infosets.items():
177 cluster_name = "cluster_{}_{}".format(player, info_state)
178 self.add_subgraph(sibblings, cluster_name,
179 **(infoset_attrs or {
180 "style": "dashed"
181 }))
182
183 for pubset, sibblings in self._pubsets.items():
184 if target_pubset == "*" or target_pubset == pubset:
185 cluster_name = "cluster_{}".format(pubset)
186 self.add_subgraph(sibblings, cluster_name,
187 **(pubset_attrs or {
188 "style": "dashed"
189 }))
190
191 if group_terminal:
192 self.add_subgraph(self._terminal_nodes, rank="same")

194 def state_to_str(self, state):
195 """Unique string representation of a state.
196
197 Args:
198 state: The state.
199
200 Returns:
201 String representation of state.
202 """
203 assert not state.is_simultaneous_node()
204 # AGraph nodes can't have empty string == None as a key, thus we prepend " "
205 return " " + state.history_str()
206
207 def _build_tree(self, state, depth, depth_limit):
208 """Recursively builds the game tree."""
209 state_str = self.state_to_str(state)
210
211 if state.is_terminal():
212 self._terminal_nodes.append(state_str)
213 return
214 if depth > depth_limit >= 0:
215 return
216
217 for action in state.legal_actions():
218 child = state.child(action)
219 child_str = self.state_to_str(child)
220 self.add_node(child_str, **self._node_decorator(child))
221 self.add_edge(state_str, child_str,
222 **self._edge_decorator(state, child, action))
223
224 if (self._group_infosets and not child.is_chance_node() and
225 not child.is_terminal()):
226 player = child.current_player()
227 info_state = child.information_state_string()
228 self._infosets[(player, info_state)].append(child_str)
229
230 if self._group_pubsets:
231 pub_obs_history = str(pyspiel.PublicObservationHistory(child))
232 self._pubsets[pub_obs_history].append(child_str)
233
234 self._build_tree(child, depth + 1, depth_limit)
