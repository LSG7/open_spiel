35 def test_python_and_cpp_bot(self):
36 game = pyspiel.load_game("kuhn_poker")
37 bots = [
38 pyspiel.make_uniform_random_bot(0, 1234),
39 uniform_random.UniformRandomBot(1, np.random.RandomState(4321)),
40 ]
41 results = np.array([
42 pyspiel.evaluate_bots(game.new_initial_state(), bots, iteration)
43 for iteration in range(10000)
44 ])
45 average_results = np.mean(results, axis=0)
46 np.testing.assert_allclose(average_results, [0.125, -0.125], atol=0.1)

95 def test_roshambo_bot(self):
96 if hasattr(pyspiel, "make_roshambo_bot"):
97 game = pyspiel.load_game("repeated_game(stage_game=matrix_rps()," +
98 "num_repetitions=" +
99 f"{pyspiel.ROSHAMBO_NUM_THROWS})")
100 num_players = 2
101 bots = [
102 pyspiel.make_roshambo_bot(0, "rotatebot",
103 pyspiel.ROSHAMBO_NUM_THROWS),
104 pyspiel.make_roshambo_bot(1, "copybot", pyspiel.ROSHAMBO_NUM_THROWS)
105 ]
106 state = game.new_initial_state()
107 for i in range(pyspiel.ROSHAMBO_NUM_THROWS):
108 joint_action = [-1] * num_players
109 for p in range(num_players):
110 joint_action[p] = bots[p].step(state)
111 state.apply_actions(joint_action)
112 if i == 0:
113 # copybot wins the first round
114 self.assertListEqual(state.returns(), [-1, 1])
115 else:
116 # the rest are a draw
117 self.assertListEqual(state.rewards(), [0, 0])
118 self.assertTrue(state.is_terminal())
119 self.assertListEqual(state.returns(), [-1, 1])
