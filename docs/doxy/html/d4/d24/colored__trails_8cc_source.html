68bool IsLegalTrade(
69 const Board& board, const Trade& trade,
70 const std::vector<int>& proposer_chips,
71 const std::vector<int>& responder_chips) {
72 if (trade.giving.empty() || trade.receiving.empty()) {
73 // pass trade is always legal.
74 return true;
75 }
76
77 for (int i = 0; i < board.num_colors; ++i) {
78 if (trade.giving[i] > proposer_chips[i]) {
79 return false;
80 }
81
82 if (trade.receiving[i] > responder_chips[i]) {
83 return false;
84 }
85 }
86
87 // Try to reduce the trade, if it's not valid or not equal to the same trade
88 // then not a legal trade.
89 Trade copy = trade;
90 bool valid = copy.reduce();
91 return (valid && copy == trade);
92}

95std::vector<Action> GenerateLegalActionsForChips(
96 const ColoredTrailsGame* game,
97 const Board& board,
98 const std::vector<int>& player_chips,
99 const std::vector<int>& responder_chips) {
100 std::vector<Action> actions;
101 ChipComboIterator proposer_iter(player_chips);
102 while (!proposer_iter.IsFinished()) {
103 std::vector<int> proposer_chips = proposer_iter.Next();
104 ChipComboIterator receiver_iter(responder_chips);
105 while (!receiver_iter.IsFinished()) {
106 std::vector<int> receiver_chips = receiver_iter.Next();
107 Trade trade(proposer_chips, receiver_chips);
108 if (IsLegalTrade(board, trade, proposer_chips, responder_chips)) {
109 int trade_id = game->LookupTradeId(trade.ToString());
110 actions.push_back(trade_id);
111 }
112 }
113 }
114 // Sort and remove duplicates.
115 absl::c_sort(actions);
116 auto last = std::unique(actions.begin(), actions.end());
117 actions.erase(last, actions.end());
118
119 // Add pass trade.
120 actions.push_back(game->PassAction());
121 return actions;
122}

164void Board::ApplyTrade(std::pair<int, int> players, const Trade& trade) {
165 if (trade.giving.empty()) {
166 // This is a pass, so don't change the board.
167 return;
168 }
169 SPIEL_CHECK_EQ(trade.giving.size(), num_colors);
170 SPIEL_CHECK_EQ(trade.receiving.size(), num_colors);
171 for (int i = 0; i < num_colors; ++i) {
172 SPIEL_CHECK_LE(trade.giving[i], chips[players.first][i]);
173 SPIEL_CHECK_LE(trade.receiving[i], chips[players.second][i]);
174 chips[players.first][i] -= trade.giving[i];
175 chips[players.second][i] += trade.giving[i];
176 chips[players.first][i] += trade.receiving[i];
177 chips[players.second][i] -= trade.receiving[i];
178 }
179}

205void Board::ParseFromLine(const std::string& line) {
206 // Example: 4 5 3 AAEDCABCDAAABBEE AACCCD AAAC BBCEE 14 7 0 2
207 std::vector<std::string> parts = absl::StrSplit(line, ' ');
208 SPIEL_CHECK_EQ(parts.size(), 3 + 2 * num_players + 2);
209
210 int _size, _colors, _players;
211 SPIEL_CHECK_TRUE(absl::SimpleAtoi(parts[0], &_size));
212 SPIEL_CHECK_TRUE(absl::SimpleAtoi(parts[1], &_colors));
213 SPIEL_CHECK_TRUE(absl::SimpleAtoi(parts[2], &_players));
214 SPIEL_CHECK_EQ(_size, size);
215 SPIEL_CHECK_EQ(_colors, num_colors);
216 SPIEL_CHECK_EQ(_players, num_players);
217
218 SPIEL_CHECK_EQ(parts[3].size(), size * size);
219 for (int i = 0; i < parts[3].size(); ++i) {
220 board[i] = CharToColor(parts[3].at(i));
221 }
222
223 for (Player p = 0; p < num_players; ++p) {
224 num_chips[p] = parts[4 + p].length();
225 for (int i = 0; i < parts[4 + p].length(); ++i) {
226 int chip_color = CharToColor(parts[4 + p].at(i));
227 chips[p][chip_color]++;
228 }
229 }
230
231 for (int i = 0; i < num_players + 1; ++i) {
232 SPIEL_CHECK_TRUE(
233 absl::SimpleAtoi(parts[4 + num_players + i], &positions[i]));
234 }
235}

244int Trade::DistanceTo(const Trade& other) const {
245 int sum = 0;
246 if (other.giving.empty() || other.receiving.empty()) {
247 // Pass trade is the furthest possible distance.
248 return kDefaultTradeDistanceUpperBound + 1;
249 }
250 for (int i = 0; i < giving.size(); ++i) {
251 sum += std::abs(other.giving[i] - giving[i]);
252 sum += std::abs(other.receiving[i] - receiving[i]);
253 }
254 return sum;
255}

257bool Trade::reduce() {
258 for (int i = 0; i < giving.size(); ++i) {
259 int min_val = std::min(giving[i], receiving[i]);
260 giving[i] -= min_val;
261 receiving[i] -= min_val;
262 }
263 return (std::accumulate(giving.begin(), giving.end(), 0) > 0 &&
264 std::accumulate(receiving.begin(), receiving.end(), 0) > 0);
265}

273std::string ColoredTrailsState::ActionToString(Player player,
274 Action move_id) const {
275 if (player == kChancePlayerId) {
276 return absl::StrCat("Chance outcome ", move_id);
277 } else if (player < kResponderId) {
278 return absl::StrCat("Proposer ", player, ": ",
279 parent_game_->LookupTrade(move_id).ToString());
280 } else if (player == kResponderId) {
281 if (move_id == num_distinct_actions_ - 3) {
282 return "Deal: trade with proposer 0";
283 } else if (move_id == num_distinct_actions_ - 2) {
284 return "Deal: trade with proposer 1";
285 } else if (move_id == num_distinct_actions_ - 1) {
286 return "No Deal!";
287 } else {
288 SpielFatalError(absl::StrCat("move_id unrecognized: ", move_id));
289 }
290 } else {
291 SpielFatalError(absl::StrCat("Player and move case unrecognized: ", player,
292 " ", move_id));
293 }

338std::unique_ptr<State> ColoredTrailsState::ResampleFromInfostate(
339 int player_id, std::function<double()> rng) const {
340 std::vector<std::unique_ptr<State>> candidates;
341 const std::vector<Board>& all_boards = parent_game_->AllBoards();
342
343 for (int o = 0; o < all_boards.size(); ++o) {
344 if (board_.ToString() != all_boards[o].ToString()) {
345 continue;
346 }
347
348 std::unique_ptr<State> candidate_state = parent_game_->NewInitialState();
349 candidate_state->ApplyAction(o);
350
351 if (player_id == 0) {
352 if (candidate_state->InformationStateString(0) ==
353 InformationStateString(0)) {
354 candidates.push_back(std::move(candidate_state));
355 }
356 } else if (player_id == 1) {
357 // Enumerate legal moves.
358 for (Action action : candidate_state->LegalActions()) {
359 std::unique_ptr<State> candidate_child = candidate_state->Child(action);
360 if (candidate_child->InformationStateString(1) ==
361 InformationStateString(1)) {
362 candidates.push_back(std::move(candidate_child));
363 } else {
364 // Player 0's move is hidden. No need to keep trying actions if P1's
365 // infostate doesn't match.
366 break;
367 }
368 }
369 } else {
370 SPIEL_CHECK_EQ(player_id, 2);
371 SPIEL_CHECK_EQ(History().size(), 3);
372 Action p0_action = History()[1];
373 Action p1_action = History()[2];
374 // Receiver sees everything, so replay the moves.
375 std::vector<Action> legal_actions = candidate_state->LegalActions();
376 if (absl::c_find(legal_actions, p0_action) != legal_actions.end()) {
377 candidate_state->ApplyAction(p0_action);
378 legal_actions = candidate_state->LegalActions();
379 if (absl::c_find(legal_actions, p1_action) != legal_actions.end()) {
380 candidate_state->ApplyAction(p1_action);
381 candidates.push_back(std::move(candidate_state));
382 }
383 }
384 }
385 }
386
387 SPIEL_CHECK_GE(candidates.size(), 1);
388 if (candidates.size() == 1) {
389 return std::move(candidates[0]);
390 } else {
391 int idx = static_cast<int>(rng() * candidates.size());
392 SPIEL_CHECK_LE(idx, candidates.size());
393 return std::move(candidates[idx]);
394 }
395}

397void ColoredTrailsState::InformationStateTensor(
398 Player player, absl::Span<float> values) const {
399 SPIEL_CHECK_GE(player, 0);
400 SPIEL_CHECK_LT(player, num_players_);
401
402 SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorSize());
403 std::fill(values.begin(), values.end(), 0);
404
405 if (IsChanceNode()) {
406 // No observations at chance nodes.
407 return;
408 }
409
410 int offset = 0;
411
412 // Player.
413 values[player] = 1;
414 offset += num_players_;
415
416 // Terminal?
417 if (IsTerminal()) {
418 values[offset] = 1;
419 }
420 offset += 1;
421
422 // The board
423 for (int i = 0; i < board_.board.size(); ++i) {
424 values[offset + board_.board[i]] = 1;
425 offset += board_.num_colors;
426 }
427
428 // Positions
429 for (int i = 0; i < board_.positions.size(); ++i) {
430 values[offset + board_.positions[i]] = 1;
431 offset += board_.size * board_.size;
432 }
433
434 // Chips.
435 std::array<const std::vector<int>*, 3> chips_ptrs;
436 std::vector<int> zeros(board_.num_colors, 0);
437 if (player < kResponderId) {
438 chips_ptrs[0] = &board_.chips[player];
439 chips_ptrs[1] = &zeros;
440 chips_ptrs[2] = &board_.chips[kResponderId];
441 } else {
442 chips_ptrs[0] = &board_.chips[0];
443 chips_ptrs[1] = &board_.chips[1];
444 chips_ptrs[2] = &board_.chips[kResponderId];
445 }
446 for (int c = 0; c < 3; ++c) {
447 for (int i = 0; i < board_.num_colors; ++i) {
448 for (int j = 0; j <= chips_ptrs[c]->at(i); ++j) {
449 values[offset + j] = 1;
450 }
451 offset += (kNumChipsUpperBound + 1);
452 }
453 }
454
455 // Proposals
456 if (player == kResponderId && CurrentPlayer() == kResponderId) {
457 SPIEL_CHECK_EQ(proposals_.size(), 2);
458 for (int p : {0, 1}) {
459 if (IsPassTrade(proposals_[p])) {
460 chips_ptrs[0] = &zeros;
461 chips_ptrs[1] = &zeros;
462 } else {
463 chips_ptrs[0] = &(proposals_[p].giving);
464 chips_ptrs[1] = &(proposals_[p].receiving);
465 }
466
467 for (int c = 0; c < 2; ++c) {
468 for (int i = 0; i < board_.num_colors; ++i) {
469 for (int j = 0; j <= chips_ptrs[c]->at(i); ++j) {
470 values[offset + j] = 1;
471 }
472 offset += (kNumChipsUpperBound + 1);
473 }
474 }
475 }
476 } else {
477 // Proposers have no observations of the proposals.
478 // Responder doesn't observe the chips until its their turn.
479 offset += (kNumChipsUpperBound + 1) * board_.num_colors * 2 *
480 (num_players_ - 1);
481 }
482 SPIEL_CHECK_EQ(offset, values.size());
483}

497void ColoredTrailsState::DoApplyAction(Action action) {
498 if (IsChanceNode()) {
499 const std::vector<Board>& all_boards = parent_game_->AllBoards();
500 SPIEL_CHECK_GE(action, 0);
501 SPIEL_CHECK_LT(action, all_boards.size());
502 board_ = all_boards[action];
503 cur_player_ = 0;
504 } else if (cur_player_ < kResponderId) {
505 proposals_.push_back(parent_game_->LookupTrade(action));
506 cur_player_++;
507
508 // Special case when using SetChipsAndProposals, check the future_trade_.
509 // If it's now the second player, and there's a future trade queued, apply
510 // it.
511 if (cur_player_ == 1 &&
512 (!future_trade_.giving.empty() || !future_trade_.receiving.empty())) {
513 proposals_.push_back(future_trade_);
514 cur_player_++;
515 }
516 } else {
517 // Base scores.
518 SPIEL_CHECK_EQ(cur_player_, kResponderId);
519 for (Player p = 0; p < board_.num_players; ++p) {
520 returns_[p] = Score(p, board_).first;
521 }
522
523 if (action == parent_game_->ResponderTradeWithPlayerAction(0)) {
524 if (!IsPassTrade(proposals_[0])) {
525 board_.ApplyTrade({0, kResponderId}, proposals_[0]);
526 }
527 } else if (action == parent_game_->ResponderTradeWithPlayerAction(1)) {
528 if (!IsPassTrade(proposals_[1])) {
529 board_.ApplyTrade({1, kResponderId}, proposals_[1]);
530 }
531 } else if (action == parent_game_->PassAction()) {
532 // No trade.
533 } else {
534 std::string error = absl::StrCat("Invalid action: ", action,
535 parent_game_->ActionToString(kResponderId, action), "\n",
536 ToString());
537 SpielFatalErrorWithStateInfo(error, *parent_game_, *this);
538 }
539
540 // Gain is final score minus base score.
541 for (Player p = 0; p < board_.num_players; ++p) {
542 returns_[p] = Score(p, board_).first - returns_[p];
543 }
544
545 cur_player_ = kTerminalPlayerId;
546 }
547}

559std::vector<Action> ColoredTrailsState::LegalActionsForChips(
560 const std::vector<int>& player_chips,
561 const std::vector<int>& responder_chips) const {
562 // First, check the cache.
563 std::string key = absl::StrCat(ComboToString(player_chips), " ",
564 ComboToString(responder_chips));
565 std::vector<Action> actions = parent_game_->LookupTradesCache(key);
566 if (!actions.empty()) {
567 return actions;
568 }
569
570 actions = GenerateLegalActionsForChips(parent_game_, board_, player_chips,
571 responder_chips);
572
573 // Add these to the cache.
574 parent_game_->AddToTradesCache(key, actions);
575 return actions;
576}

578std::vector<Action> ColoredTrailsState::LegalActions() const {
579 if (IsChanceNode()) {
580 return LegalChanceOutcomes();
581 } else if (IsTerminal()) {
582 return {};
583 } else if (cur_player_ < kResponderId) {
584 return LegalActionsForChips(board_.chips[cur_player_],
585 board_.chips[kResponderId]);
586 } else {
587 SPIEL_CHECK_EQ(cur_player_, kResponderId);
588 // Last three actions correspond to "trade with 0", "trade with 1", and
589 // "no trade".
590 return {parent_game_->ResponderTradeWithPlayerAction(0),
591 parent_game_->ResponderTradeWithPlayerAction(1),
592 parent_game_->PassAction()};
593 }

596std::vector<std::pair<Action, double>> ColoredTrailsState::ChanceOutcomes()
597 const {
598 SPIEL_CHECK_TRUE(IsChanceNode());
599 std::vector<std::pair<Action, double>> outcomes;
600 const int num_boards = parent_game_->AllBoards().size();
601 outcomes.reserve(num_boards);
602 double uniform_prob = 1.0 / num_boards;
603 for (int i = 0; i < num_boards; ++i) {
604 outcomes.push_back({i, uniform_prob});
605 }
606 return outcomes;
607}

635void ColoredTrailsState::SetChipsAndTradeProposal(
636 Player player, std::vector<int> chips, Trade trade,
637 std::vector<double>& rng_rolls) {
638 // First, check the chips.
639 int rng_idx = 0;
640 int num_chips = std::accumulate(chips.begin(), chips.end(), 0);
641
642 while (num_chips < kNumChipsLowerBound) {
643 std::vector<int> indices;
644 for (int i = 0; i < chips.size(); i++) {
645 if (chips[i] == 0) {
646 indices.push_back(i);
647 }
648 }
649 SPIEL_CHECK_LT(rng_idx, rng_rolls.size());
650 int selected_idx =
651 indices[static_cast<int>(rng_rolls[rng_idx] * indices.size())];
652 chips[selected_idx]++;
653 rng_idx++;
654 num_chips = std::accumulate(chips.begin(), chips.end(), 0);
655 }
656
657 while (num_chips > kNumChipsUpperBound) {
658 std::vector<int> indices;
659 for (int i = 0; i < chips.size(); i++) {
660 if (chips[i] > 0) {
661 indices.push_back(i);
662 }
663 }
664 SPIEL_CHECK_LT(rng_idx, rng_rolls.size());
665 int selected_idx =
666 indices[static_cast<int>(rng_rolls[rng_idx] * indices.size())];
667 chips[selected_idx]--;
668 rng_idx++;
669 num_chips = std::accumulate(chips.begin(), chips.end(), 0);
670 }
671
672 board_.chips[player] = chips;
673 trade.reduce();
674
675 // Now check if the Trade is legal. If not, chose one of the closest legal
676 // ones in edit distance
677 if (!IsLegalTrade(player, trade)) {
678 std::vector<Trade> closest_trades;
679 int lowest_distance = kDefaultTradeDistanceUpperBound + 100;
680 std::vector<Action> legal_actions =
681 LegalActionsForChips(chips, board_.chips[kResponderId]);
682 for (Action action : legal_actions) {
683 const Trade& legal_trade = parent_game_->LookupTrade(action);
684 int dist = trade.DistanceTo(legal_trade);
685 if (dist == lowest_distance) {
686 closest_trades.push_back(legal_trade);
687 } else if (dist < lowest_distance) {
688 lowest_distance = dist;
689 closest_trades = {legal_trade};
690 }
691 }
692
693 if (closest_trades.empty()) {
694 std::cout << ToString() << std::endl;
695 std::cout << "Trade: " << trade.ToString() << std::endl;
696 }
697
698 SPIEL_CHECK_GT(closest_trades.size(), 0);
699 if (closest_trades.size() == 1) {
700 trade = closest_trades[0];
701 } else {
702 trade = closest_trades[static_cast<int>(rng_rolls[rng_idx] *
703 closest_trades.size())];
704 rng_idx++;
705 }
706 }
707
708 if (player == 0) {
709 SPIEL_CHECK_NE(cur_player_, 0);
710 proposals_[0] = trade;
711 } else if (player == 1) {
712 SPIEL_CHECK_NE(cur_player_, 1);
713 if (cur_player_ == 0) {
714 future_trade_ = trade;
715 } else {
716 proposals_[1] = trade;
717 }
718 }
719}

776bool CheckBoard(const Board& board) {
777 std::vector<int> base_scores(board.num_players);
778 int min_score = board.size * 100;
779 int max_score = board.size * -100;
780
781 for (Player player = 0; player < board.num_players; ++player) {
782 std::pair<int, bool> score_and_solved = Score(player, board);
783 if (score_and_solved.second) {
784 // Cannot be solvable without negotiation.
785 return false;
786 }
787 base_scores[player] = score_and_solved.first;
788 min_score = std::min(min_score, base_scores[player]);
789 max_score = std::max(max_score, base_scores[player]);
790 }
791
792 if (max_score - min_score > kBaseScoreEpsilon) {
793 return false;
794 }
795
796 // Now check that there exist two trades:
797 // - one between player 0 and 2, such that both can reach the goal
798 // - one between player 1 and 2, such that both can reach the goal
799 for (int proposer : {0, 1}) {
800 bool found_trade = false;
801 ChipComboIterator iter1(board.chips[proposer]);
802 while (!found_trade && !iter1.IsFinished()) {
803 std::vector<int> combo1 = iter1.Next();
804 ChipComboIterator iter2(board.chips[2]);
805 while (!found_trade && !iter2.IsFinished()) {
806 std::vector<int> combo2 = iter2.Next();
807 // Do the trade and check if both can reach the goal.
808 Board board_copy = board;
809 Trade trade(combo1, combo2);
810 board_copy.ApplyTrade({proposer, 2}, trade);
811 std::pair<int, bool> prop_score_and_goal = Score(proposer, board_copy);
812 if (prop_score_and_goal.second) {
813 std::pair<int, bool> rec_score_and_goal = Score(2, board_copy);
814 if (rec_score_and_goal.second) {
815 found_trade = true;
816 }
817 }
818 }
819 }
820 if (!found_trade) {
821 return false;
822 }
823 }
824
825 return true;
826}

875std::pair<Board, Action> ColoredTrailsGame::SampleRandomBoardCompletion(
876 int seed, const Board& board, Player player) const {
877 std::mt19937 rng(seed);
878 Board new_board = board;
879 const int max_tries = 1000;
880 int tries = 0;
881
882 do {
883 tries += 1;
884 for (int i = 0; i < new_board.chips[player].size(); ++i) {
885 new_board.chips[player][i] = 0;
886 }
887 int width = kNumChipsUpperBound - kNumChipsLowerBound + 1;
888 new_board.num_chips[player] =
889 kNumChipsLowerBound + absl::Uniform<int>(rng, 0, width);
890 for (int i = 0; i < new_board.num_chips[player]; ++i) {
891 int chip = absl::Uniform<int>(rng, 0, new_board.num_colors);
892 new_board.chips[player][chip]++;
893 }
894 } while (!CheckBoardForProposer(new_board, player) && tries < max_tries);
895 SPIEL_CHECK_LT(tries, max_tries);
896
897 std::string key = absl::StrCat(ComboToString(new_board.chips[player]), " ",
898 ComboToString(new_board.chips[kResponderId]));
899 std::vector<Action> actions = LookupTradesCache(key);
900 if (actions.empty()) {
901 actions = GenerateLegalActionsForChips(this, new_board,
902 new_board.chips[player],
903 new_board.chips[kResponderId]);
904 AddToTradesCache(key, actions);
905 }
906
907 Action action = actions[absl::Uniform<int>(rng, 0, actions.size())];
908 return {new_board, action};
909}
