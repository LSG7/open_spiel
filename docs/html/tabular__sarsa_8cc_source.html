33 Action TabularSarsaSolver::GetBestAction(const State& state,
34  double min_utility) {
35  vector<Action> legal_actions = state.LegalActions();
36  SPIEL_CHECK_GT(legal_actions.size(), 0);
37  const auto state_str = state.ToString();
38 
39  Action best_action = legal_actions[0];
40  double value = min_utility;
41  for (const Action& action : legal_actions) {
42  double q_val = values_[{state_str, action}];
43  if (q_val >= value) {
44  value = q_val;
45  best_action = action;
46  }
47  }
48  return best_action;
49 }

51 Action TabularSarsaSolver::SampleActionFromEpsilonGreedyPolicy(
52  const State& state, double min_utility) {
53  vector<Action> legal_actions = state.LegalActions();
54  if (legal_actions.empty()) {
55  return kInvalidAction;
56  }
57 
58  if (absl::Uniform(rng_, 0.0, 1.0) < epsilon_) {
59  // Choose a random action
60  return legal_actions[absl::Uniform<int>(rng_, 0, legal_actions.size())];
61  }
62  // Choose the best action
63  return GetBestAction(state, min_utility);
64 }

66 void TabularSarsaSolver::SampleUntilNextStateOrTerminal(State* state) {
67  // Repeatedly sample while chance node, so that we end up at a decision node
68  while (state->IsChanceNode() && !state->IsTerminal()) {
69  std::vector<std::pair<Action, double>> outcomes = state->ChanceOutcomes();
70  state->ApplyAction(SampleAction(outcomes, rng_).first);
71  }
72 }

75  : game_(game),

78  learning_rate_(kDefaultLearningRate),
79  discount_factor_(kDefaultDiscountFactor),
80  lambda_(kDefaultLambda) {
81  SPIEL_CHECK_LE(lambda_, 1);
82  SPIEL_CHECK_GE(lambda_, 0);
83 
84  // Currently only supports 1-player or 2-player zero sum games
85  SPIEL_CHECK_TRUE(game_->NumPlayers() == 1 || game_->NumPlayers() == 2);
86  if (game_->NumPlayers() == 2) {
87  SPIEL_CHECK_EQ(game_->GetType().utility, GameType::Utility::kZeroSum);
88  }
89 
90  // No support for simultaneous games (needs an LP solver). And so also must

92  SPIEL_CHECK_EQ(game_->GetType().dynamics, GameType::Dynamics::kSequential);
93  SPIEL_CHECK_EQ(game_->GetType().information,
94  GameType::Information::kPerfectInformation);

128 void TabularSarsaSolver::RunIteration() {
129  double min_utility = game_->MinUtility();
130  // Choose start state
131  std::unique_ptr<State> curr_state = game_->NewInitialState();
132  SampleUntilNextStateOrTerminal(curr_state.get());
133 
134  Player player = curr_state->CurrentPlayer();
135  // Sample action from the state using an epsilon-greedy policy
136  Action curr_action =
137  SampleActionFromEpsilonGreedyPolicy(*curr_state, min_utility);
138  std::unique_ptr<State> next_state;
139  SPIEL_CHECK_NE(curr_action, kInvalidAction);
140 
141  while (!curr_state->IsTerminal()) {
142  player = curr_state->CurrentPlayer();
143 
144  next_state = curr_state->Child(curr_action);
145  SampleUntilNextStateOrTerminal(next_state.get());
146  const double reward = next_state->Rewards()[player];
147 
148  const Action next_action =
149  next_state->IsTerminal()
150  ? kInvalidAction
151  : SampleActionFromEpsilonGreedyPolicy(*next_state, min_utility);
152 
153  // Update the new q value
154  std::string key = curr_state->ToString();
155  // Next q-value in perspective of player to play at curr_state (important
156  // note: exploits property of two-player zero-sum). Define the value of
157  // q(s', a') to be 0 if s' is terminal.
158  const double future_value =
159  next_state->IsTerminal()
160  ? 0
161  : values_[{next_state->ToString(), next_action}];
162  const double next_q_value =
163  (player != next_state->CurrentPlayer() ? -1 : 1) * future_value;
164  double new_q_value = reward + discount_factor_ * next_q_value;
165 
166  double prev_q_val = values_[{key, curr_action}];
167  if (lambda_ == 0) {
168  // If lambda_ is equal to zero, run sarsa as usual. It's not necessary
169  // to update eligibility traces.
170  values_[{key, curr_action}] +=
171  learning_rate_ * (new_q_value - prev_q_val);
172  } else {
173  double lambda =
174  player != next_state->CurrentPlayer() ? -lambda_ : lambda_;
175  eligibility_traces_[{key, curr_action}] += 1;
176 
177  for (const auto& q_cell : values_) {
178  std::string state = q_cell.first.first;
179  Action action = q_cell.first.second;
180 
181  values_[{state, action}] += learning_rate_ *
182  (new_q_value - prev_q_val) *
183  eligibility_traces_[{state, action}];
184  eligibility_traces_[{state, action}] *= discount_factor_ * lambda;
185  }
186  }
187 
188  curr_state = std::move(next_state);
189  curr_action = next_action;
190  }
191 }
