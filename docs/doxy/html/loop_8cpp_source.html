27void loop_solve(
28 boardsPBN * bop,
29 solvedBoards * solvedbdp,
30 dealPBN * deal_list,
31 futureTricks * fut_list,
32 const int number,
33 const int stepsize)
34{
35#ifdef BATCHTIMES
36 cout << setw(8) << left << "Hand no." <<
37 setw(25) << right << "Time" << "\n";
38#endif
39
40 for (int i = 0; i < number; i += stepsize)
41 {
42 int count = (i + stepsize > number ? number - i : stepsize);
43
44 bop->noOfBoards = count;
45 for (int j = 0; j < count; j++)
46 {
47 bop->deals[j] = deal_list[i + j];
48 bop->target[j] = -1;
49 bop->solutions[j] = 3;
50 bop->mode[j] = 1;
51 }
52
53 timer.start(count);
54 int ret;
55 if ((ret = SolveAllChunks(bop, solvedbdp, 1)) != RETURN_NO_FAULT)
56 {
57 cout << "loop_solve: i " << i << ", return " << ret << "\n";
58 exit(0);
59 }
60 timer.end();
61
62#ifdef BATCHTIMES
63 timer.printRunning(i+count, number);
64#endif
65
66 for (int j = 0; j < count; j++)
67 {
68 if (compare_FUT(solvedbdp->solvedBoard[j], fut_list[i + j]))
69 continue;
70
71 cout << "loop_solve: i " << i << ", j " << j << ": " <<
72 "Difference\n\n";
73 print_FUT(solvedbdp->solvedBoard[j]);
74 cout << "\n";
75 print_FUT(fut_list[i+j]);
76 cout << "\n";
77 }
78 }
79
80#ifdef BATCHTIMES
81 cout << "\n";
82#endif
83
84}
