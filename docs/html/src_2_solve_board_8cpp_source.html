74 void SolveChunkCommon(
75  const int thrId)
76 {
77  int index;
78  schedType st;
79 
80  while (1)
81  {
82  st = scheduler.GetNumber(thrId);
83  index = st.number;
84  if (index == -1)
85  break;
86 
87  // This is not a perfect repeat detector, as the hands in
88  // a group might have declarers N, S, N, N. Then the second
89  // N would not reuse the first N. However, must reuses are
90  // reasonably adjacent, and this is just an optimization anyway.
91 
92  if (st.repeatOf != -1 &&
93  param.bop->deals[index ].first ==
94  param.bop->deals[st.repeatOf].first)
95  {
96  START_THREAD_TIMER(thrId);
97  param.solvedp->solvedBoard[index] =
98  param.solvedp->solvedBoard[st.repeatOf];
99  END_THREAD_TIMER(thrId);
100  continue;
101  }
102  else
103  {
104  SolveSingleCommon(thrId, index);
105  }
106  }
107 }

186  bo.solutions[k] = bop->solutions[k];
187  bo.target[k] = bop->target[k];
188  bo.deals[k].first = bop->deals[k].first;
189  bo.deals[k].trump = bop->deals[k].trump;
190 
191  for (int i = 0; i <= 2; i++)
192  {
193  bo.deals[k].currentTrickSuit[i] = bop->deals[k].currentTrickSuit[i];
194  bo.deals[k].currentTrickRank[i] = bop->deals[k].currentTrickRank[i];
195  }
196 
197  if (ConvertFromPBN(bop->deals[k].remainCards, bo.deals[k].remainCards)
198  != 1)
199  return RETURN_PBN_FAULT;
200  }
201 
202  int res = SolveAllBoardsN(bo, * solvedp);

246 void DetectSolveDuplicates(
247  const boards& bds,
248  vector<int>& uniques,
249  vector<int>& crossrefs)
250 {
251  const unsigned nu = static_cast<unsigned>(bds.noOfBoards);
252 
253  uniques.clear();
254  crossrefs.resize(nu);
255 
256  for (unsigned i = 0; i < nu; i++)
257  crossrefs[i] = -1;
258 
259  for (unsigned i = 0; i < nu; i++)
260  {
261  if (crossrefs[i] != -1)
262  continue;
263 
264  uniques.push_back(static_cast<int>(i));
265 
266  for (unsigned index = i+1; index < nu; index++)
267  {
268  if (SameBoard(bds, i, index))
269  crossrefs[index] = static_cast<int>(i);
270  }
271  }

275 bool SameBoard(
276  const boards& bds,
277  const unsigned index1,
278  const unsigned index2)
279 {
280  for (int h = 0; h < DDS_HANDS; h++)
281  {
282  for (int s = 0; s < DDS_SUITS; s++)
283  {
284  if (bds.deals[index1].remainCards[h][s] !=
285  bds.deals[index2].remainCards[h][s])
286  return false;
287  }
288  }
289 
290  if (bds.mode[index1] != bds.mode[index2])
291  return false;
292  if (bds.solutions[index1] != bds.solutions[index2])
293  return false;
294  if (bds.target[index1] != bds.target[index2])
295  return false;
296  if (bds.deals[index1].first != bds.deals[index2].first)
297  return false;
298  if (bds.deals[index1].trump != bds.deals[index2].trump)
299  return false;
300 
301  for (int k = 0; k < 3; k++)
302  {
303  if (bds.deals[index1].currentTrickSuit[k] !=
304  bds.deals[index2].currentTrickSuit[k])
305  return false;
306  if (bds.deals[index1].currentTrickRank[k] !=
307  bds.deals[index2].currentTrickRank[k])
308  return false;
309  }
310  return true;
311 }
