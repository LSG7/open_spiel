62std::vector<VirtualPoint> HandicapStones(int num_handicap) {
63 if (num_handicap < 2 || num_handicap > 9) return {};
64
65 static std::array<VirtualPoint, 9> placement = {
66 {MakePoint("d4"), MakePoint("q16"), MakePoint("d16"), MakePoint("q4"),
67 MakePoint("d10"), MakePoint("q10"), MakePoint("k4"), MakePoint("k16"),
68 MakePoint("k10")}};
69 static VirtualPoint center = MakePoint("k10");
70
71 std::vector<VirtualPoint> points;
72 points.reserve(num_handicap);
73 for (int i = 0; i < num_handicap; ++i) {
74 points.push_back(placement[i]);
75 }
76
77 if (num_handicap >= 5 && num_handicap % 2 == 1) {
78 points[num_handicap - 1] = center;
79 }
80
81 return points;
82}

109void GoState::ObservationTensor(int player, absl::Span<float> values) const {
110 SPIEL_CHECK_GE(player, 0);
111 SPIEL_CHECK_LT(player, num_players_);
112
113 int num_cells = board_.board_size() * board_.board_size();
114 SPIEL_CHECK_EQ(values.size(), num_cells * (CellStates() + 1));
115 std::fill(values.begin(), values.end(), 0.);
116
117 // Add planes: black, white, empty.
118 int cell = 0;
119 for (VirtualPoint p : BoardPoints(board_.board_size())) {
120 int color_val = static_cast<int>(board_.PointColor(p));
121 values[num_cells * color_val + cell] = 1.0;
122 ++cell;
123 }
124 SPIEL_CHECK_EQ(cell, num_cells);
125
126 // Add a fourth binary plane for komi (whether white is to play).
127 std::fill(values.begin() + (CellStates() * num_cells), values.end(),
128 (to_play_ == GoColor::kWhite ? 1.0 : 0.0));
129}

131std::vector<Action> GoState::LegalActions() const {
132 std::vector<Action> actions{};
133 if (IsTerminal()) return actions;
134 for (VirtualPoint p : BoardPoints(board_.board_size())) {
135 if (board_.IsLegalMove(p, to_play_)) {
136 actions.push_back(board_.VirtualActionToAction(p));
137 }
138 }
139 actions.push_back(board_.pass_action());
140 return actions;
141}

157bool GoState::IsTerminal() const {
158 if (history_.size() < 2) return false;
159 return (history_.size() >= max_game_length_) || superko_ ||
160 (history_[history_.size() - 1].action == board_.pass_action() &&
161 history_[history_.size() - 2].action == board_.pass_action());
162}

164std::vector<double> GoState::Returns() const {
165 if (!IsTerminal()) return {0.0, 0.0};
166
167 if (superko_) {
168 // Superko rules (https://senseis.xmp.net/?Superko) are complex and vary
169 // between rulesets.
170 // For simplicity and because superkos are very rare, we just treat them as
171 // a draw.
172 return {DrawUtility(), DrawUtility()};
173 }
174
175 // Score with Tromp-Taylor.
176 float black_score = TrompTaylorScore(board_, komi_, handicap_);
177
178 std::vector<double> returns(go::NumPlayers());
179 if (black_score > 0) {
180 returns[ColorToPlayer(GoColor::kBlack)] = WinUtility();
181 returns[ColorToPlayer(GoColor::kWhite)] = LossUtility();
182 } else if (black_score < 0) {
183 returns[ColorToPlayer(GoColor::kBlack)] = LossUtility();
184 returns[ColorToPlayer(GoColor::kWhite)] = WinUtility();
185 } else {
186 returns[ColorToPlayer(GoColor::kBlack)] = DrawUtility();
187 returns[ColorToPlayer(GoColor::kWhite)] = DrawUtility();
188 }
189 return returns;
190}

196void GoState::UndoAction(Player player, Action action) {
197 // We don't have direct undo functionality, but copying the board and
198 // replaying all actions is still pretty fast (> 1 million undos/second).
199 history_.pop_back();
200 --move_number_;
201 ResetBoard();
202 for (auto [_, action] : history_) {
203 DoApplyAction(action);
204 }
205}

207void GoState::DoApplyAction(Action action) {
208 SPIEL_CHECK_TRUE(
209 board_.PlayMove(board_.ActionToVirtualAction(action), to_play_));
210 to_play_ = OppColor(to_play_);
211
212 bool was_inserted = repetitions_.insert(board_.HashValue()).second;
213 if (!was_inserted && action != board_.pass_action()) {
214 // We have encountered this position before.
215 superko_ = true;
216 }
217}

219void GoState::ResetBoard() {
220 board_.Clear();
221 if (handicap_ < 2) {
222 to_play_ = GoColor::kBlack;
223 } else {
224 for (VirtualPoint p : HandicapStones(handicap_)) {
225 board_.PlayMove(p, GoColor::kBlack);
226 }
227 to_play_ = GoColor::kWhite;
228 }
229
230 repetitions_.clear();
231 repetitions_.insert(board_.HashValue());
232 superko_ = false;
233}
