114bool Mkdirs(const std::string& path, int mode) {
115 struct stat info;
116 size_t pos = 0;
117 while (pos != std::string::npos) {
118 pos = path.find_first_of("\\/", pos + 1);
119 std::string sub_path = path.substr(0, pos);
120 if (stat(sub_path.c_str(), &info) == 0) {
121 if (info.st_mode & S_IFDIR) {
122 continue; // directory already exists
123 } else {
124 return false; // is a file?
125 }
126 }
127 if (!Mkdir(sub_path, mode)) {
128 return false; // permission error?
129 }
130 }
131 return true;
132}
