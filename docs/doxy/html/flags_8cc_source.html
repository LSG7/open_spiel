88ABSL_FLAG(std::string, log_backtrace_at, "",
89 "Emit a backtrace when logging at file:linenum.")
90 .OnUpdate([] {
91 const std::string log_backtrace_at =
92 absl::GetFlag(FLAGS_log_backtrace_at);
93 if (log_backtrace_at.empty()) return;
94
95 const size_t last_colon = log_backtrace_at.rfind(':');
96 if (last_colon == log_backtrace_at.npos) return;
97
98 const absl::string_view file =
99 absl::string_view(log_backtrace_at).substr(0, last_colon);
100 int line;
101 if (absl::SimpleAtoi(
102 absl::string_view(log_backtrace_at).substr(last_colon + 1),
103 &line)) {
104 absl::SetLogBacktraceLocation(file, line);
105 }
106 });
