25def _inc_average(count, average, value):
26 """Computes the incremental average, `a_n = ((n - 1)a_{n-1} + v_n) / n`."""
27 count += 1
28 average = ((count - 1) * average + value) / count
29 return (count, average)

32def from_match_results(df, consider_agents):
33 """Builds a heuristic payoff table from average win probabilities.
34
35 Args:
36 df: a Pandas dataframe of match results. Must contain a column "agents"
37 consisting of tuples of agent names, and a column "scores" consisting of
38 the score for each agent in the match.
39 consider_agents: a list of agent names. Will only consider matches in which
40 exclusively these agents appeared.
41
42 Returns:
43 A PayoffTable object.
44
45 Raises:
46 ValueError: if dataframe is empty, or columns 'agents' and 'scores' not
47 specified, or games have zero players.
48 """
49 if df.empty:
50 raise ValueError("Please provide a non-empty dataframe.")
51 if "agents" not in df.columns:
52 raise ValueError("Dataframe must contain a column 'agents'.")
53 if "scores" not in df.columns:
54 raise ValueError("Dataframe must contain a column 'scores'.")
55
56 num_strategies = len(consider_agents)
57 num_players = len(df["agents"][0])
58
59 if num_players == 0:
60 raise ValueError("Games must have > 0 players.")
61
62 count_per_distribution = {}
63 win_prob_per_distribution = {}
64
65 for i, row in df.iterrows():
66 print("Parsing row {} / {} ...".format(i, len(df)), end="\r")
67 agents = row["agents"]
68 scores = row["scores"]
69 assert len(agents) == len(scores) == num_players
70
71 if not set(agents).issubset(set(consider_agents)):
72 # Ignore agents outside those we are supposed to consider.
73 continue
74 elif len(set(agents)) == 1:
75 # Special case of self-play: deal with separately.
76 continue
77
78 # Find winner(s): In each match one must determine a winning strategy. One
79 # way of doing this is to average over the returns for each strategy and
80 # then say that the one with the greatest returns is the winner.
81
82 # Get unique score per agent by averaging.
83 count_per_agent = collections.defaultdict(int)
84 average_score_per_agent = collections.defaultdict(int)
85 for agent, score in zip(agents, scores):
86 count_per_agent[agent], average_score_per_agent[agent] = _inc_average(
87 count_per_agent[agent], average_score_per_agent[agent], score)
88
89 winner_score = max(average_score_per_agent.values())
90 winner_agents = [
91 k for k, v in average_score_per_agent.items() if v == winner_score
92 ]
93 winner_strategy_idxs = [
94 consider_agents.index(winner) for winner in winner_agents
95 ]
96
97 # Select the winner as the one maximizing the selected statistics.
98 win_probabilities = np.zeros(num_strategies)
99 for winner_strategy_idx in winner_strategy_idxs:
100 win_probabilities[winner_strategy_idx] = 1 / len(winner_strategy_idxs)
101
102 distribution = np.zeros(num_strategies)
103 for agent, count in count_per_agent.items():
104 strategy_idx = consider_agents.index(agent)
105 distribution[strategy_idx] = count
106
107 distribution = tuple(distribution)
108
109 if distribution not in count_per_distribution:
110 count_per_distribution[distribution] = 1
111 win_prob_per_distribution[distribution] = win_probabilities
112 continue
113 (count_per_distribution[distribution],
114 win_prob_per_distribution[distribution]) = _inc_average(
115 count_per_distribution[distribution],
116 win_prob_per_distribution[distribution], win_probabilities)
117
118 # Populate self-play case (strategy both wins and loses).
119 for idx, agent in enumerate(consider_agents):
120 distribution = np.zeros(num_strategies)
121 distribution[idx] = num_players
122 distribution = tuple(distribution)
123 win_prob = np.zeros(num_strategies)
124 win_prob[idx] = 0.5
125 win_prob_per_distribution[distribution] = win_prob
126
127 # Create empty (nan) payoff table.
128 table = PayoffTable(num_players, num_strategies)
129
130 # Populate with win probabilities.
131 for distribution, payoff in win_prob_per_distribution.items():
132 table[distribution] = payoff
133
134 return table

137def from_matrix_game(matrix_game):
138 """Returns a PayOffTable given a symmetric 2-player matrix game.
139
140 Args:
141 matrix_game: The payoff matrix corresponding to a 2-player symmetric game.
142 """
143
144 if not isinstance(matrix_game, np.ndarray):
145 raise ValueError("The matrix game should be a numpy array, not a {}".format(
146 type(matrix_game)))
147 num_strats_per_population = (
148 utils.get_num_strats_per_population(
149 payoff_tables=[matrix_game], payoffs_are_hpt_format=False))
150 assert len(num_strats_per_population) == 2
151 assert num_strats_per_population[0] == num_strats_per_population[1]
152 num_strategies = num_strats_per_population[0]
153
154 num_profiles = utils.get_num_profiles(num_strats_per_population)
155 table = PayoffTable(num_players=2, num_strategies=num_strategies)
156
157 # Construct the HPT by filling in the corresponding payoffs for each profile
158 for id_profile in range(num_profiles):
159 strat_profile = utils.get_strat_profile_from_id(num_strats_per_population,
160 id_profile)
161 distribution = table.get_distribution_from_profile(strat_profile)
162 # For symmetric matrix games, multiple strategy profiles correspond to the
163 # same distribution and payoffs. Thus, ensure the table entry has not
164 # already been filled by a previous strategy profile.
165 if table.item_is_uninitialized(tuple(distribution)):
166 payoffs = np.zeros(num_strategies)
167 payoffs[strat_profile[0]] = matrix_game[strat_profile[0],
168 strat_profile[1]]
169 payoffs[strat_profile[1]] = matrix_game[strat_profile[1],
170 strat_profile[0]]
171 table[tuple(distribution)] = payoffs
172
173 return table

176def from_heuristic_payoff_table(hpt):
177 """Returns a `PayoffTable` instance from a numpy 2D HPT."""
178 [num_rows, num_columns] = hpt.shape
179 assert num_columns % 2 == 0
180 num_strategies = int(num_columns / 2)
181 num_players = np.sum(hpt[0, :num_strategies])
182 obj = PayoffTable(num_players, num_strategies, initialize_payoff_table=False)
183
184 # pylint: disable=protected-access
185 for row in hpt:
186 payoff_row = np.array(row[num_strategies:])
187 obj._payoff_table[tuple(row[:num_strategies])] = payoff_row
188
189 assert len(obj._payoff_table) == num_rows
190 # pylint: enable=protected-access
191 return obj

194def _compute_win_probability_from_elo(rating_1, rating_2):
195 """Computes the win probability of 1 vs 2 based on the provided Elo ratings.
196
197 Args:
198 rating_1: The Elo rating of player 1.
199 rating_2: The Elo rating of player 2.
200
201 Returns:
202 The win probability of player 1, when playing against 2.
203 """
204 m = max(rating_1, rating_2) # We subtract the max for numerical stability.
205
206 m1 = 10**((rating_1 - m) / 400)
207 m2 = 10**((rating_2 - m) / 400)
208
209 return m1 / (m1 + m2)

212def from_elo_scores(elo_ratings, num_agents=2):
213 """Computes the Elo win probability payoff matrix `X` from the Elo scores.
214
215 Args:
216 elo_ratings: The elo scores vector of length [num_strategies].
217 num_agents: The number of agents. Only 2 agents are supported for now.
218
219 Returns:
220 The HPT associated to the Elo win probability payoff matrix `X`. The score
221 for a given agent is given by its win probability given its Elo score.
222
223 Raises:
224 ValueError: If `num_agents != 2`.
225 """
226 if num_agents != 2:
227 raise ValueError("Only 2 agents are supported, because we need to compute "
228 "the win probability and that can only be computed with "
229 "2 players.")
230 num_strategies = len(elo_ratings)
231
232 hpt_rows = []
233
234 possible_teams = utils.distribute(num_agents, num_strategies, normalize=False)
235
236 for distribution_row in possible_teams:
237 payoff_row = np.zeros([num_strategies])
238 non_zero_index = np.nonzero(distribution_row)[0] # Why [0]?
239 assert len(non_zero_index.shape) == 1
240
241 if len(non_zero_index) > 1:
242 index_first_player, index_second_player = non_zero_index
243 prob = _compute_win_probability_from_elo(elo_ratings[index_first_player],
244 elo_ratings[index_second_player])
245 payoff_row[index_first_player] = prob
246 payoff_row[index_second_player] = 1 - prob
247 elif len(non_zero_index) == 1:
248 payoff_row[non_zero_index[0]] = 0.5
249 else:
250 assert False, "Impossible case, we have at least one strategy used."
251
252 hpt_rows.append(np.hstack([distribution_row, payoff_row]))
253
254 return NumpyPayoffTable(np.vstack(hpt_rows))

276 def expected_payoff(self, strategy):
277 """The expected payoff of each pure strategy against the mixed strategy.
278
279 We define the expected payoff of a strategy A as the expected payoff of
280 that strategy over the space of 2 randomly sampled
281
282 The mixed strategy is equivalently the composition of an infinitely large
283 population. To find the expected payoff, we:
284 1. Compute the probabilities of sampling each player distribution in the
285 heuristic payoff table from the population.
286 2. Compute the expected payoff of pure strategy against the mixed
287 strategy by averaging over the payoff rows with these probabilities.
288
289 For each pure strategy we must normalize by the probability that it appeared
290 in the player distribution at all; otherwise we would be undercounting.
291
292 For more details, see https://arxiv.org/pdf/1803.06376.pdf.
293
294 Args:
295 strategy: an `np.array(shape=self._num_strategies)` of probabilities.
296
297 Returns:
298 An `np.array(shape=self._num_strategies)` of payoffs for pure strategies.
299
300 Raises:
301 ValueError: if the provided strategy probabilities do not define a valid
302 distribution over `self._num_strategies` strategies.
303 """
304 if strategy.shape != (self.num_strategies,):
305 raise ValueError("The strategy probabilities should be of shape "
306 "({},), not {}".format(self.num_strategies,
307 strategy.shape))
308 if np.around(np.sum(strategy), decimals=3) != 1.0:
309 raise ValueError("The strategy probabilities should sum to 1.")
310 if not all([p >= 0 for p in strategy]):
311 raise ValueError("The strategy probabilities should all be >= 0.")
312
313 distributions = self._distributions.astype(int)
314 if not np.all(np.isclose(self._distributions, distributions, 1e-10)):
315 raise ValueError("Conversion to integers for distributions failed.")
316
317 # Multinomial coefficients (one per distribution).
318 coefficients = _multinomial_coefficients(distributions)
319 # Probabilities of sampling each distribution given population composition.
320 probabilities = _row_probabilities(coefficients, distributions, strategy)
321
322 return _expected_payoff(probabilities, self._payoffs, strategy,
323 self._num_players)

336class NumpyPayoffTable(object):
337 """An object wrapping a Numpy array heuristic payoff table for a metagame.
338
339 NOTE: We assume the number of players to be equal to the number of
340 replicators.
341
342 """
343
344 def __init__(self, payoff_table, writeable=False):
345 """Initializes an immutable payoff table.
346
347 Let p be the number of players, k be the number of strategies. Then, there
348 are Combinations(p + k - 1, k - 1) distinct configurations for the
349 strategies of the p players.
350
351 The payoff table is of shape [(p + k - 1)! / (p! * (k - 1)!), 2 * k].
352
353 The first k columns encode the number of players playing each strategies.
354
355 The second k columns encode the average payoff of each strategy in that
356 game.
357
358 Args:
359 payoff_table: A numpy heuristic payoff table, which is assumed to be
360 correctly constructed.
361 writeable: Whether the numpy array payoff_table should be writeable. See
362 https://docs.scipy.org/doc/numpy-1.15.1/reference/generated/numpy.ndarray.flags.html.
363 However, locking a base object does not lock any views that already
364 reference it,
365 """
366 self._writeable = writeable
367 self._payoff_table = payoff_table
368
369 [self._num_rows, num_columns] = self._payoff_table.shape
370 assert num_columns % 2 == 0
371 self._num_strategies = int(num_columns / 2)
372 self._num_players = np.sum(self._payoff_table[0, :self._num_strategies])
373
374 def __call__(self):
375 """Returns a view of the table as a np.array.
376
377 The mutability of the object is controlled by `writeable`.
378 """
379 if self._writeable:
380 return self._payoff_table
381 else:
382 return np.copy(self._payoff_table)

486# The following provides utility functions to compute the expected payoff of
487# a given strategy profile.
488# See https://arxiv.org/pdf/1803.06376.pdf, page 3, left column.
489#
490# Usage:
491#
492# coefficients = _multinomial_coefficients(distributions, strategies):
493# row_probabilities = _row_probabilities(coefficients, distributions, strategy)
494# expected_payoff = _expected_payoff(row_probabilities, payoffs, composition,
495# num_players)
496#
497#
498def _multinomial_coefficients(distributions):
499 """Returns the multinomial coefficients.
500
501 Args:
502 distributions: The distributions table [num_rows, num_strategies].
503 """
504 v_factorial = np.vectorize(np.math.factorial)
505 # Multinomial coefficients (one per distribution Ni).
506 # ( P )
507 # ( Ni1, Ni1, ... Nik )
508 coefficients = (
509 v_factorial(np.sum(distributions, axis=1)) /
510 np.prod(v_factorial(distributions), axis=1))
511
512 return coefficients
513

515def _row_probabilities(coefficients, distributions, strategy):
516 """Returns the row probabilities [num_rows].
517
518 Args:
519 coefficients: The multinomial coefficients [num_rows].
520 distributions: The distributions table [num_rows, num_strategies].
521 strategy: The strategy array [num_strategies].
522 """
523 row_probabilities = coefficients * np.prod(
524 np.power(strategy, distributions), axis=1)
525 return row_probabilities

528def _expected_payoff(row_probabilities, payoffs, strategy, num_players):
529 # pylint: disable=g-doc-args
530 r"""Returns the expected payoff.
531
532 Computes (with p=num_players):
533
534 r_j = \sum_i row_probabilities[i] * payoffs[i, j] / (1 - (1-strategy[j])^p)
535 """
536 # pylint: enable=g-doc-args
537 [num_rows] = row_probabilities.shape
538 [num_rows_2, num_strategies] = payoffs.shape
539 [num_strategies_2] = strategy.shape
540 assert num_rows == num_rows_2
541 assert num_strategies == num_strategies_2
542
543 # One per pure strategy.
544 numerators = np.dot(np.transpose(payoffs), row_probabilities)
545 # One per pure strategy.
546 denominators = 1 - np.power(1 - strategy, num_players)
547 return numerators / denominators
