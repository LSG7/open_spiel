28def _sum_j_x_j_ln_x_j_over_x_i(x):
29 r"""Computes \sum_j x_j ln(x_j / x_i)."""
30 # By having a = x.reshape([1, -1]) and b = x.reshape([-1, 1]), we can use
31 # broadcasting and have:
32 # (a / b)[i, j] = x_j / x_i
33 # thus giving:
34 # \sum_j x_j * log(x_j/ x_i) = sum(a * ln (a/b), axis=1)
35
36 a = x.reshape([1, -1])
37 b = x.reshape([-1, 1])
38
39 return np.sum(a * np.log(np.divide(a, b)), axis=1)

42def _q_learning_dynamics(composition, payoff, temperature):
43 r"""An equivalent implementation of `dynamics.boltzmannq`."""
44 return 1 / temperature * dynamics.replicator(composition, payoff) + (
45 composition * _sum_j_x_j_ln_x_j_over_x_i(composition))

52 # internals are correct.
53 x = np.asarray([1., 2., 3.])

55 # We use 2 different formula to check we have the correct result.

59 expected_0 = 1. * log(1 / 1.) + 2 * log(2 / 1.) + 3 * log(3 / 1.)
60 expected_1 = 1. * log(1 / 2.) + 2 * log(2 / 2.) + 3 * log(3 / 2.)
61 expected_2 = 1. * log(1 / 3.) + 2 * log(2 / 3.) + 3 * log(3 / 3.)
