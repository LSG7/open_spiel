15"""Implements an Ephemeral Value Adjustment Agent.
16
17See https://arxiv.org/abs/1810.08163.
18The algorithm queries trajectories from a replay buffer based on similarities
19to embedding representations and uses a parametric model to compute values for
20counterfactual state-action pairs when integrating across those trajectories.
21Finally, a weighted average between the parametric (DQN in this case) and the
22non-parametric model is used to compute the policy.
23"""

45class QueryableFixedSizeRingBuffer(dqn.ReplayBuffer):
46 """ReplayBuffer of fixed size with a FIFO replacement policy.
47
48 Stored transitions can be sampled uniformly. This extends the DQN replay
49 buffer by allowing the contents to be fetched by L2 proximity to a query
50 value.
51 The underlying datastructure is a ring buffer, allowing 0(1) adding and
52 sampling.
53 """
54
55 def knn(self, key, key_name, k, trajectory_len=1):
56 """Computes top-k neighbours based on L2 distance.
57
58 Args:
59 key: (np.array) key value to query memory.
60 key_name: (str) attribute name of key in memory elements.
61 k: (int) number of neighbours to fetch.
62 trajectory_len: (int) length of trajectory to fetch from replay buffer.
63
64 Returns:
65 List of tuples (L2 negative distance, BufferElement) sorted in increasing
66 order by the negative L2 distqances from the key.
67 """
68 distances = [(np.linalg.norm(getattr(sample, key_name) - key, 2,
69 axis=0), sample) for sample in self._data]
70 return sorted(distances, key=lambda v: -v[0])[:k]

100 embedding_as_parametric_input=False):
101 """Initialize the Ephemeral VAlue Adjustment algorithm.
102
103 Args:
104 game: (rl_environment.Environment) Open Spiel game.
105 player_id: (int) Player id for this player.
106 state_size: (int) Size of info state vector.
107 num_actions: (int) number of actions.
108 embedding_network_layers: (list[int]) Layer sizes of strategy net MLP.
109 embedding_size: (int) Size of memory embeddings.
110 dqn_hidden_layers: (list(int)) MLP layer sizes of DQN network.
111 batch_size: (int) Size of batches for DQN learning steps.
112 trajectory_len: (int) Length of trajectories from replay buffer.
113 num_neighbours: (int) Number of neighbours to fetch from replay buffer.
114 learning_rate: (float) Learning rate.
115 mixing_parameter: (float) Value mixing parameter between 0 and 1.
116 memory_capacity: Number af samples that can be stored in memory.
117 discount_factor: (float) Discount factor for Q-Learning.
118 update_target_network_every: How often to update DQN target network.
119 epsilon_start: (float) Starting epsilon-greedy value.
120 epsilon_end: (float) Final epsilon-greedy value.
121 epsilon_decay_duration: (float) Number of steps over which epsilon decays.
122 embedding_as_parametric_input: (bool) Whether we use embeddings as input
123 to the parametric model.
124 """
125 assert (mixing_parameter >= 0 and mixing_parameter <= 1)
126 self._game = game
127 self.player_id = player_id
128 self._env = game
129 self._num_actions = num_actions
130 self._info_state_size = state_size
131 self._embedding_size = embedding_size
132 self._lambda = mixing_parameter
133 self._trajectory_len = trajectory_len
134 self._num_neighbours = num_neighbours
135 self._discount = discount_factor
136 self._epsilon_start = epsilon_start
137 self._epsilon_end = epsilon_end
138 self._epsilon_decay_duration = epsilon_decay_duration
139 self._last_time_step = None
140 self._last_action = None
141 self._embedding_as_parametric_input = embedding_as_parametric_input
142
143 self._embedding_network = dqn.MLP(self._info_state_size,
144 list(embedding_network_layers),
145 embedding_size)
146
147 # The DQN agent requires this be an integer.
148 if not isinstance(memory_capacity, int):
149 raise ValueError("Memory capacity not an integer.")
150
151 # Initialize the parametric & non-parametric Q-networks.
152 self._agent = dqn.DQN(
153 player_id,
154 state_representation_size=self._info_state_size,
155 num_actions=self._num_actions,
156 hidden_layers_sizes=list(dqn_hidden_layers),
157 replay_buffer_capacity=memory_capacity,
158 replay_buffer_class=QueryableFixedSizeRingBuffer,
159 batch_size=batch_size,
160 learning_rate=learning_rate,
161 update_target_network_every=update_target_network_every,
162 learn_every=batch_size,
163 discount_factor=1.0,
164 epsilon_start=1.0,
165 epsilon_end=0.1,
166 epsilon_decay_duration=int(1e6))
167 # Initialize Value Buffers - Fetch Replay buffers from agents.
168 self._value_buffer = QueryableFixedSizeRingBuffer(memory_capacity)
169 self._replay_buffer = self._agent.replay_buffer
170
171 # Initialize non-parametric & EVA Q-values.
172 self._v_np = collections.defaultdict(float)
173 self._q_np = collections.defaultdict(lambda: [0] * self._num_actions)
174 self._q_eva = collections.defaultdict(lambda: [0] * self._num_actions)

184 def _add_transition_value(self, infostate_embedding, value):
185 """Adds the embedding and value to the ValueBuffer.

187 Args:
188 infostate_embedding: (np.array) embeddig vector.

194 def _add_transition_replay(self, infostate_embedding, time_step):
195 """Adds the new transition using `time_step` to the replay buffer.
196
197 Adds the transition from `self._prev_timestep` to `time_step` by
198 `self._prev_action`.
199 Args:
200 infostate_embedding: embeddig vector.
201 time_step: an instance of rl_environment.TimeStep.
202 """
203 prev_timestep = self._last_time_step
204 assert prev_timestep is not None
205 legal_actions = (
206 prev_timestep.observations["legal_actions"][self.player_id])
207 legal_actions_mask = np.zeros(self._num_actions)
208 legal_actions_mask[legal_actions] = 1.0
209 reward = time_step.rewards[self.player_id] if time_step.rewards else 0.0
210 transition = ReplayBufferElement(
211 embedding=infostate_embedding,
212 info_state=(prev_timestep.observations["info_state"][self.player_id]),
213 action=self._last_action,
214 reward=reward,
215 next_info_state=time_step.observations["info_state"][self.player_id],
216 is_final_step=float(time_step.last()),
217 legal_actions_mask=legal_actions_mask)
218 self._replay_buffer.add(transition)

220 def step(self, time_step, is_evaluation=False):
221 """Returns the action to be taken and updates the value functions.
222
223 Args:
224 time_step: an instance of rl_environment.TimeStep.
225 is_evaluation: bool, whether this is a training or evaluation call.
226
227 Returns:
228 A `rl_agent.StepOutput` containing the action probs and chosen action.
229 """
230 # Act step: don't act at terminal info states.
231 if not time_step.last():
232 info_state = time_step.observations["info_state"][self.player_id]
233 legal_actions = time_step.observations["legal_actions"][self.player_id]
234 epsilon = self._get_epsilon(self._agent.step_counter, is_evaluation)
235
236 # Sample an action from EVA via epsilon greedy policy.
237 action, probs = self._epsilon_greedy(self._q_eva[tuple(info_state)],
238 legal_actions, epsilon)
239
240 # Update Step: Only with transitions and not when evaluating.
241 if (not is_evaluation and self._last_time_step is not None):
242 info_state = self._last_time_step.observations["info_state"][
243 self.player_id]
244 legal_actions = self._last_time_step.observations["legal_actions"][
245 self.player_id]
246 epsilon = self._get_epsilon(self._agent.step_counter, is_evaluation)
247
248 # Get embedding.
249 self._info_state = torch.Tensor(np.expand_dims(info_state, axis=0))
250 infostate_embedding = self._embedding_network(
251 self._info_state).detach()[0]
252
253 neighbours_value = self._value_buffer.knn(infostate_embedding,
254 MEM_KEY_NAME,
255 self._num_neighbours, 1)
256 # collect trace values of knn from L (value buffer) .. Q_np(s_k)
257 neighbours_replay = self._replay_buffer.knn(infostate_embedding,
258 MEM_KEY_NAME,
259 self._num_neighbours,
260 self._trajectory_len)
261
262 # Take a step with the parametric model and get q-values. Use embedding as
263 # input to the parametric meodel.
264 # TODO(author6) Recompute embeddings for buffers on learning steps.
265 if self._embedding_as_parametric_input:
266 last_time_step_copy = copy.deepcopy(self._last_time_step)
267 last_time_step_copy.observations["info_state"][
268 self.player_id] = infostate_embedding
269 self._agent.step(last_time_step_copy, add_transition_record=False)
270 else:
271 self._agent.step(self._last_time_step, add_transition_record=False)
272 q_values = self._agent._q_network(self._info_state).detach()[0]
273 # Update EVA: Q_eva = lambda q_theta(s_t) + (1-lambda) sum(Q_np(s_k, .))/K
274 for a in legal_actions:
275 q_theta = q_values[a]
276 self._q_eva[tuple(info_state)][a] = (
277 self._lambda * q_theta + (1 - self._lambda) *
278 sum([elem[1].value
279 for elem in neighbours_value]) / self._num_neighbours)
280
281 # Append (e,s,a,r,s') to Replay Buffer
282 self._add_transition_replay(infostate_embedding, time_step)
283
284 # update Q_np with Traces using TCP
285 self._trajectory_centric_planning(neighbours_replay)
286
287 # Append Q_np(s, a) to Value Buffer
288 self._add_transition_value(
289 infostate_embedding, self._q_np[tuple(info_state)][self._last_action])
290
291 # Prepare for the next episode.
292 if time_step.last():
293 self._last_time_step = None
294 self._last_action = None
295 return
296
297 self._last_time_step = time_step
298 self._last_action = action
299 return rl_agent.StepOutput(action=action, probs=probs)

301 def _trajectory_centric_planning(self, trajectories):
302 """Performs trajectory centric planning.
303
304 Uses trajectories from the replay buffer to update the non-parametric values
305 while supplying counter-factual values with the parametric model.
306 Args:
307 trajectories: Current OpenSpiel game state.
308 """
309 # Calculate non-parametric values over the trajectories.
310 # Iterate backward through trajectories
311 for t in range(len(trajectories) - 1, 0, -1):
312 elem = trajectories[t][1]
313 s_tp1 = tuple(elem.next_info_state)
314 s_t = tuple(elem.info_state)
315 a_t = elem.action
316 r_t = elem.reward
317 legal_actions = elem.legal_actions_mask
318 if t < len(trajectories) - 1:
319 for action in range(len(legal_actions)):
320 if not legal_actions[action]:
321 continue
322 if action == elem.action:
323 self._q_np[s_t][a_t] = (r_t + self._discount * self._v_np[s_tp1])
324 else:
325 self._agent.info_state = torch.Tensor(
326 np.expand_dims(elem.info_state, axis=0))
327 q_values_parametric = self._agent._q_network(
328 self._agent.info_state).detach().numpy()
329 self._q_np[s_t][a_t] = q_values_parametric[0][action]
330
331 # Set V(s_t)
332 if t == len(trajectories) - 1:
333 # Sample from the parametric model.
334 self._agent.info_state = torch.Tensor(
335 np.expand_dims(elem.info_state, axis=0))
336 q_values_parametric = self._agent._q_network(
337 self._agent.info_state).detach().numpy()
338 self._v_np[s_t] = np.max(q_values_parametric)
339 else:
340 self._v_np[s_t] = max(self._q_np[s_t])

342 def _epsilon_greedy(self, q_values, legal_actions, epsilon):
343 """Returns a valid epsilon-greedy action and valid action probs.
344
345 Action probabilities are given by a softmax over legal q-values.
346 Args:
347 q_values: list of Q-values by action.
348 legal_actions: list of legal actions at `info_state`.
349 epsilon: float, probability of taking an exploratory action.
350
351 Returns:
352 A valid epsilon-greedy action and valid action probabilities.
353 """
354 probs = np.zeros(self._num_actions)
355 q_values = np.array(q_values)
356 if np.random.rand() < epsilon:
357 action = np.random.choice(legal_actions)
358 probs[legal_actions] = 1.0 / len(legal_actions)
359 else:
360 legal_q_values = q_values[legal_actions]
361 action = legal_actions[np.argmax(legal_q_values)]
362 # Reduce max_q for numerical stability. Result is the same.
363 max_q = np.max(legal_q_values)
364 e_x = np.exp(legal_q_values - max_q)
365 probs[legal_actions] = e_x / e_x.sum(axis=0)
366 return action, probs

368 def _get_epsilon(self, step_counter, is_evaluation):
369 """Returns the evaluation or decayed epsilon value."""
370 if is_evaluation:
371 return 0.0
372 decay_steps = min(step_counter, self._epsilon_decay_duration)
373 decayed_epsilon = (
374 self._epsilon_end + (self._epsilon_start - self._epsilon_end) *
375 (1 - decay_steps / self._epsilon_decay_duration))
376 return decayed_epsilon
