15 """Implements Deep CFR Algorithm.
16 
17 See https://arxiv.org/abs/1811.00164.
18 
19 The algorithm defines an `advantage` and `strategy` networks that compute
20 advantages used to do regret matching across information sets and to approximate
21 the strategy profiles of the game. To train these networks a reservoir buffer
22 (other data structures may be used) memory is used to accumulate samples to
23 train the networks.
24 
25 This implementation uses skip connections as described in the paper if two
26 consecutive layers of the advantage or policy network have the same number
27 of units, except for the last connection. Before the last hidden layer
28 a layer normalization is applied.
29 """

54 class ReservoirBuffer(object):
55  """Allows uniform sampling over a stream of data.
56 
57  This class supports the storage of arbitrary elements, such as observation
58  tensors, integer actions, etc.
59 
60  See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
61  """
62 
63  def __init__(self, reservoir_buffer_capacity):
64  self._reservoir_buffer_capacity_reservoir_buffer_capacity = reservoir_buffer_capacity
65  self._data_data = []
66  self._add_calls_add_calls = 0
67 
68  def add(self, element):
69  """Potentially adds `element` to the reservoir buffer.
70 
71  Args:
72  element: data to be added to the reservoir buffer.
73  """
74  if len(self._data_data) < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
75  self._data_data.append(element)
76  else:
77  idx = np.random.randint(0, self._add_calls_add_calls + 1)
78  if idx < self._reservoir_buffer_capacity_reservoir_buffer_capacity:
79  self._data_data[idx] = element
80  self._add_calls_add_calls += 1
81 
82  def sample(self, num_samples):
83  """Returns `num_samples` uniformly sampled from the buffer.
84 
85  Args:
86  num_samples: `int`, number of samples to draw.
87 
88  Returns:
89  An iterable over `num_samples` random elements of the buffer.
90 
91  Raises:
92  ValueError: If there are less than `num_samples` elements in the buffer
93  """
94  if len(self._data_data) < num_samples:
95  raise ValueError('{} elements could not be sampled from size {}'.format(
96  num_samples, len(self._data_data)))
97  return random.sample(self._data_data, num_samples)
98 
99  def clear(self):
100  self._data_data = []
101  self._add_calls_add_calls = 0

104  return len(self._data_data)
105 
106  def __iter__(self):
107  return iter(self._data_data)

111  return self._data_data

263  def _get_jitted_matched_regrets(self):
264  """get jitted regret matching function."""
265 
266  @jax.jit
267  def get_matched_regrets(info_state, legal_actions_mask, params_adv):
268  advs = self._hk_adv_network_hk_adv_network.apply(params_adv, info_state,
269  legal_actions_mask)
270  advantages = jnp.maximum(advs, 0)
271  summed_regret = jnp.sum(advantages)
272  matched_regrets = jax.lax.cond(
273  summed_regret > 0, lambda _: advantages / summed_regret,
274  lambda _: jax.nn.one_hot( # pylint: disable=g-long-lambda
275  jnp.argmax(jnp.where(legal_actions_mask == 1, advs, -10e20)), self
276  ._num_actions), None)
277  return advantages, matched_regrets
278 
279  return get_matched_regrets

336  advantage_losses = collections.defaultdict(list)
337  for _ in range(self._num_iterations_num_iterations):

339  for _ in range(self._num_traversals_num_traversals):

342  # Re-initialize advantage network for p and train from scratch.
343  self._reinitialize_advantage_network_reinitialize_advantage_network(p)
344  advantage_losses[p].append(self._learn_advantage_network_learn_advantage_network(p))
345  self._iteration_iteration += 1
346  # Train policy network.
347  policy_loss = self._learn_strategy_network_learn_strategy_network()

416  def _traverse_game_tree(self, state, player):
417  """Performs a traversal of the game tree using external sampling.
418 
419  Over a traversal the advantage and strategy memories are populated with
420  computed advantage values and matched regrets respectively.
421 
422  Args:
423  state: Current OpenSpiel game state.
424  player: (int) Player index for this traversal.
425 
426  Returns:
427  Recursively returns expected payoffs for each action.
428  """
429  if state.is_terminal():
430  # Terminal state get returns.
431  return state.returns()[player]
432  elif state.is_chance_node():
433  # If this is a chance node, sample an action
434  chance_outcome, chance_proba = zip(*state.chance_outcomes())
435  action = np.random.choice(chance_outcome, p=chance_proba)
436  return self._traverse_game_tree_traverse_game_tree(state.child(action), player)
437  elif state.current_player() == player:
438  # Update the policy over the info set & actions via regret matching.
439  _, strategy = self._sample_action_from_advantage_sample_action_from_advantage(state, player)
440  strategy = np.array(strategy)
441  exp_payoff = 0 * strategy
442  for action in state.legal_actions():
443  exp_payoff[action] = self._traverse_game_tree_traverse_game_tree(
444  state.child(action), player)
445  ev = np.sum(exp_payoff * strategy)
446  samp_regret = (exp_payoff - ev) * state.legal_actions_mask(player)
447  self._advantage_memories_advantage_memories[player].add(
448  self._serialize_advantage_memory_serialize_advantage_memory(state.information_state_tensor(),
449  self._iteration_iteration, samp_regret,
450  state.legal_actions_mask(player)))
451  return ev
452  else:
453  other_player = state.current_player()
454  _, strategy = self._sample_action_from_advantage_sample_action_from_advantage(state, other_player)
455  # Recompute distribution for numerical errors.
456  probs = np.array(strategy)
457  probs /= probs.sum()
458  sampled_action = np.random.choice(range(self._num_actions_num_actions), p=probs)
459  self._add_to_strategy_memory_add_to_strategy_memory(
460  state.information_state_tensor(other_player), self._iteration_iteration, probs,
461  state.legal_actions_mask(other_player))
462  return self._traverse_game_tree_traverse_game_tree(state.child(sampled_action), player)
