62 ActionsAndProbs ISMCTSBot::RunSearch(const State& state) {
63  Reset();
64  SPIEL_CHECK_EQ(state.GetGame()->GetType().dynamics,
65  GameType::Dynamics::kSequential);
66  SPIEL_CHECK_EQ(state.GetGame()->GetType().information,
67  GameType::Information::kImperfectInformation);
68 
69  // Optimization in case of single legal action, and support for games which
70  // do not support ResampleFromInfostate in certain specific (single action)
71  // states.
72  std::vector<Action> legal_actions = state.LegalActions();
73  if (legal_actions.size() == 1) return {{legal_actions[0], 1.0}};
74 
75  root_node_ = CreateNewNode(state);
76  SPIEL_CHECK_TRUE(root_node_ != nullptr);
77 
78  auto root_infostate_key = GetStateKey(state);
79 
80  for (int sim = 0; sim < max_simulations_; ++sim) {
81  std::unique_ptr<State> sampled_root_state = SampleRootState(state);
82  SPIEL_CHECK_TRUE(root_infostate_key == GetStateKey(*sampled_root_state));
83  SPIEL_CHECK_TRUE(sampled_root_state != nullptr);
84  RunSimulation(sampled_root_state.get());
85  }
86 
87  if (allow_inconsistent_action_sets_) {
88  // Filter illegals for this state.
89  std::vector<Action> legal_actions = state.LegalActions();
90  ISMCTSNode temp_node = FilterIllegals(root_node_, legal_actions);
91  SPIEL_CHECK_GT(temp_node.total_visits, 0);
92  return GetFinalPolicy(state, &temp_node);
93  } else {
94  return GetFinalPolicy(state, root_node_);
95  }
96 }

114 ActionsAndProbs ISMCTSBot::GetFinalPolicy(const State& state,
115  ISMCTSNode* node) const {
116  ActionsAndProbs policy;
117  SPIEL_CHECK_FALSE(node == nullptr);
118 
119  switch (final_policy_type_) {
120  case ISMCTSFinalPolicyType::kNormalizedVisitCount: {
121  SPIEL_CHECK_GT(node->total_visits, 0);
122  policy.reserve(node->child_info.size());
123  double total_visits = static_cast<double>(node->total_visits);
124  for (const auto& action_and_child : node->child_info) {
125  policy.push_back({action_and_child.first,
126  action_and_child.second.visits / total_visits});
127  }
128  } break;
129 
130  case ISMCTSFinalPolicyType::kMaxVisitCount: {
131  SPIEL_CHECK_GT(node->total_visits, 0);
132  policy.reserve(node->child_info.size());
133  Action max_action = kInvalidAction;
134  int max_visits = -std::numeric_limits<int>::infinity();
135  for (const auto& action_and_child : node->child_info) {
136  if (action_and_child.second.visits > max_visits) {
137  max_visits = action_and_child.second.visits;
138  max_action = action_and_child.first;
139  }
140  }
141  SPIEL_CHECK_NE(max_action, kInvalidAction);
142  for (const auto& action_and_child : node->child_info) {
143  policy.push_back({action_and_child.first,
144  action_and_child.first == max_action ? 1.0 : 0.0});
145  }
146  } break;
147 
148  case ISMCTSFinalPolicyType::kMaxValue: {
149  SPIEL_CHECK_GT(node->total_visits, 0);
150  policy.reserve(node->child_info.size());
151  Action max_action = kInvalidAction;
152  double max_value = -std::numeric_limits<double>::infinity();
153  for (const auto& action_and_child : node->child_info) {
154  double value = action_and_child.second.value();
155  if (value > max_value) {
156  max_value = value;
157  max_action = action_and_child.first;
158  }
159  }
160  SPIEL_CHECK_NE(max_action, kInvalidAction);
161  for (const auto& action_and_child : node->child_info) {
162  policy.push_back({action_and_child.first,
163  action_and_child.first == max_action ? 1.0 : 0.0});
164  }
165  }
166  }
167 
168  // In case the search didn't cover all the legal moves, at zero probability
169  // for all the remaining actions.
170  int policy_size = policy.size();
171  std::vector<Action> legal_actions = state.LegalActions();
172  if (policy_size < legal_actions.size()) {
173  for (Action action : legal_actions) {
174  if (node->child_info.find(action) == node->child_info.end()) {
175  // Legal action not found in the node's actions: assign probability 0.
176  policy.push_back({action, 0.0});
177  }
178  }
179  }
180  return policy;
181 }

183 std::unique_ptr<State> ISMCTSBot::SampleRootState(const State& state) {
184  if (max_world_samples_ == kUnlimitedNumWorldSamples) {
185  return ResampleFromInfostate(state);
186  } else if (root_samples_.size() < max_world_samples_) {
187  root_samples_.push_back(ResampleFromInfostate(state));
188  return root_samples_.back()->Clone();
189  } else if (root_samples_.size() == max_world_samples_) {
190  int idx = absl::Uniform(rng_, 0u, root_samples_.size());
191  return root_samples_[idx]->Clone();
192  } else {
193  SpielFatalError("Case not handled (badly set max_world_samples..?)");
194  }
195 }

236 ISMCTSNode ISMCTSBot::FilterIllegals(
237  ISMCTSNode* node, const std::vector<Action>& legal_actions) const {
238  ISMCTSNode new_node = *node;
239  std::vector<Action> to_delete;
240  for (const auto& action_and_child : node->child_info) {
241  if (std::find(legal_actions.begin(), legal_actions.end(),
242  action_and_child.first) == legal_actions.end()) {
243  // Illegal action: mark for deletion.
244  new_node.total_visits -= action_and_child.second.visits;
245  new_node.child_info.erase(action_and_child.first);
246  }
247  }
248 
249  return new_node;
250 }

258 Action ISMCTSBot::SelectActionTreePolicy(
259  ISMCTSNode* node, const std::vector<Action>& legal_actions) {
260  // Check to see if we are allowing inconsistent action sets.
261  if (allow_inconsistent_action_sets_) {
262  // If so, it could mean that the node has actions with child info that are
263  // not legal in this state, so we have to remove them.
264  ISMCTSNode temp_node = FilterIllegals(node, legal_actions);
265  if (temp_node.total_visits == 0) {
266  // If we've filtered everything, return a random action.
267  Action action =
268  legal_actions[absl::Uniform(rng_, 0u, legal_actions.size())];
269  ExpandIfNecessary(node, action);
270  return action;
271  } else {
272  return SelectActionUCB(&temp_node);
273  }
274  } else {
275  return SelectActionUCB(node);
276  }
277 }

279 Action ISMCTSBot::SelectActionUCB(ISMCTSNode* node) {
280  std::vector<Action> candidates;
281  double max_value = -std::numeric_limits<double>::infinity();
282 
283  for (const auto& action_and_child : node->child_info) {
284  // Every child should have at least one visit because the child is only
285  // created when the action took it in a simulation, which then increases
286  // its visit count immediately.
287  SPIEL_CHECK_GT(action_and_child.second.visits, 0);
288 
289  Action action = action_and_child.first;
290  double uct_val = action_and_child.second.value() +
291  uct_c_ * std::sqrt(std::log(node->total_visits) /
292  action_and_child.second.visits);
293 
294  if (uct_val > max_value + kTieTolerance) {
295  candidates.clear();
296  candidates.push_back(action);
297  max_value = uct_val;
298  } else if (uct_val > max_value - kTieTolerance &&
299  uct_val < max_value + kTieTolerance) {
300  candidates.push_back(action);
301  max_value = uct_val;
302  }
303  }
304 
305  SPIEL_CHECK_GE(candidates.size(), 1);
306 
307  if (candidates.size() == 1) {
308  return candidates[0];
309  } else {
310  return candidates[absl::Uniform(rng_, 0u, candidates.size())];
311  }
312 }

314 Action ISMCTSBot::CheckExpand(ISMCTSNode* node,
315  const std::vector<Action>& legal_actions) {
316  // Fast check in the common/default case.
317  if (!allow_inconsistent_action_sets_ &&
318  node->child_info.size() == legal_actions.size()) {
319  return kInvalidAction;
320  }
321 
322  // Shuffle the legal actions to remove the bias from the move order.
323  std::vector<Action> legal_actions_copy = legal_actions;
324  std::shuffle(legal_actions_copy.begin(), legal_actions_copy.end(), rng_);
325  for (Action action : legal_actions_copy) {
326  if (node->child_info.find(action) == node->child_info.end()) {
327  return action;
328  }
329  }
330  return kInvalidAction;
331 }

333 std::vector<double> ISMCTSBot::RunSimulation(State* state) {
334  if (state->IsTerminal()) {
335  return state->Returns();
336  } else if (state->IsChanceNode()) {
337  Action chance_action =
338  SampleAction(state->ChanceOutcomes(), RandomNumber()).first;
339  state->ApplyAction(chance_action);
340  return RunSimulation(state);
341  }
342 
343  std::vector<Action> legal_actions = state->LegalActions();
344  Player cur_player = state->CurrentPlayer();
345  ISMCTSNode* node = LookupOrCreateNode(*state);
346  SPIEL_CHECK_TRUE(node != nullptr);
347 
348  if (node->total_visits == kUnexpandedVisitCount) {
349  // Newly created node, so we've just stepped out of the tree.
350  node->total_visits = 0; // Expand the node.
351  return evaluator_->Evaluate(*state);
352  } else {
353  // Apply tree policy.
354  Action chosen_action = CheckExpand(node, legal_actions);
355  if (chosen_action != kInvalidAction) {
356  // Expand.
357  ExpandIfNecessary(node, chosen_action);
358  } else {
359  // No expansion, so use the tree policy to select.
360  chosen_action = SelectActionTreePolicy(node, legal_actions);
361  }
362 
363  SPIEL_CHECK_NE(chosen_action, kInvalidAction);
364 
365  // Need to updates the visits before the recursive call. In games with
366  // imperfect recall, a node could be expanded with zero visit counts, and
367  // you might encounter the same (node, action) pair in the same simulation
368  // and the denominator for the UCT formula would be 0.
369  node->total_visits++;
370  node->child_info[chosen_action].visits++;
371 
372  state->ApplyAction(chosen_action);
373  std::vector<double> returns = RunSimulation(state);
374  node->child_info[chosen_action].return_sum += returns[cur_player];
375  return returns;
376  }
