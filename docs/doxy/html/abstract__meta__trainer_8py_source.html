30def _process_string_or_callable(string_or_callable, dictionary):
31 """Process a callable or a string representing a callable.
32
33 Args:
34 string_or_callable: Either a string or a callable
35 dictionary: Dictionary of shape {string_reference: callable}
36
37 Returns:
38 string_or_callable if string_or_callable is a callable ; otherwise,
39 dictionary[string_or_callable]
40
41 Raises:
42 NotImplementedError: If string_or_callable is of the wrong type, or has an
43 unexpected value (Not present in dictionary).
44 """
45 if callable(string_or_callable):
46 return string_or_callable
47
48 try:
49 return dictionary[string_or_callable]
50 except KeyError as e:
51 raise NotImplementedError("Input type / value not supported. Accepted types"
52 ": string, callable. Acceptable string values : "
53 "{}. Input provided : {}".format(
54 list(dictionary.keys()),
55 string_or_callable)) from e

58def sample_episode(state, policies):
59 """Samples an episode using policies, starting from state.
60
61 Args:
62 state: Pyspiel state representing the current state.
63 policies: List of policy representing the policy executed by each player.
64
65 Returns:
66 The result of the call to returns() of the final state in the episode.
67 Meant to be a win/loss integer.
68 """
69 if state.is_terminal():
70 return np.array(state.returns(), dtype=np.float32)
71
72 if state.is_simultaneous_node():
73 actions = [None] * state.num_players()
74 for player in range(state.num_players()):
75 state_policy = policies[player](state, player)
76 outcomes, probs = zip(*state_policy.items())
77 actions[player] = utils.random_choice(outcomes, probs)
78 state.apply_actions(actions)
79 return sample_episode(state, policies)
80
81 if state.is_chance_node():
82 outcomes, probs = zip(*state.chance_outcomes())
83 else:
84 player = state.current_player()
85 state_policy = policies[player](state)
86 outcomes, probs = zip(*state_policy.items())
87
88 state.apply_action(utils.random_choice(outcomes, probs))
89 return sample_episode(state, policies)

92class AbstractMetaTrainer(object):
93 """Abstract class implementing meta trainers.
94
95 If a trainer is something that computes a best response to given environment &
96 agents, a meta trainer will compute which best responses to compute (Against
97 what, how, etc)
98 This class can support PBT, Hyperparameter Evolution, etc.
99 """
100
101 # pylint:disable=dangerous-default-value
102 def __init__(self,
103 game,
104 oracle,
105 initial_policies=None,
106 meta_strategy_method=_DEFAULT_META_STRATEGY_METHOD,
107 training_strategy_selector=_DEFAULT_STRATEGY_SELECTION_METHOD,
108 symmetric_game=False,
109 number_policies_selected=1,
110 **kwargs):
111 """Abstract Initialization for meta trainers.
112
113 Args:
114 game: A pyspiel game object.
115 oracle: An oracle object, from an implementation of the AbstractOracle
116 class.
117 initial_policies: A list of initial policies, to set up a default for
118 training. Resorts to tabular policies if not set.
119 meta_strategy_method: String, or callable taking a MetaTrainer object and
120 returning a list of meta strategies (One list entry per player).
121 String value can be:
122 - "uniform": Uniform distribution on policies.
123 - "nash": Taking nash distribution. Only works for 2 player, 0-sum
124 games.
125 - "prd": Projected Replicator Dynamics, as described in Lanctot et
126 Al.
127 training_strategy_selector: A callable or a string. If a callable, takes
128 as arguments: - An instance of `PSROSolver`, - a
129 `number_policies_selected` integer. and returning a list of
130 `num_players` lists of selected policies to train from.
131 When a string, supported values are:
132 - "top_k_probabilites": selects the first
133 'number_policies_selected' policies with highest selection
134 probabilities.
135 - "probabilistic": randomly selects 'number_policies_selected'
136 with probabilities determined by the meta strategies.
137 - "exhaustive": selects every policy of every player.
138 - "rectified": only selects strategies that have nonzero chance of
139 being selected.
140 - "uniform": randomly selects 'number_policies_selected' policies
141 with uniform probabilities.
142 symmetric_game: Whether to consider the current game as symmetric (True)
143 game or not (False).
144 number_policies_selected: Maximum number of new policies to train for each
145 player at each PSRO iteration.
146 **kwargs: kwargs for meta strategy computation and training strategy
147 selection
148 """
149 self._iterations = 0
150 self._game = game
151 self._oracle = oracle
152 self._num_players = self._game.num_players()
153
154 self.symmetric_game = symmetric_game
155 self._game_num_players = self._num_players
156 self._num_players = 1 if symmetric_game else self._num_players
157
158 self._number_policies_selected = number_policies_selected
159
160 meta_strategy_method = _process_string_or_callable(
161 meta_strategy_method, meta_strategies.META_STRATEGY_METHODS)
162 print("Using {} as strategy method.".format(meta_strategy_method))
163
164 self._training_strategy_selector = _process_string_or_callable(
165 training_strategy_selector,
166 strategy_selectors.TRAINING_STRATEGY_SELECTORS)
167 print("Using {} as training strategy selector.".format(
168 self._training_strategy_selector))
169
170 self._meta_strategy_method = meta_strategy_method
171 self._kwargs = kwargs
172
173 self._initialize_policy(initial_policies)
174 self._initialize_game_state()
175 self.update_meta_strategies()

185 def iteration(self, seed=None):
186 """Main trainer loop.
187
188 Args:
189 seed: Seed for random BR noise generation.
190 """
191 self._iterations += 1
192 self.update_agents() # Generate new, Best Response agents via oracle.
193 self.update_empirical_gamestate(seed=seed) # Update gamestate matrix.
194 self.update_meta_strategies() # Compute meta strategy (e.g. Nash)

208 def sample_episodes(self, policies, num_episodes):
209 """Samples episodes and averages their returns.
210
211 Args:
212 policies: A list of policies representing the policies executed by each
213 player.
214 num_episodes: Number of episodes to execute to estimate average return of
215 policies.
216
217 Returns:
218 Average episode return over num episodes.
219 """
220 totals = np.zeros(self._num_players)
221 for _ in range(num_episodes):
222 totals += sample_episode(self._game.new_initial_state(),
223 policies).reshape(-1)
224 return totals / num_episodes

226 def get_meta_strategies(self):
227 """Returns the Nash Equilibrium distribution on meta game matrix."""
228 meta_strategy_probabilities = self._meta_strategy_probabilities
229 if self.symmetric_game:
230 meta_strategy_probabilities = (self._game_num_players *
231 meta_strategy_probabilities)
232 return [np.copy(a) for a in meta_strategy_probabilities]
233
234 def get_meta_game(self):
235 """Returns the meta game matrix."""
236 meta_games = self._meta_games
237 return [np.copy(a) for a in meta_games]

239 def get_policies(self):
240 """Returns the players' policies."""
241 policies = self._policies
242 if self.symmetric_game:
243 # Notice that the following line returns N references to the same policy
244 # This might not be correct for certain applications.
245 # E.g., a DQN BR oracle with player_id information
246 policies = self._game_num_players * policies
247 return policies
