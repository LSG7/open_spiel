94  def serialize_deserialize(self, game, state, check_pyspiel_serialization,
95  check_pickle_serialization):
96  # OpenSpiel native serialization
97  if check_pyspiel_serialization:
98  ser_str = pyspiel.serialize_game_and_state(game, state)
99  new_game, new_state = pyspiel.deserialize_game_and_state(ser_str)
100  self.assertEqual(str(game), str(new_game))
101  self.assertEqual(str(state), str(new_state))
102  if check_pickle_serialization:
103  # Pickle serialization + deserialization (of the state).
104  pickled_state = pickle.dumps(state)
105  unpickled_state = pickle.loads(pickled_state)
106  self.assertEqual(str(state), str(unpickled_state))

108  def sim_game(
109  self,
110  game,
111  check_pyspiel_serialization=True,
112  check_pickle_serialization=True,
113  make_distribution_fn=(
114  lambda states: ([1 / len(states)] * len(states) if states else []))
115  ):
116  min_utility = game.min_utility()
117  max_utility = game.max_utility()
118  self.assertLess(min_utility, max_utility)
119 
120  if check_pickle_serialization:
121  # Pickle serialization + deserialization (of the game).
122  pickled_game = pickle.dumps(game)
123  unpickled_game = pickle.loads(pickled_game)
124  self.assertEqual(str(game), str(unpickled_game))
125 
126  # Pickle serialization + deserialization (of the game type).
127  pickled_game_type = pickle.dumps(game.get_type())
128  unpickled_game_type = pickle.loads(pickled_game_type)
129  self.assertEqual(game.get_type(), unpickled_game_type)
130 
131  # Get a new state
132  for state in game.new_initial_states():
133  total_actions = 0
134 
135  next_serialize_check = 1
136 
137  while not state.is_terminal() and total_actions <= MAX_ACTIONS_PER_GAME:
138  total_actions += 1
139 
140  # Serialize/Deserialize is costly. Only do it every power of 2 actions.
141  if total_actions >= next_serialize_check:
142  self.serialize_deserializeserialize_deserialize(game, state, check_pyspiel_serialization,
143  check_pickle_serialization)
144  next_serialize_check *= 2
145 
146  # The state can be four different types: chance node,
147  # mean-field-game node, simultaneous node, or decision node
148  if state.is_chance_node():
149  # Chance node: sample an outcome
150  outcomes = state.chance_outcomes()
151  self.assertNotEmpty(outcomes)
152  action_list, prob_list = zip(*outcomes)
153  action = np.random.choice(action_list, p=prob_list)
154  state.apply_action(action)
155  elif state.is_simultaneous_node():
156  # Simultaneous node: sample actions for all players
157  chosen_actions = []
158  for pid in range(game.num_players()):
159  legal_actions = state.legal_actions(pid)
160  action = 0 if not legal_actions else np.random.choice(legal_actions)
161  chosen_actions.append(action)
162  # Apply the joint action and test cloning states.
163  self.apply_action_test_cloneapply_action_test_clone(state, chosen_actions)
164  elif state.is_mean_field_node():
165  self.assertEqual(game.get_type().dynamics,
166  pyspiel.GameType.Dynamics.MEAN_FIELD)
167  state.update_distribution(
168  make_distribution_fn(state.distribution_support()))
169  else:
170  self.assertTrue(state.is_player_node())
171  # Decision node: sample action for the single current player
172  action = np.random.choice(state.legal_actions(state.current_player()))
173  # Apply action and test state cloning.
174  self.apply_action_test_cloneapply_action_test_clone(state, action)
175 
176  # Max sure at least one action was made.
177  self.assertGreater(total_actions, 0,
178  "No actions taken in sim of " + str(game))
179 
180  # Either the game is now done, or the maximum actions has been taken.
181  if state.is_terminal():
182  # Check there are no legal actions.
183  self.assertEmpty(state.legal_actions())
184  for player in range(game.num_players()):
185  self.assertEmpty(state.legal_actions(player))
186  # Print utilities for each player.
187  utilities = state.returns()
188  # Check that player returns are correct
189  for player in range(game.num_players()):
190  self.assertEqual(state.player_return(player), utilities[player])
191  # Check that each one is in range
192  for utility in utilities:
193  self.assertGreaterEqual(utility, game.min_utility())
194  self.assertLessEqual(utility, game.max_utility())
195  print("Sim of game {} terminated with {} total actions. Utilities: {}"
196  .format(game, total_actions, utilities))
197  else:
198  print("Sim of game {} terminated after maximum number of actions {}"
199  .format(game, MAX_ACTIONS_PER_GAME))
