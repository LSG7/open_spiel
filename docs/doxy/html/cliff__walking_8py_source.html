17import numpy as np

19from open_spiel.python import rl_environment

21# Actions
22RIGHT, UP, LEFT, DOWN = range(4)
23

25class Environment(object):
26 r"""A cliff walking reinforcement learning environment.
27
28 This is a deterministic environment that can be used to test RL algorithms.
29 Note there are *no illegal moves* in this environment--if the agent is on the
30 edge of the cliff and takes an action which would yield an invalid position,
31 the action is ignored (as if there were walls surrounding the cliff).
32
33 Cliff example for height=3 and width=5:
34
35 | | | | | |
36 | | | | | |
37 | S | x | x | x | G |
38
39 where `S` is always the starting position, `G` is always the goal and `x`
40 represents the zone of high negative reward to be avoided. For this instance,
41 the optimum policy is depicted as follows:
42
43 | | | | | |
44 |-->|-->|-->|-->|\|/|
45 |/|\| x | x | x | G |
46
47 yielding a reward of -6 (minus 1 per time step).
48
49 See pages 132 of Rich Sutton's book for details:
50 http://www.incompleteideas.net/book/bookdraft2018mar21.pdf
51 """
52
53 def __init__(self, height=4, width=8, discount=1.0, max_t=100):
54 if height < 2 or width < 3:
55 raise ValueError("height must be >= 2 and width >= 3.")
56 self._height = height
57 self._width = width
58 self._legal_actions = [RIGHT, UP, LEFT, DOWN]
59 self._should_reset = True
60 self._max_t = max_t
61
62 # Discount returned at non-initial steps.
63 self._discounts = [discount] * self.num_players
64
65 def reset(self):
66 """Resets the environment."""
67 self._should_reset = False
68 self._time_counter = 0
69 self._state = np.array([self._height - 1, 0])
70
71 observations = {
72 "info_state": [self._state.copy()],
73 "legal_actions": [self._legal_actions],
74 "current_player": 0,
75 }
76
77 return rl_environment.TimeStep(
78 observations=observations,
79 rewards=None,
80 discounts=None,
81 step_type=rl_environment.StepType.FIRST)

83 def step(self, actions):
84 """Updates the environment according to `actions` and returns a `TimeStep`.
85
86 Args:
87 actions: A singleton list with an integer, or an integer, representing the
88 action the agent took.
89
90 Returns:
91 A `rl_environment.TimeStep` namedtuple containing:
92 observation: singleton list of dicts containing player observations,
93 each corresponding to `observation_spec()`.
94 reward: singleton list containing the reward at this timestep, or None
95 if step_type is `rl_environment.StepType.FIRST`.
96 discount: singleton list containing the discount in the range [0, 1], or
97 None if step_type is `rl_environment.StepType.FIRST`.
98 step_type: A `rl_environment.StepType` value.
99 """
100 if self._should_reset:
101 return self.reset()
102 self._time_counter += 1
103
104 if isinstance(actions, list):
105 action = actions[0]
106 elif isinstance(actions, int):
107 action = actions
108 else:
109 raise ValueError("Action not supported.", actions)
110
111 dx = 0
112 dy = 0
113 if action == LEFT:
114 dx -= 1
115 elif action == RIGHT:
116 dx += 1
117
118 if action == UP:
119 dy -= 1
120 elif action == DOWN:
121 dy += 1
122
123 self._state += np.array([dy, dx])
124 self._state = self._state.clip(0, [self._height - 1, self._width - 1])
125
126 done = self._is_pit(self._state) or self._is_goal(self._state)
127 done = done or self._time_counter >= self._max_t
128 # Return observation
129 step_type = (
130 rl_environment.StepType.LAST if done else rl_environment.StepType.MID)
131 self._should_reset = step_type == rl_environment.StepType.LAST
132
133 observations = {
134 "info_state": [self._state.copy()],
135 "legal_actions": [self._legal_actions],
136 "current_player": 0,
137 }
138
139 return rl_environment.TimeStep(
140 observations=observations,
141 rewards=[self._get_reward(self._state)],
142 discounts=self._discounts,
143 step_type=step_type)
144
145 def _is_goal(self, pos):
146 """Check if position is bottom right corner of grid."""
147 return pos[0] == self._height - 1 and pos[1] == self._width - 1
148
149 def _is_pit(self, pos):
150 """Check if position is in bottom row between start and goal."""
151 return (pos[1] > 0 and pos[1] < self._width - 1 and
152 pos[0] == self._height - 1)
153
154 def _get_reward(self, pos):
155 if self._is_pit(pos):
156 return -100.0
157 else:
158 return -1.0
159
160 def observation_spec(self):
161 """Defines the observation provided by the environment.
162
163 Each dict member will contain its expected structure and shape.
164
165 Returns:
166 A specification dict describing the observation fields and shapes.
167 """
168 return dict(
169 info_state=tuple([2]),
170 legal_actions=(len(self._legal_actions),),
171 current_player=(),
172 )

174 def action_spec(self):
175 """Defines action specifications.
176
177 Specifications include action boundaries and their data type.
178
179 Returns:
180 A specification dict containing action properties.

182 return dict(
183 num_actions=len(self._legal_actions),
184 min=min(self._legal_actions),
