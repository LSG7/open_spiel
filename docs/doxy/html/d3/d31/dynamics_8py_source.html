20def replicator(state, fitness):
21 """Continuous-time replicator dynamics.
22
23 This is the standard form of the continuous-time replicator dynamics also
24 known as selection dynamics.
25
26 For more details, see equation (5) page 9 in
27 https://jair.org/index.php/jair/article/view/10952
28
29 Args:
30 state: Probability distribution as an `np.array(shape=num_strategies)`.
31 fitness: Fitness vector as an `np.array(shape=num_strategies)`.
32
33 Returns:
34 Time derivative of the population state.
35 """
36 avg_fitness = state.dot(fitness)
37 return state * (fitness - avg_fitness)

40def boltzmannq(state, fitness, temperature=1.):
41 """Selection-mutation dynamics modeling Q-learning with Boltzmann exploration.
42
43 For more details, see equation (10) page 15 in
44 https://jair.org/index.php/jair/article/view/10952
45
46 Args:
47 state: Probability distribution as an `np.array(shape=num_strategies)`.
48 fitness: Fitness vector as an `np.array(shape=num_strategies)`.
49 temperature: A scalar parameter determining the rate of exploration.
50
51 Returns:
52 Time derivative of the population state.
53 """
54 exploitation = (1. / temperature) * replicator(state, fitness)
55 exploration = (np.log(state) - state.dot(np.log(state).transpose()))
56 return exploitation - state * exploration

59def qpg(state, fitness):
60 """Q-based policy gradient dynamics (QPG).
61
62 For more details, see equation (12) on page 18 in
63 https://arxiv.org/pdf/1810.09026.pdf
64
65 Args:
66 state: Probability distribution as an `np.array(shape=num_strategies)`.
67 fitness: Fitness vector as an `np.array(shape=num_strategies)`.
68
69 Returns:
70 Time derivative of the population state.
71 """
72 regret = fitness - state.dot(fitness)
73 return state * (state * regret - np.sum(state**2 * regret))

76class SinglePopulationDynamics(object):
77 """Continuous-time single population dynamics.
78
79 Attributes:
80 payoff_matrix: The payoff matrix as an `numpy.ndarray` of shape `[2, k_1,
81 k_2]`, where `k_1` is the number of strategies of the first player and
82 `k_2` for the second player. The game is assumed to be symmetric.
83 dynamics: A callback function that returns the time-derivative of the
84 population state.
85 """
86
87 def __init__(self, payoff_matrix, dynamics):
88 """Initializes the single-population dynamics."""
89 assert payoff_matrix.ndim == 3
90 assert payoff_matrix.shape[0] == 2
91 assert np.allclose(payoff_matrix[0], payoff_matrix[1].T)
92 self.payoff_matrix = payoff_matrix[0]
93 self.dynamics = dynamics
94
95 def __call__(self, state=None, time=None):
96 """Time derivative of the population state.
97
98 Args:
99 state: Probability distribution as list or
100 `numpy.ndarray(shape=num_strategies)`.
101 time: Time is ignored (time-invariant dynamics). Including the argument in
102 the function signature supports numerical integration via e.g.
103 `scipy.integrate.odeint` which requires that the callback function has
104 at least two arguments (state and time).
105
106 Returns:
107 Time derivative of the population state as
108 `numpy.ndarray(shape=num_strategies)`.
109 """
110 state = np.array(state)
111 assert state.ndim == 1
112 assert state.shape[0] == self.payoff_matrix.shape[0]
113 # (Ax')' = xA'
114 fitness = np.matmul(state, self.payoff_matrix.T)
115 return self.dynamics(state, fitness)

118class MultiPopulationDynamics(object):
119 """Continuous-time multi-population dynamics.
120
121 Attributes:
122 payoff_tensor: The payoff tensor as an numpy.ndarray of size `[n, k0, k1,
123 k2, ...]`, where n is the number of players and `k0` is the number of
124 strategies of the first player, `k1` of the second player and so forth.
125 dynamics: List of callback functions for the time-derivative of the
126 population states, where `dynamics[i]` computes the time-derivative of the
127 i-th player's population state. If at construction, only a single callback
128 function is provided, the same function is used for all populations.
129 """
130
131 def __init__(self, payoff_tensor, dynamics):
132 """Initializes the multi-population dynamics."""
133 if isinstance(dynamics, list) or isinstance(dynamics, tuple):
134 assert payoff_tensor.shape[0] == len(dynamics)
135 else:
136 dynamics = [dynamics] * payoff_tensor.shape[0]
137 self.payoff_tensor = payoff_tensor
138 self.dynamics = dynamics
139
140 def __call__(self, state, time=None):
141 """Time derivative of the population states.
142
143 Args:
144 state: Combined population state for all populations as a list or flat
145 `numpy.ndarray` (ndim=1). Probability distributions are concatenated in
146 order of the players.
147 time: Time is ignored (time-invariant dynamics). Including the argument in
148 the function signature supports numerical integration via e.g.
149 `scipy.integrate.odeint` which requires that the callback function has
150 at least two arguments (state and time).
151
152 Returns:
153 Time derivative of the combined population state as `numpy.ndarray`.
154 """
155 state = np.array(state)
156 n = self.payoff_tensor.shape[0] # number of players
157 ks = self.payoff_tensor.shape[1:] # number of strategies for each player
158 assert state.shape[0] == sum(ks)
159
160 states = np.split(state, np.cumsum(ks)[:-1])
161 dstates = [None] * n
162 for i in range(n):
163 # move i-th population to front
164 fitness = np.moveaxis(self.payoff_tensor[i], i, 0)
165 # marginalize out all other populations
166 for i_ in set(range(n)) - {i}:
167 fitness = np.tensordot(states[i_], fitness, axes=[0, 1])
168 dstates[i] = self.dynamics[i](states[i], fitness)
169
170 return np.concatenate(dstates)

173def time_average(traj):
174 """Time-averaged population state trajectory.
175
176 Args:
177 traj: Trajectory as `numpy.ndarray`. Time is along the first dimension,
178 types/strategies along the second.
179
180 Returns:
181 Time-averaged trajectory.
182 """
183 n = traj.shape[0]
184 sum_traj = np.cumsum(traj, axis=0)
185 norm = 1. / np.arange(1, n + 1)
186 return sum_traj * norm[:, np.newaxis]
