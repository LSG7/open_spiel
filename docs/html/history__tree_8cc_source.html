28 std::unique_ptr<HistoryNode> RecursivelyBuildGameTree(
29  std::unique_ptr<State> state, Player player_id,
30  absl::flat_hash_map<std::string, HistoryNode*>* state_to_node) {
31  std::unique_ptr<HistoryNode> node(
32  new HistoryNode(player_id, std::move(state)));
33  if (state_to_node == nullptr) SpielFatalError("state_to_node is null.");
34  (*state_to_node)[node->GetHistory()] = node.get();
35  State* state_ptr = node->GetState();
36  switch (node->GetType()) {
37  case StateType::kMeanField: {
38  SpielFatalError("kMeanField not supported.");
39  }
40  case StateType::kChance: {
41  double probability_sum = 0;
42  for (const auto& [outcome, prob] : state_ptr->ChanceOutcomes()) {
43  std::unique_ptr<State> child = state_ptr->Child(outcome);
44  if (child == nullptr) {
45  SpielFatalError("Can't add child; child is null.");
46  }
47  probability_sum += prob;
48  std::unique_ptr<HistoryNode> child_node = RecursivelyBuildGameTree(
49  std::move(child), player_id, state_to_node);
50  node->AddChild(outcome, {prob, std::move(child_node)});
51  }
52  SPIEL_CHECK_FLOAT_EQ(probability_sum, 1.0);
53  break;
54  }
55  case StateType::kDecision: {
56  for (const auto& legal_action : state_ptr->LegalActions()) {
57  std::unique_ptr<State> child = state_ptr->Child(legal_action);
58 
59  // Note: The probabilities here are meaningless if state.CurrentPlayer()
60  // != player_id, as we'll be getting the probabilities from the policy
61  // during the call to Value. For state.CurrentPlayer() == player_id,
62  // the probabilities are equal to 1. for every action as these are
63  // *counter-factual* probabilities, which ignore the probability of
64  // the player that we are playing as.
65  node->AddChild(legal_action,
66  {1., RecursivelyBuildGameTree(
67  std::move(child), player_id, state_to_node)});
68  }
69  break;
70  }
71  case StateType::kTerminal: {
72  // As we assign terminal utilities to node.value in the constructor of
73  // HistoryNode, we don't have anything to do here.
74  break;
75  }
76  }
77  return node;
78 }

82 HistoryNode::HistoryNode(Player player_id, std::unique_ptr<State> game_state)
83  : state_(std::move(game_state)),
84  history_(state_->HistoryString()),
85  type_(state_->GetType()) {
86  // Unless it's the opposing player's turn, we always view the game from the
87  // view of player player_id.
88  if (type_ == StateType::kDecision && state_->CurrentPlayer() != player_id) {
89  infostate_ = state_->InformationStateString();
90  } else if (type_ == StateType::kChance) {
91  infostate_ = kChanceNodeInfostateString;
92  } else if (type_ == StateType::kTerminal) {
93  infostate_ = kTerminalNodeInfostateString;
94  } else {
95  infostate_ = state_->InformationStateString(player_id);
96  }
97  // Compute & store the legal actions so we can check that all actions we're
98  // adding are legal.
99  for (Action action : state_->LegalActions()) legal_actions_.insert(action);
100  if (type_ == StateType::kTerminal) value_ = state_->PlayerReturn(player_id);
101 }

103 void HistoryNode::AddChild(
104  Action outcome, std::pair<double, std::unique_ptr<HistoryNode>> child) {
105  if (!legal_actions_.count(outcome)) SpielFatalError("Child is not legal.");
106  if (child.second == nullptr) {
107  SpielFatalError("Error inserting child; child is null.");
108  }
109  SPIEL_CHECK_PROB_TOLERANCE(child.first, ProbabilityDefaultTolerance());
110  child_info_[outcome] = std::move(child);
111  if (child_info_.size() > legal_actions_.size()) {
112  SpielFatalError("More children than legal actions.");
113  }
114 }

116 std::pair<double, HistoryNode*> HistoryNode::GetChild(Action outcome) {
117  auto it = child_info_.find(outcome);
118  if (it == child_info_.end()) {
119  SpielFatalError("Error getting child; action not found.");
120  }
121  // it->second.first is the probability associated with outcome, so as it is a
122  // probability, it must be in [0, 1].
123  SPIEL_CHECK_PROB_TOLERANCE(it->second.first, ProbabilityDefaultTolerance());
124  std::pair<double, HistoryNode*> child =
125  std::make_pair(it->second.first, it->second.second.get());
126  if (child.second == nullptr) {
127  SpielFatalError("Error getting child; child is null.");
128  }
129  return child;
130 }

160 ActionsAndProbs GetSuccessorsWithProbs(const State& state,
161  Player best_responder,
162  const Policy* policy) {
163  if (state.CurrentPlayer() == best_responder) {
164  ActionsAndProbs state_policy;
165  for (const auto& legal_action : state.LegalActions()) {
166  // Counterfactual reach probabilities exclude the player's
167  // actions, hence return probability 1.0 for every action.
168  state_policy.push_back({legal_action, 1.});
169  }
170  return state_policy;
171  } else if (state.IsChanceNode()) {
172  return state.ChanceOutcomes();
173  } else {
174  // Finally, we look at the policy we are finding a best response to, and
175  // get our probabilities from there.
176  auto state_policy = policy->GetStatePolicy(state);
177  if (state_policy.empty()) {
178  SpielFatalError(state.InformationStateString() + " not found in policy.");
179  }
180  return state_policy;
181  }
182 }

184 // TODO(author1): If this is a bottleneck, it should be possible
185 // to pass the probabilities-so-far into the call, and get everything right
186 // the first time, without recursion. The recursion is simpler, however.
187 std::vector<std::pair<std::unique_ptr<State>, double>> DecisionNodes(
188  const State& parent_state, Player best_responder, const Policy* policy) {
189  // If the state is terminal, then there are no more decisions to be made,
190  // so we're done.
191  if (parent_state.IsTerminal()) return {};
192 
193  std::vector<std::pair<std::unique_ptr<State>, double>> states_and_probs;
194  // We only consider states where the best_responder is making a decision.
195  if (parent_state.CurrentPlayer() == best_responder) {
196  states_and_probs.push_back({parent_state.Clone(), 1.});
197  }
198  ActionsAndProbs actions_and_probs =
199  GetSuccessorsWithProbs(parent_state, best_responder, policy);
200  for (open_spiel::Action action : parent_state.LegalActions()) {
201  std::unique_ptr<State> child = parent_state.Child(action);
202 
203  // We recurse here to get the correct probabilities for all children.
204  // This could probably be done in a cleaner, more performant way, but as
205  // this is only done once, at the start of the exploitability calculation,
206  // this is fine for now.
207  std::vector<std::pair<std::unique_ptr<State>, double>> children =
208  DecisionNodes(*child, best_responder, policy);
209  const double policy_prob = GetProb(actions_and_probs, action);
210  SPIEL_CHECK_PROB_TOLERANCE(policy_prob, ProbabilityDefaultTolerance());
211  for (auto& [state, prob] : children) {
212  states_and_probs.push_back(
213  {std::move(state),
214  // We weight the child probabilities by the probability of taking
215  // the action that would lead to them.
216  policy_prob * prob});
217  }
218  }
219  return states_and_probs;
220 }

222 absl::flat_hash_map<std::string, std::vector<std::pair<HistoryNode*, double>>>
223 GetAllInfoSets(std::unique_ptr<State> state, Player best_responder,
224  const Policy* policy, HistoryTree* tree) {
225  absl::flat_hash_map<std::string, std::vector<std::pair<HistoryNode*, double>>>
226  infosets;
227  // We only need decision nodes, as there's no decision to be made at chance
228  // nodes (we randomly sample from the different outcomes there).
229  std::vector<std::pair<std::unique_ptr<State>, double>> states_and_probs =
230  DecisionNodes(*state, best_responder, policy);
231  infosets.reserve(states_and_probs.size());
232  for (const auto& [state, prob] : states_and_probs) {
233  // We look at each decision from the perspective of the best_responder.
234  std::string infostate = state->InformationStateString(best_responder);
235  infosets[infostate].push_back({tree->GetByHistory(*state), prob});
236  }
237  return infosets;
238 }
