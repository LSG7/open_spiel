41Instance GenerateInstance(std::mt19937* rng) {
42 Instance instance;
43 bool valid = false;
44 while (!valid) {
45 valid = true;
46 for (int i = 0; i < kNumItemTypes; ++i) {
47 instance.pool[i] = absl::Uniform<int>(*rng, 1, kPoolMaxNumItems + 1);
48 }
49 int num_pool_items =
50 std::accumulate(instance.pool.begin(), instance.pool.end(), 0);
51 if (!(num_pool_items >= kPoolMinNumItems &&
52 num_pool_items <= kPoolMaxNumItems)) {
53 valid = false;
54 continue;
55 }
56
57 // total value to each user is 10
58 // every item has nonzero value to at least one player
59 // some items have nonzero value to both players
60 bool exists_valuable_to_both = false;
61 std::array<int, 2> total_values = {0, 0};
62 for (int i = 0; i < kNumItemTypes && valid; ++i) {
63 for (Player p : {0, 1}) {
64 instance.values[p][i] =
65 absl::Uniform<int>(*rng, 0, kTotalValueAllItems + 1);
66 }
67
68 if (instance.values[0][i] == 0 && instance.values[1][i] == 0) {
69 valid = false;
70 break;
71 } else if (instance.values[0][i] > 0 && instance.values[1][i] > 0) {
72 exists_valuable_to_both = true;
73 }
74
75 for (Player p : {0, 1}) {
76 total_values[p] += instance.values[p][i] * instance.pool[i];
77 if (total_values[p] > kTotalValueAllItems) {
78 valid = false;
79 break;
80 }
81 }
82 }
83
84 if (!valid) {
85 continue;
86 }
87
88 if (!(total_values[0] == kTotalValueAllItems &&
89 total_values[1] == kTotalValueAllItems && exists_valuable_to_both)) {
90 valid = false;
91 }
92 }
93
94 return instance;
95}

97void GenerateInstances(int num) {
98 std::string filename = absl::GetFlag(FLAGS_filename);
99 int seed = absl::GetFlag(FLAGS_seed);
100 std::mt19937 rng(seed);
101
102 std::cout << "Opening file: " << filename << std::endl;
103 open_spiel::file::File outfile(filename, "w");
104 for (int i = 0; i < num; ++i) {
105 Instance instance = GenerateInstance(&rng);
106 std::string instance_line = instance.ToString();
107 instance_line.push_back('\n');
108 std::cout << i << std::endl << instance_line;
109 outfile.Write(instance_line);
110 }
111 std::cout << "Wrote to file: " << filename << std::endl;
112}
