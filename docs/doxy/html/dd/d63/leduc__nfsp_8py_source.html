77class NFSPPolicies(policy.Policy):
78 """Joint policy to be evaluated."""
79
80 def __init__(self, env, nfsp_policies, mode):
81 game = env.game
82 player_ids = list(range(FLAGS.num_players))
83 super(NFSPPolicies, self).__init__(game, player_ids)
84 self._policies = nfsp_policies
85 self._mode = mode
86 self._obs = {
87 "info_state": [None] * FLAGS.num_players,
88 "legal_actions": [None] * FLAGS.num_players
89 }
90
91 def action_probabilities(self, state, player_id=None):
92 cur_player = state.current_player()
93 legal_actions = state.legal_actions(cur_player)
94
95 self._obs["current_player"] = cur_player
96 self._obs["info_state"][cur_player] = (
97 state.information_state_tensor(cur_player))
98 self._obs["legal_actions"][cur_player] = legal_actions
99
100 info_state = rl_environment.TimeStep(
101 observations=self._obs, rewards=None, discounts=None, step_type=None)
102
103 with self._policies[cur_player].temp_mode_as(self._mode):
104 p = self._policies[cur_player].step(info_state, is_evaluation=True).probs
105 prob_dict = {action: p[action] for action in legal_actions}
106 return prob_dict

109def main(unused_argv):
110 logging.info("Loading %s", FLAGS.game_name)
111 game = FLAGS.game_name
112 num_players = FLAGS.num_players
113
114 env_configs = {"players": num_players}
115 env = rl_environment.Environment(game, **env_configs)
116 info_state_size = env.observation_spec()["info_state"][0]
117 num_actions = env.action_spec()["num_actions"]
118
119 hidden_layers_sizes = [int(l) for l in FLAGS.hidden_layers_sizes]
120 kwargs = {
121 "replay_buffer_capacity": FLAGS.replay_buffer_capacity,
122 "reservoir_buffer_capacity": FLAGS.reservoir_buffer_capacity,
123 "min_buffer_size_to_learn": FLAGS.min_buffer_size_to_learn,
124 "anticipatory_param": FLAGS.anticipatory_param,
125 "batch_size": FLAGS.batch_size,
126 "learn_every": FLAGS.learn_every,
127 "rl_learning_rate": FLAGS.rl_learning_rate,
128 "sl_learning_rate": FLAGS.sl_learning_rate,
129 "optimizer_str": FLAGS.optimizer_str,
130 "loss_str": FLAGS.loss_str,
131 "update_target_network_every": FLAGS.update_target_network_every,
132 "discount_factor": FLAGS.discount_factor,
133 "epsilon_decay_duration": FLAGS.epsilon_decay_duration,
134 "epsilon_start": FLAGS.epsilon_start,
135 "epsilon_end": FLAGS.epsilon_end,
136 }
137
138 with tf.Session() as sess:
139 # pylint: disable=g-complex-comprehension
140 agents = [
141 nfsp.NFSP(sess, idx, info_state_size, num_actions, hidden_layers_sizes,
142 **kwargs) for idx in range(num_players)
143 ]
144 joint_avg_policy = NFSPPolicies(env, agents, nfsp.MODE.average_policy)
145
146 sess.run(tf.global_variables_initializer())
147
148 if FLAGS.use_checkpoints:
149 for agent in agents:
150 if agent.has_checkpoint(FLAGS.checkpoint_dir):
151 agent.restore(FLAGS.checkpoint_dir)
152
153 for ep in range(FLAGS.num_train_episodes):
154 if (ep + 1) % FLAGS.eval_every == 0:
155 losses = [agent.loss for agent in agents]
156 logging.info("Losses: %s", losses)
157 if FLAGS.evaluation_metric == "exploitability":
158 # Avg exploitability is implemented only for 2 players constant-sum
159 # games, use nash_conv otherwise.
160 expl = exploitability.exploitability(env.game, joint_avg_policy)
161 logging.info("[%s] Exploitability AVG %s", ep + 1, expl)
162 elif FLAGS.evaluation_metric == "nash_conv":
163 nash_conv = exploitability.nash_conv(env.game, joint_avg_policy)
164 logging.info("[%s] NashConv %s", ep + 1, nash_conv)
165 else:
166 raise ValueError(" ".join(("Invalid evaluation metric, choose from",
167 "'exploitability', 'nash_conv'.")))
168 if FLAGS.use_checkpoints:
169 for agent in agents:
170 agent.save(FLAGS.checkpoint_dir)
171 logging.info("_____________________________________________")
172
173 time_step = env.reset()
174 while not time_step.last():
175 player_id = time_step.observations["current_player"]
176 agent_output = agents[player_id].step(time_step)
177 action_list = [agent_output.action]
178 time_step = env.step(action_list)
179
180 # Episode is over, step all agents with final info state.
181 for agent in agents:
182 agent.step(time_step)
183
184
185if __name__ == "__main__":
186 app.run(main)
