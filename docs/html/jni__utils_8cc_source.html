31  std::cout << "intializing JVM" << std::endl;
32 #ifdef JNI_VERSION_1_2
33  JavaVMInitArgs vm_args;
34  JavaVMOption options[1];
35  std::string java_classpath = "-Djava.class.path=" + jar_location;
36  char *c_classpath = strdup(java_classpath.c_str());
37  options[0].optionString = c_classpath;
38  vm_args.version = 0x00010002;
39  vm_args.options = options;
40  vm_args.nOptions = 1;
41  vm_args.ignoreUnrecognized = JNI_TRUE;
42  /* Create the Java VM */
43  res = JNI_CreateJavaVM(&jvm, (void **)&env, &vm_args);
44  free(c_classpath);
45 #else

48  char* c_classpath = strdup(java_classpath.c_str());
49  vm_args.version = 0x00010001;

54  res = JNI_CreateJavaVM(&jvm, &env, &vm_args);
55  free(c_classpath);
56 #endif /* JNI_VERSION_1_2 */
