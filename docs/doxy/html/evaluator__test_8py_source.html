34 def test_evaluator_caching(self):
35 game = pyspiel.load_game("tic_tac_toe")
36 model = build_model(game)
37 evaluator = evaluator_lib.AlphaZeroEvaluator(game, model)
38
39 state = game.new_initial_state()
40 obs = state.observation_tensor()
41 act_mask = state.legal_actions_mask()
42 action = state.legal_actions()[0]
43 policy = np.zeros(len(act_mask), dtype=float)
44 policy[action] = 1
45 train_inputs = [model_lib.TrainInput(obs, act_mask, policy, value=1)]
46
47 value = evaluator.evaluate(state)
48 self.assertEqual(value[0], -value[1])
49 value = value[0]
50
51 value2 = evaluator.evaluate(state)[0]
52 self.assertEqual(value, value2)
53
54 prior = evaluator.prior(state)
55 prior2 = evaluator.prior(state)
56 np.testing.assert_array_equal(prior, prior2)
57
58 info = evaluator.cache_info()
59 self.assertEqual(info.misses, 1)
60 self.assertEqual(info.hits, 3)
61
62 for _ in range(20):
63 model.update(train_inputs)
64
65 # Still equal due to not clearing the cache
66 value3 = evaluator.evaluate(state)[0]
67 self.assertEqual(value, value3)
68
69 info = evaluator.cache_info()
70 self.assertEqual(info.misses, 1)
71 self.assertEqual(info.hits, 4)
72
73 evaluator.clear_cache()
74
75 info = evaluator.cache_info()
76 self.assertEqual(info.misses, 0)
77 self.assertEqual(info.hits, 0)
78
79 # Now they differ from before
80 value4 = evaluator.evaluate(state)[0]
81 value5 = evaluator.evaluate(state)[0]
82 self.assertNotEqual(value, value4)
83 self.assertEqual(value4, value5)
84
85 info = evaluator.cache_info()
86 self.assertEqual(info.misses, 1)
87 self.assertEqual(info.hits, 1)
88
89 value6 = evaluator.evaluate(game.new_initial_state())[0]
90 self.assertEqual(value4, value6)
91
92 info = evaluator.cache_info()
93 self.assertEqual(info.misses, 1)
94 self.assertEqual(info.hits, 2)
