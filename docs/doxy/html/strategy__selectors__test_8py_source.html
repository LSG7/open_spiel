42def rectified_alias(solver, number_policies_to_select):
43 """Returns every strategy with nonzero selection probability.
44
45 Args:
46 solver: A GenPSROSolver instance.
47 number_policies_to_select: Number policies to select
48
49 Returns:
50 used_policies: A list, each element a list of the policies used per player.
51 """
52 del number_policies_to_select
53
54 used_policies = []
55 used_policy_indexes = []
56
57 policies = solver.get_policies()
58 num_players = len(policies)
59 meta_strategy_probabilities = solver.get_meta_strategies()
60
61 for k in range(num_players):
62 current_policies = policies[k]
63 current_probabilities = meta_strategy_probabilities[k]
64
65 current_indexes = [
66 i for i in range(len(current_policies))

69 current_policies = [
70 current_policies[i]
71 for i in current_indexes
72 ]

142class StrategySelectorsTest(absltest.TestCase):

144 def test_vital(self):
145 n_tests = 1000
146 number_strategies = 50
147 number_players = 3
148 for i in range(n_tests):
149 probabilities = np.random.uniform(size=(number_players,
150 number_strategies))
151 probabilities /= np.sum(probabilities, axis=1).reshape(-1, 1)
152 probabilities = list(probabilities)
153
154 policies = [list(range(number_strategies)) for _ in range(number_players)]
155
156 solver = FakeSolver(probabilities, policies)
157
158 # To see how rectified reacts to 0 probabilities.
159 probabilities[0][0] = 0
160 probabilities[-1][-1] = 0
161 a, b = strategy_selectors.rectified(solver, 1)
162 c, d = rectified_alias(solver, 1)
163
164 self.assertEqual(a, c, "Rectified failed.")
165 self.assertEqual(b, d, "Rectified failed.")
166
167 a, b = strategy_selectors.top_k_probabilities(solver, 3)
168 c, d = top_k_probabilities_alias(solver, 3)
169
170 self.assertEqual(a, c, "Top k failed.")
171 self.assertEqual(b, d, "Top k failed.")
172
173 n_nonzero_policies = 2
174 probabilities = [np.zeros(number_strategies) for _ in range(
175 number_players)]
176
177 for player in range(number_players):
178 for _ in range(n_nonzero_policies):
179 i = np.random.randint(0, high=number_strategies)
180 while probabilities[player][i] > 1e-12:
181 i = np.random.randint(0, high=number_strategies)
182 probabilities[player][i] = 1.0 / n_nonzero_policies
183 probabilities[player] /= np.sum(probabilities[player])
184
185 solver = FakeSolver(probabilities, policies)
186 a, b = strategy_selectors.probabilistic(solver, n_nonzero_policies)
187 c, d = probabilistic_alias(solver, n_nonzero_policies)
188
189 self.assertTrue(equal_to_transposition_lists(a, c),
190 "Probabilistic failed.")
191 self.assertTrue(equal_to_transposition_lists(b, d),
192 "Probabilistic failed.")
