61 class RestrictedNashResponseObserver : public Observer {
62  public:
63  RestrictedNashResponseObserver(IIGObservationType iig_obs_type)
64  : Observer(/*has_string=*/true, /*has_tensor=*/true),
65  iig_obs_type_(iig_obs_type) {}

67  // Writes the complete observation in tensor form.
68  // The supplied allocator is responsible for providing memory to write the
69  // observation into.
70  void WriteTensor(const State& observed_state, int player,
71  Allocator *allocator) const override {
72  auto& state = open_spiel::down_cast<const RestrictedNashResponseState &>(
73  observed_state);
74  SPIEL_CHECK_GE(player, 0);
75  SPIEL_CHECK_LT(player, state.NumPlayers());
76 
77  std::shared_ptr<const Game> original_game = state.GetOriginalGame();
78  GameParameters params;
79  std::shared_ptr<Observer> observer =
80  original_game->MakeObserver(iig_obs_type_, params);
81  // Observing player.
82  auto out = allocator->Get("initial_and_fixed", {2});
83  if (iig_obs_type_.public_info) {
84  if (state.IsRestrictedNashResponseInitialState()) {
85  out.at(0) = 1;
86  }
87  }
88  if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
89  if (state.IsPlayerFixed(player)) {
90  out.at(1) = state.IsStateFixed();
91  } else {
92  out.at(1) = 0;
93  }
94  } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
95  out.at(1) = state.IsStateFixed();
96  }
97  observer->WriteTensor(*state.GetOriginalState(), player, allocator);
98  }
99 
100  // Writes an observation in string form. It would be possible just to
101  // turn the tensor observation into a string, but we prefer something
102  // somewhat human-readable.
103 
104  std::string StringFrom(const State &observed_state,
105  int player) const override {
106  auto& state = open_spiel::down_cast<const RestrictedNashResponseState &>(
107  observed_state);
108  SPIEL_CHECK_GE(player, 0);
109  SPIEL_CHECK_LT(player, state.NumPlayers());
110  std::string result;
111 
112  std::shared_ptr<const Game> original_game = state.GetOriginalGame();
113  GameParameters params;
114  std::shared_ptr<Observer> observer =
115  original_game->MakeObserver(iig_obs_type_, params);
116  if (iig_obs_type_.public_info) {
117  if (state.IsRestrictedNashResponseInitialState()) {
118  return "Initial";
119  }
120  }
121  if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
122  if (state.IsPlayerFixed(player)) {
123  result += state.IsStateFixed() ? "[Rnr: fixed]" : "[Rnr: free]";
124  }
125  } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
126  result += state.IsStateFixed() ? "[Rnr: fixed]" : "[Rnr: free]";
127  }
128 
129  result += observer->StringFrom(*state.GetOriginalState(), player);
130  return result;
131  }
132 
133  private:
134  IIGObservationType iig_obs_type_;
135 };

179 std::vector<std::pair<Action, double>>
180 RestrictedNashResponseState::ChanceOutcomes() const {
181  if (is_initial_) {
182  return {{Action(kFixedAction), p_}, {Action(kFreeAction), 1 - p_}};
183  } else {
184  if (state_->IsChanceNode()) {
185  return state_->ChanceOutcomes();
186  } else if (use_fixed_policy_ && fixed_ &&
187  state_->CurrentPlayer() == fixed_player_) {
188  return fixed_policy_->GetStatePolicy(*state_);
189  }
190  }
191  return {};
192 }

202 std::vector<Action> RestrictedNashResponseState::LegalActions(
203  Player player) const {
204  // Initial state only has two actions to fixed or free tree
205  if (is_initial_) {
206  if (player == kChancePlayerId) {
207  return {Action(kFixedAction), Action(kFreeAction)};
208  } else {
209  return {};
210  }
211  } else {
212  if (use_fixed_policy_ && fixed_ &&
213  state_->CurrentPlayer() == fixed_player_) {
214  // In other states if we exchanged fixed player nodes for chance node we
215  // return action for chance player
216  if (player == kChancePlayerId) {
217  return state_->LegalActions(fixed_player_);
218  } else {
219  return {};
220  }
221  } else {
222  // Otherwise we just use original legal actions
223  return state_->LegalActions(player);
224  }
225  }
226 }

228 std::string RestrictedNashResponseState::ActionToString(
229  Player player, Action action_id) const {
230  if (is_initial_) {
231  SPIEL_CHECK_EQ(player, kChancePlayerId);
232  return (action_id == kFixedAction ? "Fixed" : "Free");
233  } else {
234  Player action_player = player;
235  if (action_player == kChancePlayerId && use_fixed_policy_ && fixed_ &&
236  state_->CurrentPlayer() == fixed_player_) {
237  // This is a chance node in the RNR game, but a regular player node
238  // in the underlying game, so we need to use the player's true identity
239  // at this node.
240  action_player = state_->CurrentPlayer();
241  }
242  return state_->ActionToString(action_player, action_id);
243  }
244 }
