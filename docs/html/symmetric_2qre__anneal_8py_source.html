112  def gradients(self, dist, y, anneal_steps, payoff_matrices, num_players,
113  temperature=0., proj_grad=True):
114  """Computes exploitablity gradient and aux variable gradients.
115 
116  Args:
117  dist: 1-d np.array, current estimate of nash distribution
118  y: 1-d np.array (same shape as dist), current estimate of payoff gradient
119  anneal_steps: int, elapsed num steps since last anneal
120  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
121  num_players: int, number of players, in case payoff_matrices is
122  abbreviated
123  temperature: non-negative float, default 0.
124  proj_grad: bool, if True, projects dist gradient onto simplex
125  Returns:
126  gradient of exploitability w.r.t. (dist, y, anneal_steps) as tuple
127  unregularized exploitability (stochastic estimate)
128  tsallis regularized exploitability (stochastic estimate)
129  """
130  nabla = payoff_matrices[0].dot(dist)
131  if temperature >= 1e-3:
132  br = special.softmax(y / temperature)
133  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
134  log_br_safe = np.clip(np.log(br), -1e5, 0)
135  br_policy_gradient = nabla - temperature * (log_br_safe + 1)
136  else:
137  power = np.inf
138  s = np.linalg.norm(y, ord=power)
139  br = np.zeros_like(dist)
140  maxima = (y >= s)
141  br[maxima] = 1. / maxima.sum()
142  br_mat = np.zeros((br.size, br.size))
143  br_policy_gradient = np.zeros_like(br)
144 
145  unreg_exp = np.max(y) - y.dot(dist)
146  entr_br = temperature * special.entr(br).sum()
147  entr_dist = temperature * special.entr(dist).sum()
148  reg_exp = y.dot(br - dist) + entr_br - entr_dist
149 
150  policy_gradient = np.array(nabla)
151  if temperature > 0:
152  log_dist_safe = np.clip(np.log(dist), -1e5, 0)
153  policy_gradient -= temperature * (log_dist_safe + 1)
154  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
155 
156  other_player_fx_translated = payoff_matrices[1].dot(other_player_fx)
157  grad_dist = -policy_gradient
158  grad_dist += (num_players - 1) * other_player_fx_translated
159  if proj_grad:
160  grad_dist = simplex.project_grad(grad_dist)
161  grad_y = y - nabla
162 
163  _, lr_y = self.lrslrs
164  if (reg_exp < self.exp_threshexp_thresh) and (anneal_steps >= 1 / lr_y):
165  self.temperaturetemperature = np.clip(temperature / 2., 0., np.inf)
166  if self.temperaturetemperature < 1e-3:
167  self.temperaturetemperature = 0.
168  grad_anneal_steps = -anneal_steps
169  else:
170  grad_anneal_steps = 1
171 
172  return (grad_dist, grad_y, grad_anneal_steps), unreg_exp, reg_exp

174  def cheap_gradients(self, random, dist, y, anneal_steps, payoff_matrices,
175  num_players, temperature=0., proj_grad=True):
176  """Computes exploitablity gradient and aux variable gradients with samples.
177 
178  This implementation takes payoff_matrices as input so technically uses
179  O(d^2) compute but only a single column of payoff_matrices is used to
180  perform the update so can be re-implemented in O(d) if needed.
181 
182  Args:
183  random: random number generator, np.random.RandomState(seed)
184  dist: 1-d np.array, current estimate of nash distribution
185  y: 1-d np.array (same shape as dist), current estimate of payoff gradient
186  anneal_steps: int, elapsed num steps since last anneal
187  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
188  num_players: int, number of players, in case payoff_matrices is
189  abbreviated
190  temperature: non-negative float, default 0.
191  proj_grad: bool, if True, projects dist gradient onto simplex
192  Returns:
193  gradient of exploitability w.r.t. (dist, y, anneal_steps) as tuple
194  unregularized exploitability (stochastic estimate)
195  tsallis regularized exploitability (stochastic estimate)
196  """
197  del anneal_steps
198  action_1 = random.choice(dist.size, p=dist)
199  nabla = payoff_matrices[0][:, action_1]
200  if temperature >= 1e-3:
201  br = special.softmax(y / temperature)
202  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
203  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
204  else:
205  power = np.inf
206  s = np.linalg.norm(y, ord=power)
207  br = np.zeros_like(dist)
208  maxima = (y == s)
209  br[maxima] = 1. / maxima.sum()
210  br_mat = np.zeros((br.size, br.size))
211  br_policy_gradient = np.zeros_like(br)
212 
213  unreg_exp = np.max(y) - y.dot(dist)
214  entr_br = temperature * special.entr(br).sum()
215  entr_dist = temperature * special.entr(dist).sum()
216  reg_exp = y.dot(br - dist) + entr_br - entr_dist
217 
218  policy_gradient = nabla - temperature * (np.log(dist) + 1)
219  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
220 
221  action_u = random.choice(dist.size) # uniform, ~importance sampling
222  other_player_fx = dist.size * other_player_fx[action_u]
223  other_player_fx_translat = payoff_matrices[1, :, action_u] * other_player_fx
224  grad_dist = -policy_gradient + (num_players - 1) * other_player_fx_translat
225  if proj_grad:
226  grad_dist = simplex.project_grad(grad_dist)
227  grad_y = y - nabla
228 
229  return (grad_dist, grad_y, None), unreg_exp, reg_exp

241  random: random number generator, np.random.RandomState(seed)

244  anneal_steps: int, elapsed num steps since last anneal

251  version: int, default 0, two options for variance reduction

253  gradient of exploitability w.r.t. (dist, y, anneal_steps) as tuple

256  """
257  del anneal_steps
258  if pm_vr is None:
259  raise ValueError("pm_vr must be np.array of shape (num_strats,) * 2")
260  if (not isinstance(version, int)) or (version < 0) or (version > 1):
261  raise ValueError("version must be non-negative int < 2")
262 
263  action_1 = random.choice(dist.size, p=dist)
264  nabla = payoff_matrices[0][:, action_1]
265  if temperature >= 1e-3:
266  br = special.softmax(y / temperature)
267  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
268  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
269  else:
270  power = np.inf
271  s = np.linalg.norm(y, ord=power)
272  br = np.zeros_like(dist)
273  maxima = (y == s)
274  br[maxima] = 1. / maxima.sum()
275  br_mat = np.zeros((br.size, br.size))
276  br_policy_gradient = np.zeros_like(br)
277 
278  unreg_exp = np.max(y) - y.dot(dist)
279  entr_br = temperature * special.entr(br).sum()
280  entr_dist = temperature * special.entr(dist).sum()
281  reg_exp = y.dot(br - dist) + entr_br - entr_dist
282 
283  policy_gradient = nabla - temperature * (np.log(dist) + 1)
284  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
285 
286  if version == 0:
287  other_player_fx_translated = pm_vr.dot(other_player_fx)
288  action_u = random.choice(dist.size) # uniform, ~importance sampling
289  other_player_fx = other_player_fx[action_u]

298  r = dist.size * pm_vr[:, action_u]
299  other_player_fx_translated += pm_mod * other_player_fx - r
300 
301  grad_dist = -policy_gradient
302  grad_dist += (num_players - 1) * other_player_fx_translated
303  if proj_grad:

305  grad_y = y - nabla
306 
307  if version == 0:

309  elif version == 1:

314  def euc_descent_step(self, params, grads, t):
315  """Projected gradient descent on exploitability using Euclidean projection.
316 
317  Args:
318  params: tuple of variables to be updated (dist, y, anneal_steps)
319  grads: tuple of variable gradients (grad_dist, grad_y, grad_anneal_steps)
320  t: int, solver iteration
321  Returns:
322  new_params: tuple of update params (new_dist, new_y, new_anneal_steps)
323  """
324  lr_dist, lr_y = self.lrslrs
325  new_params = [params[0] - lr_dist * grads[0]]
326  lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
327  new_params += [params[1] - lr_y * grads[1]]
328  new_params = euc_project(*new_params)
329  new_params += (params[2] + grads[2],)
330  return new_params

332  def mirror_descent_step(self, params, grads, t):
333  """Entropic mirror descent on exploitability.
334 
335  Args:
336  params: tuple of variables to be updated (dist, y, anneal_steps)
337  grads: tuple of variable gradients (grad_dist, grad_y, grad_anneal_steps)
338  t: int, solver iteration
339  Returns:
340  new_params: tuple of update params (new_dist, new_y, new_anneal_steps)
341  """
342  lr_dist, lr_y = self.lrslrs
343  new_params = [np.log(np.clip(params[0], 0, np.inf)) - lr_dist * grads[0]]
344  lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
345  new_params += [params[1] - lr_y * grads[1]]
346  new_params = mirror_project(*new_params)
347  new_params += (params[2] + grads[2],)
348  return new_params
