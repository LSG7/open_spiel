24 def tf_masked_softmax(logits, legal_actions_mask):
25  """Returns the softmax over the valid actions defined by `legal_actions_mask`.
26 
27  Args:
28  logits: A tensor [..., num_actions] (e.g. [num_actions] or [B, num_actions])
29  representing the logits to mask.
30  legal_actions_mask: The legal action mask, same shape as logits. 1 means
31  it's a legal action, 0 means it's illegal. If can be a tensorflow or numpy
32  tensor.
33  """
34  # This will raise a warning as we are taking the log of 0, which sets the 0
35  # values to -inf. However, this is fine, as we then apply tf.exp, which sets
36  # tf.exp(-inf) to 0. e.g. if we have logits [5, 3, 1], with legal_mask
37  # [0, 1, 1], then masked_logits == [-inf, 3, 1], so we subtract the max to
38  # get [-inf, 0, -2], and apply tf.exp to get [0, 1, e^-2].
39  legal_actions_mask = tf.cast(legal_actions_mask, dtype=logits.dtype)
40  masked_logits = logits + tf.log(legal_actions_mask)
41  max_logit = tf.reduce_max(masked_logits, axis=-1, keepdims=True)
42  exp_logit = tf.exp(masked_logits - max_logit)
43  return exp_logit / tf.reduce_sum(exp_logit, axis=-1, keepdims=True)

46 def np_masked_softmax(logits, legal_actions_mask):
47  """Returns the softmax over the valid actions defined by `legal_actions_mask`.
48 
49  Args:
50  logits: A tensor [..., num_actions] (e.g. [num_actions] or [B, num_actions])
51  representing the logits to mask.
52  legal_actions_mask: The legal action mask, same shape as logits. 1 means
53  it's a legal action, 0 means it's illegal.
54  """
55  masked_logits = logits + np.log(legal_actions_mask)
56  max_logit = np.amax(masked_logits, axis=-1, keepdims=True)
57  exp_logit = np.exp(masked_logits - max_logit)
58  return exp_logit / np.sum(exp_logit, axis=-1, keepdims=True)
