42 def marl_path_finding_example(_):
43  """Example usage of multiagent Nash Q-learner.
44 
45  Based on https://www.jmlr.org/papers/volume4/hu03a/hu03a.pdf
46  """
47 
48  logging.info("Creating the Grid Game")
49  env = rl_environment.Environment(
50  "pathfinding", grid="B.A\n...\na.b", players=2, step_reward=-1.)
51 
52  qlearner = QLearner(0, env.game.num_distinct_actions())
53  nashqlearner = MAQLearner(1, 2, [env.game.num_distinct_actions()] * 2,
54  TwoPlayerNashSolver())
55 
56  time_step = env.reset()
57  actions = [None, None]
58 
59  while not time_step.last():
60  actions = [
61  qlearner.step(time_step).action,
62  nashqlearner.step(time_step, actions).action
63  ]
64  time_step = env.step(actions)
65  print_iteration(actions, env.get_state)
