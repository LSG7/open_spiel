31// A wrapper class to the GAMUT (http://gamut.stanford.edu) game generator.
32// See also, "Run the GAMUT: A Comprehensive Approach to Evaluating
33// Game-Theoretic Algorithms." Eugene Nudelman, Jennifer Wortman,
34// Kevin Leyton-Brown, Yoav Shoham. AAMAS-2004.
35class GamutGenerator {
36 public:
37 // Create a game generator with the specified java executable and GAMUT jar
38 // file. The seed is used for random file names (if 0, uses the current time).
39 GamutGenerator(const std::string& java_path, const std::string& jar_path,
40 int tmpfile_seed = 0);
41
42 // Create a game generator using the default path to java executable, defined
43 // in gamut.cc. The seed is used for random file names (if 0, uses the
44 // current time).
45 GamutGenerator(const std::string& jar_path, int tmpfile_seed = 0);
46
47 // Generate a game using GAMUT command-line arguments. Do not use -f nor
48 // -output, as they are added to the command-line arguments inside this
49 // function.
50 std::shared_ptr<const Game> GenerateGame(const std::string& cmdline_args);
51 std::shared_ptr<const Game> GenerateGame(
52 const std::vector<std::string>& cmdline_args);
53
54 // Same as above; returns a MatrixGame subtype for two-player games.
55 std::shared_ptr<const matrix_game::MatrixGame> GenerateMatrixGame(
56 const std::vector<std::string>& cmdline_args);
57
58 // Same as above; returns a MatrixGame subtype for games with >= 2 players.
59 std::shared_ptr<const tensor_game::TensorGame> GenerateTensorGame(
60 const std::vector<std::string>& cmdline_args);
61
62 private:
63 std::string TmpFile();
64
65 std::string java_path_;
66 std::string jar_path_;
67 absl::Mutex generation_mutex_;
68 std::mt19937 rng_;
69 std::string rand_string_; // Random string used for temp file names.
70};
