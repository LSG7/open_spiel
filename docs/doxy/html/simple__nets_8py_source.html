27class Linear(tf.Module):
28 """A simple linear module.
29
30 Always includes biases and only supports ReLU activations.
31 """
32
33 def __init__(self, in_size, out_size, activate_relu=True, name=None):
34 """Creates a linear layer.
35
36 Args:
37 in_size: (int) number of inputs
38 out_size: (int) number of outputs
39 activate_relu: (bool) whether to include a ReLU activation layer
40 name: (string): the name to give to this layer
41 """
42
43 super(Linear, self).__init__(name=name)
44 self._activate_relu = activate_relu
45 # Weight initialization inspired by Sonnet's Linear layer,
46 # which cites https://arxiv.org/abs/1502.03167v3
47 stddev = 1.0 / math.sqrt(in_size)
48 self._weights = tf.Variable(
49 tf.random.truncated_normal([in_size, out_size], mean=0.0,
50 stddev=stddev),
51 name="weights")
52 self._bias = tf.Variable(tf.zeros([out_size]), name="bias")
53
54 def __call__(self, tensor):
55 y = tf.matmul(tensor, self._weights) + self._bias
56 return tf.nn.relu(y) if self._activate_relu else y

59class Sequential(tf.Module):
60 """A simple sequential module.
61
62 Always includes biases and only supports ReLU activations.
63 """
64
65 def __init__(self, layers, name=None):
66 """Creates a model from successively applying layers.
67
68 Args:
69 layers: Iterable[tf.Module] that can be applied.
70 name: (string): the name to give to this layer
71 """
72
73 super(Sequential, self).__init__(name=name)
74 self._layers = layers
75
76 def __call__(self, tensor):
77 for layer in self._layers:
78 tensor = layer(tensor)
79 return tensor

82class MLP(tf.Module):
83 """A simple dense network built from linear layers above."""
84
85 def __init__(self,
86 input_size,
87 hidden_sizes,
88 output_size,
89 activate_final=False,
90 name=None):
91 """Create the MLP.
92
93 Args:
94 input_size: (int) number of inputs
95 hidden_sizes: (list) sizes (number of units) of each hidden layer
96 output_size: (int) number of outputs
97 activate_final: (bool) should final layer should include a ReLU
98 name: (string): the name to give to this network
99 """
100
101 super(MLP, self).__init__(name=name)
102 self._layers = []
103 with self.name_scope:
104 # Hidden layers
105 for size in hidden_sizes:
106 self._layers.append(Linear(in_size=input_size, out_size=size))
107 input_size = size
108 # Output layer
109 self._layers.append(
110 Linear(
111 in_size=input_size,
112 out_size=output_size,
113 activate_relu=activate_final))
114
115 @tf.Module.with_name_scope
116 def __call__(self, x):
117 for layer in self._layers:
118 x = layer(x)
119 return x
