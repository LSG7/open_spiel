16"""Implementation of dynamic routing game.
17
18The game is derived from https://arxiv.org/abs/2110.11943.
19This dynamic routing game models the evolution of N vehicles in a road network.
20The vehicles are described by their current link location, the time they have to
21spend on the link before exiting it, and their destination. The action of a
22vehicle is the successor link they want to reach when exiting a given link.
23Actions are encoded as integer from 0 to K. Action 0 encodes not being able to
24move on a successor link because the waiting time of the player is still
25positive. Actions 1 to K correspond to the indices of the network links. Legal
26actions for a player on link l, with a negative waiting time are the indices of
27the successors link of l. When arriving on a link, the waiting time of the
28player is assign based on the number of players on the link at this time. Over
29time steps, the waiting time linearly decrease until it is negative, the vehicle
30moves to a successor link and the waiting time get reassigned.
31The cost of the vehicle is its arrival time, it could be seen as a running cost
32where +1 is added to the cost at any time step the vehicle is not on its
33destination.
34This dynamic routing game is a mesoscopic traffic model with explicit congestion
35dynamics where vehicle minimizes their arrival time.
36
37The game is defined by:
38- a network given by the class Network.
39- a list of vehicles given by the class Vehicle.
40
41The current game is implementated as a N player game. However this game can also
42be extended to a mean field game, implemented as python_mfg_dynamic_routing.
43"""

77class DynamicRoutingGame(pyspiel.Game):
78 """Implementation of dynamic routing game.
79
80 At each simultaneous-move time, each vehicle/player with negative waiting time
81 chooses on which successor link they would like to go. When arriving on the
82 link, a waiting time is assigned to the player based on the count of players
83 on the link, after everyone has moved to their successors link. One vehicle
84 arrival time is equal to the time step when they first reach their
85 destination.
86 See module docstring for more information.
87
88 Attributes inherited from GameInfo:
89 max_chance_outcome: 0, the game is deterministic.
90 max_game_length: maximum number of time step played. Passed during
91 construction.
92 max_utility: maximum utility is the opposite of the minimum arrival time.
93 Set to 0.
94 min_utility: minimum utility is the opposite of the maximum arrival time.
95 Set to - max_game_length - 1.
96 num_distinct_actions: maximum number of possible actions. This is equal to
97 the number of links + 1 (corresponding to having no possible action
98 _NO_POSSIBLE_ACTION).
99 num_players: the number of vehicles. Choosen during by the constructor as
100 the number of vehicles.
101 Attributes:
102 network: the network of the game.
103 _vehicles: a list of the vehicle. Their origin and their destination should
104 be road sections of the game. The number of vehicles in the list sets the
105 num_players attribute.
106 time_step_length: size of the time step, used to convert travel times into
107 number of game time steps.
108 perform_sanity_checks: if true, sanity checks are done during the game,
109 should be set to false to speed up the game.
110 """
111 network: dynamic_routing_utils.Network
112 _vehicles: List[dynamic_routing_utils.Vehicle]
113 perform_sanity_checks: bool
114 time_step_length: float
115
116 def __init__(
117 self,
118 params: Mapping[str, Any],
119 network: Optional[dynamic_routing_utils.Network] = None,
120 vehicles: Optional[List[dynamic_routing_utils.Vehicle]] = None,
121 perform_sanity_checks: bool = True,
122 ):
123 """Initiliaze the game.
124
125 Args:
126 params: game parameters. It should define max_num_time_step and
127 time_step_length.
128 network: the network of the game.
129 vehicles: a list of the vehicle. Their origin and their destination should
130 be road sections of the game. The number of vehicles in the list sets
131 the num_players attribute.
132 perform_sanity_checks: set the perform_sanity_checks attribute.
133 """
134 max_num_time_step = params["max_num_time_step"]
135 time_step_length = params["time_step_length"]
136 self.network = network if network else dynamic_routing_data.BRAESS_NETWORK
137 self._vehicles_vehicles = (
138 vehicles
139 if vehicles else dynamic_routing_data.BRAESS_NETWORK_VEHICLES_DEMAND)
140 self.network.check_list_of_vehicles_is_correct(self._vehicles_vehicles)
141 self.perform_sanity_checks = perform_sanity_checks
142 self.time_step_length = time_step_length
143 game_info = pyspiel.GameInfo(
144 num_distinct_actions=self.network.num_actions(),
145 max_chance_outcomes=0,
146 num_players=len(self._vehicles_vehicles),
147 min_utility=-max_num_time_step - 1,
148 max_utility=0,
149 max_game_length=max_num_time_step)
150 super().__init__(_GAME_TYPE, game_info, params if params else {})

164class DynamicRoutingGameState(pyspiel.State):
165 """State of the DynamicRoutingGame.
166
167 One player is equal to one vehicle.
168 See docstring of the game class and of the file for more information.
169 Attributes:
170 _current_time_step: current time step of the game.
171 _is_terminal: boolean that encodes weither the game is over.
172 _time_step_length: size of the time step, used to convert travel times into
173 number of game time steps.
174 _vehicle_at_destination: set of vehicles that have reached their
175 destinations. When a vehicle has reached its destination but the game is
176 not finished, it cannot do anything.
177 _vehicle_destinations: the destination of each vehicle.
178 _vehicle_final_arrival_times: the arrival times of each vehicle, the arrival
179 is either 0 if the vehicle is still in the network or its arrival time if
180 the vehicle has reached its destination.
181 _vehicle_locations: current location of the vehicles as a network road
182 section.
183 _vehicle_without_legal_actions: list of vehicles without legal actions at
184 next time step. This is required because if no vehicle has legal actions
185 for a simultaneous node then an error if raised.
186 _waiting_times: time that each vehicle should wait before being able to move
187 to the next road section.
188 """
189 _current_time_step: int
190 _is_terminal: bool
191 _time_step_length: float
192 _vehicle_at_destination: Set[int]
193 _vehicle_destinations: List[str]
194 _vehicle_final_arrival_times: List[float]
195 _vehicle_locations: List[str]
196 _vehicle_without_legal_actions: Set[int]
197 _waiting_times: List[int]
198
199 def __init__(self, game: DynamicRoutingGame,
200 vehicles: Iterable[dynamic_routing_utils.Vehicle],
201 time_step_length: float):
202 """Constructor; should only be called by Game.new_initial_state."""
203 super().__init__(game)
204 self._current_time_step = 0
205 self._is_terminal = False
206 self._time_step_length = time_step_length
207 self._vehicle_at_destination = set()
208 self._vehicle_destinations = [vehicle.destination for vehicle in vehicles]
209 self._vehicle_final_arrival_times = [0.0 for _ in vehicles]
210 self._vehicle_locations = [vehicle.origin for vehicle in vehicles]
211 self._vehicle_without_legal_actions_vehicle_without_legal_actions = set()
212 self._waiting_times = [
213 int(veh._departure_time / self._time_step_length) for veh in vehicles
214 ]
215 self.running_cost = [0 for vehicle in vehicles]

232 def assert_valid_player(self, vehicle: int):
233 """Assert that a vehicle as a int between 0 and num_players."""
234 assert isinstance(vehicle, int), f"{vehicle} is not a int."
235 assert vehicle >= 0, f"player: {vehicle}<0."
236 assert vehicle < self.get_game().num_players(), (
237 f"player: {vehicle} >= num_players: {self.get_game().num_players()}")

239 def _legal_actions(self, vehicle: int) -> List[int]:
240 """Return the legal actions of the vehicle.
241
242 Legal actions are the succesor road section of the vehicle current road
243 section.
244 Args:
245 vehicle: the vehicle id.
246
247 Returns:
248 list_legal_actions: a list of legal actions. If the game is finished then
249 the list is empty. If the vehicle is at its destination, has a positive
250 waiting time or if it is on a node without successors then an empty list
251 is returned. Otherwise the list of successors nodes of the current
252 vehicle location is returned.
253 """
254 if self._is_terminal:
255 return []
256 if self.get_game().perform_sanity_checks:
257 self.assert_valid_player(vehicle)
258 if vehicle in self._vehicle_without_legal_actions_vehicle_without_legal_actions:
259 # If the vehicle is at destination it cannot do anything.
260 return [dynamic_routing_utils.NO_POSSIBLE_ACTION]
261 if self._waiting_times[vehicle] > 0:
262 return [dynamic_routing_utils.NO_POSSIBLE_ACTION]
263 _, end_section_node = dynamic_routing_utils._nodes_from_road_section( # pylint:disable=protected-access
264 self._vehicle_locations[vehicle])
265 successors = self.get_game().network.get_successors(end_section_node)
266 if successors:
267 assert isinstance(successors, Iterable)
268 actions = [
269 self.get_game().network.get_action_id_from_movement(
270 end_section_node, d) for d in successors
271 ]
272 if self.get_game().perform_sanity_checks:
273 map(self.get_game().network.assert_valid_action, actions)
274 return sorted(actions)
275 return []

277 def _apply_actions(self, actions: List[int]):
278 """Applies the specified action to the state.
279
280 For each vehicle's action, if the vehicle is not at a sink node, if the
281 action is valid and if the waiting time is negative, then the vehicle will
282 move to the successor link corresponding to its action.
283 The function then detects if the vehicle has reached its destination or
284 a sink node and updates _vehicle_at_destination,
285 _vehicle_without_legal_actions and _vehicle_final_arrival_times
286 accordingly.
287 The function then assigns waiting for each vehicle that have moved based on
288 the new volume of cars on the link they reach.
289 The function evolves the time and checks if the game is finished.
290 Args:
291 actions: the action chosen by each vehicle.
292 """
293 if self.get_game().perform_sanity_checks:
294 assert not self._is_terminal
295 if self.get_game().perform_sanity_checks:
296 assert isinstance(actions, Iterable)
297 assert len(actions) == self.get_game().num_players(), (
298 f"Each player does not have an actions. Actions has {len(actions)} "
299 f"elements, it should have {self.get_game().num_players()}.")
300 for vehicle_id, action in enumerate(actions):
301 if vehicle_id not in self._vehicle_at_destination:
302 self.running_cost[vehicle_id] += self._time_step_length
303 # Has the vehicle already reached a sink node?
304 if vehicle_id in self._vehicle_without_legal_actions_vehicle_without_legal_actions:
305 if self.get_game().perform_sanity_checks:
306 assert action == dynamic_routing_utils.NO_POSSIBLE_ACTION, (
307 f"{action} should be {dynamic_routing_utils.NO_POSSIBLE_ACTION}.")
308 continue
309 if self._waiting_times[vehicle_id] > 0:
310 continue
311 if self.get_game().perform_sanity_checks:
312 self.get_game().network.assert_valid_action(
313 action, self._vehicle_locations[vehicle_id])
314 self._vehicle_locations[vehicle_id] = (
315 self.get_game().network.get_road_section_from_action_id(action))
316 if (self._vehicle_locations[vehicle_id] ==
317 self._vehicle_destinations[vehicle_id]):
318 self._vehicle_final_arrival_times[vehicle_id] = self._current_time_step
319 self._vehicle_at_destination.add(vehicle_id)
320 self._vehicle_without_legal_actions_vehicle_without_legal_actions.add(vehicle_id)
321 # Will the vehicle have a legal action for next time step?
322 elif self.get_game().network.is_location_at_sink_node(
323 self._vehicle_locations[vehicle_id]):
324 self._vehicle_without_legal_actions_vehicle_without_legal_actions.add(vehicle_id)
325 self._current_time_step += 1
326 volumes = {}
327 for road_section in self._vehicle_locations:
328 if road_section not in volumes:
329 volumes[road_section] = 0
330 # Each vehicle has a weight a one.
331 volumes[road_section] += 1
332 for vehicle_id, _ in enumerate(actions):
333 # Has the vehicle already reached a sink node?
334 if vehicle_id in self._vehicle_without_legal_actions_vehicle_without_legal_actions:
335 continue
336 if self._waiting_times[vehicle_id] > 0:
337 self._waiting_times[vehicle_id] -= 1
338 else:
339 self._waiting_times[vehicle_id] = int(self.get_game(
340 ).network.get_travel_time(self._vehicle_locations[vehicle_id], volumes[
341 self._vehicle_locations[vehicle_id]]) / self._time_step_length -
342 1.0)
343 # Is the game finished?
344 if (self._current_time_step >= self.get_game().max_game_length() or len(
345 self._vehicle_without_legal_actions_vehicle_without_legal_actions) == self.get_game().num_players()):
346 self._is_terminal = True
347 for vehicle_id in range(self.get_game().num_players()):
348 if vehicle_id not in self._vehicle_at_destination:
349 self._vehicle_final_arrival_times[vehicle_id] = (
350 self._current_time_step)

352 def _action_to_string(self, player, action) -> str:
353 """Action -> string."""
354 if self.get_game().perform_sanity_checks:
355 self.assert_valid_player(player)
356 if action == dynamic_routing_utils.NO_POSSIBLE_ACTION:
357 return f"Vehicle {player} reach a sink node or its destination."
358 if self.get_game().perform_sanity_checks:
359 self.get_game().network.assert_valid_action(action)
360 return (f"Vehicle {player} would like to move to "
361 f"{self.get_game().network.get_road_section_from_action_id(action)}"

364 def is_terminal(self) -> bool:

368 def rewards(self):
369 """Reward at the previous step."""
370 if self._is_terminal or self._current_time_step == 0:
371 return [0 for _ in self._vehicle_locations]
372 reward = [-self._time_step_length for _ in self._vehicle_locations]
373 for vehicle in self._vehicle_at_destination:
374 reward[vehicle] = 0
375 return reward
376
377 def returns(self) -> List[float]:
378 """Total reward for each player over the course of the game so far."""
379 if not self._is_terminal:
380 returns = [
381 -self._time_step_length * self.current_time_step
382 for _ in self._vehicle_locations
383 ]
384 for vehicle in self._vehicle_at_destination:
385 returns[vehicle] = -(
386 self._vehicle_final_arrival_times[vehicle] * self._time_step_length)
387 return returns
388 returns = [
389 -arrival_time * self._time_step_length
390 for arrival_time in self._vehicle_final_arrival_times
391 ]
392 return returns

422class NetworkObserver:
423 """Network observer used by the learning algorithm.
424
425 The state string is the state history string. The state tensor is an array
426 of size max_game_length, num_players where each element is the location of
427 the vehicle at this time.
428 Attributes:
429 dict: dictionary {"observation": tensor}.
430 tensor: list of location for each time step.
431 """
432
433 def __init__(self, num_vehicles: int, num_time: int):
434 """Initializes an empty observation tensor."""
435 shape = (num_time + 1, num_vehicles + 1)
436 self.tensor = np.zeros(np.prod(shape), np.float32)
437 self.dict = {"observation": np.reshape(self.tensor, shape)}
438
439 def set_from(self, state, player):
440 """Update the state tensor.
441
442 Put the locations of each players in the tensor row corresponding to
443 the current time step. Insert the current player location at the
444 beginning of the row.
445 Args:
446 state: the state,
447 player: the player.
448 """
449 vehicles = state.get_current_vehicle_locations_as_int()
450 vehicles.insert(0, state.get_location_as_int(player))
451 self.dict["observation"][state.current_time_step, :] = vehicles
452
453 def string_from(self, state, player):
454 """Return the state history string."""
455 return f"{player}: {state.history_str()}"
