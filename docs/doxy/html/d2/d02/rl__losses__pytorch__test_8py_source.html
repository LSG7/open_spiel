29 @parameterized.named_parameters(('no_entropy_cost', 0.),
30 ('with_entropy_cost', 1.))
31 def test_batch_qpg_loss_with_entropy_cost(self, entropy_cost):
32 batch_qpg_loss = rl_losses.BatchQPGLoss(entropy_cost=entropy_cost)
33 q_values = torch.FloatTensor([[0., -1., 1.], [1., -1., 0]])
34 policy_logits = torch.FloatTensor([[1., 1., 1.], [1., 1., 4.]])
35 total_loss = batch_qpg_loss.loss(policy_logits, q_values)
36 # Compute expected quantities.
37 expected_policy_entropy = (1.0986 + 0.3665) / 2
38 # baseline = \sum_a pi_a * Q_a = 0.
39 # -\sum_a pi_a * (Q_a - baseline)
40 expected_policy_loss = (0.0 + 0.0) / 2
41 expected_total_loss = (
42 expected_policy_loss + entropy_cost * expected_policy_entropy)
43 np.testing.assert_allclose(total_loss, expected_total_loss, atol=1e-4)

92 # policy_loss = cross_entropy * advantages = [-0.3662, 0.04746]

94 expected_policy_loss = (-0.3662 + 0.04746) / 2
95 expected_total_loss = (
