15 """Implements an Ephemeral Value Adjustment Agent.
16 
17 See https://arxiv.org/abs/1810.08163.
18 The algorithm queries trajectories from a replay buffer based on similarities
19 to embedding representations and uses a parametric model to compute values for
20 counterfactual state-action pairs when integrating across those trajectories.
21 Finally, a weighted average between the parametric (DQN in this case) and the
22 non-parametric model is used to compute the policy.
23 """

45 class QueryableFixedSizeRingBuffer(dqn.ReplayBuffer):
46  """ReplayBuffer of fixed size with a FIFO replacement policy.
47 
48  Stored transitions can be sampled uniformly. This extends the DQN replay
49  buffer by allowing the contents to be fetched by L2 proximity to a query
50  value.
51  The underlying datastructure is a ring buffer, allowing 0(1) adding and
52  sampling.
53  """

55  def knn(self, key, key_name, k, trajectory_len=1):
56  """Computes top-k neighbours based on L2 distance.
57 
58  Args:
59  key: (np.array) key value to query memory.
60  key_name: (str) attribute name of key in memory elements.
61  k: (int) number of neighbours to fetch.
62  trajectory_len: (int) length of trajectory to fetch from replay buffer.
63 
64  Returns:
65  List of tuples (L2 negative distance, BufferElement) sorted in increasing
66  order by the negative L2 distqances from the key.
67  """
68  distances = [(np.linalg.norm(getattr(sample, key_name) - key, 2,
69  axis=0), sample) for sample in self._data_data]
70  return sorted(distances, key=lambda v: -v[0])[:k]

301  def _trajectory_centric_planning(self, trajectories):
302  """Performs trajectory centric planning.
303 
304  Uses trajectories from the replay buffer to update the non-parametric values
305  while supplying counter-factual values with the parametric model.
306  Args:
307  trajectories: Current OpenSpiel game state.
308  """
309  # Calculate non-parametric values over the trajectories.
310  # Iterate backward through trajectories
311  for t in range(len(trajectories) - 1, 0, -1):
312  elem = trajectories[t][1]
313  s_tp1 = tuple(elem.next_info_state)
314  s_t = tuple(elem.info_state)
315  a_t = elem.action
316  r_t = elem.reward
317  legal_actions = elem.legal_actions_mask
318  if t < len(trajectories) - 1:
319  for action in range(len(legal_actions)):
320  if not legal_actions[action]:
321  continue
322  if action == elem.action:
323  self._q_np_q_np[s_t][a_t] = (r_t + self._discount_discount * self._v_np_v_np[s_tp1])
324  else:
325  self._agent_agent.info_state = torch.Tensor(
326  np.expand_dims(elem.info_state, axis=0))
327  q_values_parametric = self._agent_agent._q_network(
328  self._agent_agent.info_state).detach().numpy()
329  self._q_np_q_np[s_t][a_t] = q_values_parametric[0][action]
330 
331  # Set V(s_t)
332  if t == len(trajectories) - 1:
333  # Sample from the parametric model.
334  self._agent_agent.info_state = torch.Tensor(
335  np.expand_dims(elem.info_state, axis=0))
336  q_values_parametric = self._agent_agent._q_network(
337  self._agent_agent.info_state).detach().numpy()
338  self._v_np_v_np[s_t] = np.max(q_values_parametric)
339  else:
340  self._v_np_v_np[s_t] = max(self._q_np_q_np[s_t])

342  def _epsilon_greedy(self, q_values, legal_actions, epsilon):
343  """Returns a valid epsilon-greedy action and valid action probs.
344 
345  Action probabilities are given by a softmax over legal q-values.
346  Args:
347  q_values: list of Q-values by action.
348  legal_actions: list of legal actions at `info_state`.
349  epsilon: float, probability of taking an exploratory action.
350 
351  Returns:
352  A valid epsilon-greedy action and valid action probabilities.
353  """
354  probs = np.zeros(self._num_actions_num_actions)
355  q_values = np.array(q_values)
356  if np.random.rand() < epsilon:
357  action = np.random.choice(legal_actions)
358  probs[legal_actions] = 1.0 / len(legal_actions)
359  else:
360  legal_q_values = q_values[legal_actions]
361  action = legal_actions[np.argmax(legal_q_values)]
362  # Reduce max_q for numerical stability. Result is the same.
363  max_q = np.max(legal_q_values)
364  e_x = np.exp(legal_q_values - max_q)
365  probs[legal_actions] = e_x / e_x.sum(axis=0)
366  return action, probs
