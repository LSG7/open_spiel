29 def test_biased_game(self, trials=100, atol=1e-5, rtol=1e-5, seed=1234):
30 """Test best responses to sampled opp. actions in BiasedGame are biased."""
31 game = small.BiasedGame(seed)
32 random = np.random.RandomState(seed)
33
34 successes = []
35 for _ in range(trials):
36 dirichlet_alpha = np.ones(game.num_strategies()[0])
37 dist = random.dirichlet(dirichlet_alpha) # mixed srategy
38
39 sample_best_responses = np.argmax(game.payoff_tensor()[0], axis=0)
40 estimated_best_response = np.dot(sample_best_responses, dist)
41
42 true_best_response = game.best_response(dist)
43
44 successes += [not np.allclose(estimated_best_response, true_best_response,
45 rtol, atol)]
46
47 perc = 100 * np.mean(successes)
48 logging.info('bias rate out of %d is %f', trials, perc)
49 self.assertGreaterEqual(
50 perc, 99., 'best responses should be biased more often')

52 @staticmethod
53 def simp_to_euc(a, b, center):
54 r"""Transforms a point [a, b] on the simplex to Euclidean space.
55
56 /\ ^ b
57 / \ |
58 /____\ --> a
59
60 Args:
61 a: horizonal deviation from center
62 b: vertical deviation from center
63 center: center of ref frame given in [x, y, z] Euclidean coordinates
64 Returns:
65 1-d np.array of len 3, i.e., np.array([x, y, z])
66 """
67 transform = np.array([[.5, -.5, 0], [-.5, -.5, 1], [1, 1, 1]]).T
68 transform /= np.linalg.norm(transform, axis=0)
69 return transform.dot(np.array([a, b, 0])) + center

79 def test_spiral_game(self, dx_1, dy_1, dx_2, dy_2, expected_cos_sim,
80 trials=100, eps=0.1, seed=1234):
81 """Test that gradients on simplex rotate around SpiralGame's center."""
82 random = np.random.RandomState(seed)
83
84 successes = []
85 for _ in range(trials):
86 dx, dy = eps * (random.rand(2) * 2 - 1)
87 center = self.simp_to_euc(dx, dy, np.ones(3) / 3.)
88 game = small.SpiralGame(center, seed)
89 pt = game.payoff_tensor()[0]
90
91 point_1 = self.simp_to_euc(dx_1, dy_1, game.center)
92 point_2 = self.simp_to_euc(dx_2, dy_2, game.center)
93
94 grad_1 = simplex.project_grad(pt.dot(point_1))
95 grad_2 = simplex.project_grad(pt.dot(point_2))
96 norm = np.linalg.norm(grad_1) * np.linalg.norm(grad_2)
97 cos_sim = grad_1.dot(grad_2) / norm
98
99 successes += [(np.abs(cos_sim - expected_cos_sim) < 1e-5)]
100
101 perc = 100 * np.mean(successes)
102 logging.info('alignment success rate out of %d is %f', trials, perc)
103 self.assertGreaterEqual(
104 perc, 99., 'gradient field should exhibit cycles')
