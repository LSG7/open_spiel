35 // Returns the lexicographically next permutation of the supplied bits.
36 // See https://graphics.stanford.edu/~seander/bithacks.html#NextBitPermutation
37 uint64_t bit_twiddle_permute(uint64_t v) {
38  uint64_t t = v | (v - 1);
39  uint64_t u = ((~t & -~t) - 1);
40  int shift = __builtin_ctzl(v) + 1;
41  // Shifting by 64 bits or more is undefined behaviour, so we must avoid it.
42  // See for example: http://c0x.coding-guidelines.com/6.5.7.html (1185).
43  u = (shift < 64) ? (u >> shift) : 0;
44  uint64_t w = (t + 1) | u;
45  return w;
46 }

48 CardSet::CardSet(std::string cardString) : cs() {
49  SPIEL_CHECK_LE(cardString.size(), 10); // Max 5 cards.

52  for (int i = 0; i < cardString.size(); i += 2) {

56  uint8_t rank = (uint8_t)(kRankChars.find(rankChr));

58  SPIEL_CHECK_LT(rank, MAX_RANKS);
59  SPIEL_CHECK_LT(suit, MAX_SUITS);
60  cs.bySuit[suit] |= ((uint16_t)1 << rank);

133 std::vector<CardSet> CardSet::SampleCards(int nbCards) {
134  std::vector<CardSet> combinations;
135 
136  uint64_t p = 0;
137  for (int i = 0; i < nbCards; ++i) {
138  p += (1 << i);
139  }
140  // Enumerates all the uint64_t integers that with nbCards 1-bits.
141  // The final n is ignored. It is fine as long as the rank < 16.
142  for (uint64_t n = bit_twiddle_permute(p); n > p;
143  p = n, n = bit_twiddle_permute(p)) {
144  // Checks whether the CardSet represented by p is inside the CardSet.
145  uint64_t combo = p & cs.cards;
146  if (__builtin_popcountl(combo) == nbCards) {
147  CardSet c;
148  c.cs.cards = combo;
149  combinations.emplace_back(c);
150  }
151  }
152 
153  // std::cout << "combinations.size() " << combinations.size() << std::endl;
154  return combinations;
155 }
