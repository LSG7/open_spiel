123  """Rescales payoffs to be in [0,1]."""

129  """Returns empirical payoffs for each agent."""
130  outcomes = np.zeros(self.n_playersn_players)
131  for k in range(self.n_playersn_players):
132  # compute Bernoulli probabilities
133  outcomes[k] = np.random.choice(

156 def get_payoffs_bernoulli_game(size=(2, 2, 2)):
157  """Gets randomly-generated zero-sum symmetric two-player game."""
158  too_close = True
159  while too_close:
160  M = np.random.uniform(-1, 1, size=size) # pylint: disable=invalid-name
161  M[0, :, :] = 0.5 * (M[0, :, :] - M[0, :, :].T)
162  M[1, :, :] = -M[0, :, :]
163  if np.abs(M[0, 0, 1]) < 0.1:
164  too_close = True
165  else:
166  too_close = False
167  return M

200 def get_game_for_sampler(game_name):
201  """Returns pre-processed game data for ResponseGraphUCB examples."""
202  # pylint: disable=invalid-name
203  if game_name == 'bernoulli':
204  M = get_payoffs_bernoulli_game()
205  strategy_spaces = [2, 2]
206  G = ZeroSumBernoulliGameSampler(
207  strategy_spaces, means=M, payoff_bounds=[-1., 1.])
208  elif game_name == 'soccer':
209  M = get_soccer_data()
210  M = M * 2. - 1 # Convert to zero-sum
211  strategy_spaces = np.shape(M)
212  M = np.asarray([M, M.T])
213  G = ZeroSumBernoulliGameSampler(strategy_spaces, means=M,
214  payoff_bounds=[np.min(M), np.max(M)])
215  elif game_name in ['kuhn_poker_2p', 'kuhn_poker_3p', 'kuhn_poker_4p']:
216  if '2p' in game_name:
217  num_players = 2
218  elif '3p' in game_name:
219  num_players = 3
220  elif '4p' in game_name:
221  num_players = 4
222  M = get_kuhn_poker_data(num_players, iterations=2) # pylint: disable=invalid-name
223  strategy_spaces = egt_utils.get_num_strats_per_population(M, False)
224  G = BernoulliGameSampler(
225  strategy_spaces, means=M, payoff_bounds=[np.min(M), np.max(M)])
226  else:
227  raise ValueError('Game', game_name, 'not implemented!')
228  # pylint: enable=invalid-name
229  return G
