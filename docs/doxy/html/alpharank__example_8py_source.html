30def get_kuhn_poker_data(num_players=3):
31 """Returns the kuhn poker data for the number of players specified."""
32 game = pyspiel.load_game('kuhn_poker', {'players': num_players})
33 xfp_solver = fictitious_play.XFPSolver(game, save_oracles=True)
34 for _ in range(3):
35 xfp_solver.iteration()
36
37 # Results are seed-dependent, so show some interesting cases
38 if num_players == 2:
39 meta_games = xfp_solver.get_empirical_metagame(100, seed=1)
40 elif num_players == 3:
41 meta_games = xfp_solver.get_empirical_metagame(100, seed=5)
42 elif num_players == 4:
43 meta_games = xfp_solver.get_empirical_metagame(100, seed=2)
44
45 # Metagame utility matrices for each player
46 payoff_tables = []
47 for i in range(num_players):
48 payoff_tables.append(meta_games[i])
49 return payoff_tables
