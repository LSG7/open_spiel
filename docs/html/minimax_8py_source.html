26 def _alpha_beta(state, depth, alpha, beta, value_function,
27  maximizing_player_id):
28  """An alpha-beta algorithm.
29 
30  Implements a min-max algorithm with alpha-beta pruning.
31  See for example https://en.wikipedia.org/wiki/Alpha-beta_pruning
32 
33  Arguments:
34  state: The current state node of the game.
35  depth: The maximum depth for the min/max search.
36  alpha: best value that the MAX player can guarantee (if the value is <= than
37  alpha, the MAX player will avoid it).
38  beta: the best value that the MIN currently can guarantee (if the value is
39  >= than beta, the MIN player will avoid it).
40  value_function: An optional function mapping a Spiel `State` to a numerical
41  value, to be used as the value of the maximizing player for a node when we
42  reach `maximum_depth` and the node is not terminal.
43  maximizing_player_id: The id of the MAX player. The other player is assumed
44  to be MIN.
45 
46  Returns:
47  A tuple of the optimal value of the sub-game starting in state
48  (given alpha/beta) and the move that achieved it.
49 
50  Raises:
51  NotImplementedError: If we reach the maximum depth. Given we have no value
52  function for a non-terminal node, we cannot break early.
53  """
54  if state.is_terminal():
55  return state.player_return(maximizing_player_id), None
56 
57  if depth == 0 and value_function is None:
58  raise NotImplementedError(
59  "We assume we can walk the full depth of the tree. "
60  "Try increasing the maximum_depth or provide a value_function.")
61  if depth == 0:
62  return value_function(state), None
63 
64  player = state.current_player()
65  best_action = -1
66  if player == maximizing_player_id:
67  value = -float("inf")
68  for action in state.legal_actions():
69  child_state = state.clone()
70  child_state.apply_action(action)
71  child_value, _ = _alpha_beta(child_state, depth - 1, alpha, beta,
72  value_function, maximizing_player_id)
73  if child_value > value:
74  value = child_value
75  best_action = action
76  alpha = max(alpha, value)
77  if alpha >= beta:
78  break # beta cut-off
79  return value, best_action
80  else:
81  value = float("inf")
82  for action in state.legal_actions():
83  child_state = state.clone()
84  child_state.apply_action(action)
85  child_value, _ = _alpha_beta(child_state, depth - 1, alpha, beta,
86  value_function, maximizing_player_id)
87  if child_value < value:
88  value = child_value
89  best_action = action
90  beta = min(beta, value)
91  if alpha >= beta:
92  break # alpha cut-off
93  return value, best_action

96 def alpha_beta_search(game,
97  state=None,
98  value_function=None,
99  maximum_depth=30,
100  maximizing_player_id=None):
101  """Solves deterministic, 2-players, perfect-information 0-sum game.
102 
103  For small games only! Please use keyword arguments for optional arguments.
104 
105  Arguments:
106  game: The game to analyze, as returned by `load_game`.
107  state: The state to run from, as returned by `game.new_initial_state()`. If
108  none is specified, then the initial state is assumed.
109  value_function: An optional function mapping a Spiel `State` to a numerical
110  value, to be used as the value of the maximizing player for a node when we
111  reach `maximum_depth` and the node is not terminal.
112  maximum_depth: The maximum depth to search over. When this depth is reached,
113  an exception will be raised.
114  maximizing_player_id: The id of the MAX player. The other player is assumed
115  to be MIN. The default (None) will suppose the player at the root to be
116  the MAX player.
117 
118  Returns:
119  A tuple containing the value of the game for the maximizing player when
120  both player play optimally, and the action that achieves this value.
121  """
122  game_info = game.get_type()
123 
124  if game.num_players() != 2:
125  raise ValueError("Game must be a 2-player game")
126  if game_info.chance_mode != pyspiel.GameType.ChanceMode.DETERMINISTIC:
127  raise ValueError("The game must be a Deterministic one, not {}".format(
128  game.chance_mode))
129  if game_info.information != pyspiel.GameType.Information.PERFECT_INFORMATION:
130  raise ValueError(
131  "The game must be a perfect information one, not {}".format(
132  game.information))
133  if game_info.dynamics != pyspiel.GameType.Dynamics.SEQUENTIAL:
134  raise ValueError("The game must be turn-based, not {}".format(
135  game.dynamics))
136  if game_info.utility != pyspiel.GameType.Utility.ZERO_SUM:
137  raise ValueError("The game must be 0-sum, not {}".format(game.utility))
138 
139  if state is None:
140  state = game.new_initial_state()
141  if maximizing_player_id is None:
142  maximizing_player_id = state.current_player()
143  return _alpha_beta(
144  state.clone(),
145  maximum_depth,
146  alpha=-float("inf"),
147  beta=float("inf"),
148  value_function=value_function,
149  maximizing_player_id=maximizing_player_id)

152 def expectiminimax(state, depth, value_function, maximizing_player_id):
153  """Runs expectiminimax until the specified depth.
154 
155  See https://en.wikipedia.org/wiki/Expectiminimax for details.
156 
157  Arguments:
158  state: The state to start the search from.
159  depth: The depth of the search (not counting chance nodes).
160  value_function: A value function, taking in a state and returning a value,
161  in terms of the maximizing_player_id.
162  maximizing_player_id: The player running the search (current player at root
163  of the search tree).
164 
165  Returns:
166  A tuple (value, best_action) representing the value to the maximizing player
167  and the best action that achieves that value. None is returned as the best
168  action at chance nodes, the depth limit, and terminals.
169  """
170  if state.is_terminal():
171  return state.player_return(maximizing_player_id), None
172 
173  if depth == 0:
174  return value_function(state), None
175 
176  if state.is_chance_node():
177  value = 0
178  for outcome, prob in state.chance_outcomes():
179  child_state = state.clone()
180  child_state.apply_action(outcome)
181  child_value, _ = expectiminimax(child_state, depth, value_function,
182  maximizing_player_id)
183  value += prob * child_value
184  return value, None
185  elif state.current_player() == maximizing_player_id:
186  value = -float("inf")
187  for action in state.legal_actions():
188  child_state = state.clone()
189  child_state.apply_action(action)
190  child_value, _ = expectiminimax(child_state, depth - 1, value_function,
191  maximizing_player_id)
192  if child_value > value:
193  value = child_value
194  best_action = action
195  return value, best_action
196  else:
197  value = float("inf")
198  for action in state.legal_actions():
199  child_state = state.clone()
200  child_state.apply_action(action)
201  child_value, _ = expectiminimax(child_state, depth - 1, value_function,
202  maximizing_player_id)
203  if child_value < value:
204  value = child_value
205  best_action = action
206  return value, best_action
