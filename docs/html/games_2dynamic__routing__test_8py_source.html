242  class TruePathPolicy(policy.Policy):
243 
244  def __init__(self, game):
245  super().__init__(game, list(range(num_player)))
246  self._path = {}
247 
248  def action_probabilities(self, state, player_id=None):
249  assert player_id is not None
250  legal_actions = state.legal_actions(player_id)
251  if not legal_actions:
252  return {dynamic_routing_utils.NO_POSSIBLE_ACTION: 1.0}
253  elif len(legal_actions) == 1:
254  return {legal_actions[0]: 1.0}
255  else:
256  if legal_actions[0] == 1:
257  if self._path[player_id] in ["top", "middle"]:
258  return {1: 1.0}
259  elif self._path[player_id] == "bottom":
260  return {2: 1.0}
261  else:
262  raise ValueError()
263  elif legal_actions[0] == 3:
264  if self._path[player_id] == "top":
265  return {4: 1.0}
266  elif self._path[player_id] == "middle":
267  return {3: 1.0}
268  else:
269  raise ValueError()
270  raise ValueError(f"{legal_actions} is not correct.")
