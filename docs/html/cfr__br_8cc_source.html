50 void CFRBRSolver::EvaluateAndUpdatePolicy() {
51  ++iteration_;
52 
53  std::vector<TabularPolicy> br_policies(game_->NumPlayers());
54  std::shared_ptr<Policy> current_policy = CurrentPolicy();
55 
56  // Set all the player's policies first.
57  for (int p = 0; p < game_->NumPlayers(); ++p) {
58  // Need to have an exception here because the CFR policy objects are
59  // wrappers around information that is contained in a table, and those do
60  // not exist until there's been a tree traversal to compute regrets below.
61  if (iteration_ > 1) {
62  best_response_computers_[p]->SetPolicy(current_policy.get());
63  }
64  }
65 
66  // Now, for each player compute a best response
67  for (int p = 0; p < game_->NumPlayers(); ++p) {
68  br_policies[p] = best_response_computers_[p]->GetBestResponsePolicy();
69  }
70 
71  for (int p = 0; p < game_->NumPlayers(); ++p) {
72  // Override every player except p.
73  for (int opp = 0; opp < game_->NumPlayers(); ++opp) {
74  policy_overrides_[opp] = (opp == p ? nullptr : &br_policies[opp]);
75  }
76 
77  // Then collect regret and update p's average strategy.
78  ComputeCounterFactualRegret(*root_state_, p, root_reach_probs_,
79  &policy_overrides_);
80  }
81  ApplyRegretMatching();
82 }
