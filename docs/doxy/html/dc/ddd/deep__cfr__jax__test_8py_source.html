43 def test_matching_pennies_3p(self):
44 # We don't expect Deep CFR to necessarily converge on 3-player games but
45 # it's nonetheless interesting to see this result.
46 game = pyspiel.load_game_as_turn_based('matching_pennies_3p')
47 deep_cfr_solver = deep_cfr.DeepCFRSolver(
48 game,
49 policy_network_layers=(16, 8),
50 advantage_network_layers=(32, 16),
51 num_iterations=2,
52 num_traversals=2,
53 learning_rate=1e-3,
54 batch_size_advantage=8,
55 batch_size_strategy=8,
56 memory_capacity=1e7)
57 deep_cfr_solver.solve()
58 conv = exploitability.nash_conv(
59 game,
60 policy.tabular_policy_from_callable(
61 game, deep_cfr_solver.action_probabilities))
62 print('Deep CFR in Matching Pennies 3p. NashConv: {}'.format(conv))
