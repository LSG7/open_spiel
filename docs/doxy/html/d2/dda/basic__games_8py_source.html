29class IceCreamGame(coalitional_game.CoalitionalGame):
30 """Example 2.2 from CACGT book by Chalkiadakis, Elkind, and Wooldridge."""
31
32 def __init__(self):
33 super().__init__(num_players=3)
34
35 def coalition_value(self, coalition: np.ndarray) -> float:
36 """Encodes the payoffs."""
37 # indices ordered as C M P
38 if coalition.sum() < 2:
39 return 0.0
40 elif coalition[0] == 1 and coalition[1] == 1 and coalition[2] == 0:
41 # {C, M}
42 return 500.0
43 elif coalition[0] == 1 and coalition[1] == 0 and coalition[2] == 1:
44 # {C, P}
45 return 500.0
46 elif coalition[0] == 0 and coalition[1] == 1 and coalition[2] == 1:
47 # {M, P}
48 return 750.0
49 elif coalition.sum() == 3:
50 return 1000.0
51 else:
52 raise RuntimeError("Invalid coalition")

55class TabularGame(coalitional_game.CoalitionalGame):
56 """A game represented by a table of values."""
57
58 def __init__(self, table: Dict[Tuple[int, ...], float]):
59 super().__init__(num_players=-1) # set num players to -1 for now
60 for key in table:
61 if self._num_players_num_players < 0:
62 self._num_players_num_players = len(key)
63 else:
64 assert len(key) == self._num_players_num_players
65 assert self._num_players_num_players >= 1
66 self._table = table
67
68 def coalition_value(self, coalition: np.ndarray) -> float:
69 return self._table[tuple(coalition)]
