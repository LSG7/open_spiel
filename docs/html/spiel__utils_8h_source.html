120 // Builds a string from pieces:
121 //
122 // SpielStrCat(1, " + ", 1, " = ", 2) --> "1 + 1 = 2"
123 //
124 // Converting the parameters to strings is done using the stream operator<<.
125 // This is only kept around to be used in the SPIEL_CHECK_* macros and should
126 // not be called by any code outside of this file. Prefer absl::StrCat instead.
127 // It is kept here due to support for more types, including char.
128 template <typename... Args>
129 std::string SpielStrCat(Args&&... args) {
130  std::ostringstream out;
131  SpielStrOut(out, std::forward<Args>(args)...);
132  return out.str();
133 }

188 // Returns whether the absolute difference between floating point values a and
189 // b is less than or equal to FloatingPointThresholdRatio() * max(|a|, |b|).
190 template <typename T>
191 bool Near(T a, T b) {
192  static_assert(std::is_floating_point<T>::value,
193  "Near() is only for floating point args.");
194  return fabs(a - b) <=
195  (std::max(fabs(a), fabs(b)) * FloatingPointDefaultThresholdRatio());
196 }
197 
198 // Returns whether |a - b| <= epsilon.
199 template <typename T>
200 bool Near(T a, T b, T epsilon) {
201  static_assert(std::is_floating_point<T>::value,
202  "Near() is only for floating point args.");
203  return fabs(a - b) <= epsilon;
204 }

206 template <typename T>
207 bool AllNear(const std::vector<T>& vector1, const std::vector<T>& vector2,
208  T epsilon) {
209  if (vector1.size() != vector2.size()) {
210  return false;
211  }
212  for (int i = 0; i < vector1.size(); ++i) {
213  if (!Near(vector1[i], vector2[i], epsilon)) {
214  return false;
215  }
216  }
217  return true;
218 }

225 #define SPIEL_CHECK_OP(x_exp, op, y_exp) \
226  do { \
227  auto x = x_exp; \
228  auto y = y_exp; \
229  if (!((x)op(y))) \
230  open_spiel::SpielFatalError(open_spiel::internal::SpielStrCat( \
231  __FILE__, ":", __LINE__, " ", #x_exp " " #op " " #y_exp, \
232  "\n" #x_exp, " = ", x, ", " #y_exp " = ", y)); \
233  } while (false)
234 
235 #define SPIEL_CHECK_FN2(x_exp, y_exp, fn) \
236  do { \
237  auto x = x_exp; \
238  auto y = y_exp; \
239  if (!fn(x, y)) \
240  open_spiel::SpielFatalError(open_spiel::internal::SpielStrCat( \
241  __FILE__, ":", __LINE__, " ", #fn "(" #x_exp ", " #y_exp ")\n", \
242  #x_exp " = ", x, ", " #y_exp " = ", y)); \
243  } while (false)
244 
245 #define SPIEL_CHECK_FN3(x_exp, y_exp, z_exp, fn) \
246  do { \
247  auto x = x_exp; \
248  auto y = y_exp; \
249  auto z = z_exp; \
250  if (!fn(x, y, z)) \
251  open_spiel::SpielFatalError(open_spiel::internal::SpielStrCat( \
252  __FILE__, ":", __LINE__, " ", \
253  #fn "(" #x_exp ", " #y_exp ", " #z_exp ")\n", #x_exp " = ", x, \
254  ", " #y_exp " = ", y, ", " #z_exp " = ", z)); \
255  } while (false)

389 // Utility functions intended to be used for casting
390 // from a Base class to a Derived subclass.
391 // These functions handle various use cases, such as pointers and const
392 // references. For shared or unique pointers you can get the underlying pointer.
393 // When you use debug mode, a more expensive dynamic_cast is used and it checks
394 // whether the casting has been successful. In optimized builds only static_cast
395 // is used when possible.
396 
397 // use like this: down_cast<T*>(foo);
398 template <typename To, typename From>
399 inline To down_cast(From* f) {
400 #if !defined(NDEBUG)
401  if (f != nullptr && dynamic_cast<To>(f) == nullptr) {
402  std::string from = typeid(From).name();
403  std::string to = typeid(From).name();
404  SpielFatalError(
405  absl::StrCat("Cast failure: could not cast a pointer from '", from,
406  "' to '", to, "'"));
407  }
408 #endif
409  return static_cast<To>(f);
410 }
411 
412 // use like this: down_cast<T&>(foo);
413 template <typename To, typename From>
414 inline To down_cast(From& f) {
415  typedef typename std::remove_reference<To>::type* ToAsPointer;
416 #if !defined(NDEBUG)
417  if (dynamic_cast<ToAsPointer>(&f) == nullptr) {
418  std::string from = typeid(From).name();
419  std::string to = typeid(From).name();
420  SpielFatalError(
421  absl::StrCat("Cast failure: could not cast a reference from '", from,
422  "' to '", to, "'"));
423  }
424 #endif
425  return *static_cast<ToAsPointer>(&f);
426 }
