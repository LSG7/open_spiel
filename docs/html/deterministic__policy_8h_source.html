39 // action) for each information state.
40 struct LegalsWithIndex {

45  void SetAction(Action action) {
46  auto iter = std::find(legal_actions_.begin(), legal_actions_.end(), action);
47  SPIEL_CHECK_TRUE(iter != legal_actions_.end());
48  index = std::distance(legal_actions_.begin(), iter);
49  }
50 
51  Action GetAction() const { return legal_actions_[index]; }

54  // deterministic policies (DeterministicPolicy::NextPolicy) below.
55  bool TryIncIndex() {
56  if (index + 1 < legal_actions_.size()) {
57  index += 1;
58  return true;
59  } else {
60  return false;

85  // Deterministic policies are ordered. First, we define some order to the
86  // information states (which is the order defined by the legal_actions_map
87  // for the game). Then the total order over policies is defined in a
88  // "counting order according to their associated tuple (
89  // legal_action_index[state] for state in ordered_states). The first
90  // deterministic policy in the order is the one whose action is set is the
91  // first legal action (legal action index = 0). The value of the index can be
92  // interpreted as a digit in a mixed base integer, where the value of the
93  // integer would represent the position of the deterministic policy in the
94  // total order.
95  //
96  // This function sets this policy to the next deterministic policy in this
97  // counting order. The function returns true if this changed the policy (i.e
98  // there exists a next policy in the order), otherwise returns false.
99  bool NextPolicy();

103  // list).
