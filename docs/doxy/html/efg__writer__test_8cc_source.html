30void WriteAndLoadGame(std::string game_name) {
31 std::string filename =
32 absl::StrCat(file::GetTmpDir(), "/open_spiel_test_", game_name, ".efg");
33 std::shared_ptr<const Game> base_game = LoadGame(game_name);
34
35 EFGWriter(*base_game, filename).Write();
36
37 std::shared_ptr<const Game> efg_game =
38 LoadGame("efg_game", {{"filename", GameParameter(filename)}});
39 SPIEL_CHECK_TRUE(efg_game != nullptr);
40 GameType base_game_type = base_game->GetType();
41 GameType efg_game_type = efg_game->GetType();
42 SPIEL_CHECK_EQ(efg_game_type.dynamics, GameType::Dynamics::kSequential);
43 SPIEL_CHECK_EQ(efg_game_type.information, base_game_type.information);
44 SPIEL_CHECK_EQ(efg_game_type.utility, base_game_type.utility);
45 SPIEL_CHECK_EQ(efg_game_type.chance_mode, base_game_type.chance_mode);
46 SPIEL_CHECK_EQ(efg_game->NumDistinctActions(),
47 base_game->NumDistinctActions());
48 // TODO(author11) More comprehensive tests that the games are the same.
49 testing::RandomSimTestNoSerialize(*efg_game, 100);
50
51 SPIEL_CHECK_TRUE(file::Exists(filename));
52 SPIEL_CHECK_TRUE(file::Remove(filename));
53 SPIEL_CHECK_FALSE(file::Remove(filename)); // Already gone
54 SPIEL_CHECK_FALSE(file::Exists(filename));
55}
