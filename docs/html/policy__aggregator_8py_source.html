26 class PolicyFunction(policy.Policy):
27  """A callable policy class."""

29  def __init__(self, pids, policies, game):
30  """Construct a policy function.
31 
32  Arguments:
33  pids: spiel player id of players these policies belong to.
34  policies: a list of dictionaries of keys (stringified binary observations)
35  to a list of probabilities for each move uid (between 0 and max_moves -
36  1).
37  game: OpenSpiel game.
38  """
39  super().__init__(game, pids)
40  self._policies_policies = policies
41  self._game_type_game_type = game.get_type()

43  def _state_key(self, state, player_id=None):
44  """Returns the key to use to look up this (state, player_id) pair."""
45  if self._game_type_game_type.provides_information_state_string:
46  if player_id is None:
47  return state.information_state_string()
48  else:
49  return state.information_state_string(player_id)
50  elif self._game_type_game_type.provides_observation_tensor:
51  if player_id is None:
52  return state.observation_tensor()
53  else:
54  return state.observation_tensor(player_id)

62  def action_probabilities(self, state, player_id=None):
63  """Returns the policy for a player in a state.
64 
65  Args:
66  state: A `pyspiel.State` object.
67  player_id: Optional, the player id for whom we want an action. Optional
68  unless this is a simultaneous state at which multiple players can act.
69 
70  Returns:
71  A `dict` of `{action: probability}` for the specified player in the
72  supplied state.
73  """
74  state_key = self._state_key_state_key(state, player_id=player_id)
75  if state.is_simultaneous_node():
76  # for simultaneous node, assume player id must be provided
77  assert player_id >= 0
78  return self._policies_policies[player_id][state_key]
79  if player_id is None:
80  player_id = state.current_player()
81  return self._policies_policies[player_id][state_key]

84 class PolicyPool(object):
85  """Transforms a list of list of policies (One list per player) to callable."""
86 
87  def __init__(self, policies):
88  """Transforms a list of list of policies (One list per player) to callable.
89 
90  Args:
91  policies: List of list of policies.
92  """

95  def __call__(self, state, player):

147  def _sub_aggregate(self, pid, policies, weights):
148  """Aggregate the list of policies for one player.
149 
150  Arguments:
151  pid: the spiel player id of the player the strategies belong to.
152  policies: List of list of policies (One list per player)
153  weights: the list of weights to attach to each policy.
154 
155  Returns:
156  A PolicyFunction, a callable object representing the policy.
157  """
158  self._policy_pool_policy_pool = PolicyPool(policies)
159  # ipdb.set_trace()
160 
161  assert self._policy_pool_policy_pool is not None
162  self._weights_weights = weights
163  # string of state -> probs list
164  self._policy_policy = {}
165 
166  state = self._game_game.new_initial_state()
167  my_reaches = weights[:]
168  self._rec_aggregate_rec_aggregate(pid, state, my_reaches)
169 
170  # Now normalize
171  for key in self._policy_policy:
172  actions, probabilities = zip(*self._policy_policy[key].items())
173  # Add some small proba mass to avoid divide by zero, which happens for
174  # games with low reach probabilities for certain states (keys)
175  new_probs = [prob + self._epsilon_epsilon for prob in probabilities]
176  denom = sum(new_probs)
177  for i in range(len(actions)):
178  self._policy_policy[key][actions[i]] = new_probs[i] / denom
179  return self._policy_policy

181  def _rec_aggregate(self, pid, state, my_reaches):
182  """Recursively traverse game tree to compute aggregate policy."""
183 
184  if state.is_terminal():
185  return
186  elif state.is_simultaneous_node():
187 
188  policies = self._policy_pool_policy_pool(state, pid)
189  state_key = self._state_key_state_key(state, pid)
190  self._policy_policy[state_key] = {}
191  used_moves = state.legal_actions(pid)
192 
193  for uid in used_moves:
194  new_reaches = copy.deepcopy(my_reaches)
195  for i in range(len(policies)):
196  # compute the new reach for each policy for this action
197  new_reaches[pid][i] *= policies[i].get(uid, 0)
198  # add reach * prob(a) for this policy to the computed policy
199  if uid in self._policy_policy[state_key].keys():
200  self._policy_policy[state_key][uid] += new_reaches[pid][i]
201  else:
202  self._policy_policy[state_key][uid] = new_reaches[pid][i]
203 
204  num_players = self._game_game.num_players()
205  all_other_used_moves = []
206  for player in range(num_players):
207  if player != pid:
208  all_other_used_moves.append(state.legal_actions(player))
209 
210  other_joint_actions = itertools.product(*all_other_used_moves)
211 
212  # enumerate every possible other-agent actions for next-state
213  for other_joint_action in other_joint_actions:
214  for uid in used_moves:
215  new_reaches = copy.deepcopy(my_reaches)
216  for i in range(len(policies)):
217  # compute the new reach for each policy for this action
218  new_reaches[pid][i] *= policies[i].get(uid, 0)
219 
220  joint_action = list(
221  other_joint_action[:pid] + (uid,) + other_joint_action[pid:]
222  )
223  new_state = state.clone()
224  new_state.apply_actions(joint_action)
225  self._rec_aggregate_rec_aggregate(pid, new_state, new_reaches)
226  return
227 
228  elif state.is_chance_node():
229  # do not factor in opponent reaches
230  outcomes, _ = zip(*state.chance_outcomes())
231  for i in range(0, len(outcomes)):
232  outcome = outcomes[i]
233  new_state = state.clone()
234  new_state.apply_action(outcome)
235  self._rec_aggregate_rec_aggregate(pid, new_state, my_reaches)
236  return
237  else:
238  turn_player = state.current_player()
239 
240  state_key = self._state_key_state_key(state, turn_player)
241  legal_policies = self._policy_pool_policy_pool(state, turn_player)
242  if pid == turn_player:
243  # update the current node
244  # will need the observation to query the policies
245  if state_key not in self._policy_policy:
246  self._policy_policy[state_key] = {}
247 
248  used_moves = state.legal_actions(turn_player)
249 
250  for uid in used_moves:
251  new_reaches = copy.deepcopy(my_reaches)
252  if pid == turn_player:
253  for i in range(len(legal_policies)):
254  # compute the new reach for each policy for this action
255  new_reaches[turn_player][i] *= legal_policies[i].get(uid, 0)
256  # add reach * prob(a) for this policy to the computed policy
257  if uid in self._policy_policy[state_key].keys():
258  self._policy_policy[state_key][uid] += new_reaches[turn_player][i]
259  else:
260  self._policy_policy[state_key][uid] = new_reaches[turn_player][i]
261 
262  # recurse
263  new_state = state.clone()
264  new_state.apply_action(uid)
265  self._rec_aggregate_rec_aggregate(pid, new_state, new_reaches)
