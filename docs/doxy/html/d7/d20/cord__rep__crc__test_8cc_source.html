42absl::crc_internal::CrcCordState MakeCrcCordState(uint32_t crc) {
43 crc_internal::CrcCordState state;
44 state.mutable_rep()->prefix_crc.push_back(
45 crc_internal::CrcCordState::PrefixCrc(42, crc32c_t{crc}));

49TEST(CordRepCrc, NewDestroy) {
50 CordRep* rep = cordrep_testing::MakeFlat("Hello world");
51 CordRepCrc* crc = CordRepCrc::New(rep, MakeCrcCordState(12345));
52 EXPECT_TRUE(crc->refcount.IsOne());

71TEST(CordRepCrc, NewExistingCrcShared) {
72 CordRep* rep = cordrep_testing::MakeFlat("Hello world");
73 CordRepCrc* crc = CordRepCrc::New(rep, MakeCrcCordState(12345));
74 CordRep::Ref(crc);
75 CordRepCrc* new_crc = CordRepCrc::New(crc, MakeCrcCordState(54321));
76
77 EXPECT_THAT(new_crc, Ne(crc));
78 EXPECT_TRUE(new_crc->refcount.IsOne());
79 EXPECT_TRUE(crc->refcount.IsOne());
80 EXPECT_FALSE(rep->refcount.IsOne());
81 EXPECT_THAT(crc->child, Eq(rep));
82 EXPECT_THAT(new_crc->child, Eq(rep));
83 EXPECT_THAT(crc->crc_cord_state.Checksum(), Eq(crc32c_t{12345u}));
84 EXPECT_THAT(new_crc->crc_cord_state.Checksum(), Eq(crc32c_t{54321u}));
85
86 CordRep::Unref(crc);
87 CordRep::Unref(new_crc);
88}
