23def strip_msg(text: str,
24 block_msg: str,
25 block_opt: str,
26 terminal_str: str = '') -> str:
27 """Strip email message (with header) from text block, i.e., [ (A) - (B) ).
28
29 Assumes messages adhere to the following format:
30 BLOCK_OPT
31 <-- action & info -->
32 BLOCK_MSG (A)
33 <-- e.g., sender/receiver -->
34 BLOCK_MSG
35 <-- e.g., message -->
36 BLOCK_OPT (B)
37
38 Args:
39 text: str
40 block_msg: str, string of characters delineating the message
41 block_opt: str, string of characters demarking the start of
42 the options (actions and info)
43 terminal_str: str (optional), indicates the end of a message if block_opt
44 is not found. this will be included in the stripped output.
45 Returns:
46 stripped_text: str
47 """
48 ctr = 0
49 right_ptr = 0
50 left_ptr = text.find(block_msg)
51 if left_ptr == -1:
52 return ''
53 while ctr < 2:
54 block_idx = text[right_ptr:].find(block_msg)
55 if block_idx == -1:
56 return ''
57 right_ptr += block_idx + len(block_msg)
58 ctr += 1
59 block_idx = text[right_ptr:].find(block_opt)
60 if block_idx != -1: # if find block_opt return message ending at (B)
61 right_ptr += block_idx
62 else:
63 if terminal_str: # if no block_opt, return message ending at terminal_str
64 block_idx = text[right_ptr:].find(terminal_str)
65 if block_idx != -1:
66 right_ptr += block_idx + len(terminal_str)
67 else: # if no terminal_str, return message to end of text string
68 right_ptr = len(text)
69 return text[left_ptr:right_ptr]

72def first_special_char(text: str,
73 max_idx: int,
74 special_chars: Tuple[str, ...]) -> int:
75 first_special_chars = [max_idx]
76 for char in special_chars:
77 idx = text.find(char)
78 if idx < 0:
79 first_special_chars.append(max_idx)
80 else:
81 first_special_chars.append(idx)
82 return min(first_special_chars)

85def retrieve_special_char_block(text: str,
86 special_chars: Tuple[str, ...] = ('*',),
87 useless_chars: Tuple[str, ...] = (' ', '\n')):
88 for char in special_chars:
89 text = text.strip(char)
90 idx_end = first_special_char(text, len(text), special_chars)
91 text = text[:idx_end]
92 for char in useless_chars:
93 text = text.strip(char)
94 return text

97def retrieve_alpha_block(text: str) -> str:
98 """Return the first instance of a contiguous alpha(not numeric) substring."""
99 first_alpha_char = next(filter(str.isalpha, text), -1)
100 if first_alpha_char == -1:
101 return ''
102 start = text.find(first_alpha_char)
103 sliced = text[start:]
104 last_alpha_char = next(filter(lambda s: not str.isalpha(s), sliced), -1)
105 if last_alpha_char == -1:
106 return sliced
107 finish = sliced.find(last_alpha_char)
108 return text[start:start + finish]

111def retrieve_numeric_block(text: str) -> str:
112 """Return the first instance of a contiguous numeric(not alpha) substring."""
113 first_numeric_char = next(filter(str.isnumeric, text), -1)
114 if first_numeric_char == -1:
115 return ''
116 start = text.find(first_numeric_char)
117 sliced = text[start:]
118 last_numeric_char = next(filter(lambda s: not str.isnumeric(s), sliced), -1)
119 if start > 0 and text[start - 1] == '-':
120 start -= 1
121 sliced = text[start:]
122 if last_numeric_char == -1:
123 return sliced
124 finish = sliced.find(last_numeric_char)
125 return text[start:start + finish]

128def wrap(message: List[str]) -> List[str]:
129 """Given a list of strings, returns a list of them `wrapped` (paragraphs).
130
131 Args:
132 message: list of strings
133 Returns:
134 wrapped: list of strings with each string `wrapped` so that each line only
135 contains (default) 70 characters
136 """
137 wrapped = []
138 for sub_msg in message:
139 sub_msg_wrapped = textwrap.wrap(sub_msg)
140 if len(sub_msg_wrapped) > 1:
141 sub_msg_wrapped = ['\n'.join(sub_msg_wrapped)]
142 wrapped.extend(sub_msg_wrapped)
143 return wrapped
