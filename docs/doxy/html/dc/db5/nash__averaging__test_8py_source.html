68class NashAveragingTest(parameterized.TestCase):
69
70 @parameterized.named_parameters(
71 ("transitive_game", game_trans, eq_trans, value_trans),
72 ("rps_game", game_rps, eq_rps, value_rps),
73 )
74 def test_simple_games(self, game, eq, value):
75
76 maxent_nash, nash_avg_value = nash_averaging(game)
77 with self.subTest("probability"):
78 np.testing.assert_array_almost_equal(eq, maxent_nash.reshape(-1))
79
80 with self.subTest("value"):
81 np.testing.assert_array_almost_equal(value, nash_avg_value.reshape(-1))
82
83 @parameterized.named_parameters(
84 ("game0", game0, dominated_idxs0),)
85 def test_ava_games_with_dominated_strategy(self, game, dominated_idxs):
86 maxent_nash, _ = nash_averaging(game)
87 with self.subTest("dominated strategies have zero Nash probs"):
88 for idx in dominated_idxs:
89 self.assertAlmostEqual(maxent_nash[idx].item(), 0.0)
90
91 @parameterized.named_parameters(
92 ("game1", game1, dominated_idxs1),
93 )
94 def test_avt_games_with_dominated_strategy(self, game, dominated_idxs):
95 (agent_strategy, _), _ = nash_averaging(game, a_v_a=False)
96 with self.subTest("dominated strategies have zero Nash probs"):
97 for idx in dominated_idxs:
98 self.assertAlmostEqual(agent_strategy[idx].item(), 0.0)
99
100 @parameterized.named_parameters(
101 ("game2", game2, dom_idxs2),
102 )
103 def test_avt_games_with_multiple_dominant_strategies(self, game, dom_idxs):
104 (agent_strategy, _), (agent_values, _) = nash_averaging(game, a_v_a=False)
105 with self.subTest("dominant strategies have equal Nash probs"):
106 for idx in dom_idxs:
107 self.assertAlmostEqual(agent_strategy[idx].item(), 1 / len(dom_idxs2))
108
109 with self.subTest("dominant strategies have equal Nash values"):
110 values = [agent_values[idx] for idx in dom_idxs]
111 self.assertAlmostEqual(np.abs(np.max(values) - np.min(values)), 0.0)
