59 std::string DouDizhuState::ActionToString(Player player, Action action) const {
60  if (player == kChancePlayerId) {
61  if (action < kDealingActionBase) {
62  return absl::StrCat("Decide first card up position ", action);
63  } else if (action < kDealingActionBase + kNumCards) {
64  return absl::StrCat("Deal ", CardString(action - kDealingActionBase));
65  } else {
66  SpielFatalError(
67  absl::StrFormat("Non valid ID %d for chance player", action));
68  }
69  }
70 
71  if (action == kPass) {
72  return "Pass";
73  } else if (action > kPass && action < kPlayActionBase) {
74  return absl::StrCat("Bid ", action - kBiddingActionBase);
75  } else if (action >= kPlayActionBase && action <= kRocketActionBase) {
76  // For aiplane combinations, need special treatment to resolve ambiguity
77  if (action >= kAirplaneWithSoloActionBase && action < kBombActionBase) {
78  return FormatAirplaneCombHand(action);
79  }
80  return FormatSingleHand(ActionToHand(action));
81  } else {
82  SpielFatalError("Non valid action ID!");
83  }
84 }

132 std::array<std::string, kNumRanks> FormatHand(
133  int player, bool mark_voids,
134  const std::array<std::array<int, kNumRanks>, kNumPlayers>& deal) {
135  std::array<std::string, kNumRanks> cards{};
136  for (int rank = 0; rank < kNumRanks - 2; ++rank) {
137  bool is_void = true;
138  for (int i = 0; i < deal[player][rank]; ++i) {
139  cards[rank].push_back(kRankChar[rank]);
140  is_void = false;
141  }
142  if (is_void && mark_voids) absl::StrAppend(&cards[rank], "none");
143  }
144  if (deal[player][kNumRanks - 2])
145  absl::StrAppend(&cards[kNumRanks - 2], "(BWJ)");
146  else if (mark_voids)
147  absl::StrAppend(&cards[kNumRanks - 2], "none");
148 
149  if (deal[player][kNumRanks - 1])
150  absl::StrAppend(&cards[kNumRanks - 1], "(CJ)");
151  else if (mark_voids)
152  absl::StrAppend(&cards[kNumRanks - 1], "none");
153 
154  return cards;

170 std::string DouDizhuState::FormatDeal() const {
171  std::array<std::array<std::string, kNumRanks>, kNumPlayers> cards{};
172  if (IsTerminal()) {
173  // Include all cards in the terminal state to make reviewing the deal easier
174  auto deal = OriginalDeal();
175  for (int player = 0; player < kNumPlayers; ++player) {
176  cards[player] = FormatHand(player, /*mark_voids=*/false, deal);
177  }
178  } else {
179  for (int player = 0; player < kNumPlayers; ++player) {
180  cards[player] = FormatHand(player, /*mark_voids=*/false, holds_);
181  }
182  }
183  constexpr int kColumnWidth = 8;
184  std::string padding(kColumnWidth, ' ');
185  std::string rv;
186  for (int rank = 0; rank < kNumRanks; ++rank)
187  absl::StrAppend(&rv, absl::StrFormat("%-8s", cards[1][rank]), padding,
188  cards[2][rank], "\n");
189  for (int rank = 0; rank < kNumRanks; ++rank)
190  absl::StrAppend(&rv, padding, cards[0][rank], "\n");
191  return rv;
192 }

216 void DouDizhuState::WriteObservationTensor(Player player,
217  absl::Span<float> values) const {
218  SPIEL_CHECK_GE(player, 0);
219  SPIEL_CHECK_LT(player, num_players_);
220 
221  absl::c_fill(values, 0.);
222  if (phase_ == Phase::kDeal) return;
223  auto values_iterator = values.begin();
224  const int played_deck_base = (kNumRanks - 2) * (kNumSuits + 1) + 2 * 2;
225  for (int i = 0; i < kNumRanks; ++i) {
226  values_iterator[i * (kNumSuits + 1) + holds_[player][i]] = 1;
227  values_iterator[played_deck_base + i * (kNumSuits + 1) + played_deck_[i]] =
228  1;
229  }
230 
231  if (dizhu_ != kInvalidPlayer) {
232  const int from_dizhu_base = 2 * played_deck_base;
233  const int from_dizhu = (player - dizhu_ + kNumPlayers) % kNumPlayers;
234  values_iterator[from_dizhu_base + from_dizhu] = 1;
235  }
236 
237  if (first_player_ != kInvalidPlayer) {
238  const int start_player_base = 2 * played_deck_base + kNumPlayers;
239  values_iterator[start_player_base + first_player_] = 1;
240  values_iterator[start_player_base + kNumPlayers + card_rank_face_up_] = 1;
241  }
242 }

282 std::vector<Action> DouDizhuState::PlayLegalActions() const {
283  std::vector<Action> legal_actions;
284  // the leader of a trick must play./ an action and cannot pass
285  if (!new_trick_begin_) legal_actions.push_back(kPass);
286 
287  std::array<int, kNumRanks> hand = holds_[current_player_];
288  const int prev_action = CurrentTrick().WinningAction();
289  SearchForLegalActions(&legal_actions, hand, prev_action);
290 
291  absl::c_sort(legal_actions);
292  return legal_actions;

295 std::vector<std::pair<Action, double>> DouDizhuState::ChanceOutcomes() const {
296  std::vector<std::pair<Action, double>> outcomes;
297  int num_cards_remaining = 0;
298  for (int i = 0; i < kNumCards; ++i) num_cards_remaining += dealer_deck_[i];
299  outcomes.reserve(num_cards_remaining);
300 
301  if (card_face_up_position_ == -1) {
302  for (int i = 0; i < kDealingActionBase; ++i)
303  outcomes.emplace_back(i, 1.0 / static_cast<double>(kDealingActionBase));
304  } else {
305  for (int card = 0; card < kNumCards; ++card)
306  if (dealer_deck_[card])
307  outcomes.emplace_back(card + kDealingActionBase,
308  1.0 / static_cast<double>(num_cards_remaining));
309  }
310 
311  return outcomes;
312 }

327 void DouDizhuState::ApplyDealAction(int action) {
328  // First decide the face up card
329  if (card_face_up_position_ == -1) {
330  card_face_up_position_ = action;
331  return;
332  }
333 
334  const int dealing_round = static_cast<int>(history_.size()) - 1;
335  // if the current player is dealt the face up card, make it the first one to
336  // bid
337  if (dealing_round == history_[0].action) {
338  first_player_ = dealing_round % kNumPlayers;
339  card_rank_face_up_ = CardToRank(action - kDealingActionBase);
340  }
341  const int dealt_player_idx = ((history_.size() - 1) % kNumPlayers);
342  const int dealt_rank = CardToRank(action - kDealingActionBase);
343  holds_[dealt_player_idx][dealt_rank]++;
344  dealer_deck_[action - kDealingActionBase]--;
345  if (history_.size() == kNumCards - kNumCardsLeftOver) {
346  phase_ = Phase::kAuction;
347  current_player_ = first_player_;
348  SPIEL_CHECK_GE(current_player_, 0);
349  SPIEL_CHECK_LE(current_player_, num_players_);
350  for (int card = 0; card < kNumCards; ++card)
351  if (dealer_deck_[card]) {
352  cards_left_over_.push_back(CardToRank(card));
353  }
354  }
355 }

357 void DouDizhuState::ApplyBiddingAction(int action) {
358  // Track the number of consecutive passes since the last bid (if any).
359  if (action == kPass) {
360  ++num_passes_;
361  } else {
362  num_passes_ = 0;
363  }
364 
365  bool has_winner = false;
366 
367  if (action == kPass) {
368  if (num_passes_ == kNumPlayers)
369  phase_ = Phase::kGameOver;
370  else if (num_passes_ == kNumPlayers - 1 && winning_bid_ > 0)
371  has_winner = true;
372  } else {
373  dizhu_ = current_player_;
374  winning_bid_ = action - kBiddingActionBase;
375  if (winning_bid_ == kNumBids) has_winner = true;
376  }
377  if (has_winner) {
378  for (int i = 0; i < kNumCardsLeftOver; ++i)
379  holds_[dizhu_][cards_left_over_[i]]++;
380  phase_ = Phase::kPlay;
381  current_player_ = dizhu_;
382  new_trick_begin_ = true;
383  tricks_.push_back(Trick(dizhu_, kInvalidAction));
384  num_passes_ = 0;
385  } else {
386  current_player_ = (current_player_ + 1) % kNumPlayers;
387  }
388 }

390 bool DouDizhuState::AfterPlayHand(int player, int action) {
391  std::array<int, kNumRanks> used_hand = ActionToHand(action);
392  bool flag = true;
393  for (int rank = 0; rank < kNumRanks; ++rank) {
394  SPIEL_CHECK_GE(holds_[player][rank], used_hand[rank]);
395  holds_[player][rank] -= used_hand[rank];
396  flag &= !holds_[player][rank];
397  played_deck_[rank] += used_hand[rank];
398  }
399  return flag;
400 }

402 void DouDizhuState::ApplyPlayAction(int action) {
403  num_played_++;
404 
405  if (action == kPass) {
406  ++num_passes_;
407  } else {
408  num_passes_ = 0;
409  }
410 
411  if (action == kPass) {
412  if (num_passes_ == kNumPlayers - 1) {
413  current_player_ = CurrentTrick().Winner();
414  trick_played_++;
415  num_passes_ = 0;
416  tricks_.push_back(Trick());
417  new_trick_begin_ = true;
418  return;
419  }
420  } else {
421  if (action >= kBombActionBase) bombs_played_++;
422  players_hands_played[current_player_]++;
423 
424  if (new_trick_begin_) new_trick_begin_ = false;
425 
426  CurrentTrick().Play(current_player_, action);
427 
428  bool all_played = AfterPlayHand(current_player_, action);
429  if (all_played) {
430  final_winner_ = current_player_;
431  ScoreUp();
432  phase_ = Phase::kGameOver;
433  return;
434  }
435  }
436  current_player_ = (current_player_ + 1) % kNumPlayers;
437 }

449 void DouDizhuState::ScoreUp() {
450  // If no one bids, 0 for everyone
451  if (dizhu_ == kInvalidPlayer) return;
452 
453  // if none of the farmers played, or the dizhu only played once
454  // then it is spring!
455  bool is_spring = false;
456  is_spring |= (players_hands_played[dizhu_] == 1);
457  is_spring |= ((!players_hands_played[(dizhu_ + 1) % 3]) &&
458  (!players_hands_played[(dizhu_ + 2) % 3]));
459 
460  int paying = winning_bid_;
461  for (int i = 0; i < is_spring + bombs_played_; ++i) paying *= 2;
462  const int dizhu_sign = (final_winner_ == dizhu_) ? 1 : -1;
463 
464  returns_[dizhu_] = dizhu_sign * 2 * paying;
465  returns_[(dizhu_ + 1) % 3] = -dizhu_sign * paying;
466  returns_[(dizhu_ + 2) % 3] = -dizhu_sign * paying;
467 }
