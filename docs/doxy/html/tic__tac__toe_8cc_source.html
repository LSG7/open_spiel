57CellState PlayerToState(Player player) {
58 switch (player) {
59 case 0:
60 return CellState::kCross;
61 case 1:
62 return CellState::kNought;
63 default:
64 SpielFatalError(absl::StrCat("Invalid player id ", player));
65 return CellState::kEmpty;
66 }
67}

69std::string StateToString(CellState state) {
70 switch (state) {
71 case CellState::kEmpty:
72 return ".";
73 case CellState::kNought:
74 return "o";
75 case CellState::kCross:
76 return "x";
77 default:
78 SpielFatalError("Unknown state.");
79 }
80}

82bool BoardHasLine(const std::array<CellState, kNumCells>& board,
83 const Player player) {
84 CellState c = PlayerToState(player);
85 return (board[0] == c && board[1] == c && board[2] == c) ||
86 (board[3] == c && board[4] == c && board[5] == c) ||
87 (board[6] == c && board[7] == c && board[8] == c) ||
88 (board[0] == c && board[3] == c && board[6] == c) ||
89 (board[1] == c && board[4] == c && board[7] == c) ||
90 (board[2] == c && board[5] == c && board[8] == c) ||
91 (board[0] == c && board[4] == c && board[8] == c) ||
92 (board[2] == c && board[4] == c && board[6] == c);
93}

95void TicTacToeState::DoApplyAction(Action move) {
96 SPIEL_CHECK_EQ(board_[move], CellState::kEmpty);
97 board_[move] = PlayerToState(CurrentPlayer());
98 if (HasLine(current_player_)) {
99 outcome_ = current_player_;
100 }
101 current_player_ = 1 - current_player_;
102 num_moves_ += 1;
103}

105std::vector<Action> TicTacToeState::LegalActions() const {
106 if (IsTerminal()) return {};
107 // Can move in any empty cell.
108 std::vector<Action> moves;
109 for (int cell = 0; cell < kNumCells; ++cell) {
110 if (board_[cell] == CellState::kEmpty) {
111 moves.push_back(cell);
112 }
113 }
114 return moves;
115}

132std::string TicTacToeState::ToString() const {
133 std::string str;
134 for (int r = 0; r < kNumRows; ++r) {
135 for (int c = 0; c < kNumCols; ++c) {
136 absl::StrAppend(&str, StateToString(BoardAt(r, c)));
137 }
138 if (r < (kNumRows - 1)) {
139 absl::StrAppend(&str, "\n");
140 }
141 }
142 return str;
143}

171void TicTacToeState::ObservationTensor(Player player,
172 absl::Span<float> values) const {
173 SPIEL_CHECK_GE(player, 0);
174 SPIEL_CHECK_LT(player, num_players_);
175
176 // Treat `values` as a 2-d tensor.
177 TensorView<2> view(values, {kCellStates, kNumCells}, true);
178 for (int cell = 0; cell < kNumCells; ++cell) {
179 view[{static_cast<int>(board_[cell]), cell}] = 1.0;
180 }
181}

183void TicTacToeState::UndoAction(Player player, Action move) {
184 board_[move] = CellState::kEmpty;
185 current_player_ = player;
186 outcome_ = kInvalidPlayer;
187 num_moves_ -= 1;
188 history_.pop_back();
189 --move_number_;
190}
