10 /*
11  Explanation of full memory version:
12 
13  There are some constants that only need to be calculated
14  once. In fact they are the same for all instances of the
15  object. In order to save time and memory, they share a
16  single memory.
17 
18  Each 13-bit number, aggr, represents a possible set of cards
19  remaining in a suit. For example, 0x15a2 represents
20  A(1) QT(5) 97(a) 3(2).
21 
22  TTlowestRank[aggr] gives the lowest relative rank that is in
23  play in aggr. The ace is 14, the deuce is 2. A void counts as
24  rank 15 ("not even the ace"). It would go horribly wrong
25  if this rank were chosen to be 0, as might seem intuitive.
26  This is not the same as lowestRank, the lowest absolute rank.
27 
28  maskBytes[aggr][suit] is a set of 4 32-bit integers,
29  where suit is 0 ..3 (spades .. clubs). Each integer only
30  has 8 of its 32 bits set, but these 8 bits could be either
31  in the top byte (byte 0) or any of the others (bytes 1 ..3).
32  The bytes are abbreviated as B0 .. B3 below.
33 
34  int 0 int 1 int 2 int 3
35  suit 0, spades B0 R0 B0 R1 B0 R2 B0 R3
36  suit 1, hearts B1 R0 B1 R1 B1 R2 B1 R3
37  suit 2, diamonds B2 R0 B2 R1 B2 R2 B2
38  suit 3, clubs B3 R0 B3 R1 B3 R2 B3
39 
40  R0 .. R3 are explained now. The purpose of maskBytes is
41  to generate 32-bit masks for later use with actual suits.
42  As a card can be with either of 4 players, 2 bits are needed
43  to encode the position of a card. Therefore the masks also
44  need 2 bits per card, even though the 2 bits are identical.
45 
46  In the table, R0 means the top byte (8 bits = 4 cards) of
47  a holding.
48 
49  R0 AKQJ
50  R1 T987
51  R2 6543
52  R3 2
53 
54  For example, if the ace is held by North, the king by South, the
55  queen and jack by West, then the top byte for that SUIT would be
56 
57  00(North) 10(South) 11(West) 11(West)
58 
59  The MASK for that holding would be 11 11 11 11, as all four
60  cards are in play.
61 
62  If the jack were missing, because it had already been played,
63  then the suit would be 00 10 11 00 (a missing card is also
64  encoded as 00), and the mask would be 11 11 11 00.
65 
66  Later on, when we have a specific set of spades .. clubs,
67  we want to check whether those cards are already in the
68  transposition table. As far as the necessary masks is
69  concerned, this is generated by an OR (|) of the four
70  32-bit integers in a column of the table above.
71 
72  So the first column yields four bytes which are already
73  shifted in place, all corresponding to R0. The mask
74  corresponds to the AKQJ of the four suits in order.
75 
76  It's not really AKQJ, but the four highest cards still in
77  play in that suit. So missing cards are always at the end
78  of the list.
79 */

140 void TransTableL::SetConstants()
141 {
142  unsigned int topBitRank = 1;
143  TTlowestRank[0] = 15; // Void
144  unsigned winMask[8192];
145  winMask[0] = 0;
146 
147  for (unsigned ind = 1; ind < 8192; ind++)
148  {
149  if (ind >= (topBitRank + topBitRank)) /* Next top bit */
150  topBitRank <<= 1;
151 
152  // winMask is a growing list of 11's. In the end it will
153  // have 26 bits, so 13 groups of two bits. It always
154  // consists of all 11's, then all 00's.
155 
156  winMask[ind] = (winMask[ind ^ topBitRank] >> 2) | (3 << 24);
157 
158  maskBytes[ind][0][0] = (winMask[ind] << 6) & 0xff000000;
159  maskBytes[ind][0][1] = (winMask[ind] << 14) & 0xff000000;
160  maskBytes[ind][0][2] = (winMask[ind] << 22) & 0xff000000;
161  maskBytes[ind][0][3] = (winMask[ind] << 30) & 0xff000000;
162 
163  maskBytes[ind][1][0] = (winMask[ind] >> 2) & 0x00ff0000;
164  maskBytes[ind][1][1] = (winMask[ind] << 6) & 0x00ff0000;
165  maskBytes[ind][1][2] = (winMask[ind] << 14) & 0x00ff0000;
166  maskBytes[ind][1][3] = (winMask[ind] << 22) & 0x00ff0000;
167 
168  maskBytes[ind][2][0] = (winMask[ind] >> 10) & 0x0000ff00;
169  maskBytes[ind][2][1] = (winMask[ind] >> 2) & 0x0000ff00;
170  maskBytes[ind][2][2] = (winMask[ind] << 6) & 0x0000ff00;
171  maskBytes[ind][2][3] = (winMask[ind] << 14) & 0x0000ff00;
172 
173  maskBytes[ind][3][0] = (winMask[ind] >> 18) & 0x000000ff;
174  maskBytes[ind][3][1] = (winMask[ind] >> 10) & 0x000000ff;
175  maskBytes[ind][3][2] = (winMask[ind] >> 2) & 0x000000ff;
176  maskBytes[ind][3][3] = (winMask[ind] << 6) & 0x000000ff;
177 
178  TTlowestRank[ind] = TTlowestRank[ind ^ topBitRank] - 1;

428 TransTableL::winBlockType * TransTableL::GetNextCardBlock()
429 {
430  /*
431  Spaghetti code. The basic idea is that there is a pool of
432  pages. When a page runs out, we get a next pool. But we're
433  only allowed a certain maximum number, and calloc might also
434  fail before then. We have a default number of pages that
435  we don't give back voluntarily once we have acquired them,
436  but we give back anything more than that at the end of each
437  hand. If this overall mechanism fails, then we try to harvest
438  old entries scattered throughout the TT memory. If we get
439  enough for a "page", then we use that single page, and if
440  that runs out later, we try to harvest some more, starting
441  where we left off harvesting last time. If the harvesting also
442  fails, then we reset whatever TT memory we do have, and we
443  continue with that.
444  */
445 
446  if (poolp == nullptr)
447  {
448  // Have to be able to get at least one pool.
449  poolp = static_cast<poolType *>(calloc(1, sizeof(poolType)));
450  if (poolp == nullptr)
451  exit(1);
452 
453  poolp->list = static_cast<winBlockType *>
454  (malloc(BLOCKS_PER_PAGE * sizeof(winBlockType)));
455 
456  if (! poolp->list)
457  exit(1);
458 
459  poolp->next = nullptr;
460  poolp->prev = nullptr;
461  poolp->nextBlockNo = 1;
462 
463  nextBlockp = poolp->list;
464 
465  pagesCurrent++;
466 
467  return nextBlockp++;
468  }
469  else if (memState == FROM_HARVEST)
470  {
471  // Not allowed to get more memory, so reuse old one.
472  int n = harvested.nextBlockNo;
473  if (n == BLOCKS_PER_PAGE)
474  {
475  if (! TransTableL::Harvest())
476  {
477  TransTableL::ResetMemory(TT_RESET_UNKNOWN);
478  poolp->nextBlockNo++;
479  return nextBlockp++;
480  }
481  n = 0;
482  }
483 
484  harvested.nextBlockNo++;
485  return harvested.list[n];
486  }
487  else if (poolp->nextBlockNo == BLOCKS_PER_PAGE)
488  {
489  if (poolp->next)
490  {
491  // Reuse a dormant block that has not been freed.
492  poolp = poolp->next;
493  poolp->nextBlockNo = 1;
494  nextBlockp = poolp->list;
495 
496  return nextBlockp++;
497  }
498  else if (pagesCurrent == pagesMaximum)
499  {
500  // Have to try to reclaim memory.
501  if (! TransTableL::Harvest())
502  {
503  TransTableL::ResetMemory(TT_RESET_UNKNOWN);
504  poolp->nextBlockNo++;
505  return nextBlockp++;
506  }
507 
508  memState = FROM_HARVEST;
509  harvested.nextBlockNo++;
510  return harvested.list[0];
511  }
512  else
513  {
514  // Make a new pool.
515  poolType * newpoolp = static_cast<poolType *>
516  (calloc(1, sizeof(poolType)));
517 
518  if (newpoolp == nullptr)
519  {
520  // Unexpected, but try harvesting before we give up
521  // and start over.
522  if (! TransTableL::Harvest())
523  {
524  TransTableL::ResetMemory(TT_RESET_UNKNOWN);
525  poolp->nextBlockNo++;
526  return nextBlockp++;
527  }
528 
529  memState = FROM_HARVEST;
530  harvested.nextBlockNo++;
531  return harvested.list[0];
532  }
533 
534  newpoolp->list = static_cast<winBlockType *>
535  (malloc(BLOCKS_PER_PAGE * sizeof(winBlockType)));
536 
537  if (! newpoolp->list)
538  {
539  if (! TransTableL::Harvest())
540  {
541  TransTableL::ResetMemory(TT_RESET_UNKNOWN);
542  poolp->nextBlockNo++;
543  return nextBlockp++;
544  }
545 
546  memState = FROM_HARVEST;
547  harvested.nextBlockNo++;
548  return harvested.list[0];
549  }
550 
551  newpoolp->nextBlockNo = 1;
552  newpoolp->next = nullptr;
553  newpoolp->prev = poolp;
554 
555  poolp->next = newpoolp;
556  poolp = newpoolp;
557 
558  nextBlockp = newpoolp->list;
559 
560  pagesCurrent++;
561 
562  return nextBlockp++;
563  }
564  }
565 
566  poolp->nextBlockNo++;
567  return nextBlockp++;
568 }

571 bool TransTableL::Harvest()
572 {
573  distHashType * rootptr = TTroot[harvestTrick][harvestHand];
574  distHashType * ptr;
575  winBlockType * bp;
576 
577  int trick = harvestTrick;
578  int hand = harvestHand;
579  int hash, suit, hno = 0;
580 
581  while (1)
582  {
583  for (hash = 0; hash < 256; hash++)
584  {
585  ptr = &rootptr[hash];
586  for (suit = ptr->nextNo - 1; suit >= 0; suit--)
587  {
588  bp = ptr->list[suit].posBlock;
589  if (timestamp - bp->timestampRead > HARVEST_AGE)
590  {
591  bp->nextMatchNo = 0;
592  bp->nextWriteNo = 0;
593  bp->timestampRead = timestamp;
594  harvested.list[hno] = bp;
595 
596  // Swap the last element down.
597  if (suit != ptr->nextNo - 1)
598  ptr->list[suit] = ptr->list[ ptr->nextNo - 1 ];
599 
600  ptr->nextNo--;
601  ptr->nextWriteNo = ptr->nextNo;
602 
603  if (++hno == BLOCKS_PER_PAGE)
604  {
605  if (++harvestHand >= DDS_HANDS)
606  {
607  // Skip rest of this [trick][hand] for simplicity.
608  harvestHand = 0;
609  if (--harvestTrick < 0)
610  harvestTrick = FIRST_HARVEST_TRICK;
611  }
612 
613  harvested.nextBlockNo = 0;
614  pageStats.numHarvests++;
615  return true;
616  }
617  }
618  }
619  }
620 
621  if (++harvestHand >= DDS_HANDS)
622  {
623  harvestHand = 0;
624  if (--harvestTrick < 0)
625  harvestTrick = FIRST_HARVEST_TRICK;
626  }
627 
628  if (harvestTrick == trick && harvestHand == hand)
629  return false;
630 
631  rootptr = TTroot[harvestTrick][harvestHand];
632  }
633 }

636 int TransTableL::hash8(const int handDist[]) const
637 {
638  /*
639  handDist is an array of hand distributions, North .. West.
640  Each entry is a 12-bit number with 3 groups of 4 bits.
641  Each group is the binary representation of the number of
642  cards held in that suit. The suits are in order spades,
643  hearts, diamonds. Clubs can be neglected, as the total
644  number of cards in a hand is given by the trick number.
645 
646  For example, if handDist[1] equals 0x0433, then East holds
647  4 spades, 3 hearts, 3 diamonds and the rest in clubs.
648  If this is after the second trick, there are 11 cards, so
649  East must hold 1 club.
650 
651  The hash function turns all 4 hand distributions into a
652  single 8-bit number. The numbers should be spread as
653  evenly as possible across the 256 possibilities. I've not
654  done extensive research into finding the best hash function,
655  but this one seems OK. It uses a small prime, 5, and its
656  powers. The shift at the end is in order to get some use
657  out of the bits above the first 8 ones.
658  */
659 
660  int h =
661  (handDist[0] ^
662  ((handDist[1] * 5) ) ^
663  ((handDist[2] * 25) ) ^
664  ((handDist[3] * 125) ) );
665 
666  return (h ^ (h >> 5)) & 0xff;
667 }

729  }
730  }
731 
732  empty = true;
733  int m;
734 
735  if (n == DISTS_PER_ENTRY)
736  {
737  // No room for new exact suits at this hash position.
738  // Have to reuse an existing posBlock.
739  if (dp->nextWriteNo == DISTS_PER_ENTRY)
740  {
741  m = 0;
742  dp->nextWriteNo = 1;
743  }
744  else
745  m = dp->nextWriteNo++;
746  }
747  else
748  {
749  // Didn't find an exact match, but there is still room.
750  // The following looks a bit odd because it is possible that
751  // GetNextCardBlock wipes out the whole memory, so we
752  // have to use the up-to-date location, not m from above.
753 
754  winBlockType * bp = GetNextCardBlock();
755  m = dp->nextWriteNo++;
756  dp->list[m].posBlock = bp;
757  dp->list[m].posBlock->timestampRead = timestamp;
758  dp->nextNo++;
759  }
760 
761  // As long as the secondary Lookup loop in ABsearch exists,
762  // it will cause spurious extra blocks to be created here
763  // which are not useful, because nothing is ever Add'ed.
764  // This is not a memory leak, as the memory is properly freed,
765  // but it is also a small waste of about 0.5%. I don't mind.
766 
767  dp->list[m].key = key;
768  dp->list[m].posBlock->nextMatchNo = 0;
769  dp->list[m].posBlock->nextWriteNo = 0;
770 
771  return dp->list[m].posBlock;
772 }

775 nodeCardsType * TransTableL::LookupCards(
776  const winMatchType& search,
777  winBlockType * bp,
778  const int limit,
779  bool& lowerFlag)
780 {
781  const int n = bp->nextWriteNo - 1;
782  winMatchType * wp = &bp->list[n];
783 
784  // It may be a bit silly to duplicate the code like this.
785  // It could be combined to one loop with a slight overhead.
786 
787  for (int i = n; i >= 0; i--, wp--)
788  {
789  if ((wp->topSet1 ^ search.topSet1) & wp->topMask1)
790  continue;
791 
792  if (wp->lastMaskNo != 1)
793  {
794  if ((wp->topSet2 ^ search.topSet2) & wp->topMask2)
795  continue;
796 
797  if (wp->lastMaskNo != 2)
798  {
799  if ((wp->topSet3 ^ search.topSet3) & wp->topMask3)
800  continue;
801  }
802  }
803 
804  // Check bounds.
805  nodeCardsType * nodep = &wp->first;
806  if (nodep->lbound > limit)
807  {
808  bp->timestampRead = ++timestamp;
809  lowerFlag = true;
810  return nodep;
811  }
812  else if (nodep->ubound <= limit)
813  {
814  bp->timestampRead = ++timestamp;
815  lowerFlag = false;
816  return nodep;
817  }
818  }
819 
820  const int n2 = bp->nextMatchNo - 1;
821  wp = &bp->list[n2];
822 
823  for (int i = n2; i > n; i--, wp--)
824  {
825  if ((wp->topSet1 ^ search.topSet1) & wp->topMask1)
826  continue;
827 
828  if (wp->lastMaskNo != 1)
829  {
830  if ((wp->topSet2 ^ search.topSet2) & wp->topMask2)
831  continue;
832 
833  if (wp->lastMaskNo != 2)
834  {
835  if ((wp->topSet3 ^ search.topSet3) & wp->topMask3)
836  continue;
837  }
838  }
839 
840  nodeCardsType * nodep = &wp->first;
841  if (nodep->lbound > limit)
842  {
843  lowerFlag = true;
844  bp->timestampRead = ++timestamp;
845  return nodep;
846  }
847  else if (nodep->ubound <= limit)
848  {
849  lowerFlag = false;
850  bp->timestampRead = ++timestamp;
851  return nodep;
852  }
853  }
854 
855  return nullptr;
856 }

859 void TransTableL::CreateOrUpdate(
860  winBlockType * bp,
861  const winMatchType& search,
862  const bool flag)
863 {
864  // Either updates an existing SOP or creates a new one.
865  // A new one is created at the end of the bp list if this
866  // is not already full, or the oldest one in the list is
867  // overwritten.
868 
869  winMatchType * wp = bp->list;
870  int n = bp->nextMatchNo;
871 
872  for (int i = 0; i < n; i++, wp++)
873  {
874  if (wp->xorSet != search.xorSet ) continue;
875  if (wp->maskIndex != search.maskIndex) continue;
876  if (wp->topSet1 != search.topSet1 ) continue;
877  if (wp->topSet2 != search.topSet2 ) continue;
878  if (wp->topSet3 != search.topSet3 ) continue;
879 
880  nodeCardsType& node = wp->first;
881  if (search.first.lbound > node.lbound)
882  node.lbound = search.first.lbound;
883  if (search.first.ubound < node.ubound)
884  node.ubound = search.first.ubound;
885 
886  node.bestMoveSuit = search.first.bestMoveSuit;
887  node.bestMoveRank = search.first.bestMoveRank;
888  return;
889  }
890 
891  if (n == BLOCKS_PER_ENTRY)
892  {
893  if (bp->nextWriteNo >= BLOCKS_PER_ENTRY)
894  bp->nextWriteNo = 0;
895  }
896  else
897  bp->nextMatchNo++;
898 
899 
900  wp = &bp->list[ bp->nextWriteNo++ ];
901  *wp = search;
902 
903  if (!flag)
904  {
905  wp->first.bestMoveSuit = 0;
906  wp->first.bestMoveRank = 0;
907  }
908 }

911 void TransTableL::Add(
912  const int tricks,
913  const int hand,
914  const unsigned short aggrTarget[],
915  const unsigned short ourWinRanks[],
916  const nodeCardsType& first,
917  const bool flag)
918 {
919  if (lastBlockSeen[tricks][hand] == nullptr)
920  {
921  // We have recently reset the entire memory, and we were
922  // in the middle of a recursion. So we'll just have to
923  // drop this entry that we were supposed to be adding.
924  return;
925  }
926 
927  unsigned * ab[DDS_SUITS];
928  unsigned * mb[DDS_SUITS];
929  char low[DDS_SUITS];
930  unsigned short int ag;
931  int w;
932  winMatchType TTentry;
933 
934  // Inefficient, as it also copies leastWin.
935  // In fact I'm not quite happy with the treatment of
936  // leastWin in general.
937 
938  TTentry.first = first;
939 
940  TTentry.xorSet = 0;
941 
942  for (int ss = 0; ss < DDS_SUITS; ss++)
943  {
944  w = static_cast<int>(ourWinRanks[ss]);
945  if (w == 0)
946  {
947  ab[ss] = aggr[0].aggrBytes[ss];
948  mb[ss] = maskBytes[0][ss];
949  low[ss] = 15;
950  TTentry.first.leastWin[ss] = 0;
951  }
952  else
953  {
954  w = w & (-w); /* Only lowest win */
955  ag = static_cast<unsigned short>(aggrTarget[ss] & (-w));
956 
957  ab[ss] = aggr[ag].aggrBytes[ss];
958  mb[ss] = maskBytes[ag][ss];
959  low[ss] = static_cast<char>(TTlowestRank[ag]);
960 
961  TTentry.first.leastWin[ss] = 15 - low[ss];
962  TTentry.xorSet ^= aggr[ag].aggrRanks[ss];
963  }
964  }
965 
966  // It's a bit annoying that we may be regenerating these.
967  // But winRanks can cause them to change after lookup().
968 
969  TTentry.topSet1 = ab[0][0] | ab[1][0] | ab[2][0] | ab[3][0];
970  TTentry.topSet2 = ab[0][1] | ab[1][1] | ab[2][1] | ab[3][1];
971  TTentry.topSet3 = ab[0][2] | ab[1][2] | ab[2][2] | ab[3][2];
972  TTentry.topSet4 = ab[0][3] | ab[1][3] | ab[2][3] | ab[3][3];
973 
974  TTentry.topMask1 = mb[0][0] | mb[1][0] | mb[2][0] | mb[3][0];
975  TTentry.topMask2 = mb[0][1] | mb[1][1] | mb[2][1] | mb[3][1];
976  TTentry.topMask3 = mb[0][2] | mb[1][2] | mb[2][2] | mb[3][2];
977  TTentry.topMask4 = mb[0][3] | mb[1][3] | mb[2][3] | mb[3][3];
978 
979  TTentry.maskIndex =
980  (low[0] << 12) | (low[1] << 8) | (low[2] << 4) | low[3];
981 
982  if (TTentry.topMask2 == 0)
983  TTentry.lastMaskNo = 1;
984  else if (TTentry.topMask3 == 0)
985  TTentry.lastMaskNo = 2;
986  else if (TTentry.topMask4 == 0)
987  TTentry.lastMaskNo = 3;
988  else
989  TTentry.lastMaskNo = 4;
990 
991  TransTableL::CreateOrUpdate(lastBlockSeen[tricks][hand],
992  TTentry, flag);
993 }
