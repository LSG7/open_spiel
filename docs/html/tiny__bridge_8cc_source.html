93 // Requires card0 > card1
94 int CardsToChanceOutcome(int card0, int card1) {
95  return (card0 * (card0 - 1)) / 2 + card1;
96 }

98 // Returns first > second
99 std::pair<int, int> ChanceOutcomeToCards(int outcome) {
100  int card0 = 1;
101  while (CardsToChanceOutcome(card0 + 1, 0) <= outcome) ++card0;
102  return {card0, outcome - CardsToChanceOutcome(card0, 0)};
103 }

125 // Computes the abstraction lookup.
126 std::vector<int> ConcreteToAbstract() {
127  std::vector<int> concrete_to_abstract(kNumPrivates, -1);
128  for (int c = 0; c < kNumPrivates; ++c) {
129  auto hand = HandString(c);
130  for (int ah = 0; ah < kNumAbstractHands; ++ah) {
131  if (absl::StrContains(kAbstraction[ah], hand)) {
132  concrete_to_abstract[c] = ah;
133  break;
134  }
135  }
136  if (concrete_to_abstract[c] == -1) {
137  SpielFatalError(
138  absl::StrCat("Abstraction not found for concrete hand '", hand, "'"));
139  }
140  }
141  return concrete_to_abstract;
142 }

236 // Score a played-out hand.
237 int Score(int contract, int tricks, bool doubled, bool redoubled, int trumps) {
238  // -20 per undertrick
239  // +10 for 1H/S/NT (+10 extra if overtrick)
240  // +30 for 2H/S
241  // +35 for 2NT
242  const int contract_tricks = 1 + (contract - 1) / 3;
243  const int contract_result = tricks - contract_tricks;
244  const int double_factor = (1 + doubled) * (1 + redoubled);
245  if (contract_result < 0) return 20 * double_factor * contract_result;
246  int score = tricks * 10;
247  if (contract_tricks == 2) score += 10;
248  if (contract_tricks == 2 && trumps == 2) score += 5;
249  return score * double_factor;
250 }

323 TinyBridgeAuctionState::AuctionState TinyBridgeAuctionState::AnalyzeAuction()
324  const {
325  AuctionState rv;
326  rv.last_bid = Call::kPass;
327  rv.last_bidder = kInvalidSeat;
328  rv.doubler = kInvalidSeat;
329  rv.redoubler = kInvalidSeat;
330  for (int i = num_players_; i < actions_.size(); ++i) {
331  if (actions_[i] == Call::kDouble) {
332  rv.doubler = PlayerToSeat(i % num_players_);
333  } else if (actions_[i] == Call::kRedouble) {
334  rv.redoubler = PlayerToSeat(i % num_players_);
335  } else if (actions_[i] != Call::kPass) {
336  rv.last_bid = actions_[i];
337  rv.last_bidder = PlayerToSeat(i % num_players_);
338  rv.doubler = kInvalidSeat;
339  rv.redoubler = kInvalidSeat;
340  }
341  }
342  return rv;
343 }

345 int Score_p0(std::array<Seat, kDeckSize> holder,
346  const TinyBridgeAuctionState::AuctionState& state) {
347  if (state.last_bid == Call::kPass) return 0;
348  std::shared_ptr<Game> game(new TinyBridgePlayGame({}));
349  int trumps = (state.last_bid - 1) % 3;
350  Seat leader = Seat((state.last_bidder + 3) % 4);
351  Seat decl = Seat(state.last_bidder % 2);
352  TinyBridgePlayState play{game, trumps, leader, holder};
353  const double tricks =
354  algorithms::AlphaBetaSearch(*game, &play, nullptr, -1, decl).first;
355  SPIEL_CHECK_GE(tricks, 0);
356  SPIEL_CHECK_LE(tricks, kNumTricks);
357  const int declarer_score =
358  Score(state.last_bid, tricks, state.doubler != kInvalidSeat,
359  state.redoubler != kInvalidSeat, trumps);
360  return (decl == 0) ? declarer_score : -declarer_score;
361 }

363 // Score indexed by [WestHand][EastHand][Contract][LastBidder]
364 using ScoringTable = std::array<
365  std::array<std::array<std::array<double, 2>, kNumActions2p>, kNumPrivates>,
366  kNumPrivates>;
367 
368 // Calculates a single score.
369 double Score_2p_(Action hand0, Action hand1,
370  const TinyBridgeAuctionState::AuctionState& state) {
371  if (state.last_bid == kPass) return 0;
372  const double freq = 1. / 6;
373  double utility_p0 = 0;
374  std::array<Seat, kDeckSize> holders_2p;
375  std::fill(holders_2p.begin(), holders_2p.end(), kInvalidSeat);
376  const auto cards0 = ChanceOutcomeToCards(hand0);
377  holders_2p[cards0.first] = kWest;
378  holders_2p[cards0.second] = kWest;
379  const auto cards1 = ChanceOutcomeToCards(hand1);
380  holders_2p[cards1.first] = kEast;
381  holders_2p[cards1.second] = kEast;
382  std::array<Seat, kDeckSize> holders_4p;
383  for (int n0 = 0; n0 < 3; ++n0) {
384  for (int n1 = n0 + 1; n1 < 4; ++n1) {
385  int n = 0;
386  for (int i = 0; i < kDeckSize; ++i) {
387  if (holders_2p[i] == kInvalidSeat) {
388  holders_4p[i] = (n == n0 || n == n1) ? kNorth : kSouth;
389  ++n;
390  } else {
391  holders_4p[i] = holders_2p[i];
392  }
393  }
394  utility_p0 += Score_p0(holders_4p, state) * freq;
395  }
396  }
397  return utility_p0;
398 }

400 // Returns a cache of scores.
401 ScoringTable MakeScores() {
402  ScoringTable scores;
403  for (int hand0 = 0; hand0 < kNumPrivates; ++hand0) {
404  for (int hand1 = 0; hand1 < kNumPrivates; ++hand1) {
405  if (!IsConsistent(hand0, hand1)) continue;
406  for (int contract = k1H; contract < kNumActions2p; ++contract) {
407  for (Seat last_bidder : {kWest, kEast}) {
408  scores[hand0][hand1][contract][last_bidder / 2] =
409  Score_2p_(hand0, hand1,
410  {contract, last_bidder, kInvalidSeat, kInvalidSeat});
411  }
412  }
413  }
414  }
415  return scores;
416 }

418 double Score_2p(Action hand0, Action hand1,

420  if (state.last_bid == kPass) return 0;

439 void TinyBridgeAuctionState::DoApplyAction(Action action) {
440  actions_.push_back(action);
441  if (num_players_ == 2) {
442  if (actions_.size() >= 2 * num_players_ && actions_.back() == Call::kPass) {
443  is_terminal_ = true;
444  utility_p0 = Score_2p(actions_[0], actions_[1], AnalyzeAuction());
445  }
446  } else {
447  if (actions_.size() >= 2 * num_players_ &&
448  actions_[actions_.size() - 1] == Call::kPass &&
449  actions_[actions_.size() - 2] == Call::kPass &&
450  actions_[actions_.size() - 3] == Call::kPass) {
451  is_terminal_ = true;
452  utility_p0 = Score_p0(CardHolders(), AnalyzeAuction());
453  }
454  }
455 }

457 std::vector<Action> TinyBridgeAuctionState::LegalActions() const {
458  std::vector<Action> actions;
459  if (IsChanceNode()) {
460  return LegalChanceOutcomes();
461  } else if (IsTerminal()) {
462  return {};
463  } else {
464  auto state = AnalyzeAuction();
465  actions.push_back(Call::kPass);
466  for (int bid = state.last_bid + 1; bid <= Call::k2NT; ++bid) {
467  actions.push_back(bid);
468  }
469  if (num_players_ == 4 && state.last_bidder != kInvalidSeat) {
470  if (state.last_bidder % 2 != CurrentPlayer() % 2) {
471  if (state.doubler == kInvalidSeat) actions.push_back(Call::kDouble);
472  } else {
473  if (state.doubler != kInvalidSeat && state.redoubler == kInvalidSeat)
474  actions.push_back(Call::kRedouble);
475  }
476  }
477  }
478  return actions;
479 }

481 std::vector<std::pair<Action, double>> TinyBridgeAuctionState::ChanceOutcomes()
482  const {
483  std::vector<Action> actions;
484  auto holder = CardHolders();
485  for (int card1 = 0; card1 < kDeckSize; ++card1) {
486  if (holder[card1] != kInvalidSeat) continue;
487  for (int card2 = card1 + 1; card2 < kDeckSize; ++card2) {
488  if (holder[card2] != kInvalidSeat) continue;
489  actions.push_back(CardsToChanceOutcome(card2, card1));
490  }
491  }
492  const int num_actions = actions.size();
493  std::vector<std::pair<Action, double>> outcomes;
494  outcomes.reserve(num_actions);
495  for (auto action : actions) {
496  outcomes.emplace_back(action, 1.0 / num_actions);
497  }
498  return outcomes;
499 }

563 // Observation string is the player's cards plus the most recent bid,
564 // plus any doubles or redoubles. E.g. "HJSA 2NT:Us Dbl:RH RDbl:Pd"
565 // This is an observation for a player who holds HJ and SA.
566 // and redoubled by West.
567 // The most recent bid is 2NT by this player, which has been doubled by their
568 // right-hand-opponent and redoubled by their partner. So the most recent few
569 // calls must be: 2NT-Pass-Pass-Dbl-Pass-Pass-RDbl-Pass.
570 std::string TinyBridgeAuctionState::ObservationString(Player player) const {
571  SPIEL_CHECK_GE(player, 0);
572  SPIEL_CHECK_LT(player, num_players_);
573 
574  std::string observation = PlayerHandString(player, is_abstracted_);
575  if (HasAuctionStarted()) {
576  auto state = AnalyzeAuction();
577  if (state.last_bid != Call::kPass) {
578  absl::StrAppend(
579  &observation, " ", ActionToString(state.last_bidder, state.last_bid),
580  ":", RelativeSeatString(state.last_bidder, PlayerToSeat(player)));
581  }
582  if (state.doubler != kInvalidSeat)
583  absl::StrAppend(&observation, " ", "Dbl:",
584  RelativeSeatString(state.doubler, PlayerToSeat(player)));
585  if (state.redoubler != kInvalidSeat)
586  absl::StrAppend(
587  &observation, " ",
588  "RDbl:", RelativeSeatString(state.redoubler, PlayerToSeat(player)));
589  }
590  return observation;
591 }

593 // Information state vector consists of:
594 // kNumCards bits showing which cards the observing player holds
595 // For 2p, kNumActions2p*2 bits showing which actions have been taken:
596 // For each action, the bits are [1, 0] if we took the action,
597 // [0, 1] if our partner took the action, and otherwise [0, 0].
598 // For 4p, (kNumBids*3+1)*num_players bits showing which actions have been
599 // taken:
600 // For each player, 1 bit showing if they passed before the first bid
601 // For each bid, 4 bits showing who made it, 4 bits showing who doubled it,
602 // and 4 bits showing who redoubled it.
603 // Each set of 4 bits is relative the the current player.
604 void TinyBridgeAuctionState::InformationStateTensor(
605  Player player, absl::Span<float> values) const {
606  SPIEL_CHECK_GE(player, 0);
607  SPIEL_CHECK_LT(player, num_players_);
608 
609  const int hand_size = is_abstracted_ ? kNumAbstractHands : kDeckSize;
610  const int auction_size = (num_players_ == 2)
611  ? kNumActions2p * num_players_
612  : num_players_ + kNumBids * num_players_ * 3;
613  std::fill(values.begin(), values.end(), 0);
614  SPIEL_CHECK_EQ(values.size(), hand_size + auction_size);
615  if (IsDealt(player)) {
616  if (is_abstracted_) {
617  const int abstraction = ChanceOutcomeToHandAbstraction(actions_[player]);
618  values.at(abstraction) = 1;
619  } else {
620  const auto cards = ChanceOutcomeToCards(actions_[player]);
621  values.at(cards.first) = 1;
622  values.at(cards.second) = 1;
623  }
624  }
625  if (num_players_ == 2) {
626  for (int i = num_players_; i < actions_.size(); ++i) {
627  values.at(hand_size + actions_[i] * 2 + (i - player) % num_players_) = 1;
628  }
629  } else {
630  auto last_bid = Call::kPass;
631  auto observer = PlayerToSeat(player);
632  for (int i = num_players_; i < actions_.size(); ++i) {
633  int bidder = RelativeSeatIndex(Seat(i % num_players_), observer);
634  if (actions_[i] == Call::kPass) {
635  if (last_bid == Call::kPass) {
636  values.at(hand_size + bidder) = 1;
637  }
638  } else if (actions_[i] == Call::kDouble) {
639  values.at(hand_size + num_players_ +
640  (last_bid - 1) * (3 * num_players_) + bidder) = 1;
641  } else if (actions_[i] == Call::kRedouble) {
642  values.at(hand_size + num_players_ +
643  (last_bid - 1) * (3 * num_players_) + num_players_ + bidder) =
644  1;
645  } else {
646  last_bid = Call(actions_[i]);
647  values.at(hand_size + num_players_ +
648  (last_bid - 1) * (3 * num_players_) + num_players_ * 2 +
649  bidder) = 1;
650  }
651  }
652  }
653 }
654 
655 // Information state vector consists of:
656 // kNumCards bits showing which cards the observing player holds
657 // For 2p:
658 // kNumActions2p bits showing the most recent action (one-hot)
659 // For 4p:
660 // kNumBids bits showing the most recent bid (one-hot)
661 // 4 bits showing who bid it (relative to the observing player)
662 // 4 bits showing who doubled it (relative to the observing player)
663 // 4 bits showing who redoubled it (relative to the observing player)
664 // 4 bits for the dealer
665 void TinyBridgeAuctionState::ObservationTensor(Player player,
666  absl::Span<float> values) const {
667  SPIEL_CHECK_GE(player, 0);
668  SPIEL_CHECK_LT(player, num_players_);
669 
670  const int hand_size = is_abstracted_ ? kNumAbstractHands : kDeckSize;
671  const int auction_size =
672  num_players_ == 2 ? kNumActions2p : kNumBids + 4 * num_players_;
673  std::fill(values.begin(), values.end(), 0);
674  SPIEL_CHECK_EQ(values.size(), hand_size + auction_size);
675  if (IsDealt(player)) {
676  if (is_abstracted_) {
677  const int abstraction = ChanceOutcomeToHandAbstraction(actions_[player]);
678  values.at(abstraction) = 1;
679  } else {
680  const auto cards = ChanceOutcomeToCards(actions_[player]);
681  values.at(cards.first) = 1;
682  values.at(cards.second) = 1;
683  }
684  }
685  if (num_players_ == 2) {
686  if (HasAuctionStarted()) {
687  values.at(hand_size + actions_.back()) = 1;
688  }
689  } else {
690  auto state = AnalyzeAuction();
691  auto seat = PlayerToSeat(player);
692  if (state.last_bidder != kInvalidSeat)
693  values.at(hand_size + RelativeSeatIndex(state.last_bidder, seat)) = 1;
694  if (state.doubler != kInvalidSeat)
695  values.at(hand_size + kNumSeats +
696  RelativeSeatIndex(state.doubler, seat)) = 1;
697  if (state.redoubler != kInvalidSeat)
698  values.at(hand_size + kNumSeats * 2 +
699  RelativeSeatIndex(state.redoubler, seat)) = 1;
700  values.at(hand_size + kNumSeats * 3 +
701  RelativeSeatIndex(Seat::kWest, seat)) = 1;
702  if (state.last_bidder != kInvalidSeat)
703  values.at(hand_size + kNumSeats * 4 + state.last_bid - 1) = 1;
704  }
705 }
