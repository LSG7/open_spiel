109  def euc_descent_step(self, params, grads, t):
110  """Projected gradient descent on exploitability using Euclidean projection.
111 
112  Args:
113  params: tuple of variables to be updated (dist, y)
114  grads: tuple of variable gradients (grad_dist, grad_y)
115  t: int, solver iteration
116  Returns:
117  new_params: tuple of update params (new_dist, new_y)
118  """
119  lr_dist, lr_y = self.lrslrs
120  new_params = [params[0] - lr_dist * grads[0]]
121  lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
122  new_params += [params[1] - lr_y * grads[1]]
123  new_params = euc_project(*new_params)
124  return new_params

126  def mirror_descent_step(self, params, grads, t):
127  """Entropic mirror descent on exploitability.
128 
129  Args:
130  params: tuple of variables to be updated (dist, y)
131  grads: tuple of variable gradients (grad_dist, grad_y)
132  t: int, solver iteration
133  Returns:
134  new_params: tuple of update params (new_dist, new_y)
135  """
136  lr_dist, lr_y = self.lrslrs

138  lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
139  new_params += [params[1] - lr_y * grads[1]]

144 def gradients(dist, y, payoff_matrices, num_players, temperature=0.,
145  proj_grad=True):
146  """Computes exploitablity gradient and aux variable gradients.
147 
148  Args:
149  dist: 1-d np.array, current estimate of nash distribution
150  y: 1-d np.array (same shape as dist), current estimate of payoff gradient
151  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
152  num_players: int, number of players, in case payoff_matrices is abbreviated
153  temperature: non-negative float, default 0.
154  proj_grad: bool, if True, projects dist gradient onto simplex
155  Returns:
156  gradient of exploitability w.r.t. (dist, y) as tuple
157  unregularized exploitability (stochastic estimate)
158  tsallis regularized exploitability (stochastic estimate)
159  """
160  nabla = payoff_matrices[0].dot(dist)
161  y = nabla
162  if temperature > 0:
163  br = special.softmax(y / temperature)
164  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
165  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
166  else:
167  power = np.inf
168  s = np.linalg.norm(y, ord=power)
169  br = np.zeros_like(dist)
170  maxima = (y == s)
171  br[maxima] = 1. / maxima.sum()
172  br_mat = np.zeros((br.size, br.size))
173  br_policy_gradient = np.zeros_like(br)
174 
175  unreg_exp = np.max(y) - y.dot(dist)
176  entr_br = temperature * special.entr(br).sum()
177  entr_dist = temperature * special.entr(dist).sum()
178  reg_exp = y.dot(br - dist) + entr_br - entr_dist
179 
180  policy_gradient = nabla
181  if temperature > 0:
182  policy_gradient -= temperature * (np.log(dist) + 1)
183  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
184 
185  other_player_fx_translated = payoff_matrices[1].dot(other_player_fx)
186  grad_dist = -policy_gradient + (num_players - 1) * other_player_fx_translated
187  if proj_grad:
188  grad_dist = simplex.project_grad(grad_dist)
189  grad_y = y - nabla
190 
191  return (grad_dist, grad_y), unreg_exp, reg_exp

194 def cheap_gradients(random, dist, y, payoff_matrices, num_players,
195  temperature=0., proj_grad=True):

197

200  update so can be re-implemented in O(d) if needed.
201 
202  Args:
203  random: random number generator, np.random.RandomState(seed)
204  dist: 1-d np.array, current estimate of nash distribution
205  y: 1-d np.array (same shape as dist), current estimate of payoff gradient
206  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
207  num_players: int, number of players, in case payoff_matrices is abbreviated
208  temperature: non-negative float, default 0.
209  proj_grad: bool, if True, projects dist gradient onto simplex

211  gradient of exploitability w.r.t. (dist, y) as tuple
212  unregularized exploitability (stochastic estimate)
213  tsallis regularized exploitability (stochastic estimate)
214  """
215  action_1 = random.choice(dist.size, p=dist)
216  nabla = payoff_matrices[0][:, action_1]
217  if temperature > 0:
218  br = special.softmax(y / temperature)
219  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
220  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
221  else:
222  power = np.inf
223  s = np.linalg.norm(y, ord=power)
224  br = np.zeros_like(dist)
225  maxima = (y == s)

228  br_policy_gradient = np.zeros_like(br)
229

232  entr_dist = temperature * special.entr(dist).sum()
233  reg_exp = y.dot(br - dist) + entr_br - entr_dist
234 
235  policy_gradient = nabla

237  policy_gradient -= temperature * (np.log(dist) + 1)

241  other_player_fx = dist.size * other_player_fx[action_u]
242  other_player_fx_translated = payoff_matrices[1, :, action_u] * other_player_fx
243  grad_dist = -policy_gradient + (num_players - 1) * other_player_fx_translated
244  if proj_grad:

251 def cheap_gradients_vr(random, dist, y, payoff_matrices, num_players, pm_vr,
252  temperature=0., proj_grad=True, version=0):
253  """Computes exploitablity gradient and aux variable gradients with samples.
254 
255  This implementation takes payoff_matrices as input so technically uses O(d^2)
256  compute but only a single column of payoff_matrices is used to perform the
257  update so can be re-implemented in O(d) if needed.
258 
259  Args:
260  random: random number generator, np.random.RandomState(seed)
261  dist: 1-d np.array, current estimate of nash distribution
262  y: 1-d np.array (same shape as dist), current estimate of payoff gradient
263  payoff_matrices: (>=2 x A x A) np.array, payoffs for each joint action
264  num_players: int, number of players, in case payoff_matrices is abbreviated
265  pm_vr: approximate payoff_matrix for variance reduction
266  temperature: non-negative float, default 0.
267  proj_grad: bool, if True, projects dist gradient onto simplex
268  version: int, default 0, two options for variance reduction
269  Returns:
270  gradient of exploitability w.r.t. (dist, y) as tuple
271  unregularized exploitability (stochastic estimate)
272  tsallis regularized exploitability (stochastic estimate)
273  """
274  if pm_vr is None:
275  raise ValueError("pm_vr must be np.array of shape (num_strats, num_strats)")
276  if (not isinstance(version, int)) or (version < 0) or (version > 1):
277  raise ValueError("version must be non-negative int < 2")
278 
279  action_1 = random.choice(dist.size, p=dist)
280  nabla = payoff_matrices[0][:, action_1]
281  if temperature > 0:
282  br = special.softmax(y / temperature)
283  br_mat = (np.diag(br) - np.outer(br, br)) / temperature
284  br_policy_gradient = nabla - temperature * (np.log(br) + 1)
285  else:
286  power = np.inf
287  s = np.linalg.norm(y, ord=power)
288  br = np.zeros_like(dist)
289  maxima = (y == s)
290  br[maxima] = 1. / maxima.sum()
291  br_mat = np.zeros((br.size, br.size))
292  br_policy_gradient = np.zeros_like(br)
293 
294  unreg_exp = np.max(y) - y.dot(dist)
295  entr_br = temperature * special.entr(br).sum()
296  entr_dist = temperature * special.entr(dist).sum()
297  reg_exp = y.dot(br - dist) + entr_br - entr_dist
298 
299  policy_gradient = nabla
300  if temperature > 0:
301  policy_gradient -= temperature * (np.log(dist) + 1)
302  other_player_fx = (br - dist) + br_mat.dot(br_policy_gradient)
303 
304  if version == 0:
305  other_player_fx_translated = pm_vr.dot(other_player_fx)
306  action_u = random.choice(dist.size) # uniform, ~importance sampling
307  other_player_fx = other_player_fx[action_u]
308  pm_mod = dist.size * (payoff_matrices[1, :, action_u] - pm_vr[:, action_u])
309  other_player_fx_translated += pm_mod * other_player_fx
310  elif version == 1:
311  other_player_fx_translated = np.sum(pm_vr, axis=1)
312  action_u = random.choice(dist.size) # uniform, ~importance sampling
313  other_player_fx = other_player_fx[action_u]
314  pm_mod = dist.size * payoff_matrices[1, :, action_u]
315  r = dist.size * pm_vr[:, action_u]
316  other_player_fx_translated += pm_mod * other_player_fx - r
317 
318  grad_dist = -policy_gradient + (num_players - 1) * other_player_fx_translated
319  if proj_grad:
320  grad_dist = simplex.project_grad(grad_dist)
321  grad_y = y - nabla
322 
323  if version == 0:
324  pm_vr[:, action_u] = payoff_matrices[1, :, action_u]
325  elif version == 1:
326  pm_vr[:, action_u] = payoff_matrices[1, :, action_u] * other_player_fx
327 
328  return (grad_dist, grad_y), pm_vr, unreg_exp, reg_exp
