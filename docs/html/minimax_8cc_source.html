27 // An alpha-beta algorithm.
28 //
29 // Implements a min-max algorithm with alpha-beta pruning.
30 // See for example https://en.wikipedia.org/wiki/Alpha-beta_pruning
31 //
32 // Arguments:
33 // state: The current state node of the game.
34 // depth: The maximum depth for the min/max search.
35 // alpha: best value that the MAX player can guarantee (if the value is <=
36 // alpha, the MAX player will avoid it).
37 // beta: the best value that the MIN currently can guarantee (if the value is
38 // >= than beta, the MIN player will avoid it).
39 // value_function: An optional function mapping a Spiel `State` to a
40 // numerical value, to be used as the value for a node when we reach
41 // `depth_limit` and the node is not terminal.
42 // maximizing_player_id: The id of the MAX player. The other player is assumed
43 // to be MIN.
44 //
45 // Returns:
46 // The optimal value of the sub-game starting in state (given alpha/beta).
47 double _alpha_beta(State* state, int depth, double alpha, double beta,
48  std::function<double(const State&)> value_function,
49  Player maximizing_player, Action* best_action) {
50  if (state->IsTerminal()) {
51  return state->PlayerReturn(maximizing_player);
52  }
53 
54  if (depth == 0 && !value_function) {
55  SpielFatalError(
56  "We assume we can walk the full depth of the tree. "
57  "Try increasing depth or provide a value_function.");
58  }
59 
60  if (depth == 0) {
61  return value_function(*state);
62  }
63 
64  Player player = state->CurrentPlayer();
65  if (player == maximizing_player) {
66  double value = -std::numeric_limits<double>::infinity();
67 
68  for (Action action : state->LegalActions()) {
69  state->ApplyAction(action);
70  double child_value =
71  _alpha_beta(state, /*depth=*/depth - 1, /*alpha=*/alpha,
72  /*beta=*/beta, value_function, maximizing_player,
73  /*best_action=*/nullptr);
74  state->UndoAction(player, action);
75 
76  if (child_value > value) {
77  value = child_value;
78  if (best_action != nullptr) {
79  *best_action = action;
80  }
81  }
82 
83  alpha = std::max(alpha, value);
84  if (alpha >= beta) {
85  break; // beta cut-off
86  }
87  }
88 
89  return value;
90  } else {
91  double value = std::numeric_limits<double>::infinity();
92 
93  for (Action action : state->LegalActions()) {
94  state->ApplyAction(action);
95  double child_value =
96  _alpha_beta(state, /*depth=*/depth - 1, /*alpha=*/alpha,
97  /*beta=*/beta, value_function, maximizing_player,
98  /*best_action=*/nullptr);
99  state->UndoAction(player, action);
100 
101  if (child_value < value) {
102  value = child_value;
103  if (best_action != nullptr) {
104  *best_action = action;
105  }
106  }
107 
108  beta = std::min(beta, value);
109  if (alpha >= beta) {
110  break; // alpha cut-off
111  }
112  }
113 
114  return value;
115  }
116 }

118 // Expectiminimax algorithm.
119 //
120 // Runs expectiminimax until the specified depth.
121 // See https://en.wikipedia.org/wiki/Expectiminimax for details.
122 //
123 // Arguments:
124 // state: The state to start the search from.
125 // depth: The depth of the search (not counting chance nodes).
126 // value_function: A value function, taking in a state and returning a value,
127 // in terms of the maximizing_player_id.
128 // maximizing_player_id: The player running the search (current player at root
129 // of the search tree).
130 //
131 // Returns:
132 // The optimal value of the sub-game starting in state.
133 double _expectiminimax(const State* state, int depth,
134  std::function<double(const State&)> value_function,
135  Player maximizing_player, Action* best_action) {
136  if (state->IsTerminal()) {
137  return state->PlayerReturn(maximizing_player);
138  }
139 
140  if (depth == 0 && !value_function) {
141  SpielFatalError(
142  "We assume we can walk the full depth of the tree. "
143  "Try increasing depth or provide a value_function.");
144  }
145 
146  if (depth == 0) {
147  return value_function(*state);
148  }
149 
150  Player player = state->CurrentPlayer();
151  if (state->IsChanceNode()) {
152  double value = 0;
153  for (const auto& actionprob : state->ChanceOutcomes()) {
154  std::unique_ptr<State> child_state = state->Child(actionprob.first);
155  double child_value =
156  _expectiminimax(child_state.get(), depth, value_function,
157  maximizing_player, /*best_action=*/nullptr);
158  value += actionprob.second * child_value;
159  }
160  return value;
161  } else if (player == maximizing_player) {
162  double value = -std::numeric_limits<double>::infinity();
163 
164  for (Action action : state->LegalActions()) {
165  std::unique_ptr<State> child_state = state->Child(action);
166  double child_value = _expectiminimax(child_state.get(),
167  /*depth=*/depth - 1, value_function,
168  maximizing_player,
169  /*best_action=*/nullptr);
170 
171  if (child_value > value) {
172  value = child_value;
173  if (best_action != nullptr) {
174  *best_action = action;
175  }
176  }
177  }
178  return value;
179  } else {
180  double value = std::numeric_limits<double>::infinity();
181 
182  for (Action action : state->LegalActions()) {
183  std::unique_ptr<State> child_state = state->Child(action);
184  double child_value = _expectiminimax(child_state.get(),
185  /*depth=*/depth - 1, value_function,
186  maximizing_player,
187  /*best_action=*/nullptr);
188 
189  if (child_value < value) {
190  value = child_value;
191  if (best_action != nullptr) {
192  *best_action = action;
193  }
194  }
195  }
196  return value;
197  }

201 std::pair<double, Action> AlphaBetaSearch(
202  const Game& game, const State* state,
203  std::function<double(const State&)> value_function, int depth_limit,
204  Player maximizing_player) {
205  SPIEL_CHECK_LE(game.NumPlayers(), 2);
206 
207  GameType game_info = game.GetType();
208  SPIEL_CHECK_EQ(game_info.chance_mode, GameType::ChanceMode::kDeterministic);
209  SPIEL_CHECK_EQ(game_info.information,
210  GameType::Information::kPerfectInformation);
211  SPIEL_CHECK_EQ(game_info.dynamics, GameType::Dynamics::kSequential);
212  SPIEL_CHECK_EQ(game_info.utility, GameType::Utility::kZeroSum);
213  SPIEL_CHECK_EQ(game_info.reward_model, GameType::RewardModel::kTerminal);
214 
215  std::unique_ptr<State> search_root;
216  if (state == nullptr) {
217  search_root = game.NewInitialState();
218  } else {
219  search_root = state->Clone();
220  }
221 
222  if (maximizing_player == kInvalidPlayer) {
223  maximizing_player = search_root->CurrentPlayer();
224  }
225 
226  double infinity = std::numeric_limits<double>::infinity();
227  Action best_action = kInvalidAction;
228  double value = _alpha_beta(
229  search_root.get(), /*depth=*/depth_limit, /*alpha=*/-infinity,
230  /*beta=*/infinity, value_function, maximizing_player, &best_action);
231 
232  return {value, best_action};
233 }
