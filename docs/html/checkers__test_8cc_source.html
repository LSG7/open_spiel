87 void MultipleJumpTest() {
88  std::shared_ptr<const Game> game = LoadGame("checkers");
89  std::unique_ptr<State> state = game->NewInitialState();
90  CheckersState* cstate = static_cast<CheckersState*>(state.get());
91  cstate->SetCustomBoard(
92  "0..........*.................+.o......o..+..........+....o.o.....");
93  cstate->ApplyAction(cstate->LegalActions()[0]);
94  // Confirm that player 0 is given only one action (f4 token is in the middle
95  // of a multiple jump) and there's a capture opportunity for c1 piece as well
96  // (which cannot be moved in this extra move)
97  SPIEL_CHECK_EQ(cstate->LegalActions().size(), 1);
98  cstate->ApplyAction(cstate->LegalActions()[0]);
99  SPIEL_CHECK_EQ(cstate->BoardAt(0, 1), CellState::kWhiteKing);
100  SPIEL_CHECK_EQ(cstate->BoardAt(1, 2), CellState::kEmpty);
101  SPIEL_CHECK_EQ(cstate->BoardAt(3, 4), CellState::kEmpty);
102 }
