92 class BreakthroughGame : public Game {
93  public:
94  explicit BreakthroughGame(const GameParameters& params);
95  int NumDistinctActions() const override;
96  std::unique_ptr<State> NewInitialState() const override {
97  return std::unique_ptr<State>(
98  new BreakthroughState(shared_from_this(), rows_, cols_));
99  }
100  int NumPlayers() const override { return kNumPlayers; }
101  double MinUtility() const override { return -1; }
102  absl::optional<double> UtilitySum() const override { return 0; }
103  double MaxUtility() const override { return 1; }
104  std::vector<int> ObservationTensorShape() const override {
105  return {kCellStates, rows_, cols_};
106  }
107 
108  // Each piece must move forward from its current position, so the second last
109  // row on the opponent's side is a distance of rows_ - 2 for the front row,
110  // and rows_ - 1 for the back row (= 2*rows_ - 3). This can be done for each
111  // column, and for both players, and there is one final move to step onto the
112  // last winning row. As such, the formula for maximum game length is:
113  int MaxGameLength() const override {
114  return (2 * (2 * rows_ - 3) * cols_) + 1;
115  }
116 
117  std::unique_ptr<State> DeserializeState(
118  const std::string& str) const override;
119 
120  private:
121  int rows_ = -1;
122  int cols_ = -1;
123 };
