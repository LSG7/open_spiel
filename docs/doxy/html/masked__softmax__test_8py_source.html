108class MaskedSoftmaxTest(parameterized.TestCase, absltest.TestCase):
109
110 @parameterized.parameters(_ALL_TESTS_INPUTS)
111 def test_np_masked_softmax(self, logits, legal_actions, expected):
112 np.testing.assert_array_almost_equal(
113 expected, masked_softmax.np_masked_softmax(logits, legal_actions))
114
115 @parameterized.parameters(_ALL_TESTS_INPUTS)
116 def test_tf_masked_softmax(self, np_logits, np_legal_actions, expected):
117 logits = tf.Variable(np_logits, tf.float32)
118 mask = tf.Variable(np_legal_actions, tf.float32)
119
120 policy = masked_softmax.tf_masked_softmax(logits, mask)
121
122 with tf.Session() as sess:
123 sess.run(tf.global_variables_initializer())
124 np_policy = sess.run(policy)
125
126 np.testing.assert_array_almost_equal(expected, np_policy)
127
128 def test_masked_softmax_on_all_invalid_moves(self):
129 # If all actions are illegal, the behavior is undefined (it can be nan
130 # or can be 0. We add this test to document this behavior and know if we
131 # change it.
132 np_logits = np.asarray([[
133 [1.0, 1.0, 1.0],
134 [0.0, 0.0, 0.0],
135 [0.0, 0.0, 0.0],
136 ]])
137 logits = tf.Variable(np_logits, tf.float32)
138 np_mask = np.asarray([[
139 [1.0, 1.0, 1.0],
140 [1.0, 0.0, 1.0],
141 [0.0, 0.0, 0.0],
142 ]])
143 mask = tf.Variable(np_mask, tf.float32)
144
145 expected = np.asarray([[
146 [1 / 3, 1 / 3, 1 / 3],
147 [1 / 2, 0.0, 1 / 2],
148 [np.nan, np.nan, np.nan],
149 ]])
150
151 policy = masked_softmax.tf_masked_softmax(logits, mask)
152
153 with tf.Session() as sess:
154 sess.run(tf.global_variables_initializer())
155 np_policy = sess.run(policy)
156 np.testing.assert_array_almost_equal(expected, np_policy)
157
158 # Numpy behaves similarly.
159 np.testing.assert_array_almost_equal(
160 expected, masked_softmax.np_masked_softmax(np_logits, np_mask))
