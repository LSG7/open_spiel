29 // 8 adjacent directions.
30 //
31 // 405
32 // 1 2
33 // 637
34 //
35 // The order is important because it is used to index 3x3 patterns!
36 //
37 inline constexpr std::array<int, 9> Dir8 = {{
38  kVirtualBoardSize, // new line
39  -1, // new line
40  +1, // new line
41  -static_cast<int>(kVirtualBoardSize),
42  +static_cast<int>(kVirtualBoardSize) - 1,
43  +static_cast<int>(kVirtualBoardSize) + 1,
44  -static_cast<int>(kVirtualBoardSize) - 1,
45  -static_cast<int>(kVirtualBoardSize) + 1,
46  0 // Dummy element.
47 }};

49 // Calls f for all 4 direct neighbours of p.
50 // f should have type void f(VirtualPoint n), but is passed as a template so we
51 // can elide the function call overhead.
52 template <typename F>
53 void Neighbours(VirtualPoint p, const F& f) {
54  f(p + kVirtualBoardSize);
55  f(p + 1);
56  f(p - 1);
57  f(p - kVirtualBoardSize);
58 }

93 std::string MoveAsAscii(VirtualPoint p, GoColor c) {
94  static std::string code = "0123456789abcdefghijklmnopqrstuvwxyz";
95  static int mask = 31;
96  // 1 bit for color, 9 bits for the point.
97  uint16_t value = static_cast<int>(c) | (p << 1);
98  // Encode in 2 characters of 5 bit each.
99  std::string encoded;
100  encoded.push_back(code[(value >> 5) & mask]);
101  encoded.push_back(code[value & mask]);
102  return encoded;
103 }

119 std::pair<int, int> VirtualPointTo2DPoint(VirtualPoint p) {
120  if (p == kInvalidPoint || p == kVirtualPass) return std::make_pair(-1, -1);
121 
122  const int row = static_cast<int>(p) / kVirtualBoardSize;
123  const int col = static_cast<int>(p) % kVirtualBoardSize;
124  return std::make_pair(row - 1, col - 1);

132 // Internally, the board is *always* 21*21 with a border of guard stones around
133 // all sides of the board. Thus we need to map a coordinate in that space
134 // to a coordinate in the normal board.
135 Action VirtualActionToAction(int virtual_action, int board_size) {
136  if (virtual_action == kVirtualPass) return board_size * board_size;
137  const int virtual_row = static_cast<int>(virtual_action) / kVirtualBoardSize;
138  const int virtual_col = static_cast<int>(virtual_action) % kVirtualBoardSize;
139  return board_size * (virtual_row - 1) + (virtual_col - 1);
140 }

142 int ActionToVirtualAction(Action action, int board_size) {
143  if (action == board_size * board_size) return kVirtualPass;
144  int row = action / board_size;
145  int column = action % board_size;
146  return (row + 1) * kVirtualBoardSize + (column + 1);
147 }

235 VirtualPoint MakePoint(std::string s) {
236  std::transform(s.begin(), s.end(), s.begin(), ::tolower);
237 
238  if (s == "pass") return kVirtualPass;
239  if (s.size() < 2 || s.size() > 3) return kInvalidPoint;
240 
241  int col = s[0] < 'i' ? s[0] - 'a' : s[0] - 'a' - 1;
242  int row = s[1] - '0';
243  if (s.size() == 3) {
244  row *= 10;
245  row += s[2] - '0';
246  }
247  return VirtualPointFrom2DPoint({row - 1, col});
248 }

290 bool GoBoard::PlayMove(VirtualPoint p, GoColor c) {
291  if (p == kVirtualPass) {
292  last_ko_point_ = kInvalidPoint;
293  return true;
294  }
295 
296  if (board_[p].color != GoColor::kEmpty) {
297  SpielFatalError(absl::StrCat("Trying to play the move ", GoColorToString(c),
298  ": ", VirtualPointToString(p), " (", p,
299  ") but the cell is already filled with ",
300  GoColorToString(board_[p].color)));
301  }
302  SPIEL_CHECK_EQ(GoColor::kEmpty, board_[p].color);
303 
304  // Preparation for ko checking.
305  bool played_in_enemy_eye = true;
306  Neighbours(p, [this, c, &played_in_enemy_eye](VirtualPoint n) {
307  GoColor s = PointColor(n);
308  if (s == c || s == GoColor::kEmpty) {
309  played_in_enemy_eye = false;
310  }
311  });
312 
313  JoinChainsAround(p, c);
314  SetStone(p, c);
315  RemoveLibertyFromNeighbouringChains(p);
316  int stones_captured = CaptureDeadChains(p, c);
317 
318  if (played_in_enemy_eye && stones_captured == 1) {
319  last_ko_point_ = last_captures_[0];
320  } else {
321  last_ko_point_ = kInvalidPoint;
322  }
323 
324  SPIEL_CHECK_GT(chain(p).num_pseudo_liberties, 0);
325 
326  return true;
327 }

329 VirtualPoint GoBoard::SingleLiberty(VirtualPoint p) const {
330  VirtualPoint head = ChainHead(p);
331  VirtualPoint liberty = chain(p).single_liberty();
332 
333  // Check it is really a liberty.
334  SPIEL_CHECK_TRUE(IsInBoardArea(liberty));
335  SPIEL_CHECK_TRUE(IsEmpty(liberty));
336 
337  // Make sure the liberty actually borders the group.
338  for (auto n = Neighbours4(liberty); n; ++n) {
339  if (ChainHead(*n) == head) return liberty;
340  }
341 
342  SpielFatalError(
343  absl::StrCat("liberty", liberty, " does not actually border group ", p));
344 }

357 // Combines the groups around the newly placed stone at vertex. If no groups
358 // are available for joining, the new stone is placed as a new group.
359 void GoBoard::JoinChainsAround(VirtualPoint p, GoColor c) {
360  VirtualPoint largest_chain_head = kInvalidPoint;
361  int largest_chain_size = 0;
362  Neighbours(
363  p, [this, c, &largest_chain_head, &largest_chain_size](VirtualPoint n) {
364  if (PointColor(n) == c) {
365  Chain& c = chain(n);
366  if (c.num_stones > largest_chain_size) {
367  largest_chain_size = c.num_stones;
368  largest_chain_head = ChainHead(n);
369  }
370  }
371  });
372  if (largest_chain_size == 0) {
373  InitNewChain(p);
374  return;
375  }
376 
377  Neighbours(p, [this, c, &largest_chain_head](VirtualPoint n) {
378  if (PointColor(n) == c) {
379  VirtualPoint chain_head = ChainHead(n);
380  if (chain_head != largest_chain_head) {
381  chain(largest_chain_head).merge(chain(n));
382 
383  // Set all stones in the smaller string to be part of the larger
384  // chain.
385  VirtualPoint cur = n;
386  do {
387  board_[cur].chain_head = largest_chain_head;
388  cur = board_[cur].chain_next;
389  } while (cur != n);
390 
391  // Connect the 2 linked lists representing the stones in the two
392  // chains.
393  std::swap(board_[largest_chain_head].chain_next, board_[n].chain_next);
394  }
395  }
396  });
397 
398  board_[p].chain_next = board_[largest_chain_head].chain_next;
399  board_[largest_chain_head].chain_next = p;
400  board_[p].chain_head = largest_chain_head;
401  chain(largest_chain_head).num_stones += 1;
402 
403  Neighbours(p, [this, largest_chain_head](VirtualPoint n) {
404  if (IsEmpty(n)) {
405  chain(largest_chain_head).add_liberty(n);
406  }
407  });
408 }

414 int GoBoard::CaptureDeadChains(VirtualPoint p, GoColor c) {
415  int stones_captured = 0;
416  int capture_index = 0;
417  Neighbours(p, [this, c, &capture_index, &stones_captured](VirtualPoint n) {
418  if (PointColor(n) == OppColor(c) && chain(n).num_pseudo_liberties == 0) {
419  last_captures_[capture_index++] = ChainHead(n);
420  stones_captured += chain(n).num_stones;
421  RemoveChain(n);
422  }
423  });
424 
425  for (; capture_index < last_captures_.size(); ++capture_index) {
426  last_captures_[capture_index] = kInvalidPoint;
427  }
428 
429  return stones_captured;
430 }

432 void GoBoard::RemoveChain(VirtualPoint p) {
433  VirtualPoint this_chain_head = ChainHead(p);
434  VirtualPoint cur = p;
435  do {
436  VirtualPoint next = board_[cur].chain_next;
437 
438  SetStone(cur, GoColor::kEmpty);
439  InitNewChain(cur);
440 
441  Neighbours(cur, [this, this_chain_head, cur](VirtualPoint n) {
442  if (ChainHead(n) != this_chain_head || IsEmpty(n)) {
443  chain(n).add_liberty(cur);
444  }
445  });
446 
447  cur = next;
448  } while (cur != p);
449 }

472 bool GoBoard::IsLegalMove(VirtualPoint p, GoColor c) const {
473  if (p == kVirtualPass) return true;
474  if (!IsInBoardArea(p)) return false;
475  if (!IsEmpty(p) || p == LastKoPoint()) return false;
476  if (chain(p).num_pseudo_liberties > 0) return true;
477 
478  // For all checks below, the newly placed stone is completely surrounded by
479  // enemy and friendly stones.
480 
481  // Allow to play if the placed stones connects to a group that still has at
482  // least one other liberty after connecting.
483  bool has_liberty = false;
484  Neighbours(p, [this, c, &has_liberty](VirtualPoint n) {
485  has_liberty |= (PointColor(n) == c && !chain(n).in_atari());
486  });
487  if (has_liberty) return true;
488 
489  // Allow to play if the placed stone will kill at least one group.
490  bool kills_group = false;
491  Neighbours(p, [this, c, &kills_group](VirtualPoint n) {
492  kills_group |= (PointColor(n) == OppColor(c) && chain(n).in_atari());
493  });
494  if (kills_group) return true;
495 
496  return false;
497 }

536 VirtualPoint GoBoard::Chain::single_liberty() const {
537  SPIEL_CHECK_TRUE(in_atari());
538  // A point is in Atari if it has only a single liberty, i.e. all pseudo
539  // liberties are for the same point.
540  // This is true exactly when
541  // liberty_vertex_sum**2 == liberty_vertex_sum_squared * num_pseudo_liberties
542  // Since all pseudo liberties are for the same point, this is equivalent to
543  // (taking n = num_pseudo_liberties):
544  // (n * p)**2 = (n * p**2) * n
545  // Thus to obtain p, we simple need to divide out the number of pseudo
546  // liberties.
547  SPIEL_CHECK_EQ(liberty_vertex_sum % num_pseudo_liberties, 0);
548  return static_cast<VirtualPoint>(liberty_vertex_sum / num_pseudo_liberties);
549 }

586 void GoBoard::GroupIter::step() {
587  --lib_i_;
588  while (lib_i_ < 0 && !marked_[chain_cur_]) {
589  Neighbours(chain_cur_, [this](VirtualPoint n) {
590  VirtualPoint head = board_->ChainHead(n);
591  if (board_->PointColor(head) == group_color_ && !marked_[head]) {
592  cur_libs_[++lib_i_] = head;
593  marked_[head] = true;
594  }
595  });
596  marked_[chain_cur_] = true;
597  chain_cur_ = board_->board_[chain_cur_].chain_next;
598  }
599 }

601 // Returns the number of points surrounded entirely by one color.
602 // Aborts early and returns 0 if the area borders both black and white stones.
603 int NumSurroundedPoints(const GoBoard& board, const VirtualPoint p,
604  std::array<bool, kVirtualBoardPoints>* marked,
605  bool* reached_black, bool* reached_white) {
606  if ((*marked)[p]) return 0;
607  (*marked)[p] = true;
608 
609  int num_points = 1;
610  Neighbours(p, [&board, &num_points, marked, reached_black,
611  reached_white](VirtualPoint n) {
612  switch (board.PointColor(n)) {
613  case GoColor::kBlack:
614  *reached_black = true;
615  break;
616  case GoColor::kWhite:
617  *reached_white = true;
618  break;
619  case GoColor::kEmpty:
620  num_points +=
621  NumSurroundedPoints(board, n, marked, reached_black, reached_white);
622  break;
623  case GoColor::kGuard:
624  // Ignore the border.
625  break;
626  }
627  });
628 
629  return num_points;
630 }

632 float TrompTaylorScore(const GoBoard& board, float komi, int handicap) {
633  // The delta of how many points on the board black and white have occupied,
634  // from black's point of view, i.e. Black points - White points.
635  int occupied_delta = 0;
636 
637  // We need to keep track of which empty points we've already counted as part
638  // of a larger territory.
639  std::array<bool, kVirtualBoardPoints> marked;
640  marked.fill(false);
641 
642  for (VirtualPoint p : BoardPoints(board.board_size())) {
643  switch (board.PointColor(p)) {
644  case GoColor::kBlack:
645  ++occupied_delta;
646  break;
647  case GoColor::kWhite:
648  --occupied_delta;
649  break;
650  case GoColor::kEmpty: {
651  if (marked[p]) continue;
652  // If some empty points are surrounded entirely by one player, they
653  // count as that player's territory.
654  bool reached_black = false, reached_white = false;
655  int n = NumSurroundedPoints(board, p, &marked, &reached_black,
656  &reached_white);
657  if (reached_black && !reached_white) {
