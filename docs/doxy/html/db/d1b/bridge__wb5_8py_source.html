57def net_fn(x):
58 """Haiku module for our network."""
59 net = hk.Sequential([
60 hk.Linear(1024),
61 jax.nn.relu,
62 hk.Linear(1024),
63 jax.nn.relu,
64 hk.Linear(1024),
65 jax.nn.relu,
66 hk.Linear(1024),
67 jax.nn.relu,
68 hk.Linear(NUM_ACTIONS),
69 jax.nn.log_softmax,
70 ])
71 return net(x)

81def ai_action(state, net, params):
82 observation = np.array(state.observation_tensor(), np.float32)
83 policy = np.exp(net.apply(params, observation))
84 probs_actions = [(p, a + MIN_ACTION) for a, p in enumerate(policy)]
85 pred = max(probs_actions)[1]
86 return pred

89def _run_once(state, bots, net, params):
90 """Plays bots with each other, returns terminal utility for each player."""
91 for bot in bots:
92 bot.restart()
93 while not state.is_terminal():
94 if state.is_chance_node():
95 outcomes, probs = zip(*state.chance_outcomes())
96 state.apply_action(np.random.choice(outcomes, p=probs))
97 else:
98 if FLAGS.sleep:
99 time.sleep(FLAGS.sleep) # wait for the human to see how it goes
100 if state.current_player() % 2 == 1:
101 # Have simplest play for now
102 action = state.legal_actions()[0]
103 if action > 51:
104 # TODO(ed2k) extend beyond just bidding
105 action = ai_action(state, net, params)
106 state.apply_action(action)
107 else:
108 result = bots[state.current_player() // 2].step(state)
109 state.apply_action(result)
110 return state

113def main(argv):
114 if len(argv) > 1:
115 raise app.UsageError("Too many command-line arguments.")
116 game = pyspiel.load_game("bridge(use_double_dummy_result=false)")
117 net, params = load_model()
118 bots = [
119 bluechip_bridge.BlueChipBridgeBot(game, 0, controller_factory),
120 bluechip_bridge.BlueChipBridgeBot(game, 2, controller_factory)
121 ]
122
123 results = []
124
125 for i_deal in range(FLAGS.num_deals):
126 state = _run_once(game.new_initial_state(), bots, net, params)
127 print("Deal #{}; final state:\n{}".format(i_deal, state))
128 results.append(state.returns())
129
130 stats = np.array(results)
131 mean = np.mean(stats, axis=0)
132 stderr = np.std(stats, axis=0, ddof=1) / np.sqrt(FLAGS.num_deals)
133 print(u"Absolute score: {:+.1f}\u00b1{:.1f}".format(mean[0], stderr[0]))
134 print(u"Relative score: {:+.1f}\u00b1{:.1f}".format(mean[1], stderr[1]))

144class _WBridge5Client(object):
145 """Manages the connection to a WBridge5 bot."""
146
147 def __init__(self, command):
148 self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
149 self.sock.bind(("", 0))
150 self.port = self.sock.getsockname()[1]
151 self.sock.listen(1)
152 self.process = None
153 self.command = command.format(port=self.port)
154
155 def start(self):
156 if self.process is not None:
157 self.process.kill()
158 self.process = subprocess.Popen(self.command.split(" "))
159 self.conn, self.addr = self.sock.accept()
160
161 def read_line(self):
162 line = ""
163 while True:
164 self.conn.settimeout(FLAGS.timeout_secs)
165 data = self.conn.recv(1024)
166 if not data:
167 raise EOFError("Connection closed")
168 line += data.decode("ascii")
169 if line.endswith("\n"):
170 return re.sub(r"\s+", " ", line).strip()
171
172 def send_line(self, line):
173 self.conn.send((line + "\r\n").encode("ascii"))
174
175 def terminate(self):
176 self.process.kill()
177 self.process = None
