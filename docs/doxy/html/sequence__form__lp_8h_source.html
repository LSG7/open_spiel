28// An implementation of a sequence-form linear program for computing Nash
29// equilibria in sequential games, based on [1]. The implementation constructs
30// infostate trees for both players, connects them through the terminals and
31// recursively specifies constraints on reach probability of the player and
32// counterfactual values of the opponent.
33//
34// [1]: Efficient Computation of Equilibria for Extensive Two-Person Games
35// http://www.maths.lse.ac.uk/Personal/stengel/TEXTE/geb1996b.pdf

46template <class T>
47struct BijectiveContainer {
48 std::map<T, T> x2y;
49 std::map<T, T> y2x;
50
51 void put(std::pair<T, T> xy) {
52 const T& x = xy.first;
53 const T& y = xy.second;
54 SPIEL_CHECK_TRUE(x2y.find(x) == x2y.end());
55 SPIEL_CHECK_TRUE(y2x.find(y) == y2x.end());
56 x2y[x] = y;
57 y2x[y] = x;
58 }
59 // Direction is equivalent to player id.
60 const std::map<T, T>& association(int direction) const {
61 SPIEL_CHECK_TRUE(direction == 0 || direction == 1);
62 if (direction == 0) {
63 return x2y;
64 } else {
65 return y2x;
66 }
67 }
68};
