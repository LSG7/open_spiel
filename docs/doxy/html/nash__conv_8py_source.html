14"""Implementation of Nash Conv metric for a policy.
15
16In the context of mean field games, the Nash Conv is the difference between:
17- the value of a policy against the distribution of that policy,
18- and the best response against the distribution of the policy.
19"""

28class NashConv(object):
29 """Computes the Nash Conv of a policy."""
30
31 def __init__(self, game, policy: policy_std.Policy, root_state=None):
32 """Initializes the nash conv.
33
34 Args:
35 game: The game to analyze.
36 policy: A `policy.Policy` object.
37 root_state: The state of the game at which to start. If `None`, the game
38 root state is used.
39 """
40 self._game = game
41 self._policy = policy
42 if root_state is None:
43 self._root_states = game.new_initial_states()
44 else:
45 self._root_states = [root_state]
46 self._distrib = distribution.DistributionPolicy(
47 self._game, self._policy, root_state=root_state)
48 self._pi_value = policy_value.PolicyValue(
49 self._game,
50 self._distrib,
51 self._policy,
52 value.TabularValueFunction(self._game),
53 root_state=root_state)
54 self._br_value = best_response_value.BestResponse(
55 self._game,
56 self._distrib,
57 value.TabularValueFunction(self._game),
58 root_state=root_state)
59
60 def nash_conv(self):
61 """Returns the nash conv.
62
63 Returns:
64 A float representing the nash conv for the policy.
65 """
66 return sum([
67 self._br_value.eval_state(state) - self._pi_value.eval_state(state)
68 for state in self._root_states
69 ])
70
71 def br_values(self):
72 """Returns the best response values to the policy distribution.
73
74 Returns:
75 A List[float] representing the best response values for a policy
76 distribution.
77 """
78 return [self._br_value.eval_state(state) for state in self._root_states]
