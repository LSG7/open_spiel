15 """Wraps third-party bridge bots to make them usable in OpenSpiel.
16 
17 This code enables OpenSpiel interoperation for bots which implement the BlueChip
18 bridge protocol. This is widely used, e.g. in the World computer bridge
19 championships. For a rough outline of the protocol, see:
20 http://www.bluechipbridge.co.uk/protocol.htm
21 
22 No formal specification is available. This implementation has been verified
23 to work correctly with WBridge5.
24 
25 This bot controls a single player in the game of uncontested bridge bidding. It
26 chooses its actions by invoking an external bot which plays the full game of
27 bridge. This means that each time the bot is asked for an action, it sends up to
28 three actions (forced passes from both opponents, plus partner's most recent
29 action) to the external bridge bot, and obtains an action in return.
30 
31 Since we are restricting ourselves to the uncontested bidding game, we have
32 no support for Doubling, Redoubling, or the play of the cards.
33 """

81 # BlueChip bridge protocol message constants
82 _SEATS = ["WEST", "EAST"]
83 _OPPONENTS = ["NORTH", "SOUTH"]
84 _TRUMP_SUIT = ["C", "D", "H", "S", "NT"]
85 _NUMBER_TRUMP_SUITS = len(_TRUMP_SUIT)
86 _RANKS = ["2", "3", "4", "5", "6", "7", "8", "9", "T", "J", "Q", "K", "A"]

92 def _string_to_action(call_str):
93  """Converts a BlueChip bid string to an OpenSpiel action id (an integer).
94 
95  Args:
96  call_str: string representing a bid in the BlueChip format, i.e. "[level]
97  (as a digit) + [trump suit (S, H, D, C or NT)]", e.g. "1C".
98 
99  Returns:
100  An integer action id - see `bridge_uncontested_bidding.cc`, functions
101  `Denomination` and `Level`.
102  0 is reserved for Pass, so bids are in order from 1 upwards: 1 = 1C,
103  2 = 1D, etc.
104  """
105  level = int(call_str[0])
106  trumps = _TRUMP_SUIT.index(call_str[1:])
107  return (level - 1) * _NUMBER_TRUMP_SUITS + trumps + 1

110 def _action_to_string(action):
111  """Converts OpenSpiel action id (an integer) to a BlueChip bid string.
112 
113  Args:
114  action: an integer action id corresponding to a bid.
115 
116  Returns:
117  A string in BlueChip format.
118 
119  Inverse of `_string_to_action`. See documentation there.
120  """
121  level = str((action - 1) // _NUMBER_TRUMP_SUITS + 1)
122  trumps = _TRUMP_SUIT[(action - 1) % _NUMBER_TRUMP_SUITS]
123  return level + trumps

126 def _expect_regex(client, regex):
127  """Reads a line from the client, parses it using the regular expression."""
128  line = client.read_line()
129  match = re.match(regex, line)
130  if not match:
131  raise ValueError("Received '{}' which does not match regex '{}'".format(
132  line, regex))
133  return match.groupdict()

143 def _hand_string(state_vec):
144  """Returns the hand of the to-play player in the state in BlueChip format."""
145  # See UncontestedBiddingState::InformationStateTensor
146  # The first 52 elements are whether or not we hold the given card (cards
147  # ordered suit-by-suit, in ascending order of rank).
148  suits = []
149  for suit in reversed(range(4)):
150  cards = []
151  for rank in reversed(range(13)):
152  if state_vec[rank * 4 + suit]:
153  cards.append(_RANKS[rank])
154  suits.append(_TRUMP_SUIT[suit] + " " + (" ".join(cards) if cards else "-") +
155  ".")
156  return " ".join(suits)
157

159 def _actions(state_vec):
160  """Returns the player actions that have been taken in the game so far."""
161  # See UncontestedBiddingState::InformationStateTensor
162  # The first 52 elements are the cards held, then two elements for each
163  # possible action, specifying which of the two players has taken it (if
164  # either player has). Then two elements specifying which player we are.
165  actions = state_vec[52:-2]
166  return [index // 2 for index, value in enumerate(actions) if value]

169 def _connect(client, seat, state_vec):
170  """Performs the initial handshake with a BlueChip bot."""
171  client.start()
172  client_name = _expect_regex(client, _CONNECT)["client_name"]
173  client.send_line(_SEATED.format(seat=seat, client_name=client_name))
174  _expect(client, _READY_FOR_TEAMS.format(seat=seat))
175  client.send_line(_TEAMS)
176  _expect(client, _READY_TO_START.format(seat=seat))
177  client.send_line(_START_BOARD)
178  _expect(client, _READY_FOR_DEAL.format(seat=seat))
179  client.send_line(_DEAL)
180  _expect(client, _READY_FOR_CARDS.format(seat=seat))
181  client.send_line(_CARDS.format(seat=seat, hand=_hand_string(state_vec)))

184 class BlueChipBridgeBot(pyspiel.Bot):
185  """An OpenSpiel bot, wrapping a BlueChip bridge bot implementation."""

187  def __init__(self, game, player_id, client):
188  """Initializes an OpenSpiel `Bot` wrapping a BlueChip-compatible bot.
189 
190  Args:
191  game: The OpenSpiel game object, should be an instance of
192  bridge_uncontested_bidding, without forced actions.
193  player_id: The id of the player the bot will act as, 0 = West (dealer), 1
194  = East.
195  client: The BlueChip bot; must support methods `start`, `read_line`, and
196  `send_line`.
197  """
198  pyspiel.Bot.__init__(self)
199  self._game_game = game
200  self._player_id_player_id = player_id
201  self._client_client = client
202  self._seat_seat = _SEATS[player_id]
203  self._partner_partner = _SEATS[1 - player_id]
204  self._left_hand_opponent_left_hand_opponent = _OPPONENTS[player_id]
205  self._right_hand_opponent_right_hand_opponent = _OPPONENTS[1 - player_id]
206  self._connected_connected = False
207 
208  def player_id(self):
209  return self._player_id_player_id
210

212  """Indicates that the next step may be from a non-sequential state."""
213  self._connected_connected = False
214 
215  def restart_at(self, state):
216  """Indicates that the next step may be from a non-sequential state."""
217  self._connected_connected = False
218 
219  def step(self, state):
220  """Returns the action and policy for the bot in this state."""
221  state_vec = state.information_state_tensor(self.player_idplayer_id())
222 
223  # Connect if necessary.
224  if not self._connected_connected:
225  _connect(self._client_client, self._seat_seat, state_vec)
226  self._connected_connected = True
227 
228  # Get the actions in the game so far.
229  actions = _actions(state_vec)
230 
231  # Unless this is the first or second action in the game, our
232  # left-hand-opponent will have passed since our last turn.
233  if len(actions) > 1:
234  _expect(
235  self._client_client,
236  _READY_FOR_BID.format(
237  seat=self._seat_seat, other=self._left_hand_opponent_left_hand_opponent))
238  self._client_client.send_line(
239  _OTHER_PLAYER_PASS.format(player=self._left_hand_opponent_left_hand_opponent))
240 
241  # Unless there aren't any prior actions, our partner will have bid
242  # or passed since our last turn, and so we need to send partner's action
243  # to the bot.
244  if actions:
245  _expect(self._client_client,
246  _READY_FOR_BID.format(seat=self._seat_seat, other=self._partner_partner))
247  if actions[-1] == _PASS_ACTION:
248  self._client_client.send_line(_OTHER_PLAYER_PASS.format(player=self._partner_partner))
249  else:
250  self._client_client.send_line(
251  _OTHER_PLAYER_BID.format(
252  player=self._partner_partner, bid=_action_to_string(actions[-1])))
253 
254  # Unless there aren't any prior actions, our right-hand-opponent will have
255  # passed since our last turn.
256  if actions:
257  _expect(
258  self._client_client,
259  _READY_FOR_BID.format(
260  seat=self._seat_seat, other=self._right_hand_opponent_right_hand_opponent))
261  self._client_client.send_line(
262  _OTHER_PLAYER_PASS.format(player=self._right_hand_opponent_right_hand_opponent))
263 
264  # Get our action from the bot.
265  our_action = _expect_regex(self._client_client,
266  _SELF_BID_OR_PASS.format(seat=self._seat_seat))
267  action = 0 if our_action["pass"] else _string_to_action(our_action["bid"])
268  return (action, 1.0), action
