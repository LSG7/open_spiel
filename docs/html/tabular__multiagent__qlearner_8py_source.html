51  res_values: a list of expected utilities for each agent

55 class TwoPlayerNashSolver(JointActionSolver):
56  """A joint action solver solving for Nash for two-player games.
57 
58  Uses python.algorithms.matrix_nash.lemke_howson_solve
59  """
60 
61  def __call__(self, payoffs_array):
62  assert len(payoffs_array) == 2
63 
64  row_payoffs, col_payoffs = payoffs_array[0], payoffs_array[1]
65  a0, a1 = payoffs_array.shape[1:]
66 
67  nashpy_game = nash.Game(row_payoffs, col_payoffs)
68 
69  best_value = float("-inf")
70  res_mixtures, res_values = None, None
71 
72  for (row_mixture, col_mixture) in nashpy_game.support_enumeration():
73  # TO-DO: handle the case where the LH solver gave ineligible answer
74  if np.sum(np.isnan(row_mixture)) or np.sum(np.isnan(col_mixture)):
75  continue
76  row_mixture_, col_mixture_ = row_mixture.reshape(
77  (-1, 1)), col_mixture.reshape((-1, 1))
78  row_value, col_value = (
79  row_mixture_.T.dot(row_payoffs).dot(col_mixture_)).item(), (
80  row_mixture_.T.dot(col_payoffs).dot(col_mixture_)).item()
81  # Currently using maximizing social welfare for equilibrium selection
82  if row_value + col_value > best_value:
83  best_value = row_value + col_value
84  res_mixtures = [row_mixture, col_mixture]
85  res_values = [row_value, col_value]
86 
87  # If no plauisble nash found, use uniform mixed strategies
88  if not res_mixtures:
89  res_mixtures = [np.ones(a0) / a0, np.ones(a1) / a1]
90  row_mixture_, col_mixture_ = res_mixtures[0].reshape(
91  (-1, 1)), res_mixtures[1].reshape((-1, 1))
92  res_values = [(row_mixture_.T.dot(row_payoffs).dot(col_mixture_)).item(),
93  (row_mixture_.T.dot(col_payoffs).dot(col_mixture_)).item()]
94 
95  return res_mixtures, res_values

98 class CorrelatedEqSolver(JointActionSolver):
99  """A joint action solver solving for correlated equilibrium.
100 
101  Uses python.algorithms.jspro._mgce and _mgcce for solving (coarse) correlated
102  equilibrium.
103  """
104 
105  def __init__(self, is_cce=False):
106  self._is_cce_is_cce = is_cce
107 
108  def __call__(self, payoffs_array):
109  num_players = len(payoffs_array)
110  assert num_players > 0
111  num_strategies_per_player = payoffs_array.shape[1:]
112  mixture, _ = (
113  _mgcce( # pylint: disable=g-long-ternary
114  payoffs_array,
115  [np.ones([ns], dtype=np.int32) for ns in num_strategies_per_player],
116  ignore_repeats=True)
117  if self._is_cce_is_cce else _mgcce(
118  payoffs_array,
119  [np.ones([ns], dtype=np.int32) for ns in num_strategies_per_player],
120  ignore_repeats=True))
121  mixtures, values = [], []
122  for n in range(num_players):
123  values.append(np.sum(payoffs_array[n] * mixture))
124  mixtures.append(
125  np.sum(
126  mixture,
127  axis=tuple([n_ for n_ in range(num_players) if n_ != n])))
128  return mixtures, values

131 class StackelbergEqSolver(JointActionSolver):
132  """A joint action solver solving for Stackelverg equilibrium.
133 
134  Uses python.algorithms.stackelberg_lp.py.
135  """
136 
137  def __init__(self, is_first_leader=True):
138  self._is_first_leader_is_first_leader = is_first_leader
139 
140  def __call__(self, payoffs_array):
141  assert len(payoffs_array) == 2
142  game = pyspiel.create_matrix_game(payoffs_array[0], payoffs_array[1])
143  try:
144  player0_strategy, player1_strategy, player0_value, player1_value = solve_stackelberg(
145  game, self._is_first_leader_is_first_leader)
146  return [player0_strategy,
147  player1_strategy], [player0_value, player1_value]
148  except: # pylint: disable=bare-except
149  # if the game matrix is degenerated and cannot solve for an SSE,
150  # return uniform strategy
151  num_player0_strategies, num_player1_strategies = payoffs_array[0].shape
152  player0_strategy, player1_strategy = np.ones(
153  num_player0_strategies) / num_player0_strategies, np.ones(
154  num_player1_strategies) / num_player1_strategies
155  player0_value, player1_value = player0_strategy.reshape(1, -1).dot(
156  payoffs_array[0]).dot(player1_strategy.reshape(
157  -1, 1)), player0_strategy.reshape(1, -1).dot(
158  payoffs_array[1]).dot(player1_strategy.reshape(-1, 1))
159  return [player0_strategy,
160  player1_strategy], [player0_value, player1_value]

163 class MultiagentQLearner(rl_agent.AbstractAgent):
164  """A multiagent joint action learner."""
165 
166  def __init__(self,
167  player_id,
168  num_players,
169  num_actions,
170  joint_action_solver,
171  step_size=0.1,
172  epsilon_schedule=rl_tools.ConstantSchedule(0.2),
173  discount_factor=1.0):
174  """Initialize the Multiagent joint-action Q-Learning agent.
175 
176  The joint_action_solver solves for one-step matrix game defined by Q-tables.
177 
178  Args:
179  player_id: the player id this agent will play as,
180  num_players: the number of players in the game,
181  num_actions: the number of distinct actions in the game,
182  joint_action_solver: the joint action solver class to use to solve the
183  one-step matrix games
184  step_size: learning rate for Q-learning,
185  epsilon_schedule: exploration parameter,
186  discount_factor: the discount factor as in Q-learning.
187  """
188  self._player_id_player_id = player_id
189  self._num_players_num_players = num_players
190  self._num_actions_num_actions = num_actions
191  self._joint_action_solver_joint_action_solver = joint_action_solver
192  self._step_size_step_size = step_size
193  self._epsilon_schedule_epsilon_schedule = epsilon_schedule
194  self._epsilon_epsilon = epsilon_schedule.value
195  self._discount_factor_discount_factor = discount_factor
196  self._q_values_q_values = [
197  collections.defaultdict(valuedict) for _ in range(num_players)
198  ]
199  self._prev_info_state_prev_info_state = None
200 
201  def _get_payoffs_array(self, info_state):
202  payoffs_array = np.zeros((self._num_players_num_players,) + tuple(self._num_actions_num_actions))
203  for joint_action in itertools.product(
204  *[range(dim) for dim in self._num_actions_num_actions]):
205  for n in range(self._num_players_num_players):
206  payoffs_array[
207  (n,) + joint_action] = self._q_values_q_values[n][info_state][joint_action]
208  return payoffs_array
209 
210  def _epsilon_greedy(self, info_state, legal_actions, epsilon):
211  """Returns a valid epsilon-greedy action and valid action probs.
212 
213  If the agent has not been to `info_state`, a valid random action is chosen.
214  Args:
215  info_state: hashable representation of the information state.
216  legal_actions: list of actions at `info_state`.
217  epsilon: float, prob of taking an exploratory action.
218 
219  Returns:
220  A valid epsilon-greedy action and valid action probabilities.
221  """
222  probs = np.zeros(self._num_actions_num_actions[self._player_id_player_id])
223 
224  state_probs, _ = self._joint_action_solver_joint_action_solver(
225  self._get_payoffs_array_get_payoffs_array(info_state))
226 
227  probs[legal_actions[self._player_id_player_id]] = (
228  epsilon / len(legal_actions[self._player_id_player_id]))
229  probs += (1 - epsilon) * state_probs[self._player_id_player_id]
230  action = np.random.choice(
231  range(self._num_actions_num_actions[self._player_id_player_id]), p=probs)
232  return action, probs
233 
234  def step(self, time_step, actions=None, is_evaluation=False):
235  """Returns the action to be taken and updates the Q-values if needed.
236 
237  Args:
238  time_step: an instance of rl_environment.TimeStep,
239  actions: list of actions taken by all agents from the previous step,
240  is_evaluation: bool, whether this is a training or evaluation call,
241 
242  Returns:
243  A `rl_agent.StepOutput` containing the action probs and chosen action.
244  """
245  info_state = str(time_step.observations["info_state"])
246  legal_actions = time_step.observations["legal_actions"]
247 
248  # Prevent undefined errors if this agent never plays until terminal step
249  action, probs = None, None
250 
251  # Act step: don't act at terminal states.
252  if not time_step.last():
253  epsilon = 0.0 if is_evaluation else self._epsilon_epsilon
254  # select according to the joint action solver
255  action, probs = self._epsilon_greedy_epsilon_greedy(
256  info_state, legal_actions, epsilon=epsilon)
257 
258  # Learn step: don't learn during evaluation or at first agent steps.
259  actions = tuple(actions)
260 
261  if self._prev_info_state_prev_info_state and not is_evaluation:
262  _, next_state_values = (
263  self._joint_action_solver_joint_action_solver(self._get_payoffs_array_get_payoffs_array(info_state)))
264  # update Q values for every agent
265  for n in range(self._num_players_num_players):
266  target = time_step.rewards[n]
267  if not time_step.last(): # Q values are zero for terminal.
268  target += self._discount_factor_discount_factor * next_state_values[n]
269 
270  prev_q_value = self._q_values_q_values[n][self._prev_info_state_prev_info_state][actions]
271 
272  self._q_values_q_values[n][self._prev_info_state_prev_info_state][actions] += (
273  self._step_size_step_size * (target - prev_q_value))
274 
275  # Decay epsilon, if necessary.
276  self._epsilon_epsilon = self._epsilon_schedule_epsilon_schedule.step()
277 
278  if time_step.last(): # prepare for the next episode.
279  self._prev_info_state_prev_info_state = None
280  return
281 
282  # Don't mess up with the state during evaluation.
283  if not is_evaluation:
284  self._prev_info_state_prev_info_state = info_state
285 
286  return rl_agent.StepOutput(action=action, probs=probs)
