94  def update(self, params, grads, t):
95  """Projected gradient descent on exploitability using Euclidean projection.
96 
97  Args:
98  params: tuple of variables to be updated (dist, y, regret)
99  grads: tuple of variable gradients (grad_dist, grad_y, regret_delta)
100  t: int, solver iteration
101  Returns:
102  new_params: tuple of update params (new_dist, new_y, new_regret)
103  """
104  dist, y, regret = params
105  _, y_grad, regret_delta = grads
106 
107  lr_y = np.clip(1 / float(t + 1), self.lrslrs[0], np.inf)
108  new_y = []
109  for y_i, y_grad_i in zip(y, y_grad):
110  new_y_i = y_i - lr_y * y_grad_i
111  new_y_i = np.clip(new_y_i, 0., np.inf)
112  new_y.append(new_y_i)
113 
114  if self.discountdiscount:
115  gamma = t / float(t + 1)
116  else:
117  gamma = 1
118 
119  new_dist = []
120  new_regret = []
121  for dist_i, regret_i, regret_delta_i in zip(dist, regret, regret_delta):
122  new_regret_i = gamma * regret_i + regret_delta_i
123  new_clipped_regrets_i = np.clip(
124  new_regret_i + self.optimismoptimism * regret_delta_i, 0., np.inf)
125  if np.sum(new_clipped_regrets_i) > 0:
126  new_dist_i = new_clipped_regrets_i / new_clipped_regrets_i.sum()
127  else:
128  new_dist_i = np.ones_like(dist_i) / dist_i.size
129  new_dist.append(new_dist_i)
130  new_regret.append(new_regret_i)
131 
132  return (new_dist, new_y, new_regret)

135 def gradients(dist, y, regret, payoff_matrices, num_players, p=1):
136  """Computes exploitablity gradient and aux variable gradients.
137 
138  Args:
139  dist: list of 1-d np.arrays, current estimate of nash distribution
140  y: list 1-d np.arrays (same shape as dist), current est. of payoff gradient
141  regret: list of 1-d np.arrays (same shape as dist), exploitability regrets
142  payoff_matrices: dictionary with keys as tuples of agents (i, j) and
143  values of (2 x A x A) np.arrays, payoffs for each joint action. keys
144  are sorted and arrays should be indexed in the same order
145  num_players: int, number of players, in case payoff_matrices is abbreviated
146  p: float in [0, 1], Tsallis entropy-regularization --> 0 as p --> 0
147  Returns:
148  gradient of exploitability w.r.t. (dist, y) as tuple
149  unregularized exploitability (stochastic estimate)
150  tsallis regularized exploitability (stochastic estimate)
151  """
152  del regret
153 
154  # first compute policy gradients and player effects (fx)
155  policy_gradient = []
156  other_player_fx = []
157  grad_y = []
158  unreg_exp = []
159  reg_exp = []
160  for i in range(num_players):
161 
162  nabla_i = np.zeros_like(dist[i])
163  for j in range(num_players):
164  if j == i:
165  continue
166  if i < j:
167  hess_i_ij = payoff_matrices[(i, j)][0]
168  else:
169  hess_i_ij = payoff_matrices[(j, i)][1].T
170 
171  nabla_ij = hess_i_ij.dot(dist[j])
172  nabla_i += nabla_ij / float(num_players - 1)
173 
174  grad_y.append(y[i] - nabla_i)
175  y[i] = nabla_i # TODO(imgemp): overwriting temporarily to test something
176 
177  if p > 0:
178  power = 1. / float(p)
179  s_i = np.linalg.norm(y[i], ord=power)
180  if s_i == 0:
181  br_i = misc.uniform_dist(y[i])
182  else:
183  br_i = (y[i] / s_i)**power
184  else:
185  power = np.inf
186  s_i = np.linalg.norm(y[i], ord=power)
187  br_i = np.zeros_like(dist[i])
188  maxima_i = (y[i] == s_i)
189  br_i[maxima_i] = 1. / maxima_i.sum()
190 
191  policy_gradient_i = nabla_i - s_i * dist[i]**p
192  policy_gradient.append(policy_gradient_i)
193 
194  unreg_exp.append(np.max(y[i]) - y[i].dot(dist[i]))
195 
196  br_i_inv_sparse = 1 - np.sum(br_i**(p + 1))
197  dist_i_inv_sparse = 1 - np.sum(dist[i]**(p + 1))
198  entr_br_i = s_i / (p + 1) * br_i_inv_sparse
199  entr_dist_i = s_i / (p + 1) * dist_i_inv_sparse
200 
201  reg_exp.append(y[i].dot(br_i - dist[i]) + entr_br_i - entr_dist_i)
202 
203  entr_br_vec_i = br_i_inv_sparse * br_i**(1 - p)
204  entr_dist_vec_i = dist_i_inv_sparse * dist[i]**(1 - p)
205  other_player_fx_i = (br_i - dist[i]) + 1 / (p + 1) * (
206  entr_br_vec_i - entr_dist_vec_i)
207  other_player_fx.append(other_player_fx_i)
208 
209  # then construct exploitability gradient
210  grad_dist = []
211  regret_delta = []
212  for i in range(num_players):
213 
214  grad_dist_i = -policy_gradient[i]
215  for j in range(num_players):
216  if j == i:
217  continue
218  if i < j:
219  hess_j_ij = payoff_matrices[(i, j)][1]
220  else:
221  hess_j_ij = payoff_matrices[(j, i)][0].T
222 
223  grad_dist_i += hess_j_ij.dot(other_player_fx[j])
224 
225  regret_delta_i = -(grad_dist_i - grad_dist_i.dot(dist[i]))
226  # regret_delta_i = y[i] - y[i].dot(dist[i])
227 
228  grad_dist.append(grad_dist_i)
229  regret_delta.append(regret_delta_i)
230 
231  return (grad_dist, grad_y, regret_delta), np.mean(unreg_exp), np.mean(reg_exp)
