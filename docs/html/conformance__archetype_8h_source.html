19 // This file contains a facility for generating "archetypes" of out of
20 // "Conformance Profiles" (see "conformance_profiles.h" for more information
21 // about Conformance Profiles). An archetype is a type that aims to support the
22 // bare minimum requirements of a given Conformance Profile. For instance, an
23 // archetype that corresponds to an ImmutableProfile has exactly a nothrow
24 // move-constructor, a potentially-throwing copy constructor, a nothrow
25 // destructor, with all other special-member-functions deleted. These archetypes
26 // are useful for testing to make sure that templates are able to work with the
27 // kinds of types that they claim to support (i.e. that they do not accidentally
28 // under-constrain),
29 //
30 // The main type template in this file is the Archetype template, which takes
31 // a Conformance Profile as a template argument and its instantiations are a
32 // minimum-conforming model of that profile.

389 // An alias to the top of the chain of bases for special-member functions.
390 // NOTE: move_constructible::maybe, move_assignable::maybe, and
391 // destructible::maybe are handled in the top-level type by way of SFINAE.
392 // Because of this, we never instantiate the base classes with
393 // move_constructible::maybe, move_assignable::maybe, or destructible::maybe so
394 // that we minimize the number of different possible type-template
395 // instantiations.

397  move_constructible MoveConstructibleValue,
398  copy_constructible CopyConstructibleValue,
399  move_assignable MoveAssignableValue,
400  copy_assignable CopyAssignableValue, destructible DestructibleValue>

402  DefaultConstructibleValue,
403  MoveConstructibleValue != move_constructible::maybe
404  ? MoveConstructibleValue
405  : move_constructible::nothrow,
406  CopyConstructibleValue,
407  MoveAssignableValue != move_assignable::maybe ? MoveAssignableValue
408  : move_assignable::nothrow,
409  CopyAssignableValue,
410  DestructibleValue != destructible::maybe ? DestructibleValue
411  : destructible::nothrow>;

413 // A function that is used to create an archetype with some associated state.
414 template <class Arch>
415 Arch MakeArchetype(ArchetypeState state) noexcept {
416  static_assert(IsArchetype<Arch>::value,
417  "The explicit template argument to MakeArchetype is required "
418  "to be an Archetype.");
419  return Arch(MakeArchetypeState(), state);
420 }

422 // This is used to conditionally delete "copy" and "move" constructors in a way
423 // that is consistent with what the ConformanceProfile requires and that also
424 // strictly enforces the arguments to the copy/move to not come from implicit
425 // conversions when dealing with the Archetype.
426 template <class Prof, class T>
427 constexpr bool ShouldDeleteConstructor() {
428  return !((PropertiesOfT<Prof>::move_constructible_support !=
429  move_constructible::maybe &&
430  std::is_same<T, Archetype<Prof>>::value) ||
431  (PropertiesOfT<Prof>::copy_constructible_support !=
432  copy_constructible::maybe &&
433  (std::is_same<T, const Archetype<Prof>&>::value ||
434  std::is_same<T, Archetype<Prof>&>::value ||
435  std::is_same<T, const Archetype<Prof>>::value)));
436 }
437 
438 // This is used to conditionally delete "copy" and "move" assigns in a way
439 // that is consistent with what the ConformanceProfile requires and that also
440 // strictly enforces the arguments to the copy/move to not come from implicit
441 // conversions when dealing with the Archetype.
442 template <class Prof, class T>
443 constexpr bool ShouldDeleteAssign() {
444  return !(
445  (PropertiesOfT<Prof>::move_assignable_support != move_assignable::maybe &&
446  std::is_same<T, Archetype<Prof>>::value) ||
447  (PropertiesOfT<Prof>::copy_assignable_support != copy_assignable::maybe &&
448  (std::is_same<T, const Archetype<Prof>&>::value ||
449  std::is_same<T, Archetype<Prof>&>::value ||
450  std::is_same<T, const Archetype<Prof>>::value)));
451 }
