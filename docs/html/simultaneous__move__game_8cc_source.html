28 std::vector<Action> SimMoveState::FlatJointActionToActions(
29  Action flat_action) const {
30  std::vector<Action> actions(num_players_, kInvalidAction);
31  for (Player player = 0; player < num_players_; ++player) {
32  // For each player with legal actions available:
33  const auto legal_actions = LegalActions(player);
34  int num_actions = legal_actions.size();
35  if (num_actions > 0) {
36  // Extract the least-significant digit (radix = the number legal actions
37  // for the current player) from flat_action. Use the digit as an index
38  // into the player's set of legal actions.
39  actions[player] = legal_actions[flat_action % num_actions];
40  // Update the flat_action to be for the remaining players only.
41  flat_action /= num_actions;
42  }
43  }
44  return actions;
45 }

62 std::string SimMoveState::FlatJointActionToString(Action flat_action) const {
63  // Assembles the string for each individual player action into a single
64  // string. For example, [Heads, Tails] would mean than player 0 chooses Heads,
65  // and player 1 chooses Tails.
66  std::string str;
67  for (auto player = Player{0}; player < num_players_; ++player) {
68  if (!str.empty()) str.append(", ");
69  const auto legal_actions = LegalActions(player);
70  int num_actions = legal_actions.size();
71  str.append(
72  ActionToString(player, legal_actions[flat_action % num_actions]));
73  flat_action /= num_actions;
74  }
75  return absl::StrCat("[", str, "]");
76 }
