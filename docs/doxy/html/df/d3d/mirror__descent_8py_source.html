26def softmax_projection(logits):
27 max_l = max(logits)
28 exp_l = [np.exp(l - max_l) for l in logits]
29 norm_exp = sum(exp_l)
30 return [l / norm_exp for l in exp_l]

33class ProjectedPolicy(policy_lib.Policy):
34 """Project values on the policy simplex."""
35
36 def __init__(
37 self,
38 game: pyspiel.Game,
39 player_ids: List[int],
40 state_value: value.ValueFunction,
41 coeff: float = 1.0,
42 ):
43 """Initializes the projected policy.
44
45 Args:
46 game: The game to analyze.
47 player_ids: list of player ids for which this policy applies; each should
48 be in the range 0..game.num_players()-1.
49 state_value: The (cumulative) state value to project.
50 coeff: Coefficient for the values of the states.
51 """
52 super(ProjectedPolicy, self).__init__(game, player_ids)
53 self._state_value = state_value
54 self._coeff = coeff
55
56 def value(self, state: pyspiel.State, action: Optional[int] = None) -> float:
57 if action is None:
58 return self._state_value(
59 state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID))
60 else:
61 new_state = state.child(action)
62 return state.rewards()[0] + self._state_value(
63 new_state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID))
64
65 def action_probabilities(self,
66 state: pyspiel.State,
67 player_id: Optional[int] = None) -> Dict[int, float]:
68 del player_id
69 action_logit = [(a, self._coeff * self.value(state, action=a))
70 for a in state.legal_actions()]
71 action, logit = zip(*action_logit)
72 return dict(zip(action, softmax_projection(logit)))

75class MirrorDescent(object):
76 """The mirror descent algorithm."""
77
78 def __init__(self,
79 game: pyspiel.Game,
80 state_value: Optional[value.ValueFunction] = None,
81 lr: float = 0.01,
82 root_state: Optional[pyspiel.State] = None):
83 """Initializes mirror descent.
84
85 Args:
86 game: The game,
87 state_value: A state value function. Default to TabularValueFunction.
88 lr: The learning rate of mirror descent,
89 root_state: The state of the game at which to start. If `None`, the game
90 root state is used.
91 """
92 self._game = game
93 if root_state is None:
94 self._root_states = game.new_initial_states()
95 else:
96 self._root_states = [root_state]
97 self._policy = policy_lib.UniformRandomPolicy(game)
98 self._distribution = distribution.DistributionPolicy(game, self._policy)
99 self._md_step = 0
100 self._lr = lr
101
102 self._state_value = (
103 state_value if state_value else value.TabularValueFunction(game))
104 self._cumulative_state_value = value.TabularValueFunction(game)
105
106 def get_state_value(self, state: pyspiel.State,
107 learning_rate: float) -> float:
108 """Returns the value of the state."""
109 if state.is_terminal():
110 return state.rewards()[state.mean_field_population()]
111
112 if state.current_player() == pyspiel.PlayerId.CHANCE:
113 v = 0.0
114 for action, prob in state.chance_outcomes():
115 new_state = state.child(action)
116 v += prob * self.eval_state(new_state, learning_rate)
117 return v
118
119 if state.current_player() == pyspiel.PlayerId.MEAN_FIELD:
120 dist_to_register = state.distribution_support()
121 dist = [
122 self._distribution.value_str(str_state, 0.0)
123 for str_state in dist_to_register
124 ]
125 new_state = state.clone()
126 new_state.update_distribution(dist)
127 return (state.rewards()[state.mean_field_population()] +
128 self.eval_state(new_state, learning_rate))
129
130 assert int(state.current_player()) >= 0, "The player id should be >= 0"
131 v = 0.0
132 for action, prob in self._policy.action_probabilities(state).items():
133 new_state = state.child(action)
134 v += prob * self.eval_state(new_state, learning_rate)
135 return state.rewards()[state.mean_field_population()] + v
136
137 def eval_state(self, state: pyspiel.State, learning_rate: float) -> float:
138 """Evaluate the value of a state and update the cumulative sum."""
139 state_str = state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)
140 # Return the already calculated value if present.
141 if self._state_value.has(state_str):
142 return self._state_value(state_str)
143 # Otherwise, calculate the value of the state.
144 v = self.get_state_value(state, learning_rate)
145 self._state_value.set_value(state_str, v)
146 # Update the cumulative value of the state.
147 self._cumulative_state_value.add_value(state_str, learning_rate * v)
148 return v
149
150 def get_projected_policy(self) -> policy_lib.Policy:
151 """Returns the projected policy."""
152 return ProjectedPolicy(self._game, list(range(self._game.num_players())),
153 self._cumulative_state_value)
154
155 def iteration(self, learning_rate: Optional[float] = None):
156 """An iteration of Mirror Descent."""
157 self._md_step += 1
158 # TODO(sertan): Fix me.
159 self._state_value = value.TabularValueFunction(self._game)
160 for state in self._root_states:
161 self.eval_state(state, learning_rate if learning_rate else self._lr)
162 self._policy = self.get_projected_policy()
163 self._distribution = distribution.DistributionPolicy(
164 self._game, self._policy)
165
166 def get_policy(self) -> policy_lib.Policy:
167 return self._policy

171 return self._distribution
