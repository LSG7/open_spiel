278struct contractType
279{
280 int underTricks; /* 0 = make 1-13 = sacrifice */
281 int overTricks; /* 0-3, e.g. 1 for 4S + 1. */
282 int level; /* 1-7 */
283 int denom; /* 0 = No Trumps, 1 = trump Spades, 2 = trump Hearts,
284 3 = trump Diamonds, 4 = trump Clubs */
285 int seats; /* One of the cases N, E, W, S, NS, EW;
286 0 = N 1 = E, 2 = S, 3 = W, 4 = NS, 5 = EW */
287};
