25 def test_ranked_pairs_wikipedia_example(self):
26 alternatives = ["Memphis", "Nashville", "Chattanooga", "Knoxville"]
27 votes = [
28 base.WeightedVote(42,
29 ["Memphis", "Nashville", "Chattanooga", "Knoxville"]),
30 base.WeightedVote(26,
31 ["Nashville", "Chattanooga", "Knoxville", "Memphis"]),
32 base.WeightedVote(15,
33 ["Chattanooga", "Knoxville", "Nashville", "Memphis"]),
34 base.WeightedVote(17,
35 ["Knoxville", "Chattanooga", "Nashville", "Memphis"]),
36 ]
37 profile = base.PreferenceProfile(votes=votes, alternatives=alternatives)
38 method = kemeny_young.KemenyYoungVoting()
39 outcome = method.run_election(profile)
40 self.assertListEqual(outcome.ranking,
41 ["Nashville", "Chattanooga", "Knoxville", "Memphis"])
42 self.assertListEqual(outcome.scores, [194, 141, 58, 0])
