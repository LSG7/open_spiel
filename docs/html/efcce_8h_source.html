29 // This is an EFCCE extended game similar to the one described in von Stengel
30 // and Forges 2008, Definition 2.2. The incentive to deviate to a best response
31 // is computed by running NashConv on this auxiliary game.
32 
33 // The main difference in the EFCCE auxiliary game (from the EFCE game) is that
34 // players must decide whether to accept or reject the recommendation *before*
35 // seeing it. This changes the action space of the game, adding two new actions
36 // that must be taken at each state before the actual decision is made.
37 class EFCCEState : public WrappedState {
38  public:
39  EFCCEState(std::shared_ptr<const Game> game, std::unique_ptr<State> state,
40  CorrDistConfig config, const CorrelationDevice& mu,
41  Action follow_action, Action defect_action);
42 
43  std::unique_ptr<State> Clone() const override {
44  return std::make_unique<EFCCEState>(*this);
45  }
46 
47  // Need to override this because otherwise WrappedState forwards the
48  // implementation to the underlying state, which calls the wrong
49  // ChanceOutcomes
50  std::vector<Action> LegalChanceOutcomes() const override {
51  return State::LegalChanceOutcomes();
52  }
53 
54  Player CurrentPlayer() const override;
55  ActionsAndProbs ChanceOutcomes() const override;
56  std::vector<Action> LegalActions() const override;
57  std::string InformationStateString(Player player) const override;
58  std::string ToString() const override;
59 
60  bool HasDefected(Player player) const;
61 
62  protected:
63  Action CurRecommendation() const;
64  void DoApplyAction(Action action_id) override;
65 
66  private:
67  const CorrDistConfig config_;
68  const CorrelationDevice& mu_;
69 
70  Action follow_action_;
71  Action defect_action_;
72 
73  // Which joint policy was chosen?
74  int rec_index_;
75 
76  // Has the player defected?
77  std::vector<int> defected_;
78 
79  // The sequence of recommendations, indexed by player
80  std::vector<std::vector<Action>> recommendation_seq_;
81 };

83 class EFCCEGame : public WrappedGame {
84  public:
85  EFCCEGame(std::shared_ptr<const Game> game, CorrDistConfig config,
86  const CorrelationDevice& mu)
87  : WrappedGame(game, game->GetType(), game->GetParameters()),
88  config_(config),
89  mu_(mu),
90  orig_num_distinct_actions_(game->NumDistinctActions()) {}
91 
92  std::unique_ptr<State> NewInitialState() const override {
93  return std::make_unique<EFCCEState>(shared_from_this(),
94  game_->NewInitialState(), config_, mu_,
95  FollowAction(), DefectAction());
96  }
97 
98  int NumDistinctActions() const override {
99  // 2 extra actions: cooperate/follow or defect
100  return orig_num_distinct_actions_ + 2;
101  }
102 
103  int FollowAction() const { return orig_num_distinct_actions_; }
104  int DefectAction() const { return orig_num_distinct_actions_ + 1; }
105 
106  private:
107  const CorrDistConfig config_;
108  const CorrelationDevice& mu_;
109 
110  // Number of distinct actions in the original game.
111  int orig_num_distinct_actions_;
112 };
