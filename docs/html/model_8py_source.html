35 def batch_norm(training, updates, name):
36  """A batch norm layer.
37 
38  Args:
39  training: A placeholder of whether this is done in training or not.
40  updates: A list to be extended with this layer's updates.
41  name: Name of the layer.
42 
43  Returns:
44  A function to apply to the previous layer.
45  """
46  bn = tfkl.BatchNormalization(name=name)
47  def batch_norm_layer(x):
48  # This emits a warning that training is a placeholder instead of a concrete
49  # bool, but seems to work anyway.
50  applied = bn(x, training)
51  updates.extend(bn.updates)
52  return applied
53  return batch_norm_layer

102 class Model(object):
103  """An AlphaZero style model with a policy and value head.
104 
105  This supports three types of models: mlp, conv2d and resnet.
106 
107  All models have a shared torso stack with two output heads: policy and value.
108  They have same meaning as in the AlphaGo Zero and AlphaZero papers. The resnet
109  model copies the one in that paper when set with width 256 and depth 20. The
110  conv2d model is the same as the resnet except uses a conv+batchnorm+relu
111  instead of the res blocks. The mlp model uses dense layers instead of conv,
112  and drops batch norm.
113 
114  Links to relevant articles/papers:
115  https://deepmind.com/blog/article/alphago-zero-starting-scratch has an open
116  access link to the AlphaGo Zero nature paper.
117  https://deepmind.com/blog/article/alphazero-shedding-new-light-grand-games-chess-shogi-and-go
118  has an open access link to the AlphaZero science paper.
119 
120  All are parameterized by their input (observation) shape and output size
121  (number of actions), though the conv2d and resnet might only work with games
122  that have spatial data (ie 3 non-batch dimensions, eg: connect four would
123  work, but not poker).
124 
125  The depth is the number of blocks in the torso, where the definition of a
126  block varies by model. For a resnet it's a resblock which is two conv2ds,
127  batch norms and relus, and an addition. For conv2d it's a conv2d, a batch norm
128  and a relu. For mlp it's a dense plus relu.
129 
130  The width is the number of filters for any conv2d and the number of hidden
131  units for any dense layer.
132 
133  Note that this uses an explicit graph so that it can be used for inference
134  and training from C++. It seems to also be 20%+ faster than using eager mode,
135  at least for the unit test.
136  """

211  @staticmethod
212  def _define_graph(model_type, input_shape, output_size,
213  nn_width, nn_depth, weight_decay, learning_rate):
214  """Define the model graph."""
215  # Inference inputs
216  input_size = int(np.prod(input_shape))
217  observations = tf.placeholder(tf.float32, [None, input_size], name="input")
218  legals_mask = tf.placeholder(tf.bool, [None, output_size],
219  name="legals_mask")
220  training = tf.placeholder(tf.bool, name="training")
221 
222  bn_updates = []
223 
224  # Main torso of the network
225  if model_type == "mlp":
226  torso = observations # Ignore the input shape, treat it as a flat array.
227  for i in range(nn_depth):
228  torso = cascade(torso, [
229  tfkl.Dense(nn_width, name=f"torso_{i}_dense"),
230  tfkl.Activation("relu"),
231  ])
232  elif model_type == "conv2d":
233  torso = tfkl.Reshape(input_shape)(observations)
234  for i in range(nn_depth):
235  torso = cascade(torso, [
236  conv_2d(nn_width, 3, name=f"torso_{i}_conv"),
237  batch_norm(training, bn_updates, f"torso_{i}_batch_norm"),
238  tfkl.Activation("relu"),
239  ])
240  elif model_type == "resnet":
241  torso = cascade(observations, [
242  tfkl.Reshape(input_shape),
243  conv_2d(nn_width, 3, name="torso_in_conv"),
244  batch_norm(training, bn_updates, "torso_in_batch_norm"),
245  tfkl.Activation("relu"),
246  ])
247  for i in range(nn_depth):
248  torso = residual_layer(torso, nn_width, 3, training, bn_updates,
249  f"torso_{i}")
250  else:
251  raise ValueError("Unknown model type.")
252 
253  # The policy head
254  if model_type == "mlp":
255  policy_head = cascade(torso, [
256  tfkl.Dense(nn_width, name="policy_dense"),
257  tfkl.Activation("relu"),
258  ])
259  else:
260  policy_head = cascade(torso, [
261  conv_2d(filters=2, kernel_size=1, name="policy_conv"),
262  batch_norm(training, bn_updates, "policy_batch_norm"),
263  tfkl.Activation("relu"),
264  tfkl.Flatten(),
265  ])
266  policy_logits = tfkl.Dense(output_size, name="policy")(policy_head)
267  policy_logits = tf.where(legals_mask, policy_logits,
268  -1e32 * tf.ones_like(policy_logits))
269  unused_policy_softmax = tf.identity(tfkl.Softmax()(policy_logits),
270  name="policy_softmax")
271  policy_targets = tf.placeholder(
272  shape=[None, output_size], dtype=tf.float32, name="policy_targets")
273  policy_loss = tf.reduce_mean(
274  tf.nn.softmax_cross_entropy_with_logits_v2(
275  logits=policy_logits, labels=policy_targets),
276  name="policy_loss")
277 
278  # The value head
279  if model_type == "mlp":
280  value_head = torso # Nothing specific before the shared value head.
281  else:
282  value_head = cascade(torso, [
283  conv_2d(filters=1, kernel_size=1, name="value_conv"),
284  batch_norm(training, bn_updates, "value_batch_norm"),
285  tfkl.Activation("relu"),
286  tfkl.Flatten(),
287  ])
288  value_out = cascade(value_head, [
289  tfkl.Dense(nn_width, name="value_dense"),
290  tfkl.Activation("relu"),
291  tfkl.Dense(1, name="value"),
292  tfkl.Activation("tanh"),
293  ])
294  # Need the identity to name the single value output from the dense layer.
295  value_out = tf.identity(value_out, name="value_out")
296  value_targets = tf.placeholder(
297  shape=[None, 1], dtype=tf.float32, name="value_targets")
298  value_loss = tf.identity(tf.losses.mean_squared_error(
299  value_out, value_targets), name="value_loss")
300 
301  l2_reg_loss = tf.add_n([
302  weight_decay * tf.nn.l2_loss(var)
303  for var in tf.trainable_variables()
304  if "/bias:" not in var.name
305  ], name="l2_reg_loss")
306 
307  total_loss = policy_loss + value_loss + l2_reg_loss
308  optimizer = tf.train.AdamOptimizer(learning_rate)
309  with tf.control_dependencies(bn_updates):
310  unused_train = optimizer.minimize(total_loss, name="train")
