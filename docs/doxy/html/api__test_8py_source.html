451def _assert_is_perfect_recall(game):
452 """Raises an AssertionError if the game is not perfect recall.
453
454 We are willing to ensure the following property (perfect recall):
455 - define X_i(h) be the sequence of information states and actions from the
456 start of the game observed by player i (i.e. excluding the states and
457 actions taken by the opponents unless those actions are included in player
458 i's information state), along h but not including the state at h:
459 X_i(h) = (s_1, a_1), (s_2, a_2), ... , (s_{t-1}, a_{t-1})
460 then player i has perfect recall in this game iff: forall s in S_i,
461 forall h1, h2 in s X_i(h1) == X_i(h2). Here, we check that the game has
462 perfect recall if this is true for all players i (excluding chance).
463
464 For more detail and context, see page 11 of
465 http://mlanctot.info/files/papers/PhD_Thesis_MarcLanctot.pdf
466
467 In particular, note that:
468 - we want to check that holds both for
469 + `std::string information_state_string(current_player)`
470 + `information_state_tensor`.
471 - we check that currently only from the point of view of the current
472 player at the information state (i.e. we compare
473 `prev_state.information_state_string(current_player)` but not
474 `prev_state.information_state_string(opponent_player)`
475
476 The strategy is the following: we traverse the full tree (of states, not
477 infostates), and make sure for each node that the history we get for
478 the infostate associated to that node, that is is unique with respect to
479 the infostate.
480
481 Args:
482 game: A Spiel game to check.
483
484 Returns:
485 The internal cache mapping (infostate_str, player_id) to a list of one
486 history leading to this infostate.
487 """
488 game_info = game.get_type()
489 if game_info.dynamics != pyspiel.GameType.Dynamics.SEQUENTIAL:
490 raise ValueError("The game is expected to be sequential")
491
492 infostate_player_to_history = {}
493 _assert_is_perfect_recall_recursive(
494 game.new_initial_state(),
495 current_history=[],
496 infostate_player_to_history=infostate_player_to_history)
497
498 return infostate_player_to_history
499

501def _assert_is_perfect_recall_recursive(state, current_history,
502 infostate_player_to_history):
503 """Raises an AssertionError if the game is not perfect recall.
504
505 The strategy is the following: we traverse the full tree (of states, not
506 infostates), and make sure for each node that the history we get for
507 the infostate associated to that node, that is is unique with respect to
508 the infostate.
509
510 Args:
511 state: The current state during the recursive tree traversal.
512 current_history: The current list of strictly preceding `SpielState` objects
513 that lead to the current `state` (excluded).
514 infostate_player_to_history: A dictionnary mapping (infostate string
515 representation, current_player) to the list of one instance of actual
516 predecessor nodes.
517 """
518
519 if not state.is_chance_node() and not state.is_terminal():
520 current_player = state.current_player()
521 infostate_str = state.information_state_string(current_player)
522 key = (infostate_str, current_player)
523
524 if key not in infostate_player_to_history:
525 # First time we see the node.
526 infostate_player_to_history[key] = list(current_history)
527 else:
528 previous_history = infostate_player_to_history[key]
529
530 if len(previous_history) != len(current_history):
531 raise AssertionError("We found 2 history leading to the same state:\n"
532 "State: {!r}\n"
533 "InfoState str: {}\n"
534 "First history ({} states): {!r}\n"
535 "Second history ({} states): {!r}\n".format(
536 state.history(), infostate_str,
537 len(previous_history),
538 "|".join([str(sa) for sa in previous_history]),
539 len(current_history),
540 "|".join([str(sa) for sa in current_history])))
541
542 # Check for `information_state`
543 # pylint: disable=g-complex-comprehension
544 expected_infosets_history = [(s.information_state_string(current_player),
545 a)
546 for s, a in previous_history
547 if s.current_player() == current_player]
548 # pylint: disable=g-complex-comprehension
549 infosets_history = [(s.information_state_string(current_player), a)
550 for s, a in current_history
551 if s.current_player() == current_player]
552
553 if expected_infosets_history != infosets_history:
554 # pyformat: disable
555 raise AssertionError("We found 2 history leading to the same state:\n"
556 "history: {!r}\n"
557 "info_state str: {}\n"
558 "**First history ({} states)**\n"
559 "states: {!r}\n"
560 "info_sets: {!r}\n"
561 "**Second history ({} states)**\n"
562 "Second info_state history: {!r}\n"
563 "Second history: {!r}\n".format(
564 state.history(),
565 infostate_str,
566 len(previous_history),
567 "|".join([str(sa) for sa in previous_history]),
568 expected_infosets_history,
569 len(current_history), infosets_history,
570 "|".join([str(sa) for sa in current_history])))
571 # pyformat: enable
572
573 # Check for `information_state_tensor`
574 expected_infosets_history = [
575 (s.information_state_tensor(s.current_player()), a)
576 for s, a in previous_history
577 if s.current_player() == current_player
578 ]
579 infosets_history = [(s.information_state_tensor(s.current_player()), a)
580 for s, a in current_history
581 if s.current_player() == current_player]
582
583 if infosets_history != expected_infosets_history:
584 raise ValueError("The history as tensor in the same infoset "
585 "are different:\n"
586 "History: {!r}\n".format(state.history()))
587
588 # Recursion
589
590 # TODO(author2): We often use a `give me the next node` function and we
591 # probably want a utility method for that, which works for all games.
592 if state.is_terminal():
593 return
594 else:
595 for action in state.legal_actions():
596 state_for_search = state.child(action)
597 _assert_is_perfect_recall_recursive(
598 state_for_search,
599 current_history=current_history + [(state, action)],
600 infostate_player_to_history=infostate_player_to_history)
