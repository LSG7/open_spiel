31 // Counts the number of states in the game according to various measures.
32 // - histories is a sequence of moves (for all players) and chance outcomes
33 // - states is for imperfect information games, information states (i.e.
34 // sets of histories which are indistinguishable to the acting player);
35 // for example in poker, the acting player's private cards plus the sequence
36 // of bets and public cards, for perfect information games, Markov states
37 // (i.e. sets of histories which yield the same result with the same actions
38 // applied), e.g. in tic-tac-toe the current state of the board, regardless
39 // of the order in which the moves were played.
40 int main(int argc, char** argv) {
41  for (const std::string& game_name :
42  {std::string("tic_tac_toe"), std::string("kuhn_poker"),
43  std::string("leduc_poker"), std::string("liars_dice"),
44  TurnBasedGoofspielGameString(4), TurnBasedGoofspielGameString(5),
45  TurnBasedGoofspielGameString(6)}) {
46  std::shared_ptr<const open_spiel::Game> game =
47  LoadGame(std::string(game_name));
48  std::vector<std::unique_ptr<open_spiel::State>> all_histories =
49  GetAllHistories(*game, /*depth_limit=*/-1, /*include_terminals=*/true,
50  /*include_chance_states=*/true);
51  absl::flat_hash_set<std::string> nonterminal_states;
52  absl::flat_hash_set<std::string> terminal_states;
53  const int num_histories = all_histories.size();
54  int num_terminal_histories = 0;
55  int num_chance_nodes = 0;
56  for (const auto& state : all_histories) {
57  switch (state->GetType()) {
58  case StateType::kDecision:
59  if (game->GetType().information ==
60  GameType::Information::kPerfectInformation) {
61  nonterminal_states.insert(state->ToString());
62  } else {
63  nonterminal_states.insert(state->InformationStateString());
64  }
65  break;
66  case StateType::kTerminal:
67  ++num_terminal_histories;
68  terminal_states.insert(state->ToString());
69  break;
70  case StateType::kChance:
71  ++num_chance_nodes;
72  break;
73  case StateType::kMeanField:
74  open_spiel::SpielFatalError("kMeanField not handeled.");
75  }
76  }
77  const int num_nonterminal_states = nonterminal_states.size();
78  const int num_terminal_states = terminal_states.size();
79  std::cout << "Game: " << game_name
80  << ", num_histories: " << num_histories
81  << ", num_terminal_histories: " << num_terminal_histories
82  << ", num_chance_nodes: " << num_chance_nodes
83  << ", num_nonterminal_states: " << num_nonterminal_states
84  << ", num_terminal_states: " << num_terminal_states
85  << std::endl;
86  }
87 }
