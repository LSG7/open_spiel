37  def test_consistent(self):
38  """Checks the Python and C++ game implementations are the same."""
39  py_game = pyspiel.load_game("python_kuhn_poker")
40  cc_game = pyspiel.load_game("kuhn_poker")
41  obs_types = [None, pyspiel.IIGObservationType(perfect_recall=True)]
42  py_observations = [make_observation(py_game, o) for o in obs_types]
43  cc_observations = [make_observation(cc_game, o) for o in obs_types]
44  py_states = get_all_states(py_game)
45  cc_states = get_all_states(cc_game)
46  self.assertCountEqual(list(cc_states), list(py_states))
47  for key, cc_state in cc_states.items():
48  py_state = py_states[key]
49  np.testing.assert_array_equal(py_state.history(), cc_state.history())
50  np.testing.assert_array_equal(py_state.returns(), cc_state.returns())
51  for py_obs, cc_obs in zip(py_observations, cc_observations):
52  for player in (0, 1):
53  py_obs.set_from(py_state, player)
54  cc_obs.set_from(cc_state, player)
55  np.testing.assert_array_equal(py_obs.tensor, cc_obs.tensor)

57  def test_nash_value_sequence_form_lp(self):
