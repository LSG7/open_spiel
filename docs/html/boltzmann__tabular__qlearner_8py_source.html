54  def _softmax(self, info_state, legal_actions, temperature):
55  """Action selection based on boltzmann probability interpretation of Q-values.
56 
57  For more details, see equation (2) page 2 in
58  https://arxiv.org/pdf/1109.1528.pdf
59 
60  Args:
61  info_state: hashable representation of the information state.
62  legal_actions: list of actions at `info_state`.
63  temperature: temperature used for softmax.
64 
65  Returns:
66  A valid soft-max selected action and valid action probabilities.
67  """
68  probs = np.zeros(self._num_actions_num_actions)
69 
70  if temperature > 0.0:
71  probs += [
72  np.exp((1 / temperature) * self._q_values_q_values[info_state][i])
73  for i in range(self._num_actions_num_actions)
74  ]
75  probs /= np.sum(probs)
76  else:
77  # Temperature = 0 causes normal greedy action selection
78  greedy_q = max([self._q_values_q_values[info_state][a] for a in legal_actions])
79  greedy_actions = [
80  a for a in legal_actions if self._q_values_q_values[info_state][a] == greedy_q
81  ]
82 
83  probs[greedy_actions] += 1 / len(greedy_actions)
84 
85  action = np.random.choice(range(self._num_actions_num_actions), p=probs)
86  return action, probs

89  """Returns a selected action and the probabilities of legal actions."""
90  return self._softmax_softmax(info_state, legal_actions, temperature=epsilon)
