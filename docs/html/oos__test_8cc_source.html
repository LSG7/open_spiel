24 // A helper to create a zipped vector from two vectors.
25 // The resulting vector has the size of xs, possibly omitting any longer ys.
26 template <typename X, typename Y>
27 std::vector<std::pair<X, Y>> Zip(const std::vector<X>& xs,
28  const std::vector<Y>& ys) {
29  SPIEL_CHECK_LE(xs.size(), ys.size());
30  std::vector<std::pair<X, Y>> zipped;
31  zipped.reserve(xs.size());
32  for (int i = 0; i < xs.size(); ++i) {
33  zipped.emplace_back(std::make_pair(xs[i], ys[i]));
34  }
35  return zipped;
36 }

44 void EpsExploreSamplingPolicyTest() {
45  std::shared_ptr<const Game> game = LoadGame("kuhn_poker");
46 
47  std::unique_ptr<State> card_to_player0 = game->NewInitialState();
48  SPIEL_CHECK_EQ(card_to_player0->CurrentPlayer(), kChancePlayerId);
49  std::unique_ptr<State> card_to_player1 = card_to_player0->Child(0);
50  SPIEL_CHECK_EQ(card_to_player1->CurrentPlayer(), kChancePlayerId);
51  std::unique_ptr<State> player0_plays = card_to_player1->Child(0);
52  SPIEL_CHECK_EQ(player0_plays->CurrentPlayer(), 0);
53  std::unique_ptr<State> player1_plays = player0_plays->Child(0);
54  SPIEL_CHECK_EQ(player1_plays->CurrentPlayer(), 1);
55 
56  auto chn_3cards_dist = ZipActionsProbs(card_to_player0->LegalActions(),
57  {1 / 3., 1 / 3., 1 / 3.});
58  auto chn_2cards_dist =
59  ZipActionsProbs(card_to_player1->LegalActions(), {1 / 2., 1 / 2.});
60  auto player0_dist =
61  ZipActionsProbs(player0_plays->LegalActions(), {1 / 2., 1 / 2.});
62  auto player1_dist =
63  ZipActionsProbs(player1_plays->LegalActions(), {1 / 2., 1 / 2.});
64 
65  std::vector<double> current_policy = {0.2, 0.8};
66  auto expected_mix = ZipActionsProbs(player0_plays->LegalActions(),
67  {
68  0.4 * 0.5 + 0.6 * current_policy[0],
69  0.4 * 0.5 + 0.6 * current_policy[1],
70  });
71 
72  OOSInfoStateValuesTable table;
73  auto pl0_info_state = player0_plays->InformationStateString();
74  auto pl1_info_state = player1_plays->InformationStateString();
75  table[pl0_info_state] = CFRInfoStateValues(player0_plays->LegalActions());
76  table[pl1_info_state] = CFRInfoStateValues(player1_plays->LegalActions());
77  table[pl0_info_state].current_policy = current_policy;
78  table[pl1_info_state].current_policy = current_policy;
79 
80  auto p = ExplorativeSamplingPolicy(table, 0.4);
81  SPIEL_CHECK_EQ(p.GetStatePolicy(*card_to_player0), chn_3cards_dist);
82  SPIEL_CHECK_EQ(p.GetStatePolicy(*card_to_player1), chn_2cards_dist);
83  SPIEL_CHECK_EQ(p.GetStatePolicy(*player0_plays), expected_mix);
84  SPIEL_CHECK_EQ(p.GetStatePolicy(*player1_plays), expected_mix);
85 }

87 std::vector<std::unique_ptr<State>> CollectStatesInGame(
88  std::shared_ptr<const Game> game) {
89  std::vector<std::unique_ptr<State>> state_collection;
90 
91  std::function<void(State*)> walk = [&](State* s) {
92  for (auto action : s->LegalActions()) {
93  auto child = s->Child(action);
94  walk(child.get());
95  state_collection.push_back(std::move(child));
96  }
97  };
98 
99  auto root_state = game->NewInitialState();
100  walk(root_state.get());
101  state_collection.push_back(std::move(root_state));
102  return state_collection;
103 }

121 void BiasedIterationsConverge(std::shared_ptr<const Game> game, int iterations,
122  double expl_bound, int max_test_states = 100) {
123  // Check that we can target any state in the game.
124  std::vector<std::unique_ptr<State>> states = CollectStatesInGame(game);
125  for (int i = 0; i < std::fmin(states.size(), max_test_states); i++) {
126  // Action-Observation history targeting:
127  for (int player = 0; player < game->NumPlayers(); player++) {
128  auto alg = OOSAlgorithm(game);
129  auto policy = alg.AveragePolicy();
130 
131  ActionObservationHistory aoh(player, *states[i]);
132  std::cout << "Targeting " << aoh << " with " << iterations << " iters.\n";
133  alg.RunTargetedIterations(aoh, iterations);
134  double actual_expl = Exploitability(*game, *policy);
135 
136  std::cout << alg.GetStats();
137  std::cout << "Exploitability: " << actual_expl << "\n";
138  std::cout << "----" << std::endl;
139  SPIEL_CHECK_LT(actual_expl, expl_bound);
140  alg.GetStats().CheckConsistency();
141  }
142 
143  // Public-Observation history targeting:
144  {
145  auto alg = OOSAlgorithm(game);
146  auto policy = alg.AveragePolicy();
147 
148  PublicObservationHistory poh(*states[i]);
149  std::cout << "Targeting " << poh << " with " << iterations << " iters.\n";
150  alg.RunTargetedIterations(poh, iterations);
151  double actual_expl = Exploitability(*game, *policy);

153  std::cout << alg.GetStats();
154  std::cout << "Exploitability: " << actual_expl << "\n";
155  std::cout << "----" << std::endl;

157  alg.GetStats().CheckConsistency();
158  }
159  }
