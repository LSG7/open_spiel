96 void ConcreteActionsAreUsedInTheAPI() {
97  std::shared_ptr<const Game> game = LoadGame(
98  "phantom_go", {{"board_size", open_spiel::GameParameter(kBoardSize)}});
99  std::unique_ptr<State> state = game->NewInitialState();
100 
101  SPIEL_CHECK_EQ(state->NumDistinctActions(), kBoardSize * kBoardSize + 1);
102  SPIEL_CHECK_EQ(state->LegalActions().size(), state->NumDistinctActions());
103  for (Action action : state->LegalActions()) {
104  SPIEL_CHECK_GE(action, 0);
105  SPIEL_CHECK_LE(action, kBoardSize * kBoardSize);
106  }
107 }
