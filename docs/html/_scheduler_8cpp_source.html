40 void Scheduler::InitHighCards()
41 {
42  // highCards[i] is a point value of a given suit holding i.
43  // This can be HCP, for instance. Currently it is close to
44  // 6 - 4 - 2 - 1 - 0.5 for A-K-Q-J-T, but with 6.5 for the ace
45  // in order to make the sum come out to 28, an even number, so
46  // that the average number is an integer.
47 
48  highCards.resize(1 << 13);
49  const unsigned pA = 1 << 12;
50  const unsigned pK = 1 << 11;
51  const unsigned pQ = 1 << 10;
52  const unsigned pJ = 1 << 9;
53  const unsigned pT = 1 << 8;
54 
55  for (unsigned suit = 0; suit < (1 << 13); suit++)
56  {
57  int j = 0;
58  if (suit & pA) j += 13;
59  if (suit & pK) j += 8;
60  if (suit & pQ) j += 4;
61  if (suit & pJ) j += 2;
62  if (suit & pT) j += 1;
63  highCards[suit] = j;
64  }
65 }

182 void Scheduler::MakeGroups(const boards& bds)
183 {
184  deal const * dl;
185  listType * lp;
186 
187  for (int b = 0; b < numHands; b++)
188  {
189  dl = &bds.deals[b];
190 
191  int strain = dl->trump;
192 
193  unsigned dlXor =
194  dl->remainCards[0][0] ^
195  dl->remainCards[1][1] ^
196  dl->remainCards[2][2] ^
197  dl->remainCards[3][3];
198 
199  int key = static_cast<int>(((dlXor >> 2) ^ (dlXor >> 6)) & 0x7f);
200 
201  hands[b].spareKey = static_cast<int>(
202  (dl->remainCards[1][0] << 17) ^
203  (dl->remainCards[2][1] << 11) ^
204  (dl->remainCards[3][2] << 5) ^
205  (dl->remainCards[0][3] >> 2));
206 
207  for (int h = 0; h < DDS_HANDS; h++)
208  for (int s = 0; s < DDS_SUITS; s++)
209  hands[b].remainCards[h][s] = dl->remainCards[h][s];
210 
211  hands[b].NTflag = (strain == 4 ? 1 : 0);
212  hands[b].first = dl->first;
213  hands[b].strain = strain;
214  hands[b].fanout = Scheduler::Fanout(* dl);
215  // hands[b].strength = Scheduler::Strength(* dl);
216 
217  lp = &list[strain][key];
218 
219  if (lp->first == -1)
220  {
221  lp->first = b;
222  lp->last = b;
223  lp->length = 1;
224 
225  group[numGroups].strain = strain;
226  group[numGroups].hash = key;
227  numGroups++;
228  }
229  else
230  {
231  int l = lp->last;
232  hands[l].next = b;
233 
234  lp->last = b;
235  lp->length++;
236  }
237  }
238 }

241 void Scheduler::FinetuneGroups()
242 {
243  listType * lp;
244  int strain, key, b1, b2;
245  int numGroupsOrig = numGroups;
246 
247  for (int g = 0; g < numGroupsOrig; g++)
248  {
249  strain = group[g].strain;
250  key = group[g].hash;
251 
252  lp = &list[strain][key];
253 
254  if (lp->length == 1)
255  continue;
256 
257  else if (lp->length == 2)
258  {
259  // This happens quite often, so worth optimizing.
260 
261  b1 = lp->first;
262  b2 = hands[lp->first].next;
263 
264  bool match = false;
265  if (hands[b1].spareKey == hands[b2].spareKey)
266  {
267  // It is now extremely likely that it is a repeat hand,
268  // but we have to be sure.
269  match = true;
270  for (int h = 0; h < DDS_HANDS && match; h++)
271  for (int s = 0; s < DDS_SUITS && match; s++)
272  if (hands[b1].remainCards[h][s] != hands[b2].remainCards[h][s])
273  match = false;
274  }
275 
276  if (match)
277  continue;
278 
279  // Leave the first hand in place.
280  hands[lp->first].next = -1;
281  lp->last = lp->first;
282  lp->length = 1;
283 
284  // Move the second hand to the special list.
285  lp = &list[5][extraGroups];
286 
287  lp->first = b2;
288  lp->last = b2;
289  lp->length = 1;
290 
291  group[numGroups].strain = 5;
292  group[numGroups].hash = extraGroups;
293 
294  numGroups++;
295  extraGroups++;
296  }
297 
298  else
299  {
300  // This is the general case. The comparison is not quite
301  // as thorough here, but it's better than above and it uses
302  // a different hand.
303 
304  sortType st;
305  sortLen = lp->length;
306  int index = lp->first;
307 
308  for (int i = 0; i < sortLen; i++)
309  {
310  sortList[i].number = index;
311  sortList[i].value = hands[index].spareKey;
312 
313  index = hands[index].next;
314  }
315 
316  // Sort the list heuristically by spareKey value.
317 
318  for (int i = 1; i < sortLen; i++)
319  {
320  st = sortList[i];
321  int j = i;
322  for (; j && st.value > sortList[j - 1].value; --j)
323  sortList[j] = sortList[j - 1];
324  sortList[j] = st;
325  }
326 
327  // First group stays where it is, but shorter and rejigged.
328  // From here on, hand comparisons are completely rigorous.
329  // We might miss duplicates, but we won't let different
330  // hands through as belonging to the same group.
331 
332  int l = 0;
333  while (l < sortLen-1 &&
334  Scheduler::SameHand(sortList[l].number, sortList[l+1].number))
335  l++;
336 
337  if (l == sortLen-1)
338  continue;
339 
340  lp->first = sortList[0].number;
341  lp->last = sortList[l].number;
342  lp->length = l + 1;
343 
344  index = lp->first;
345 
346  for (int i = 0; i < l; i++)
347  {
348  hands[index].next = sortList[i + 1].number;
349  index = hands[index].next;
350  }
351 
352  hands[index].next = -1;
353 
354  // The rest is moved to special groups.
355  l++;
356 
357  while (l < sortLen)
358  {
359  if (Scheduler::SameHand(sortList[l].number, sortList[l-1].number))
360  {
361  // Same group
362  int nOld = sortList[l - 1].number;
363  int nNew = sortList[l].number;
364  hands[nOld].next = nNew;
365  hands[nNew].next = -1;
366 
367  lp->last = nNew;
368  lp->length++;
369  }
370  else
371  {
372  // New group
373  int n = sortList[l].number;
374  hands[n].next = -1;
375 
376  lp = &list[5][extraGroups];
377  lp->first = n;
378  lp->last = n;
379  lp->length = 1;
380 
381  group[numGroups].strain = 5;
382  group[numGroups].hash = extraGroups;
383 
384  numGroups++;
385  extraGroups++;
386  }
387  l++;
388  }
389  }
390  }
391 }

434 void Scheduler::SortSolve()
435 {
436  listType * lp;
437  handType * hp;
438  int strain, key, index;
439 
440  for (int g = 0; g < numGroups; g++)
441  {
442  strain = group[g].strain;
443  key = group[g].hash;
444  lp = &list[strain][key];
445  index = lp->first;
446  hp = &hands[index];
447 
448  // Taking into account repeat times saves 1-2%.
449 
450  int repeatNo = 0;
451  int firstPrev = -1;
452  group[g].pred = 0;
453  do
454  {
455  // Skip complete duplicates, as we won't solve them again.
456  if (hands[index].first != firstPrev)
457  {
458  group[g].pred += SORT_SOLVE_TIMES[hp->NTflag][repeatNo];
459  if (repeatNo < 7)
460  repeatNo++;
461  firstPrev = hands[index].first;
462  }
463 
464  index = hands[index].next;
465  }
466  while (index != -1);
467 
468  // Taking into account fanout saves 4-6%.
469 
470  int fanout = hp->fanout;
471  double * slist = SORT_SOLVE_FANOUT[hp->NTflag];
472  double fanoutFactor;
473 
474  if (fanout < slist[0])
475  fanoutFactor = 0.; // A bit extreme...
476  else if (fanout < slist[1])
477  fanoutFactor = slist[2] * (fanout - slist[0]);
478  else
479  fanoutFactor = slist[3] * exp( (fanout - slist[1]) / slist[4] );
480 
481  group[g].pred = static_cast<int>(
482  (fanoutFactor * static_cast<double>(group[g].pred)));
483  }
484 
485  // Sort groups using merge sort.
486  groupType gp;
487  for (int g = 0; g < numGroups; g++)
488  {
489  gp = group[g];
490  int j = g;
491  for (; j && gp.pred > group[j - 1].pred; --j)
492  group[j] = group[j - 1];
493  group[j] = gp;
494  }
495 }

666 int Scheduler::Strength(const deal& dl) const
667 {
668  // If the strength in all suits is evenly split, then the
669  // "strength" returned is close to 0. Maximum is 49.
670 
671  const unsigned sp = (dl.remainCards[0][0] | dl.remainCards[2][0]) >> 2;
672  const unsigned he = (dl.remainCards[0][1] | dl.remainCards[2][1]) >> 2;
673  const unsigned di = (dl.remainCards[0][2] | dl.remainCards[2][2]) >> 2;
674  const unsigned cl = (dl.remainCards[0][3] | dl.remainCards[2][3]) >> 2;
675 
676  const int hsp = highCards[sp];
677  const int hhe = highCards[he];
678  const int hdi = highCards[di];
679  const int hcl = highCards[cl];
680 
681  int dev = (hsp >= 14 ? hsp - 14 : 14 - hsp) +
682  (hhe >= 14 ? hhe - 14 : 14 - hhe) +
683  (hdi >= 14 ? hdi - 14 : 14 - hdi) +
684  (hcl >= 14 ? hcl - 14 : 14 - hcl);
685 
686  if (dev >= 50) dev = 49;
687 
688  return dev;
689 }

692 int Scheduler::Fanout(const deal& dl) const
693 {
694  // The fanout for a given suit and a given player is the number
695  // of bit groups, so KT982 has 3 groups. In a given suit the
696  // maximum number over all four players is 13.
697  // A void counts as the sum of the other players' groups.
698 
699  int fanout = 0;
700  int fanoutSuit, numVoids, c;
701 
702  for (int h = 0; h < DDS_HANDS; h++)
703  {
704  fanoutSuit = 0;
705  numVoids = 0;
706  for (int s = 0; s < DDS_SUITS; s++)
707  {
708  c = static_cast<int>(dl.remainCards[h][s] >> 2);
709  fanoutSuit += groupData[c].lastGroup + 1;
710  if (c == 0)
711  numVoids++;
712  }
713  fanoutSuit += numVoids * fanoutSuit;
714  fanout += fanoutSuit;
715  }
716 
717  return fanout;
718 }

721 schedType Scheduler::GetNumber(const int thrId)
722 {
723  const unsigned tu = static_cast<unsigned>(thrId);
724  int g = threadGroup[tu];
725  listType * lp;
726  schedType st;
727 
728  if (g == -1)
729  {
730  // Find a new group
731 
732  if (currGroup >= numGroups - 1)
733  {
734  // Out of groups. Just an optimization not to touch the
735  // shared variable unnecessarily.
736  st.number = -1;
737  return st;
738  }
739 
740  // Atomic.
741  g = ++currGroup;
742 
743  if (g >= numGroups)
744  {
745  // Out of groups. currGroup could have changed in the
746  // meantime in another thread, so test again.
747 
748  st.number = -1;
749  return st;
750  }
751 
752  // A bit inelegant to duplicate this, but seems better than
753  // the alternative, as threadGroup must get set to -1 in some
754  // cases.
755  threadGroup[tu] = g;
756  threadCurrGroup[tu] = g;
757  group[g].repeatNo = 0;
758  group[g].actual = 0;
759  }
760 
761  // Continue with existing or new group
762 
763  int strain = group[g].strain;
764  int key = group[g].hash;
765 
766  lp = &list[strain][key];
767  st.number = lp->first;
768  lp->first = hands[lp->first].next;
769 
770  if (group[g].repeatNo == 0)
771  {
772  group[g].head = st.number;
773  st.repeatOf = -1;
