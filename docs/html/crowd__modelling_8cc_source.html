59 std::string StateToString(int x, int t, Player player_id, bool is_chance_init) {
60  if (is_chance_init) {
61  return "initial";
62  }
63  if (player_id == 0) {
64  return absl::Substitute("($0, $1)", x, t);
65  }
66  if (player_id == kMeanFieldPlayerId) {
67  return absl::Substitute("($0, $1)_a", x, t);
68  }
69  if (player_id == kChancePlayerId) {
70  return absl::Substitute("($0, $1)_a_mu", x, t);
71  }
72  SpielFatalError(absl::Substitute(
73  "Unexpected state (player_id: $0, is_chance_init: $1)",
74  player_id, is_chance_init));
75 }

113 ActionsAndProbs CrowdModellingState::ChanceOutcomes() const {
114  if (is_chance_init_) {
115  ActionsAndProbs outcomes;
116  for (int i = 0; i < size_; ++i) {
117  outcomes.push_back({i, 1. / size_});

124 void CrowdModellingState::DoApplyAction(Action action) {
125  SPIEL_CHECK_NE(current_player_, kMeanFieldPlayerId);
126  return_value_ += Rewards()[0];
127  if (is_chance_init_) {
128  SPIEL_CHECK_GE(action, 0);
129  SPIEL_CHECK_LT(action, size_);
130  SPIEL_CHECK_EQ(current_player_, kChancePlayerId);
131  x_ = action;
132  is_chance_init_ = false;
133  current_player_ = 0;
134  } else if (current_player_ == kChancePlayerId) {
135  x_ = (x_ + kActionToMove.at(action) + size_) % size_;
136  ++t_;
137  current_player_ = kMeanFieldPlayerId;
138  } else {
139  SPIEL_CHECK_EQ(current_player_, 0);
140  x_ = (x_ + kActionToMove.at(action) + size_) % size_;
141  last_action_ = action;
142  current_player_ = kChancePlayerId;
143  }
144 }

173 std::vector<double> CrowdModellingState::Rewards() const {
174  if (current_player_ != 0) {
175  return {0.};
176  }
177  double r_x = 1 - 1.0 * std::abs(x_ - size_ / 2) / (size_ / 2);
178  double r_a = -1.0 * std::abs(kActionToMove.at(last_action_)) / size_;
179  double r_mu = -std::log(distribution_[x_]+kEpsilon);
180  return {r_x + r_a + r_mu};
181 }

203 void CrowdModellingState::ObservationTensor(Player player,
204  absl::Span<float> values) const {
205  SPIEL_CHECK_GE(player, 0);
206  SPIEL_CHECK_LT(player, num_players_);
207  SPIEL_CHECK_EQ(values.size(), size_ + horizon_ + 1);
208  SPIEL_CHECK_LT(x_, size_);
209  SPIEL_CHECK_GE(t_, 0);
210  // Allow t_ == horizon_.
211  SPIEL_CHECK_LE(t_, horizon_);
212  std::fill(values.begin(), values.end(), 0.);
213  if (x_ >= 0) {
214  values[x_] = 1.;
215  }
216  // x_ equals -1 for the initial (blank) state, don't set any
217  // position bit in that case.
218  values[size_ + t_] = 1.;
219 }

243 std::unique_ptr<State> CrowdModellingGame::DeserializeState(
244  const std::string& str) const {
245  std::vector<std::string> lines = absl::StrSplit(str, '\n');
246  if (lines.size() != 2) {
247  SpielFatalError(absl::StrCat("Expected 2 lines in serialized state, got: ",
248  lines.size()));
249  }
250  Player current_player;
251  int is_chance_init;
252  int x;
253  int t;
254  int last_action;
255  double return_value;
256  std::vector<std::string> properties = absl::StrSplit(lines[0], ',');
257  if (properties.size() != 6) {
258  SpielFatalError(
259  absl::StrCat("Expected 6 properties for serialized state, got: ",
260  properties.size()));
261  }
262  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[0], &current_player));
263  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[1], &is_chance_init));
264  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[2], &x));
265  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[3], &t));
266  SPIEL_CHECK_TRUE(absl::SimpleAtoi(properties[4], &last_action));
267  SPIEL_CHECK_TRUE(absl::SimpleAtod(properties[5], &return_value));
268  std::vector<std::string> serialized_distrib = absl::StrSplit(lines[1], ',');
269  std::vector<double> distribution;
270  distribution.reserve(serialized_distrib.size());
271  for (std::string& v : serialized_distrib) {
272  double parsed_weight;
273  SPIEL_CHECK_TRUE(absl::SimpleAtod(v, &parsed_weight));
274  distribution.push_back(parsed_weight);
275  }
276  return absl::make_unique<CrowdModellingState>(
277  shared_from_this(), size_, horizon_, current_player, is_chance_init, x, t,
278  last_action, return_value, distribution);
279 }
