76def sym(pt):
77 """Symmetrize stack of payoff tensors (stacked along first dimension).
78
79 A payoff tensor can be `symmetrized' by averaging over all possible
80 permutations of the players. This means permuting the axes corresponding to
81 the player strategies as well as the payoffs assigned to the players. E.g.,
82 player A playing strategy 1 and player B playing strategy 3 is no different
83 from player A playing strategy 3 and player B playing strategy 1 in a
84 symmetric game. Note we permuted the strategies, but we must also permute the
85 payoffs.
86
87 Args:
88 pt: tensor of shape: (num_players,) + (num_strategies,) * num_players
89 Returns:
90 pt_sym: symmetrized payoff tensor of same shape
91 """
92 num_players = len(pt.shape[1:])
93 num_perms = math.factorial(num_players)
94 pt_sym = np.zeros_like(pt)
95 for _, perm_players in enumerate(itertools.permutations(range(num_players))):
96 perm_axes = tuple([pi + 1 for pi in perm_players])
97 permuted_tensor = np.transpose(pt, (0,) + perm_axes)[list(perm_players)]
98 pt_sym += permuted_tensor / float(num_perms)
99 return pt_sym

102def random_policy(rnd, state):
103 # all actions are legal for now
104 rnd_action = tuple([rnd.choice(a) for a in state.num_actions])
105 return np.ravel_multi_index(rnd_action, state.num_actions)

108def fixed_prompt_policy(rnd, state, prompt_action_dict):
109 # all actions are legal for now
110 action = [rnd.choice(a) for a in state.num_actions]
111 for prompt_key, prompt_action in prompt_action_dict.items():
112 prompt_key_idx = 1 + state.header.action_keys.index(prompt_key)
113 prompt_val_idx = state.prompt_actions[prompt_key].index(prompt_action)
114 action[prompt_key_idx] = prompt_val_idx
115 action = tuple(action)
116 return np.ravel_multi_index(action, state.num_actions)

119def mixed_prompt_policy(rnd, state, prompt_keys, mixture):
120 # all actions are legal for now
121 action = [rnd.choice(a) for a in state.num_actions]
122 for prompt_key in prompt_keys:
123 prompt_key_idx = 1 + state.header.action_keys.index(prompt_key)
124 actions = state.prompt_actions[prompt_key]
125 num_actions = len(actions)
126 prompt_val_idx = rnd.choice(num_actions, p=mixture)
127 action[prompt_key_idx] = prompt_val_idx
128 action = tuple(action)
129 return np.ravel_multi_index(action, state.num_actions)

138def simulate_dialogue(game, policy):
139 """Simulate a dialogue and returns payoffs for each player."""
140
141 state = game.new_initial_state()
142
143 while not state.is_terminal():
144 if state.is_chance_node():
145 # Chance node: sample an outcome
146 outcomes = state.chance_outcomes()
147 action_list, prob_list = zip(*outcomes)
148 action = np.random.choice(action_list, p=prob_list)
149 state.apply_action(action)
150 else:
151 # Decision node: sample action for the single current player
152 action = policy(state.current_player(), state)
153 state.apply_action(action)
154
155 # Game is now done. Print utilities for each player
156 returns = state.returns()
157
158 return returns

161def estimate_payoff_tensor(game, rnd, num_trials=5):
162 """Simulate a batch of dialogues and returns payoffs for each player."""
163
164 num_players = game.num_players()
165 num_actions = len(game.given_prompt_actions["tone"])
166 payoff_tensor = np.zeros(
167 (num_trials, num_players) + (num_actions,) * num_players
168 )
169
170 joint_actions = list(itertools.product(range(num_actions),
171 repeat=num_players))
172
173 for trial in range(num_trials):
174 for joint_action_idx in joint_actions:
175 policies = []
176 for _, tone_idx in zip(range(num_players), joint_action_idx):
177 fixed_tone = {"tone": game.given_prompt_actions["tone"][tone_idx]}
178 policy = lambda state: fixed_prompt_policy(rnd, state, fixed_tone) # pylint:disable=cell-var-from-loop
179 policies.append(policy)
180 player_policy = build_player_policy(policies)
181
182 returns = simulate_dialogue(game, player_policy)
183
184 pt_index = (trial, slice(None)) + joint_action_idx
185
186 payoff_tensor[pt_index] = returns
187
188 return payoff_tensor

191def score_candidate_responses(game_str, config, load_dict, rnd,
192 background_policies, candidates,
193 player_ids=(0,), num_trials=5):
194 """Simulate a batch of dialogues and returns payoffs for each player."""
195
196 num_players = config.params["num_players"]
197
198 num_candidates = len(candidates)
199
200 config.game.given_prompt_actions["tone"] += candidates
201 num_actions = len(config.game.given_prompt_actions["tone"])
202 config.params["num_distinct_actions"] = num_players * num_actions
203
204 game = pyspiel.load_game(game_str, config.params.to_dict())
205
206 game.load_chat_game(**load_dict, **config.game)
207
208 payoffs = np.zeros((num_trials, len(player_ids), num_candidates))
209
210 for player_id in player_ids:
211 for trial in range(num_trials):
212 for candidate_idx in range(num_candidates):
213 policies = []
214 for i in range(num_players):
215 if player_id == i:
216 fixed_tone = {"tone": candidates[candidate_idx]}
217 policy = lambda state: fixed_prompt_policy(rnd, state, fixed_tone) # pylint:disable=cell-var-from-loop
218 policies.append(policy)
219 else:
220 policies.append(background_policies[i])
221 player_policy = build_player_policy(policies)
222
223 returns = simulate_dialogue(game, player_policy)
224
225 payoffs[trial, player_id, candidate_idx] = returns[player_id]
226
227 # undo changes to config (is this inplace?)
228 config.game.given_prompt_actions["tone"] = config.game.given_prompt_actions[
229 "tone"
230 ][:-num_candidates]
231 num_tones = len(config.game.given_prompt_actions["tone"])
232 config.params["num_distinct_actions"] = num_players * num_tones
233
234 return payoffs, candidates

237def compute_sym_eq(pt):
238 game = nashpy.Game(pt[0], pt[1])
239 p1_traj, p2_traj = game.asymmetric_replicator_dynamics()
240 p1_strat = np.mean(p1_traj, axis=0)
241 p2_strat = np.mean(p2_traj, axis=0)
242 return 0.5 * p1_strat + 0.5 * p2_strat

245class PSRO():
246 """Run prompt-space response oracle algorithm on chat game."""
247
248 def __init__(self, save_path, config):
249 self.save_path = save_path
250 self.game_string = config.game_string
251 self.seed = config.seed
252 self.num_iters = config.num_iters
253 self.num_trials = config.num_trials
254 self.num_candidates = config.num_candidates
255 self.domain = config.domain.value
256 self.config = config.env_config
257
258 self.rnd = np.random.RandomState(self.seed)
259
260 self.num_players = self.config.params["num_players"]
261
262 self.game = pyspiel.load_game(self.game_string,
263 self.config.params.to_dict())
264
265 vectorizer = chat_test_utils.MockVectorizer()
266 vectorize = vectorizer.vectorize
267
268 self.load_dict = {"llm_type": LLM_TYPE,
269 "vectorize": vectorize,
270 "seed": self.seed}
271
272 self.game.load_chat_game(**self.load_dict, **self.config.game)
273
274 self.reporting = PSROReporting(
275 save_path=self.save_path,
276 experiment_name="psro",
277 game_string=self.game_string,
278 seed=self.seed,
279 num_iters=self.num_iters,
280 num_trials=self.num_trials,
281 num_candidates=self.num_candidates,
282 domain=self.domain,
283 base_candidates=list(self.config.game.given_prompt_actions["tone"]))
284
285 def run(self):
286 """Evaluate an imitation-learned policy."""
287
288 for psro_iter in range(self.num_iters):
289
290 pt = estimate_payoff_tensor(self.game,
291 self.rnd,
292 num_trials=self.num_trials)
293 pt = pt.mean(axis=0) # mean over trials
294 pt = sym(pt) # symmetrize the pt
295
296 # compute eq
297 sub_eq = compute_sym_eq(pt) # assume symmetric ne
298
299 # generate num_candidate tones
300 actions = self.config.game.given_prompt_actions["tone"]
301 candidates = self.game.generate_prompts("tone",
302 actions,
303 self.num_candidates,
304 text.retrieve_alpha_block)
305 new_actions = actions + candidates
306 new_num_actions = len(new_actions)
307
308 eq = np.zeros(new_num_actions) / float(new_num_actions)
309 eq[:pt.shape[1]] = sub_eq
310
311 background_policies = []
312 for _ in range(self.num_players):
313 bg_policy = lambda state: mixed_prompt_policy(self.rnd,
314 state,
315 ["tone"],
316 eq) # pylint:disable=cell-var-from-loop
317 background_policies.append(bg_policy)
318
319 scores, candidates = score_candidate_responses(
320 self.game_string,
321 self.config,
322 self.load_dict,
323 self.rnd,
324 background_policies,
325 candidates,
326 player_ids=(0,),
327 num_trials=self.num_trials)
328
329 mean_scores = np.mean(scores, axis=0)[0] # only need player 0's scores
330 br_idx = np.argmax(mean_scores)
331 br = candidates[br_idx]
332
333 self.config.game.given_prompt_actions["tone"] += [br]
334 new_num_tones = len(self.config.game.given_prompt_actions["tone"])
335 self.num_players = self.config.params["num_players"]
336 new_num_distinct_actions = self.num_players * new_num_tones
337 self.config.params["num_distinct_actions"] = new_num_distinct_actions
338
339 self.game = pyspiel.load_game(self.game_string,
340 self.config.params.to_dict())
341
342 self.game.load_chat_game(**self.load_dict, **self.config.game)
343
344 self.reporting.report(psro_iter,
345 pt,
346 br,
347 mean_scores,
348 candidates,
349 sub_eq)
