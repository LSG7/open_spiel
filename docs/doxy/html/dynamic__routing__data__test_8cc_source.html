27void TestGetDynamicRoutingDataLine() {
28 std::unique_ptr<DynamicRoutingData> data =
29 DynamicRoutingData::Create(DynamicRoutingDataName::kLine);
30 Network* network = data->network_.get();
31 OriginDestinationDemand od_demand = data->od_demand_->at(0);
32 SPIEL_CHECK_EQ(network->num_links(), 4);
33 SPIEL_CHECK_EQ(network->GetSuccessors("bef_O"),
34 std::vector<std::string>{"O"});
35 SPIEL_CHECK_EQ(network->GetSuccessors("O"), std::vector<std::string>{"A"});
36 SPIEL_CHECK_EQ(network->GetSuccessors("A"), std::vector<std::string>{"D"});
37 SPIEL_CHECK_EQ(network->GetSuccessors("D"),
38 std::vector<std::string>{"aft_D"});
39 SPIEL_CHECK_EQ(network->GetSuccessors("aft_D"), std::vector<std::string>{});
40 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("bef_O->O"));
41 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("O->A"));
42 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("A->D"));
43 SPIEL_CHECK_TRUE(network->IsLocationASinkNode("D->aft_D"));
44 SPIEL_CHECK_EQ(od_demand.vehicle.origin, "bef_O->O");
45 SPIEL_CHECK_EQ(od_demand.vehicle.destination, "D->aft_D");
46 SPIEL_CHECK_EQ(od_demand.vehicle.departure_time, 0);
47 SPIEL_CHECK_EQ(od_demand.counts, 100);
48}

50void TestGetDynamicRoutingDataBraess() {
51 std::unique_ptr<DynamicRoutingData> data =
52 DynamicRoutingData::Create(DynamicRoutingDataName::kBraess);
53 Network* network = data->network_.get();
54 OriginDestinationDemand od_demand = data->od_demand_->at(0);
55 SPIEL_CHECK_EQ(network->num_links(), 7);
56 SPIEL_CHECK_EQ(network->GetSuccessors("O"), (std::vector<std::string>{"A"}));
57 SPIEL_CHECK_EQ(network->GetSuccessors("A"),
58 (std::vector<std::string>{"B", "C"}));
59 SPIEL_CHECK_EQ(network->GetSuccessors("B"),
60 (std::vector<std::string>{"C", "D"}));
61 SPIEL_CHECK_EQ(network->GetSuccessors("C"), (std::vector<std::string>{"D"}));
62 SPIEL_CHECK_EQ(network->GetSuccessors("D"), (std::vector<std::string>{"E"}));
63 SPIEL_CHECK_EQ(network->GetSuccessors("E"), (std::vector<std::string>{}));
64 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("A->B"));
65 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("B->C"));
66 SPIEL_CHECK_FALSE(network->IsLocationASinkNode("C->D"));
67 SPIEL_CHECK_TRUE(network->IsLocationASinkNode("D->E"));
68 SPIEL_CHECK_EQ(od_demand.vehicle.origin, "O->A");
69 SPIEL_CHECK_EQ(od_demand.vehicle.destination, "D->E");
70 SPIEL_CHECK_EQ(od_demand.vehicle.departure_time, 0);
71 SPIEL_CHECK_EQ(od_demand.counts, 5);
72 SPIEL_CHECK_EQ(network->GetTravelTime("O->A", 1.0), 0);
73 SPIEL_CHECK_EQ(network->GetTravelTime("A->B", 1.0),
74 GetTravelTime(1.0, 1.0, 1.0, 5.0, 1.0));
75 SPIEL_CHECK_EQ(network->GetTravelTime("A->C", 1.0),
76 GetTravelTime(2.0, 0, 1.0, 5.0, 1.0));
77 SPIEL_CHECK_EQ(network->GetTravelTime("B->C", 1.0),
78 GetTravelTime(0.25, 0, 1.0, 5.0, 1.0));
79 SPIEL_CHECK_EQ(network->GetTravelTime("B->D", 1.0),
80 GetTravelTime(2.0, 0, 1.0, 5.0, 1.0));
81 SPIEL_CHECK_EQ(network->GetTravelTime("C->D", 1.0),
82 GetTravelTime(1.0, 1.0, 1.0, 5.0, 1.0));
83 SPIEL_CHECK_EQ(network->GetTravelTime("D->E", 1.0), 0);
84}
