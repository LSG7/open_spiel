40 def step(self, step_outputs, reset_if_done=False):
41 """Apply one step.
42
43 Args:
44 step_outputs: the step outputs
45 reset_if_done: if True, automatically reset the environment
46 when the epsiode ends
47
48 Returns:
49 time_steps: the time steps,
50 reward: the reward
51 done: done flag
52 unreset_time_steps: unreset time steps
53 """
54 time_steps = [
55 self.envs[i].step([step_outputs[i].action])
56 for i in range(len(self.envs))
57 ]
58 reward = [step.rewards for step in time_steps]
59 done = [step.last() for step in time_steps]
60 unreset_time_steps = time_steps # Copy these because you may want to look
61 # at the unreset versions to extract
62 # information from them
63
64 if reset_if_done:
65 time_steps = self.reset(envs_to_reset=done)
66
67 return time_steps, reward, done, unreset_time_steps

69 def reset(self, envs_to_reset=None):
70 if envs_to_reset is None:
71 envs_to_reset = [True for _ in range(len(self.envs))]

73 time_steps = [
74 self.envs[i].reset()
75 if envs_to_reset[i] else self.envs[i].get_time_step()
76 for i in range(len(self.envs))
77 ]
78 return time_steps
