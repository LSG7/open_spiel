18 // Implementation of the popular game 2048.
19 // https://en.wikipedia.org/wiki/2048_(video_game)
20 // https://github.com/gabrielecirulli/2048
21 //
22 // The objective of the game is to slide numbered tiles on a grid to combine
23 // them to create bigger tiles.
24 //
25 // Some notes about this implementation:
26 // - End condition:
27 // The game ends when a player has no more valid actions, or a maximum tile
28 // value is reached (default: 2048).
29 //
30 // Parameters:
31 // max_tile int End the game when max_tile is reached?
32 // (default: 2048)

141  int NumDistinctActions() const override;
142  std::unique_ptr<State> NewInitialState() const override {
143  return absl::make_unique<TwentyFortyEightState>(shared_from_this());
144  }
145  int NumPlayers() const override { return kNumPlayers; }
146  double MinUtility() const override { return 0; }
147 
148  std::vector<int> ObservationTensorShape() const override {
149  return {kRows, kColumns};
150  }
151  int MaxChanceOutcomes() const override {
152  return kRows * kColumns * kChanceTiles.size() + 1;
153  }
154 
155  // Using analysis here to derive these bounds:
156  // https://www.reddit.com/r/2048/comments/214njx/highest_possible_score_for_2048_warning_math/
157  double MaxUtility() const override {
158  return (std::log2(max_tile_) - 1) * max_tile_;
159  }
160  // First 2 is for the chance actions, second 2 for all the action required
161  // to get the max tile.
162  int MaxGameLength() const override { return 2 * 2 * max_tile_; }
163 
164  const int max_tile() const { return max_tile_; }
165 
166  private:
167  const int max_tile_;
