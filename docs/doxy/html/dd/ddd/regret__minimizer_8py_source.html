24class RegretMinimizer(object):
25 """Base class for Regret Minimizers.
26
27 Implements base functions for regret minimizers to implement.
28
29 Attributes:
30 _game: Pyspiel game.
31 _regret_steps_per_step: Number of regret steps per `step` call (Maximum
32 number in case `stop_early` is true)
33 _rho_tol: If `_compress_nus` is true, minimum probability threshold (
34 Probabilities below `rho_tol` will be filtered out).
35 _compress_nus: Whether to compress nus (Remove nus with low selection
36 probability) or not.
37 _compress_lbd: Penalty term in L1 minimization when compressing nus.
38 _stop_early: Whether to stop regret computation when average regret is lower
39 than `_stop_regret_threshold` or to keep going until
40 `_regret_steps_per_step` steps have been accomplished.
41 _stop_regret_threshold: If `stop_early` is true, average regret threshold
42 under which the algorithm will stop.
43 _policies: List of Policies
44 _value_estimator: Value estimation function.
45 _value_estimation_n: Number of runs to average _value_estimator's result on.
46 """

101 def step_for(
102 self,
103 T, # pylint: disable=invalid-name
104 initial_welfare_bonus=None,
105 welfare_decay=None,
106 use_true_rewards_when_compressing=True,
107 ):
108 """Call `step` method `T` times maximum, potentially stop early.
109
110 Args:
111 T: Maximum number of `step` calls to run.
112 initial_welfare_bonus: How much to initially reward high-welfare-inducing
113 actions.
114 welfare_decay: Welfare decay term.
115 use_true_rewards_when_compressing: Compress and compute optimal (C)CE
116 according to true rewards (= True) or according to modified rewards (=
117 False)
118 """
119 welfare_bonus = 0.0
120 if initial_welfare_bonus is not None:
121 assert welfare_decay is not None
122 welfare_bonus = initial_welfare_bonus
123
124 weights = None
125 for t in range(T):
126 if welfare_decay is not None:
127 welfare_bonus = max(0.0, welfare_bonus - welfare_decay * t / T)
128 self.step(welfare_bonus=welfare_bonus)
129 if self._stop_early and (t % self._compress_every == 0):
130 try:
131 regret, weights = self.get_post_compression_regret_and_weights(
132 use_true_rewards_when_compressing=use_true_rewards_when_compressing
133 )
134 # print("\t\t{}".format(regret))
135 assert np.abs(np.sum(weights) - 1.0) < 1e-8, np.sum(weights)
136 except: # pylint: disable=bare-except
137 print("Simplex method encountered an error.")
138 continue
139 if regret < self._stop_regret_threshold:
140 break
141 if weights is None and self._compress_nus:
142 regret, weights = self.get_post_compression_regret_and_weights(
143 use_true_rewards_when_compressing=use_true_rewards_when_compressing
144 )
145 if self._compress_nus:
146 self.compress_nus_and_weights(weights)

148 def get_post_compression_regret_and_weights(
149 self, use_true_rewards_when_compressing=True
150 ):
151 """Computes optimized (C)CE by varying the temporal weight on each `nu`.
152
153 Args:
154 use_true_rewards_when_compressing: compute optimal (C)CE according to true
155 rewards (= True) or according to modified rewards (= False)
156
157 Returns:
158 Regret for new temporal weights, new temporal weights
159 """
160 if self._compute_internal_regret:
161 nu_weights = c_ce_optimization.compress_internal_weights(
162 self.get_nus(),
163 self.compute_regrets(
164 use_true_rewards=use_true_rewards_when_compressing
165 ),
166 rewards=self._rewards,
167 lbd=self._compress_lbd,
168 )
169 regret = np.max([
170 np.max(np.sum(nu_weights.reshape(-1, 1) * a, axis=0))
171 for a in self.compute_regrets(
172 use_true_rewards=use_true_rewards_when_compressing
173 )
174 ])
175 else:
176 nu_weights = c_ce_optimization.compress_external_weights(
177 self.get_nus(),
178 self.compute_regrets(
179 use_true_rewards=use_true_rewards_when_compressing
180 ),
181 rewards=self._rewards,
182 lbd=self._compress_lbd,
183 )
184 regret = np.max(
185 np.sum(
186 nu_weights.reshape(-1, 1)
187 * self.compute_regrets(
188 use_true_rewards=use_true_rewards_when_compressing
189 ),
190 axis=0,
191 )
192 )
193 return regret, nu_weights

195 def compress_nus_and_weights(self, nu_weights):
196 """Run L1 optimization to only keep important members of `nus`."""
197 if self._compress_nus:
198 if np.abs(np.sum(nu_weights) - 1.0) > 1e-8:
199 # If the optimization was unsuccessful, do *not* compress.
200 print(
201 "Unsuccessful optimization, weights sum to {}".format(
202 np.sum(nu_weights)
203 )
204 )
205 return
206 new_nus = [
207 nu
208 for weight, nu in zip(nu_weights, self._nus)
209 if weight > self._rho_tol
210 ]
211 new_rewards = [
212 reward
213 for weight, reward in zip(nu_weights, self._rewards)
214 if weight > self._rho_tol
215 ]
216 new_true_rewards = [
217 reward
218 for weight, reward in zip(nu_weights, self._true_rewards)
219 if weight > self._rho_tol
220 ]
221
222 new_nu_weights = [
223 weight for weight in nu_weights if weight > self._rho_tol
224 ]
225 new_nu_weights = np.array(new_nu_weights) / np.sum(new_nu_weights)
226
227 self._nus = new_nus
228 self._rewards = new_rewards
229 self._true_rewards = new_true_rewards
230 self._nu_weights = new_nu_weights

241 def compute_p(self):
242 """Computes `p` as presented in Blum's External to Internal Regret."""
243 assert (
244 self._compute_internal_regret
245 ), "`p` does not exist when computing external regret."
246 w_nus = np.array(self.get_all_w_nus())
247
248 p = np.ones(len(self._policies))
249 pprime = np.dot(p, w_nus)
250 n_trials = 100000
251 i = 0
252 while np.sum(np.abs(pprime - p)) > 1e-8 and i < n_trials:
253 p = pprime
254 pprime = np.dot(p, w_nus)
255 i += 1
256
257 if np.sum(np.abs(pprime - p)) > 1e-8 and i >= n_trials:
258 raise ValueError(
259 "Power method did not converge after {} trials.".format(n_trials)
260 )
261 self._p = p / np.sum(p)

267 def compute_regrets(self, use_true_rewards=False):
268 """Computes the algorithm's current external/internal regrets.
269
270 Args:
271 use_true_rewards: Whether to use altered game rewards, or true game
272 rewards.
273
274 Returns:
275 Internal regret of shape [T, P, P] if `self._compute_internal_regret` is
276 true, otherwise external regret of shape [T, P], where T is the current
277 number of iterations and P the number of policies.
278 """
279 if use_true_rewards:
280 rewards = self._true_rewards
281 else:
282 rewards = self._rewards
283
284 if self._compute_internal_regret:
285 regrets = []
286 nus = np.array(self._nus)
287 rewards = np.array(rewards)
288 for action in range(rewards.shape[1]):
289 on_policy_values = (rewards[:, action] * nus[:, action]).reshape(-1, 1)
290 action_values = rewards * nus[:, action].reshape(-1, 1)
291 regrets.append(action_values - on_policy_values)
292 else:
293 on_policy_value = np.sum(
294 rewards * np.array(self._nus), axis=1, keepdims=True
295 )
296 policy_value = rewards
297 regrets = policy_value - on_policy_value
298 return regrets

300 def compute_average_regret(self, use_true_rewards=True):
301 """Computes the algorithm's average external/internal regrets.
302
303 Args:
304 use_true_rewards: Whether to use altered game rewards, or true game
305 rewards.
306
307 Returns:
308 Internal regret if `self._compute_internal_regret` is true, otherwise
309 external regret.
310 """
311
312 if use_true_rewards:
313 rewards = self._true_rewards
314 else:
315 rewards = self._rewards
316
317 nu_weights = self.get_normalized_nu_weights()
318 if self._compute_internal_regret:
319 regrets = 0.0
320 nus = np.array(self._nus)
321 rewards = np.array(rewards)
322 for action in range(rewards.shape[1]):
323 on_policy_values = (rewards[:, action] * nus[:, action]).reshape(-1, 1)
324 action_values = rewards * nus[:, action].reshape(-1, 1)
325 regrets += np.max(
326 np.sum(
327 nu_weights.reshape(-1, 1) * (action_values - on_policy_values),
328 axis=0,
329 )
330 )
331 else:
332 regrets = np.sum(
333 nu_weights.reshape(-1, 1)
334 * self.compute_regrets(use_true_rewards=use_true_rewards),
335 axis=0,
336 )
337 return np.max(regrets) / len(self._nus)
