15"""Implements Deep CFR Algorithm.
16
17See https://arxiv.org/abs/1811.00164.
18
19The algorithm defines an `advantage` and `strategy` networks that compute
20advantages used to do regret matching across information sets and to approximate
21the strategy profiles of the game. To train these networks a reservoir buffer
22(other data structures may be used) memory is used to accumulate samples to
23train the networks.
24
25This implementation uses skip connections as described in the paper if two
26consecutive layers of the advantage or policy network have the same number
27of units, except for the last connection. Before the last hidden layer
28a layer normalization is applied.
29"""

58class ReservoirBuffer(object):
59 """Allows uniform sampling over a stream of data.
60
61 This class supports the storage of arbitrary elements, such as observation
62 tensors, integer actions, etc.
63
64 See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
65 """
66
67 def __init__(self, reservoir_buffer_capacity):
68 self._reservoir_buffer_capacity = reservoir_buffer_capacity
69 self._data = []
70 self._add_calls = 0
71
72 def add(self, element):
73 """Potentially adds `element` to the reservoir buffer.
74
75 Args:
76 element: data to be added to the reservoir buffer.
77 """
78 if len(self._data) < self._reservoir_buffer_capacity:
79 self._data.append(element)
80 else:
81 idx = np.random.randint(0, self._add_calls + 1)
82 if idx < self._reservoir_buffer_capacity:
83 self._data[idx] = element
84 self._add_calls += 1
85
86 def sample(self, num_samples):
87 """Returns `num_samples` uniformly sampled from the buffer.
88
89 Args:
90 num_samples: `int`, number of samples to draw.
91
92 Returns:
93 An iterable over `num_samples` random elements of the buffer.
94
95 Raises:
96 ValueError: If there are less than `num_samples` elements in the buffer
97 """
98 if len(self._data) < num_samples:
99 raise ValueError('{} elements could not be sampled from size {}'.format(
100 num_samples, len(self._data)))
101 return random.sample(self._data, num_samples)
102
103 def clear(self):
104 self._data = []
105 self._add_calls = 0
106
107 def __len__(self):
108 return len(self._data)
109
110 def __iter__(self):
111 return iter(self._data)
112
113 @property
114 def data(self):
115 return self._data
116
117 def shuffle_data(self):
118 random.shuffle(self._data)

121class SkipDense(tf.keras.layers.Layer):
122 """Dense Layer with skip connection."""
123
124 def __init__(self, units, **kwargs):
125 super().__init__(**kwargs)
126 self.hidden = tf.keras.layers.Dense(units, kernel_initializer='he_normal')
127
128 def call(self, x):
129 return self.hidden(x) + x

132class PolicyNetwork(tf.keras.Model):
133 """Implements the policy network as an MLP.
134
135 Implements the policy network as a MLP with skip connections in adjacent
136 layers with the same number of units, except for the last hidden connection
137 where a layer normalization is applied.
138 """
139
140 def __init__(self,
141 input_size,
142 policy_network_layers,
143 num_actions,
144 activation='leakyrelu',
145 **kwargs):
146 super().__init__(**kwargs)
147 self._input_size = input_size
148 self._num_actions = num_actions
149 if activation == 'leakyrelu':
150 self.activation = tf.keras.layers.LeakyReLU(alpha=0.2)
151 elif activation == 'relu':
152 self.activation = tf.keras.layers.ReLU()
153 else:
154 self.activation = activation
155
156 self.softmax = tf.keras.layers.Softmax()
157
158 self.hidden = []
159 prevunits = 0
160 for units in policy_network_layers[:-1]:
161 if prevunits == units:
162 self.hidden.append(SkipDense(units))
163 else:
164 self.hidden.append(
165 tf.keras.layers.Dense(units, kernel_initializer='he_normal'))
166 prevunits = units
167 self.normalization = tf.keras.layers.LayerNormalization()
168 self.lastlayer = tf.keras.layers.Dense(
169 policy_network_layers[-1], kernel_initializer='he_normal')
170
171 self.out_layer = tf.keras.layers.Dense(num_actions)
172
173 @tf.function
174 def call(self, inputs):
175 """Applies Policy Network.
176
177 Args:
178 inputs: Tuple representing (info_state, legal_action_mask)
179
180 Returns:
181 Action probabilities
182 """
183 x, mask = inputs
184 for layer in self.hidden:
185 x = layer(x)
186 x = self.activation(x)
187
188 x = self.normalization(x)
189 x = self.lastlayer(x)
190 x = self.activation(x)
191 x = self.out_layer(x)
192 x = tf.where(mask == 1, x, -10e20)
193 x = self.softmax(x)
194 return x

236 @tf.function
237 def call(self, inputs):
238 """Applies Policy Network.
239
240 Args:
241 inputs: Tuple representing (info_state, legal_action_mask)
242
243 Returns:
244 Cumulative regret for each info_state action
245 """
246 x, mask = inputs
247 for layer in self.hidden:
248 x = layer(x)
249 x = self.activation(x)
250
251 x = self.normalization(x)
252 x = self.lastlayer(x)
253 x = self.activation(x)
254 x = self.out_layer(x)
255 x = mask * x
256
257 return x

260class DeepCFRSolver(policy.Policy):
261 """Implements a solver for the Deep CFR Algorithm.
262
263 See https://arxiv.org/abs/1811.00164.
264
265 Define all networks and sampling buffers/memories. Derive losses & learning
266 steps. Initialize the game state and algorithmic variables.
267 """
268
269 def __init__(self,
270 game,
271 policy_network_layers=(256, 256),
272 advantage_network_layers=(128, 128),
273 num_iterations: int = 100,
274 num_traversals: int = 100,
275 learning_rate: float = 1e-3,
276 batch_size_advantage: int = 2048,
277 batch_size_strategy: int = 2048,
278 memory_capacity: int = int(1e6),
279 policy_network_train_steps: int = 5000,
280 advantage_network_train_steps: int = 750,
281 reinitialize_advantage_networks: bool = True,
282 save_advantage_networks: str = None,
283 save_strategy_memories: str = None,
284 infer_device='cpu',
285 train_device='cpu'):
286 """Initialize the Deep CFR algorithm.
287
288 Args:
289 game: Open Spiel game.
290 policy_network_layers: (list[int]) Layer sizes of strategy net MLP.
291 advantage_network_layers: (list[int]) Layer sizes of advantage net MLP.
292 num_iterations: Number of iterations.
293 num_traversals: Number of traversals per iteration.
294 learning_rate: Learning rate.
295 batch_size_advantage: (int) Batch size to sample from advantage memories.
296 batch_size_strategy: (int) Batch size to sample from strategy memories.
297 memory_capacity: Number of samples that can be stored in memory.
298 policy_network_train_steps: Number of policy network training steps (one
299 policy training iteration at the end).
300 advantage_network_train_steps: Number of advantage network training steps
301 (per iteration).
302 reinitialize_advantage_networks: Whether to re-initialize the advantage
303 network before training on each iteration.
304 save_advantage_networks: If provided, all advantage network itearations
305 are saved in the given folder. This can be useful to implement SD-CFR
306 https://arxiv.org/abs/1901.07621
307 save_strategy_memories: saves the collected strategy memories as a
308 tfrecords file in the given location. This is not affected by
309 memory_capacity. All memories are saved to disk and not kept in memory
310 infer_device: device used for TF-operations in the traversal branch.
311 Format is anything accepted by tf.device
312 train_device: device used for TF-operations in the NN training steps.
313 Format is anything accepted by tf.device
314 """
315 all_players = list(range(game.num_players()))
316 super(DeepCFRSolver, self).__init__(game, all_players)
317 self._game = game
318 if game.get_type().dynamics == pyspiel.GameType.Dynamics.SIMULTANEOUS:
319 # `_traverse_game_tree` does not take into account this option.
320 raise ValueError('Simulatenous games are not supported.')
321 self._batch_size_advantage = batch_size_advantage
322 self._batch_size_strategy = batch_size_strategy
323 self._policy_network_train_steps = policy_network_train_steps
324 self._advantage_network_train_steps = advantage_network_train_steps
325 self._policy_network_layers = policy_network_layers
326 self._advantage_network_layers = advantage_network_layers
327 self._num_players = game.num_players()
328 self._root_node = self._game.new_initial_state()
329 self._embedding_size = len(self._root_node.information_state_tensor(0))
330 self._num_iterations = num_iterations
331 self._num_traversals = num_traversals
332 self._reinitialize_advantage_networks = reinitialize_advantage_networks
333 self._num_actions = game.num_distinct_actions()
334 self._iteration = 1
335 self._learning_rate = learning_rate
336 self._save_advantage_networks = save_advantage_networks
337 self._save_strategy_memories = save_strategy_memories
338 self._infer_device = infer_device
339 self._train_device = train_device
340 self._memories_tfrecordpath = None
341 self._memories_tfrecordfile = None
342
343 # Initialize file save locations
344 if self._save_advantage_networks:
345 os.makedirs(self._save_advantage_networks, exist_ok=True)
346
347 if self._save_strategy_memories:
348 if os.path.isdir(self._save_strategy_memories):
349 self._memories_tfrecordpath = os.path.join(
350 self._save_strategy_memories, 'strategy_memories.tfrecord')
351 else:
352 os.makedirs(
353 os.path.split(self._save_strategy_memories)[0], exist_ok=True)
354 self._memories_tfrecordpath = self._save_strategy_memories
355
356 # Initialize policy network, loss, optmizer
357 self._reinitialize_policy_network()
358
359 # Initialize advantage networks, losses, optmizers
360 self._adv_networks = []
361 self._adv_networks_train = []
362 self._loss_advantages = []
363 self._optimizer_advantages = []
364 self._advantage_train_step = []
365 for player in range(self._num_players):
366 self._adv_networks.append(
367 AdvantageNetwork(self._embedding_size, self._advantage_network_layers,
368 self._num_actions))
369 with tf.device(self._train_device):
370 self._adv_networks_train.append(
371 AdvantageNetwork(self._embedding_size,
372 self._advantage_network_layers, self._num_actions))
373 self._loss_advantages.append(tf.keras.losses.MeanSquaredError())
374 self._optimizer_advantages.append(
375 tf.keras.optimizers.Adam(learning_rate=learning_rate))
376 self._advantage_train_step.append(
377 self._get_advantage_train_graph(player))
378
379 self._create_memories(memory_capacity)

433 def solve(self):
434 """Solution logic for Deep CFR."""
435 advantage_losses = collections.defaultdict(list)
436 with tf.device(self._infer_device):
437 with contextlib.ExitStack() as stack:
438 if self._save_strategy_memories:
439 self._memories_tfrecordfile = stack.enter_context(
440 tf.io.TFRecordWriter(self._memories_tfrecordpath))
441 for _ in range(self._num_iterations):
442 for p in range(self._num_players):
443 for _ in range(self._num_traversals):
444 self._traverse_game_tree(self._root_node, p)
445 if self._reinitialize_advantage_networks:
446 # Re-initialize advantage network for p and train from scratch.
447 self._reinitialize_advantage_network(p)
448 advantage_losses[p].append(self._learn_advantage_network(p))
449 if self._save_advantage_networks:
450 os.makedirs(self._save_advantage_networks, exist_ok=True)
451 self._adv_networks[p].save(
452 os.path.join(self._save_advantage_networks,
453 f'advnet_p{p}_it{self._iteration:04}'))
454 self._iteration += 1
455 # Train policy network.
456 policy_loss = self._learn_strategy_network()
457 return self._policy_network, advantage_losses, policy_loss

552 def _traverse_game_tree(self, state, player):
553 """Performs a traversal of the game tree using external sampling.
554
555 Over a traversal the advantage and strategy memories are populated with
556 computed advantage values and matched regrets respectively.
557
558 Args:
559 state: Current OpenSpiel game state.
560 player: (int) Player index for this traversal.
561
562 Returns:
563 Recursively returns expected payoffs for each action.
564 """
565 if state.is_terminal():
566 # Terminal state get returns.
567 return state.returns()[player]
568 elif state.is_chance_node():
569 # If this is a chance node, sample an action
570 chance_outcome, chance_proba = zip(*state.chance_outcomes())
571 action = np.random.choice(chance_outcome, p=chance_proba)
572 return self._traverse_game_tree(state.child(action), player)
573 elif state.current_player() == player:
574 # Update the policy over the info set & actions via regret matching.
575 _, strategy = self._sample_action_from_advantage(state, player)
576 exp_payoff = 0 * strategy
577 for action in state.legal_actions():
578 exp_payoff[action] = self._traverse_game_tree(
579 state.child(action), player)
580 ev = np.sum(exp_payoff * strategy)
581 samp_regret = (exp_payoff - ev) * state.legal_actions_mask(player)
582 self._advantage_memories[player].add(
583 self._serialize_advantage_memory(state.information_state_tensor(),
584 self._iteration, samp_regret,
585 state.legal_actions_mask(player)))
586 return ev
587 else:
588 other_player = state.current_player()
589 _, strategy = self._sample_action_from_advantage(state, other_player)
590 # Recompute distribution for numerical errors.
591 probs = strategy
592 probs /= probs.sum()
593 sampled_action = np.random.choice(range(self._num_actions), p=probs)
594 self._add_to_strategy_memory(
595 state.information_state_tensor(other_player), self._iteration,
596 strategy, state.legal_actions_mask(other_player))
597 return self._traverse_game_tree(state.child(sampled_action), player)

599 @tf.function
600 def _get_matched_regrets(self, info_state, legal_actions_mask, player):
601 """TF-Graph to calculate regret matching."""
602 advs = self._adv_networks[player](
603 (tf.expand_dims(info_state, axis=0), legal_actions_mask),
604 training=False)[0]
605 advantages = tf.maximum(advs, 0)
606 summed_regret = tf.reduce_sum(advantages)
607 if summed_regret > 0:
608 matched_regrets = advantages / summed_regret
609 else:
610 matched_regrets = tf.one_hot(
611 tf.argmax(tf.where(legal_actions_mask == 1, advs, -10e20)),
612 self._num_actions)
613 return advantages, matched_regrets

615 def _sample_action_from_advantage(self, state, player):
616 """Returns an info state policy by applying regret-matching.
617
618 Args:
619 state: Current OpenSpiel game state.
620 player: (int) Player index over which to compute regrets.
621
622 Returns:
623 1. (np-array) Advantage values for info state actions indexed by action.
624 2. (np-array) Matched regrets, prob for actions indexed by action.
625 """
626 info_state = tf.constant(
627 state.information_state_tensor(player), dtype=tf.float32)
628 legal_actions_mask = tf.constant(
629 state.legal_actions_mask(player), dtype=tf.float32)
630 advantages, matched_regrets = self._get_matched_regrets(
631 info_state, legal_actions_mask, player)
632 return advantages.numpy(), matched_regrets.numpy()

634 def action_probabilities(self, state):
635 """Returns action probabilities dict for a single batch."""
636 cur_player = state.current_player()
637 legal_actions = state.legal_actions(cur_player)
638 legal_actions_mask = tf.constant(
639 state.legal_actions_mask(cur_player), dtype=tf.float32)
640 info_state_vector = tf.constant(
641 state.information_state_tensor(), dtype=tf.float32)
642 if len(info_state_vector.shape) == 1:
643 info_state_vector = tf.expand_dims(info_state_vector, axis=0)
644 probs = self._policy_network((info_state_vector, legal_actions_mask),
645 training=False)
646 probs = probs.numpy()
647 return {action: probs[0][action] for action in legal_actions}

649 def _get_advantage_dataset(self, player):
650 """Returns the collected regrets for the given player as a dataset."""
651 self._advantage_memories[player].shuffle_data()
652 data = tf.data.Dataset.from_tensor_slices(
653 self._advantage_memories[player].data)
654 data = data.shuffle(ADVANTAGE_TRAIN_SHUFFLE_SIZE)
655 data = data.repeat()
656 data = data.batch(self._batch_size_advantage)
657 data = data.map(self._deserialize_advantage_memory)

661 def _get_advantage_train_graph(self, player):
662 """Return TF-Graph to perform advantage network train step."""
663 @tf.function
664 def train_step(info_states, advantages, iterations, masks, iteration):
665 model = self._adv_networks_train[player]
666 with tf.GradientTape() as tape:
667 preds = model((info_states, masks), training=True)
668 main_loss = self._loss_advantages[player](
669 advantages, preds, sample_weight=iterations * 2 / iteration)
670 loss = tf.add_n([main_loss], model.losses)
671 gradients = tape.gradient(loss, model.trainable_variables)
672 self._optimizer_advantages[player].apply_gradients(
673 zip(gradients, model.trainable_variables))
674 return main_loss
675
676 return train_step

678 def _learn_advantage_network(self, player):
679 """Compute the loss on sampled transitions and perform a Q-network update.
680
681 If there are not enough elements in the buffer, no loss is computed and
682 `None` is returned instead.
683
684 Args:
685 player: (int) player index.
686
687 Returns:
688 The average loss over the advantage network of the last batch.
689 """
690
691 with tf.device(self._train_device):
692 tfit = tf.constant(self._iteration, dtype=tf.float32)
693 data = self._get_advantage_dataset(player)
694 for d in data.take(self._advantage_network_train_steps):
695 main_loss = self._advantage_train_step[player](*d, tfit)
696
697 self._adv_networks[player].set_weights(
698 self._adv_networks_train[player].get_weights())
699 return main_loss

702 """Returns the collected strategy memories as a dataset."""
703 if self._memories_tfrecordpath:
704 data = tf.data.TFRecordDataset(self._memories_tfrecordpath)
705 else:
706 self._strategy_memories.shuffle_data()

708 data = data.shuffle(STRATEGY_TRAIN_SHUFFLE_SIZE)
709 data = data.repeat()
710 data = data.batch(self._batch_size_strategy)
711 data = data.map(self._deserialize_strategy_memory)
712 data = data.prefetch(tf.data.experimental.AUTOTUNE)
713 return data

717
718 Returns:

720 or `None`.
721 """
722
723 @tf.function
724 def train_step(info_states, action_probs, iterations, masks):
725 model = self._policy_network

727 preds = model((info_states, masks), training=True)
728 main_loss = self._loss_policy(
729 action_probs, preds, sample_weight=iterations * 2 / self._iteration)
730 loss = tf.add_n([main_loss], model.losses)
731 gradients = tape.gradient(loss, model.trainable_variables)
732 self._optimizer_policy.apply_gradients(
733 zip(gradients, model.trainable_variables))
734 return main_loss

738 for d in data.take(self._policy_network_train_steps):
739 main_loss = train_step(*d)
