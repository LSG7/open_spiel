47 def init_vars(self, num_strats, num_players):
48 """Initialize solver parameters."""
49 self.num_players = num_players
50 if len(num_strats) != num_players:
51 raise ValueError('Must specify num strategies for each player')
52 init_dist = []
53 for num_strats_i in num_strats:
54 if self.rnd_init:
55 init_dist_i = self.random.rand(num_strats_i)
56 else:
57 init_dist_i = np.ones(num_strats_i)
58 init_dist_i /= init_dist_i.sum()
59 init_dist.append(init_dist_i)
60 return (init_dist,)

87 def euc_descent_step(self, params, grads, t, eps=0.):
88 """Projected gradient descent on exploitability using Euclidean projection.
89
90 Args:
91 params: tuple of variables to be updated (dist,)
92 grads: tuple of variable gradients (grad_dist,)
93 t: int, solver iteration (unused)
94 eps: float > 0, force all probabilities >= eps / dim(dist)
95 Returns:
96 new_params: tuple of update params (new_dist,)
97 """
98 del t
99 lr_dist = self.lrs[0]
100 new_params = []
101 for dist_i, dist_grad_i in zip(params[0], grads[0]):
102 new_dist_i = dist_i - lr_dist * dist_grad_i
103 new_dist_i = simplex.euclidean_projection_onto_simplex(new_dist_i)
104 if eps > 0:
105 new_dist_i = simplex.project_to_interior(new_dist_i, eps)
106 new_params.append(new_dist_i)
107 return (new_params,)

109 def mirror_descent_step(self, params, grads, t, eps=0.):
110 """Entropic mirror descent on exploitability.
111
112 Args:
113 params: tuple of variables to be updated (dist - a list of np.arrays)
114 grads: tuple of variable gradients (grad_dist - a list of np.arrays)
115 t: int, solver iteration (unused)
116 eps: float > 0, force all probabilities >= eps / dim(dist)
117 Returns:
118 new_params: tuple of update params (new_dist)
119 """
120 del t
121 lr_dist = self.lrs[0]
122 new_params = []
123 for dist_i, dist_grad_i in zip(params[0], grads[0]):
124 new_dist_i = np.clip(dist_i, 0, np.inf)
125 new_dist_i = special.softmax(np.log(new_dist_i) - lr_dist * dist_grad_i)
126 if eps > 0:
127 new_dist_i = simplex.project_to_interior(new_dist_i, eps)
128 new_params.append(new_dist_i)
129 return (new_params,)
