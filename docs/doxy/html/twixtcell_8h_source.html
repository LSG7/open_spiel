20struct Position {
21 int x;
22 int y;
23 Position operator+(const Position &p) { return {x + p.x, y + p.y}; }
24 bool operator==(const Position &p) const { return x == p.x && y == p.y; }
25 bool operator<(const Position &p) const {
26 return x < p.x || (x == p.x && y < p.y);
27 }
28};

30struct Link {
31 Position position;
32 int direction;
33 bool operator==(const Link &l) const {
34 return position == l.position && direction == l.direction;
35 }
36 bool operator<(const Link &l) const {
37 return position < l.position ||
38 (position == l.position && direction < l.direction);
39 }
40};
