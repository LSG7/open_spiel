16 """Representation of a policy for a game.
17 
18 This is a standard representation for passing policies into algorithms,
19 with currently the following implementations:
20 
21  TabularPolicy - an explicit policy per state, stored in an array
22  of shape `(num_states, num_actions)`, convenient for tabular policy
23  solution methods.
24  UniformRandomPolicy - a uniform distribution over all legal actions for
25  the specified player. This is computed as needed, so can be used for
26  games where a tabular policy would be unfeasibly large.
27 
28 The main way of using a policy is to call `action_probabilities(state,
29 player_id`), to obtain a dict of {action: probability}. `TabularPolicy`
30 objects expose a lower-level interface, which may be more efficient for
31 some use cases.
32 """

53 def joint_action_probabilities_aux(state, policy):
54  """Auxiliary function for joint_action_probabilities.
55 
56  Args:
57  state: a game state at a simultaneous decision node.
58  policy: policy that gives the probability distribution over the legal
59  actions for each players.
60 
61  Returns:
62  actions_per_player: list of list of actions for each player
63  probs_per_player: list of list of probabilities do the corresponding action
64  in actions_per_player for each player.
65  """

68  policy.action_probabilities(state, player)

76 def joint_action_probabilities(state, policy):
77  """Yields action, probability pairs for a joint policy in simultaneous state.
78 
79  Args:
80  state: a game state at a simultaneous decision node.
81  policy: policy that gives the probability distribution over the legal
82  actions for each players.
83 
84  Yields:
85  (action, probability) pairs. An action is a tuple of individual
86  actions for each player of the game. The probability is a single joint
87  probability (product of all the individual probabilities).
88  """
89  actions_per_player, probs_per_player = joint_action_probabilities_aux(
90  state, policy)
91  for actions, probs in zip(
92  itertools.product(*actions_per_player),
93  itertools.product(*probs_per_player)):
94  yield actions, np.prod(probs)

124  - All probabilities are >=0 and sum to 1

126  actions, but algorithms should not rely on this (yet).

178 class TabularPolicy(Policy):
179  """Policy implementation where the policy is in explicit tabular form.
180 
181  In addition to implementing the `Policy` interface, this class exposes
182  details of the policy representation for easy manipulation.
183 
184  The states are guaranteed to be grouped by player, which can simplify
185  code for users of this class, i.e. `action_probability_array` contains
186  states for player 0 first, followed by states for player 1, etc.
187 
188  The policy uses `state.information_state_string` as the keys if available,
189  otherwise `state.observation_string`.
190 
191  Usages:
192 
193  - Set `policy(info_state, action)`:
194  ```
195  tabular_policy = TabularPolicy(game)
196  info_state_str = state.information_state_string(<optional player>)
197  state_policy = tabular_policy.policy_for_key(info_state_str)
198  state_policy[action] = <value>
199  ```
200  - Set `policy(info_state)`:
201  ```
202  tabular_policy = TabularPolicy(game)
203  info_state_str = state.information_state_string(<optional player>)
204  state_policy = tabular_policy.policy_for_key(info_state_str)
205  state_policy[:] = <list or numpy.array>
206  ```
207 
208  Attributes:
209  action_probability_array: array of shape `(num_states, num_actions)`, where
210  `action_probability_array[s, a]` is the probability of choosing action `a`
211  when at state `s`.
212  state_lookup: `dict` mapping state key string to index into the
213  `tabular_policy` array. If information state strings overlap, e.g. for
214  different players or if the information state string has imperfect recall,
215  then those states will be mapped to the same policy.
216  legal_actions_mask: array of shape `(num_states, num_actions)`, each row
217  representing which of the possible actions in the game are valid in this
218  particular state, containing 1 for valid actions, 0 for invalid actions.
219  states_per_player: A `list` per player of the state key strings at which
220  they have a decision to make.
221  states: A `list` of the states as ordered in the `action_probability_array`.
222  state_in: array of shape `(num_states, state_vector_size)` containing the
223  normalised vector representation of each information state. Populated only
224  for games which support information_state_tensor(), and is None otherwise.
225  game_type: The game attributes as returned by `Game::GetType`; used to
226  determine whether to use information state or observation as the key in
227  the tabular policy.
228  """

369  def copy_with_noise(self,
370  alpha=0.0,
371  beta=0.0,
372  random_state=np.random.RandomState()):
373  """Returns a copy of this policy perturbed with noise.
374 
375  Generates a new random distribution using a softmax on normal random
376  variables with temperature beta, and mixes it with the old distribution
377  using 1-alpha * old_distribution + alpha * random_distribution.
378  Args:
379  alpha: Parameter characterizing the mixture amount between new and old
380  distributions. Between 0 and 1.
381  alpha = 0: keep old table.
382  alpha = 1: keep random table.
383  beta: Temperature of the softmax. Makes for more extreme policies.
384  random_state: A numpy `RandomState` object. If not provided, a shared
385  random state will be used.
386 
387  Returns:
388  Perturbed copy.
389  """
390  copied_instance = self.__copy____copy__(False)
391  probability_array = self.action_probability_arrayaction_probability_array
392  noise_mask = random_state.normal(size=probability_array.shape)
393  noise_mask = np.exp(beta * noise_mask) * self.legal_actions_masklegal_actions_mask
394  noise_mask = noise_mask / (np.sum(noise_mask, axis=1).reshape(-1, 1))
395  copied_instance.action_probability_array = (
396  1 - alpha) * probability_array + alpha * noise_mask
397  return copied_instance
