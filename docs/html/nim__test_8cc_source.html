52 void SinglePileNormalTest() {
53  std::shared_ptr<const Game> game =
54  LoadGame("nim", {
55  {"pile_sizes", GameParameter("100")},
56  {"is_misere", GameParameter(false)},
57  });
58  std::unique_ptr<State> state = game->NewInitialState();
59  std::vector<Action> actions = state->LegalActions();
60  SPIEL_CHECK_EQ(actions.size(), 100);
61 
62  state->ApplyAction(actions.back());
63  SPIEL_CHECK_EQ(state->IsTerminal(), 1);
64  SPIEL_CHECK_EQ(state->PlayerReturn(0), 1);
65  SPIEL_CHECK_EQ(state->PlayerReturn(1), -1);
66 }
