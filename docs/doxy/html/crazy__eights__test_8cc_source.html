37void SpecialCardTests() {
38 std::shared_ptr<const Game> game =
39 LoadGame("crazy_eights", {{"players", GameParameter(4)},
40 {"use_special_cards", GameParameter(true)}});
41
42 std::unique_ptr<State> state = game->NewInitialState();
43 // 0 is the dealer
44 state->ApplyAction(kDecideDealerActionBase);
45 // Player0 has (S2)(H8)(DQ)(SK)(SA)
46 // Player1 has (C2)(C3)(S8)(HQ)(CA)
47 // Player2 has (D2)(C8)(C9)(SQ)(DA)
48 // Player3 has (H2)(D8)(CQ)(CK)(HA)
49 std::vector<int> dealt_cards = {0, 1, 2, 3, 4, 24, 25, 26, 27, 28,
50 40, 41, 42, 43, 44, 47, 48, 49, 50, 51};
51
52 for (auto card : dealt_cards) state->ApplyAction(card);
53
54 // The first card is D3
55 state->ApplyAction(5);
56
57 // Player 1 plays C3
58 state->ApplyAction(4);
59
60 // Player 2 plays C8
61 state->ApplyAction(24);
62
63 // Check the current actions are color nomination
64 SPIEL_CHECK_EQ(state->CurrentPlayer(), 2);
65 std::vector<Action> legal_actions = state->LegalActions();
66 SPIEL_CHECK_EQ(static_cast<int>(legal_actions.size()), kNumSuits);
67
68 for (int i = 0; i < kNumSuits; ++i) {
69 SPIEL_CHECK_GE(legal_actions[i], kNominateSuitActionBase);
70 SPIEL_CHECK_LT(legal_actions[i], kNominateSuitActionBase + kNumSuits);
71 }
72
73 // The next suit is H
74 state->ApplyAction(kNominateSuitActionBase + 2);
75
76 SPIEL_CHECK_EQ(state->CurrentPlayer(), 3);
77 // Player 3 plays HA
78 state->ApplyAction(50);
79 // Reverse direction to player 2
80 SPIEL_CHECK_EQ(state->CurrentPlayer(), 2);
81 // Player 2 plays DA
82 state->ApplyAction(49);
83 SPIEL_CHECK_EQ(state->CurrentPlayer(), 3);
84 // Reverse direction to player 3
85 // Player 3 plays D8
86 state->ApplyAction(25);
87 // Player 3 nominates D
88 state->ApplyAction(kNominateSuitActionBase + 1);
89
90 SPIEL_CHECK_EQ(state->CurrentPlayer(), 0);
91 // Player 0 plays DQ
92 state->ApplyAction(41);
93
94 // Player 1 is skipped, next is player 2
95 SPIEL_CHECK_EQ(state->CurrentPlayer(), 2);
96
97 // Player 2 plays D2!
98 state->ApplyAction(1);
99 // Player 3 only has two actions: H2 or start drawing
100 legal_actions = state->LegalActions();
101 SPIEL_CHECK_EQ(static_cast<int>(legal_actions.size()), 2);
102 SPIEL_CHECK_EQ(legal_actions[0], 2);
103 SPIEL_CHECK_EQ(legal_actions[1], kDraw);
104 // Let's stack the twos!
105 state->ApplyAction(2);
106 SPIEL_CHECK_EQ(state->CurrentPlayer(), 0);
107
108 // Keep stacking
109 state->ApplyAction(3);
110 SPIEL_CHECK_EQ(state->CurrentPlayer(), 1);
111
112 // Keep stacking
113 state->ApplyAction(0);
114 SPIEL_CHECK_EQ(state->CurrentPlayer(), 2);
115 legal_actions = state->LegalActions();
116 SPIEL_CHECK_EQ(static_cast<int>(legal_actions.size()), 1);
117 // Player 2 has to draw 8 cards
118
119 state->ApplyAction(kDraw);
120 std::vector<int> draw_cards = {6, 7, 8, 9, 10, 11, 12, 13};
121 for (auto card : draw_cards) state->ApplyAction(card);
122 // Then it is player 3's turn
123 SPIEL_CHECK_EQ(state->CurrentPlayer(), 3);
124}
