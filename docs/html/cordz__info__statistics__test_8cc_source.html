50 // Creates a flat of the specified allocated size
51 CordRepFlat* Flat(size_t size) {
52  // Round up to a tag size, as we are going to poke an exact tag size back into
53  // the allocated flat. 'size returning allocators' could grant us more than we
54  // wanted, but we are ok to poke the 'requested' size in the tag, even in the
55  // presence of sized deletes, so we need to make sure the size rounds
56  // perfectly to a tag value.
57  assert(size >= kMinFlatSize);
58  size = RoundUpForTag(size);
59  CordRepFlat* flat = CordRepFlat::New(size - kFlatOverhead);
60  flat->tag = AllocatedSizeToTag(size);
61  flat->length = size - kFlatOverhead;
62  return flat;
63 }

80 // Reference count helper
81 struct RefHelper {
82  std::vector<CordRep*> refs;
83 
84  ~RefHelper() {
85  for (CordRep* rep : refs) {
86  CordRep::Unref(rep);
87  }
88  }
89 
90  // Invokes CordRep::Unref() on `rep` when this instance is destroyed.
91  template <typename T>
92  T* NeedsUnref(T* rep) {
93  refs.push_back(rep);
94  return rep;
95  }
96 
97  // Adds `n` reference counts to `rep` which will be unreffed when this
98  // instance is destroyed.
99  template <typename T>
100  T* Ref(T* rep, size_t n = 1) {
101  while (n--) {
102  NeedsUnref(CordRep::Ref(rep));
103  }
104  return rep;
105  }
106 };

131 // Computes fair share memory used in a naive 'we dare to recurse' way.
132 double FairShareImpl(CordRep* rep, size_t ref) {
133  double self = 0.0, children = 0.0;
134  ref *= rep->refcount.Get();
135  if (rep->tag >= FLAT) {
136  self = SizeOf(rep->flat());
137  } else if (rep->tag == EXTERNAL) {
138  self = SizeOf(rep->external());
139  } else if (rep->tag == SUBSTRING) {
140  self = SizeOf(rep->substring());
141  children = FairShareImpl(rep->substring()->child, ref);
142  } else if (rep->tag == BTREE) {
143  self = SizeOf(rep->btree());
144  for (CordRep*edge : rep->btree()->Edges()) {
145  children += FairShareImpl(edge, ref);
146  }
147  } else if (rep->tag == RING) {
148  self = SizeOf(rep->ring());
149  rep->ring()->ForEach([&](CordRepRing::index_type i) {
150  self += FairShareImpl(rep->ring()->entry_child(i), 1);
151  });
152  } else {
153  assert(false);
154  }
155  return self / ref + children;
156 }
157 
158 // Returns the fair share memory size from `ShareFhareImpl()` as a size_t.
159 size_t FairShare(CordRep* rep, size_t ref = 1) {
160  return static_cast<size_t>(FairShareImpl(rep, ref));
161 }

405  } else {
406  tree = CordRepBtree::Append(tree, flat1);
407  }
408 
409  auto* flat2 = Flat(200);

411  mem_size += SizeOf(flat2) + SizeOf(substr);
412  tree = CordRepBtree::Append(tree, substr);
413 
414  auto* external = External(30);
415  mem_size += SizeOf(external);
416  tree = CordRepBtree::Append(tree, external);
417

419  auto* flat3 = Flat(70);

444  expected.node_counts.flat_256 = leaf_count;
445  expected.node_counts.external = leaf_count;
446  expected.node_counts.substring = leaf_count;
447  expected.node_counts.btree = 1 + leaf_count;

470 TEST(CordzInfoStatisticsTest, ThreadSafety) {
471  Notification stop;
472  static constexpr int kNumThreads = 8;
473  int64_t sampled_node_count = 0;
474 
475  {
476  absl::synchronization_internal::ThreadPool pool(kNumThreads);
477 
478  // Run analyzer thread emulating a CordzHandler collection.
479  pool.Schedule([&]() {
480  while (!stop.HasBeenNotified()) {
481  // Run every 10us (about 100K total collections).
482  absl::SleepFor(absl::Microseconds(10));
483  CordzSampleToken token;
484  for (const CordzInfo& cord_info : token) {
485  CordzStatistics stats = cord_info.GetCordzStatistics();
486  sampled_node_count += stats.node_count;
487  }
488  }
489  });
490 
491  // Run 'application threads'
492  for (int i = 0; i < kNumThreads; ++i) {
493  pool.Schedule([&]() {
494  // Track 0 - 2 cordz infos at a time, providing permutations of 0, 1
495  // and 2 CordzHandle and CordzInfo queues being active, with plenty of
496  // 'empty to non empty' transitions.
497  InlineData cords[2];
498  std::minstd_rand gen;
499  std::uniform_int_distribution<int> coin_toss(0, 1);
500 
501  while (!stop.HasBeenNotified()) {
502  for (InlineData& cord : cords) {
503  // 50/50 flip the state of the cord
504  if (coin_toss(gen) != 0) {
505  if (cord.is_tree()) {
506  // 50/50 simulate delete (untrack) or 'edit to empty'
507  if (coin_toss(gen) != 0) {
508  CordzInfo::MaybeUntrackCord(cord.cordz_info());
509  } else {
510  CordzUpdateScope scope(cord.cordz_info(),
511  CordzUpdateTracker::kUnknown);
512  scope.SetCordRep(nullptr);
513  }
514  CordRep::Unref(cord.as_tree());
515  cord.set_inline_size(0);
516  } else {
517  // Coin toss to 25% ring, 25% btree, and 50% flat.
518  CordRep* rep = Flat(256);
519  if (coin_toss(gen) != 0) {
520  if (coin_toss(gen) != 0) {
521  rep = CordRepRing::Create(rep);
522  } else {
523  rep = CordRepBtree::Create(rep);
524  }
525  }
526  cord.make_tree(rep);
527 
528  // 50/50 sample
529  if (coin_toss(gen) != 0) {
530  CordzInfo::TrackCord(cord, CordzUpdateTracker::kUnknown);
531  }
532  }
533  }
534  }
535  }
536  for (InlineData& cord : cords) {
537  if (cord.is_tree()) {
538  CordzInfo::MaybeUntrackCord(cord.cordz_info());
539  CordRep::Unref(cord.as_tree());
540  }
541  }
542  });
543  }
544 
545  // Run for 1 second to give memory and thread safety analyzers plenty of
546  // time to detect any mishaps or undefined behaviors.
547  absl::SleepFor(absl::Seconds(1));
548  stop.Notify();
549  }
550 
551  std::cout << "Sampled " << sampled_node_count << " nodes\n";
552 }
