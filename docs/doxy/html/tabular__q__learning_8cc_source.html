31Action TabularQLearningSolver::GetBestAction(const State& state,
32 double min_utility) {
33 vector<Action> legal_actions = state.LegalActions();
34 SPIEL_CHECK_GT(legal_actions.size(), 0);
35 const auto state_str = state.ToString();
36
37 Action best_action = legal_actions[0];
38 double value = min_utility;
39 for (const Action& action : legal_actions) {
40 double q_val = values_[{state_str, action}];
41 if (q_val >= value) {
42 value = q_val;
43 best_action = action;
44 }
45 }
46 return best_action;
47}

49double TabularQLearningSolver::GetBestActionValue(const State& state,
50 double min_utility) {
51 if (state.IsTerminal()) {
52 // q(s,a) is 0 when s is terminal.
53 return 0;
54 }
55 return values_[{state.ToString(), GetBestAction(state, min_utility)}];
56}

58std::pair<Action, bool>
59TabularQLearningSolver::SampleActionFromEpsilonGreedyPolicy(
60 const State& state, double min_utility) {
61 vector<Action> legal_actions = state.LegalActions();
62 if (legal_actions.empty()) {
63 return {kInvalidAction, false};
64 }
65
66 if (absl::Uniform(rng_, 0.0, 1.0) < epsilon_) {
67 // Choose a random action
68 return {legal_actions[absl::Uniform<int>(rng_, 0, legal_actions.size())],
69 true};
70 }
71 // Choose the best action
72 return {GetBestAction(state, min_utility), false};
73}

75void TabularQLearningSolver::SampleUntilNextStateOrTerminal(State* state) {
76 // Repeatedly sample while chance node, so that we end up at a decision node
77 while (state->IsChanceNode() && !state->IsTerminal()) {
78 std::vector<std::pair<Action, double>> outcomes = state->ChanceOutcomes();
79 state->ApplyAction(SampleAction(outcomes, rng_).first);
80 }
81}

83TabularQLearningSolver::TabularQLearningSolver(std::shared_ptr<const Game> game)
84 : game_(game),
85 depth_limit_(kDefaultDepthLimit),
86 epsilon_(kDefaultEpsilon),
87 learning_rate_(kDefaultLearningRate),
88 discount_factor_(kDefaultDiscountFactor),
89 lambda_(kDefaultLambda) {
90 SPIEL_CHECK_LE(lambda_, 1);
91 SPIEL_CHECK_GE(lambda_, 0);
92
93 // Currently only supports 1-player or 2-player zero sum games
94 SPIEL_CHECK_TRUE(game_->NumPlayers() == 1 || game_->NumPlayers() == 2);
95 if (game_->NumPlayers() == 2) {
96 SPIEL_CHECK_EQ(game_->GetType().utility, GameType::Utility::kZeroSum);
97 }
98
99 // No support for simultaneous games (needs an LP solver). And so also must
100 // be a perfect information game.
101 SPIEL_CHECK_EQ(game_->GetType().dynamics, GameType::Dynamics::kSequential);
102 SPIEL_CHECK_EQ(game_->GetType().information,
103 GameType::Information::kPerfectInformation);
104}

136void TabularQLearningSolver::RunIteration() {
137 const double min_utility = game_->MinUtility();
138 // Choose start state
139 std::unique_ptr<State> curr_state = game_->NewInitialState();
140 SampleUntilNextStateOrTerminal(curr_state.get());
141
142 while (!curr_state->IsTerminal()) {
143 const Player player = curr_state->CurrentPlayer();
144
145 // Sample action from the state using an epsilon-greedy policy
146 auto [curr_action, chosen_uniformly] =
147 SampleActionFromEpsilonGreedyPolicy(*curr_state, min_utility);
148
149 std::unique_ptr<State> next_state = curr_state->Child(curr_action);
150 SampleUntilNextStateOrTerminal(next_state.get());
151
152 const double reward = next_state->Rewards()[player];
153 // Next q-value in perspective of player to play at curr_state (important
154 // note: exploits property of two-player zero-sum)
155 const double next_q_value =
156 (player != next_state->CurrentPlayer() ? -1 : 1) *
157 GetBestActionValue(*next_state, min_utility);
158
159 // Update the q value
160 std::string key = curr_state->ToString();
161 double new_q_value = reward + discount_factor_ * next_q_value;
162
163 double prev_q_val = values_[{key, curr_action}];
164 if (lambda_ == 0) {
165 // If lambda_ is equal to zero run Q-learning as usual.
166 // It's not necessary to update eligibility traces.
167 values_[{key, curr_action}] +=
168 learning_rate_ * (new_q_value - prev_q_val);
169 } else {
170 double lambda =
171 player != next_state->CurrentPlayer() ? -lambda_ : lambda_;
172 eligibility_traces_[{key, curr_action}] += 1;
173
174 for (const auto& q_cell : values_) {
175 std::string state = q_cell.first.first;
176 Action action = q_cell.first.second;
177
178 values_[{state, action}] += learning_rate_ *
179 (new_q_value - prev_q_val) *
180 eligibility_traces_[{state, action}];
181 if (chosen_uniformly) {
182 eligibility_traces_[{state, action}] = 0;
183 } else {
184 eligibility_traces_[{state, action}] *= discount_factor_ * lambda;
185 }
186 }
187 }
188
189 curr_state = std::move(next_state);
190 }
191}
