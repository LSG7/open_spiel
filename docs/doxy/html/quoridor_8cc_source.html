74class QuoridorState::SearchState {
75 using DistanceAndMove = std::pair<int, Move>;
76
77 class SearchQueue
78 : public std::priority_queue<DistanceAndMove,
79 std::vector<DistanceAndMove>,
80 std::greater<DistanceAndMove>> {
81 public:
82 void clear() { c.clear(); }
83 void reserve(int capacity) { c.reserve(capacity); }
84 };
85
86 public:
87 explicit SearchState(int board_diameter) {
88 int size = board_diameter * board_diameter;
89 mark_.resize(size, false);
90 on_shortest_path_.resize(size, false);
91 distance_.resize(size, UndefinedDistance());
92 queue_.reserve(size);
93 }
94
95 bool IsEmpty() const { return queue_.empty(); }
96
97 void ClearSearchQueue() { queue_.clear(); }
98
99 bool Push(int dist, Move move) {
100 if (mark_[move.xy] == false) {
101 mark_[move.xy] = true;
102 queue_.emplace(dist, move);
103 return true;
104 } else {
105 return false;
106 }
107 }
108
109 Move Pop() {
110 Move move = queue_.top().second;
111 queue_.pop();
112 return move;
113 }
114
115 void ResetSearchQueue() {
116 std::fill(mark_.begin(), mark_.end(), false);
117 queue_.clear();
118 }
119
120 void ResetDists() {
121 std::fill(distance_.begin(), distance_.end(), UndefinedDistance());
122 }
123
124 void SetDist(Move move, int dist) { distance_[move.xy] = dist; }
125 int GetDist(Move move) const { return distance_[move.xy]; }
126 void SetOnShortestPath(Move move) { on_shortest_path_[move.xy] = true; }
127 bool IsOnShortestPath(Move move) const { return on_shortest_path_[move.xy]; }
128
129 static constexpr int UndefinedDistance() { return -1; }
130
131 private:
132 SearchQueue queue_;
133 std::vector<bool> mark_; // Whether this position has been pushed before.
134 std::vector<int> distance_; // Distance from player.
135 std::vector<bool> on_shortest_path_; // Is this position on a shortest path?
136};

151QuoridorState::QuoridorState(std::shared_ptr<const Game> game, int board_size,
152 int wall_count, bool ansi_color_output)
153 : State(game),
154 board_size_(board_size),
155 board_diameter_(board_size * 2 - 1),
156 ansi_color_output_(ansi_color_output),
157 // See ActionToMove for explanation of the below
158 base_for_relative_(2, 2, board_diameter_) {
159 board_.resize(board_diameter_ * board_diameter_, kPlayerNone);
160 players_.resize(num_players_);
161 // Account for order of turns (order of play is clockwise)
162 if (num_players_ == 2) {
163 players_[0] = kPlayer1;
164 players_[1] = kPlayer2;
165 } else if (num_players_ == 3) {
166 players_[0] = kPlayer1;
167 players_[1] = kPlayer3;
168 players_[2] = kPlayer2;
169 } else if (num_players_ == 4) {
170 players_[0] = kPlayer1;
171 players_[1] = kPlayer3;
172 players_[2] = kPlayer2;
173 players_[3] = kPlayer4;
174 }
175 wall_count_.resize(num_players_);
176 player_loc_.resize(num_players_);
177 end_zone_.resize(num_players_);
178 for (int i = 0; i < num_players_; ++i) {
179 wall_count_[players_[i]] = wall_count;
180 InitializePlayer(players_[i]);
181 }
182}

184void QuoridorState::InitializePlayer(QuoridorPlayer p) {
185 int center_field = board_size_ - (board_size_ % 2);
186 if (p == kPlayer1) {
187 player_loc_[p] = GetMove(center_field, board_diameter_ - 1);
188 SetPlayer(player_loc_[p], p, kPlayerNone);
189 end_zone_[p] = 0;
190 return;
191 }
192 if (p == kPlayer2) {
193 player_loc_[p] = GetMove(center_field, 0);
194 SetPlayer(player_loc_[p], kPlayer2, kPlayerNone);
195 end_zone_[p] = board_diameter_ - 1;
196 return;
197 }
198 if (p == kPlayer3) {
199 player_loc_[p] = GetMove(0, center_field);
200 SetPlayer(player_loc_[p], p, kPlayerNone);
201 end_zone_[p] = board_diameter_ - 1;
202 return;
203 }
204 if (p == kPlayer4) {
205 player_loc_[p] = GetMove(board_diameter_ - 1, center_field);
206 SetPlayer(player_loc_[p], p, kPlayerNone);
207 end_zone_[p] = 0;
208 return;
209 }
210}

212/*
213 * The original implementation mapped action IDs to absolute board positions.
214 * This meant that moving "north" had a different ID for every pawn position.
215 * Now action IDs are encoded in the same virtual space as absolute board
216 * positions, but they indicate the pawn's relative move as if it were in
217 * square (1,1). So when we get those action IDs in, we need to convert them
218 * back into the absolute position into which we need to place the pawn.
219 */
220Move QuoridorState::ActionToMove(Action action_id) const {
221 Move move = GetMove(action_id % board_diameter_, action_id / board_diameter_);
222 if (!move.IsWall()) {
223 Move target = player_loc_[current_player_] + (move - base_for_relative_);
224 if (GetPlayer(target) == kPlayerNone) {
225 return target;
226 } else {
227 // Jumping over a player is inferred - it has the same action ID as just
228 // stepping
229 return player_loc_[current_player_] + ((move - base_for_relative_) * 2);
230 }
231 }
232 return move;
233}

235std::vector<Action> QuoridorState::LegalActions() const {
236 std::vector<Action> moves;
237 if (IsTerminal()) return moves;
238 int max_moves =
239 num_players_ > 2 ? 6 : 5; // Max legal pawn moves, including jumps.
240 if (wall_count_[current_player_] > 0) {
241 max_moves += 2 * (board_size_ - 1) * (board_size_ - 1); // Max wall moves.
242 }
243 moves.reserve(max_moves);
244
245 // Pawn moves.
246 Move cur = player_loc_[current_player_];
247 AddActions(cur, Offset(1, 0), &moves);
248 AddActions(cur, Offset(0, 1), &moves);
249 AddActions(cur, Offset(-1, 0), &moves);
250 AddActions(cur, Offset(0, -1), &moves);
251
252 // Wall placements.
253 if (wall_count_[current_player_] > 0) {
254 SearchState search_state(board_diameter_);
255 for (int i = 0; i < num_players_; ++i) {
256 SearchShortestPath(players_[i], &search_state);
257 }
258 for (int y = 0; y < board_diameter_ - 2; y += 2) {
259 for (int x = 0; x < board_diameter_ - 2; x += 2) {
260 Move h = GetMove(x, y + 1);
261 if (IsValidWall(h, &search_state)) {
262 moves.push_back(h.xy);
263 }
264 Move v = GetMove(x + 1, y);
265 if (IsValidWall(v, &search_state)) {
266 moves.push_back(v.xy);
267 }
268 }
269 }
270 }
271
272 // If no action is possible add 'pass' action to list of moves
273 if (moves.empty()) {
274 moves.push_back(cur.xy);
275 }
276
277 std::sort(moves.begin(), moves.end());
278 return moves;
279}

281void QuoridorState::AddActions(Move cur, Offset offset,
282 std::vector<Action>* moves) const {
283 SPIEL_CHECK_FALSE(cur.IsWall());
284
285 if (IsWall(cur + offset)) {
286 // Hit a wall or edge in this direction.
287 return;
288 }
289
290 Move forward = cur + offset * 2;
291 if (GetPlayer(forward) == kPlayerNone) {
292 // Normal single step in this direction.
293 moves->push_back((base_for_relative_ + offset * 2).xy);
294 return;
295 }
296
297 // Other player, so which jumps are valid?
298
299 if (!IsWall(cur + offset * 3)) {
300 // In two-players: A normal jump is allowed. We know that spot is empty.
301 // In >2 players, must check.
302 if (GetPlayer(cur + offset * 4) == kPlayerNone) {
303 // The relative action ID for jumping directly over is the same as moving
304 moves->push_back((base_for_relative_ + offset * 2).xy);
305 return;
306 } else {
307 return;
308 }
309 }
310 // We are jumping over the other player against a wall, which side jumps are
311 // valid?
312
313 Offset left = offset.rotate_left();
314 if (!IsWall(forward + left)) {
315 if (GetPlayer(forward + left * 2) == kPlayerNone) {
316 moves->push_back((base_for_relative_ + offset * 2 + left * 2).xy);
317 }
318 }
319 Offset right = offset.rotate_right();
320 if (!IsWall(forward + right)) {
321 if (GetPlayer(forward + right * 2) == kPlayerNone) {
322 moves->push_back((base_for_relative_ + offset * 2 + right * 2).xy);
323 }
324 }
325}

327bool QuoridorState::IsValidWall(Move m, SearchState* search_state) const {
328 Offset offset = (m.IsHorizontalWall() ? Offset(1, 0) : Offset(0, 1));
329
330 if (IsWall(m + offset * 0) || IsWall(m + offset * 1) ||
331 IsWall(m + offset * 2)) {
332 // Already blocked by a wall.
333 return false;
334 }
335
336 // Any wall that doesn't intersect with a shortest path is clearly legal.
337 // Walls that do intersect might still be legal because there's another way
338 // around, but that's more expensive to check.
339 if (!search_state->IsOnShortestPath(m) &&
340 !search_state->IsOnShortestPath(m + offset * 2)) {
341 return true;
342 }
343
344 // If this wall doesn't connect two existing walls/edges, then it can't cut
345 // any paths. Even connecting to a node where 3 other walls meet, but without
346 // connecting them to anything else, can't cut any paths.
347 int count = (
348 // The 3 walls near the close end.
349 (IsWall(m - offset * 2) || IsWall(m - offset + offset.rotate_left()) ||
350 IsWall(m - offset + offset.rotate_right())) +
351 // The 3 walls near the far end.
352 (IsWall(m + offset * 4) ||
353 IsWall(m + offset * 3 + offset.rotate_left()) ||
354 IsWall(m + offset * 3 + offset.rotate_right())) +
355 // The 2 walls in the middle.
356 (IsWall(m + offset + offset.rotate_left()) ||
357 IsWall(m + offset + offset.rotate_right())));
358 if (count <= 1) return true;
359
360 // Do a full search to verify both players can get to their respective goals.
361 bool pathExists = true;
362 for (int i = 0; i < num_players_; ++i) {
363 pathExists = pathExists &&
364 SearchEndZone(players_[i], m, m + offset * 2, search_state);
365 }
366 return pathExists;
367}

369bool QuoridorState::SearchEndZone(QuoridorPlayer p, Move wall1, Move wall2,
370 SearchState* search_state) const {
371 search_state->ResetSearchQueue();
372 Offset dir(1, 0); // Direction is arbitrary. Queue will make it fast.
373 int goal = end_zone_[p];
374 int goal_dir = (goal == 0 ? -1 : 1); // Sort for shortest dist in a min-heap.
375 search_state->Push(0, player_loc_[p]);
376 while (!search_state->IsEmpty()) {
377 Move c = search_state->Pop();
378 for (int i = 0; i < 4; ++i) {
379 Move wall = c + dir;
380 if (!IsWall(wall) && wall != wall1 && wall != wall2) {
381 Move move = c + dir * 2;
382 int moveCoord;
383 if (p == kPlayer1 || p == kPlayer2) {
384 moveCoord = move.y;
385 } else if (p == kPlayer3 || p == kPlayer4) {
386 moveCoord = move.x;
387 } else {
388 SpielFatalError("Case not handled for player in SearchEndZone.");
389 }
390 if (moveCoord == goal) {
391 return true;
392 }
393 search_state->Push(goal_dir * (goal - move.y), move);
394 }
395 dir = dir.rotate_left();
396 }
397 }
398
399 return false;
400}

402void QuoridorState::SearchShortestPath(QuoridorPlayer p,
403 SearchState* search_state) const {
404 search_state->ResetSearchQueue();
405 search_state->ResetDists();
406 Offset dir(1, 0); // Direction is arbitrary. Queue will make it fast.
407 int goal = end_zone_[p];
408 int goal_dir = (goal == 0 ? -1 : 1); // Sort for shortest dist in a min-heap.
409 search_state->Push(0, player_loc_[p]);
410 search_state->SetDist(player_loc_[p], 0);
411 Move goal_found = GetMove(-1, -1); // invalid
412
413 // A* search for the end-zone, keeping distances to each cell.
414 while (!search_state->IsEmpty()) {
415 Move c = search_state->Pop();
416 int dist = search_state->GetDist(c);
417 for (int i = 0; i < 4; ++i) {
418 Move wall = c + dir;
419 if (!IsWall(wall)) {
420 Move move = c + dir * 2;
421 int moveCoord;
422 if (p == kPlayer1 || p == kPlayer2) {
423 moveCoord = move.y;
424 } else if (p == kPlayer3 || p == kPlayer4) {
425 moveCoord = move.x;
426 } else {
427 SpielFatalError("Case not handled for player in SearchShortestPath");
428 }
429 if (moveCoord == goal) {
430 search_state->SetDist(move, dist + 1);
431 search_state->ClearSearchQueue(); // Break out of the search.
432 goal_found = move;
433 break;
434 }
435 if (search_state->Push(dist + 1 + goal_dir * (goal - moveCoord),
436 move)) {
437 search_state->SetDist(move, dist + 1);
438 }
439 }
440 dir = dir.rotate_left();
441 }
442 }
443
444 // Trace the way back, setting them to be on a shortest path.
445 Move current = goal_found;
446 int dist = search_state->GetDist(current);
447 while (current != player_loc_[p]) {
448 for (int i = 0; i < 4; ++i) {
449 Move wall = current + dir;
450 if (!IsWall(wall)) {
451 Move move = current + dir * 2;
452 int dist2 = search_state->GetDist(move);
453 if (dist2 != search_state->UndefinedDistance() && dist2 + 1 == dist) {
454 search_state->SetOnShortestPath(wall);
455 current = move;
456 dist = dist2;
457 break;
458 }
459 }
460 dir = dir.rotate_left();
461 }
462 }

470std::string QuoridorState::ToString() const {
471 // Generates something like:
472 // Board size: 5, walls: 0, 0
473 // a b c d e
474 // 1 . | . . . . 1
475 // + ---+---
476 // 2 . | . | . . . 2
477 // +
478 // 3 . . | O @ . 3
479 // ---+---
480 // 4 . | . . . . 4
481 // + ---+---
482 // 5 . | . . . . 5
483 // a b c d e
484
485 std::string reset;
486 std::array<std::string, 4> colors, coords;
487 if (ansi_color_output_) {
488 std::string esc = "\033";
489 reset = esc + "[0m";
490 coords[0] = esc + "[1;33m";
491 coords[1] = esc + "[1;34m";
492 coords[2] = esc + "[1;35m";
493 coords[3] = esc + "[1;36m";
494 colors[0] = esc + "[1;33m" + " O " + reset;
495 colors[1] = esc + "[1;34m" + " @ " + reset;
496 colors[2] = esc + "[1;35m" + " # " + reset;
497 colors[3] = esc + "[1;36m" + " % " + reset;
498 } else {
499 std::string reset = "";
500 coords[0] = "";
501 coords[1] = "";
502 coords[2] = "";
503 coords[3] = "";
504 colors[0] = " 0 ";
505 colors[1] = " @ ";
506 colors[2] = " # ";
507 colors[3] = " % ";
508 }
509
510 std::ostringstream out;
511 out << "Board size: " << board_size_ << ", walls: ";
512 for (int i = 0; i < num_players_; ++i) {
513 out << wall_count_[players_[i]];
514 if (i < num_players_ - 1) out << ", ";
515 }
516 out << "\n";
517
518 // Top x coords.
519 for (int x = 0; x < board_size_; ++x) {
520 out << " " << coords[1] << static_cast<char>('a' + x);
521 }
522 out << reset << '\n';
523
524 for (int y = 0; y < board_diameter_; ++y) {
525 if (y % 2 == 0) {
526 if (y / 2 + 1 < 10) out << " ";
527 out << coords[2] << (y / 2 + 1) << reset; // Leading y coord.
528 } else {
529 out << " "; // Wall lines.
530 }
531
532 for (int x = 0; x < board_diameter_; ++x) {
533 QuoridorPlayer p = GetPlayer(GetMove(x, y));
534 if (x % 2 == 0 && y % 2 == 0) {
535 bool playerFound = false;
536 for (int i = 0; i < num_players_; ++i) {
537 if (p == players_[i]) {
538 out << colors[players_[i]];
539 playerFound = true;
540 }
541 }
542 if (!playerFound) {
543 out << " . ";
544 }
545 } else if (x % 2 == 1 && y % 2 == 1) {
546 out << (p == kPlayerWall ? "+" : " ");
547 } else if (x % 2 == 1) {
548 out << (p == kPlayerWall ? "|" : " ");
549 } else if (y % 2 == 1) {
550 out << (p == kPlayerWall ? "---" : " ");
551 }
552 }
553 if (y % 2 == 0) {
554 if (y / 2 + 1 < 10) out << " ";
555 out << coords[3] << (y / 2 + 1) << reset; // y coord on the right.
556 } else {
557 out << " "; // Wall lines.
558 }
559 out << '\n';
560 }
561 // Bottom x coords.
562 for (int x = 0; x < board_size_; ++x) {
563 out << " " << coords[0] << static_cast<char>('a' + x);
564 }
565 out << reset << '\n';
566 return out.str();
567}

569std::vector<double> QuoridorState::Returns() const {
570 std::vector<double> res(num_players_, 0.0);
571 for (int i = 0; i < num_players_; ++i) {
572 if (outcome_ == players_[i]) {
573 // If someone as won, set their reward to +1 and all the others to
574 // -1 / (num_players - 1).
575 std::fill(res.begin(), res.end(), -1.0 / (num_players_ - 1));
576 res[i] = 1.0;
577 break;
578 }
579 }
580 return res;
581}

595void QuoridorState::ObservationTensor(Player player,
596 absl::Span<float> values) const {
597 SPIEL_CHECK_GE(player, 0);
598 SPIEL_CHECK_LT(player, num_players_);
599
600 TensorView<2> view(
601 values, {NumCellStates() + num_players_, static_cast<int>(board_.size())},
602 true);
603
604 for (int i = 0; i < board_.size(); ++i) {
605 if (board_[i] < NumCellStates()) {
606 view[{static_cast<int>(board_[i]), i}] = 1.0;
607 }
608 for (int j = 0; j < num_players_; ++j) {
609 view[{NumCellStates() + players_[j], i}] = wall_count_[players_[j]];
610 }
611 }

614void QuoridorState::DoApplyAction(Action action) {
615 Move move = ActionToMove(action);
616 // If players is forced to pass it is valid to stay in place, on a field where
617 // there is already a player
618 if (board_[move.xy] != current_player_) {
619 SPIEL_CHECK_EQ(board_[move.xy], kPlayerNone);
620 }
621 SPIEL_CHECK_EQ(outcome_, kPlayerNone);
622
623 SPIEL_CHECK_TRUE(move.IsValid());
624
625 if (move.IsWall()) {
626 Offset offset = (move.IsHorizontalWall() ? Offset(1, 0) : Offset(0, 1));
627 SetPlayer(move + offset * 0, kPlayerWall, kPlayerNone);
628 SetPlayer(move + offset * 1, kPlayerWall, kPlayerNone);
629 SetPlayer(move + offset * 2, kPlayerWall, kPlayerNone);
630 wall_count_[current_player_] -= 1;
631 } else {
632 SetPlayer(player_loc_[current_player_], kPlayerNone, current_player_);
633 SetPlayer(move, current_player_, kPlayerNone);
634 player_loc_[current_player_] = move;
635
636 int end_zone_coord;
637 if (current_player_ == kPlayer1 || current_player_ == kPlayer2) {
638 end_zone_coord = move.y;
639 } else {
640 end_zone_coord = move.x;
641 }
642
643 outcome_ = kPlayerNone;
644 if (end_zone_coord == end_zone_[current_player_]) {
645 outcome_ = current_player_;
646 }
647 }
648
649 ++moves_made_;
650 if (moves_made_ >= kMaxGameLengthFactor * board_size_ * board_size_) {
651 outcome_ = kPlayerDraw;
652 }
653
654 current_player_index_ += 1;
655 if (current_player_index_ == num_players_) current_player_index_ = 0;
656 current_player_ = players_[current_player_index_];
657}

663QuoridorGame::QuoridorGame(const GameParameters& params)
664 : Game(kGameType, params),
665 board_size_(ParameterValue<int>("board_size")),
666 wall_count_(
667 ParameterValue<int>("wall_count", board_size_ * board_size_ / 8)),
668 ansi_color_output_(ParameterValue<bool>("ansi_color_output")),
669 num_players_(ParameterValue<int>("players")) {
670 if (board_size_ < 3) {
671 // For relative moves, we need to be able to describe moves using a 3x3 grid
672 // and since we use the board to number the moves (see above), we need the
673 // playing board to be at least that big.
674 SpielFatalError("Board size must be at least 3x3.");
675 }
