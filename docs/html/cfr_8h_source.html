41 // A basic structure to store the relevant quantities.
42 struct CFRInfoStateValues {
43  CFRInfoStateValues() {}
44  CFRInfoStateValues(std::vector<Action> la, double init_value)
45  : legal_actions(la),
46  cumulative_regrets(la.size(), init_value),
47  cumulative_policy(la.size(), init_value),
48  current_policy(la.size(), 1.0 / la.size()) {}
49  CFRInfoStateValues(std::vector<Action> la) : CFRInfoStateValues(la, 0) {}
50 
51  // For randomized initial regrets.
52  CFRInfoStateValues(std::vector<Action> la,
53  std::mt19937* rng,
54  double magnitude_scale) : CFRInfoStateValues(la, 0) {
55  for (int i = 0; i < cumulative_policy.size(); ++i) {
56  cumulative_regrets[i] = magnitude_scale *
57  absl::Uniform<double>(*rng, 0.0, 1.0);
58  }
59  ApplyRegretMatching();
60  }

173 // Base class supporting different flavours of the Counterfactual Regret
174 // Minimization (CFR) algorithm.
175 //
176 // see https://webdocs.cs.ualberta.ca/~bowling/papers/07nips-regretpoker.pdf
177 // and http://modelai.gettysburg.edu/2013/cfr/cfr.pdf
178 //
179 // The implementation is similar to the Python version:
180 // open_spiel/python/algorithms/cfr.py
181 //
182 // The algorithm computes an approximate Nash policy for 2 player zero-sum
183 // games.
184 //
185 // CFR can be view as a policy iteration algorithm. Importantly, the policies
186 // themselves do not converge to a Nash policy, but their average does.
187 //
188 class CFRSolverBase {
189  public:
190  CFRSolverBase(const Game& game, bool alternating_updates,
191  bool linear_averaging, bool regret_matching_plus,
192  bool random_initial_regrets = false, int seed = 0);
193  // The constructor below is used for deserialization purposes.
194  CFRSolverBase(std::shared_ptr<const Game> game, bool alternating_updates,
195  bool linear_averaging, bool regret_matching_plus, int iteration,
196  bool random_initial_regrets = false, int seed = 0);
197  virtual ~CFRSolverBase() = default;
198

203  // The returned policy instance should only be used during the lifetime of
204  // the CFRSolver object.

206  return std::make_shared<CFRAveragePolicy>(info_states_, nullptr);
207  }
208  // Note: This can be quite large.
