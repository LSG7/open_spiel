38def get_method_tuple_linespecs(method):
39 """Gets plot linespecs for the specified ResponseGraphUCB method."""
40 sampling_strats = [
41 'uniform-exhaustive', 'uniform', 'valence-weighted', 'count-weighted'
42 ]
43 conf_methods = ['ucb-standard', 'clopper-pearson-ucb']
44 method_to_id_map = dict(
45 (m, i)
46 for i, m in enumerate(itertools.product(sampling_strats, conf_methods)))
47
48 # Create palette
49 num_colors = len(method_to_id_map.keys())
50 colors = plt.get_cmap('Set1', num_colors).colors
51
52 # Spec out the linestyle
53 base_method = (method[0], method[1].replace('-relaxed', '')
54 ) # Method name without -relaxed suffix
55 linespecs = {
56 'color': colors[method_to_id_map[base_method]]
57 } # Use base method for color (ignoring relaxed vs non-relaxed)
58 if 'relaxed' in method[1]: # Use actual method for linestyle
59 linespecs['linestyle'] = 'dashed'
60 else:
61 linespecs['linestyle'] = 'solid'
62
63 return linespecs

66def get_method_acronym(method):
67 """Gets pretty acronym for specified ResponseGraphUCB method."""
68 if method == 'uniform-exhaustive':
69 return r'$\mathcal{S}$: UE'
70 elif method == 'uniform':
71 return r'$\mathcal{S}$: U'
72 elif method == 'valence-weighted':
73 return r'$\mathcal{S}$: VW'
74 elif method == 'count-weighted':
75 return r'$\mathcal{S}$: CW'
76 elif method == 'ucb-standard':
77 return r'$\mathcal{C}(\delta)$: UCB'
78 elif method == 'ucb-standard-relaxed':
79 return r'$\mathcal{C}(\delta)$: R-UCB'
80 elif method == 'clopper-pearson-ucb':
81 return r'$\mathcal{C}(\delta)$: CP-UCB'
82 elif method == 'clopper-pearson-ucb-relaxed':
83 return r'$\mathcal{C}(\delta)$: R-CP-UCB'
84 elif method == 'fixedbudget-uniform':
85 return r'$\mathcal{S}$: U, $\mathcal{C}(\delta)$: FB'
86 else:

90def digraph_edge_hamming_dist(g1, g2):
91 """Returns number of directed edge mismatches between digraphs g1 and g2."""
92 dist = 0
93 for e1 in g1.edges:
94 if e1 not in g2.edges:
95 dist += 1
96 return dist

99class BernoulliGameSampler(object):
100 """A sampler for a game with Bernoulli-distributed payoffs."""
101
102 def __init__(self, strategy_spaces, means, payoff_bounds):
103 """Initializes the Bernoulli game sampler.
104
105 Payoffs are automatically scaled to lie between 0 and 1.
106
107 Args:
108 strategy_spaces: a list of sizes of player strategy spaces.
109 means: 1+num_players dimensional array of mean payoffs.
110 payoff_bounds: min/max observable value of payoffs, necessary since one
111 may seek Bernoulli-sampling for games with different payoff ranges.
112 """
113 self.strategy_spaces = strategy_spaces
114 self.n_players = len(strategy_spaces)
115 self.raw_means = means
116 self.payoff_bounds = payoff_bounds
117 self.means = self.rescale_payoff(means)
118
119 # Specific to the Bernoulli case. Probas in [0,1], proportional to payoffs
120 self.p_max = self.means
121
122 def rescale_payoff(self, payoff):
123 """Rescales payoffs to be in [0,1]."""
124 # Assumes payoffs are bounded between [-payoff_bound, payoff_bound]
125 return (payoff - self.payoff_bounds[0]) / (
126 self.payoff_bounds[1] - self.payoff_bounds[0])
127
128 def observe_result(self, strat_profile):
129 """Returns empirical payoffs for each agent."""
130 outcomes = np.zeros(self.n_players)
131 for k in range(self.n_players):
132 # compute Bernoulli probabilities
133 outcomes[k] = np.random.choice(
134 [1, 0],
135 p=[self.p_max[k][strat_profile], 1. - self.p_max[k][strat_profile]])
136 return outcomes

139class ZeroSumBernoulliGameSampler(BernoulliGameSampler):
140 """A sampler for a zero-sum game with Bernoulli-distributed payoffs."""

144 payoff_bounds)
145 # Verify the game is zero-sum
146 assert np.allclose(np.sum(self.meansmeans, axis=0), 1.)

149 outcomes = np.zeros(self.n_playersn_players)
150 win_ix = np.random.choice(
151 self.n_playersn_players, p=self.meansmeans[(slice(None),) + strat_profile])
152 outcomes[win_ix] = 1.
153 return outcomes

156def get_payoffs_bernoulli_game(size=(2, 2, 2)):
157 """Gets randomly-generated zero-sum symmetric two-player game."""
158 too_close = True
159 while too_close:
160 M = np.random.uniform(-1, 1, size=size) # pylint: disable=invalid-name
161 M[0, :, :] = 0.5 * (M[0, :, :] - M[0, :, :].T)
162 M[1, :, :] = -M[0, :, :]
163 if np.abs(M[0, 0, 1]) < 0.1:
164 too_close = True
165 else:
166 too_close = False
167 return M

178def get_kuhn_poker_data(num_players=4, iterations=3):
179 """Returns the kuhn poker data for the number of players specified."""
180 game = pyspiel.load_game('kuhn_poker', {'players': num_players})
181 xfp_solver = fictitious_play.XFPSolver(game, save_oracles=True)
182 for _ in range(iterations):
183 xfp_solver.iteration()
184
185 # Results are seed-dependent, so show some interesting cases
186 if num_players == 2:
187 meta_games = xfp_solver.get_empirical_metagame(100, seed=1)
188 elif num_players == 3:
189 meta_games = xfp_solver.get_empirical_metagame(100, seed=5)
190 elif num_players == 4:
191 meta_games = xfp_solver.get_empirical_metagame(100, seed=2)
192
193 # Metagame utility matrices for each player
194 payoff_tables = []
195 for i in range(num_players):
196 payoff_tables.append(meta_games[i])
197 return payoff_tables

200def get_game_for_sampler(game_name):
201 """Returns pre-processed game data for ResponseGraphUCB examples."""
202 # pylint: disable=invalid-name
203 if game_name == 'bernoulli':
204 M = get_payoffs_bernoulli_game()
205 strategy_spaces = [2, 2]
206 G = ZeroSumBernoulliGameSampler(
207 strategy_spaces, means=M, payoff_bounds=[-1., 1.])
208 elif game_name == 'soccer':
209 M = get_soccer_data()
210 M = M * 2. - 1 # Convert to zero-sum
211 strategy_spaces = np.shape(M)
212 M = np.asarray([M, M.T])
213 G = ZeroSumBernoulliGameSampler(strategy_spaces, means=M,
214 payoff_bounds=[np.min(M), np.max(M)])
215 elif game_name in ['kuhn_poker_2p', 'kuhn_poker_3p', 'kuhn_poker_4p']:
216 if '2p' in game_name:
217 num_players = 2
218 elif '3p' in game_name:
219 num_players = 3
220 elif '4p' in game_name:
221 num_players = 4
222 M = get_kuhn_poker_data(num_players, iterations=2) # pylint: disable=invalid-name
223 strategy_spaces = egt_utils.get_num_strats_per_population(M, False)
224 G = BernoulliGameSampler(
225 strategy_spaces, means=M, payoff_bounds=[np.min(M), np.max(M)])
226 else:
227 raise ValueError('Game', game_name, 'not implemented!')
228 # pylint: enable=invalid-name
229 return G

232def plot_timeseries(ax, id_ax, data, xticks, xlabel='', ylabel='',
233 label='', logx=False, logy=False, zorder=10,
234 linespecs=None):
235 """Plots timeseries data with error bars."""
236 if logx:
237 ax[id_ax].set_xscale('log')
238 if logy:
239 ax[id_ax].set_yscale('log')
240 if linespecs:
241 kwargs = {'color': linespecs['color']}
242 else:
243 kwargs = {}
244
245 # Seaborn's bootstrapped confidence intervals were used in the original paper
246 se = scipy.stats.sem(data, axis=0)
247 ax[id_ax].fill_between(xticks, data.mean(0)-se, data.mean(0)+se,
248 zorder=zorder, alpha=0.2, **kwargs)
249 ax[id_ax].plot(xticks, data.mean(0), label=label, zorder=zorder, **kwargs)
250
251 # There may be multiple lines on the current axis, some from previous calls to
252 # plot_timeseries, so reference just the latest
253 if linespecs:
254 ax[id_ax].get_lines()[-1].set_dashes([5, 5])
255 ax[id_ax].get_lines()[-1].set_linestyle(linespecs['linestyle'])
256
257 ax[id_ax].set(xlabel=xlabel, ylabel=ylabel)
258 ax[id_ax].set_axisbelow(True)
259 ax[id_ax].grid(True)
260 for _, spine in ax[id_ax].spines.items():
261 spine.set_zorder(-1)
