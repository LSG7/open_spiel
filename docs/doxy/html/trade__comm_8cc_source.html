60std::pair<int, int> DecodeAllocation(Action chance_action, int num_items) {
61 return {chance_action / num_items, chance_action % num_items};
62}
63
64std::pair<int, int> DecodeTrade(Action trade_action, int num_items) {
65 std::pair<int, int> trade = {(trade_action - num_items) / num_items,
66 (trade_action - num_items) % num_items};
67 return trade;
68}

71std::string TradeCommState::ActionToString(Player player,
72 Action move_id) const {

75 return absl::StrCat("Allocate ", allocation.first, " ", allocation.second);
76 } else {

80 std::pair<int, int> trade = DecodeTrade(move_id, num_items_);
81 return absl::StrCat("Trade ", trade.first, ":", trade.second);

90std::vector<double> TradeCommState::Returns() const {
91 if (!IsTerminal()) {
92 return {0.0, 0.0};
93 } else {
94 // Check for a compatible trade. A compatible trade satisfies:
95 // - Agent X has item A, and offers A for B
96 // - Agent Y has item B, and offers B for A
97 std::pair<int, int> trade0 = DecodeTrade(trade_history_[0], num_items_);
98 std::pair<int, int> trade1 = DecodeTrade(trade_history_[1], num_items_);
99 if (items_[0] == trade0.first && items_[1] == trade1.first &&
100 trade0.first == trade1.second && trade1.first == trade0.second) {
101 return {1.0, 1.0};
102 } else {
103 return {0.0, 0.0};
104 }
105 }

108std::string TradeCommState::ObservationString(Player player) const {
109 SPIEL_CHECK_GE(player, 0);
110 SPIEL_CHECK_LT(player, num_players_);
111 if (IsChanceNode()) {
112 return "ChanceNode -- no observation";
113 }
114
115 std::string str = "";
116
117 // Whose turn is it?
118 absl::StrAppend(&str, "Current turn: ", cur_player_, "\n");
119
120 // A player can see their own item.
121 absl::StrAppend(&str, "My item: ", items_[player], "\n");
122
123 // A player see all the utterances, in the right order:
124 absl::StrAppend(&str, "Phase: ", phase_ == Phase::kTrade ? "trade" : "comm");
125 absl::StrAppend(&str, "\nComm history: ");
126 for (int comm : comm_history_) {
127 absl::StrAppend(&str, " ", comm);
128 }
129 absl::StrAppend(&str, "\n");
130
131 // Trade proposals are treated as simultaneous, so not included in the
132 // observation, but we do mark how many trade actions have happened to agents
133 // can work out what trading round they're on.
134 absl::StrAppend(&str, "Trade history size: ", trade_history_.size(), "\n");
135
136 // Players can see their own trades if they were made.
137 if (player < trade_history_.size()) {
138 absl::StrAppend(&str, "Observer's trade offer: ");
139 std::pair<int, int> trade = DecodeTrade(trade_history_[player], num_items_);
140 absl::StrAppend(&str, " ", trade.first, ":", trade.second, "\n");
141 }
142
143 // Players can see the other trade offers after the round.
144 if (IsTerminal()) {
145 SPIEL_CHECK_LT(1 - player, trade_history_.size());
146 absl::StrAppend(&str, "Other players's trade offer: ");
147 std::pair<int, int> trade =
148 DecodeTrade(trade_history_[1 - player], num_items_);
149 absl::StrAppend(&str, " ", trade.first, ":", trade.second, "\n");
150 }
151
152 return str;
153}

162void TradeCommState::ObservationTensor(Player player,
163 absl::Span<float> values) const {
164 SPIEL_CHECK_GE(player, 0);
165 SPIEL_CHECK_LT(player, num_players_);
166
167 SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorSize());
168 std::fill(values.begin(), values.end(), 0);
169
170 if (IsChanceNode()) {
171 // No observations at chance nodes.
172 return;
173 }
174
175 SPIEL_CHECK_TRUE(player == 0 || player == 1);
176
177 // 2 bits to indicate whose turn it is.
178 int offset = 0;
179 values[cur_player_] = 1;
180 offset += 2;
181
182 // 1 bit to indicate whether it's terminal
183 values[offset] = IsTerminal() ? 1 : 0;
184 offset += 1;
185
186 // Single bit for the phase: 0 = comm, 1 = trade.
187 values[offset] = (phase_ == Phase::kCommunication ? 0 : 1);
188 offset += 1;
189
190 // one-hot vector for the item the observing player got
191 values[offset + items_[player]] = 1;
192 offset += num_items_;
193
194 if (player < comm_history_.size()) {
195 // one-hot vector for the utterance the observing player made
196 values[offset + comm_history_[player]] = 1;
197 }
198 offset += num_items_;
199
200 // one-hot vector for the utterance the observing player observed
201 if (1 - player < comm_history_.size()) {
202 values[offset + comm_history_[1 - player]] = 1;
203 }
204 offset += num_items_;
205
206 // one-hot vector for the size of the trade history
207 values[offset + trade_history_.size()] = 1;
208 offset += 3;
209
210 // one-hot vector for observing player's trade history if it has been made.
211 if (player < trade_history_.size()) {
212 const auto& trade = DecodeTrade(trade_history_[player], num_items_);
213 values[offset + trade.first] = 1;
214 values[offset + num_items_ + trade.second] = 1;
215 }
216 offset += 2 * num_items_;
217
218 SPIEL_CHECK_EQ(offset, values.size());
219}

239void TradeCommState::DoApplyAction(Action action) {
240 if (IsChanceNode()) {
241 std::pair<int, int> allocation = DecodeAllocation(action, num_items_);
242 items_.push_back(allocation.first);
243 items_.push_back(allocation.second);
244 cur_player_ = 0;
245 } else {
246 if (phase_ == Phase::kCommunication) {
247 comm_history_.push_back(action);
248 if (comm_history_.size() == 2) {
249 phase_ = Phase::kTrade;
250 }
251 cur_player_ = NextPlayerRoundRobin(cur_player_, num_players_);
252 } else {
253 trade_history_.push_back(action);
254 cur_player_ = NextPlayerRoundRobin(cur_player_, num_players_);
255 }
256 }
257}

259std::vector<Action> TradeCommState::LegalActions() const {
260 if (IsChanceNode()) {
261 return LegalChanceOutcomes();
262 } else if (IsTerminal()) {
263 return {};
264 } else if (phase_ == Phase::kCommunication) {
265 // Can utter anything. Utterances are actions 0 to num_items_ - 1 for now.
266 std::vector<Action> legal_actions;
267 legal_actions.reserve(num_items_);
268 for (int i = 0; i < num_items_; ++i) {
269 legal_actions.push_back(i);
270 }
271 return legal_actions;
272 } else if (phase_ == Phase::kTrade) {
273 // 1:1 trades for k items = k*k actions (includes trading an item for the
274 // same item) starting at num_items_.
275 std::vector<Action> legal_actions;
276 int num_trade_actions = num_items_ * num_items_;
277 legal_actions.reserve(num_trade_actions);
278 for (int i = 0; i < num_trade_actions; ++i) {
279 legal_actions.push_back(num_items_ + i);
280 }
281 return legal_actions;
282 } else {
283 SpielFatalError("Invalid phase?");
284 }
285}

287std::vector<std::pair<Action, double>> TradeCommState::ChanceOutcomes() const {
288 SPIEL_CHECK_TRUE(IsChanceNode());
289 int num_outcomes = num_items_ * num_items_;
290 std::vector<std::pair<Action, double>> outcomes;
291 outcomes.reserve(num_outcomes);
292 for (int i = 0; i < num_outcomes; ++i) {
293 outcomes.push_back({i, 1.0 / num_outcomes});
294 }
295 return outcomes;
296}

298std::string TradeCommState::ToString() const {
299 if (IsChanceNode()) {
300 return "Initial chance node";
301 }
302
303 std::string str = absl::StrCat("Items: ", absl::StrJoin(items_, " "));
304 absl::StrAppend(&str,
305 "\nPhase: ", phase_ == Phase::kTrade ? "trade" : "comm");
306 absl::StrAppend(&str, "\nComm history: ", absl::StrJoin(comm_history_, " "));
307 absl::StrAppend(&str, "\nTrade history:");
308 for (Action trade_action : trade_history_) {
309 std::pair<int, int> trade = DecodeTrade(trade_action, num_items_);
310 absl::StrAppend(&str, " ", trade.first, ":", trade.second);
311 }
312 absl::StrAppend(&str, "\n");
313
314 return str;

330std::vector<int> TradeCommGame::ObservationTensorShape() const {
331 return {
332 2 + // one hot vector for whose turn it is
333 1 + // one bit to indicate whether the state is terminal
334 1 + // a single bit indicating the phase (comm or trade)
335 num_items_ + // one-hot vector for the item the player got
336 num_items_ + // one-hot vector for the utterance the player made
337 num_items_ + // one-hot vector for the utterance the player observed
338 3 + // trade history size
339 2 * num_items_ // observer's trade if made.
340 };
341}
