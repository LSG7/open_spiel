29// SARSA algorithm: solves for the optimal action value function of a game.
30// It considers all states with depth at most depth_limit from the
31// initial state (so if depth_limit is 0, only the root is considered).
32// If depth limit is negative, all states are considered.
33//
34// Currently works for sequential 1-player or 2-player zero-sum games.
35//
36// Based on the implementation in Sutton and Barto, Intro to RL. Second Edition,
37// 2018. Section 6.4.
38//
39// Includes implementation of SARSA(lambda) which can be found in
40// Sutton and Barto, Intro to RL. Second Edition, 2018. Section 12.7.
41// (E.g. https://www.andrew.cmu.edu/course/10-703/textbook/BartoSutton.pdf)
42// Eligibility traces are implemented with the "accumulate"
43// method (+1 at each iteration) instead of "replace" implementation
44// (doesn't sum trace values). Parameter lambda_ determines the level
45// of bootstraping.
46
47class TabularSarsaSolver {
48 static inline constexpr double kDefaultDepthLimit = -1;
49 static inline constexpr double kDefaultEpsilon = 0.1;
50 static inline constexpr double kDefaultLearningRate = 0.01;
51 static inline constexpr double kDefaultDiscountFactor = 0.99;
52 static inline constexpr double kDefaultLambda = 0;

55 TabularSarsaSolver(std::shared_ptr<const Game> game);
