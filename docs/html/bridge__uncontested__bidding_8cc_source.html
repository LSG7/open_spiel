80 bool Is2NTDeal(const Deal& deal) {
81  int lengths[kNumSuits] = {0, 0, 0, 0};
82  int hcp = 0;
83  for (int i = 0; i < kNumCardsPerHand; ++i) {
84  int suit = deal.Suit(i);
85  int rank = deal.Rank(i);
86  lengths[suit]++;
87  if (rank > 8) hcp += (rank - 8);
88  }
89  // Balanced means 4333, 4432 or 5332
90  bool is_balanced = (lengths[0] * lengths[1] * lengths[2] * lengths[3] >= 90);
91  return is_balanced && (20 <= hcp) && (hcp <= 21);
92 }

119  if (str == "Pass") return kPass;

122  auto denomination = std::string(kDenominationChar).find(str[1]);

124  return (level - 1) * kNumDenominations + denomination + 1;

127 std::string Deal::HandString(int begin, int end) const {
128  bool cards[kNumSuits][kNumCardsPerSuit] = {{false}};
129  for (int i = begin; i < end; ++i) {
130  cards[Suit(i)][Rank(i)] = true;
131  }
132  std::string hand;
133  for (int s = 3; s >= 0; --s) {
134  for (int r = 12; r >= 0; --r) {
135  if (cards[s][r]) {
136  hand.push_back(kRankChar[r]);
137  }
138  }
139  if (s) hand.push_back('.');
140  }
141  return hand;

193 void UncontestedBiddingState::InformationStateTensor(
194  Player player, absl::Span<float> values) const {
195  SPIEL_CHECK_GE(player, 0);
196  SPIEL_CHECK_LT(player, num_players_);
197 
198  SPIEL_CHECK_EQ(values.size(), kStateSize);
199  std::fill(values.begin(), values.end(), 0.);
200  auto ptr = values.begin();
201 
202  for (int i = kNumCardsPerHand * player; i < kNumCardsPerHand * (1 + player);
203  ++i) {
204  ptr[deal_.Card(i)] = 1.;
205  }
206  ptr += kNumCards;
207 
208  // What actions have been taken, and by whom
209  for (int i = 0; i < actions_.size(); ++i) {
210  ptr[actions_[i] * kNumPlayers + (i % kNumPlayers)] = 1;
211  }
212  ptr += kNumActions * kNumPlayers;
213 
214  // Which player we are
215  ptr[player] = 1;
216  ptr += kNumPlayers;
217 }

236 void UncontestedBiddingState::ScoreDeal() {
237  // If both Pass, the score is zero.
238  const bool passed_out = (actions_.size() == 2);
239  if (passed_out && reference_contracts_.empty()) {
240  score_ = 0;
241  return;
242  }
243 
244  // Determine the final contract and declarer
245  const Action bid = actions_[actions_.size() - 2];
246  Contract contract{passed_out ? 0 : Level(bid),
247  passed_out ? kNoTrump : Denomination(bid), kUndoubled};
248  for (int i = 0; i < actions_.size(); ++i) {
249  if (actions_[i] > 0 && Denomination(actions_[i]) == contract.trumps) {
250  contract.declarer = i % 2;
251  break;
252  }
253  }
254 
255  // Populate East-West cards
256  ddTableDeal dd_table_deal{};
257  for (Player player = 0; player < kNumPlayers; ++player) {
258  for (int i = kNumCardsPerHand * player; i < kNumCardsPerHand * (1 + player);
259  ++i) {
260  dd_table_deal.cards[player * 2][deal_.Suit(i)] += 1
261  << (2 + deal_.Rank(i));
262  }
263  }
264 
265  // Initialize scores to zero
266  score_ = 0;
267  reference_scores_.resize(reference_contracts_.size());
268  std::fill(reference_scores_.begin(), reference_scores_.end(), 0);
269 
270  // For each redeal
271  for (int ideal = 0; ideal < num_redeals_; ++ideal) {
272  if (ideal > 0) deal_.Shuffle(&rng_, kNumCardsPerHand * 2, kNumCards);
273 
274  // Populate (reshuffled) North-South cards
275  for (int opponent = 0; opponent < kNumPlayers; ++opponent) {
276  std::fill(dd_table_deal.cards[1 + opponent * 2],
277  dd_table_deal.cards[1 + opponent * 2] + 4, 0);
278  for (int i = kNumCardsPerHand * (2 + opponent);
279  i < kNumCardsPerHand * (3 + opponent); ++i) {
280  dd_table_deal.cards[1 + opponent * 2][deal_.Suit(i)] +=
281  1 << (2 + deal_.Rank(i));
282  }
283  }
284 
285  // Analyze the deal.
286  DDS_EXTERNAL(SetMaxThreads)(0);
287  struct ddTableResults results;
288  const int return_code = DDS_EXTERNAL(CalcDDtable)(dd_table_deal, &results);
289 
290  // Check for errors.
291  if (return_code != RETURN_NO_FAULT) {
292  char error_message[80];
293  DDS_EXTERNAL(ErrorMessage)(return_code, error_message);
294  SpielFatalError(absl::StrCat("double_dummy_solver:", error_message));
295  }
296 
297  // Compute the score and update the total.
298  if (!passed_out) {
299  const int declarer_tricks =
300  results.resTable[contract.trumps][2 * contract.declarer];
301  const int declarer_score =
302  Score(contract, declarer_tricks, /*is_vulnerable=*/false);
303  score_ += static_cast<double>(declarer_score) / num_redeals_;
304  }
305 
306  // Compute the scores for reference contracts.
307  for (int i = 0; i < reference_contracts_.size(); ++i) {
308  const int declarer_tricks =
309  results.resTable[reference_contracts_[i].trumps]
310  [2 * reference_contracts_[i].declarer];
311  const int declarer_score = Score(reference_contracts_[i], declarer_tricks,
312  /*is_vulnerable=*/false);
313  reference_scores_[i] +=
314  static_cast<double>(declarer_score) / num_redeals_;
315  }
316  }
317 }

319 void UncontestedBiddingState::DoApplyAction(Action action_id) {
320  if (dealt_) {
321  actions_.push_back(action_id);
322  if (IsTerminal()) ScoreDeal();
323  } else {
324  do {
325  deal_.Shuffle(&rng_);
326  } while (!deal_filter_(deal_));
327  dealt_ = true;
328  }
329 }

396 // Deserialize the deal and auction
397 // e.g. "AKQJ.543.QJ8.T92 97532.A2.9.QJ853 2N-3C"
398 std::unique_ptr<State> UncontestedBiddingGame::DeserializeState(
399  const std::string& str) const {
400  if (str.empty()) {
401  return absl::make_unique<UncontestedBiddingState>(
402  shared_from_this(), reference_contracts_, deal_filter_, forced_actions_,
403  rng_seed_, num_redeals_);
404  }
405  SPIEL_CHECK_GE(str.length(),
406  kNumPlayers * (kNumCardsPerHand + kNumSuits) - 1);
407  std::array<int, kNumCards> cards{};
408  std::array<int, kNumCards> cards_dealt{};
409  for (Player player = 0; player < kNumPlayers; ++player) {
410  int suit = 0;
411  int start = player * (kNumCardsPerHand + kNumSuits);
412  for (int i = 0; i < kNumCardsPerHand; ++i) {
413  char ch = str[start + i + suit];
414  while (ch == '.') {
415  ++suit;
416  ch = str[start + i + suit];
417  }
418  const int rank = (std::strchr(kRankChar, ch) - kRankChar);
419  const int card = rank * 4 + (3 - suit);
420  SPIEL_CHECK_FALSE(cards_dealt[card]);
421  cards[player * kNumCardsPerHand + i] = card;
422  cards_dealt[card] = true;
423  }
424  }
425  int i = kNumPlayers * kNumCardsPerHand;
426  for (int c = 0; c < 52; ++c) {
427  if (!cards_dealt[c]) cards[i++] = c;
428  }
429 
430  // Get any actions there may be.
431  std::vector<Action> actions;
432  int start = kNumPlayers * (kNumCardsPerHand + kNumSuits);
433  while (start < str.length()) {
434  auto end = str.find('-', start);
435  if (end == std::string::npos) end = str.length();
436  actions.push_back(ActionFromString(str.substr(start, end - start)));
437  start = end + 1;
438  }
439 
440  // Check that early actions agree with the forced actions in this game.
441  SPIEL_CHECK_GE(actions.size(), forced_actions_.size());
442  for (int i = 0; i < forced_actions_.size(); ++i) {
443  SPIEL_CHECK_EQ(actions[i], forced_actions_[i]);
444  }
445 
446  return absl::make_unique<UncontestedBiddingState>(
447  shared_from_this(), reference_contracts_, Deal(cards), actions, rng_seed_,
448  num_redeals_);
449 }
