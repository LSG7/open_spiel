16"""Implementation of a mean field routing game.
17
18The game is derived from https://arxiv.org/abs/2110.11943.
19It is the extension of the dynamic routing game python_dynamic_routing.
20The list of vehicles decribing the N player of the dynamic routing game is
21replaced by a list of OriginDestinationDemand. One OriginDestinationDemand
22corresponds to one population of vehicles (with the same origin, destination and
23departure time).
24
25This game is a variant of the mean field route choice game as the vehicle
26movement depends on the current network congestion. In the mean field route
27choice game, the number of time step to reach the destination is constant and
28does not depend on the network congestion, neither of the vehicle cost function.
29In the dynamic driving and routing games the vehicle choose its speed to travel
30on each link in order to minimize its cost function. Therefore the congestion is
31encoded in the cost function.
32
33More context can be found on the docstring of the python_dynamic_routing class.
34"""

71@functools.lru_cache(maxsize=None)
72def _state_to_str(
73 is_chance_init: bool,
74 location: str,
75 time_step: int,
76 player_id: int,
77 waiting_time: int,
78 destination: str,
79 final_arrival_time: float,
80) -> str:
81 """Convert the state to a string representation.
82
83 As the string representation will be used in dictionaries for various
84 algorithms that computes the state value, expected return, best response or
85 find the mean field Nash equilibrium.
86 The state is uniquely define by the current time, the type of node
87 (decision, mean field or chance), the vehicle location, its destination and
88 its waiting time.
89 Args:
90 is_chance_init: True if at chance initialization.
91 location: the location of the representative player.
92 time_step: the current time step.
93 player_id: the current node type as a player id.
94 waiting_time: the representative player waiting time.
95 destination: the destination of the representative player.
96 final_arrival_time: time of arrival.
97
98 Returns:
99 state_string: string representing uniquely the mean field game.
100 """
101 if is_chance_init:
102 return "initial chance node"
103 if player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
104 time = str(time_step)
105 elif player_id == pyspiel.PlayerId.MEAN_FIELD:
106 time = f"{time_step}_mean_field"
107 elif player_id == pyspiel.PlayerId.CHANCE:
108 time = f"{time_step}_chance"
109 else:
110 raise ValueError(
111 "Player id should be DEFAULT_PLAYER_ID, MEAN_FIELD or CHANCE")
112 if final_arrival_time:
113 return (f"Arrived at {location}, with arrival time "
114 f"{final_arrival_time}, t={time}")
115 return (f"Location={location}, waiting_time={waiting_time},"
116 f" t={time}, destination='{destination}'")

119class MeanFieldRoutingGame(pyspiel.Game):
120 """Implementation of mean field routing game.
121
122 The representative vehicle/player is represented as a tuple current location,
123 current waiting time and destination. When the waiting time is negative, the
124 vehicle choose on with successor link it would like to go. When arriving on
125 the link, a waiting time is assigned to the player based on the distribution
126 of players on the link. The vehicle arrival time is equal to the time step
127 when they first reach their destination. See module docstring for more
128 information.
129
130 Attributes inherited from GameInfo:
131 max_chance_outcomes: maximum number of chance actions. Set to the length of
132 od_demand, i.e. the number of `OriginDestinationDemand`s.
133 max_game_length: maximum number of time step played. Passed during
134 construction.
135 max_utility: maximum utility is the opposite of the minimum arrival
136 time. Set to 0.
137 min_utility: minimum utility is the opposite of the maximum arrival
138 time. Set to - max_game_length - 1.
139 num_distinct_actions: maximum number of possible actions. This is
140 equal to the number of links + 1 (corresponding to having no
141 possible action _NO_POSSIBLE_ACTION).
142 num_players: the number of vehicles. Should be 1 as this mean field
143 game is a one population game.
144 Attributes:
145 network: the network of the game.
146 od_demand: a list of the vehicle. Their origin and their destination should
147 be road sections of the game.
148 time_step_length: size of the time step, used to convert travel times into
149 number of game time steps.
150 perform_sanity_checks: if true, sanity checks are done during the game,
151 should be set to false to speed up the game.
152 total_num_vehicle: total number of vehicles as the sum of the od_demand.
153 chance_outcomes: chance outcomes based on the initial probability
154 distribution and their probabilities.
155 """
156 network: dynamic_routing_utils.Network
157 od_demand: List[dynamic_routing_utils.OriginDestinationDemand]
158 perform_sanity_checks: bool
159 time_step_length: float
160
161 def __init__(self,
162 params: Mapping[str, Any],
163 network: Optional[dynamic_routing_utils.Network] = None,
164 od_demand: Optional[List[
165 dynamic_routing_utils.OriginDestinationDemand]] = None,
166 perform_sanity_checks: bool = True):
167 """Initiliaze the game.
168
169 Args:
170 params: game parameters. It should define max_num_time_step and
171 time_step_length.
172 network: the network of the game.
173 od_demand: a list of the vehicle. Their origin and their destination
174 should be road sections of the game.
175 perform_sanity_checks: set the perform_sanity_checks attribute.
176 """
177 max_num_time_step = params["max_num_time_step"]
178 time_step_length = params["time_step_length"]
179 self.network = network if network else dynamic_routing_data.BRAESS_NETWORK
180 self.od_demandod_demand = (
181 od_demand
182 if od_demand else dynamic_routing_data.BRAESS_NETWORK_OD_DEMAND)
183 self.network.check_list_of_od_demand_is_correct(self.od_demandod_demand)
184 self.perform_sanity_checks = perform_sanity_checks
185 self.time_step_length = time_step_length
186 self.total_num_vehicle = sum(
187 [od_demand_item.counts for od_demand_item in self.od_demandod_demand])
188 self.chance_outcomes = [(i, od_demand_item.counts / self.total_num_vehicle)
189 for i, od_demand_item in enumerate(self.od_demandod_demand)]
190 game_info = pyspiel.GameInfo(
191 num_distinct_actions=self.network.num_actions(),
192 max_chance_outcomes=len(self.od_demandod_demand),
193 num_players=1,
194 min_utility=-max_num_time_step - 1,
195 max_utility=0,
196 max_game_length=max_num_time_step)
197 super().__init__(_GAME_TYPE, game_info, params if params else {})

223class MeanFieldRoutingGameState(pyspiel.State):
224 """State of the DynamicRoutingGame.
225
226 One player is equal to one vehicle.
227 See docstring of the game class and of the file for more information.
228 Attributes:
229 _current_time_step: current time step of the game.
230 _is_chance_init: boolean that encodes weither the current node is the
231 initial chance node.
232 _is_terminal: boolean that encodes weither the game is over.
233 _max_arrival_time: int that encodes maximum arrival time on any link in
234 number of time steps. Needed to enumerate all the possible state of a
235 vehicle being on a link to compute volume of cars on the link.
236 _max_waiting_time: maximum time a vehicle can wait on a time. This is done
237 in order to limit the number of possible state with a vehicle on a
238 specific link.
239 _normed_density_on_vehicle_link: density of vehicles on the link that is
240 used by the representative vehicle. This is given by the mean field
241 distribution.
242 _time_step_length: size of the time step, used to convert travel times into
243 number of game time steps.
244 _vehicle_at_destination: boolean that encodes if the representative vehicle
245 has reached its destination.
246 _vehicle_destination: the destination of the representative vehicle
247 corresponding to this state. It is associated to the representative
248 vehicle after the initial chance node according to the od_demand
249 distribution.
250 _vehicle_final_arrival_time: the arrival time of the representative vehicle,
251 the arrival is either 0 if the vehicle is still in the network or its
252 arrival time if the vehicle has reached its destination.
253 _vehicle_location: current location of the vehicle as a network road
254 section.
255 _vehicle_without_legal_action: boolean that encodes if the representative
256 vehicle has reach a sink node, meaning that it will not be able to move
257 anymore.
258 _waiting_time: time that the vehicle has to wait before moving to the next
259 link (equal to the link travel time when the vehicle just reached the
260 link).
261 """
262 _current_time_step: int
263 _is_chance_init: bool
264 _is_terminal: bool
265 _max_arrival_time: int
266 _max_waiting_time: int
267 _normed_density_on_vehicle_link: float
268 _time_step_length: float
269 _vehicle_at_destination: bool
270 _vehicle_destination: Optional[str]
271 _vehicle_final_arrival_time: float
272 _vehicle_location: Optional[str]
273 _vehicle_without_legal_action: bool
274 _waiting_time: int

325 def distribution_support(self) -> List[str]:
326 """Returns the state that should be used for update_distribution.
327
328 The distribution of the vehicle is used to determined the number of
329 cars on the same link of the representative vehicle in order to define
330 the waiting time of the representative vehicle when joining a link.
331 Therefore, only the states corresponding to be on the link of the
332 representative vehicle at this current time are useful.
333 Returns:
334 list of the two state: being on the link of the representative vehicle at
335 the current time and being stuck in traffic or not.
336 """
337 if self._vehicle_without_legal_action:
338 return []
339 od_demand = self.get_game().od_demand
340 dist = [
341 self.state_to_str( # pylint:disable=g-complex-comprehension
342 self._vehicle_location_vehicle_location,
343 self._current_time_step,
344 player_id=pyspiel.PlayerId.MEAN_FIELD,
345 waiting_time=waiting_time,
346 destination=destination)
347 for waiting_time in range(WAITING_TIME_NOT_ASSIGNED,
348 self._max_arrival_time)
349 for destination in {od._destination for od in od_demand} # pylint:disable=protected-access
350 ]
351 assert len(set(dist)) == len(dist), (
352 f"Distribution should not have duplicated states: {dist}.")
353 return dist

355 def update_distribution(self, distribution: List[float]):
356 """Get the number of cars on the same link as the representative player.
357
358 _normed_density_on_vehicle_link stores the number of cars on the link
359 where the representative player is.
360 Args:
361 distribution: the probability for a vehicle to be in the states in
362 distribution_support. The distribution is a list of probabilities.
363 """
364 game = self.get_game()
365 if game.perform_sanity_checks:
366 if self._player_id != pyspiel.PlayerId.MEAN_FIELD:
367 raise ValueError(("update_distribution should only be called at"
368 " a MEAN_FIELD state."))
369 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
370 if not self._vehicle_without_legal_action:
371 self._normed_density_on_vehicle_link = sum(distribution)
372 if game.perform_sanity_checks:
373 assert 0 <= self._normed_density_on_vehicle_link <= 1 + 1e-4, (
374 f"{self._normed_density_on_vehicle_link} is not in [0, 1].")
375 if self._waiting_time == WAITING_TIME_NOT_ASSIGNED:
376 volume = (game.total_num_vehicle * self._normed_density_on_vehicle_link)
377 self._waiting_time = int(
378 game.network.get_travel_time(self._vehicle_location_vehicle_location, volume) /
379 self._time_step_length) - 1
380 self._waiting_time = max(0, self._waiting_time)

397 def _legal_actions(self, player: pyspiel.PlayerId) -> List[int]:
398 """Return the legal actions of the vehicle.
399
400 Legal actions are the succesor road section of the vehicle current road
401 section.
402 Args:
403 player: the vehicle id.
404
405 Returns:
406 list_legal_actions: a list of legal actions. If the game is finished then
407 the list is empty. If the vehicle is at its destination, has a positive
408 waiting time or if it is on a node without successors then an empty list
409 is returned. Otherwise the list of successors nodes of the current
410 vehicle location is returned.
411 """
412 if self._is_terminal:
413 return []
414 if self.get_game().perform_sanity_checks:
415 assert player == pyspiel.PlayerId.DEFAULT_PLAYER_ID, str(player)
416 if self._vehicle_without_legal_action:
417 # If the vehicle is at destination it cannot do anything.
418 return [dynamic_routing_utils.NO_POSSIBLE_ACTION]
419 if self._waiting_time > 0:
420 return [dynamic_routing_utils.NO_POSSIBLE_ACTION]
421 _, end_section_node = dynamic_routing_utils._nodes_from_road_section( # pylint:disable=protected-access
422 self._vehicle_location_vehicle_location)
423 successors = self.get_game().network.get_successors(end_section_node)
424 if self.get_game().perform_sanity_checks:
425 if not successors:
426 raise ValueError(("If a vehicle is not without legal action, it"
427 " should have an action."))
428 assert isinstance(successors, Iterable)
429 actions = [
430 self.get_game().network.get_action_id_from_movement(
431 end_section_node, d) for d in successors
432 ]
433 map(self.get_game().network.assert_valid_action, actions)
434 return sorted(actions)

436 def _apply_action(self, action: int):
437 """Apply the action to the state.
438
439 This function can be either called on a chance node or on a decision
440 node. If called on the initial chance node, the action gives in which OD
441 demand the representative vehicle belongs too (it put the vehicle at
442 this location and define its destination).
443 If called on decision node, the action defines on which link the vehicle
444 will move (if it is not stuck in traffic) and assign a waiting time to the
445 vehicle.
446 Args:
447 action: the action to apply.
448 """
449 if self._player_id == pyspiel.PlayerId.CHANCE:
450 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
451 assert self._is_chance_init
452 # Apply action is called on initial chance node to initialized
453 # the vehicle position based on the initial location
454 # distribution.
455 od_demand = self.get_game().od_demand
456 self._vehicle_destination_vehicle_destination = od_demand[action].destination
457 self._vehicle_location_vehicle_location = od_demand[action].origin
458 self._waiting_time = int(od_demand[action].departure_time /
459 self._time_step_length)
460 self._is_chance_init = False
461 self._normed_density_on_vehicle_link = 0
462 elif self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
463 self._player_id = pyspiel.PlayerId.MEAN_FIELD
464 # Apply action is called on a descision node. If the vehicle can
465 # move, then it will move to the next road section.
466 # Has the vehicle already reached a sink node?
467 if not self._vehicle_without_legal_action:
468 # If the vehicle is stuck in traffic it cannot move.
469 if self._waiting_time > 0:
470 self._waiting_time -= 1
471 else:
472 if self.get_game().perform_sanity_checks:
473 self.get_game().network.assert_valid_action(action,
474 self._vehicle_location_vehicle_location)
475 self._vehicle_location_vehicle_location = (
476 self.get_game().network.get_road_section_from_action_id(action))
477 # Has the vehicle just reached its destination?
478 if self._vehicle_location_vehicle_location == self._vehicle_destination_vehicle_destination:
479 self._vehicle_final_arrival_time = self._current_time_step
480 self._vehicle_at_destination = True
481 self._vehicle_without_legal_action = True
482 # Will the vehicle have a legal action for next time step?
483 elif self.get_game().network.is_location_at_sink_node(
484 self._vehicle_location_vehicle_location):
485 self._vehicle_without_legal_action = True
486 self._vehicle_final_arrival_time = -self.get_game().min_utility()
487 else:
488 self._waiting_time = WAITING_TIME_NOT_ASSIGNED
489 self._current_time_step += 1
490 elif self.get_game().perform_sanity_checks:
491 if self._is_terminal:
492 raise ValueError(
493 "_apply_action should not be called at a end of the game.")
494 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
495 raise ValueError(
496 "_apply_action should not be called at a MEAN_FIELD state.")
497 # Is the game finished?
498 if self._current_time_step >= self.get_game().max_game_length():
499 self._is_terminal = True
500 if not self._vehicle_at_destination:
501 self._vehicle_final_arrival_time = -self.get_game().min_utility()

561class NetworkObserver:
562 """Network observer used by the learning algorithm.
563
564 The state string is the state history string. The state tensor is an array
565 of size number of locations * 2 + maximum number of time steps + 2, which is
566 the concatenation of one-hot encodings of the location, destination (1-based;
567 if location or destination is None, then the 0th element will be set to 1) and
568 the current time (0-based). The last element of the array will be set to 1 if
569 waiting time is positive, or 0 otherwise.
570
571 Attributes:
572 dict: Dictionary of tensors for the components of the observation
573 corresponding to the location, destination and time.
574 tensor: The concatenated form of the observation.
575 """
576
577 def __init__(self, num_locations: int, max_num_time_step: int):
578 """Initializes an empty observation tensor."""
579 self.tensor = np.zeros(num_locations * 2 + max_num_time_step + 1 + 1,
580 np.float32)
581 self.dict = {
582 "location": self.tensor[:num_locations],
583 "destination": self.tensor[num_locations:num_locations * 2],
584 "time": self.tensor[num_locations * 2:-1],
585 "waiting": self.tensor[-1:]
586 }
587
588 def set_from(self, state, player):
589 """Sets the state tensor based on the specified state.
590
591 Note that the function may be called with arbitrary states of the game, e.g.
592 from different runs, and therefore the tensor should be cleared and updated
593 instead of preserving any earlier values.
594
595 Args:
596 state: state of the game.
597 player: player id that should play.
598 """
