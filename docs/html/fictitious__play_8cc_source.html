35 ActionsAndProbs UniformSequencePolicy(int num_actions) {
36  ActionsAndProbs actions_and_probs;
37  actions_and_probs.reserve(num_actions);
38  for (Action a = 0; a < num_actions; ++a) {
39  actions_and_probs.push_back({a, 1.0 / num_actions});
40  }
41  return actions_and_probs;
42 }

44 ActionsAndProbs RandomStatePolicy(int num_actions,
45  const std::function<double()>& rng) {
46  ActionsAndProbs actions_and_probs;
47  actions_and_probs.reserve(num_actions);
48  double total_weight = 0.0;
49  for (Action a = 0; a < num_actions; ++a) {
50  double weight = rng();
51  total_weight += weight;
52  actions_and_probs.push_back({a, weight});
53  }
54  for (Action a = 0; a < num_actions; ++a) {
55  actions_and_probs[a].second /= total_weight;
56  }
57  return actions_and_probs;
58 }

60 FictitiousPlayProcess::FictitiousPlayProcess(std::shared_ptr<const Game> game,
61  int seed,
62  bool randomize_initial_policies)
63  : rng_(seed),
64  dist_(0.0, 1.0),
65  game_(game),
66  num_players_(game->NumPlayers()),
67  num_actions_(game->NumDistinctActions()),
68  iterations_(0),
69  total_time_(absl::ZeroDuration()) {
70  // Get the info states strings.
71  infostate_strings_.reserve(num_players_);
72  std::unique_ptr<State> state = game->NewInitialState();
73  for (Player p = 0; p < num_players_; ++p) {
74  SPIEL_CHECK_EQ(state->CurrentPlayer(), p);
75  std::vector<Action> legal_actions = state->LegalActions();
76  std::string infostate_str = state->InformationStateString();
77  infostate_strings_.push_back(infostate_str);
78  state->ApplyAction(legal_actions[0]);
79  }
80  SPIEL_CHECK_TRUE(state->IsTerminal());
81 
82  // Get number of joint actions
83  num_joint_actions_ = 1;
84  for (Player p = 0; p < num_players_; ++p) {
85  num_joint_actions_ *= num_actions_;
86  }
87 
88  // Set initial cumulative policies to uniform.
89  if (randomize_initial_policies) {
90  InitPoliciesRandom();
91  } else {
92  InitPolicies();
93  }
94 
95  GetMarginalJointPolicy(&joint_policy_);
96  CheckJointUtilitiesCache();
97 
98  // Make the best response computers (for full FP)
99  for (int p = 0; p < num_players_; ++p) {
100  best_response_computers_.push_back(std::unique_ptr<TabularBestResponse>(
101  new TabularBestResponse(*game_, p, &joint_policy_)));
102  }
103 }

105 void FictitiousPlayProcess::GetMarginalJointPolicy(
106  TabularPolicy* policy,
107  const std::vector<std::vector<double>>* weight_table) const {
108  if (weight_table == nullptr) {
109  weight_table = &cumulative_policies_;
110  }
111 
112  for (Player p = 0; p < num_players_; ++p) {
113  double prob_sum = 0.0;
114  double total_weight = std::accumulate((*weight_table)[p].begin(),
115  (*weight_table)[p].end(), 0.0);
116  for (Action a = 0; a < num_actions_; ++a) {
117  double prob = (*weight_table)[p][a] / total_weight;
118  SPIEL_CHECK_PROB(prob);
119  policy->SetProb(infostate_strings_[p], a, prob);
120  prob_sum += prob;
121  }
122  SPIEL_CHECK_TRUE(Near(prob_sum, 1.0));
123  }
124 }

126 // Get the marginalized joint policy by marginalizing the empirical joint
127 // policy.
128 void FictitiousPlayProcess::GetMarginalJointPolicyFromEmpirical(
129  TabularPolicy* policy) const {
130  std::vector<std::vector<double>> marginal_weights;
131  marginal_weights.reserve(num_players_);
132  for (Player p = 0; p < num_players_; ++p) {
133  marginal_weights.push_back(std::vector<double>(num_actions_, 0));
134  }
135 
136  NormalFormCorrelationDevice mu = GetEmpiricalJointPolicy();
137 
138  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
139  for (Player p = 0; p < num_players_; ++p) {
140  marginal_weights[p][mu[ja_idx].actions[p]] += mu[ja_idx].probability;
141  }
142  }
143 
144  for (Player p = 0; p < num_players_; ++p) {
145  double total_weight = std::accumulate(marginal_weights[p].begin(),
146  marginal_weights[p].end(), 0.0);
147  for (Action a = 0; a < num_actions_; ++a) {
148  marginal_weights[p][a] /= total_weight;
149  }
150  }
151 
152  GetMarginalJointPolicy(policy, &marginal_weights);
153 }

155 TabularPolicy FictitiousPlayProcess::GetLatestPolicy() const {
156  TabularPolicy policy;
157  for (Player p = 0; p < num_players_; ++p) {

169 NormalFormCorrelationDevice FictitiousPlayProcess::GetEmpiricalJointPolicy()
170  const {
171  double prob_sum = 0.0;
172  NormalFormCorrelationDevice corr_dev;
173  corr_dev.reserve(num_joint_actions_);
174  double total_weight = std::accumulate(cumulative_joint_policy_.begin(),
175  cumulative_joint_policy_.end(), 0.0);
176  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
177  double prob = cumulative_joint_policy_[ja_idx] / total_weight;
178  SPIEL_CHECK_PROB(prob);
179  corr_dev.push_back({prob, IndexToJointAction(ja_idx)});
180  prob_sum += prob;
181  }
182  SPIEL_CHECK_TRUE(Near(prob_sum, 1.0));
183  SPIEL_CHECK_EQ(corr_dev.size(), num_joint_actions_);
184  return corr_dev;
185 }

187 void FictitiousPlayProcess::InitPoliciesRandom() {
188  // Cumulative policies per player.
189  cumulative_policies_.reserve(num_players_);
190  for (Player p = 0; p < num_players_; ++p) {
191  std::vector<double> policy(num_actions_, 0);
192  double total_weight = 0.0;
193  for (int a = 0; a < num_actions_; ++a) {
194  policy[a] = dist_(rng_);
195  total_weight += policy[a];
196  }
197  for (int a = 0; a < num_actions_; ++a) {
198  policy[a] /= total_weight;
199  }
200  cumulative_policies_.push_back(policy);
201  }
202 
203  // Initial, current, and past policies
204  current_joint_policy_counts_.resize(num_joint_actions_);
205  std::vector<ActionsAndProbs> initial_policies;
206  initial_policies.reserve(num_players_);
207  for (Player p = 0; p < num_players_; ++p) {
208  initial_policies.push_back(
209  RandomStatePolicy(num_actions_, [this]() { return dist_(rng_); }));
210  }
211  past_policies_.push_back(initial_policies);
212 
213  // Cumulative joint policy.
214  cumulative_joint_policy_ = std::vector<double>(num_joint_actions_, 0);
215  double total_weight = 0.0;
216  for (int idx = 0; idx < num_joint_actions_; ++idx) {
217  cumulative_joint_policy_[idx] = dist_(rng_);
218  total_weight += cumulative_joint_policy_[idx];
219  }
220  for (int idx = 0; idx < num_joint_actions_; ++idx) {
221  cumulative_joint_policy_[idx] /= total_weight;
222  }
223 }

227  cumulative_policies_.reserve(num_players_);

229  std::vector<double> uniform_policy(num_actions_, 1.0 / num_actions_);
230  cumulative_policies_.push_back(uniform_policy);

236  initial_policies.reserve(num_players_);
237  for (Player p = 0; p < num_players_; ++p) {
238  initial_policies.push_back(UniformSequencePolicy(num_actions_));

242  // Cumulative joint policy.

244  std::vector<double>(num_joint_actions_, 1.0 / num_joint_actions_);

247 // Add appropriate weights given each players' (potentially mixed) policy
248 void FictitiousPlayProcess::UpdateCumulativeJointPolicy(
249  const std::vector<std::vector<double>>& policies) {
250  double sum_weights = 0.0;
251  double sum_joint_policy = 0.0;
252  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
253  std::vector<Action> joint_action = IndexToJointAction(ja_idx);
254  double weight = 1.0;
255  for (Player p = 0; p < num_players_; ++p) {
256  Action action = joint_action[p];
257  weight *= policies[p][action];
258  }
259  cumulative_joint_policy_[ja_idx] += weight;
260  sum_weights += weight;
261  sum_joint_policy += cumulative_joint_policy_[ja_idx];
262  }
263  SPIEL_CHECK_FLOAT_NEAR(sum_weights, 1.0, 1e-12);
264  SPIEL_CHECK_FLOAT_NEAR(sum_joint_policy, iterations_ + 2.0, 1e-12);
265 }

267 void FictitiousPlayProcess::UpdateCumulativeJointPolicySampled(
268  const std::vector<std::vector<double>>& policies, int num_samples) {
269  double weight = 1.0 / num_samples;
270  std::vector<absl::discrete_distribution<int>> dists;
271  dists.reserve(policies.size());
272  for (Player p = 0; p < policies.size(); ++p) {
273  dists.push_back(absl::discrete_distribution<int>(policies[p].begin(),
274  policies[p].end()));
275  }
276  for (int sample = 0; sample < num_samples; ++sample) {
277  std::vector<Action> joint_action;
278  joint_action.reserve(num_players_);
279  for (Player p = 0; p < policies.size(); ++p) {
280  joint_action.push_back(dists[p](rng_));
281  }
282  int ja_idx = JointActionToIndex(joint_action);
283  cumulative_joint_policy_[ja_idx] += weight;
284  }

287 std::vector<double> FictitiousPlayProcess::Softmax(
288  const std::vector<double>& values, double lambda) const {
289  std::vector<double> new_values = values;
290  for (double& new_value : new_values) {
291  new_value *= lambda;
292  }
293  double max = *std::max_element(new_values.begin(), new_values.end());
294 
295  double denom = 0;
296  for (int idx = 0; idx < values.size(); ++idx) {
297  new_values[idx] = std::exp(new_values[idx] - max);
298  denom += new_values[idx];
299  }
300 
301  SPIEL_CHECK_GT(denom, 0);
302  double prob_sum = 0.0;
303  std::vector<double> policy;
304  policy.reserve(new_values.size());
305  for (int idx = 0; idx < values.size(); ++idx) {
306  double prob = new_values[idx] / denom;
307  SPIEL_CHECK_PROB(prob);
308  prob_sum += prob;
309  policy.push_back(prob);
310  }
311 
312  SPIEL_CHECK_FLOAT_NEAR(prob_sum, 1.0, 1e-12);
313  return policy;
314 }

368 void FictitiousPlayProcess::CheckJointUtilitiesCache() {
369  if (cached_joint_utilities_.empty()) {
370  cached_joint_utilities_.reserve(num_players_);
371  for (Player p = 0; p < num_players_; ++p) {
372  cached_joint_utilities_.push_back(
373  std::vector<double>(num_joint_actions_, 0));
374  }
375 
376  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
377  std::vector<Action> joint_action = IndexToJointAction(ja_idx);
378  std::unique_ptr<State> state = game_->NewInitialState();
379  for (Action action : joint_action) {
380  state->ApplyAction(action);
381  }
382  SPIEL_CHECK_TRUE(state->IsTerminal());
383  std::vector<double> returns = state->Returns();
384  for (Player p = 0; p < num_players_; ++p) {
385  cached_joint_utilities_[p][ja_idx] = returns[p];
386  }
387  }
388  }

391 double FictitiousPlayProcess::CCEDist() const {
392  double dist = 0;
393  std::vector<double> max_deviation(num_players_, -10);
394  std::vector<double> exp_values(num_players_, 0);
395  NormalFormCorrelationDevice corr_dev = GetEmpiricalJointPolicy();
396 
397  // First compute expected values for everyone
398  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
399  if (corr_dev[ja_idx].probability > 0) {
400  for (Player p = 0; p < num_players_; ++p) {
401  exp_values[p] +=
402  corr_dev[ja_idx].probability * cached_joint_utilities_[p][ja_idx];
403  }
404  }
405  }
406 
407  // Now for each player, find the maximal deviation
408  for (Player p = 0; p < num_players_; ++p) {
409  for (Action a = 0; a < num_actions_; ++a) {
410  double action_value = 0;
411  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
412  if (corr_dev[ja_idx].probability > 0) {
413  // Player p is changing to choose a instead.
414  std::vector<Action> joint_action = IndexToJointAction(ja_idx);
415  joint_action[p] = a;
416  int other_index = JointActionToIndex(joint_action);
417 
418  action_value += corr_dev[ja_idx].probability *
419  cached_joint_utilities_[p][other_index];
420  }
421  }
422  if (action_value > max_deviation[p]) {
423  max_deviation[p] = action_value;
424  }
425  }
426  }
427 
428  for (Player p = 0; p < num_players_; ++p) {
429  double delta = std::max(max_deviation[p] - exp_values[p], 0.0);
430  SPIEL_CHECK_GE(delta, 0);
431  dist += delta;
432  }
433 
434  return dist;
435 }

446 int FictitiousPlayProcess::JointActionToIndex(
447  const std::vector<Action>& joint_action) const {
448  // Convert to a number from base num_actions_

562 void FictitiousPlayProcess::SFPIteration(double lambda) {
563  absl::Time start = absl::Now();
564 
565  std::vector<std::vector<double>> softmax_brs;
566  softmax_brs.reserve(num_players_);

570  // BestResponseAgainstEmpiricalMarginalizedPolicies(p, &values);

581  }
582 
583  // Update empirical cumulative dist with these mixed policies
584  UpdateCumulativeJointPolicy(softmax_brs);
585 
586  iterations_++;
587  total_time_ += absl::Now() - start;
588 }

596  std::vector<Action> joint_action(num_players_, kInvalidAction);

598  // Sample the base profiles: number by player
599  std::vector<std::vector<Action>> base_samples;
600  base_samples.reserve(num_base_samples);
601  for (int i = 0; i < num_base_samples; ++i) {
602  std::vector<Action> base_profile;
603  base_profile.reserve(num_players_);
604  int past_idx = static_cast<int>(dist_(rng_) * past_policies_.size());
605  for (Player p = 0; p < num_players_; ++p) {
606  base_profile.push_back(
607  SampleAction(past_policies_[past_idx][p], dist_(rng_)).first);
608  }
609  base_samples.push_back(base_profile);
610  }
611 
612  // Each player computes a sampled BR.
613  for (Player p = 0; p < num_players_; ++p) {
614  double max_return = -std::numeric_limits<double>::infinity();
615  Action best_candidate = kInvalidAction;
616 
617  for (int i = 0; i < num_candidates; ++i) {
618  Action sampled_candidate =
619  absl::Uniform(rng_, 0u, static_cast<unsigned int>(num_actions_));
620  // Compute the action's expectation.
621  double return_sum = 0.0;
622  for (const std::vector<Action>& base_joint_action : base_samples) {
623  std::vector<Action> joint_action = base_joint_action;
624  joint_action[p] = sampled_candidate;
625  std::unique_ptr<State> state = game_->NewInitialState();
626  // Turn-based simultaneous game, so must apply them in order.
627  for (Player pp = 0; pp < num_players_; ++pp) {
628  state->ApplyAction(joint_action[pp]);
629  }
630  return_sum += state->PlayerReturn(p);
631  }
632  if (return_sum / num_base_samples > max_return) {
633  max_return = return_sum / num_base_samples;
634  best_candidate = sampled_candidate;
635  }
636  }
637 
638  SPIEL_CHECK_TRUE(best_candidate != kInvalidAction);
639  cumulative_policies_[p][best_candidate] += 1.0;
640  joint_action[p] = best_candidate;
641  }
642 
643  // Add to past policies
644  std::vector<ActionsAndProbs> new_policy;
645  new_policy.reserve(num_players_);
646  for (Player p = 0; p < num_players_; ++p) {
647  new_policy.push_back({{joint_action[p], 1.0}});
648  }
649  past_policies_.push_back(new_policy);
650 
651  cumulative_joint_policy_[JointActionToIndex(joint_action)] += 1.0;
652 
653  iterations_++;
654  total_time_ += absl::Now() - start;
655 }

687 Action FictitiousPlayProcess::SampleCandidate(
688  Player player, CandidatesSamplerType sampler_type) {
689  if (sampler_type == CandidatesSamplerType::kCandidatesInitial) {
690  return absl::Uniform(rng_, 0u, static_cast<unsigned int>(num_actions_));
691  } else if (sampler_type == CandidatesSamplerType::kCandidatesUniform ||
692  sampler_type == CandidatesSamplerType::kCandidatesLatest) {
693  int past_idx =
694  sampler_type == CandidatesSamplerType::kCandidatesUniform
695  ? absl::Uniform(rng_, 0u,
696  static_cast<unsigned int>(past_policies_.size()))
697  : past_policies_.size() - 1;
698  return SampleAction(past_policies_[past_idx][player], dist_(rng_)).first;
699  } else if (sampler_type == CandidatesSamplerType::kCandidatesInitialUniform) {
700  int bit = absl::Uniform(rng_, 0u, 2u);

703  } else {
704  return SampleCandidate(player, CandidatesSamplerType::kCandidatesUniform);
705  }
706  } else if (sampler_type == CandidatesSamplerType::kCandidatesInitialLatest) {
707  int bit = absl::Uniform(rng_, 0u, 2u);
708  if (bit == 0) {
709  return SampleCandidate(player, CandidatesSamplerType::kCandidatesInitial);
710  } else {
711  return SampleCandidate(player, CandidatesSamplerType::kCandidatesLatest);
712  }
713  } else {
714  SpielFatalError("Candidate sampling method unrecognized.");
715  }

728 Action FictitiousPlayProcess::GetBestCandidate(
729  Player player, const std::vector<std::vector<Action>>& base_samples,
730  int num_candidates, CandidatesSamplerType sampler_type) {
731  std::vector<Action> best_candidates;
732  double max_return = -std::numeric_limits<double>::infinity();
733 
734  for (int i = 0; i < num_candidates; ++i) {
735  Action sampled_candidate = SampleCandidate(player, sampler_type);
736  // Compute the action's expectation.
737  double return_sum = 0.0;
738  for (const std::vector<Action>& base_joint_action : base_samples) {
739  std::vector<Action> ja_prime = base_joint_action;
740  ja_prime[player] = sampled_candidate;
741  int ja_idx = JointActionToIndex(ja_prime);
742  return_sum += cached_joint_utilities_[player][ja_idx];
743  }
744 
745  // Consider values within [ -kTieTolerance, kTieTolerance ] as tied.
746  if (return_sum > max_return + kTieTolerance) {
747  max_return = return_sum;
748  best_candidates = {sampled_candidate};
749  } else if (return_sum > max_return - kTieTolerance) {
750  best_candidates.push_back(sampled_candidate);
751  }
752  }
753 
754  SPIEL_CHECK_GE(best_candidates.size(), 0);
755  if (best_candidates.size() == 1) {
756  return best_candidates[0];
757  } else {
758  int idx = absl::Uniform(rng_, 0u,
759  static_cast<unsigned int>(best_candidates.size()));
760  return best_candidates[idx];
761  }
762 }

764 // pi_i^t = \frac{1}{N} \sum_{n = 1}^N 1(a_i), where a_i ~ SBR(\pi_b, \pi_c).
765 void FictitiousPlayProcess::BRPIIteration(
766  BaseSamplerType base_sampling, CandidatesSamplerType candidates_sampling,
767  int num_base_samples, int num_candidates, int brpi_N) {
768  absl::Time start = absl::Now();
769 
770  // Clear policy counts
771  std::fill(current_joint_policy_counts_.begin(),
772  current_joint_policy_counts_.end(), 0.0);
773 
774  // N trials
775  for (int n = 0; n < brpi_N; ++n) {
776  std::vector<Action> joint_action(num_players_, kInvalidAction);
777 
778  // Sample the base profiles: number by player
779  std::vector<std::vector<Action>> base_samples =
780  SampleBaseProfiles(base_sampling, num_base_samples);
781 
782  // Each player computes a sampled BR.
783  for (Player p = 0; p < num_players_; ++p) {
784  Action best_candidate = GetBestCandidate(p, base_samples, num_candidates,
785  candidates_sampling);
786  joint_action[p] = best_candidate;
787  }
788  current_joint_policy_counts_[JointActionToIndex(joint_action)] += 1.0;
789 
790  // End of trial.
791  }
792 
793  // Apply the 1/N to the emprical estimate of the joint policy and add them to
794  // the past policies.
795  std::vector<ActionsAndProbs> policies(num_players_);
796  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
797  if (current_joint_policy_counts_[ja_idx] > 0) {
798  double weight = current_joint_policy_counts_[ja_idx] / brpi_N;
799  std::vector<Action> joint_action = IndexToJointAction(ja_idx);
800  for (Player p = 0; p < num_players_; ++p) {
801  AddWeight(&policies[p], joint_action[p], weight);
802  }
803  }
804  }
805  past_policies_.push_back(policies);
806 
807  std::fill(cumulative_joint_policy_.begin(), cumulative_joint_policy_.end(),
808  0.0);
809  for (int ja_idx = 0; ja_idx < num_joint_actions_; ++ja_idx) {
810  std::vector<Action> joint_action = IndexToJointAction(ja_idx);
811  double joint_prob = 1.0;
812  for (Player p = 0; p < num_players_ && joint_prob > 0; ++p) {
813  double prob = std::max(0.0, GetProb(policies[p], joint_action[p]));
814  if (prob == 0) {
815  joint_prob = 0;
816  } else {
817  joint_prob *= prob;
818  }
819  }
820  if (joint_prob > 0) {
821  cumulative_joint_policy_[ja_idx] = joint_prob;
822  }
823  }
824 
825  iterations_++;
826  total_time_ += absl::Now() - start;
827 }
