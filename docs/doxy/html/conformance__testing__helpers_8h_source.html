53// Return a readable name for type T.
54template <class T>
55absl::string_view NameOfImpl() {
56// TODO(calabrese) Investigate using debugging:internal_demangle as a fallback.
57#if ABSL_TYPES_INTERNAL_HAS_CXA_DEMANGLE
58 int status = 0;
59 char* demangled_name = nullptr;
60
61 demangled_name =
62 abi::__cxa_demangle(typeid(T).name(), nullptr, nullptr, &status);
63
64 if (status == 0 && demangled_name != nullptr) {
65 return demangled_name;
66 } else {
67 return typeid(T).name();
68 }
69#else
70 return typeid(T).name();
71#endif
72 // NOTE: We intentionally leak demangled_name so that it remains valid
73 // throughout the remainder of the program.
74}

76// Given a type, returns as nice of a type name as we can produce (demangled).
77//
78// Note: This currently strips cv-qualifiers and references, but that is okay
79// because we only use this internally with unqualified object types.
80template <class T>
81std::string NameOf() {
82 static const absl::string_view result = NameOfImpl<T>();
83 return std::string(result);
84}

109// A type that contains a function object that returns an instance of a type
110// that is undergoing conformance testing. This function is required to always
111// return the same value upon invocation.
112template <class Fun>
113struct GeneratorType;
114
115// A type that contains a tuple of GeneratorType<Fun> where each Fun has the
116// same return type. The result of each of the different generators should all
117// be equal values, though the underlying object representation may differ (such
118// as if one returns 0.0 and another return -0.0, or if one returns an empty
119// vector and another returns an empty vector with a different capacity.
120template <class... Funs>
121struct EquivalenceClassType;
122
124//
125// A metafunction to check if a type is a specialization of EquivalenceClassType
126template <class T>
127struct IsEquivalenceClass : std::false_type {};
128
129template <>
130struct IsEquivalenceClass<EquivalenceClassType<>> : std::true_type {
131 using self = IsEquivalenceClass;
132
133 // A metafunction to check if this EquivalenceClassType is a valid
134 // EquivalenceClassType for a type `ValueType` that is undergoing testing
135 template <class ValueType>
136 using for_type = std::true_type;
137};
138
139template <class Head, class... Tail>
140struct IsEquivalenceClass<EquivalenceClassType<Head, Tail...>>
141 : std::true_type {
142 using self = IsEquivalenceClass;
143
144 // The type undergoing conformance testing that this EquivalenceClass
145 // corresponds to
146 using result_type = typename IsNullaryCallable<Head>::result_type;
147
148 // A metafunction to check if this EquivalenceClassType is a valid
149 // EquivalenceClassType for a type `ValueType` that is undergoing testing
150 template <class ValueType>
151 using for_type = std::is_same<ValueType, result_type>;
152};

155
156// A type that contains an ordered series of EquivalenceClassTypes, where the
157// the function object of each underlying GeneratorType has the same return type
158//
159// These equivalence classes are required to be in a logical ascending order
160// that is consistent with comparison operators that are defined for the return
161// type of each GeneratorType, if any.
162template <class... EqClasses>
163struct OrderedEquivalenceClasses;

205//
206// A metafunction that yields true iff each of Funs is an EquivalenceClassType
207// specialization and they all contain GeneratorType specializations that have
208// the same return type
209template <class... EqClasses>
210struct AreEquivalenceClassesOfTheSameType {
211 static_assert(sizeof...(EqClasses) != sizeof...(EqClasses), "");
212};

214template <>
215struct AreEquivalenceClassesOfTheSameType<> : std::true_type {
216 using self = AreEquivalenceClassesOfTheSameType;
217
218 // Metafunction to check that a type is the same as all of the equivalence
219 // classes, if any.
220 // Note: In this specialization there are no equivalence classes, so the
221 // value type is always compatible.
222 template <class /*ValueType*/>
223 using for_type = std::true_type;
224};

226template <class... Funs>
227struct AreEquivalenceClassesOfTheSameType<EquivalenceClassType<Funs...>>
228 : std::true_type {
229 using self = AreEquivalenceClassesOfTheSameType;

231 // Metafunction to check that a type is the same as all of the equivalence
232 // classes, if any.
233 template <class ValueType>
234 using for_type = typename IsEquivalenceClass<

274// Execute a function for each passed-in parameter.
275template <class Fun, class... Cases>
276void ForEachParameter(const Fun& fun, const Cases&... cases) {
277 const std::initializer_list<bool> results = {
278 (static_cast<void>(fun(cases)), true)...};
279
280 (void)results;
281}
282
283// Execute a function on each passed-in parameter (using a bound function).
284template <class Fun>
285struct ForEachParameterFun {
286 template <class... T>
287 void operator()(const T&... cases) const {
288 (ForEachParameter)(fun, cases...);
289 }
290
291 Fun fun;
292};
293
294// Execute a function on each element of a tuple.
295template <class Fun, class Tup>
296void ForEachTupleElement(const Fun& fun, const Tup& tup) {
297 absl::apply(ForEachParameterFun<Fun>{fun}, tup);
298}

301//
302// Execute a function for each combination of two elements of a tuple, including
303// combinations of an element with itself.
304template <class Fun, class... T>
305struct ForEveryTwoImpl {
306 template <class Lhs>
307 struct WithBoundLhs {
308 template <class Rhs>
309 void operator()(const Rhs& rhs) const {
310 fun(lhs, rhs);
311 }
312
313 Fun fun;
314 Lhs lhs;
315 };
316
317 template <class Lhs>
318 void operator()(const Lhs& lhs) const {
319 (ForEachTupleElement)(WithBoundLhs<Lhs>{fun, lhs}, args);
320 }
321
322 Fun fun;
323 std::tuple<T...> args;
324};
325
326template <class Fun, class... T>
327void ForEveryTwo(const Fun& fun, std::tuple<T...> args) {
328 (ForEachTupleElement)(ForEveryTwoImpl<Fun, T...>{fun, args}, args);
329}

348// passed-in Callable when `Condition` is true, otherwise it ignores the
