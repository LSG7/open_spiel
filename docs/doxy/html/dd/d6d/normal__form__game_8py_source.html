26def coop_reward(last_action, distribution):
27 """A game incentivising cooperation."""
28 nu_a, nu_b, nu_c, *_ = distribution
29 if last_action == 0:
30 return 10 * nu_a - 200 / 9 * (nu_a - nu_c) * nu_c - 20 * nu_b
31 elif last_action == 1:
32 return 20 * (nu_a - nu_b) - 2380 * nu_c
33 elif last_action == 2:
34 return 2000 / 9 * (nu_a - nu_c) * nu_c
35 else:
36 raise ValueError("Unknown last action " + str(last_action))

39def biased_indirect_rps(last_action, distribution):
40 """Biased indirect Rock Paper Scissors."""
41 nu_a = 0.7 * distribution[0]
42 nu_b = 0.5 * distribution[1]
43 nu_c = 0.3 * distribution[2]
44 if last_action == 0:
45 return nu_b - nu_c
46 elif last_action == 1:
47 return nu_c - nu_a
48 elif last_action == 2:
49 return nu_a - nu_b
50 else:
51 raise ValueError("Unknown last action " + str(last_action))

54def dominated_reward_source(last_action, distribution):
55 nu_a, nu_b, nu_c, *_ = distribution
56 if last_action == 0:
57 return nu_a + nu_c
58 elif last_action == 1:
59 return nu_b
60 elif last_action == 2:
61 return nu_a + nu_c - 0.25
62 else:
63 raise ValueError("Unknown last action " + str(last_action))

87class MFGNormalFormGame(pyspiel.Game):
88 """A Mean Field Normal Form game.
89
90 A game starts by an initial chance node that select the initial state
91 of the MFG.
92 Then the game sequentially alternates between:
93 - An action selection node (Where the player Id >= 0)
94 - A chance node (the player id is pyspiel.PlayerId.CHANCE)
95 - A Mean Field node (the player id is pyspiel.PlayerId.MEAN_FIELD)
96 """
97
98 # pylint:disable=dangerous-default-value
99 def __init__(self, params: Mapping[str, Any] = _DEFAULT_PARAMS):
100 game_info = pyspiel.GameInfo(
101 num_distinct_actions=_NUM_ACTIONS,
102 max_chance_outcomes=_NUM_ACTIONS,
103 num_players=_NUM_PLAYERS,
104 min_utility=-np.inf,
105 max_utility=+np.inf,
106 utility_sum=0.0,
107 max_game_length=2,
108 )
109 super().__init__(_GAME_TYPE, game_info, params)
110 if params["reward_function"] == "coop":
111 self.reward_function = coop_reward
112 elif params["reward_function"] == "dom":
113 self.reward_function = dominated_reward_source
114 elif params["reward_function"] == "biased_indirect_rps":
115 self.reward_function = biased_indirect_rps
116 else:
117 raise ValueError("Unknown reward function " + params["reward_function"])
118 self.num_actions = params["num_actions"]
119 self.size = 1 + self.num_actions
120
121 def new_initial_state(self):
122 """Returns a state corresponding to the start of a game."""
123 return MFGNormalFormState(self)
124
125 def make_py_observer(self, iig_obs_type=None, params=None):
126 """Returns an object used for observing game state."""
127 if (iig_obs_type is None) or (
128 iig_obs_type.public_info and not iig_obs_type.perfect_recall
129 ):
130 return Observer(params, self)
131 return IIGObserverForPublicInfoGame(iig_obs_type, params)
132
133 def max_chance_nodes_in_history(self):
134 """Maximun chance nodes in game history."""
135 return 0

138class MFGNormalFormState(pyspiel.State):
139 """A Mean Field Normal-Form state."""
140
141 def __init__(self, game, last_action=None):
142 """Constructor; should only be called by Game.new_initial_state."""
143 super().__init__(game)
144 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
145 self._last_action = last_action
146 self._num_actions = game.num_actions
147 self.reward_function = game.reward_function
148 self.size = game.size
149 self._terminal = False
150
151 # Represents the current probability distribution over game states.
152 # Initialized with a uniform distribution.
153 self._distribution = [1.0 / self.size for _ in range(self.size)]
154
155 def state_to_str(self, player_id=pyspiel.PlayerId.DEFAULT_PLAYER_ID):
156 """A string that uniquely identify a triplet x, t, player_id."""
157 if self._last_action is None:
158 return "initial"
159 else:
160 bonus = "_final" if self.is_terminal() else ""
161 return str(self._last_action) + bonus
162
163 # OpenSpiel (PySpiel) API functions are below. This is the standard set that
164 # should be implemented by every perfect-information sequential-move game.
165
166 def _legal_actions(self, player):
167 """Returns a list of legal actions for player and MFG nodes."""
168 if player == pyspiel.PlayerId.MEAN_FIELD:
169 return []
170 if (
171 player == pyspiel.PlayerId.DEFAULT_PLAYER_ID
172 and player == self.current_player()
173 ):
174 return list(range(self._num_actions))
175 raise ValueError(
176 f"Unexpected player {player}. "
177 "Expected a mean field or current player 0."
178 )
179
180 def _apply_action(self, action):
181 """Applies the specified action to the state."""
182 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
183 raise ValueError(
184 "_apply_action should not be called at a MEAN_FIELD state."
185 )
186 self.return_value = self._rewards()
187
188 assert self._player_id == 0
189 # Here the action is between 0 and N-1
190 if action < 0 or action > self._num_actions - 1:
191 raise ValueError(
192 "The action is between 0 and {} at any node".format(
193 self._num_actions - 1
194 )
195 )
196 self._last_action = action
197 self._player_id = pyspiel.PlayerId.MEAN_FIELD
198
199 def _action_to_string(self, player, action):
200 """Action -> string."""
201 del player
202 return str(action)
203
204 def distribution_support(self):
205 """return a list of state string."""
206 if self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
207 return [self.state_to_str()]
208 elif self._player_id == pyspiel.PlayerId.MEAN_FIELD:
209 return [str(i) for i in range(self._num_actions)]
210
211 def update_distribution(self, distribution):
212 """This function is central and specific to the logic of the MFG.
213
214 Args:
215 distribution: a distribution to register. - function should be called
216 when the node is in MEAN_FIELD state. - distribution are probabilities
217 that correspond to each game state given by distribution_support.
218 """
219 if self._player_id != pyspiel.PlayerId.MEAN_FIELD:
220 raise ValueError(
221 "update_distribution should only be called at a MEAN_FIELD state."
222 )
223 self._distribution = distribution.copy()
224 self._player_id = pyspiel.PlayerId.TERMINAL
225
226 def is_terminal(self):
227 """Returns True if the game is over."""
228 return self._player_id == pyspiel.PlayerId.TERMINAL
229
230 def current_player(self):
231 """Returns id of the next player to move, or TERMINAL if game is over."""
232 if self.is_terminal():
233 return pyspiel.PlayerId.TERMINAL
234 return self._player_id
235
236 def _rewards(self):
237 """Reward for the player for this state."""
238 reward = 0.0
239 if self._player_id == pyspiel.PlayerId.TERMINAL:
240 reward = self.reward_function(self._last_action, self._distribution)
241 return reward
242
243 def rewards(self) -> List[float]:
244 """Rewards for all players."""
245 # For now, only single-population (single-player) mean field games
246 # are supported.
247 return [self._rewards()]
248
249 def _returns(self):
250 """Returns is the sum of all payoffs collected so far."""
251 return self._rewards()
252
253 def returns(self) -> List[float]:
254 """Returns for all players."""
255 # For now, only single-population (single-player) mean field games
256 # are supported.
257 return [self._returns()]
258
259 def __str__(self):
260 """A string that uniquely identify the current state."""
261 return self.state_to_str(player_id=self._player_id)

264class Observer:
265 """Observer, conforming to the PyObserver interface (see observation.py)."""
266
267 def __init__(self, params, game):
268 """Initializes an empty observation tensor."""
269 del params
270
271 self.size = game.size
272 # +1 to allow t == horizon.
273 self.tensor = np.array([])
274 self.dict = {}
275
276 def set_from(self, state: MFGNormalFormState, player: int):
277 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
278 del player
279 # We update the observation via the shaped tensor since indexing is more
280 # convenient than with the 1-D tensor. Both are views onto the same memory.
281 del state
282 self.tensor.fill(0)
283 # state.x is None for the initial (blank) state, don't set any
284 # position bit in that case.
285 pass

288 """Observation of `state` from the PoV of `player`, as a string."""
289 del player
