14"""Implementation of the Deep Average-network Fictitious Play.
15
16Coupled with agents that compute a best-response (BR) at each iteration, instead
17of keeping in memory all the BRs from past iterations Deep Average-network
18Fictitious Play learns along the way the policy generating the average
19distribution. This is done by keeping a buffer of state-action pairs generated
20by past BRs and learning the average policy (represented by a neural network) by
21minimizing a categorical loss. This approach is inspired by the Neural
22Fictitious Self Play (NFSP) method (Heinrich & Silver, 2016), developed
23initially for imperfect information games with a finite number of players, and
24adapted here to the MFG setting.
25"""

53class AveragePolicy(rl_agent.AbstractAgent):
54 """NFSP-like agent that learns an average policy using a single network."""
55
56 def __init__(self,
57 player_id: int,
58 br_rl_agent: rl_agent.AbstractAgent,
59 state_representation_size: int,
60 num_actions: int,
61 hidden_layers_sizes: List[int],
62 params_avg_network: Optional[jnp.ndarray] = None,
63 reservoir_buffer_capacity: int = 100000,
64 batch_size: int = 128,
65 learning_rate: float = 0.01,
66 min_buffer_size_to_learn: int = 1000,
67 optimizer_str: str = 'sgd',
68 gradient_clipping: Optional[float] = None,
69 seed: int = 42,
70 tau: float = 1.0):
71 """Initialize the AveragePolicy agent."""
72 self._br_rl_agent = br_rl_agent
73 self._player_id = player_id
74 self._num_actions = num_actions
75 self._batch_size = batch_size
76 self._min_buffer_size_to_learn = min_buffer_size_to_learn
77
78 self._reservoir_buffer = reservoir_buffer.ReservoirBuffer(
79 reservoir_buffer_capacity)
80
81 # Keep track of the last training loss achieved in an update step.
82 self._last_loss_value = None
83
84 # Average policy network.
85 def network(x):
86 mlp = hk.nets.MLP(hidden_layers_sizes + [num_actions])
87 return mlp(x)
88
89 self.avg_network = hk.without_apply_rng(hk.transform(network))
90
91 def avg_network_policy(param, info_state):
92 action_values = self.avg_network.apply(param, info_state)
93 return jax.nn.softmax(action_values / tau, axis=1)
94
95 self._avg_network_policy = jax.jit(avg_network_policy)
96
97 rng = jax.random.PRNGKey(seed)
98 x = jnp.ones([1, state_representation_size])
99 # Use the specified parameters if any, or initialize the network with random
100 # weights.
101 if params_avg_network is None:
102 self._params_avg_network = self.avg_network.init(rng, x)
103 else:
104 self._params_avg_network = jax.tree_util.tree_map(lambda x: x.copy(),
105 params_avg_network)
106 self._params_avg_network = jax.device_put(self._params_avg_network)
107
108 if optimizer_str == 'adam':
109 optimizer = optax.adam(learning_rate)
110 elif optimizer_str == 'sgd':
111 optimizer = optax.sgd(learning_rate)
112 else:
113 raise ValueError('Not implemented, choose from "adam" and "sgd".')
114
115 if gradient_clipping:
116 optimizer = optax.chain(optimizer,
117 optax.clip_by_global_norm(gradient_clipping))
118
119 opt_init, opt_update = optimizer.init, optimizer.update
120
121 def opt_update_fn(params, opt_state, gradient):
122 """Learning rule (stochastic gradient descent)."""
123 updates, opt_state = opt_update(gradient, opt_state)
124 new_params = optax.apply_updates(params, updates)
125 return new_params, opt_state
126
127 self._opt_update_fn = opt_update_fn
128 self._opt_state = opt_init(self._params_avg_network)
129 self._loss_and_grad = jax.value_and_grad(self._loss_avg, has_aux=False)
130
131 self._jit_update = jax.jit(self._get_update_fn())
132
133 def _get_update_fn(self):
134 """Returns the function that updates the parameters."""
135
136 def update(param_avg, opt_state_avg, info_states, action_probs):
137 loss_val, grad_val = self._loss_and_grad(param_avg, info_states,
138 action_probs)
139 new_param_avg, new_opt_state_avg = self._opt_update_fn(
140 param_avg, opt_state_avg, grad_val)
141 return new_param_avg, new_opt_state_avg, loss_val
142
143 return update
144
145 def _act(self, info_state, legal_actions) -> Tuple[int, np.ndarray]:
146 """Returns an action and the action probabilities."""
147 info_state = np.reshape(info_state, [1, -1])
148 action_probs = self._avg_network_policy(self._params_avg_network,
149 info_state)
150 # Remove illegal actions and normalize probs
151 probs = np.zeros(self._num_actions)
152 action_probs = np.asarray(action_probs)
153 probs[legal_actions] = action_probs[0][legal_actions]
154 probs /= sum(probs)
155 action = np.random.choice(len(probs), p=probs)
156 return action, probs
157
158 @property
159 def loss(self) -> Optional[float]:
160 """Return the latest loss."""
161 return self._last_loss_value
162
163 def step(self,
164 time_step: rl_environment.TimeStep,
165 is_evaluation: bool = True) -> Optional[rl_agent.StepOutput]:
166 """Returns the action to be taken by following the average network policy.
167
168 Note that unlike most other algorithms, this method doesn't train the agent.
169 Instead, we add new samples to the reservoir buffer and the training happens
170 at a later stage.
171
172 Args:
173 time_step: an instance of rl_environment.TimeStep.
174 is_evaluation: bool, whether this is a training or evaluation call.
175
176 Returns:
177 A `rl_agent.StepOutput` containing the action probs and chosen action.
178 """
179
180 # Prepare for the next episode.
181 if time_step.last():
182 return
183
184 if is_evaluation:
185 # Use the average policy network.
186 info_state = time_step.observations['info_state'][self._player_id]
187 legal_actions = time_step.observations['legal_actions'][self._player_id]
188 action, probs = self._act(info_state, legal_actions)
189 return rl_agent.StepOutput(action=action, probs=probs)
190
191 # Use the best response agent and add the transition in the reservoir
192 # buffer.
193 br_agent_output = self._br_rl_agent.step(time_step, is_evaluation=True)
194 self._add_transition(time_step, br_agent_output)
195 return br_agent_output
196
197 def _add_transition(self, time_step, agent_output):
198 """Adds the new transition using `time_step` to the reservoir buffer.
199
200 Transitions are in the form (time_step, agent_output.probs, legal_mask).
201
202 Args:
203 time_step: an instance of rl_environment.TimeStep.
204 agent_output: an instance of rl_agent.StepOutput.
205 """
206 legal_actions = time_step.observations['legal_actions'][self._player_id]
207 legal_actions_mask = np.zeros(self._num_actions)
208 legal_actions_mask[legal_actions] = 1.0
209 transition = Transition(
210 info_state=(time_step.observations['info_state'][self._player_id][:]),
211 action_probs=agent_output.probs,
212 legal_actions_mask=legal_actions_mask)
213 self._reservoir_buffer.add(transition)
214
215 def _loss_avg(self, param_avg, info_states, action_probs):
216 avg_logit = self.avg_network.apply(param_avg, info_states)
217 loss_value = -jnp.sum(
218 action_probs * jax.nn.log_softmax(avg_logit)) / avg_logit.shape[0]
219 return loss_value
220
221 def learn(self) -> Optional[float]:
222 """Compute the loss on sampled transitions and perform a avg-network update.
223
224 If there are not enough elements in the buffer, no loss is computed and
225 `None` is returned instead.
226
227 Returns:
228 The average loss obtained on this batch of transitions or `None`.
229 """
230 if (len(self._reservoir_buffer) < self._batch_size or
231 len(self._reservoir_buffer) < self._min_buffer_size_to_learn):
232 return None
233
234 transitions = self._reservoir_buffer.sample(self._batch_size)
235 info_states = np.asarray([t.info_state for t in transitions])
236 action_probs = np.asarray([t.action_probs for t in transitions])
237
238 self._params_avg_network, self._opt_state, loss_val_avg = self._jit_update(
239 self._params_avg_network, self._opt_state, info_states, action_probs)
240 self._last_loss_value = float(loss_val_avg)
241 return loss_val_avg

244class AverageNetworkFictitiousPlay(object):
245 """Deep Average-network Fictitious Play.
246
247 See the file description for more information.
248 """

250 def __init__(self,
251 game: pyspiel.Game,
252 envs: Sequence[rl_environment.Environment],
253 br_rl_agents: Sequence[rl_agent.AbstractAgent],
254 num_episodes_per_iteration: int,
255 num_training_steps_per_iteration: int,
256 eval_every: int = 200,
257 logging_fn: Optional[Callable[[int, int, Dict[str, Any]],
258 None]] = None,
259 **kwargs):
260 """Initializes the greedy policy.
261
262 Args:
263 game: The game to analyze.
264 envs: RL environment for each player.
265 br_rl_agents: Best response, e.g. DQN, agents for each player.
266 num_episodes_per_iteration: Number of episodes to collect samples that are
267 added to the reservoir buffer.
268 num_training_steps_per_iteration: Number of steps to train the average
269 policy in each iteration.
270 eval_every: Number of training steps between two evaluations.
271 logging_fn: Callable for logging the metrics. The arguments will be the
272 current iteration, episode and a dictionary of metrics to log.
273 **kwargs: kwargs passed to the AveragePolicy() constructor.
274 """
275 self._game = game
276 self._envs = envs
277 self._num_episodes_per_iteration = num_episodes_per_iteration
278 self._num_training_steps_per_iteration = num_training_steps_per_iteration
279 self._eval_every = eval_every
280 self._logging_fn = logging_fn
281
282 self._num_players = game.num_players()
283 self._fp_iteration = 0
284
285 env = self._envs[0]
286 info_state_size = env.observation_spec()['info_state'][0]
287 num_actions = env.action_spec()['num_actions']
288
289 self._avg_rl_agents = [
290 AveragePolicy(p, br_rl_agents[p], info_state_size, num_actions,
291 **kwargs) for p in range(self._num_players)
292 ]
293 self._policy = rl_agent_policy.JointRLAgentPolicy(
294 self._game,
295 {idx: agent for idx, agent in enumerate(self._avg_rl_agents)},
296 use_observation=env.use_observation)
297 self._update_distribution()
298
299 def _update_distribution(self):
300 """Calculates the current distribution and updates the environments."""
301 self._distribution = distribution.DistributionPolicy(
302 self._game, self._policy)
303 for env in self._envs:
304 env.update_mfg_distribution(self._distribution)
305
306 @property
307 def policy(self) -> rl_agent_policy.JointRLAgentPolicy:
308 return self._policy

310 def iteration(self):
311 """An average-network fictitious play step."""
312 # Generate samples using latest best-response and add them to the reservoir
313 # buffer. Note that the algorithm is agnostic to the best-response policies
314 # as we only use them to collect new samples. They can be approximate (e.g.
315 # backed by a deep algorithm) or exact.
316 training.run_episodes(
317 self._envs,
318 self._avg_rl_agents,
319 num_episodes=self._num_episodes_per_iteration,
320 is_evaluation=False)
321
322 # Train the average policy.
323 for step in range(self._num_training_steps_per_iteration):
324 for avg_rl_agent in self._avg_rl_agents:
325 avg_rl_agent.learn()
326
327 if self._logging_fn and (step + 1) % self._eval_every == 0:
328 self._logging_fn(
329 self._fp_iteration, step, {
330 f'avg_agent{i}/loss': float(agent.loss)
331 for i, agent in enumerate(self._avg_rl_agents)
332 })
333
334 # Update the distribution.
335 self._update_distribution()
336 self._fp_iteration += 1
