46def assign_dictionary_input_to_object(dict_object: dict[str, Any],
47 road_sections: Collection[str],

60class Network:
61 """Network implementation.
62
63 A network is basically a directed graph with a volume delay function on each
64 of its edges. Each vertex is refered to as a string (for example "A") and each
65 edge as a string f"{node1}->{node2}" (for example "A->B"). The network is
66 created from a adjacency list. Each road section is mapped to an action index
67 (positive integer) in _action_by_road_section. The volume delay function on
68 each road section rs is given by
69 _free_flow_travel_time[rs]*(1+ _a[rs]*(v/_capacity[rs])**_b[rs])
70 where v is the volume on the road section rs, according to the U.S. Bureau of
71 Public Road (BPR). Such functions are called fundamental diagram of traffic
72 flow.
73
74 If one would like to plot the network then node position should be passed
75 in the constructor. Then return_list_for_matplotlib_quiver can be used with
76 Matplotlib:
77 ```python3
78 fig, ax = plt.subplots()
79 o_xs, o_ys, d_xs, d_ys = g.return_list_for_matplotlib_quiver()
80 ax.quiver(o_xs, o_ys, np.subtract(d_xs, o_xs), np.subtract(d_ys, o_ys),
81 color="b", angles='xy', scale_units='xy', scale=1)
82 ```
83
84 See the Network tests for an example.
85 Attributes: _a, _b, _capacity, _free_flow_travel_time: dictionary that maps
86 road section string representation to its a, b, relative capacity and free
87 flow travel time coefficient in its BPR function.
88 _action_by_road_section: dictionary that maps road section to action id.
89 _adjacency_list: adjacency list of the line graph of the road network.
90 _node_position: dictionary that maps node to couple of float encoding x and
91 y position of the node. None by default.
92 _road_section_by_action: dictionary that maps action id to road section.
93 """
94 _a: dict[str, float]
95 _b: dict[str, float]
96 _action_by_road_section: dict[str, int]
97 _adjacency_list: dict[str, Collection[str]]
98 _capacity: dict[str, float]
99 _free_flow_travel_time: dict[str, float]
100 _node_position: dict[str, tuple[float, float]]
101 _road_section_by_action: dict[int, str]
102
103 def __init__(self,
104 adjacency_list: dict[str, Collection[str]],
105 node_position: Optional[dict[str, tuple[float, float]]] = None,
106 bpr_a_coefficient: Optional[dict[str, float]] = None,
107 bpr_b_coefficient: Optional[dict[str, float]] = None,
108 capacity: Optional[dict[str, float]] = None,
109 free_flow_travel_time: Optional[dict[str, float]] = None):
110 self._adjacency_list_adjacency_list = adjacency_list
111 self._action_by_road_section_action_by_road_section = self._create_action_by_road_section()
112 self._road_section_by_action = {
113 v: k for k, v in self._action_by_road_section_action_by_road_section.items()
114 }
115
116 nodes = set(adjacency_list)
117 # pylint: disable=g-complex-comprehension
118 assert all(destination_node in nodes
119 for destination_nodes in self._adjacency_list_adjacency_list.values()
120 for destination_node in destination_nodes), (
121 "Adjacency list is not correct.")
122 # pylint: enable=g-complex-comprehension
123 if node_position:
124 assert set(node_position) == nodes
125 self._node_position = node_position
126 else:
127 self._node_position = None
128 self._a = assign_dictionary_input_to_object(bpr_a_coefficient,
129 self._action_by_road_section_action_by_road_section, 0)
130 self._b = assign_dictionary_input_to_object(bpr_b_coefficient,
131 self._action_by_road_section_action_by_road_section, 1)
132 self._capacity = assign_dictionary_input_to_object(
133 capacity, self._action_by_road_section_action_by_road_section, 1)
134 self._free_flow_travel_time = assign_dictionary_input_to_object(
135 free_flow_travel_time, self._action_by_road_section_action_by_road_section, 1)
136 assert hasattr(self, "_adjacency_list")
137 assert hasattr(self, "_node_position")
138 assert hasattr(self, "_a")
139 assert hasattr(self, "_b")
140 assert hasattr(self, "_capacity")
141 assert hasattr(self, "_free_flow_travel_time")

143 def _create_action_by_road_section(self) -> tuple[set[str], dict[int, str]]:
144 """Create dictionary that maps movement to action.
145
146 The dictionary that maps movement to action is used to define the action
147 from a movement that a vehicle would like to do.
148 Returns:
149 action_by_road_section: dictionary with key begin a movement for example
150 "O->A" and value the action numbers. Action numbers are succesive
151 integers indexed from 1.
152 """
153 action_by_road_section = {}
154 action_number = NO_POSSIBLE_ACTION + 1
155 for origin, successors in sorted(self._adjacency_list_adjacency_list.items()):
156 for destination in successors:
157 road_section = _road_section_from_nodes(origin, destination)
158 if road_section in action_by_road_section:
159 raise ValueError((
160 f"{road_section} exists twice in the adjacency list. The current "
161 "network implementation does not enable parallel links."))
162 action_by_road_section[road_section] = action_number
163 action_number += 1
164 return action_by_road_section

166 def num_links(self) -> int:
167 """Returns the number of road sections."""
168 return len(self._action_by_road_section_action_by_road_section)
169
170 def num_actions(self) -> int:
171 """Returns the number of possible actions.
172
173 Equal to the number of road section + 1. An action could either be moving to
174 a specific road section or not move.
175 """
176 return 1 + self.num_links()

178 def links(self) -> list[str]:
179 """Returns the road sections as a list."""
180 return list(self._action_by_road_section_action_by_road_section)

182 def get_successors(self, node: str) -> Collection[str]:
183 """Returns the successor nodes of the node."""
184 return self._adjacency_list_adjacency_list[node]

186 def get_action_id_from_movement(self, origin: str, destination: str) -> int:
187 """Maps two connected nodes to an action."""
188 return self._action_by_road_section_action_by_road_section[_road_section_from_nodes(
189 origin, destination)]

191 def get_road_section_from_action_id(self, action_id: int) -> str:
192 """Maps a action to the corresponding road section."""
193 return self._road_section_by_action[action_id]

195 def is_location_at_sink_node(self, road_section: str) -> bool:
196 """Returns True if the road section has no successors."""
197 start_section, end_section_node = _nodes_from_road_section(road_section)
198 if start_section not in self._adjacency_list_adjacency_list:
199 raise KeyError(f"{start_section} is not a network node.")
200 return not self.get_successors(end_section_node)

203 """Assert that vehicles have valid origin and destination."""

220 def get_travel_time(self, road_section: str, volume: float) -> int:
221 """Returns travel time on the road section given the volume on it.
222
223 Volume unit should be the same as the capacity unit.
224 Travel time unit is the free flow travel time unit.
225 Args:
226 road_section: the road section.
227 volume: the volume on the road section.
228 """
229 return self._free_flow_travel_time[road_section] * (
230 1.0 + self._a[road_section] *
231 (volume / self._capacity[road_section])**self._b[road_section])

233 def assert_valid_action(self, action: int, road_section: str = None):
234 """Assert that an action as a int is valid.
235
236 The action should be a int between 1 and num_actions. In case road_section
237 is not None then it is test if the action correspond to going on a road
238 section which is a successor of road_section.
239
240 Args:
241 action: the action,
242 road_section: the road section.
243 """
244 assert isinstance(action, int), f"{action} is not a int."
245 assert 1 <= action < self.num_actions(), str(action)
246 if road_section is not None:
247 new_road_section = self.get_road_section_from_action_id(action)
248 origin_new_section, end_new_section = _nodes_from_road_section(
249 new_road_section)
250 _, end_section_node = _nodes_from_road_section(road_section)
251 assert end_section_node == origin_new_section, (
252 f"The action is not legal, trying to go to {new_road_section} "
253 f"from {road_section} without going through {end_section_node}"
254 ".")
255 successors = self.get_successors(origin_new_section)
256 assert end_new_section in successors, (
257 f"Invalid action {new_road_section}. It is not a successors of"
258 f" {end_section_node}: {successors}.")

260 def return_position_of_road_section(self,
261 road_section: str) -> tuple[float, float]:
262 """Returns position of the middle of theroad section as (x,y)."""
263 assert self._node_position is not None, (
264 "The network should have node positions in order to be plot.")
265 o_link, d_link = _nodes_from_road_section(road_section)
266 o_x, o_y = self._node_position[o_link]
267 d_x, d_y = self._node_position[d_link]
268 return (o_x + d_x) / 2, (o_y + d_y) / 2

270 def return_list_for_matplotlib_quiver(
271 self) -> tuple[list[float], list[float], list[float], list[float]]:
272 """Returns 4 list of encoding the positions of the road sections.
273
274 ```python3
275 fig, ax = plt.subplots()
276 o_xs, o_ys, d_xs, d_ys = g.return_list_for_matplotlib_quiver()
277 ax.quiver(o_xs, o_ys, np.subtract(d_xs, o_xs), np.subtract(d_ys, o_ys),
278 color="b", angles='xy', scale_units='xy', scale=1)
279 ```
280 will show the network.
281 Returns:
282 o_xs, o_ys, d_xs, d_ys: list of the start x and y positions and of the end
283 x and y postions of each road section. Each element of each list
284 corresponds to one road section.
285 """
286 assert self._node_position is not None, (
287 "The network should have node positions in order to be plot.")
288 o_xs = []
289 o_ys = []
290 d_xs = []
291 d_ys = []
292 for road_section in self._action_by_road_section_action_by_road_section:
293 o_link, d_link = _nodes_from_road_section(road_section)
294 o_x, o_y = self._node_position[o_link]
295 d_x, d_y = self._node_position[d_link]
296 o_xs.append(o_x)
297 o_ys.append(o_y)
298 d_xs.append(d_x)
299 d_ys.append(d_y)
300 return o_xs, o_ys, d_xs, d_ys
