65 def _format_vec(vec):
66  """Returns a readable format for a vector."""
67  full_fmt = "".join(_format_value(v) for v in vec)
68  short_fmt = None
69  max_len = 250
70  vec2int = lambda vec: int("".join("1" if b else "0" for b in vec), 2)
71  if len(vec) > max_len:
72  if all(v == 0 for v in vec):
73  short_fmt = f"zeros({len(vec)})"
74  elif all(v in (0, 1) for v in vec):
75  sz = (len(vec) + 15) // 16
76  # To reconstruct the original vector:
77  # binvec = lambda n, x: [int(x) for x in f"{x:0>{n}b}"]
78  short_fmt = f"binvec({len(vec)}, 0x{vec2int(vec):0>{sz}x})"
79  if short_fmt and len(short_fmt) < len(full_fmt):
80  return short_fmt
81  else:
82  return full_fmt

102 def _format_tensor(tensor, tensor_name, max_cols=120):
103  """Formats a tensor in an easy-to-view format as a list of lines."""
104  if ((not tensor.shape) or (tensor.shape == (0,)) or (len(tensor.shape) > 3) or
105  not np.logical_or(tensor == 0, tensor == 1).all()):
106  vec = ", ".join(str(round(v, 5)) for v in tensor.ravel())
107  return ["{} = [{}]".format(tensor_name, vec)]
108  elif len(tensor.shape) == 1:
109  return ["{}: {}".format(tensor_name, _format_vec(tensor))]
110  elif len(tensor.shape) == 2:
111  if len(tensor_name) + tensor.shape[1] + 2 < max_cols:
112  lines = ["{}: {}".format(tensor_name, _format_vec(tensor[0]))]
113  prefix = " " * (len(tensor_name) + 2)
114  else:
115  lines = ["{}:".format(tensor_name), _format_vec(tensor[0])]
116  prefix = ""
117  for row in tensor[1:]:
118  lines.append(prefix + _format_vec(row))
119  return lines
120  elif len(tensor.shape) == 3:
121  lines = ["{}:".format(tensor_name)]
122  rows = []
123  for m in tensor:
124  formatted_matrix = _format_matrix(m)
125  if (not rows) or (len(rows[-1][0] + formatted_matrix[0]) + 2 > max_cols):
126  rows.append(formatted_matrix)
127  else:
128  rows[-1] = rows[-1] + " " + formatted_matrix
129  for i, big_row in enumerate(rows):
130  if i > 0:
131  lines.append("")
132  for row in big_row:
133  lines.append("".join(row))
134  return lines

189 class ShouldDisplayStateTracker:
190  """Determines whether a state is interesting enough to display."""
191 
192  def __init__(self):
193  self.states_by_playerstates_by_player = collections.defaultdict(int)
194 
195  def __call__(self, state) -> bool:
196  """Returns True if a state is sufficiently interesting to display."""
197  player = state.current_player()
198  count = self.states_by_playerstates_by_player[player]
199  self.states_by_playerstates_by_player[player] += 1
200  if count == 0:
201  # Always display the first state for a player
202  return True
203  elif player == -1:
204  # For chance moves, display the first two only
205  return count < 2
206  else:
207  # For regular player moves, display the first three and selected others
208  return (count < 3) or (count % 10 == 0)

211 def playthrough_lines(game_string, alsologtostdout=False, action_sequence=None,

214  """Returns a playthrough of the specified game as a list of lines.
215 
216  Actions are selected uniformly at random, including chance actions.
217 
218  Args:
219  game_string: string, e.g. 'markov_soccer' or 'kuhn_poker(players=4)'.
220  alsologtostdout: Whether to also print the trace to stdout. This can be
221  useful when an error occurs, to still be able to get context information.
222  action_sequence: A (possibly partial) list of action choices to make.
223  observation_params_string: Optional observation parameters for constructing
224  an observer.
225  seed: A(n optional) seed to initialize the random number generator from.
226  """

259  )

266  # The default observation is the same as the public observation, while private
267  # observations are always empty.

380  if s is not None:
381  add_line(f'PrivateObservationString({player}) = "{_escape(s)}"')
382  if default_observation and default_observation.tensor is not None:
383  for player in players:
384  default_observation.set_from(state, player)

387  label += f".{name}" if name != "observation" else ""
388  for line in _format_tensor(tensor, label):
389  add_line(line)

391  add_line('SerializeState() = "{}"'.format(_escape(state.serialize())))
392  if not state.is_chance_node():
393  add_line("Rewards() = {}".format(_format_float_vector(state.rewards())))
394  add_line("Returns() = {}".format(_format_float_vector(state.returns())))
395  if state.is_terminal():
396  break
397  if state.is_chance_node():
398  add_line("ChanceOutcomes() = {}".format(
399  _format_chance_outcomes(state.chance_outcomes())))

401  add_line("DistributionSupport() = {}".format(
402  state.distribution_support()))
403  num_states = len(state.distribution_support())

405  [1. / num_states] * num_states if num_states else [])

407  assert action_sequence[state_idx] == "update_distribution", (
408  f"Unexpected action at MFG node: {action_sequence[state_idx]}, "

412  add_line("action: update_distribution", force=True)

414  for player in players:
415  add_line("LegalActions({}) = [{}]".format(
416  player, ", ".join(str(x) for x in state.legal_actions(player))))

419  player, ", ".join('"{}"'.format(state.action_to_string(player, x))

433  format(", ".join(
434  '"{}"'.format(state.action_to_string(player, action))

441  str(x) for x in state.legal_actions())))
442  add_line("StringLegalActions() = [{}]".format(", ".join(
443  '"{}"'.format(state.action_to_string(state.current_player(), x))

465 def _playthrough_params(lines):
466  """Returns the playthrough parameters from a playthrough record.
467 
468  Args:
469  lines: The playthrough as a list of lines.
470 
471  Returns:
472  A `dict` with entries:
473  game_string: string, e.g. 'markov_soccer'.
474  action_sequence: a list of action choices made in the playthrough.
475  Suitable for passing to playthrough to re-generate the playthrough.
476 
477  Raises:
478  ValueError if the playthrough is not valid.
479  """
480  params = {"action_sequence": []}
481  use_action_ids = _USE_ACTION_IDS.value
482  for line in lines:
483  match_game = re.fullmatch(r"game: (.*)", line)
484  match_observation_params = re.fullmatch(r"observation_params: (.*)", line)
485  match_update_distribution = (line == "action: update_distribution")
486  if use_action_ids:
487  match_action = re.fullmatch(r"action: (.*)", line)
488  match_actions = re.fullmatch(r"actions: \[(.*)\]", line)
489  else:
490  match_action = re.fullmatch(r'# Apply action "(.*)"', line)
491  match_actions = re.fullmatch(r"# Apply joint action \[(.*)\]", line)
492  if match_game:
493  params["game_string"] = match_game.group(1)
494  elif match_observation_params:
495  params["observation_params_string"] = match_observation_params.group(1)
496  elif match_update_distribution:
497  params["action_sequence"].append("update_distribution")
498  elif match_action:
499  matched = match_action.group(1)
500  if use_action_ids:
501  params["action_sequence"].append(int(matched))
502  else:
503  params["action_sequence"].append(matched)
504  elif match_actions:
505  if use_action_ids:
506  params["action_sequence"].append(
507  [int(x) for x in match_actions.group(1).split(", ")])
508  else:
509  params["action_sequence"].append(
510  [x[1:-1] for x in match_actions.group(1).split(", ")])
511  if "game_string" in params:
512  return params
513  raise ValueError("Could not find params")

530 def update_path(path, shard_index=0, num_shards=1):
531  """Regenerates all playthroughs in the path."""
532  if os.path.isfile(path):
533  file_list = [path]
534  else:
535  file_list = sorted(os.listdir(path))
536  for filename in file_list[shard_index::num_shards]:
537  try:
538  original, kwargs = _read_playthrough(os.path.join(path, filename))
539  try:
540  pyspiel.load_game(kwargs["game_string"])
541  except pyspiel.SpielError as e:
542  if "Unknown game" in str(e):
543  print(f"\x1b[0J[Skipped] Skipping game {filename} as ",
544  f"{kwargs['game_string']} is not available.")
545  continue
546  else:
547  raise
548  new = playthrough(**kwargs)
549  if original == new:
550  print(f"\x1b[0J {filename}", end="\r")
551  else:
552  with open(os.path.join(path, filename), "w") as f:
553  f.write(new)
554  print(f"\x1b[0JUpdated {filename}")
555  except Exception as e: # pylint: disable=broad-except
556  print(f"\x1b[0J{filename} failed: {e}")
