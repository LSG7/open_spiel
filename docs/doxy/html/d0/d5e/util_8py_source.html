22def compute_payoff_epsilon(
23 game: coalitional_game.CoalitionValueCalculator,
24 p: np.ndarray
25) -> float:
26 """For a payoff vector p, get max_e s.t. p dot c + e >= V(c).
27
28 Warning! Enumerates all coalitions.
29
30 Args:
31 game: the game to enumerate.
32 p: the payoff vector.
33
34 Returns:
35 the value max_e s.t. p dot c + e >= V(C) for all subsets C subseteq N.
36 """
37 epsilon = 0
38 for c in itertools.product([0, 1], repeat=game.get_num_players()):
39 coalition = np.asarray(c)
40 val_c = game.get_coalition_values(coalition)
41 payoffs_to_coalition = np.inner(p, coalition)
42 epsilon = max(epsilon, val_c - payoffs_to_coalition)
43 return epsilon
