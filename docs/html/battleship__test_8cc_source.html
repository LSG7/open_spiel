247 void TestNashEquilibriumInSmallBoard() {
248  // We replicate the same setting as page 7 of [1].
249  //
250  // There, each player has a 1x3 board with a single ship of size 1 and value
251  // 1.0. Each player has two shots available. The loss multiplier is 2.0, so
252  // this is a *general-sum* game.
253  //
254  // The only Nash equilibrium of the game is for all players to place their
255  // ship at random, and then shoot at random.
256  //
257  //
258  // [1]:
259  // https://papers.nips.cc/paper/9122-correlation-in-extensive-form-games-saddle-point-formulation-and-benchmarks.pdf#page=7
260 
261  const std::shared_ptr<const Game> game =
262  LoadGame("battleship", {{"board_width", GameParameter(3)},
263  {"board_height", GameParameter(1)},
264  {"ship_sizes", GameParameter("[1]")},
265  {"ship_values", GameParameter("[1.0]")},
266  {"num_shots", GameParameter(2)},
267  {"allow_repeated_shots", GameParameter(false)},
268  {"loss_multiplier", GameParameter(2.0)}});
269  SPIEL_CHECK_EQ(game->GetType().utility, GameType::Utility::kGeneralSum);
270 
271  const TabularPolicy policy = GetUniformPolicy(*game);
272  const std::vector<double> expected_utilities = algorithms::ExpectedReturns(
273  *game->NewInitialState(), policy,
274  /* depth_limit = */ std::numeric_limits<int>::max());
275 
276  // Under the uniformly random policy, we expect that Player 0 and Player 1
277  // will sink their opponent's ship with probability 5/9 and 1/3, respectively.
278  //
279  // Correspondingly, Player 0's expected utility is 5/9 - 2 * 1/3 = -1/9 (the 2
280  // comes from the loss multiplier), while Player 1's expected utility is 1/3 -
281  // 2 * 5/9 = -7/9.
282  SPIEL_CHECK_FLOAT_EQ(expected_utilities[Player{0}], -1.0 / 9);
283  SPIEL_CHECK_FLOAT_EQ(expected_utilities[Player{1}], -7.0 / 9);
284 
285  // We check that this joint policy is a best response, by computing the Nash
286  // gap.
287  SPIEL_CHECK_FLOAT_NEAR(algorithms::NashConv(*game, policy), 0.0, 1e-9);
288 
289  // TODO(gfarina): When OpenSpiel implements algorithms for optimal
290  // EFCE/EFCCE/NFCCE,finish checking that we exactly replicate the same
291  // results as [1] in this game.
292 }

294 struct GameSize {
295  uint32_t num_sequences[2] = {0, 0}; // Layout: [Pl.0, Pl.1].
296  uint32_t num_infostates[2] = {0, 0}; // Layout: [Pl.0, Pl.1].
297  uint32_t num_terminal_states = 0;
298 };
299 
300 GameSize ComputeGameSize(const std::shared_ptr<const Game> game) {
301  std::map<std::string, std::unique_ptr<open_spiel::State>> all_states =
302  open_spiel::algorithms::GetAllStates(
303  *game, /* depth_limit = */ std::numeric_limits<int>::max(),
304  /* include_terminals = */ true,
305  /* include_chance_states = */ false);
306 
307  GameSize size;
308 
309  // Account for empty sequence.
310  size.num_sequences[Player{0}] = 1;
311  size.num_sequences[Player{1}] = 1;
312 
313  absl::flat_hash_set<std::string> infosets;
314  for (const auto& [_, state] : all_states) {
315  if (state->IsTerminal()) {
316  ++size.num_terminal_states;
317  } else {
318  const Player player = state->CurrentPlayer();
319  SPIEL_CHECK_TRUE(player == Player{0} || player == Player{1});
320 
321  // NOTE: there is no requirement that infostates strings be unique across
322  // players. So, we disambiguate the player by prepending it.
323  const std::string infostate_string =
324  absl::StrCat(player, state->InformationStateString());
325 
326  if (infosets.insert(infostate_string).second) {
327  // The infostate string was not present in the hash set. We update the
328  // tally of infosets and sequences for the player.
329  size.num_infostates[player] += 1;
330  size.num_sequences[player] += state->LegalActions().size();
331  }
332  }
333  }
334 
335  return size;
336 }

338 void TestGameSizes() {
339  // We expect these game sizes when using allow_repeated_shots = False:
340  //
341  // +-------+-------+-------+-----------------+----------------+----------+
342  // | Grid | Shots | Ship | Num sequences | Num infosets | Terminal |
343  // | | | sizes | pl 0 | pl 1 | pl 0 | pl 1 | states |
344  // +-------+-------+-------+--------+--------+-------+--------+----------+
345  // | 2 x 2 | 2 | [1] | 165 | 341 | 53 | 109 | 1072 |
346  // | 2 x 2 | 3 | [1] | 741 | 917 | 341 | 397 | 2224 |
347  // | 2 x 2 | 2 | [1;2] | 1197 | 3597 | 397 | 1189 | 9216 |
348  // | 2 x 2 | 3 | [1;2] | 13485 | 22029 | 6541 | 10405 | 32256 |
349  // +-------+-------+-------+--------+--------+-------+--------+----------+
350  // | 2 x 3 | 2 | [1] | 943 | 3787 | 187 | 751 | 19116 |
351  // | 2 x 3 | 3 | [1] | 15343 | 46987 | 3787 | 11551 | 191916 |
352  // | 2 x 3 | 4 | [1] | 144943 | 306187 | 46987 | 97951 | 969516 |
353  // +-------+-------+-------+--------+--------+-------+--------+----------+
354 
355  // To simplify the construction of game instance we introduce a lambda.
356  //
357  // Since the value of the ships and the loss multiplier do not affect the game
358  // size, the lambda fills those parameters with 2
359  const auto ConstructInstance =
360  [](const std::string& grid, const int num_shots,
361  const std::string& ship_sizes_str) -> std::shared_ptr<const Game> {
362  std::vector<std::string> grid_dimensions = absl::StrSplit(grid, 'x');
363  SPIEL_CHECK_EQ(grid_dimensions.size(), 2);
364 
365  const GameParameter board_width(std::stoi(grid_dimensions[1]));
366  const GameParameter board_height(std::stoi(grid_dimensions[0]));
367  const GameParameter ship_sizes(ship_sizes_str);
368 
369  // We reuse the ship sizes as ship values. The values of the ships do not
370  // affect the game size.
371  const GameParameter ship_values(ship_sizes_str);
372 
373  return LoadGame("battleship",
374  {{"board_width", board_width},
375  {"board_height", board_height},
376  {"ship_sizes", ship_sizes},
377  {"ship_values", ship_values},
378  {"num_shots", GameParameter(num_shots)},
379  {"allow_repeated_shots", GameParameter(false)},
380  {"loss_multiplier", GameParameter(2.0)}});
381  };
382 
383  // 2x2 grid, 2 shots, ships sizes [1].
384  GameSize size = ComputeGameSize(ConstructInstance("2x2", 2, "[1]"));
385  SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 165);
386  SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 341);
387  SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 53);
388  SPIEL_CHECK_EQ(size.num_infostates[Player{1}], 109);
389  SPIEL_CHECK_EQ(size.num_terminal_states, 1072);
390 
391  // 2x2 grid, 3 shots, ships sizes [1].
392  size = ComputeGameSize(ConstructInstance("2x2", 3, "[1]"));
393  SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 741);
394  SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 917);
395  SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 341);
396  SPIEL_CHECK_EQ(size.num_infostates[Player{1}], 397);
397  SPIEL_CHECK_EQ(size.num_terminal_states, 2224);
398 
399  // 2x2 grid, 2 shots, ships sizes [1;2].
400  size = ComputeGameSize(ConstructInstance("2x2", 2, "[1;2]"));
401  SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 1197);
402  SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 3597);
403  SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 397);
404  SPIEL_CHECK_EQ(size.num_infostates[Player{1}], 1189);
405  SPIEL_CHECK_EQ(size.num_terminal_states, 9216);
406 
407  // 2x2 grid, 3 shots, ships sizes [1;2].
408  size = ComputeGameSize(ConstructInstance("2x2", 3, "[1;2]"));
409  SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 13485);
410  SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 22029);
