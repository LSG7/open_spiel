87 void WriteTensor(const State& observed_state, int player,
88 Allocator* allocator) const override {
89 auto& state = open_spiel::down_cast<const GinRummyState&>(observed_state);
90 SPIEL_CHECK_GE(player, 0);
91 SPIEL_CHECK_LT(player, state.num_players_);
92
93 if (iig_obs_type_.perfect_recall) {
94 SpielFatalError(
95 "GinRummyObserver: tensor with perfect recall not implemented.");
96 }
97
98 // Observing player.
99 WriteObservingPlayer(state, player, allocator);
100
101 // Private hand(s).
102 if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
103 WriteSinglePlayerHand(state, player, allocator);
104 } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
105 WriteAllPlayerHands(state, allocator);
106 }
107
108 // Public information.
109 if (iig_obs_type_.public_info) {
110 WriteCurrentPlayer(state, allocator);
111 WriteKnockCard(state, allocator);
112 WriteUpcard(state, allocator);
113 WriteDiscardPile(state, allocator);
114 WriteStockSize(state, allocator);
115 WriteLayedMelds(state, allocator);
116 }

119 std::string StringFrom(const State& observed_state,
120 int player) const override {
121 auto& state = open_spiel::down_cast<const GinRummyState&>(observed_state);
122 SPIEL_CHECK_GE(player, 0);
123 SPIEL_CHECK_LT(player, state.num_players_);
124
125 if (iig_obs_type_.perfect_recall) {
126 if (iig_obs_type_.public_info &&
127 iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
128 return state.aohs_[player].ToString();
129 } else {
130 SpielFatalError(
131 "GinRummyObserver: string with perfect recall is implemented only"
132 " for the (default) info state observation type.");
133 }
134 }
135
136 std::string rv;
137 absl::StrAppend(&rv, "\nKnock card: ", state.knock_card_);
138 absl::StrAppend(&rv, "\nPrev upcard: ",
139 state.utils_.CardString(state.prev_upcard_));
140 absl::StrAppend(&rv, "\nRepeated move: ", state.repeated_move_);
141 absl::StrAppend(&rv, "\nCurrent player: ", state.cur_player_);
142 absl::StrAppend(&rv, "\nPhase: ",
143 GinRummyState::kPhaseString[static_cast<int>(state.phase_)],
144 "\n");
145 if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers ||
146 (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer &&
147 player == 0)) {
148 absl::StrAppend(&rv, "\nPlayer0: Deadwood=", state.deadwood_[0]);
149 } else {
150 absl::StrAppend(&rv, "\nPlayer0:");
151 }
152 if (state.knocked_[1] && !state.layoffs_.empty()) {
153 absl::StrAppend(&rv, "\nLayoffs: ");
154 for (int card : state.layoffs_) {
155 absl::StrAppend(&rv, state.utils_.CardString(card));
156 }
157 }
158 if (!state.layed_melds_[0].empty()) {
159 absl::StrAppend(&rv, "\nLayed melds:");
160 for (int meld_id : state.layed_melds_[0]) {
161 absl::StrAppend(&rv, " ");
162 std::vector<int> meld = state.utils_.int_to_meld.at(meld_id);
163 for (int card : meld) {
164 absl::StrAppend(&rv, state.utils_.CardString(card));
165 }
166 }
167 }
168 if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers ||
169 (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer &&
170 player == 0)) {
171 absl::StrAppend(&rv, "\n", state.utils_.HandToString(state.hands_[0]));
172 } else {
173 absl::StrAppend(&rv, "\n", state.utils_.HandToString(std::vector<int>()));
174 }
175
176 absl::StrAppend(&rv, "\nStock size: ", state.stock_size_);
177 absl::StrAppend(&rv, " Upcard: ", state.utils_.CardString(state.upcard_));
178 absl::StrAppend(&rv, "\nDiscard pile: ");
179 for (int card : state.discard_pile_) {
180 absl::StrAppend(&rv, state.utils_.CardString(card));
181 }
182 if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers ||
183 (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer &&
184 player == 1)) {
185 absl::StrAppend(&rv, "\n\nPlayer1: Deadwood=", state.deadwood_[1]);
186 } else {
187 absl::StrAppend(&rv, "\n\nPlayer1:");
188 }
189 if (state.knocked_[0] && !state.layoffs_.empty()) {
190 absl::StrAppend(&rv, "\nLayoffs: ");
191 for (int card : state.layoffs_) {
192 absl::StrAppend(&rv, state.utils_.CardString(card));
193 }
194 }
195 if (!state.layed_melds_[1].empty()) {
196 absl::StrAppend(&rv, "\nLayed melds:");
197 for (int meld_id : state.layed_melds_[1]) {
198 absl::StrAppend(&rv, " ");
199 std::vector<int> meld = state.utils_.int_to_meld.at(meld_id);
200 for (int card : meld) {
201 absl::StrAppend(&rv, state.utils_.CardString(card));
202 }
203 }
204 }
205
206 if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers ||
207 (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer &&
208 player == 1)) {
209 absl::StrAppend(&rv, "\n", state.utils_.HandToString(state.hands_[1]));
210 } else {
211 absl::StrAppend(&rv, "\n", state.utils_.HandToString(std::vector<int>()));
212 }
213 return rv;
214 }

277GinRummyState::GinRummyState(std::shared_ptr<const Game> game, bool oklahoma,
278 int knock_card, int gin_bonus, int undercut_bonus,
279 int num_ranks, int num_suits, int hand_size)
280 : State(std::move(game)),
281 oklahoma_(oklahoma),
282 knock_card_(knock_card),
283 gin_bonus_(gin_bonus),
284 undercut_bonus_(undercut_bonus),
285 num_ranks_(num_ranks),
286 num_suits_(num_suits),
287 num_cards_(num_ranks * num_suits),
288 hand_size_(hand_size),
289 utils_(GinRummyUtils(num_ranks, num_suits, hand_size)),
290 stock_size_(num_ranks * num_suits),
291 deck_(num_ranks * num_suits, true) {
292 aohs_.reserve(kNumPlayers);
293 for (Player player = 0; player < kNumPlayers; ++player) {
294 std::vector<std::pair<absl::optional<Action>, std::string>> aoh;
295 aoh.push_back({{}, ObservationString(player)});
296 aohs_.push_back(open_spiel::ActionObservationHistory(player, aoh));
297 }

308void GinRummyState::DoApplyAction(Action action) {
309 Player current_player = CurrentPlayer();
310 switch (phase_) {
311 case Phase::kDeal:
312 ApplyDealAction(action);
313 break;
314 case Phase::kFirstUpcard:
315 ApplyFirstUpcardAction(action);
316 break;
317 case Phase::kDraw:
318 ApplyDrawAction(action);
319 break;
320 case Phase::kDiscard:
321 ApplyDiscardAction(action);
322 break;
323 case Phase::kKnock:
324 ApplyKnockAction(action);
325 break;
326 case Phase::kLayoff:
327 ApplyLayoffAction(action);
328 break;
329 case Phase::kWall:
330 ApplyWallAction(action);
331 break;
332 case Phase::kGameOver:
333 SpielFatalError("Cannot act in terminal states.");
334 break;
335 default:
336 SpielFatalError("Invalid game phase.");
337 }
338 for (Player player = 0; player < NumPlayers(); ++player) {
339 absl::optional<Action> a = {};
340 if (current_player == player) a = action;
341 aohs_[player].Extend(a, ObservationString(player));
342 }
343}

345void GinRummyState::ApplyDealAction(Action action) {
346 SPIEL_CHECK_TRUE(IsChanceNode());
347 SPIEL_CHECK_GE(action, 0);
348 SPIEL_CHECK_LT(action, num_cards_);
349 // Deal cards to player 0.
350 if (stock_size_ > num_cards_ - hand_size_) {
351 StockToHand(0, action);
352 } else if (stock_size_ > num_cards_ - 2 * hand_size_) {
353 // Next deal to player 1.
354 StockToHand(1, action);
355 } else if (stock_size_ == num_cards_ - 2 * hand_size_) {
356 // Set upcard.
357 StockToUpcard(action);
358 for (int i = 0; i < kNumPlayers; ++i) {
359 deadwood_[i] = utils_.MinDeadwood(hands_[i]);
360 }
361 // Initial upcard determines the knock card if playing Oklahoma.
362 if (oklahoma_) {
363 knock_card_ = utils_.CardValue(action);
364 // Ace upcard means we must play for gin!
365 if (knock_card_ == 1) knock_card_ = 0;
366 }
367 prev_player_ = kChancePlayerId;
368 // This implementation always starts the action with player 0.
369 cur_player_ = 0;
370 phase_ = Phase::kFirstUpcard;
371 } else {
372 // Previous player drew from stock, let's deal them a card.
373 StockToHand(prev_player_, action);
374 // Update deadwood total, used to see if knock is legal.
375 deadwood_[prev_player_] = utils_.MinDeadwood(hands_[prev_player_]);
376 cur_player_ = prev_player_;
377 prev_player_ = kChancePlayerId;
378 phase_ = Phase::kDiscard;
379 }
380}

382// Unique rules apply to the first upcard. If the first player to act does not
383// draw the upcard, the second player then has the option to pick it up. If
384// both players pass, the first player draws from the stock.
385void GinRummyState::ApplyFirstUpcardAction(Action action) {
386 if (action == kDrawUpcardAction) {
387 SPIEL_CHECK_TRUE(upcard_.has_value());
388 prev_upcard_ = upcard_;
389 UpcardToHand(cur_player_);
390 deadwood_[cur_player_] = utils_.MinDeadwood(hands_[cur_player_]);
391 prev_player_ = cur_player_;
392 phase_ = Phase::kDiscard;
393 } else if (action == kDrawStockAction) {
394 SPIEL_CHECK_TRUE(pass_on_first_upcard_[0] && pass_on_first_upcard_[1]);
395 prev_upcard_ = upcard_;
396 discard_pile_.push_back(upcard_.value());
397 upcard_ = absl::nullopt;
398 prev_player_ = cur_player_;
399 cur_player_ = kChancePlayerId;
400 phase_ = Phase::kDeal;
401 } else if (action == kPassAction) {
402 SPIEL_CHECK_FALSE(pass_on_first_upcard_[0] && pass_on_first_upcard_[1]);
403 pass_on_first_upcard_[cur_player_] = true;
404 prev_player_ = cur_player_;
405 cur_player_ = Opponent(prev_player_);
406 phase_ = Phase::kFirstUpcard;
407 } else {
408 SpielFatalError("Invalid Action");
409 }
410}

412void GinRummyState::ApplyDrawAction(Action action) {
413 if (action == kDrawUpcardAction) {
414 SPIEL_CHECK_TRUE(upcard_.has_value());
415 if (++num_draw_upcard_actions_ == kMaxNumDrawUpcardActions) {
416 phase_ = Phase::kGameOver;
417 return;
418 }
419 prev_upcard_ = upcard_;
420 UpcardToHand(cur_player_);
421 deadwood_[cur_player_] = utils_.MinDeadwood(hands_[cur_player_]);
422 prev_player_ = cur_player_;
423 phase_ = Phase::kDiscard;
424 } else if (action == kDrawStockAction) {
425 // When a player chooses to draw from stock the upcard is no
426 // longer in play and goes to the top of the discard pile.
427 prev_upcard_ = upcard_;
428 if (upcard_.has_value()) discard_pile_.push_back(upcard_.value());
429 upcard_ = absl::nullopt;
430 prev_player_ = cur_player_;
431 cur_player_ = kChancePlayerId;
432 phase_ = Phase::kDeal;
433 } else {
434 SpielFatalError("Invalid DrawAction");
435 }
436}

438void GinRummyState::ApplyDiscardAction(Action action) {
439 if (action == kKnockAction) {
440 SPIEL_CHECK_LE(deadwood_[cur_player_], knock_card_);
441 // The hand has been knocked, so now deadwood tracks the total card value.
442 for (int i = 0; i < kNumPlayers; ++i)
443 deadwood_[i] = utils_.TotalCardValue(hands_[i]);
444 knocked_[cur_player_] = true;
445 prev_player_ = cur_player_;
446 phase_ = Phase::kKnock;
447 } else {
448 SPIEL_CHECK_TRUE(absl::c_linear_search(hands_[cur_player_], action));
449 RemoveFromHand(cur_player_, action);
450 deadwood_[cur_player_] = utils_.MinDeadwood(hands_[cur_player_]);
451 upcard_ = action;
452 prev_player_ = cur_player_;
453 cur_player_ = Opponent(prev_player_);
454 if (upcard_ == prev_upcard_) {
455 if (repeated_move_) {
456 phase_ = Phase::kGameOver;
457 return;
458 } else {
459 repeated_move_ = true;
460 }
461 } else {
462 repeated_move_ = false;
463 }
464 if (stock_size_ == kWallStockSize) {
465 phase_ = Phase::kWall;
466 } else {
467 phase_ = Phase::kDraw;
468 }
469 }
470}

472void GinRummyState::ApplyKnockAction(Action action) {
473 // First the knocking player must discard.
474 if (hands_[cur_player_].size() == hand_size_ + 1) {
475 SPIEL_CHECK_TRUE(absl::c_linear_search(hands_[cur_player_], action));
476 RemoveFromHand(cur_player_, action);
477 discard_pile_.push_back(action);
478 deadwood_[cur_player_] = utils_.TotalCardValue(hands_[cur_player_]);
479 phase_ = Phase::kKnock;
480 } else if (action == kPassAction) {
481 // Here the pass action indicates knocking player is finished laying the
482 // hand. The player's deadwood is now final, and any cards not layed in a
483 // meld are counted towards the deadwood total.
484 deadwood_[cur_player_] = utils_.TotalCardValue(hands_[cur_player_]);
485 // Make sure the knocking player has completed a legal knock.
486 SPIEL_CHECK_LE(deadwood_[cur_player_], knock_card_);
487 // If deadwood equals 0 then the player has gin. The opponent is not
488 // allowed to lay off on gin.
489 if (deadwood_[cur_player_] == 0) finished_layoffs_ = true;
490 cur_player_ = Opponent(prev_player_);
491 phase_ = Phase::kLayoff;
492 } else {
493 // Knocking player must declare or "lay" melds, one action at a time.
494 SPIEL_CHECK_LE(action - kMeldActionBase, kNumMeldActions);
495 SPIEL_CHECK_GE(action - kMeldActionBase, 0);
496 layed_melds_[cur_player_].push_back(action - kMeldActionBase);
497 // Upon laying a meld the cards are removed from the player's hand.
498 for (int card : utils_.int_to_meld.at(action - kMeldActionBase)) {
499 RemoveFromHand(cur_player_, card);
500 }
501 deadwood_[cur_player_] = utils_.TotalCardValue(hands_[cur_player_]);
502 phase_ = Phase::kKnock;
503 }
504}
505
506void GinRummyState::ApplyLayoffAction(Action action) {
507 if (!finished_layoffs_) {
508 if (action == kPassAction) {
509 finished_layoffs_ = true;
510 phase_ = Phase::kLayoff;
511 } else {
512 SPIEL_CHECK_TRUE(absl::c_linear_search(hands_[cur_player_], action));
513 layoffs_.push_back(action);
514 RemoveFromHand(cur_player_, action);
515 deadwood_[cur_player_] = utils_.TotalCardValue(hands_[cur_player_]);
516 phase_ = Phase::kLayoff;
517 }
518 } else {
519 // Finished laying off individual cards, now lay melds.
520 if (action == kPassAction) {
521 deadwood_[cur_player_] = utils_.TotalCardValue(hands_[cur_player_]);
522 phase_ = Phase::kGameOver;
523 } else {
524 // Lay melds one action at a time.
525 SPIEL_CHECK_LE(action - kMeldActionBase, kNumMeldActions);
526 SPIEL_CHECK_GE(action - kMeldActionBase, 0);
527 layed_melds_[cur_player_].push_back(action - kMeldActionBase);
528 // Upon laying a meld the cards are removed from the player's hand.
529 for (int card : utils_.int_to_meld.at(action - kMeldActionBase))
530 RemoveFromHand(cur_player_, card);
531 deadwood_[cur_player_] = utils_.TotalCardValue(hands_[cur_player_]);
532 phase_ = Phase::kLayoff;
533 }
534 }
535}

537void GinRummyState::ApplyWallAction(Action action) {
538 if (action == kKnockAction) {
539 // When we've reached the wall, a knock automatically includes upcard.
540 UpcardToHand(cur_player_);
541 deadwood_[cur_player_] = utils_.MinDeadwood(hands_[cur_player_]);
542 // Make sure knock is legal.
543 SPIEL_CHECK_LE(deadwood_[cur_player_], knock_card_);
544 knocked_[cur_player_] = true;
545 prev_player_ = cur_player_;
546 phase_ = Phase::kKnock;
547 } else if (action == kPassAction) {
548 phase_ = Phase::kGameOver;
549 } else {
550 SpielFatalError("Invalid WallAction");
551 }

554std::vector<Action> GinRummyState::LegalActions() const {
555 switch (phase_) {
556 case Phase::kDeal:
557 return DealLegalActions();
558 case Phase::kFirstUpcard:
559 return FirstUpcardLegalActions();
560 case Phase::kDraw:
561 return DrawLegalActions();
562 case Phase::kDiscard:
563 return DiscardLegalActions();
564 case Phase::kKnock:
565 return KnockLegalActions();
566 case Phase::kLayoff:
567 return LayoffLegalActions();
568 case Phase::kWall:
569 return WallLegalActions();
570 default:
571 return {};
572 }
573}

575std::vector<Action> GinRummyState::DealLegalActions() const {
576 std::vector<Action> legal_actions;
577 for (int card = 0; card < num_cards_; ++card) {
578 if (deck_[card]) legal_actions.push_back(card);
579 }
580 return legal_actions;

583std::vector<Action> GinRummyState::FirstUpcardLegalActions() const {
584 std::vector<Action> legal_actions;
585 // If both players have passed then must draw from stock.
586 if (pass_on_first_upcard_[0] && pass_on_first_upcard_[1]) {
587 legal_actions.push_back(kDrawStockAction);
588 } else {
589 // Options are to draw upcard or pass to opponent.
590 legal_actions.push_back(kDrawUpcardAction);
591 legal_actions.push_back(kPassAction);
592 }
593 return legal_actions;
594}

596std::vector<Action> GinRummyState::DrawLegalActions() const {
597 std::vector<Action> legal_actions;
598 legal_actions.push_back(kDrawUpcardAction);
599 legal_actions.push_back(kDrawStockAction);
600 return legal_actions;
601}

603std::vector<Action> GinRummyState::DiscardLegalActions() const {
604 // All cards in hand are legal discards.
605 std::vector<Action> legal_actions(hands_[cur_player_].begin(),
606 hands_[cur_player_].end());
607 if (deadwood_[cur_player_] <= knock_card_) {
608 legal_actions.push_back(kKnockAction);
609 }
610 std::sort(legal_actions.begin(), legal_actions.end());
611 return legal_actions;
612}

614std::vector<Action> GinRummyState::KnockLegalActions() const {
615 std::vector<Action> legal_actions;
616 // After knocking, the player discards. This discard must not prevent
617 // the player from arranging the hand in such a way that the deadwood
618 // total is less than the knock card.
619 if (hands_[cur_player_].size() == hand_size_ + 1) {
620 for (int card : utils_.LegalDiscards(hands_[cur_player_], knock_card_)) {
621 legal_actions.push_back(card);
622 }
623 } else {
624 for (int meld_id : utils_.LegalMelds(hands_[cur_player_], knock_card_)) {
625 legal_actions.push_back(meld_id + kMeldActionBase);
626 }
627 // Must keep laying melds until remaining deadwood is less than knock card.
628 // Once that has been accomplished, the knocking player can pass.
629 if (utils_.TotalCardValue(hands_[cur_player_]) <= knock_card_) {
630 legal_actions.push_back(kPassAction);
631 }
632 }
633 std::sort(legal_actions.begin(), legal_actions.end());
634 return legal_actions;
635}
636
637std::vector<Action> GinRummyState::LayoffLegalActions() const {
638 std::vector<Action> legal_actions;
639 if (!finished_layoffs_) {
640 // Always have the option not to lay off any cards.
641 legal_actions.push_back(kPassAction);
642 std::vector<int> all_possible_layoffs =
643 utils_.AllLayoffs(layed_melds_[prev_player_], layoffs_);
644 for (int card : all_possible_layoffs) {
645 if (absl::c_linear_search(hands_[cur_player_], card)) {
646 legal_actions.push_back(card);
647 }
648 }
649 } else {
650 // After laying off individual cards, now the player lays melds.
651 // Always have the option not to declare any melds.
652 legal_actions.push_back(kPassAction);
653 // The non-knocking player does not have to arrange melds in a particular
654 // way to get under the knock card. Therefore we use kMaxPossibleDeadwood
655 // to ensure that all melds are legal.
656 for (int meld_id : utils_.LegalMelds(hands_[cur_player_],
657 kMaxPossibleDeadwood)) {
658 legal_actions.push_back(meld_id + kMeldActionBase);
659 }
660 }
661 std::sort(legal_actions.begin(), legal_actions.end());
662 return legal_actions;
663}
664
665std::vector<Action> GinRummyState::WallLegalActions() const {
666 std::vector<Action> legal_actions;
667 legal_actions.push_back(kPassAction);
668 // Player can either pass or knock (if legal).
669 int deadwood = utils_.MinDeadwood(hands_[cur_player_], upcard_);
670 if (deadwood <= knock_card_) {
671 legal_actions.push_back(kKnockAction);
672 }
673 return legal_actions;
674}

676std::vector<std::pair<Action, double>> GinRummyState::ChanceOutcomes() const {
677 SPIEL_CHECK_TRUE(IsChanceNode());
678 std::vector<std::pair<Action, double>> outcomes;
679 outcomes.reserve(stock_size_);
680 const double p = 1.0 / stock_size_;
681 for (int card = 0; card < num_cards_; ++card) {
682 // This card is still in the deck, prob is 1/stock_size_.
683 if (deck_[card]) outcomes.push_back({card, p});
684 }
685 return outcomes;
686}

688std::string GinRummyState::ActionToString(Player player, Action action) const {
689 if (player == kChancePlayerId) {
690 return absl::StrCat("Chance outcome: ", utils_.CardString(action));
691 } else {
692 std::string action_str;
693 if (action < num_cards_) {
694 action_str = utils_.CardString(action);
695 } else if (action == kDrawUpcardAction) {
696 action_str = "Draw upcard";
697 } else if (action == kDrawStockAction) {
698 action_str = "Draw stock";
699 } else if (action == kPassAction) {
700 action_str = "Pass";
701 } else if (action == kKnockAction) {
702 action_str = "Knock";
703 } else if (action < kMeldActionBase + kNumMeldActions) {
704 std::vector<int> meld = utils_.int_to_meld.at(action - kMeldActionBase);
705 std::vector<std::string> meld_str = utils_.CardIntsToCardStrings(meld);
706 action_str = absl::StrJoin(meld_str, "");
707 } else {
708 SpielFatalError(
709 absl::StrCat("Error in GinRummyState::ActionToString()."));
710 }
711 return absl::StrCat("Player: ", player, " Action: ", action_str);
712 }

758std::vector<double> GinRummyState::Returns() const {
759 if (!IsTerminal()) {
760 return std::vector<double>(kNumPlayers, 0.0);
761 }
762 std::vector<double> returns(kNumPlayers, 0.0);
763 // player 0 knocked
764 if (knocked_[0]) {
765 returns[0] = deadwood_[1] - deadwood_[0];
766 if (deadwood_[0] == 0) {
767 returns[0] += gin_bonus_;
768 }
769 if (returns[0] < 0) {
770 returns[0] -= undercut_bonus_;
771 }
772 returns[1] = -returns[0];
773 } else if (knocked_[1]) {
774 // player 1 knocked
775 returns[1] = deadwood_[0] - deadwood_[1];
776 if (deadwood_[1] == 0) {
777 returns[1] += gin_bonus_;
778 }
779 if (returns[1] < 0) {
780 returns[1] -= undercut_bonus_;
781 }
782 returns[0] = -returns[1];
783 }
784 // If neither player knocked both players get 0.
785 return returns;
786}
