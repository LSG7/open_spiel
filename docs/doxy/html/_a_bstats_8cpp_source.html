92void ABstats::IncrPos(
93 const ABCountType no,
94 const bool side,
95 const int depth)
96{
97 if (no < 0 || no >= AB_SIZE)
98 return;
99
100 ABplaces[no].list[depth]++;
101 ABplaces[no].sum++;
102 ABplaces[no].sumWeighted += depth;
103 ABplaces[no].sumCum++;
104 ABplaces[no].sumCumWeighted += depth;
105
106 const int iside = (side ? 1 : 0);
107
108 ABsides[iside].list[depth]++;
109 ABsides[iside].sum++;
110 ABsides[iside].sumWeighted += depth;
111 ABsides[iside].sumCum++;
112 ABsides[iside].sumCumWeighted += depth;
113
114}

150void ABstats::PrintStatsPosition(
151 ofstream& fout,
152 const int no,
153 const string& text,
154 const ABtracker& abt,
155 const ABtracker& divisor) const
156{
157 if (! abt.sumCum)
158 return;
159
160 fout << setw(2) << (no == -1 ? "" : to_string(no)) << " " <<
161 setw(20) << left << text <<
162 setw(9) << right << abt.sum <<
163 setw(6) << setprecision(1) << fixed <<
164 100. * abt.sum / static_cast<double>(divisor.sum);
165
166 if (abt.sum)
167 fout << setw(6) << setprecision(1) << fixed <<
168 abt.sumWeighted / static_cast<double>(abt.sum);
169 else
170 fout << setw(6) << "";
171
172 fout << setw(9) << abt.sumCum <<
173 setw(6) << setprecision(1) << fixed <<
174 100. * abt.sumCum / static_cast<double>(divisor.sumCum) <<
175 setw(6) << setprecision(1) << fixed <<
176 abt.sumCumWeighted / static_cast<double>(abt.sumCum) << "\n";
177}

193void ABstats::PrintStatsDepth(
194 ofstream& fout,
195 const int depth,
196 const int cum) const
197{
198 fout << setw(5) << depth <<
199 setw(7) << ABnodes.list[depth] <<
200 setw(7) << ABnodesCum.list[depth] <<
201 setw(6) << setprecision(1) << fixed <<
202 100. * ABnodesCum.list[depth] /
203 static_cast<double>(ABnodesCum.sumCum) <<
204 setw(6) << setprecision(1) << fixed <<
205 100. * cum / static_cast<double>(ABnodesCum.sumCum);
206
207 // "Branching factor" from end of one trick to end of
208 // the previous trick.
209 if ((depth % 4 == 1) &&
210 (depth < DDS_MAXDEPTH - 4) &&
211 (ABnodesCum.list[depth + 4] > 0))
212 fout << setw(6) << setprecision(2) << fixed <<
213 ABnodesCum.list[depth] /
214 static_cast<double>(ABnodesCum.list[depth + 4]);
215 fout << "\n";
216}

302void ABstats::PrintStats(ofstream& fout)
303{
304 ABtracker ABsidesSum;
305 ABsidesSum.sum = ABsides[1].sum + ABsides[0].sum;
306 ABsidesSum.sumCum = ABsides[1].sumCum + ABsides[0].sumCum;
307
308 if (ABsidesSum.sum)
309 {
310 // First table: By side and position.
311
312 ABstats::PrintHeaderPosition(fout);
313
314 ABstats::PrintStatsPosition(fout, -1, "Side1", ABsides[1], ABsidesSum);
315 ABstats::PrintStatsPosition(fout, -1, "Side0", ABsides[0], ABsidesSum);
316 fout << "\n";
317
318 for (int p = 0; p < AB_SIZE; p++)
319 ABstats::PrintStatsPosition(fout, p, name[p], ABplaces[p], ABsidesSum);
320 fout << "\n";
321 }
322
323 ABstats::PrintHeaderDepth(fout);
324
325
326 // Second table: By depth.
327
328 int c = 0;
329 for (int d = DDS_MAXDEPTH - 1; d >= 0; d--)
330 {
331 if (ABnodesCum.list[d] == 0)
332 continue;
333
334 c += ABnodesCum.list[d];
335 ABstats::PrintStatsDepth(fout, d, c);
336 }
337
338 ABstats::PrintAverageDepth(fout, ABsidesSum);
339
340
341#ifdef DDS_AB_DETAILS
342 // Third table: All the detail.
343
344 ABstats::PrintHeaderDetail(fout);
345
346 for (int d = DDS_MAXDEPTH - 1; d >= 0; d--)
347 ABstats::PrintStatsDetail(fout, d);
348
349 ABstats::PrintSumDetail(fout);
350#endif
351}
