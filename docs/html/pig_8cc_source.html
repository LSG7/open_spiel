86 bool PigState::IsTerminal() const {
87  if (total_moves_ >= horizon_) {
88  return true;
89  }
90 
91  for (auto p = Player{0}; p < num_players_; p++) {
92  if (scores_[p] >= win_score_) {
93  return true;
94  }
95  }
96  return false;

99 std::vector<double> PigState::Returns() const {
100  if (!IsTerminal()) {
101  return std::vector<double>(num_players_, 0.0);
102  }
103 
104  // For (n>2)-player games, must keep it zero-sum.
105  std::vector<double> returns(num_players_, -1.0 / (num_players_ - 1));
106 
107  for (auto player = Player{0}; player < num_players_; ++player) {
108  if (scores_[player] >= win_score_) {
109  returns[player] = 1.0;
110  return returns;
111  }
112  }
113 
114  // Nobody has won? (e.g. over horizon length.) Then everyone gets 0.
115  return std::vector<double>(num_players_, 0.0);
116 }

129 void PigState::ObservationTensor(Player player,
130  absl::Span<float> values) const {
131  SPIEL_CHECK_GE(player, 0);
132  SPIEL_CHECK_LT(player, num_players_);
133 
134  // One extra bin for when value is >= max.
135  // So for win_score_ 100, binSize = 1 -> 0, 1, ..., 99, >= 100.
136  int num_bins = (win_score_ / kBinSize) + 1;
137 
138  // One-hot encoding: turn total (#bin) followed by p1, p2, ...
139  // Treat `values` as a 2-d tensor.
140  TensorView<2> view(values, {1 + num_players_, num_bins}, true);
141 
142  // One-hot encoding:
143  // - turn total (#bins)
144  // - player 0 (#bins)
145  // - player 1 (#bins)
146  // .
147  // .
148  // .
149 
150  // turn total
151  view[{0, std::min(turn_total_ / kBinSize, num_bins - 1)}] = 1;
152 
153  for (auto p = Player{0}; p < num_players_; p++) {
154  // score of each player
155  view[{1 + p, std::min(scores_[p] / kBinSize, num_bins - 1)}] = 1;
156  }
157 }

177 void PigState::DoApplyAction(Action move) {
178  // For decision node: 0 means roll, 1 means stop.
179  // For chance node: outcome of the dice (x-1, piglet: [x != 1]).
180  if (cur_player_ >= 0 && move == kRoll) {
181  // Player roll -> chance node.
182  cur_player_ = kChancePlayerId;
183  total_moves_++;
184  } else if (cur_player_ >= 0 && move == kStop) {
185  // Player stops. Take turn total and pass to next player.
186  scores_[turn_player_] += turn_total_;
187  turn_total_ = 0;
188  turn_player_ = NextPlayerRoundRobin(turn_player_, num_players_);
189  cur_player_ = turn_player_;
190  total_moves_++;
191  } else if (IsChanceNode()) {
192  // Resolve chance node outcome. If 1, reset turn total and change players;
193  // else, add to total and keep going.
194  if (move == 0) {
195  // Reset turn total and loses turn!
196  turn_total_ = 0;
197  turn_player_ = NextPlayerRoundRobin(turn_player_, num_players_);
198  cur_player_ = turn_player_;
199  } else {
200  // Add to the turn total.
201  turn_total_ += (piglet_ ? 1 : move + 1);
202  cur_player_ = turn_player_;
203  }
204  } else {
205  SpielFatalError(absl::StrCat("Move ", move, " is invalid."));
206  }
207 }

223 std::vector<std::pair<Action, double>> PigState::ChanceOutcomes() const {
224  SPIEL_CHECK_TRUE(IsChanceNode());
225  std::vector<std::pair<Action, double>> outcomes;
226 
227  if (piglet_) {
228  // Chance outcomes are labelled 0 or 1, corresponding to rolling 1 or not 1
229  // respectively
230  outcomes.reserve(2);
231  outcomes.push_back(std::make_pair(0, 1.0 / dice_outcomes_));
232  outcomes.push_back(std::make_pair(1, 1.0 - (1.0 / dice_outcomes_)));
233  } else {
234  // Chance outcomes are labelled 0+, corresponding to rolling 1+x.
235  outcomes.reserve(dice_outcomes_);
236  for (int i = 0; i < dice_outcomes_; i++) {
237  outcomes.push_back(std::make_pair(i, 1.0 / dice_outcomes_));
238  }
239  }
240 
241  return outcomes;
242 }
