33def argmax(random, z):
34 """Returns argmax of flattened z with ties split randomly.
35
36 Args:
37 random: Random number generator, e.g., np.random.RandomState()
38 z: np.array
39 Returns:
40 integer representing index of argmax
41 """
42 inds = np.arange(z.size)
43 random.shuffle(inds)
44 z_shuffled = z[inds]
45 ind_max = np.argmax(z_shuffled)

49def pt_reduce(payoff_tensor, strats, remove_players):
50 """Computes possible payoffs for remove_players with others' strats fixed.
51
52 This is equivalent to the Jacobian of the payoff w.r.t. remove_players:
53 sum_{a...z} A_k * x_1a * ... * x_nz for player k.
54 Args:
55 payoff_tensor: a single player k's payoff tensor, i.e.,
56 a num action x ... x num action (num player) np.array
57 strats: list of distributions over strategies for each player
58 remove_players: players to NOT sum over in expectation
59 Returns:
60 payoff tensor of shape: num_action x ... x num_action,
61 num_action for each player in remove_players
62 """
63 result = np.copy(payoff_tensor)
64 result_dims = list(range(len(result.shape)))
65 other_player_idxs = list(result_dims)
66 for remove_player in remove_players:
67 other_player_idxs.remove(remove_player)
68 for other_player_idx in other_player_idxs:
69 new_result_dims = list(result_dims)
70 new_result_dims.remove(other_player_idx)
71 result = np.einsum(result, result_dims, strats[other_player_idx],
72 [other_player_idx], new_result_dims)
73 result_dims = new_result_dims
74 return result

77def isnan(x):
78 """Checks for NaN's in nested objects."""
79 if isinstance(x, float):
80 return np.isnan(x)
81 elif isinstance(x, int):
82 return np.isnan(x)
83 elif isinstance(x, np.ndarray):
84 return np.any(np.isnan(x))
85 elif isinstance(x, list):
86 return np.any([isnan(xi) for xi in x])
87 elif isinstance(x, tuple):
88 return np.any([isnan(xi) for xi in x])
89 elif isinstance(x, dict):
90 return np.any([isnan(xi) for xi in x.values()])
91 else:
92 typ = repr(type(x))
93 err_string = 'type(x)={:s} not recognized when checking for NaN'.format(typ)
94 raise NotImplementedError(err_string)
