22def compute_shapley_values(
23 game: coalitional_game.CoalitionalGame,
24) -> np.ndarray:
25 """Compute the Shapley values exactly.
26
27 Uses Eq (2) of Mitchell et al. "Sampling Permutations for Shapley Value
28 Estimation". https://people.math.sc.edu/cooper/shapley.pdf
29
30 Args:
31 game: the game to compute Shapley values for.
32
33 Returns:
34 shapley_values: a numpy array of Shapley values per player.
35 """
36
37 shapley_values_sum = np.zeros(game.num_players(), dtype=float)
38 coalition = np.zeros(game.num_players(), dtype=int)
39 empty_coalition_value = game.coalition_value(coalition)
40 num_perms = 0
41 for perm_tup in itertools.permutations(range(game.num_players())):
42 perm = list(perm_tup)
43 value_with = empty_coalition_value
44 coalition.fill(0)
45 for idx in range(game.num_players()):
46 value_without = value_with # re-use the one computed from the last iter
47 i = perm[idx]
48 coalition[i] = 1
49 value_with = game.coalition_value(coalition)
50 shapley_values_sum[i] += value_with - value_without
51 num_perms += 1
52 return shapley_values_sum / num_perms

55def compute_approximate_shapley_values(
56 game: coalitional_game.CoalitionalGame,
57 num_samples: int,
58) -> np.ndarray:
59 """Compute the Shapley values using Monte Carlo estimation.
60
61 Specifically, applies the implementation described in Section 2.3 of Mitchell
62 et al. "Sampling Permutations for Shapley Value Estimation".
63 https://people.math.sc.edu/cooper/shapley.pdf
64
65 Args:
66 game: the game to compute Shapley values for.
67 num_samples: number of permutations to sample
68
69 Returns:
70 shapley_values: a numpy array of Shapley values per player.
71 """
72
73 shapley_values_sum = np.zeros(game.num_players(), dtype=float)
74 coalition = np.zeros(game.num_players(), dtype=int)
75 empty_coalition_value = game.coalition_value(coalition)
76 for _ in range(num_samples):
77 perm = np.random.permutation(game.num_players())
78 value_with = empty_coalition_value
79 coalition.fill(0)
80 for idx in range(game.num_players()):
81 value_without = value_with # re-use the one computed from the last iter
82 i = perm[idx]
83 coalition[i] = 1
84 value_with = game.coalition_value(coalition)
85 shapley_values_sum[i] += value_with - value_without
86 return shapley_values_sum / num_samples
