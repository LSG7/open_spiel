118 void TestLegalActions() {
119  auto game = LoadGame(
120  "mfg_crowd_modelling_2d(size=5,horizon=10,forbidden_states=[0|0;0|1;1|0]"
121  ",initial_distribution=[1|1],initial_distribution_value=[1.0])");
122  auto state = game->NewInitialState();
123  SPIEL_CHECK_EQ(state->CurrentPlayer(), kChancePlayerId);
124  // Legal action will be the state in the initial distribution.
125  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({6}));
126  state->ApplyAction(6);
127  // Legal actions are moving right, down or no movement.
128  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({2, 3, 4}));
129  // Go right.
130  state->ApplyAction(3);
131  // Chance node. No forbidden states around and all actions are legal.
132  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({0, 1, 2, 3, 4}));
133  // Go left.
134  state->ApplyAction(1);
135  // Mean field node; legal actions will be empty.
136  SPIEL_CHECK_TRUE(state->LegalActions().empty());
137  std::vector<double> dist(25);
138  state->UpdateDistribution(dist);
139  // Back to starting point.
140  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({2, 3, 4}));
141  // Stay in the same position.
142  state->ApplyAction(2);
143  // Chance node. The legal actions should be the same.
144  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({2, 3, 4}));
145 }

147 void TestNoiseIntensity() {
148  // Same as the setting in TestLegalActions() test above, except the noise
149  // intensity.
150  auto game = LoadGame(
151  "mfg_crowd_modelling_2d(size=5,horizon=10,forbidden_states=[0|0;0|1;1|0]"
152  ",initial_distribution=[1|1],initial_distribution_value=[1.0]"
153  ",noise_intensity=0.5)");
154  auto state = game->NewInitialState();
155  SPIEL_CHECK_EQ(state->CurrentPlayer(), kChancePlayerId);
156  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({6}));
157  state->ApplyAction(6);
158  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({2, 3, 4}));
159  state->ApplyAction(3);
160  // Now we are at a chance node.
161  SPIEL_CHECK_EQ(state->LegalActions(), std::vector<Action>({0, 1, 2, 3, 4}));
162  // Neutral action should have a probability of 0.5+0.1 and others 0.1, i.e.
163  // 0.5 / 5.
164  ActionsAndProbs expected_outcomes(
165  {{0, 0.1}, {1, 0.1}, {2, 0.6}, {3, 0.1}, {4, 0.1}});
166  SPIEL_CHECK_EQ(state->ChanceOutcomes(), expected_outcomes);
167 }
