56 std::vector<Action> AFCEState::LegalActions() const {
57  SPIEL_CHECK_FALSE(IsSimultaneousNode());
58 
59  if (IsTerminal()) {
60  return {};
61  } else if (IsChanceNode()) {
62  return LegalChanceOutcomes();
63  }
64 
65  if (!HasDefected(CurrentPlayer())) {
66  // If the player has not defected, they are unrestricted.
67  return state_->LegalActions();
68  } else {
69  Action recommended_action = CurRecommendation();
70 
71  // Check that it's a legal recommendation.
72  std::vector<Action> legal_actions = state_->LegalActions();
73  SPIEL_CHECK_TRUE(std::find(legal_actions.begin(), legal_actions.end(),
74  recommended_action) != legal_actions.end());
75 
76  // It is the only allowed action once this player has deviated.
77  return {recommended_action};
78  }
79 }

81 std::string AFCEState::InformationStateString(Player player) const {
82  // should look like <infoset string> <delimiter> <defected? true | false>
83  // <recommendation sequence, (including current recommendation)>
84  // <defection infoset, if available>
85  SPIEL_CHECK_FALSE(IsChanceNode());
86  std::string rec_str = absl::StrJoin(recommendation_seq_[player], ",");
87 
88  // Always add the recommendation.
89  absl::StrAppend(&rec_str, ",", CurRecommendation());
90 
91  std::string infoset_str = state_->InformationStateString(player);
92  SPIEL_CHECK_EQ(infoset_str.find(config_.recommendation_delimiter),
93  std::string::npos);
94 
95  // Note that there are two infosets for a defection:
96  // - The first one where a player chooses the defect action. Here, defected?
97  // is false and there is no defection infoset set yet
98  // - Directly after defection by the same player. Here defected? is true but
99  // the infoset is not yet set.
100  // After the defection, defected? is set to to true and the defection infoset
101  // is included in the infoset string.
102  return absl::StrCat(infoset_str, config_.recommendation_delimiter,
103  HasDefected(player) ? "true " : "false ", rec_str,
104  defection_infoset_[player].has_value()
105  ? defection_infoset_[player].value()
106  : "");
107 }

111  "%s\nCur player: %i\nRec index %i\nDefected %s", state_->ToString(),

118  "\n");
119  }
120  for (Player p = 0; p < state_->NumPlayers(); ++p) {
121  absl::StrAppend(&state_str, "\nPlayer ", p, " recommendation seq: ",
122  absl::StrJoin(recommendation_seq_[p], ","), "\n");
123  }
124  return state_str;

133 Action AFCEState::CurRecommendation() const {
134  SPIEL_CHECK_GE(rec_index_, 0);
135  SPIEL_CHECK_LT(rec_index_, mu_.size());
136  ActionsAndProbs actions_and_probs =

138  Action rec_action = GetAction(actions_and_probs);

143 void AFCEState::DoApplyAction(Action action_id) {
144  if (rec_index_ < 0) {
145  // Pick the joint policy which will provide recommendations.
146  rec_index_ = action_id;
147  SPIEL_CHECK_GE(rec_index_, 0);
148  SPIEL_CHECK_LT(rec_index_, mu_.size());
149  } else if (state_->IsChanceNode()) {
150  // Regular chance node
151  state_->ApplyAction(action_id);
152  } else {
153  // Check for defection at this point.
154  const Action recommendation = CurRecommendation();
155 
156  Player cur_player = CurrentPlayer();
157  SPIEL_CHECK_GE(cur_player, 0);
158  SPIEL_CHECK_LT(cur_player, game_->NumPlayers());
159 
160  if (HasDefected(cur_player)) {
161  // The recommendation should be the only legal action.
162  SPIEL_CHECK_EQ(action_id, recommendation);
163  }
164 
165  // In this auxiliary game the agent always gets a recommendation.
166  recommendation_seq_[cur_player].push_back(recommendation);
167 
168  if (action_id != recommendation) {
169  // Cannot defect more than once.
170  SPIEL_CHECK_TRUE(!HasDefected(cur_player));
171  defected_[cur_player] = 1;
172  defection_infoset_[cur_player] =
173  state_->InformationStateString(cur_player);
174  }
175 
176  state_->ApplyAction(action_id);
177  }
178 }

180 ActionsAndProbs AFCETabularPolicy::GetStatePolicy(const State& state) const {
181  // The best response code has to have a policy defined everywhere when it
182  // builds its initial tree. For the fixed policies, the players will not
183  // defect, so we define a uniform policy in the regions where players have
184  // defected (which will not affect the best responding player, since the
185  // opponents will never reach these regions).
186  const auto* AFCE_state = dynamic_cast<const AFCEState*>(&state);
187  SPIEL_CHECK_TRUE(AFCE_state != nullptr);
188  if (AFCE_state->HasDefected(state.CurrentPlayer())) {
189  return UniformStatePolicy(state);
190  }
191 
192  // Otherwise: simply returns a fixed policy with prob 1 on the recommended
193  // action (extrapolated from the information state string) and 0 on the
194  // others.
195  std::string info_state = state.InformationStateString();
196  const size_t idx = info_state.find(config_.recommendation_delimiter);
197  SPIEL_CHECK_NE(idx, std::string::npos);
198 
199  // String looks like ... <delimiter> <true | false> <recommendation sequence>
200  std::vector<std::string> suffix_parts = absl::StrSplit(
201  info_state.substr(idx + config_.recommendation_delimiter.length()), ' ');
202  SPIEL_CHECK_TRUE(suffix_parts[0] == "true" || suffix_parts[0] == "false");
203  std::vector<std::string> rec_seq = absl::StrSplit(suffix_parts[1], ',');
204  SPIEL_CHECK_GE(rec_seq.size(), 1);
205  Action rec_action;
206  ActionsAndProbs state_policy;
207  std::vector<Action> legal_actions = state.LegalActions();
208  state_policy.reserve(legal_actions.size());
209  SPIEL_CHECK_TRUE(absl::SimpleAtoi(rec_seq.back(), &rec_action));
210  for (Action action : legal_actions) {
211  state_policy.push_back({action, action == rec_action ? 1.0 : 0.0});
212  }
213  return state_policy;
214 }
