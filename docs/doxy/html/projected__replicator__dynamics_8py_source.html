26def _partial_multi_dot(player_payoff_tensor, strategies, index_avoided):
27 """Computes a generalized dot product avoiding one dimension.
28
29 This is used to directly get the expected return of a given action, given
30 other players' strategies, for the player indexed by index_avoided.
31 Note that the numpy.dot function is used to compute this product, as it ended
32 up being (Slightly) faster in performance tests than np.tensordot. Using the
33 reduce function proved slower for both np.dot and np.tensordot.
34
35 Args:
36 player_payoff_tensor: payoff tensor for player[index_avoided], of dimension
37 (dim(vector[0]), dim(vector[1]), ..., dim(vector[-1])).
38 strategies: Meta strategy probabilities for each player.
39 index_avoided: Player for which we do not compute the dot product.
40
41 Returns:
42 Vector of expected returns for each action of player [the player indexed by
43 index_avoided].
44 """
45 new_axis_order = [index_avoided] + [
46 i for i in range(len(strategies)) if (i != index_avoided)
47 ]
48 accumulator = np.transpose(player_payoff_tensor, new_axis_order)
49 for i in range(len(strategies) - 1, -1, -1):
50 if i != index_avoided:
51 accumulator = np.dot(accumulator, strategies[i])
52 return accumulator

55def _project_distribution(updated_strategy, gamma):
56 """Projects the distribution in updated_x to have minimal probabilities.
57
58 Minimal probabilities are set as gamma, and the probabilities are then
59 renormalized to sum to 1.
60
61 Args:
62 updated_strategy: New distribution value after being updated by update rule.
63 gamma: minimal probability value when divided by number of actions.
64
65 Returns:
66 Projected distribution.
67 """
68 # Epsilon approximation of L2-norm projection onto the Delta_gamma space.
69 updated_strategy[updated_strategy < gamma] = gamma
70 updated_strategy = updated_strategy / np.sum(updated_strategy)
71 return updated_strategy

93def _simplex_projection(updated_strategy, gamma=0.0):
94 """Project updated_strategy on the closest point in L2-norm on gamma-simplex.
95
96 Based on: https://eng.ucmerced.edu/people/wwang5/papers/SimplexProj.pdf
97
98 Args:
99 updated_strategy: New distribution value after being updated by update rule.
100 gamma: minimal probability value when divided by number of actions.
101
102 Returns:
103 Projected distribution
104
105 Algorithm description:
106 It aims to find a scalar lam to be substracted by each dimension of v
107 with the restriction that the resulted quantity should lie in [gamma, 1]
108 until the resulted vector summed up to 1
109 Example: [0.4, 0.7, 0.6], 0.2 -- > find lam=0.25
110 --> [max(0.4-0.25, 0.2), max(0.7-0.25, 0.2), max(0.6-0.25, 0.2)]
111 --> [0.2, 0.45, 0.35]
112 """
113
114 n = len(updated_strategy)
115 idx = np.arange(1, n + 1)
116 u = np.sort(updated_strategy)[::-1]
117 u_tmp = (1 - np.cumsum(u) - (n - idx) * gamma) / idx
118 rho = np.searchsorted(u + u_tmp <= gamma, True)
119 return np.maximum(updated_strategy + u_tmp[rho - 1], gamma)

122def _projected_replicator_dynamics_step(payoff_tensors, strategies, dt, gamma,
123 use_approx=False):
124 """Does one step of the projected replicator dynamics algorithm.
125
126 Args:
127 payoff_tensors: List of payoff tensors for each player.
128 strategies: List of the strategies used by each player.
129 dt: Update amplitude term.
130 gamma: Minimum exploratory probability term.
131 use_approx: use approximate simplex projection.
132
133 Returns:
134 A list of updated strategies for each player.
135 """
136
137 # TODO(author4): Investigate whether this update could be fully vectorized.
138 new_strategies = []
139 for player in range(len(payoff_tensors)):
140 current_payoff_tensor = payoff_tensors[player]
141 current_strategy = strategies[player]
142
143 values_per_strategy = _partial_multi_dot(current_payoff_tensor, strategies,
144 player)
145 average_return = np.dot(values_per_strategy, current_strategy)
146 delta = current_strategy * (values_per_strategy - average_return)
147
148 updated_strategy = current_strategy + dt * delta
149 updated_strategy = (
150 _approx_simplex_projection(updated_strategy, gamma) if use_approx
151 else _simplex_projection(updated_strategy, gamma))
152 new_strategies.append(updated_strategy)
153 return new_strategies

156def projected_replicator_dynamics(payoff_tensors,
157 prd_initial_strategies=None,
158 prd_iterations=int(1e5),
159 prd_dt=1e-3,
160 prd_gamma=1e-6,
161 average_over_last_n_strategies=None,
162 use_approx=False,
163 **unused_kwargs):
164 """The Projected Replicator Dynamics algorithm.
165
166 Args:
167 payoff_tensors: List of payoff tensors for each player.
168 prd_initial_strategies: Initial list of the strategies used by each player,
169 if any. Could be used to speed up the search by providing a good initial
170 solution.
171 prd_iterations: Number of algorithmic steps to take before returning an
172 answer.
173 prd_dt: Update amplitude term.
174 prd_gamma: Minimum exploratory probability term.
175 average_over_last_n_strategies: Running average window size for average
176 policy computation. If None, use the whole trajectory.
177 use_approx: use the approximate simplex projection.
178 **unused_kwargs: Convenient way of exposing an API compatible with other
179 methods with possibly different arguments.
180
181 Returns:
182 PRD-computed strategies.
183 """
184 number_players = len(payoff_tensors)
185 # Number of actions available to each player.
186 action_space_shapes = payoff_tensors[0].shape
187
188 # If no initial starting position is given, start with uniform probabilities.
189 new_strategies = prd_initial_strategies or [
190 np.ones(action_space_shapes[k]) / action_space_shapes[k]
191 for k in range(number_players)
192 ]
193
194 averager = nfg_utils.StrategyAverager(number_players, action_space_shapes,
195 average_over_last_n_strategies)
196 averager.append(new_strategies)
197
198 for _ in range(prd_iterations):
199 new_strategies = _projected_replicator_dynamics_step(
200 payoff_tensors, new_strategies, prd_dt, prd_gamma, use_approx)
201 averager.append(new_strategies)
202 return averager.average_strategies()
