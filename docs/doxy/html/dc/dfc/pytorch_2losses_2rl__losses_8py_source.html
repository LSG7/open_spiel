15"""Reinforcement learning loss functions.
16
17All the loss functions implemented here compute the loss for the policy (actor).
18The critic loss functions are typically regression loss are omitted for their
19simplicity.
20
21For the batch QPG, RM and RPG loss, please refer to the paper:
22https://papers.nips.cc/paper/7602-actor-critic-policy-optimization-in-partially-observable-multiagent-environments.pdf
23
24The BatchA2C loss uses code from the `TRFL` library:
25https://github.com/deepmind/trfl/blob/master/trfl/discrete_policy_gradient_ops.py
26"""

32def _assert_rank_and_shape_compatibility(tensors, rank):
33 if not tensors:
34 raise ValueError("List of tensors cannot be empty")
35
36 tmp_shape = tensors[0].shape
37 for tensor in tensors:
38 if tensor.ndim != rank:
39 raise ValueError("Shape %s must have rank %d" % (tensor.ndim, rank))
40 if tensor.shape != tmp_shape:
41 raise ValueError("Shapes %s and %s are not compatible" %
42 (tensor.shape, tmp_shape))

45def thresholded(logits, regrets, threshold=2.0):
46 """Zeros out `regrets` where `logits` are too negative or too large."""
47 can_decrease = logits.gt(-threshold).float()
48 can_increase = logits.lt(threshold).float()
49 regrets_negative = regrets.clamp(max=0.0)
50 regrets_positive = regrets.clamp(min=0.0)
51 return can_decrease * regrets_negative + can_increase * regrets_positive
52
53
54def compute_baseline(policy, action_values):
55 # V = pi * Q, backprop through pi but not Q.
56 return torch.sum(torch.mul(policy, action_values.detach()), dim=1)
57
58
59def compute_regrets(policy_logits, action_values):
60 """Compute regrets using pi and Q."""
61 # Compute regret.
62 policy = F.softmax(policy_logits, dim=1)
63 # Avoid computing gradients for action_values.
64 action_values = action_values.detach()
65
66 baseline = compute_baseline(policy, action_values)
67
68 regrets = torch.sum(
69 F.relu(action_values - torch.unsqueeze(baseline, 1)), dim=1)
70
71 return regrets

74def compute_advantages(policy_logits,
75 action_values,
76 use_relu=False,
77 threshold_fn=None):
78 """Compute advantages using pi and Q."""
79 # Compute advantage.
80 policy = F.softmax(policy_logits, dim=1)
81 # Avoid computing gradients for action_values.
82 action_values = action_values.detach()
83
84 baseline = compute_baseline(policy, action_values)
85
86 advantages = action_values - torch.unsqueeze(baseline, 1)
87 if use_relu:
88 advantages = F.relu(advantages)
89
90 if threshold_fn:
91 # Compute thresholded advanteges weighted by policy logits for NeuRD.
92 policy_logits = policy_logits - policy_logits.mean(-1, keepdim=True)
93 advantages = threshold_fn(policy_logits, advantages)
94 policy_advantages = -torch.mul(policy_logits, advantages.detach())
95 else:
96 # Compute advantage weighted by policy.
97 policy_advantages = -torch.mul(policy, advantages.detach())
98 return torch.sum(policy_advantages, dim=1)

101def compute_a2c_loss(policy_logits, actions, advantages):
102 cross_entropy = F.cross_entropy(policy_logits, actions, reduction="none")
103 advantages = advantages.detach()
104 if advantages.ndim != cross_entropy.ndim:
105 raise ValueError("Shapes %s and %s are not compatible" %
106 (advantages.ndim, cross_entropy.ndim))
107 return torch.mul(cross_entropy, advantages)
108
109
110def compute_entropy(policy_logits):
111 return torch.sum(
112 -F.softmax(policy_logits, dim=1) * F.log_softmax(policy_logits, dim=1),
113 dim=-1)

116class BatchQPGLoss(object):
117 """Defines the batch QPG loss op."""
118
119 def __init__(self, entropy_cost=None, name="batch_qpg_loss"):
120 self._entropy_cost = entropy_cost
121 self._name = name
122
123 def loss(self, policy_logits, action_values):
124 """Constructs a PyTorch Crierion that computes the QPG loss for batches.
125
126 Args:
127 policy_logits: `B x A` tensor corresponding to policy logits.
128 action_values: `B x A` tensor corresponding to Q-values.
129
130 Returns:
131 loss: A 0-D `float` tensor corresponding the loss.
132 """
133 _assert_rank_and_shape_compatibility([policy_logits, action_values], 2)
134 advantages = compute_advantages(policy_logits, action_values)
135 _assert_rank_and_shape_compatibility([advantages], 1)
136 total_adv = torch.mean(advantages, dim=0)
137
138 total_loss = total_adv
139 if self._entropy_cost:
140 policy_entropy = torch.mean(compute_entropy(policy_logits))
141 entropy_loss = torch.mul(float(self._entropy_cost), policy_entropy)
142 total_loss = torch.add(total_loss, entropy_loss)
143
144 return total_loss

245 self._entropy_cost = entropy_cost

248 def loss(self, policy_logits, baseline, actions, returns):
249 """Constructs a PyTorch Crierion that computes the A2C loss for batches.
250
251 Args:
252 policy_logits: `B x A` tensor corresponding to policy logits.
253 baseline: `B` tensor corresponding to baseline (V-values).
254 actions: `B` tensor corresponding to actions taken.
255 returns: `B` tensor corresponds to returns accumulated.
256
257 Returns:

259 """

261 _assert_rank_and_shape_compatibility([baseline, actions, returns], 1)
262 advantages = returns - baseline
