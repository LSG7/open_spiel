96 DealtCards DealCards(int num_players, int seed) {
97  std::vector<Action> cards(54);
98  std::iota(cards.begin(), cards.end(), 0);
99  Shuffle(&cards, std::mt19937(seed));
100 
101  // first six cards are talon
102  auto begin = cards.begin();
103  auto end = begin + 6;
104  std::vector<Action> talon(begin, end);
105 
106  // deal the rest of the cards to players
107  int num_cards_per_player = 48 / num_players;
108  std::vector<std::vector<Action>> players_cards;
109  players_cards.reserve(num_players);
110 
111  std::advance(begin, 6);
112  for (int i = 0; i < num_players; i++) {
113  std::advance(end, num_cards_per_player);
114  std::vector<Action> player_cards(begin, end);
115  // player's cards are sorted since legal actions need to be returned in
116  // ascending order
117  std::sort(player_cards.begin(), player_cards.end());
118  players_cards.push_back(player_cards);
119  std::advance(begin, num_cards_per_player);
120  }
121 
122  return {talon, players_cards};
123 }

125 void Shuffle(std::vector<Action>* actions, std::mt19937&& rng) {
126  for (int i = actions->size() - 1; i > 0; i--) {
127  std::swap(actions->at(i), actions->at(rng() % (i + 1)));
128  }
129 }

131 int CardPoints(const std::vector<Action>& actions,
132  const std::array<Card, 54>& deck) {
133  // counting is done in batches of three (for every batch we sum up points from
134  // three cards and subtract 2 points, if the last batch has less than three
135  // cards we subtract 1 point), mathematically, this is equevalent to
136  // subtracting 2/3 from each card
137  float points = 0;
138  for (auto const& action : actions) {
139  points += deck.at(action).points;
140  }
141  points -= actions.size() * 0.666f;
142  return static_cast<int>(round(points));
143 }
