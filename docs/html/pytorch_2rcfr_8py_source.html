15 """Regression counterfactual regret minimization (RCFR) [Waugh et al., 2015; Morrill, 2016].
16 
17 In contrast to (tabular) counterfactual regret minimization (CFR)
18 [Zinkevich et al., 2007], RCFR replaces the table of regrets that generate the
19 current policy profile with a profile of regression models. The average
20 policy is still tracked exactly with a full game-size table. The exploitability
21 of the average policy in zero-sum games decreases as the model accuracy and
22 the number of iterations increase [Waugh et al., 2015; Morrill, 2016]. As long
23 as the regression model errors decrease across iterations, the average policy
24 converges toward a Nash equilibrium in zero-sum games.
25 
26 # References
27 
28 Dustin Morrill. Using Regret Estimation to Solve Games Compactly.
29  M.Sc. thesis, Computing Science Department, University of Alberta,
30  Apr 1, 2016, Edmonton Alberta, Canada.
31 Kevin Waugh, Dustin Morrill, J. Andrew Bagnell, and Michael Bowling.
32  Solving Games with Functional Regret Estimation. At the Twenty-Ninth AAAI
33  Conference on Artificial Intelligence, January 25-29, 2015, Austin Texas,
34  USA. Pages 2138-2145.
35 Martin Zinkevich, Michael Johanson, Michael Bowling, and Carmelo Piccione.
36  Regret Minimization in Games with Incomplete Information.
37  At Advances in Neural Information Processing Systems 20 (NeurIPS). 2007.
38 """

46 def tensor_to_matrix(tensor):
47  """Converts `tensor` to a matrix (a rank-2 tensor) or raises an exception.
48 
49  Args:
50  tensor: The tensor to convert.
51 
52  Returns:
53  A PyTorch matrix (rank-2 `torch.Tensor`).
54 
55  Raises:
56  ValueError: If `tensor` cannot be trivially converted to a matrix, i.e.
57  `tensor` has a rank > 2.
58  """
59  tensor = torch.Tensor(tensor)
60  rank = tensor.ndim
61  # rank = len(list(tensor.shape))
62  if rank > 2:
63  raise ValueError(
64  ("Tensor {} cannot be converted into a matrix as it is rank "
65  "{} > 2.").format(tensor, rank))
66  elif rank < 2:
67  num_columns = 1 if rank == 0 else tensor.shape[0]
68  tensor = torch.reshape(tensor, [1, num_columns])
69  return tensor

72 def with_one_hot_action_features(state_features, legal_actions,
73  num_distinct_actions):
74  """Constructs features for each sequence by extending state features.
75 
76  Sequences features are constructed by concatenating one-hot features
77  indicating each action to the information state features and stacking them.
78 
79  Args:
80  state_features: The features for the information state alone. Must be a
81  `torch.Tensor` with a rank less than or equal to (if batched) 2.
82  legal_actions: The list of legal actions in this state. Determines the
83  number of rows in the returned feature matrix.
84  num_distinct_actions: The number of globally distinct actions in the game.
85  Determines the length of the action feature vector concatenated onto the
86  state features.
87 
88  Returns:
89  A `torch.Tensor` feature matrix with one row for each sequence and # state
90  features plus `num_distinct_actions`-columns.
91 
92  Raises:
93  ValueError: If `state_features` has a rank > 2.
94  """
95  state_features = tensor_to_matrix(state_features)

259  def counterfactual_regrets_and_reach_weights(self, regret_player,
260  reach_weight_player,
261  *sequence_weights):
262  """Returns counterfactual regrets and reach weights as a tuple.
263 
264  Args:
265  regret_player: The player for whom counterfactual regrets are computed.
266  reach_weight_player: The player for whom reach weights are computed.
267  *sequence_weights: A list of non-negative sequence weights for each player
268  determining the policy profile. Behavioral policies are generated by
269  normalizing sequence weights corresponding to actions in each
270  information state by their sum.
271 
272  Returns:
273  The counterfactual regrets and reach weights as an `np.array`-`np.array`
274  tuple.
275 
276  Raises:
277  ValueError: If there are too few sequence weights at any information state
278  for any player.
279  """
280  num_players = len(sequence_weights)
281  regrets = np.zeros(self.num_player_sequencesnum_player_sequences[regret_player])
282  reach_weights = np.zeros(self.num_player_sequencesnum_player_sequences[reach_weight_player])

284  def _walk_descendants(state, reach_probabilities, chance_reach_probability):
285  """Compute `state`'s counterfactual regrets and reach weights.
286 
287  Args:
288  state: An OpenSpiel `State`.
289  reach_probabilities: The probability that each player plays to reach
290  `state`'s history.
291  chance_reach_probability: The probability that all chance outcomes in
292  `state`'s history occur.
293 
294  Returns:
295  The counterfactual value of `state`'s history.
296  Raises:
297  ValueError if there are too few sequence weights at any information
298  state for any player.
299  """
300 
301  if state.is_terminal():
302  player_reach = (
303  np.prod(reach_probabilities[:regret_player]) *
304  np.prod(reach_probabilities[regret_player + 1:]))
305 
306  counterfactual_reach_prob = player_reach * chance_reach_probability
307  u = self.terminal_valuesterminal_values[state.history_str()]
308  return u[regret_player] * counterfactual_reach_prob
309 
310  elif state.is_chance_node():
311  v = 0.0
312  for action, action_prob in state.chance_outcomes():
313  v += _walk_descendants(
314  state.child(action), reach_probabilities,
315  chance_reach_probability * action_prob)
316  return v
317 
318  player = state.current_player()
319  info_state = state.information_state_string(player)
320  sequence_idx_offset = self.info_state_to_sequence_idxinfo_state_to_sequence_idx[info_state]
321  actions = state.legal_actions(player)
322 
323  sequence_idx_end = sequence_idx_offset + len(actions)
324  my_sequence_weights = sequence_weights[player][
325  sequence_idx_offset:sequence_idx_end]
326 
327  if len(my_sequence_weights) < len(actions):
328  raise ValueError(
329  ("Invalid policy: Policy {player} at sequence offset "
330  "{sequence_idx_offset} has only {policy_len} elements but there "
331  "are {num_actions} legal actions.").format(
332  player=player,
333  sequence_idx_offset=sequence_idx_offset,
334  policy_len=len(my_sequence_weights),
335  num_actions=len(actions)))
336 
337  policy = normalized_by_sum(my_sequence_weights)
338  action_values = np.zeros(len(actions))
339  state_value = 0.0
340 
341  is_reach_weight_player_node = player == reach_weight_player
342  is_regret_player_node = player == regret_player
343 
344  reach_prob = reach_probabilities[player]
345  for action_idx, action in enumerate(actions):
346  action_prob = policy[action_idx]
347  next_reach_prob = reach_prob * action_prob
348 
349  if is_reach_weight_player_node:
350  reach_weight_player_plays_down_this_line = next_reach_prob > 0
351  if not reach_weight_player_plays_down_this_line:
352  continue
353  sequence_idx = sequence_idx_offset + action_idx
354  reach_weights[sequence_idx] += next_reach_prob
355 
356  reach_probabilities[player] = next_reach_prob
357 
358  action_value = _walk_descendants(
359  state.child(action), reach_probabilities, chance_reach_probability)
360 
361  if is_regret_player_node:
362  state_value = state_value + action_prob * action_value
363  else:
364  state_value = state_value + action_value
365  action_values[action_idx] = action_value
366 
367  reach_probabilities[player] = reach_prob
368 
369  if is_regret_player_node:
370  regrets[sequence_idx_offset:sequence_idx_end] += (
371  action_values - state_value)
372  return state_value
373 
374  # End of _walk_descendants
375 
376  _walk_descendants(self.rootroot, np.ones(num_players), 1.0)
377  return regrets, reach_weights

380 def normalized_by_sum(v, axis=0, mutate=False):
381  """Divides each element of `v` along `axis` by the sum of `v` along `axis`.
382 
383  Assumes `v` is non-negative. Sets of `v` elements along `axis` that sum to
384  zero are normalized to `1 / v.shape[axis]` (a uniform distribution).
385 
386  Args:
387  v: Non-negative array of values.
388  axis: An integer axis.
389  mutate: Whether or not to store the result in `v`.
390 
391  Returns:
392  The normalized array.
393  """
394  v = np.asarray(v)
395  denominator = v.sum(axis=axis, keepdims=True)
396  denominator_is_zero = denominator == 0
397 
398  # Every element of `denominator_is_zero` that is true corresponds to a
399  # set of elements in `v` along `axis` that are all zero. By setting these
400  # denominators to `v.shape[axis]` and adding 1 to each of the corresponding
401  # elements in `v`, these elements are normalized to `1 / v.shape[axis]`
402  # (a uniform distribution).
403  denominator += v.shape[axis] * denominator_is_zero
404  if mutate:
405  v += denominator_is_zero
406  v /= denominator
407  else:
408  v = (v + denominator_is_zero) / denominator
409  return v

417 def _descendant_states(state, depth_limit, depth, include_terminals,
418  include_chance_states):
419  """Recursive descendant state generator.
420 
421  Decision states are always yielded.
422 
423  Args:
424  state: The current state.
425  depth_limit: The descendant depth limit. Zero will ensure only
426  `initial_state` is generated and negative numbers specify the absence of a
427  limit.
428  depth: The current descendant depth.
429  include_terminals: Whether or not to include terminal states.
430  include_chance_states: Whether or not to include chance states.
431 
432  Yields:
433  `State`, a state that is `initial_state` or one of its descendants.
434  """
435  if state.is_terminal():
436  if include_terminals:
437  yield state
438  return
439 
440  if depth > depth_limit >= 0:
441  return
442 
443  if not state.is_chance_node() or include_chance_states:
444  yield state
445 
446  for action in state.legal_actions():
447  state_for_search = state.child(action)
448  for substate in _descendant_states(state_for_search, depth_limit, depth + 1,
449  include_terminals,
450  include_chance_states):
451  yield substate

503 def feedforward_evaluate(layers,
504  x,
505  use_skip_connections=False,
506  hidden_are_factored=False,
507  hidden_activation=nn.ReLU):
508  """Evaluates `layers` as a feedforward neural network on `x`.
509 
510  Args:
511  layers: The neural network layers (`torch.Tensor` -> `torch.Tensor`
512  callables).
513  x: The array-like input to evaluate. Must be trivially convertible to a
514  matrix (tensor rank <= 2).
515  use_skip_connections: Whether or not to use skip connections between layers.
516  If the layer input has too few features to be added to the layer output,
517  then the end of input is padded with zeros. If it has too many features,
518  then the input is truncated.
519  hidden_are_factored: Whether or not hidden logical layers are factored into
520  two separate linear transformations stored as adjacent elements of
521  `layers`.
522  hidden_activation: the activation function following the hidden layers.
523 
524  Returns:
525  The `torch.Tensor` evaluation result.
526 
527  Raises:
528  ValueError: If `x` has a rank greater than 2.
529  """
530  x = tensor_to_matrix(x)
531  i = 0
532  while i < len(layers) - 1:
533  if isinstance(layers[i], hidden_activation):
534  x = layers[i](x)
535  i += 1
536  continue
537  y = layers[i](x)
538  i += 1
539  if hidden_are_factored:
540  y = layers[i](y)
541  i += 1
542  if use_skip_connections:
543  my_num_features = x.shape[1]
544  padding = y.shape[1] - my_num_features
545  if padding > 0:
546  zeros = torch.zeros([x.shape[0], padding])
547  x = torch.cat([x, zeros], axis=1)
548  elif padding < 0:
549  x = x[0:x.shape[0], 0:y.shape[1]]
550  y = x + y
551  x = y
552  return layers[-1](x)

555 class DeepRcfrModel(nn.Module):
556  """A flexible deep feedforward RCFR model class.
557 
558  Properties:
559  layers: The `torch.keras.Layer` layers describing this model.
560  """
561 
562  def __init__(self,
563  game,
564  num_hidden_units,
565  num_hidden_layers=1,
566  num_hidden_factors=0,
567  hidden_activation=nn.ReLU,
568  use_skip_connections=False,
569  regularizer=None):
570  """Creates a new `DeepRcfrModel.
571 
572  Args:
573  game: The OpenSpiel game being solved.
574  num_hidden_units: The number of units in each hidden layer.
575  num_hidden_layers: The number of hidden layers. Defaults to 1.
576  num_hidden_factors: The number of hidden factors or the matrix rank of the
577  layer. If greater than zero, hidden layers will be split into two
578  separate linear transformations, the first with
579  `num_hidden_factors`-columns and the second with
580  `num_hidden_units`-columns. The result is that the logical hidden layer
581  is a rank-`num_hidden_units` matrix instead of a rank-`num_hidden_units`
582  matrix. When `num_hidden_units < num_hidden_units`, this is effectively
583  implements weight sharing. Defaults to 0.
584  hidden_activation: The activation function to apply over hidden layers.
585  Defaults to `torch.nn.ReLU`.
586  use_skip_connections: Whether or not to apply skip connections (layer
587  output = layer(x) + x) on hidden layers. Zero padding or truncation is
588  used to match the number of columns on layer inputs and outputs.
589  regularizer: A regularizer to apply to each layer. Defaults to `None`.
590  """
591  super(DeepRcfrModel, self).__init__()
592  self._use_skip_connections_use_skip_connections = use_skip_connections
593  self._hidden_are_factored_hidden_are_factored = num_hidden_factors > 0
594  self._hidden_activation_hidden_activation = hidden_activation
595  input_size = num_features(game)
596 
597  self.layerslayers = []
598  for _ in range(num_hidden_layers):
599  if self._hidden_are_factored_hidden_are_factored:
600  self.layerslayers.append(nn.Linear(input_size, num_hidden_factors, bias=True))
601 
602  self.layerslayers.append(
603  nn.Linear(
604  num_hidden_factors if self._hidden_are_factored_hidden_are_factored else input_size,
605  num_hidden_units,
606  bias=True))
607  if hidden_activation:
608  self.layerslayers.append(hidden_activation())
609 
610  self.layerslayers.append(nn.Linear(num_hidden_units, 1, bias=True))
611 
612  self.layerslayers = nn.ModuleList(self.layerslayers)
613  # Construct variables for all layers by exercising the network.
614  x = torch.zeros([1, num_features(game)])
615  for layer in self.layerslayers:
616  x = layer(x)
617 
618  def __call__(self, x):
619  """Evaluates this model on `x`."""
620  return feedforward_evaluate(
621  layers=self.layerslayers,
622  x=x,
623  use_skip_connections=self._use_skip_connections_use_skip_connections,
624  hidden_are_factored=self._hidden_are_factored_hidden_are_factored,
625  hidden_activation=self._hidden_activation_hidden_activation)

796  self._num_candidates_num_candidates += 1
797  if self.num_elementsnum_elements < self.sizesize:
798  self._buffer_buffer[self.num_elementsnum_elements] = candidate
799  self.num_elementsnum_elements += 1

802  if idx < self.sizesize:
803  self._buffer_buffer[idx] = candidate

807  for candidate in candidates:

809
