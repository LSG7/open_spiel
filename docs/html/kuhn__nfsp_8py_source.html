55  def action_probabilities(self, state, player_id=None):
56  cur_player = state.current_player()
57  legal_actions = state.legal_actions(cur_player)
58 
59  self._obs_obs["current_player"] = cur_player
60  self._obs_obs["info_state"][cur_player] = (
61  state.information_state_tensor(cur_player))
62  self._obs_obs["legal_actions"][cur_player] = legal_actions

64  info_state = rl_environment.TimeStep(
65  observations=self._obs_obs, rewards=None, discounts=None, step_type=None)
66 
67  with self._policies_policies[cur_player].temp_mode_as(self._mode_mode):
68  p = self._policies_policies[cur_player].step(info_state, is_evaluation=True).probs
69  prob_dict = {action: p[action] for action in legal_actions}
