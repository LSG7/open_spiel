16"""Representation of a policy for a game.
17
18This is a standard representation for passing policies into algorithms,
19with currently the following implementations:
20
21 TabularPolicy - an explicit policy per state, stored in an array
22 of shape `(num_states, num_actions)`, convenient for tabular policy
23 solution methods.
24 UniformRandomPolicy - a uniform distribution over all legal actions for
25 the specified player. This is computed as needed, so can be used for
26 games where a tabular policy would be unfeasibly large.
27
28The main way of using a policy is to call `action_probabilities(state,
29player_id`), to obtain a dict of {action: probability}. `TabularPolicy`
30objects expose a lower-level interface, which may be more efficient for
31some use cases.
32"""

53def joint_action_probabilities_aux(state, policy):
54 """Auxiliary function for joint_action_probabilities.
55
56 Args:
57 state: a game state at a simultaneous decision node.
58 policy: policy that gives the probability distribution over the legal
59 actions for each players.
60
61 Returns:
62 actions_per_player: list of list of actions for each player
63 probs_per_player: list of list of probabilities do the corresponding action
64 in actions_per_player for each player.
65 """
66 assert state.is_simultaneous_node()
67 action_probs_per_player = [
68 policy.action_probabilities(state, player)
69 for player in range(state.get_game().num_players())
70 ]
71 actions_per_player = [pi.keys() for pi in action_probs_per_player]
72 probs_per_player = [pi.values() for pi in action_probs_per_player]
73 return actions_per_player, probs_per_player

76def joint_action_probabilities(state, policy):
77 """Yields action, probability pairs for a joint policy in simultaneous state.
78
79 Args:
80 state: a game state at a simultaneous decision node.
81 policy: policy that gives the probability distribution over the legal
82 actions for each players.
83
84 Yields:
85 (action, probability) pairs. An action is a tuple of individual
86 actions for each player of the game. The probability is a single joint
87 probability (product of all the individual probabilities).
88 """
89 actions_per_player, probs_per_player = joint_action_probabilities_aux(
90 state, policy)
91 for actions, probs in zip(
92 itertools.product(*actions_per_player),
93 itertools.product(*probs_per_player)):
94 yield actions, np.prod(probs)

97class Policy:
98 """Base class for policies.
99
100 A policy is something that returns a distribution over possible actions
101 given a state of the world.
102
103 Attributes:
104 game: the game for which this policy applies
105 player_ids: list of player ids for which this policy applies; each in the
106 interval [0..game.num_players()-1].
107 """
108
109 def __init__(self, game, player_ids):
110 """Initializes a policy.
111
112 Args:
113 game: the game for which this policy applies
114 player_ids: list of player ids for which this policy applies; each should
115 be in the range 0..game.num_players()-1.
116 """
117 self.game = game
118 self.player_ids = player_ids
119
120 def action_probabilities(self, state, player_id=None):
121 """Returns a dictionary {action: prob} for all legal actions.
122
123 IMPORTANT: We assume the following properties hold:
124 - All probabilities are >=0 and sum to 1
125 - TLDR: Policy implementations should list the (action, prob) for all legal
126 actions, but algorithms should not rely on this (yet).
127 Details: Before May 2020, only legal actions were present in the mapping,
128 but it did not have to be exhaustive: missing actions were considered to
129 be associated to a zero probability.
130 For example, a deterministic state-poliy was previously {action: 1.0}.
131 Given this change of convention is new and hard to enforce, algorithms
132 should not rely on the fact that all legal actions should be present.
133
134 Args:
135 state: A `pyspiel.State` object.
136 player_id: Optional, the player id for whom we want an action. Optional
137 unless this is a simultaneous state at which multiple players can act.
138
139 Returns:
140 A `dict` of `{action: probability}` for the specified player in the
141 supplied state.
142 """
143 raise NotImplementedError()
144
145 def __call__(self, state, player_id=None):
146 """Turns the policy into a callable.
147
148 Args:
149 state: The current state of the game.
150 player_id: Optional, the player id for whom we want an action. Optional
151 unless this is a simultaneous state at which multiple players can act.
152
153 Returns:
154 Dictionary of action: probability.
155 """
156 return self.action_probabilities(state, player_id)

158 def to_tabular(self, states=None):
159 """Returns a new `TabularPolicy` equivalent to this policy.
160
161 Args:
162 states: States of the game that will be used for the tabular policy. If
163 None, then get_tabular_policy_states() method will be used to generate
164 them.
165
166 Returns:
167 a TabularPolicy.
168 """
169 states = states or get_tabular_policy_states(self.game)
170 tabular_policy = TabularPolicy(self.game, self.player_ids, states=states)
171 for index, state in enumerate(tabular_policy.states):
172 tabular_policy.action_probability_array[index, :] = 0
173 for action, probability in self.action_probabilities(state).items():
174 tabular_policy.action_probability_array[index, action] = probability
175 return tabular_policy

178class TabularPolicy(Policy):
179 """Policy implementation where the policy is in explicit tabular form.
180
181 In addition to implementing the `Policy` interface, this class exposes
182 details of the policy representation for easy manipulation.
183
184 The states are guaranteed to be grouped by player, which can simplify
185 code for users of this class, i.e. `action_probability_array` contains
186 states for player 0 first, followed by states for player 1, etc.
187
188 The policy uses `state.information_state_string` as the keys if available,
189 otherwise `state.observation_string`.
190
191 Usages:
192
193 - Set `policy(info_state, action)`:
194 ```
195 tabular_policy = TabularPolicy(game)
196 info_state_str = state.information_state_string(<optional player>)
197 state_policy = tabular_policy.policy_for_key(info_state_str)
198 state_policy[action] = <value>
199 ```
200 - Set `policy(info_state)`:
201 ```
202 tabular_policy = TabularPolicy(game)
203 info_state_str = state.information_state_string(<optional player>)
204 state_policy = tabular_policy.policy_for_key(info_state_str)
205 state_policy[:] = <list or numpy.array>
206 ```
207
208 Attributes:
209 action_probability_array: array of shape `(num_states, num_actions)`, where
210 `action_probability_array[s, a]` is the probability of choosing action `a`
211 when at state `s`.
212 state_lookup: `dict` mapping state key string to index into the
213 `tabular_policy` array. If information state strings overlap, e.g. for
214 different players or if the information state string has imperfect recall,
215 then those states will be mapped to the same policy.
216 legal_actions_mask: array of shape `(num_states, num_actions)`, each row
217 representing which of the possible actions in the game are valid in this
218 particular state, containing 1 for valid actions, 0 for invalid actions.
219 states_per_player: A `list` per player of the state key strings at which
220 they have a decision to make.
221 states: A `list` of the states as ordered in the `action_probability_array`.
222 state_in: array of shape `(num_states, state_vector_size)` containing the
223 normalised vector representation of each information state. Populated only
224 for games which support information_state_tensor(), and is None otherwise.
225 game_type: The game attributes as returned by `Game::GetType`; used to
226 determine whether to use information state or observation as the key in
227 the tabular policy.
228 """
229
230 def __init__(self,
231 game,
232 players=None,
233 to_string=lambda s: s.history_str(),
234 states=None):
235 """Initializes a uniform random policy for all players in the game."""
236 players = sorted(players or range(game.num_players()))
237 super().__init__(game, players)
238 self.game_type = game.get_type()
239
240 # Get all states in the game at which players have to make decisions unless
241 # they are explicitly specified.
242 states = states or get_all_states.get_all_states(
243 game,
244 depth_limit=-1,
245 include_terminals=False,
246 include_chance_states=False,
247 include_mean_field_states=False,
248 to_string=to_string)
249
250 # Assemble legal actions for every valid (state, player) pair, keyed by
251 # information state string.
252 self.state_lookup = {}
253 self.states_per_player = [[] for _ in range(game.num_players())]
254 self.states = []
255 legal_actions_list = []
256 state_in_list = []
257 for player in players:
258 # States are ordered by their history.
259 for _, state in sorted(states.items(), key=lambda pair: pair[0]):
260 if state.is_simultaneous_node() or player == state.current_player():
261 legal_actions = state.legal_actions_mask(player)
262 if any(legal_actions):
263 key = self._state_key(state, player)
264 if key not in self.state_lookup:
265 state_index = len(legal_actions_list)
266 self.state_lookup[key] = state_index
267 legal_actions_list.append(legal_actions)
268 self.states_per_player[player].append(key)
269 self.states.append(state)
270 if self.game_type.provides_information_state_tensor:
271 state_in_list.append(state.information_state_tensor(player))
272 elif self.game_type.provides_observation_tensor:
273 state_in_list.append(state.observation_tensor(player))
274
275 # Put legal action masks in a numpy array and create the uniform random
276 # policy.
277 self.state_in = None
278 if state_in_list:
279 self.state_in = np.array(state_in_list)
280 self.legal_actions_mask = np.array(legal_actions_list)
281 self.action_probability_array = (
282 self.legal_actions_mask /
283 np.sum(self.legal_actions_mask, axis=-1, keepdims=True))
284
285 def _state_key(self, state, player):
286 """Returns the key to use to look up this (state, player) pair."""
287 if self.game_type.provides_information_state_string:
288 if player is None:
289 return state.information_state_string()
290 return state.information_state_string(player)
291 if self.game_type.provides_observation_string:
292 if player is None:
293 return state.observation_string()
294 return state.observation_string(player)
295 return str(state)
296
297 def action_probabilities(self, state, player_id=None):
298 """Returns an {action: probability} dict, covering all legal actions."""
299 legal_actions = (
300 state.legal_actions()
301 if player_id is None else state.legal_actions(player_id))
302 if not legal_actions:
303 return {0: 1.0}
304 probability = self.policy_for_key(self._state_key(state, player_id))
305 return {action: probability[action] for action in legal_actions}
306
307 def state_index(self, state):
308 """Returns the index in the TabularPolicy associated to `state`."""
309 return self.state_lookup[self._state_key(state, state.current_player())]
310
311 def policy_for_key(self, key):
312 """Returns the policy as a vector given a state key string.
313
314 Args:
315 key: A key for the specified state.
316
317 Returns:
318 A vector of probabilities, one per action. This is a slice of the
319 backing policy array, and so slice or index assignment will update the
320 policy. For example:
321 ```
322 tabular_policy.policy_for_key(s)[:] = [0.1, 0.5, 0.4]
323 ```
324 """
325 return self.action_probability_array[self.state_lookup[key]]
326
327 def to_dict(self):
328 """Returns a single dictionary representing the tabular policy.
329
330 Returns:
331 A dictionary of string keys to lists of (action, prob) pairs.
332 """
333 policy_dict = {}
334 num_actions = self.action_probability_array.shape[1]
335 for infostate_key, index in self.state_lookup.items():
336 probs = self.action_probability_array[index]
337 actions_and_probs = [(a, probs[a]) for a in range(num_actions)]
338 policy_dict[infostate_key] = actions_and_probs
339 return policy_dict

341 def __copy__(self, copy_action_probability_array=True):
342 """Returns a shallow copy of self.
343
344 Most class attributes will be pointers to the copied object's attributes,
345 and therefore altering them could lead to unexpected behavioural changes.
346 Only action_probability_array is expected to be modified.
347
348 Args:
349 copy_action_probability_array: Whether to also include
350 action_probability_array in the copy operation.
351
352 Returns:
353 Copy.
354 """
355 result = TabularPolicy.__new__(TabularPolicy)
356 result.state_lookup = self.state_lookup
357 result.game_type = self.game_type
358 result.legal_actions_mask = self.legal_actions_mask
359 result.state_in = self.state_in
360 result.state_lookup = self.state_lookup
361 result.states_per_player = self.states_per_player
362 result.states = self.states
363 result.game = self.game
364 result.player_ids = self.player_ids
365 if copy_action_probability_array:
366 result.action_probability_array = np.copy(self.action_probability_array)
367 return result

369 def copy_with_noise(self,
370 alpha=0.0,
371 beta=0.0,
372 random_state=np.random.RandomState()):
373 """Returns a copy of this policy perturbed with noise.
374
375 Generates a new random distribution using a softmax on normal random
376 variables with temperature beta, and mixes it with the old distribution
377 using 1-alpha * old_distribution + alpha * random_distribution.
378 Args:
379 alpha: Parameter characterizing the mixture amount between new and old
380 distributions. Between 0 and 1.
381 alpha = 0: keep old table.
382 alpha = 1: keep random table.
383 beta: Temperature of the softmax. Makes for more extreme policies.
384 random_state: A numpy `RandomState` object. If not provided, a shared
385 random state will be used.
386
387 Returns:
388 Perturbed copy.
389 """
390 copied_instance = self.__copy__(False)
391 probability_array = self.action_probability_array
392 noise_mask = random_state.normal(size=probability_array.shape)
393 noise_mask = np.exp(beta * noise_mask) * self.legal_actions_mask
394 noise_mask = noise_mask / (np.sum(noise_mask, axis=1).reshape(-1, 1))
395 copied_instance.action_probability_array = (
396 1 - alpha) * probability_array + alpha * noise_mask
397 return copied_instance

400class UniformRandomPolicy(Policy):
401 """Policy where the action distribution is uniform over all legal actions.
402
403 This is computed as needed, so can be used for games where a tabular policy
404 would be unfeasibly large, but incurs a legal action computation every time.
405 """
406
407 def __init__(self, game):
408 """Initializes a uniform random policy for all players in the game."""
409 all_players = list(range(game.num_players()))
410 super().__init__(game, all_players)
411
412 def action_probabilities(self, state, player_id=None):
413 """Returns a uniform random policy for a player in a state.
414
415 Args:
416 state: A `pyspiel.State` object.
417 player_id: Optional, the player id for which we want an action. Optional
418 unless this is a simultaneous state at which multiple players can act.
419
420 Returns:
421 A `dict` of `{action: probability}` for the specified player in the
422 supplied state. This will contain all legal actions, each with the same
423 probability, equal to 1 / num_legal_actions.
424 """
425 legal_actions = (
426 state.legal_actions()
427 if player_id is None else state.legal_actions(player_id))
428 if not legal_actions:
429 return {0: 1.0}
430 probability = 1 / len(legal_actions)
431 return {action: probability for action in legal_actions}

434class FirstActionPolicy(Policy):

437 def __init__(self, game):

445 if not legal_actions:
446 return {0: 1.0}

475def tabular_policy_from_callable(game, callable_policy, players=None):
476 """Converts a legacy callable policy into a TabularPolicy.
477
478 Recommendation - instead of using this to convert your policy for evaluation
479 purposes, work directly with a `TabularPolicy` if possible.
480 Second choice - work with a `Policy` class and call `to_tabular` as needed.
481
482 Args:
483 game: The game for which we want a TabularPolicy.
484 callable_policy: A callable: state -> action probabilities dict or list.
485 players: List of players this policy applies to. If `None`, applies to all
486 players.
487
488 Returns:
489 A TabularPolicy that materializes the callable policy.
490 """
491 tabular_policy = TabularPolicy(game, players)
492 for state_index, state in enumerate(tabular_policy.states):
493 action_probabilities = dict(callable_policy(state))
494 infostate_policy = [
495 action_probabilities.get(action, 0.)
496 for action in range(game.num_distinct_actions())
497 ]
498 tabular_policy.action_probability_array[state_index, :] = infostate_policy
499 return tabular_policy

502def pyspiel_policy_to_python_policy(game, pyspiel_tabular_policy, players=None):
503 """Converts a pyspiel.TabularPolicy to a TabularPolicy.
504
505 Args:
506 game: The OpenSpiel game.
507 pyspiel_tabular_policy: Pyspiel tabular policy to copy from.
508 players: List of integer player ids to copy policy from. For example,
509 `players=[0]` will only copy player 0's policy over into the python policy
510 (the other player's policies will be undefined). Default value of `None`
511 will copy all players' policies.
512
513 Returns:
514 python_policy
515 """
516 policy = TabularPolicy(game, players=players)
517 for item in pyspiel_tabular_policy.policy_table().items():
518 info_state_str, actions_probs = item
519 # If requested, only populate a policy for particular players.
520 if players is not None and info_state_str not in policy.state_lookup:
521 continue
522 state_policy = policy.policy_for_key(info_state_str)
523 for action, prob in actions_probs:
524 state_policy[action] = prob
525 return policy

528def python_policy_to_pyspiel_policy(python_tabular_policy):
529 """Converts a TabularPolicy to a pyspiel.TabularPolicy."""
530 infostates_to_probabilities = dict()
531 for infostate, index in python_tabular_policy.state_lookup.items():
532 probs = python_tabular_policy.action_probability_array[index]
533 legals = python_tabular_policy.legal_actions_mask[index]
534
535 action_probs = []
536 for action, (prob, is_legal) in enumerate(zip(probs, legals)):
537 if is_legal == 1:
538 action_probs.append((action, prob))
539 infostates_to_probabilities[infostate] = action_probs
540 return pyspiel.TabularPolicy(infostates_to_probabilities)

555def merge_tabular_policies(tabular_policies, game):
556 """Merges n_player policies into single joint policy.
557
558 Missing states are filled with a valid uniform policy.
559
560 Args:
561 tabular_policies: List of python TabularPolicy (one for each player).
562 game: The game corresponding to the resulting TabularPolicy.
563
564 Returns:
565 merged_policy: A TabularPolicy with each player i's policy taken from the
566 ith joint_policy.
567 """
568 if len(tabular_policies) != game.num_players():
569 raise ValueError("len(tabular_policies) != num_players: %d != %d" %
570 (len(tabular_policies), game.num_players()))
571 merged_policy = TabularPolicy(game)
572 for p, p_states in enumerate(merged_policy.states_per_player):
573 for p_state in p_states:
574 to_index = merged_policy.state_lookup[p_state]
575 # Only copy if the state exists, otherwise fall back onto uniform.
576 if p_state in tabular_policies[p].state_lookup:
577 from_index = tabular_policies[p].state_lookup[p_state]
578 merged_policy.action_probability_array[to_index] = (
579 tabular_policies[p].action_probability_array[from_index])
580 return merged_policy
