15"""Example: MMD with dilated entropy to compute approx. Nash in Kuhn poker."""
16
17from absl import app
18from absl import flags
19
20from open_spiel.python.algorithms import exploitability
21from open_spiel.python.algorithms import mmd_dilated
22import pyspiel
23
24FLAGS = flags.FLAGS
25
26flags.DEFINE_integer("iterations", 1000, "Number of iterations")
27flags.DEFINE_string("game", "kuhn_poker", "Name of the game")
28flags.DEFINE_integer("print_freq", 100, "How often to print the exploitability")
29
30
31def main(_):
32 game = pyspiel.load_game(FLAGS.game)
33 # need to manually set stepsize if alpha = 0
34 mmd = mmd_dilated.MMDDilatedEnt(game, alpha=0, stepsize=1)
35
36 for i in range(FLAGS.iterations):
37 mmd.update_sequences()
38 if i % FLAGS.print_freq == 0:
39 conv = exploitability.exploitability(game, mmd.get_avg_policies())
40 print("Iteration {} exploitability {}".format(i, conv))
41
42
43if __name__ == "__main__":
44 app.run(main)
