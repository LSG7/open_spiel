23class PolicyValue(value.ValueFunction):
24 """Computes the value of a specified strategy."""
25
26 def __init__(self,
27 game,
28 distribution: distribution_std.Distribution,
29 policy: policy_std.Policy,
30 state_value: Optional[value.ValueFunction] = None,
31 root_state=None):
32 """Initializes the value calculation.
33
34 Args:
35 game: The game to analyze.
36 distribution: A `distribution.Distribution` object.
37 policy: A `policy.Policy` object.
38 state_value: A state value function. Defaults to Tabular.
39 root_state: The state of the game at which to start. If `None`, the game
40 root state is used.
41 """
42 super(PolicyValue, self).__init__(game)
43 if root_state is None:
44 self._root_states = game.new_initial_states()
45 else:
46 self._root_states = [root_state]
47 self._distribution = distribution
48 self._policy = policy
49
50 self._state_value = (state_value if state_value is not None
51 else value.TabularValueFunction(game))
52
53 self.evaluate()
54
55 def eval_state(self, state):
56 """Evaluate the value of a state."""
57 state_str = state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)
58 if self._state_value.has(state_str):
59 return self._state_value(state_str)
60 elif state.is_terminal():
61 self._state_value.set_value(
62 state_str,
63 state.rewards()[state.mean_field_population()])
64 return self._state_value(state_str)
65 elif state.current_player() == pyspiel.PlayerId.CHANCE:
66 self._state_value.set_value(state_str, 0.0)
67 for action, prob in state.chance_outcomes():
68 new_state = state.child(action)
69 self._state_value.add_value(state_str,
70 prob * self.eval_state(new_state))
71 return self._state_value(state_str)
72 elif state.current_player() == pyspiel.PlayerId.MEAN_FIELD:
73 dist_to_register = state.distribution_support()
74 dist = [
75 self._distribution.value_str(str_state, 0.)
76 for str_state in dist_to_register
77 ]
78 new_state = state.clone()
79 new_state.update_distribution(dist)
80 self._state_value.set_value(
81 state_str,
82 state.rewards()[state.mean_field_population()] +
83 self.eval_state(new_state))
84 return self._state_value(state_str)
85 else:
86 assert int(state.current_player()) >= 0, "The player id should be >= 0"
87 v = 0.0
88 for action, prob in self._policy.action_probabilities(state).items():
89 new_state = state.child(action)
90 v += prob * self.eval_state(new_state)
91 self._state_value.set_value(
92 state_str,
93 state.rewards()[state.mean_field_population()] + v)
94 return self._state_value(state_str)
95
96 def evaluate(self):
97 """Evaluate the value over states of self._policy."""
98 for state in self._root_states:
99 self.eval_state(state)
100
101 def value(self, state, action=None):
102 if action is None:
103 return self._state_value(
104 state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID))
105 new_state = state.child(action)
106 return state.rewards()[state.mean_field_population()] + self._state_value(
107 new_state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID))
