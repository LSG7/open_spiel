50// This templated function avoids compiler warnings about tautological
51// comparisons when log_internal::Tid is unsigned. It can be replaced with a
52// constexpr if once the minimum C++ version Abseil suppports is C++17.
53template <typename T>
54inline std::enable_if_t<!std::is_signed<T>::value>
55PutLeadingWhitespace(T tid, char*& p) {
56 if (tid < 10) *p++ = ' ';
57 if (tid < 100) *p++ = ' ';
58 if (tid < 1000) *p++ = ' ';
59 if (tid < 10000) *p++ = ' ';
60 if (tid < 100000) *p++ = ' ';
61 if (tid < 1000000) *p++ = ' ';
62}

75// The fields before the filename are all fixed-width except for the thread ID,
76// which is of bounded width.
77size_t FormatBoundedFields(absl::LogSeverity severity, absl::Time timestamp,
78 log_internal::Tid tid, absl::Span<char>& buf) {
79 constexpr size_t kBoundedFieldsMaxLen =
80 sizeof("SMMDD HH:MM:SS.NNNNNN ") +
81 (1 + std::numeric_limits<log_internal::Tid>::digits10 + 1) - sizeof("");
82 if (ABSL_PREDICT_FALSE(buf.size() < kBoundedFieldsMaxLen)) {
83 // We don't bother trying to truncate these fields if the buffer is too
84 // short (or almost too short) because it would require doing a lot more
85 // length checking (slow) and it should never happen. A 15kB buffer should
86 // be enough for anyone. Instead we mark `buf` full without writing
87 // anything.
88 buf.remove_suffix(buf.size());
89 return 0;
90 }
91
92 // We can't call absl::LocalTime(), localtime_r(), or anything else here that
93 // isn't async-signal-safe. We can only use the time zone if it has already
94 // been loaded.
95 const absl::TimeZone* tz = absl::log_internal::TimeZone();
96 if (ABSL_PREDICT_FALSE(tz == nullptr)) {
97 // If a time zone hasn't been set yet because we are logging before the
98 // logging library has been initialized, we fallback to a simpler, slower
99 // method. Just report the raw Unix time in seconds. We cram this into the
100 // normal time format for the benefit of parsers.
101 auto tv = absl::ToTimeval(timestamp);
102 int snprintf_result = absl::SNPrintF(
103 buf.data(), buf.size(), "%c0000 00:00:%02d.%06d %7d ",
104 absl::LogSeverityName(severity)[0], static_cast<int>(tv.tv_sec),
105 static_cast<int>(tv.tv_usec), static_cast<int>(tid));
106 if (snprintf_result >= 0) {
107 buf.remove_prefix(static_cast<size_t>(snprintf_result));
108 return static_cast<size_t>(snprintf_result);
109 }
110 return 0;
111 }
112
113 char* p = buf.data();
114 *p++ = absl::LogSeverityName(severity)[0];
115 const absl::TimeZone::CivilInfo ci = tz->At(timestamp);
116 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(ci.cs.month()), p);
117 p += 2;
118 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(ci.cs.day()), p);
119 p += 2;
120 *p++ = ' ';
121 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(ci.cs.hour()), p);
122 p += 2;
123 *p++ = ':';
124 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(ci.cs.minute()), p);
125 p += 2;
126 *p++ = ':';
127 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(ci.cs.second()), p);
128 p += 2;
129 *p++ = '.';
130 const int64_t usecs = absl::ToInt64Microseconds(ci.subsecond);
131 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(usecs / 10000), p);
132 p += 2;
133 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(usecs / 100 % 100),
134 p);
135 p += 2;
136 absl::numbers_internal::PutTwoDigits(static_cast<size_t>(usecs % 100), p);
137 p += 2;
138 *p++ = ' ';
139 PutLeadingWhitespace(tid, p);
140 p = absl::numbers_internal::FastIntToBuffer(tid, p);
141 *p++ = ' ';
142 const size_t bytes_formatted = static_cast<size_t>(p - buf.data());
143 buf.remove_prefix(bytes_formatted);
144 return bytes_formatted;
145}

147size_t FormatLineNumber(int line, absl::Span<char>& buf) {
148 constexpr size_t kLineFieldMaxLen =
149 sizeof(":] ") + (1 + std::numeric_limits<int>::digits10 + 1) - sizeof("");
150 if (ABSL_PREDICT_FALSE(buf.size() < kLineFieldMaxLen)) {
151 // As above, we don't bother trying to truncate this if the buffer is too
152 // short and it should never happen.
153 buf.remove_suffix(buf.size());
154 return 0;
155 }
156 char* p = buf.data();
157 *p++ = ':';
158 p = absl::numbers_internal::FastIntToBuffer(line, p);
159 *p++ = ']';
160 *p++ = ' ';
161 const size_t bytes_formatted = static_cast<size_t>(p - buf.data());
162 buf.remove_prefix(bytes_formatted);
163 return bytes_formatted;
164}

168std::string FormatLogMessage(absl::LogSeverity severity,
169 absl::CivilSecond civil_second,
170 absl::Duration subsecond, log_internal::Tid tid,
171 absl::string_view basename, int line,
172 PrefixFormat format, absl::string_view message) {
173 return absl::StrFormat(
174 "%c%02d%02d %02d:%02d:%02d.%06d %7d %s:%d] %s%s",
175 absl::LogSeverityName(severity)[0], civil_second.month(),
176 civil_second.day(), civil_second.hour(), civil_second.minute(),
177 civil_second.second(), absl::ToInt64Microseconds(subsecond), tid,
178 basename, line, format == PrefixFormat::kRaw ? "RAW: " : "", message);
179}

181// This method is fairly hot, and the library always passes a huge `buf`, so we
182// save some bounds-checking cycles by not trying to do precise truncation.
183// Truncating at a field boundary is probably a better UX anyway.
184//
185// The prefix is written in three parts, each of which does a single
186// bounds-check and truncation:
187// 1. severity, timestamp, and thread ID
188// 2. filename
189// 3. line number and bracket
190size_t FormatLogPrefix(absl::LogSeverity severity, absl::Time timestamp,
191 log_internal::Tid tid, absl::string_view basename,
192 int line, PrefixFormat format, absl::Span<char>& buf) {
193 auto prefix_size = FormatBoundedFields(severity, timestamp, tid, buf);
194 prefix_size += log_internal::AppendTruncated(basename, buf);
195 prefix_size += FormatLineNumber(line, buf);
196 if (format == PrefixFormat::kRaw)
197 prefix_size += log_internal::AppendTruncated("RAW: ", buf);
198 return prefix_size;
199}
