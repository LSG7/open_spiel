37void init_pyspiel_games_euchre(py::module& m) {
38 py::module_ euchre = m.def_submodule("euchre");
39
40 euchre.attr("JACK_RANK") = py::int_(euchre::kJackRank);
41 euchre.attr("NUM_SUITS") = py::int_(euchre::kNumSuits);
42 euchre.attr("NUM_CARDS_PER_SUIT") = py::int_(euchre::kNumCardsPerSuit);
43 euchre.attr("NUM_CARDS") = py::int_(euchre::kNumCards);
44 euchre.attr("PASS_ACTION") = py::int_(euchre::kPassAction);
45 euchre.attr("CLUBS_TRUMP_ACTION") = py::int_(euchre::kClubsTrumpAction);
46 euchre.attr("DIAMONDS_TRUMP_ACTION") = py::int_(euchre::kDiamondsTrumpAction);
47 euchre.attr("HEARTS_TRUMP_ACTION") = py::int_(euchre::kHeartsTrumpAction);
48 euchre.attr("SPADES_TRUMP_ACTION") = py::int_(euchre::kSpadesTrumpAction);
49 euchre.attr("GO_ALONE_ACTION") = py::int_(euchre::kGoAloneAction);
50 euchre.attr("PLAY_WITH_PARTNER_ACTION") = py::int_(
51 euchre::kPlayWithPartnerAction);
52 euchre.attr("MAX_BIDS") = py::int_(euchre::kMaxBids);
53 euchre.attr("NUM_TRICKS") = py::int_(euchre::kNumTricks);
54 euchre.attr("FULL_HAND_SIZE") = py::int_(euchre::kFullHandSize);
55
56 euchre.def("card_string", euchre::CardString);
57 euchre.def("card_rank", py::overload_cast<int>(euchre::CardRank));
58 euchre.def("card_rank",
59 py::overload_cast<int, euchre::Suit>(euchre::CardRank));
60 euchre.def("card_suit", py::overload_cast<int>(euchre::CardSuit));
61 euchre.def("card_suit",
62 py::overload_cast<int, euchre::Suit>(euchre::CardSuit));
63
64 py::enum_<euchre::Suit>(euchre, "Suit")
65 .value("INVALID_SUIT", euchre::Suit::kInvalidSuit)

67 .value("DIAMONDS", euchre::Suit::kDiamonds)
68 .value("HEARTS", euchre::Suit::kHearts)

74 .value("DEAL", euchre::Phase::kDeal)
75 .value("BIDDING", euchre::Phase::kBidding)
76 .value("DISCARD", euchre::Phase::kDiscard)
77 .value("GO_ALONE", euchre::Phase::kGoAlone)
78 .value("PLAY", euchre::Phase::kPlay)
79 .value("GAME_OVER", euchre::Phase::kGameOver)
80 .export_values();
81
82 py::classh<EuchreState, State> state_class(euchre, "EuchreState");
83 state_class
84 .def("num_cards_dealt", &EuchreState::NumCardsDealt)
85 .def("num_cards_played", &EuchreState::NumCardsPlayed)
86 .def("num_passes", &EuchreState::NumPasses)
87 .def("upcard", &EuchreState::Upcard)
88 .def("discard", &EuchreState::Discard)
89 .def("trump_suit", &EuchreState::TrumpSuit)
90 .def("left_bower", &EuchreState::LeftBower)
91 .def("right_bower", &EuchreState::RightBower)
92 .def("declarer", &EuchreState::Declarer)
93 .def("declarer_partner", &EuchreState::DeclarerPartner)
94 .def("first_defender", &EuchreState::FirstDefender)
95 .def("second_defender", &EuchreState::SecondDefender)
96 .def("declarer_go_alone", &EuchreState::DeclarerGoAlone)
97 .def("lone_defender", &EuchreState::LoneDefender)
98 .def("active_players", &EuchreState::ActivePlayers)
99 .def("dealer", &EuchreState::Dealer)
100 .def("current_phase", &EuchreState::CurrentPhase)
101 .def("current_trick_index", &EuchreState::CurrentTrickIndex)
102 .def("current_trick",
103 py::overload_cast<>(&EuchreState::CurrentTrick, py::const_))
104 .def("card_holder", &EuchreState::CardHolder)
105 .def("tricks", &EuchreState::Tricks)
106 // Pickle support
107 .def(py::pickle(
108 [](const EuchreState& state) { // __getstate__
109 return SerializeGameAndState(*state.GetGame(), state);
110 },
111 [](const std::string& data) { // __setstate__
112 std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>
113 game_and_state = DeserializeGameAndState(data);
114 return dynamic_cast<EuchreState*>(game_and_state.second.release());
115 }));
116
117 py::class_<euchre::Trick>(state_class, "Trick")
118 .def("winning_card", &euchre::Trick::WinningCard)
119 .def("led_suit", &euchre::Trick::LedSuit)
120 .def("trump_suit", &euchre::Trick::TrumpSuit)
121 .def("trump_played", &euchre::Trick::TrumpPlayed)
122 .def("leader", &euchre::Trick::Leader)
123 .def("winner", &euchre::Trick::Winner)
124 .def("cards", &euchre::Trick::Cards);
125
126 py::classh<EuchreGame, Game>(m, "EuchreGame")
127 // Pickle support
128 .def(py::pickle(
129 [](std::shared_ptr<const EuchreGame> game) { // __getstate__
130 return game->ToString();
131 },
132 [](const std::string& data) { // __setstate__
133 return std::dynamic_pointer_cast<EuchreGame>(
134 std::const_pointer_cast<Game>(LoadGame(data)));
135 }));
136}
