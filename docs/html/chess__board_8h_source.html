163  // Converts to standard algebraic notation, as required by portable game
164  // notation (PGN). It is a chess move notation that is designed to be
165  // human-readable and concise.
166  //
167  // Unlike the LAN format, generating a SAN string requires the board the move
168  // is generated from.
169  //
170  // There are 3 types of SAN moves -
171  // 1. O-O (short castle)
172  // 2. O-O-O (long castle)
173  // 3. [piece type][from file][from rank][x][to square][=Promo][annotations]
174  //
175  // [piece type] is omitted for pawns
176  // [from file] is only included if 1) move is a pawn capture, or 2) it's
177  // required for disambiguation (see below).
178  // [from rank] is only included if it's required for disambiguation.
179  // [x] is only included for captures
180  // [to square] is always included
181  // [=Promo] is only included for promotions ("=N", "=B", "=R", "=Q" depending
182  // on type promoting to).
183  // [annotations] are a list of 0 or more characters added with different
184  // meanings. The ones we care about are '+' for check, and '#' for
185  // checkmate. All others are optional.
186  //
187  // Disambiguation:
188  // If a move is not uniquely-identified otherwise, file and/or rank of the
189  // from square is inserted to disambiguate. When either one will disambiguate,
190  // file should be used. If file is unique, file is used. Otherwise if rank is
191  // unique, rank is used. If neither is unique (this happens rarely, usually
192  // after under-promoting to a minor piece with both original pieces still
193  // intact, or double queen promotions with original queen still intact), both
194  // are used.
195  //
196  // Examples:
197  // * e4 (pawn to e4)
198  // * exd5 (pawn on file e capture the piece on d5)
199  // * Nf3 (knight to f3)
200  // * Nxd5 (knight captures piece on d5)
201  // * Bed5 (bishop on file e to d5)
202  // * B5xc3 (bishop on rank 5 capture piece on c3)
203  // * Ne5f7 (knight on e5 to f7, when there are 3 knights on the board, one on
204  // e file, and one on 5th rank)
205  // * exd8=N#!! (pawn on e file capture piece on d8 and promote to knight
206  // resulting in checkmate in a surprisingly good move)
207  // * O-O-O!!N+/- (a surprisingly good long castle that is a theoretical
208  // novelty that gives white a clear but not winning advantage)
209  std::string ToSAN(const ChessBoard& board) const;
210 
211  bool operator==(const Move& other) const {
212  return from == other.from && to == other.to && piece == other.piece &&
213  promotion_type == other.promotion_type &&
214  is_castling == other.is_castling;
215  }
216 };

235 // Specifies policy for pseudo legal moves generation.
236 enum PseudoLegalMoveSettings {
237  // Standard legal moves (do not allow to move past enemy pieces).
238  kAcknowledgeEnemyPieces,
239  // Pseudo-legal moves, where a piece can move anywhere (according to the rules
240  // for that piece), except if it was blocked from doing so by other player's
241  // pieces. This is used in games, where the player may not know the position
242  // of an enemy piece (like Kriegspiel or RBC) and it can try to move past the
243  // enemy (for example a rook can try to move the other side of the board, even
244  // if it is in fact blocked by an unseen opponent's pawn).

246 };
