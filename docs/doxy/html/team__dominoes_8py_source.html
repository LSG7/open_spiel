112class DominoesState(pyspiel.State):
113 """A python version of the Block Dominoes state."""
114
115 def __init__(self, game):
116 """Constructor; should only be called by Game.new_initial_state."""
117 super().__init__(game)
118 self.actions_history = []
119 self.open_edges = []
120 self.hands = [[] for _ in range(_NUM_PLAYERS)]
121 self.deck = copy.deepcopy(_DECK)
122 self._game_over = False
123 self._next_player = pyspiel.PlayerId.CHANCE
124 self._current_deal_player = 0 # NEW ATTRIBUTE

126 # OpenSpiel (PySpiel) API functions are below. This is the standard set that
127 # should be implemented by every sequential-move game with chance.

129 def current_player(self):
130 """Returns id of the next player to move, or TERMINAL if game is over."""
131 if self._game_over:
132 return pyspiel.PlayerId.TERMINAL
133 if self.deck: # deal phase
134 return pyspiel.PlayerId.CHANCE
135 return self._next_player
136
137 def _legal_actions(self, player):
138 """Returns a list of legal actions, sorted in ascending order."""
139 assert player >= 0
140 assert player == self._next_player
141 return self.get_legal_actions(player)
142
143 def get_legal_actions(self, player):
144 """Returns a list of legal actions."""
145 assert player >= 0
146
147 actions = []
148 hand = self.hands[player]
149
150 # first move, no open edges
151 if not self.open_edges:
152 for tile in hand:
153 actions.append(Action(player, tile, None))
154 else:
155 for tile in hand:
156 if tile[0] in self.open_edges:
157 actions.append(Action(player, tile, tile[0]))
158 if tile[0] != tile[1] and tile[1] in self.open_edges:
159 actions.append(Action(player, tile, tile[1]))
160
161 actions_idx = [_ACTIONS_STR.index(str(action)) for action in actions]
162 actions_idx.sort()
163 return actions_idx
164
165 def chance_outcomes(self):
166 """Returns the possible chance outcomes and their probabilities."""
167 assert self.is_chance_node()
168 p = 1.0 / len(self.deck)
169 return [(_DECK.index(i), p) for i in self.deck]

171 def _apply_action(self, action):
172 """Applies the specified action to the state."""
173 if self.is_chance_node():
174 # Deal tiles to players in order (0, 1, 2, 3)
175 hand_to_add_tile = self.hands[self._current_deal_player]
176 tile = _DECK[action]
177 self.deck.remove(tile)
178 hand_to_add_tile.append(tile)
179 self._current_deal_player = (self._current_deal_player + 1) % 4
180
181 # Check if all hands are of _HAND_SIZE
182 if not all(len(hand) == _HAND_SIZE for hand in self.hands):
183 return # more tiles to deal
184
185 for hand in self.hands:
186 hand.sort()
187
188 self._next_player = 0
189 else:
190 action = _ACTIONS[action]
191 self.actions_history.append(action)
192 my_idx = self.current_player()
193 my_hand = self.hands[my_idx]
194 my_hand.remove(action.tile)
195 self.update_open_edges(action)
196
197 if not my_hand:
198 self._game_over = True # player played his last tile
199 return
200
201 for i in range(1, 5):
202 next_idx = (my_idx + i) % 4
203 next_legal_actions = self.get_legal_actions(next_idx)
204
205 if next_legal_actions:
206 self._next_player = next_idx
207 return
208
209 # Check if a team has played all their tiles.
210 if not (self.hands[0] or self.hands[2]) or not (
211 self.hands[1] or self.hands[3]
212 ):
213 self._game_over = True
214 return
215
216 # all players are blocked. Game is stuck.
217 self._game_over = True
218
219 def update_open_edges(self, action):
220 if not self.open_edges:
221 self.open_edges = list(action.tile)
222 else:
223 self.open_edges.remove(action.edge)
224 new_edge = (
225 action.tile[0] if action.tile[0] != action.edge else action.tile[1]
226 )
227 self.open_edges.append(new_edge)
228
229 self.open_edges.sort()

231 def _action_to_string(self, player, action):
232 """Action -> string."""
233 if player == pyspiel.PlayerId.CHANCE:
234 return f"Deal {_DECK[action]}"
235 return _ACTIONS_STR[action]

237 def is_terminal(self):
238 """Returns True if the game is over."""
239 return self._game_over

241 def returns(self):
242 """Total reward for each player over the course of the game so far."""
243 if not self.is_terminal():
244 return [0 for _ in range(_NUM_PLAYERS)]
245
246 sum_of_pips0 = sum(t[0] + t[1] for t in (self.hands[0] + self.hands[2]))
247 sum_of_pips1 = sum(t[0] + t[1] for t in (self.hands[1] + self.hands[3]))
248
249 if sum_of_pips1 == sum_of_pips0:
250 return [0 for _ in range(_NUM_PLAYERS)]
251
252 if sum_of_pips1 > sum_of_pips0:
253 return [sum_of_pips1, -sum_of_pips1, sum_of_pips1, -sum_of_pips1]
254 return [-sum_of_pips0, sum_of_pips0, -sum_of_pips0, sum_of_pips0]

256 def __str__(self):
257 """String for debug purposes. No particular semantics are required."""
258 hand0 = [str(c) for c in self.hands[0]]
259 hand1 = [str(c) for c in self.hands[1]]
260 hand2 = [str(c) for c in self.hands[2]]
261 hand3 = [str(c) for c in self.hands[3]]
262 board = self.draw_board()
263 return (
264 f"hand0:{hand0}\n"
265 f"hand1:{hand1}\n"
266 f"hand2:{hand2}\n"
267 f"hand3:{hand3}\n\n"
268 f"board: {board}"

271 def draw_board(self):
272 """Draw the board' in a human readable format."""
273 board = collections.deque()
274 current_open_edges = None
275 for action in self.actions_history:
276 # check if action is played on an empty board
277 if action.edge is None:
278 board.append(action.tile)
279 # pylint: disable=unused-variable
280 current_open_edges = list(action.tile)
281 # check if action edge matches last played edge in the left or right
282 elif action.edge == current_open_edges[0]:
283 # invert the tile if the edge is on the right:
284 tile = (
285 (action.tile[1], action.tile[0])
286 if action.tile[0] == current_open_edges[0]
287 else action.tile
288 )
289 board.appendleft(tile)
290
291 elif action.edge == current_open_edges[1]:
292 # invert the tile if the edge is on the left:
293 tile = (
294 (action.tile[1], action.tile[0])
295 if action.tile[1] == current_open_edges[1]
296 else action.tile
297 )
298 board.append(tile)
299
300 current_open_edges = board[0][0], board[-1][1]
301
302 # TODO(someone): move this to a test
303 assert len(board) == len(self.actions_history)
304 return list(board)

307class DominoesObserver:
308 """Observer, conforming to the PyObserver interface (see observation.py)."""
309
310 def __init__(self, iig_obs_type, params):
311 """Initializes an empty observation tensor."""
312 if params:
313 raise ValueError(f"Observation parameters not supported; passed {params}")
314
315 # Determine which observation pieces we want to include.
316 pieces = [("player", 4, (4,))]
317
318 if iig_obs_type.private_info == pyspiel.PrivateInfoType.SINGLE_PLAYER:
319 # each tile is represented using 3 integers:
320 # 2 for the pips, and 1 to distinguish between (0,0) to empty slot for
321 # a tile.
322 pieces.append(("hand", 21, (7, 3))) # 7 tiles per hand
323 if iig_obs_type.public_info:
324 if iig_obs_type.perfect_recall:
325 # list of all played actions, each action is represented using 5
326 # integers:
327 # 2 for the played tile (0-6),
328 # 1 for the covered edge (0-6),
329 # 1 for which player (0,1,3,4),
330 # 1 to distinguish between actual move and empty slot for a move (0/1).
331 # the None (play on an empty board) edge represented using 0.
332 pieces.append(("actions_history", 125, (25, 5)))
333 else:
334 # last action, represented in the same way as in "actions_history"
335 # but without the last integer.
336 pieces.append(("last_action", 4, (4,)))
337 pieces.append(("hand_sizes", 4, (4,)))
338
339 # Build the single flat tensor.
340 total_size = sum(size for name, size, shape in pieces)
341 self.tensor = np.zeros(total_size, np.float32)
342
343 # Build the named & reshaped views of the bits of the flat tensor.
344 self.dict = {}
345 index = 0
346 for name, size, shape in pieces:
347 self.dict[name] = self.tensor[index : index + size].reshape(shape)
348 index += size
349
350 def copy_indices(self, dest, source, index_list):
351 for idx in index_list:
352 dest[idx] = source[idx]
353
354 def set_from(self, state, player):
355 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
356
357 self.tensor.fill(0)
358
359 if "player" in self.dict:
360 self.dict["player"][player] = 1
361 self.dict["player"][1 - player] = 0
362
363 if "hand_sizes" in self.dict:
364 my_hand_size = len(state.hands[player])
365 opp_hand_size = len(state.hands[1 - player])
366 self.dict["hand_sizes"][0] = my_hand_size
367 self.dict["hand_sizes"][1] = opp_hand_size
368
369 if "edges" in self.dict:
370 if state.open_edges:
371 self.copy_indices(self.dict["edges"], state.open_edges, [0, 1])
372 else:
373 self.dict["edges"][0] = 0.0
374 self.dict["edges"][1] = 0.0
375
376 if "hand" in self.dict:
377 for i, tile in enumerate(state.hands[player]):
378 self.copy_indices(self.dict["hand"][i], tile, [0, 1])
379 self.dict["hand"][i][2] = 1.0
380
381 if "actions_history" in self.dict:
382 for i, action in enumerate(state.actions_history):
383 self.copy_indices(self.dict["actions_history"][i], action.tile, [0, 1])
384 self.dict["actions_history"][i][2] = (
385 action.edge if action.edge is not None else 0.0
386 )
387 self.dict["actions_history"][i][3] = action.player
388 self.dict["actions_history"][i][4] = 1.0
389
390 if "last_action" in self.dict:
391 if state.actions_history:
392 action = state.actions_history[-1]
393 self.copy_indices(self.dict["last_action"], action.tile, [0, 1])
394 self.dict["last_action"][2] = (
395 action.edge if action.edge is not None else 0.0
396 )
397 self.dict["last_action"][3] = action.player
398
399 def string_from(self, state, player):
400 """Observation of `state` from the PoV of `player`, as a string."""
401 pieces = []
402 if "player" in self.dict:
403 pieces.append(f"p{player}")
404 if "hand" in self.dict:
405 pieces.append(f"hand:{state.hands[player]}")
406 if "actions_history" in self.dict:
407 pieces.append(f"history:{str(state.actions_history)}")
408 if "last_action" in self.dict and state.actions_history:
409 pieces.append(f"last_action:{str(state.actions_history[-1])}")
410 return " ".join(str(p) for p in pieces)
