16"""Find Nash equilibria for constant- or general-sum 2-player games.
17
18Non-matrix games are handled by computing the normal (bimatrix) form.
19The algorithms used are:
20* direct computation of pure equilibria.
21* linear programming to find equilibria for constant-sum games.
22* iterated dominance to reduce the action space.
23* reverse search vertex enumeration (if using lrsnash) to find all general-sum
24 equilibria.
25* support enumeration (if using nashpy) to find all general-sum equilibria.
26* Lemke-Howson enumeration (if using nashpy) to find one general-sum
27 equilibrium.
28The general-sum mixed-equilibrium algorithms are likely to work well for tens of
29actions, but less likely to scale beyond that.
30"""

47def lrs_solve(row_payoffs, col_payoffs, lrsnash_max_denom, lrsnash_path):
48 """Find all Nash equilibria using the lrsnash solver.
49
50 `lrsnash` uses reverse search vertex enumeration on rational polytopes.
51 For more info, see: http://cgm.cs.mcgill.ca/~avis/C/lrslib/USERGUIDE.html#nash
52
53 Args:
54 row_payoffs: payoffs for row player
55 col_payoffs: payoffs for column player
56 lrsnash_max_denom: maximum denominator
57 lrsnash_path: path for temporary files
58
59 Yields:
60 (row_mixture, col_mixture), numpy vectors of float64s.
61 """
62 num_rows, num_cols = row_payoffs.shape
63 game_file, game_file_path = tempfile.mkstemp()
64 try:
65 game_file = os.fdopen(game_file, "w")
66
67 # write dimensions
68 game_file.write("%d %d\n\n" % (num_rows, num_cols))
69
70 # write row-player payoff matrix as fractions
71 for row in range(num_rows):
72 game_file.write(
73 " ".join(to_fraction_str(row_payoffs[row], lrsnash_max_denom)) + "\n")
74 game_file.write("\n")
75
76 # write col-player payoff matrix as fractions
77 for row in range(num_rows):
78 game_file.write(
79 " ".join(to_fraction_str(col_payoffs[row], lrsnash_max_denom)) + "\n")
80 game_file.write("\n")
81 game_file.close()
82 lrs = subprocess.Popen([lrsnash_path or "lrsnash", "-s", game_file_path],
83 stdin=subprocess.PIPE,
84 stdout=subprocess.PIPE,
85 stderr=subprocess.STDOUT)
86 col_mixtures = []
87 for line in lrs.stdout:
88 if len(line) <= 1 or line[:1] == b"*":
89 continue
90 line = np.asfarray([fractions.Fraction(x) for x in line.decode().split()])
91 if line[0] == 2: # col-player
92 col_mixtures.append(line[1:-1])
93 else: # row-player
94 row_mixture = line[1:-1]
95 # row-mixture forms a Nash with every col-mixture listed directly above
96 for col_mixture in col_mixtures:
97 yield (row_mixture, col_mixture)
98 col_mixtures = []
99 finally:
100 os.remove(game_file_path)

103def lemke_howson_solve(row_payoffs, col_payoffs):
104 """Find Nash equilibria using the Lemke-Howson algorithm.
105
106 The algorithm is not guaranteed to find all equilibria. Also it can yield
107 wrong answers if the game is degenerate (but raises warnings in that case).
108 Args:
109 row_payoffs: payoffs for row player
110 col_payoffs: payoffs for column player
111 Yields:
112 (row_mixture, col_mixture), numpy vectors of float64s.
113 """
114
115 showwarning = warnings.showwarning
116 warned_degenerate = [False]
117
118 def showwarning_check_degenerate(message, *args, **kwargs):
119 if "Your game could be degenerate." in str(message):
120 warned_degenerate[0] = True
121 showwarning(message, *args, **kwargs)
122
123 try:
124 warnings.showwarning = showwarning_check_degenerate
125 for row_mixture, col_mixture in nashpy.Game(
126 row_payoffs, col_payoffs).lemke_howson_enumeration():
127 if warned_degenerate[0]:
128 # attempt to discard obviously-wrong results
129 if (row_mixture.shape != row_payoffs.shape[:1] or
130 col_mixture.shape != row_payoffs.shape[1:]):
131 warnings.warn("Discarding ill-shaped solution.")
132 continue
133 if (not np.isfinite(row_mixture).all() or
134 not np.isfinite(col_mixture).all()):
135 warnings.warn("Discarding non-finite solution.")
136 continue
137 yield row_mixture, col_mixture
138 finally:
139 warnings.showwarning = showwarning
