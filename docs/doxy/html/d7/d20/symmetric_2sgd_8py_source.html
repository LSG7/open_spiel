29 def __init__(self, temperature=0., proj_grad=True, euclidean=False,
30 lrs=(1e-1,), rnd_init=False, seed=None, **kwargs):
31 """Ctor."""
32 del kwargs
33 super().__init__(proj_grad, euclidean, rnd_init, seed)
34 if temperature < 0.:
35 raise ValueError('temperature must be non-negative')
36 self.temperature = temperature
37 self.lrs = lrs
38 self.num_estimates = 2

40 if temperature > 0:
41 self.eps = np.exp(-1 / temperature) # ensure dist[i] >= eps / dim(dist)

49 def init_vars(self, num_strats, num_players):
50 """Initialize solver parameters."""
51 self.num_players = num_players
52 if self.rnd_init:
53 init_dist = self.random.rand(num_strats)

56 init_dist /= init_dist.sum()

87def gradients(dist, payoff_matrices, num_players, temperature=0.,
88 proj_grad=True):
89 """Computes exploitablity gradient.
90
91 Assumption: eta_k = 1 for all k
92
93 Args:
94 dist: 1-d np.array, current estimate of nash distribution
95 payoff_matrices: 2 (>=2 x A x A) np.arrays, payoffs for each joint action
96 num_players: int, number of players, in case payoff_matrices is abbreviated
97 temperature: non-negative float, default 0.
98 proj_grad: bool, if True, projects dist gradient onto simplex
99 Returns:
100 gradient of exploitability w.r.t. (dist) as tuple
101 unregularized exploitability (stochastic estimate)
102 shannon regularized exploitability (stochastic estimate)
103 """
104 del num_players
105 tau = temperature
106
107 a, b = 0, 1 # 2 samples needed for unbiased estimation
108 p_0, p_1 = 0, 1 # player 0 index, player 1 index
109 hess_0_01_a = payoff_matrices[a][p_0]
110 hess_1_01_a = payoff_matrices[a][p_1]
111 hess_0_01_b = payoff_matrices[b][p_0]
112
113 pg_0_a = simplex.project_grad(hess_0_01_a.dot(dist))
114 pg_0_b = simplex.project_grad(hess_0_01_b.dot(dist))
115
116 unreg_exp = np.dot(pg_0_a, pg_0_b)
117
118 if tau > 0.:
119 log_dist_safe = np.clip(np.log(dist), -40, 0)
120 entr_grad_proj = simplex.project_grad(-tau * (log_dist_safe + 1))
121 else:
122 entr_grad_proj = 0.
123 pg_0_a_entr = pg_0_a + entr_grad_proj
124 pg_0_b_entr = pg_0_b + entr_grad_proj
125 pg_0_entr = 0.5 * (pg_0_a_entr + pg_0_b_entr)
126 pg_1_b_entr = pg_0_b_entr
127
128 reg_exp = np.dot(pg_0_a_entr, pg_0_b_entr)
129
130 # then construct unbiased stochastic gradient
131 grad_dist = 2. * hess_1_01_a.dot(pg_1_b_entr)
132 if tau > 0.:
133 grad_dist += 2. * -tau * pg_0_entr / dist
134
135 if proj_grad:
136 grad_dist = simplex.project_grad(grad_dist)
137
138 return (grad_dist,), unreg_exp, reg_exp
