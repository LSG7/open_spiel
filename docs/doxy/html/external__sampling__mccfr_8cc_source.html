71void ExternalSamplingMCCFRSolver::RunIteration(std::mt19937* rng) {
72 for (auto p = Player{0}; p < game_->NumPlayers(); ++p) {
73 UpdateRegrets(*game_->NewInitialState(), p, rng);
74 }
75
76 if (avg_type_ == AverageType::kFull) {
77 std::vector<double> reach_probs(game_->NumPlayers(), 1.0);
78 FullUpdateAverage(*game_->NewInitialState(), reach_probs);
79 }
80}

82std::string ExternalSamplingMCCFRSolver::Serialize(
83 int double_precision, std::string delimiter) const {
84 SPIEL_CHECK_GE(double_precision, -1);
85 std::string str = "";
86 // Meta section
87 absl::StrAppend(&str,
88 "# Automatically generated by OpenSpiel "
89 "ExternalSamplingMCCFRSolver::Serialize\n");
90 absl::StrAppend(&str, kSerializeMetaSectionHeader, "\n");
91 absl::StrAppend(&str, "Version: ", kSerializationVersion, "\n");
92 absl::StrAppend(&str, "\n");
93 // Game section
94 absl::StrAppend(&str, kSerializeGameSectionHeader, "\n");
95 absl::StrAppend(&str, game_->Serialize(), "\n");
96 // Internal solver state section
97 absl::StrAppend(&str, kSerializeSolverTypeSectionHeader, "\n");
98 absl::StrAppend(&str, "ExternalSamplingMCCFRSolver", "\n");
99 absl::StrAppend(&str, kSerializeSolverSpecificStateSectionHeader, "\n");
100 // RNG section
101 absl::StrAppend(&str, kSerializeSolverRNGSectionHeader, "\n");
102 std::ostringstream rng_stream;
103 rng_stream << *rng_;
104 absl::StrAppend(&str, rng_stream.str(), "\n");
105 // Average type section
106 absl::StrAppend(&str, kSerializeSolverAverageTypeSectionHeader, "\n");
107 if (avg_type_ == AverageType::kFull)
108 absl::StrAppend(&str, "FullAverageType", "\n");
109 else if (avg_type_ == AverageType::kSimple)
110 absl::StrAppend(&str, "SimpleAverageType", "\n");
111 // Default policy section
112 absl::StrAppend(&str, kSerializeSolverDefaultPolicySectionHeader, "\n");
113 absl::StrAppend(&str, default_policy_->Serialize(double_precision, delimiter),
114 "\n");
115 // Info state values table section
116 absl::StrAppend(&str, kSerializeSolverValuesTableSectionHeader, "\n");
117 SerializeCFRInfoStateValuesTable(info_states_, &str, double_precision,
118 delimiter);
119 return str;
120}

122double ExternalSamplingMCCFRSolver::UpdateRegrets(const State& state,
123 Player player,
124 std::mt19937* rng) {
125 if (state.IsTerminal()) {
126 return state.PlayerReturn(player);
127 } else if (state.IsChanceNode()) {
128 Action action = SampleAction(state.ChanceOutcomes(), dist_(*rng)).first;
129 return UpdateRegrets(*state.Child(action), player, rng);
130 } else if (state.IsSimultaneousNode()) {
131 SpielFatalError(
132 "Simultaneous moves not supported. Use "
133 "TurnBasedSimultaneousGame to convert the game first.");
134 }
135
136 Player cur_player = state.CurrentPlayer();
137 std::string is_key = state.InformationStateString(cur_player);
138 std::vector<Action> legal_actions = state.LegalActions();
139
140 // The insert here only inserts the default value if the key is not found,
141 // otherwise returns the entry in the map.
142 auto iter_and_result = info_states_.insert(
143 {is_key, CFRInfoStateValues(legal_actions, kInitialTableValues)});
144
145 CFRInfoStateValues info_state_copy = iter_and_result.first->second;
146 info_state_copy.ApplyRegretMatching();
147
148 double value = 0;
149 std::vector<double> child_values(legal_actions.size(), 0);
150
151 if (cur_player != player) {
152 // Sample at opponent nodes.
153 int aidx = info_state_copy.SampleActionIndex(0.0, dist_(*rng));
154 value = UpdateRegrets(*state.Child(legal_actions[aidx]), player, rng);
155 } else {
156 // Walk over all actions at my nodes
157 for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
158 child_values[aidx] =
159 UpdateRegrets(*state.Child(legal_actions[aidx]), player, rng);
160 value += info_state_copy.current_policy[aidx] * child_values[aidx];
161 }
162 }
163
164 // Now the regret and avg strategy updates.
165 CFRInfoStateValues& info_state = info_states_[is_key];
166
167 if (cur_player == player) {
168 // Update regrets
169 for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
170 info_state.cumulative_regrets[aidx] += (child_values[aidx] - value);
171 }
172 }
173
174 // Simple average does averaging on the opponent node. To do this in a game
175 // with more than two players, we only update the player + 1 mod num_players,
176 // which reduces to the standard rule in 2 players.
177 if (avg_type_ == AverageType::kSimple &&
178 cur_player == ((player + 1) % game_->NumPlayers())) {
179 for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
180 info_state.cumulative_policy[aidx] +=
181 info_state_copy.current_policy[aidx];
182 }
183 }
184
185 return value;
186}

188void ExternalSamplingMCCFRSolver::FullUpdateAverage(
189 const State& state, const std::vector<double>& reach_probs) {
190 if (state.IsTerminal()) {
191 return;
192 } else if (state.IsChanceNode()) {
193 for (Action action : state.LegalActions()) {
194 FullUpdateAverage(*state.Child(action), reach_probs);
195 }
196 return;
197 } else if (state.IsSimultaneousNode()) {
198 SpielFatalError(
199 "Simultaneous moves not supported. Use "
200 "TurnBasedSimultaneousGame to convert the game first.");
201 }
202
203 // If all the probs are zero, no need to keep going.
204 double sum = std::accumulate(reach_probs.begin(), reach_probs.end(), 0.0);
205 if (sum == 0.0) return;
206
207 Player cur_player = state.CurrentPlayer();
208 std::string is_key = state.InformationStateString(cur_player);
209 std::vector<Action> legal_actions = state.LegalActions();
210
211 // The insert here only inserts the default value if the key is not found,
212 // otherwise returns the entry in the map.
213 auto iter_and_result = info_states_.insert(
214 {is_key, CFRInfoStateValues(legal_actions, kInitialTableValues)});
215
216 CFRInfoStateValues info_state_copy = iter_and_result.first->second;
217 info_state_copy.ApplyRegretMatching();
218
219 for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
220 std::vector<double> new_reach_probs = reach_probs;
221 new_reach_probs[cur_player] *= info_state_copy.current_policy[aidx];
222 FullUpdateAverage(*state.Child(legal_actions[aidx]), new_reach_probs);
223 }
224
225 // Now update the cumulative policy.
226 CFRInfoStateValues& info_state = info_states_[is_key];
227 for (int aidx = 0; aidx < legal_actions.size(); ++aidx) {
228 info_state.cumulative_policy[aidx] +=
229 (reach_probs[cur_player] * info_state_copy.current_policy[aidx]);
230 }
231}

233std::unique_ptr<ExternalSamplingMCCFRSolver>
234DeserializeExternalSamplingMCCFRSolver(const std::string& serialized,
235 std::string delimiter) {
236 auto partial = PartiallyDeserializeCFRSolver(serialized);
237 SPIEL_CHECK_EQ(partial.solver_type, "ExternalSamplingMCCFRSolver");
238
239 enum Section {
240 kInvalid = -1,
241 kRNG = 0,
242 kAverageType = 1,
243 kDefaultPolicy = 2
244 };
245 std::array<std::string, 3> section_strings = {"", "", ""};
246 Section current_section = kInvalid;
247
248 std::vector<absl::string_view> lines =
249 absl::StrSplit(partial.solver_specific_state, '\n');
250 for (int i = 0; i < lines.size(); i++) {
251 if (lines[i] == kSerializeSolverRNGSectionHeader) {
252 SPIEL_CHECK_EQ(current_section, kInvalid);
253 current_section = kRNG;
254 } else if (lines[i] == kSerializeSolverAverageTypeSectionHeader) {
255 SPIEL_CHECK_EQ(current_section, kRNG);
256 current_section = kAverageType;
257 } else if (lines[i] == kSerializeSolverDefaultPolicySectionHeader) {
258 SPIEL_CHECK_EQ(current_section, kAverageType);
259 current_section = kDefaultPolicy;
260 } else {
261 SPIEL_CHECK_NE(current_section, kInvalid);
262 absl::StrAppend(&section_strings[current_section], lines[i], "\n");
263 }
264 }
265 // Remove trailing newline characters
266 for (int i = 0; i < section_strings.size(); i++) {
267 section_strings[i].erase(section_strings[i].length() - 1);
268 }
269
270 // Do the actual deserialization for all internal state values
271 std::unique_ptr<std::mt19937> rng = std::make_unique<std::mt19937>();
272 std::istringstream rng_stream(section_strings[kRNG]);
273 rng_stream >> *rng;
274
275 AverageType average_type;
276 if (section_strings[kAverageType] == "FullAverageType")
277 average_type = AverageType::kFull;
278 else if (section_strings[kAverageType] == "SimpleAverageType")
279 average_type = AverageType::kSimple;
280
281 auto solver = std::make_unique<ExternalSamplingMCCFRSolver>(
282 partial.game, DeserializePolicy(section_strings[kDefaultPolicy]),
283 std::move(rng), average_type);
284 DeserializeCFRInfoStateValuesTable(partial.serialized_cfr_values_table,
285 &solver->InfoStateValuesTable(),
286 delimiter);
287 return solver;
288}
