53 def test_draw_hands(self):
54 """Tests hand drawing functions."""
55 game = liars_poker.LiarsPoker({"hand_length": 3, "num_digits": 3})
56 state = game.new_initial_state()
57 expected_hands = [[] for _ in range(game.num_players())]
58 for i in range(game.num_players() * game.hand_length):
59 # Verify we have chance nodes until all player hands are filled.
60 self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
61
62 # Draw a digit.
63 outcomes_with_probs = state.chance_outcomes()
64 action_list, prob_list = zip(*outcomes_with_probs)
65 action = np.random.choice(action_list, p=prob_list)
66
67 # Verify players' hands are filled correctly.
68 cur_player = i % game.num_players()
69 expected_hands[cur_player].append(action)
70 state.apply_action(action)
71 self.assertEqual(state.hands, expected_hands)
72 # Assert after all hands are filled, we have non-chance nodes.
73 cur_player = state.current_player()
74 self.assertNotEqual(cur_player, pyspiel.PlayerId.CHANCE)
75 self.assertEqual(cur_player, 0)

85 def test_basic_bid(self):
86 """Tests a single bid."""
87 game = liars_poker.LiarsPoker({"hand_length": 3, "num_digits": 3})
88 state = game.new_initial_state()
89 expected_bid_history = np.zeros(
90 (state.total_possible_bids, state.num_players())
91 )
92
93 # Fill players hands.
94 self._populate_game_hands(game, state)
95 # After all hands are filled, have player 0 bid.
96 cur_player = state.current_player()
97 action = 2
98 state.apply_action(action)
99
100 # Verify bid history is updated correctly.
101 bid_offset = liars_poker.BID_ACTION_OFFSET
102 expected_bid_history[action - bid_offset][cur_player] = 1
103 self.assertTrue((state.bid_history == expected_bid_history).all())
104
105 # Verify next set of legal bids is greater than the current bid.
106 for next_action in state.legal_actions():
107 if next_action == liars_poker.CHALLENGE_ACTION:
108 continue
109 self.assertGreater(next_action, action)

111 def _verify_returns(self, game, state):
112 self.assertTrue(state.winner() != -1 or state.loser() != -1)
113 actual_returns = state.returns()
114 if state.winner() != -1:
115 expected_returns = [-1.0 for _ in range(game.num_players())]
116 expected_returns[state.winner()] = game.num_players() - 1
117 else:
118 expected_returns = [1.0 for _ in range(game.num_players())]
119 expected_returns[state.loser()] = -1.0 * (game.num_players() - 1)
120 self.assertEqual(actual_returns, expected_returns)

122 def test_single_random_round(self):
123 """Runs a single round of bidding followed by a challenge."""
124 game = liars_poker.LiarsPoker({"hand_length": 3, "num_digits": 3})
125 state = game.new_initial_state()
126 expected_challenge_history = np.zeros(
127 (state.total_possible_bids, state.num_players())
128 )
129
130 # Fill players hands.
131 self._populate_game_hands(game, state)
132 # Have player 0 bid.
133 action = 2
134 state.apply_action(action)
135 # Verify challenge action is available to the next player.
136 challenge = liars_poker.CHALLENGE_ACTION
137 self.assertIn(challenge, state.legal_actions())
138 # Player 1 challenges.
139 cur_player = state.current_player()
140 state.apply_action(challenge)
141 bid_offset = liars_poker.BID_ACTION_OFFSET
142 expected_challenge_history[action - bid_offset][cur_player] = 1
143 # Verify challenge history is updated correctly.
144 self.assertTrue(
145 (state.challenge_history == expected_challenge_history).all()
146 )
147 # Original bidder challenges, thus agreeing to a count.
148 cur_player = state.current_player()
149 state.apply_action(challenge)
150 expected_challenge_history[action - bid_offset][cur_player] = 1
151 # Verify challenge history is updated correctly.
152 self.assertTrue(
153 (state.challenge_history == expected_challenge_history).all()
154 )
155
156 # Verify game is over.
157 self.assertTrue(state.is_terminal())
158 # Verify returns.
159 self._verify_returns(game, state)

161 def test_single_deterministic_round(self):
162 """Runs a single round where cards are dealt deterministically."""
163 game = liars_poker.LiarsPoker({"hand_length": 3, "num_digits": 3})
164 state = game.new_initial_state()
165
166 # Deal player 0 all "1" cards and player 1 all "2" cards.
167 for i in range(game.num_players() * game.hand_length):
168 if i % 2 == 0:
169 # Deal card to player 0
170 state.apply_action(1)
171 else:
172 # Deal card to player 1
173 state._apply_action(2)
174
175 # Have player 0 bid that there are four 1's.
176 state.apply_action(state.encode_bid(4, 1) + liars_poker.BID_ACTION_OFFSET)
177 # Player 1 challenges.
178 state.apply_action(liars_poker.CHALLENGE_ACTION)
179 # Player 0 accepts the challenge.
180 state.apply_action(liars_poker.CHALLENGE_ACTION)
181 # Verify game ends with player 0 losing.
182 self.assertTrue(state.is_terminal())
183 self.assertEqual(state.loser(), 0)
184 expected_returns = [1.0 for _ in range(game.num_players())]
185 expected_returns[state.loser()] = -1.0 * (game.num_players() - 1)
186 self.assertEqual(state.returns(), expected_returns)

265 def test_cloned_state_matches_original_state(self):

268 state = game.new_initial_state()

275 self.assertEqual(state.move_number(), clone.move_number())
276 self.assertEqual(state.num_distinct_actions(), clone.num_distinct_actions())
277
278 self.assertEqual(state._current_player, clone._current_player)
279 self.assertEqual(state._current_action, clone._current_action)
