26U256 LeftShift(U256 v, int s) {
27 if (s == 0) {
28 return v;
29 } else if (s < 128) {
30 return {(v.hi << s) | (v.lo >> (128 - s)), v.lo << s};
31 } else {
32 return {v.lo << (s - 128), 0};
33 }
34}

36MATCHER_P2(Eq256, hi, lo, "") { return arg.hi == hi && arg.lo == lo; }
37MATCHER_P(Eq256, v, "") { return arg.hi == v.hi && arg.lo == v.lo; }

39TEST(WideMultiplyTest, MultiplyU128ToU256Test) {
40 using absl::uint128;
41 constexpr uint128 k1 = 1;
42 constexpr uint128 kMax = ~static_cast<uint128>(0);
43
44 EXPECT_THAT(MultiplyU128ToU256(0, 0), Eq256(0, 0));
45
46 // Max uin128_t
47 EXPECT_THAT(MultiplyU128ToU256(kMax, kMax), Eq256(kMax << 1, 1));
48 EXPECT_THAT(MultiplyU128ToU256(kMax, 1), Eq256(0, kMax));
49 EXPECT_THAT(MultiplyU128ToU256(1, kMax), Eq256(0, kMax));
50 for (int i = 0; i < 64; ++i) {
51 SCOPED_TRACE(i);
52 EXPECT_THAT(MultiplyU128ToU256(kMax, k1 << i),
53 Eq256(LeftShift({0, kMax}, i)));
54 EXPECT_THAT(MultiplyU128ToU256(k1 << i, kMax),
55 Eq256(LeftShift({0, kMax}, i)));
56 }
57
58 // 1-bit x 1-bit.

60 for (int j = 0; j < 64; ++j) {
61 EXPECT_THAT(MultiplyU128ToU256(k1 << i, k1 << j),
