29 def eval_agent(env, agent, num_episodes):
30  """Evaluates `agent` for `num_episodes`."""
31  rewards = 0.0
32  for _ in range(num_episodes):
33  time_step = env.reset()
34  episode_reward = 0
35  while not time_step.last():
36  agent_output = agent.step(time_step, is_evaluation=True)
37  time_step = env.step([agent_output.action])
38  episode_reward += time_step.rewards[0]
39  rewards += episode_reward
40  return rewards / num_episodes
