145 def main(_):
146  setup_logging()
147 
148  batch_size = int(FLAGS.num_envs * FLAGS.num_steps)
149 
150  if FLAGS.game_name == "atari":
151  # pylint: disable=unused-import
152  # pylint: disable=g-import-not-at-top
153  import open_spiel.python.games.atari
154 
155  current_day = datetime.now().strftime("%d")
156  current_month_text = datetime.now().strftime("%h")
157  run_name = f"{FLAGS.game_name}__{FLAGS.exp_name}__"
158  if FLAGS.game_name == "atari":
159  run_name += f"{FLAGS.gym_id}__"
160  run_name += f"{FLAGS.seed}__{current_month_text}__{current_day}__{int(time.time())}"
161 
162  writer = SummaryWriter(f"runs/{run_name}")
163  writer.add_text(
164  "hyperparameters",
165  "|param|value|\n|-|-|\n%s" %
166  ("\n".join([f"|{key}|{value}|" for key, value in vars(FLAGS).items()])),
167  )
168 
169  random.seed(FLAGS.seed)
170  np.random.seed(FLAGS.seed)
171  torch.manual_seed(FLAGS.seed)
172  torch.backends.cudnn.deterministic = FLAGS.torch_deterministic
173 
174  device = torch.device(
175  "cuda" if torch.cuda.is_available() and FLAGS.cuda else "cpu")
176  logging.info("Using device: %s", str(device))
177 
178  if FLAGS.game_name == "atari":
179  envs = SyncVectorEnv([
180  make_single_atari_env(FLAGS.gym_id, FLAGS.seed + i, i, False,
181  run_name)() for i in range(FLAGS.num_envs)
182  ])
183  agent_fn = PPOAtariAgent
184  else:
185  envs = SyncVectorEnv([
186  make_single_env(FLAGS.game_name, FLAGS.seed + i)()
187  for i in range(FLAGS.num_envs)
188  ])
189  agent_fn = PPOAgent
190 
191  game = envs.envs[0]._game # pylint: disable=protected-access
192  info_state_shape = game.observation_tensor_shape()
193 
194  num_updates = FLAGS.total_timesteps // batch_size
195  agent = PPO(
196  input_shape=info_state_shape,
197  num_actions=game.num_distinct_actions(),
198  num_players=game.num_players(),
199  player_id=0,
200  num_envs=FLAGS.num_envs,
201  steps_per_batch=FLAGS.num_steps,
202  num_minibatches=FLAGS.num_minibatches,
203  update_epochs=FLAGS.update_epochs,
204  learning_rate=FLAGS.learning_rate,
205  gae=FLAGS.gae,
206  gamma=FLAGS.gamma,
207  gae_lambda=FLAGS.gae_lambda,
208  normalize_advantages=FLAGS.norm_adv,
209  clip_coef=FLAGS.clip_coef,
210  clip_vloss=FLAGS.clip_vloss,
211  entropy_coef=FLAGS.ent_coef,
212  value_coef=FLAGS.vf_coef,
213  max_grad_norm=FLAGS.max_grad_norm,
214  target_kl=FLAGS.target_kl,
215  device=device,
216  writer=writer,
217  agent_fn=agent_fn,
218  )
219 
220  n_reward_window = 50
221  recent_rewards = collections.deque(maxlen=n_reward_window)
222  time_step = envs.reset()
223  for update in range(num_updates):
224  for _ in range(FLAGS.num_steps):
225  agent_output = agent.step(time_step)
226  time_step, reward, done, unreset_time_steps = envs.step(
227  agent_output, reset_if_done=True)
228 
229  if FLAGS.game_name == "atari":
230  # Get around the fact that
231  # stable_baselines3.common.atari_wrappers.EpisodicLifeEnv will modify
232  # rewards at the LIFE and not GAME level by only counting
233  # rewards of finished episodes
234  for ts in unreset_time_steps:
235  info = ts.observations.get("info")
236  if info and "episode" in info:
237  real_reward = info["episode"]["r"]
238  writer.add_scalar("charts/player_0_training_returns", real_reward,
239  agent.total_steps_done)
240  recent_rewards.append(real_reward)
241  else:
242  for ts in unreset_time_steps:
243  if ts.last():
244  real_reward = ts.rewards[0]
245  writer.add_scalar("charts/player_0_training_returns", real_reward,
246  agent.total_steps_done)
247  recent_rewards.append(real_reward)
248 
249  agent.post_step(reward, done)
250 
251  if FLAGS.anneal_lr:
252  agent.anneal_learning_rate(update, num_updates)
253 
254  agent.learn(time_step)
255 
256  if update % FLAGS.eval_every == 0:
257  logging.info("-" * 80)
258  logging.info("Step %s", agent.total_steps_done)
259  logging.info("Summary of past %i rewards\n %s",
260  n_reward_window,
261  pd.Series(recent_rewards).describe())
262 
263  writer.close()
264  logging.info("All done. Have a pleasant day :)")

268  app.run(main)
