15"""Implementation of Fictitious Play from Perrin & al.
16
17Reference: https://arxiv.org/abs/2007.03458.
18As presented, the Fictitious Play algorithm provides a robust approximation
19scheme for Nash equilibrium by iteratively computing the best response
20against the distribution induced by the average of the past best responses.
21The provided formulation of Deep Fictitious Play mirrors this procedure,
22but substitutes out the exact best reponse computation with an approximation
23of best response values through a Reinforcement Learning approach (where
24the RL method in question is a user-determined parameter for each iteration).
25
26Policy is initialized to uniform policy.
27Each iteration:
28 1. Compute best response against policy
29 2. Update policy as weighted average of best response and current policy
30 (default learning rate is 1 / num_iterations + 1).
31
32To use fictitious play one should initialize it and run multiple iterations:
33fp = FictitiousPlay(game)
34for _ in range(num_iterations):
35 fp.iteration()
36policy = fp.get_policy()
37"""

53class MergedPolicy(policy_std.Policy):
54 """Merge several policies."""
55
56 def __init__(
57 self,
58 game,
59 player_ids: List[int],
60 policies: List[policy_std.Policy],
61 distributions: List[distribution_std.Distribution],
62 weights: List[float],
63 ):
64 """Initializes the merged policy.
65
66 Args:
67 game: The game to analyze.
68 player_ids: list of player ids for which this policy applies; each should
69 be in the range 0..game.num_players()-1.
70 policies: A `List[policy_std.Policy]` object.
71 distributions: A `List[distribution_std.Distribution]` object.
72 weights: A `List[float]` object. The elements should sum to 1.
73 """
74 super().__init__(game, player_ids)
75 self._policies = policies
76 self._distributions = distributions
77 self._weights = weights
78 assert len(policies) == len(distributions), (
79 f'Length mismatch {len(policies)} != {len(distributions)}')
80 assert len(policies) == len(weights), (
81 f'Length mismatch {len(policies)} != {len(weights)}')
82 assert math.isclose(
83 sum(weights),
84 1.0), (f'Weights should sum to 1, but instead sum to {sum(weights)}')
85
86 def action_probabilities(self, state, player_id=None):
87 action_prob = []
88 legal = state.legal_actions()
89 num_legal = len(legal)
90 for a in legal:
91 merged_pi = 0.0
92 norm_merged_pi = 0.0
93 for p, d, w in zip(self._policies, self._distributions, self._weights):
94 try:
95 merged_pi += w * d(state) * p(state)[a]
96 norm_merged_pi += w * d(state)
97 except (KeyError, ValueError):
98 # This happens when the state was not observed in the merged
99 # distributions or policies.
100 pass
101 if norm_merged_pi > 0.0:
102 action_prob.append((a, merged_pi / norm_merged_pi))
103 else:
104 action_prob.append((a, 1.0 / num_legal))
105 return dict(action_prob)

108class FictitiousPlay(object):
109 """Computes the value of a specified strategy."""
110
111 def __init__(self,
112 game: pyspiel.Game,
113 lr: Optional[float] = None,
114 temperature: Optional[float] = None):
115 """Initializes the greedy policy.
116
117 Args:
118 game: The game to analyze.
119 lr: The learning rate of mirror descent. If None, at iteration i it will
120 be set to 1/i.
121 temperature: If set, then instead of the greedy policy a softmax policy
122 with the specified temperature will be used to update the policy at each
123 iteration.
124 """
125 self._game = game
126 self._lr = lr
127 self._temperature = temperature
128 self._policy = policy_std.UniformRandomPolicy(self._game)
129
130 self._correlating_policy = self._policy
131 self._distribution = distribution.DistributionPolicy(
132 self._game, self._correlating_policy
133 )
134 self._fp_step = 0
135
136 def get_policy(self):
137 return self._policy
138
139 def get_correlating_policy(self):
140 return self._policy
141
142 def get_correlating_distribution(self):
143 return distribution.DistributionPolicy(self._game, self._policy)
144
145 def iteration(self, br_policy=None, learning_rate=None):
146 """Returns a new `TabularPolicy` equivalent to this policy.
147
148 Args:
149 br_policy: Policy to compute the best response value for each iteration.
150 If none provided, the exact value is computed.
151 learning_rate: The learning rate.
152 """
153 self._fp_step += 1
154
155 distrib = distribution.DistributionPolicy(self._game, self._policy)
156
157 if br_policy:
158 br_value = policy_value.PolicyValue(self._game, distrib, br_policy)
159 else:
160 br_value = best_response_value.BestResponse(
161 self._game, distrib, value.TabularValueFunction(self._game))
162
163 # Policy is either greedy or softmax with respect to the best response if
164 # temperature is specified.
165 player_ids = list(range(self._game.num_players()))
166 if self._temperature is None:
167 pi = greedy_policy.GreedyPolicy(self._game, player_ids, br_value)
168 else:
169 pi = softmax_policy.SoftmaxPolicy(self._game, player_ids,
170 self._temperature, br_value)
171 pi = pi.to_tabular()
172
173 distrib_pi = distribution.DistributionPolicy(self._game, pi)
174
175 if learning_rate:
176 weight = learning_rate
177 else:
178 weight = self._lr if self._lr else 1.0 / (self._fp_step + 1)
179
180 self._correlating_policy = pi
181 self._distribution = distrib_pi
182
183 if math.isclose(weight, 1.0):
184 self._policy = pi
185 else:
186 self._policy = MergedPolicy(self._game, player_ids, [self._policy, pi],
187 [distrib, distrib_pi],
188 [1.0 - weight, weight]).to_tabular()
