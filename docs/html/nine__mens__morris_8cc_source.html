169 void NineMensMorrisState::GetCurrentLegalActions() {
170  cur_legal_actions_.clear();
171 
172  if (capture_) {
173  Player opp = 1 - current_player_;
174  bool all_mills = CheckAllMills(opp);
175  for (int p = 0; p < kNumPoints; ++p) {
176  if (StateToPlayer(board_[p]) == opp) {
177  if (all_mills || !CheckInMill(p)) {
178  cur_legal_actions_.push_back(p);
179  }
180  }
181  }
182  } else {
183  if (men_to_deploy_[current_player_] > 0) {
184  // Still in phase 1.
185  for (int p = 0; p < kNumPoints; ++p) {
186  if (board_[p] == CellState::kEmpty) {
187  cur_legal_actions_.push_back(p);
188  }
189  }
190  } else if (num_men_[current_player_] > 3) {
191  // Phase 2.
192  for (int p = 0; p < kNumPoints; ++p) {
193  Player player = StateToPlayer(board_[p]);
194  if (player == current_player_) {
195  for (int dir = 0; dir < 4; ++dir) {
196  int np = kPointNeighbors[p][dir];
197  if (np > 0 && board_[np] == CellState::kEmpty) {
198  cur_legal_actions_.push_back(ToMoveAction(p, np));
199  }
200  }
201  }
202  }
203  absl::c_sort(cur_legal_actions_);
204  } else {
205  // Phase 3.
206  for (int p = 0; p < kNumPoints; ++p) {
207  Player player = StateToPlayer(board_[p]);
208  if (player == current_player_) {
209  for (int np = 0; np < kNumPoints; ++np) {
210  if (p == np) {
211  continue;
212  }
213 
214  if (board_[np] == CellState::kEmpty) {
215  cur_legal_actions_.push_back(ToMoveAction(p, np));
216  }
217  }
218  }
219  }
220  absl::c_sort(cur_legal_actions_);
221  }
222  }
223 }

225 bool NineMensMorrisState::CheckAllMills(Player player) const {
226  for (int p = 0; p < kNumPoints; ++p) {
227  if (StateToPlayer(board_[p]) == player) {
228  if (!CheckInMill(p)) {
229  return false;

231  }

236 bool NineMensMorrisState::CheckInMill(int pos) const {
237  Player player = StateToPlayer(board_[pos]);
238  if (player == kInvalidPlayer) {
239  return false;
240  }
241 
242  int cp = pos;
243 
244  // Direction base: North or East.
245  for (int dir_base = 0; dir_base < 2; ++dir_base) {
246  int total_matches = 0;
247 
248  // Try North + South, then East + West
249  for (int dir : {dir_base, dir_base + 2}) {
250  cp = pos;
251 
252  for (int i = 0; i < 2; ++i) {
253  cp = kPointNeighbors[cp][dir];
254  if (cp < 0 || StateToPlayer(board_[cp]) != player) {
255  break;
256  } else {
257  total_matches++;
258  }
259  }
260  }
261 
262  if (total_matches == 2) {
263  return true;
264  }
265  }
266 
267  return false;
268 }

270 void NineMensMorrisState::DoApplyAction(Action move) {
271  cur_legal_actions_.clear();
272  if (move < kNumPoints) {
273  if (capture_) {
274  // Capture move: choosing which piece to remove.
275  SPIEL_CHECK_TRUE(board_[move] != CellState::kEmpty);
276  Player opp = StateToPlayer(board_[move]);
277  SPIEL_CHECK_TRUE(opp == 1 - current_player_);
278  num_men_[opp]--;
279  board_[move] = CellState::kEmpty;
280  capture_ = false;
281  current_player_ = 1 - current_player_;
282  num_turns_++;
283  } else {
284  // Regular move in phase 1 (deployment)
285  SPIEL_CHECK_TRUE(board_[move] == CellState::kEmpty);
286  board_[move] = PlayerToState(current_player_);
287  SPIEL_CHECK_GT(men_to_deploy_[current_player_], 0);
288  men_to_deploy_[current_player_]--;
289  bool mill = CheckInMill(move);
290  if (mill) {
291  capture_ = true;
292  } else {
293  current_player_ = 1 - current_player_;
294  num_turns_++;
295  }
296  }
297  } else {
298  // Movement move (phase 2 or 3).
299  int from_pos = -1, to_pos = -1;
300  FromMoveAction(move, &from_pos, &to_pos);
301  SPIEL_CHECK_TRUE(StateToPlayer(board_[from_pos]) == current_player_);
302  SPIEL_CHECK_TRUE(board_[to_pos] == CellState::kEmpty);
303  board_[to_pos] = board_[from_pos];
304  board_[from_pos] = CellState::kEmpty;
305  bool mill = CheckInMill(to_pos);
306  if (mill) {
307  capture_ = true;
308  } else {
309  current_player_ = 1 - current_player_;
310  num_turns_++;
311  }
312  }
313 
314  if (cur_legal_actions_.empty()) {
315  GetCurrentLegalActions();
316  }
317 }

402 void NineMensMorrisState::ObservationTensor(Player player,
403  absl::Span<float> values) const {
404  SPIEL_CHECK_GE(player, 0);
405  SPIEL_CHECK_LT(player, num_players_);
406 
407  std::string templ =
408  ".--.--.\n"
409  "|.-.-.|\n"
410  "||...||\n"
411  "... ...\n"
412  "||...||\n"
413  "|.-.-.|\n"
414  ".--.--.\n";
415  int pos = 0;
416  TensorView<3> view(
417  values, {kCellStates + 2, kObservationSize, kObservationSize}, true);
418  for (int r = 0; r < kObservationSize; ++r) {
419  for (int c = 0; c < kObservationSize; ++c) {
420  int char_idx = r * 8 + c;
421  int plane = -1;
422  if (templ[char_idx] == '.') {
423  if (board_[pos] == CellState::kWhite) {
424  plane = 0;
425  } else if (board_[pos] == CellState::kBlack) {
426  plane = 1;
427  } else {
428  plane = 2;
429  }
430  pos++;
431  } else if (templ[char_idx] == '-') {
432  plane = 3;
433  } else if (templ[char_idx] == '|') {
434  plane = 4;
435  }
436 
437  if (plane >= 0) {
438  view[{plane, r, c}] = 1.0;
439  }
440  }
441  }
