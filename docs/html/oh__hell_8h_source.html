18 // The game of Oh Hell!.
19 // https://en.wikipedia.org/wiki/Oh_Hell
20 
21 // This is played by 3-7 players on a deck of up to 52 cards. It consists of a
22 // bidding phase followed by a play phase.
23 //
24 // Games start with a dealer dealing a specified number of cards to each player
25 // and then overturning a final card that is placed face up in the middle. The
26 // suit of this card becomes the 'trump' suit for the remainder of the game.
27 //
28 // In the bidding phase, players proceed clockwise, starting from the player
29 // to the left of the dealer, announcing how many tricks they think they can
30 // win. There is one catch: the total number of tricks bid by players cannot
31 // be equal to the actual number of tricks that will follow. For example, if
32 // 4 players are each dealt 5 cards, and the first 3 players bid 1, 2, 1, then
33 // the last player cannot bid 1.
34 //
35 // This is followed by the play phase, which proceeds as is standard in
36 // trick-taking games. Scoring is based on whether a player won exactly the
37 // number of tricks they bid. In this implementation, a player scores 1 point
38 // for every trick won and an additional 10 points if they won the exact number
39 // bid.
40 //

83  public:

86  num_cards_per_suit_(num_cards_per_suit) {}
87  int NumSuits() const { return num_suits_; }
88  int NumCardsPerSuit() const { return num_cards_per_suit_; }
89  int NumCards() const { return num_suits_ * num_cards_per_suit_; }
90  Suit CardSuit(int card) const {
91  if (num_suits_ <= 0) return Suit::kInvalidSuit;
92  return Suit(card % num_suits_);
93  }
94  int CardRank(int card) const {
95  if (num_suits_ <= 0) return kInvalidRank;
96  return card / num_suits_;
97  }
98  int Card(Suit suit, int rank) const {
99  return rank * num_suits_ + static_cast<int>(suit);
100  }
101  std::string CardString(int card) const {
102  return {kSuitChar[static_cast<int>(CardSuit(card))],
103  kRankChar[CardRank(card)]};
104  }

107  int num_suits_;
108  int num_cards_per_suit_;

216 class OhHellGame : public Game {
217  public:
218  explicit OhHellGame(const GameParameters& params);
219  int NumDistinctActions() const override {
220  return deck_props_.NumCards() + MaxNumTricks() + 1;
221  }
222  int MaxChanceOutcomes() const override { return deck_props_.NumCards(); }
223  std::unique_ptr<State> NewInitialState() const override {
224  return std::unique_ptr<State>(new OhHellState(
225  shared_from_this(),
226  /*num_players=*/num_players_,
227  /*deck_props=*/deck_props_,
228  /*num_tricks_fixed=*/num_tricks_fixed_,
229  /*off_bid_penalty=*/off_bid_penalty_,
230  /*points_per_trick=*/points_per_trick_));
231  }
232  int NumPlayers() const override { return num_players_; }
233  double MinUtility() const override {
234  if (off_bid_penalty_) return (- MaxNumTricks() * points_per_trick_);
235  return 0;
236  }
237  double MaxUtility() const override {
238  return MaxNumTricks() * points_per_trick_ + kMadeBidBonus;
239  }
240  // select dealer and number of tricks (kNumPreDealChanceActions)
241  // deal (MaxNumTricks() * num_players + kNumTrumpDeal)
242  // bidding (num_players)
243  // play (MaxNumTricks() * num_players)
244  int MaxGameLength() const override {
245  return 2 * MaxNumTricks() * num_players_ + num_players_
246  + kNumPreDealChanceActions + kNumTrumpDeal;
247  }
248  int MaxChanceNodesInHistory() const override {
249  return kNumPreDealChanceActions + MaxNumTricks() * num_players_
250  + kNumTrumpDeal;
251  }
252  std::vector<int> InformationStateTensorShape() const override;
253  // Given deck size, We can deal at most this many cards to each player and
254  // have an extra card to choose trump
255  int MaxNumTricks() const {
256  if (num_tricks_fixed_ > 0) return num_tricks_fixed_;
257  return (deck_props_.NumCards() - kNumTrumpDeal) / num_players_;
258  }
259 
260  private:
261  const int num_players_;
262  const DeckProperties deck_props_;
263  const int num_tricks_fixed_;
264  const bool off_bid_penalty_;
265  const int points_per_trick_;
