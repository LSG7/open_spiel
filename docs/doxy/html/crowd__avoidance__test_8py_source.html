156 def test_rewards(
157 self,
158 coef_congestion,
159 coef_target,
160 congestion_matrix,
161 players,
162 population,
163 initial_pos,
164 distributions,
165 expected_rewards,
166 init_distrib,
167 ):
168 game = pyspiel.load_game(
169 'python_mfg_crowd_avoidance',
170 {
171 'size': 2,
172 'coef_congestion': coef_congestion,
173 'coef_target': coef_target,
174 'congestion_matrix': ' '.join(
175 str(v) for v in congestion_matrix.flatten()
176 ),
177 'players': players,
178 'init_distrib': ' '.join(str(v) for v in init_distrib.flatten()),
179 'forbidden_states': '[]',
180 },
181 )
182 state = game.new_initial_state_for_population(population)
183 # Initial chance node.
184 self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
185 state.apply_action(crowd_avoidance.pos_to_merged(initial_pos, state.size))
186 self.assertEqual(state.current_player(), population)
187 npt.assert_array_equal(state.pos, initial_pos)
188 state.apply_action(state._NEUTRAL_ACTION)
189 npt.assert_array_equal(state.pos, initial_pos)
190 self.assertEqual(state.current_player(), pyspiel.PlayerId.CHANCE)
191 state.apply_action(state._NEUTRAL_ACTION)
192 self.assertEqual(state.current_player(), pyspiel.PlayerId.MEAN_FIELD)
193
194 # Maps states (in string representation) to their proba.
195 dist = {}
196 for x in range(state.size):
197 for y in range(state.size):
198 for pop in range(len(congestion_matrix)):
199 state_str = state.state_to_str(
200 np.array([x, y]),
201 state.t,
202 pop,
203 player_id=pyspiel.PlayerId.MEAN_FIELD,
204 )
205 dist[state_str] = distributions[pop][y][x]
206 support = state.distribution_support()
207 state.update_distribution([dist[s] for s in support])
208
209 # Decision node where we get a reward.
210 self.assertEqual(state.current_player(), population)
211 npt.assert_array_equal(state.rewards(), expected_rewards)
