43def batch_norm(training, updates, name):
44 """A batch norm layer.
45
46 Args:
47 training: A placeholder of whether this is done in training or not.
48 updates: A list to be extended with this layer's updates.
49 name: Name of the layer.
50
51 Returns:
52 A function to apply to the previous layer.
53 """
54 bn = tfkl.BatchNormalization(name=name)
55 def batch_norm_layer(x):
56 # This emits a warning that training is a placeholder instead of a concrete
57 # bool, but seems to work anyway.
58 applied = bn(x, training)
59 updates.extend(bn.updates)
60 return applied
61 return batch_norm_layer

64def residual_layer(inputs, num_filters, kernel_size, training, updates, name):
65 return cascade(inputs, [
66 conv_2d(num_filters, kernel_size, name=f"{name}_res_conv1"),
67 batch_norm(training, updates, f"{name}_res_batch_norm1"),
68 tfkl.Activation("relu"),
69 conv_2d(num_filters, kernel_size, name=f"{name}_res_conv2"),
70 batch_norm(training, updates, f"{name}_res_batch_norm2"),
71 lambda x: tfkl.add([x, inputs]),
72 tfkl.Activation("relu"),
73 ])

90class Losses(collections.namedtuple("Losses", "policy value l2")):
91 """Losses from a training step."""

93 @property

97 def __str__(self):
98 return ("Losses(total: {:.3f}, policy: {:.3f}, value: {:.3f}, "
99 "l2: {:.3f})").format(self.totaltotal, self.policy, self.value, self.l2)

101 def __add__(self, other):
102 return Losses(self.policy + other.policy,

110class Model(object):
111 """An AlphaZero style model with a policy and value head.
112
113 This supports three types of models: mlp, conv2d and resnet.
114
115 All models have a shared torso stack with two output heads: policy and value.
116 They have same meaning as in the AlphaGo Zero and AlphaZero papers. The resnet
117 model copies the one in that paper when set with width 256 and depth 20. The
118 conv2d model is the same as the resnet except uses a conv+batchnorm+relu
119 instead of the res blocks. The mlp model uses dense layers instead of conv,
120 and drops batch norm.
121
122 Links to relevant articles/papers:
123 https://deepmind.com/blog/article/alphago-zero-starting-scratch has an open
124 access link to the AlphaGo Zero nature paper.
125 https://deepmind.com/blog/article/alphazero-shedding-new-light-grand-games-chess-shogi-and-go
126 has an open access link to the AlphaZero science paper.
127
128 All are parameterized by their input (observation) shape and output size
129 (number of actions), though the conv2d and resnet might only work with games
130 that have spatial data (ie 3 non-batch dimensions, eg: connect four would
131 work, but not poker).
132
133 The depth is the number of blocks in the torso, where the definition of a
134 block varies by model. For a resnet it's a resblock which is two conv2ds,
135 batch norms and relus, and an addition. For conv2d it's a conv2d, a batch norm
136 and a relu. For mlp it's a dense plus relu.
137
138 The width is the number of filters for any conv2d and the number of hidden
139 units for any dense layer.
140
141 Note that this uses an explicit graph so that it can be used for inference
142 and training from C++. It seems to also be 20%+ faster than using eager mode,
143 at least for the unit test.
144 """

169 @classmethod
170 def build_model(cls, model_type, input_shape, output_size, nn_width, nn_depth,
171 weight_decay, learning_rate, path):
172 """Build a model with the specified params."""
173 if model_type not in cls.valid_model_types:
174 raise ValueError(f"Invalid model type: {model_type}, "
175 f"expected one of: {cls.valid_model_types}")
176
177 # The order of creating the graph, init, saver, and session is important!
178 # https://stackoverflow.com/a/40788998
179 g = tf.Graph() # Allow multiple independent models and graphs.
180 with g.as_default():
181 cls._define_graph(model_type, input_shape, output_size, nn_width,
182 nn_depth, weight_decay, learning_rate)
183 init = tf.variables_initializer(tf.global_variables(),
184 name="init_all_vars_op")
185 with tf.device("/cpu:0"): # Saver only works on CPU.
186 saver = tf.train.Saver(
187 max_to_keep=10000, sharded=False, name="saver")
188 session = tf.Session(graph=g)
189 session.__enter__()
190 session.run(init)
191 return cls(session, saver, path)

219 @staticmethod
220 def _define_graph(model_type, input_shape, output_size,
221 nn_width, nn_depth, weight_decay, learning_rate):
222 """Define the model graph."""
223 # Inference inputs
224 input_size = int(np.prod(input_shape))
225 observations = tf.placeholder(tf.float32, [None, input_size], name="input")
226 legals_mask = tf.placeholder(tf.bool, [None, output_size],
227 name="legals_mask")
228 training = tf.placeholder(tf.bool, name="training")
229
230 bn_updates = []
231
232 # Main torso of the network
233 if model_type == "mlp":
234 torso = observations # Ignore the input shape, treat it as a flat array.
235 for i in range(nn_depth):
236 torso = cascade(torso, [
237 tfkl.Dense(nn_width, name=f"torso_{i}_dense"),
238 tfkl.Activation("relu"),
239 ])
240 elif model_type == "conv2d":
241 torso = tfkl.Reshape(input_shape)(observations)
242 for i in range(nn_depth):
243 torso = cascade(torso, [
244 conv_2d(nn_width, 3, name=f"torso_{i}_conv"),
245 batch_norm(training, bn_updates, f"torso_{i}_batch_norm"),
246 tfkl.Activation("relu"),
247 ])
248 elif model_type == "resnet":
249 torso = cascade(observations, [
250 tfkl.Reshape(input_shape),
251 conv_2d(nn_width, 3, name="torso_in_conv"),
252 batch_norm(training, bn_updates, "torso_in_batch_norm"),
253 tfkl.Activation("relu"),
254 ])
255 for i in range(nn_depth):
256 torso = residual_layer(torso, nn_width, 3, training, bn_updates,
257 f"torso_{i}")
258 else:
259 raise ValueError("Unknown model type.")
260
261 # The policy head
262 if model_type == "mlp":
263 policy_head = cascade(torso, [
264 tfkl.Dense(nn_width, name="policy_dense"),
265 tfkl.Activation("relu"),
266 ])
267 else:
268 policy_head = cascade(torso, [
269 conv_2d(filters=2, kernel_size=1, name="policy_conv"),
270 batch_norm(training, bn_updates, "policy_batch_norm"),
271 tfkl.Activation("relu"),
272 tfkl.Flatten(),
273 ])
274 policy_logits = tfkl.Dense(output_size, name="policy")(policy_head)
275 policy_logits = tf.where(legals_mask, policy_logits,
276 -1e32 * tf.ones_like(policy_logits))
277 unused_policy_softmax = tf.identity(tfkl.Softmax()(policy_logits),
278 name="policy_softmax")
279 policy_targets = tf.placeholder(
280 shape=[None, output_size], dtype=tf.float32, name="policy_targets")
281 policy_loss = tf.reduce_mean(
282 tf.nn.softmax_cross_entropy_with_logits_v2(
283 logits=policy_logits, labels=policy_targets),
284 name="policy_loss")
285
286 # The value head
287 if model_type == "mlp":
288 value_head = torso # Nothing specific before the shared value head.
289 else:
290 value_head = cascade(torso, [
291 conv_2d(filters=1, kernel_size=1, name="value_conv"),
292 batch_norm(training, bn_updates, "value_batch_norm"),
293 tfkl.Activation("relu"),
294 tfkl.Flatten(),
295 ])
296 value_out = cascade(value_head, [
297 tfkl.Dense(nn_width, name="value_dense"),
298 tfkl.Activation("relu"),
299 tfkl.Dense(1, name="value"),
300 tfkl.Activation("tanh"),
301 ])
302 # Need the identity to name the single value output from the dense layer.
303 value_out = tf.identity(value_out, name="value_out")
304 value_targets = tf.placeholder(
305 shape=[None, 1], dtype=tf.float32, name="value_targets")
306 value_loss = tf.identity(tf.losses.mean_squared_error(
307 value_out, value_targets), name="value_loss")
308
309 l2_reg_loss = tf.add_n([
310 weight_decay * tf.nn.l2_loss(var)
311 for var in tf.trainable_variables()
312 if "/bias:" not in var.name
313 ], name="l2_reg_loss")
314
315 total_loss = policy_loss + value_loss + l2_reg_loss
316 optimizer = tf.train.AdamOptimizer(learning_rate)
317 with tf.control_dependencies(bn_updates):
318 unused_train = optimizer.minimize(total_loss, name="train")

342 def update(self, train_inputs: Sequence[TrainInput]):
343 """Runs a training step."""
344 batch = TrainInput.stack(train_inputs)
345
346 # Run a training step and get the losses.
347 _, policy_loss, value_loss, l2_reg_loss = self._session.run(
348 [self._train, self._policy_loss, self._value_loss, self._l2_reg_loss],
349 feed_dict={self._input: batch.observation,
350 self._legals_mask: batch.legals_mask,
351 self._policy_targets: batch.policy,
352 self._value_targets: batch.value,
353 self._training: True})
354
355 return Losses(policy_loss, value_loss, l2_reg_loss)
