44def eval_agents(env, agents, num_episodes):
45 """Evaluate the agents, returning a numpy array of average returns."""
46 rewards = np.array([0] * env.num_players, dtype=np.float64)
47 for _ in range(num_episodes):
48 time_step = env.reset()
49 while not time_step.last():
50 player_id = time_step.observations["current_player"]
51 agent_output = agents[player_id].step(time_step, is_evaluation=True)
52 time_step = env.step([agent_output.action])
53 for i in range(env.num_players):
54 rewards[i] += time_step.rewards[i]
55 rewards /= num_episodes
56 return rewards

59def create_epsilon_schedule(sched_str):
60 """Creates an epsilon schedule from the string as desribed in the flags."""
61 values = FLAGS.epsilon_schedule.split(",")
62 if values[0] == "linear":
63 assert len(values) == 4
64 return rl_tools.LinearSchedule(
65 float(values[1]), float(values[2]), int(values[3]))
66 elif values[0] == "constant":
67 assert len(values) == 2
68 return rl_tools.ConstantSchedule(float(values[1]))
69 else:
70 print("Unrecognized schedule string: {}".format(sched_str))
71 sys.exit()

74def main(_):
75 env = rl_environment.Environment(FLAGS.game)
76 num_players = env.num_players
77 num_actions = env.action_spec()["num_actions"]
78
79 agents = []
80 if FLAGS.epsilon_schedule is not None:
81 for idx in range(num_players):
82 agents.append(
83 tabular_qlearner.QLearner(
84 player_id=idx,
85 num_actions=num_actions,
86 epsilon_schedule=create_epsilon_schedule(FLAGS.epsilon_schedule)))
87 else:
88 agents = [
89 tabular_qlearner.QLearner(player_id=idx, num_actions=num_actions)
90 for idx in range(num_players)
91 ]
92
93 # 1. Train the agents
94 training_episodes = FLAGS.num_train_episodes
95 for cur_episode in range(training_episodes):
96 if cur_episode % int(FLAGS.eval_freq) == 0:
97 avg_rewards = eval_agents(env, agents, FLAGS.num_eval_episodes)
98 print("Training episodes: {}, Avg rewards: {}".format(
99 cur_episode, avg_rewards))
100 time_step = env.reset()
101 while not time_step.last():
102 player_id = time_step.observations["current_player"]
103 agent_output = agents[player_id].step(time_step)
104 time_step = env.step([agent_output.action])
105
106 # Episode is over, step all agents with final info state.
107 for agent in agents:
108 agent.step(time_step)
