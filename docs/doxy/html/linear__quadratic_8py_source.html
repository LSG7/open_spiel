76class MFGLinearQuadraticGame(pyspiel.Game):
77 """A Mean-Field Linear Quadratic game.
78
79 For now, only single-population setting is covered. A game starts by an
80 initial chance node that selects the initial state of the player in the MFG.
81 Then the game sequentially alternates between:
82 - An action selection node (where the player id is >= 0)
83 - A chance node (the player id is pyspiel.PlayerId.CHANCE)
84 - A Mean Field node (the player id is pyspiel.PlayerId.MEAN_FIELD)
85 """
86
87 # pylint:disable=dangerous-default-value
88 def __init__(self, params: Mapping[str, Any] = _DEFAULT_PARAMS):
89 self.size = params.get("size", _SIZE)
90 self.horizon = params.get("horizon", _HORIZON)
91 self.dt = params.get("dt", _DELTA_T)
92 self.n_actions_per_side = params.get(
93 "n_actions_per_side", _N_ACTIONS_PER_SIDE
94 )
95 self.volatility = params.get("volatility", _VOLATILITY)
96 self.mean_revert = params.get("mean_revert", _MEAN_REVERT)
97 self.cross_q = params.get("cross_q", _CROSS_Q)
98 self.kappa = params.get("kappa", _KAPPA)
99 self.terminal_cost = params.get("terminal_cost", _TERMINAL_COST)
100 self.spatial_bias = params.get("spatial_bias", _SPATIAL_BIAS)
101
102 game_info = pyspiel.GameInfo(
103 num_distinct_actions=2 * self.n_actions_per_side + 1,
104 max_chance_outcomes=2 * self.n_actions_per_side + 1,
105 num_players=_NUM_PLAYERS,
106 min_utility=-np.inf,
107 max_utility=+np.inf,
108 utility_sum=0.0,
109 max_game_length=self.horizon,

162 def state_to_str(self, x, tick, player_id=pyspiel.PlayerId.DEFAULT_PLAYER_ID):
163 """A string that uniquely identifies a triplet x, t, player_id."""
164 if self.x is None:
165 return "initial"
166
167 if self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
168 return "({}, {})".format(x, tick)
169 elif self._player_id == pyspiel.PlayerId.MEAN_FIELD:
170 return "({}, {})_a".format(x, tick)
171 elif self._player_id == pyspiel.PlayerId.CHANCE:
172 return "({}, {})_a_mu".format(x, tick)
173 raise ValueError(
174 "player_id is not mean field, chance or default player id."

184 def _legal_actions(self, player):
185 """Returns a list of legal actions for player and MFG nodes."""
186 if player == pyspiel.PlayerId.MEAN_FIELD:
187 return []
188 if (
189 player == pyspiel.PlayerId.DEFAULT_PLAYER_ID
190 and player == self.current_player()
191 ):
192 return list(range(self.n_actionsn_actions))
193 raise ValueError(
194 f"Unexpected player {player}. "
195 "Expected a mean field or current player 0."
196 )

198 def _apply_action(self, action):
199 """Applies the specified action to the state."""
200 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
201 raise ValueError(
202 "_apply_action should not be called at a MEAN_FIELD state."
203 )
204 self.return_value = self._rewards()
205
206 assert (
207 self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID
208 or self._player_id == pyspiel.PlayerId.CHANCE
209 )
210
211 if self.x is None:
212 self.x = action
213 self._player_id = pyspiel.PlayerId.DEFAULT_PLAYER_ID
214 return
215
216 if action < 0 or action >= self.n_actionsn_actions:
217 raise ValueError(
218 "The action is between 0 and {} at any node".format(self.n_actionsn_actions)
219 )
220
221 move = self.action_to_move(action)
222 if self._player_id == pyspiel.PlayerId.CHANCE:
223 self.x += move * math.sqrt(self.dt) * self.volatility
224 self.x = round(self.x) % self.size
225 self._player_id = pyspiel.PlayerId.MEAN_FIELD
226 self.tick += 1
227 elif self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
228 dist_mean = self.distribution_average() - self.x
229 full_move = move
230 full_move += self.mean_revert * dist_mean
231 full_move *= self.dt
232 self.x += round(full_move)
233 self.x = round(self.x) % self.size
234
235 self._last_action = action
236 self._player_id = pyspiel.PlayerId.CHANCE

238 def _action_to_string(self, player, action):
239 """Action -> string."""
240 del player
241 return str(action)

243 def action_to_move(self, action):
244 return action - self.n_actions_per_side
245
246 def actions_to_position(self):
247 return [a - self.n_actions_per_side for a in range(self.n_actionsn_actions)]

249 def chance_outcomes(self):
250 """Returns the possible chance outcomes and their probabilities."""
251 if self.x is None:
252 return list(enumerate(self._distribution))
253
254 a = np.array(self.actions_to_position())
255 gaussian_vals = scipy.stats.norm.cdf(
256 a + 0.5, scale=self.volatility
257 ) - scipy.stats.norm.cdf(a - 0.5, scale=self.volatility)
258 gaussian_vals[0] += (
259 scipy.stats.norm.cdf(a[0] - 0.5, scale=self.volatility) - 0.0
260 )
261 gaussian_vals[-1] += 1.0 - scipy.stats.norm.cdf(
262 a[-1] + 0.5, scale=self.volatility
263 )
264 return [
265 (act, p) for act, p in zip(list(range(self.n_actionsn_actions)), gaussian_vals)
266 ]

268 def distribution_support(self):
269 """return a list of state string."""
270 return [
271 self.state_to_str(i, self.tick, player_id=pyspiel.PlayerId.MEAN_FIELD)

275 def distribution_average(self):
276 """return the average of the distribution over the states: 0, ..., Size."""
277 states = np.arange(self.size)
278 pos = states * (self._distribution)
279 return np.sum(pos)

281 def update_distribution(self, distribution):
282 """This function is central and specific to the logic of the MFG.
283
284 Args:
285 distribution: a distribution to register. - function should be called
286 when the node is in MEAN_FIELD state. - distribution are probabilities
287 that correspond to each game state given by distribution_support.

290 raise ValueError(
291 "update_distribution should only be called at a MEAN_FIELD state."

310 def eta_t(self):
311 """Computes the theoretical policy's `eta_t` term."""
312 # pylint: disable=invalid-name
313 kappa = self.kappa
314 K = self.mean_revert
315 q = self.cross_q
316 c = self.terminal_cost
317 T = self.horizon
318 t = self.t
319
320 R = (K + q) ** 2 + (kappa - q**2)
321 deltap = -(K + q) + math.sqrt(R)
322 deltam = -(K + q) - math.sqrt(R)
323 numerator = -(kappa - q**2) * (
324 math.exp((deltap - deltam) * (T - t)) - 1
325 ) - c * (deltap * math.exp((deltap - deltam) * (T - t)) - deltam)
326 denominator = (
327 deltam * math.exp((deltap - deltam) * (T - t)) - deltap
328 ) - c * (math.exp((deltap - deltam) * (T - t)) - 1)
329 return numerator / denominator
330
331 def _rewards(self):
332 """Reward for the player for this state."""
333 if self._player_id == pyspiel.PlayerId.DEFAULT_PLAYER_ID:
334 dist_mean = self.distribution_average() - self.x
335
336 move = self.action_to_move(self._last_action)
337 action_reward = (
338 self.dt
339 / 2
340 * (
341 -(move**2)
342 + 2 * self.cross_q * move * dist_mean
343 - self.kappa * dist_mean**2
344 )
345 )
346
347 if self.is_terminal():
348 terminal_reward = -self.terminal_cost * dist_mean**2 / 2.0
349 return action_reward + terminal_reward
350 return action_reward
351
352 return 0.0

375class Observer:

377
378 def __init__(self, params, game):
379 """Initializes an empty observation tensor."""
380 del params
381
382 self.size = game.size
383 self.horizon = game.horizon
384 self.tensor = np.zeros(2, np.float32)
385 self.dict = {
386 "x": self.tensor[0],
387 "t": self.tensor[1],
388 "observation": self.tensor,
389 }
390
391 def set_from(self, state, player: int):
392 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
393 del player
394 # We update the observation via the shaped tensor since indexing is more
395 # convenient than with the 1-D tensor. Both are views onto the same memory.
396 self.tensor[0] = state.x
397 self.tensor[1] = state.t
398 # state.x is None for the initial (blank) state, don't set any
399 # position bit in that case.
400 if state.x is not None:
401 if not 0 <= state.x < self.size:
402 raise ValueError(
403 f"Expected {state} x position to be in [0, {self.size})"
404 )
405 self.dict["x"] = np.array([state.x])
406 if not 0 <= state.t <= self.horizon:
407 raise ValueError(f"Expected {state} time to be in [0, {self.horizon}]")
408 self.dict["t"] = np.array([state.t])

410 def string_from(self, state, player):
411 """Observation of `state` from the PoV of `player`, as a string."""
412 del player
413 return state.to_string()
