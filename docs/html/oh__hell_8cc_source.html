32 const GameType kGameType{
33  /*short_name=*/"oh_hell",
34  /*long_name=*/"Oh Hell!",
35  GameType::Dynamics::kSequential,
36  GameType::ChanceMode::kExplicitStochastic,
37  GameType::Information::kImperfectInformation,
38  GameType::Utility::kGeneralSum,
39  GameType::RewardModel::kTerminal,
40  /*max_num_players=*/kMaxNumPlayers,
41  /*min_num_players=*/kMinNumPlayers,
42  /*provides_information_state_string=*/true,
43  /*provides_information_state_tensor=*/true,
44  /*provides_observation_string=*/false,
45  /*provides_observation_tensor=*/false,
46  /*parameter_specification=*/
47  {
48  {"players", GameParameter(kMinNumPlayers)},
49  {"num_suits", GameParameter(kMaxNumSuits)},
50  {"num_cards_per_suit", GameParameter(kMaxNumCardsPerSuit)},
51  // number of tricks in the game, must be between 1 and
52  // (num_suits * num_cards_per_suit - 1) / num_players,
53  // default is to choose randomly in the legal range every game
54  {"num_tricks_fixed", GameParameter(kRandomNumTricks)},
55  // In case of no off-bid penalty, players receive `points_per_trick`
56  // per trick made, plus a bonus if their bid was correct.
57  // In case of an off-bid penalty, if a player missed their bid, they
58  // receive a penalty of `points_per_trick` times the number of tricks
59  // they are above or below their bid and only if the bid was correct
60  // they receive `points_per_trick` per trick made plus a bonus.
61  {"off_bid_penalty", GameParameter(false)},
62  {"points_per_trick", GameParameter(1)},
63  }};

75 OhHellGame::OhHellGame(const GameParameters& params)
76  : Game(kGameType, params),
77  num_players_(ParameterValue<int>("players")),
78  deck_props_(ParameterValue<int>("num_suits"),
79  ParameterValue<int>("num_cards_per_suit")),
80  num_tricks_fixed_(ParameterValue<int>("num_tricks_fixed")),
81  off_bid_penalty_(ParameterValue<bool>("off_bid_penalty")),
82  points_per_trick_(ParameterValue<int>("points_per_trick")) {
83  SPIEL_CHECK_TRUE(num_players_ >= kMinNumPlayers &&
84  num_players_ <= kMaxNumPlayers);
85  SPIEL_CHECK_TRUE(deck_props_.NumSuits() >= kMinNumSuits &&
86  deck_props_.NumSuits() <= kMaxNumSuits);
87  SPIEL_CHECK_TRUE(deck_props_.NumCardsPerSuit() >= kMinNumCardsPerSuit &&
88  deck_props_.NumCardsPerSuit() <= kMaxNumCardsPerSuit);
89  // need at least num_players + 1 cards
90  SPIEL_CHECK_TRUE(num_players_ <= deck_props_.NumCards() - kNumTrumpDeal);
91  SPIEL_CHECK_TRUE(num_tricks_fixed_ == kRandomNumTricks ||
92  (num_tricks_fixed_ >= kMinNumTricks &&
93  num_tricks_fixed_ <= MaxNumTricks()));
94 }

96 std::vector<int> OhHellGame::InformationStateTensorShape() const {
97  // initial chance actions (incl trump dealing)
98  int len = MaxNumTricks() + num_players_ + deck_props_.NumCards();
99  // initial hand and current hand
100  len += 2 * deck_props_.NumCards();
101  // bids, legal range is [no bid, 0, 1, ..., max legal bid]
102  len += num_players_ * (MaxNumTricks() + 2);
103  // tricks won so far
104  len += MaxNumTricks() * num_players_;
105  // tricks
106  len += MaxNumTricks() * (2 * num_players_ - 1) * deck_props_.NumCards();
107  return {len};
108 }

286 std::vector<Action> OhHellState::BiddingLegalActions() const {
287  int bid_sum = 0;
288  bool last_bidder = true;
289  for (Player player = 0; player < num_players_; ++player) {
290  if (player != current_player_) last_bidder &= bids_[player] != kInvalidBid;
291  bid_sum += std::max(0, bids_[player]);
292  }
293  std::vector<Action> legal_actions;
294  for (Action bid = 0; bid <= num_tricks_; ++bid) {
295  if (!last_bidder || bid + bid_sum != num_tricks_) {
296  legal_actions.push_back(bid + deck_props_.NumCards());
297  }
298  }
299  return legal_actions;
300 }

302 std::vector<Action> OhHellState::PlayLegalActions() const {
303  std::vector<Action> legal_actions;
304  legal_actions.reserve(num_tricks_ - num_cards_played_ / num_players_);
305 
306  // Check if we can follow suit.
307  if (num_cards_played_ % num_players_ != 0) {
308  auto suit = CurrentTrick().LedSuit();
309  for (int rank = 0; rank < deck_props_.NumCardsPerSuit(); ++rank) {
310  if (holder_[deck_props_.Card(suit, rank)] == current_player_) {
311  legal_actions.push_back(deck_props_.Card(suit, rank));
312  }
313  }
314  }
315  if (!legal_actions.empty()) return legal_actions;
316 
317  // Otherwise, we can play any of our cards.
318  for (int card = 0; card < deck_props_.NumCards(); ++card) {
319  if (holder_[card] == current_player_) legal_actions.push_back(card);
320  }
321  return legal_actions;
322 }

324 std::vector<std::pair<Action, double>> OhHellState::ChanceOutcomes() const {
325  std::vector<std::pair<Action, double>> outcomes;
326  double p;
327  if (phase_ == Phase::kChooseNumTricks) {
328  // uniform randomly select between all legal numbers of tricks possible
329  // given the number of players and size of the deck
330  if (num_tricks_fixed_ < kMinNumTricks) {
331  p = 1.0 / static_cast<double>(MaxNumTricks());
332  for (int i = 0; i < MaxNumTricks(); ++i) outcomes.emplace_back(i + 1, p);
333  } else {
334  outcomes.emplace_back(num_tricks_fixed_, 1.0);
335  }
336  } else if (phase_ == Phase::kDealer) {
337  // uniform randomly select a player
338  p = 1.0 / static_cast<double>(num_players_);
339  for (int i = 0; i < num_players_; ++i) outcomes.emplace_back(i, p);
340  } else if (num_cards_dealt_ < num_players_ * num_tricks_ + kNumTrumpDeal) {
341  // the only other chance nodes are when cards are dealt
342  int num_cards_rem = deck_props_.NumCards() - num_cards_dealt_;
343  outcomes.reserve(num_cards_rem);
344  p = 1.0 / static_cast<double>(num_cards_rem);
345  for (int card = 0; card < deck_props_.NumCards(); ++card) {
346  if (!initial_deal_[card].has_value()) outcomes.emplace_back(card, p);
347  }
348  }
349  return outcomes;
350 }

352 void OhHellState::DoApplyAction(Action action) {
353  switch (phase_) {
354  case Phase::kChooseNumTricks:
355  return ApplyChooseNumTricksAction(action);
356  case Phase::kDealer:
357  return ApplyDealerAction(action);
358  case Phase::kDeal:
359  return ApplyDealAction(action);
360  case Phase::kBid:
361  return ApplyBiddingAction(action - deck_props_.NumCards());
362  case Phase::kPlay:
363  return ApplyPlayAction(action);
364  case Phase::kGameOver:
365  SpielFatalError("Cannot act in terminal states");
366  }
367 }

380 void OhHellState::ApplyDealAction(int card) {
381  // dealer_ is ignored for dealing (player 0 always gets the first card)
382  // dealer is only used to determine who will go first during bid and play
383  int num_player_cards = num_players_ * num_tricks_;
384  if (num_cards_dealt_ < num_player_cards) {
385  holder_[card] = (num_cards_dealt_ % num_players_);
386  initial_deal_[card] = (num_cards_dealt_ % num_players_);
387  } else {
388  // last card dealt tells us the trump suit
389  trump_ = card;
390  phase_ = Phase::kBid;
391  current_player_ = (dealer_ + 1) % num_players_;
392  }
393  ++num_cards_dealt_;
394 }

396 void OhHellState::ApplyBiddingAction(int bid) {
397  bids_[current_player_] = bid;
398  current_player_ = (current_player_ + 1) % num_players_;
399  if (current_player_ == (dealer_ + 1) % num_players_) phase_ = Phase::kPlay;
400 }
401 
402 void OhHellState::ApplyPlayAction(int card) {
403  SPIEL_CHECK_TRUE(holder_[card] == current_player_);
404 
405  holder_[card] = absl::nullopt;
406  if (num_cards_played_ % num_players_ == 0) {
407  CurrentTrick() = Trick(current_player_, deck_props_.CardSuit(trump_),
408  card, deck_props_);
409  } else {
410  CurrentTrick().Play(current_player_, card);
411  }
412  const Player winner = CurrentTrick().Winner();
413  ++num_cards_played_;
414  if (num_cards_played_ % num_players_ == 0) {
415  ++num_tricks_won_[winner];
416  current_player_ = winner;
417  } else {
418  current_player_ = (current_player_ + 1) % num_players_;
419  }
420  if (num_cards_played_ == num_players_ * num_tricks_) {
421  phase_ = Phase::kGameOver;
422  ComputeScore();
423  }
424 }

436 void OhHellState::ComputeScore() {
437  SPIEL_CHECK_TRUE(IsTerminal());
438  for (Player player = 0; player < num_players_; ++player) {
439  if (off_bid_penalty_) {
440  if (num_tricks_won_[player] == bids_[player]) {
441  returns_[player] =
442  points_per_trick_ * num_tricks_won_[player] + kMadeBidBonus;
443  } else {
444  int diff = num_tricks_won_[player] - bids_[player];
445  returns_[player] = -(points_per_trick_ * abs(diff));
446  }
447  } else {
448  returns_[player] = points_per_trick_ * num_tricks_won_[player];
449  if (num_tricks_won_[player] == bids_[player]) {
450  returns_[player] += kMadeBidBonus;
451  }
452  }
453  }
454 }

475 void OhHellState::InformationStateTensor(Player player,
476  absl::Span<float> values) const {
477  SPIEL_CHECK_GE(player, 0);
478  SPIEL_CHECK_LT(player, num_players_);
479  std::fill(values.begin(), values.end(), 0.0);
480  SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorSize());
481  if (phase_ != Phase::kBid && phase_ != Phase::kPlay) return;
482  auto ptr = values.begin();
483  // total number of tricks
484  ptr[num_tricks_ - 1] = 1;
485  ptr += MaxNumTricks();
486  // which player is dealer
487  ptr[dealer_] = 1;
488  ptr += num_players_;
489  // trump
490  ptr[trump_] = 1;
491  ptr += deck_props_.NumCards();
492  // initial hand
493  for (int i = 0; i < deck_props_.NumCards(); ++i)
494  if (initial_deal_[i] == player) ptr[i] = 1;
495  ptr += deck_props_.NumCards();
496  // Current hand
497  for (int i = 0; i < deck_props_.NumCards(); ++i)
498  if (holder_[i] == player) ptr[i] = 1;
499  ptr += deck_props_.NumCards();
500  // all bids
501  for (Player p = 0; p < num_players_; ++p) {
502  ptr[bids_[p] + 1] = 1;
503  // need to account for bid of 0 and if player hasn't bid yet
504  ptr += MaxNumTricks() + 2;
505  }
506  // each player's number of tricks won so far (temperature encoding)
507  for (Player p = 0; p < num_players_; ++p) {
508  for (int i = 0; i < MaxNumTricks(); ++i) {
509  if (num_tricks_won_[p] > i) ptr[i] = 1;
510  }
511  ptr += MaxNumTricks();
512  }
513  // History of tricks, each in the format: 0 1 ... n 0 1 ... n-1
514  int current_trick = num_cards_played_ / num_players_;
515  auto play_hist = history_.begin() + NumChanceActions() + num_players_;
516  for (int i = 0; i <= current_trick; ++i) {
517  Player leader = tricks_[i].Leader();
518  ptr += std::max(leader, 0) * deck_props_.NumCards();
519  for (int i = 0; i < num_players_; i++) {
520  if (play_hist < history_.end()) {
521  ptr[play_hist->action] = 1;
522  ++play_hist;
523  }
524  ptr += deck_props_.NumCards();
525  }
526  ptr += (num_players_ - std::max(leader, 0) - 1) * deck_props_.NumCards();
527  }
528  // Skip over unplayed tricks.
529  int trick_tensor_size = (2 * num_players_ - 1) * deck_props_.NumCards();
530  ptr += (MaxNumTricks() - current_trick - 1) * trick_tensor_size;
531  SPIEL_CHECK_EQ(ptr, values.end());
532 }

534 // This implementation produces samples that may be inconsistent w.r.t. voids.
535 // i.e. if a player has played another suit when a diamond was lead,
536 // this player cannot have any diamonds according to the rules of the game, but
537 // the generated sample could be a state that contradicts this rule.
538 std::unique_ptr<State> OhHellState::ResampleFromInfostate(
539  int player_id, std::function<double()> rng) const {
540  std::unique_ptr<State> clone = game_->NewInitialState();
541  if (phase_ != Phase::kBid && phase_ != Phase::kPlay) return clone;
542 
543  // initial chance actions (choose num tricks and dealer)
544  clone->ApplyAction(num_tricks_);
545  clone->ApplyAction(dealer_);
546 
547  // deal needs to be consistent with the player's hand, and the opponent's
548  // played cards
549  std::vector<std::vector<int>> known(num_players_);
550  for (int card = 0; card < deck_props_.NumCards(); ++card) {
551  absl::optional<Player> p = initial_deal_[card];
552  if (p.has_value() && (*p == player_id || !holder_[card].has_value())) {
553  // if player_id was initially dealt the card, or if anyone was but no
554  // longer holds it (because it was played), player_id knows where it was
555  // dealt
556  known[*p].push_back(card);
557  }
558  }
559 
560  // the only other known card is trump
561  // apply num_tricks * num_players deal actions
562  std::vector<int> known_deal_counter(num_players_, 0);
563  for (int i = 0; i < num_players_ * num_tricks_; ++i) {
564  Player deal_to = i % num_players_;
565  if (known_deal_counter[deal_to] < known[deal_to].size()) {
566  clone->ApplyAction(known[deal_to][known_deal_counter[deal_to]]);
567  known_deal_counter[deal_to]++;
568  } else {
569  // deal randomly from the remaining unknown cards
570  Action candidate = kInvalidAction;
571  while (candidate == kInvalidAction) {
572  candidate = SampleAction(clone->ChanceOutcomes(), rng()).first;
573  absl::optional<Player> p = initial_deal_[candidate];
574  if (candidate == trump_ || (p.has_value() &&

577  // any player
578  candidate = kInvalidAction;
579  }
580  }
581  clone->ApplyAction(candidate);
582  }
583  }
584

586  clone->ApplyAction(trump_);
587

589  // original state

591  for (size_t i = start; i < history_.size(); i++) {
592  clone->ApplyAction(history_.at(i).action);
593  }

597  clone->InformationStateString(player_id));
