47int CharToRank(char c) {
48 switch (c) {
49 case 'J':
50 return 0;
51 case 'Q':
52 return 1;
53 case 'K':
54 return 2;
55 case 'A':
56 return 3;
57 }
58 SpielFatalError(absl::StrCat("Unknown rank '", std::string(1, c), "'"));
59}

61int CharToTrumps(char c) {
62 switch (c) {
63 case 'H':
64 return 0;
65 case 'S':
66 return 1;
67 case 'N': // No-trump
68 return 2;
69 }
70 SpielFatalError(absl::StrCat("Unknown trump suit '", std::string(1, c), "'"));
71}

73Seat CharToSeat(char c) {
74 switch (c) {
75 case 'W':

77 case 'N':

79 case 'E':

96// Requires card0 > card1
97int CardsToChanceOutcome(int card0, int card1) {
98 return (card0 * (card0 - 1)) / 2 + card1;
99}

101// Returns first > second
102std::pair<int, int> ChanceOutcomeToCards(int outcome) {
103 int card0 = 1;
104 while (CardsToChanceOutcome(card0 + 1, 0) <= outcome) ++card0;
105 return {card0, outcome - CardsToChanceOutcome(card0, 0)};
106}

128// Computes the abstraction lookup.
129std::vector<int> ConcreteToAbstract() {
130 std::vector<int> concrete_to_abstract(kNumPrivates, -1);
131 for (int c = 0; c < kNumPrivates; ++c) {
132 auto hand = HandString(c);
133 for (int ah = 0; ah < kNumAbstractHands; ++ah) {
134 if (absl::StrContains(kAbstraction[ah], hand)) {
135 concrete_to_abstract[c] = ah;
136 break;
137 }
138 }
139 if (concrete_to_abstract[c] == -1) {
140 SpielFatalError(
141 absl::StrCat("Abstraction not found for concrete hand '", hand, "'"));
142 }
143 }
144 return concrete_to_abstract;
145}

147// Returns an abstraction.
148int ChanceOutcomeToHandAbstraction(int outcome) {
149 static std::vector<int> concrete_to_abstract = ConcreteToAbstract();
150 return concrete_to_abstract[outcome];
151}

239// Score a played-out hand.
240int Score(int contract, int tricks, bool doubled, bool redoubled, int trumps) {
241 // -20 per undertrick
242 // +10 for 1H/S/NT (+10 extra if overtrick)
243 // +30 for 2H/S
244 // +35 for 2NT
245 const int contract_tricks = 1 + (contract - 1) / 3;
246 const int contract_result = tricks - contract_tricks;
247 const int double_factor = (1 + doubled) * (1 + redoubled);
248 if (contract_result < 0) return 20 * double_factor * contract_result;
249 int score = tricks * 10;
250 if (contract_tricks == 2) score += 10;
251 if (contract_tricks == 2 && trumps == 2) score += 5;
252 return score * double_factor;
253}

284std::unique_ptr<State> TinyBridgePlayGame::NewInitialState() const {
285 int trumps = CharToTrumps(ParameterValue<std::string>("trumps")[0]);
286 Seat leader = CharToSeat(ParameterValue<std::string>("leader")[0]);
287 std::array<Seat, kDeckSize> holder;
288 for (Seat i : {kWest, kNorth, kEast, kSouth}) {
289 std::string hand = ParameterValue<std::string>(
290 absl::StrCat("hand_", std::string(1, kSeatChar[i])));
291 for (int j = 0; j < kNumTricks; ++j) {
292 int c = StringToCard(hand.substr(j * 2, 2));
293 holder[c] = i;
294 }
295 }
296 return std::unique_ptr<State>(
297 new TinyBridgePlayState(shared_from_this(), trumps, leader, holder));
298}

326TinyBridgeAuctionState::AuctionState TinyBridgeAuctionState::AnalyzeAuction()
327 const {
328 AuctionState rv;
329 rv.last_bid = Call::kPass;
330 rv.last_bidder = kInvalidSeat;
331 rv.doubler = kInvalidSeat;
332 rv.redoubler = kInvalidSeat;
333 for (int i = num_players_; i < actions_.size(); ++i) {
334 if (actions_[i] == Call::kDouble) {
335 rv.doubler = PlayerToSeat(i % num_players_);
336 } else if (actions_[i] == Call::kRedouble) {
337 rv.redoubler = PlayerToSeat(i % num_players_);
338 } else if (actions_[i] != Call::kPass) {
339 rv.last_bid = actions_[i];
340 rv.last_bidder = PlayerToSeat(i % num_players_);
341 rv.doubler = kInvalidSeat;
342 rv.redoubler = kInvalidSeat;
343 }
344 }
345 return rv;
346}

348int Score_p0(std::array<Seat, kDeckSize> holder,
349 const TinyBridgeAuctionState::AuctionState& state) {
350 if (state.last_bid == Call::kPass) return 0;
351 std::shared_ptr<Game> game(new TinyBridgePlayGame({}));
352 int trumps = (state.last_bid - 1) % 3;
353 Seat leader = Seat((state.last_bidder + 3) % 4);
354 Seat decl = Seat(state.last_bidder % 2);
355 TinyBridgePlayState play{game, trumps, leader, holder};
356 const double tricks =
357 algorithms::AlphaBetaSearch(*game, &play, nullptr, -1, decl).first;
358 SPIEL_CHECK_GE(tricks, 0);
359 SPIEL_CHECK_LE(tricks, kNumTricks);
360 const int declarer_score =
361 Score(state.last_bid, tricks, state.doubler != kInvalidSeat,
362 state.redoubler != kInvalidSeat, trumps);
363 return (decl == 0) ? declarer_score : -declarer_score;
364}

366// Score indexed by [WestHand][EastHand][Contract][LastBidder]
367using ScoringTable = std::array<
368 std::array<std::array<std::array<double, 2>, kNumActions2p>, kNumPrivates>,
369 kNumPrivates>;
370
371// Calculates a single score.
372double Score_2p_(Action hand0, Action hand1,
373 const TinyBridgeAuctionState::AuctionState& state) {
374 if (state.last_bid == kPass) return 0;
375 const double freq = 1. / 6;
376 double utility_p0 = 0;
377 std::array<Seat, kDeckSize> holders_2p;
378 std::fill(holders_2p.begin(), holders_2p.end(), kInvalidSeat);
379 const auto cards0 = ChanceOutcomeToCards(hand0);
380 holders_2p[cards0.first] = kWest;
381 holders_2p[cards0.second] = kWest;
382 const auto cards1 = ChanceOutcomeToCards(hand1);
383 holders_2p[cards1.first] = kEast;
384 holders_2p[cards1.second] = kEast;
385 std::array<Seat, kDeckSize> holders_4p;
386 for (int n0 = 0; n0 < 3; ++n0) {
387 for (int n1 = n0 + 1; n1 < 4; ++n1) {
388 int n = 0;
389 for (int i = 0; i < kDeckSize; ++i) {
390 if (holders_2p[i] == kInvalidSeat) {
391 holders_4p[i] = (n == n0 || n == n1) ? kNorth : kSouth;
392 ++n;
393 } else {
394 holders_4p[i] = holders_2p[i];
395 }
396 }
397 utility_p0 += Score_p0(holders_4p, state) * freq;
398 }
399 }
400 return utility_p0;
401}

403// Returns a cache of scores.
404ScoringTable MakeScores() {
405 ScoringTable scores;
406 for (int hand0 = 0; hand0 < kNumPrivates; ++hand0) {
407 for (int hand1 = 0; hand1 < kNumPrivates; ++hand1) {
408 if (!IsConsistent(hand0, hand1)) continue;
409 for (int contract = k1H; contract < kNumActions2p; ++contract) {
410 for (Seat last_bidder : {kWest, kEast}) {
411 scores[hand0][hand1][contract][last_bidder / 2] =
412 Score_2p_(hand0, hand1,
413 {contract, last_bidder, kInvalidSeat, kInvalidSeat});
414 }
415 }
416 }
417 }
418 return scores;
419}
420
421double Score_2p(Action hand0, Action hand1,
422 const TinyBridgeAuctionState::AuctionState& state) {
423 if (state.last_bid == kPass) return 0;
424 static const ScoringTable scoring_table = MakeScores();
425 const double score =
426 scoring_table[hand0][hand1][state.last_bid][state.last_bidder / 2];
427 return score;
428}

430std::array<Seat, kDeckSize> TinyBridgeAuctionState::CardHolders() const {
431 std::array<Seat, kDeckSize> holder;
432 std::fill(holder.begin(), holder.end(), kInvalidSeat);
433 for (int i = 0; i < actions_.size() && i < num_players_; ++i) {
434 int action_id = actions_[i];
435 const auto cards = ChanceOutcomeToCards(action_id);
436 holder[cards.first] = Seat(i);
437 holder[cards.second] = Seat(i);
438 }
439 return holder;
440}

442void TinyBridgeAuctionState::DoApplyAction(Action action) {
443 actions_.push_back(action);
444 if (num_players_ == 2) {
445 if (actions_.size() >= 2 * num_players_ && actions_.back() == Call::kPass) {
446 is_terminal_ = true;
447 utility_p0 = Score_2p(actions_[0], actions_[1], AnalyzeAuction());
448 }
449 } else {
450 if (actions_.size() >= 2 * num_players_ &&
451 actions_[actions_.size() - 1] == Call::kPass &&
452 actions_[actions_.size() - 2] == Call::kPass &&
453 actions_[actions_.size() - 3] == Call::kPass) {
454 is_terminal_ = true;
455 utility_p0 = Score_p0(CardHolders(), AnalyzeAuction());
456 }
457 }
458}

460std::vector<Action> TinyBridgeAuctionState::LegalActions() const {
461 std::vector<Action> actions;
462 if (IsChanceNode()) {
463 return LegalChanceOutcomes();
464 } else if (IsTerminal()) {
465 return {};
466 } else {
467 auto state = AnalyzeAuction();
468 actions.push_back(Call::kPass);
469 for (int bid = state.last_bid + 1; bid <= Call::k2NT; ++bid) {
470 actions.push_back(bid);
471 }
472 if (num_players_ == 4 && state.last_bidder != kInvalidSeat) {
473 if (state.last_bidder % 2 != CurrentPlayer() % 2) {
474 if (state.doubler == kInvalidSeat) actions.push_back(Call::kDouble);
475 } else {
476 if (state.doubler != kInvalidSeat && state.redoubler == kInvalidSeat)
477 actions.push_back(Call::kRedouble);
478 }
479 }
480 }
481 return actions;
482}

484std::vector<std::pair<Action, double>> TinyBridgeAuctionState::ChanceOutcomes()
485 const {
486 std::vector<Action> actions;
487 auto holder = CardHolders();
488 for (int card1 = 0; card1 < kDeckSize; ++card1) {
489 if (holder[card1] != kInvalidSeat) continue;
490 for (int card2 = card1 + 1; card2 < kDeckSize; ++card2) {
491 if (holder[card2] != kInvalidSeat) continue;
492 actions.push_back(CardsToChanceOutcome(card2, card1));
493 }
494 }
495 const int num_actions = actions.size();
496 std::vector<std::pair<Action, double>> outcomes;
497 outcomes.reserve(num_actions);
498 for (auto action : actions) {
499 outcomes.emplace_back(action, 1.0 / num_actions);
500 }
501 return outcomes;
502}

566// Observation string is the player's cards plus the most recent bid,
567// plus any doubles or redoubles. E.g. "HJSA 2NT:Us Dbl:RH RDbl:Pd"
568// This is an observation for a player who holds HJ and SA.
569// and redoubled by West.
570// The most recent bid is 2NT by this player, which has been doubled by their
571// right-hand-opponent and redoubled by their partner. So the most recent few
572// calls must be: 2NT-Pass-Pass-Dbl-Pass-Pass-RDbl-Pass.
573std::string TinyBridgeAuctionState::ObservationString(Player player) const {
574 SPIEL_CHECK_GE(player, 0);
575 SPIEL_CHECK_LT(player, num_players_);
576
577 std::string observation = PlayerHandString(player, is_abstracted_);
578 if (HasAuctionStarted()) {
579 auto state = AnalyzeAuction();
580 if (state.last_bid != Call::kPass) {
581 absl::StrAppend(
582 &observation, " ", ActionToString(state.last_bidder, state.last_bid),
583 ":", RelativeSeatString(state.last_bidder, PlayerToSeat(player)));
584 }
585 if (state.doubler != kInvalidSeat)
586 absl::StrAppend(&observation, " ", "Dbl:",
587 RelativeSeatString(state.doubler, PlayerToSeat(player)));
588 if (state.redoubler != kInvalidSeat)
589 absl::StrAppend(
590 &observation, " ",
591 "RDbl:", RelativeSeatString(state.redoubler, PlayerToSeat(player)));
592 }
593 return observation;
594}

596// Information state vector consists of:
597// kNumCards bits showing which cards the observing player holds
598// For 2p, kNumActions2p*2 bits showing which actions have been taken:
599// For each action, the bits are [1, 0] if we took the action,
600// [0, 1] if our partner took the action, and otherwise [0, 0].
601// For 4p, (kNumBids*3+1)*num_players bits showing which actions have been
602// taken:
603// For each player, 1 bit showing if they passed before the first bid
604// For each bid, 4 bits showing who made it, 4 bits showing who doubled it,
605// and 4 bits showing who redoubled it.
606// Each set of 4 bits is relative the the current player.
607void TinyBridgeAuctionState::InformationStateTensor(
608 Player player, absl::Span<float> values) const {
609 SPIEL_CHECK_GE(player, 0);
610 SPIEL_CHECK_LT(player, num_players_);
611
612 const int hand_size = is_abstracted_ ? kNumAbstractHands : kDeckSize;
613 const int auction_size = (num_players_ == 2)
614 ? kNumActions2p * num_players_
615 : num_players_ + kNumBids * num_players_ * 3;
616 std::fill(values.begin(), values.end(), 0);
617 SPIEL_CHECK_EQ(values.size(), hand_size + auction_size);
618 if (IsDealt(player)) {
619 if (is_abstracted_) {
620 const int abstraction = ChanceOutcomeToHandAbstraction(actions_[player]);
621 values.at(abstraction) = 1;
622 } else {
623 const auto cards = ChanceOutcomeToCards(actions_[player]);
624 values.at(cards.first) = 1;
625 values.at(cards.second) = 1;
626 }
627 }
628 if (num_players_ == 2) {
629 for (int i = num_players_; i < actions_.size(); ++i) {
630 values.at(hand_size + actions_[i] * 2 + (i - player) % num_players_) = 1;
631 }
632 } else {
633 auto last_bid = Call::kPass;
634 auto observer = PlayerToSeat(player);
635 for (int i = num_players_; i < actions_.size(); ++i) {
636 int bidder = RelativeSeatIndex(Seat(i % num_players_), observer);
637 if (actions_[i] == Call::kPass) {
638 if (last_bid == Call::kPass) {
639 values.at(hand_size + bidder) = 1;
640 }
641 } else if (actions_[i] == Call::kDouble) {
642 values.at(hand_size + num_players_ +
643 (last_bid - 1) * (3 * num_players_) + bidder) = 1;
644 } else if (actions_[i] == Call::kRedouble) {
645 values.at(hand_size + num_players_ +
646 (last_bid - 1) * (3 * num_players_) + num_players_ + bidder) =
647 1;
648 } else {
649 last_bid = Call(actions_[i]);
650 values.at(hand_size + num_players_ +
651 (last_bid - 1) * (3 * num_players_) + num_players_ * 2 +
652 bidder) = 1;
653 }
654 }
655 }
656}

658// Information state vector consists of:
659// kNumCards bits showing which cards the observing player holds
660// For 2p:
661// kNumActions2p bits showing the most recent action (one-hot)
662// For 4p:
663// kNumBids bits showing the most recent bid (one-hot)
664// 4 bits showing who bid it (relative to the observing player)
665// 4 bits showing who doubled it (relative to the observing player)
666// 4 bits showing who redoubled it (relative to the observing player)
667// 4 bits for the dealer
668void TinyBridgeAuctionState::ObservationTensor(Player player,
669 absl::Span<float> values) const {
670 SPIEL_CHECK_GE(player, 0);
671 SPIEL_CHECK_LT(player, num_players_);
672
673 const int hand_size = is_abstracted_ ? kNumAbstractHands : kDeckSize;
674 const int auction_size =
675 num_players_ == 2 ? kNumActions2p : kNumBids + 4 * num_players_;
676 std::fill(values.begin(), values.end(), 0);
677 SPIEL_CHECK_EQ(values.size(), hand_size + auction_size);
678 if (IsDealt(player)) {
679 if (is_abstracted_) {
680 const int abstraction = ChanceOutcomeToHandAbstraction(actions_[player]);
681 values.at(abstraction) = 1;
682 } else {
683 const auto cards = ChanceOutcomeToCards(actions_[player]);
684 values.at(cards.first) = 1;
685 values.at(cards.second) = 1;
686 }
687 }
688 if (num_players_ == 2) {
689 if (HasAuctionStarted()) {
690 values.at(hand_size + actions_.back()) = 1;
691 }
692 } else {
693 auto state = AnalyzeAuction();
694 auto seat = PlayerToSeat(player);
695 if (state.last_bidder != kInvalidSeat)
696 values.at(hand_size + RelativeSeatIndex(state.last_bidder, seat)) = 1;
697 if (state.doubler != kInvalidSeat)
698 values.at(hand_size + kNumSeats +
699 RelativeSeatIndex(state.doubler, seat)) = 1;
700 if (state.redoubler != kInvalidSeat)
701 values.at(hand_size + kNumSeats * 2 +
702 RelativeSeatIndex(state.redoubler, seat)) = 1;
703 values.at(hand_size + kNumSeats * 3 +
704 RelativeSeatIndex(Seat::kWest, seat)) = 1;
705 if (state.last_bidder != kInvalidSeat)
706 values.at(hand_size + kNumSeats * 4 + state.last_bid - 1) = 1;
707 }
708}

721void TinyBridgePlayState::DoApplyAction(Action action) {
722 actions_.emplace_back(CurrentHand(), action);
723 if (actions_.size() % 4 == 0) {
724 Seat win_hand = actions_[actions_.size() - 4].first;
725 int win_card = actions_[actions_.size() - 4].second;
726 for (int i = actions_.size() - 3; i < actions_.size(); ++i) {
727 Seat hand = actions_[i].first;
728 int card = actions_[i].second;
729 if (Suit(card) == Suit(win_card)) {
730 if (Rank(card) > Rank(win_card)) {
731 win_card = card;
732 win_hand = hand;
733 }
734 } else if (Suit(card) == trumps_) {
735 win_card = card;
736 win_hand = hand;
737 }
738 }
739 winner_[actions_.size() / 4 - 1] = win_hand;
740 }
741}

743std::vector<Action> TinyBridgePlayState::LegalActions() const {
744 std::vector<Action> actions;
745 const int hand = CurrentHand();
746 for (int i = 0; i < kDeckSize; ++i) {
747 if (holder_[i] == hand &&
748 (actions_.size() < 4 ||
749 actions_[(4 + hand - leader_) % 4].second != i)) {
750 actions.push_back(i);
751 }
752 }
753 // Have to follow suit if we have two cards of different suits.
754 if (!actions_.empty() && actions.size() == 2 &&
755 Suit(actions[0]) != Suit(actions[1])) {
756 return {Suit(actions[0]) == Suit(actions_[0].second) ? actions[0]
757 : actions[1]};
758 } else {
759 return actions;
760 }
761}
