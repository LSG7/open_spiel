66std::string CEState::InformationStateString(Player player) const {
67 // should look like <infoset string> <delimiter> <signal id>
68 SPIEL_CHECK_FALSE(IsChanceNode());
69 std::string infoset_str = state_->InformationStateString(player);
70 SPIEL_CHECK_EQ(infoset_str.find(config_.recommendation_delimiter),
71 std::string::npos);
72 const auto* parent_game = down_cast<const CEGame*>(game_.get());
73 SPIEL_CHECK_GE(rec_index_, 0);
74 int signal_id = parent_game->GetSignalId(rec_index_, player);
75 return absl::StrCat(infoset_str, config_.recommendation_delimiter, signal_id);
76}

85void CEState::DoApplyAction(Action action_id) {
86 if (rec_index_ < 0) {
87 // Pick the joint policy which will provide recommendations.
88 rec_index_ = action_id;
89 SPIEL_CHECK_LT(rec_index_, mu_.size());
90 } else if (state_->IsChanceNode()) {
91 // Regular chance node
92 state_->ApplyAction(action_id);
93 } else {
94 // Regular decision node
95 state_->ApplyAction(action_id);
96 }

105CEGame::CEGame(std::shared_ptr<const Game> game, CorrDistConfig config,
106 const CorrelationDevice& mu)
107 : WrappedGame(game, game->GetType(), game->GetParameters()),
108 config_(config),
109 mu_(mu),
110 orig_num_distinct_actions_(game->NumDistinctActions()),
111 signal_ids_(game->NumPlayers(), 0) {
112 // First, build the map that will identify which information states belong
113 // to which player.
114 {
115 std::vector<std::vector<std::string>> all_infostates =
116 GetAllInformationStates(*game);
117 SPIEL_CHECK_EQ(all_infostates.size(), game->NumPlayers());
118 for (Player p = 0; p < all_infostates.size(); ++p) {
119 for (const std::string& info_state : all_infostates[p]) {
120 const auto iter = info_state_to_player_.find(info_state);
121 if (iter != info_state_to_player_.end()) {
122 SpielFatalError("Duplicate information set found!");
123 }
124 info_state_to_player_[info_state] = p;
125 }
126 }
127 }
128
129 // Now, go through each joint policy in the correlation device, splitting them
130 // among players.
131 for (int rec_index = 0; rec_index < mu_.size(); ++rec_index) {
132 const TabularPolicy& joint_policy = mu_[rec_index].second;
133
134 // Split the policies into individual player policies.
135 std::vector<TabularPolicy> player_policies(game->NumPlayers());
136 for (const auto& [info_state, action_probs] : joint_policy.PolicyTable()) {
137 const auto player_iter = info_state_to_player_.find(info_state);
138 SPIEL_CHECK_TRUE(player_iter != info_state_to_player_.end());
139 Player player = player_iter->second;
140 player_policies[player].SetStatePolicy(info_state, action_probs);
141 }
142
143 // Lookup / assign signals to each individual policy.
144 for (Player p = 0; p < player_policies.size(); ++p) {
145 std::string sorted_policy_string = player_policies[p].ToStringSorted();
146 std::pair<std::string, Player> key = {sorted_policy_string, p};
147 const auto iter = policy_player_to_signal_id_.find(key);
148 int signal_id = -1;
149
150 if (iter == policy_player_to_signal_id_.end()) {
151 // Signal for this policy does not exist yet, use the next one.
152 signal_id = signal_ids_[p]++;
153 policy_player_to_signal_id_[key] = signal_id;
154 } else {
155 signal_id = iter->second;
156 }
157
158 recidx_player_to_signal_id_[{rec_index, p}] = signal_id;
159 }
160 }
161}

163ActionsAndProbs CETabularPolicy::GetStatePolicy(const State& state) const {
164 // Here we must scrape off the signal id so that the BR code does a proper
165 // lookup on the orginal info state string.
166 const auto* ce_state = dynamic_cast<const CEState*>(&state);
167 SPIEL_CHECK_TRUE(ce_state != nullptr);
168
169 std::string info_state = state.InformationStateString();
170 const size_t idx = info_state.find(config_.recommendation_delimiter);
171 SPIEL_CHECK_NE(idx, std::string::npos);
172
173 std::string orig_info_state = info_state.substr(0, idx);
174 return ce_state->RecommendedStatePolicy(orig_info_state);
175}
