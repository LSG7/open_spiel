110  def euc_descent_step(self, params, grads, t):
111  """Projected gradient descent on exploitability using Euclidean projection.
112 
113  Args:
114  params: tuple of variables to be updated (dist, y, anneal_steps)
115  grads: tuple of variable gradients (grad_dist, grad_y, grad_anneal_steps)
116  t: int, solver iteration (unused)
117  Returns:
118  new_params: tuple of update params (new_dist, new_y, new_anneal_steps)
119  """
120  lr_dist, lr_y = self.lrslrs
121  new_dist = []
122  for dist_i, dist_grad_i in zip(params[0], grads[0]):
123  new_dist_i = dist_i - lr_dist * dist_grad_i
124  new_dist_i = simplex.euclidean_projection_onto_simplex(new_dist_i)
125  new_dist.append(new_dist_i)
126  lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
127  new_y = []
128  for y_i, y_grad_i in zip(params[1], grads[1]):
129  new_y_i = y_i - lr_y * y_grad_i
130  new_y_i = np.clip(new_y_i, 0., np.inf)
131  new_y.append(new_y_i)
132  new_anneal_steps = params[2] + grads[2]
133  return (new_dist, new_y, new_anneal_steps)

135  def mirror_descent_step(self, params, grads, t):
136  """Entropic mirror descent on exploitability.
137 
138  Args:
139  params: tuple of variables to be updated (dist, y)
140  grads: tuple of variable gradients (grad_dist, grad_y)
141  t: int, solver iteration (unused)
142  Returns:
143  new_params: tuple of update params (new_dist, new_y)
144  """
145  lr_dist, lr_y = self.lrslrs
146  new_dist = []
147  for dist_i, dist_grad_i in zip(params[0], grads[0]):
148  new_dist_i = np.log(np.clip(dist_i, 0., np.inf)) - lr_dist * dist_grad_i
149  new_dist_i = special.softmax(new_dist_i)
150  new_dist.append(new_dist_i)
151  lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
152  new_y = []
153  for y_i, y_grad_i in zip(params[1], grads[1]):
154  new_y_i = y_i - lr_y * y_grad_i
155  new_y_i = np.clip(new_y_i, 0., np.inf)
156  new_y.append(new_y_i)
157  new_anneal_steps = params[2] + grads[2]
158  return (new_dist, new_y, new_anneal_steps)

160  def gradients(self, dist, y, anneal_steps, payoff_matrices, num_players, p=1,
161  proj_grad=True):
162  """Computes exploitablity gradient and aux variable gradients.
163 
164  Args:
165  dist: list of 1-d np.arrays, current estimate of nash distribution
166  y: list 1-d np.arrays (same shape as dist), current est. of payoff grad
167  anneal_steps: int, elapsed num steps since last anneal
168  payoff_matrices: dictionary with keys as tuples of agents (i, j) and
169  values of (2 x A x A) np.arrays, payoffs for each joint action. keys
170  are sorted and arrays should be indexed in the same order
171  num_players: int, number of players, in case payoff_matrices is abbrev'd
172  p: float in [0, 1], Tsallis entropy-regularization --> 0 as p --> 0
173  proj_grad: bool, if True, projects dist gradient onto simplex
174  Returns:
175  gradient of exploitability w.r.t. (dist, y) as tuple
176  unregularized exploitability (stochastic estimate)
177  tsallis regularized exploitability (stochastic estimate)
178  """
179  # first compute policy gradients and player effects (fx)
180  policy_gradient = []
181  other_player_fx = []
182  grad_y = []
183  unreg_exp = []
184  reg_exp = []
185  for i in range(num_players):
186 
187  nabla_i = np.zeros_like(dist[i])
188  for j in range(num_players):
189  if j == i:
190  continue
191  if i < j:
192  hess_i_ij = payoff_matrices[(i, j)][0]
193  else:
194  hess_i_ij = payoff_matrices[(j, i)][1].T
195 
196  nabla_ij = hess_i_ij.dot(dist[j])
197  nabla_i += nabla_ij / float(num_players - 1)
198 
199  grad_y.append(y[i] - nabla_i)
200 
201  if p > 1e-2: # encounter numerical under/overflow when power > 100.
202  power = 1. / float(p)
203  s_i = np.linalg.norm(y[i], ord=power)
204  if s_i == 0:
205  br_i = misc.uniform_dist(y[i])
206  else:
207  br_i = (y[i] / s_i)**power
208  else:
209  power = np.inf
210  s_i = np.linalg.norm(y[i], ord=power)
211  br_i = np.zeros_like(dist[i])
212  maxima_i = (y[i] == s_i)
213  br_i[maxima_i] = 1. / maxima_i.sum()
214 
215  policy_gradient_i = nabla_i - s_i * dist[i]**p
216  policy_gradient.append(policy_gradient_i)
217 
218  unreg_exp.append(np.max(y[i]) - y[i].dot(dist[i]))
219 
220  br_i_inv_sparse = 1 - np.sum(br_i**(p + 1))
221  dist_i_inv_sparse = 1 - np.sum(dist[i]**(p + 1))
222  entr_br_i = s_i / (p + 1) * br_i_inv_sparse
223  entr_dist_i = s_i / (p + 1) * dist_i_inv_sparse
224 
225  reg_exp.append(y[i].dot(br_i - dist[i]) + entr_br_i - entr_dist_i)
226 
227  entr_br_vec_i = br_i_inv_sparse * br_i**(1 - p)
228  entr_dist_vec_i = dist_i_inv_sparse * dist[i]**(1 - p)
229  other_player_fx_i = (br_i - dist[i]) + 1 / (p + 1) * (
230  entr_br_vec_i - entr_dist_vec_i)
231  other_player_fx.append(other_player_fx_i)
232 
233  # then construct exploitability gradient
234  grad_dist = []
235  for i in range(num_players):
236 
237  grad_dist_i = -policy_gradient[i]
238  for j in range(num_players):
239  if j == i:
240  continue
241  if i < j:
242  hess_j_ij = payoff_matrices[(i, j)][1]
243  else:
244  hess_j_ij = payoff_matrices[(j, i)][0].T
245 
246  grad_dist_i += hess_j_ij.dot(other_player_fx[j])
247 
248  if proj_grad:
249  grad_dist_i = simplex.project_grad(grad_dist_i)
250 
251  grad_dist.append(grad_dist_i)
252 
253  unreg_exp_mean = np.mean(unreg_exp)
254  reg_exp_mean = np.mean(reg_exp)
255 
256  _, lr_y = self.lrslrs
257  if (reg_exp_mean < self.exp_threshexp_thresh) and (anneal_steps >= 1 / lr_y):
258  self.pp = np.clip(p / 2., 0., 1.)
259  grad_anneal_steps = -anneal_steps
260  else:
261  grad_anneal_steps = 1
262 
263  return (grad_dist, grad_y, grad_anneal_steps), unreg_exp_mean, reg_exp_mean

265  def cheap_gradients(self, random, dist, y, anneal_steps, payoff_matrices,

268 
269  This implementation takes payoff_matrices as input so technically uses
270  O(d^2) compute but only a single column of payoff_matrices is used to
271  perform the update so can be re-implemented in O(d) if needed.
272

274  random: random number generator, np.random.RandomState(seed)
275  dist: list of 1-d np.arrays, current estimate of nash distribution
276  y: list 1-d np.arrays (same shape as dist), current est. of payoff grad
277  anneal_steps: int, elapsed num steps since last anneal
278  payoff_matrices: dictionary with keys as tuples of agents (i, j) and
279  values of (2 x A x A) np.arrays, payoffs for each joint action. keys
280  are sorted and arrays should be indexed in the same order
281  num_players: int, number of players, in case payoff_matrices is abbrev'd
282  p: float in [0, 1], Tsallis entropy-regularization --> 0 as p --> 0
283  proj_grad: bool, if True, projects dist gradient onto simplex
284  Returns:
285  gradient of exploitability w.r.t. (dist, y) as tuple
286  unregularized exploitability (stochastic estimate)
287  tsallis regularized exploitability (stochastic estimate)
288  """

291  other_player_fx = []

294  reg_exp = []
295  for i in range(num_players):
296 
297  others = list(range(num_players))
298  others.remove(i)
299  j = np.random.choice(others)
300  action_j = random.choice(dist[j].size, p=dist[j])
301  if i < j:
302  hess_i_ij = payoff_matrices[(i, j)][0]
303  else:
304  hess_i_ij = payoff_matrices[(j, i)][1].T

306 
307  grad_y.append(y[i] - nabla_i)
308

310  power = 1. / float(p)

312  if s_i == 0:
313  br_i = misc.uniform_dist(y[i])
314  else:
315  br_i = (y[i] / s_i)**power
316  else:
317  power = np.inf
318  s_i = np.linalg.norm(y[i], ord=power)
319  br_i = np.zeros_like(dist[i])
320  maxima_i = (y[i] == s_i)
321  br_i[maxima_i] = 1. / maxima_i.sum()
322 
323  policy_gradient_i = nabla_i - s_i * dist[i]**p
324  policy_gradient.append(policy_gradient_i)
325 
326  unreg_exp.append(np.max(y[i]) - y[i].dot(dist[i]))
327 
328  br_i_inv_sparse = 1 - np.sum(br_i**(p + 1))
329  dist_i_inv_sparse = 1 - np.sum(dist[i]**(p + 1))
330  entr_br_i = s_i / (p + 1) * br_i_inv_sparse
331  entr_dist_i = s_i / (p + 1) * dist_i_inv_sparse
332 
333  reg_exp.append(y[i].dot(br_i - dist[i]) + entr_br_i - entr_dist_i)
334 
335  entr_br_vec_i = br_i_inv_sparse * br_i**(1 - p)
336  entr_dist_vec_i = dist_i_inv_sparse * dist[i]**(1 - p)
337  other_player_fx_i = (br_i - dist[i]) + 1 / (p + 1) * (
338  entr_br_vec_i - entr_dist_vec_i)
339  other_player_fx.append(other_player_fx_i)
340 
341  # then construct exploitability gradient
342  grad_dist = []
343  for i in range(num_players):
344 
345  grad_dist_i = -policy_gradient[i]
346  for j in range(num_players):
347  if j == i:
348  continue
349  if i < j:
350  hess_j_ij = payoff_matrices[(i, j)][1]
351  else:
352  hess_j_ij = payoff_matrices[(j, i)][0].T
353 
354  action_u = random.choice(dist[j].size) # uniform, ~importance sampling
355  other_player_fx_j = dist[j].size * other_player_fx[j][action_u]
356  grad_dist_i += hess_j_ij[:, action_u] * other_player_fx_j
357 
358  if proj_grad:
359  grad_dist_i = simplex.project_grad(grad_dist_i)
360 
361  grad_dist.append(grad_dist_i)
362
