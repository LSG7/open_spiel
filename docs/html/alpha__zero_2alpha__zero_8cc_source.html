50 struct Trajectory {
51  struct State {
52  std::vector<float> observation;
53  open_spiel::Player current_player;
54  std::vector<open_spiel::Action> legal_actions;
55  open_spiel::Action action;
56  open_spiel::ActionsAndProbs policy;
57  double value;
58  };
59 
60  std::vector<State> states;
61  std::vector<double> returns;
62 };

64 Trajectory PlayGame(
65  Logger* logger,
66  int game_num,
67  const open_spiel::Game& game,
68  std::vector<std::unique_ptr<MCTSBot>>* bots,
69  std::mt19937* rng, double temperature, int temperature_drop,
70  double cutoff_value, bool verbose = false) {
71  std::unique_ptr<open_spiel::State> state = game.NewInitialState();
72  std::vector<std::string> history;
73  Trajectory trajectory;
74 
75  while (true) {
76  open_spiel::Player player = state->CurrentPlayer();
77  std::unique_ptr<SearchNode> root = (*bots)[player]->MCTSearch(*state);
78  open_spiel::ActionsAndProbs policy;
79  policy.reserve(root->children.size());
80  for (const SearchNode& c : root->children) {
81  policy.emplace_back(
82  c.action, std::pow(c.explore_count, 1.0 / temperature));
83  }
84  NormalizePolicy(&policy);
85  open_spiel::Action action;
86  if (history.size() >= temperature_drop) {
87  action = root->BestChild().action;
88  } else {
89  action = open_spiel::SampleAction(policy, *rng).first;
90  }
91 
92  double root_value = root->total_reward / root->explore_count;
93  trajectory.states.push_back(Trajectory::State{
94  state->ObservationTensor(), player,
95  state->LegalActions(), action, std::move(policy), root_value});
96  std::string action_str = state->ActionToString(player, action);
97  history.push_back(action_str);
98  state->ApplyAction(action);
99  if (verbose) {
100  logger->Print("Player: %d, action: %s", player, action_str);
101  }
102  if (state->IsTerminal()) {
103  trajectory.returns = state->Returns();
104  break;
105  } else if (std::abs(root_value) > cutoff_value) {
106  trajectory.returns.resize(2);
107  trajectory.returns[player] = root_value;
108  trajectory.returns[1 - player] = -root_value;
109  break;
110  }
111  }
112 
113  logger->Print(
114  "Game %d: Returns: %s; Actions: %s", game_num,
115  absl::StrJoin(trajectory.returns, " "),
116  absl::StrJoin(history, " "));
117  return trajectory;
118 }

137 // An actor thread runner that generates games and returns trajectories.
138 void actor(const open_spiel::Game& game, const AlphaZeroConfig& config, int num,
139  ThreadedQueue<Trajectory>* trajectory_queue,
140  std::shared_ptr<VPNetEvaluator> vp_eval,
141  StopToken* stop) {
142  std::unique_ptr<Logger> logger;
143  if (num < 20) { // Limit the number of open files.
144  logger.reset(new FileLogger(config.path, absl::StrCat("actor-", num)));
145  } else {
146  logger.reset(new NoopLogger());
147  }
148  std::mt19937 rng;
149  absl::uniform_real_distribution<double> dist(0.0, 1.0);
150  std::vector<std::unique_ptr<MCTSBot>> bots;
151  bots.reserve(2);
152  for (int player = 0; player < 2; player++) {
153  bots.push_back(InitAZBot(config, game, vp_eval, false));
154  }
155  for (int game_num = 1; !stop->StopRequested(); ++game_num) {
156  double cutoff = (dist(rng) < config.cutoff_probability
157  ? config.cutoff_value : game.MaxUtility() + 1);
158  if (!trajectory_queue->Push(
159  PlayGame(logger.get(), game_num, game, &bots, &rng,
160  config.temperature, config.temperature_drop, cutoff),
161  absl::Seconds(10))) {
162  logger->Print("Failed to push a trajectory after 10 seconds.");
163  }
164  }
165  logger->Print("Got a quit.");
166 }

168 class EvalResults {
169  public:
170  explicit EvalResults(int count, int evaluation_window) {
171  results_.reserve(count);
172  for (int i = 0; i < count; ++i) {
173  results_.emplace_back(evaluation_window);
174  }
175  }
176 
177  // How many evals per difficulty.
178  int EvalCount() {
179  absl::MutexLock lock(&m_);
180  return eval_num_ / results_.size();
181  }
182 
183  // Which eval to do next: difficulty, player0.
184  std::pair<int, bool> Next() {
185  absl::MutexLock lock(&m_);
186  int next = eval_num_ % (results_.size() * 2);
187  eval_num_ += 1;
188  return {next / 2, next % 2};
189  }
190 
191  void Add(int i, double value) {
192  absl::MutexLock lock(&m_);
193  results_[i].Add(value);
194  }
195 
196  std::vector<double> AvgResults() {
197  absl::MutexLock lock(&m_);
198  std::vector<double> out;
199  out.reserve(results_.size());
200  for (const auto& result : results_) {
201  out.push_back(result.Empty() ? 0
202  : (absl::c_accumulate(result.Data(), 0.0) /
203  result.Size()));
204  }
205  return out;
206  }
207 
208  private:
209  std::vector<CircularBuffer<double>> results_;
210  int eval_num_ = 0;
211  absl::Mutex m_;
212 };

214 // A thread that plays vs standard MCTS.
215 void evaluator(const open_spiel::Game& game, const AlphaZeroConfig& config,
216  int num, EvalResults* results,
217  std::shared_ptr<VPNetEvaluator> vp_eval, StopToken* stop) {
218  FileLogger logger(config.path, absl::StrCat("evaluator-", num));
219  std::mt19937 rng;
220  auto rand_evaluator = std::make_shared<RandomRolloutEvaluator>(1, num);
221 
222  for (int game_num = 1; !stop->StopRequested(); ++game_num) {
223  auto [difficulty, first] = results->Next();
224  int az_player = first ? 0 : 1;
225  int rand_max_simulations = config.max_simulations * std::pow(
226  10, difficulty / 2.0);
227  std::vector<std::unique_ptr<MCTSBot>> bots;
228  bots.reserve(2);
229  bots.push_back(InitAZBot(config, game, vp_eval, true));
230  bots.push_back(std::make_unique<MCTSBot>(
231  game,
232  rand_evaluator,
233  config.uct_c,
234  rand_max_simulations,
235  /*max_memory_mb=*/1000,
236  /*solve=*/true,
237  /*seed=*/num * 1000 + game_num,
238  /*verbose=*/false,
239  ChildSelectionPolicy::UCT));
240  if (az_player == 1) {
241  std::swap(bots[0], bots[1]);
242  }
243 
244  logger.Print("Running MCTS with %d simulations", rand_max_simulations);
245  Trajectory trajectory = PlayGame(
246  &logger, game_num, game, &bots, &rng, /*temperature=*/ 1,
247  /*temperature_drop=*/ 0, /*cutoff_value=*/ game.MaxUtility() + 1);
248 
249  results->Add(difficulty, trajectory.returns[az_player]);
250  logger.Print("Game %d: AZ: %5.2f, MCTS: %5.2f, MCTS-sims: %d, length: %d",
251  game_num, trajectory.returns[az_player],
252  trajectory.returns[1 - az_player], rand_max_simulations,
253  trajectory.states.size());
254  }
255  logger.Print("Got a quit.");
256 }

258 void learner(const open_spiel::Game& game,
259  const AlphaZeroConfig& config,
260  DeviceManager* device_manager,
261  std::shared_ptr<VPNetEvaluator> eval,
262  ThreadedQueue<Trajectory>* trajectory_queue,
263  EvalResults* eval_results,
264  StopToken* stop) {
265  FileLogger logger(config.path, "learner");
266  DataLoggerJsonLines data_logger(config.path, "learner", true);
267  std::mt19937 rng;
268 
269  int device_id = 0;
270  logger.Print("Running the learner on device %d: %s", device_id,
271  device_manager->Get(0, device_id)->Device());
272 
273  CircularBuffer<VPNetModel::TrainInputs> replay_buffer(
274  config.replay_buffer_size);
275  int learn_rate = config.replay_buffer_size / config.replay_buffer_reuse;
276  int64_t total_trajectories = 0;
277 
278  const int stage_count = 7;
279  std::vector<open_spiel::BasicStats> value_accuracies(stage_count);
280  std::vector<open_spiel::BasicStats> value_predictions(stage_count);
281  open_spiel::BasicStats game_lengths;
282  open_spiel::HistogramNumbered game_lengths_hist(game.MaxGameLength() + 1);
283 
284  open_spiel::HistogramNamed outcomes({"Player1", "Player2", "Draw"});
285  // Actor threads have likely been contributing for a while, so put `last` in
286  // the past to avoid a giant spike on the first step.
287  absl::Time last = absl::Now() - absl::Seconds(60);
288  for (int step = 1; !stop->StopRequested() &&
289  (config.max_steps == 0 || step <= config.max_steps);
290  ++step) {
291  outcomes.Reset();
292  game_lengths.Reset();
293  game_lengths_hist.Reset();
294  for (auto& value_accuracy : value_accuracies) {
295  value_accuracy.Reset();
296  }
297  for (auto& value_prediction : value_predictions) {
298  value_prediction.Reset();
299  }
300 
301  // Collect trajectories
302  int queue_size = trajectory_queue->Size();
303  int num_states = 0;
304  int num_trajectories = 0;
305  while (!stop->StopRequested() && num_states < learn_rate) {
306  absl::optional<Trajectory> trajectory = trajectory_queue->Pop();
307  if (trajectory) {
308  num_trajectories += 1;
309  total_trajectories += 1;
310  game_lengths.Add(trajectory->states.size());
311  game_lengths_hist.Add(trajectory->states.size());
312 
313  double p1_outcome = trajectory->returns[0];
314  outcomes.Add(p1_outcome > 0 ? 0 : (p1_outcome < 0 ? 1 : 2));
315 
316  for (const Trajectory::State& state : trajectory->states) {
317  replay_buffer.Add(
318  VPNetModel::TrainInputs{
319  state.legal_actions,
320  state.observation,
321  state.policy,
322  p1_outcome});
323  num_states += 1;
324  }
325 
326  for (int stage = 0; stage < stage_count; ++stage) {
327  // Scale for the length of the game
328  int index = (trajectory->states.size() - 1) *
329  static_cast<double>(stage) / (stage_count - 1);
330  const Trajectory::State& s = trajectory->states[index];
331  value_accuracies[stage].Add(
332  (s.value >= 0) == (trajectory->returns[s.current_player] >= 0));
333  value_predictions[stage].Add(abs(s.value));
334  }
335  }
336  }
337  absl::Time now = absl::Now();
338  double seconds = absl::ToDoubleSeconds(now - last);
339  logger.Print("Step: %d", step);
340  logger.Print(
341  "Collected %5d states from %3d games, %.1f states/s; "
342  "%.1f states/(s*actor), game length: %.1f",
343  num_states, num_trajectories, num_states / seconds,
344  num_states / (config.actors * seconds),
345  static_cast<double>(num_states) / num_trajectories);
346  logger.Print("Queue size: %d. Buffer size: %d. States seen: %d",
347  queue_size, replay_buffer.Size(), replay_buffer.TotalAdded());
348 
349  if (stop->StopRequested()) {
350  break;
351  }
352 
353  last = now;
354 
355  VPNetModel::LossInfo losses;
356  { // Extra scope to return the device for use for inference asap.
357  DeviceManager::DeviceLoan learn_model =
358  device_manager->Get(config.train_batch_size, device_id);
359 
360  // Learn from them.
361  for (int i = 0; i < replay_buffer.Size() / config.train_batch_size; i++) {
362  losses += learn_model->Learn(replay_buffer.Sample(
363  &rng, config.train_batch_size));
364  }
365  }
366 
367  // Always save a checkpoint, either for keeping or for loading the weights
368  // to the other sessions. It only allows numbers, so use -1 as "latest".
369  std::string checkpoint_path =
370  device_manager->Get(0, device_id)->SaveCheckpoint(
371  step % config.checkpoint_freq == 0 ? step : -1);
372  if (device_manager->Count() > 0) {
373  for (int i = 0; i < device_manager->Count(); ++i) {
374  if (i != device_id) {
375  device_manager->Get(0, i)->LoadCheckpoint(checkpoint_path);
376  }
377  }
378  }
379  logger.Print("Checkpoint saved: %s", checkpoint_path);
380 
381  DataLogger::Record record = {
382  {"step", step},
383  {"total_states", replay_buffer.TotalAdded()},
384  {"states_per_s", num_states / seconds},
385  {"states_per_s_actor", num_states / (config.actors * seconds)},
386  {"total_trajectories", total_trajectories},
387  {"trajectories_per_s", num_trajectories / seconds},
388  {"queue_size", queue_size},
389  {"game_length", game_lengths.ToJson()},
390  {"game_length_hist", game_lengths_hist.ToJson()},
391  {"outcomes", outcomes.ToJson()},
392  {"value_accuracy", json::TransformToArray(
393  value_accuracies, [](auto v){ return v.ToJson(); })},
394  {"value_prediction", json::TransformToArray(
395  value_predictions, [](auto v){ return v.ToJson(); })},
396  {"eval", json::Object({
397  {"count", eval_results->EvalCount()},
398  {"results", json::CastToArray(eval_results->AvgResults())},
399  })},
400  {"batch_size", eval->BatchSizeStats().ToJson()},
401  {"batch_size_hist", eval->BatchSizeHistogram().ToJson()},
402  {"loss", json::Object({
403  {"policy", losses.Policy()},
404  {"value", losses.Value()},
405  {"l2reg", losses.L2()},
406  {"sum", losses.Total()},
407  })},
408  };
409  eval->ResetBatchSizeStats();
410  logger.Print("Losses: policy: %.4f, value: %.4f, l2: %.4f, sum: %.4f",
411  losses.Policy(), losses.Value(), losses.L2(), losses.Total());
412 
413  LRUCacheInfo cache_info = eval->CacheInfo();
414  if (cache_info.size > 0) {
415  logger.Print(absl::StrFormat(
416  "Cache size: %d/%d: %.1f%%, hits: %d, misses: %d, hit rate: %.3f%%",
417  cache_info.size, cache_info.max_size, 100.0 * cache_info.Usage(),
418  cache_info.hits, cache_info.misses, 100.0 * cache_info.HitRate()));
419  eval->ClearCache();
420  }
421  record.emplace("cache", json::Object({
422  {"size", cache_info.size},
423  {"max_size", cache_info.max_size},
424  {"usage", cache_info.Usage()},
425  {"requests", cache_info.Total()},
426  {"requests_per_s", cache_info.Total() / seconds},
427  {"hits", cache_info.hits},
428  {"misses", cache_info.misses},
429  {"misses_per_s", cache_info.misses / seconds},
430  {"hit_rate", cache_info.HitRate()},
431  }));
432 
433  data_logger.Write(record);
434  logger.Print("");
435  }
436 }

438 bool AlphaZero(AlphaZeroConfig config, StopToken* stop) {
439  std::shared_ptr<const open_spiel::Game> game =
440  open_spiel::LoadGame(config.game);
441 
442  open_spiel::GameType game_type = game->GetType();
443  if (game->NumPlayers() != 2)
444  open_spiel::SpielFatalError("AlphaZero can only handle 2-player games.");
445  if (game_type.reward_model != open_spiel::GameType::RewardModel::kTerminal)
446  open_spiel::SpielFatalError("Game must have terminal rewards.");
447  if (game_type.dynamics != open_spiel::GameType::Dynamics::kSequential)
448  open_spiel::SpielFatalError("Game must have sequential turns.");
449  if (game_type.chance_mode != open_spiel::GameType::ChanceMode::kDeterministic)
450  open_spiel::SpielFatalError("Game must be deterministic.");
451 
452  file::Mkdirs(config.path);
453  if (!file::IsDirectory(config.path)) {
454  std::cerr << config.path << " is not a directory." << std::endl;
455  return false;
456  }
457 
458  std::cout << "Logging directory: " << config.path << std::endl;
459 
460  if (config.graph_def.empty()) {
461  config.graph_def = "vpnet.pb";
462  std::string model_path = absl::StrCat(config.path, "/", config.graph_def);
463  if (file::Exists(model_path)) {
464  std::cout << "Overwriting existing model: " << model_path << std::endl;
465  } else {
466  std::cout << "Creating model: " << model_path << std::endl;
467  }
468  SPIEL_CHECK_TRUE(CreateGraphDef(
469  *game, config.learning_rate, config.weight_decay,
470  config.path, config.graph_def,
471  config.nn_model, config.nn_width, config.nn_depth));
472  } else {
473  std::string model_path = absl::StrCat(config.path, "/", config.graph_def);
474  if (file::Exists(model_path)) {
475  std::cout << "Using existing model: " << model_path << std::endl;
476  } else {
477  std::cout << "Model not found: " << model_path << std::endl;
478  }
479  }
480 
481  std::cout << "Playing game: " << config.game << std::endl;
482 
483  config.inference_batch_size = std::max(1, std::min(
484  config.inference_batch_size, config.actors + config.evaluators));
485 
486  config.inference_threads = std::max(1, std::min(
487  config.inference_threads, (1 + config.actors + config.evaluators) / 2));
488 
489  {
490  file::File fd(config.path + "/config.json", "w");
491  fd.Write(json::ToString(config.ToJson(), true) + "\n");
492  }
493 
494  DeviceManager device_manager;
495  for (const absl::string_view& device : absl::StrSplit(config.devices, ',')) {
496  device_manager.AddDevice(VPNetModel(
497  *game, config.path, config.graph_def, std::string(device)));
498  }
499 
500  if (device_manager.Count() == 0) {
501  std::cerr << "No devices specified?" << std::endl;
502  return false;
503  }
504 
505  { // Make sure they're all in sync.
506  std::string first_checkpoint = device_manager.Get(0)->SaveCheckpoint(0);
507  for (int i = 1; i < device_manager.Count(); ++i) {
508  device_manager.Get(0, i)->LoadCheckpoint(first_checkpoint);
509  }
510  }
511 
512  auto eval = std::make_shared<VPNetEvaluator>(
513  &device_manager, config.inference_batch_size, config.inference_threads,
514  config.inference_cache, (config.actors + config.evaluators) / 16);
515 
516  ThreadedQueue<Trajectory> trajectory_queue(
517  config.replay_buffer_size / config.replay_buffer_reuse);
518 
519  EvalResults eval_results(config.eval_levels, config.evaluation_window);
520 
521  std::vector<Thread> actors;
522  actors.reserve(config.actors);
523  for (int i = 0; i < config.actors; ++i) {
524  actors.emplace_back(
525  [&, i]() { actor(*game, config, i, &trajectory_queue, eval, stop); });
526  }
527  std::vector<Thread> evaluators;
528  evaluators.reserve(config.evaluators);
529  for (int i = 0; i < config.evaluators; ++i) {
530  evaluators.emplace_back(
531  [&, i]() { evaluator(*game, config, i, &eval_results, eval, stop); });
532  }
533  learner(*game, config, &device_manager, eval, &trajectory_queue,
534  &eval_results, stop);
535 
536  if (!stop->StopRequested()) {
537  stop->Stop();
538  }
539 
540  // Empty the queue so that the actors can exit.
541  trajectory_queue.BlockNewValues();
542  trajectory_queue.Clear();
543 
544  std::cout << "Joining all the threads." << std::endl;
545  for (auto& t : actors) {
546  t.join();
547  }
548  for (auto& t : evaluators) {
549  t.join();
550  }
551  std::cout << "Exiting cleanly." << std::endl;
552  return true;
553 }
