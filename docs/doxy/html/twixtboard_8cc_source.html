178void Board::InitializeBlockerMap(Position position, int dir,
179 const LinkDescriptor& ld) {
180 Link link = {position, dir};
181 for (auto&& entry : ld.blocking_links) {
182 Position fromPosition = position + entry.position;
183 if (!PositionIsOffBoard(fromPosition)) {
184 const LinkDescriptor& oppLd = kLinkDescriptorTable[entry.direction];
185 Position toPosition = position + entry.position + oppLd.offsets;
186 if (!PositionIsOffBoard(toPosition)) {
187 BlockerMap::PushBlocker(link, {fromPosition, entry.direction});
188 BlockerMap::PushBlocker(link, {toPosition, OppDir(entry.direction)});
189 }
190 }
191 }

194void Board::UpdateResult(Player player, Position position) {
195 // check for WIN
196 bool connected_to_start = GetCell(position).IsLinkedToBorder(player, kStart);
197 bool connected_to_end = GetCell(position).IsLinkedToBorder(player, kEnd);
198 if (connected_to_start && connected_to_end) {
199 // peg is linked to both boarder lines
200 set_result(player == kRedPlayer ? kRedWin : kBlueWin);
201 return;
202 }
203
204 // check if opponent (player to turn next) has any legal moves left
205 if (!HasLegalActions(1 - player)) {
206 set_result(kDraw);
207 return;
208 }
209}

211void Board::InitializeCells(bool init_blocker_map) {
212 cell_.resize(size(), std::vector<Cell>(size()));
213 BlockerMap::ClearBlocker();
214
215 for (int x = 0; x < size(); x++) {
216 for (int y = 0; y < size(); y++) {
217 Position position = {x, y};
218 Cell& cell = GetCell(position);
219
220 // set color to EMPTY or OFFBOARD
221 if (PositionIsOffBoard(position)) {
222 cell.set_color(kOffBoard);
223 } else { // regular board
224 cell.set_color(kEmpty);
225 if (x == 0) {
226 cell.SetLinkedToBorder(kBluePlayer, kStart);
227 } else if (x == size() - 1) {
228 cell.SetLinkedToBorder(kBluePlayer, kEnd);
229 } else if (y == 0) {
230 cell.SetLinkedToBorder(kRedPlayer, kStart);
231 } else if (y == size() - 1) {
232 cell.SetLinkedToBorder(kRedPlayer, kEnd);
233 }
234 InitializeNeighbors(position, cell, init_blocker_map);
235 }
236 }
237 }

240void Board::InitializeNeighbors(Position position, Cell& cell,
241 bool init_blocker_map) {
242 for (int dir = 0; dir < kMaxCompass; dir++) {
243 const LinkDescriptor& ld = kLinkDescriptorTable[dir];
244 Position target_position = position + ld.offsets;
245 if (!PositionIsOffBoard(target_position)) {
246 if (init_blocker_map) {
247 InitializeBlockerMap(position, dir, ld);
248 }
249 cell.SetNeighbor(dir, target_position);
250 }
251 }

254void Board::InitializeLegalActions() {
255 int num_legal_actions_per_player = size() * (size() - 2);
256
257 for (Player p = 0; p < kNumPlayers; p++) {
258 legal_actions_[p].resize(num_legal_actions_per_player);
259 legal_actions_[p].clear();
260 }
261
262 for (int col = 0; col < size(); col++) {
263 for (int row = 0; row < size(); row++) {
264 Position pos = {col, row};
265 Action action = col * size() + row;
266 if (PositionIsOffBoard(pos)) {
267 continue;
268 } else if (PositionIsOnBorder(kRedPlayer, pos)) {
269 legal_actions_[kRedPlayer].push_back(action);
270 } else if (PositionIsOnBorder(kBluePlayer, pos)) {
271 legal_actions_[kBluePlayer].push_back(action);
272 } else {
273 legal_actions_[kRedPlayer].push_back(action);
274 legal_actions_[kBluePlayer].push_back(action);
275 }
276 }
277 }

280std::string Board::ToString() const {
281 std::string s = "";
282
283 // head line
284 s.append(" ");
285 for (int y = 0; y < size(); y++) {
286 std::string letter = "";
287 letter += static_cast<int>('a') + y;
288 letter += " ";
289 AppendColorString(s, kAnsiRed, letter);
290 }
291 s.append("\n");
292
293 for (int y = size() - 1; y >= 0; y--) {
294 // print "before" row
295 s.append(" ");
296 for (int x = 0; x < size(); x++) {
297 AppendBeforeRow(s, {x, y});
298 }
299 s.append("\n");
300
301 // print "peg" row
302 size() - y < 10 ? s.append(" ") : s.append(" ");
303 AppendColorString(s, kAnsiBlue, std::to_string(size() - y) + " ");
304 for (int x = 0; x < size(); x++) {
305 AppendPegRow(s, {x, y});
306 }
307 s.append("\n");
308
309 // print "after" row
310 s.append(" ");
311 for (int x = 0; x < size(); x++) {
312 AppendAfterRow(s, {x, y});
313 }
314 s.append("\n");
315 }
316 s.append("\n");
317
318 if (swapped_) s.append("[swapped]");
319
320 switch (result_) {
321 case kOpen: {
322 break;
323 }
324 case kRedWin: {
325 s.append("[x has won]");
326 break;
327 }
328 case kBlueWin: {
329 s.append("[o has won]");
330 break;
331 }
332 case kDraw: {
333 s.append("[draw]");
334 break;
335 }
336 default: {
337 break;
338 }
339 }
340
341 return s;
342}

454void Board::ApplyAction(Player player, Action action) {
455 Position position = ActionToPosition(action);
456
457 if (move_counter() == 1) {
458 // it's the second position
459 if (position == move_one()) {
460 // blue player swapped
461 set_swapped(true);
462
463 // undo the first move: (remove peg and restore legal actions)
464 UndoFirstMove();
465
466 // turn position 90° clockwise:
467 // [2,3]->[3,5]; [1,4]->[4,6]; [3,2]->[2,4]
468 int x = position.y;
469 int y = size() - position.x - 1;
470 position = {x, y};
471
472 } else {
473 // blue player hasn't swapped => regular move
474 // remove move one from legal moves
475 RemoveLegalAction(kRedPlayer, move_one());
476 RemoveLegalAction(kBluePlayer, move_one());
477 }
478 }
479
480 SetPegAndLinks(player, position);
481
482 if (move_counter() == 0) {
483 // do not remove the move from legal actions but store it
484 // because second player might want to swap, by choosing the same move
485 set_move_one(position);
486 } else {
487 // otherwise remove move from legal actions
488 RemoveLegalAction(kRedPlayer, position);
489 RemoveLegalAction(kBluePlayer, position);
490 }
491
492 IncMoveCounter();
493
494 // Update the predicted result and update current_player_...
495 UpdateResult(player, position);
496}

498void Board::SetPegAndLinks(Player player, Position position) {
499 bool linked_to_neutral = false;
500 bool linked_to_start = false;
501 bool linked_to_end = false;
502
503 // set peg
504 Cell& cell = GetCell(position);
505 cell.set_color(player);
506
507 int dir = 0;
508 bool newLinks = false;
509 // check all neigbors that are empty or have same color)
510 for (dir = 0; dir < kMaxCompass; dir++) {
511 Position target_position = position + kLinkDescriptorTable[dir].offsets;
512 if (!PositionIsOffBoard(target_position)) {
513 Cell& target_cell = GetCell(target_position);
514 if (target_cell.color() == cell.color()) {
515 // check if there are blocking links before setting link
516 const std::set<Link>& blockers =
517 BlockerMap::GetBlockers((Link){position, dir});
518 bool blocked = false;
519 for (auto& bl : blockers) {
520 if (GetCell(bl.position).HasLink(bl.direction)) {
521 blocked = true;
522 break;
523 }
524 }
525
526 if (!blocked) {
527 // we set the link, and set the flag that there is at least one new
528 // link
529 cell.set_link(dir);
530 target_cell.set_link(OppDir(dir));
531
532 newLinks = true;
533
534 // check if cell we link to is linked to START border / END border
535 if (target_cell.IsLinkedToBorder(player, kStart)) {
536 cell.SetLinkedToBorder(player, kStart);
537 linked_to_start = true;
538 } else if (target_cell.IsLinkedToBorder(player, kEnd)) {
539 cell.SetLinkedToBorder(player, kEnd);
540 linked_to_end = true;
541 } else {
542 linked_to_neutral = true;
543 }
544 } else {
545 // we store the fact that these two pegs of the same color cannot be
546 // linked this info is used for the ObservationTensor
547 cell.SetBlockedNeighbor(dir);
548 target_cell.SetBlockedNeighbor(OppDir(dir));
549 }
550 } // same color
551 } // is on board
552 } // range of directions
553
554 // check if we need to explore further
555 if (newLinks) {
556 std::set<Cell*> visited = {};
557 if (cell.IsLinkedToBorder(player, kStart) && linked_to_neutral) {
558 // case: new cell is linked to START and linked to neutral cells
559 // => explore neutral graph and add all its cells to START
560 ExploreLocalGraph(player, cell, kStart, visited);
561 }
562 if (cell.IsLinkedToBorder(player, kEnd) && linked_to_neutral) {
563 // case: new cell is linked to END and linked to neutral cells
564 // => explore neutral graph and add all its cells to END
565 ExploreLocalGraph(player, cell, kEnd, visited);
566 }
567 }

570void Board::ExploreLocalGraph(Player player, Cell& cell, enum Border border,
571 std::set<Cell*> visited) {
572 visited.insert(&cell);
573 for (int dir = 0; dir < kMaxCompass; dir++) {
574 if (cell.HasLink(dir)) {
575 Cell& target_cell = GetCell(cell.GetNeighbor(dir));
576 if ((visited.find(&target_cell) == visited.end()) &&
577 !target_cell.IsLinkedToBorder(player, border)) {
578 // linked neighbor has not been visited yet
579 // => add it and explore
580 target_cell.SetLinkedToBorder(player, border);
581 ExploreLocalGraph(player, target_cell, border, visited);
582 }
583 }
584 }

611bool Board::PositionIsOnBorder(Player player, Position position) const {
612 if (player == kRedPlayer) {
613 return ((position.y == 0 || position.y == size() - 1) &&
614 (position.x > 0 && position.x < size() - 1));
615 } else {
616 return ((position.x == 0 || position.x == size() - 1) &&
617 (position.y > 0 && position.y < size() - 1));
618 }
619}

621bool Board::PositionIsOffBoard(Position position) const {
622 return (position.y < 0 || position.y > size() - 1 || position.x < 0 ||
623 position.x > size() - 1 ||
624 // corner case
625 ((position.x == 0 || position.x == size() - 1) &&
626 (position.y == 0 || position.y == size() - 1)));
627}

629void Board::RemoveLegalAction(Player player, Position position) {
630 Action action = PositionToAction(position);

632 std::vector<Action>::iterator it;
633 it = find(la.begin(), la.end(), action);
634 if (it != la.end()) la.erase(it);
