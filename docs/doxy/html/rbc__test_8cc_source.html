28void TestPassMove() {
29 auto game = LoadGame("rbc");
30 std::unique_ptr<State> s = game->NewInitialState();
31 SPIEL_CHECK_EQ(s->ToString(), chess::kDefaultStandardFEN);
32
33 // First move
34 SPIEL_CHECK_EQ(s->ActionToString(Player{0}, 0), "Sense a1");
35 SPIEL_CHECK_EQ(s->StringToAction("Sense a1"), 0);
36 s->ApplyAction(0); // Sense phase
37 SPIEL_CHECK_EQ(s->ActionToString(Player{0}, chess::kPassAction), "pass");
38 SPIEL_CHECK_EQ(s->StringToAction("pass"), chess::kPassAction);
39 s->ApplyAction(chess::kPassAction); // Move phase
40 std::string black_fen = chess::kDefaultStandardFEN;
41 std::replace(black_fen.begin(), black_fen.end(), 'w', 'b'); // Switch sides.
42 SPIEL_CHECK_EQ(s->ToString(), black_fen);
43
44 // Second move
45 SPIEL_CHECK_EQ(s->ActionToString(Player{1}, 0), "Sense a1");
46 s->ApplyAction(0); // Sense phase
47 SPIEL_CHECK_EQ(s->ActionToString(Player{1}, chess::kPassAction), "pass");
48 s->ApplyAction(chess::kPassAction); // Move phase
49 std::string white_fen = chess::kDefaultStandardFEN;
50 std::replace(white_fen.begin(), white_fen.end(), '1', '2'); // Update clock.
51 SPIEL_CHECK_EQ(s->ToString(), white_fen);
52}

54void TestRepetitionDraw() {
55 auto game = LoadGame("rbc");
56 auto state = game->NewInitialState();
57 auto rbc_state = down_cast<RbcState*>(state.get());
58 for (int i = 0; i < 2 * 2; ++i) { // 2 players, 2 repetitions.
59 SPIEL_CHECK_FALSE(state->IsTerminal());
60 SPIEL_CHECK_FALSE(rbc_state->IsRepetitionDraw());
61 state->ApplyAction(state->StringToAction("Sense a1"));
62 state->ApplyAction(chess::kPassAction);
63 }
64 SPIEL_CHECK_TRUE(state->IsTerminal());
65 SPIEL_CHECK_TRUE(rbc_state->IsRepetitionDraw());
66}

76void TestIllegalMovesFlag() {
77 // Setup test code.
78 auto game = LoadGame("rbc");
79 Observation observation(*game, game->MakeObserver(kDefaultObsType, {}));
80 SpanTensor illegal_move_span = GetSpanTensor(observation, "illegal_move");
81 SPIEL_CHECK_EQ(illegal_move_span.info().size(), 2); // Binary observation.
82 auto CHECK_OBSERVATION = [&](const State& s, bool illegal) {
83 observation.SetFrom(s, kDefaultPlayerId);
84 SPIEL_CHECK_EQ(illegal_move_span.at(0), !illegal);
85 SPIEL_CHECK_EQ(illegal_move_span.at(1), illegal);
86 };
87
88 { // No move has been made.
89 auto state = game->NewInitialState();
90 CHECK_OBSERVATION(*state, /*illegal=*/false);
91 }
92 { // Legal pawn move.
93 auto state = game->NewInitialState();
94 state->ApplyAction(state->StringToAction("Sense a1"));
95 state->ApplyAction(state->StringToAction("a2a4"));
96 CHECK_OBSERVATION(*state, /*illegal=*/false);
97 }
98 { // Illegal pawn attack.
99 auto state = game->NewInitialState();
100 state->ApplyAction(state->StringToAction("Sense a1"));
101 state->ApplyAction(state->StringToAction("a2b3"));
102 CHECK_OBSERVATION(*state, /*illegal=*/true);
103 }
104 { // Illegal pawn forward move.
105 auto state = game->NewInitialState();
106 state->ApplyAction(state->StringToAction("Sense a1"));
107 state->ApplyAction(state->StringToAction("a2a4"));
108 CHECK_OBSERVATION(*state, /*illegal=*/false);
109 state->ApplyAction(state->StringToAction("Sense a1"));
110 state->ApplyAction(state->StringToAction("a7a5"));
111 CHECK_OBSERVATION(*state, /*illegal=*/false);
112 state->ApplyAction(state->StringToAction("Sense a1"));
113 state->ApplyAction(state->StringToAction("a4a5"));
114 CHECK_OBSERVATION(*state, /*illegal=*/true);
115 }
116 { // Allow castling when king is in check.
117 auto state = game->NewInitialState(
118 "rnbqkb1r/pppppp1p/6p1/8/8/3n1NPB/PPP1PP1P/RNBQK2R w KQkq - 0 1");
119 state->ApplyAction(state->StringToAction("Sense a1"));
120 state->ApplyAction(state->StringToAction("e1g1"));
121 CHECK_OBSERVATION(*state, /*illegal=*/false);
122 }
123 { // Allow castling through a square controlled by the enemy.
124 auto state = game->NewInitialState(
125 "rnbqkb1r/pppppp1p/6p1/8/8/5NPB/PPPnPP1P/RNBQK2R w KQkq - 0 1");
126 state->ApplyAction(state->StringToAction("Sense a1"));
127 state->ApplyAction(state->StringToAction("e1g1"));
128 CHECK_OBSERVATION(*state, /*illegal=*/false);
129 }
130 { // Allow castling when king will be in check after castling.
131 auto state = game->NewInitialState(
132 "rnbqkb1r/pppppp1p/6p1/8/8/5nPB/PPP1PP1P/RNBQK2R w KQkq - 0 1");
133 state->ApplyAction(state->StringToAction("Sense a1"));
134 state->ApplyAction(state->StringToAction("e1g1"));
135 CHECK_OBSERVATION(*state, /*illegal=*/false);
136 }
137 { // Illegal castling:
138 // There is an opponent piece between the king and the rook.
139 auto state = game->NewInitialState(
140 "rnbqkb1r/pppppp1p/6p1/8/8/6PB/PPP1PP1P/RNBQK1nR w KQkq - 0 1");
141 state->ApplyAction(state->StringToAction("Sense a1"));
142 state->ApplyAction(state->StringToAction("e1g1"));
143 CHECK_OBSERVATION(*state, /*illegal=*/true);
144 }
145}

147void TestKingCaptureEndsTheGame() {

150 "rnbqk1nr/pppp1ppp/4p3/8/4P3/3P1Pb1/PPP3PP/RNBQKBNR b KQkq - 0 1");
