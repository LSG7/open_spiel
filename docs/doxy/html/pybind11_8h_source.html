217// Custom caster for GameParameter (essentially a variant).
218namespace pybind11 {
219namespace detail {
220
221template <>
222struct type_caster<open_spiel::GameParameter> {
223 public:
224 PYBIND11_TYPE_CASTER(open_spiel::GameParameter, _("GameParameter"));
225
226 bool load(handle src, bool convert) {
227 if (src.is_none()) {
228 // value is default-constructed to an unset value
229 return true;
230 } else if (PyBool_Check(src.ptr())) {
231 value = open_spiel::GameParameter(src.cast<bool>());
232 return true;
233 } else if (auto str_val = maybe_load<std::string>(src, convert)) {
234 value = open_spiel::GameParameter(*str_val);
235 return true;
236 } else if (PyFloat_Check(src.ptr())) {
237 value = open_spiel::GameParameter(src.cast<double>());
238 return true;
239 } else if (PyLong_Check(src.ptr())) {
240 value = open_spiel::GameParameter(src.cast<int>());
241 return true;
242 } else {
243 auto dict = src.cast<pybind11::dict>();
244 std::map<std::string, open_spiel::GameParameter> d;
245 for (const auto& [k, v] : dict) {
246 d[k.cast<std::string>()] = v.cast<open_spiel::GameParameter>();
247 }
248 value = open_spiel::GameParameter(d);
249 return true;
250 }
251 }
252
253 static handle cast(const open_spiel::GameParameter& gp,
254 return_value_policy policy, handle parent) {
255 if (gp.has_bool_value()) {
256 return pybind11::bool_(gp.bool_value()).release();
257 } else if (gp.has_double_value()) {
258 return pybind11::float_(gp.double_value()).release();
259 } else if (gp.has_string_value()) {
260 return pybind11::str(gp.string_value()).release();
261 } else if (gp.has_int_value()) {
262 return pybind11::int_(gp.int_value()).release();
263 } else if (gp.has_game_value()) {
264 pybind11::dict d;
265 for (const auto& [k, v] : gp.game_value()) {
266 d[pybind11::str(k)] = pybind11::cast(v);
267 }
268 return d.release();
269 } else {
270 return pybind11::none();
271 }
272 }
273
274 private:
275 template <typename T>
276 absl::optional<T> maybe_load(handle src, bool convert) {
277 auto caster = pybind11::detail::make_caster<T>();
278 if (caster.load(src, convert)) {
279 return cast_op<T>(caster);
280 } else {
281 return absl::nullopt;
282 }
283 }
284};
