59 // We want to enable conversion from vector<T*> to Span<const T* const> but
60 // disable conversion from vector<Derived> to Span<Base>. Here we use
61 // the fact that U** is convertible to Q* const* if and only if Q is the same
62 // type or a more cv-qualified version of U. We also decay the result type of
63 // data() to avoid problems with classes which have a member function data()
64 // which returns a reference.
65 template <typename T, typename C>
66 using HasData =
67  std::is_convertible<absl::decay_t<decltype(GetData(std::declval<C&>()))>*,
68  T* const*>;

106 // IsView is true for types where the return type of .data() is the same for
107 // mutable and const instances. This isn't foolproof, but it's only used to

114 template <typename T>
115 struct IsView<
116  T, absl::void_t<decltype(span_internal::GetData(std::declval<const T&>()))>,
117  absl::void_t<decltype(span_internal::GetData(std::declval<T&>()))>> {
118  private:
119  using Container = std::remove_const_t<T>;
120  using ConstData =
121  decltype(span_internal::GetData(std::declval<const Container&>()));
122  using MutData = decltype(span_internal::GetData(std::declval<Container&>()));
123  public:
