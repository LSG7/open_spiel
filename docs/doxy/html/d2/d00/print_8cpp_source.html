96string equals_to_string(const int equals)
97{
98 string st = "";
99 for (unsigned i = 15; i >= 2; i--)
100 {
101 if (equals & dbitMapRank[i])
102 st += static_cast<char>(dcardRank[i]);
103 }
104 return (st == "" ? "-" : st);
105}

180
181 const int m = min(solved.number, ref.number);
182 for (int i = 0; i < m; i++)
183 {
184 cout << "Trick " << i << ": " <<
185 solved.tricks[i] << " vs " <<
186 ref.tricks[i] <<
187 (solved.tricks[i] == ref.tricks[i] ? "" : " ERROR") << "\n";
188 }
189
190 if (solved.number > m)
191 {
192 for (int i = m; i < solved.number; i++)
193 cout << "Solved " << i << ": " << solved.tricks[i] << "\n";
194 }
195 else if (ref.number > m)

197 for (int i = m; i < ref.number; i++)

199 }
