28class EVATest(parameterized.TestCase):
29
30 @parameterized.parameters("tic_tac_toe", "kuhn_poker", "liars_dice")
31 def test_run_games(self, game):
32 env = rl_environment.Environment(game)
33 num_players = env.num_players
34 eva_agents = []
35 num_actions = env.action_spec()["num_actions"]
36 state_size = env.observation_spec()["info_state"][0]
37 for player in range(num_players):
38 eva_agents.append(
39 eva.EVAAgent(
40 env,
41 player,
42 state_size,
43 num_actions,
44 embedding_network_layers=(64, 32),
45 embedding_size=12,
46 learning_rate=1e-4,
47 mixing_parameter=0.5,
48 memory_capacity=int(1e6),
49 discount_factor=1.0,
50 epsilon_start=1.0,
51 epsilon_end=0.1,
52 epsilon_decay_duration=int(1e6)))
53 time_step = env.reset()
54 while not time_step.last():
55 current_player = time_step.observations["current_player"]
56 current_agent = eva_agents[current_player]
57 # 1. Step the agent.
58 # 2. Step the Environment.
59 agent_output = current_agent.step(time_step)
60 time_step = env.step([agent_output.action])
61 for agent in eva_agents:
62 agent.step(time_step)
