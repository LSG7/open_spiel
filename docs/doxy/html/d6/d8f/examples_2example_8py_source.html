15"""Python spiel example."""
16
17import random
18from absl import app
19from absl import flags
20import numpy as np
21
22from open_spiel.python import games # pylint: disable=unused-import
23import pyspiel
24
25FLAGS = flags.FLAGS
26
27# Game strings can just contain the name or the name followed by parameters
28# and arguments, e.g. "breakthrough(rows=6,columns=6)"
29flags.DEFINE_string("game_string", "tic_tac_toe", "Game string")
30
31
32def main(_):
33 games_list = pyspiel.registered_games()
34 print("Registered games:")
35 print(games_list)
36
37 action_string = None
38
39 print("Creating game: " + FLAGS.game_string)
40 game = pyspiel.load_game(FLAGS.game_string)
41
42 # Create the initial state
43 state = game.new_initial_state()
44
45 # Print the initial state
46 print(str(state))
47
48 while not state.is_terminal():
49 # The state can be three different types: chance node,
50 # simultaneous node, or decision node
51 if state.is_chance_node():
52 # Chance node: sample an outcome
53 outcomes = state.chance_outcomes()
54 num_actions = len(outcomes)
55 print("Chance node, got " + str(num_actions) + " outcomes")
56 action_list, prob_list = zip(*outcomes)
57 action = np.random.choice(action_list, p=prob_list)
58 print("Sampled outcome: ",
59 state.action_to_string(state.current_player(), action))
60 state.apply_action(action)
61 elif state.is_simultaneous_node():
62 # Simultaneous node: sample actions for all players.
63 random_choice = lambda a: np.random.choice(a) if a else [0]
64 chosen_actions = [
65 random_choice(state.legal_actions(pid))
66 for pid in range(game.num_players())
67 ]
68 print("Chosen actions: ", [
69 state.action_to_string(pid, action)
70 for pid, action in enumerate(chosen_actions)
71 ])
72 state.apply_actions(chosen_actions)
73 else:
74 # Decision node: sample action for the single current player
75 action = random.choice(state.legal_actions(state.current_player()))
76 action_string = state.action_to_string(state.current_player(), action)
77 print("Player ", state.current_player(), ", randomly sampled action: ",
78 action_string)
79 state.apply_action(action)
80 print(str(state))
81
82 # Game is now done. Print utilities for each player
83 returns = state.returns()
84 for pid in range(game.num_players()):
85 print("Utility for player {} is {}".format(pid, returns[pid]))

88if __name__ == "__main__":
89 app.run(main)
