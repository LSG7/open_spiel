38 std::vector<int> ParsePayoffString(const std::string& str) {
39  std::vector<std::string> pieces = absl::StrSplit(str, ';');
40  std::vector<int> payoff;
41  for (const auto& piece : pieces) {
42  int val;
43  if (!absl::SimpleAtoi(piece, &val)) {
44  SpielFatalError(absl::StrCat("Could not parse piece '", piece,
45  "' of payoff string '", str,
46  "' as an integer"));
47  }
48  payoff.push_back(val);
49  }
50  return payoff;
51 }

113 std::vector<std::pair<Action, double>> TinyHanabiState::ChanceOutcomes() const {
114  if (!IsChanceNode()) return {};
115  std::vector<std::pair<Action, double>> outcomes;
116  const int num_outcomes = payoff_.NumChance();
117  const double p = 1.0 / num_outcomes;
118  outcomes.reserve(num_outcomes);
119  for (int i = 0; i < num_outcomes; ++i) outcomes.emplace_back(i, p);
120  return outcomes;
121 }

170 void TinyHanabiState::InformationStateTensor(Player player,
171  absl::Span<float> values) const {
172  SPIEL_CHECK_GE(player, 0);
173  SPIEL_CHECK_LT(player, num_players_);
174 
175  SPIEL_CHECK_EQ(values.size(), payoff_.NumChance() + payoff_.NumActions() *
176  payoff_.NumPlayers());
177  std::fill(values.begin(), values.end(), 0);
178  if (history_.size() > player) values.at(history_[player].action) = 1;
179  for (int i = payoff_.NumPlayers(); i < history_.size(); ++i) {
180  values.at(payoff_.NumChance() +
181  (i - payoff_.NumPlayers()) * payoff_.NumActions() +
182  history_[i].action) = 1;
183  }
184 }
