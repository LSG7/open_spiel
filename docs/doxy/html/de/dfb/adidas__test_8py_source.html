30 def test_adidas_on_prisoners_dilemma(self):
31 """Tests ADIDAS on a 2-player prisoner's dilemma game."""
32 # pylint:disable=bad-whitespace
33 pt_r = np.array([[-1, -3],
34 [0, -2]])
35 # pylint:enable=bad-whitespace
36 # shift tensor to ensure positivity required if run adidas w/ Tsallis entrpy
37 pt_r -= pt_r.min()
38 pt_c = pt_r.T # symmetric game
39 pt = np.stack((pt_r, pt_c), axis=0).astype(float)
40 pt /= pt.max() # arbitrary design choice to upper bound entries to 1
41 game = MatrixGame(pt, seed=0)
42 # for games with more than 2 players, see adidas_utils/games/big.py
43 solver = qre_anneal_sym.Solver(temperature=100,
44 proj_grad=False, euclidean=True,
45 lrs=(1e-4, 1e-4), exp_thresh=0.01,
46 rnd_init=True, seed=0)
47 # note we set rnd_init to True which initializes adidas' initial
48 # approximation to nash to a random point on the simplex. if rnd_init is
49 # False, adidas is initialized to uniform which is the Nash equilibrium
50 # of the prisoner's dilemma, in which case adidas trivially solves this
51 # game in 0 iterations.
52 lle = adidas.ADIDAS(seed=0)
53 lle.approximate_nash(game, solver, sym=True, num_iterations=1,
54 num_samples=1, num_eval_samples=int(1e5),
55 approx_eval=True, exact_eval=True,
56 avg_trajectory=False)
57 self.assertLess(lle.results['exps_exact'][-1], 0.2)

59 def test_adidas_on_elfarol(self):
60 """Test ADIDAS on a 10-player, symmetric El Farol bar game."""
61 game = ElFarol(n=10, c=0.7)
62 solver = qre_anneal_sym.Solver(temperature=100,
63 proj_grad=False, euclidean=False,
64 lrs=(1e-4, 1e-2), exp_thresh=0.01,
65 seed=0)
66 lle = adidas.ADIDAS(seed=0)
67 lle.approximate_nash(game, solver, sym=True, num_iterations=1,
68 num_samples=np.inf, num_eval_samples=int(1e5),
69 approx_eval=True, exact_eval=True,
70 avg_trajectory=False)
71 self.assertLess(lle.results['exps_exact'][-1], 0.5)
