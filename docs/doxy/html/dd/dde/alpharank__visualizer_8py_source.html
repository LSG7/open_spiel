45 def __init__(self,
46 payoff_tables,
47 rhos,
48 rho_m,
49 pi,
50 state_labels,
51 num_top_profiles=None):
52 """Initializes a network plotting object.
53
54 Args:
55 payoff_tables: List of game payoff tables, one for each agent identity.
56 Each payoff_table may be either a 2D numpy array, or a
57 _PayoffTableInterface object.
58 rhos: Fixation probabilities.
59 rho_m: Neutral fixation probability.
60 pi: Stationary distribution of fixation Markov chain defined by rhos.
61 state_labels: Labels corresponding to Markov states. For the
62 single-population case, state_labels should be a list of pure strategy
63 names. For the multi-population case, it
64 should be a dict with (key,value) pairs: (population
65 index,list of strategy names)
66 num_top_profiles: Set to (int) to show only the graph nodes corresponding
67 to the top k elements of stationary distribution, or None to show all.
68 """
69 self.fig = plt.figure(figsize=(10, 10))
70 self.num_populations = len(payoff_tables)
71 payoffs_are_hpt_format = utils.check_payoffs_are_hpt(payoff_tables)
72 self.num_strats_per_population = (
73 utils.get_num_strats_per_population(payoff_tables,
74 payoffs_are_hpt_format))
75 self.rhos = rhos
76 self.rho_m = rho_m
77 self.pi = pi
78 self.num_profiles = len(pi)
79 self.state_labels = state_labels
80 self.first_run = True
81 self.num_top_profiles = num_top_profiles
82
83 if self.num_top_profiles:
84 # More than total number of strats requested for plotting
85 if self.num_top_profiles > self.num_profiles:
86 self.num_top_profiles = self.num_profiles
87 # Skip the bottom num_profiles-k stationary strategies.
88 self.nodes_to_skip = list(self.pi.argsort()[:self.num_profiles -
89 self.num_top_profiles])
90 else:
91 self.nodes_to_skip = []
92
93 self._reset_cycle_counter()

98 def _draw_network(self):
99 """Draws the NetworkX object representing the underlying graph."""
100 plt.clf()
101
102 if self.num_populations == 1:
103 node_sizes = 5000
104 node_border_width = 1.
105 else:
106 node_sizes = 15000
107 node_border_width = 3.
108
109 vmin, vmax = 0, np.max(self.pi) + 0.1
110
111 nx.draw_networkx_nodes(
112 self.g,
113 self.pos,
114 node_size=node_sizes,
115 node_color=self.node_colors,
116 edgecolors="k",
117 cmap=plt.cm.Blues,
118 vmin=vmin,
119 vmax=vmax,
120 linewidths=node_border_width)
121
122 nx.draw_networkx_edges(
123 self.g,
124 self.pos,
125 node_size=node_sizes,
126 arrowstyle="->",
127 arrowsize=10,
128 edge_color=self.edge_colors,
129 edge_cmap=plt.cm.Blues,
130 width=5)
131
132 nx.draw_networkx_edge_labels(self.g, self.pos, edge_labels=self.edge_labels)
133
134 if self.num_populations > 1:
135 subnode_separation = 0.1
136 subgraph = nx.Graph()
137 for i_population in range(self.num_populations):
138 subgraph.add_node(i_population)
139
140 for i_strat_profile in self.g:
141 x, y = self.pos[i_strat_profile]
142 if self.num_populations == 1:
143 node_text = "$\\pi_{" + self.state_labels[i_strat_profile] + "}=$"
144 node_text += str(np.round(self.pi[i_strat_profile], decimals=2))
145 else:
146 node_text = "" # No text for multi-population case as plot gets messy
147 txt = plt.text(
148 x,
149 y,
150 node_text,
151 horizontalalignment="center",
152 verticalalignment="center",
153 fontsize=12)
154 txt.set_path_effects(
155 [PathEffects.withStroke(linewidth=3, foreground="w")])
156
157 if self.num_populations > 1:
158 sub_pos = nx.circular_layout(subgraph)
159 subnode_labels = dict()
160 strat_profile = utils.get_strat_profile_from_id(
161 self.num_strats_per_population, i_strat_profile)
162 for i_population in subgraph.nodes():
163 i_strat = strat_profile[i_population]
164 subnode_labels[i_population] = "$s^{" + str(i_population + 1) + "}="
165 subnode_labels[i_population] += (
166 self.state_labels[i_population][i_strat] + "$")
167 # Adjust the node positions generated by NetworkX's circular_layout(),
168 # such that the node for the 1st strategy starts on the left.
169 sub_pos[i_population] = (-sub_pos[i_population] * subnode_separation +
170 self.pos[i_strat_profile])
171 nx.draw(
172 subgraph,
173 pos=sub_pos,
174 with_labels=True,
175 width=0.,
176 node_color="w",
177 labels=subnode_labels,
178 node_size=2500)

180 def compute_and_draw_network(self):
181 """Computes the various node/edge connections of the graph and draws it."""
182
183 if np.max(self.rhos) < self.rho_m:
184 print("All node-to-node fixation probabilities (not including self-cycles"
185 " are lower than neutral. Thus, no graph will be drawn.")
186 return
187
188 self.g = nx.MultiDiGraph()
189 self.edge_labels = {}
190 self.edge_alphas = []
191 rho_max = np.max(self.rhos / self.rho_m)
192 rho_m_alpha = 0.1 # Transparency of neutral selection edges
193
194 for i in range(self.num_profiles):
195 for j in range(self.num_profiles):
196 # Do not draw edge if any node involved is skipped
197 if j not in self.nodes_to_skip and i not in self.nodes_to_skip:
198 rate = self.rhos[i][j] / self.rho_m
199 # Draws edges when fixation from one strategy to another occurs (i.e.,
200 # rate > 1), or with fixation equal to neutral selection probability
201 # (i.e., rate == 1). This is consistent with visualizations used in
202 # finite-population literature.
203 if rate > 1:
204 # Compute alphas. Clip needed due to numerical precision.
205 alpha = np.clip(rho_m_alpha + (1 - rho_m_alpha) * rate / rho_max,
206 None, 1.)
207 self.g.add_edge(i, j, weight=alpha, label="{:.01f}".format(rate))
208 self.edge_alphas.append(alpha)
209 elif np.isclose(rate, 1):
210 alpha = rho_m_alpha
211 self.g.add_edge(i, j, weight=alpha, label="{:.01f}".format(rate))
212 self.edge_alphas.append(alpha)
213 # Label edges for non-self-loops with sufficient flowrate
214 if i != j and rate > 1:
215 edge_string = "$" + str(np.round(rate, decimals=2)) + "\\rho_m$"
216 else:
217 edge_string = ""
218 self.edge_labels[(i, j)] = edge_string
219
220 # MultiDiGraph nodes are not ordered, so order the node colors accordingly
221 self.node_colors = [self.pi[node] for node in self.g.nodes()]
222
223 self.cycles = list(nx.simple_cycles(self.g))
224 self.num_cycles = len(self.cycles)
225
226 # Color the edges of cycles if user requested it
227 if self.i_cycle_to_show >= 0:
228 all_cycle_edges = [
229 zip(nodes, (nodes[1:] + nodes[:1])) for nodes in self.cycles
230 ]
231 cur_cycle_edges = all_cycle_edges[self.i_cycle_to_show]
232 self.edge_colors = []
233 for u, v in self.g.edges():
234 if (u, v) in cur_cycle_edges:
235 self.edge_colors.append([1., 0., 0.])
236 else:
237 self.edge_colors.append([1. - self.g[u][v][0]["weight"]] * 3)
238 else:
239 self.edge_colors = [
240 [1. - self.g[u][v][0]["weight"]] * 3 for u, v in self.g.edges()
241 ]
242 self.edge_alphas = [self.g[u][v][0]["weight"] for u, v in self.g.edges()]
243
244 ax = plt.gca()
245
246 # Centered circular pose
247 self.pos = nx.layout.circular_layout(self.g)
248 all_x = [node_pos[0] for node, node_pos in self.pos.items()]
249 all_y = [node_pos[1] for node, node_pos in self.pos.items()]
250 min_x = np.min(all_x)
251 max_x = np.max(all_x)
252 min_y = np.min(all_y)
253 max_y = np.max(all_y)
254 for _, node_pos in self.pos.items():
255 node_pos[0] -= (max_x + min_x) / 2
256 node_pos[1] -= (max_y + min_y) / 2
257
258 # Rendering
259 self._draw_network()
260 if self.first_run:
261 ax.autoscale_view()
262 ax.set_axis_off()
263 ax.set_aspect("equal")
264 plt.ylim(-1.3, 1.3)
265 plt.xlim(-1.3, 1.3)
266 if self.first_run:
267 self.first_run = False
268 plt.axis("off")
269 plt.show()

272def _draw_pie(ax,
273 ratios,
274 colors,
275 x_center=0,
276 y_center=0,
277 size=100,
278 clip_on=True,
279 zorder=0):
280 """Plots a pie chart.
281
282 Args:
283 ax: plot axis.
284 ratios: list indicating size of each pie slice, with elements summing to 1.
285 colors: list indicating color of each pie slice.
286 x_center: x coordinate of pie center.
287 y_center: y coordinate of pie center.
288 size: pie size.
289 clip_on: control clipping of pie (e.g., to show it when it's out of axis).
290 zorder: plot z order (e.g., to show pie on top of other plot elements).
291 """
292 xy = []
293 start = 0.
294 for ratio in ratios:
295 x = [0] + np.cos(
296 np.linspace(2 * np.pi * start, 2 * np.pi *
297 (start + ratio), 30)).tolist()
298 y = [0] + np.sin(
299 np.linspace(2 * np.pi * start, 2 * np.pi *
300 (start + ratio), 30)).tolist()
301 xy.append(list(zip(x, y)))
302 start += ratio
303
304 for i, xyi in enumerate(xy):
305 ax.scatter([x_center], [y_center],
306 marker=xyi,
307 s=size,
308 facecolor=colors[i],
309 edgecolors="none",
310 clip_on=clip_on,
311 zorder=zorder)

314def generate_sorted_masses_strats(pi_list, curr_alpha_idx, strats_to_go):
315 """Generates a sorted list of (mass, strats) tuples.
316
317 Args:
318 pi_list: List of stationary distributions, pi
319 curr_alpha_idx: Index in alpha_list for which to start clustering
320 strats_to_go: List of strategies that still need to be ordered
321
322 Returns:
323 Sorted list of (mass, strats) tuples.
324 """
325 if curr_alpha_idx > 0:
326 sorted_masses_strats = list()
327 masses_to_strats = utils.cluster_strats(pi_list[curr_alpha_idx,
328 strats_to_go])
329
330 for mass, strats in sorted(masses_to_strats.items(), reverse=True):
331 if len(strats) > 1:
332 to_append = generate_sorted_masses_strats(pi_list, curr_alpha_idx - 1,
333 strats)
334
335 to_append = [(mass, [strats_to_go[s]
336 for s in strats_list])
337 for (mass, strats_list) in to_append]
338
339 sorted_masses_strats.extend(to_append)
340 else:
341 sorted_masses_strats.append((mass, [
342 strats_to_go[strats[0]],
343 ]))
344
345 return sorted_masses_strats
346 else:
347 to_return = sorted(
348 utils.cluster_strats(pi_list[curr_alpha_idx, strats_to_go]).items(),
349 reverse=True)
350 to_return = [(mass, [strats_to_go[s]
351 for s in strats_list])
352 for (mass, strats_list) in to_return]
353 return to_return

356def plot_pi_vs_alpha(pi_list,
357 alpha_list,
358 num_populations,
359 num_strats_per_population,
360 strat_labels,
361 num_strats_to_label,
362 plot_semilogx=True,
363 xlabel=r"Ranking-intensity $\alpha$",
364 ylabel=r"Strategy mass in stationary distribution $\pi$",
365 legend_sort_clusters=False):
366 """Plots stationary distributions, pi, against selection intensities, alpha.
367
368 Args:
369 pi_list: List of stationary distributions, pi.
370 alpha_list: List of selection intensities, alpha.
371 num_populations: The number of populations.
372 num_strats_per_population: List of the number of strategies per population.
373 strat_labels: Human-readable strategy labels.
374 num_strats_to_label: The number of top strategies to label in the legend.
375 plot_semilogx: Boolean set to enable/disable semilogx plot.
376 xlabel: Plot xlabel.
377 ylabel: Plot ylabel.
378 legend_sort_clusters: If true, strategies in the same cluster are sorted in
379 the legend according to orderings for earlier alpha values. Primarily for
380 visualization purposes! Rankings for lower alpha values should be
381 interpreted carefully.
382 """
383
384 # Cluster strategies for which the stationary distribution has similar masses
385 masses_to_strats = utils.cluster_strats(pi_list[-1, :])
386
387 # Set colors
388 num_strat_profiles = np.shape(pi_list)[1]
389 num_strats_to_label = min(num_strats_to_label, num_strat_profiles)
390 cmap = plt.get_cmap("Paired")
391 colors = [cmap(i) for i in np.linspace(0, 1, num_strat_profiles)]
392
393 # Plots stationary distribution vs. alpha series
394 plt.figure(facecolor="w")
395 axes = plt.gca()
396
397 legend_line_objects = []
398 legend_labels = []
399
400 rank = 1
401 num_strats_printed = 0
402 add_legend_entries = True
403
404 if legend_sort_clusters:
405 sorted_masses_strats = generate_sorted_masses_strats(
406 pi_list, pi_list.shape[0] - 1, range(pi_list.shape[1]))
407 else:
408 sorted_masses_strats = sorted(masses_to_strats.items(), reverse=True)
409
410 for mass, strats in sorted_masses_strats:
411 for profile_id in strats:
412 if num_populations == 1:
413 strat_profile = profile_id
414 else:
415 strat_profile = utils.get_strat_profile_from_id(
416 num_strats_per_population, profile_id)
417
418 if plot_semilogx:
419 series = plt.semilogx(
420 alpha_list,
421 pi_list[:, profile_id],
422 color=colors[profile_id],
423 linewidth=2)
424 else:
425 series = plt.plot(
426 alpha_list,
427 pi_list[:, profile_id],
428 color=colors[profile_id],
429 linewidth=2)
430
431 if add_legend_entries:
432 if num_strats_printed >= num_strats_to_label:
433 # Placeholder blank series for remaining entries
434 series = plt.semilogx(np.NaN, np.NaN, "-", color="none")
435 label = "..."
436 add_legend_entries = False
437 else:
438 label = utils.get_label_from_strat_profile(num_populations,
439 strat_profile,
440 strat_labels)
441 legend_labels.append(label)
442 legend_line_objects.append(series[0])
443 num_strats_printed += 1
444 rank += 1
445
446 # Plots pie charts on far right of figure to indicate clusters of strategies
447 # with identical rank
448 for mass, strats in iter(masses_to_strats.items()):
449 _draw_pie(
450 axes,
451 ratios=[1 / len(strats)] * len(strats),
452 colors=[colors[i] for i in strats],
453 x_center=alpha_list[-1],
454 y_center=mass,
455 size=200,
456 clip_on=False,
457 zorder=10)
458
459 # Axes ymax set slightly above highest stationary distribution mass
460 max_mass = np.amax(pi_list)
461 axes_y_max = np.ceil(
462 10. * max_mass) / 10 # Round upward to nearest first decimal
463 axes_y_max = np.clip(axes_y_max, 0., 1.)
464
465 # Plots a rectangle highlighting the rankings on the far right of the figure
466 box_x_min = alpha_list[-1] * 0.7
467 box_y_min = np.min(pi_list[-1, :]) - 0.05 * axes_y_max
468 width = 0.7 * alpha_list[-1]
469 height = np.max(pi_list[-1, :]) - np.min(
470 pi_list[-1, :]) + 0.05 * axes_y_max * 2
471 axes.add_patch(
472 patches.Rectangle((box_x_min, box_y_min),
473 width,
474 height,
475 edgecolor="b",
476 facecolor=(1, 0, 0, 0),
477 clip_on=False,
478 linewidth=5,
479 zorder=20))
480
481 # Plot formatting
482 axes.set_xlim(np.min(alpha_list), np.max(alpha_list))
483 axes.set_ylim([0.0, axes_y_max])
484 axes.set_xlabel(xlabel)
485 axes.set_ylabel(ylabel)
486 axes.set_axisbelow(True) # Axes appear below data series in terms of zorder
487
488 # Legend on the right side of the current axis
489 box = axes.get_position()
490 axes.set_position([box.x0, box.y0, box.width * 0.8, box.height])
491 axes.legend(
492 legend_line_objects,
493 legend_labels,
494 loc="center left",
495 bbox_to_anchor=(1.05, 0.5))
496 plt.grid()
497 plt.show()
