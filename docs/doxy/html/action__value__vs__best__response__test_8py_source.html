28 def test_kuhn_poker_uniform(self):
29 game = pyspiel.load_game("kuhn_poker")
30 calc = action_value_vs_best_response.Calculator(game)
31 (expl, avvbr, cfrp,
32 player_reach_probs) = calc(0, policy.UniformRandomPolicy(game),
33 ["0", "1", "2", "0pb", "1pb", "2pb"])
34 self.assertAlmostEqual(expl, 15 / 36)
35 np.testing.assert_allclose(
36 avvbr,
37 [
38 [-1.5, -2.0], # 0 (better to pass)
39 [-0.5, -0.5], # 1 (same)
40 [0.5, 1.5], # 2 (better to bet)
41 [-1.0, -2.0], # 0pb - losing
42 [-1.0, 0.0], # 1pb - best response is bet always
43 [-1.0, 2.0], # 2pb - winning
44 ])
45 np.testing.assert_allclose(cfrp, [1 / 3, 1 / 3, 1 / 3, 1 / 3, 1 / 3, 1 / 3])
46 np.testing.assert_allclose([1, 1, 1, 1 / 2, 1 / 2, 1 / 2],
47 player_reach_probs)

49 def test_kuhn_poker_always_pass_p0(self):
50 game = pyspiel.load_game("kuhn_poker")
51 calc = action_value_vs_best_response.Calculator(game)
52 (expl, avvbr, cfrp, player_reach_probs) = calc(
53 0, policy.FirstActionPolicy(game),
54 ["0", "1", "2", "0pb", "1pb", "2pb"])
55 self.assertAlmostEqual(expl, 1.)
56 np.testing.assert_allclose(
57 avvbr,
58 [
59 # Opening bet. If we pass, we always lose (pass-pass with op's K,
60 # otherwise pass-bet-pass).
61 # If we bet, we always win (because op's best response is to pass,
62 # because this is an unreachable state and we break ties in favour
63 # of the lowest action).
64 [-1, 1],
65 [-1, 1],
66 [-1, 1],
67 # We pass, opp bets into us. This can be either J or Q (K will pass
68 # because of the tie-break rules).
69 # So we are guaranteed to be winning with Q or K.
70 [-1, -2], # 0pb
71 [-1, 2], # 1pb
72 [-1, 2], # 2pb
73 ])
74 np.testing.assert_allclose(cfrp, [1 / 3, 1 / 3, 1 / 3, 1 / 6, 1 / 6, 1 / 3])
75 np.testing.assert_allclose([1., 1., 1., 1., 1., 1.], player_reach_probs)
