76void CCEState::DoApplyAction(Action action_id) {
77 if (rec_index_ < 0) {
78 // Pick the joint policy which will provide recommendations.
79 rec_index_ = action_id;
80 SPIEL_CHECK_LT(rec_index_, mu_.size());
81 } else if (state_->IsChanceNode()) {
82 // Regular chance node
83 state_->ApplyAction(action_id);
84 } else {
85 // Regular decision node
86 state_->ApplyAction(action_id);
87 }
