22// Transforms a game into its Misere version by inverting the sign of the
23// rewards / utilities. This is a self-inverse operation.
24// https://en.wikipedia.org/wiki/Mis%C3%A8re
25
26namespace open_spiel {
27
28// Flips the sign of a vector.
29inline std::vector<double> Negative(std::vector<double>&& vector) {
30 std::vector<double> neg = std::move(vector);
31 for (auto& item : neg) item = -item;
32 return neg;
33}

35class MisereState : public WrappedState {
36 public:
37 MisereState(std::shared_ptr<const Game> game, std::unique_ptr<State> state)
38 : WrappedState(game, std::move(state)) {}
39 MisereState(const MisereState& other) = default;
40
41 std::vector<double> Rewards() const override {
42 return Negative(state_->Rewards());
43 }
44
45 std::vector<double> Returns() const override {
46 return Negative(state_->Returns());
47 }
48
49 std::unique_ptr<State> Clone() const override {
50 return std::unique_ptr<State>(new MisereState(*this));
51 }
52};

54class MisereGame : public WrappedGame {
55 public:
56 MisereGame(std::shared_ptr<const Game> game, GameType game_type,
57 GameParameters game_parameters);
58 MisereGame(const MisereGame& other) = default;
59
60 std::unique_ptr<State> NewInitialState() const override {
61 return std::unique_ptr<State>(
62 new MisereState(shared_from_this(), game_->NewInitialState()));
63 }
64
65 double MinUtility() const override { return -game_->MaxUtility(); }
66 double MaxUtility() const override { return -game_->MinUtility(); }
67 absl::optional<double> UtilitySum() const override {
68 auto base_game_utility_sum = game_->UtilitySum();
69 return !base_game_utility_sum.has_value() ? base_game_utility_sum
70 : -base_game_utility_sum.value();
71 }
72};
