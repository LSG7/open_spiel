27void init_pyspiel_observer(py::module& m) {
28 // C++ Observer, intended only for the Python Observation class, not
29 // for general Python code.
30 py::class_<Observer, std::shared_ptr<Observer>>(m, "Observer")
31 .def("__str__", [](const Observer& self) { return "Observer()"; });
32
33 py::class_<SpanTensorInfo>(m, "SpanTensorInfo")
34 .def_property_readonly(
35 "name", [](const SpanTensorInfo& info) { return info.name(); })
36 .def_property_readonly(
37 "shape",
38 [](const SpanTensorInfo& info) { return info.vector_shape(); })
39 .def("__str__", &SpanTensorInfo::DebugString);
40
41 py::class_<SpanTensor>(m, "SpanTensor")
42 .def_property_readonly(
43 "name", [](const SpanTensor& tensor) { return tensor.info().name(); })
44 .def_property_readonly(
45 "shape",
46 [](const SpanTensor& tensor) { return tensor.info().vector_shape(); })
47 .def_property_readonly("data",
48 [](const SpanTensor& tensor) {
49 // absl::Span requires pybind11_abseil which
50 // open spiel forbids. Thus copy the data
51 // and expose a vector through pybind.
52 std::vector<float> data(tensor.data().begin(),
53 tensor.data().end());
54 return data;
55 })
56 .def("__str__", &SpanTensor::DebugString);
57
58 // C++ Observation, intended only for the Python Observation class, not
59 // for general Python code.
60 py::class_<Observation>(m, "_Observation", py::buffer_protocol())
61 .def(py::init([](std::shared_ptr<const Game> game,
62 std::shared_ptr<Observer> observer) {
63 return new Observation(*game, observer);
64 }),
65 py::arg("game"), py::arg("observer"))
66 .def("tensors", &Observation::tensors)
67 .def("tensors_info", &Observation::tensors_info)
68 .def("string_from", &Observation::StringFrom)
69 .def("set_from", &Observation::SetFrom)
70 .def("has_string", &Observation::HasString)
71 .def("has_tensor", &Observation::HasTensor)
72 .def("compress",
73 [](const Observation& self) { return py::bytes(self.Compress()); })
74 .def("decompress", &Observation::Decompress)
75 .def_buffer([](Observation& buffer_observer) -> py::buffer_info {
76 return py::buffer_info(
77 buffer_observer.Tensor().data(), // Pointer to buffer
78 sizeof(float), // Size of one scalar
79 py::format_descriptor<float>::format(), // Format descriptor
80 1, // Num dimensions
81 {buffer_observer.Tensor().size()}, // Dimensions
82 {sizeof(float)} // Stride
83 );
84 });
85}
