31std::vector<double> _maxn(
32 const State* state, int depth,
33 std::function<double(const State&, Player player)> value_function,
34 Action* best_action) {
35 const int num_players = state->NumPlayers();
36
37 if (state->IsTerminal()) {
38 return state->Returns();
39 }
40
41 if (depth == 0 && !value_function) {
42 SpielFatalError(
43 "We assume we can walk the full depth of the tree. "
44 "Try increasing depth or provide a value_function.");
45 }
46
47 if (depth == 0) {
48 std::vector<double> values(num_players);
49 for (Player p = 0; p < num_players; ++p) {
50 values[p] = value_function(*state, p);
51 }
52 return values;
53 }
54
55 Player player = state->CurrentPlayer();
56 if (state->IsChanceNode()) {
57 std::vector<double> values(num_players, 0.0);
58 for (const auto& actionprob : state->ChanceOutcomes()) {
59 std::unique_ptr<State> child_state = state->Child(actionprob.first);
60 std::vector<double> child_values =
61 _maxn(child_state.get(), depth, value_function,
62 /*best_action=*/nullptr);
63 for (Player p = 0; p < num_players; ++p) {
64 values[p] += actionprob.second * child_values[p];
65 }
66 }
67 return values;
68 } else {
69 double value = -std::numeric_limits<double>::infinity();
70 std::vector<double> values(num_players, 0);
71
72 for (Action action : state->LegalActions()) {
73 std::unique_ptr<State> child_state = state->Child(action);
74 std::vector<double> child_values =
75 _maxn(child_state.get(),
76 /*depth=*/depth - 1, value_function,
77 /*best_action=*/nullptr);
78
79 if (child_values[player] > value) {
80 value = child_values[player];
81 values = child_values;
82 if (best_action != nullptr) {
83 *best_action = action;
84 }
85 }
86 }
87 return values;
88 }
89}

92std::pair<std::vector<double>, Action> MaxNSearch(
93 const Game& game, const State* state,
94 std::function<double(const State&, Player player)> value_function,
95 int depth_limit) {
96 GameType game_info = game.GetType();
97 SPIEL_CHECK_TRUE(
98 game_info.chance_mode == GameType::ChanceMode::kDeterministic ||
99 game_info.chance_mode == GameType::ChanceMode::kExplicitStochastic);
100 // Do not check perfect information. Used by PIMC.
101 SPIEL_CHECK_EQ(game_info.dynamics, GameType::Dynamics::kSequential);
102 SPIEL_CHECK_EQ(game_info.reward_model, GameType::RewardModel::kTerminal);
103
104 std::unique_ptr<State> search_root;
105 if (state == nullptr) {
106 search_root = game.NewInitialState();
107 } else {
108 search_root = state->Clone();
109 }
110
111 SPIEL_CHECK_FALSE(search_root->IsChanceNode());
112
113 Action best_action = kInvalidAction;
114 std::vector<double> values = _maxn(search_root.get(), /*depth=*/depth_limit,
115 value_function, &best_action);
116
117 return {values, best_action};
118}
