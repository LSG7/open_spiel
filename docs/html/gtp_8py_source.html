30  The Go Text Protocol, GTP, is a text based protocol for communication with
31  computer Go programs (https://www.lysator.liu.se/~gunnar/gtp/). It has also
32  been adopted by agents in other games including Hex and Havannah. If you need
33  to configure your agent in some specific way (eg time/resource limits), you

83  def gtp_cmd(self, *args):
84  """Send commands directly to the game, and get the response as a string."""
85  cmd = " ".join([str(a) for a in args]).encode()
86  self._process_process.stdin.write(cmd + b"\n")
87  response = ""
88  while True:
89  line = self._process_process.stdout.readline().decode()
90  if not line:
91  raise IOError("Engine closed the connection.")
92  if line == "\n":
93  if response:
94  break # A blank line signifies end of response.
95  else:
96  continue # Ignore leading newlines, possibly left from prev response.
97  response += line
98  if response.startswith("="):
99  return response[1:].strip()
100  else:
101  raise CommandError(response[1:].strip())

146 def _shutdown_proc(p, timeout):
147  """Waits for a proc to shut down; terminates or kills it after `timeout`."""
148  freq = 10 # how often to check per second
149  for _ in range(1 + timeout * freq):
150  p.terminate()
