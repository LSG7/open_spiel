42class NFSP(rl_agent.AbstractAgent):
43 """NFSP Agent implementation in TensorFlow.
44
45 See open_spiel/python/examples/kuhn_nfsp.py for an usage example.
46 """

48 def __init__(self,
49 session,
50 player_id,
51 state_representation_size,
52 num_actions,
53 hidden_layers_sizes,
54 reservoir_buffer_capacity,
55 anticipatory_param,
56 batch_size=128,
57 rl_learning_rate=0.01,
58 sl_learning_rate=0.01,
59 min_buffer_size_to_learn=1000,
60 learn_every=64,
61 optimizer_str="sgd",
62 **kwargs):
63 """Initialize the `NFSP` agent."""
64 self.player_id = player_id
65 self._session = session
66 self._num_actions = num_actions
67 self._layer_sizes = hidden_layers_sizes
68 self._batch_size = batch_size
69 self._learn_every = learn_every
70 self._anticipatory_param = anticipatory_param
71 self._min_buffer_size_to_learn = min_buffer_size_to_learn
72
73 self._reservoir_buffer = ReservoirBuffer(reservoir_buffer_capacity)
74 self._prev_timestep = None
75 self._prev_action = None
76
77 # Step counter to keep track of learning.
78 self._step_counter = 0
79
80 # Inner RL agent
81 kwargs.update({
82 "batch_size": batch_size,
83 "learning_rate": rl_learning_rate,
84 "learn_every": learn_every,
85 "min_buffer_size_to_learn": min_buffer_size_to_learn,
86 "optimizer_str": optimizer_str,
87 })
88 self._rl_agent = dqn.DQN(session, player_id, state_representation_size,
89 num_actions, hidden_layers_sizes, **kwargs)
90
91 # Keep track of the last training loss achieved in an update step.
92 self._last_rl_loss_value = lambda: self._rl_agent.loss
93 self._last_sl_loss_value = None
94
95 # Placeholders.
96 self._info_state_ph = tf.placeholder(
97 shape=[None, state_representation_size],
98 dtype=tf.float32,
99 name="info_state_ph")
100
101 self._action_probs_ph = tf.placeholder(
102 shape=[None, num_actions], dtype=tf.float32, name="action_probs_ph")
103
104 self._legal_actions_mask_ph = tf.placeholder(
105 shape=[None, num_actions],
106 dtype=tf.float32,
107 name="legal_actions_mask_ph")
108
109 # Average policy network.
110 self._avg_network = simple_nets.MLP(state_representation_size,
111 self._layer_sizes, num_actions)
112 self._avg_policy = self._avg_network(self._info_state_ph)
113 self._avg_policy_probs = tf.nn.softmax(self._avg_policy)
114
115 self._savers = [
116 ("q_network", tf.train.Saver(self._rl_agent._q_network.variables)),
117 ("avg_network", tf.train.Saver(self._avg_network.variables))
118 ]
119
120 # Loss
121 self._loss = tf.reduce_mean(
122 tf.nn.softmax_cross_entropy_with_logits_v2(
123 labels=tf.stop_gradient(self._action_probs_ph),
124 logits=self._avg_policy))
125
126 if optimizer_str == "adam":
127 optimizer = tf.train.AdamOptimizer(learning_rate=sl_learning_rate)
128 elif optimizer_str == "sgd":
129 optimizer = tf.train.GradientDescentOptimizer(
130 learning_rate=sl_learning_rate)
131 else:
132 raise ValueError("Not implemented. Choose from ['adam', 'sgd'].")
133
134 self._learn_step = optimizer.minimize(self._loss)
135 self._sample_episode_policy()
136
137 @contextlib.contextmanager
138 def temp_mode_as(self, mode):
139 """Context manager to temporarily overwrite the mode."""
140 previous_mode = self._mode
141 self._mode = mode
142 yield
143 self._mode = previous_mode
144
145 def get_step_counter(self):
146 return self._step_counter
147
148 def _sample_episode_policy(self):
149 if np.random.rand() < self._anticipatory_param:
150 self._mode = MODE.best_response
151 else:
152 self._mode = MODE.average_policy
153
154 def _act(self, info_state, legal_actions):
155 info_state = np.reshape(info_state, [1, -1])
156 action_values, action_probs = self._session.run(
157 [self._avg_policy, self._avg_policy_probs],
158 feed_dict={self._info_state_ph: info_state})
159
160 self._last_action_values = action_values[0]
161 # Remove illegal actions, normalize probs
162 probs = np.zeros(self._num_actions)
163 probs[legal_actions] = action_probs[0][legal_actions]
164 probs /= sum(probs)
165 action = np.random.choice(len(probs), p=probs)
166 return action, probs
167
168 @property
169 def mode(self):
170 return self._mode
171
172 @property
173 def loss(self):
174 return (self._last_sl_loss_value, self._last_rl_loss_value())
175
176 def step(self, time_step, is_evaluation=False):
177 """Returns the action to be taken and updates the Q-networks if needed.
178
179 Args:
180 time_step: an instance of rl_environment.TimeStep.
181 is_evaluation: bool, whether this is a training or evaluation call.
182
183 Returns:
184 A `rl_agent.StepOutput` containing the action probs and chosen action.
185 """
186 if self._mode == MODE.best_response:
187 agent_output = self._rl_agent.step(time_step, is_evaluation)
188 if not is_evaluation and not time_step.last():
189 self._add_transition(time_step, agent_output)
190
191 elif self._mode == MODE.average_policy:
192 # Act step: don't act at terminal info states.
193 if not time_step.last():
194 info_state = time_step.observations["info_state"][self.player_id]
195 legal_actions = time_step.observations["legal_actions"][self.player_id]
196 action, probs = self._act(info_state, legal_actions)
197 agent_output = rl_agent.StepOutput(action=action, probs=probs)
198
199 if self._prev_timestep and not is_evaluation:
200 self._rl_agent.add_transition(self._prev_timestep, self._prev_action,
201 time_step)
202 else:
203 raise ValueError("Invalid mode ({})".format(self._mode))
204
205 if not is_evaluation:
206 self._step_counter += 1
207
208 if self._step_counter % self._learn_every == 0:
209 self._last_sl_loss_value = self._learn()
210 # If learn step not triggered by rl policy, learn.
211 if self._mode == MODE.average_policy:
212 self._rl_agent.learn()
213
214 # Prepare for the next episode.
215 if time_step.last():
216 self._sample_episode_policy()
217 self._prev_timestep = None
218 self._prev_action = None
219 return
220 else:
221 self._prev_timestep = time_step
222 self._prev_action = agent_output.action
223
224 return agent_output
225
226 def _add_transition(self, time_step, agent_output):
227 """Adds the new transition using `time_step` to the reservoir buffer.
228
229 Transitions are in the form (time_step, agent_output.probs, legal_mask).
230
231 Args:
232 time_step: an instance of rl_environment.TimeStep.
233 agent_output: an instance of rl_agent.StepOutput.
234 """
235 legal_actions = time_step.observations["legal_actions"][self.player_id]
236 legal_actions_mask = np.zeros(self._num_actions)
237 legal_actions_mask[legal_actions] = 1.0
238 transition = Transition(
239 info_state=(time_step.observations["info_state"][self.player_id][:]),
240 action_probs=agent_output.probs,
241 legal_actions_mask=legal_actions_mask)
242 self._reservoir_buffer.add(transition)
243
244 def _learn(self):
245 """Compute the loss on sampled transitions and perform a avg-network update.
246
247 If there are not enough elements in the buffer, no loss is computed and
248 `None` is returned instead.
249
250 Returns:
251 The average loss obtained on this batch of transitions or `None`.
252 """
253 if (len(self._reservoir_buffer) < self._batch_size or
254 len(self._reservoir_buffer) < self._min_buffer_size_to_learn):
255 return None
256
257 transitions = self._reservoir_buffer.sample(self._batch_size)
258 info_states = [t.info_state for t in transitions]
259 action_probs = [t.action_probs for t in transitions]
260 legal_actions_mask = [t.legal_actions_mask for t in transitions]
261
262 loss, _ = self._session.run(
263 [self._loss, self._learn_step],
264 feed_dict={
265 self._info_state_ph: info_states,
266 self._action_probs_ph: action_probs,
267 self._legal_actions_mask_ph: legal_actions_mask,
268 })
269 return loss

279 def save(self, checkpoint_dir):
280 """Saves the average policy network and the inner RL agent's q-network.
281
282 Note that this does not save the experience replay buffers and should
283 only be used to restore the agent's policy, not resume training.
284
285 Args:
286 checkpoint_dir: directory where checkpoints will be saved.
287 """
288 for name, saver in self._savers:
289 path = saver.save(
290 self._session,
291 self._full_checkpoint_name(checkpoint_dir, name),
292 latest_filename=self._latest_checkpoint_filename(name))
293 logging.info("Saved to path: %s", path)
294
295 def has_checkpoint(self, checkpoint_dir):
296 for name, _ in self._savers:
297 if tf.train.latest_checkpoint(
298 self._full_checkpoint_name(checkpoint_dir, name),
299 os.path.join(checkpoint_dir,
300 self._latest_checkpoint_filename(name))) is None:
301 return False
302 return True
303
304 def restore(self, checkpoint_dir):
305 """Restores the average policy network and the inner RL agent's q-network.
306
307 Note that this does not restore the experience replay buffers and should
308 only be used to restore the agent's policy, not resume training.
309
310 Args:
311 checkpoint_dir: directory from which checkpoints will be restored.
312 """
313 for name, saver in self._savers:
314 full_checkpoint_dir = self._full_checkpoint_name(checkpoint_dir, name)
315 logging.info("Restoring checkpoint: %s", full_checkpoint_dir)
316 saver.restore(self._session, full_checkpoint_dir)

319class ReservoirBuffer(object):
320 """Allows uniform sampling over a stream of data.
321
322 This class supports the storage of arbitrary elements, such as observation
323 tensors, integer actions, etc.
324
325 See https://en.wikipedia.org/wiki/Reservoir_sampling for more details.
326 """
327
328 def __init__(self, reservoir_buffer_capacity):
329 self._reservoir_buffer_capacity = reservoir_buffer_capacity
330 self._data = []
331 self._add_calls = 0
332
333 def add(self, element):
334 """Potentially adds `element` to the reservoir buffer.
335
336 Args:
337 element: data to be added to the reservoir buffer.
338 """
339 if len(self._data) < self._reservoir_buffer_capacity:
340 self._data.append(element)
341 else:
342 idx = np.random.randint(0, self._add_calls + 1)
343 if idx < self._reservoir_buffer_capacity:
344 self._data[idx] = element
345 self._add_calls += 1
346
347 def sample(self, num_samples):
348 """Returns `num_samples` uniformly sampled from the buffer.
349
350 Args:
351 num_samples: `int`, number of samples to draw.
352
353 Returns:
354 An iterable over `num_samples` random elements of the buffer.
355
356 Raises:
357 ValueError: If there are less than `num_samples` elements in the buffer
358 """
359 if len(self._data) < num_samples:
360 raise ValueError("{} elements could not be sampled from size {}".format(
361 num_samples, len(self._data)))
362 return random.sample(self._data, num_samples)
363
364 def clear(self):
365 self._data = []
366 self._add_calls = 0
367
368 def __len__(self):
369 return len(self._data)
370
371 def __iter__(self):
372 return iter(self._data)
