47 def setup_optimizer(self, num_actions, num_infostates):
48 if FLAGS.use_infostate_representation:
49 dummy_input = np.zeros(
50 shape=[FLAGS.batch_size, 1, num_actions + num_infostates])
51 else:
52 dummy_input = np.zeros(shape=[FLAGS.batch_size, 1, num_actions])
53
54 def mlp_forward(dummy_input):
55 mlp = hk.nets.MLP([10, num_actions])
56 return mlp(dummy_input)
57 forward = hk.transform(mlp_forward)
58
59 rng_seq = jax.random.PRNGKey(10)
60 params = forward.init(rng_seq, dummy_input)
61 lr_scheduler_fn = optax.polynomial_schedule(
62 init_value=0.2, end_value=0.0001, power=1., transition_steps=100)
63 opt_init, opt_update = optax.chain(
64 optax.scale_by_adam(), optax.scale_by_schedule(lr_scheduler_fn),
65 optax.scale(-0.2))
66 net_apply = forward.apply
67 opt_state = opt_init(params)
68 return params, net_apply, opt_state, opt_update
