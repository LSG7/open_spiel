28 VPNetEvaluator::VPNetEvaluator(DeviceManager* device_manager, int batch_size,
29  int threads, int cache_size, int cache_shards)
30  : device_manager_(*device_manager),
31  batch_size_(batch_size),
32  queue_(batch_size * threads * 4),
33  batch_size_hist_(batch_size + 1) {
34  cache_shards = std::max(1, cache_shards);
35  cache_.reserve(cache_shards);
36  for (int i = 0; i < cache_shards; ++i) {
37  cache_.push_back(
38  std::make_unique<LRUCache<uint64_t, VPNetModel::InferenceOutputs>>(
39  cache_size / cache_shards));
40  }
41  if (batch_size_ <= 1) {
42  threads = 0;
43  }
44  inference_threads_.reserve(threads);
45  for (int i = 0; i < threads; ++i) {
46  inference_threads_.emplace_back([this]() { this->Runner(); });
47  }
48 }

87 VPNetModel::InferenceOutputs VPNetEvaluator::Inference(const State& state) {
88  VPNetModel::InferenceInputs inputs = {state.LegalActions(),
89  state.ObservationTensor()};
90 
91  uint64_t key;
92  int cache_shard;
93  if (!cache_.empty()) {
94  key = absl::Hash<VPNetModel::InferenceInputs>{}(inputs);
95  cache_shard = key % cache_.size();
96  absl::optional<const VPNetModel::InferenceOutputs> opt_outputs =
97  cache_[cache_shard]->Get(key);
98  if (opt_outputs) {
99  return *opt_outputs;
100  }
101  }
102  VPNetModel::InferenceOutputs outputs;
103  if (batch_size_ <= 1) {
104  outputs = device_manager_.Get(1)->Inference(std::vector{inputs})[0];
105  } else {
106  std::promise<VPNetModel::InferenceOutputs> prom;
107  std::future<VPNetModel::InferenceOutputs> fut = prom.get_future();
108  queue_.Push(QueueItem{inputs, &prom});
109  outputs = fut.get();
110  }
111  if (!cache_.empty()) {
112  cache_[cache_shard]->Set(key, outputs);
113  }
114  return outputs;
115 }

117 void VPNetEvaluator::Runner() {
118  std::vector<VPNetModel::InferenceInputs> inputs;
119  std::vector<std::promise<VPNetModel::InferenceOutputs>*> promises;
120  inputs.reserve(batch_size_);
121  promises.reserve(batch_size_);
122  while (!stop_.StopRequested()) {
123  {
124  // Only one thread at a time should be listening to the queue to maximize
125  // batch size and minimize latency.
126  absl::MutexLock lock(&inference_queue_m_);
127  absl::Time deadline = absl::InfiniteFuture();
128  for (int i = 0; i < batch_size_; ++i) {
129  absl::optional<QueueItem> item = queue_.Pop(deadline);
130  if (!item) { // Hit the deadline.
131  break;
132  }
133  if (inputs.empty()) {
134  deadline = absl::Now() + absl::Milliseconds(1);
135  }
136  inputs.push_back(item->inputs);
137  promises.push_back(item->prom);
138  }
139  }
140 
141  if (inputs.empty()) { // Almost certainly StopRequested.
142  continue;
143  }
144 
145  {
146  absl::MutexLock lock(&stats_m_);
147  batch_size_stats_.Add(inputs.size());
148  batch_size_hist_.Add(inputs.size());
149  }
150 
151  std::vector<VPNetModel::InferenceOutputs> outputs =
152  device_manager_.Get(inputs.size())->Inference(inputs);
153  for (int i = 0; i < promises.size(); ++i) {
154  promises[i]->set_value(outputs[i]);
155  }
156  inputs.clear();
157  promises.clear();
158  }
159 }
