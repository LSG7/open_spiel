35 // Start all players and wait for ready messages from all them simultaneously.
36 std::vector<bool> Referee::StartPlayers() {
37  SPIEL_CHECK_EQ(game_->NumPlayers(), num_bots());
38 
39  // Launch players and create communication channels.
40  log_ << "Starting players." << std::endl;
41  for (int pl = 0; pl < num_bots(); ++pl) {
42  const std::string& bot_command = bot_commands_[pl];
43  log_ << "Bot#" << pl << ": " << bot_command << std::endl;
44  errors_.push_back(BotErrors());
45  channels_.push_back(MakeBotChannel(pl, bot_command));
46  // Read from bot's stdout/stderr in separate threads.
47  threads_stdout_.push_back(std::make_unique<std::thread>(
48  ReadLineFromChannelStdout, channels_.back().get()));
49  threads_stderr_.push_back(std::make_unique<std::thread>(
50  ReadLineFromChannelStderr, channels_.back().get()));
51  }
52 
53  // Send setup information.
54  for (int pl = 0; pl < num_bots(); ++pl) {
55  BotChannel* chn = channels_[pl].get();
56  chn->Write(game_name_ + "\n");
57  chn->Write(std::to_string(pl) + "\n");
58  chn->StartRead(settings_.timeout_ready);
59  }
60 
61  sleep_ms(
62  settings_.timeout_ready); // Blocking sleep to give time to the bots.
63  return CheckResponses(kReadyMessage);
64 }

111 std::unique_ptr<State> Referee::PlayMatch() {
112  SPIEL_CHECK_EQ(num_bots(), game_->NumPlayers());
113  std::unique_ptr<State> state = game_->NewInitialState();
114 
115  std::vector<int> player_order(num_bots());
116  std::vector<bool> is_acting(num_bots(), false);
117  bool only_ponder = false; // Whether all bots only ponder (i.e chance node)
118  for (int i = 0; i < num_bots(); ++i) player_order[i] = i;
119 
120  // Check start of match message.
121  for (int pl = 0; pl < num_bots(); ++pl) {
122  BotChannel* chn = channels_[pl].get();
123  chn->StartRead(settings_.timeout_start);
124  }
125  sleep_ms(settings_.timeout_start);
126  CheckResponses(kStartMessage);
127 
128  while (!state->IsTerminal()) {
129  log_ << "\nHistory: " << absl::StrJoin(state->History(), " ") << std::endl;
130 
131  only_ponder = state->IsChanceNode();
132  // Cache whether player is acting.
133  for (int pl = 0; pl < num_bots(); ++pl) {
134  is_acting[pl] = state->IsPlayerActing(pl);
135  }
136  // Make sure no player is preferred when we communicate with it.
137  std::shuffle(player_order.begin(), player_order.end(), rng_);
138 
139  // Send players' observation and possibly a set of legal actions
140  // available to the players.
141  for (int pl : player_order) {
142  BotChannel* chn = channels_[pl].get();
143  public_observation_->SetFrom(*state, pl);
144  private_observation_->SetFrom(*state, pl);
145  std::string public_tensor = public_observation_->Compress();
146  std::string private_tensor = private_observation_->Compress();
147 
148  // Send observations.
149  absl::string_view public_string(
150  reinterpret_cast<char* const>(public_tensor.data()),
151  public_tensor.size());
152  chn->Write(absl::Base64Escape(public_string));
153  chn->Write(" ");
154  absl::string_view private_string(
155  reinterpret_cast<char* const>(private_tensor.data()),
156  private_tensor.size());
157  chn->Write(absl::Base64Escape(private_string));
158  // Send actions.
159  if (is_acting[pl]) {
160  std::vector<Action> legal_actions = state->LegalActions(pl);
161  for (Action a : legal_actions) {
162  chn->Write(" ");
163  chn->Write(std::to_string(a));
164  }
165  }
166  chn->Write("\n");
167  }
168 
169  // Start waiting for response within the time limits.
170  for (int pl : player_order) {
171  BotChannel* chn = channels_[pl].get();
172  chn->StartRead(is_acting[pl] ? settings_.timeout_act
173  : settings_.timeout_ponder);
174  }
175 
176  // Wait for ponder messages.
177  WaitForPonderingBots(is_acting);
178  for (int pl = 0; pl < num_bots(); ++pl) {
179  if (is_acting[pl]) continue;
180  BotChannel* chn = channels_[pl].get();
181  std::string response = chn->response();
182  if (response != kPonderMessage) {
183  log_ << "Bot#" << pl << " ponder bad response: '" << response << "'"
184  << std::endl;
185  errors_[pl].ponder_error++;
186  if (chn->is_time_out()) {
187  log_ << "Bot#" << pl << " ponder also timed out." << std::endl;
188  errors_[pl].time_over++;
189  }
190  } else {
191  log_ << "Bot#" << pl << " ponder ok." << std::endl;
192  }
193  }
194 
195  // Wait for response(s) from acting player(s).
196  // If (all) response(s) arrive before the time limit,
197  // we don't have to wait to apply the action(s).
198  WaitForActingBots(is_acting);
199 
200  // Parse submitted actions based on the bot responses.
201  std::vector<Action> bot_actions(num_bots(), kInvalidAction);
202  for (int pl = 0; pl < num_bots(); ++pl) {
203  if (!is_acting[pl]) continue; // Ponders have been already processed.
204 
205  BotChannel* chn = channels_[pl].get();
206  std::vector<Action> legal_actions = state->LegalActions(pl);
207 
208  if (chn->comm_error() != 0) {
209  log_ << "Bot#" << pl
210  << " act communication error: " << chn->comm_error() << std::endl;
211  errors_[pl].protocol_error++;
212  } else if (chn->is_time_out()) {
213  log_ << "Bot#" << pl << " act timed out. " << std::endl;
214  errors_[pl].time_over++;
215  } else if (!chn->has_read()) {
216  log_ << "Bot#" << pl << " act no response. " << std::endl;
217  errors_[pl].protocol_error++;
218  } else {
219  std::string response = chn->response();
220  log_ << "Bot#" << pl << " act response: '" << response << "'"
221  << std::endl;
222 
223  int action = -1;
224  bool success = absl::SimpleAtoi(response, &action);
225  if (!success) {
226  log_ << "Bot#" << pl << " act invalid action. " << std::endl;
227  errors_[pl].protocol_error++;
228  } else if (std::find(legal_actions.begin(), legal_actions.end(),
229  action) == legal_actions.end()) {
230  log_ << "Bot#" << pl << " act illegal action. " << std::endl;
231  errors_[pl].illegal_actions++;
232  } else {
233  log_ << "Bot#" << pl << " act ok. " << std::endl;
234  if (errors_[pl].total_errors() > settings_.max_invalid_behaviors) {
235  log_ << "Bot#" << pl << " act randomly (exceeded illegal behaviors)"
236  << std::endl;
237  } else {
238  bot_actions[pl] = action;
239  }
240  }
241  }
242 
243  if (bot_actions[pl] == kInvalidAction) { // Pick a random action.
244  log_ << "Picking random action for Bot#" << pl << std::endl;
245  std::uniform_int_distribution<int> dist(0, legal_actions.size() - 1);
246  int random_idx = dist(rng_);
247  bot_actions[pl] = legal_actions[random_idx];
248  }
249  }
250  log_ << "Submitting actions:";
251  for (Action a : bot_actions) log_ << ' ' << a;
252  log_ << std::endl;
253 
254  // Apply actions.
255  if (state->IsChanceNode()) {
256  ActionsAndProbs actions_and_probs = state->ChanceOutcomes();
257  std::uniform_real_distribution<double> dist;
258  const auto& [chance_action, prob] =
259  SampleAction(actions_and_probs, dist(rng_));
260  log_ << "Chance action: " << chance_action << " with prob " << prob
261  << std::endl;
262  state->ApplyAction(chance_action);
263  } else if (state->IsSimultaneousNode()) {
264  state->ApplyActions(bot_actions);
265  } else {
266  state->ApplyAction(bot_actions[state->CurrentPlayer()]);
267  }
268  }
269 
270  std::vector<double> returns = state->Returns();
271 
272  log_ << "\nMatch over!" << std::endl;
273  log_ << "History: " << absl::StrJoin(state->History(), " ") << std::endl;
274 
275  for (int pl = 0; pl < num_bots(); ++pl) {
276  int score = returns[pl];
277  channels_[pl]->Write(absl::StrCat(kMatchOverMessage, " ",
278  score, "\n"));
279  channels_[pl]->StartRead(settings_.timeout_match_over);
280  }
281 
282  for (int pl = 0; pl < num_bots(); ++pl) {
283  log_ << "Bot#" << pl << " returns " << returns[pl] << std::endl;
284  log_ << "Bot#" << pl << " protocol errors " << errors_[pl].protocol_error
285  << std::endl;
286  log_ << "Bot#" << pl << " illegal actions " << errors_[pl].illegal_actions
287  << std::endl;
288  log_ << "Bot#" << pl << " ponder errors " << errors_[pl].ponder_error
289  << std::endl;
290  log_ << "Bot#" << pl << " time overs " << errors_[pl].time_over
291  << std::endl;
292  }
293 
294  sleep_ms(settings_.timeout_match_over);
295  CheckResponses(kMatchOverMessage);
296 
297  return state;
298 }

313 bool Referee::CheckResponse(const std::string& expected_response, int pl) {
314  BotChannel* chn = channels_[pl].get();
315  chn->CancelReadBlocking();
316  std::string response = chn->response();
317  if (response != expected_response) {
318  log_ << "Bot#" << pl << " did not respond '" << expected_response << "'"
319  << std::endl;
320  log_ << "Bot#" << pl << " response was: '" << response << "'" << std::endl;
321  if (chn->comm_error() < 0) {
322  log_ << "Bot#" << pl
323  << " also had a communication error: " << chn->comm_error()
324  << std::endl;
325  }
326  errors_[pl].protocol_error++;
327  if (chn->is_time_out()) {
328  errors_[pl].time_over++;
329  log_ << "Bot#" << pl << " also timed out." << std::endl;
330  }
331  return false;
332  } else {
333  log_ << "Bot#" << pl << " " << expected_response << " ok." << std::endl;
334  return true;
335  }

382 std::unique_ptr<TournamentResults> Referee::PlayTournament(int num_matches) {
383  auto results = std::make_unique<TournamentResults>(num_bots());
384  std::vector<bool> start_ok = StartPlayers();
385  bool all_ok = true;
386  for (int pl = 0; pl < num_bots(); ++pl) {
387  all_ok = all_ok && start_ok[pl];
388  if (!start_ok[pl]) results->disqualified[pl] = true;
389  }
390  if (!all_ok) {
391  log_ << "Could not start all players correctly, "
392  "cannot play the tournament."
393  << std::endl;
394  return results;
395  }
396 
397  const int corruption_threshold =
398  num_matches * settings().disqualification_rate;
399  int match;
400  for (match = 0; match < num_matches; ++match) {
401  log_ << "\n";
402  for (int j = 0; j < 80; ++j) log_ << '-';
403  log_ << "\nPlaying match " << match + 1 << " / " << num_matches
404  << std::endl;
405  for (int j = 0; j < 80; ++j) log_ << '-';
406  log_ << std::endl;
407 
408  ResetErrorTracking();
409  std::unique_ptr<State> state = PlayMatch();
410  std::vector<double> returns = state->Returns();
411 
412  // Update mean,var statistics.
413  results->history_len_mean +=
414  (state->FullHistory().size() - results->history_len_mean) /
415  (match + 1.);
416  for (int pl = 0; pl < num_bots(); ++pl) {
417  double delta = returns[pl] - results->returns_mean[pl];
418  results->returns_mean[pl] += delta / (match + 1.);
419  double delta2 = returns[pl] - results->returns_mean[pl];
420  results->returns_agg[pl] += delta * delta2;
421  }
422  // Disqualifications update.
423  bool tournament_over = false;
424  for (int pl = 0; pl < num_bots(); ++pl) {
425  if (!corrupted_match_due(pl)) continue;
426  log_ << "Bot#" << pl << " exceeded illegal behaviors in match " << match
427  << std::endl;
428  results->corrupted_matches[pl]++;
429 
430  if (results->corrupted_matches[pl] > corruption_threshold) {
431  log_ << "Bot#" << pl << " is disqualified!" << std::endl;
432  results->disqualified[pl] = true;
433  tournament_over = true;
434  } else {
435  log_ << "Bot#" << pl << " is going to restart!" << std::endl;
436  ++results->restarts[pl];
437  RestartPlayer(pl);
438  }
439  }
440 
441  results->matches.push_back(
442  MatchResult{.terminal = std::move(state), .errors = errors_});
443 
444  if (tournament_over) {
445  break;
446  }
447  }
448 
449  log_ << "\n";
450  for (int j = 0; j < 80; ++j) log_ << '-';
451  log_ << "\nTournament is over!" << std::endl;
452  for (int j = 0; j < 80; ++j) log_ << '-';
453  log_ << std::endl;
454 
455  results->PrintVerbose(log_);
456  TournamentOver();
457  log_ << "Shutting down players." << std::endl;
458  ShutDownPlayers();
459 
460  return results;
461 }

463 void Referee::WaitForBots(const std::vector<bool>& is_acting, bool mask) {
464  int num_bots_to_wait_for = 0;
465  for (int pl = 0; pl < is_acting.size(); ++pl) {
466  if (is_acting[pl] == mask) num_bots_to_wait_for++;
467  }
468  if (num_bots_to_wait_for == 0) return;
469 
470  while (true) {
471  sleep_ms(1);
472 
473  int arrived_bots = 0;
474  for (int pl = 0; pl < is_acting.size(); ++pl) {
475  if (is_acting[pl] == mask && channels_[pl]->is_waiting_for_referee()) {
476  arrived_bots++;
477  }
478  }
479  if (arrived_bots == num_bots_to_wait_for) return;
480  }
481 }
