83void signal_handler(int s) {
84 if (stop_token.StopRequested()) {
85 exit(1);
86 } else {
87 stop_token.Stop();
88 }

91void signal_installer() {
92 struct sigaction sigIntHandler;
93 sigIntHandler.sa_handler = signal_handler;
94 sigemptyset(&sigIntHandler.sa_mask);
95 sigIntHandler.sa_flags = 0;
96 sigaction(SIGINT, &sigIntHandler, nullptr);
97}
