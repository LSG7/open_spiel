143// An indexed sequence of distinct structures holding a single
144// value of type T
145template<typename T, size_t N>
146struct ValueHolder {
147 explicit ValueHolder(const T& x) : value(x) {}
148 typedef T value_type;
149 value_type value;
150 static const size_t kIndex = N;
151};
152template<typename T, size_t N>
153const size_t ValueHolder<T, N>::kIndex;
154
155// The following three functions make ValueHolder compatible with
156// EXPECT_EQ and EXPECT_NE
157template<typename T, size_t N>
158inline bool operator==(const ValueHolder<T, N>& left,
159 const ValueHolder<T, N>& right) {
160 return left.value == right.value;
161}
162
163template<typename T, size_t N>
164inline bool operator!=(const ValueHolder<T, N>& left,
165 const ValueHolder<T, N>& right) {
166 return left.value != right.value;
167}
168
169template<typename T, size_t N>
170inline std::ostream& operator<<(
171 std::ostream& stream, const ValueHolder<T, N>& object) {
172 return stream << object.value;
173}

188// Increments the provided counter pointer in the destructor
189struct IncrementInDtor {
190 explicit IncrementInDtor(int* counter) : counter(counter) {}
191 ~IncrementInDtor() { *counter += 1; }
192 int* counter;
193};
194
195struct IncrementInDtorCopyCanThrow {
196 explicit IncrementInDtorCopyCanThrow(int* counter) : counter(counter) {}
197 IncrementInDtorCopyCanThrow(IncrementInDtorCopyCanThrow&& other) noexcept =
198 default;
199 IncrementInDtorCopyCanThrow(const IncrementInDtorCopyCanThrow& other)
200 : counter(other.counter) {}
201 IncrementInDtorCopyCanThrow& operator=(
202 IncrementInDtorCopyCanThrow&&) noexcept = default;
203 IncrementInDtorCopyCanThrow& operator=(
204 IncrementInDtorCopyCanThrow const& other) {
205 counter = other.counter;
206 return *this;
207 }
208 ~IncrementInDtorCopyCanThrow() { *counter += 1; }
209 int* counter;
210};
