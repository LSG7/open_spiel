30 Action GetOptimalAction(
31  absl::flat_hash_map<std::pair<std::string, Action>, double> q_values,
32  const std::unique_ptr<State> &state) {
33  std::vector<Action> legal_actions = state->LegalActions();
34  const auto state_str = state->ToString();
35 
36  Action optimal_action = open_spiel::kInvalidAction;
37  double value = -1;
38  for (const Action &action : legal_actions) {
39  double q_val = q_values[{state_str, action}];
40  if (q_val >= value) {
41  value = q_val;
42  optimal_action = action;
43  }
44  }
45  return optimal_action;
46 }

48 Action GetRandomAction(const std::unique_ptr<State> &state, int seed) {
49  std::vector<Action> legal_actions = state->LegalActions();
50  if (legal_actions.empty()) {
51  return kInvalidAction;
52  }
53  std::mt19937 rng(seed);
54  return legal_actions[absl::Uniform<int>(rng, 0, legal_actions.size())];
55 }

57 double PlayCatch(
58  absl::flat_hash_map<std::pair<std::string, Action>, double> q_values,
59  const std::unique_ptr<State> &state, double seed) {
60  // First action determines the starting column. Do the first action before the
61  // main loop, where the optimal action is chosen.
62  // Example: Initial state with random seed 42
63  // ...o.
64  // .....
65  // .....
66  // .....
67  // .....
68  // .....
69  // .....
70  // .....
71  // .....
72  // ..x..
73  std::mt19937 gen(seed);
74  std::uniform_int_distribution<int> distribution(0,
75  catch_::kDefaultColumns - 1);
76  int ball_starting_column = distribution(gen);
77  state->ApplyAction(ball_starting_column);
78 
79  while (!state->IsTerminal()) {
80  Action optimal_action = GetOptimalAction(q_values, state);
81  state->ApplyAction(optimal_action);
82  }
83 
84  return state->Rewards()[0];
85 }

98 void TabularQLearningTest_Catch_Lambda00_Loss() {
99  // Classic Q-learning. No bootstraping (lambda=0.0)
100  // Player loses after only 1 train iteration.
101  std::shared_ptr<const Game> game = LoadGame("catch");
102  auto tabular_q_learning_solver = QLearningSolver(game, 0);
103 
104  tabular_q_learning_solver->RunIteration();
105  const absl::flat_hash_map<std::pair<std::string, Action>, double>& q_values =
106  tabular_q_learning_solver->GetQValueTable();
107  std::unique_ptr<State> state = game->NewInitialState();
108 
109  double reward = PlayCatch(q_values, state, 42);
110  SPIEL_CHECK_EQ(reward, -1);

146 void TabularQLearningTest_Catch_Lambda01FasterThanLambda00() {
147  // Eligibility traces (lambda > 0.0) always achieves victory with less
148  // training steps w.r.t. Q-learning(lambda=0.0)
149  std::shared_ptr<const Game> game = LoadGame("catch");
150  auto tabular_q_learning_solver_lambda00 = QLearningSolver(game, 0);
151  auto tabular_q_learning_solver_lambda01 = QLearningSolver(game, 0.1);
152 
153  for (int seed = 0; seed < 100; seed++) {
154  int lambda_00_train_iter = 0;
155  int lambda_01_train_iter = 0;
156  double lambda_00_reward = -1.0;
157  double lambda_01_reward = -1.0;
158 
159  while (lambda_00_reward == -1.0) {
160  tabular_q_learning_solver_lambda00->RunIteration();
161  std::unique_ptr<State> state = game->NewInitialState();
162  lambda_00_reward = PlayCatch(
163  tabular_q_learning_solver_lambda00->GetQValueTable(), state, seed);
164  lambda_00_train_iter++;
165  }
166  while (lambda_01_reward == -1.0) {
167  tabular_q_learning_solver_lambda01->RunIteration();
168  std::unique_ptr<State> state = game->NewInitialState();
169  lambda_01_reward = PlayCatch(
170  tabular_q_learning_solver_lambda01->GetQValueTable(), state, seed);
171  lambda_01_train_iter++;
172  }
173  SPIEL_CHECK_GE(lambda_00_train_iter, lambda_01_train_iter);
174  }
175 }
