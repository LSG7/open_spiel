204 def test_play_kuhn_poker(self):
205 game = pyspiel.load_game("kuhn_poker")
206 state = game.new_initial_state()
207 self.assertEqual(state.is_chance_node(), True)
208 self.assertEqual(state.chance_outcomes(), [(0, 1 / 3), (1, 1 / 3),
209 (2, 1 / 3)])
210 state.apply_action(1)
211 self.assertEqual(state.is_chance_node(), True)
212 self.assertEqual(state.chance_outcomes(), [(0, 0.5), (2, 0.5)])
213 state.apply_action(2)
214 self.assertEqual(state.is_chance_node(), False)
215 self.assertEqual(state.legal_actions(), [0, 1])
216 sampler = pyspiel.UniformProbabilitySampler(0., 1.)
217 clone = state.resample_from_infostate(1, sampler)
218 self.assertEqual(
219 clone.information_state_string(1), state.information_state_string(1))
