30class AverageNetworkFictitiousPlayTest(parameterized.TestCase):
31
32 @parameterized.named_parameters(('cpp', 'mfg_crowd_modelling'),
33 ('python', 'python_mfg_crowd_modelling'))
34 def test_train(self, name):
35 """Checks that the training works."""
36 game = pyspiel.load_game(name)
37 assert game.num_players() == 1
38 uniform_policy = policy.UniformRandomPolicy(game)
39 uniform_dist = distribution.DistributionPolicy(game, uniform_policy)
40 env = rl_environment.Environment(
41 game, mfg_distribution=uniform_dist, mfg_population=0)
42 info_state_size = env.observation_spec()['info_state'][0]
43 num_actions = env.action_spec()['num_actions']
44 np.random.seed(0)
45
46 dqn_args = {
47 'batch_size': 32,
48 'epsilon_end': 0.1,
49 'epsilon_start': 0.1,
50 'hidden_layers_sizes': [128],
51 'learn_every': 32,
52 'learning_rate': 0.01,
53 'min_buffer_size_to_learn': 32,
54 'optimizer_str': 'adam',
55 'replay_buffer_capacity': 2000,
56 'update_target_network_every': 32,
57 }
58 br_agent = dqn.DQN(0, info_state_size, num_actions, **dqn_args)
59
60 args = {
61 'batch_size': 32,
62 'hidden_layers_sizes': [128],
63 'reservoir_buffer_capacity': 100000,
64 'learning_rate': 0.01,
65 'min_buffer_size_to_learn': 32,
66 'optimizer_str': 'adam',
67 'seed': 0,
68 'tau': 1.0,
69 }
70 fp = average_network_fictitious_play.AverageNetworkFictitiousPlay(
71 game, [env], [br_agent],
72 num_episodes_per_iteration=50,
73 num_training_steps_per_iteration=10,
74 **args)
75
76 # Run several iterations.
77 for _ in range(5):
78 training.run_episodes([env], [br_agent],
79 num_episodes=50,
80 is_evaluation=False)
81 fp.iteration()
82
83 # Just sanity check.
84 nash_conv_fp = nash_conv.NashConv(game, fp.policy)
85 self.assertLessEqual(nash_conv_fp.nash_conv(), 15)
86
