56int CalcXY(int x, int y, int board_size) {
57 if (x >= 0 && y >= 0 && x < board_size && y < board_size &&
58 (x + y < board_size)) {
59 return x + y * board_size;
60 } else {
61 return kMoveUnknown;
62 }
63}

65struct Move {
66 int8_t x, y; // The x,y coordinates
67 int16_t xy; // precomputed x + y * board_size as an index into the array.
68
69 inline constexpr Move(MoveSpecial m = kMoveUnknown) : x(-1), y(-1), xy(m) {}
70 inline constexpr Move(int x_, int y_, MoveSpecial m) : x(x_), y(y_), xy(m) {}
71 Move(int x_, int y_, int board_size)
72 : x(x_), y(y_), xy(CalcXY(x_, y_, board_size)) {}
73
74 std::string ToString() const;
75
76 bool operator==(const Move& b) const { return xy == b.xy; }
77 bool operator!=(const Move& b) const { return xy != b.xy; }
78 bool operator==(const MoveSpecial& b) const { return xy == b; }
79 bool operator!=(const MoveSpecial& b) const { return xy != b; }
80
81 // Whether the move is valid and on the board. May be invalid because it is
82 // a MoveSpecial, in the cut-off corner, or otherwise off the board.
83 bool OnBoard() const { return xy >= 0; }
84
85 // Flags for which edge this move is part of.
86 int Edge(int board_size) const;
87};

92// State of an in-play game.
93class YState : public State {
94 // Represents a single cell on the board, as well as the structures needed for
95 // groups of cells. Groups of cells are defined by a union-find structure
96 // embedded in the array of cells. Following the `parent` indices will lead to
97 // the group leader which has the up to date size and edge
98 // connectivity of that group. Size and edge are not valid for any
99 // cell that is not a group leader.
100 struct Cell {
101 // Who controls this cell.
102 YPlayer player;
103
104 // A parent index to allow finding the group leader. It is the leader of the
105 // group if it points to itself. Allows path compression to shorten the path
106 // from a direct parent to the leader.
107 uint16_t parent;
108
109 // These three are only defined for the group leader's cell.
110 uint16_t size; // Size of this group of cells.
111 uint8_t edge; // A bitset of which edges this group is connected to.
112
113 Cell() {}
114 Cell(YPlayer player_, int parent_, int edge_)
115 : player(player_), parent(parent_), size(1), edge(edge_) {}
116 };
