15"""Outputs value of best response policy against set of distributions."""
16import collections
17from typing import List
18from open_spiel.python.mfg import distribution as distribution_std
19from open_spiel.python.mfg import value
20import pyspiel
21
22
23class JointBestResponse(value.ValueFunction):
24 """Computes a best response value."""
25
26 def __init__(
27 self,
28 game,
29 distributions: List[distribution_std.Distribution],
30 weights,
31 root_state=None,
32 ):
33 """Initializes the joint best response computation.
34
35 The joint best response is computed under the following premisse : the
36 player does not know which distribution it is playing against. It only knows
37 their probabilities, and thus tries to find a best response against their
38 mixture.
39
40 This is accomplished by recursively computing the action that maximizes the
41 marginalized value of each node over each distribution.
42
43 Warning : This version only works on games whose observation space &
44 dynamics do NOT depend on state distribution.
45
46 Args:
47 game: The game to analyze.
48 distributions: A list of `distribution_std.Distribution`.
49 weights: A list of floats the same length as `distributions`. Represents
50 the mixture weight of each member of `distributions`.
51 root_state: The state of the game at which to start. If `None`, the game
52 root state is used.
53 """
54 super().__init__(game)
55 if root_state is None:
56 self._root_states = game.new_initial_states()
57 else:
58 self._root_states = [root_state]
59 self._distributions = distributions
60 self._weights = weights
61 # Maps states (in string format) to the value of the optimal policy given
62 # 'self._distribution'.
63 self._state_value = collections.defaultdict(float)
64 self.evaluate()
65
66 def get_state_rewards(self, mu_states):
67 return sum([
68 weight * mu_state.rewards()[mu_state.mean_field_population()]
69 for weight, mu_state in zip(self._weights, mu_states)
70 ])
71
72 def get_new_mu_states(self, mu_states):
73 new_mu_states = []
74 for mu_ind, mu_state in enumerate(mu_states):
75 dist = [
76 self._distributions[mu_ind].value_str(str_state, 0.0)
77 for str_state in mu_state.distribution_support()
78 ]
79 new_mu_state = mu_state.clone()
80 new_mu_state.update_distribution(dist)
81 new_mu_states.append(new_mu_state)
82 return new_mu_states
83
84 def eval_state(self, mu_states):
85 """Evaluate the value of a state.
86
87 Args:
88 mu_states: A list of game states, one for each `distributions` member.
89
90 Returns:
91 The optimal value of the state.
92
93 Recursively computes the value of the optimal policy given the fixed state
94 distributions. `self._state_value` is used as a cache for pre-computed
95 values.
96 """
97 state = mu_states[0]
98 state_str = state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)
99 if state_str in self._state_value:
100 return self._state_value[state_str]
101 if state.is_terminal():
102 self._state_value[state_str] = self.get_state_rewards(mu_states)
103 return self._state_value[state_str]
104 if state.current_player() == pyspiel.PlayerId.CHANCE:
105 self._state_value[state_str] = 0.0
106 for action, prob in state.chance_outcomes():
107 new_mu_states = [mu_state.child(action) for mu_state in mu_states]
108 self._state_value[state_str] += prob * self.eval_state(new_mu_states)
109 return self._state_value[state_str]
110 if state.current_player() == pyspiel.PlayerId.MEAN_FIELD:
111 new_mu_states = self.get_new_mu_states(mu_states)
112 self._state_value[state_str] = self.get_state_rewards(
113 mu_states
114 ) + self.eval_state(new_mu_states)
115 return self._state_value[state_str]
116 else:
117 assert int(state.current_player()) >= 0, "The player id should be >= 0"
118 max_q = max(
119 self.eval_state([mu_state.child(action) for mu_state in mu_states])
120 for action in state.legal_actions()
121 )
122 self._state_value[state_str] = self.get_state_rewards(mu_states) + max_q
123 return self._state_value[state_str]
124
125 def evaluate(self):
126 """Evaluate the best response value on all states."""
127 for state in self._root_states:
128 self.eval_state([state.clone() for _ in self._distributions])
129
130 def value(self, state, action=None):
131 if action is None:
132 return self._state_value[state.observation_string(
133 pyspiel.PlayerId.DEFAULT_PLAYER_ID)]
134 new_state = state.child(action)
135 return state.rewards()[state.mean_field_population()] + self._state_value[
136 new_state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)]
