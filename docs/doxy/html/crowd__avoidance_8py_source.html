75def grid_to_forbidden_states(grid):
76 """Converts a grid into string representation of forbidden states.
77
78 Args:
79 grid: Rows of the grid. '#' character denotes a forbidden state. All rows
80 should have the same number of columns, i.e. cells.
81
82 Returns:
83 String representation of forbidden states in the form of x (column) and y
84 (row) pairs, e.g. [1|1;0|2].
85 """
86 forbidden_states = []
87 num_cols = len(grid[0])
88 for y, row in enumerate(grid):
89 assert len(row) == num_cols, f"Number of columns should be {num_cols}."
90 for x, cell in enumerate(row):
91 if cell == "#":
92 forbidden_states.append(f"{x}|{y}")
93 return "[" + ";".join(forbidden_states) + "]"

183class MFGCrowdAvoidanceGame(pyspiel.Game):
184 """Multi-population MFG."""
185
186 # pylint:disable=dangerous-default-value
187 def __init__(self, params: Mapping[str, Any] = _DEFAULT_PARAMS):
188 self.size = get_param("size", params)
189 self.horizon = get_param("horizon", params)
190 flat_congestion_matrix = np.fromstring(
191 get_param("congestion_matrix", params), dtype=np.float64, sep=" "
192 )
193 num_players = get_param("players", params)
194 if len(flat_congestion_matrix) != num_players**2:
195 raise ValueError(
196 "Congestion matrix passed in flat representation does not represent "
197 f"a square matrix: {flat_congestion_matrix}"
198 )
199 self.congestion_matrix = flat_congestion_matrix.reshape(
200 [num_players, num_players]
201 )
202 self.geometry = get_param("geometry", params)
203 num_states = self.size**2
204 game_info = pyspiel.GameInfo(
205 num_distinct_actions=_NUM_ACTIONS,
206 max_chance_outcomes=max(num_states, _NUM_CHANCE),
207 num_players=num_players,
208 min_utility=-np.inf,
209 max_utility=+np.inf,
210 utility_sum=None,
211 max_game_length=self.horizon,
212 )
213 self.proba_noise = get_param("proba_noise", params)
214 self.coef_congestion = get_param("coef_congestion", params)
215 self.forbidden_states = pairs_string_to_list(
216 get_param("forbidden_states", params)
217 )
218 self.coef_target = get_param("coef_target", params)
219 # TODO(lauriere): should be given as a parameter of the model.
220 self.target_positions = np.array([[5, 3], [1, 3]])
221
222 # Represents the current probability distribution over game states
223 # (when grouped for each population).
224 str_init_distrib = get_param("init_distrib", params)
225 if str_init_distrib:
226 flat_init_distrib = np.fromstring(
227 str_init_distrib, dtype=np.float64, sep=" "
228 )
229 if len(flat_init_distrib) != num_players * self.size**2:
230 raise ValueError(
231 "Initial distribution matrix passed in flat representation does"
232 f" not represent a sequence of square matrices: {flat_init_distrib}"
233 )
234 self.initial_distribution = flat_init_distrib
235 else:
236 # Initialized with a uniform distribution.
237 self.initial_distribution = [1.0 / num_states] * (
238 num_states * num_players
239 )
240 super().__init__(_GAME_TYPE, game_info, params)

285 # Maps legal actions to the corresponding move on the grid of the game.
286 _ACTION_TO_MOVE = {
287 0: np.array([0, 0]),
288 1: np.array([1, 0]),
289 2: np.array([0, 1]),
290 3: np.array([0, -1]),
291 4: np.array([-1, 0]),

294 _NEUTRAL_ACTION = 0

296 def __init__(self, game, population=None):
297 """Constructor; should only be called by Game.new_initial_state.*.
298
299 Args:
300 game: MFGCrowdAvoidanceGame for which a state should be created.
301 population: ID of the population to create this state for. Must be in [0,
302 num_players()) or None. States with population=None cannot be used to
303 perform game actions.
304 """
305 super().__init__(game)
306 # Initial state where the initial position is chosen according to
307 # an initial distribution.
308 self._is_position_init = True
309 self._player_id = pyspiel.PlayerId.CHANCE
310 # Population this state corresponds to. Can be None, in which
311 # case, ApplyAction() is forbidden.
312 self._population = population
313 if self._population is not None:
314 assert 0 <= self._population < self.num_players()
315 # When set, <int>[2] numpy array representing the x, y position on the grid.
316 self._pos = None # type: Optional[np.ndarray]
317 self._t = 0
318 self.size = game.size
319 # Number of states in the grid.
320 self.num_states = self.size**2
321 self.horizon = game.horizon
322 self.congestion_matrix = game.congestion_matrix
323 self.geometry = game.geometry
324 self._returns = np.zeros([self.num_players()], dtype=np.float64)
325 self._distribution = shared_value.SharedValue(game.initial_distribution)
326 self.proba_noise = game.proba_noise
327 self.coef_congestion = game.coef_congestion
328 self.forbidden_states = game.forbidden_states
329 self.coef_target = game.coef_target

370 def chance_outcomes(self) -> List[Tuple[int, float]]:
371 """Returns the possible chance outcomes and their probabilities."""
372 if self._is_position_init:
373 if (
374 self._population is None
375 or not 0 <= self._population < self.num_players()
376 ):
377 raise ValueError(f"Invalid population {self._population}")
378 p = self._population % 2
379 dist = self._distribution.value
380 dist_p = dist[p * self.num_states : (p + 1) * self.num_states]
381 pos_indices_flat = np.nonzero(dist_p)[0]
382 pos_indices = [
383 np.array([i % self.size, (i - i % self.size) // self.size])
384 for i in pos_indices_flat
385 ]
386 # Beware: In the initial distribution representation, x and y correspond
387 # respectively to the row and the column, but in the state representation,
388 # they correspond to the column and the row.
389 return [
390 (pos_to_merged(i, self.size), dist_p[i[1] * self.size + i[0]])
391 for i in pos_indices
392 ]
393 return [
394 (0, 1.0 - self.proba_noise),
395 (1, self.proba_noise / 4.0),
396 (2, self.proba_noise / 4.0),
397 (3, self.proba_noise / 4.0),
398 (4, self.proba_noise / 4.0),
399 ]

401 def update_pos(self, action):
402 """Updates the position of the player given a move action."""
403 if action < 0 or action >= len(self._ACTION_TO_MOVE_ACTION_TO_MOVE):
404 raise ValueError(
405 f"The action must be between 0 and {len(self._ACTION_TO_MOVE)}, "
406 f"got {action}"
407 )
408 candidate_pos = self._pos + self._ACTION_TO_MOVE_ACTION_TO_MOVE[action]
409 # if candidate_pos in self.forbidden_states:
410 # if np.any(np.all(candidate_pos == self.forbidden_states, axis=1)):
411 if any(np.array_equal(candidate_pos, x) for x in self.forbidden_states):
412 candidate_pos = self._pos
413 elif self.geometry == Geometry.TORUS:
414 candidate_pos += self.size
415 candidate_pos %= self.size
416 else:
417 assert (
418 self.geometry == Geometry.SQUARE
419 ), f"Invalid geometry {self.geometry}"
420 # Keep the position within the bounds of the square.
421 candidate_pos = np.minimum(candidate_pos, self.size - 1)
422 candidate_pos = np.maximum(candidate_pos, 0)
423 self._pos = candidate_pos

425 def _apply_action(self, action):
426 """Applies the specified action to the state."""
427 if self._population is None:
428 raise ValueError(
429 "Attempting to perform an action with a population-less state."
430 )
431 if self._player_id == pyspiel.PlayerId.MEAN_FIELD:
432 raise ValueError(
433 "_apply_action should not be called at a MEAN_FIELD state."
434 )
435 self._returns += np.array(self.rewards())
436 if self._is_position_init:
437 self._pos = merged_to_pos(action, self.size)
438 self._is_position_init = False
439 self._player_id = self._population
440 elif self._player_id == pyspiel.PlayerId.CHANCE:
441 self.update_pos(action)
442 self._t += 1
443 self._player_id = pyspiel.PlayerId.MEAN_FIELD
444 elif int(self._player_id) >= 0:
445 assert self._player_id == self._population, (
446 f"Invalid decision player id {self._player_id} "
447 f"expected {self._population}"
448 )
449 self.update_pos(action)
450 self._player_id = pyspiel.PlayerId.CHANCE
451 else:
452 raise ValueError(f"Unexpected state. Player id: {self._player_id}")

477 def get_pos_proba(self, pos: np.ndarray, population: int) -> float:
478 """Gets the probability of a pos and population in the current distrib.
479
480 Args:
481 pos: 2D position.
482 population: Population requested.
483
484 Returns:
485 The probability for the provided position and population.
486 """
487 assert (pos >= 0).all(), pos
488 assert (pos < self.size).all(), pos
489 assert 0 <= population < self.num_players(), population
490 # This logic needs to match the ordering defined in distribution_support().
491 index = population + self.num_players() * (pos[1] + self.size * pos[0])
492 assert 0 <= index < len(self._distribution.value), (
493 f"Invalid index {index} vs dist length:"
494 f" {len(self._distribution.value)}, population={population}, pos={pos},"
495 f" state={self}"
496 )
497 return self._distribution.value[index]

499 def update_distribution(self, distribution):
500 """This function is central and specific to the logic of the MFG.
501
502 It should only be called when the node is in MEAN_FIELD state.
503
504 Args:
505 distribution: List of floats that should contain the probability of each
506 state returned by distribution_support().
507 """
508 expected_dist_size = self.num_states * self.num_players()
509 assert len(distribution) == expected_dist_size, (
510 "Unexpected distribution length "
511 f"{len(distribution)} != {expected_dist_size}"
512 )
513 if self._player_id != pyspiel.PlayerId.MEAN_FIELD:
514 raise ValueError(
515 "update_distribution should only be called at a MEAN_FIELD state."
516 )
517 self._distribution = shared_value.SharedValue(distribution)
518 self._player_id = self._population

530 def rewards(self) -> List[float]:
531 """Crowd avoidance rewards for all populations.
532
533 Returns:
534 One float per population.
535 """
536 if int(self._player_id) < 0:
537 return [0.0] * self.num_players()
538 densities = np.array(
539 [
540 self.get_pos_proba(self._pos, population)
541 for population in range(self.num_players())
542 ],
543 dtype=np.float64,
544 )
545 rew = -self.coef_congestion * np.dot(self.congestion_matrix, densities)
546 # Rewards for target positions.
547 rew[0] += self.coef_target * np.array_equal(
548 self._pos, self.target_positions[0]
549 )
550 rew[1] += self.coef_target * np.array_equal(
551 self._pos, self.target_positions[1]
552 )
553 return list(rew)
554
555 def returns(self) -> List[float]:
556 """Returns is the sum of all payoffs collected so far."""
557 return list(self._returns + np.array(self.rewards()))

569 def __init__(self, params, game):
570 """Initializes an empty observation tensor."""
571 del params
572
573 self.size = game.size
574 self.horizon = game.horizon
575 # +1 to allow t == horizon.
576 self.tensor = np.zeros(2 * self.size + self.horizon + 1, np.float32)
577 self.dict = {
578 "x": self.tensor[: self.size],
579 "y": self.tensor[self.size : self.size * 2],
580 "t": self.tensor[self.size * 2 :],
581 }

583 def set_from(self, state: MFGCrowdAvoidanceState, player: int):
584 """Updates `tensor` and `dict` to reflect `state` from PoV of `player`."""
585 del player
586 # We update the observation via the shaped tensor since indexing is more
587 # convenient than with the 1-D tensor. Both are views onto the same memory.
588 self.tensor.fill(0)
589 # state.pos is None for the initial (blank) state, don't set any
590 # position bit in that case.
591 if state.pos is not None:
592 if not (state.pos >= 0).all() or not (state.pos < self.size).all():
593 raise ValueError(
594 f"Expected {state} positions to be in [0, {self.size})"
595 )
596 self.dict["x"][state.pos[0]] = 1
597 self.dict["y"][state.pos[1]] = 1
598 if not 0 <= state.t <= self.horizon:
599 raise ValueError(f"Expected {state} time to be in [0, {self.horizon}]")
600 self.dict["t"][state.t] = 1
601
602 def string_from(self, state, player):
603 """Observation of `state` from the PoV of `player`, as a string."""
604 del player
605 return str(state)
