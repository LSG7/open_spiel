37 // This example check to see how different the approximate equilibria CFR find
38 // are, based on random initial regrets and Monte Carlo sampling.
39 
40 // Example code for using CFR+ to solve Kuhn Poker.
41 int main(int argc, char** argv) {
42  open_spiel::Init("", &argc, &argv, false);
43  absl::ParseCommandLine(argc, argv);
44  std::shared_ptr<const open_spiel::Game> game =
45  open_spiel::LoadGame(absl::GetFlag(FLAGS_game));
46  std::mt19937 rng(absl::GetFlag(FLAGS_seed));
47  absl::uniform_int_distribution<int> dist;
48  std::string file_prefix = absl::GetFlag(FLAGS_file_prefix);
49 
50  // Random initial regrets
51  for (int i = 0; i < absl::GetFlag(FLAGS_repeats); ++i) {
52  std::string filename = absl::StrCat(file_prefix, "/cfr_rir_", i, ".txt");
53  std::cout << "Random initial regrets, repeat number " << i
54  << ", generating " << filename << std::endl;
55  int seed = dist(rng);
56  open_spiel::algorithms::CFRSolverBase solver(*game,
57  /*alternating_updates*/true, /*linear_averaging*/false,
58  /*regret_matching_plus*/false, /*random_initial_regrets*/true,
59  /*seed*/seed);
60 
61  for (int i = 0; i < absl::GetFlag(FLAGS_num_cfr_iters); ++i) {
62  solver.EvaluateAndUpdatePolicy();
63  }
64  open_spiel::TabularPolicy avg_policy = solver.TabularAveragePolicy();
65  open_spiel::file::File outfile(filename, "w");
66  outfile.Write(avg_policy.ToStringSorted());
67  }
68 
69  // Outcome Sampling MCCFR
70  for (int i = 0; i < absl::GetFlag(FLAGS_repeats); ++i) {
71  std::string filename = absl::StrCat(file_prefix, "/cfr_cfros_", i, ".txt");
72  std::cout << "Outcome Sampling MCCFR, repeat number " << i
73  << ", generating " << filename << std::endl;
74  int seed = dist(rng);
75  open_spiel::algorithms::OutcomeSamplingMCCFRSolver solver(*game,
76  /*epsilon*/0.6, seed);
77 
78  for (int i = 0; i < absl::GetFlag(FLAGS_num_cfros_iters); ++i) {
79  solver.RunIteration();
80  }
81  open_spiel::TabularPolicy avg_policy = solver.TabularAveragePolicy();
82  open_spiel::file::File outfile(filename, "w");
83  outfile.Write(avg_policy.ToStringSorted());
84  }
