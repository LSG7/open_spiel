33 def run_election(self, profile: base.PreferenceProfile) -> base.RankOutcome:
34 assert self.is_valid_profile(profile)
35 alternatives = profile.alternatives
36 num_alternatives = profile.num_alternatives()
37 pref_mat = profile.pref_matrix()
38 strongest_paths = np.zeros(shape=(num_alternatives, num_alternatives),
39 dtype=np.float32)
40 # calculating the direct paths
41 for i in range(num_alternatives):
42 for j in range(num_alternatives):
43 if i != j:
44 if pref_mat[i, j] > pref_mat[j, i]:
45 strongest_paths[i, j] = pref_mat[i, j]
46 else:
47 strongest_paths[i, j] = 0
48 # checking if any indirect paths are better
49 for i in range(num_alternatives):
50 for j in range(num_alternatives):
51 if i != j and strongest_paths[j, i] > 0:
52 for k in range(num_alternatives):
53 if i != k and j != k:
54 # if the path from j to k through i is better, replace
55 strongest_paths[j, k] = max(strongest_paths[j, k],
56 min(strongest_paths[j, i],
57 strongest_paths[i, k]))
58
59 def compare(x, y):
60 return strongest_paths[x, y] - strongest_paths[y, x]
61 ranking_idx = np.arange(num_alternatives)
62 sorted_ranking_idx = sorted(ranking_idx, key=functools.cmp_to_key(compare),
63 reverse=True)
64 # Define the scores as the sum of preferences for everything it beats in
65 # the order.
66 cumul_score = 0
67 # start at the end and work backwards
68 ranking_alts = [alternatives[sorted_ranking_idx[-1]]]
69 scores = [0]
70 i = num_alternatives - 2
71 while i >= 0:
72 alt_idx_i = sorted_ranking_idx[i]
73 alt_idx_j = sorted_ranking_idx[i+1]
74 ranking_alts.insert(0, alternatives[alt_idx_i])
75 cumul_score += pref_mat[alt_idx_i, alt_idx_j]
76 scores.insert(0, cumul_score)
77 i -= 1
78 return base.RankOutcome(rankings=ranking_alts, scores=scores)
