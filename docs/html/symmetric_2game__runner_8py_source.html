23 def construct_game_queries(base_profile, num_checkpts):
24  """Constructs a list of checkpoint selection tuples to query value function.
25 
26  Each query tuple (p1's selected checkpt, ..., p7's selected checkpt)
27  fixes the players in the game of diplomacy to be played. It may be necessary
28  to play several games with the same players to form an accurate estimate of
29  the value or payoff for each player as checkpts contain stochastic policies.
30 
31  Args:
32  base_profile: list of selected checkpts for each player, i.e.,
33  a sample from the player strategy profile ([x_i ~ p(x_i)])
34  num_checkpts: number of checkpts available to each player
35  Returns:
36  Set of query tuples containing a selected checkpoint index for each player.
37  """
38  new_queries = set([])
39 
40  pi, pj = 0, 1
41  new_profile = list(base_profile)
42  for ai in range(num_checkpts):
43  new_profile[pi] = ai
44  for aj in range(num_checkpts):
45  new_profile[pj] = aj
46  query = tuple(new_profile)
47  new_queries.update([query])
48 
49  return new_queries

79 def run_games_and_record_payoffs(game_queries, evaluate_game, ckpt_to_policy):
80  """Simulate games according to game queries and return results.
81 
82  Args:
83  game_queries: set of tuples containing indices specifying each players strat
84  evaluate_game: callable function that takes a list of policies as argument
85  ckpt_to_policy: maps a strat (or checkpoint) to a policy
86  Returns:
87  dictionary: key=query, value=np.array of payoffs (1 for each player)
88  """
89  game_results = {}
90  for query in game_queries:
91  policies = [ckpt_to_policy[ckpt] for ckpt in query]
92  payoffs = evaluate_game(policies)
93  game_results.update({query: payoffs})
94  return game_results

97 def form_payoff_matrices(game_results, num_checkpts):
98  """Packages dictionary of game results into a payoff tensor.
99 
100  Args:
101  game_results: dictionary of payoffs for each game evaluated
102  num_checkpts: int, number of strats (or ckpts) per player
103  Returns:
104  payoff_matrices: np.array (2 x num_checkpts x num_checkpts) with payoffs for
105  two players (assumes symmetric game and only info for 2 players is needed
106  for stochastic gradients)
107  """
108  payoff_matrices = np.zeros((2, num_checkpts, num_checkpts))
109  for profile, payoffs in game_results.items():
110  i, j = profile[:2]
111  payoff_matrices[:, i, j] = payoffs[:2]
112  return payoff_matrices
