16r"""Policy Gradient based agents implemented in TensorFlow.
17
18This class is composed of three policy gradient (PG) algorithms:
19
20- Q-based Policy Gradient (QPG): an "all-actions" advantage actor-critic
21algorithm differing from A2C in that all action values are used to estimate the
22policy gradient (as opposed to only using the action taken into account):
23
24 baseline = \sum_a pi_a * Q_a
25 loss = - \sum_a pi_a * (Q_a - baseline)
26
27where (Q_a - baseline) is the usual advantage. QPG is also known as Mean
28Actor-Critic (https://arxiv.org/abs/1709.00503).
29
30
31- Regret policy gradient (RPG): a PG algorithm inspired by counterfactual regret
32minimization (CFR). Unlike standard actor-critic methods (e.g. A2C), the loss is
33defined purely in terms of thresholded regrets as follows:
34
35 baseline = \sum_a pi_a * Q_a
36 loss = regret = \sum_a relu(Q_a - baseline)
37
38where gradients only flow through the action value (Q_a) part and are blocked on
39the baseline part (which is trained separately by usual MSE loss).
40The lack of negative sign in the front of the loss represents a switch from
41gradient ascent on the score to descent on the loss.
42
43
44- Regret Matching Policy Gradient (RMPG): inspired by regret-matching, the
45policy gradient is by weighted by the thresholded regret:
46
47 baseline = \sum_a pi_a * Q_a
48 loss = - \sum_a pi_a * relu(Q_a - baseline)
49
50
51These algorithms were published in NeurIPS 2018. Paper title: "Actor-Critic
52Policy Optimization in Partially Observable Multiagent Environment", the paper
53is available at: https://arxiv.org/abs/1810.09026.
54
55- Advantage Actor Critic (A2C): The popular advantage actor critic (A2C)
56algorithm. The algorithm uses the baseline (Value function) as a control variate
57to reduce variance of the policy gradient. The loss is only computed for the
58actions actually taken in the episode as opposed to a loss computed for all
59actions in the variants above.
60
61 advantages = returns - baseline
62 loss = -log(pi_a) * advantages
63
64The algorithm can be found in the textbook:
65https://incompleteideas.net/book/RLbook2018.pdf under the chapter on
66`Policy Gradients`.
67
68See open_spiel/python/algorithms/losses/rl_losses_test.py for an example of the
69loss computation.
70"""

98 def __init__(self,
99 session,
100 player_id,
101 info_state_size,
102 num_actions,
103 loss_str="a2c",
104 loss_class=None,
105 hidden_layers_sizes=(128,),
106 batch_size=16,
107 critic_learning_rate=0.01,
108 pi_learning_rate=0.001,
109 entropy_cost=0.01,
110 num_critic_before_pi=8,
111 additional_discount_factor=1.0,
112 max_global_gradient_norm=None,
113 optimizer_str="sgd"):
114 """Initialize the PolicyGradient agent.
115
116 Args:
117 session: Tensorflow session.
118 player_id: int, player identifier. Usually its position in the game.
119 info_state_size: int, info_state vector size.
120 num_actions: int, number of actions per info state.
121 loss_str: string or None. If string, must be one of ["rpg", "qpg", "rm",
122 "a2c"] and defined in `_get_loss_class`. If None, a loss class must be
123 passed through `loss_class`. Defaults to "a2c".
124 loss_class: Class or None. If Class, it must define the policy gradient
125 loss. If None a loss class in a string format must be passed through
126 `loss_str`. Defaults to None.
127 hidden_layers_sizes: iterable, defines the neural network layers. Defaults
128 to (128,), which produces a NN: [INPUT] -> [128] -> ReLU -> [OUTPUT].
129 batch_size: int, batch size to use for Q and Pi learning. Defaults to 128.
130 critic_learning_rate: float, learning rate used for Critic (Q or V).
131 Defaults to 0.01.
132 pi_learning_rate: float, learning rate used for Pi. Defaults to 0.001.
133 entropy_cost: float, entropy cost used to multiply the entropy loss. Can
134 be set to None to skip entropy computation. Defaults to 0.01.
135 num_critic_before_pi: int, number of Critic (Q or V) updates before each
136 Pi update. Defaults to 8 (every 8th critic learning step, Pi also
137 learns).
138 additional_discount_factor: float, additional discount to compute returns.
139 Defaults to 1.0, in which case, no extra discount is applied. None that
140 users must provide *only one of* `loss_str` or `loss_class`.
141 max_global_gradient_norm: float or None, maximum global norm of a gradient
142 to which the gradient is shrunk if its value is larger. Defaults to
143 None.
144 optimizer_str: String defining which optimizer to use. Supported values
145 are {sgd, adam}. Defaults to sgd
146 """
147 assert bool(loss_str) ^ bool(loss_class), "Please provide only one option."
148 self._kwargs = locals()
149 loss_class = loss_class if loss_class else self._get_loss_class(loss_str)
150 self._loss_class = loss_class
151
152 self.player_id = player_id
153 self._session = session
154 self._num_actions = num_actions
155 self._layer_sizes = hidden_layers_sizes
156 self._batch_size = batch_size
157 self._extra_discount = additional_discount_factor
158 self._num_critic_before_pi = num_critic_before_pi
159
160 self._episode_data = []
161 self._dataset = collections.defaultdict(list)
162 self._prev_time_step = None
163 self._prev_action = None
164
165 # Step counters
166 self._step_counter = 0
167 self._episode_counter = 0
168 self._num_learn_steps = 0
169
170 # Keep track of the last training loss achieved in an update step.
171 self._last_loss_value = None
172
173 # Placeholders
174 self._info_state_ph = tf.placeholder(
175 shape=[None, info_state_size], dtype=tf.float32, name="info_state_ph")
176 self._action_ph = tf.placeholder(
177 shape=[None], dtype=tf.int32, name="action_ph")
178 self._return_ph = tf.placeholder(
179 shape=[None], dtype=tf.float32, name="return_ph")
180
181 # Network
182 # activate final as we plug logit and qvalue heads afterwards.
183 self._net_torso = simple_nets.MLPTorso(info_state_size, self._layer_sizes)
184 torso_out = self._net_torso(self._info_state_ph)
185 torso_out_size = self._layer_sizes[-1]
186 self._policy_logits_layer = simple_nets.Linear(
187 torso_out_size,
188 self._num_actions,
189 activate_relu=False,
190 name="policy_head")
191 # Do not remove policy_logits_network. Even if it's not used directly here,
192 # other code outside this file refers to it.
193 self.policy_logits_network = simple_nets.Sequential(
194 [self._net_torso, self._policy_logits_layer])
195 self._policy_logits = self._policy_logits_layer(torso_out)
196 self._policy_probs = tf.nn.softmax(self._policy_logits)
197
198 self._savers = []
199
200 # Add baseline (V) head for A2C (or Q-head for QPG / RPG / RMPG)
201 if loss_class.__name__ == "BatchA2CLoss":
202 self._baseline_layer = simple_nets.Linear(
203 torso_out_size, 1, activate_relu=False, name="baseline")
204 self._baseline = tf.squeeze(self._baseline_layer(torso_out), axis=1)
205 else:
206 self._q_values_layer = simple_nets.Linear(
207 torso_out_size,
208 self._num_actions,
209 activate_relu=False,
210 name="q_values_head")
211 self._q_values = self._q_values_layer(torso_out)
212
213 # Critic loss
214 # Baseline loss in case of A2C
215 if loss_class.__name__ == "BatchA2CLoss":
216 self._critic_loss = tf.reduce_mean(
217 tf.losses.mean_squared_error(
218 labels=self._return_ph, predictions=self._baseline))
219 else:
220 # Q-loss otherwise.
221 action_indices = tf.stack(
222 [tf.range(tf.shape(self._q_values)[0]), self._action_ph], axis=-1)
223 value_predictions = tf.gather_nd(self._q_values, action_indices)
224 self._critic_loss = tf.reduce_mean(
225 tf.losses.mean_squared_error(
226 labels=self._return_ph, predictions=value_predictions))
227 if optimizer_str == "adam":
228 self._critic_optimizer = tf.train.AdamOptimizer(
229 learning_rate=critic_learning_rate)
230 elif optimizer_str == "sgd":
231 self._critic_optimizer = tf.train.GradientDescentOptimizer(
232 learning_rate=critic_learning_rate)
233 else:
234 raise ValueError("Not implemented, choose from 'adam' and 'sgd'.")

236 def minimize_with_clipping(optimizer, loss):
237 grads_and_vars = optimizer.compute_gradients(loss)
238 if max_global_gradient_norm is not None:
239 grads, variables = zip(*grads_and_vars)
240 grads, _ = tf.clip_by_global_norm(grads, max_global_gradient_norm)
241 grads_and_vars = list(zip(grads, variables))
242
243 return optimizer.apply_gradients(grads_and_vars)
244
245 self._critic_learn_step = minimize_with_clipping(self._critic_optimizer,
246 self._critic_loss)
247
248 # Pi loss
249 pg_class = loss_class(entropy_cost=entropy_cost)
250 if loss_class.__name__ == "BatchA2CLoss":
251 self._pi_loss = pg_class.loss(
252 policy_logits=self._policy_logits,
253 baseline=self._baseline,
254 actions=self._action_ph,
255 returns=self._return_ph)
256 else:
257 self._pi_loss = pg_class.loss(
258 policy_logits=self._policy_logits, action_values=self._q_values)
259 if optimizer_str == "adam":
260 self._pi_optimizer = tf.train.AdamOptimizer(
261 learning_rate=pi_learning_rate)
262 elif optimizer_str == "sgd":
263 self._pi_optimizer = tf.train.GradientDescentOptimizer(
264 learning_rate=pi_learning_rate)
265
266 self._pi_learn_step = minimize_with_clipping(self._pi_optimizer,
267 self._pi_loss)
268 self._loss_str = loss_str
269 self._initialize()

281 def _act(self, info_state, legal_actions):
282 # Make a singleton batch for NN compatibility: [1, info_state_size]
283 info_state = np.reshape(info_state, [1, -1])
284 policy_probs = self._session.run(
285 self._policy_probs, feed_dict={self._info_state_ph: info_state})
286
287 # Remove illegal actions, re-normalize probs
288 probs = np.zeros(self._num_actions)
289 probs[legal_actions] = policy_probs[0][legal_actions]
290 if sum(probs) != 0:
291 probs /= sum(probs)
292 else:
293 probs[legal_actions] = 1 / len(legal_actions)
294 action = np.random.choice(len(probs), p=probs)
295 return action, probs

297 def step(self, time_step, is_evaluation=False):
298 """Returns the action to be taken and updates the network if needed.
299
300 Args:
301 time_step: an instance of rl_environment.TimeStep.
302 is_evaluation: bool, whether this is a training or evaluation call.
303 Defaults to False.
304
305 Returns:
306 A `rl_agent.StepOutput` containing the action probs and chosen action.
307 """
308 # Act step: don't act at terminal info states or if its not our turn.
309 if (not time_step.last()) and (
310 time_step.is_simultaneous_move() or
311 self.player_id == time_step.current_player()):
312 info_state = time_step.observations["info_state"][self.player_id]
313 legal_actions = time_step.observations["legal_actions"][self.player_id]
314 action, probs = self._act(info_state, legal_actions)
315 else:
316 action = None
317 probs = []
318
319 if not is_evaluation:
320 self._step_counter += 1
321
322 # Add data points to current episode buffer.
323 if self._prev_time_step:
324 self._add_transition(time_step)
325
326 # Episode done, add to dataset and maybe learn.
327 if time_step.last():
328 self._add_episode_data_to_dataset()
329 self._episode_counter += 1
330
331 if len(self._dataset["returns"]) >= self._batch_size:
332 self._critic_update()
333 self._num_learn_steps += 1
334 if self._num_learn_steps % self._num_critic_before_pi == 0:
335 self._pi_update()
336 self._dataset = collections.defaultdict(list)
337
338 self._prev_time_step = None
339 self._prev_action = None
340 return
341 else:
342 self._prev_time_step = time_step
343 self._prev_action = action
344
345 return rl_agent.StepOutput(action=action, probs=probs)

384 def _add_episode_data_to_dataset(self):
385 """Add episode data to the buffer."""
386 info_states = [data.info_state for data in self._episode_data]
387 rewards = [data.reward for data in self._episode_data]
388 discount = [data.discount for data in self._episode_data]
389 actions = [data.action for data in self._episode_data]
390
391 # Calculate returns
392 returns = np.array(rewards)
393 for idx in reversed(range(len(rewards[:-1]))):
394 returns[idx] = (
395 rewards[idx] +
396 discount[idx] * returns[idx + 1] * self._extra_discount)
397
398 # Add flattened data points to dataset
399 self._dataset["actions"].extend(actions)
400 self._dataset["returns"].extend(returns)
401 self._dataset["info_states"].extend(info_states)
402 self._episode_data = []

404 def _add_transition(self, time_step):
405 """Adds intra-episode transition to the `_episode_data` buffer.
406
407 Adds the transition from `self._prev_time_step` to `time_step`.
408
409 Args:
410 time_step: an instance of rl_environment.TimeStep.
411 """
412 assert self._prev_time_step is not None
413 legal_actions = (
414 self._prev_time_step.observations["legal_actions"][self.player_id])
415 legal_actions_mask = np.zeros(self._num_actions)
416 legal_actions_mask[legal_actions] = 1.0
417 transition = Transition(
418 info_state=(
419 self._prev_time_step.observations["info_state"][self.player_id][:]),
420 action=self._prev_action,
421 reward=time_step.rewards[self.player_id],
422 discount=time_step.discounts[self.player_id],
423 legal_actions_mask=legal_actions_mask)
424
425 self._episode_data.append(transition)

502 def copy_with_noise(self, sigma=0.0, copy_weights=True):
503 """Copies the object and perturbates its network's weights with noise.
504
505 Args:
506 sigma: gaussian dropout variance term : Multiplicative noise following
507 (1+sigma*epsilon), epsilon standard gaussian variable, multiplies each
508 model weight. sigma=0 means no perturbation.
509 copy_weights: Boolean determining whether to copy model weights (True) or
510 just model hyperparameters.
511
512 Returns:
513 Perturbated copy of the model.
514 """
515 _ = self._kwargs.pop("self", None)
516 copied_object = PolicyGradient(**self._kwargs)
517
518 net_torso = getattr(copied_object, "_net_torso")
519 policy_logits_layer = getattr(copied_object, "_policy_logits_layer")
520 if hasattr(copied_object, "_q_values_layer"):
521 q_values_layer = getattr(copied_object, "_q_values_layer")
522 if hasattr(copied_object, "_baseline_layer"):
523 baseline_layer = getattr(copied_object, "_baseline_layer")
524
525 if copy_weights:
526 copy_mlp_weights = tf.group(*[
527 va.assign(vb * (1 + sigma * tf.random.normal(vb.shape)))
528 for va, vb in zip(net_torso.variables, self._net_torso.variables)
529 ])
530 self._session.run(copy_mlp_weights)
531
532 copy_logit_weights = tf.group(*[
533 va.assign(vb * (1 + sigma * tf.random.normal(vb.shape)))
534 for va, vb in zip(policy_logits_layer.variables,
535 self._policy_logits_layer.variables)
536 ])
537 self._session.run(copy_logit_weights)
538 if hasattr(copied_object, "_q_values_layer"):
539 copy_q_value_weights = tf.group(*[
540 va.assign(vb * (1 + sigma * tf.random.normal(vb.shape))) for va, vb
541 in zip(q_values_layer.variables, self._q_values_layer.variables)
542 ])
543 self._session.run(copy_q_value_weights)
544 if hasattr(copied_object, "_baseline_layer"):
545 copy_baseline_weights = tf.group(*[
546 va.assign(vb * (1 + sigma * tf.random.normal(vb.shape))) for va, vb
547 in zip(baseline_layer.variables, self._baseline_layer.variables)
548 ])
549 self._session.run(copy_baseline_weights)
550
551 for var in getattr(copied_object, "_critic_optimizer").variables():
552 self._session.run(var.initializer)
553 for var in getattr(copied_object, "_pi_optimizer").variables():
554 self._session.run(var.initializer)
555
556 return copied_object
