28// Phantom Tic-Tac-Toe is a phantom version of the classic game of Tic-Tac-Toe
29// (Noughts and Crosses). For some perfect information game X", the game
30// "phantom X" is a version of the game X where the players do not observe the
31// other players' pieces. Only a referee knows the full state of the board.
32// So, on a player's turn, a chosen moves may fail because it is illegal given
33// the true state of the board; in this case, a player can continue to try moves
34// until one succeeds.
35//
36// Common phantom games include Kriegspiel (Phantom chess), e.g. see
37// https://en.wikipedia.org/wiki/Kriegspiel_(chess), and Phantom Go.
38// See also http://mlanctot.info/files/papers/PhD_Thesis_MarcLanctot.pdf, Ch 3.
39//

131// Implements the FOE abstraction from Lanctot et al. '12
132// http://mlanctot.info/files/papers/12icml-ir.pdf
133class ImperfectRecallPTTTState : public PhantomTTTState {
134 public:
135 ImperfectRecallPTTTState(std::shared_ptr<const Game> game,
136 ObservationType obs_type)
137 : PhantomTTTState(game, obs_type) {}
138 std::string InformationStateString(Player player) const override {
139 SPIEL_CHECK_GE(player, 0);
140 SPIEL_CHECK_LT(player, num_players_);
141 return absl::StrCat("P", player, " ", ViewToString(player));
142 }
143 std::unique_ptr<State> Clone() const override {
144 return std::unique_ptr<State>(new ImperfectRecallPTTTState(*this));
