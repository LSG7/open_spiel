71 CellState HexState::PlayerAndActionToState(Player player, Action move) const {
72  // This function returns the CellState resulting from the given move.
73  // The cell state tells us:
74  // - The colour of the stone.
75  // - If the stone results in a winning connection.
76  // - If the stone connects to one of the edges needed for that colour's
77  // winning connection.
78  //
79  // We know the colour from the argument player
80  // For connectedness to the edges, we check if the move is in first/last
81  // row/column, or if any of the neighbours are the same colour and connected.
82  if (player == 0) {
83  bool north_connected = false;
84  bool south_connected = false;
85  if (move < num_cols_) { // First row
86  north_connected = true;
87  } else if (move >= (board_.size() - num_cols_)) { // Last row
88  south_connected = true;
89  }
90  for (int neighbour : AdjacentCells(move)) {
91  if (board_[neighbour] == CellState::kBlackNorth) {
92  north_connected = true;
93  } else if (board_[neighbour] == CellState::kBlackSouth) {
94  south_connected = true;
95  }
96  }
97  if (north_connected && south_connected) {
98  return CellState::kBlackWin;
99  } else if (north_connected) {
100  return CellState::kBlackNorth;
101  } else if (south_connected) {
102  return CellState::kBlackSouth;
103  } else {
104  return CellState::kBlack;
105  }
106  } else if (player == 1) {
107  bool west_connected = false;
108  bool east_connected = false;
109  if (move % num_cols_ == 0) { // First column
110  west_connected = true;
111  } else if (move % num_cols_ == num_cols_ - 1) { // Last column
112  east_connected = true;
113  }
114  for (int neighbour : AdjacentCells(move)) {
115  if (board_[neighbour] == CellState::kWhiteWest) {
116  west_connected = true;
117  } else if (board_[neighbour] == CellState::kWhiteEast) {
118  east_connected = true;
119  }
120  }
121  if (west_connected && east_connected) {
122  return CellState::kWhiteWin;
123  } else if (west_connected) {
124  return CellState::kWhiteWest;
125  } else if (east_connected) {
126  return CellState::kWhiteEast;
127  } else {
128  return CellState::kWhite;
129  }
130  } else {
131  SpielFatalError(absl::StrCat("Invalid player id ", player));
132  return CellState::kEmpty;
133  }
134 }

162 void HexState::DoApplyAction(Action move) {
163  SPIEL_CHECK_EQ(board_[move], CellState::kEmpty);
164  CellState move_cell_state = PlayerAndActionToState(CurrentPlayer(), move);
165  board_[move] = move_cell_state;
166  if (move_cell_state == CellState::kBlackWin) {
167  result_black_perspective_ = 1;
168  } else if (move_cell_state == CellState::kWhiteWin) {
169  result_black_perspective_ = -1;
170  } else if (move_cell_state != CellState::kBlack &&
171  move_cell_state != CellState::kWhite) {
172  // Move is connected to an edge but not winning.
173  // Update edge-connected groups with a flood-fill, to maintain that all edge
174  // connected nodes are known about.
175  // We don't do flood fill if a player has won, so it's impossible for a cell
176  // connected to an edge to be changed by the flood-fill.
177  CellState cell_state_to_change =
178  (current_player_ == 0 ? CellState::kBlack : CellState::kWhite);
179  // We assume that move can safely be cast to int
180  std::vector<int> flood_stack = {static_cast<int>(move)};
181  int latest_cell;
182  while (!flood_stack.empty()) {
183  latest_cell = flood_stack.back();
184  flood_stack.pop_back();
185  for (int neighbour : AdjacentCells(latest_cell)) {
186  if (board_[neighbour] == cell_state_to_change) {
187  // We make the change before putting the cell on the queue to avoid
188  // putting the same cell on the queue multiple times
189  board_[neighbour] = move_cell_state;
190  flood_stack.push_back(neighbour);
191  }
192  }
193  }
194  }
195  current_player_ = 1 - current_player_;
196 }

198 std::vector<Action> HexState::LegalActions() const {

218 std::vector<int> HexState::AdjacentCells(int cell) const {
219  std::vector<int> neighbours = {};
220  bool north_edge = (cell < num_cols_);
221  bool south_edge = (cell >= (board_.size() - num_cols_));
222  bool west_edge = (cell % num_cols_ == 0);
223  bool east_edge = (cell % num_cols_ == num_cols_ - 1);
224  if (!north_edge) { neighbours.push_back(cell - num_cols_); }
225  if (!north_edge && !east_edge) { neighbours.push_back(cell - num_cols_ + 1); }
226  if (!east_edge) { neighbours.push_back(cell + 1); }
227  if (!south_edge) { neighbours.push_back(cell + num_cols_); }
228  if (!south_edge && !west_edge) { neighbours.push_back(cell + num_cols_ - 1); }
229  if (!west_edge) { neighbours.push_back(cell - 1); }
230  return neighbours;
231 }

239 std::string HexState::ToString() const {
240  std::string str;
241  // Each cell has the cell plus a space
242  // nth line has n spaces, and 1 "\n", except last line has no "\n"
243  str.reserve(num_cols_ * num_rows_ * 2 + num_rows_ * (num_rows_ + 1) / 2 - 1);
244  int line_num = 0;
245  for (int cell = 0; cell < board_.size(); ++cell) {
246  // if it's the first cell in a new row
247  if (cell && cell % num_cols_ == 0) {
248  absl::StrAppend(&str, "\n");
249  line_num++;
250  absl::StrAppend(&str, std::string(line_num, ' '));
251  }
252  absl::StrAppend(&str, StateToString(board_[cell]));
253  absl::StrAppend(&str, " ");
254  }
255  return str;
256 }
