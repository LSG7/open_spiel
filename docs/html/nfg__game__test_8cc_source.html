97 void NFGLoadSampleFromFile() {
98  absl::optional<std::string> file = FindFile(kSampleNFGFile, 2);
99  if (file.has_value()) {
100  std::cout << "Found file: " << file.value() << "; running sim test.";
101  std::shared_ptr<const Game> game =
102  LoadGame("nfg_game", {{"filename", GameParameter(file.value())}});
103  SPIEL_CHECK_TRUE(game != nullptr);
104  GameType type = game->GetType();
105  SPIEL_CHECK_EQ(type.dynamics, GameType::Dynamics::kSimultaneous);
106  SPIEL_CHECK_EQ(type.information, GameType::Information::kOneShot);
107  SPIEL_CHECK_EQ(type.utility, GameType::Utility::kGeneralSum);
108  SPIEL_CHECK_EQ(type.chance_mode, GameType::ChanceMode::kDeterministic);
109  SPIEL_CHECK_EQ(game->NumPlayers(), 2);
110  SPIEL_CHECK_EQ(game->NumDistinctActions(), 3);
111  SPIEL_CHECK_EQ(game->MaxChanceOutcomes(), 0);
112  testing::RandomSimTestNoSerialize(*game, 100);
113  }
114 }

137 void NFGExportReloadTestInternalGames() {
138  std::vector<std::string> game_strings = {
139  "matrix_rps",
140  "matrix_shapleys_game",
141  "matrix_pd",
142  "matrix_sh",
143  "blotto(players=2,coins=5,fields=3)",
144  "blotto(players=3,coins=5,fields=3)",
145  };
146 
147  for (const std::string& game_string : game_strings) {
148  // Load a native game, write it to NFG, parse the NFG, and export again.
149  // Both .nfg strings should be identical.
150  std::shared_ptr<const Game> general_game = LoadGame(game_string);
151  std::shared_ptr<const Game> game;
152  if (general_game->NumPlayers() == 2) {
153  game = algorithms::LoadMatrixGame(game_string);
154  } else {
155  game = algorithms::LoadTensorGame(game_string);
156  }
157  std::string nfg_string = open_spiel::GameToNFGString(*game);
158  std::shared_ptr<const Game> game2 = LoadNFGGame(nfg_string);
159 
160  if (game->NumPlayers() == 2) {
161  const auto* matrix_game = dynamic_cast<const MatrixGame*>(game.get());
162  const auto* matrix_game2 = dynamic_cast<const MatrixGame*>(game2.get());
163  SPIEL_CHECK_TRUE(matrix_game != nullptr);
164  SPIEL_CHECK_TRUE(matrix_game2 != nullptr);
165  SPIEL_CHECK_TRUE(matrix_game->ApproxEqual(*matrix_game2, 1e-10));
166  } else {
167  const auto* tensor_game = dynamic_cast<const TensorGame*>(game.get());
168  const auto* tensor_game2 = dynamic_cast<const TensorGame*>(game2.get());
169  SPIEL_CHECK_TRUE(tensor_game != nullptr);
170  SPIEL_CHECK_TRUE(tensor_game2 != nullptr);
171  SPIEL_CHECK_TRUE(tensor_game->ApproxEqual(*tensor_game2, 1e-10));
172  }
173  }
174 }
