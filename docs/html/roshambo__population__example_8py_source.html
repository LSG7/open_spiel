48 class BotAgent(rl_agent.AbstractAgent):
49  """Agent class that wraps a bot.
50 
51  Note, the environment must include the OpenSpiel state in its observations,
52  which means it must have been created with use_full_state=True.
53  """
54 
55  def __init__(self, num_actions, bot, name="bot_agent"):
56  assert num_actions > 0
57  self._bot_bot = bot
58  self._num_actions_num_actions = num_actions
59 
60  def restart(self):
61  self._bot_bot.restart()
62 
63  def step(self, time_step, is_evaluation=False):
64  # If it is the end of the episode, don't select an action.
65  if time_step.last():
66  return
67 
68  _, state = pyspiel.deserialize_game_and_state(
69  time_step.observations["serialized_state"])
70 
71  action = self._bot_bot.step(state)
72  probs = np.zeros(self._num_actions_num_actions)
73  probs[action] = 1.0
74 
75  return rl_agent.StepOutput(action=action, probs=probs)

78 def eval_agents(env, agents, num_players, num_episodes):
79  """Evaluate the agent."""
80  sum_episode_rewards = np.zeros(num_players)
81  for ep in range(num_episodes):
82  for agent in agents:
83  # Bots need to be restarted at the start of the episode.
84  if hasattr(agent, "restart"):
85  agent.restart()
86  time_step = env.reset()
87  episode_rewards = np.zeros(num_players)
88  while not time_step.last():
89  agents_output = [
90  agent.step(time_step, is_evaluation=True) for agent in agents
91  ]
92  action_list = [agent_output.action for agent_output in agents_output]
93  time_step = env.step(action_list)
94  episode_rewards += time_step.rewards
95  sum_episode_rewards += episode_rewards
96  print(f"Finished episode {ep}, "
97  + f"avg returns: {sum_episode_rewards / num_episodes}")
98 
99  return sum_episode_rewards / num_episodes

117 def analyze_bot_table(filename):
118  """Do some analysis on the payoff cross-table."""
119  print(f"Opening bot table file: {filename}")
120  bot_table_file = open(filename, "r")
121  table = np.zeros(shape=(pyspiel.ROSHAMBO_NUM_BOTS,
122  pyspiel.ROSHAMBO_NUM_BOTS), dtype=np.float64)
123  print("Parsing file...")
124  values = {}
125  bot_names_map = {}
126  for line in bot_table_file:
127  line = line.strip()
128  # ('driftbot', 'driftbot', -0.571)
129  myre = re.compile(r"\'(.*)\', \'(.*)\', (.*)\‍)")
130  match_obj = myre.search(line)
131  row_agent, col_agent, value = match_obj.groups()
132  values[f"{row_agent},{col_agent}"] = value
133  bot_names_map[row_agent] = True
134  bot_names_list = list(bot_names_map.keys())
135  bot_names_list.sort()
136  print(len(bot_names_list))
137  assert len(bot_names_list) == pyspiel.ROSHAMBO_NUM_BOTS
138  print(bot_names_list)
139  for i in range(pyspiel.ROSHAMBO_NUM_BOTS):
140  for j in range(pyspiel.ROSHAMBO_NUM_BOTS):
141  key = f"{bot_names_list[i]},{bot_names_list[j]}"
142  assert key in values
143  table[i][j] = float(values[key])
144  print("Population returns:")
145  pop_returns = np.zeros(pyspiel.ROSHAMBO_NUM_BOTS)
146  pop_aggregate = np.zeros(pyspiel.ROSHAMBO_NUM_BOTS)
147  for i in range(pyspiel.ROSHAMBO_NUM_BOTS):
148  pop_eval = 0
149  for j in range(pyspiel.ROSHAMBO_NUM_BOTS):
150  pop_eval += table[i][j]
151  pop_eval /= pyspiel.ROSHAMBO_NUM_BOTS
152  # print(f" {bot_names_list[i]}: {pop_eval}")
153  pop_returns[i] = pop_eval
154  pop_aggregate[i] += pop_eval
155  print(f" {pop_eval},")
156  print("Population exploitabilities: ")
157  pop_expls = np.zeros(pyspiel.ROSHAMBO_NUM_BOTS)
158  avg_pop_expl = 0
159  for i in range(pyspiel.ROSHAMBO_NUM_BOTS):
160  pop_expl = -float(pyspiel.ROSHAMBO_NUM_THROWS)
161  for j in range(pyspiel.ROSHAMBO_NUM_BOTS):
162  pop_expl = max(pop_expl, -table[i][j])
163  avg_pop_expl += pop_expl
164  pop_expls[i] = pop_expl
165  pop_aggregate[i] -= pop_expl
166  print(f" {pop_expl},")
167  avg_pop_expl /= pyspiel.ROSHAMBO_NUM_BOTS
168  print(f"Avg within-pop expl: {avg_pop_expl}")
169  print("Aggregate: ")
170  indices = np.argsort(pop_aggregate)
171  for i in range(pyspiel.ROSHAMBO_NUM_BOTS):
172  idx = indices[pyspiel.ROSHAMBO_NUM_BOTS - i - 1]
173  print(f" {i+1} & \\textsc{{{bot_names_list[idx]}}} & " +
174  f" ${pop_returns[idx]:0.3f}$ " +
175  f"& ${pop_expls[idx]:0.3f}$ & ${pop_aggregate[idx]:0.3f}$ \\\\")
176  print("Dominance:")
177  for i in range(pyspiel.ROSHAMBO_NUM_BOTS):
178  for j in range(pyspiel.ROSHAMBO_NUM_BOTS):
179  if np.all(np.greater(table[i], table[j])):
180  print(f"{bot_names_list[i]} dominates {bot_names_list[j]}")
