32int RandomSimulation(std::mt19937* rng, const Game& game, bool verbose) {
33 std::unique_ptr<State> state = game.NewInitialState();
34
35 if (verbose) {
36 std::cout << "Initial state:" << std::endl
37 << "State:" << std::endl
38 << state->ToString() << std::endl;
39 }
40
41 bool provides_info_state_tensor =
42 game.GetType().provides_information_state_tensor;
43 bool provides_observations_tensor =
44 game.GetType().provides_observation_tensor;
45 std::vector<float> obs;
46 if (provides_info_state_tensor) {
47 obs = std::vector<float>(game.InformationStateTensorSize());
48 } else if (provides_observations_tensor) {
49 obs = std::vector<float>(game.ObservationTensorSize());
50 }
51
52 int game_length = 0;
53 while (!state->IsTerminal()) {
54 if (provides_info_state_tensor && state->CurrentPlayer() >= 0) {
55 state->InformationStateTensor(state->CurrentPlayer(),
56 absl::MakeSpan(obs));
57 } else if (provides_observations_tensor && state->CurrentPlayer() >= 0) {
58 state->ObservationTensor(state->CurrentPlayer(), absl::MakeSpan(obs));
59 }
60 ++game_length;
61 if (state->IsChanceNode()) {
62 std::vector<std::pair<Action, double>> outcomes = state->ChanceOutcomes();
63 Action action;
64 if (game.GetType().chance_mode ==
65 GameType::ChanceMode::kSampledStochastic) {
66 action = outcomes.front().first;
67 } else {
68 // Explicit chance node; sample one according to underlying
69 // distribution.
70 action = SampleAction(outcomes, *rng).first;
71 }
72 if (verbose) {
73 std::cout << "Sampled outcome: "
74 << state->ActionToString(kChancePlayerId, action)
75 << std::endl;
76 }
77 state->ApplyAction(action);
78 } else if (state->CurrentPlayer() == kSimultaneousPlayerId) {
79 // Sample an action for each player
80 std::vector<Action> joint_action;
81 for (int p = 0; p < game.NumPlayers(); p++) {
82 std::vector<Action> actions;
83 actions = state->LegalActions(p);
84 Action action = 0;
85 if (!actions.empty()) {
86 std::uniform_int_distribution<int> dis(0, actions.size() - 1);
87 action = actions[dis(*rng)];
88 }
89 joint_action.push_back(action);
90 if (verbose) {
91 std::cout << "Player " << p
92 << " chose action:" << state->ActionToString(p, action)
93 << std::endl;
94 }
95 }
96 state->ApplyActions(joint_action);
97 } else {
98 // Sample an action uniformly.
99 std::vector<Action> actions = state->LegalActions();
100 std::uniform_int_distribution<int> dis(0, actions.size() - 1);
101 Action action = actions[dis(*rng)];
102 if (verbose) {
103 int p = state->CurrentPlayer();
104 std::cout << "Player " << p
105 << " chose action: " << state->ActionToString(p, action)
106 << std::endl;
107 }
108 state->ApplyAction(action);
109 }
110 if (verbose) {
111 std::cout << "State: " << std::endl << state->ToString() << std::endl;
112 std::cout << "Observation: " << obs << std::endl;
113 }
114 }
115 return game_length;
116}

118// Perform num_sims random simulations of the specified game, and output the
119// time taken.
120void RandomSimBenchmark(const std::string& game_def, int num_sims,
121 bool verbose) {
122 std::mt19937 rng;
123 std::cout << absl::StrFormat("Benchmark: game: %s, num_sims: %d. ", game_def,
124 num_sims);
125
126 auto game = LoadGame(game_def);
127
128 absl::Time start = absl::Now();
129 int num_moves = 0;
130 for (int sim = 0; sim < num_sims; ++sim) {
131 num_moves += RandomSimulation(&rng, *game, verbose);
132 }
133 absl::Time end = absl::Now();
134 double seconds = absl::ToDoubleSeconds(end - start);
135
136 std::cout << absl::StrFormat(
137 "Finished %d moves in %.1f ms: %.1f sim/s, %.1f moves/s",
138 num_moves, seconds * 1000, num_sims / seconds,
139 num_moves / seconds)
140 << std::endl;
141}
