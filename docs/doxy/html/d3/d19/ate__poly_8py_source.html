56 def init_vars(self, num_strats, num_players):
57 """Initialize solver parameters."""
58 self.num_strats = num_strats
59 self.num_players = num_players
60 if len(num_strats) != num_players:
61 raise ValueError('Must specify num strategies for each player')
62 init_dist = []
63 for num_strats_i in num_strats:
64 if self.rnd_init:
65 init_dist_i = self.random.rand(num_strats_i)
66 else:
67 init_dist_i = np.ones(num_strats_i)
68 init_dist_i /= init_dist_i.sum()
69 init_dist.append(init_dist_i)
70 init_y = self.init_polymatrix(num_strats, num_players)
71 return (init_dist, init_y)

73 def init_polymatrix(self, num_strats, num_players):
74 """Initialize all pairwise bimatrix games to zero and return as dict."""
75 init_pm = dict()
76 for i, j in itertools.combinations(range(num_players), 2):
77 init_pm[(i, j)] = np.zeros((2, num_strats[i], num_strats[j])) # i < j
78 return init_pm

80 def record_aux_errors(self, grads):
81 """Record errors for the auxiliary variables."""
82 grad_y = grads[1]
83 # call ravel in case use y to track entire payoff matrices in future
84 grad_y_flat = np.concatenate([np.ravel(g) for g in grad_y.values()])
85 self.aux_errors.append([np.linalg.norm(grad_y_flat)])

111 def euc_descent_step(self, params, grads, t):
112 """Projected gradient descent on exploitability using Euclidean projection.
113
114 Args:
115 params: tuple of variables to be updated (dist, y)
116 grads: tuple of variable gradients (grad_dist, grad_y)
117 t: int, solver iteration (unused)
118 Returns:
119 new_params: tuple of update params (new_dist, new_y)
120 """
121 lr_dist, lr_y = self.lrs
122 new_dist = []
123 for dist_i, dist_grad_i in zip(params[0], grads[0]):
124 new_dist_i = dist_i - lr_dist * dist_grad_i
125 new_dist_i = simplex.euclidean_projection_onto_simplex(new_dist_i)
126 new_dist.append(new_dist_i)
127 lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
128 new_y = dict()
129 for i, j in params[1]:
130 y_ij = params[1][(i, j)]
131 y_grad_ij = grads[1][(i, j)]
132 new_y_ij = y_ij - lr_y * y_grad_ij
133 new_y_ij = np.clip(new_y_ij, 0., np.inf)
134 new_y[(i, j)] = new_y_ij
135 return (new_dist, new_y)

137 def mirror_descent_step(self, params, grads, t):
138 """Entropic mirror descent on exploitability.
139
140 Args:
141 params: tuple of variables to be updated (dist, y)
142 grads: tuple of variable gradients (grad_dist, grad_y)
143 t: int, solver iteration (unused)
144 Returns:
145 new_params: tuple of update params (new_dist, new_y)
146 """
147 lr_dist, lr_y = self.lrs
148 new_dist = []
149 for dist_i, dist_grad_i in zip(params[0], grads[0]):
150 new_dist_i = np.log(np.clip(dist_i, 0., np.inf)) - lr_dist * dist_grad_i
151 new_dist_i = special.softmax(new_dist_i)
152 new_dist.append(new_dist_i)
153 lr_y = np.clip(1 / float(t + 1), lr_y, np.inf)
154 new_y = dict()
155 for i, j in params[1]:
156 y_ij = params[1][(i, j)]
157 y_grad_ij = grads[1][(i, j)]
158 new_y_ij = y_ij - lr_y * y_grad_ij
159 new_y_ij = np.clip(new_y_ij, 0., np.inf)
160 new_y[(i, j)] = new_y_ij
161 return (new_dist, new_y)

163 def gradients(self, dist, y, payoff_matrices, p=1, proj_grad=True):
164 """Computes exploitablity gradient and aux variable gradients.
165
166 Args:
167 dist: list of 1-d np.arrays, current estimate of nash distribution
168 y: dict of 2-d np.arrays, current est. of players (i, j)'s payoff matrix
169 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
170 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
171 are sorted and arrays should be indexed in the same order
172 p: float in [0, 1], Tsallis entropy-regularization --> 0 as p --> 0
173 proj_grad: bool, if True, projects dist gradient onto simplex
174 Returns:
175 gradient of exploitability w.r.t. (dist, y) as tuple
176 unregularized exploitability (stochastic estimate)
177 tsallis regularized exploitability (stochastic estimate)
178 """
179 # first compute policy gradients and player effects (fx)
180 policy_gradient = []
181 other_player_fx = []
182 grad_y = self.init_polymatrix(self.num_strats, self.num_players)
183 unreg_exp = []
184 reg_exp = []
185 for i in range(self.num_players):
186
187 nabla_i = np.zeros_like(dist[i])
188 for j in range(self.num_players):
189 if j == i:
190 continue
191 if i < j:
192 hess_i_ij = payoff_matrices[(i, j)][0]
193 hess_i_ij_from_y = y[(i, j)][0]
194 grad_y[(i, j)][0] = hess_i_ij_from_y - hess_i_ij
195 else:
196 hess_i_ij = payoff_matrices[(j, i)][1].T
197 hess_i_ij_from_y = y[(j, i)][1].T
198 grad_y[(j, i)][1] = hess_i_ij_from_y.T - hess_i_ij.T
199
200 nabla_ij = hess_i_ij_from_y.dot(dist[j])
201 nabla_i += nabla_ij / float(self.num_players - 1)
202
203 if p > 0:
204 power = 1. / float(p)
205 s_i = np.linalg.norm(nabla_i, ord=power)
206 if s_i == 0:
207 br_i = misc.uniform_dist(nabla_i)
208 else:
209 br_i = (nabla_i / s_i)**power
210 else:
211 power = np.inf
212 s_i = np.linalg.norm(nabla_i, ord=power)
213 br_i = np.zeros_like(dist[i])
214 maxima_i = (nabla_i == s_i)
215 br_i[maxima_i] = 1. / maxima_i.sum()
216
217 policy_gradient_i = nabla_i - s_i * dist[i]**p
218 policy_gradient.append(policy_gradient_i)
219
220 unreg_exp.append(np.max(nabla_i) - nabla_i.dot(dist[i]))
221
222 br_i_inv_sparse = 1 - np.sum(br_i**(p + 1))
223 dist_i_inv_sparse = 1 - np.sum(dist[i]**(p + 1))
224 entr_br_i = s_i / (p + 1) * br_i_inv_sparse
225 entr_dist_i = s_i / (p + 1) * dist_i_inv_sparse
226
227 reg_exp.append(nabla_i.dot(br_i - dist[i]) + entr_br_i - entr_dist_i)
228
229 entr_br_vec_i = br_i_inv_sparse * br_i**(1 - p)
230 entr_dist_vec_i = dist_i_inv_sparse * dist[i]**(1 - p)
231 other_player_fx_i = (br_i - dist[i]) + 1 / (p + 1) * (
232 entr_br_vec_i - entr_dist_vec_i)
233 other_player_fx.append(other_player_fx_i)
234
235 # then construct exploitability gradient
236 grad_dist = []
237 for i in range(self.num_players):
238
239 grad_dist_i = -policy_gradient[i]
240 for j in range(self.num_players):
241 if j == i:
242 continue
243 if i < j:
244 hess_j_ij_from_y = y[(i, j)][1]
245 else:
246 hess_j_ij_from_y = y[(j, i)][0].T
247
248 grad_dist_i += hess_j_ij_from_y.dot(other_player_fx[j])
249
250 if proj_grad:
251 grad_dist_i = simplex.project_grad(grad_dist_i)
252
253 grad_dist.append(grad_dist_i)
254
255 return (grad_dist, grad_y), np.mean(unreg_exp), np.mean(reg_exp)
