21class BasicStats(object):
22 """A set of statistics about a single value series."""
23 __slots__ = ("_num", "_min", "_max", "_sum", "_sum_sq")
24
25 def __init__(self):
26 self.reset()
27
28 def reset(self):
29 self._num = 0
30 self._min = float("inf")
31 self._max = float("-inf")
32 self._sum = 0
33 self._sum_sq = 0
34
35 def add(self, val: float):
36 self._num += 1
37 if self._min > val:
38 self._min = val
39 if self._max < val:
40 self._max = val
41 self._sum += val
42 self._sum_sq += val**2
43
44 @property
45 def num(self):
46 return self._num
47
48 @property
49 def min(self):
50 return 0 if self._num == 0 else self._min
51
52 @property
53 def max(self):
54 return 0 if self._num == 0 else self._max
55
56 @property
57 def avg(self):
58 return 0 if self._num == 0 else self._sum / self._num
59
60 @property
61 def std_dev(self):
62 """Standard deviation."""
63 if self._num == 0:
64 return 0
65 return math.sqrt(
66 max(0, self._sum_sq / self._num - (self._sum / self._num)**2))

69 # pylint: disable=protected-access

71 self._min = min(self._min, other._min)
72 self._max = max(self._max, other._max)
73 self._sum += other._sum
74 self._sum_sq += other._sum_sq

94class HistogramNumbered:
95 """Track a histogram of occurences for `count` buckets.
96
97 You need to decide how to map your data into the buckets. Mainly useful for
98 scalar values.
99 """
100
101 def __init__(self, num_buckets: int):
102 self._counts = [0] * num_buckets
103
104 def reset(self):
105 self._counts = [0] * len(self._counts)
106
107 def add(self, bucket_id: int):
108 self._counts[bucket_id] += 1

110 @property
