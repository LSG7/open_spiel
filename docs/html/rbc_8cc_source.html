58 chess::ObservationTable ComputeObservationTable(const chess::ChessBoard& board,
59  chess::Color color,
60  int sense_location,
61  int sense_size) {
62  const int board_size = board.BoardSize();
63  chess::ObservationTable observability_table{false};
64 
65  // Player pieces.
66  for (int8_t y = 0; y < board_size; ++y) {
67  for (int8_t x = 0; x < board_size; ++x) {
68  chess::Square sq{x, y};
69  const auto& piece = board.at(sq);
70  if (piece.color == color) {
71  size_t index = chess::SquareToIndex(sq, board_size);
72  observability_table[index] = true;
73  }
74  }
75  }
76 
77  // No sense window specified.
78  if (sense_location < 0) return observability_table;
79 
80  // All pieces under the sense window.
81  int inner_size = board_size - sense_size + 1;
82  chess::Square sense_sq = chess::IndexToSquare(sense_location, inner_size);
83  SPIEL_DCHECK_LE(sense_sq.x + sense_size, board_size);
84  SPIEL_DCHECK_LE(sense_sq.y + sense_size, board_size);
85  for (int8_t x = sense_sq.x; x < sense_sq.x + sense_size; ++x) {
86  for (int8_t y = sense_sq.y; y < sense_sq.y + sense_size; ++y) {
87  const chess::Square sq{x, y};
88  size_t index = chess::SquareToIndex(sq, board_size);
89  observability_table[index] = true;
90  }
91  }
92 
93  return observability_table;
94 }

115  Allocator* allocator) const override {

162  private:
163  // Encode the private observations as a FEN-like string.
164  std::string StringPrivateInfoObservation(const RbcState& state,
165  const RbcGame& game,
166  int player) const {
167  chess::Color color = chess::PlayerToColor(player);
168  const int sense_location =
169  (state.phase_ == MovePhase::kMoving && state.CurrentPlayer() == player)
170  ? state.sense_location_[player]
171  // Make sure that sense from last round does not
172  // reveal a new hidden move: allow players to
173  // perceive only results of the last sensing.
174  : kSenseLocationNonSpecified;
175  const chess::ChessBoard& board = state.Board();
176  chess::ObservationTable observability_table = ComputeObservationTable(
177  board, color, sense_location, game.sense_size());
178  const int board_size = game.board_size();
179  std::string str = "";
180 
181  // 1. Encode the board based on what can be observed for the player.
182  for (int8_t rank = board_size - 1; rank >= 0; --rank) {
183  int num_unknown = 0;
184  for (int8_t file = 0; file < board_size; ++file) {
185  const size_t index =
186  chess::SquareToIndex(chess::Square{file, rank}, board_size);
187  if (!observability_table[index]) {
188  num_unknown++;
189  } else {
190  if (num_unknown > 0) {
191  absl::StrAppend(&str, num_unknown);
192  num_unknown = 0;
193  }
194  const chess::Piece& piece = board.at(chess::Square{file, rank});
195  absl::StrAppend(&str, piece.ToString());
196  }
197  }
198  if (num_unknown > 0) {
199  absl::StrAppend(&str, num_unknown);
200  }
201  if (rank > 0) {
202  absl::StrAppend(&str, "/");
203  }
204  }
205 
206  // 2. Castling rights of the player.
207  absl::StrAppend(&str, " ");
208  std::string castling_rights;
209  if (board.CastlingRight(color, chess::CastlingDirection::kRight)) {
210  absl::StrAppend(&castling_rights, "K");
211  }
212  if (board.CastlingRight(color, chess::CastlingDirection::kLeft)) {
213  castling_rights.push_back('Q');
214  }
215  absl::StrAppend(&str, castling_rights.empty() ? "-" : castling_rights);
216 
217  // 3. Phase.
218  absl::StrAppend(&str, " ", state.phase_ == MovePhase::kSensing ? "s" : "m");
219 
220  // 4. Capture (but no information about what was captured).
221  absl::StrAppend(&str, " ", state.move_captured_ ? "c" : "-");
222 
223  // 5. Side to play.
224  absl::StrAppend(&str, " ",
225  board.ToPlay() == chess::Color::kWhite ? "w" : "b");
226 
227  // 6. Illegal move.
228  const bool can_show = state.CurrentPlayer() == player;
229  absl::StrAppend(&str, " ",
230  can_show && state.illegal_move_attempted_ ? "i" : "-");
231  return str;
232  }

234  void WritePieces(chess::Color color, chess::PieceType piece_type,
235  const chess::ChessBoard& board, int sense_location,
236  int sense_size, const std::string& prefix,
237  Allocator* allocator) const {
238  const std::string type_string = chess::PieceTypeToString(
239  piece_type, /*uppercase=*/color == chess::Color::kWhite);
240  const int board_size = board.BoardSize();
241  int inner_size = board_size - sense_size + 1;
242  chess::Square sense_square =
243  chess::IndexToSquare(sense_location, inner_size);
244  auto out = allocator->Get(prefix + "_" + type_string + "_pieces",
245  {board_size, board_size});
246 
247  if (sense_location < 0) return; // No sense window specified.
248  SPIEL_DCHECK_LE(sense_square.x + sense_size, board_size);
249  SPIEL_DCHECK_LE(sense_square.y + sense_size, board_size);
250  for (int8_t x = sense_square.x; x < sense_square.x + sense_size; ++x) {
251  for (int8_t y = sense_square.y; y < sense_square.y + sense_size; ++y) {
252  const chess::Square square{x, y};
253  const chess::Piece& piece_on_board = board.at(square);
254  const bool write_square =
255  piece_on_board.color == color && piece_on_board.type == piece_type;
256  out.at(x, y) = write_square ? 1.0f : 0.0f;
257  }
258  }
259  }

276  void WritePrivateInfoTensor(const RbcState& state, int player,
277  const std::string& prefix,
278  Allocator* allocator) const {
279  chess::Color color = chess::PlayerToColor(player);
280 
281  // Illegal move (pawn attack or pawn forward-move or castle through
282  // opponent pieces).
283  const bool can_show = state.CurrentPlayer() == player;
284  WriteBinary(can_show && state.illegal_move_attempted_, "illegal_move",
285  allocator);
286 
287  // Piece configuration.
288  for (const chess::PieceType& piece_type : chess::kPieceTypes) {
289  WritePieces(static_cast<chess::Color>(player), piece_type, state.Board(),
290  0, state.game()->board_size(), prefix, allocator);
291  }
292 
293  // Castling rights.
294  WriteBinary(
295  state.Board().CastlingRight(color, chess::CastlingDirection::kLeft),
296  prefix + "_left_castling", allocator);
297  WriteBinary(
298  state.Board().CastlingRight(color, chess::CastlingDirection::kRight),
299  prefix + "_right_castling", allocator);
300 
301  // Last sensing
302  for (const chess::PieceType& piece_type : chess::kPieceTypes) {
303  int sense_location = (state.phase_ == MovePhase::kMoving &&
304  state.CurrentPlayer() == player)
305  ? state.sense_location_[player]
306  // Make sure that sense from last round does not
307  // reveal a new hidden move: allow players to
308  // perceive only results of the last sensing.
309  : kSenseLocationNonSpecified;
310  WritePieces(static_cast<chess::Color>(1 - player), piece_type,
311  state.Board(), sense_location, state.game()->sense_size(),
312  prefix + "_sense", allocator);
313  }
314  }

316  void WritePublicInfoTensor(const RbcState& state,
317  Allocator* allocator) const {
318  // Compute number of pieces of each player.
319  const int board_size = state.game()->board_size();
320  std::array<int, 2> num_pieces = {0, 0};
321  for (int x = 0; x < board_size; ++x) {
322  for (int y = 0; y < board_size; ++y) {
323  for (int pl = 0; pl < 2; ++pl) {
324  num_pieces[pl] +=
325  state.Board().IsFriendly(
326  chess::Square{static_cast<int8_t>(x),
327  static_cast<int8_t>(y)},
328  static_cast<chess::Color>(pl));
329  }
330  }
331  }
332 
333  WriteScalar(num_pieces[0], 0, board_size * 2, "pieces_black", allocator);
334  WriteScalar(num_pieces[1], 0, board_size * 2, "pieces_white", allocator);
335  WriteBinary(state.phase_ == MovePhase::kSensing, "phase", allocator);
336  WriteBinary(state.move_captured_, "capture", allocator);
337  WriteBinary(state.CurrentPlayer(), "side_to_play", allocator);
338  }

356 void RbcState::DoApplyAction(Action action) {
357  // Reset common flags.
358  illegal_move_attempted_ = false;
359  move_captured_ = false;
360 
361  if (phase_ == MovePhase::kSensing) {
362  sense_location_[CurrentPlayer()] = action;
363  phase_ = MovePhase::kMoving;
364  } else {
365  SPIEL_CHECK_TRUE(phase_ == MovePhase::kMoving);
366  chess::Move move = ActionToMove(action, Board());
367 
368  // Handle special cases for RBC.
369 
370  if (move == chess::kPassMove) {
371  // The RBC's pass move is handled via ChessBoard flag allow_pass_move.
372  // Nothing here. Values set above.
373  } else if (Board().IsBreachingMove(move)) {
374  SPIEL_DCHECK_FALSE(Board().IsMoveLegal(move));
375  Board().BreachingMoveToCaptureMove(&move);
376  // Transformed move must be legal.
377  SPIEL_DCHECK_TRUE(Board().IsMoveLegal(move));
378  // And it must be a capture, since we breached unseen opponent pieces.
379  SPIEL_DCHECK_NE(Board().at(move.from).color, Board().at(move.to).color);
380  move_captured_ = true;
381  } else if (!Board().IsMoveLegal(move)) {
382  // Illegal move was chosen.
383  illegal_move_attempted_ = true;
384 
385  // Check why the move was illegal:
386  // if it is pawn two-squares-forward move,
387  // and there is an enemy piece blocking it, the attempt to move only one
388  // square forward (if that would be a legal move).
389  // if it is pawn move to last rank, change to pawn move & queen promotion
390  // (if that would be a legal move)
391  if (move.piece.type == chess::PieceType::kPawn &&
392  abs(move.from.y - move.to.y) == 2) {
393  const int dy = move.to.y - move.from.y > 0 ? 1 : -1;
394  chess::Move one_forward_move = move;
395  one_forward_move.to.y -= dy;
396  move = Board().IsMoveLegal(one_forward_move) ? one_forward_move
397  : chess::kPassMove;
398  } else if (move.piece.type == chess::PieceType::kPawn &&
399  Board().IsPawnPromotionRank(move.to)) {
400  chess::Move promote_move = move;
401  promote_move.promotion_type = chess::PieceType::kQueen;
402  move = Board().IsMoveLegal(promote_move) ? promote_move
403  : chess::kPassMove;
404  } else {
405  // Treat the illegal move as a pass.
406  move = chess::kPassMove;
407  }
408  } else {
409  // All other moves
410  SPIEL_DCHECK_EQ(Board().at(move.from).color, Board().ToPlay());
411  move_captured_ =
412  Board().at(move.to).color == chess::OppColor(Board().ToPlay());
413  }
414 
415  SPIEL_DCHECK_TRUE(Board().IsMoveLegal(move));
416  moves_history_.push_back(move);
417  Board().ApplyMove(move);
418 
419  ++repetitions_[current_board_.HashValue()];
420  phase_ = MovePhase::kSensing;
421  }
422  cached_legal_actions_.reset();
423 }

425 void RbcState::MaybeGenerateLegalActions() const {
426  if (!cached_legal_actions_) {
427  cached_legal_actions_ = std::vector<Action>();
428 
429  if (phase_ == MovePhase::kSensing) {
430  int num_possible_sense_locations =
431  game()->inner_size() * game()->inner_size();
432  cached_legal_actions_->resize(num_possible_sense_locations);
433  absl::c_iota(*cached_legal_actions_, 0);
434  } else {
435  SPIEL_CHECK_TRUE(phase_ == MovePhase::kMoving);
436  Board().GeneratePseudoLegalMoves(
437  [this](const chess::Move& move) -> bool {
438  cached_legal_actions_->push_back(MoveToAction(move, BoardSize()));
439  return true;
440  },
441  Board().ToPlay(), chess::PseudoLegalMoveSettings::kBreachEnemyPieces);
442  absl::c_sort(*cached_legal_actions_);
443  }
444  }
445 }

518 absl::optional<std::vector<double>> RbcState::MaybeFinalReturns() const {
519  const auto to_play_color = Board().ToPlay();
520  const auto opp_color = chess::OppColor(to_play_color);
521 
522  const auto to_play_king =
523  chess::Piece{to_play_color, chess::PieceType::kKing};
524  const auto opp_king = chess::Piece{opp_color, chess::PieceType::kKing};
525 
526  if (Board().find(to_play_king) == chess::kInvalidSquare) {
527  std::vector<double> returns(NumPlayers());
528  returns[chess::ColorToPlayer(to_play_color)] = LossUtility();
529  returns[chess::ColorToPlayer(opp_color)] = WinUtility();
530  return returns;
531 
532  } else if (Board().find(opp_king) == chess::kInvalidSquare) {
533  std::vector<double> returns(NumPlayers());
534  returns[chess::ColorToPlayer(to_play_color)] = WinUtility();
535  returns[chess::ColorToPlayer(opp_color)] = LossUtility();
536  return returns;
537  }
538 
539  if (!Board().HasSufficientMaterial()) {
540  return std::vector<double>{DrawUtility(), DrawUtility()};
541  }
542 
543  if (IsRepetitionDraw()) {
544  return std::vector<double>{DrawUtility(), DrawUtility()};
545  }
546  // Compute and cache the legal actions.
547  MaybeGenerateLegalActions();
548  SPIEL_CHECK_TRUE(cached_legal_actions_);
549  const bool have_legal_moves = !cached_legal_actions_->empty();
550 
551  // If we don't have legal moves we are stalemated
552  if (!have_legal_moves) {
553  return std::vector<double>{DrawUtility(), DrawUtility()};
554  }
555 
556  if (Board().IrreversibleMoveCounter() >= kNumReversibleMovesToDraw) {
557  // This is theoretically a draw that needs to be claimed, but we implement
558  // it as a forced draw for now.
559  return std::vector<double>{DrawUtility(), DrawUtility()};
560  }
561 
562  return absl::nullopt;
563 }
