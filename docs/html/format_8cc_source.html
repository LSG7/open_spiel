51 // Splits a Time into seconds and femtoseconds, which can be used with CCTZ.
52 // Requires that 't' is finite. See duration.cc for details about rep_hi and
53 // rep_lo.
54 cctz_parts Split(absl::Time t) {
55  const auto d = time_internal::ToUnixDuration(t);
56  const int64_t rep_hi = time_internal::GetRepHi(d);
57  const int64_t rep_lo = time_internal::GetRepLo(d);
58  const auto sec = unix_epoch() + cctz::seconds(rep_hi);
59  const auto fem = cctz::detail::femtoseconds(rep_lo * (1000 * 1000 / 4));
60  return {sec, fem};
61 }
62 
63 // Joins the given seconds and femtoseconds into a Time. See duration.cc for
64 // details about rep_hi and rep_lo.
65 absl::Time Join(const cctz_parts& parts) {
66  const int64_t rep_hi = (parts.sec - unix_epoch()).count();
67  const uint32_t rep_lo =
68  static_cast<uint32_t>(parts.fem.count() / (1000 * 1000 / 4));
69  const auto d = time_internal::MakeDuration(rep_hi, rep_lo);
70  return time_internal::FromUnixDuration(d);
71 }

99 bool ParseTime(absl::string_view format, absl::string_view input,

108  // Portable toolchains means we don't get nice constexpr here.
109  struct Literal {
110  const char* name;
111  size_t size;
112  absl::Time value;
113  };
114  static Literal literals[] = {
115  {kInfiniteFutureStr, strlen(kInfiniteFutureStr), InfiniteFuture()},
116  {kInfinitePastStr, strlen(kInfinitePastStr), InfinitePast()},
117  };
118  strip_leading_space(&input);
119  for (const auto& lit : literals) {
120  if (absl::StartsWith(input, absl::string_view(lit.name, lit.size))) {
121  absl::string_view tail = input;
122  tail.remove_prefix(lit.size);
123  strip_leading_space(&tail);
124  if (tail.empty()) {
125  *time = lit.value;
126  return true;
127  }
128  }
129  }
130 
131  std::string error;
132  cctz_parts parts;
133  const bool b =
134  cctz::detail::parse(std::string(format), std::string(input),
135  cctz::time_zone(tz), &parts.sec, &parts.fem, &error);
136  if (b) {
137  *time = Join(parts);
138  } else if (err != nullptr) {
139  *err = error;
140  }
141  return b;
142 }
