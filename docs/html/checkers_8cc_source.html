188 CheckersState::CheckersState(std::shared_ptr<const Game> game, int rows,
189  int columns)
190  : State(game), rows_(rows), columns_(columns) {
191  SPIEL_CHECK_GE(rows_, 1);
192  SPIEL_CHECK_GE(columns_, 1);
193  SPIEL_CHECK_LE(rows_, 99); // Only supports 1 and 2 digit row numbers.
194  SPIEL_CHECK_LE(columns_, 26); // Only 26 letters to represent columns.
195 
196  moves_without_capture_ = 0;
197  board_ = std::vector<CellState>(rows_ * columns_, CellState::kEmpty);
198  turn_history_info_ = {};
199 
200  for (int row = rows_ - 1; row >= 0; row--) {
201  for (int column = 0; column < columns_; column++) {
202  if ((row + column) % 2 == 1) {
203  if (row >= 0 && row < kNumRowsWithPieces) {
204  SetBoard(row, column, CellState::kBlack);
205  } else if (row >= (rows_ - kNumRowsWithPieces)) {
206  SetBoard(row, column, CellState::kWhite);
207  }
208  }
209  }
210  }
211 }

223 void CheckersState::SetCustomBoard(const std::string board_string) {
224  SPIEL_CHECK_EQ(rows_ * columns_, board_string.length() - 1);
225  current_player_ = board_string[0] - '0';
226  SPIEL_CHECK_GE(current_player_, 0);
227  SPIEL_CHECK_LE(current_player_, 1);
228  // Create the board from the board string. The characters 'o', '8' are White
229  // (first player) & '+', '*' are Black (second player), and the character '.'
230  // is an Empty cell. Population goes from top left to bottom right.
231  for (int row = 0; row < rows_; row++) {
232  for (int column = 0; column < columns_; column++) {
233  char state_character = board_string[1 + row * columns_ + column];
234  CellState state = StringToState(state_character);
235  SetBoard(row, column, state);
236  }
237  }
238 }

253 void CheckersState::DoApplyAction(Action action) {
254  CheckersAction checkers_action = SpielActionToCheckersAction(action);
255  SPIEL_CHECK_TRUE(InBounds(checkers_action.row, checkers_action.column));
256 
257  int end_row, end_column;
258  multiple_jump_piece_ = kNoMultipleJumpsPossible;
259  moves_without_capture_++;
260 
261  switch (checkers_action.move_type) {
262  case MoveType::kNormal:
263  end_row = checkers_action.row + kDirRowOffsets[checkers_action.direction];
264  end_column =
265  checkers_action.column + kDirColumnOffsets[checkers_action.direction];
266  SPIEL_CHECK_TRUE(InBounds(end_row, end_column));
267  SPIEL_CHECK_EQ(BoardAt(end_row, end_column), CellState::kEmpty);
268  turn_history_info_.push_back(TurnHistoryInfo(
269  action, current_player_, PieceType::kMan,
270  StateToPiece(BoardAt(checkers_action.row, checkers_action.column))));
271  SetBoard(
272  end_row, end_column,
273  CrownStateIfLastRowReached(
274  end_row, BoardAt(checkers_action.row, checkers_action.column)));
275  SetBoard(checkers_action.row, checkers_action.column, CellState::kEmpty);
276  break;
277  case MoveType::kCapture:
278  end_row =
279  checkers_action.row + kDirRowOffsets[checkers_action.direction] * 2;
280  end_column = checkers_action.column +
281  kDirColumnOffsets[checkers_action.direction] * 2;
282  SPIEL_CHECK_TRUE(InBounds(end_row, end_column));
283  SPIEL_CHECK_EQ(BoardAt(end_row, end_column), CellState::kEmpty);
284  PieceType captured_piece =
285  StateToPiece(BoardAt((checkers_action.row + end_row) / 2,
286  (checkers_action.column + end_column) / 2));
287  turn_history_info_.push_back(TurnHistoryInfo(
288  action, current_player_, captured_piece,
289  StateToPiece(BoardAt(checkers_action.row, checkers_action.column))));
290  SetBoard((checkers_action.row + end_row) / 2,
291  (checkers_action.column + end_column) / 2, CellState::kEmpty);
292  CellState end_state = CrownStateIfLastRowReached(
293  end_row, BoardAt(checkers_action.row, checkers_action.column));
294  SetBoard(end_row, end_column, end_state);
295  bool piece_crowned =
296  BoardAt(checkers_action.row, checkers_action.column) != end_state;
297  SetBoard(checkers_action.row, checkers_action.column, CellState::kEmpty);
298  moves_without_capture_ = 0;
299 
300  // Check if multiple jump is possible for the piece that made the
301  // last capture. If that is the case, then the current player gets
302  // to move again with LegalActions restricted to multiple jump moves
303  // for this piece.
304  if (!piece_crowned) {
305  std::vector<Action> moves = LegalActions();
306  for (Action action : moves) {
307  CheckersAction move = SpielActionToCheckersAction(action);
308  if (move.row == end_row && move.column == end_column &&
309  move.move_type == MoveType::kCapture) {
310  multiple_jump_piece_ = end_row * rows_ + end_column;
311  break;
312  }
313  }
314  }
315  break;
316  }
317 
318  if (multiple_jump_piece_ == kNoMultipleJumpsPossible) {
319  current_player_ = 1 - current_player_;
320  }
321 
322  if (LegalActions().empty()) {
323  outcome_ = 1 - current_player_;
324  }
325 }

340

344 std::vector<Action> CheckersState::LegalActions() const {
345  if (moves_without_capture_ >= kMaxMovesWithoutCapture) {
346  return {};
347  }
348  std::vector<Action> move_list, capture_move_list;
349  CellState current_player_state = PlayerToState(current_player_);
350  CellState current_player_crowned = CrownState(current_player_state);
351 
352  for (int row = 0; row < rows_; row++) {
353  for (int column = 0; column < columns_; column++) {
354  if (BoardAt(row, column) == current_player_state ||
355  BoardAt(row, column) == current_player_crowned) {
356  for (int direction = 0; direction < kNumDirections; direction++) {
357  // Only crowned pieces can move in all 4 directions.
358  if (BoardAt(row, column) == current_player_state &&
359  ((current_player_ == 0 && direction > 1) ||
360  (current_player_ == 1 && direction < 2))) {
361  continue;
362  }
363  int adjacent_row = row + kDirRowOffsets[direction];
364  int adjacent_column = column + kDirColumnOffsets[direction];
365 
366  if (InBounds(adjacent_row, adjacent_column)) {
367  CellState adjacent_state = BoardAt(adjacent_row, adjacent_column);
368  CellState opponent_state = OpponentState(current_player_state);
369  CellState opponent_state_crowned = CrownState(opponent_state);
370 
371  if (adjacent_state == CellState::kEmpty) {
372  CheckersAction move =
373  CheckersAction(row, column, direction, MoveType::kNormal);
374  move_list.push_back(CheckersActionToSpielAction(move));
375  } else if (adjacent_state == opponent_state ||
376  adjacent_state == opponent_state_crowned) {
377  int jumping_row = adjacent_row + kDirRowOffsets[direction];
378  int jumping_column =
379  adjacent_column + kDirColumnOffsets[direction];
380  if (InBounds(jumping_row, jumping_column) &&
381  BoardAt(jumping_row, jumping_column) == CellState::kEmpty) {
382  CheckersAction move =
383  CheckersAction(row, column, direction, MoveType::kCapture);
384  capture_move_list.push_back(CheckersActionToSpielAction(move));
385  }
386  }
387  }
388  }
389  }
390  }
391  }
392 
393  // If capture moves are possible, it's mandatory to play them.
394  if (!capture_move_list.empty()) {
395  if (multiple_jump_piece_ != kNoMultipleJumpsPossible) {
396  int multiple_jump_piece_row = multiple_jump_piece_ / rows_;
397  int multiple_jump_piece_column = multiple_jump_piece_ % rows_;
398  std::vector<Action> multiple_move_list;
399  for (Action action : capture_move_list) {
400  CheckersAction move = SpielActionToCheckersAction(action);
401  if (move.row == multiple_jump_piece_row &&
402  move.column == multiple_jump_piece_column) {
403  multiple_move_list.push_back(action);
404  }
405  }
406  SPIEL_CHECK_GT(multiple_move_list.size(), 0);
407  return multiple_move_list;
408  }
409  return capture_move_list;
410  }
411  return move_list;
412 }

503 void CheckersState::ObservationTensor(Player player,
504  absl::Span<float> values) const {
505  SPIEL_CHECK_GE(player, 0);
506  SPIEL_CHECK_LT(player, num_players_);
507 
508  TensorView<3> view(values, {kCellStates, rows_, columns_}, true);
509 
510  // Observation Tensor Representation:
511  // Plane 0: 1's where the current player's pieces are, 0's elsewhere.
512  // Plane 1: 1's where the oppponent's pieces are, 0's elsewhere.
513  // Plane 2: 1's where the current player's crowned pieces are, 0's elsewhere.
514  // Plane 3: 1's where the oppponent's crowned pieces are, 0's elsewhere.
515  // Plane 4: 1's where the empty cells are, 0's elsewhere.
516  for (int row = 0; row < rows_; row++) {
517  for (int column = 0; column < columns_; column++) {
518  int plane = ObservationPlane(BoardAt(row, column), player);
519  view[{plane, row, column}] = 1.0;
520  }
521  }
522 }

529 void CheckersState::UndoAction(Player player, Action action) {
530  CheckersAction move = SpielActionToCheckersAction(action);
531  const TurnHistoryInfo& thi = turn_history_info_.back();
532  SPIEL_CHECK_EQ(thi.player, player);
533  SPIEL_CHECK_EQ(thi.action, action);
534  current_player_ = player;
535  outcome_ = kInvalidPlayer;
536  move_number_--;
537 
538  int end_row, end_column;
539  CellState player_piece =
540  GetPieceStateFromTurnHistory(player, thi.player_piece_type);
541 
542  switch (move.move_type) {
543  case MoveType::kNormal:
544  end_row = move.row + kDirRowOffsets[move.direction];
545  end_column = move.column + kDirColumnOffsets[move.direction];
546  SetBoard(move.row, move.column, player_piece);
547  SetBoard(end_row, end_column, CellState::kEmpty);
548  break;
549  case MoveType::kCapture:
550  end_row = move.row + kDirRowOffsets[move.direction] * 2;
551  end_column = move.column + kDirColumnOffsets[move.direction] * 2;
552  SetBoard(move.row, move.column, player_piece);
553  SetBoard(end_row, end_column, CellState::kEmpty);
554  CellState captured_piece =
555  GetPieceStateFromTurnHistory(1 - player, thi.captured_piece_type);
556  SetBoard((move.row + end_row) / 2, (move.column + end_column) / 2,
557  captured_piece);
558  break;
559  }
560  turn_history_info_.pop_back();
561  history_.pop_back();
562 }
