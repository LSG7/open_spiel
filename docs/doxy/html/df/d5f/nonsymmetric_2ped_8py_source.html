51def gradients(dist, payoff_matrices, num_players, proj_grad=True):
52 """Computes exploitablity gradient.
53
54 Args:
55 dist: list of 1-d np.arrays, current estimate of nash distribution
56 payoff_matrices: dictionary with keys as tuples of agents (i, j) and
57 values of (2 x A x A) np.arrays, payoffs for each joint action. keys
58 are sorted and arrays should be indexed in the same order
59 num_players: int, number of players, in case payoff_matrices is abbreviated
60 proj_grad: bool, if True, projects dist gradient onto simplex
61 Returns:
62 gradient of exploitability w.r.t. (dist) as tuple
63 unregularized exploitability (stochastic estimate)
64 unregularized exploitability (stochastic estimate) *duplicate
65 """
66 # first compute best responses and payoff gradients
67 nabla = []
68 br = []
69 unreg_exp = []
70 for i in range(num_players):
71
72 nabla_i = np.zeros_like(dist[i])
73 for j in range(num_players):
74 if j == i:
75 continue
76 if i < j:
77 hess_i_ij = payoff_matrices[(i, j)][0]
78 else:
79 hess_i_ij = payoff_matrices[(j, i)][1].T
80
81 nabla_ij = hess_i_ij.dot(dist[j])
82 nabla_i += nabla_ij / float(num_players - 1)
83
84 nabla.append(nabla_i)
85
86 power = np.inf
87 s_i = np.linalg.norm(nabla_i, ord=power)
88 br_i = np.zeros_like(nabla_i)
89 maxima_i = (nabla_i == s_i)
90 br_i[maxima_i] = 1. / maxima_i.sum()
91 br.append(br_i)
92
93 unreg_exp.append(np.max(nabla_i) - nabla_i.dot(dist[i]))
94
95 # then construct exploitability gradient
96 grad_dist = []
97 for i in range(num_players):
98
99 grad_dist_i = -nabla[i]
100 for j in range(num_players):
101 if j == i:
102 continue
103 if i < j:
104 hess_j_ij = payoff_matrices[(i, j)][1]
105 else:
106 hess_j_ij = payoff_matrices[(j, i)][0].T
107
108 grad_dist_i += hess_j_ij.dot(br[j] - dist[j])
109
110 if proj_grad:
111 grad_dist_i = simplex.project_grad(grad_dist_i)
112
113 grad_dist.append(grad_dist_i)
114
115 return (grad_dist,), np.mean(unreg_exp), np.mean(unreg_exp)
