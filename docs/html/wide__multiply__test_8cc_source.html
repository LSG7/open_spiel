26 U256 LeftShift(U256 v, int s) {
27  if (s == 0) {
28  return v;
29  } else if (s < 128) {
30  return {(v.hi << s) | (v.lo >> (128 - s)), v.lo << s};
31  } else {
32  return {v.lo << (s - 128), 0};
33  }
34 }
