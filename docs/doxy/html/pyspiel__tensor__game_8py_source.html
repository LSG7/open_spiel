28 def __init__(self, string_specifier='blotto(coins=10,fields=3,players=3)',

42 def num_players(self):

45 def num_strategies(self):

48 def payoff_tensor(self):
49 if self.pt is None:
50 if not self.tensor_game:
51 logging.info('reloading pyspiel game as tensor_game')
52 self.game = pyspiel.load_tensor_game(self.string_specifier)
53 self.tensor_game = True
54 pt = np.asarray(game_payoffs_array(self.game))
55 self.pt = pt - self.game.min_utility()
56 return self.pt

58 def get_payoffs_for_strategies(self, policies):
59 """Return vector of payoffs for all players given list of strategies.
60
61 Args:
62 policies: list of integers indexing strategies for each player
63 Returns:
64 np.array (length num players) of payoffs
65 """
66 state = self.game.new_initial_state()
67 state.apply_actions(policies)
68 return np.asarray(state.returns()) - self.game.min_utility()
