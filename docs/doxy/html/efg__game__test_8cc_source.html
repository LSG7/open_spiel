57void EFGGameSimTestsKuhnFromData() {
58 std::shared_ptr<const Game> game = LoadEFGGame(GetKuhnPokerEFGData());
59 SPIEL_CHECK_TRUE(game != nullptr);
60 GameType type = game->GetType();
61 SPIEL_CHECK_EQ(type.dynamics, GameType::Dynamics::kSequential);
62 SPIEL_CHECK_EQ(type.information,
63 GameType::Information::kImperfectInformation);
64 SPIEL_CHECK_EQ(type.utility, GameType::Utility::kZeroSum);
65 SPIEL_CHECK_EQ(type.chance_mode, GameType::ChanceMode::kExplicitStochastic);
66 SPIEL_CHECK_EQ(game->NumDistinctActions(), 2);
67 SPIEL_CHECK_EQ(game->MaxChanceOutcomes(), 3);
68
69 // EFG games loaded directly via string cannot be properly deserialized

97void EFGGameCommasFromFile() {
98 absl::optional<std::string> file = FindFile(kCommasFilename, 2);
99 if (file != absl::nullopt) {
100 std::cout << "Found file: " << file.value() << "; running sim test.";
101 std::shared_ptr<const Game> game =
102 LoadGame("efg_game", {{"filename", GameParameter(file.value())}});
103 SPIEL_CHECK_TRUE(game != nullptr);
104 GameType type = game->GetType();
105 SPIEL_CHECK_EQ(type.dynamics, GameType::Dynamics::kSequential);
106 SPIEL_CHECK_EQ(type.information,
107 GameType::Information::kImperfectInformation);
108 SPIEL_CHECK_EQ(type.utility, GameType::Utility::kGeneralSum);
109 SPIEL_CHECK_EQ(type.chance_mode, GameType::ChanceMode::kExplicitStochastic);
110 SPIEL_CHECK_EQ(game->NumDistinctActions(), 4);
111 SPIEL_CHECK_EQ(game->NumPlayers(), 2);
112 }
113}
