30 def test_single_deterministic_game_1(self):
31 """Runs a single game where tiles and actions chose deterministically."""
32 game = pyspiel.load_game("python_block_dominoes")
33 state = game.new_initial_state()
34 hand0 = [
35 (6.0, 6.0),
36 (0.0, 2.0),
37 (4.0, 4.0),
38 (3.0, 3.0),
39 (2.0, 2.0),
40 (1.0, 1.0),
41 (0.0, 0.0),
42 ]
43 hand1 = [
44 (5.0, 6.0),
45 (4.0, 5.0),
46 (3.0, 4.0),
47 (2.0, 3.0),
48 (1.0, 2.0),
49 (0.0, 1.0),
50 (4.0, 6.0),
51 ]
52 self.deal_hands(state, [hand0, hand1])
53
54 self.apply_action(state, block_dominoes.Action(0, (6.0, 6.0), None))
55 self.apply_action(state, block_dominoes.Action(1, (5.0, 6.0), 6.0))
56 # player 0 don't hold any tile with 6 or 5, player 1 turn again
57 self.apply_action(state, block_dominoes.Action(1, (4.0, 5.0), 5.0))
58 self.apply_action(state, block_dominoes.Action(0, (4.0, 4.0), 4.0))
59 self.apply_action(state, block_dominoes.Action(1, (3.0, 4.0), 4.0))
60 self.apply_action(state, block_dominoes.Action(0, (3.0, 3.0), 3.0))
61 self.apply_action(state, block_dominoes.Action(1, (2.0, 3.0), 3.0))
62 self.apply_action(state, block_dominoes.Action(0, (2.0, 2.0), 2.0))
63 self.apply_action(state, block_dominoes.Action(1, (1.0, 2.0), 2.0))
64 self.apply_action(state, block_dominoes.Action(0, (1.0, 1.0), 1.0))
65 self.apply_action(state, block_dominoes.Action(1, (0.0, 1.0), 1.0))
66 self.apply_action(state, block_dominoes.Action(0, (0.0, 0.0), 0.0))
67 self.apply_action(state, block_dominoes.Action(1, (4.0, 6.0), 6.0))
68
69 # player 1 played all is tile and player 0 hold the tile (0, 2)
70 self.assertTrue(state.is_terminal())
71 self.assertEqual(state.returns()[0], -2)
72 self.assertEqual(state.returns()[1], 2)

74 def test_single_deterministic_game_2(self):
75 """Runs a single game where tiles and actions chose deterministically."""

78 hand0 = [
79 (6.0, 6.0),

81 (1.0, 5.0),
82 (2.0, 5.0),
83 (3.0, 5.0),
84 (4.0, 5.0),

88 (0.0, 4.0),
89 (1.0, 4.0),
90 (2.0, 4.0),

92 (4.0, 4.0),
93 (0.0, 3.0),
94 (1.0, 3.0),
95 ]
96 self.deal_hands(state, [hand0, hand1])
97
98 self.apply_action(state, block_dominoes.Action(0, (6.0, 6.0), None))
99 # Both players don't hold tile with 6, therefore both blocked and the

101 self.assertTrue(state.is_terminal())
102 self.assertEqual(state.returns()[0], -45)
