36int GetBeliefHistorySize(const HistoryDistribution& beliefs) {
37 int belief_history_size = 0;
38 for (int i = 0; i < beliefs.first.size(); ++i) {
39 belief_history_size =
40 std::max(belief_history_size,
41 static_cast<int>(beliefs.first[i]->FullHistory().size()));
42 }
43 return belief_history_size;
44}

46std::unique_ptr<HistoryDistribution> AdvanceBeliefHistoryOneAction(
47 std::unique_ptr<HistoryDistribution> previous, Action action,
48 Player player_id, const Policy& opponent_policy) {
49 auto dist = absl::make_unique<HistoryDistribution>();
50 for (int i = 0; i < previous->first.size(); ++i) {
51 std::unique_ptr<State>& state = previous->first[i];
52 const double& prob = previous->second[i];
53 if (Near(prob, 0.)) continue;
54 switch (state->GetType()) {
55 case StateType::kChance: {
56 // If we can't find the action in the policy, then set it to 0.
57 const double action_prob = GetProb(state->ChanceOutcomes(), action);
58
59 // Then, skip all actions with 0 probability, as they don't matter
60 // moving forward.
61 if (Near(std::max(action_prob, 0.0), 0.0)) continue;
62 SPIEL_CHECK_PROB(action_prob);
63
64 // If we don't find the chance outcome, then the state we're in is
65 // impossible, so we set it to zero.
66 state->ApplyAction(action);
67
68 dist->first.push_back(std::move(state));
69 dist->second.push_back(prob * std::max(0.0, action_prob));
70 break;
71 }
72 case StateType::kDecision: {
73 if (state->CurrentPlayer() == player_id) {
74 state->ApplyAction(action);
75 dist->first.push_back(std::move(state));
76 dist->second.push_back(prob);
77 } else {
78 // We have to add all actions as we don't know if the opponent is
79 // taking a private or public action.
80 // TODO(author1): Add method to open_spiel::State that lets us
81 // only loop over the actions that are consistent with a given private
82 // action.
83 for (const auto& [candidate, action_prob] :
84 opponent_policy.GetStatePolicy(*state)) {
85 if (Near(std::max(0.0, action_prob), 0.0)) continue;
86 SPIEL_CHECK_PROB(action_prob);
87 std::unique_ptr<State> child = state->Child(candidate);
88 if (child->IsTerminal()) continue;
89 dist->first.push_back(std::move(child));
90 dist->second.push_back(prob * action_prob);
91 }
92 }
93 break;
94 }
95 case StateType::kTerminal:
96 // If the state is terminal, and we have to advance by an action, we
97 // discard the terminal histories from our beliefs.
98 continue;
99 // SpielFatalError("State is terminal, should not call
100 // AdvanceBeliefs.");
101 default:
102 SpielFatalError(absl::StrCat("Unknown state type: ", state->GetType(),
103 ", state: ", state->ToString()));
104 }
105 }
106 return dist;
107}

109// Filters out all beliefs that do not belong to infostate.
110std::unique_ptr<HistoryDistribution> FilterOutBeliefs(
111 const State& state, std::unique_ptr<HistoryDistribution> dist,
112 int player_id) {
113 const std::string infostate = state.InformationStateString(player_id);
114 auto new_dist = absl::make_unique<HistoryDistribution>();
115 std::vector<int> good_indices;
116 for (int i = 0; i < dist->first.size(); ++i) {
117 if (dist->first[i]->InformationStateString(player_id) == infostate) {
118 good_indices.push_back(i);
119 }
120 }
121 new_dist->first.reserve(good_indices.size());
122 new_dist->second.reserve(good_indices.size());
123 for (int i : good_indices) {
124 new_dist->first.push_back(std::move(dist->first[i]));
125 new_dist->second.push_back(dist->second[i]);
126 }
127 return new_dist;
128}

132std::unique_ptr<open_spiel::HistoryDistribution> CloneBeliefs(
133 const open_spiel::HistoryDistribution& beliefs) {
134 auto beliefs_copy = absl::make_unique<open_spiel::HistoryDistribution>();
135 for (int i = 0; i < beliefs.first.size(); ++i) {
136 beliefs_copy->first.push_back(beliefs.first[i]->Clone());
137 beliefs_copy->second.push_back(beliefs.second[i]);
138 }
139 return beliefs_copy;

142HistoryDistribution GetStateDistribution(const State& state,
143 const Policy& opponent_policy) {
144 std::shared_ptr<const Game> game = state.GetGame();
145 GameType game_type = game->GetType();
146 if (game_type.information == GameType::Information::kPerfectInformation) {
147 HistoryDistribution dist;
148 // We can't use brace initialization here as it triggers the copy ctor.
149 dist.first.push_back(state.Clone());
150 dist.second.push_back(1.);
151 return dist;
152 }
153 SPIEL_CHECK_EQ(game_type.information,
154 GameType::Information::kImperfectInformation);
155 SPIEL_CHECK_EQ(game_type.dynamics, GameType::Dynamics::kSequential);
156 SPIEL_CHECK_NE(game_type.chance_mode,
157 GameType::ChanceMode::kSampledStochastic);
158 SPIEL_CHECK_FALSE(state.IsChanceNode());
159 SPIEL_CHECK_FALSE(state.IsTerminal());
160
161 Player player = state.CurrentPlayer();
162 std::string info_state_string = state.InformationStateString();
163
164 // Generate the (info state, action) map for the current player using
165 // the state's history.
166 absl::btree_map<std::string, Action> infostate_action_map;
167 std::vector<Action> history = state.History();
168 std::unique_ptr<State> tmp_state = game->NewInitialState();
169 for (Action action : history) {
170 if (tmp_state->CurrentPlayer() == player) {
171 infostate_action_map[tmp_state->InformationStateString()] = action;
172 }
173 tmp_state->ApplyAction(action);
174 }
175 // Add the current one to this list with an invalid action so that the
176 // information state is included.
177 infostate_action_map[info_state_string] = kInvalidAction;
178
179 // Should get to the exact same state by re-applying the history.
180 SPIEL_CHECK_EQ(tmp_state->ToString(), state.ToString());
181
182 // Now, do a breadth-first search of all the candidate histories, removing
183 // them whenever their (infostate, action) is not contained in the map above.
184 // The search finishes when all the information states of the states in
185 // the list have been found. We use two lists: final_states contains the ones
186 // that have been found, while states are the current candidates.
187 std::vector<std::unique_ptr<State>> final_states;
188 std::vector<double> final_probs;
189 std::vector<std::unique_ptr<State>> states;
190 std::vector<double> probs;
191 states.push_back(game->NewInitialState());
192 probs.push_back(1.0);
193
194 while (!states.empty()) {
195 for (int idx = 0; idx < states.size();) {
196 if (states[idx]->IsTerminal()) {
197 // Terminal cannot be a valid history in an information state, so stop
198 // considering this line.
199 } else if (states[idx]->IsChanceNode()) {
200 // At chance nodes, just add all the children and delete the state.
201 for (std::pair<Action, double> action_and_prob :
202 states[idx]->ChanceOutcomes()) {
203 states.push_back(states[idx]->Child(action_and_prob.first));
204 probs.push_back(probs[idx] * action_and_prob.second);
205 }
206 } else if (states[idx]->CurrentPlayer() != player) {
207 // At opponent nodes, similar to chance nodes but get the probability
208 // from the policy instead.
209 std::string opp_infostate_str = states[idx]->InformationStateString();
210 ActionsAndProbs state_policy =
211 opponent_policy.GetStatePolicy(*states[idx]);
212 for (Action action : states[idx]->LegalActions()) {
213 double action_prob = GetProb(state_policy, action);
214 states.push_back(states[idx]->Child(action));
215 probs.push_back(probs[idx] * action_prob);
216 }
217 } else if (states[idx]->CurrentPlayer() == player) {
218 std::string my_infostate_str = states[idx]->InformationStateString();
219 // First check if this state is in the target information state. If
220 // add it to the final set and don't check for expansion.
221 if (my_infostate_str == info_state_string) {
222 final_states.push_back(states[idx]->Clone());
223 final_probs.push_back(probs[idx]);
224 } else {
225 // Check for expansion of this candidate. To expand this candidate,
226 // the (infostate, action) pair must be contained in the map.
227 auto iter = infostate_action_map.find(my_infostate_str);
228 if (iter != infostate_action_map.end() && iter->second) {
229 states.push_back(states[idx]->Child(iter->second));
230 probs.push_back(probs[idx]);
231 }
232 }
233 } else {
234 SpielFatalError(
235 absl::StrCat("Unknown player: ", states[idx]->CurrentPlayer()));
236 }
237
238 // Delete entries at the index i. Rather than call erase, which would
239 // shift everything, simply swap with the last element and call
240 // pop_back(), which can be done in constant time.
241 std::swap(states[idx], states.back());
242 std::swap(probs[idx], probs.back());
243 states.pop_back();
244 probs.pop_back();
245
246 // Do not increment the counter index here because the current one points
247 // to a valid state that was just expanded.
248 }
249 }
250
251 // Now normalize the probs
252 Normalize(absl::MakeSpan(final_probs));
253 HistoryDistribution dist = {std::move(final_states), std::move(final_probs)};
254
255 // Note: We do not call CheckBeliefs here as the beliefs are _wrong_ until we
256 // perform the filter step.
257 return dist;
258}

260std::unique_ptr<HistoryDistribution> UpdateIncrementalStateDistribution(
261 const State& state, const Policy& opponent_policy, int player_id,
262 std::unique_ptr<HistoryDistribution> previous) {
263 std::unique_ptr<HistoryDistribution> dist;
264 if (previous) {
265 dist = std::move(previous);
266 }
267 // If we don't have a previous set of beliefs, create it.
268 if (!dist || dist->first.empty()) {
269 // This allows for games to special case this scenario. It only works if
270 // this is only called at the first decision node after chance nodes. We
271 // leave it to the caller to verify this is the case.
272 dist = state.GetHistoriesConsistentWithInfostate();
273
274 // If the game didn't implement GetHistoriesConsistentWithInfostate, then
275 // this is empty, otherwise, we're good.
276 if (!dist || dist->first.empty()) {
277 // If the previous pair is empty, then we have to do a BFS to find all
278 // relevant nodes:
279 dist = absl::make_unique<HistoryDistribution>(
280 GetStateDistribution(state, opponent_policy));
281 }
282 }
283 // Now, we verify that the beliefs match the current infostate.
284 const std::vector<State::PlayerAction>& history = state.FullHistory();
285 int belief_history_size = GetBeliefHistorySize(*dist);
286 std::unique_ptr<State> new_state = state.GetGame()->NewInitialState();
287 for (int i = 0; i < belief_history_size; ++i) {
288 new_state->ApplyAction(history[i].action);
289 }
290 SPIEL_DCHECK_TRUE(CheckBeliefs(*new_state, *dist, player_id));
291 while (belief_history_size < history.size()) {
292 dist = AdvanceBeliefHistoryOneAction(std::move(dist),
293 history[belief_history_size].action,
294 player_id, opponent_policy);
295 new_state->ApplyAction(history[belief_history_size].action);
296 dist = FilterOutBeliefs(*new_state, std::move(dist), player_id);
297 SPIEL_CHECK_FALSE(dist->first.empty());
298 if (!new_state->IsChanceNode()) {
299 SPIEL_DCHECK_TRUE(CheckBeliefs(*new_state, *dist, player_id));
300 }
301 const int new_belief_history_size = GetBeliefHistorySize(*dist);
302 SPIEL_CHECK_LT(belief_history_size, new_belief_history_size);
303 belief_history_size = new_belief_history_size;
304 }
305 SPIEL_CHECK_EQ(belief_history_size, history.size());
306 SPIEL_CHECK_EQ(new_state->FullHistory(), state.FullHistory());
307 dist = FilterOutBeliefs(state, std::move(dist), player_id);
308 SPIEL_CHECK_FALSE(dist->first.empty());
309
310 // We only normalize after filtering out invalid infostates.
311 Normalize(absl::MakeSpan(dist->second));
312
313 SPIEL_DCHECK_TRUE(CheckBeliefs(state, *dist, player_id));
314 return dist;
315}

317std::string PrintBeliefs(const HistoryDistribution& beliefs, int player_id) {
318 const int num_states = beliefs.first.size();
319 SPIEL_CHECK_EQ(num_states, beliefs.second.size());
320 std::string str;
321 for (int i = 0; i < num_states; ++i) {
322 absl::StrAppend(
323 &str,
324 absl::StrFormat("(%s, %f)",
325 beliefs.first[i]->InformationStateString(player_id),
326 beliefs.second[i]));
327 if (i < num_states - 1) absl::StrAppend(&str, "\n");
328 }
329 return str;
330}

332bool CheckBeliefs(const State& ground_truth_state,
333 const HistoryDistribution& beliefs, int player_id) {
334 const std::string infostate =
335 ground_truth_state.InformationStateString(player_id);
336 for (int i = 0; i < beliefs.first.size(); ++i) {
337 if (Near(beliefs.second[i], 0.0, 1e-5)) {
338 continue;
339 }
340 SPIEL_CHECK_EQ(infostate,
341 beliefs.first[i]->InformationStateString(player_id));
342 SPIEL_CHECK_EQ(ground_truth_state.FullHistory().size(),
343 beliefs.first[i]->FullHistory().size());
344 SPIEL_CHECK_EQ(ground_truth_state.IsTerminal(),
345 beliefs.first[i]->IsTerminal());
346 }
347 return true;
348}
