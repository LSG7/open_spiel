51def _get_next_states(state, next_states, to_string):
52 """Extract non-chance states for a subgame into the all_states dict."""
53 is_mean_field = state.current_player() == pyspiel.PlayerId.MEAN_FIELD
54 if state.is_chance_node():
55 # Add only if not already present
56
57 for action, _ in state.chance_outcomes():
58 next_state = state.child(action)
59 state_str = to_string(next_state)
60 if state_str not in next_states:
61 next_states[state_str] = next_state
62
63 if is_mean_field:
64 support = state.distribution_support()
65 next_state = state.clone()
66 support_length = len(support)
67 # update with a dummy distribution
68 next_state.update_distribution(
69 [1.0 / support_length for _ in range(support_length)])
70 state_str = to_string(next_state)
71 if state_str not in next_states:
72 next_states[state_str] = next_state
73
74 if int(state.current_player()) >= 0:
75 for action in state.legal_actions():
76 next_state = state.child(action)
77 state_str = to_string(next_state)
78 if state_str not in next_states:
79 next_states[state_str] = next_state

82def _next_states(states, to_string):
83 next_states = {}
84 for state in states:
85 _get_next_states(state, next_states, to_string)
86 return set(next_states.keys()), set(next_states.values())

89def type_from_states(states):
90 """Get node type of a list of states and assert they are the same."""
91 types = [state.get_type() for state in states]
92 assert len(set(types)) == 1
93 return types[0]

98 @parameterized.parameters(
99 {'game_name': 'python_mfg_crowd_modelling'},
100 {'game_name': 'mfg_crowd_modelling'},
101 {'game_name': 'mfg_garnet'},
102 {'game_name': 'mfg_crowd_modelling_2d'},
103 {'game_name': 'python_mfg_periodic_aversion'},
104 {'game_name': 'python_mfg_predator_prey'},
105 )
106 def test_is_finite_horizon(self, game_name):
107 """Check that the game has no loop."""
108 game = pyspiel.load_game(game_name)
109 states = set(game.new_initial_states())
110 def to_string(s):
111 return s.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)
112 all_states_key = set(to_string(state) for state in states)
113 while type_from_states(states) != pyspiel.StateType.TERMINAL:
114 new_states_key, states = _next_states(states, to_string)
115 self.assertEmpty(all_states_key.intersection(new_states_key))
116 all_states_key.update(new_states_key)

149 def test_rl_environment(self, game_name):
150 """Check that the RL environment runs for a few trajectories."""
151 game = pyspiel.load_game(game_name)
152 uniform_policy = policy.UniformRandomPolicy(game)
153 mfg_dist = distribution.DistributionPolicy(game, uniform_policy)
154
155 envs = [
156 rl_environment.Environment(
157 game, mfg_distribution=mfg_dist, mfg_population=p)
158 for p in range(game.num_players())
159 ]
160 for p, env in enumerate(envs):
161 for _ in range(FLAGS.rl_env_simulations):
162 time_step = env.reset()
163 while not time_step.last():
164 a = random.choice(time_step.observations['legal_actions'][p])
165 time_step = env.step([a])
166
167 env = envs[0]
168 self.assertEqual(env.mfg_distribution, mfg_dist)
169
170 # Update the distribution.
171 new_mfg_dist = distribution.DistributionPolicy(game, uniform_policy)
172 env.update_mfg_distribution(new_mfg_dist)
173 self.assertEqual(env.mfg_distribution, new_mfg_dist)
