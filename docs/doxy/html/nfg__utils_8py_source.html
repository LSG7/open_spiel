21class StrategyAverager(object):
22 """A helper class for averaging strategies for players."""
23
24 def __init__(self, num_players, action_space_shapes, window_size=None):
25 """Initialize the average strategy helper object.
26
27 Args:
28 num_players (int): the number of players in the game,
29 action_space_shapes: an vector of n integers, where each element
30 represents the size of player i's actions space,
31 window_size (int or None): if None, computes the players' average
32 strategies over the entire sequence, otherwise computes the average
33 strategy over a finite-sized window of the k last entries.
34 """
35 self._num_players = num_players
36 self._action_space_shapes = action_space_shapes
37 self._window_size = window_size
38 self._num = 0
39 if self._window_size is None:
40 self._sum_meta_strategies = [
41 np.zeros(action_space_shapes[p]) for p in range(num_players)
42 ]
43 else:
44 self._window = collections.deque(maxlen=self._window_size)
45
46 def append(self, meta_strategies):
47 """Append the meta-strategies to the averaged sequence.
48
49 Args:
50 meta_strategies: a list of strategies, one per player.
51 """
52 if self._window_size is None:
53 for p in range(self._num_players):
54 self._sum_meta_strategies[p] += meta_strategies[p]
55 else:
56 self._window.append(meta_strategies)
57 self._num += 1
58
59 def average_strategies(self):
60 """Return each player's average strategy.
61
62 Returns:
63 The averaged strategies, as a list containing one strategy per player.
64 """
65
66 if self._window_size is None:
67 avg_meta_strategies = [
68 np.copy(x) for x in self._sum_meta_strategies
69 ]
70 num_strategies = self._num
71 else:
72 avg_meta_strategies = [
73 np.zeros(self._action_space_shapes[p])
74 for p in range(self._num_players)
75 ]
76 for i in range(len(self._window)):
77 for p in range(self._num_players):
78 avg_meta_strategies[p] += self._window[i][p]
79 num_strategies = len(self._window)
80 for p in range(self._num_players):
81 avg_meta_strategies[p] /= num_strategies
82 return avg_meta_strategies
