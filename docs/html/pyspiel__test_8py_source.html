199  def test_play_kuhn_poker(self):
200  game = pyspiel.load_game("kuhn_poker")
201  state = game.new_initial_state()
202  self.assertEqual(state.is_chance_node(), True)
203  self.assertEqual(state.chance_outcomes(), [(0, 1 / 3), (1, 1 / 3),
204  (2, 1 / 3)])
205  state.apply_action(1)
206  self.assertEqual(state.is_chance_node(), True)
207  self.assertEqual(state.chance_outcomes(), [(0, 0.5), (2, 0.5)])
208  state.apply_action(2)
209  self.assertEqual(state.is_chance_node(), False)
210  self.assertEqual(state.legal_actions(), [0, 1])
211  sampler = pyspiel.UniformProbabilitySampler(0., 1.)
212  clone = state.resample_from_infostate(1, sampler)
213  self.assertEqual(
214  clone.information_state_string(1), state.information_state_string(1))
