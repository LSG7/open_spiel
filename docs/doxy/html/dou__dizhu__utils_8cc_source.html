23int CardToRank(int card) {
24 if (card == kNumCards - 2 || card == kNumCards - 1) {
25 return card - kNumCards + kNumRanks;
26 }
27 return card % (kNumRanks - 2);
28}

30int CardToSuit(int card) {
31 if (card == kNumCards - 2 || card == kNumCards - 1) {
32 SpielFatalError("No Suit defined for Jokers");
33 }
34 return card / (kNumRanks - 2);
35}

37std::string RankString(int rank) {
38 if (rank < kNumRanks - 2)
39 return std::string(1, kRankChar[rank]);
40 else if (rank == kNumRanks - 2)
41 return "(BWJ)";
42 else if (rank == kNumRanks - 1)
43 return "(CJ)";
44 else
45 SpielFatalError("Non valid rank");
46}

48std::string CardString(int card) {
49 int rank = CardToRank(card);
50 if (rank >= kNumRanks - 2) {
51 return RankString(rank);
52 } else {
53 int suit = CardToSuit(card);
54 SPIEL_CHECK_GE(suit, 0);
55 SPIEL_CHECK_LT(suit, kNumSuits);
56 return absl::StrFormat("%c%c", kSuitChar[suit], kRankChar[rank]);
57 }
58}

60std::string FormatSingleHand(absl::Span<const int> hand) {
61 std::string hand_format;
62 for (int rank = 0; rank < kNumRanks; ++rank) {
63 for (int i = 0; i < hand[rank]; ++i)
64 absl::StrAppend(&hand_format, RankString(rank));
65 }

69// resolve ambiguity for cases like 333444555666
70std::string FormatAirplaneCombHand(int action) {
71 TrioCombParams params = GetAirplaneCombParams(action);
72 std::array<int, kNumRanks> hand = ActionToHand(action);
73 std::string airplane_comb_str;
74 // specify which is chain
75 for (int rank = params.chain_head;
76 rank < params.chain_head + params.chain_length; ++rank) {
77 for (int i = 0; i < 3; ++i)
78 absl::StrAppend(&airplane_comb_str, RankString(rank));
79 }
80 absl::StrAppend(&airplane_comb_str, "-");
81 // kickers
82 for (int rank = 0; rank < kNumRanks; ++rank) {
83 if (rank >= params.chain_head &&
84 rank < params.chain_head + params.chain_length)
85 continue;
86 if (!hand[rank]) continue;
87 for (int i = 0; i < hand[rank]; ++i)
88 absl::StrAppend(&airplane_comb_str, RankString(rank));
89 }
90 return airplane_comb_str;
91}

94int GetNumCardsPerRank(int action) {
95 int num_cards;
96 if (action >= kPlayActionBase && action < kPairActionBase) {
97 num_cards = 1;
98 } else if (action >= kPairActionBase && action < kTrioActionBase) {
99 num_cards = 2;
100 } else if ((action >= kTrioActionBase && action < kTrioWithSoloActionBase) ||
101 (action >= kAirplaneActionBase &&
102 action < kAirplaneWithSoloActionBase)) {
103 num_cards = 3;
104 } else if (action >= kBombActionBase && action < kRocketActionBase) {
105 num_cards = 4;
106 } else {
107 SpielFatalError("Invalid action ID");
108 }
109
110 return num_cards;
111}

149// given a single-rank hand, map it to action id
150int SingleRankHandToActionId(absl::Span<const int> hand) {
151 int the_rank;
152 int counter = 0;
153
154 for (int rank = 0; rank < kNumRanks; ++rank) {
155 if (hand[rank] != 0) {
156 the_rank = rank;
157 counter++;
158 }
159 }
160 SPIEL_CHECK_EQ(counter, 1);
161 const int num_cards_same_rank = hand[the_rank];
162 int action = GetSingleRankActionBase(num_cards_same_rank);
163 action += the_rank;
164 return action;
165}

167// given an arbitrary hand, search for possible single-rank hands
168// if prev_action = kInvalidAction, search for all possible such hands
169// otherwise, only search for those that are ranked higher than prev_action
170void SearchSingleRankActions(std::vector<Action>* actions,
171 absl::Span<const int> hand,
172 int prev_action = kInvalidAction) {
173 std::array<int, kNumRanks> used_hands{};
174 SingleRankHandParams prev_action_params;
175 int start_rank;
176 if (prev_action == kInvalidAction) {
177 start_rank = 0;
178 } else {
179 prev_action_params = GetSingleRankHandParams(prev_action);
180 start_rank = prev_action_params.rank + 1;
181 }
182 for (int rank = start_rank; rank < kNumRanks; ++rank) {
183 SPIEL_CHECK_LE(hand[rank], kNumSuits);
184 SPIEL_CHECK_GE(hand[rank], 0);
185 if (rank == kNumRanks - 2 || rank == kNumRanks - 1)
186 SPIEL_CHECK_LE(hand[rank], 1);
187 if (prev_action == kInvalidAction) {
188 for (int i = 0; i < hand[rank]; ++i) {
189 used_hands[rank]++;
190 actions->push_back(SingleRankHandToActionId(used_hands));
191 }
192 } else if (hand[rank] >= prev_action_params.num_cards) {
193 used_hands[rank] = prev_action_params.num_cards;
194 actions->push_back(SingleRankHandToActionId(used_hands));
195 }
196 used_hands[rank] = 0;
197 }
198}

236ChainOnlyHandParams GetChainOnlyHandParams(int action) {
237 const int num_cards_same_rank = GetNumCardsPerRank(action);
238 const int action_base = GetChainOnlyActionBase(num_cards_same_rank);
239 const int min_length = GetChainOnlyMinLength(num_cards_same_rank);
240 SPIEL_CHECK_GE(action, action_base);
241 const int hand_id = action - action_base;
242 int chain_length = min_length;
243 int base = 0;
244 // we label the action Ids by increasing length of the chain
245 for (chain_length = min_length; chain_length <= kNumRanks; ++chain_length) {
246 int num_chains = kNumRanks - chain_length - 2;
247 if (base <= hand_id && hand_id < base + num_chains) break;
248 base += num_chains;
249 }
250 const int chain_head = hand_id - base;
251 return ChainOnlyHandParams(chain_head, num_cards_same_rank, chain_length);
252}

263int ChainOnlyHandToActionId(absl::Span<const int> hand) {
264 int chain_head = -1;
265 int chain_length = 0;
266 int chain_counter = 0;
267 int num_cards_same_rank = 0;
268 bool chain_stopped = true;
269
270 if (hand[kNumRanks - 3] || hand[kNumRanks - 2] || hand[kNumRanks - 1])
271 SpielFatalError("2s and Jokers cannot be in a chain");
272
273 for (int rank = 0; rank < kNumRanks - 3; ++rank) {
274 if (hand[rank] == 0) {
275 chain_stopped = true;
276 } else {
277 if (chain_stopped) {
278 chain_head = rank;
279 num_cards_same_rank = hand[rank];
280 chain_length = 1;
281 chain_counter++;
282 chain_stopped = false;
283 } else if (hand[rank] != num_cards_same_rank) {
284 SpielFatalError("Invalid pattern");
285 } else {
286 chain_length++;
287 }
288 }
289 }
290
291 SPIEL_CHECK_EQ(chain_counter, 1);
292 const int min_length = GetChainOnlyMinLength(num_cards_same_rank);
293 const int action_base = GetChainOnlyActionBase(num_cards_same_rank);
294
295 if (chain_length < min_length)
296 SpielFatalError(absl::StrFormat("The length of chain should be at least %d",
297 min_length));
298 int action = action_base;
299 for (int length = min_length; length < chain_length; ++length)
300 action += kNumRanks - length - 2;
301 action += chain_head;
302 return action;
303}

305void SearchChainOnlyActions(std::vector<Action>* actions,
306 absl::Span<const int> hand,
307 int prev_action = kInvalidAction) {
308 ChainOnlyHandParams prev_action_params;
309
310 int start_rank;
311 if (prev_action == kInvalidAction) {
312 start_rank = 0;
313 } else {
314 prev_action_params = GetChainOnlyHandParams(prev_action);
315 start_rank = prev_action_params.chain_head + 1;
316 }
317
318 for (int chain_head = start_rank; chain_head < kNumRanks - 4; ++chain_head) {
319 if (!hand[chain_head] || hand[chain_head] == kNumSuits) continue;
320 int num_cards = hand[chain_head];
321 // 2-s and Jokers cannot be in chain
322 for (int chain_length = 2; chain_head + chain_length - 1 < kNumRanks - 3;
323 ++chain_length) {
324 int chain_tail = chain_head + chain_length - 1;
325 num_cards = std::min(num_cards, hand[chain_tail]);
326 if (!num_cards) break;
327 std::vector<int> all_nums;
328 if (prev_action != kInvalidAction) {
329 if (num_cards < prev_action_params.num_cards_per_rank) break;
330 if (chain_length > prev_action_params.chain_length) break;
331 if (chain_length == prev_action_params.chain_length) {
332 all_nums.push_back(prev_action_params.num_cards_per_rank);
333 }
334 } else {
335 for (int n = 1; n <= num_cards; ++n) {
336 all_nums.push_back(n);
337 }
338 }
339
340 for (auto n : all_nums) {
341 const int min_length = GetChainOnlyMinLength(n);
342 if (chain_length >= min_length) {
343 std::array<int, kNumRanks> used_rank{};
344 for (int i = 0; i < chain_length; ++i) used_rank[chain_head + i] = n;
345 actions->push_back(ChainOnlyHandToActionId(used_rank));
346 }
347 }
348 }
349 }
350}

390// single trio comb includes trio+solo and trio+pair (excluding airplanes)
391TrioCombParams GetSingleTrioCombParams(int action) {
392 if (action < kTrioWithSoloActionBase || action >= kAirplaneActionBase)
393 SpielFatalError("Must be single trio pattern");
394
395 const int action_base = GetTrioCombActionBase(action);
396 const KickerType kicker_type = GetTrioCombKickerType(action);
397 const int hand_id = (action - action_base);
398 const int num_kickers = kicker_type == kSolo ? kNumRanks - 1 : kNumRanks - 3;
399 const int head = hand_id / num_kickers;
400 const int kicker_steps = hand_id % num_kickers;
401
402 return TrioCombParams(head, 1, kicker_type, kicker_steps);
403}

500TrioCombParams GetAirplaneCombParams(int action) {
501 if (action < kAirplaneWithSoloActionBase || action >= kBombActionBase)
502 SpielFatalError("Must be airplane pattern");
503
504 int action_base = kInvalidAction;
505 KickerType kicker_type;
506
507 SPIEL_CHECK_GE(action, kAirplaneWithSoloActionBase);
508 SPIEL_CHECK_LT(action, kBombActionBase);
509 int start_length = 2, end_length, end_base;
510
511 int (*GetActionBaseFunc)(int), (*GetKickersNumFunc)(int);
512 if (kAirplaneWithSoloActionBase <= action &&
513 action < kAirplaneWithPairActionBase) {
514 kicker_type = kSolo;
515 GetActionBaseFunc = &GetAirplaneSoloActionBase;
516 GetKickersNumFunc = &GetNumKickersAirplaneSoloComb;
517 end_length = 5;
518 end_base = kAirplaneWithPairActionBase;
519 } else {
520 kicker_type = kPair;
521 GetActionBaseFunc = &GetAirplanePairActionBase;
522 GetKickersNumFunc = &GetNumKickersAirplanePairComb;
523 end_length = 4;
524 end_base = kBombActionBase;
525 }
526 int chain_length;
527 // label the action Ids in increasing length of chain
528 for (chain_length = start_length; chain_length <= end_length;
529 ++chain_length) {
530 int start_base = GetActionBaseFunc(chain_length);
531 int next_base = chain_length == end_length
532 ? end_base
533 : GetActionBaseFunc(chain_length + 1);
534 if (start_base <= action && action < next_base) {
535 action_base = start_base;
536 break;
537 }
538 }
539 const int hand_id = (action - action_base);
540 const int num_kickers = GetKickersNumFunc(chain_length);
541 const int chain_head = hand_id / num_kickers;
542 const int kicker_steps = hand_id % num_kickers;
543 SPIEL_CHECK_FALSE(action_base == kInvalidAction);
544 return TrioCombParams(chain_head, chain_length, kicker_type, kicker_steps);
545}

547std::array<int, kNumRanks> SingleTrioCombHand(int action) {
548 std::array<int, kNumRanks> hand{};
549
550 TrioCombParams params = GetSingleTrioCombParams(action);
551
552 hand[params.chain_head] = 3;
553 const int kicker_steps = params.kicker_id;
554 int kicker_rank, counter = 0;
555
556 for (kicker_rank = 0; kicker_rank < kNumRanks; ++kicker_rank) {
557 // kicker cannot be the same rank as trio
558 if (kicker_rank == params.chain_head) continue;
559 if (counter++ == kicker_steps) break;
560 }
561
562 hand[kicker_rank] = (params.kicker_type == kSolo ? 1 : 2);
563 return hand;
564}

566int SingleTrioCombHandToActionId(absl::Span<const int> hand) {
567 int trio_rank, kicker_rank;
568 int trio_counter = 0, kicker_counter = 0;
569 for (int rank = 0; rank < kNumRanks; ++rank) {
570 if (hand[rank] == 3) {
571 trio_counter++;
572 trio_rank = rank;
573 } else if (hand[rank] == 1 || hand[rank] == 2) {
574 kicker_counter++;
575 kicker_rank = rank;
576 } else if (hand[rank] == 4) {
577 SpielFatalError("There cannot be a bomb");
578 }
579 }
580 SPIEL_CHECK_EQ(trio_counter, 1);
581 SPIEL_CHECK_EQ(kicker_counter, 1);
582
583 int action;
584 if (hand[kicker_rank] == 1)
585 action = kTrioWithSoloActionBase;
586 else
587 action = kTrioWithPairActionBase;
588 // one of the rank had already been taken by the trio
589 if (hand[kicker_rank] == 1)
590 action += trio_rank * (kNumRanks - 1);
591 else
592 action += trio_rank * (kNumRanks - 3); // the jokers cannot be the pair
593 int kicker_steps = 0;
594 for (int rank = 0; rank < kNumRanks; ++rank) {
595 if (rank == trio_rank) continue;
596 if (rank == kicker_rank) break;
597 kicker_steps++;
598 }
599 action += kicker_steps;
600 return action;
601}

603void SearchSingleTrioCombActions(std::vector<Action>* actions,
604 absl::Span<const int> hand,
605 int prev_action = kInvalidAction) {
606 TrioCombParams prev_action_params;
607 int start_rank;
608 if (prev_action == kInvalidAction) {
609 start_rank = 0;
610 } else {
611 prev_action_params = GetSingleTrioCombParams(prev_action);
612 start_rank = prev_action_params.chain_head + 1;
613 }
614 // enumerate possible trio
615 for (int rank = start_rank; rank < kNumRanks - 2; ++rank) {
616 if (hand[rank] < 3) continue;
617 for (int kicker = 0; kicker < kNumRanks; ++kicker) {
618 if (!hand[kicker] || kicker == rank) continue;
619 std::vector<KickerType> all_kicker_types;
620 if (prev_action != kInvalidAction) {
621 if (hand[kicker] >= prev_action_params.kicker_type)
622 all_kicker_types.push_back(prev_action_params.kicker_type);
623 } else {
624 for (int i = 1; i <= std::min(hand[kicker], 2); ++i)
625 all_kicker_types.push_back(static_cast<KickerType>(i));
626 }
627 for (auto n : all_kicker_types) {
628 std::array<int, kNumRanks> used_hand{};
629 used_hand[rank] = 3;
630 used_hand[kicker] = static_cast<int>(n);
631 actions->push_back(SingleTrioCombHandToActionId(used_hand));
632 }
633 }
634 }
635}

637// a dfs backtrack algorithm to compute action ids / hands for airplane
638// combinations if target_count = -1, then the goal of this algorithm is to find
639// the kicker_id of ans_hand, stored in count reference otherwise, the goal is
640// to find a hand whose kicker_id is target_count and the result hand is stored
641// in ans_hand
642bool dfs_airplane_kicker(int chain_length, int depth, int target_count,
643 int& count, int max_search_rank,
644 absl::Span<int> used_rank, absl::Span<int> ans_hand,
645 KickerType kicker_type) {
646 if (chain_length == depth) {
647 if (target_count == -1) {
648 bool found = true;
649 for (int rank = 0; rank < kNumRanks; ++rank)
650 found = found & (used_rank[rank] == ans_hand[rank]);
651 if (found) return true;
652 } else if (target_count == count) {
653 for (int rank = 0; rank < kNumRanks; ++rank)
654 ans_hand[rank] = used_rank[rank];
655 return true;
656 }
657 count++;
658 } else {
659 for (int rank = 0; rank <= max_search_rank; ++rank) {
660 SPIEL_CHECK_NE(used_rank[rank], kNumSuits);
661 if (used_rank[rank] == 3) continue;
662 if (kicker_type == kPair) {
663 SPIEL_CHECK_NE(used_rank[rank], 1);
664 if (used_rank[rank] == 2) continue;
665 }
666 if (rank == kNumRanks - 1 || rank == kNumRanks - 2) {
667 if (kicker_type == kPair) continue;
668 if (used_rank[rank]) continue;
669 // Rocket cannot be kickers
670 if (used_rank[2 * kNumRanks - 3 - rank]) continue;
671 }
672 used_rank[rank] += kicker_type == kSolo ? 1 : 2;
673 if (dfs_airplane_kicker(chain_length, depth + 1, target_count, count,
674 rank, used_rank, ans_hand, kicker_type))
675 return true;
676 used_rank[rank] -= kicker_type == kSolo ? 1 : 2;
677 }
678 }
679 return false;
680}

683 std::array<int, kNumRanks> hand{};

700// for aiplane combination, we have to specify the chain head
701// to resolve ambiguity such as 333444555666
702int AirplaneCombHandToActionId(absl::Span<const int> hand, int chain_head,
703 KickerType kicker_type) {
704 int chain_length = 0;
705 bool chain_begun = false;
706 std::vector<int> kickers;
707 for (int rank = 0; rank < kNumRanks; ++rank) {
708 SPIEL_CHECK_LT(hand[rank], kNumSuits);
709 if (!hand[rank]) continue;
710 if (!chain_begun && rank != chain_head) {
711 if (kicker_type == kSolo) {
712 for (int i = 0; i < hand[rank]; ++i) {
713 kickers.push_back(rank);
714 }
715 } else {
716 SPIEL_CHECK_EQ(hand[rank], 2);
717 kickers.push_back(rank);
718 }
719 } else if (rank == chain_head) {
720 SPIEL_CHECK_EQ(hand[rank], 3);
721 chain_begun = true;
722 chain_length++;
723 } else if (chain_begun && hand[rank] == 3) {
724 chain_length++;
725 } else if (chain_begun && hand[rank] != 3) {
726 chain_begun = false;
727 if (kicker_type == kSolo) {
728 for (int i = 0; i < hand[rank]; ++i) kickers.push_back(rank);
729 } else {
730 SPIEL_CHECK_EQ(hand[rank], 2);
731 kickers.push_back(rank);
732 }
733 }
734 }
735
736 // handle case where 333444555666 and chain_head=3
737 // in this case, the above linear scan algorithm will view 3-4-5-6 as the
738 // chain where 6s should be the kickers
739 if (chain_length - 1 == static_cast<int>(kickers.size()) + 3) {
740 chain_length--;
741 for (int i = 0; i < 3; ++i) kickers.push_back(chain_head + chain_length);
742 }
743 SPIEL_CHECK_EQ(chain_length, static_cast<int>(kickers.size()));
744
745 if (chain_head + chain_length - 1 >= kNumRanks - 3)
746 SpielFatalError("2s, Joker cannot be in a chain");
747 int action_base;
748 if (kicker_type == kSolo)
749 action_base = GetAirplaneSoloActionBase(chain_length) +
750 chain_head * GetNumKickersAirplaneSoloComb(chain_length);
751 else
752 action_base = GetAirplanePairActionBase(chain_length) +
753 chain_head * GetNumKickersAirplanePairComb(chain_length);
754
755 int count = 0;
756 std::array<int, kNumRanks> used_rank{};
757 for (int i = 0; i < chain_length; ++i) used_rank[chain_head + i] = 3;
758
759 std::array<int, kNumRanks> hand_copy{};
760 for (int i = 0; i < kNumRanks; ++i) hand_copy[i] = hand[i];
761 bool found = dfs_airplane_kicker(chain_length, 0, -1, count, kNumRanks - 1,
762 absl::MakeSpan(used_rank),
763 absl::MakeSpan(hand_copy), kicker_type);
764 SPIEL_CHECK_TRUE(found);
765
766 return action_base + count;
767}

769// a dfs backtrack algorithm that found the action ids of all possible airplane
770// combination the action ids are stored in action_ids
771void dfs_add_all_airplane_kickers(int chain_head, int chain_length, int depth,
772 int max_search_rank,
773 absl::Span<int> used_rank,
774 absl::Span<const int> ans_hand,
775 std::vector<Action>* action_ids,
776 KickerType kicker_type) {
777 if (chain_length == depth) {
778 std::array<int, kNumRanks> final_hand{};
779 for (int i = 0; i < kNumRanks; ++i) final_hand[i] = used_rank[i];
780 action_ids->push_back(static_cast<Action>(
781 AirplaneCombHandToActionId(final_hand, chain_head, kicker_type)));
782 } else {
783 for (int rank = 0; rank <= max_search_rank; ++rank) {
784 if (rank >= chain_head && rank <= chain_head + chain_length - 1) continue;
785 SPIEL_CHECK_NE(used_rank[rank], kNumSuits);
786 if (used_rank[rank] == 3) continue;
787 if (kicker_type == kPair) {
788 SPIEL_CHECK_NE(used_rank[rank], 1);
789 if (used_rank[rank] == 2) continue;
790 }
791 if (rank == kNumRanks - 1 || rank == kNumRanks - 2) {
792 if (kicker_type == kPair) continue;
793 if (used_rank[rank]) continue;
794 if (used_rank[2 * kNumRanks - 3 - rank]) continue;
795 }
796 int num_use_cards = kicker_type == kSolo ? 1 : 2;
797 if (ans_hand[rank] < num_use_cards + used_rank[rank]) continue;
798 used_rank[rank] += num_use_cards;
799 dfs_add_all_airplane_kickers(chain_head, chain_length, depth + 1, rank,
800 used_rank, ans_hand, action_ids,
801 kicker_type);
802 used_rank[rank] -= num_use_cards;
803 }
804 }
805}

807void SearchAirplaneCombActions(std::vector<Action>* actions,
808 absl::Span<const int> hand,
809 int prev_action = kInvalidAction) {
810 TrioCombParams prev_action_params;
811 int start_rank;
812 if (prev_action == kInvalidAction) {
813 start_rank = 0;
814 } else {
815 prev_action_params = GetAirplaneCombParams(prev_action);
816 start_rank = prev_action_params.chain_head + 1;
817 }
818 for (int chain_head = start_rank; chain_head < kNumRanks - 4; ++chain_head) {
819 if (hand[chain_head] < 3) continue;
820 int num_cards = hand[chain_head];
821 for (int chain_length = 2; chain_head + chain_length - 1 < kNumRanks - 3;
822 ++chain_length) {
823 int chain_tail = chain_head + chain_length - 1;
824 num_cards = std::min(num_cards, hand[chain_tail]);
825 if (num_cards < 3) break;
826 std::vector<KickerType> all_kicker_types;
827 if (prev_action != kInvalidAction) {
828 if (chain_length > prev_action_params.chain_length) break;
829 if (chain_length == prev_action_params.chain_length) {
830 all_kicker_types.push_back(prev_action_params.kicker_type);
831 }
832 } else {
833 all_kicker_types.push_back(kSolo);
834 all_kicker_types.push_back(kPair);
835 }
836 for (auto kicker_type : all_kicker_types) {
837 std::array<int, kNumRanks> used_hand{};
838 for (int i = 0; i < chain_length; ++i) used_hand[chain_head + i] = 3;
839 dfs_add_all_airplane_kickers(chain_head, chain_length, 0, kNumRanks - 1,
840 absl::MakeSpan(used_hand),
841 absl::MakeSpan(hand), actions,
842 kicker_type);
843 }
844 }
845 }

848std::array<int, kNumRanks> ActionToHand(int action) {
849 std::array<int, kNumRanks> hand{};
850 if ((action >= kPlayActionBase && action < kSoloChainActionBase) ||
851 (action >= kPairActionBase && action < kPairChainActionBase) ||
852 (action >= kTrioActionBase && action < kTrioWithSoloActionBase) ||
853 (action >= kBombActionBase && action < kRocketActionBase)) {
854 hand = SingleRankHand(action);
855 } else if ((action >= kSoloChainActionBase && action < kPairActionBase) ||
856 (action >= kPairChainActionBase && action < kTrioActionBase) ||
857 (action >= kAirplaneActionBase &&
858 action < kAirplaneWithSoloActionBase)) {
859 hand = ChainOnlyHand(action);
860 } else if (action >= kTrioWithSoloActionBase &&
861 action < kAirplaneActionBase) {
862 hand = SingleTrioCombHand(action);
863 } else if (action >= kAirplaneWithSoloActionBase &&
864 action < kBombActionBase) {
865 hand = AirplaneCombHand(action);
866 } else if (action == kRocketActionBase) {
867 hand[kNumRanks - 1] = hand[kNumRanks - 2] = 1;
868 } else {
869 SpielFatalError("Non valid Action Ids");
870 }
871 return hand;

874void SearchForLegalActions(std::vector<Action>* legal_actions,
875 absl::Span<const int> hand, int prev_action) {
876 if (hand[kNumRanks - 2] && hand[kNumRanks - 1])
877 legal_actions->push_back(kRocketActionBase);
878 if (prev_action == kInvalidAction) {
879 // search for all possible actions
880 SearchSingleRankActions(legal_actions, hand, prev_action);
881 SearchChainOnlyActions(legal_actions, hand, prev_action);
882 SearchSingleTrioCombActions(legal_actions, hand, prev_action);
883 SearchAirplaneCombActions(legal_actions, hand, prev_action);
884 } else if (prev_action >= kBombActionBase &&
885 prev_action < kRocketActionBase) {
886 // if previous action is a bomb, then only higher bomb or rocket can be
887 // played
888 SearchSingleRankActions(legal_actions, hand, prev_action);
889 } else {
890 // check for bombs
891 for (int rank = 0; rank < kNumRanks - 2; ++rank) {
892 if (hand[rank] == kNumSuits) {
893 std::array<int, kNumRanks> used_rank{};
894 used_rank[rank] = kNumSuits;
895 legal_actions->push_back(SingleRankHandToActionId(used_rank));
896 }
897 }
898
899 // then search within each category
900 if ((prev_action >= kPlayActionBase &&
901 prev_action < kSoloChainActionBase) ||
902 (prev_action >= kPairActionBase &&
903 prev_action < kPairChainActionBase) ||
904 (prev_action >= kTrioActionBase &&
905 prev_action < kTrioWithSoloActionBase)) {
906 SearchSingleRankActions(legal_actions, hand, prev_action);
907 } else if ((prev_action >= kSoloChainActionBase &&
908 prev_action < kPairActionBase) ||
909 (prev_action >= kPairChainActionBase &&
910 prev_action < kTrioActionBase) ||
911 (prev_action >= kAirplaneActionBase &&
912 prev_action < kAirplaneWithSoloActionBase)) {
913 SearchChainOnlyActions(legal_actions, hand, prev_action);
914 } else if (prev_action >= kTrioWithSoloActionBase &&
915 prev_action < kAirplaneActionBase) {
916 SearchSingleTrioCombActions(legal_actions, hand, prev_action);
917 } else if (prev_action >= kAirplaneWithSoloActionBase &&
918 prev_action < kBombActionBase) {
919 SearchAirplaneCombActions(legal_actions, hand, prev_action);
920 } else if (prev_action == kRocketActionBase) {
921 } else {
922 SpielFatalError("Previous actions invalid");
923 }
924 }
