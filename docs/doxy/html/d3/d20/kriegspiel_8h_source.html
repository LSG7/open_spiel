32// Kriegspiel - imperfect information version of chess:
33// https://en.wikipedia.org/wiki/Kriegspiel
34// The implementation follows ICC rules (with a few exceptions):
35// https://www.chessclub.com/help/Kriegspiel
36// One of the exceptions is that ICC does not notify opponent about player's
37// illegal move. Here he has to be notified about it because tests don't allow
38// player to not recognise the difference between states with different move
39// number. And Illegal attempt is considered a move.
40// Other exceptions are 50-move rule and threefold repetition, which under ICC
41// rules are not automatically enforced, but can be claimed by the player. This
42// implementation does not support claiming or offering draws so these rules'
43// automatic enforcement can be turned on and off
44//
45// Parameters:
46// "board_size" int Number of squares in each row and column
47// (default: 8)
48// "fen" string String describing the chess board position
49// in Forsyth-Edwards Notation. The FEN has to
50// match the board size. Default values are
51// available for board sizes 4 and 8.
52// "threefold_repetition" bool Whether threefold repetition rule should be
53// automatically enforced (default: true)
54// "50_move_rule" bool Whether 50 move rule should be automatically
55// enforced (default: true)
56

92struct KriegspielUmpireMessage {
93 bool illegal = false;
94 KriegspielCaptureType capture_type = KriegspielCaptureType::kNoCapture;
95 chess::Square square = chess::kInvalidSquare;
96 // there can be max two checks at a time so a pair is enough
97 std::pair<KriegspielCheckType, KriegspielCheckType> check_types = {
98 KriegspielCheckType::kNoCheck, KriegspielCheckType::kNoCheck};
99 chess::Color to_move = chess::Color::kEmpty;
100 int pawn_tries = 0;
101
102 std::string ToString() const;
103
104 bool operator==(KriegspielUmpireMessage& other) const {
105 return illegal == other.illegal && capture_type == other.capture_type &&
106 square == other.square && check_types == other.check_types &&
107 to_move == other.to_move && pawn_tries == other.pawn_tries;
108 }
109};
