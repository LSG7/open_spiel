79 double TabularBestResponse::HandleDecisionCase(HistoryNode* node) {
80  if (node == nullptr) SpielFatalError("HandleDecisionCase: node is null.");
81  if (node->GetState()->CurrentPlayer() == best_responder_) {
82  // If we're playing as the best responder, we look at every child node,
83  if (action_value_tolerance_ < 0) {
84  // Pick the one with the highest expected utility to play.
85  BestResponseAction(node->GetInfoState());
86  } else {
87  // Or spread support over all best_actions.
88  BestResponseActions(node->GetInfoState(), action_value_tolerance_);
89  }
90 
91  auto action_prob = best_response_policy_[node->GetInfoState()];
92  double value = 0.0;
93  for (const auto& [action, prob] : action_prob) {
94  HistoryNode* child = node->GetChild(action).second;
95  if (child == nullptr)
96  SpielFatalError("HandleDecisionCase: node is null.");
97  double child_value = Value(child->GetHistory());
98  value += child_value * prob;
99  }
100  return value;
101  }
102  // If the other player is playing, then we can recursively compute the
103  // expected utility of that node by looking at their policy.
104  // We take child probabilities from the policy as that is what we are
105  // calculating a best response to.
106  ActionsAndProbs state_policy = policy_->GetStatePolicy(*node->GetState());
107  if (state_policy.empty())
108  SpielFatalError(absl::StrCat("InfoState ", node->GetInfoState(),
109  " not found in policy."));
110  if (state_policy.size() > node->NumChildren()) {
111  int num_zeros = 0;
112  for (const auto& a_and_p : state_policy) {
113  if (Near(a_and_p.second, 0.)) ++num_zeros;
114  }
115  // We check here that the policy is valid, i.e. that it doesn't contain
116  // too many (invalid) actions. This can only happen when the policy is
117  // built incorrectly. If this is failing, you are building the policy
118  // wrong.
119  if (state_policy.size() > node->NumChildren() + num_zeros) {
120  std::vector<std::string> action_probs_str_vector;
121  action_probs_str_vector.reserve(state_policy.size());
122  for (const auto& action_prob : state_policy) {
123  // TODO(b/127423396): Use absl::StrFormat.
124  action_probs_str_vector.push_back(absl::StrCat(
125  "(", action_prob.first, ", ", action_prob.second, ")"));
126  }
127  std::string action_probs_str =
128  absl::StrJoin(action_probs_str_vector, " ");
129  SpielFatalError(absl::StrCat(
130  "Policies don't match in size, in state ",
131  node->GetState()->HistoryString(), ".\nThe tree has '",
132  node->NumChildren(), "' valid children, but ", state_policy.size(),
133  " valid (action, prob) are available: [", action_probs_str, "]"));
134  }
135  }
136  double value = 0;
137  for (const auto& action : node->GetState()->LegalActions()) {
138  const double prob = GetProb(state_policy, action);
139  if (prob <= prob_cut_threshold_) continue;
140  // We discard the probability here that's returned by GetChild as we
141  // immediately load the probability for the given child from the policy.
142  HistoryNode* child = node->GetChild(action).second;
143  if (child == nullptr) SpielFatalError("HandleDecisionCase: node is null.");
144  // Finally, we update value by the policy weighted value of the child.
145  SPIEL_CHECK_PROB_TOLERANCE(prob, ProbabilityDefaultTolerance());
146  value += prob * Value(child->GetHistory());
147  }
148  return value;
149 }
150 double TabularBestResponse::HandleChanceCase(HistoryNode* node) {
151  double value = 0;
152  double prob_sum = 0;
153  for (const auto& action : node->GetChildActions()) {
154  std::pair<double, HistoryNode*> prob_and_child = node->GetChild(action);
155  double prob = prob_and_child.first;
156  prob_sum += prob;
157  if (prob <= prob_cut_threshold_) continue;
158  HistoryNode* child = prob_and_child.second;
159  if (child == nullptr) SpielFatalError("Child is null.");
160  // Verify that the probability is valid. This should always be true.
161  SPIEL_CHECK_PROB_TOLERANCE(prob, ProbabilityDefaultTolerance());
162  value += prob * Value(child->GetHistory());
163  }
164  // Verify that the sum of the probabilities is 1, within tolerance.
165  SPIEL_CHECK_FLOAT_EQ(prob_sum, 1.0);
166  return value;
167 }
168 double TabularBestResponse::Value(const std::string& history) {
169  auto it = value_cache_.find(history);
170  if (it != value_cache_.end()) return it->second;
171  HistoryNode* node = tree_.GetByHistory(history);
172  if (node == nullptr) SpielFatalError("node returned is null.");
173  double cache_value = 0;
174  switch (node->GetType()) {
175  case StateType::kTerminal: {
176  cache_value = HandleTerminalCase(*node);
177  break;
178  }
179  case StateType::kDecision: {
180  cache_value = HandleDecisionCase(node);
181  break;
182  }
183  case StateType::kChance: {
184  cache_value = HandleChanceCase(node);
185  break;
186  }
187  case StateType::kMeanField: {
188  SpielFatalError("kMeanField not supported.");
189  }
190  }
191  value_cache_[history] = cache_value;
192  return value_cache_[history];
193 }
194 Action TabularBestResponse::BestResponseAction(const std::string& infostate) {
195  auto it = best_response_policy_.find(infostate);
196  if (it != best_response_policy_.end()) return it->second.begin()->first;
197  std::vector<std::pair<HistoryNode*, double>> infoset = infosets_[infostate];
198  Action best_action = -1;
199  double best_value = std::numeric_limits<double>::lowest();
200  // The legal actions are the same for all children, so we arbitrarily pick
201  // the first one to get the legal actions from.
202  for (const auto& action : infoset[0].first->GetChildActions()) {
203  double value = 0;
204  // Prob here is the counterfactual reach-weighted probability.
205  for (const auto& state_and_prob : infoset) {
206  if (state_and_prob.second <= prob_cut_threshold_) continue;
207  HistoryNode* state_node = state_and_prob.first;
208  HistoryNode* child_node = state_node->GetChild(action).second;
209  SPIEL_CHECK_TRUE(child_node != nullptr);
210  value += state_and_prob.second * Value(child_node->GetHistory());
211  }
212  if (value > best_value) {
213  best_value = value;
214  best_action = action;
215  }
216  }
217  if (best_action == -1) SpielFatalError("No action was chosen.");
218 
219  ActionsAndProbs actions_and_probs;
220  for (const auto& action : infoset[0].first->GetChildActions()) {
221  double prob = 0.0;
222  if (action == best_action) prob = 1.0;
223  actions_and_probs.push_back(std::make_pair(action, prob));
224  }
225  best_response_policy_[infostate] = actions_and_probs;
226  best_response_actions_[infostate] = best_action;
227  return best_action;
228 }
229 std::vector<Action> TabularBestResponse::BestResponseActions(
230  const std::string& infostate, double tolerance) {
231  absl::btree_set<Action> best_actions;
232  std::vector<std::pair<Action, double>> action_values;
233  std::vector<std::pair<HistoryNode*, double>> infoset =
234  infosets_.at(infostate);
235  double best_value = std::numeric_limits<double>::lowest();
236  // The legal actions are the same for all children, so we arbitrarily pick
237  // the first one to get the legal actions from.
238  for (const Action& action : infoset[0].first->GetChildActions()) {
239  double value = 0;
240  // Prob here is the counterfactual reach-weighted probability.
241  for (const auto& [state_node, prob] : infoset) {
242  if (prob <= prob_cut_threshold_) continue;
243  HistoryNode* child_node = state_node->GetChild(action).second;
244  SPIEL_CHECK_TRUE(child_node != nullptr);
245  value += prob * Value(child_node->GetHistory());
246  }
247  action_values.push_back({action, value});
248  if (value > best_value) {
249  best_value = value;
250  }
251  }
252  for (const auto& [action, value] : action_values) {
253  if (value >= best_value - tolerance) {
254  best_actions.insert(action);
255  }
256  }
257  if (best_actions.empty()) SpielFatalError("No action was chosen.");
258  ActionsAndProbs actions_and_probs;
259  for (const auto& action : infoset[0].first->GetChildActions()) {
260  double prob = 0.0;
261  if (best_actions.count(action)) {
262  prob = 1.0 / best_actions.size();
263  }
264  actions_and_probs.push_back(std::make_pair(action, prob));
265  }
266  best_response_policy_[infostate] = actions_and_probs;
267  return std::vector<Action>(best_actions.begin(), best_actions.end());
268 }
269 std::vector<std::pair<Action, double>>
270 TabularBestResponse::BestResponseActionValues(const std::string& infostate) {
271  std::vector<std::pair<Action, double>> action_values;
272  std::vector<std::pair<HistoryNode*, double>> infoset =
273  infosets_.at(infostate);
274  action_values.reserve(infoset[0].first->GetChildActions().size());
275  for (Action action : infoset[0].first->GetChildActions()) {
276  double value = 0;
277  double normalizer = 0;
278  // Prob here is the counterfactual reach-weighted probability.
279  for (const auto& [state_node, prob] : infoset) {
280  if (prob <= prob_cut_threshold_) continue;
281  HistoryNode* child_node = state_node->GetChild(action).second;
282  SPIEL_CHECK_TRUE(child_node != nullptr);
283  value += prob * Value(child_node->GetHistory());
284  normalizer += prob;
285  }
286  SPIEL_CHECK_GT(normalizer, 0);
287  action_values.push_back({action, value / normalizer});
288  }
289  return action_values;
290 }
