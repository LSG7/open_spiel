30 class SoftmaxPolicyTest(parameterized.TestCase):

32  @parameterized.named_parameters(('python', 'python_mfg_crowd_modelling'),
33  ('cpp', 'mfg_crowd_modelling'))
34  def test_softmax(self, name):
35  """Check if the softmax policy works as expected.
36 
37  The test checks that:
38  - uniform prior policy gives the same results than no prior.
39  - very high temperature gives almost a uniform policy.
40  - very low temperature gives almost a deterministic policy for the best
41  action.
42 
43  Args:
44  name: Name of the game.
45  """
46 
47  game = pyspiel.load_game(name)
48  uniform_policy = policy.UniformRandomPolicy(game)
49  dist = distribution.DistributionPolicy(game, uniform_policy)
50  br_value = best_response_value.BestResponse(
51  game, dist, value.TabularValueFunction(game))
52  br_init_val = br_value(game.new_initial_state())
53 
54  # uniform prior policy gives the same results than no prior.
55  softmax_pi_uniform_prior = softmax_policy.SoftmaxPolicy(
56  game, None, 1.0, br_value, uniform_policy).to_tabular()
57  softmax_pi_uniform_prior_value = policy_value.PolicyValue(
58  game, dist, softmax_pi_uniform_prior, value.TabularValueFunction(game))
59  softmax_pi_uniform_prior_init_val = softmax_pi_uniform_prior_value(
60  game.new_initial_state())
61  softmax_pi_no_prior = softmax_policy.SoftmaxPolicy(game, None, 1.0,
62  br_value, None)
63  softmax_pi_no_prior_value = policy_value.PolicyValue(
64  game, dist, softmax_pi_no_prior, value.TabularValueFunction(game))
65  softmax_pi_no_prior_init_val = softmax_pi_no_prior_value(
66  game.new_initial_state())
67 
68  self.assertAlmostEqual(softmax_pi_uniform_prior_init_val,
69  softmax_pi_no_prior_init_val)
70 
71  # very high temperature gives almost a uniform policy.
72  uniform_policy = uniform_policy.to_tabular()
73  uniform_value = policy_value.PolicyValue(game, dist, uniform_policy,
74  value.TabularValueFunction(game))
75  uniform_init_val = uniform_value(game.new_initial_state())
76 
77  softmax_pi_no_prior = softmax_policy.SoftmaxPolicy(game, None, 100000000,
78  br_value, None)
79  softmax_pi_no_prior_value = policy_value.PolicyValue(
80  game, dist, softmax_pi_no_prior, value.TabularValueFunction(game))
81  softmax_pi_no_prior_init_val = softmax_pi_no_prior_value(
82  game.new_initial_state())
83 
84  self.assertAlmostEqual(uniform_init_val, softmax_pi_no_prior_init_val)
85 
86  # very low temperature gives almost a best response policy.
87  softmax_pi_no_prior = softmax_policy.SoftmaxPolicy(game, None, 0.0001,
88  br_value, None)
89  softmax_pi_no_prior_value = policy_value.PolicyValue(
90  game, dist, softmax_pi_no_prior, value.TabularValueFunction(game))
91  softmax_pi_no_prior_init_val = softmax_pi_no_prior_value(
92  game.new_initial_state())
93 
94  self.assertAlmostEqual(br_init_val, softmax_pi_no_prior_init_val)
