18 // Full implementation of the classic trick taking game Hearts.
19 // https://www.pagat.com/reverse/hearts.html
20 //
21 // Some notes on this implementation:
22 //
23 // - Pass Direction
24 // The direction of the pass is decided by the first chance action. If the
25 // "pass_cards" game parameter is set to false, the "No Pass" action will be
26 // the only legal action at the first chance node.
27 //
28 // In standard play, the direction of the pass alternates in
29 // a fixed pattern. Here, however, state is not preserved between hands, so
30 // the game itself cannot enforce that pattern. By using the first chance
31 // action to set the pass direction, the game can be dropped in to pre-existing
32 // algorithms without requiring modifications to coordinate training.
33 //
34 // - Returns
35 // Hearts is a trick-avoidance game in which the goal is to accumulate the
36 // fewest number of points. Because RL algorithms are designed to maximize
37 // reward, returns are calculated by subtracting the in-game points from an
38 // upper bound.

74 enum class Suit { kClubs = 0, kDiamonds = 1, kHearts = 2, kSpades = 3 };
75 enum class PassDir { kNoPass = 0, kLeft = 1, kAcross = 2, kRight = 3 };
76 enum Seat { kNorth, kEast, kSouth, kWest };
77 // Cards are represented as rank * kNumSuits + suit.
78 inline Suit CardSuit(int card) { return Suit(card % kNumSuits); }
79 inline int CardRank(int card) { return card / kNumSuits; }
80 inline int Card(Suit suit, int rank) {
81  return rank * kNumSuits + static_cast<int>(suit);
82 }
83 inline int CardPoints(int card, bool jd_bonus) {
84  if (CardSuit(card) == Suit::kHearts) {
85  return kPointsForHeart;
86  } else if (card == Card(Suit::kSpades, 10)) {
87  return kPointsForQS;
88  } else if (card == Card(Suit::kDiamonds, 9) && jd_bonus) {
89  return kPointsForJD;
90  } else {
91  return 0;
92  }
93 }
