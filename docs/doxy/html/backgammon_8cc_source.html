151int BackgammonState::AugmentCheckerMove(CheckerMove* cmove, int player,
152 int start) const {
153 int end = cmove->num;
154 if (end != kPassPos) {
155 // Not a pass, so work out where the piece finished
156 end = start - cmove->num;
157 if (end <= 0) {
158 end = kNumOffPosHumanReadable; // Off
159 } else if (board_[Opponent(player)]
160 [player == kOPlayerId ? (end - 1) : (kNumPoints - end)] ==
161 1) {
162 cmove->hit = true; // Check to see if move is a hit
163 }
164 }
165 return end;
166}

168std::string BackgammonState::ActionToString(Player player,
169 Action move_id) const {
170 if (player == kChancePlayerId) {
171 if (turns_ >= 0) {
172 // Normal chance roll.
173 return absl::StrCat("chance outcome ", move_id,
174 " (roll: ", kChanceOutcomeValues[move_id][0],
175 kChanceOutcomeValues[move_id][1], ")");
176 } else {
177 // Initial roll to determine who starts.
178 const char* starter = (move_id < kNumNonDoubleOutcomes ?
179 "X starts" : "O starts");
180 if (move_id >= kNumNonDoubleOutcomes) {
181 move_id -= kNumNonDoubleOutcomes;
182 }
183 return absl::StrCat("chance outcome ", move_id, " ", starter, ", ",
184 "(roll: ", kChanceOutcomeValues[move_id][0],
185 kChanceOutcomeValues[move_id][1], ")");
186 }
187 } else {
188 // Assemble a human-readable string representation of the move using
189 // standard backgammon notation:
190 //
191 // - Always show the numbering going from Bar->24->0->Off, irrespective of
192 // which player is moving.
193 // - Show the start position followed by end position.
194 // - Show hits with an asterisk, e.g. 9/7*.
195 // - Order the moves by highest number first, e.g. 22/7 10/8 not 10/8 22/7.
196 // Not an official requirement, but seems to be standard convention.
197 // - Show duplicate moves as 10/8(2).
198 // - Show moves on a single piece as 10/8/5 not 10/8 8/5
199 //
200 // Note that there are tests to ensure the ActionToString follows this
201 // output format. Any changes would need to be reflected in the tests as
202 // well.
203 std::vector<CheckerMove> cmoves = SpielMoveToCheckerMoves(player, move_id);
204
205 int cmove0_start;
206 int cmove1_start;
207 if (player == kOPlayerId) {
208 cmove0_start = (cmoves[0].pos == kBarPos ? kNumBarPosHumanReadable
209 : cmoves[0].pos + 1);
210 cmove1_start = (cmoves[1].pos == kBarPos ? kNumBarPosHumanReadable
211 : cmoves[1].pos + 1);
212 } else {
213 // swap the board numbering round for Player X so player is moving
214 // from 24->0
215 cmove0_start = (cmoves[0].pos == kBarPos ? kNumBarPosHumanReadable
216 : kNumPoints - cmoves[0].pos);
217 cmove1_start = (cmoves[1].pos == kBarPos ? kNumBarPosHumanReadable
218 : kNumPoints - cmoves[1].pos);
219 }
220
221 // Add hit information and compute whether the moves go off the board.
222 int cmove0_end = AugmentCheckerMove(&cmoves[0], player, cmove0_start);
223 int cmove1_end = AugmentCheckerMove(&cmoves[1], player, cmove1_start);
224
225 // check for 2 pieces hitting on the same point.
226 bool double_hit =
227 (cmoves[1].hit && cmoves[0].hit && cmove1_end == cmove0_end);
228
229 std::string returnVal = "";
230 if (cmove0_start == cmove1_start &&
231 cmove0_end == cmove1_end) { // same move, show as (2).
232 if (cmoves[1].num == kPassPos) { // Player can't move at all!
233 returnVal = "Pass";
234 } else {
235 returnVal = absl::StrCat(move_id, " - ",
236 PositionToStringHumanReadable(cmove0_start),
237 "/", PositionToStringHumanReadable(cmove0_end),
238 cmoves[0].hit ? "*" : "", "(2)");
239 }
240 } else if ((cmove0_start < cmove1_start ||
241 (cmove0_start == cmove1_start && cmove0_end < cmove1_end) ||
242 cmoves[0].num == kPassPos) &&
243 cmoves[1].num != kPassPos) {
244 // tradition to start with higher numbers first,
245 // so swap moves round if this not the case. If
246 // there is a pass move, put it last.
247 if (cmove1_end == cmove0_start) {
248 // Check to see if the same piece is moving for both
249 // moves, as this changes the format of the output.
250 returnVal = absl::StrCat(
251 move_id, " - ", PositionToStringHumanReadable(cmove1_start), "/",
252 PositionToStringHumanReadable(cmove1_end), cmoves[1].hit ? "*" : "",
253 "/", PositionToStringHumanReadable(cmove0_end),
254 cmoves[0].hit ? "*" : "");
255 } else {
256 returnVal = absl::StrCat(
257 move_id, " - ", PositionToStringHumanReadable(cmove1_start), "/",
258 PositionToStringHumanReadable(cmove1_end), cmoves[1].hit ? "*" : "",
259 " ",
260 (cmoves[0].num != kPassPos)
261 ? PositionToStringHumanReadable(cmove0_start)
262 : "",
263 (cmoves[0].num != kPassPos) ? "/" : "",
264 PositionToStringHumanReadable(cmove0_end),
265 (cmoves[0].hit && !double_hit) ? "*" : "");
266 }
267 } else {
268 if (cmove0_end == cmove1_start) {
269 // Check to see if the same piece is moving for both
270 // moves, as this changes the format of the output.
271 returnVal = absl::StrCat(
272 move_id, " - ", PositionToStringHumanReadable(cmove0_start), "/",
273 PositionToStringHumanReadable(cmove0_end), cmoves[0].hit ? "*" : "",
274 "/", PositionToStringHumanReadable(cmove1_end),
275 cmoves[1].hit ? "*" : "");
276 } else {
277 returnVal = absl::StrCat(
278 move_id, " - ", PositionToStringHumanReadable(cmove0_start), "/",
279 PositionToStringHumanReadable(cmove0_end), cmoves[0].hit ? "*" : "",
280 " ",
281 (cmoves[1].num != kPassPos)
282 ? PositionToStringHumanReadable(cmove1_start)
283 : "",
284 (cmoves[1].num != kPassPos) ? "/" : "",
285 PositionToStringHumanReadable(cmove1_end),
286 (cmoves[1].hit && !double_hit) ? "*" : "");
287 }
288 }
289
290 return returnVal;
291 }

300void BackgammonState::ObservationTensor(Player player,
301 absl::Span<float> values) const {
302 SPIEL_CHECK_GE(player, 0);
303 SPIEL_CHECK_LT(player, num_players_);
304
305 int opponent = Opponent(player);
306 SPIEL_CHECK_EQ(values.size(), kStateEncodingSize);
307 auto value_it = values.begin();
308 // The format of this vector is described in Section 3.4 of "G. Tesauro,
309 // Practical issues in temporal-difference learning, 1994."
310 // https://link.springer.com/article/10.1007/BF00992697
311 // The values of the dice are added in the last two positions of the vector.
312 for (int count : board_[player]) {
313 *value_it++ = ((count == 1) ? 1 : 0);
314 *value_it++ = ((count == 2) ? 1 : 0);
315 *value_it++ = ((count == 3) ? 1 : 0);
316 *value_it++ = ((count > 3) ? (count - 3) : 0);
317 }
318 for (int count : board_[opponent]) {
319 *value_it++ = ((count == 1) ? 1 : 0);
320 *value_it++ = ((count == 2) ? 1 : 0);
321 *value_it++ = ((count == 3) ? 1 : 0);
322 *value_it++ = ((count > 3) ? (count - 3) : 0);
323 }
324 *value_it++ = (bar_[player]);
325 *value_it++ = (scores_[player]);
326 *value_it++ = ((cur_player_ == player) ? 1 : 0);
327
328 *value_it++ = (bar_[opponent]);
329 *value_it++ = (scores_[opponent]);
330 *value_it++ = ((cur_player_ == opponent) ? 1 : 0);
331
332 *value_it++ = ((!dice_.empty()) ? dice_[0] : 0);
333 *value_it++ = ((dice_.size() > 1) ? dice_[1] : 0);
334
335 SPIEL_CHECK_EQ(value_it, values.end());
336}

402 SPIEL_CHECK_GE(i, 0);
403 SPIEL_CHECK_LT(i, dice_.size());
404

416void BackgammonState::DoApplyAction(Action move) {
417 if (IsChanceNode()) {
418 turn_history_info_.push_back(TurnHistoryInfo(kChancePlayerId, prev_player_,
419 dice_, move, double_turn_,
420 false, false));
421
422 if (turns_ == -1) {
423 // The first chance node determines who goes first: X or O.
424 // The move is between 0 and 29 and the range determines whether X starts
425 // or O starts. The value is then converted to a number between 0 and 15,
426 // which represents the non-double chance outcome that the first player
427 // starts with (see RollDice(move) below). These 30 possibilities are
428 // constructed in GetChanceOutcomes().
429 SPIEL_CHECK_TRUE(dice_.empty());
430 if (move < kNumNonDoubleOutcomes) {
431 // X starts.
432 cur_player_ = prev_player_ = kXPlayerId;
433 } else {
434 // O Starts
435 cur_player_ = prev_player_ = kOPlayerId;
436 move -= kNumNonDoubleOutcomes;
437 }
438 RollDice(move);
439 turns_ = 0;
440 return;
441 } else {
442 // Normal chance node.
443 SPIEL_CHECK_TRUE(dice_.empty());
444 RollDice(move);
445 cur_player_ = Opponent(prev_player_);
446 return;
447 }
448 }
449
450 // Normal move action.
451 std::vector<CheckerMove> moves = SpielMoveToCheckerMoves(cur_player_, move);
452 bool first_move_hit = ApplyCheckerMove(cur_player_, moves[0]);
453 bool second_move_hit = ApplyCheckerMove(cur_player_, moves[1]);
454
455 turn_history_info_.push_back(
456 TurnHistoryInfo(cur_player_, prev_player_, dice_, move, double_turn_,
457 first_move_hit, second_move_hit));
458
459 if (!double_turn_) {
460 turns_++;
461 if (cur_player_ == kXPlayerId) {
462 x_turns_++;
463 } else if (cur_player_ == kOPlayerId) {
464 o_turns_++;
465 }
466 }
467
468 prev_player_ = cur_player_;
469
470 // Check for doubles.
471 bool extra_turn = false;
472 if (!double_turn_ && dice_[0] == dice_[1]) {
473 // Check the dice, and unuse them if they are used.
474 int dice_used = 0;
475 for (int i = 0; i < 2; i++) {
476 if (dice_[i] > 6) {
477 dice_[i] -= 6;
478 dice_used++;
479 }
480 SPIEL_CHECK_GE(dice_[i], 1);
481 SPIEL_CHECK_LE(dice_[i], 6);
482 }
483
484 if (dice_used == 2) {
485 extra_turn = true;
486 }
487 }
488
489 if (extra_turn) {
490 // Dice have been unused above.
491 double_turn_ = true;
492 } else {
493 cur_player_ = kChancePlayerId;
494 dice_.clear();
495 double_turn_ = false;
496 }
497}

499void BackgammonState::UndoAction(int player, Action action) {
500 {
501 const TurnHistoryInfo& thi = turn_history_info_.back();
502 SPIEL_CHECK_EQ(thi.player, player);
503 SPIEL_CHECK_EQ(action, thi.action);
504 cur_player_ = thi.player;
505 prev_player_ = thi.prev_player;
506 dice_ = thi.dice;
507 double_turn_ = thi.double_turn;
508 if (player != kChancePlayerId) {
509 std::vector<CheckerMove> moves = SpielMoveToCheckerMoves(player, action);
510 SPIEL_CHECK_EQ(moves.size(), 2);
511 moves[0].hit = thi.first_move_hit;
512 moves[1].hit = thi.second_move_hit;
513 UndoCheckerMove(player, moves[1]);
514 UndoCheckerMove(player, moves[0]);
515 turns_--;
516 if (!double_turn_) {
517 if (player == kXPlayerId) {
518 x_turns_--;
519 } else if (player == kOPlayerId) {
520 o_turns_--;
521 }
522 }
523 }
524 }
525 turn_history_info_.pop_back();
526 history_.pop_back();
527 --move_number_;
528}

554Action BackgammonState::CheckerMovesToSpielMove(
555 const std::vector<CheckerMove>& moves) const {
556 SPIEL_CHECK_LE(moves.size(), 2);
557 int dig0 = EncodedPassMove();
558 int dig1 = EncodedPassMove();
559 bool high_roll_first = false;
560 int high_roll = DiceValue(0) >= DiceValue(1) ? DiceValue(0) : DiceValue(1);
561
562 if (!moves.empty()) {
563 int pos1 = moves[0].pos;
564 if (pos1 == kBarPos) {
565 pos1 = EncodedBarMove();
566 }
567 if (pos1 != kPassPos) {
568 int num1 = moves[0].num;
569 dig0 = pos1;
570 high_roll_first = num1 == high_roll;
571 }
572 }
573
574 if (moves.size() > 1) {
575 int pos2 = moves[1].pos;
576 if (pos2 == kBarPos) {
577 pos2 = EncodedBarMove();
578 }
579 if (pos2 != kPassPos) {
580 dig1 = pos2;
581 }
582 }
583
584 Action move = dig1 * 26 + dig0;
585 if (!high_roll_first) {
586 move += 676; // 26**2
587 }
588 SPIEL_CHECK_GE(move, 0);
589 SPIEL_CHECK_LT(move, kNumDistinctActions);
590 return move;
591}

593std::vector<CheckerMove> BackgammonState::SpielMoveToCheckerMoves(
594 int player, Action spiel_move) const {
595 SPIEL_CHECK_GE(spiel_move, 0);
596 SPIEL_CHECK_LT(spiel_move, kNumDistinctActions);
597
598 bool high_roll_first = spiel_move < 676;
599 if (!high_roll_first) {
600 spiel_move -= 676;
601 }
602
603 std::vector<Action> digits = {spiel_move % 26, spiel_move / 26};
604 std::vector<CheckerMove> cmoves;
605 int high_roll = DiceValue(0) >= DiceValue(1) ? DiceValue(0) : DiceValue(1);
606 int low_roll = DiceValue(0) < DiceValue(1) ? DiceValue(0) : DiceValue(1);
607
608 for (int i = 0; i < 2; ++i) {
609 SPIEL_CHECK_GE(digits[i], 0);
610 SPIEL_CHECK_LE(digits[i], 25);
611
612 int num = -1;
613 if (i == 0) {
614 num = high_roll_first ? high_roll : low_roll;
615 } else {
616 num = high_roll_first ? low_roll : high_roll;
617 }
618 SPIEL_CHECK_GE(num, 1);
619 SPIEL_CHECK_LE(num, 6);
620
621 if (digits[i] == EncodedPassMove()) {
622 cmoves.push_back(CheckerMove(kPassPos, -1, false));
623 } else {
624 cmoves.push_back(CheckerMove(
625 digits[i] == EncodedBarMove() ? kBarPos : digits[i], num, false));
626 }
627 }
628
629 return cmoves;
630}

632std::vector<CheckerMove> BackgammonState::AugmentWithHitInfo(

802// Basic from_to check (including bar checkers).
803bool BackgammonState::IsLegalFromTo(int player, int from_pos, int to_pos,
804 int my_checkers_from,
805 int opp_checkers_to) const {
806 // Must have at least one checker the from position.
807 if (my_checkers_from == 0) {
808 return false;
809 }
810
811 if (opp_checkers_to > 1) {
812 return false;
813 }
814
815 // Quick validity checks out of the way. This appears to be a valid move.
816 // Now, must check: if there are moves on this player's bar, they must move
817 // them first, and if there are no legal moves out of the bar, the player
818 // loses their turn.
819 int my_bar_checkers = board(player, kBarPos);
820 if (my_bar_checkers > 0 && from_pos != kBarPos) {
821 return false;
822 }
823
824 // If this is a scoring move, then check that all this player's checkers are
825 // either scored or home.
826 if (to_pos < 0 || to_pos > 23) {
827 if ((CheckersInHome(player) + scores_[player]) != 15) {
828 return false;
829 }
830
831 // If it's not *exactly* the right amount, then we have to do a check to see
832 // if there exist checkers further from home, as those must be moved first.
833 if (player == kXPlayerId && to_pos > 24) {
834 for (int pos = from_pos - 1; pos >= 18; pos--) {
835 if (board(player, pos) > 0) {
836 return false;
837 }
838 }
839 } else if (player == kOPlayerId && to_pos < -1) {
840 for (int pos = from_pos + 1; pos <= 5; pos++) {
841 if (board(player, pos) > 0) {
842 return false;
843 }
844 }
845 }
846 }
847
848 return true;
849}

916std::set<CheckerMove> BackgammonState::LegalCheckerMoves(int player) const {
917 std::set<CheckerMove> moves;
918
919 if (bar_[player] > 0) {
920 // If there are any checkers are the bar, must move them out first.
921 for (int outcome : dice_) {
922 if (UsableDiceOutcome(outcome)) {
923 int pos = PositionFromBar(player, outcome);
924 if (NumOppCheckers(player, pos) <= 1) {
925 bool hit = NumOppCheckers(player, pos) == 1;
926 moves.insert(CheckerMove(kBarPos, outcome, hit));
927 }
928 }
929 }
930 return moves;
931 }
932
933 // Regular board moves.
934 bool all_in_home = AllInHome(player);
935 for (int i = 0; i < kNumPoints; ++i) {
936 if (board_[player][i] > 0) {
937 for (int outcome : dice_) {
938 if (UsableDiceOutcome(outcome)) {
939 int pos = PositionFrom(player, i, outcome);
940 if (pos == kScorePos && all_in_home) {
941 // Check whether a bear off move is legal.
942
943 // It is ok to bear off if all the checkers are at home and the
944 // point being used to move from exactly matches the distance from
945 // just stepping off the board.
946 if ((player == kXPlayerId && i + outcome == 24) ||
947 (player == kOPlayerId && i - outcome == -1)) {
948 moves.insert(CheckerMove(i, outcome, false));
949 } else {
950 // Otherwise, a die can only be used to move a checker off if
951 // there are no checkers further than it in the player's home.
952 if (i == FurthestCheckerInHome(player)) {
953 moves.insert(CheckerMove(i, outcome, false));
954 }
955 }
956 } else if (pos != kScorePos && NumOppCheckers(player, pos) <= 1) {
957 // Regular move.
958 bool hit = NumOppCheckers(player, pos) == 1;
959 moves.insert(CheckerMove(i, outcome, hit));
960 }
961 }
962 }
963 }
964 }
965 return moves;
966}

968bool BackgammonState::ApplyCheckerMove(int player, const CheckerMove& move) {
969 // Pass does nothing.
970 if (move.pos < 0) {
971 return false;
972 }
973
974 // First, remove the checker.
975 int next_pos = -1;
976 if (move.pos == kBarPos) {
977 bar_[player]--;
978 next_pos = PositionFromBar(player, move.num);
979 } else {
980 board_[player][move.pos]--;
981 next_pos = PositionFrom(player, move.pos, move.num);
982 }
983
984 // Mark the die as used.
985 for (int i = 0; i < 2; ++i) {
986 if (dice_[i] == move.num) {
987 dice_[i] += 6;
988 break;
989 }
990 }
991
992 // Now add the checker (or score).
993 if (next_pos == kScorePos) {
994 scores_[player]++;
995 } else {
996 board_[player][next_pos]++;
997 }
998
999 bool hit = false;
1000 // If there was a hit, remove opponent's piece and add to bar.
1001 // Note: the move.hit will only be properly set during the legal moves search,
1002 // so we have to also check here if there is a hit candidate.
1003 if (move.hit ||
1004 (next_pos != kScorePos && board_[Opponent(player)][next_pos] == 1)) {
1005 hit = true;
1006 board_[Opponent(player)][next_pos]--;
1007 bar_[Opponent(player)]++;
1008 }
1009
1010 return hit;
1011}

1013// Undoes a checker move. Important note: this checkermove needs to have
1014// move.hit set from the history to properly undo a move (this information is
1015// not tracked in the action value).
1016void BackgammonState::UndoCheckerMove(int player, const CheckerMove& move) {
1017 // Undoing a pass does nothing
1018 if (move.pos < 0) {
1019 return;
1020 }
1021
1022 // First, figure out the next position.
1023 int next_pos = -1;
1024 if (move.pos == kBarPos) {
1025 next_pos = PositionFromBar(player, move.num);
1026 } else {
1027 next_pos = PositionFrom(player, move.pos, move.num);
1028 }
1029
1030 // If there was a hit, take it out of the opponent's bar and put it back
1031 // onto the next position.
1032 if (move.hit) {
1033 bar_[Opponent(player)]--;
1034 board_[Opponent(player)][next_pos]++;
1035 }
1036
1037 // Remove the moved checker or decrement score.
1038 if (next_pos == kScorePos) {
1039 scores_[player]--;
1040 } else {
1041 board_[player][next_pos]--;
1042 }
1043
1044 // Mark the die as unused.
1045 for (int i = 0; i < 2; ++i) {
1046 if (dice_[i] == move.num + 6) {
1047 dice_[i] -= 6;
1048 break;
1049 }
1050 }
1051
1052 // Finally, return back the checker to its original position.
1053 if (move.pos == kBarPos) {
1054 bar_[player]++;
1055 } else {
1056 board_[player][move.pos]++;
1057 }
1058}

1060// Returns the maximum move size (2, 1, or 0)
1061int BackgammonState::RecLegalMoves(
1062 std::vector<CheckerMove> moveseq,
1063 std::set<std::vector<CheckerMove>>* movelist) {
1064 if (moveseq.size() == 2) {
1065 movelist->insert(moveseq);
1066 return moveseq.size();
1067 }
1068
1069 std::set<CheckerMove> moves_here = LegalCheckerMoves(cur_player_);
1070
1071 if (moves_here.empty()) {
1072 movelist->insert(moveseq);
1073 return moveseq.size();
1074 }
1075
1076 int max_moves = -1;
1077 for (const auto& move : moves_here) {
1078 moveseq.push_back(move);
1079 ApplyCheckerMove(cur_player_, move);
1080 int child_max = RecLegalMoves(moveseq, movelist);
1081 UndoCheckerMove(cur_player_, move);
1082 max_moves = std::max(child_max, max_moves);
1083 moveseq.pop_back();
1084 }
1085
1086 return max_moves;
1087}

1089std::vector<Action> BackgammonState::ProcessLegalMoves(
1090 int max_moves, const std::set<std::vector<CheckerMove>>& movelist) const {
1091 if (max_moves == 0) {
1092 SPIEL_CHECK_EQ(movelist.size(), 1);
1093 SPIEL_CHECK_TRUE(movelist.begin()->empty());
1094
1095 // Passing is always a legal move!
1096 return {CheckerMovesToSpielMove(
1097 {{kPassPos, -1, false}, {kPassPos, -1, false}})};
1098 }
1099
1100 // Rule 2 in Movement of Checkers:
1101 // A player must use both numbers of a roll if this is legally possible (or
1102 // all four numbers of a double). When only one number can be played, the
1103 // player must play that number. Or if either number can be played but not
1104 // both, the player must play the larger one. When neither number can be used,
1105 // the player loses his turn. In the case of doubles, when all four numbers
1106 // cannot be played, the player must play as many numbers as he can.
1107 std::vector<Action> legal_actions;
1108 int max_roll = -1;
1109 for (const auto& move : movelist) {
1110 if (max_moves == 2) {
1111 // Only add moves that are size 2.
1112 if (move.size() == 2) {
1113 legal_actions.push_back(CheckerMovesToSpielMove(move));
1114 }
1115 } else if (max_moves == 1) {
1116 // We are just finding the maximum roll.
1117 max_roll = std::max(max_roll, move[0].num);
1118 }
1119 }
1120
1121 if (max_moves == 1) {
1122 // Another round to add those that have the max die roll.
1123 for (const auto& move : movelist) {
1124 if (move[0].num == max_roll) {
1125 legal_actions.push_back(CheckerMovesToSpielMove(move));
1126 }
1127 }
1128 }
1129
1130 SPIEL_CHECK_FALSE(legal_actions.empty());
1131 return legal_actions;
1132}

1134std::vector<Action> BackgammonState::LegalActions() const {
1135 if (IsChanceNode()) return LegalChanceOutcomes();
1136 if (IsTerminal()) return {};
1137
1138 SPIEL_CHECK_EQ(CountTotalCheckers(kXPlayerId),
1139 NumCheckersPerPlayer(game_.get()));
1140 SPIEL_CHECK_EQ(CountTotalCheckers(kOPlayerId),
1141 NumCheckersPerPlayer(game_.get()));
1142
1143 std::unique_ptr<State> cstate = this->Clone();
1144 BackgammonState* state = dynamic_cast<BackgammonState*>(cstate.get());
1145 std::set<std::vector<CheckerMove>> movelist;
1146 int max_moves = state->RecLegalMoves({}, &movelist);
1147 SPIEL_CHECK_GE(max_moves, 0);
1148 SPIEL_CHECK_LE(max_moves, 2);
1149 std::vector<Action> legal_actions = ProcessLegalMoves(max_moves, movelist);
1150 std::sort(legal_actions.begin(), legal_actions.end());
1151 return legal_actions;
1152}

1154std::vector<std::pair<Action, double>> BackgammonState::ChanceOutcomes() const {
1155 SPIEL_CHECK_TRUE(IsChanceNode());
1156 if (turns_ == -1) {
1157 // Doubles not allowed for the initial roll to determine who goes first.
1158 // Range 0-14: X goes first, range 15-29: O goes first.
1159 std::vector<std::pair<Action, double>> outcomes;
1160 int num_outcomes = kNumNonDoubleOutcomes * 2;
1161 outcomes.reserve(num_outcomes);
1162 const double uniform_prob = 1.0 / num_outcomes;
1163 for (Action action = 0; action < num_outcomes; ++action) {
1164 outcomes.push_back({action, uniform_prob});
1165 }
1166 return outcomes;
1167 } else {
1168 return kChanceOutcomes;
1169 }

1174 "+------|------+", "|......|......|", "|......|......|",
1175 "|......|......|", "|......|......|", "|......|......|",

1177 "|......|......|", "|......|......|", "|......|......|",

1179

1181 for (int pos = 0; pos < 24; pos++) {
1182 if (board_[kXPlayerId][pos] > 0 || board_[kOPlayerId][pos] > 0) {
1183 int start_row = (pos < 12 ? 11 : 1);
1184 int col = (pos < 12 ? (pos >= 6 ? 12 - pos : 13 - pos)
1185 : (pos < 18 ? pos - 11 : pos - 10));
1186
1187 int row_offset = (pos < 12 ? -1 : 1);
1188

1199 char lsd = std::to_string(my_checkers % 10)[0];
1200 // Make sure it reads downward.
1201 if (pos < 12) {
1202 board_array[start_row + row_offset][col] = '1';
1203 board_array[start_row][col] = lsd;
1204 } else {
1205 board_array[start_row][col] = '1';
1206 board_array[start_row + row_offset][col] = lsd;
1207 }
1208 } else if (my_checkers > 5) {
1209 board_array[start_row][col] = std::to_string(my_checkers)[0];
1210 }
1211 }
1212 }
1213

1215
1216 // Extra info like whose turn it is etc.
1217 absl::StrAppend(&board_str, "Turn: ");

1225 absl::StrAppend(&board_str,

1227 for (int p = 0; p < 2; p++) {
1228 for (int n = 0; n < bar_[p]; n++) {
1229 absl::StrAppend(&board_str, (p == kXPlayerId ? "x" : "o"));
1230 }
1231 }
1232 absl::StrAppend(&board_str, "\n");
1233 absl::StrAppend(&board_str, "Scores, X: ", scores_[kXPlayerId]);
1234 absl::StrAppend(&board_str, ", O: ", scores_[kOPlayerId], "\n");

1237}

1244std::vector<double> BackgammonState::Returns() const {
1245 int winner = -1;
1246 int loser = -1;
1247 if (scores_[kXPlayerId] == 15) {
1248 winner = kXPlayerId;
1249 loser = kOPlayerId;
1250 } else if (scores_[kOPlayerId] == 15) {
1251 winner = kOPlayerId;
1252 loser = kXPlayerId;
1253 } else {
1254 return {0.0, 0.0};
1255 }
1256
1257 // Magnify the util based on the scoring rules for this game.
1258 int util_mag = 1;
1259 switch (scoring_type_) {
1260 case ScoringType::kWinLossScoring:
1261 default:
1262 break;
1263
1264 case ScoringType::kEnableGammons:
1265 util_mag = (IsGammoned(loser) ? 2 : 1);
1266 break;
1267
1268 case ScoringType::kFullScoring:
1269 util_mag = (IsBackgammoned(loser) ? 3 : IsGammoned(loser) ? 2 : 1);
1270 break;
1271 }
1272
1273 std::vector<double> returns(kNumPlayers);
1274 returns[winner] = util_mag;
1275 returns[loser] = -util_mag;
1276 return returns;
1277}
