28 std::unordered_map<std::string, int> GetStatesToIndices(const Game& game) {
29  std::unordered_map<std::string, int> state_index;
30  std::vector<std::unique_ptr<State>> to_visit;
31  to_visit.push_back(game.NewInitialState());
32  int index = 0;
33  while (!to_visit.empty()) {
34  std::unique_ptr<State> state = std::move(to_visit.back());
35  to_visit.pop_back();
36  if (!state->IsChanceNode() && !state->IsTerminal()) {
37  state_index[state->InformationStateString()] = index;
38  }
39  ++index;
40  for (Action action : state->LegalActions()) {
41  to_visit.push_back(state->Child(action));
42  }
43  }
44  return state_index;
45 }

213 void BatchedTrajectoryResizesCorrectly(const std::string& game_name) {
214  std::shared_ptr<const Game> game = LoadGame(game_name);
215  const std::vector<TabularPolicy> policies(2, GetUniformPolicy(*game));
216  std::unordered_map<std::string, int> states_to_indices =
217  GetStatesToIndices(*game);
218  std::mt19937 rng;
219  BatchedTrajectory trajectory = RecordBatchedTrajectory(
220  *game, policies, states_to_indices, kBatchSize,
221  /*include_full_observations=*/false, /*rng_ptr=*/&rng);
222  for (int b = 0; b < trajectory.batch_size; ++b) {
223  SPIEL_CHECK_EQ(trajectory.valid[b].size(), trajectory.actions[b].size());
224  }
225  trajectory.ResizeFields(game->MaxGameLength());
226  SPIEL_CHECK_EQ(trajectory.batch_size, kBatchSize);
227  SPIEL_CHECK_EQ(trajectory.actions.size(), kBatchSize);
228  SPIEL_CHECK_EQ(trajectory.player_ids.size(), kBatchSize);
229  SPIEL_CHECK_EQ(trajectory.rewards.size(), kBatchSize);
230  SPIEL_CHECK_EQ(trajectory.legal_actions.size(), kBatchSize);
231  SPIEL_CHECK_EQ(trajectory.player_policies.size(), kBatchSize);
232  SPIEL_CHECK_EQ(trajectory.next_is_terminal.size(), kBatchSize);
233  SPIEL_CHECK_EQ(trajectory.valid.size(), kBatchSize);
234  for (int b = 0; b < trajectory.batch_size; ++b) {
235  SPIEL_CHECK_EQ(trajectory.actions[b].size(),
236  trajectory.max_trajectory_length);
237  SPIEL_CHECK_EQ(trajectory.valid[b].size(),
238  trajectory.max_trajectory_length);
239  SPIEL_CHECK_EQ(trajectory.player_ids[b].size(),
240  trajectory.max_trajectory_length);
241  SPIEL_CHECK_EQ(trajectory.next_is_terminal[b].size(),
242  trajectory.max_trajectory_length);
243  SPIEL_CHECK_EQ(trajectory.rewards[b].size(), game->NumPlayers());
244  for (int t = 0; t < trajectory.max_trajectory_length; ++t) {
245  SPIEL_CHECK_EQ(trajectory.legal_actions[b][t].size(),
246  game->NumDistinctActions());
247 
248  // We have to check for <= as some policies omit actions with zero
249  // probability.
250  SPIEL_CHECK_LE(trajectory.player_policies[b][t].size(),
251  game->NumDistinctActions());
252  }
253  }
