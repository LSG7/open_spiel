80 template <>
81 struct type_caster<open_spiel::GameParameter> {
82  public:
83  PYBIND11_TYPE_CASTER(open_spiel::GameParameter, _("GameParameter"));
84 
85  bool load(handle src, bool convert) {
86  if (src.is_none()) {
87  // value is default-constructed to an unset value
88  return true;
89  } else if (PyBool_Check(src.ptr())) {
90  value = open_spiel::GameParameter(src.cast<bool>());
91  return true;
92  } else if (auto str_val = maybe_load<std::string>(src, convert)) {
93  value = open_spiel::GameParameter(*str_val);
94  return true;
95  } else if (PyFloat_Check(src.ptr())) {
96  value = open_spiel::GameParameter(src.cast<double>());
97  return true;
98  } else if (PyLong_Check(src.ptr())) {
99  value = open_spiel::GameParameter(src.cast<int>());
100  return true;
101  } else {
102  auto dict = src.cast<pybind11::dict>();
103  std::map<std::string, open_spiel::GameParameter> d;
104  for (const auto& [k, v] : dict) {
105  d[k.cast<std::string>()] = v.cast<open_spiel::GameParameter>();
106  }
107  value = open_spiel::GameParameter(d);
108  return true;
109  }
110  }
111 
112  static handle cast(const open_spiel::GameParameter& gp,
113  return_value_policy policy, handle parent) {
114  if (gp.has_bool_value()) {
115  return pybind11::bool_(gp.bool_value()).release();
116  } else if (gp.has_double_value()) {
117  return pybind11::float_(gp.double_value()).release();
118  } else if (gp.has_string_value()) {
119  return pybind11::str(gp.string_value()).release();
120  } else if (gp.has_int_value()) {
121  return pybind11::int_(gp.int_value()).release();
122  } else if (gp.has_game_value()) {
123  pybind11::dict d;
124  for (const auto& [k, v] : gp.game_value()) {
125  d[pybind11::str(k)] = pybind11::cast(v);
126  }
127  return d.release();
128  } else {
129  return pybind11::none();
130  }
131  }
132 
133  private:
134  template <typename T>
135  absl::optional<T> maybe_load(handle src, bool convert) {
136  auto caster = pybind11::detail::make_caster<T>();
137  if (caster.load(src, convert)) {
138  return cast_op<T>(caster);
139  } else {
140  return absl::nullopt;
141  }
142  }
143 };
