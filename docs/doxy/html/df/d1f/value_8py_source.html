16"""Representation of a value for a game.
17
18This is a standard representation for passing value functions into algorithms,
19with currently the following implementations:
20
21The main way of using a value is to call `value(state)`
22or `value(state, action)`.
23
24We will prevent calling a value on a state action on a MEAN_FIELD state.
25
26The state can be a pyspiel.State object or its string representation. For a
27particular ValueFunction instance, you should use only one or the other. The
28behavior may be undefined for mixed usage depending on the implementation.
29"""

39class ValueFunction(object):
40 """Base class for values.
41
42 A ValueFunction is something that returns a value given
43 a state of the world or a state and an action.
44
45 Attributes:
46 game: the game for which this ValueFunction derives
47 """
48
49 def __init__(self, game):
50 """Initializes a value.
51
52 Args:
53 game: the game for which this value derives
54 """
55 self.game = game
56
57 def value(self, state: ValueFunctionState, action=None) -> float:
58 """Returns a float representing a value.
59
60 Args:
61 state: A `pyspiel.State` object or its string representation.
62 action: may be None or a legal action
63
64 Returns:
65 A value for the state (and eventuallu state action pair).
66 """
67 raise NotImplementedError()
68
69 def __call__(self, state: ValueFunctionState, action=None) -> float:
70 """Turns the value into a callable.
71
72 Args:
73 state: A `pyspiel.State` object or its string representation.
74 action: may be None or a legal action
75
76 Returns:
77 Float: the value of the state or the state action pair.
78 """
79 return self.value(state, action=action)
80
81 def set_value(self, state: ValueFunctionState, value: float, action=None):
82 """Sets the value of the state.
83
84 Args:
85 state: A `pyspiel.State` object or its string representation.
86 value: Value of the state.
87 action: may be None or a legal action
88 """
89 raise NotImplementedError()
90
91 def has(self, state: ValueFunctionState, action=None) -> bool:
92 """Returns true if state(-action) has an explicit value.
93
94 Args:
95 state: A `pyspiel.State` object or its string representation.
96 action: may be None or a legal action
97
98 Returns:
99 True if there is an explicitly specified value.
100 """
101 raise NotImplementedError()
102
103 def add_value(self, state, value: float, action=None):
104 """Adds the value to the current value of the state.
105
106 Args:
107 state: A `pyspiel.State` object or its string representation.
108 value: Value to add.
109 action: may be None or a legal action
110 """
111 self.set_value(
112 state, self.value(state, action=action) + value, action=action)

115class TabularValueFunction(ValueFunction):
116 """Tabular value function backed by a dictionary."""
117
118 def __init__(self, game):

120 self._values = collections.defaultdict(float)

122 def value(self, state: ValueFunctionState, action=None):
