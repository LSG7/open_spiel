28// This is the payoff matrix from the bayesian Action Decoder paper.
29constexpr char kDefaultPayoffString[] =
30 // Cards: 1, 1
31 "10;0;0;4;8;4;10;0;0;"
32 // Cards: 1, 2
33 "0;0;10;4;8;4;0;0;10;"
34 // Cards: 2, 1
35 "0;0;10;4;8;4;0;0;0;"
36 // Cards: 2, 2
37 "10;0;0;4;8;4;10;0;0";

39std::vector<int> ParsePayoffString(const std::string& str) {
40 std::vector<std::string> pieces = absl::StrSplit(str, ';');
41 std::vector<int> payoff;
42 for (const auto& piece : pieces) {
43 int val;
44 if (!absl::SimpleAtoi(piece, &val)) {
45 SpielFatalError(absl::StrCat("Could not parse piece '", piece,
46 "' of payoff string '", str,
47 "' as an integer"));
48 }
49 payoff.push_back(val);
50 }
51 return payoff;
52}

99Player TinyHanabiState::CurrentPlayer() const {
100 const int history_size = history_.size();
101 if (history_size < num_players_) return kChancePlayerId;
102 if (history_size == 2 * num_players_) return kTerminalPlayerId;

114std::vector<std::pair<Action, double>> TinyHanabiState::ChanceOutcomes() const {
115 if (!IsChanceNode()) return {};
116 std::vector<std::pair<Action, double>> outcomes;
117 const int num_outcomes = payoff_.NumChance();
118 const double p = 1.0 / num_outcomes;
119 outcomes.reserve(num_outcomes);
120 for (int i = 0; i < num_outcomes; ++i) outcomes.emplace_back(i, p);
121 return outcomes;
122}

124std::string TinyHanabiState::ToString() const {
125 std::string rv;
126 for (int i = 0; i < payoff_.NumPlayers() && i < history_.size(); ++i) {
127 if (i != 0) absl::StrAppend(&rv, " ");
128 absl::StrAppend(&rv, "p", i, ":d", history_[i].action);
129 }
130 for (int i = payoff_.NumPlayers(); i < history_.size(); ++i) {
131 absl::StrAppend(&rv, " p", history_[i].player, ":a", history_[i].action);
132 }
133 return rv;
134}

157std::string TinyHanabiState::InformationStateString(Player player) const {
158 SPIEL_CHECK_GE(player, 0);

160
161 std::string rv = absl::StrCat("p", player);
162 if (history_.size() > player)
163 absl::StrAppend(&rv, ":d", history_[player].action);
164 for (int i = payoff_.NumPlayers(); i < history_.size(); ++i) {
165 absl::StrAppend(&rv, " p", i - payoff_.NumPlayers(), ":a",
166 history_[i].action);

171void TinyHanabiState::InformationStateTensor(Player player,
172 absl::Span<float> values) const {
173 SPIEL_CHECK_GE(player, 0);
174 SPIEL_CHECK_LT(player, num_players_);
175
176 SPIEL_CHECK_EQ(values.size(), payoff_.NumChance() + payoff_.NumActions() *
177 payoff_.NumPlayers());
178 std::fill(values.begin(), values.end(), 0);
179 if (history_.size() > player) values.at(history_[player].action) = 1;
180 for (int i = payoff_.NumPlayers(); i < history_.size(); ++i) {
181 values.at(payoff_.NumChance() +
182 (i - payoff_.NumPlayers()) * payoff_.NumActions() +
183 history_[i].action) = 1;
184 }
185}
