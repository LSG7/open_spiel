26class MaximalLotteriesVoting(base.AbstractVotingMethod):
27 """Implements Copeland's method."""

29 def __init__(self,
30 iterative: bool = False,
31 verbose: bool = False,
32 zero_tolerance: float = 1e-6):
33 self._iterative = iterative
34 self._verbose = verbose
35 self._zero_tolerance = zero_tolerance
36
37 def name(self) -> str:
38 return f"maximal_lotteries(iterative={self._iterative})"

40 def _create_matrix_game(self, matrix: np.ndarray):
41 return pyspiel.create_tensor_game([matrix, -matrix]).as_matrix_game()
42
43 def _solve_game(
44 self, margin_matrix: np.ndarray
45 ) -> np.ndarray:
46 matrix_game = self._create_matrix_game(margin_matrix)
47 p0_sol, _, _, _ = lp_solver.solve_zero_sum_matrix_game(matrix_game)
48 return p0_sol

50 def run_election(self, profile: base.PreferenceProfile) -> base.RankOutcome:
51 margin_matrix = profile.margin_matrix()
52 alternatives = profile.alternatives
53 m = profile.num_alternatives()
54 if self._verbose:
55 print(f"Margin matrix: \n{margin_matrix}")
56 print(f"Alternatives: {alternatives}")
57 p0_sol = self._solve_game(margin_matrix)
58
59 # For now define scores as the probabilities.
60 scores = {}
61 if not self._iterative:
62 # and negligible noise to break ties
63 noise = 1e-10 * np.random.uniform(size=m)
64 for i in range(m):
65 scores[alternatives[i]] = p0_sol[i] + noise[i]
66 sorted_scores = sorted(scores.items(), key=lambda item: item[1])
67 sorted_scores.reverse()
68 outcome = base.RankOutcome()
69 outcome.unpack_from(sorted_scores)
70 return outcome
71 else:
72 # Continue to iteratively solve all the remaining subgames.
73 return self._iterate(alternatives, margin_matrix, p0_sol)

75 def _iterate(
76 self,
77 alternatives: List[base.AlternativeId],
78 margin_matrix: np.ndarray,
79 p0_sol: np.ndarray,
80 ):
81 remaining_alternatives = alternatives[:]
82 leveled_ranking = []
83 leveled_scores = []
84 while remaining_alternatives:
85 # Pull out the nonzero entries and make them winners of this level.
86 m = len(remaining_alternatives)
87 if self._verbose:
88 print(f"\nRemaining alternatives: {remaining_alternatives}")
89 cur_level = len(leveled_ranking)
90 print(f"IML Level {cur_level}")
91 print(f"Remaining alternatives: {remaining_alternatives}")
92 print(f"Margin matrix: \n{margin_matrix}\n")
93 if m == 1:
94 leveled_ranking.append(remaining_alternatives[:])
95 leveled_scores.append([1])
96 break
97 noise = 1e-10 * np.random.uniform(size=m)
98 for i in range(m):
99 p0_sol[i] += noise[i]
100 values = -1 * np.ones(m, dtype=np.float64)
101 level_winners_idxs = []
102 for i in range(m):
103 if p0_sol[i] > self._zero_tolerance:
104 # print(f"p0_sol[{i}] = {p0_sol[i]}")
105 level_winners_idxs.append(i)
106 values[i] = p0_sol[i]
107 num_level_winners = len(level_winners_idxs)
108 assert num_level_winners >= 1
109 indices = np.argsort(-values)
110 level_winners_ranked = []
111 level_winners_scores = []
112 for j in range(num_level_winners):
113 idx = int(indices[j])
114 level_winners_ranked.append(remaining_alternatives[idx])
115 level_winners_scores.append(p0_sol[idx])
116 leveled_ranking.append(level_winners_ranked)
117 leveled_scores.append(level_winners_scores)
118 if self._verbose:
119 print(f"Level winners: {level_winners_ranked}")
120 print(f"Level scores: {level_winners_scores}")
121 # Now, take them out of the margin matrix and remaining alternatives
122 # Delete in reverse order.
123 for j in range(num_level_winners):
124 idx = level_winners_idxs[num_level_winners - 1 - j]
125 del remaining_alternatives[idx]
126 margin_matrix = np.delete(margin_matrix, (idx), axis=0)
127 margin_matrix = np.delete(margin_matrix, (idx), axis=1)
128 if len(remaining_alternatives) > 1:
129 p0_sol = self._solve_game(margin_matrix)
130 # Now bump up the scores by level, and put them in the outcome.
131 scores = {}
132 num_levels = len(leveled_ranking)
133 if self._verbose:
134 print(f"Num levels: {num_levels}")
135 level_base_points = num_levels - 1
136 for level in range(num_levels):
137 for j in range(len(leveled_ranking[level])):
138 alternative = leveled_ranking[level][j]
139 score = level_base_points + leveled_scores[level][j]
140 scores[alternative] = score
141 level_base_points -= 1
142 sorted_scores = sorted(scores.items(), key=lambda item: item[1])
143 sorted_scores.reverse()
144 outcome = base.RankOutcome()
145 outcome.unpack_from(sorted_scores)
146 return outcome
