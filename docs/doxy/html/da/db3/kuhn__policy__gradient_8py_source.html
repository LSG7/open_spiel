35class PolicyGradientPolicies(policy.Policy):
36 """Joint policy to be evaluated."""
37
38 def __init__(self, env, nfsp_policies):
39 game = env.game
40 player_ids = [0, 1]
41 super(PolicyGradientPolicies, self).__init__(game, player_ids)
42 self._policies = nfsp_policies
43 self._obs = {"info_state": [None, None], "legal_actions": [None, None]}
44
45 def action_probabilities(self, state, player_id=None):
46 cur_player = state.current_player()
47 legal_actions = state.legal_actions(cur_player)
48
49 self._obs["current_player"] = cur_player
50 self._obs["info_state"][cur_player] = (
51 state.information_state_tensor(cur_player))
52 self._obs["legal_actions"][cur_player] = legal_actions
53
54 info_state = rl_environment.TimeStep(
55 observations=self._obs, rewards=None, discounts=None, step_type=None)
56
57 p = self._policies[cur_player].step(info_state, is_evaluation=True).probs
58 prob_dict = {action: p[action] for action in legal_actions}
59 return prob_dict

62def main(_):
63 game = "kuhn_poker"
64 num_players = 2
65
66 env_configs = {"players": num_players}
67 env = rl_environment.Environment(game, **env_configs)
68 info_state_size = env.observation_spec()["info_state"][0]
69 num_actions = env.action_spec()["num_actions"]
70
71 with tf.Session() as sess:
72 # pylint: disable=g-complex-comprehension
73 agents = [
74 policy_gradient.PolicyGradient(
75 sess,
76 idx,
77 info_state_size,
78 num_actions,
79 loss_str=FLAGS.loss_str,
80 hidden_layers_sizes=(128,)) for idx in range(num_players)
81 ]
82 expl_policies_avg = PolicyGradientPolicies(env, agents)
83
84 sess.run(tf.global_variables_initializer())
85 for ep in range(FLAGS.num_episodes):
86
87 if (ep + 1) % FLAGS.eval_every == 0:
88 losses = [agent.loss for agent in agents]
89 expl = exploitability.exploitability(env.game, expl_policies_avg)
90 msg = "-" * 80 + "\n"
91 msg += "{}: {}\n{}\n".format(ep + 1, expl, losses)
92 logging.info("%s", msg)
93
94 time_step = env.reset()
95 while not time_step.last():
96 player_id = time_step.observations["current_player"]
97 agent_output = agents[player_id].step(time_step)
98 action_list = [agent_output.action]
99 time_step = env.step(action_list)
100
101 # Episode is over, step all agents with final info state.
102 for agent in agents:
103 agent.step(time_step)
104
105
106if __name__ == "__main__":
107 app.run(main)
