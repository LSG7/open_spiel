32 double NashConvTest(const std::string& game_string, const Policy& policy,
33  absl::optional<double> expected_nash_conv = absl::nullopt) {
34  std::shared_ptr<const Game> game = LoadGame(game_string);
35  TabularBestResponseMDP tbr(*game, policy);
36  TabularBestResponseMDPInfo br_info = tbr.NashConv();
37  if (expected_nash_conv.has_value()) {
38  SPIEL_CHECK_FLOAT_NEAR(br_info.nash_conv, expected_nash_conv.value(),
39  kFloatTolerance);
40  }
41  return br_info.nash_conv;
42 }

79 void TicTacToeTests() {

82  TabularBestResponseMDP tbr1(*game, uniform_policy);

159 void ImperfectRecallLiarsDiceGameTests() {
160  std::shared_ptr<const Game> ir_game = LoadGame("liars_dice_ir");
161  std::shared_ptr<const Game> pr_game = LoadGame("liars_dice");
162 
163  std::cout << ir_game->GetType().short_name << std::endl;
164 
165  {
166  UniformPolicy uniform_policy;
167  TabularBestResponseMDP tbr1(*pr_game, uniform_policy);
168  TabularBestResponseMDPInfo br_info1 = tbr1.NashConv();
169  std::cout << "PR uniform: " << br_info1.nash_conv << std::endl;
170  SPIEL_CHECK_FLOAT_NEAR(br_info1.nash_conv, 1.561488646384479,
171  kFloatTolerance);
172 
173  TabularBestResponseMDP tbr2(*ir_game, uniform_policy);
174  TabularBestResponseMDPInfo br_info2 = tbr2.NashConv();
175  std::cout << "IR uniform: " << br_info2.nash_conv << std::endl;
176  }
177 
178  // For a reference, see Figure 1 from Lanctot et al. '12
179  // http://mlanctot.info/files/papers/12icml-ir.pdf
180  CFRSolver pr_solver(*pr_game);
181  CFRSolver ir_solver(*ir_game);
182  for (int i = 0; i < kNumLiarsDiceCFRIterations; i++) {
183  pr_solver.EvaluateAndUpdatePolicy();
184  ir_solver.EvaluateAndUpdatePolicy();
185  if (i % 10 == 0) {
186  const std::shared_ptr<Policy> pr_avg_policy = pr_solver.AveragePolicy();
187  const std::shared_ptr<Policy> ir_avg_policy = ir_solver.AveragePolicy();
188 
189  TabularBestResponseMDP pr_tbr(*pr_game, *pr_avg_policy);
190  TabularBestResponseMDPInfo pr_br_info = pr_tbr.NashConv();
191 
192  TabularBestResponseMDP ir_tbr(*ir_game, *ir_avg_policy);
193  TabularBestResponseMDPInfo ir_br_info = ir_tbr.NashConv();
194 
195  printf("%3d %0.15lf %0.15lf\n", i, pr_br_info.nash_conv,
196  ir_br_info.nash_conv);
197  }
198  }
