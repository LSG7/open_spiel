33// Online Outcome Sampling (OOS) algorithm
34//
35// Online algorithm for solving 2-player imperfect-information 0-sum games.
36//
37// Based on the original implementation of paper
38//
39// [1] Online monte carlo counterfactual regret minimization for search
40// in imperfect information games, Lis√Ω, Lanctot and Bowling
41//
42// The original implementation can be found at
43// https://github.com/aicenter/gtlibrary-java/tree/master/src/cz/agents/gtlibrary/algorithms/mcts/oos
44//
45// # Description of the algorithm:
46//
47// OOS is a modification of Outcome Sampling Monte Carlo CFR for online setting.
48// The player receives its current play position (infostate) and it has some
49// time (or iteration) budget to improve his strategy. On a high level,
50// OOS changes MCCFR sampling scheme to bias it towards the current info state.
51//
52// Additionally, it incrementally builds the search tree by doing rollouts
53// whenever leafs of the lookahead search tree are hit, and it expands this
54// search tree.
55//
56// If the algorithm is unable to retrieve strategy for the current infostate,
57// i.e. it "gets lost" in the game, it continues to play uniformly randomly.
58// This can happen because targeting sampling of infostate / public state
59// by using Action-Observation or Public observation histories respectively
60// is not always successful, and therefore no strategy may be computed at the
61// requested target.
62//
63// When this algorithm is instantiated with target_biasing = 0, it becomes
64// Outcome Sampling MCCFR with incremental tree building. If you also prebuild
65// the tree you get the MCCFR algorithm.
66//
67// The implementation supports both information state and public state
68// targeting.
69//
70// It is possible to provide custom sampling schemes that are implemented
71// on the level of infostate strategies.
72//
73// There is a small difference to the original implementation:
74// It used a "target compensation", a weighting factor according
75// to equation (3) in [1]. This compensation is not implemented. According
76// to conversation with the original author it did not influence the results
77// significantly, and it makes the implementation unnecessarily cluttered.
78//
79// Internally, the algorithm uses a large number of various variables, so there
80// are some important conventions in variable naming, in the format of: A_B_C
81//
82// # A corresponds to strategy:
83//
84// rm regret matching (or also called current) strategy
85// avg average strategy
86// unif uniform strategy
87// bs biased sampling strategy
88// us unbiased sampling strategy
89// s sampling strategy (combination of biased and unbiased strategy)
90//
91// # B corresponds to a specific history or trajectory
92//
93// h current history
94// z terminal history (of the game)
95// zh from current history to the terminal, i.e. z|h
96// zha from current history and playing action a with 100% prob
97// to the terminal, i.e. z|h.a
98// ha play action a at the current history, i.e. resulting
99// to child history h.a
100//
101// # C corresponds to player
102//
103// pl current player
104// opp opponent player (without chance)
105// cn chance player
106// both current and opponent player (without chance)
107// all all the players (including chance)
108//
109// # Examples:
110//
111// s_z_all: is the probability of sampling terminal history
112// rm_h_pl: reach probability of the searching player to the current history
113// using RM strategy
114
115enum Targeting {
116 kDoNotUseTargeting,
117
118 // Target samples to the current information state.
119 // More precisely, target the current Action-Observation history (AOH),
120 // which coincides with the notion of information states on the player
121 // states.
122 kInfoStateTargeting,
123
124 // Target samples to the current public state.
125 // More precisely, target the current Public-Observation history (POH).
126 kPublicStateTargeting,
127};

156// Epsilon-on-policy exploration sampling.
157//
158// The sampling distribution is an epsilon convex combination
159// of the current policy (from Regret Matching) and uniform strategy.
160class ExplorativeSamplingPolicy : public CFRCurrentPolicy {
161 public:
162 const double exploration_; // AKA epsilon
163
164 ExplorativeSamplingPolicy(const OOSInfoStateValuesTable& table,
165 double exploration = kDefaultExploration)
166 : CFRCurrentPolicy(table, std::make_shared<UniformPolicy>()),
167 exploration_(exploration) {
168 // We need the exploration to be positive to guarantee all leaves are
169 // reachable.
170 SPIEL_CHECK_GT(exploration_, 0);
171 SPIEL_CHECK_LE(exploration_, 1);
172 }
173
174 ActionsAndProbs GetStatePolicy(const State& state) const override;
175 ActionsAndProbs GetStatePolicy(const std::string& info_state) const override;
176};

182// Biased sampling policy.
183//
184// The policy will return a convex combination of bias_exploration between
185// regret matching strategy and uniform strategy on actions to do lead
186// to the target. If an action does not lead to the target, the value of
187// kBannedAction (negative zero) is used.
188//
189// The targeting is done on top of cached entries from the main algorithm,
190// i.e. the OOSValues table is shared between the algorithm and this sampling
191// policy.
192class TargetedPolicy : public CFRCurrentPolicy {
193 protected:
194 const std::shared_ptr<const Game> game_;
195 const double bias_exploration_; // AKA epsilon
196
197 Targeting targeting_ = kDoNotUseTargeting;
198 // Current target for which we should be doing the biasing.
199 const ActionObservationHistory* target_info_state_ =
200 kNoActionObsTargetSpecified;
201 const PublicObservationHistory* target_public_state_ =
202 kNoPublicObsTargetSpecified;
203
204 // Externally keep track of how many targets have been missed.
205 OnlineStats* stats_;
206
207 public:
208 TargetedPolicy(std::shared_ptr<const Game> game,
209 const OOSInfoStateValuesTable& table,
210 double bias_exploration = kDefaultExploration)
211 : CFRCurrentPolicy(table, std::make_shared<UniformPolicy>()),
212 game_(game),
213 bias_exploration_(bias_exploration) {}
214
215 void NoTargeting();
216 void UpdateTarget(const ActionObservationHistory* info_state);
217 void UpdateTarget(const PublicObservationHistory* public_state);
218 bool IsTargetHit(const State& h);
219
220 ActionsAndProbs GetStatePolicy(const State& h) const;
221 void TrackStatistics(OnlineStats* stats) { stats_ = stats; }
222
223 private:
224 bool IsAllowedAction(const State& h, const Action& action) const;
225};
