71  def __init__(self, objective):
72  assert objective == OBJ_MIN or objective == OBJ_MAX
73  self._valid_constraint_types_valid_constraint_types = [CONS_TYPE_EQ, CONS_TYPE_LEQ, CONS_TYPE_GEQ]
74  self._objective_objective = objective
75  self._obj_coeffs_obj_coeffs = {} # var label -> value
76  self._vars_vars = {} # var label -> var
77  self._cons_cons = {} # cons label -> constraint
78  self._var_list_var_list = []

82  def add_or_reuse_variable(self, label, lb=None, ub=None):
83  """Adds a variable to this LP, or reuses one if the label exists.
84 
85  If the variable already exists, simply checks that the upper and lower
86  bounds are the same as previously specified.
87 
88  Args:
89  label: a label to assign to this constraint
90  lb: a lower-bound value for this variable
91  ub: an upper-bound value for this variable
92  """
93  var = self._vars_vars.get(label)
94  if var is not None:
95  # Do not re-add, but ensure it's the same
96  assert var.lb == lb and var.ub == ub
97  return
98  var = _Variable(len(self._var_list_var_list), lb, ub)
99  self._vars_vars[label] = var
100  self._var_list_var_list.append(var)

102  def add_or_reuse_constraint(self, label, ctype):
103  """Adds a constraint to this LP, or reuses one if the label exists.
104 
105  If the constraint is already present, simply checks it's the same type as
106  previously specified.
107 
108  Args:
109  label: a label to assign to this constraint
110  ctype: the constraint type (in CONS_TYPE_{LEQ,GEQ,EQ})
111  """
112  assert ctype in self._valid_constraint_types_valid_constraint_types
113  cons = self._cons_cons.get(label)
114  if cons is not None:
115  # Do not re-add, but ensure it's the same type
116  assert cons.ctype == ctype
117  return
118  if ctype == CONS_TYPE_LEQ or ctype == CONS_TYPE_GEQ:
119  cons = _Constraint(len(self._leq_cons_list_leq_cons_list), ctype)
120  self._cons_cons[label] = cons
121  self._leq_cons_list_leq_cons_list.append(cons)
122  elif ctype == CONS_TYPE_EQ:
123  cons = _Constraint(len(self._eq_cons_list_eq_cons_list), ctype)
124  self._cons_cons[label] = cons
125  self._eq_cons_list_eq_cons_list.append(cons)
126  else:
127  assert False, "Unknown constraint type"

156  def solve(self, solver=None):
157  """Solves the LP.
158 
159  Args:
160  solver: the solver to use ('blas', 'lapack', 'glpk'). Defaults to None,
161  which then uses the cvxopt internal default.
162 
163  Returns:
164  The solution as a dict of var label -> value, one for each variable.
165  """
166  # From http://cvxopt.org/userguide/coneprog.html#linear-programming,
167  # CVXOPT uses the formulation:
168  # minimize: c^t x
169  # s.t. Gx <= h
170  # Ax = b
171  #
172  # Here:
173  # - x is the vector the variables
174  # - c is the vector of objective coefficients
175  # - G is the matrix of LEQ (and GEQ) constraint coefficients
176  # - h is the vector or right-hand side values of the LEQ/GEQ constraints
177  # - A is the matrix of equality constraint coefficients
178  # - b is the vector of right-hand side values of the equality constraints
179  #
180  # This function builds these sparse matrices from the information it has
181  # gathered, flipping signs where necessary, and adding equality constraints
182  # for the upper and lower bounds of variables. It then calls the cvxopt
183  # solver and maps back the values.
184  num_vars = len(self._var_list_var_list)
185  num_eq_cons = len(self._eq_cons_list_eq_cons_list)
186  num_leq_cons = len(self._leq_cons_list_leq_cons_list)
187  for var in self._var_list_var_list:
188  if var.lb is not None:
189  num_leq_cons += 1
190  if var.ub is not None:
191  num_leq_cons += 1
192  # Make the matrices (some need to be dense).
193  c = cvxopt.matrix([0.0] * num_vars)
194  h = cvxopt.matrix([0.0] * num_leq_cons)
195  g_mat = cvxopt.spmatrix([], [], [], (num_leq_cons, num_vars))
196  a_mat = None
197  b = None
198  if num_eq_cons > 0:
199  a_mat = cvxopt.spmatrix([], [], [], (num_eq_cons, num_vars))
200  b = cvxopt.matrix([0.0] * num_eq_cons)
201  # Objective coefficients: c
202  for var_label in self._obj_coeffs_obj_coeffs:
203  value = self._obj_coeffs_obj_coeffs[var_label]
204  vid = self._vars_vars[var_label].vid
205  if self._objective_objective == OBJ_MAX:
206  c[vid] = -value # negate the value because it's a max
207  else:
208  c[vid] = value # min objective matches cvxopt
209  # Inequality constraints: G, h
210  row = 0
211  for cons in self._leq_cons_list_leq_cons_list:
212  # If it's >= then need to negate all coeffs and the rhs
213  if cons.rhs is not None:
214  h[row] = cons.rhs if cons.ctype == CONS_TYPE_LEQ else -cons.rhs
215  for var_label in cons.coeffs:
216  value = cons.coeffs[var_label]
217  vid = self._vars_vars[var_label].vid
218  g_mat[(row, vid)] = value if cons.ctype == CONS_TYPE_LEQ else -value
219  row += 1
220  # Inequality constraints: variables upper and lower bounds
221  for var in self._var_list_var_list:
222  if var.lb is not None: # x_i >= lb has to be -x_i <= -lb
223  g_mat[(row, var.vid)] = -1.0
224  h[row] = -var.lb
225  row += 1
226  if var.ub is not None: # x_i <= ub
227  g_mat[(row, var.vid)] = 1.0
228  h[row] = var.ub
229  row += 1
230  # Equality constraints: A, b
231  if num_eq_cons > 0:
232  row = 0
233  for cons in self._eq_cons_list_eq_cons_list:
234  b[row] = cons.rhs if cons.rhs is not None else 0.0
235  for var_label in cons.coeffs:
236  value = cons.coeffs[var_label]
237  vid = self._vars_vars[var_label].vid
238  a_mat[(row, vid)] = value
239  row += 1
240  # Solve!
241  if num_eq_cons > 0:
242  sol = cvxopt.solvers.lp(c, g_mat, h, a_mat, b, solver=solver)
243  else:
244  sol = cvxopt.solvers.lp(c, g_mat, h, solver=solver)
245  return sol["x"]

248 def solve_zero_sum_matrix_game(game):
249  """Solves a matrix game by using linear programming.
250 
251  Args:
252  game: a pyspiel MatrixGame
253 
254  Returns:
255  A 4-tuple containing:
256  - p0_sol (array-like): probability distribution over row actions
257  - p1_sol (array-like): probability distribution over column actions
258  - p0_sol_value, expected value to the first player
259  - p1_sol_value, expected value to the second player
260  """
261 
262  # Solving a game for player i (e.g. row player) requires finding a mixed
263  # policy over player i's pure strategies (actions) such that a value of the
264  # mixed strategy against every opponent pure strategy is maximized.
265  #
266  # For more detail, please refer to Sec 4.1 of Shoham & Leyton-Brown, 2009:
267  # Multiagent Systems: Algorithmic, Game-Theoretic, and Logical Foundations
268  # http://www.masfoundations.org/mas.pdf
269  #
270  # For the row player the LP looks like:
271  # max V
272  # st. sigma_a1 \dot col_0 >= V
273  # sigma_a2 \dot col_1 >= V
274  # .
275  # .
276  # sigma_am \cot col_n >= V
277  # for all i, sigma_ai >= 0
278  # sigma \dot 1 = 1
279  assert isinstance(game, pyspiel.MatrixGame)
280  assert game.get_type().information == pyspiel.GameType.Information.ONE_SHOT
281  assert game.get_type().utility == pyspiel.GameType.Utility.ZERO_SUM
282  num_rows = game.num_rows()
283  num_cols = game.num_cols()
284  cvxopt.solvers.options["show_progress"] = False
285 
286  # First, do the row player (player 0).
287  lp0 = LinearProgram(OBJ_MAX)
288  for r in range(num_rows): # one var per action / pure strategy
289  lp0.add_or_reuse_variable(r, lb=0)
290  lp0.add_or_reuse_variable(num_rows) # V
291  lp0.set_obj_coeff(num_rows, 1.0) # max V
292  for c in range(num_cols):
293  lp0.add_or_reuse_constraint(c, CONS_TYPE_GEQ)
294  for r in range(num_rows):
295  lp0.set_cons_coeff(c, r, game.player_utility(0, r, c))
296  lp0.set_cons_coeff(c, num_rows, -1.0) # -V >= 0
297  lp0.add_or_reuse_constraint(num_cols + 1, CONS_TYPE_EQ)
298  lp0.set_cons_rhs(num_cols + 1, 1.0)
299  for r in range(num_rows):
300  lp0.set_cons_coeff(num_cols + 1, r, 1.0)
301  sol = lp0.solve()
302  p0_sol = sol[:-1]
303  p0_sol_val = sol[-1]
304 
305  # Now, the column player (player 1).
306  lp1 = LinearProgram(OBJ_MAX)
307  for c in range(num_cols): # one var per action / pure strategy
308  lp1.add_or_reuse_variable(c, lb=0)
309  lp1.add_or_reuse_variable(num_cols) # V
310  lp1.set_obj_coeff(num_cols, 1) # max V
311  for r in range(num_rows):
312  lp1.add_or_reuse_constraint(r, CONS_TYPE_GEQ)
313  for c in range(num_cols):
314  lp1.set_cons_coeff(r, c, game.player_utility(1, r, c))
315  lp1.set_cons_coeff(r, num_cols, -1.0) # -V >= 0
316  lp1.add_or_reuse_constraint(num_rows + 1, CONS_TYPE_EQ)
317  lp1.set_cons_rhs(num_rows + 1, 1.0)
318  for c in range(num_cols):
319  lp1.set_cons_coeff(num_rows + 1, c, 1.0)
320  sol = lp1.solve()
321  p1_sol = sol[:-1]
322  p1_sol_val = sol[-1]
323 
324  return p0_sol, p1_sol, p0_sol_val, p1_sol_val

327 def is_dominated(action,
328  game_or_payoffs,
329  player,
330  mode=DOMINANCE_STRICT,
331  tol=1e-7,
332  return_mixture=False):
333  """Determines whether a pure strategy is dominated by any mixture strategies.
334 
335  Args:
336  action: index of an action for `player`
337  game_or_payoffs: either a pyspiel matrix- or normal-form game, or a payoff
338  tensor for `player` with ndim == number of players
339  player: index of the player (an integer)
340  mode: dominance criterion: strict, weak, or very weak
341  tol: tolerance
342  return_mixture: whether to return the dominating strategy if one exists
343 
344  Returns:
345  If `return_mixture`:
346  a dominating mixture strategy if one exists, or `None`.
347  the strategy is provided as a 1D numpy array of mixture weights.
348  Otherwise: True if a dominating strategy exists, False otherwise.
349  """
350  # For more detail, please refer to Sec 4.5.2 of Shoham & Leyton-Brown, 2009:
351  # Multiagent Systems: Algorithmic, Game-Theoretic, and Logical Foundations
352  # http://www.masfoundations.org/mas.pdf
353  assert mode in (DOMINANCE_STRICT, DOMINANCE_VERY_WEAK, DOMINANCE_WEAK)
354  payoffs = utils.game_payoffs_array(game_or_payoffs)[player] if isinstance(
355  game_or_payoffs, pyspiel.NormalFormGame) else np.asfarray(game_or_payoffs)
356 
357  # Reshape payoffs so rows correspond to `player` and cols to the joint action
358  # of all other players
359  payoffs = np.moveaxis(payoffs, player, 0)
360  payoffs = payoffs.reshape((payoffs.shape[0], -1))
361  num_rows, num_cols = payoffs.shape
362 
363  cvxopt.solvers.options["show_progress"] = False
364  cvxopt.solvers.options["maxtol"] = tol
365  cvxopt.solvers.options["feastol"] = tol
366  lp = LinearProgram(OBJ_MAX)
367 
368  # One var for every row probability, fixed to 0 if inactive
369  for r in range(num_rows):
370  if r == action:
371  lp.add_or_reuse_variable(r, lb=0, ub=0)
372  else:
373  lp.add_or_reuse_variable(r, lb=0)
374 
375  # For the strict LP we normalize the payoffs to be strictly positive
376  if mode == DOMINANCE_STRICT:
377  to_subtract = payoffs.min() - 1
378  else:
379  to_subtract = 0
380  # For non-strict LPs the probabilities must sum to 1
381  lp.add_or_reuse_constraint(num_cols, CONS_TYPE_EQ)
382  lp.set_cons_rhs(num_cols, 1)
383  for r in range(num_rows):
384  if r != action:
385  lp.set_cons_coeff(num_cols, r, 1)
386 
387  # The main dominance constraint
388  for c in range(num_cols):
389  lp.add_or_reuse_constraint(c, CONS_TYPE_GEQ)
390  lp.set_cons_rhs(c, payoffs[action, c] - to_subtract)
391  for r in range(num_rows):
392  if r != action:
393  lp.set_cons_coeff(c, r, payoffs[r, c] - to_subtract)
394 
395  if mode == DOMINANCE_STRICT:
396  # Minimize sum of probabilities
397  for r in range(num_rows):
398  if r != action:
399  lp.set_obj_coeff(r, -1)
400  mixture = lp.solve()
401  if mixture is not None and np.sum(mixture) < 1 - tol:
402  mixture = np.squeeze(mixture, 1) / np.sum(mixture)
403  else:
404  mixture = None
405 
406  if mode == DOMINANCE_VERY_WEAK:
407  # Check feasibility
408  mixture = lp.solve()
409  if mixture is not None:
410  mixture = np.squeeze(mixture, 1)
411 
412  if mode == DOMINANCE_WEAK:
413  # Check feasibility and whether there's any advantage
414  for r in range(num_rows):
415  lp.set_obj_coeff(r, payoffs[r].sum())
416  mixture = lp.solve()
417  if mixture is not None:
418  mixture = np.squeeze(mixture, 1)
419  if (np.dot(mixture, payoffs) - payoffs[action]).sum() <= tol:
420  mixture = None
421 
422  return mixture if return_mixture else (mixture is not None)

425 def _pure_dominated_from_advantages(advantages, mode, tol=1e-7):
426  if mode == DOMINANCE_STRICT:
427  return (advantages > tol).all(1)
428  if mode == DOMINANCE_WEAK:
429  return (advantages >= -tol).all(1) & (advantages.sum(1) > tol)
430  if mode == DOMINANCE_VERY_WEAK:
431  return (advantages >= -tol).all(1)

434 def iterated_dominance(game_or_payoffs, mode, tol=1e-7):
435  """Reduces a strategy space using iterated dominance.
436 
437  See: http://www.smallparty.com/yoram/classes/principles/nash.pdf
438 
439  Args:
440  game_or_payoffs: either a pyspiel matrix- or normal-form game, or a payoff
441  tensor of dimension `num_players` + 1. First dimension is the player,
442  followed by the actions of all players, e.g. a 3x3 game (2 players) has
443  dimension [2,3,3].
444  mode: DOMINANCE_STRICT, DOMINANCE_WEAK, or DOMINANCE_VERY_WEAK
445  tol: tolerance
446 
447  Returns:
448  A tuple (`reduced_game`, `live_actions`).
449  * if `game_or_payoffs` is an instance of `pyspiel.MatrixGame`, so is
450  `reduced_game`; otherwise `reduced_game` is a payoff tensor.
451  * `live_actions` is a tuple of length `num_players`, where
452  `live_actions[player]` is a boolean vector of shape `num_actions`;
453  `live_actions[player][action]` is `True` if `action` wasn't dominated for
454  `player`.
455  """
456  payoffs = utils.game_payoffs_array(game_or_payoffs) if isinstance(
457  game_or_payoffs, pyspiel.NormalFormGame) else np.asfarray(game_or_payoffs)
458  live_actions = [
459  np.ones(num_actions, bool) for num_actions in payoffs.shape[1:]
460  ]
461  progress = True
462  while progress:
463  progress = False
464  # trying faster method first
465  for method in ("pure", "mixed"):
466  if progress:
467  continue
468  for player, live in enumerate(live_actions):
469  if live.sum() == 1:
470  # one action is dominant
471  continue
472 
473  # discarding all dominated opponent actions
474  payoffs_live = payoffs[player]
475  for opponent in range(payoffs.shape[0]):
476  if opponent != player:
477  payoffs_live = payoffs_live.compress(live_actions[opponent],
478  opponent)
479 
480  # reshaping to (player_actions, joint_opponent_actions)
481  payoffs_live = np.moveaxis(payoffs_live, player, 0)
482  payoffs_live = payoffs_live.reshape((payoffs_live.shape[0], -1))
483 
484  for action in range(live.size):
485  if not live[action]:
486  continue
487  if method == "pure":
488  # mark all actions that `action` dominates
489  advantage = payoffs_live[action] - payoffs_live
490  dominated = _pure_dominated_from_advantages(advantage, mode, tol)
491  dominated[action] = False
492  dominated &= live
493  if dominated.any():
494  progress = True
495  live &= ~dominated
496  if live.sum() == 1:
497  break
498  if method == "mixed":
499  # test if `action` is dominated by a mixed policy
500  mixture = is_dominated(
501  live[:action].sum(),
502  payoffs_live[live],
503  0,
504  mode,
505  tol,
506  return_mixture=True)
507  if mixture is None:
508  continue
509  # if it is, mark any other actions dominated by that policy
510  progress = True
511  advantage = mixture.dot(payoffs_live[live]) - payoffs_live[live]
512  dominated = _pure_dominated_from_advantages(advantage, mode, tol)
513  dominated[mixture > tol] = False
514  assert dominated[live[:action].sum()]
515  live.put(live.nonzero()[0], ~dominated)
516  if live.sum() == 1:
517  break
518 
519  for player, live in enumerate(live_actions):
520  payoffs = payoffs.compress(live, player + 1)
521 
522  if isinstance(game_or_payoffs, pyspiel.MatrixGame):
523  return pyspiel.MatrixGame(game_or_payoffs.get_type(),
524  game_or_payoffs.get_parameters(), [
525  game_or_payoffs.row_action_name(action)
526  for action in live_actions[0].nonzero()[0]
527  ], [
528  game_or_payoffs.col_action_name(action)
529  for action in live_actions[1].nonzero()[0]
530  ], *payoffs), live_actions
531  else:
532  return payoffs, live_actions
