27 VPNetEvaluator::VPNetEvaluator(DeviceManager* device_manager, int batch_size,
28  int threads, int cache_size, int cache_shards)
29  : device_manager_(*device_manager), batch_size_(batch_size),
30  queue_(batch_size * threads * 4), batch_size_hist_(batch_size + 1) {
31  cache_shards = std::max(1, cache_shards);
32  cache_.reserve(cache_shards);
33  for (int i = 0; i < cache_shards; ++i) {
34  cache_.push_back(
35  std::make_unique<LRUCache<uint64_t, VPNetModel::InferenceOutputs>>(
36  cache_size / cache_shards));
37  }
38  if (batch_size_ <= 1) {
39  threads = 0;
40  }
41  inference_threads_.reserve(threads);
42  for (int i = 0; i < threads; ++i) {
43  inference_threads_.emplace_back([this]() { this->Runner(); });
44  }
45 }

80 VPNetModel::InferenceOutputs VPNetEvaluator::Inference(const State& state) {
81  VPNetModel::InferenceInputs inputs = {
82  state.LegalActions(), state.ObservationTensor()};
83 
84  uint64_t key;
85  int cache_shard;
86  if (!cache_.empty()) {
87  key = absl::Hash<VPNetModel::InferenceInputs>{}(inputs);
88  cache_shard = key % cache_.size();
89  absl::optional<const VPNetModel::InferenceOutputs> opt_outputs =
90  cache_[cache_shard]->Get(key);
91  if (opt_outputs) {
92  return *opt_outputs;
93  }
94  }
95  VPNetModel::InferenceOutputs outputs;
96  if (batch_size_ <= 1) {
97  outputs = device_manager_.Get(1)->Inference(std::vector{inputs})[0];
98  } else {
99  std::promise<VPNetModel::InferenceOutputs> prom;
100  std::future<VPNetModel::InferenceOutputs> fut = prom.get_future();
101  queue_.Push(QueueItem{inputs, &prom});
102  outputs = fut.get();
103  }
104  if (!cache_.empty()) {
105  cache_[cache_shard]->Set(key, outputs);
106  }
107  return outputs;
108 }

110 void VPNetEvaluator::Runner() {
111  std::vector<VPNetModel::InferenceInputs> inputs;
112  std::vector<std::promise<VPNetModel::InferenceOutputs>*> promises;
113  inputs.reserve(batch_size_);
114  promises.reserve(batch_size_);
115  while (!stop_.StopRequested()) {
116  {
117  // Only one thread at a time should be listening to the queue to maximize
118  // batch size and minimize latency.
119  absl::MutexLock lock(&inference_queue_m_);
120  absl::Time deadline = absl::InfiniteFuture();
121  for (int i = 0; i < batch_size_; ++i) {
122  absl::optional<QueueItem> item = queue_.Pop(deadline);
123  if (!item) { // Hit the deadline.
124  break;
125  }
126  if (inputs.empty()) {
127  deadline = absl::Now() + absl::Milliseconds(1);
128  }
129  inputs.push_back(item->inputs);
130  promises.push_back(item->prom);
131  }
132  }
133 
134  if (inputs.empty()) { // Almost certainly StopRequested.
135  continue;
136  }
137 
138  {
139  absl::MutexLock lock(&stats_m_);
140  batch_size_stats_.Add(inputs.size());
141  batch_size_hist_.Add(inputs.size());
142  }
143 
144  std::vector<VPNetModel::InferenceOutputs> outputs =
145  device_manager_.Get(inputs.size())->Inference(inputs);
146  for (int i = 0; i < promises.size(); ++i) {
147  promises[i]->set_value(outputs[i]);
148  }
149  inputs.clear();
150  promises.clear();
151  }
152 }
