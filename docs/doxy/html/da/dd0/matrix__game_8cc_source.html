29// Check the utilities to see if the game is constant-sum or identical
30// (cooperative).
31GameType::Utility GetUtilityType(const std::vector<double>& row_player_utils,
32 const std::vector<double>& col_player_utils) {
33 double util_sum = 0;
34 // Assume both are true until proven otherwise.
35 bool constant_sum = true;
36 bool identical = true;
37 for (int i = 0; i < row_player_utils.size(); ++i) {
38 if (i == 0) {
39 util_sum = row_player_utils[i] + col_player_utils[i];
40 } else {
41 if (constant_sum &&
42 !Near(row_player_utils[i] + col_player_utils[i], util_sum)) {
43 constant_sum = false;
44 }
45 }
46
47 if (identical && row_player_utils[i] != col_player_utils[i]) {
48 identical = false;
49 }
50 }
51
52 if (constant_sum && Near(util_sum, 0.0)) {
53 return GameType::Utility::kZeroSum;
54 } else if (constant_sum) {
55 return GameType::Utility::kConstantSum;
56 } else if (identical) {
57 return GameType::Utility::kIdentical;
58 } else {
59 return GameType::Utility::kGeneralSum;
60 }
61}

63absl::optional<double> GetUtilitySum(
64 const std::vector<double>& row_player_utils,
65 const std::vector<double>& col_player_utils) {
66 double util_sum = 0;
67 bool constant_sum = true;
68 for (int i = 0; i < row_player_utils.size(); ++i) {
69 if (i == 0) {
70 util_sum = row_player_utils[i] + col_player_utils[i];
71 } else {
72 if (constant_sum &&
73 !Near(row_player_utils[i] + col_player_utils[i], util_sum)) {
74 constant_sum = false;
75 }
76 }
77 }
78
79 if (constant_sum) {
80 return Near(util_sum, 0.0) ? 0 : util_sum;
81 } else {
82 return absl::nullopt;
83 }
84}

92std::string MatrixState::ToString() const {
93 std::string result = "";
94 absl::StrAppend(&result, "Terminal? ", IsTerminal() ? "true" : "false", "\n");
95 if (IsTerminal()) {
96 absl::StrAppend(&result, "History: ", HistoryString(), "\n");
97 absl::StrAppend(&result, "Returns: ", absl::StrJoin(Returns(), ","), "\n");
98 }
99 absl::StrAppend(&result, "Row actions: ");
100 for (auto move : LegalActions(0)) {
101 absl::StrAppend(&result, ActionToString(0, move), " ");
102 }
103 absl::StrAppend(&result, "\nCol actions: ");
104 for (auto move : LegalActions(1)) {
105 absl::StrAppend(&result, ActionToString(1, move), " ");
106 }
107 absl::StrAppend(&result, "\nUtility matrix:\n");
108 for (int r = 0; r < matrix_game_->NumRows(); r++) {
109 for (int c = 0; c < matrix_game_->NumCols(); c++) {
110 absl::StrAppend(&result, matrix_game_->RowUtility(r, c), ",",
111 matrix_game_->ColUtility(r, c), " ");
112 }
113 absl::StrAppend(&result, "\n");
114 }
115 return result;
116}

122std::vector<double> FlattenMatrix(
123 const std::vector<std::vector<double>>& matrix_rows) {
124 std::vector<double> utilities;
125 int total_size = 0;
126 int row_size = -1;
127 int i = 0;
128
129 for (int r = 0; r < matrix_rows.size(); ++r) {
130 if (row_size < 0) {
131 row_size = matrix_rows[r].size();
132 }
133 SPIEL_CHECK_GT(row_size, 0);
134 SPIEL_CHECK_EQ(row_size, matrix_rows[r].size());
135 total_size += row_size;
136 utilities.resize(total_size, 0);
137
138 for (int c = 0; c < matrix_rows[r].size(); ++c) {
139 utilities[i] = matrix_rows[r][c];
140 ++i;
141 }
142 }
143
144 return utilities;
145}

147std::shared_ptr<const MatrixGame> CreateMatrixGame(
148 const std::vector<std::vector<double>>& row_player_utils,
149 const std::vector<std::vector<double>>& col_player_utils) {
150 SPIEL_CHECK_GT(row_player_utils.size(), 0);
151 int num_rows = row_player_utils.size();
152 int num_columns = row_player_utils[0].size();
153 std::vector<std::string> row_names(num_rows);
154 std::vector<std::string> col_names(num_columns);
155 for (int i = 0; i < num_rows; ++i) {
156 row_names[i] = absl::StrCat("row", i);
157 }
158 for (int i = 0; i < num_columns; ++i) {
159 col_names[i] = absl::StrCat("col", i);
160 }
161 return CreateMatrixGame("short_name", "Long Name", row_names, col_names,
162 row_player_utils, col_player_utils);
163}

165// Create a matrix game with the specified utilities and row/column names.
166// Utilities must be in row-major form.
167std::shared_ptr<const MatrixGame> CreateMatrixGame(
168 const std::string& short_name, const std::string& long_name,
169 const std::vector<std::string>& row_names,
170 const std::vector<std::string>& col_names,
171 const std::vector<std::vector<double>>& row_player_utils,
172 const std::vector<std::vector<double>>& col_player_utils) {
173 int rows = row_names.size();
174 int columns = col_names.size();
175 std::vector<double> flat_row_utils = FlattenMatrix(row_player_utils);
176 std::vector<double> flat_col_utils = FlattenMatrix(col_player_utils);
177 SPIEL_CHECK_EQ(flat_row_utils.size(), rows * columns);
178 SPIEL_CHECK_EQ(flat_col_utils.size(), rows * columns);
179 return CreateMatrixGame(short_name, long_name, row_names, col_names,
180 flat_row_utils, flat_col_utils);
181}

183std::shared_ptr<const MatrixGame> CreateMatrixGame(
184 const std::string& short_name, const std::string& long_name,
185 const std::vector<std::string>& row_names,
186 const std::vector<std::string>& col_names,
187 const std::vector<double>& flat_row_utils,
188 const std::vector<double>& flat_col_utils) {
189 // Detect the utility type from the utilities.
190 GameType::Utility utility = GetUtilityType(flat_row_utils, flat_col_utils);
191
192 GameType game_type{
193 /*short_name=*/short_name,
194 /*long_name=*/long_name,
195 GameType::Dynamics::kSimultaneous,
196 GameType::ChanceMode::kDeterministic,
197 GameType::Information::kOneShot,
198 utility,
199 GameType::RewardModel::kTerminal,
200 /*max_num_players=*/2,
201 /*min_num_players=*/2,
202 /*provides_information_state_string=*/true,
203 /*provides_information_state_tensor=*/true,
204 /*provides_observation_string=*/true,
205 /*provides_observation_tensor=*/true,
206 /*parameter_specification=*/{} // no parameters
207 };
208
209 return std::shared_ptr<const MatrixGame>(new MatrixGame(
210 game_type, {}, row_names, col_names, flat_row_utils, flat_col_utils));
211}
