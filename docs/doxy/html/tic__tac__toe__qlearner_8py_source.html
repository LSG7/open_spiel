15"""Tabular Q-Learner example on Tic Tac Toe.
16
17Two Q-Learning agents are trained by playing against each other. Then, the game
18can be played against the agents from the command line.
19
20After about 10**5 training episodes, the agents reach a good policy: win rate
21against random opponents is around 99% for player 0 and 92% for player 1.
22"""

44def pretty_board(time_step):
45 """Returns the board in `time_step` in a human readable format."""
46 info_state = time_step.observations["info_state"][0]
47 x_locations = np.nonzero(info_state[9:18])[0]
48 o_locations = np.nonzero(info_state[18:])[0]
49 board = np.full(3 * 3, ".")
50 board[x_locations] = "X"
51 board[o_locations] = "0"
52 board = np.reshape(board, (3, 3))
53 return board

56def command_line_action(time_step):
57 """Gets a valid action from the user on the command line."""
58 current_player = time_step.observations["current_player"]
59 legal_actions = time_step.observations["legal_actions"][current_player]
60 action = -1
61 while action not in legal_actions:
62 print("Choose an action from {}:".format(legal_actions))
63 sys.stdout.flush()
64 action_str = input()
65 try:
66 action = int(action_str)
67 except ValueError:
68 continue
69 return action

72def eval_against_random_bots(env, trained_agents, random_agents, num_episodes):
73 """Evaluates `trained_agents` against `random_agents` for `num_episodes`."""
74 wins = np.zeros(2)
75 for player_pos in range(2):
76 if player_pos == 0:
77 cur_agents = [trained_agents[0], random_agents[1]]
78 else:
79 cur_agents = [random_agents[0], trained_agents[1]]
80 for _ in range(num_episodes):
81 time_step = env.reset()
82 while not time_step.last():
83 player_id = time_step.observations["current_player"]
84 agent_output = cur_agents[player_id].step(time_step, is_evaluation=True)
85 time_step = env.step([agent_output.action])
86 if time_step.rewards[player_pos] > 0:
87 wins[player_pos] += 1
88 return wins / num_episodes

91def main(_):
92 game = "tic_tac_toe"
93 num_players = 2
94
95 env = rl_environment.Environment(game)
96 num_actions = env.action_spec()["num_actions"]
97
98 agents = [
99 tabular_qlearner.QLearner(player_id=idx, num_actions=num_actions)
100 for idx in range(num_players)
101 ]
102
103 # random agents for evaluation
104 random_agents = [
105 random_agent.RandomAgent(player_id=idx, num_actions=num_actions)
106 for idx in range(num_players)
107 ]
108
109 # 1. Train the agents
110 training_episodes = FLAGS.num_episodes
111 for cur_episode in range(training_episodes):
112 if cur_episode % int(1e4) == 0:
113 win_rates = eval_against_random_bots(env, agents, random_agents, 1000)
114 logging.info("Starting episode %s, win_rates %s", cur_episode, win_rates)
115 time_step = env.reset()
116 while not time_step.last():
117 player_id = time_step.observations["current_player"]
118 agent_output = agents[player_id].step(time_step)
119 time_step = env.step([agent_output.action])
120
121 # Episode is over, step all agents with final info state.
122 for agent in agents:
123 agent.step(time_step)
124
125 if not FLAGS.interactive_play:
126 return
127
128 # 2. Play from the command line against the trained agent.
129 human_player = 1
130 while True:
131 logging.info("You are playing as %s", "O" if human_player else "X")
132 time_step = env.reset()
133 while not time_step.last():
134 player_id = time_step.observations["current_player"]
135 if player_id == human_player:
136 agent_out = agents[human_player].step(time_step, is_evaluation=True)
137 logging.info("\n%s", agent_out.probs.reshape((3, 3)))
138 logging.info("\n%s", pretty_board(time_step))
139 action = command_line_action(time_step)
140 else:
141 agent_out = agents[1 - human_player].step(time_step, is_evaluation=True)
142 action = agent_out.action
143 time_step = env.step([action])
144
145 logging.info("\n%s", pretty_board(time_step))
146
147 logging.info("End of game!")
148 if time_step.rewards[human_player] > 0:
149 logging.info("You win")
150 elif time_step.rewards[human_player] < 0:
151 logging.info("You lose")
152 else:
153 logging.info("Draw")
154 # Switch order of players
155 human_player = 1 - human_player
156
157
158if __name__ == "__main__":
159 app.run(main)
