27class JointPolicyAggregatorTest(parameterized.TestCase):
28
29 @parameterized.named_parameters(
30 {
31 "testcase_name": "kuhn_poker",
32 "game_name": "kuhn_poker"
33 }, {
34 "testcase_name": "leduc_poker",
35 "game_name": "leduc_poker"
36 })
37 def test_policy_aggregation_random(self, game_name):
38 env = rl_environment.Environment(game_name)
39 num_players = 2
40 num_joint_policies = 4
41
42 joint_policies = [[
43 policy.UniformRandomPolicy(env.game) for _ in range(num_players)
44 ] for _ in range(num_joint_policies)]
45 probabilities = np.ones(len(joint_policies))
46 probabilities /= np.sum(probabilities)
47
48 pol_ag = policy_aggregator_joint.JointPolicyAggregator(env.game)
49 aggr_policy = pol_ag.aggregate([0, 1], joint_policies, probabilities)
50
51 self.assertLen(aggr_policy.policies, num_players)
52 for player in range(num_players):
53 player_policy = aggr_policy.policies[player]
54 self.assertNotEmpty(player_policy)
55 for state_action_probs in player_policy.values():
56 probs = list(state_action_probs.values())
57 expected_prob = 1. / len(probs)
58 for prob in probs:
59 self.assertAlmostEqual(expected_prob, prob, places=10)

63 absltest.main()
