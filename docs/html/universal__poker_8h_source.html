138  logic::CardSet HoleCards(Player player) const {
139  logic::CardSet hole_cards;
140  const int num_players = acpc_game_->GetNbPlayers();
141  const int num_cards_dealt_to_all = hole_cards_dealt_ / num_players;
142  int num_cards_dealt_to_player = num_cards_dealt_to_all;
143  // We deal to players in order from 0 to n - 1. So if the number of cards
144  // dealt % num_players is > the player, we haven't dealt them a card yet;
145  // otherwise we have.
146  if (player < (hole_cards_dealt_ % num_players) &&
147  num_cards_dealt_to_all < acpc_game_->GetNbHoleCardsRequired()) {
148  ++num_cards_dealt_to_player;
149  }
150  SPIEL_CHECK_LT(player, acpc_game_->GetNbPlayers());
151  SPIEL_CHECK_LE(num_cards_dealt_to_player,
152  static_cast<int>(acpc_game_->GetNbHoleCardsRequired()));
153  for (int i = 0; i < num_cards_dealt_to_player; ++i) {
154  hole_cards.AddCard(acpc_state_.hole_cards(player, i));
155  }
156  return hole_cards;
157  }

231 // Only supported for UniversalPoker. Randomly plays an action from a fixed list
232 // of actions. If none of the actions are legal, checks/calls.
233 class UniformRestrictedActions : public Policy {
234  public:
235  // Actions will be restricted to this list when legal. If no such action is
236  // legal, checks/calls.
237  explicit UniformRestrictedActions(absl::Span<const ActionType> actions)
238  : actions_(actions.begin(), actions.end()),
239  max_action_(*absl::c_max_element(actions)) {}
240 
241  ActionsAndProbs GetStatePolicy(const State &state) const {
242  ActionsAndProbs policy;
243  policy.reserve(actions_.size());
244  const std::vector<Action> legal_actions = state.LegalActions();
245  for (Action action : legal_actions) {
246  if (actions_.contains(static_cast<ActionType>(action))) {
247  policy.emplace_back(action, 1.);
248  }
249  if (policy.size() >= actions_.size() || action > max_action_) break;
250  }
251 
252  // It is always legal to check/call.
253  if (policy.empty()) {
254  SPIEL_DCHECK_TRUE(absl::c_find(legal_actions, ActionType::kCall) !=
255  legal_actions.end());
256  policy.push_back({static_cast<Action>(ActionType::kCall), 1.});
257  }
258 
259  // If we have a non-empty policy, normalize it!
260  if (policy.size() > 1) NormalizePolicy(&policy);
261  return policy;
262  }
263 
264  private:
265  const absl::flat_hash_set<ActionType> actions_;
266  const ActionType max_action_;
267 };
