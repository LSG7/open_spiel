15"""Implements ResponseGraphUCB algorithm from the below paper.
16
17 "Multiagent Evaluation under Incomplete Information" (Rowland et al., 2019)
18 See https://arxiv.org/abs/1909.09849 for details.
19"""

33class ResponseGraphUCB(object):
34 """ResponseGraphUCB sampler class."""
35
36 def __init__(self,
37 game,
38 exploration_strategy='uniform-exhaustive',
39 confidence_method='ucb-standard',
40 delta=0.01,
41 ucb_eps=0,
42 per_payoff_confidence=True,
43 time_dependent_delta=False):
44 """Initializes ResponseGraphUCB instance.
45
46 Assumes that all payoffs fall in the interval [0,1].
47
48 Args:
49 game: an instance of the BernoulliGameSampler class.
50 exploration_strategy: string specifying the exploration strategy.
51 confidence_method: string specifying the confidence method.
52 delta: float specifying the UCB delta parameter.
53 ucb_eps: float specifying the UCB epsilon parameter.
54 per_payoff_confidence: bool specifying whether confidence level applies
55 on a per-payoff basis, or to all payoffs simultaneously.
56 time_dependent_delta: bool specifying whether the confidence parameter
57 varies with the number of interactions so that a union bound holds.
58 """
59 self.exploration_strategy = exploration_strategy
60 self.confidence_method = confidence_method
61 self.ucb_eps = ucb_eps
62 self.G = game # pylint: disable=invalid-name
63 self.per_payoff_confidence = per_payoff_confidence
64 self.time_dependent_delta = time_dependent_delta
65 if self.per_payoff_confidence:
66 self._delta = delta
67 else:
68 self._delta = delta / (
69 self.G.n_players *
70 functools.reduce(operator.mul, self.G.strategy_spaces, 1))
71
72 # Compute the graph
73 self.V = list( # pylint: disable=invalid-name
74 itertools.product(*[range(smax) for smax in self.G.strategy_spaces]))
75 self.E = [] # pylint: disable=invalid-name
76 for v in self.V:
77 adj_strats = [
78 list(range(v[k] + 1, self.G.strategy_spaces[k]))
79 for k in range(self.G.n_players)
80 ]
81 for k in range(self.G.n_players):
82 for new_s in adj_strats[k]:
83 second_vertex = list(v)
84 second_vertex[k] = new_s
85 second_vertex = tuple(second_vertex)
86 self.E.append((v, second_vertex))
87 self.count_history = {v: [] for v in self.V}
88 self.total_interactions = 0
89
90 def delta(self, k, s):
91 """Returns the confidence parameter for a given player and profile."""
92 if not self.time_dependent_delta:
93 return self._delta
94 else:
95 return self._delta * (6 / (np.pi**2 * self.count[k][s] **2))
96
97 def initialise_mean_and_count(self):
98 """Initializes means and counts for all response graph profiles."""
99 self.mu = [
100 np.zeros(tuple(self.G.strategy_spaces)) for _ in range(self.G.n_players)
101 ]
102 self.count = [
103 np.zeros(tuple(self.G.strategy_spaces)) for _ in range(self.G.n_players)
104 ]
105
106 def update_mean_and_count(self, strat_profile, game_outcome):
107 """Updates means and counts for strat_profile given game_outcome."""
108 self.total_interactions += 1
109 for k in range(self.G.n_players):
110 self.mu[k][strat_profile] *= self.count[k][strat_profile]
111 self.mu[k][strat_profile] += game_outcome[k]
112 self.count[k][strat_profile] += 1
113 self.mu[k][strat_profile] /= self.count[k][strat_profile]
114
115 for s in self.V:
116 self.count_history[s].append(self.count[0][s] /
117 float(self.total_interactions))
118
119 def _find_focal_coord(self, s1, s2):
120 num_deviations = tuple(s1[l] != s2[l] for l in range(len(s1)))
121 assert np.sum(num_deviations) == 1, ('Invalid profile pair s1, s2: ({},{}).'
122 'Exactly one player should'
123 'deviate!'.format(s1, s2))
124 return np.argmax(num_deviations)

126 def _initialise_queue_uniform(self):
127 self.remaining_edges = copy.deepcopy(self.E)
128
129 def _add_to_queue_uniform(self, edges_removed):
130 """Adds edge to sampling queue using uniform sampling."""
131 for e in edges_removed:
132 self.remaining_edges.remove(e)
133 self.profile_queue.append(
134 random.choice(random.choice(self.remaining_edges)))
135
136 def _initialise_queue_uniform_exhaustive(self):
137 self.edge_order = copy.deepcopy(self.E)
138 random.shuffle(self.edge_order)
139
140 def _add_to_queue_uniform_exhaustive(self, edges_removed):
141 """Adds edge to sampling queue using uniform-exhausitive sampling."""
142 for e in edges_removed:
143 self.edge_order.remove(e)
144 self.profile_queue.append(random.choice(self.edge_order[0]))
145
146 def _initialise_queue_valence_weighted(self):
147 self.vertex_valences = {
148 v: np.sum(self.G.strategy_spaces) - self.G.n_players for v in self.V
149 }
150 self.sum_valences = sum(self.vertex_valences.values())
151
152 def _add_to_queue_valence_weighted(self, edges_removed):
153 """Adds edge to sampling queue using valence-weighted sampling."""
154 # Deal with removed edges
155 for e in edges_removed:
156 for s in e:
157 self.vertex_valences[s] -= 1
158 self.sum_valences -= 1
159
160 # Calculate probabilities
161 probs = np.array([self.vertex_valences[v]**2 for v in self.V])
162 probs = probs / np.sum(probs)
163 s_ix = np.random.choice(np.arange(len(self.V)), p=probs)
164 self.profile_queue.append(self.V[s_ix])
165
166 def _initialise_queue_count_weighted(self):
167 # Keep track of which vertices have non-zero valence in graph
168 self.vertex_valences = {
169 v: np.sum(self.G.strategy_spaces) - self.G.n_players for v in self.V
170 }
171 self.sum_valences = sum(self.vertex_valences.values())
172
173 def _add_to_queue_count_weighted(self, edges_removed):
174 """Adds edge to sampling queue using count-weighted sampling."""
175 # Update vertex valences
176 for e in edges_removed:
177 for s in e:
178 self.vertex_valences[s] -= 1
179 self.sum_valences -= 1
180 # Check counts
181 eligible_vertices = {
182 v: self.count[0][v] for v in self.V if self.vertex_valences[v] != 0
183 }
184 strat = min(eligible_vertices, key=eligible_vertices.get)
185 self.profile_queue.append(strat)
186
187 def initialise_queue(self):
188 """Initializes sampling queue."""
189 self.edges_remaining = copy.deepcopy(self.E)
190 if self.exploration_strategy == 'uniform':
191 self._initialise_queue_uniform()
192 elif self.exploration_strategy == 'uniform-exhaustive':
193 self._initialise_queue_uniform_exhaustive()
194 elif self.exploration_strategy == 'valence-weighted':
195 self._initialise_queue_valence_weighted()
196 elif self.exploration_strategy == 'count-weighted':
197 self._initialise_queue_count_weighted()
198 else:
199 raise ValueError('Did not recognise exploration strategy: {}'.format(
200 self.exploration_strategy))
201
202 self.profile_queue = []
203
204 def add_to_queue(self, removed):
205 """Update the sampling queue and the list of resolved edges.
206
207 Args:
208 removed: the list of edges resolved in the previous round, which should be
209 removed from the sampling list in subsequent rounds.
210 """
211 if self.exploration_strategy == 'uniform':
212 self._add_to_queue_uniform(removed)
213 elif self.exploration_strategy == 'uniform-exhaustive':
214 self._add_to_queue_uniform_exhaustive(removed)
215 elif self.exploration_strategy == 'valence-weighted':
216 self._add_to_queue_valence_weighted(removed)
217 elif self.exploration_strategy == 'count-weighted':
218 self._add_to_queue_count_weighted(removed)
219 else:
220 raise ValueError('Did not recognise exploration strategy: {}'.format(
221 self.exploration_strategy))

223 def evaluate_strategy_profile(self, yield_outcomes=False):
224 """Evaluates a strategy profile on the sampling queue.
225
226 Specifically, this:
227 1. Removes a strategy profile from the queue.
228 2. Evaluates it.
229 3. Updates internal statistics.
230 4. Adjusts list of strategy profiles whose statistics have been updated
231 since last confidence bound check.
232
233 Args:
234 yield_outcomes: set True to yield the outcomes as well.
235
236 Yields:
237 s: profile evaluated.
238 game_outcome: outcomes (player payoffs) for profile s.
239 """
240 if self.profile_queue:
241 s = self.profile_queue.pop(0)
242 if s not in self.active_strategy_profiles:
243 self.active_strategy_profiles.append(s)
244 game_outcome = self.G.observe_result(s)
245 if yield_outcomes:
246 yield s, game_outcome
247 self.update_mean_and_count(s, game_outcome)

249 def _ucb_standard_factor(self, s, k):
250 return np.sqrt(np.log(2 / self.delta(k, s)) / (2 * self.count[k][s]))
251
252 def _bernoulli_upper(self, p, n, delta):
253 """Returns upper confidence bound for proportion p successes of n trials.
254
255 Uses exact Clopper-Pearson interval.
256
257 Args:
258 p: proportion of successes.
259 n: number of trials.
260 delta: confidence parameter.
261 """
262 if p > 1 - 1e-6:
263 return 1.
264 else:
265 upper = scipy.stats.beta.ppf(1. - delta / 2, p * n + 1, n - p * n)
266 return upper
267
268 def _bernoulli_lower(self, p, n, delta):
269 """Returns lower confidence bound for proportion p successes of n trials.
270
271 Uses exact Clopper-Pearson interval.
272
273 Args:
274 p: proportion of successes.
275 n: number of trials.
276 delta: confidence parameter.
277 """
278 if p < 1e-6:
279 return 0.
280 else:
281 lower = scipy.stats.beta.ppf(delta / 2, p * n, n - p * n + 1)
282 return lower

284 def _ucb(self, s, k):
285 """Returns k-th player's payoff upper-confidence-bound given profile s."""
286 if self.confidence_method == 'ucb-standard':
287 ucb_factor = self._ucb_standard_factor(s, k)
288 return self.mu[k][s] + ucb_factor
289 elif self.confidence_method == 'ucb-standard-relaxed':
290 ucb_factor = self._ucb_standard_factor(s, k) - self.ucb_eps
291 return self.mu[k][s] + ucb_factor
292 elif self.confidence_method == 'clopper-pearson-ucb':
293 return self._bernoulli_upper(self.mu[k][s], self.count[k][s],
294 self.delta(k, s))
295 elif self.confidence_method == 'clopper-pearson-ucb-relaxed':
296 return self._bernoulli_upper(self.mu[k][s], self.count[k][s],
297 self.delta(k, s)) - self.ucb_eps
298 else:
299 raise ValueError('Did not recognise confidence method {}'.format(
300 self.confidence_method))
301
302 def _lcb(self, s, k):
303 """Returns k-th player's payoff lower-confidence-bound given profile s."""
304 if self.confidence_method == 'ucb-standard':
305 ucb_factor = self._ucb_standard_factor(s, k)
306 return self.mu[k][s] - ucb_factor
307 elif self.confidence_method == 'ucb-standard-relaxed':
308 ucb_factor = self._ucb_standard_factor(s, k) + self.ucb_eps
309 return self.mu[k][s] - ucb_factor
310 elif self.confidence_method == 'clopper-pearson-ucb':
311 return self._bernoulli_lower(self.mu[k][s], self.count[k][s],
312 self.delta(k, s))
313 elif self.confidence_method == 'clopper-pearson-ucb-relaxed':
314 return self._bernoulli_lower(self.mu[k][s], self.count[k][s],
315 self.delta(k, s)) + self.ucb_eps
316 else:
317 raise ValueError('Did not recognise confidence method {}'.format(
318 self.confidence_method))

320 def ucb_check(self, e):
321 """Conducts a UCB check on response graph edge e.
322
323 Specifically, given edge e connecting two strategy profiles s1 and s2, this:
324 1. Determines the dominating strategy.
325 2. Checks whether the payoff_UCB(worse_strategy) is less than
326 the payoff_LCB of the better strategy; if this is true, the confidence
327 intervals are disjoint, and the edge e is considered 'resolved'.
328
329 Args:
330 e: response graph edge.
331
332 Returns:
333 A bool indicating whether the edge is resolved,
334 and also a tuple specifying the worse and better strategies.
335 """
336
337 s1, s2 = e
338 k = self._find_focal_coord(s1, s2)
339 if self.mu[k][s1] > self.mu[k][s2]:
340 better_strat = s1
341 worse_strat = s2
342 else:
343 better_strat = s2
344 worse_strat = s1
345
346 ucb = self._ucb(worse_strat, k)
347 lcb = self._lcb(better_strat, k)
348
349 return (ucb < lcb), (worse_strat, better_strat)
350
351 def check_confidence(self):
352 """Returns the edges that are 'resolved' given a confidence bound check."""
353 edges_to_check = []
354
355 for e in self.edges_remaining:
356 for s in self.active_strategy_profiles:
357 if s in e:
358 if e not in edges_to_check:
359 edges_to_check.append(e)
360
361 edges_removed = []
362 for e in edges_to_check:
363 removed, ordered_edge = self.ucb_check(e)
364 if removed:
365 edges_removed.append(e)
366 self.edges_remaining.remove(e)
367 self.directed_edges.append(ordered_edge)
368
369 self.active_strategy_profiles = []
370
371 return edges_removed

373 def real_edge_direction(self, e):
374 s1, s2 = e
375 k = self._find_focal_coord(s1, s2)
376 if self.G.means[k][s1] > self.G.means[k][s2]:
377 return (s2, s1)
378 else:
379 return (s1, s2)

381 def construct_real_graph(self):
382 directed_edges = []
383 for e in self.E:
384 ordered_edge = self.real_edge_direction(e)
385 directed_edges.append(ordered_edge)
386
387 return self._construct_digraph(directed_edges)

389 def compute_graph(self):
390 for e in self.E:
391 s1, s2 = e[0], e[1]
392 k = self._find_focal_coord(s1, s2)
393 if self.mu[k][s1] > self.mu[k][s2]:
394 directed_edge = (s2, s1)
395 else:
396 directed_edge = (s1, s2)
397 if directed_edge not in self.directed_edges:
398 self.directed_edges.append(directed_edge)

400 def forced_exploration(self):
401 for v in self.V:
402 game_outcome = self.G.observe_result(v)
403 self.update_mean_and_count(v, game_outcome)

405 def run(self, verbose=True, max_total_iterations=50000):
406 """Runs the ResponseGraphUCB algorithm."""
407 self.verbose = verbose
408
409 # Upper bounds on number of evaluations
410 self.max_total_iterations = max_total_iterations
411
412 self.initialise_mean_and_count()
413 self.directed_edges = []
414 self.active_strategy_profiles = []
415 self.initialise_queue()
416
417 # Forced initial exploration
418 self.forced_exploration()
419
420 # Keep evaluating nodes until check method declares that we're finished
421 iterations = 0
422 edges_resolved_this_round = []
423 while self.total_interactions < max_total_iterations:
424 # Add nodes to queue
425 self.add_to_queue(removed=edges_resolved_this_round)
426
427 # Evaluate the nodes and log results
428 for v, _ in self.evaluate_strategy_profile():
429 if verbose:
430 print(v)
431
432 # Recompute confidence bounds, eliminate, stop etc.
433 edges_resolved_this_round = self.check_confidence()
434
435 if not self.edges_remaining:
436 break
437 iterations += 1
438
439 # Fill in missing edges if max iters reached without resolving all edges
440 self.compute_graph()
441
442 # Compute objects to be returned
443 if verbose:
444 total_steps = self.compute_total_steps()
445 print('\nTotal steps taken = {}'.format(total_steps))
446 results = {}
447 results['interactions'] = int(np.sum(self.count[0]))
448 graph = self._construct_digraph(self.directed_edges)
449 results['graph'] = graph
450 return results
451
452 def compute_total_steps(self):
453 return int(np.sum(self.count[0]))
454
455 def _construct_digraph(self, edges):
456 graph = nx.DiGraph()
457 graph.add_nodes_from(self.V)
458 for e in edges:
459 graph.add_edge(e[0], e[1])
460 return graph
