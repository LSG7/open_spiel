52std::unique_ptr<open_spiel::Bot> InitBot(
53 std::string type, const open_spiel::Game& game, open_spiel::Player player,
54 std::shared_ptr<open_spiel::algorithms::Evaluator> evaluator) {
55 if (type == "random") {
56 return open_spiel::MakeUniformRandomBot(player, Seed());
57 }
58
59 if (type == "mcts") {
60 return std::make_unique<open_spiel::algorithms::MCTSBot>(
61 game, std::move(evaluator), absl::GetFlag(FLAGS_uct_c),
62 absl::GetFlag(FLAGS_max_simulations),
63 absl::GetFlag(FLAGS_max_memory_mb), absl::GetFlag(FLAGS_solve), Seed(),
64 absl::GetFlag(FLAGS_verbose));
65 }
66 open_spiel::SpielFatalError("Bad player type. Known types: mcts, random");
67}

69open_spiel::Action GetAction(const open_spiel::State& state,
70 std::string action_str) {
71 for (open_spiel::Action action : state.LegalActions()) {
72 if (action_str == state.ActionToString(state.CurrentPlayer(), action))
73 return action;
74 }
75 return open_spiel::kInvalidAction;
76}

78std::pair<std::vector<double>, std::vector<std::string>> PlayGame(
79 const open_spiel::Game& game,
80 std::vector<std::unique_ptr<open_spiel::Bot>>& bots, std::mt19937& rng,
81 const std::vector<std::string>& initial_actions) {
82 bool quiet = absl::GetFlag(FLAGS_quiet);
83 std::unique_ptr<open_spiel::State> state = game.NewInitialState();
84 std::vector<std::string> history;
85
86 for (const auto& action_str : initial_actions) {
87 open_spiel::Action action = GetAction(*state, action_str);
88 if (action == open_spiel::kInvalidAction)
89 open_spiel::SpielFatalError(absl::StrCat("Invalid action: ", action_str));
90
91 history.push_back(action_str);
92 state->ApplyAction(action);
93 if (!quiet) {
94 std::cerr << "Forced action" << action_str << std::endl;
95 std::cerr << "Next state:\n" << state->ToString() << std::endl;
96 }
97 }
98
99 while (!state->IsTerminal()) {
100 open_spiel::Player player = state->CurrentPlayer();
101 if (!quiet) std::cerr << "player turn: " << player << std::endl;
102
103 open_spiel::Action action;
104 if (state->IsChanceNode()) {
105 // Chance node; sample one according to underlying distribution.
106 open_spiel::ActionsAndProbs outcomes = state->ChanceOutcomes();
107 action = open_spiel::SampleAction(outcomes, rng).first;
108 if (!quiet)
109 std::cerr << "Sampled action: " << state->ActionToString(player, action)
110 << std::endl;
111 } else if (state->IsSimultaneousNode()) {
112 open_spiel::SpielFatalError(
113 "MCTS not supported for games with simultaneous actions.");
114 } else {
115 // Decision node, ask the right bot to make its action
116 action = bots[player]->Step(*state);
117 if (!quiet)
118 std::cerr << "Chose action: " << state->ActionToString(player, action)
119 << std::endl;
120 }
121 for (open_spiel::Player p = 0; p < bots.size(); ++p) {
122 if (p != player) {
123 bots[p]->InformAction(*state, player, action);
124 }
125 }
126 history.push_back(state->ActionToString(player, action));
127 state->ApplyAction(action);
128
129 if (!quiet)
130 std::cerr << "State: " << std::endl << state->ToString() << std::endl;
131 }
132
133 std::cerr << "Returns: " << absl::StrJoin(state->Returns(), ",")
134 << " Game actions: " << absl::StrJoin(history, " ") << std::endl;
135 return {state->Returns(), history};
136}

138// Example code for using MCTS agent to play a game
139int main(int argc, char** argv) {
140 std::vector<char*> positional_args = absl::ParseCommandLine(argc, argv);
141 std::mt19937 rng(Seed()); // Random number generator.
142
143 // Create the game
144 std::string game_name = absl::GetFlag(FLAGS_game);
145 std::cerr << "game: " << game_name << std::endl;
146 std::shared_ptr<const open_spiel::Game> game =
147 open_spiel::LoadGame(game_name);
148
149 // MCTS supports arbitrary number of players, but this example assumes
150 // 2-player games.
151 SPIEL_CHECK_TRUE(game->NumPlayers() <= 2);
152
153 auto evaluator =
154 std::make_shared<open_spiel::algorithms::RandomRolloutEvaluator>(
155 absl::GetFlag(FLAGS_rollout_count), Seed());
156
157 std::vector<std::unique_ptr<open_spiel::Bot>> bots;
158 bots.push_back(InitBot(absl::GetFlag(FLAGS_player1), *game, 0, evaluator));
159 bots.push_back(InitBot(absl::GetFlag(FLAGS_player2), *game, 1, evaluator));
160
161 std::vector<std::string> initial_actions;
162 for (int i = 1; i < positional_args.size(); ++i) {
163 initial_actions.push_back(positional_args[i]);
164 }
165
166 absl::btree_map<std::string, int> histories;
167 std::vector<double> overall_returns(2, 0);
168 std::vector<int> overall_wins(2, 0);
169 int num_games = absl::GetFlag(FLAGS_num_games);
170 for (int game_num = 0; game_num < num_games; ++game_num) {
171 auto [returns, history] = PlayGame(*game, bots, rng, initial_actions);
172 histories[absl::StrJoin(history, " ")] += 1;
173 for (int i = 0; i < returns.size(); ++i) {
174 double v = returns[i];
175 overall_returns[i] += v;
176 if (v > 0) {
177 overall_wins[i] += 1;
178 }
179 }
180 }
181
182 std::cerr << "Number of games played: " << num_games << std::endl;
183 std::cerr << "Number of distinct games played: " << histories.size()
184 << std::endl;
185 std::cerr << "Overall wins: " << absl::StrJoin(overall_wins, ",")
186 << std::endl;
187 std::cerr << "Overall returns: " << absl::StrJoin(overall_returns, ",")
188 << std::endl;
189
190 return 0;
191}
