54std::shared_ptr<const TensorGame> ExtensiveToTensorGame(const Game& game) {
55 std::vector<std::vector<std::string>> action_names(game.NumPlayers());
56
57 GameType type = game.GetType();
58
59 std::vector<algorithms::DeterministicTabularPolicy> policies;
60 for (Player player = 0; player < game.NumPlayers(); ++player) {
61 algorithms::DeterministicTabularPolicy policy(game, player);
62 do {
63 action_names[player].push_back(policy.ToString(/*delimiter=*/" --- "));
64 } while (policy.NextPolicy());
65 policy.ResetDefaultPolicy();
66 policies.push_back(policy);
67 }
68 std::vector<const Policy*> policy_ptrs(policies.size());
69 for (Player player = 0; player < game.NumPlayers(); ++player) {
70 policy_ptrs[player] = &policies[player];
71 }
72 const std::unique_ptr<State> initial_state = game.NewInitialState();
73 std::vector<std::vector<double>> utils(game.NumPlayers());
74 bool last_entry;
75 do {
76 std::vector<double> returns = algorithms::ExpectedReturns(
77 *initial_state, policy_ptrs, /*depth_limit=*/-1);
78 for (Player player = 0; player < game.NumPlayers(); ++player) {
79 utils[player].push_back(returns[player]);
80 }
81 last_entry = true;
82 for (auto policy = policies.rbegin(); policy != policies.rend(); ++policy) {
83 if (policy->NextPolicy()) {
84 last_entry = false;
85 break;
86 } else {
87 policy->ResetDefaultPolicy();
88 }
89 }
90 } while (!last_entry);
91
92 return tensor_game::CreateTensorGame(kGameType.short_name,
93 "Normal-form " + type.long_name,
94 action_names, utils);
95}
