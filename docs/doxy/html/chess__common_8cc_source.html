37template <typename KnightOffsets>
38int OffsetToDestinationIndexImpl(const Offset& offset,
39 const KnightOffsets& knight_offsets,
40 int board_size) {
41 // Encodes chess queen moves + knight moves.
42 int move_type = -1;
43 int destination_index = -1;
44 if (offset.x_offset == 0) {
45 // vertical moves
46 move_type = 0;
47 destination_index = DiffToDestinationIndex(offset.y_offset, board_size);
48 } else if (offset.y_offset == 0) {
49 // horizontal moves
50 move_type = 1;
51 destination_index = DiffToDestinationIndex(offset.x_offset, board_size);
52 } else if (offset.x_offset == offset.y_offset) {
53 // left downward or right upward diagonal moves.
54 move_type = 2;
55 destination_index = DiffToDestinationIndex(offset.x_offset, board_size);
56 } else if (offset.x_offset == -offset.y_offset) {
57 // left upward or right downward diagonal moves.
58 move_type = 3;
59 destination_index = DiffToDestinationIndex(offset.x_offset, board_size);
60 } else {
61 // knight moves.
62 move_type = 4;
63 auto itr = std::find(knight_offsets.begin(), knight_offsets.end(), offset);
64 if (itr != knight_offsets.end()) {
65 destination_index = std::distance(knight_offsets.begin(), itr);
66 } else {
67 SpielFatalError(absl::StrCat("Unexpected offset (",
68 static_cast<int>(offset.x_offset), ", ",
69 static_cast<int>(offset.y_offset), ")"));
70 }
71 }
72
73 return move_type * 2 * (board_size - 1) + destination_index;
74}

76template <typename KnightOffsets>
77Offset DestinationIndexToOffsetImpl(int destination_index,
78 const KnightOffsets& knight_offsets,
79 int board_size) {
80 int move_type = destination_index / (2 * (board_size - 1));
81 destination_index = destination_index % (2 * (board_size - 1));
82 int8_t diff = DestinationIndexToDiff(destination_index, board_size);
83
84 if (move_type == 0) {
85 return {0, diff};
86 } else if (move_type == 1) {
87 return {diff, 0};
88 } else if (move_type == 2) {
89 return {diff, diff};
90 } else if (move_type == 3) {
91 return {diff, static_cast<int8_t>(-diff)};
92 } else if (move_type == 4) {
93 SPIEL_CHECK_GE(destination_index, 0);
94 SPIEL_CHECK_LT(destination_index, knight_offsets.size());
95 return knight_offsets[destination_index];
96 } else {
97 SpielFatalError(absl::StrCat("Unexpected move type (", move_type, ")"));
98 }
99}

129std::pair<Square, int> DecodeNetworkTarget(int i, int board_size,
130 int num_actions_destinations) {
131 int xy = i / num_actions_destinations;
132 SPIEL_CHECK_GE(xy, 0);
133 SPIEL_CHECK_LT(xy, board_size * board_size);
134 int8_t x = xy / board_size;
135 int8_t y = xy % board_size;
136 int destination_index = i % num_actions_destinations;
137 SPIEL_CHECK_GE(destination_index, 0);
138 SPIEL_CHECK_LT(destination_index, num_actions_destinations);
139 return std::make_pair(Square{x, y}, destination_index);
140}

143 int board_size, int num_actions_destinations) {
144 return (from_square.x * board_size + from_square.y) *
145 num_actions_destinations +
146 destination_index;
