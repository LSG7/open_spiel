98 class ElFarol(TensorGame):
99  """N-Player, 2-Action symmetric game with unique symmetric Nash."""
100 
101  def __init__(self, n=2, c=0.5, B=0, S=1, G=2, seed=None):
102  """Ctor. Initializes payoff tensor (N x (2,) * N np.array).
103 
104  See Section 3.1, The El Farol Stage Game in
105  http://www.econ.ed.ac.uk/papers/id186_esedps.pdf
106 
107  action 0: go to bar
108  action 1: avoid bar
109 
110  Args:
111  n: int, number of players
112  c: float, threshold for `crowded' as a fraction of number of players
113  B: float, payoff for going to a crowded bar
114  S: float, payoff for staying at home
115  G: float, payoff for going to an uncrowded bar
116  seed: seed for random number generator, used if computing best responses
117  """
118  assert G > S > B, "Game parameters must satisfy G > S > B."
119  pt = np.zeros((n,) + (2,) * n)
120  for idx in np.ndindex(pt.shape):
121  p = idx[0]
122  a = idx[1:]
123  a_i = a[p]
124  go_to_bar = (a_i < 1)
125  crowded = (n - 1 - sum(a) + a_i) >= (c * n)
126  if go_to_bar and not crowded:
127  pt[idx] = G
128  elif go_to_bar and crowded:
129  pt[idx] = B
130  else:
131  pt[idx] = S
132  super().__init__(pt, seed)
