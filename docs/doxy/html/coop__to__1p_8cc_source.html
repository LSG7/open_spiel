95// String representation of the current possible hands for every player and the
96// assignment of hands to actions for the current player.
97std::string CoopTo1pState::Assignments() const {
98 std::string str = "";
99 Player current_player = state_->CurrentPlayer();
100 for (int player = 0; player < privates_.size(); ++player) {
101 auto possible_assignments = state_->LegalActions(player);
102 possible_assignments.push_back(PlayerPrivate::kUnassigned);
103 for (auto asignment : possible_assignments) {
104 absl::StrAppend(&str, "Player ", player);
105 if (player == current_player) {
106 absl::StrAppend(&str, " ", AssignmentToString(player, asignment), ":");
107 } else {
108 absl::StrAppend(&str, " possible:");
109 }
110 bool found = false;
111 for (int pvt = 0; pvt < privates_[player].assignments.size(); ++pvt) {
112 if (privates_[player].assignments[pvt] == asignment) {
113 absl::StrAppend(&str, " ", privates_[player].names[pvt]);
114 found = true;
115 }
116 }
117 if (!found) absl::StrAppend(&str, " none");
118 absl::StrAppend(&str, "\n");
119 }
120 }
121 return str;
122}

149void CoopTo1pState::ObservationTensor(Player player,
150 absl::Span<float> values) const {
151 SPIEL_CHECK_GE(player, 0);
152 SPIEL_CHECK_LT(player, num_players_);
153 const int num_actions = state_->NumDistinctActions();
154 const int num_players = state_->NumPlayers();
155 SPIEL_CHECK_EQ(values.size(),
156 num_privates_ * (num_players + num_actions + 1) + num_actions);
157 std::fill(values.begin(), values.end(), 0);
158 if (IsChanceNode()) return;
159
160 // Last action in the underlying game
161 int base = 0;
162 if (prev_action_ != kInvalidAction) values.at(prev_action_) = 1;
163 base += num_actions;
164
165 // Possible privates for every player (multi-hot)
166 for (int p = 0; p < num_players; ++p) {
167 const auto& pvt = privates_[p];
168 for (int i = 0; i < num_privates_; ++i) {
169 values.at(base + i) = (pvt.assignments[i] != PlayerPrivate::kImpossible);
170 }
171 base += num_privates_;
172 }
173
174 // For terminal states, we don't need anything else.
175 if (state_->IsTerminal()) return;
176
177 // Currently-assigned privates for every action (multi-hot)
178 Player current_player = state_->CurrentPlayer();
179 const auto& pvt = privates_[current_player];
180 for (Action a = 0; a < num_actions; ++a) {
181 for (int i = 0; i < num_privates_; ++i) {
182 values.at(base + i) = (pvt.assignments[i] == a);
183 }
184 base += num_privates_;
185 }
186
187 // The private we are currently considering (one-hot)
188 if (!pvt.AssignmentsComplete()) values.at(base + pvt.next_unassigned) = 1;
189 base += num_privates_;
190}

192void CoopTo1pState::DoApplyAction(Action action_id) {
193 if (IsChanceNode()) {
194 // Assume this is the dealing of a private state. Capture info on possible
195 // privates here.
196 privates_.push_back(PlayerPrivate(num_privates_));
197 actual_private_.push_back(action_id);
198 for (int i = 0; i < num_privates_; ++i) {
199 privates_.back().names[i] = state_->ActionToString(kChancePlayerId, i);
200 }
201 state_->ApplyAction(action_id);
202 } else {
203 // Update the assignment and maybe act in the underlying game.
204 Player player = state_->CurrentPlayer();
205 privates_[player].Assign(action_id);
206 if (privates_[player].AssignmentsComplete()) {
207 Action underlying_action =
208 privates_[player].assignments[actual_private_[player]];
209 state_->ApplyAction(underlying_action);
210 prev_player_ = player;
211 prev_action_ = underlying_action;
212 privates_[player].Reset(underlying_action);
213 }
214 }
215}

217std::vector<int> CoopTo1pGame::ObservationTensorShape() const {
218 // State of the underlying game (represented as the last action)
219 // Possible privates for every player (multi-hot)
220 // Currently-assigned privates for every action (multi-hot)
221 // The private we are currently considering (one-hot)
222 const int num_actions = game_->NumDistinctActions();
223 const int num_players = game_->NumPlayers();
224 return {NumPrivates() * (num_players + num_actions + 1) + num_actions};
225}
