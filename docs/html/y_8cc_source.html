57 // The board is represented as a flattened 2d array of the form:
58 // 1 2 3
59 // A 0 1 2 0 1 2 0 1 2
60 // B 3 4 5 <=> 3 4 <=> 3 4
61 // C 6 7 8 6 6
62 //
63 // Neighbors are laid out in this pattern:
64 // 0 1 0 1
65 // 5 X 2 <=> 5 X 2
66 // 4 3 4 3
67 
68 // Direct neighbors of a cell, clockwise.
69 constexpr std::array<Move, kMaxNeighbors> neighbor_offsets = {
70  Move(0, -1, kMoveOffset), Move(1, -1, kMoveOffset),
71  Move(1, 0, kMoveOffset), Move(0, 1, kMoveOffset),
72  Move(-1, 1, kMoveOffset), Move(-1, 0, kMoveOffset),
73 };
74 
75 // Precomputed list of neighbors per board_size: [board_size][cell][direction]
76 std::vector<NeighborList> neighbor_list;
77 
78 NeighborList gen_neighbors(int board_size) {
79  NeighborList out;
80  out.resize(board_size * board_size);
81  for (int y = 0; y < board_size; y++) {
82  for (int x = 0; x < board_size; x++) {
83  int xy = x + y * board_size; // Don't use Move.xy so it works off-board.
84  for (int dir = 0; dir < neighbor_offsets.size(); dir++) {
85  Move offset = neighbor_offsets[dir];
86  out[xy][dir] = Move(x + offset.x, y + offset.y, board_size);
87  }
88  }
89  }
90  return out;
91 }
92 
93 const NeighborList& get_neighbors(int board_size) {
94  if (board_size >= neighbor_list.size()) {
95  neighbor_list.resize(board_size + 1);
96  }
97  if (neighbor_list[board_size].empty()) {
98  neighbor_list[board_size] = gen_neighbors(board_size);
99  }
100  return neighbor_list[board_size];
101 }

105 int Move::Edge(int board_size) const {
106  if (!OnBoard()) return 0;
107 
108  return (x == 0 ? (1 << 0) : 0) | (y == 0 ? (1 << 1) : 0) |
109  (x + y == board_size - 1 ? (1 << 2) : 0);
110 }

153 std::string YState::ToString() const {
154  // Generates something like:
155  // a b c d e f g h i j k
156  // 1 O @ O O . @ @ O O @ O
157  // 2 . O O . O @ @ . O O
158  // 3 . O @ @ O @ O O @
159  // 4 O O . @ . @ O O
160  // 5 . . . @[@]@ O
161  // 6 @ @ @ O O @
162  // 7 @ . O @ O
163  // 8 . @ @ O
164  // 9 @ @ .
165  // 10 O .
166  // 11 @
167 
168  std::string white = "O";
169  std::string black = "@";
170  std::string empty = ".";
171  std::string coord = "";
172  std::string reset = "";
173  if (ansi_color_output_) {
174  std::string esc = "\033";
175  reset = esc + "[0m";
176  coord = esc + "[1;37m"; // bright white
177  empty = reset + ".";
178  white = esc + "[1;33m" + "@"; // bright yellow
179  black = esc + "[1;34m" + "@"; // bright blue
180  }
181 
182  std::ostringstream out;
183 
184  // Top x coords.
185  out << ' ';
186  for (int x = 0; x < board_size_; x++) {
187  out << ' ' << coord << static_cast<char>('a' + x);
188  }
189  out << '\n';
190 
191  for (int y = 0; y < board_size_; y++) {
192  out << std::string(y + ((y + 1) < 10), ' '); // Leading space.
193  out << coord << (y + 1); // Leading y coord.
194 
195  bool found_last = false;
196  for (int x = 0; x < board_size_ - y; x++) {
197  Move pos(x, y, board_size_);
198 
199  // Spacing and last-move highlight.
200  if (found_last) {
201  out << coord << ']';
202  found_last = false;
203  } else if (last_move_ == pos) {
204  out << coord << '[';
205  found_last = true;
206  } else {
207  out << ' ';
208  }
209 
210  // Actual piece.
211  Player p = board_[pos.xy].player;
212  if (p == kPlayerNone) out << empty;
213  if (p == kPlayer1) out << white;
214  if (p == kPlayer2) out << black;
215  }
216  if (found_last) {
217  out << coord << ']';
218  }
219  out << '\n';
220  }
221  out << reset;
222  return out.str();
223 }

269 void YState::DoApplyAction(Action action) {
270  SPIEL_CHECK_EQ(board_[action].player, kPlayerNone);
271  SPIEL_CHECK_EQ(outcome_, kPlayerNone);
272 
273  Move move = ActionToMove(action);
274  SPIEL_CHECK_TRUE(move.OnBoard());
275 
276  last_move_ = move;
277  board_[move.xy].player = current_player_;
278  moves_made_++;
279 
280  for (const Move& m : neighbors[move.xy]) {
281  if (m.OnBoard() && current_player_ == board_[m.xy].player) {
282  JoinGroups(move.xy, m.xy);
283  }
284  }
285 
286  if (board_[FindGroupLeader(move.xy)].edge == 0x7) { // ie all 3 edges.
287  outcome_ = current_player_;
288  }
289 
290  current_player_ = (current_player_ == kPlayer1 ? kPlayer2 : kPlayer1);
291 }

293 int YState::FindGroupLeader(int cell) {
294  int parent = board_[cell].parent;
295  if (parent != cell) {
296  do { // Follow the parent chain up to the group leader.
297  parent = board_[parent].parent;
298  } while (parent != board_[parent].parent);
299  // Do path compression, but only the current one to avoid recursion.
300  board_[cell].parent = parent;
301  }
302  return parent;
303 }

305 bool YState::JoinGroups(int cell_a, int cell_b) {
306  int leader_a = FindGroupLeader(cell_a);
307  int leader_b = FindGroupLeader(cell_b);
308 
309  if (leader_a == leader_b) // Already the same group.
310  return true;
311 
312  if (board_[leader_a].size < board_[leader_b].size) {
313  // Force group a's subtree to be bigger.
314  std::swap(leader_a, leader_b);
315  }
316 
317  // Group b joins group a.
318  board_[leader_b].parent = leader_a;
319  board_[leader_a].size += board_[leader_b].size;
320  board_[leader_a].edge |= board_[leader_b].edge;
321 
322  return false;
323 }
