73 void RepeatedState::DoApplyActions(const std::vector<Action>& actions) {
74  SPIEL_CHECK_EQ(actions.size(), num_players_);
75  // Faster to clone the reference stage_game_state_ than call
76  // game_->NewInitialState().
77  std::unique_ptr<State> stage_game_state = stage_game_state_->Clone();
78  stage_game_state->ApplyActions(actions);
79  SPIEL_CHECK_TRUE(stage_game_state->IsTerminal());
80  actions_history_.push_back(actions);
81  rewards_history_.push_back(stage_game_state->Returns());
82 }

84 std::string RepeatedState::ToString() const {
85  std::string rv;
86  for (int i = 0; i < actions_history_.size(); ++i) {
87  absl::StrAppend(&rv, "Round ", i, ":\n");
88  absl::StrAppend(&rv, "Actions: ");
89  for (int j = 0; j < num_players_; ++j) {
90  absl::StrAppend(
91  &rv, stage_game_state_->ActionToString(j, actions_history_[i][j]),
92  " ");
93  }
94  absl::StrAppend(&rv, "\n");
95  absl::StrAppend(&rv, "Rewards: ");
96  for (auto player_reward : rewards_history_[i])
97  absl::StrAppend(&rv, player_reward, " ");
98  absl::StrAppend(&rv, "\n");
99  }
100  absl::StrAppend(&rv, "Total Returns: ");
101  for (auto player_return : Returns()) absl::StrAppend(&rv, player_return, " ");
102  return rv;
103 }

138 std::string RepeatedState::ObservationString(Player /*player*/) const {
139  std::string rv;
140  if (actions_history_.empty()) { return rv; }
141 
142  // Starting from the back of the history, show each player's moves:
143  for (int j = 0;
144  j < recall_ && static_cast<int>(actions_history_.size()) - 1 - j >= 0;
145  ++j) {
146  int hist_idx = actions_history_.size() - 1 - j;
147  SPIEL_CHECK_GE(hist_idx, 0);
148  SPIEL_CHECK_LT(hist_idx, actions_history_.size());
149  for (int i = 0; i < num_players_; ++i) {
150  absl::StrAppend(&rv,
151  stage_game_state_->ActionToString(i, actions_history_[hist_idx][i]),
152  " ");
153  }
154  }
155  return rv;
156 }

158 void RepeatedState::InformationStateTensor(Player player,
159  absl::Span<float> values) const {
160  SPIEL_CHECK_GE(player, 0);
161  SPIEL_CHECK_LT(player, num_players_);
162 
163  SPIEL_CHECK_EQ(values.size(), game_->InformationStateTensorSize());
164  std::fill(values.begin(), values.end(), 0.0);
165  if (actions_history_.empty()) return;
166 
167  auto ptr = values.begin();
168  for (int j = 0; j < actions_history_.size(); ++j) {
169  for (int i = 0; i < num_players_; ++i) {
170  ptr[actions_history_[j][i]] = 1;
171  ptr += stage_game_state_->LegalActions(i).size();
172  }
173  }
174 }

257 std::shared_ptr<const Game> CreateRepeatedGame(const Game& stage_game,
258  const GameParameters& params) {
259  // The stage game must be a deterministic normal-form (one-shot) game.
260  SPIEL_CHECK_EQ(stage_game.MaxGameLength(), 1);
261  SPIEL_CHECK_EQ(stage_game.GetType().dynamics,
262  GameType::Dynamics::kSimultaneous);
263  SPIEL_CHECK_EQ(stage_game.GetType().chance_mode,
264  GameType::ChanceMode::kDeterministic);
265  return std::make_shared<const RepeatedGame>(stage_game.shared_from_this(),
266  params);
267 }
