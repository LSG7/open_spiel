29 @parameterized.parameters(["kuhn_poker", "leduc_poker"])
30 def test_cpp_and_python_implementations_are_identical(self, game_name):
31 game = pyspiel.load_game(game_name)
32
33 policy = openspiel_policy.UniformRandomPolicy(game)
34
35 all_states = get_all_states.get_all_states(
36 game,
37 depth_limit=-1,
38 include_terminals=False,
39 include_chance_states=False,
40 to_string=lambda s: s.information_state_string())
41
42 for current_player in range(game.num_players()):
43 noise = noisy_policy.NoisyPolicy(
44 policy, player_id=current_player, alpha=0.5, beta=10.)
45 for state in all_states.values():
46 if state.current_player() < 0:
47 continue
48
49 if state.current_player() != current_player:
50 self.assertEqual(
51 policy.action_probabilities(state),
52 noise.action_probabilities(state))
