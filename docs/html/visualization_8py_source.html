56 def _eval_dynamics_2x2_grid(dynamics, num_points):
57  """Evaluates dynamics on a 2-D mesh-grid.
58 
59  Args:
60  dynamics: Population dynamics of type `dynamics.MultiPopulationDynamics`.
61  num_points: Number of points along each dimension of the grid.
62 
63  Returns:
64  Mesh-grid (x, y) and corresponding derivatives of the first action for
65  player 1 and 2 (u, v).
66  """
67  assert dynamics.payoff_tensor.shape == (2, 2, 2)
68 
69  x = np.linspace(0., 1., num_points + 2)[1:-1]
70  x, y = np.meshgrid(x, x)
71  u = np.empty(x.shape)
72  v = np.empty(x.shape)
73 
74  for i in range(num_points):
75  for j in range(num_points):
76  row_state = np.array([x[i, j], 1. - x[i, j]])
77  col_state = np.array([y[i, j], 1. - y[i, j]])
78  state = np.concatenate((row_state, col_state))
79  dstate = dynamics(state)
80  u[i][j] = dstate[0]
81  v[i][j] = dstate[2]
82  return x, y, u, v

85 def _rk12_step(func, y0, dt):
86  """Improved Euler-Integration step to integrate dynamics.
87 
88  Args:
89  func: Function handle to time derivative.
90  y0: Current state.
91  dt: Integration step.
92 
93  Returns:
94  Next state.
95  """
96  dy = func(y0)
97  y_ = y0 + dt * dy
98  return y0 + dt / 2. * (dy + func(y_))

191 class SimplexTransform(transforms.Transform):
192  """Affine transform to project the 2-simplex to 2D Cartesian space."""
193  input_dims = 3
194  output_dims = 2

196  _MATRIX = np.array([[0., 0.], [1., 0.], [0.5, np.sqrt(3) / 2.]])

202 class SimplexStreamMask(object):
203  """Mask of regular discrete cells to track trajectories/streamlines.
204 
205  Also see `matplotlib.streamplot.StreamMask`.
206  """
207 
208  def __init__(self, density=1.):
209  self._n_n = int(30. * density)
210  self._mask_mask = np.zeros([self._n_n + 1] * 2 + [2], dtype=bool)
211  self.shapeshape = self._mask_mask.shape
212 
213  def index(self, point):
214  """Computes index given a point on the simplex."""
215  point = np.array(point)
216  idx = np.floor(point[:2] * self._n_n).astype(int)
217  x, y = point[:2] * self._n_n - idx
218  z = int(x + y > 1)
219  return tuple(idx.tolist() + [z])
220 
221  def point(self, index):
222  """Computes point on the simplex given an index."""
223  p = np.empty((3,))
224  p[0] = (index[0] + (1 + index[2]) / 3.) / float(self._n_n)
225  p[1] = (index[1] + (1 + index[2]) / 3.) / float(self._n_n)
226  p[2] = 1. - p[0] - p[1]
227  return p if p[2] > 0. else None
228 
229  def __getitem__(self, point):
230  return self._mask_mask.__getitem__(self.indexindex(point))
231 
232  def __setitem__(self, point, val):
233  return self._mask_mask.__setitem__(self.indexindex(point), val)

461  def _integrate(self, x, func, mask, dt, min_dist=0.01):
462  cells = []
463  trajectory = [x]
464  x_ = x
465  for dt in [dt, -dt]:
466  while not mask[x]:
467  cell = mask.index(x)
468  cells.append(cell)
469  while mask.index(x) == cell:
470  # integrate up to cell boundary
471  if np.sqrt(np.sum((x_ - x)**2)) > min_dist:
472  x_ = x
473  if dt > 0:
474  trajectory.append(x)
475  else:
476  trajectory.insert(0, x)
477 
478  x = _rk12_step(func, x, dt=dt)
479 
480  if dt > 0:
481  mask[trajectory[-1]] = True
482  else:
483  mask[trajectory[0]] = True
484 
485  # restore to integrate backwards
486  if dt > 0. and len(cells):
487  trajectory.append(_rk12_step(func, x, dt=-dt))
488  mask[mask.point(cells[0])] = False
489  x = trajectory[0]
490  x_ = x
491  else:
492  trajectory.insert(0, _rk12_step(func, x, dt=-dt))
493  return (np.array(trajectory), cells) if len(trajectory) > 2 else None

495  def streamplot(self,
496  dynamics,
497  initial_points=None,
498  dt=0.01,
499  density=1.,
500  min_length=0.4,
501  linewidth=None,
502  color="k",
503  **kwargs):
504  """Visualizes the dynamics as a streamline plot.
505 
506  Mimics the visuals of `Axes.streamplot` for simplex plots.
507 
508  Args:
509  dynamics: Population dynamics of type `dynamics.SinglePopulationDynamics`.
510  initial_points: Starting points for streamlines
511  dt: Integration step.
512  density: Controls the density of streamlines in the plot.
513  min_length: Streamlines with length < min_length will be discarded.
514  linewidth: In `{None, float, "velocity"}`, optional, default: None. If
515  `linewidth="velocity"`, line width is scaled by the velocity of the
516  dynamics. Defaults to `rcParams` if `linewidth=None`.
517  color: In `{None, string, (r,g,b), (r,g,b,a), "velocity"}`, default: None.
518  If `color="velocity"`, velocity of dynamics is used to color the
519  streamlines. Defaults to `rcParams` if `color=None`.
520  **kwargs: Additional keyword arguments passed on to `Axes.streamplot`.
521 
522  Returns:
523  The `SimplexStreamMask`.
524  """
525  mask = SimplexStreamMask(density=density)
526  trajectories = []
527 
528  if initial_points is None:
529  eps = 0.1
530  initial_points = np.array([[1. - eps, eps / 2., eps / 2.],
531  [eps / 2., 1. - eps, eps / 2.],
532  [eps / 2., eps / 2., 1. - eps]])
533  initial_points = np.vstack(
534  (initial_points, utils.sample_from_simplex(100)))
535  # TODO(author10): add heuristic for initial points
536 
537  else:
538  initial_points = np.array(initial_points)
539  assert initial_points.ndim == 2
540  assert initial_points.shape[1] == 3
541 
542  # generate trajectories
543  for p in initial_points:
544  # center initial point on grid cell
545  p = mask.point(mask.index(p))
546  res = self._integrate_integrate(p, dynamics, mask, dt=dt)
547  if res is not None:
548  t, cells = res # pylint: disable=unpacking-non-sequence
549  cum_len = np.cumsum(
550  np.sqrt(
551  np.diff(t[:, 0])**2 + np.diff(t[:, 1])**2 +
552  np.diff(t[:, 2])**2))
553  if cum_len[-1] < min_length:
554  for cell in cells:
555  mask[mask.point(cell)] = False
556  continue
557  trajectories.append(t)
558 
559  lc_color = arrow_color = color
560  lc_linewidth = linewidth
561 
562  if linewidth == "velocity" or color == "velocity":
563  vel_max = 0
564  vel_min = float("inf")
565  velocities = []
566  for t in trajectories:
567  dx = np.apply_along_axis(dynamics, 1, t)
568  vel = np.sqrt(np.sum(dx**2, axis=1))
569  vel_max = max(np.max(vel), vel_max)
570  vel_min = min(np.min(vel), vel_min)
571  velocities.append(vel)
572 
573  # add trajectories to plot
574  for i, t in enumerate(trajectories):
575  cum_len = np.cumsum(
576  np.sqrt(
577  np.diff(t[:, 0])**2 + np.diff(t[:, 1])**2 + np.diff(t[:, 2])**2))
578  mid_idx = np.searchsorted(cum_len, cum_len[-1] / 2.)
579 
580  if linewidth == "velocity" or color == "velocity":
581  vel = (velocities[i] - vel_min) / vel_max
582 
583  if linewidth == "velocity":
584  lc_linewidth = 3. * vel + 0.5
585 
586  if color == "velocity":
587  cmap = matplotlib.cm.get_cmap(rcParams["image.cmap"])
588  lc_color = cmap(vel)
589  arrow_color = cmap(vel[mid_idx])
590 
591  lc = self._linecollection_linecollection(t, linewidth=lc_linewidth, color=lc_color)
592  self.add_collection(lc)
593 
594  # add arrow centered on trajectory
595  arrow_tail = self._simplex_transform_simplex_transform.transform(t[mid_idx - 1])
596  arrow_head = self._simplex_transform_simplex_transform.transform(t[mid_idx])
597  arrow_kw = dict(arrowstyle="-|>", mutation_scale=10 * 1.)
598  arrow_patch = FancyArrowPatch(
599  arrow_tail,
600  arrow_head,
601  linewidth=None,
602  color=arrow_color,
603  zorder=3,
604  **arrow_kw)
605  self.add_patch(arrow_patch)
606  return mask
