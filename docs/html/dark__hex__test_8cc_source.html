76 void GameBlackWinsMaximumCollisions() {
77  // White follows exact moves black is playing, black does the same
78  // B . . R . . B W . B R . B W B
79  // . . . . . . . . . . . . . . . ...
80  // . . . . . . . . . . . . . . .
81  std::shared_ptr<const Game> game = LoadGame(
82  "dark_hex",
83  {{"num_cols", GameParameter(3)}, {"num_rows", GameParameter(3)}});
84  std::unique_ptr<State> state = game->NewInitialState();
85  std::array play_seq = {0, 1, 4, 2, 7, 5, 8, 6}; // 3 is the terminal move
86  for (int i = 0; i < play_seq.size(); ++i) {
87  state->ApplyAction(play_seq[i]); // player moves
88  state->ApplyAction(play_seq[i]); // other player tries to move
89  }
90  state->ApplyAction(3);
91  // Black wins
92  SPIEL_CHECK_TRUE(state->IsTerminal());
93  SPIEL_CHECK_EQ(state->PlayerReturn(0), 1.0);
94  SPIEL_CHECK_EQ(state->PlayerReturn(1), -1.0);
95 }
