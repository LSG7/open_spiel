55struct Move {
56 int x, y, xy; // xy = x + y * kBoardSize
57 int r; // rotation
58 int dir; // which direction to rotate
59 int quadrant; // which quadrant to rotate
60
61 constexpr Move(int x_, int y_, int r_)
62 : x(x_),
63 y(y_),
64 xy(x_ + y_ * kBoardSize),
65 r(r_),
66 dir(r_ & 1),
67 quadrant(r_ >> 1) {}
68 constexpr Move(Action a)
69 : Move((a / kPossibleRotations) % kBoardSize,
70 (a / (kPossibleRotations * kBoardSize)) % kBoardSize,
71 a % kPossibleRotations) {}
72
73 Action ToAction() const {
74 return ((y * kBoardSize) + x) * kPossibleRotations + r;
75 }
76
77 std::string ToString() const {
78 return absl::StrCat(std::string(1, static_cast<char>('a' + x)),
79 std::string(1, static_cast<char>('1' + y)),
80 std::string(1, static_cast<char>('s' + r)));
81 }
82};

84// Order the bits such that quadrant rotations are easy.
85constexpr int xy_to_bit[kBoardPositions] = {
86 0, 1, 2, 15, 16, 9, // Comment
87 7, 8, 3, 14, 17, 10, // to force
88 6, 5, 4, 13, 12, 11, // clang-format
89 29, 30, 31, 22, 23, 24, // to keep the
90 28, 35, 32, 21, 26, 25, // square spatial
91 27, 34, 33, 20, 19, 18, // alignment.
92};
93
94// The bit mask for reading from an xy location.
95constexpr uint64_t xym(int xy) { return 1ull << xy_to_bit[xy]; }
96constexpr uint64_t xym(int x, int y) { return xym(x + y * kBoardSize); }
97constexpr uint64_t xy_bit_mask[kBoardPositions] = {
98 xym(0, 0), xym(1, 0), xym(2, 0), xym(3, 0), xym(4, 0), xym(5, 0),
99 xym(0, 1), xym(1, 1), xym(2, 1), xym(3, 1), xym(4, 1), xym(5, 1),
100 xym(0, 2), xym(1, 2), xym(2, 2), xym(3, 2), xym(4, 2), xym(5, 2),
101 xym(0, 3), xym(1, 3), xym(2, 3), xym(3, 3), xym(4, 3), xym(5, 3),
102 xym(0, 4), xym(1, 4), xym(2, 4), xym(3, 4), xym(4, 4), xym(5, 4),
103 xym(0, 5), xym(1, 5), xym(2, 5), xym(3, 5), xym(4, 5), xym(5, 5),
104};
105
106// Helpers for creating the win mask.
107constexpr uint64_t pattern(int x, int y, int ox, int oy) {
108 return (xym(x + ox * 0, y + oy * 0) | // Comment
109 xym(x + ox * 1, y + oy * 1) | // to force
110 xym(x + ox * 2, y + oy * 2) | // clang-format
111 xym(x + ox * 3, y + oy * 3) | // to keep
112 xym(x + ox * 4, y + oy * 4)); // aligntment.
113}
114constexpr uint64_t horizontal(int x, int y) { return pattern(x, y, 1, 0); }
115constexpr uint64_t vertical(int x, int y) { return pattern(x, y, 0, 1); }
116constexpr uint64_t tl_br(int x, int y) { return pattern(x, y, 1, 1); }
117constexpr uint64_t bl_tr(int x, int y) { return pattern(x, y, 1, -1); }
118
119// The mask of 5 bits for each of the win conditions.
120constexpr uint64_t win_mask[kPossibleWinConditions] = {
121 horizontal(0, 0), horizontal(1, 0), // Row 0
122 horizontal(0, 1), horizontal(1, 1), // Row 1
123 horizontal(0, 2), horizontal(1, 2), // Row 2
124 horizontal(0, 3), horizontal(1, 3), // Row 3
125 horizontal(0, 4), horizontal(1, 4), // Row 4
126 horizontal(0, 5), horizontal(1, 5), // Row 5
127 vertical(0, 0), vertical(0, 1), // Column 0
128 vertical(1, 0), vertical(1, 1), // Column 1
129 vertical(2, 0), vertical(2, 1), // Column 2
130 vertical(3, 0), vertical(3, 1), // Column 3
131 vertical(4, 0), vertical(4, 1), // Column 4
132 vertical(5, 0), vertical(5, 1), // Column 5
133 tl_br(0, 0), tl_br(1, 1), // Center diagonals from top-left to bottom-right
134 tl_br(0, 1), tl_br(1, 0), // Offset diagonals
135 bl_tr(0, 5), bl_tr(1, 4), // Center diagonals from bottom-left to top-right
136 bl_tr(0, 4), bl_tr(1, 5), // Offset diagonals
137};
138
139// Rotate a quadrant clockwise or counter-clockwise.
140// Pulls a 8-bit segment and rotates it by 2 bits.
141uint64_t rotate_quadrant_cw(uint64_t b, int quadrant) {
142 uint64_t m = 0xFFull << (quadrant * 9);
143 return (b & ~m) | (((b & m) >> 6) & m) | (((b & m) << 2) & m);
144}
145uint64_t rotate_quadrant_ccw(uint64_t b, int quadrant) {
146 uint64_t m = 0xFFull << (quadrant * 9);
147 return (b & ~m) | (((b & m) >> 2) & m) | (((b & m) << 6) & m);
148}

159std::vector<Action> PentagoState::LegalActions() const {
160 // Can move in any empty cell, and do all rotations.
161 std::vector<Action> moves;
162 if (IsTerminal()) return moves;
163 moves.reserve((kBoardPositions - moves_made_) * kPossibleRotations);
164 for (int y = 0; y < kBoardSize; y++) {
165 for (int x = 0; x < kBoardSize; x++) {
166 if (get(x, y) == kPlayerNone) {
167 for (int r = 0; r < kPossibleRotations; r++) {
168 moves.push_back(Move(x, y, r).ToAction());
169 }
170 }
171 }
172 }
173 return moves;
174}

181std::string PentagoState::ToString() const {
182 // Generates something like:
183 // > t u <
184 // a b c d e f
185 // v 1 . . O @ . O v
186 // s 2 . . O . . @ v
187 // 3 . @ @ . @ O
188 // 4 . @ @ . O .
189 // z 5 @ . O @ O . w
190 // ^ 6 @ O @ O O O ^
191 // > y x <
192
193 std::string white = "O";
194 std::string black = "@";
195 std::string empty = ".";
196 std::string coord = "";
197 std::string reset = "";
198 if (ansi_color_output_) {
199 std::string esc = "\033";
200 reset = esc + "[0m";
201 coord = esc + "[1;37m"; // bright white
202 empty = reset + ".";
203 white = esc + "[1;33m" + "@"; // bright yellow
204 black = esc + "[1;34m" + "@"; // bright blue
205 }
206
207 // Enable the arrows if/when open_spiel allows unicode in strings.
208 // constexpr char const* arrows[] = {"", "", "", "", "", "", "", ""};
209 constexpr char const* arrows[] = {"v", ">", "<", "v", "^", "<", ">", "^"};
210 constexpr char const* left[] = {arrows[0], "s", " ", " ", "z", arrows[7]};
211 constexpr char const* right[] = {arrows[3], "v", " ", " ", "w", arrows[4]};
212
213 std::ostringstream out;
214 out << coord;
215 out << " " << arrows[1] << " t u " << arrows[2] << "\n";
216 out << " a b c d e f\n";
217 for (int y = 0; y < kBoardSize; y++) {
218 out << left[y] << " " << (y + 1) << " ";
219 for (int x = 0; x < kBoardSize; x++) {
220 Player p = get(x, y);
221 if (p == kPlayerNone) out << empty;
222 if (p == kPlayer1) out << white;
223 if (p == kPlayer2) out << black;
224 out << " ";
225 }
226 out << coord << right[y] << "\n";
227 }
228 out << " " << arrows[6] << " y x " << arrows[5] << reset << "\n";
229 return out.str();
230}

281void PentagoState::DoApplyAction(Action action) {
282 SPIEL_CHECK_EQ(outcome_, kPlayerNone);
283
284 Move move(action);
285 SPIEL_CHECK_EQ(get(move.xy), kPlayerNone);
286
287 // Apply the move.
288 board_[current_player_] |= xy_bit_mask[move.xy];
289 if (move.dir == 0) {
290 board_[0] = rotate_quadrant_ccw(board_[0], move.quadrant);
291 board_[1] = rotate_quadrant_ccw(board_[1], move.quadrant);
292 } else {
293 board_[0] = rotate_quadrant_cw(board_[0], move.quadrant);
294 board_[1] = rotate_quadrant_cw(board_[1], move.quadrant);
295 }
296 moves_made_++;
297
298 // Check the win conditions.
299 bool p1_won = false;
300 bool p2_won = false;
301 for (int i = 0; i < kPossibleWinConditions; i++) {
302 uint64_t wm = win_mask[i];
303 if ((board_[0] & wm) == wm) p1_won = true;
304 if ((board_[1] & wm) == wm) p2_won = true;
305 }
306
307 // Note that you can rotate such that you cause your opponent to win.
308 if (p1_won && p2_won) {
309 outcome_ = kPlayerDraw;
310 } else if (p1_won) {
311 outcome_ = kPlayer1;
312 } else if (p2_won) {
313 outcome_ = kPlayer2;
314 } else if (moves_made_ == kBoardPositions) {
315 outcome_ = kPlayerDraw;
316 }
317
318 current_player_ = (current_player_ == kPlayer1 ? kPlayer2 : kPlayer1);
319}
