94 GameParameter DeserializeGameParameter(const std::string& data,
95  const std::string& delimiter) {
96  std::vector<std::string> parts = absl::StrSplit(data, delimiter);
97  SPIEL_CHECK_EQ(parts.size(), 3);
98  bool mandatory = (parts[2] == "True" || parts[2] == "true");
99  if (parts[0] == "kUnset") {
100  return GameParameter(GameParameter::Type::kUnset, mandatory);
101  } else if (parts[0] == "kInt") {
102  int value;
103  SPIEL_CHECK_TRUE(absl::SimpleAtoi(parts[1], &value));
104  return GameParameter(value, mandatory);
105  } else if (parts[0] == "kDouble") {
106  double value;
107  SPIEL_CHECK_TRUE(absl::SimpleAtod(parts[1], &value));
108  return GameParameter(value, mandatory);
109  } else if (parts[0] == "kString") {
110  return GameParameter(
111  absl::StrReplaceAll(parts[1], {{kSerializedNewline, "\n"}}), mandatory);
112  } else if (parts[0] == "kBool") {
113  return GameParameter(parts[1] == "True" || parts[1] == "true", mandatory);
114  } else if (parts[0] == "kGame") {
115  return GameParameter(DeserializeGameParameters(parts[1]), mandatory);
116  } else {
117  SpielFatalError(absl::StrCat("Unrecognized type: ", parts[0]));
118  }
119 }

154 std::string GameParametersToString(const GameParameters& game_params) {
155  std::string str;
156  if (game_params.empty()) return "";
157  if (game_params.count("name")) str = game_params.at("name").string_value();
158  str.push_back('(');
159  bool first = true;
160  for (const auto& key_val : game_params) {
161  if (key_val.first != "name") {
162  if (!first) str.push_back(',');
163  str.append(key_val.first);
164  str.append("=");
165  str.append(key_val.second.ToString());
166  first = false;
167  }
168  }
169  str.push_back(')');
170  return str;
171 }

173 GameParameter GameParameterFromString(const std::string& str) {
174  if (str == "True" || str == "true") {
175  return GameParameter(true);
176  } else if (str == "False" || str == "false") {
177  return GameParameter(false);
178  } else if (str.find_first_not_of("+-0123456789") == std::string::npos) {
179  int value;
180  bool success = absl::SimpleAtoi(str, &value);
181  SPIEL_CHECK_TRUE(success);
182  return GameParameter(value);
183  } else if (str.find_first_not_of("+-0123456789.") == std::string::npos) {
184  double value;
185  bool success = absl::SimpleAtod(str, &value);
186  SPIEL_CHECK_TRUE(success);
187  return GameParameter(value);
188  } else if (str.back() == ')') {
189  return GameParameter(GameParametersFromString(str));
190  } else {
191  return GameParameter(str);
192  }
193 }

195 GameParameters GameParametersFromString(const std::string& game_string) {
196  GameParameters params;
197  if (game_string.empty()) return params;
198  int first_paren = game_string.find('(');
199  if (first_paren == std::string::npos) {
200  params["name"] = GameParameter(game_string);
201  return params;
202  }
203  params["name"] = GameParameter(game_string.substr(0, first_paren));
204  int start = first_paren + 1;
205  int parens = 1;
206  int equals = -1;
207  for (int i = start; i < game_string.length(); ++i) {
208  if (game_string[i] == '(') {
209  ++parens;
210  } else if (game_string[i] == ')') {
211  --parens;
212  } else if (game_string[i] == '=' && parens == 1) {
213  equals = i;
214  }
215  if ((game_string[i] == ',' && parens == 1) ||
216  (game_string[i] == ')' && parens == 0 && i > start + 1)) {
217  params[game_string.substr(start, equals - start)] =
218  GameParameterFromString(
219  game_string.substr(equals + 1, i - equals - 1));
220  start = i + 1;
221  equals = -1;
222  }
223  }
224  if (parens > 0) SpielFatalError("Missing closing bracket ')'.");
225  return params;
226 }
