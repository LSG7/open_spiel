105  ]
106  state = game.new_initial_state()
107  for i in range(pyspiel.ROSHAMBO_NUM_THROWS):
108  joint_action = [-1] * num_players
109  for p in range(num_players):
110  joint_action[p] = bots[p].step(state)
111  state.apply_actions(joint_action)
112  if i == 0:
113  # copybot wins the first round
114  self.assertListEqual(state.returns(), [-1, 1])
115  else:
116  # the rest are a draw
117  self.assertListEqual(state.rewards(), [0, 0])
118  self.assertTrue(state.is_terminal())
119  self.assertListEqual(state.returns(), [-1, 1])
