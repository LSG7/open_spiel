35  def test_kuhn_poker_always_pass_p0(self):
36  game = pyspiel.load_game("kuhn_poker")
37  calc = action_value.TreeWalkCalculator(game)
38  uniform_policy = policy.TabularPolicy(game)
39  always_pass_policy = policy.FirstActionPolicy(game).to_tabular()
40  returned_values = calc([always_pass_policy, uniform_policy],
41  always_pass_policy)
42  root_node_values = calc.get_root_node_values(
43  [always_pass_policy, uniform_policy])
44  self.assertTrue(
45  np.allclose(root_node_values, returned_values.root_node_values))
46 
47  # Action 0 == Pass. Action 1 == Bet

49  # value of that node is undefined.
