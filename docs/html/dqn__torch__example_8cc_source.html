33 void DQNCatch(int seed, int total_episodes, int report_every,
34  int num_eval_episodes) {
35  std::cout << "Running DQN on catch" << std::endl;
36  std::mt19937 rng(seed);
37  std::shared_ptr<const Game> game = open_spiel::LoadGame("catch");
38 
39  int dqn_agent_seed = absl::Uniform<int>(rng, 0, 1000000);
40 
41  // Values copied from: python/examples/single_agent_catch.py
42  DQNSettings settings = {
43  /*seed*/ dqn_agent_seed,
44  /*use_observation*/ game->GetType().provides_observation_tensor,
45  /*player_id*/ 0,
46  /*state_representation_size*/ game->ObservationTensorSize(),
47  /*num_actions*/ game->NumDistinctActions(),
48  /*hidden_layers_sizes*/ {32, 32},
49  /*replay_buffer_capacity*/ 10000,
50  /*batch_size*/ 128,
51  /*learning_rate*/ 0.1,
52  /*update_target_network_every*/ 250,
53  /*learn_every*/ 10,
54  /*discount_factor*/ 0.99,
55  /*min_buffer_size_to_learn*/ 1000,
56  /*epsilon_start*/ 1.0,
57  /*epsilon_end*/ 0.1,
58  /*epsilon_decay_duration*/ 2000};
59  auto dqn = std::make_unique<DQN>(settings);
60  std::vector<Agent*> agents = {dqn.get()};
61 
62  for (int num_episodes = 0; num_episodes < total_episodes;
63  num_episodes += report_every) {
64  // Training
65  RunEpisodes(&rng, *game, agents,
66  /*num_episodes*/ report_every, /*is_evaluation*/ false);
67 
68  std::vector<double> avg_returns =
69  RunEpisodes(&rng, *game, agents,
70  /*num_episodes*/ num_eval_episodes, /*is_evaluation*/ true);
71 
72  std::cout << num_episodes + report_every << " " << avg_returns[0]
73  << std::endl;
74  }
75 }

77 void SelfPlayTicTacToe(int seed, int total_episodes, int report_every,
78  int num_eval_episodes) {
79  std::cout << "Running self-play Tic-Tac-Toe" << std::endl;
80  std::mt19937 rng(seed);
81  std::shared_ptr<const Game> game = open_spiel::LoadGame("tic_tac_toe");
82 
83  std::vector<std::unique_ptr<DQN>> dqn_agents;
84  std::vector<std::unique_ptr<RandomAgent>> random_agents;
85  std::vector<Agent*> agents(game->NumPlayers(), nullptr);
86 
87  for (Player p = 0; p < game->NumPlayers(); ++p) {
88  int dqn_agent_seed = absl::Uniform<int>(rng, 0, 1000000);
89  DQNSettings settings = {
90  /*seed*/ dqn_agent_seed,
91  /*use_observation*/ game->GetType().provides_observation_tensor,
92  /*player_id*/ p,
93  /*state_representation_size*/ game->ObservationTensorSize(),
94  /*num_actions*/ game->NumDistinctActions(),
95  /*hidden_layers_sizes*/ {32, 32},
96  /*replay_buffer_capacity*/ 100000,
97  /*batch_size*/ 128,
98  /*learning_rate*/ 0.01,
99  /*update_target_network_every*/ 250,
100  /*learn_every*/ 10,
101  /*discount_factor*/ 0.99,
102  /*min_buffer_size_to_learn*/ 1000,
103  /*epsilon_start*/ 1.0,
104  /*epsilon_end*/ 0.1,
105  /*epsilon_decay_duration*/ 50000};
106  dqn_agents.push_back(std::make_unique<DQN>(settings));
107  int rand_agent_seed = absl::Uniform<int>(rng, 0, 1000000);
108  random_agents.push_back(std::make_unique<RandomAgent>(p, rand_agent_seed));
109  }
110 
111  for (int num_episodes = 0; num_episodes < total_episodes;
112  num_episodes += report_every) {
113  for (Player p = 0; p < game->NumPlayers(); ++p) {
114  agents[p] = dqn_agents[p].get();
115  }
116 
117  // Training
118  RunEpisodes(&rng, *game, agents,
119  /*num_episodes*/ report_every, /*is_evaluation*/ false);
120 
121  // Self-play eval.
122  std::vector<double> avg_self_play_returns =
123  RunEpisodes(&rng, *game, agents,
124  /*num_episodes*/ num_eval_episodes, /*is_evaluation*/ true);
125 
126  std::vector<double> avg_returns_vs_random(game->NumPlayers(), 0);
127  // Eval vs. random.
128  for (Player p = 0; p < game->NumPlayers(); ++p) {
129  for (Player pp = 0; pp < game->NumPlayers(); ++pp) {
130  if (pp == p) {
131  agents[pp] = dqn_agents[pp].get();
132  } else {
133  agents[pp] = random_agents[pp].get();
134  }
135  }
136  std::vector<double> avg_returns = RunEpisodes(
137  &rng, *game, agents,
138  /*num_episodes*/ num_eval_episodes, /*is_evaluation*/ true);
139  avg_returns_vs_random[p] = avg_returns[p];
140  }
141 
142  std::cout << num_episodes + report_every << " self-play returns: ";
143  for (Player p = 0; p < game->NumPlayers(); ++p) {
144  std::cout << avg_self_play_returns[p] << " ";
145  }
146  std::cout << "returns vs random: ";
147  for (Player p = 0; p < game->NumPlayers(); ++p) {
148  std::cout << avg_returns_vs_random[p] << " ";
149  }
150  std::cout << std::endl;
151  }
152 }
