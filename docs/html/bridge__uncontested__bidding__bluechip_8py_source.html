45 def _run_once(state, bots):
46  """Plays bots with each other, returns terminal utility for each player."""
47  for bot in bots:
48  bot.restart_at(state)
49  while not state.is_terminal():
50  if state.is_chance_node():
51  outcomes, probs = zip(*state.chance_outcomes())
52  state.apply_action(np.random.choice(outcomes, p=probs))
53  else:
54  state.apply_action(bots[state.current_player()].step(state)[1])
55  return state
