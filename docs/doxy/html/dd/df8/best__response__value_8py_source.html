15"""Does a backward pass to output a value of a best response policy."""
16from typing import Optional

19from open_spiel.python.mfg import value
20import pyspiel

23class BestResponse(value.ValueFunction):
24 """Computes a best response value."""
25
26 def __init__(self,
27 game,
28 distribution: distribution_std.Distribution,
29 state_value: Optional[value.ValueFunction] = None,
30 root_state=None):
31 """Initializes the best response calculation.
32
33 Args:
34 game: The game to analyze.
35 distribution: A `distribution_std.Distribution` object.
36 state_value: A state value function. Default to TabularValueFunction.
37 root_state: The state of the game at which to start. If `None`, the game
38 root state is used.
39 """
40 super().__init__(game)
41 if root_state is None:
42 self._root_states = game.new_initial_states()
43 else:
44 self._root_states = [root_state]
45 self._distribution = distribution
46 self._state_value = (state_value if state_value
47 else value.TabularValueFunction(game))
48
49 self.evaluate()
50
51 def eval_state(self, state):
52 """Evaluate the value of a state.
53
54 Args:
55 state: a game state.
56
57 Returns:
58 the optimal value of the state
59
60 Recursively computes the value of the optimal policy given the fixed state
61 distribution. `self._state_value` is used as a cache for pre-computed
62 values.
63 """
64 state_str = state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID)
65 if self._state_value.has(state_str):
66 return self._state_value(state_str)
67 if state.is_terminal():
68 self._state_value.set_value(
69 state_str,
70 state.rewards()[state.mean_field_population()])
71 return self._state_value(state_str)
72 if state.current_player() == pyspiel.PlayerId.CHANCE:
73 self._state_value.set_value(state_str, 0.0)
74 for action, prob in state.chance_outcomes():
75 new_state = state.child(action)
76 self._state_value.add_value(state_str,
77 prob * self.eval_state(new_state))
78 return self._state_value(state_str)
79 if state.current_player() == pyspiel.PlayerId.MEAN_FIELD:
80 dist = [
81 # We need to default to 0, because
82 # `state.distribution_support()` might contain states that
83 # we did not reach yet. These states should be given a
84 # probability of 0.
85 self._distribution.value_str(str_state, 0.)
86 for str_state in state.distribution_support()
87 ]
88 new_state = state.clone()
89 new_state.update_distribution(dist)
90 self._state_value.set_value(
91 state_str,
92 state.rewards()[state.mean_field_population()] +
93 self.eval_state(new_state))
94 return self._state_value(state_str)
95 else:
96 assert int(state.current_player()) >= 0, "The player id should be >= 0"
97 max_q = max(
98 self.eval_state(state.child(action))
99 for action in state.legal_actions())
100 self._state_value.set_value(
101 state_str,
102 state.rewards()[state.mean_field_population()] + max_q)
103 return self._state_value(state_str)
104
105 def evaluate(self):
106 """Evaluate the best response value on all states."""
107 for state in self._root_states:
108 self.eval_state(state)
109
110 def value(self, state, action=None):
111 if action is None:
112 return self._state_value(
113 state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID))
114 new_state = state.child(action)
115 return state.rewards()[state.mean_field_population()] + self._state_value(
116 new_state.observation_string(pyspiel.PlayerId.DEFAULT_PLAYER_ID))
