28 def test_symmetrize_tensor(self, trials=100, seed=1234):
29 random = np.random.RandomState(seed)
30
31 successes = []
32 for _ in range(trials):
33 pt = random.rand(3, 2, 2, 2)
34
35 pt_sym_man = np.zeros_like(pt)
36 for p in range(3):
37 for i in range(2):
38 for j in range(2):
39 for k in range(2):
40 if p == 0:
41 # read: if player 0 plays i and its two opponents play j and k
42 # this should return the same payoff as when
43 # player 1 plays i and its two opponents play j and k
44 # player 2 plays i and its two opponents play j and k
45 # solution is to add up all these payoffs and replace with avg
46 pt_sym_man[p, i, j, k] = (pt[0, i, j, k] + pt[0, i, k, j] +
47 pt[1, j, i, k] + pt[1, k, i, j] +
48 pt[2, j, k, i] + pt[2, k, j, i]) / 6.
49 elif p == 1:
50 # same rationale, but with player 1 playing j
51 pt_sym_man[p, i, j, k] = (pt[0, j, i, k] + pt[0, j, k, i] +
52 pt[1, i, j, k] + pt[1, k, j, i] +
53 pt[2, i, k, j] + pt[2, k, i, j]) / 6.
54 else:
55 # same rationale, but with player 2 playing k
56 pt_sym_man[p, i, j, k] = (pt[0, k, i, j] + pt[0, k, j, i] +
57 pt[1, i, k, j] + pt[1, j, k, i] +
58 pt[2, i, j, k] + pt[2, j, i, k]) / 6.
59 pt_sym = utils.sym(pt)
60
61 successes += [np.allclose(pt_sym, pt_sym_man)]
62
63 perc = 100 * np.mean(successes)
64 logging.info('symmetrizing success rate out of %d is %f', trials, perc)
65 self.assertGreaterEqual(
66 perc, 100., 'symmetrizing failed')
