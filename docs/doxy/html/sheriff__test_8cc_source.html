48struct GameSize {
49 uint32_t num_sequences[2] = {0, 0}; // Layout: [Pl.0, Pl.1].
50 uint32_t num_infostates[2] = {0, 0}; // Layout: [Pl.0, Pl.1].
51 uint32_t num_terminal_states = 0;
52};
53
54GameSize ComputeGameSize(const std::shared_ptr<const Game> game) {
55 std::map<std::string, std::unique_ptr<open_spiel::State>> all_states =
56 open_spiel::algorithms::GetAllStates(
57 *game, /* depth_limit = */ std::numeric_limits<int>::max(),
58 /* include_terminals = */ true,
59 /* include_chance_states = */ false);
60
61 GameSize size;
62
63 // Account for empty sequence.
64 size.num_sequences[Player{0}] = 1;
65 size.num_sequences[Player{1}] = 1;
66
67 absl::flat_hash_set<std::string> infosets;
68 for (const auto& [_, state] : all_states) {
69 if (state->IsTerminal()) {
70 ++size.num_terminal_states;
71 } else {
72 const Player player = state->CurrentPlayer();
73 SPIEL_CHECK_TRUE(player == Player{0} || player == Player{1});
74
75 // NOTE: there is no requirement that infostates strings be unique across
76 // players. So, we disambiguate the player by prepending it.
77 const std::string infostate_string =
78 absl::StrCat(player, state->InformationStateString());
79
80 if (infosets.insert(infostate_string).second) {
81 // The infostate string was not present in the hash set. We update the
82 // tally of infosets and sequences for the player.
83 size.num_infostates[player] += 1;
84 size.num_sequences[player] += state->LegalActions().size();
85 }
86 }
87 }
88
89 return size;
90}

92void TestGameSizes() {
93 // We expect these game sizes:
94 //
95 // +-------+-------+--------+-----------------+----------------+----------+
96 // | Max | Max | Num | Num sequences | Num infosets | Terminal |
97 // | bribe | items | rounds | pl 0 | pl 1 | pl 0 | pl 1 | states |
98 // +-------+-------+--------+--------+--------+-------+--------+----------+
99 // | 3 | 3 | 1 | 21 | 9 | 5 | 4 | 32 |
100 // | 3 | 5 | 2 | 223 | 73 | 55 | 36 | 384 |
101 // | 3 | 3 | 3 | 1173 | 585 | 293 | 292 | 2048 |
102 // | 3 | 5 | 4 | 14047 | 4681 | 3511 | 2340 | 24576 |
103 // +-------+-------+--------+--------+--------+-------+--------+----------+
104 // | 5 | 3 | 1 | 29 | 13 | 5 | 6 | 48 |
105 // | 5 | 3 | 2 | 317 | 157 | 53 | 78 | 576 |
106 // | 5 | 5 | 3 | 5659 | 1885 | 943 | 942 | 10368 |
107 // +-------+-------+--------+--------+--------+-------+--------+----------+
108
109 // To simplify the construction of game instance we introduce a lambda.
110 const auto ConstructInstance =
111 [](const uint32_t& max_bribe, const uint32_t max_items,
112 const uint32_t num_rounds) -> std::shared_ptr<const Game> {
113 return LoadGame(
114 "sheriff",
115 {{"max_bribe", GameParameter(static_cast<int>(max_bribe))},
116 {"max_items", GameParameter(static_cast<int>(max_items))},
117 {"num_rounds", GameParameter(static_cast<int>(num_rounds))}});
118 };
119
120 GameSize size = ComputeGameSize(ConstructInstance(3, 3, 1));
121 SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 21);
122 SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 9);
123 SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 5);
124 SPIEL_CHECK_EQ(size.num_infostates[Player{1}], 4);
125 SPIEL_CHECK_EQ(size.num_terminal_states, 32);
126
127 size = ComputeGameSize(ConstructInstance(3, 5, 2));
128 SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 223);
129 SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 73);
130 SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 55);
131 SPIEL_CHECK_EQ(size.num_infostates[Player{1}], 36);
132 SPIEL_CHECK_EQ(size.num_terminal_states, 384);
133
134 size = ComputeGameSize(ConstructInstance(3, 3, 3));
135 SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 1173);
136 SPIEL_CHECK_EQ(size.num_sequences[Player{1}], 585);
137 SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 293);
138 SPIEL_CHECK_EQ(size.num_infostates[Player{1}], 292);
139 SPIEL_CHECK_EQ(size.num_terminal_states, 2048);
140
141 size = ComputeGameSize(ConstructInstance(3, 5, 4));
142 SPIEL_CHECK_EQ(size.num_sequences[Player{0}], 14047);

144 SPIEL_CHECK_EQ(size.num_infostates[Player{0}], 3511);
