67 bool BotChannel::ReadLineAsync() {
68  int chars_read = 0;
69  bool line_read = false;
70  response_.clear();
71 
72  do {
73  // Read a single character (non-blocking).
74  char c;
75  chars_read = read(out(), &c, 1);
76  if (chars_read == 1) {
77  if (c == '\n') {
78  response_ = buf_;
79  buf_ = "";
80  line_read = true;
81  } else {
82  buf_.append(1, c);
83  }
84  }
85  } while (chars_read > 0 && !line_read && buf_.size() < kMaxLineLength);
86 
87  if (buf_.size() >= kMaxLineLength) {
88  comm_error_ = EMSGSIZE;
89  }
90 
91  return line_read;
92 }

105 // Read a response message from the bot in a separate thread.
106 void ReadLineFromChannelStdout(BotChannel* c) {
107  SPIEL_CHECK_TRUE(c);
108  // Outer loop for repeated match playing.
109  while (!c->shutdown_) {
110  // Wait until referee sends a message to the bot.
111  while (c->wait_for_referee_) {
112  sleep_ms(1);
113  if (c->shutdown_) return;
114  }
115 
116  {
117  std::lock_guard<std::mutex> lock(c->mx_read);
118 
119  auto time_start = std::chrono::system_clock::now();
120  while ( // Keep reading the current line,
121  !c->ReadLineAsync()
122  // if there is no error,
123  && c->comm_error() == 0
124  // no timeout,
125  && !(c->time_out_ = (time_elapsed(time_start) > c->time_limit_))
126  // and no reading cancellation.
127  && !c->cancel_read_) {
128  sleep_ms(1);
129  if (c->shutdown_) return;
130  }
131 
132  c->wait_for_referee_ = true;
133  }
134  }
135 }

140 // Read a stderr output from the bot in a separate thread.
141 // Forward all bot's stderr to the referee's stderr.
142 // Makes sure that lines are not tangled together by using a mutex.
143 void ReadLineFromChannelStderr(BotChannel* c) {
144  SPIEL_CHECK_TRUE(c);
145  int read_bytes;
146  std::array<char, 1024> buf;
147  while (!c->shutdown_) {
148  read_bytes = read(c->err(), &buf[0], 1024);
149  if (read_bytes > 0) {
150  std::lock_guard<std::mutex> lock(mx_cerr); // Have nice stderr outputs.
151  std::cerr << "Bot#" << c->bot_index_ << ": ";
152  for (int i = 0; i < read_bytes; ++i) std::cerr << buf[i];
153  std::cerr << std::flush;
154  }
155  sleep_ms(1);
156  }
