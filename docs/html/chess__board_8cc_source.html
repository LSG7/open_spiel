165 bool IsLongDiagonal(const chess::Square &from_sq, const chess::Square &to_sq,
166  int board_size) {
167  if (from_sq == to_sq) {
168  return false;
169  }
170  int half_board_size = board_size / 2;
171  if ((to_sq.y < half_board_size && to_sq.x < half_board_size) ||
172  (to_sq.y >= half_board_size && to_sq.x >= half_board_size)) {
173  return from_sq.y - to_sq.y == from_sq.x - to_sq.x;
174  } else {
175  return from_sq.y - to_sq.y == to_sq.x - from_sq.x;
176  }
177 }

200 std::string Move::ToSAN(const ChessBoard &board) const {
201  std::string move_text;
202  PieceType piece_type = board.at(from).type;
203  if (is_castling) {
204  if (from.x < to.x) {
205  move_text = "O-O";
206  } else {
207  move_text = "O-O-O";
208  }
209  } else {
210  switch (piece_type) {
211  case PieceType::kKing:
212  case PieceType::kQueen:
213  case PieceType::kRook:
214  case PieceType::kBishop:
215  case PieceType::kKnight:
216  move_text += PieceTypeToString(piece_type);
217  break;
218  case PieceType::kPawn:
219  // No piece type required.
220  break;
221  case PieceType::kEmpty:
222  std::cerr << "Move doesn't have a piece type" << std::endl;
223  }
224 
225  // Now we generate all moves from this position, and see if our file and
226  // rank are unique.
227  bool file_unique = true;
228  bool rank_unique = true;
229  bool disambiguation_required = false;
230 
231  board.GenerateLegalMoves([&](const Move &move) -> bool {
232  if (move.piece.type != piece.type) {
233  return true; // Continue generating moves.
234  }
235  if (move.to != to) {
236  return true;
237  }
238  if (move.from == from) {
239  // This is either us, or a promotion to a different type. We don't count
240  // them as ambiguous in either case.
241  return true;
242  }
243  disambiguation_required = true;
244  if (move.from.x == from.x) {
245  file_unique = false;
246  } else if (move.from.y == from.y) {
247  rank_unique = false;
248  }
249  return true;
250  });
251 
252  bool file_required = false;
253  bool rank_required = false;
254 
255  if (piece_type == PieceType::kPawn && from.x != to.x) {
256  // Pawn captures always need file, and they will never require rank dis-
257  // ambiguation.
258  file_required = true;
259  } else if (disambiguation_required) {
260  if (file_unique) {
261  // This includes when both will disambiguate, in which case we have to
262  // use file. [FIDE Laws of Chess (2018): C.10.3].
263  file_required = true;
264  } else if (rank_unique) {
265  rank_required = true;
266  } else {
267  // We have neither unique file nor unique rank. This is only possible
268  // with 3 or more pieces of the same type.
269  file_required = true;
270  rank_required = true;
271  }
272  }
273 
274  if (file_required) {
275  absl::StrAppend(&move_text, FileToString(from.x));
276  }
277 
278  if (rank_required) {
279  absl::StrAppend(&move_text, RankToString(from.y));
280  }
281 
282  // We have a capture if either 1) the destination square has a piece, or
283  // 2) we are making a diagonal pawn move (which can also be an en-passant
284  // capture, where the destination square would not have a piece).
285  auto piece_at_to_square = board.at(to);
286  if ((piece_at_to_square.type != PieceType::kEmpty) ||
287  (piece_type == PieceType::kPawn && from.x != to.x)) {
288  absl::StrAppend(&move_text, "x");
289  }
290 
291  // Destination square is always fully encoded.
292  absl::StrAppend(&move_text, SquareToString(to));
293 
294  // Encode the promotion type if we have a promotion.
295  switch (promotion_type) {
296  case PieceType::kEmpty:
297  break;
298  case PieceType::kQueen:
299  case PieceType::kRook:
300  case PieceType::kBishop:
301  case PieceType::kKnight:
302  absl::StrAppend(&move_text, "=", PieceTypeToString(promotion_type));
303  break;
304  case PieceType::kKing:
305  case PieceType::kPawn:
306  std::cerr << "Cannot promote to " << PieceTypeToString(promotion_type)
307  << "! Only Q, R, B, N are allowed" << std::endl;
308  break;
309  }
310  }
311 
312  // Figure out if this is a check / checkmating move or not.
313  if (!board.KingInCheckAllowed()) {
314  auto board_copy = board;
315  board_copy.ApplyMove(*this);
316  if (board_copy.InCheck()) {
317  bool has_escape = false;
318  board_copy.GenerateLegalMoves([&](const Move &) -> bool {
319  has_escape = true;
320  return false; // No need to keep generating moves.
321  });
322 
323  if (has_escape) {
324  // Check.
325  absl::StrAppend(&move_text, "+");
326  } else {
327  // Checkmate.
328  absl::StrAppend(&move_text, "#");
329  }
330  }
331  }
332 
333  return move_text;
334 }

350 /*static*/ absl::optional<ChessBoard> ChessBoard::BoardFromFEN(
351  const std::string &fen, int board_size,
352  bool king_in_check_allowed, bool allow_pass_move) {
353  /* An FEN string includes a board position, side to play, castling
354  * rights, ep square, 50 moves clock, and full move number. In that order.
355  *
356  * Eg. start position is:
357  * rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1
358  *
359  * Board is described from rank 8 to rank 1, and files from a to h. Empty
360  * squares are encoded as number of consecutive empty squares.
361  *
362  * Many FEN strings don't have the last two fields.
363  */
364  ChessBoard board(board_size, king_in_check_allowed, allow_pass_move);
365 
366  for (auto color : {Color::kBlack, Color::kWhite}) {
367  for (auto dir : {CastlingDirection::kLeft, CastlingDirection::kRight}) {
368  board.SetCastlingRight(color, dir, false);
369  }
370  }
371 
372  std::vector<std::string> fen_parts = absl::StrSplit(fen, ' ');
373 
374  if (fen_parts.size() != 6 && fen_parts.size() != 4) {
375  std::cerr << "Invalid FEN: " << fen << std::endl;
376  return absl::nullopt;
377  }
378 
379  std::string &piece_configuration = fen_parts[0];
380  std::string &side_to_move = fen_parts[1];
381  std::string &castling_rights = fen_parts[2];
382  std::string &ep_square = fen_parts[3];
383 
384  // These are defaults if the FEN string doesn't have these fields.
385  std::string fifty_clock = "0";
386  std::string move_number = "1";
387 
388  if (fen_parts.size() == 6) {
389  fifty_clock = fen_parts[4];
390  move_number = fen_parts[5];
391  }
392 
393  std::vector<std::string> piece_config_by_rank =
394  absl::StrSplit(piece_configuration, '/');
395 
396  for (int8_t current_y = board_size - 1; current_y >= 0; --current_y) {
397  std::string &rank = piece_config_by_rank[board_size - current_y - 1];
398  int8_t current_x = 0;
399  for (char c : rank) {
400  if (current_x >= board_size) {
401  std::cerr << "Too many things on FEN rank: " << rank << std::endl;
402  return absl::nullopt;
403  }
404 
405  if (c >= '1' && c <= '8') {
406  current_x += c - '0';
407  } else {
408  auto piece_type = PieceTypeFromChar(c);
409  if (!piece_type) {
410  std::cerr << "Invalid piece type in FEN: " << c << std::endl;
411  return absl::nullopt;
412  }
413 
414  Color color = isupper(c) ? Color::kWhite : Color::kBlack;
415  board.set_square(Square{current_x, current_y},
416  Piece{color, *piece_type});
417 
418  ++current_x;
419  }
420  }
421  }
422 
423  if (side_to_move == "b") {
424  board.SetToPlay(Color::kBlack);
425  } else if (side_to_move == "w") {
426  board.SetToPlay(Color::kWhite);
427  } else {
428  std::cerr << "Invalid side to move in FEN: " << side_to_move << std::endl;
429  return absl::nullopt;
430  }
431 
432  if (castling_rights.find('K') != std::string::npos) { // NOLINT
433  board.SetCastlingRight(Color::kWhite, CastlingDirection::kRight, true);
434  }
435 
436  if (castling_rights.find('Q') != std::string::npos) { // NOLINT
437  board.SetCastlingRight(Color::kWhite, CastlingDirection::kLeft, true);
438  }
439 
440  if (castling_rights.find('k') != std::string::npos) { // NOLINT
441  board.SetCastlingRight(Color::kBlack, CastlingDirection::kRight, true);
442  }
443 
444  if (castling_rights.find('q') != std::string::npos) { // NOLINT
445  board.SetCastlingRight(Color::kBlack, CastlingDirection::kLeft, true);
446  }
447 
448  if (ep_square != "-") {
449  auto maybe_ep_square = SquareFromString(ep_square);
450  if (!maybe_ep_square) {
451  std::cerr << "Invalid en passant square in FEN: " << ep_square
452  << std::endl;
453  return absl::nullopt;
454  }
455  board.SetEpSquare(*maybe_ep_square);
456  }
457 
458  board.SetIrreversibleMoveCounter(std::stoi(fifty_clock));
459  board.SetMovenumber(std::stoi(move_number));
460 
461  return board;
462 }

477 void ChessBoard::GenerateLegalMoves(const MoveYieldFn &yield,
478  Color color) const {
479  // We do not need to filter moves that would result for King to move / stay
480  // in check, so we can yield all pseudo legal moves
481  if (king_in_check_allowed_) {
482  GeneratePseudoLegalMoves(yield, color);
483  } else {
484  auto king_square = find(Piece{color, PieceType::kKing});
485 
486  GeneratePseudoLegalMoves(
487  [this, &king_square, &yield, color](const Move &move) {
488  // See if the move is legal by applying, checking whether the king is
489  // under attack, and undoing the move.
490  // TODO: Optimize this.
491  auto board_copy = *this;
492  board_copy.ApplyMove(move);
493 
494  auto ks =
495  at(move.from).type == PieceType::kKing ? move.to : king_square;
496 
497  if (board_copy.UnderAttack(ks, color)) {
498  return true;
499  } else {
500  return yield(move);
501  }
502  },
503  color);
504  }

507 void ChessBoard::GeneratePseudoLegalMoves(
508  const MoveYieldFn &yield, Color color,
509  PseudoLegalMoveSettings settings) const {
510  bool generating = true;
511 
512 #define YIELD(move) \
513  if (!yield(move)) { \
514  generating = false; \
515  }
516 
517  if (allow_pass_move_) YIELD(kPassMove);
518 
519  for (int8_t y = 0; y < board_size_ && generating; ++y) {
520  for (int8_t x = 0; x < board_size_ && generating; ++x) {
521  Square sq{x, y};
522  auto &piece = at(sq);
523  if (piece.type != PieceType::kEmpty && piece.color == color) {
524  switch (piece.type) {
525  case PieceType::kKing:
526  GenerateKingDestinations_(
527  sq, color,
528  [&yield, &piece, &sq, &generating](const Square &to) {
529  YIELD(Move(sq, to, piece));
530  });
531  GenerateCastlingDestinations_(
532  sq, color, settings,
533  [&yield, &piece, &sq, &generating](const Square &to) {
534  YIELD(Move(sq, to, piece, PieceType::kEmpty, true));
535  });
536  break;
537  case PieceType::kQueen:
538  GenerateQueenDestinations_(
539  sq, color, settings,
540  [&yield, &sq, &piece, &generating](const Square &to) {
541  YIELD(Move(sq, to, piece));
542  });
543  break;
544  case PieceType::kRook:
545  GenerateRookDestinations_(
546  sq, color, settings,
547  [&yield, &sq, &piece, &generating](const Square &to) {
548  YIELD(Move(sq, to, piece));
549  });
550  break;
551  case PieceType::kBishop:
552  GenerateBishopDestinations_(
553  sq, color, settings,
554  [&yield, &sq, &piece, &generating](const Square &to) {
555  YIELD(Move(sq, to, piece));
556  });
557  break;
558  case PieceType::kKnight:
559  GenerateKnightDestinations_(
560  sq, color,
561  [&yield, &sq, &piece, &generating](const Square &to) {
562  YIELD(Move(sq, to, piece));
563  });
564  break;
565  case PieceType::kPawn:
566  GeneratePawnDestinations_(
567  sq, color, settings,
568  [&yield, &sq, &piece, &generating, this](const Square &to) {
569  if (IsPawnPromotionRank(to)) {
570  YIELD(Move(sq, to, piece, PieceType::kQueen));
571  YIELD(Move(sq, to, piece, PieceType::kRook));
572  YIELD(Move(sq, to, piece, PieceType::kBishop));
573  YIELD(Move(sq, to, piece, PieceType::kKnight));
574  } else {
575  YIELD(Move(sq, to, piece));
576  }
577  });
578  GeneratePawnCaptureDestinations_(
579  sq, color, settings, true, /* include enpassant */
580  [&yield, &sq, &piece, &generating, this](const Square &to) {
581  if (IsPawnPromotionRank(to)) {
582  YIELD(Move(sq, to, piece, PieceType::kQueen));
583  YIELD(Move(sq, to, piece, PieceType::kRook));
584  YIELD(Move(sq, to, piece, PieceType::kBishop));
585  YIELD(Move(sq, to, piece, PieceType::kKnight));
586  } else {
587  YIELD(Move(sq, to, piece));
588  }
589  });
590  break;
591  default:
592  std::cerr << "Unknown piece type: " << static_cast<int>(piece.type)
593  << std::endl;
594  }
595  }
596  }
597  }
598 
599 #undef YIELD
600 }

666 bool ChessBoard::IsBreachingMove(Move tested_move) const {
667  if (tested_move == kPassMove) return false;
668 
669  const Piece& piece = tested_move.piece;
670  if (piece.type == PieceType::kEmpty) return false;
671  if (piece.type == PieceType::kKnight) return false;
672  if (piece.type == PieceType::kPawn) return false;
673  // King never makes breaching moves: a castling that would be breaching
674  // is considered an illegal move.
675  if (piece.type == PieceType::kKing) return false;
676 
677  SPIEL_DCHECK_TRUE(piece.type == PieceType::kQueen ||
678  piece.type == PieceType::kRook ||
679  piece.type == PieceType::kBishop);
680 
681  // The move is not breaching, if it is generated with
682  // PseudoLegalMoveSettings::kAcknowledgeEnemyPieces
683 
684  bool is_breaching = true;
685  const auto check_breaching = [&](const Square &to) {
686  if (to == tested_move.to) is_breaching = false;
687  };
688 
689  // Queen moves are a combination of rook and bishop moves.
690  if (piece.type == PieceType::kRook || piece.type == PieceType::kQueen) {
691  GenerateRookDestinations_(tested_move.from, piece.color,
692  kAcknowledgeEnemyPieces, check_breaching);
693  }
694  if (piece.type == PieceType::kBishop || piece.type == PieceType::kQueen) {
695  GenerateBishopDestinations_(tested_move.from, piece.color,
696  kAcknowledgeEnemyPieces, check_breaching);
697  }
698 
699  return is_breaching;
700 }

702 void ChessBoard::BreachingMoveToCaptureMove(Move* move) const {
703  SPIEL_CHECK_TRUE(move);
704  SPIEL_DCHECK_TRUE(IsBreachingMove(*move));
705  int dx = move->to.x - move->from.x;
706  int dy = move->to.y - move->from.y;
707  SPIEL_DCHECK_TRUE(dx == 0 || dy == 0 || std::abs(dx) == std::abs(dy));
708 
709  // Cap values to [-1, 1] range to make a proper step size.
710  dx = std::max(-1, dx);
711  dx = std::min(1, dx);
712  dy = std::max(-1, dy);
713  dy = std::min(1, dy);
714  const Offset step{static_cast<int8_t>(dx),
715  static_cast<int8_t>(dy)};
716 
717  Square sq;
718  for (sq = move->from + step; sq != move->to; sq += step) {
719  if (at(sq).type != PieceType::kEmpty) break;
720  }
721  move->to = sq;
722 }

724 bool ChessBoard::HasSufficientMaterial() const {
725  // Try to detect these 4 conditions.
726  // 1. K vs K
727  // 2. K+B vs K
728  // 3. K+N vs K
729  // 4. K+B* vs K+B* (all bishops on same coloured squares)
730 
731  // If king is allowed to move to/stay in check, any material is sufficient
732  // material. If there is no material, then there is also no opponent king and
733  // that means the game had already ended.
734  if (king_in_check_allowed_) {
735  return true;
736  }
737 
738  // Indexed by colour.
739  int knights[2] = {0, 0};
740  int dark_bishops[2] = {0, 0};
741  int light_bishops[2] = {0, 0};
742 
743  for (int8_t y = 0; y < board_size_; ++y) {
744  for (int8_t x = 0; x < board_size_; ++x) {
745  const auto &piece = at(Square{x, y});
746  // If we have a queen, rook, or pawn, we have sufficient material.
747  // This is early exit for almost all positions. We check rooks first
748  // because they tend to appear on the corners of boards.
749  if (piece.color != Color::kEmpty) {
750  if (piece.type == PieceType::kRook || piece.type == PieceType::kPawn ||
751  piece.type == PieceType::kQueen) {
752  return true;
753  }
754 
755  // We don't care about kings.
756  if (piece.type == PieceType::kKing) {
757  continue;
758  }
759 
760  if (piece.type == PieceType::kKnight) {
761  ++knights[static_cast<size_t>(piece.color)];
762  }
763 
764  if (piece.type == PieceType::kBishop) {
765  bool is_dark = ((x + y) % 2 == 0);
766  if (is_dark) {
767  ++dark_bishops[static_cast<size_t>(piece.color)];
768  } else {
769  ++light_bishops[static_cast<size_t>(piece.color)];
770  }
771  }
772  }
773  }
774  }
775 
776  // Having two knights allows helpmate.
777  if (knights[0] > 1 || knights[1] > 1) {
778  return true;
779  }
780 
781  if (knights[0] == 1) {
782  // If we have anything else, mate is possible.
783  if (light_bishops[0] > 0 || dark_bishops[0] > 0) {
784  return true;
785  } else {
786  // If one side only has a knight, the other side must have something (#3).
787  return knights[1] > 0 || dark_bishops[1] > 0 || light_bishops[1] > 0;
788  }
789  }
790 
791  if (knights[1] == 1) {
792  // If we have anything else, mate is possible.
793  if (light_bishops[1] > 0 || dark_bishops[1] > 0) {
794  return true;
795  } else {
796  // If one side only has a knight, the other side must have something (#3).
797  return knights[0] > 0 || dark_bishops[0] > 0 || light_bishops[0] > 0;
798  }
799  }
800 
801  // Now we only have bishops and kings. We must have two bishops on opposite
802  // coloured squares (from either side) to not be a draw.
803  // This covers #1, #2, and #4.
804  bool dark_bishop_exists = (dark_bishops[0] + dark_bishops[1]) > 0;
805  bool light_bishop_exists = (light_bishops[0] + light_bishops[1]) > 0;
806  return dark_bishop_exists && light_bishop_exists;
807 }

828 absl::optional<Move> ChessBoard::ParseSANMove(
829  const std::string &move_str) const {
830  std::string move = move_str;
831 
832  if (move.empty()) return absl::nullopt;
833 
834  if (absl::StartsWith(move, "O-O-O")) {
835  // Queenside / left castling.
836  std::vector<Move> candidates;
837  GenerateLegalMoves([&candidates](const Move &move) {
838  if (move.is_castling && move.to.x == 2) {
839  candidates.push_back(move);
840  }
841  return true;
842  });
843  if (candidates.size() == 1) return candidates[0];
844  std::cerr << "Invalid O-O-O" << std::endl;
845  return absl::nullopt;
846  }
847 
848  if (absl::StartsWith(move, "O-O")) {
849  // Kingside / right castling.
850  std::vector<Move> candidates;
851  GenerateLegalMoves([&candidates](const Move &move) {
852  if (move.is_castling && move.to.x == 6) {
853  candidates.push_back(move);
854  }
855  return true;
856  });
857  if (candidates.size() == 1) return candidates[0];
858  std::cerr << "Invalid O-O" << std::endl;
859  return absl::nullopt;
860  }
861 
862  auto move_annotation = SplitAnnotations(move);
863  move = move_annotation.first;
864  if (move.empty()) { return absl::nullopt; }
865 
866  auto annotation = move_annotation.second;
867 
868  // A move starts with a single letter identifying the piece. This may be
869  // omitted for pawns.
870  PieceType piece_type = PieceType::kPawn;
871  std::string pieces = "PNBRQK";
872  if (pieces.find(move[0]) != std::string::npos) { // NOLINT
873  auto maybe_piece_type = PieceTypeFromChar(move[0]);
874  if (!maybe_piece_type) {
875  std::cerr << "Invalid piece type: " << move[0] << std::endl;
876  return absl::nullopt;
877  }
878  piece_type = *maybe_piece_type;
879  move = std::string(absl::ClippedSubstr(move, 1));
880  }
881 
882  // A move always ends with the destination square.
883  if (move.size() < 2) {
884  std::cerr << "Missing destination square" << std::endl;
885  return absl::nullopt;
886  }
887  auto destination = std::string(absl::ClippedSubstr(move, move.size() - 2));
888  move = move.substr(0, move.size() - 2);
889 
890  auto dest_file = ParseFile(destination[0]);
891  auto dest_rank = ParseRank(destination[1]);
892 
893  if (!dest_file || !dest_rank) {
894  std::cerr << "Failed to parse destination square: " << destination
895  << std::endl;
896  return absl::nullopt;
897  }
898 
899  Square destination_square{*dest_file, *dest_rank};
900 
901  // Captures are indicated by a 'x' immediately preceding the destination.
902  // This is irrelevant for parsing, so we just drop it.
903  if (!move.empty() && move[move.size() - 1] == 'x') {
904  move = move.substr(0, move.size() - 1);
905  }
906 
907  // If necessary, source rank and/or file are also included for
908  // disambiguation.
909  absl::optional<int8_t> source_file, source_rank;
910  if (!move.empty()) {
911  source_file = ParseFile(move[0]);
912  if (source_file) {
913  move = std::string(absl::ClippedSubstr(move, 1));
914  }
915  }
916  if (!move.empty()) {
917  source_rank = ParseRank(move[0]);
918  if (source_rank) {
919  move = std::string(absl::ClippedSubstr(move, 1));
920  }
921  }
922 
923  if (!move.empty()) { return absl::nullopt; }
924 
925  // Pawn promations are annotated with =Q to indicate the promotion type.
926  absl::optional<PieceType> promotion_type;
927  if (!annotation.empty() && annotation[0] == '=') {
928  if (annotation.size() < 2) { return absl::nullopt; }
929  auto maybe_piece = PieceTypeFromChar(annotation[1]);
930  if (!maybe_piece) return absl::optional<Move>();
931  promotion_type = maybe_piece;
932  }
933 
934  std::vector<Move> candidates;
935  GenerateLegalMoves([&candidates, destination_square, piece_type, source_file,
936  source_rank, promotion_type, this](const Move &move) {
937  PieceType moving_piece_type = at(move.from).type;
938  if (move.to == destination_square && moving_piece_type == piece_type &&
939  (!source_file || move.from.x == *source_file) &&
940  (!source_rank || move.from.y == *source_rank) &&
941  (!promotion_type || move.promotion_type == *promotion_type)) {
942  candidates.push_back(move);
943  }
944  return true;
945  });
946 
947  if (candidates.size() == 1) return candidates[0];
948  std::cerr << "expected exactly one matching move, got " << candidates.size()
949  << std::endl;
950  return absl::optional<Move>();
951 }

953 absl::optional<Move> ChessBoard::ParseLANMove(const std::string &move) const {
954  if (move.empty()) { return absl::nullopt; }
955 
956  // Long algebraic notation moves (of the variant we care about) are in one of
957  // two forms -
958  // "anan" (eg. "e2e4") or "anana" (eg. "f7f8q")
959  if (move.size() == 4 || move.size() == 5) {
960  if (move[0] < 'a' || move[0] >= ('a' + board_size_) || move[1] < '1' ||
961  move[1] >= ('1' + board_size_) || move[2] < 'a' ||
962  move[2] >= ('a' + board_size_) || move[3] < '1' ||
963  move[3] >= ('1' + board_size_)) {
964  return absl::nullopt;
965  }
966 
967  if (move.size() == 5 && move[4] != 'q' && move[4] != 'r' &&
968  move[4] != 'b' && move[4] != 'n') {
969  return absl::nullopt;
970  }
971 
972  auto from = SquareFromString(move.substr(0, 2));
973  auto to = SquareFromString(std::string(absl::ClippedSubstr(move, 2, 2)));
974  if (from && to) {
975  absl::optional<PieceType> promotion_type;
976  if (move.size() == 5) {
977  promotion_type = PieceTypeFromChar(move[4]);
978  if (!promotion_type) {
979  std::cerr << "Invalid promotion type" << std::endl;
980  return absl::nullopt;
981  }
982  }
983 
984  std::vector<Move> candidates;
985  GenerateLegalMoves(
986  [&to, &from, &promotion_type, &candidates](const Move &move) {
987  if (move.from == *from && move.to == *to &&
988  (!promotion_type || (move.promotion_type == *promotion_type))) {
989  candidates.push_back(move);
990  }
991  return true;
992  });
993 
994  if (candidates.empty()) {
995  std::cerr << "Illegal move - " << move << " on " << ToUnicodeString()
996  << std::endl;
997  return Move();
998  } else if (candidates.size() > 1) {
999  std::cerr << "Multiple matches (is promotion type missing?) - " << move
1000  << std::endl;
1001  return Move();
1002  }
1003 
1004  return candidates[0];
1005  }
1006  } else {
1007  return absl::nullopt;
1008  }
1009  SpielFatalError("All conditionals failed; this is a bug.");
1010 }

1012 void ChessBoard::ApplyMove(const Move &move) {
1013  // Skip applying a move if it's a pass.
1014  if (move == kPassMove) {
1015  if (to_play_ == Color::kBlack) ++move_number_;
1016  SetToPlay(OppColor(to_play_));
1017  SetEpSquare(chess::kInvalidSquare);
1018  return;
1019  }
1020 
1021  // Most moves are simple - we remove the moving piece from the original
1022  // square, and put it on the destination square, overwriting whatever was
1023  // there before, update the 50 move counter, and update castling rights.
1024  //
1025  // There are a few exceptions - castling, en passant, promotions, double pawn
1026  // pushes. They require special adjustments in addition to those things. We
1027  // do them after the basic apply move.
1028 
1029  Piece moving_piece = at(move.from);
1030  Piece destination_piece = at(move.to);
1031 
1032  // We have to do it in this order because in Chess960 the king can castle
1033  // in-place! That's the only possibility for move.from == move.to.
1034  set_square(move.from, kEmptyPiece);
1035  set_square(move.to, moving_piece);
1036 
1037  // Whether the move is irreversible for the purpose of the 50-moves rule. Note
1038  // that although castling (and losing castling rights) should be irreversible,
1039  // it is counted as reversible here.
1040  // Irreversible moves are pawn moves and captures. We don't have to make a
1041  // special case for en passant, since they are pawn moves anyways.
1042  bool irreversible = (moving_piece.type == PieceType::kPawn) ||
1043  (destination_piece.type != PieceType::kEmpty);
1044 
1045  if (irreversible) {
1046  SetIrreversibleMoveCounter(0);
1047  } else {
1048  SetIrreversibleMoveCounter(IrreversibleMoveCounter() + 1);
1049  }
1050 
1051  // Castling rights can be lost in a few different ways -
1052  // 1. The king moves (loses both rights), including castling.
1053  // 2. A rook moves (loses the right on that side).
1054  // 3. Captures an opponent rook (OPPONENT loses the right on that side).
1055  if (moving_piece.type == PieceType::kKing) {
1056  SetCastlingRight(to_play_, CastlingDirection::kLeft, false);
1057  SetCastlingRight(to_play_, CastlingDirection::kRight, false);
1058  }
1059  if (moving_piece.type == PieceType::kRook) {
1060  // TODO(author12): Fix this for Chess960, which requires storing initial
1061  // positions of rooks.
1062  if ((to_play_ == Color::kWhite && move.from == Square{0, 0}) ||
1063  (to_play_ == Color::kBlack && move.from == Square{0, 7})) {
1064  SetCastlingRight(to_play_, CastlingDirection::kLeft, false);
1065  } else if ((to_play_ == Color::kWhite && move.from == Square{7, 0}) ||
1066  (to_play_ == Color::kBlack && move.from == Square{7, 7})) {
1067  SetCastlingRight(to_play_, CastlingDirection::kRight, false);
1068  }
1069  }
1070  if (destination_piece.type == PieceType::kRook) {
1071  if ((to_play_ == Color::kWhite && move.to == Square{0, 7}) ||
1072  (to_play_ == Color::kBlack && move.to == Square{0, 0})) {
1073  SetCastlingRight(OppColor(to_play_), CastlingDirection::kLeft, false);
1074  } else if ((to_play_ == Color::kWhite && move.to == Square{7, 7}) ||
1075  (to_play_ == Color::kBlack && move.to == Square{7, 0})) {
1076  SetCastlingRight(OppColor(to_play_), CastlingDirection::kRight, false);
1077  }
1078  }
1079 
1080  // Special cases that require adjustment -
1081  // 1. Castling
1082  if (move.is_castling) {
1083  SPIEL_CHECK_EQ(moving_piece.type, PieceType::kKing);
1084  // We can tell which side we are castling to using "to" square.
1085  if (to_play_ == Color::kWhite) {
1086  if (move.to == Square{2, 0}) {
1087  // left castle
1088  // TODO(author12): In Chess960, rooks can be anywhere, so delete the
1089  // correct squares.
1090  set_square(Square{0, 0}, kEmptyPiece);
1091  set_square(Square{2, 0}, Piece{Color::kWhite, PieceType::kKing});
1092  set_square(Square{3, 0}, Piece{Color::kWhite, PieceType::kRook});
1093  } else if (move.to == Square{6, 0}) {
1094  // right castle
1095  set_square(Square{7, 0}, kEmptyPiece);
1096  set_square(Square{6, 0}, Piece{Color::kWhite, PieceType::kKing});
1097  set_square(Square{5, 0}, Piece{Color::kWhite, PieceType::kRook});
1098  } else {
1099  std::cerr << "Trying to castle but destination is not valid."
1100  << std::endl;
1101  }
1102  } else {
1103  if (move.to == Square{2, 7}) {
1104  // left castle
1105  set_square(Square{0, 7}, kEmptyPiece);
1106  set_square(Square{2, 7}, Piece{Color::kBlack, PieceType::kKing});
1107  set_square(Square{3, 7}, Piece{Color::kBlack, PieceType::kRook});
1108  } else if (move.to == Square{6, 7}) {
1109  // right castle
1110  set_square(Square{7, 7}, kEmptyPiece);
1111  set_square(Square{6, 7}, Piece{Color::kBlack, PieceType::kKing});
1112  set_square(Square{5, 7}, Piece{Color::kBlack, PieceType::kRook});
1113  } else {
1114  std::cerr << "Trying to castle but destination is not valid.";
1115  }
1116  }
1117  }
1118 
1119  // 2. En-passant
1120  if (moving_piece.type == PieceType::kPawn && move.from.x != move.to.x &&
1121  destination_piece.type == PieceType::kEmpty) {
1122  if (move.to != EpSquare()) {
1123  std::cerr << "We are trying to capture an empty square "
1124  << "with a pawn, but the square is not the en passant square:\n"
1125  << *this << "\n"
1126  << "Move: " << move.ToString() << std::endl;
1127  SpielFatalError("Trying to apply an invalid move");
1128  }
1129  Square captured_pawn_square = move.to;
1130  if (to_play_ == Color::kWhite) {
1131  --captured_pawn_square.y;
1132  } else {
1133  ++captured_pawn_square.y;
1134  }
1135  SPIEL_CHECK_EQ(at(captured_pawn_square),
1136  (Piece{OppColor(to_play_), PieceType::kPawn}));
1137  set_square(captured_pawn_square, kEmptyPiece);
1138  }
1139 
1140  // 3. Promotions
1141  if (moving_piece.type == PieceType::kPawn && IsPawnPromotionRank(move.to)) {
1142  set_square(move.to, Piece{at(move.to).color, move.promotion_type});
1143  }
1144 
1145  // 4. Double push
1146  if (moving_piece.type == PieceType::kPawn &&
1147  abs(move.from.y - move.to.y) == 2) {
1148  SetEpSquare(Square{move.from.x,
1149  static_cast<int8_t>((move.from.y + move.to.y) / 2)});
1150  } else {
1151  SetEpSquare(kInvalidSquare);
1152  }
1153 
1154  if (to_play_ == Color::kBlack) {
1155  ++move_number_;
1156  }
1157 
1158  SetToPlay(OppColor(to_play_));
1159 }

1167 bool ChessBoard::UnderAttack(const Square &sq, Color our_color) const {
1168  SPIEL_CHECK_NE(sq, kInvalidSquare);
1169 
1170  bool under_attack = false;
1171  Color opponent_color = OppColor(our_color);
1172 
1173  // We do this by pretending we are a piece of different types, and seeing if
1174  // we can attack opponent pieces. Eg. if we pretend we are a knight, and can
1175  // attack an opponent knight, that means the knight can also attack us.
1176 
1177  // King moves (this is possible because we use this function for checking
1178  // whether we are moving into check, and we can be trying to move the king
1179  // into a square attacked by opponent king).
1180  GenerateKingDestinations_(
1181  sq, our_color, [this, &under_attack, &opponent_color](const Square &to) {
1182  if (at(to) == Piece{opponent_color, PieceType::kKing}) {
1183  under_attack = true;
1184  }
1185  });
1186  if (under_attack) {
1187  return true;
1188  }
1189 
1190  // Rook moves (for rooks and queens)
1191  GenerateRookDestinations_(
1192  sq, our_color, PseudoLegalMoveSettings::kAcknowledgeEnemyPieces,
1193  [this, &under_attack, &opponent_color](const Square &to) {
1194  if ((at(to) == Piece{opponent_color, PieceType::kRook}) ||
1195  (at(to) == Piece{opponent_color, PieceType::kQueen})) {
1196  under_attack = true;
1197  }
1198  });
1199  if (under_attack) {
1200  return true;
1201  }
1202 
1203  // Bishop moves (for bishops and queens)
1204  GenerateBishopDestinations_(
1205  sq, our_color, PseudoLegalMoveSettings::kAcknowledgeEnemyPieces,
1206  [this, &under_attack, &opponent_color](const Square &to) {
1207  if ((at(to) == Piece{opponent_color, PieceType::kBishop}) ||
1208  (at(to) == Piece{opponent_color, PieceType::kQueen})) {
1209  under_attack = true;
1210  }
1211  });
1212  if (under_attack) {
1213  return true;
1214  }
1215 
1216  // Knight moves
1217  GenerateKnightDestinations_(
1218  sq, our_color, [this, &under_attack, &opponent_color](const Square &to) {
1219  if (at(to) == Piece{opponent_color, PieceType::kKnight}) {
1220  under_attack = true;
1221  }
1222  });
1223  if (under_attack) {
1224  return true;
1225  }
1226 
1227  // Pawn captures.
1228  GeneratePawnCaptureDestinations_(
1229  sq, our_color, PseudoLegalMoveSettings::kAcknowledgeEnemyPieces,

1304 // optionally safe (not under attack).

1313  const int x_end = std::max(sq1.x, sq2.x);

1324  if (x_in_between && IsFriendly(test_square, our_color)) return false;

1329 template <typename YieldFn>
1330 void ChessBoard::GenerateCastlingDestinations_(Square sq, Color color,
1331  PseudoLegalMoveSettings settings,
1332  const YieldFn &yield) const {
1333  // There are 8 conditions for castling -
1334  // 1. The rook involved must not have moved.
1335  // 2. The king must not have moved.
1336  // 3. The rook involved must still be alive.
1337  // 4. All squares king jumps over must be empty.
1338  // 5. All squares the rook jumps over must be empty.
1339  // 6. The squares the king jumps over must not be under attack.
1340  // 7. The king must not be in check.
1341  // (8). The square the king ends up in must not be under attack.
1342  //
1343  // We don't check for (8) here because this is not unique to castling, and
1344  // we will check for it later.
1345  //
1346  // We use the generalized definition of castling from Chess960, instead of
1347  // hard-coding starting squares.
1348  // By Chess960 rules, the king and rook end up in the same positions as in
1349  // standard chess, but they can start from any squares.
1350  //
1351  // Castling to one side doesn't necessarily mean the king will move towards
1352  // that side.
1353  // Eg.
1354  // |RK...R..| + long castle (to the left) =>
1355  // |..KR.R..|
1356 
1357  // castling is not defined for other chessboards than the standard one
1358  if (board_size_ != 8) {
1359  return;
1360  }
1361 
1362  const auto check_castling_conditions =
1363  [this, &sq, &color, &settings](int8_t x_direction) -> bool {
1364  // First we need to find the rook.
1365  Square rook_sq = sq + Offset{x_direction, 0};
1366  bool rook_found = false;
1367 
1368  // Yes, we do actually have to check colour -
1369  // https://github.com/official-stockfish/Stockfish/issues/356
1370  for (; InBoardArea(rook_sq); rook_sq.x += x_direction) {
1371  if (at(rook_sq) == Piece{color, PieceType::kRook}) {
1372  rook_found = true;
1373  break;
1374  }
1375  }
1376 
1377  if (!rook_found) {
1378  std::cerr << "Where did our rook go?" << *this << "\n"
1379  << "Square: " << SquareToString(sq) << std::endl;
1380  SpielFatalError("Rook not found");
1381  }
1382 
1383  int8_t rook_final_x = x_direction == -1 ? 3 /* d-file */ : 5 /* f-file */;
1384  Square rook_final_sq = Square{rook_final_x, sq.y};
1385  int8_t king_final_x = x_direction == -1 ? 2 /* c-file */ : 6 /* g-file */;
1386  Square king_final_sq = Square{king_final_x, sq.y};
1387 
1388  // 4. 5. 6. All squares the king and rook jump over, including the final
1389  // squares, must be empty. Squares king jumps over must additionally be
1390  // safe.
1391  const bool make_king_jump_check =
1392  !king_in_check_allowed_ &&
1393  settings == PseudoLegalMoveSettings::kAcknowledgeEnemyPieces;
1394  if (!CanCastleBetween(rook_sq, rook_final_sq, false, settings) ||
1395  !CanCastleBetween(sq, king_final_sq, make_king_jump_check, settings)) {
1396  return false;
1397  }
1398 
1399  return true;
1400  };
1401 
1402  // 1. 2. 3. Moving the king, moving the rook, or the rook getting captured
1403  // will reset the flag.
1404  bool can_left_castle = CastlingRight(color, CastlingDirection::kLeft) &&
1405  check_castling_conditions(-1);
1406  bool can_right_castle = CastlingRight(color, CastlingDirection::kRight) &&
1407  check_castling_conditions(1);
1408 
1409  if (can_left_castle || can_right_castle) {
1410  // 7. No castling to escape from check.
1411  if (UnderAttack(sq, color) &&
1412  !(king_in_check_allowed_ ||
1413  settings == PseudoLegalMoveSettings::kBreachEnemyPieces)) {
1414  return;
1415  }
1416  if (can_left_castle) {
1417  yield(Square{static_cast<int8_t>(2), sq.y});
1418  }
1419 
1420  if (can_right_castle) {
1421  yield(Square{static_cast<int8_t>(6), sq.y});
1422  }
1423  }
1424 }
