67std::vector<VirtualPoint> HandicapStones(int num_handicap) {
68 if (num_handicap < 2 || num_handicap > 9) return {};
69
70 static std::array<VirtualPoint, 9> placement = {
71 {MakePoint("d4"), MakePoint("q16"), MakePoint("d16"), MakePoint("q4"),
72 MakePoint("d10"), MakePoint("q10"), MakePoint("k4"), MakePoint("k16"),
73 MakePoint("k10")}};
74 static VirtualPoint center = MakePoint("k10");
75
76 std::vector<VirtualPoint> points;
77 points.reserve(num_handicap);
78 for (int i = 0; i < num_handicap; ++i) {
79 points.push_back(placement[i]);
80 }
81
82 if (num_handicap >= 5 && num_handicap % 2 == 1) {
83 points[num_handicap - 1] = center;
84 }
85
86 return points;
87}

114void CursorGoState::ObservationTensor(Player player,
115 absl::Span<float> values) const {
116 SPIEL_CHECK_GE(player, 0);
117 SPIEL_CHECK_LT(player, num_players_);
118
119 int num_cells = board_.board_size() * board_.board_size();
120 SPIEL_CHECK_EQ(values.size(), num_cells * (kCellStates + 3));
121 std::fill(values.begin(), values.end(), 0.);
122
123 // Add planes: black, white, empty.
124 int cell = 0;
125 for (VirtualPoint p : BoardPoints(board_.board_size())) {
126 int color_val = static_cast<int>(board_.PointColor(p));
127 values[num_cells * color_val + cell] = 1.0;
128 ++cell;
129 }
130 SPIEL_CHECK_EQ(cell, num_cells);
131
132 // Fourth plane for cursor position.
133 const auto [row, col] = cursor_[ColorToPlayer(to_play_)];
134 const int cursor_cell = row * board_.board_size() + col;
135 values[num_cells * kCellStates + cursor_cell] = 1.0;
136
137 // Add a fifth binary plane for komi (whether white is to play).
138 std::fill(values.begin() + ((1 + kCellStates) * num_cells),
139 values.begin() + ((2 + kCellStates) * num_cells),
140 (to_play_ == GoColor::kWhite ? 1.0 : 0.0));
141
142 // Add a sixth binary plane for the number of cursor moves.
143 std::fill(values.begin() + ((2 + kCellStates) * num_cells), values.end(),
144 static_cast<float>(cursor_moves_count_) / max_cursor_moves_);
145}

147std::vector<Action> CursorGoState::LegalActions() const {
148 std::vector<Action> actions{};
149 if (is_terminal_) return actions;
150 const auto cursor = cursor_[ColorToPlayer(to_play_)];
151 if (cursor_moves_count_ < max_cursor_moves_) {
152 const auto [row, col] = cursor;
153 if (row < board_.board_size() - 1) actions.push_back(kActionUp);
154 if (row > 0) actions.push_back(kActionDown);
155 if (col > 0) actions.push_back(kActionLeft);
156 if (col < board_.board_size() - 1) actions.push_back(kActionRight);
157 }
158 if (board_.IsLegalMove(VirtualPointFrom2DPoint(cursor), to_play_))
159 actions.push_back(kActionPlaceStone);
160 actions.push_back(kActionPass);
161 return actions;
162}

187std::vector<double> CursorGoState::Returns() const {
188 if (!is_terminal_) return {0.0, 0.0};
189
190 if (superko_) {
191 // Superko rules (https://senseis.xmp.net/?Superko) are complex and vary
192 // between rulesets.
193 // For simplicity and because superkos are very rare, we just treat them as
194 // a draw.
195 return {kDrawUtility, kDrawUtility};
196 }
197
198 // Score with Tromp-Taylor.
199 float black_score = TrompTaylorScore(board_, komi_, handicap_);
200
201 std::vector<double> returns(kNumPlayers);
202 if (black_score > 0) {
203 returns[ColorToPlayer(GoColor::kBlack)] = kWinUtility;
204 returns[ColorToPlayer(GoColor::kWhite)] = kLossUtility;
205 } else if (black_score < 0) {
206 returns[ColorToPlayer(GoColor::kBlack)] = kLossUtility;
207 returns[ColorToPlayer(GoColor::kWhite)] = kWinUtility;
208 } else {
209 returns[ColorToPlayer(GoColor::kBlack)] = kDrawUtility;
210 returns[ColorToPlayer(GoColor::kWhite)] = kDrawUtility;
211 }
212 return returns;
213}

219void CursorGoState::DoApplyAction(Action action) {
220 if (action == kActionPlaceStone || action == kActionPass) {
221 VirtualPoint point =
222 (action == kActionPass)
223 ? go::kVirtualPass
224 : VirtualPointFrom2DPoint(cursor_[ColorToPlayer(to_play_)]);
225 SPIEL_CHECK_TRUE(board_.PlayMove(point, to_play_));
226 is_terminal_ = last_move_was_pass_ && (action == kActionPass);
227 last_move_was_pass_ = (action == kActionPass);
228 to_play_ = OppColor(to_play_);
229 cursor_moves_count_ = 0;
230
231 bool was_inserted = repetitions_.insert(board_.HashValue()).second;
232 if (!was_inserted && action == kActionPlaceStone) {
233 // We have encountered this position before.
234 superko_ = true;
235 }
236 } else {
237 switch (action) {
238 case kActionUp:
239 cursor_[ColorToPlayer(to_play_)].first++;
240 break;
241 case kActionDown:
242 cursor_[ColorToPlayer(to_play_)].first--;
243 break;
244 case kActionLeft:
245 cursor_[ColorToPlayer(to_play_)].second--;
246 break;
247 case kActionRight:
248 cursor_[ColorToPlayer(to_play_)].second++;
249 break;
250 default:
251 SpielFatalError(absl::StrCat("Invalid action ", action));
252 }
253 ++cursor_moves_count_;
254 }

257void CursorGoState::ResetBoard() {
258 board_.Clear();
259 const int middle = board_.board_size() / 2;
260 cursor_[0] = {middle, middle};
261 cursor_[1] = {middle, middle};
262 cursor_moves_count_ = 0;
263 if (handicap_ < 2) {
264 to_play_ = GoColor::kBlack;
265 } else {
266 for (VirtualPoint p : HandicapStones(handicap_)) {
267 board_.PlayMove(p, GoColor::kBlack);
268 }
269 to_play_ = GoColor::kWhite;
270 }
271
272 repetitions_.clear();
273 repetitions_.insert(board_.HashValue());
274 superko_ = false;
275 is_terminal_ = false;
276 last_move_was_pass_ = false;
277}
