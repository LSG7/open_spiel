68  def action_probabilities(self, state, player_id=None):
69  mu_bar_t = state.distribution_average()
70  x_t = state.x
71  q = state.cross_q
72  n_actions_per_side = state.n_actions_per_side
73  lin_action = (q + state.eta_t()) * (mu_bar_t - x_t)
74  action = n_actions_per_side + min(
75  n_actions_per_side, max(round(lin_action), -n_actions_per_side))
76  action_prob = [(a, 0.0) for a in state.legal_actions()]
77  action_prob[action] = (action, 1.0)
78  return dict(action_prob)
