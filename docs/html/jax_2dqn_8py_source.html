233  def _epsilon_greedy(self, info_state, legal_actions, epsilon):
234  """Returns a valid epsilon-greedy action and valid action probs.
235 
236  Action probabilities are given by a softmax over legal q-values.
237 
238  Args:
239  info_state: hashable representation of the information state.
240  legal_actions: list of legal actions at `info_state`.
241  epsilon: float, probability of taking an exploratory action.
242 
243  Returns:
244  A valid epsilon-greedy action and valid action probabilities.
245  """
246  probs = np.zeros(self._num_actions_num_actions)
247  legal_one_hot = np.zeros(self._num_actions_num_actions)
248  legal_one_hot[legal_actions] = 1
249  if np.random.rand() < epsilon:
250  action = np.random.choice(legal_actions)
251  probs[legal_actions] = 1.0 / len(legal_actions)
252  else:
253  info_state = np.reshape(info_state, [1, -1])
254  q_values = self.hk_network_applyhk_network_apply(self.params_q_networkparams_q_network, info_state)
255  legal_q_values = q_values[0] + (
256  1 - legal_one_hot) * ILLEGAL_ACTION_LOGITS_PENALTY
257  action = int(np.argmax(legal_q_values))
258  probs[action] = 1.0
259  return action, probs

261  def _get_epsilon(self, is_evaluation, power=1.0):
262  """Returns the evaluation or decayed epsilon value."""
263  if is_evaluation:
264  return 0.0
265  decay_steps = min(self._step_counter_step_counter, self._epsilon_decay_duration_epsilon_decay_duration)
266  decayed_epsilon = (
267  self._epsilon_end_epsilon_end + (self._epsilon_start_epsilon_start - self._epsilon_end_epsilon_end) *
268  (1 - decay_steps / self._epsilon_decay_duration_epsilon_decay_duration)**power)

271  def _loss(self, param, param_target, info_states, actions, rewards,
272  next_info_states, are_final_steps, legal_actions_mask):
273 
274  q_values = self.hk_networkhk_network.apply(param, info_states)
275  target_q_values = self.hk_networkhk_network.apply(param_target, next_info_states)
276  # Sum a large negative constant to illegal action logits before taking the
277  # max. This prevents illegal action values from being considered as target.
278  max_next_q = jnp.max(
279  target_q_values +
280  (1 - legal_actions_mask) * ILLEGAL_ACTION_LOGITS_PENALTY,
281  axis=-1)
282  max_next_q = jax.numpy.where(
283  1 - are_final_steps, x=max_next_q, y=jnp.zeros_like(max_next_q))
284  target = (
285  rewards + (1 - are_final_steps) * self._discount_factor_discount_factor * max_next_q)
286  target = jax.lax.stop_gradient(target)
287  predictions = jnp.sum(q_values * actions, axis=-1)
288  loss_value = self.loss_funcloss_func(predictions - target)
289  return loss_value
