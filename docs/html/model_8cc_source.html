194 std::vector<torch::Tensor> ResOutputBlockImpl::forward(torch::Tensor x,
195  torch::Tensor mask) {
196  torch::Tensor value_output = torch::relu(value_batch_norm_(value_conv_(x)));
197  value_output = value_output.view({-1, value_observation_size_});
198  value_output = torch::relu(value_linear1_(value_output));
199  value_output = torch::tanh(value_linear2_(value_output));
200 
201  torch::Tensor policy_logits =
202  torch::relu(policy_batch_norm_(policy_conv_(x)));
203  policy_logits = policy_logits.view({-1, policy_observation_size_});
204  policy_logits = policy_linear_(policy_logits);
205  policy_logits = torch::where(mask, policy_logits,
206  -(1 << 16) * torch::ones_like(policy_logits));
207 
208  return {value_output, policy_logits};
209 }

260 ModelImpl::ModelImpl(const ModelConfig& config, const std::string& device)
261  : device_(device),
262  num_torso_blocks_(config.nn_depth),
263  weight_decay_(config.weight_decay) {
264  // Save config.nn_model to class
265  nn_model_ = config.nn_model;
266 
267  int input_size = 1;
268  for (const auto& num : config.observation_tensor_shape) {
269  if (num > 0) {
270  input_size *= num;
271  }
272  }
273  // Decide if resnet or MLP
274  if (config.nn_model == "resnet") {
275  int channels = config.observation_tensor_shape[0];
276  int height = config.observation_tensor_shape[1];
277  int width = config.observation_tensor_shape[2];
278 
279  ResInputBlockConfig input_config = {/*input_channels=*/channels,
280  /*input_height=*/height,
281  /*input_width=*/width,
282  /*filters=*/config.nn_width,
283  /*kernel_size=*/3,
284  /*padding=*/1};
285 
286  ResTorsoBlockConfig residual_config = {/*input_channels=*/config.nn_width,
287  /*filters=*/config.nn_width,
288  /*kernel_size=*/3,
289  /*padding=*/1};
290 
291  ResOutputBlockConfig output_config = {
292  /*input_channels=*/config.nn_width,
293  /*value_filters=*/1,
294  /*policy_filters=*/2,
295  /*kernel_size=*/1,
296  /*padding=*/0,
297  /*value_linear_in_features=*/1 * width * height,
298  /*value_linear_out_features=*/config.nn_width,
299  /*policy_linear_in_features=*/2 * width * height,
300  /*policy_linear_out_features=*/config.number_of_actions,
301  /*value_observation_size=*/1 * width * height,
302  /*policy_observation_size=*/2 * width * height};
303 
304  layers_->push_back(ResInputBlock(input_config));
305  for (int i = 0; i < num_torso_blocks_; i++) {
306  layers_->push_back(ResTorsoBlock(residual_config, i));
307  }
308  layers_->push_back(ResOutputBlock(output_config));
309 
310  register_module("layers", layers_);
311 
312  } else if (config.nn_model == "mlp") {
313  layers_->push_back(MLPBlock(input_size, config.nn_width));
314  for (int i = 0; i < num_torso_blocks_; i++) {
315  layers_->push_back(MLPBlock(config.nn_width, config.nn_width));
316  }
317  layers_->push_back(
318  MLPOutputBlock(config.nn_width, config.number_of_actions));
319 
320  register_module("layers", layers_);
321  } else {
322  throw std::runtime_error("Unknown nn_model: " + config.nn_model);
323  }
324 }

332 std::vector<torch::Tensor> ModelImpl::losses(torch::Tensor inputs,
333  torch::Tensor masks,
334  torch::Tensor policy_targets,
335  torch::Tensor value_targets) {
336  std::vector<torch::Tensor> output = this->forward_(inputs, masks);
337 
338  torch::Tensor value_predictions = output[0];
339  torch::Tensor policy_predictions = output[1];
340 
341  // Policy loss (cross-entropy).
342  torch::Tensor policy_loss = torch::sum(
343  -policy_targets * torch::log_softmax(policy_predictions, 1), -1);
344  policy_loss = torch::mean(policy_loss);
345 
346  // Value loss (mean-squared error).
347  torch::nn::MSELoss mse_loss;
348  torch::Tensor value_loss = mse_loss(value_predictions, value_targets);
349 
350  // L2 regularization loss (weights only).
351  torch::Tensor l2_regularization_loss = torch::full(
352  {1, 1}, 0, torch::TensorOptions().dtype(torch::kFloat32).device(device_));
353  for (auto& named_parameter : this->named_parameters()) {
354  // named_parameter is essentially a key-value pair:
355  // {key, value} == {std::string name, torch::Tensor parameter}
356  std::string parameter_name = named_parameter.key();
357 
358  // Do not include bias' in the loss.
359  if (parameter_name.find("bias") != std::string::npos) {
360  continue;
361  }
362 
363  // Copy TensorFlow's l2_loss function.
364  // https://www.tensorflow.org/api_docs/python/tf/nn/l2_loss
365  l2_regularization_loss +=
366  weight_decay_ * torch::sum(torch::square(named_parameter.value())) / 2;
367  }
368 
369  return {policy_loss, value_loss, l2_regularization_loss};
370 }

372 std::vector<torch::Tensor> ModelImpl::forward_(torch::Tensor x,
373  torch::Tensor mask) {
374  std::vector<torch::Tensor> output;
375  if (this->nn_model_ == "resnet") {
376  for (int i = 0; i < num_torso_blocks_ + 2; i++) {
377  if (i == 0) {
378  x = layers_[i]->as<ResInputBlock>()->forward(x);
379  } else if (i >= num_torso_blocks_ + 1) {
380  output = layers_[i]->as<ResOutputBlock>()->forward(x, mask);
381  } else {
382  x = layers_[i]->as<ResTorsoBlock>()->forward(x);
383  }
384  }
385  } else if (this->nn_model_ == "mlp") {
386  for (int i = 0; i < num_torso_blocks_ + 1; i++) {
387  x = layers_[i]->as<MLPBlock>()->forward(x);
388  }
389  output = layers_[num_torso_blocks_ + 1]->as<MLPOutputBlockImpl>()
390  ->forward(x, mask);
391  } else {
392  throw std::runtime_error("Unknown nn_model: " + this->nn_model_);
393  }
394  return output;
395 }
