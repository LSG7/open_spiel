135 std::array<std::string, kNumSuits> HeartsState::FormatHand(
136  int player, bool mark_voids) const {
137  // Current hand, except in the terminal state when we use the original hand
138  // to enable an easy review of the whole deal.
139  auto deal = IsTerminal() ? initial_deal_ : holder_;
140  std::array<std::string, kNumSuits> cards;
141  for (int suit = 0; suit < kNumSuits; ++suit) {
142  cards[suit].push_back(kSuitChar[suit]);
143  cards[suit].push_back(' ');
144  bool is_void = true;
145  for (int rank = kNumCardsPerSuit - 1; rank >= 0; --rank) {
146  if (player == deal[Card(Suit(suit), rank)]) {
147  cards[suit].push_back(kRankChar[rank]);
148  is_void = false;
149  }
150  }
151  if (is_void && mark_voids) absl::StrAppend(&cards[suit], "none");
152  }
153  return cards;
154 }

156 std::string HeartsState::FormatDeal() const {
157  std::string rv;
158  std::array<std::array<std::string, kNumSuits>, kNumPlayers> cards;
159  for (auto player : {kNorth, kEast, kSouth, kWest})
160  cards[player] = FormatHand(player, /*mark_voids=*/false);
161  constexpr int kColumnWidth = 8;
162  std::string padding(kColumnWidth, ' ');
163  for (int suit = kNumSuits - 1; suit >= 0; --suit)
164  absl::StrAppend(&rv, padding, cards[kNorth][suit], "\n");
165  for (int suit = kNumSuits - 1; suit >= 0; --suit)
166  absl::StrAppend(&rv, absl::StrFormat("%-8s", cards[kWest][suit]), padding,
167  cards[kEast][suit], "\n");
168  for (int suit = kNumSuits - 1; suit >= 0; --suit)
169  absl::StrAppend(&rv, padding, cards[kSouth][suit], "\n");
170  return rv;
171 }

173 std::string HeartsState::FormatPass() const {
174  std::string rv = "\n\nPassed Cards:";
175  for (int player = 0; player < kNumPlayers; ++player) {
176  absl::StrAppend(&rv, "\n", DirString(player), ": ");
177  for (int card : passed_cards_[player]) {
178  absl::StrAppend(&rv, CardString(card), " ");
179  }
180  }
181  // Cards are not received until all players have completed passing.
182  // West is the last player to pass.
183  if (passed_cards_[static_cast<int>(kWest)].size() == kNumCardsInPass) {
184  absl::StrAppend(&rv, "\n\nReceived Cards:");
185  for (int receiver = 0; receiver < kNumPlayers; ++receiver) {
186  absl::StrAppend(&rv, "\n", DirString(receiver), ": ");
187  int passer =
188  (receiver + kNumPlayers - static_cast<int>(pass_dir_)) % kNumPlayers;
189  for (int card : passed_cards_[passer]) {
190  absl::StrAppend(&rv, CardString(card), " ");
191  }
192  }
193  }
194  absl::StrAppend(&rv, "\n");
195  return rv;
196 }

198 std::string HeartsState::FormatPass(Player player) const {
199  std::string rv = "\nPassed Cards: ";
200  std::vector<int> passed_cards = passed_cards_[player];
201  // Sort cards because players don't have access to the order in which the
202  // cards were selected to be passed. Knowing the order could allow for
203  // information leakage.
204  absl::c_sort(passed_cards);
205  for (int card : passed_cards) absl::StrAppend(&rv, CardString(card), " ");
206  // Cards are not received until all players have completed passing.
207  // West is the last player to pass.
208  if (passed_cards_[static_cast<int>(kWest)].size() == kNumCardsInPass) {
209  absl::StrAppend(&rv, "\n\nReceived Cards: ");
210  int passing_player =
211  (player + kNumPlayers - static_cast<int>(pass_dir_)) % kNumPlayers;
212  std::vector<int> received_cards = passed_cards_[passing_player];
213  absl::c_sort(received_cards);
214  for (int card : received_cards) absl::StrAppend(&rv, CardString(card), " ");
215  }
216  absl::StrAppend(&rv, "\n");
217  return rv;
218 }

241 void HeartsState::InformationStateTensor(Player player,
242  absl::Span<float> values) const {
243  SPIEL_CHECK_GE(player, 0);
244  SPIEL_CHECK_LT(player, num_players_);
245 
246  std::fill(values.begin(), values.end(), 0.0);
247  SPIEL_CHECK_EQ(values.size(), kInformationStateTensorSize);
248  if (phase_ == Phase::kPassDir || phase_ == Phase::kDeal) return;
249  auto ptr = values.begin();
250  // Pass direction
251  ptr[static_cast<int>(pass_dir_)] = 1;
252  ptr += kNumPlayers;
253  // Dealt hand
254  for (int i = 0; i < kNumCards; ++i)
255  if (initial_deal_[i] == player) ptr[i] = 1;
256  ptr += kNumCards;
257  // Passed cards
258  for (int card : passed_cards_[player]) ptr[card] = 1;
259  ptr += kNumCards;
260  // Received cards
261  // Cards are not received until all players have completed passing.
262  // West is the last player to pass.
263  if (passed_cards_[static_cast<int>(kWest)].size() == kNumCardsInPass) {
264  int passer =
265  (player - static_cast<int>(pass_dir_) + kNumPlayers) % kNumPlayers;
266  for (int card : passed_cards_[passer]) ptr[card] = 1;
267  }
268  ptr += kNumCards;
269  // Current hand
270  for (int i = 0; i < kNumCards; ++i)
271  if (holder_[i] == player) ptr[i] = 1;
272  ptr += kNumCards;
273  // Point totals
274  for (int i = 0; i < kNumPlayers; ++i) {
275  // Use thermometer representation instead of one-hot for point totals.
276  // Players can have negative points so we need to offset
277  for (int j = 0; j < points_[i] + std::abs(kPointsForJD); ++j) ptr[j] = 1;
278  ptr += kMaxScore;
279  }
280  // History of tricks, presented in the format: N E S W N E S
281  int current_trick = std::min(num_cards_played_ / kNumPlayers,
282  static_cast<int>(tricks_.size() - 1));
283  for (int i = 0; i < current_trick; ++i) {
284  Player leader = tricks_[i].Leader();
285  ptr += leader * kNumCards;
286  for (auto card : tricks_[i].Cards()) {
287  ptr[card] = 1;
288  ptr += kNumCards;
289  }
290  ptr += (kNumPlayers - leader - 1) * kNumCards;
291  }
292  Player leader = tricks_[current_trick].Leader();
293  if (leader != kInvalidPlayer) {
294  auto cards = tricks_[current_trick].Cards();
295  ptr += leader * kNumCards;
296  for (auto card : cards) {
297  ptr[card] = 1;
298  ptr += kNumCards;
299  }
300  }
301  // Current trick may contain less than four cards.
302  if (num_cards_played_ < kNumCards) {
303  ptr += (kNumPlayers - (num_cards_played_ % kNumPlayers)) * kNumCards;
304  }
305  // Move to the end of current trick.
306  ptr += (kNumPlayers - std::max(leader, 0) - 1) * kNumCards;
307  // Skip over unplayed tricks.
308  ptr += (kNumTricks - current_trick - 1) * kTrickTensorSize;
309  SPIEL_CHECK_EQ(ptr, values.end());
310 }

358 std::vector<Action> HeartsState::PlayLegalActions() const {
359  std::vector<Action> legal_actions;
360  legal_actions.reserve(kNumTricks - num_cards_played_ / kNumPlayers);
361 
362  // Check if we can follow suit.
363  if (num_cards_played_ % kNumPlayers != 0) {
364  auto suit = CurrentTrick().LedSuit();
365  for (int rank = 0; rank < kNumCardsPerSuit; ++rank) {
366  if (holder_[Card(suit, rank)] == current_player_) {
367  legal_actions.push_back(Card(suit, rank));
368  }
369  }
370  }
371  if (!legal_actions.empty()) return legal_actions;
372 
373  // Special rules apply to the first card played.
374  // Must play 2C unless option to play any club is enabled.
375  if (num_cards_played_ == 0) {
376  SPIEL_CHECK_TRUE(holder_[Card(Suit::kClubs, 0)] == current_player_);
377  legal_actions.push_back(Card(Suit::kClubs, 0));
378  if (can_lead_any_club_) {
379  for (int rank = 1; rank < kNumCardsPerSuit; ++rank) {
380  if (holder_[Card(Suit::kClubs, rank)] == current_player_) {
381  legal_actions.push_back(Card(Suit::kClubs, rank));
382  }
383  }
384  }
385  return legal_actions;
386  }
387 
388  // Special rules apply to the first trick.
389  // By default, cannot play hearts or QS on first trick.
390  if (no_pts_on_first_trick_ && num_cards_played_ < kNumPlayers) {
391  for (int card = 0; card < kNumCards; ++card) {
392  if (holder_[card] == current_player_ && card != Card(Suit::kSpades, 10) &&
393  CardSuit(card) != Suit::kHearts) {
394  legal_actions.push_back(card);
395  }
396  }
397  }
398  if (!legal_actions.empty()) return legal_actions;
399 
400  // Player must lead. By default, cannot lead hearts until broken.
401  if (num_cards_played_ % kNumPlayers == 0) {
402  for (int card = 0; card < kNumCards; ++card) {
403  if (holder_[card] == current_player_) {
404  if (CardSuit(card) != Suit::kHearts || hearts_broken_) {
405  legal_actions.push_back(card);
406  }
407  }
408  }
409  // Don't force player into leading the QS when hearts have not
410  // been broken.
411  if (can_lead_hearts_instead_of_qs_ && legal_actions.size() == 1 &&
412  legal_actions[0] == Card(Suit::kSpades, 10)) {
413  legal_actions.pop_back();
414  }
415  }
416  if (!legal_actions.empty()) return legal_actions;
417 
418  // Otherwise, we can play any of our cards.
419  for (int card = 0; card < kNumCards; ++card) {
420  if (holder_[card] == current_player_) legal_actions.push_back(card);
421  }
422  return legal_actions;
423 }

425 std::vector<std::pair<Action, double>> HeartsState::ChanceOutcomes() const {
426  std::vector<std::pair<Action, double>> outcomes;
427  if (history_.empty()) {
428  outcomes.reserve(kNumPlayers);
429  const double p = 1.0 / kNumPlayers;
430  for (int dir = 0; dir < kNumPlayers; ++dir) {
431  outcomes.emplace_back(dir, p);
432  }
433  return outcomes;
434  }
435  int num_cards_remaining = kNumCards - num_cards_dealt_;
436  outcomes.reserve(num_cards_remaining);
437  const double p = 1.0 / num_cards_remaining;
438  for (int card = 0; card < kNumCards; ++card) {
439  if (!holder_[card].has_value()) outcomes.emplace_back(card, p);
440  }
441  return outcomes;
442 }

466 void HeartsState::ApplyDealAction(int card) {
467  holder_[card] = num_cards_dealt_ % kNumPlayers;
468  ++num_cards_dealt_;
469  if (num_cards_dealt_ == kNumCards) {
470  // Preserve the initial deal for easy retrieval
471  initial_deal_ = holder_;
472  if (pass_dir_ == PassDir::kNoPass) {
473  phase_ = Phase::kPlay;
474  // Play starts with the holder of the 2C
475  current_player_ = holder_[Card(Suit::kClubs, 0)].value();
476  } else {
477  phase_ = Phase::kPass;
478  current_player_ = 0;
479  }
480  }
481 }

483 void HeartsState::ApplyPassAction(int card) {
484  passed_cards_[current_player_].push_back(card);
485  holder_[card] = absl::nullopt;
486  if (passed_cards_[current_player_].size() % kNumCardsInPass == 0)
487  ++current_player_;
488  if (current_player_ == kNumPlayers) {
489  // Players have completed passing. Now let's distribute the passed cards.
490  for (int player = 0; player < kNumPlayers; ++player) {
491  for (int card : passed_cards_[player]) {
492  holder_[card] = (player + static_cast<int>(pass_dir_)) % kNumPlayers;
493  }
494  }
495  phase_ = Phase::kPlay;
496  // Play starts with the holder of the 2C
497  current_player_ = holder_[Card(Suit::kClubs, 0)].value();
498  }
499 }

501 void HeartsState::ApplyPlayAction(int card) {
502  SPIEL_CHECK_TRUE(holder_[card] == current_player_);
503  holder_[card] = absl::nullopt;
504  if (num_cards_played_ % kNumPlayers == 0) {
505  CurrentTrick() = Trick(current_player_, card, jd_bonus_);
506  } else {
507  CurrentTrick().Play(current_player_, card);
508  }
509  // Check if action breaks hearts.
510  if (CardSuit(card) == Suit::kHearts) hearts_broken_ = true;
511  if (qs_breaks_hearts_ && card == Card(Suit::kSpades, 10))
512  hearts_broken_ = true;
513  // Update player and point totals.
514  Trick current_trick = CurrentTrick();
515  ++num_cards_played_;
516  if (num_cards_played_ % kNumPlayers == 0) {
517  current_player_ = current_trick.Winner();
518  points_[current_player_] += current_trick.Points();
519  } else {
520  current_player_ = (current_player_ + 1) % kNumPlayers;
521  }
522  if (num_cards_played_ == kNumCards) {
523  phase_ = Phase::kGameOver;
524  current_player_ = kTerminalPlayerId;
525  ComputeScore();
526  }
527 }

534 void HeartsState::ComputeScore() {
535  SPIEL_CHECK_TRUE(IsTerminal());
536  // Did anyone shoot the moon?
537  Player potential_shooter = kInvalidPlayer;
538  bool moon_shot = true;
539  for (int i = 0; i < kNumTricks; ++i) {
540  int points = tricks_[i].Points();
541  // JD not required to shoot the moon.
542  if (points != 0 && points != kPointsForJD) {
543  // This trick must be taken by the shooter.
544  if (potential_shooter == kInvalidPlayer) {
545  potential_shooter = tricks_[i].Winner();
546  } else if (potential_shooter != tricks_[i].Winner()) {
547  moon_shot = false;
548  break;
549  }
550  }
551  }
552  // Shooting the moon sets the shooter's points to 0, and adds 26 pts to each
553  // opponent's score.
554  if (moon_shot) {
555  for (int i = 0; i < kNumPlayers; ++i) {
556  points_[i] += (i == potential_shooter) ? -kTotalPositivePoints
557  : kTotalPositivePoints;
558  }
559  }
560  // Did anyone avoid taking any tricks?
561  if (avoid_all_tricks_bonus_ && !moon_shot) {
562  std::vector<int> tricks_taken(kNumPlayers, 0);
563  for (int i = 0; i < kNumTricks; ++i) {
564  tricks_taken[tricks_[i].Winner()] += 1;
565  }

572 // Hearts is a trick-avoidance game in which the goal is to accumulate the
573 // fewest number of points. Because RL algorithms are designed to maximize
574 // reward, returns are calculated by subtracting the in-game points from an
575 // upper bound.
576 std::vector<double> HeartsState::Returns() const {

578  return std::vector<double>(kNumPlayers, 0.0);
579  }

582  returns[i] = kTotalPositivePoints - returns[i];

600 bool HeartsState::KnowsLocation(Player player, int card) const {

602  dealt = initial_deal_[card] == player;

613 // Does not account for void suit information exposed by other players during
614 // the play phase
615 std::unique_ptr<State> HeartsState::ResampleFromInfostate(
616  int player_id, std::function<double()> rng) const {
617  std::unique_ptr<State> clone = game_->NewInitialState();
618  Action pass_dir = static_cast<int>(pass_dir_);
619  clone->ApplyAction(pass_dir);
620 
621  // start by gathering all public and private info known to player_id to
622  // simplify the logic for applying deal / pass actions
623  // first thing we know is the player's entire hand
624  std::vector<int> initial_hand;
625  for (int card = 0; card < kNumCards; card++) {
626  if (initial_deal_[card] == player_id) initial_hand.push_back(card);
627  }
628 
629  // collect cards that have been revealed through the play phase
630  std::vector<std::vector<int>> play_known(kNumPlayers);
631  if (phase_ == Phase::kPlay) {
632  for (int card = 0; card < kNumCards; card++) {
633  absl::optional<Player> p = Played(card);
634  if (p && *p != player_id) {
635  play_known[*p].push_back(card);
636  }
637  }
638  }
639 
640  // the two of clubs is also known once a player has played first
641  absl::optional<Player> two_clubs_holder = holder_[Card(Suit::kClubs, 0)];
642  if (phase_ == Phase::kPlay && two_clubs_holder) {
643  play_known[*two_clubs_holder].push_back(Card(Suit::kClubs, 0));
644  }
645 
646  // set up pass cards greedily using known cards so that passes are
647  // consistent
648  // this shouldn't affect the distribution of resampled states much because
649  // we have no way to model unobserved opponent pass actions anyway
650  std::vector<std::vector<int>> pass_actions(kNumPlayers);
651  for (Player p = 0; p < kNumPlayers; p++) {
652  for (int pass_num = 0; pass_num < passed_cards_[p].size(); pass_num++) {
653  if (p == player_id) {
654  pass_actions[p].push_back(passed_cards_[p][pass_num]);
655  } else {
656  Player pass_to = (p + pass_dir) % kNumPlayers;
657  // once the play phase has started, player_id knows the cards that were
658  // passed to them
659  if (phase_ == Phase::kPlay && pass_to == player_id) {
660  pass_actions[p].push_back(passed_cards_[p][pass_num]);
661  } else if (pass_num < play_known[pass_to].size()) {
662  pass_actions[p].push_back(play_known[pass_to][pass_num]);
663  }
664  }
665  }
666  }
667 
668  // at this point we have all the information we need about which card
669  // locations are known to player_id, so we can start applying deal actions
670  Player deal_to, pass_to, recv_from;
671  int card_num;
672  std::vector<bool> dealt(kNumCards, false);
673  std::vector<int> known_dealt_counter(kNumPlayers, 0);
674  for (int num_dealt = 0; num_dealt < kNumCards; num_dealt++) {
675  card_num = num_dealt / kNumPlayers;
676  deal_to = num_dealt % kNumPlayers;
677  pass_to = (deal_to + pass_dir) % kNumPlayers;
678  recv_from = (deal_to + kNumPlayers - pass_dir) % kNumPlayers;
679  Action action = kInvalidAction;
680  // deal out the pass moves first so those constraints are satisfied
681  if (card_num < pass_actions[deal_to].size()) {
682  action = pass_actions[deal_to][card_num];
683  } else {
684  // now try to find any cards that deal_to has shown they have that
685  // haven't already been allocated as a pass action for recv_from and have
686  // not already been dealt
687  auto& known = (deal_to == player_id) ? initial_hand : play_known[deal_to];
688  while ((action == kInvalidAction || dealt[action]) &&
689  known_dealt_counter[deal_to] < known.size()) {
690  action = known[known_dealt_counter[deal_to]];
691  auto it = std::find(pass_actions[recv_from].begin(),
692  pass_actions[recv_from].end(), action);
693  if (it != pass_actions[recv_from].end()) action = kInvalidAction;
694  known_dealt_counter[deal_to]++;
695  }
696  }
697 
698  // all known card constraints for to_deal have been satisfied, so we can
699  // deal them a random card that does not violate other player constraints
700  while (action == kInvalidAction) {
701  Action candidate = SampleAction(clone->ChanceOutcomes(), rng()).first;
702  if (!KnowsLocation(player_id, candidate)) {
703  action = candidate;
704  // we can also use this card as a pass action later because its
705  // location is unknown
706  if (pass_actions[deal_to].size() < passed_cards_[deal_to].size()) {
707  pass_actions[deal_to].push_back(action);
708  }
709  }
710  }
711 
712  clone->ApplyAction(action);
713  dealt[action] = true;
714  }
715 
716  // now handle the pass phase
717  if (pass_dir_ != PassDir::kNoPass) {
718  for (Player to_move = 0; to_move < kNumPlayers; to_move++) {
719  SPIEL_CHECK_EQ(pass_actions[to_move].size(),
720  passed_cards_[to_move].size());
721  pass_to = (to_move + pass_dir) % kNumPlayers;
722  for (int passes = 0; passes < passed_cards_[to_move].size(); passes++) {
723  Action action = kInvalidAction;
724  if (to_move == player_id || pass_to == player_id) {
725  // player_id knows exactly which cards were passed by player_id and
726  // the player who passed cards to player_id
727  action = passed_cards_[to_move][passes];
728  } else {
729  action = pass_actions[to_move][passes];
730  }
731  clone->ApplyAction(action);
732  }
733  }
734  }
735 
736  // given that we should now have a state consistent with the public actions
737  // and player_id's private cards, we can just copy the action sequence in
738  // the play phase
739  int play_start_index = kNumCards + 1;
740  if (pass_dir_ != PassDir::kNoPass)
741  play_start_index += kNumPlayers * kNumCardsInPass;
742  for (size_t i = play_start_index; i < history_.size(); i++) {
743  clone->ApplyAction(history_.at(i).action);
744  }
745 
746  SPIEL_CHECK_EQ(FullHistory().size(), clone->FullHistory().size());
747  SPIEL_CHECK_EQ(InformationStateString(player_id),
748  clone->InformationStateString(player_id));
749  return clone;
750 }
