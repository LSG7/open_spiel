23class MutableVote(object):
24 """A mutable vote annotated with the current preferred alternative.
25
26 This is used to keep track of votes and which index (into the preference list)
27 is currently active, i.e. the most preferred. When votes get used to determine
28 winners or elimintations, some of these votes get "transfered" down to the
29 next alternative. To transfer the vote, the index here is incremented to
30 indicate that this vote is now representing a vote for the next highest
31 alternative.
32 """
33
34 def __init__(self, idx: int, weight: int, vote: List[base.AlternativeId]):
35 self.idx = idx
36 self.weight = weight
37 self.vote = vote

40class STVVoting(base.AbstractVotingMethod):
41 """Implements STV method."""

43 def __init__(
44 self, num_winners: Union[int, None] = None, verbose: bool = False
45 ):
46 """Construct an instance of STV with the specified number of winners.
47
48 Args:
49 num_winners: number of winners. Should be less than number of
50 alternatives (m). If not specified, defaults to int(m/2).
51 verbose: whether or not to print debug information as STV is running.
52 """

59 def _is_still_active(
60 self,
61 alternative: base.AlternativeId,
62 winners: List[base.AlternativeId],
63 losers: List[base.AlternativeId],
64 ) -> bool:
65 """Returns whether the alternative is still in the running."""
66 return alternative not in winners and alternative not in losers
67
68 def _next_idx_in_the_running(
69 self,
70 mutable_vote: MutableVote,
71 winners: List[base.AlternativeId],
72 losers: List[base.AlternativeId],
73 ) -> int:
74 """"Returns the next index in the list that is still in the running."""
75 new_idx = mutable_vote.idx + 1
76 while (new_idx < len(mutable_vote.vote) and
77 not self._is_still_active(mutable_vote.vote[new_idx], winners,
78 losers)):
79 new_idx += 1
80 return new_idx

82 def _initial_scores_for_round(
83 self,
84 profile: base.PreferenceProfile,
85 winners: List[base.AlternativeId],
86 losers: List[base.AlternativeId],
87 ) -> Dict[base.AlternativeId, float]:
88 """Returns round's initial scores for alternatives still in the running."""
89 alt_scores = {}
90 for alt in profile.alternatives:
91 if self._is_still_active(alt, winners, losers):
92 alt_scores[alt] = 0
93 return alt_scores

95 def _remove_winning_votes(
96 self,
97 winning_alt: base.AlternativeId,
98 num_to_remove: int,
99 all_votes: List[MutableVote],
100 ):
101 while num_to_remove > 0:
102 for mutable_vote in all_votes:
103 if (mutable_vote.idx < len(mutable_vote.vote) and
104 mutable_vote.vote[mutable_vote.idx] == winning_alt):
105 removing_now = min(mutable_vote.weight, num_to_remove)
106 mutable_vote.weight -= removing_now
107 num_to_remove -= removing_now
108 if num_to_remove == 0:
109 break

111 def run_election(self, profile: base.PreferenceProfile) -> base.RankOutcome:
112 assert self.is_valid_profile(profile)
113 winners = []
114 losers = []
115 winner_scores = []
116 loser_scores = []
117 votes = profile.votes
118 total_votes = profile.total_weight()
119 m = profile.num_alternatives()
120 num_winners = self._num_winners
121 if num_winners is None:
122 num_winners = int(m/2)
123 if self._verbose:
124 print("Warning: number of winners not specified." +
125 f"Choosing {num_winners}")
126 assert num_winners < m
127 quota = int(total_votes / float(num_winners + 1) + 1)
128 # Table holds a list of the IndexAndWeightedVote. The index corresponds to
129 # the current alternative that this vote is representing. They all start at
130 # 0 at the start, corresponding to their highest preference, and they get
131 # incremented as they become used up.
132 all_votes: List[MutableVote] = []
133 for vote in votes:
134 all_votes.append(MutableVote(idx=0, weight=vote.weight, vote=vote.vote))
135 while len(winners) + len(losers) < m:
136 scores = self._initial_scores_for_round(profile, winners, losers)
137 for mutable_vote in all_votes:
138 if (mutable_vote.idx < len(mutable_vote.vote) and
139 mutable_vote.weight > 0):
140 alt = mutable_vote.vote[mutable_vote.idx]
141 scores[alt] += mutable_vote.weight
142 sorted_scores = sorted(scores.items(), key=lambda item: item[1],
143 reverse=True)
144 best_score = sorted_scores[0][1]
145 if best_score >= quota:
146 # Quota reached. A candidate wins!
147 if self._verbose:
148 print(f"Quota {quota} reached. Candidate {sorted_scores[0][0]} wins!")
149 winning_alt = sorted_scores[0][0]
150 winners.append(winning_alt)
151 winner_scores.append(best_score)
152 surplus = sorted_scores[0][1] - quota
153 # Remove votes that contributed to the winner, up to the quota.
154 self._remove_winning_votes(winning_alt, quota, all_votes)
155 # Then, convert all the rest.
156 num_converted = 0
157 for mutable_vote in all_votes:
158 if (mutable_vote.idx < len(mutable_vote.vote) and
159 mutable_vote.vote[mutable_vote.idx] == winning_alt and
160 mutable_vote.weight > 0):
161 # find the next one in the list still in the running.
162 new_idx = self._next_idx_in_the_running(mutable_vote, winners,
163 losers)
164 mutable_vote.idx = new_idx
165 num_converted += mutable_vote.weight
166 assert num_converted == surplus
167 else:
168 # No winner, eliminate the bottom candidate.
169 eliminated_alt = sorted_scores[-1][0]
170 eliminated_score = sorted_scores[-1][1]
171 if self._verbose:
172 print(f"No winner. Quota = {quota}. Eliminating candidate: " +
173 f"{eliminated_alt} with score: {eliminated_score}")
174 elim_count = sorted_scores[-1][1]
175 losers.insert(0, eliminated_alt)
176 loser_scores.insert(0, eliminated_score)
177 # All of the votes with this alternative as the top is converted.
178 votes_counted = 0
179 for mutable_vote in all_votes:
180 if (mutable_vote.idx < len(mutable_vote.vote) and
181 mutable_vote.vote[mutable_vote.idx] == eliminated_alt and
182 mutable_vote.weight > 0):
183 # find the next one in the list still in the running.
184 new_idx = self._next_idx_in_the_running(mutable_vote, winners,
185 losers)
186 mutable_vote.idx = new_idx
187 votes_counted += mutable_vote.weight
188 assert votes_counted == elim_count
189 ranking = winners + losers
190 scores = []
191 win_score_base = profile.num_alternatives() * 2
192 lose_score_base = profile.num_alternatives()
193 for winner_score in winner_scores:
194 scores.append(float(str(win_score_base) + "." + str(winner_score)))
195 win_score_base -= 1
196 for loser_score in loser_scores:
197 scores.append(float(str(lose_score_base) + "." + str(loser_score)))
198 lose_score_base -= 1
199 outcome = base.RankOutcome(rankings=ranking, scores=scores)
200 return outcome
