40  acpc_action.size = size;

51  default:
52  acpc_action.type = project_acpc_server::a_invalid;
53  break;

80 // We compare the values for all the fields. For arrays, note that only the
81 // first `numPlayers` or `numRounds` values are meaningful, the rest being
82 // non-initialized.
83 bool ACPCGame::operator==(const ACPCGame &other) const {
84  // See project_acpc_server/game.h:42. 12 fields total.
85  // int32_t stack[ MAX_PLAYERS ];
86  // int32_t blind[ MAX_PLAYERS ];
87  // int32_t raiseSize[ MAX_ROUNDS ];
88  // enum BettingType bettingType;
89  // uint8_t numPlayers;
90  // uint8_t numRounds;
91  // uint8_t firstPlayer[ MAX_ROUNDS ];
92  // uint8_t maxRaises[ MAX_ROUNDS ];
93  // uint8_t numSuits;
94  // uint8_t numRanks;
95  // uint8_t numHoleCards;
96  // uint8_t numBoardCards[ MAX_ROUNDS ];
97  const project_acpc_server::Game *first = &acpc_game_;
98  const project_acpc_server::Game &second = other.Game();
99  const int num_players = first->numPlayers;
100  const int num_rounds = first->numRounds;
101 
102  // We check for `raiseSize` only for limit betting.
103  if (first->bettingType != second.bettingType) {
104  return false;
105  }
106  if (first->bettingType == project_acpc_server::limitBetting) {
107  if (!std::equal(first->raiseSize, first->raiseSize + num_rounds,
108  second.raiseSize)) {
109  return false;
110  }
111  }
112  return ( // new line
113  first->numPlayers == second.numPlayers &&
114  first->numRounds == second.numRounds &&
115  std::equal(first->stack, first->stack + num_players, second.stack) &&
116  std::equal(first->blind, first->blind + num_players, second.blind) &&
117  std::equal(first->firstPlayer, first->firstPlayer + num_rounds,
118  second.firstPlayer) &&
119  std::equal(first->maxRaises, first->maxRaises + num_rounds,
120  second.maxRaises) &&
121  first->numSuits == second.numSuits &&
122  first->numRanks == second.numRanks &&
123  first->numHoleCards == second.numHoleCards &&
124  std::equal(first->numBoardCards, first->numBoardCards + num_rounds,
125  second.numBoardCards));
126 }

279 void ACPCState::SetHoleAndBoardCards(uint8_t holeCards[10][3],
280  uint8_t boardCards[7],
281  uint8_t nbHoleCards[10],
282  uint8_t nbBoardCards) {
283  for (int p = 0; p < game_->GetNbPlayers(); ++p) {
284  SPIEL_CHECK_EQ(nbHoleCards[p], game_->GetNbHoleCardsRequired());
285  for (int c = 0; c < nbHoleCards[p]; ++c) {
286  acpcState_.holeCards[p][c] = holeCards[p][c];
287  }
288  }
289 
290  SPIEL_CHECK_EQ(nbBoardCards, game_->GetNbBoardCardsRequired(GetRound()));
291  for (int c = 0; c < nbBoardCards; ++c) {
292  acpcState_.boardCards[c] = boardCards[c];
293  }
