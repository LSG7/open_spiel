37 void TestGameTree() {
38  std::vector<std::string> game_names = {"leduc_poker", "kuhn_poker",
39  "liars_dice"};
40  absl::flat_hash_map<std::string, int> num_histories = {
41  // Not sure if these are correct. Chosen to make test pass. They seem to
42  // have the right order of magnitude.
43  {"kuhn_poker", 58},
44  {"leduc_poker", 9457},
45  {"liars_dice", 294883}};
46 
47  for (const auto& game_name : game_names) {
48  std::shared_ptr<const Game> game = LoadGame(game_name);
49  for (Player player_id : {Player{0}, Player{1}}) {
50  HistoryTree tree(game->NewInitialState(), player_id);
51  if (tree.NumHistories() != num_histories[game_name]) {
52  // TODO(b/126764761): Replace calls to SpielFatalError with more
53  // appropriate test macros once they support logging.
54  SpielFatalError(absl::StrCat(
55  "In the game ", game_name,
56  ", tree has wrong number of nodes: ", tree.NumHistories(), "but ",
57  num_histories[game_name], "nodes were expected."));
58  }
59 
60  // Check that the root is not null.
61  if (tree.Root() == nullptr) {
62  SpielFatalError("Root of HistoryTree is null for game: " + game_name);
63  }
64  for (const std::string& history : tree.GetHistories()) {
65  HistoryNode* node = tree.GetByHistory(history);
66  if (node == nullptr) {
67  SpielFatalError(absl::StrCat("node is null for history: ", history,
68  " in game: ", game_name));
69  }
70  if (node->GetState() == nullptr) {
71  SpielFatalError(absl::StrCat("state is null for history: ", history,
72  " in game: ", game_name));
73  }
74  if (node->GetState()->HistoryString() != node->GetHistory()) {
75  SpielFatalError(
76  "history generated by state does not match history"
77  " stored in HistoryNode.");
78  }
79  if (history != node->GetHistory()) {
80  SpielFatalError(
81  "history key does not match history stored in "
82  "HistoryNode.");
83  }
84  if (node->GetType() != StateType::kTerminal) {
85  std::vector<Action> legal_actions = node->GetState()->LegalActions();
86  std::vector<Action> child_actions = node->GetChildActions();
87  if (legal_actions.size() != child_actions.size()) {
88  SpielFatalError(absl::StrCat(
89  "For state ", history, ", child actions has a different size (",
90  child_actions.size(), ") than legal actions (",
91  legal_actions.size(), ")."));
92  }
93  for (int i = 0; i < legal_actions.size(); ++i) {
94  if (legal_actions[i] != child_actions[i]) {
95  SpielFatalError(absl::StrCat(
96  "legal_actions[i] != child_actions[i]: ", legal_actions[i],
97  " != ", child_actions[i]));
98  }
99  }
100  }
101 
102  if (node->GetType() != StateType::kTerminal &&
103  node->NumChildren() != node->GetState()->LegalActions().size()) {
104  SpielFatalError(absl::StrCat(
105  "number of child nodes does not match number of legal"
106  " actions in history: ",
107  history, " in game: ", game_name));
108  }
109  if (node->GetType() == StateType::kDecision &&
110  node->GetState()->CurrentPlayer() != player_id) {
111  if (node->GetInfoState() !=
112  node->GetState()->InformationStateString()) {
113  SpielFatalError(absl::StrCat(
114  "infostate generated by state does not match ",
115  "infostate stored in HistoryNode for history: ", history,
116  "in game: ", game_name));
117  }
118  } else if (node->GetType() == StateType::kChance) {
119  if (node->GetInfoState() != HistoryNode::kChanceNodeInfostateString) {
120  SpielFatalError(absl::StrCat(
121  "Chance node's infostate string not properly set for history: ",
122  history, " in game: ", game_name));
123  }
124  } else if (node->GetType() == StateType::kTerminal) {
125  if (node->GetInfoState() !=
126  HistoryNode::kTerminalNodeInfostateString) {
127  SpielFatalError(absl::StrCat(
128  "Chance node's infostate string not properly set for history: ",
129  history, " in game: ", game_name));
130  }
131  } else {
132  if (node->GetInfoState() !=
133  node->GetState()->InformationStateString(player_id)) {
134  SpielFatalError(absl::StrCat(
135  "infostate generated by state does not match ",
136  "infostate stored in HistoryNode for history: ", history,
137  "in game: ", game_name));
138  }
139  }
140  }
141  }
142  }

145 void TestInfoSetsHaveRightNumberOfGameStates() {
146  std::shared_ptr<const Game> game = LoadGame("kuhn_poker");
147  std::unique_ptr<State> state = game->NewInitialState();
148  TabularPolicy policy = GetUniformPolicy(*game);
149  auto best_responder = Player{0};
150  HistoryTree tree(game->NewInitialState(), best_responder);
151  auto infosets =
152  GetAllInfoSets(game->NewInitialState(), best_responder, &policy, &tree);
153  for (const auto& kv : infosets) {
154  const std::string& infostate = kv.first;
155  const std::vector<std::pair<HistoryNode*, double>>& histories = kv.second;
156  int num_histories = histories.size();
157  // The infostate represented by the empty string corresponds to the root
158  // infoset, which only has one history associated with it.
159  if (infostate.empty()) {
160  if (num_histories != 1) {
161  SpielFatalError(
162  absl::StrCat("Wrong number of histories in infoset at root;"
163  " expected 1, but found ",
164  num_histories));
165  }
166  } else {
167  if (num_histories != 2) {
168  SpielFatalError(
169  absl::StrCat("Wrong number of histories in infoset at infostate ",
170  infostate, " expected 2, but found ", num_histories));
171  }
172  }

176 void TestGetAllInfoSetsMatchesInfoStates() {
177  std::shared_ptr<const Game> game = LoadGame("kuhn_poker");
178  std::unique_ptr<State> state = game->NewInitialState();
179  TabularPolicy policy = GetUniformPolicy(*game);
180  for (const auto& best_responder : {Player{0}, Player{1}}) {
181  HistoryTree tree(game->NewInitialState(), best_responder);
182  auto infosets =
183  GetAllInfoSets(game->NewInitialState(), best_responder, &policy, &tree);
184  for (const auto& kv : infosets) {
185  const std::string& infostate = kv.first;
186  for (const auto& state_and_prob : kv.second) {
187  HistoryNode* node = state_and_prob.first;
188  if (node == nullptr) SpielFatalError("Node is null.");
189  std::string node_infostate = node->GetInfoState();
190  if (infostate != node_infostate) {
191  SpielFatalError(
192  absl::StrCat("infostate key (", infostate, ") does not match ",
193  "infostate stored in node (", node_infostate, ")."));
194  }
195  State* node_state = node->GetState();
196  std::string state_infostate =
197  node_state->InformationStateString(best_responder);
198  if (node_infostate != state_infostate) {
199  SpielFatalError(
200  absl::StrCat("infostate stored in node (", node_infostate, ") ",
201  "does not match infostate calculated from state ",
202  "stored in node (", state_infostate, ")."));
203  }
204  if (node->GetType() == StateType::kDecision) {
205  if (node_state->CurrentPlayer() != best_responder) {
206  SpielFatalError(
207  absl::StrCat("CurrentPlayer for state stored in node (",
208  node_state->CurrentPlayer(), ") does not match ",
209  "best_responder (", best_responder, ")."));
210  }
211  } else if (node->GetType() == StateType::kDecision) {
212  if (node_state->CurrentPlayer() == best_responder) {
213  SpielFatalError(absl::StrCat(
214  "CurrentPlayer for state stored in node (",
215  node_state->CurrentPlayer(), ") matches best_responder (",
216  best_responder, ") but has type kDecision."));
217  }
218  }
219  std::vector<Action> child_actions_vector = node->GetChildActions();
220  absl::flat_hash_set<Action> child_actions(child_actions_vector.begin(),
221  child_actions_vector.end());
222  std::vector<Action> legal_actions_vector = node_state->LegalActions();
223  absl::node_hash_set<Action> legal_actions(legal_actions_vector.begin(),
224  legal_actions_vector.end());
225  for (const auto& child_action : child_actions) {
226  if (legal_actions.count(child_action) == 0) {
227  SpielFatalError("Child action found that's not a legal action.");
228  }
229  }
230  for (const auto& legal_action : node_state->LegalActions()) {
231  if (child_actions.count(legal_action) == 0) {
232  SpielFatalError("Legal action found that's not a child action.");
233  }
234  std::unique_ptr<State> child = node_state->Child(legal_action);
235  HistoryNode child_node = HistoryNode(Player{0}, std::move(child));
236  if (node->GetType() != StateType::kChance) {
237  Player child_player = child_node.GetState()->CurrentPlayer();
238  if (node_state->CurrentPlayer() == child_player) {
239  SpielFatalError(absl::StrCat(
240  "Child and parent have the same current player (",
241  child_player, ")."));
242  }
243  if (infostate == child_node.GetInfoState()) {
244  SpielFatalError(
245  absl::StrCat("Child and parent have the same infostate (",
246  infostate, ")."));
247  }
248  }
249  }
250  }
251  }
252  }

275 // This is a common test that we want to make. We want to validate the
276 // counter-factual probabilities produced by this implementation against the
277 // golden values produced by existing implementations.
278 // best_responder is the player from who's view the infostate strings are
279 // calculated from, and represents the player for whom we are calculating a
280 // best response as. It can be any value in the range [0, game.NumPlayers()).
281 void CheckCounterFactualProbs(
282  const Game& game, const TabularPolicy& policy,
283  const absl::flat_hash_map<std::string, double>& histories_and_probs,
284  Player best_responder) {
285  HistoryTree tree(game.NewInitialState(), best_responder);
286 
287  // Infosets maps infostate strings to a list of all histories that map to that
288  // same infostate, along with corresponding counter-factual reach
289  // probabilities. The counter-factual reach probability of a history is
290  // defined recursively:
291  // - At the root, the reach probability is 1.
292  // - At a chance node, you multiply the parent's reach probability by the
293  // probability of having that chance outcome.
294  // - At a decision node, if the current player is the one making the decision,
295  // you multiply the reach probability by 1.
296  // - If another player is making a decision, you multiply the parent's reach
297  // probability by the probability that player makes that decision (taken
298  // here from their policy).
299  // Infostate strings here are assumed to be those that are returned from
300  // open_spiel::State::InformationState(best_responder), which are
301  // equivalent to those returned by HistoryNode::GetInfoState.
302  absl::flat_hash_map<std::string, std::vector<std::pair<HistoryNode*, double>>>
303  infosets = GetAllInfoSets(game.NewInitialState(), best_responder, &policy,
304  &tree);
305 
306  // We check this for every infoset in the game.
307  for (const auto& infoset : infosets) {
308  for (const auto& state_and_prob : infoset.second) {
309  HistoryNode* node = state_and_prob.first;
310  // We only check for nodes where the best responder is playing. This is
311  // because the counter-factual probability calculations assign a
312  // probability of 1. to all of the best responder's actions, so by
313  // checking the nodes where the best responder plays, we remove spurious
314  // failures (as the probability would be wrong at a different decision
315  // node iff the probability is wrong at a decision node where the best
316  // responder is playing).
317  if (node->GetState()->CurrentPlayer() != best_responder) continue;
318  double prob = state_and_prob.second;
319  auto it = histories_and_probs.find(node->GetHistory());
320  if (it == histories_and_probs.end())
321  SpielFatalError(absl::StrCat("Missing history: ", node->GetHistory()));
322  SPIEL_CHECK_FLOAT_EQ(prob, it->second);
323  }
324  }
325 }
