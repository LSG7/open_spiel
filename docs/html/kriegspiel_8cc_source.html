55 chess::ObservationTable ComputePrivateInfoTable(const chess::ChessBoard &board,
56  chess::Color color) {
57  const int board_size = board.BoardSize();
58  chess::ObservationTable observability_table{false};
59 
60  for (int8_t y = 0; y < board_size; ++y) {
61  for (int8_t x = 0; x < board_size; ++x) {
62  chess::Square sq{x, y};
63  if (board.IsFriendly({x, y}, color)) {
64  size_t index = chess::SquareToIndex(sq, board_size);
65  observability_table[index] = true;
66  }
67  }
68  }
69  return observability_table;
70 }

92  void WriteTensor(const State &observed_state, int player,
93  Allocator *allocator) const override {
94  auto &state =
95  open_spiel::down_cast<const KriegspielState &>(observed_state);
96  auto &game =
97  open_spiel::down_cast<const KriegspielGame &>(*state.GetGame());
98  SPIEL_CHECK_GE(player, 0);
99  SPIEL_CHECK_LT(player, game.NumPlayers());
100  chess::Color color = chess::PlayerToColor(player);
101 
102  if (iig_obs_type_.perfect_recall) {
103  SpielFatalError(
104  "KriegspielObserver: tensor with perfect recall not implemented.");
105  }
106 
107  if (iig_obs_type_.public_info) {
108  WritePublicInfoTensor(state, "public", allocator);
109  }
110  if (iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
111  std::string prefix = "private";
112  WritePrivateInfoTensor(state, player, prefix, allocator);
113  } else if (iig_obs_type_.private_info == PrivateInfoType::kAllPlayers) {
114  for (int i = 0; i < chess::NumPlayers(); ++i) {
115  std::string prefix = chess::ColorToString(color);
116  WritePrivateInfoTensor(state, i, prefix, allocator);
117  }
118  }
119  }

121  std::string StringFrom(const State &observed_state,
122  int player) const override {
123  auto &state =
124  open_spiel::down_cast<const KriegspielState &>(observed_state);
125  auto &game =
126  open_spiel::down_cast<const KriegspielGame &>(*state.GetGame());
127  SPIEL_CHECK_GE(player, 0);
128  SPIEL_CHECK_LT(player, game.NumPlayers());
129 
130  if (iig_obs_type_.perfect_recall) {
131  SpielFatalError(
132  "KriegspielObserver: string with perfect recall is unimplemented");
133  }
134 
135  if (iig_obs_type_.public_info &&
136  iig_obs_type_.private_info == PrivateInfoType::kSinglePlayer) {
137  // No observation before the first move
138  if (state.MoveMsgHistory().empty()) {
139  return std::string();
140  }
141 
142  // Write last umpire message
143  return state.last_umpire_msg_->ToString();
144  } else {
145  SpielFatalError(
146  "KriegspielObserver: string with imperfect recall is implemented only"
147  " for the (default) observation type.");
148  }
149  }

151  private:
152  void WritePieces(chess::Color color, chess::PieceType piece_type,
153  const chess::ChessBoard &board,
154  const chess::ObservationTable &observability_table,
155  const std::string &prefix, Allocator *allocator) const {
156  const std::string type_string =
157  color == chess::Color::kEmpty
158  ? "empty"
159  : chess::PieceTypeToString(
160  piece_type,
161  /*uppercase=*/color == chess::Color::kWhite);
162  const int board_size = board.BoardSize();
163 
164  auto out = allocator->Get(prefix + "_" + type_string + "_pieces",
165  {board_size, board_size});
166  for (int8_t y = 0; y < board_size; ++y) {
167  for (int8_t x = 0; x < board_size; ++x) {
168  const chess::Square square{x, y};
169  const chess::Piece &piece_on_board = board.at(square);
170  const bool write_square =
171  piece_on_board.color == color &&
172  piece_on_board.type == piece_type &&
173  observability_table[chess::SquareToIndex(square, board_size)];
174  out.at(x, y) = write_square ? 1.0f : 0.0f;
175  }
176  }
177  }

179  void WriteUnknownSquares(const chess::ChessBoard &board,

183  const int board_size = board.BoardSize();
184  auto out =
185  allocator->Get(prefix + "_unknown_squares", {board_size, board_size});

189  const bool write_square =

191  out.at(x, y) = write_square ? 0.0f : 1.0f;
192  }
193  }

196  void WriteMove(const chess::Move &move, const chess::ChessBoard &board,
197  const std::string &prefix, Allocator *allocator) const {
198  const int board_size = board.BoardSize();
199  auto from_out = allocator->Get(prefix + "_from", {board_size, board_size});
200  if (IsValid(move.from)) from_out.at(move.from.x, move.from.y) = 1;
201  auto to_out = allocator->Get(prefix + "_to", {board_size, board_size});
202  if (IsValid(move.to)) to_out.at(move.to.x, move.to.y) = 1;
203  // 5 is maximum because we can't promote to a pawn.
204  WriteScalar(static_cast<int>(move.promotion_type), 0, 5,
205  prefix + "_promotion", allocator);
206  }

208  void WriteUmpireMessage(const KriegspielUmpireMessage &msg,
209  const chess::ChessBoard &board,
210  const std::string &prefix,
211  Allocator *allocator) const {
212  int board_size = board.BoardSize();
213  WriteBinary(msg.illegal, prefix + "_illegal", allocator);
214  WriteScalar(static_cast<int>(msg.capture_type), 0, 2,
215  prefix + "_capture_type", allocator);
216  auto square_out =
217  allocator->Get(prefix + "_captured_square", {board_size, board_size});
218  if (IsValid(msg.square)) square_out.at(msg.square.x, msg.square.y) = 1;
219  WriteScalar(static_cast<int>(msg.check_types.first), 0, 5,
220  prefix + "_check_one", allocator);
221  WriteScalar(static_cast<int>(msg.check_types.second), 0, 5,
222  prefix + "_check_two", allocator);
223  WriteScalar(static_cast<int>(msg.to_move), 0, 2, prefix + "_to_move",
224  allocator);
225  WriteScalar(msg.pawn_tries, 0, 16, prefix + "_pawn_tries", allocator);
226  }

228  void WriteScalar(int val, int min, int max, const std::string &field_name,
229  Allocator *allocator) const {
230  SPIEL_DCHECK_LT(min, max);
231  SPIEL_DCHECK_GE(val, min);
232  SPIEL_DCHECK_LE(val, max);
233  auto out = allocator->Get(field_name, {max - min + 1});
234  out.at(val - min) = 1;
235  }

243  void WritePrivateInfoTensor(const KriegspielState &state, int player,
244  const std::string &prefix,
245  Allocator *allocator) const {
246  chess::Color color = chess::PlayerToColor(player);
247  chess::ObservationTable private_info_table =
248  ComputePrivateInfoTable(state.Board(), color);
249 
250  // Piece configuration.
251  for (const chess::PieceType &piece_type : chess::kPieceTypes) {
252  WritePieces(chess::Color::kWhite, piece_type, state.Board(),
253  private_info_table, prefix, allocator);
254  WritePieces(chess::Color::kBlack, piece_type, state.Board(),
255  private_info_table, prefix, allocator);
256  }
257  WritePieces(chess::Color::kEmpty, chess::PieceType::kEmpty, state.Board(),
258  private_info_table, prefix, allocator);
259  WriteUnknownSquares(state.Board(), private_info_table, prefix, allocator);
260 
261  // Castling rights.
262  WriteBinary(
263  state.Board().CastlingRight(color, chess::CastlingDirection::kLeft),
264  prefix + "_left_castling", allocator);
265  WriteBinary(
266  state.Board().CastlingRight(color, chess::CastlingDirection::kRight),
267  prefix + "_right_castling", allocator);
268 
269  // Write observer's last move
270  chess::Move last_move = {chess::kInvalidSquare, chess::kInvalidSquare,
271  chess::kEmptyPiece};
272 
273  for (auto move_msg = state.MoveMsgHistory().rbegin();
274  move_msg != state.MoveMsgHistory().rend(); ++move_msg) {
275  if (move_msg->first.piece.color == color) {
276  last_move = move_msg->first;
277  break;
278  }
279  }
280  WriteMove(last_move, state.Board(), prefix + "_last_move", allocator);
281  }

283  void WritePublicInfoTensor(const KriegspielState &state,
284  const std::string &prefix,
285  Allocator *allocator) const {
286  const auto entry = state.repetitions_.find(state.Board().HashValue());
287  SPIEL_CHECK_FALSE(entry == state.repetitions_.end());
288  int repetitions = entry->second;
289 
290  // Num repetitions for the current board.
291  WriteScalar(/*val=*/repetitions, /*min=*/1, /*max=*/3,
292  prefix + "_repetitions", allocator);
293 
294  // Side to play.
295  WriteScalar(/*val=*/ColorToPlayer(state.Board().ToPlay()),
296  /*min=*/0, /*max=*/1, prefix + "_side_to_play", allocator);
297 
298  // Irreversible move counter.
299  auto out = allocator->Get(prefix + "_irreversible_move_counter", {1});
300  out.at(0) = state.Board().IrreversibleMoveCounter() / 100.f;
301 
302  // Write last umpire message
303  if (state.last_umpire_msg_) {
304  WriteUmpireMessage(*state.last_umpire_msg_, state.Board(), prefix,
305  allocator);
306  } else {
307  WriteUmpireMessage(KriegspielUmpireMessage(), state.Board(), prefix,
308  allocator);
309  }
310  }

342 std::pair<KriegspielCheckType, KriegspielCheckType> GetCheckType(
343  const chess::ChessBoard &board) {
344  chess::Square king_sq =
345  board.find(chess::Piece{board.ToPlay(), chess::PieceType::kKing});
346 
347  std::pair<KriegspielCheckType, KriegspielCheckType> check_type_pair = {
348  KriegspielCheckType::kNoCheck, KriegspielCheckType::kNoCheck};
349 
350  board.GeneratePseudoLegalMoves(
351  [&king_sq, &check_type_pair, &board](const chess::Move &move) {
352  if (move.to != king_sq) {
353  return true;
354  }
355  KriegspielCheckType check_type;
356  if (move.piece.type == chess::PieceType::kKnight)
357  check_type = KriegspielCheckType::kKnight;
358  else if (move.from.x == move.to.x)
359  check_type = KriegspielCheckType::kFile;
360  else if (move.from.y == move.to.y)
361  check_type = KriegspielCheckType::kRank;
362  else if (chess::IsLongDiagonal(move.from, move.to, board.BoardSize()))
363  check_type = KriegspielCheckType::kLongDiagonal;
364  else
365  check_type = KriegspielCheckType::kShortDiagonal;
366 
367  if (check_type_pair.first != KriegspielCheckType::kNoCheck) {
368  // There cannot be more than two checks at the same time
369  check_type_pair.second = check_type;
370  return false;
371  } else {
372  check_type_pair.first = check_type;
373  }
374  return true;
375  },
376  board.ToPlay(), chess::PseudoLegalMoveSettings::kAcknowledgeEnemyPieces);
377 
378  return check_type_pair;
379 }

381 std::string KriegspielUmpireMessage::ToString() const {
382  if (illegal) {
383  return "Illegal move.";
384  }
385 
386  std::string msg;
387  bool put_comma = false;
388 
389  if (capture_type != KriegspielCaptureType::kNoCapture) {
390  msg += CaptureTypeToString(capture_type) + " at " +
391  chess::SquareToString(square) + " captured";
392  put_comma = true;
393  }
394  if (check_types.first != KriegspielCheckType::kNoCheck) {
395  if (put_comma) msg += ", ";
396  msg += CheckTypeToString(check_types.first) + " check";
397  put_comma = true;
398  }
399  if (check_types.second != KriegspielCheckType::kNoCheck) {
400  if (put_comma) msg += ", ";
401  msg += CheckTypeToString(check_types.second) + " check";
402  put_comma = true;
403  }
404  if (put_comma) msg += ", ";
405 
406  msg += chess::ColorToString(to_move) + "'s move";
407  if (pawn_tries > 0) {
408  msg += ", ";
409  msg += pawn_tries == 1 ? "1 pawn try"
410  : std::to_string(pawn_tries) + " pawn tries";
411  }
412  msg += ".";
413  return msg;
414 }

416 KriegspielUmpireMessage GetUmpireMessage(const chess::ChessBoard &chess_board,
417  const chess::Move &move) {
418  KriegspielUmpireMessage msg{};
419  if (!chess_board.IsMoveLegal(move)) {
420  // If the move is illegal, the player is notified about it and can play
421  // again
422  msg.illegal = true;
423  msg.to_move = chess_board.ToPlay();
424  return msg;
425  }
426  msg.illegal = false;
427 
428  chess::PieceType capture_type = chess_board.at(move.to).type;
429  switch (capture_type) {
430  case chess::PieceType::kEmpty:
431  msg.capture_type = KriegspielCaptureType::kNoCapture;
432  msg.square = chess::kInvalidSquare;
433  break;
434  case chess::PieceType::kPawn:
435  msg.capture_type = KriegspielCaptureType::kPawn;
436  msg.square = move.to;
437  break;
438  default:
439  msg.capture_type = KriegspielCaptureType::kPiece;
440  msg.square = move.to;
441  }
442 
443  // todo optimze when undo is optimized
444  chess::ChessBoard board_copy = chess_board;
445  board_copy.ApplyMove(move);
446 
447  msg.check_types = GetCheckType(board_copy);
448 
449  int pawnTries = 0;
450  board_copy.GenerateLegalPawnCaptures(
451  [&pawnTries](const chess::Move &move) {
452  pawnTries++;
453  return true;
454  },
455  board_copy.ToPlay());
456  msg.pawn_tries = pawnTries;
457  msg.to_move = board_copy.ToPlay();
458 
459  return msg;
460 }

462 bool GeneratesUmpireMessage(const chess::ChessBoard &chess_board,
463  const chess::Move &move,
464  const KriegspielUmpireMessage &orig_msg) {
465  if (!chess_board.IsMoveLegal(move)) {
466  // If the move is illegal, the player is notified about it and can play
467  // again
468  return orig_msg.illegal;
469  }
470 
471  chess::PieceType capture_type = chess_board.at(move.to).type;
472  switch (capture_type) {
473  case chess::PieceType::kEmpty:
474  if (orig_msg.capture_type != KriegspielCaptureType::kNoCapture) {
475  return false;
476  }
477  break;
478  case chess::PieceType::kPawn:
479  if (orig_msg.capture_type != KriegspielCaptureType::kPawn) {
480  return false;
481  }
482  break;
483  default:
484  if (orig_msg.capture_type != KriegspielCaptureType::kPiece) {
485  return false;
486  }
487  }
488 
489  // todo optimize when undo is optimized
490  chess::ChessBoard board_copy = chess_board;
491  board_copy.ApplyMove(move);
492 
493  if (orig_msg.check_types != GetCheckType(board_copy)) {
494  return false;
495  }
496 
497  int pawnTries = 0;
498  board_copy.GenerateLegalPawnCaptures(
499  [&pawnTries](const chess::Move &move) {
500  pawnTries++;
501  return true;
502  },
503  board_copy.ToPlay());
504  if (orig_msg.pawn_tries != pawnTries) {
505  return false;
506  }
507  if (orig_msg.to_move != board_copy.ToPlay()) {
508  return false;
509  }
510 
511  return true;
512 }

526 void KriegspielState::DoApplyAction(Action action) {
527  cached_legal_actions_.reset();
528 
529  chess::Move move = ActionToMove(action, Board());
530 
531  KriegspielUmpireMessage msg = GetUmpireMessage(Board(), move);
532 
533  move_msg_history_.emplace_back(move, msg);
534  last_umpire_msg_ = msg;
535 
536  if (msg.illegal) {
537  // If the move is illegal, the player is notified about it and can play
538  // again
539  illegal_tried_moves_.emplace_back(move);
540  cached_legal_actions_.reset();
541  return;
542  }
543 
544  Board().ApplyMove(move);
545  illegal_tried_moves_.clear();
546  ++repetitions_[current_board_.HashValue()];
547 }

549 void KriegspielState::MaybeGenerateLegalActions() const {
550  if (!cached_legal_actions_) {
551  cached_legal_actions_ = std::vector<Action>();
552  Board().GeneratePseudoLegalMoves(
553  [this](const chess::Move &move) -> bool {
554  bool is_illegal_tried = false;
555  for (const chess::Move &illegal_tried_move : illegal_tried_moves_) {
556  if (illegal_tried_move == move) {
557  is_illegal_tried = true;
558  break;
559  }
560  }
561  if (!is_illegal_tried) {
562  cached_legal_actions_->push_back(MoveToAction(move, BoardSize()));
563  }
564  return true;
565  },
566  Board().ToPlay(), chess::PseudoLegalMoveSettings::kBreachEnemyPieces);
567  absl::c_sort(*cached_legal_actions_);
568  }
569 }

605 void KriegspielState::UndoAction(Player player, Action action) {
606  // TODO: Make this fast by storing undo info in another stack.
607  SPIEL_CHECK_GE(move_msg_history_.size(), 1);
608  --repetitions_[current_board_.HashValue()];
609  move_msg_history_.pop_back();
610  history_.pop_back();

624 absl::optional<std::vector<double>> KriegspielState::MaybeFinalReturns() const {
625  if (!Board().HasSufficientMaterial()) {
626  return std::vector<double>{kDrawUtility, kDrawUtility};
627  }
628 
629  if (threefold_repetition_ && IsThreefoldRepetitionDraw()) {
630  return std::vector<double>{kDrawUtility, kDrawUtility};
631  }
632 
633  // Compute and cache the legal actions.
634  MaybeGenerateLegalActions();
635  SPIEL_CHECK_TRUE(cached_legal_actions_);
636  const bool have_legal_moves = !cached_legal_actions_->empty();
637 
638  // If we don't have legal moves we are stalemated or mated
639  if (!have_legal_moves) {
640  if (!Board().InCheck()) {
641  return std::vector<double>{kDrawUtility, kDrawUtility};
642  } else {
643  std::vector<double> returns(NumPlayers());
644  auto next_to_play = ColorToPlayer(Board().ToPlay());
645  returns[next_to_play] = kLossUtility;
646  returns[chess::OtherPlayer(next_to_play)] = kWinUtility;
647  return returns;
648  }
649  }
650 
651  if (rule_50_move_ && Board().IrreversibleMoveCounter() >= 50) {
652  return std::vector<double>{kDrawUtility, kDrawUtility};
653  }
654 
655  return absl::nullopt;
