36class SonnetLinear(nn.Module):
37 """A Sonnet linear module.
38
39 Always includes biases and only supports ReLU activations.
40 """
41
42 def __init__(self, in_size, out_size, activate_relu=True):
43 """Creates a Sonnet linear layer.
44
45 Args:
46 in_size: (int) number of inputs
47 out_size: (int) number of outputs
48 activate_relu: (bool) whether to include a ReLU activation layer
49 """
50 super(SonnetLinear, self).__init__()
51 self._activate_relu = activate_relu
52 stddev = 1.0 / math.sqrt(in_size)
53 mean = 0
54 lower = (-2 * stddev - mean) / stddev
55 upper = (2 * stddev - mean) / stddev
56 # Weight initialization inspired by Sonnet's Linear layer,
57 # which cites https://arxiv.org/abs/1502.03167v3
58 # pytorch default: initialized from
59 # uniform(-sqrt(1/in_features), sqrt(1/in_features))
60 self._weight = nn.Parameter(
61 torch.Tensor(
62 stats.truncnorm.rvs(
63 lower, upper, loc=mean, scale=stddev, size=[out_size,
64 in_size])))
65 self._bias = nn.Parameter(torch.zeros([out_size]))
66
67 def forward(self, tensor):
68 y = F.linear(tensor, self._weight, self._bias)
69 return F.relu(y) if self._activate_relu else y

72class MLP(nn.Module):
73 """A simple network built from nn.linear layers."""
74
75 def __init__(self,
76 input_size,
77 hidden_sizes,
78 output_size,
79 activate_final=False):
80 """Create the MLP.
81
82 Args:
83 input_size: (int) number of inputs
84 hidden_sizes: (list) sizes (number of units) of each hidden layer
85 output_size: (int) number of outputs
86 activate_final: (bool) should final layer should include a ReLU
87 """
88
89 super(MLP, self).__init__()
90 self._layers = []
91 # Hidden layers
92 for size in hidden_sizes:
93 self._layers.append(SonnetLinear(in_size=input_size, out_size=size))
94 input_size = size
95 # Output layer
96 self._layers.append(
97 SonnetLinear(
98 in_size=input_size,
99 out_size=output_size,
100 activate_relu=activate_final))
101
102 self.model = nn.ModuleList(self._layers)
103
104 def forward(self, x):
105 for layer in self.model:
106 x = layer(x)
107 return x

116 def __init__(self,
117 player_id,
118 state_representation_size,
119 num_actions,
120 hidden_layers_sizes=128,
121 replay_buffer_capacity=10000,
122 batch_size=128,
123 replay_buffer_class=ReplayBuffer,
124 learning_rate=0.01,
125 update_target_network_every=1000,
126 learn_every=10,
127 discount_factor=1.0,
128 min_buffer_size_to_learn=1000,
129 epsilon_start=1.0,
130 epsilon_end=0.1,
131 epsilon_decay_duration=int(1e6),
132 optimizer_str="sgd",
133 loss_str="mse"):
134 """Initialize the DQN agent."""
135
136 # This call to locals() is used to store every argument used to initialize
137 # the class instance, so it can be copied with no hyperparameter change.
138 self._kwargs = locals()
139
140 self.player_id = player_id
141 self._num_actions = num_actions
142 if isinstance(hidden_layers_sizes, int):
143 hidden_layers_sizes = [hidden_layers_sizes]
144 self._layer_sizes = hidden_layers_sizes
145 self._batch_size = batch_size
146 self._update_target_network_every = update_target_network_every
147 self._learn_every = learn_every
148 self._min_buffer_size_to_learn = min_buffer_size_to_learn
149 self._discount_factor = discount_factor
150
151 self._epsilon_start = epsilon_start
152 self._epsilon_end = epsilon_end
153 self._epsilon_decay_duration = epsilon_decay_duration
154
155 # TODO(author6) Allow for optional replay buffer config.
156 if not isinstance(replay_buffer_capacity, int):
157 raise ValueError("Replay buffer capacity not an integer.")
158 self._replay_buffer = replay_buffer_class(replay_buffer_capacity)
159 self._prev_timestep = None
160 self._prev_action = None
161
162 # Step counter to keep track of learning, eps decay and target network.
163 self._step_counter = 0
164
165 # Keep track of the last training loss achieved in an update step.
166 self._last_loss_value = None
167
168 # Create the Q-network instances
169 self._q_network = MLP(state_representation_size, self._layer_sizes,
170 num_actions)
171
172 self._target_q_network = MLP(state_representation_size, self._layer_sizes,
173 num_actions)
174
175 if loss_str == "mse":
176 self.loss_class = F.mse_loss
177 elif loss_str == "huber":
178 self.loss_class = F.smooth_l1_loss
179 else:
180 raise ValueError("Not implemented, choose from 'mse', 'huber'.")
181
182 if optimizer_str == "adam":
183 self._optimizer = torch.optim.Adam(
184 self._q_network.parameters(), lr=learning_rate)
185 elif optimizer_str == "sgd":
186 self._optimizer = torch.optim.SGD(
187 self._q_network.parameters(), lr=learning_rate)
188 else:
189 raise ValueError("Not implemented, choose from 'adam' and 'sgd'.")

191 def step(self, time_step, is_evaluation=False, add_transition_record=True):
192 """Returns the action to be taken and updates the Q-network if needed.
193
194 Args:
195 time_step: an instance of rl_environment.TimeStep.
196 is_evaluation: bool, whether this is a training or evaluation call.
197 add_transition_record: Whether to add to the replay buffer on this step.
198
199 Returns:
200 A `rl_agent.StepOutput` containing the action probs and chosen action.
201 """
202
203 # Act step: don't act at terminal info states or if its not our turn.
204 if (not time_step.last()) and (
205 time_step.is_simultaneous_move() or
206 self.player_id == time_step.current_player()):
207 info_state = time_step.observations["info_state"][self.player_id]
208 legal_actions = time_step.observations["legal_actions"][self.player_id]
209 epsilon = self._get_epsilon(is_evaluation)
210 action, probs = self._epsilon_greedy(info_state, legal_actions, epsilon)
211 else:
212 action = None
213 probs = []
214
215 # Don't mess up with the state during evaluation.
216 if not is_evaluation:
217 self._step_counter += 1
218
219 if self._step_counter % self._learn_every == 0:
220 self._last_loss_value = self.learn()
221
222 if self._step_counter % self._update_target_network_every == 0:
223 # state_dict method returns a dictionary containing a whole state of the
224 # module.
225 self._target_q_network.load_state_dict(self._q_network.state_dict())
226
227 if self._prev_timestep and add_transition_record:
228 # We may omit record adding here if it's done elsewhere.
229 self.add_transition(self._prev_timestep, self._prev_action, time_step)
230
231 if time_step.last(): # prepare for the next episode.
232 self._prev_timestep = None
233 self._prev_action = None
234 return
235 else:
236 self._prev_timestep = time_step
237 self._prev_action = action
238
239 return rl_agent.StepOutput(action=action, probs=probs)

241 def add_transition(self, prev_time_step, prev_action, time_step):
242 """Adds the new transition using `time_step` to the replay buffer.
243
244 Adds the transition from `self._prev_timestep` to `time_step` by
245 `self._prev_action`.
246
247 Args:
248 prev_time_step: prev ts, an instance of rl_environment.TimeStep.
249 prev_action: int, action taken at `prev_time_step`.
250 time_step: current ts, an instance of rl_environment.TimeStep.
251 """
252 assert prev_time_step is not None
253 legal_actions = (time_step.observations["legal_actions"][self.player_id])
254 legal_actions_mask = np.zeros(self._num_actions)
255 legal_actions_mask[legal_actions] = 1.0
256 transition = Transition(
257 info_state=(
258 prev_time_step.observations["info_state"][self.player_id][:]),
259 action=prev_action,
260 reward=time_step.rewards[self.player_id],
261 next_info_state=time_step.observations["info_state"][self.player_id][:],
262 is_final_step=float(time_step.last()),
263 legal_actions_mask=legal_actions_mask)
264 self._replay_buffer.add(transition)

266 def _epsilon_greedy(self, info_state, legal_actions, epsilon):
267 """Returns a valid epsilon-greedy action and valid action probs.
268
269 Action probabilities are given by a softmax over legal q-values.
270
271 Args:
272 info_state: hashable representation of the information state.
273 legal_actions: list of legal actions at `info_state`.
274 epsilon: float, probability of taking an exploratory action.
275
276 Returns:
277 A valid epsilon-greedy action and valid action probabilities.
278 """
279 probs = np.zeros(self._num_actions)
280 if np.random.rand() < epsilon:
281 action = np.random.choice(legal_actions)
282 probs[legal_actions] = 1.0 / len(legal_actions)
283 else:
284 info_state = torch.Tensor(np.reshape(info_state, [1, -1]))
285 q_values = self._q_network(info_state).detach()[0]
286 legal_q_values = q_values[legal_actions]
287 action = legal_actions[torch.argmax(legal_q_values)]
288 probs[action] = 1.0
289 return action, probs

291 def _get_epsilon(self, is_evaluation, power=1.0):
292 """Returns the evaluation or decayed epsilon value."""
293 if is_evaluation:
294 return 0.0
295 decay_steps = min(self._step_counter, self._epsilon_decay_duration)
296 decayed_epsilon = (
297 self._epsilon_end + (self._epsilon_start - self._epsilon_end) *
298 (1 - decay_steps / self._epsilon_decay_duration)**power)
299 return decayed_epsilon

301 def learn(self):
302 """Compute the loss on sampled transitions and perform a Q-network update.
303
304 If there are not enough elements in the buffer, no loss is computed and
305 `None` is returned instead.
306
307 Returns:
308 The average loss obtained on this batch of transitions or `None`.
309 """
310
311 if (len(self._replay_buffer) < self._batch_size or
312 len(self._replay_buffer) < self._min_buffer_size_to_learn):
313 return None
314
315 transitions = self._replay_buffer.sample(self._batch_size)
316 info_states = torch.Tensor([t.info_state for t in transitions])
317 actions = torch.LongTensor([t.action for t in transitions])
318 rewards = torch.Tensor([t.reward for t in transitions])
319 next_info_states = torch.Tensor([t.next_info_state for t in transitions])
320 are_final_steps = torch.Tensor([t.is_final_step for t in transitions])
321 legal_actions_mask = torch.Tensor(
322 np.array([t.legal_actions_mask for t in transitions]))
323
324 self._q_values = self._q_network(info_states)
325 self._target_q_values = self._target_q_network(next_info_states).detach()
326
327 illegal_actions_mask = 1 - legal_actions_mask
328 legal_target_q_values = self._target_q_values.masked_fill(
329 illegal_actions_mask.bool(), ILLEGAL_ACTION_LOGITS_PENALTY
330 )
331 max_next_q = torch.max(legal_target_q_values, dim=1)[0]
332
333 target = (
334 rewards + (1 - are_final_steps) * self._discount_factor * max_next_q)
335 action_indices = torch.stack([
336 torch.arange(self._q_values.shape[0], dtype=torch.long), actions
337 ],
338 dim=0)
339 predictions = self._q_values[list(action_indices)]
340
341 loss = self.loss_class(predictions, target)
342
343 self._optimizer.zero_grad()
344 loss.backward()
345 self._optimizer.step()
346
347 return loss

378 def copy_with_noise(self, sigma=0.0, copy_weights=True):
379 """Copies the object and perturbates it with noise.
380
381 Args:
382 sigma: gaussian dropout variance term : Multiplicative noise following
383 (1+sigma*epsilon), epsilon standard gaussian variable, multiplies each
384 model weight. sigma=0 means no perturbation.
385 copy_weights: Boolean determining whether to copy model weights (True) or
386 just model hyperparameters.
387
388 Returns:
389 Perturbated copy of the model.
390 """
391 _ = self._kwargs.pop("self", None)
392 copied_object = DQN(**self._kwargs)
393
394 q_network = getattr(copied_object, "_q_network")
395 target_q_network = getattr(copied_object, "_target_q_network")
396
397 if copy_weights:
398 with torch.no_grad():
399 for q_model in q_network.model:
400 q_model.weight *= (1 + sigma * torch.randn(q_model.weight.shape))
401 for tq_model in target_q_network.model:
402 tq_model.weight *= (1 + sigma * torch.randn(tq_model.weight.shape))
403 return copied_object
