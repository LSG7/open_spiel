49 bool AllActionsInOtherActions(const std::vector<Action>& actions,
50  const std::vector<Action>& other_actions) {
51  for (auto const& action : actions) {
52  if (std::find(other_actions.begin(), other_actions.end(), action) ==

78 template <typename T>
79 bool AllEq(const std::vector<T>& xs0, const std::vector<T>& xs1) {
80  if (xs0.size() != xs1.size()) return false;
81  for (int i = 0; i < xs0.size(); i++) {
82  if (xs0.at(i) != xs1.at(i)) return false;
83  }
84  return true;
85 }
