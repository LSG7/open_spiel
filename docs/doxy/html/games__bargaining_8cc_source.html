32void open_spiel::init_pyspiel_games_bargaining(py::module& m) {
33 py::class_<Instance>(m, "Instance")
34 .def(py::init<>())
35 .def_readwrite("pool", &Instance::pool)
36 .def_readwrite("values", &Instance::values);
37
38 py::class_<Offer>(m, "Offer")
39 .def(py::init<>())
40 .def_readwrite("quantities", &Offer::quantities);
41
42 py::classh<BargainingState, State>(m, "BargainingState")
43 .def("instance", &BargainingState::GetInstance)
44 .def("offers", &BargainingState::Offers)
45 .def("agree_action", &BargainingState::AgreeAction)
46 // set_instance(instance)
47 .def("set_instance", &BargainingState::SetInstance)
48 // Pickle support
49 .def(py::pickle(
50 [](const BargainingState& state) { // __getstate__
51 return SerializeGameAndState(*state.GetGame(), state);
52 },
53 [](const std::string& data) { // __setstate__
54 std::pair<std::shared_ptr<const Game>, std::unique_ptr<State>>
55 game_and_state = DeserializeGameAndState(data);
56 return dynamic_cast<BargainingState*>(
57 game_and_state.second.release());
58 }));
59
60 py::classh<BargainingGame, Game>(m, "BargainingGame")
61 .def("all_instances", &BargainingGame::AllInstances)
62 // get_offer_by_quantities(quantities: List[int]). Returns a tuple
63 // of (offer, OpenSpiel action)
64 .def("get_offer_by_quantities", &BargainingGame::GetOfferByQuantities)
65 // Pickle support
66 .def(py::pickle(
67 [](std::shared_ptr<const BargainingGame> game) { // __getstate__
68 return game->ToString();
69 },
70 [](const std::string& data) { // __setstate__
71 return std::dynamic_pointer_cast<BargainingGame>(
72 std::const_pointer_cast<Game>(LoadGame(data)));
73 }));
74}
