60 kSequential, // Turn-based games.
61 // Mean field game. In particular, this adds mean field nodes. Support for

67 // Are there any chance nodes? If so, how is chance treated?
68 // Either all possible chance outcomes are explicitly returned as
69 // ChanceOutcomes(), and the result of ApplyAction() is deterministic. Or
70 // just one ChanceOutcome is returned, and the result of ApplyAction() is
71 // stochastic. If in doubt, it is better to implement stochastic games with
72 // kExplicitStochastic, as this makes more information available to any
73 // learning algorithms you choose to use (i.e. the whole chance outcome
74 // distribution is visible to the algorithm, rather than just the sampled
75 // outcome). For more discussion of this field, see the github issue:
76 // https://github.com/deepmind/open_spiel/issues/792.
77 enum class ChanceMode {
78 kDeterministic, // No chance nodes
79 kExplicitStochastic, // Has at least one chance node, all with
80 // deterministic ApplyAction()
81 kSampledStochastic, // At least one chance node with non-deterministic
82 // ApplyAction()
83 };
84 ChanceMode chance_mode;

245 // `LegalActions(Player player)` is valid for all nodes in all games,

325 // Returns reward from the most recent state transition (s, a, s') for all
326 // players. This is provided so that RL-style games with intermediate rewards
327 // (along the episode, rather than just one value at the end) can be properly
328 // implemented. The default is to return 0 except at terminal states, where
329 // the terminal returns are returned.
330 //
331 // Note 1: should not be called at chance nodes (undefined and crashes).
332 // Note 2: This must agree with Returns(). That is, for any state S_t,
333 // Returns(St) = Sum(Rewards(S_0), Rewards(S_1)... Rewards(S_t)).
334 // The default implementation is only correct for games that only
335 // have a final reward. Games with intermediate rewards must override
336 // both this method and Returns().

338 if (IsTerminal()) {
339 return Returns();

341 SPIEL_CHECK_FALSE(IsChanceNode());
342 return std::vector<double>(num_players_, 0.0);
343 }
344 }

433 // For imperfect information games. Returns an identifier for the current
434 // information state for the specified player.
435 // Different ground states can yield the same information state for a player
436 // when the only part of the state that differs is not observable by that
437 // player (e.g. opponents' cards in Poker.)
438 //
439 // The identifiers must be unique across all players.
440 // This allows an algorithm to maintain a single table of identifiers
441 // instead of maintaining a table per player to avoid name collisions.
442 //
443 // A simple way to do so is for example, in a card game, if both players can
444 // hold the card Jack, the identifier can contain player identification as
445 // well, like P1Jack and P2Jack. However prefixing by player number is not
446 // a requirement. The only thing that is necessary is that it is unambiguous
447 // who is the observer.
448 //
449 // Games that do not have imperfect information do not need to implement
450 // these methods, but most algorithms intended for imperfect information
451 // games will work on perfect information games provided the InformationState
452 // is returned in a form they support. For example, InformationState()
453 // could simply return the history for a perfect information game.
454 //
455 // A valid InformationStateString must be returned at terminal states, since
456 // this is required in some applications (e.g. final observation in an RL
457 // environment).
458 //
459 // The information state should be perfect-recall, i.e. if two states
460 // have a different InformationState, then all successors of one must have
461 // a different InformationState to all successors of the other.
462 // For example, in tic-tac-toe, the current state of the board would not be
463 // a perfect-recall representation, but the sequence of moves played would
464 // be.
465 //
466 // If you implement both InformationState and Observation, the two must be
467 // consistent for all the players (even the non-acting player(s)).
468 // By consistency we mean that when you maintain an Action-Observation
469 // history (AOH) for different ground states, the (in)equality of two AOHs
470 // implies the (in)equality of two InformationStates. In other words, AOH is a
471 // factored representation of InformationState.
472 //
473 // For details, see Section 3.1 of https://arxiv.org/abs/1908.09453
474 // or Section 2.1 of https://arxiv.org/abs/1906.11110
475
476 // There are currently no use-case for calling this function with

983 // Access to game parameters. Returns the value provided by the user. If not:
984 // - Defaults to the value stored as the default in
985 // game_type.parameter_specification if the `default_value` is absl::nullopt
986 // - Returns `default_value` if provided.
987 template <typename T>
988 T ParameterValue(const std::string& key,
989 absl::optional<T> default_value = absl::nullopt) const {
990 // Return the value if found.
991 auto iter = game_parameters_.find(key);
992 if (iter != game_parameters_.end()) {
993 return iter->second.value<T>();
994 }
995
996 // Pick the defaulted value.
997 GameParameter default_game_parameter;
998 if (default_value.has_value()) {
999 default_game_parameter = GameParameter(default_value.value());
1000 } else {
1001 auto default_iter = game_type_.parameter_specification.find(key);
1002 if (default_iter == game_type_.parameter_specification.end()) {
1003 SpielFatalError(absl::StrCat("The parameter for ", key,
1004 " is missing in game ", ToString()));
1005 }
1006 default_game_parameter = default_iter->second;
1007 }
1008
1009 // Return the default value, storing it.
1010 absl::MutexLock lock(&mutex_defaulted_parameters_);
1011 iter = defaulted_parameters_.find(key);
1012 if (iter == defaulted_parameters_.end()) {
1013 // We haven't previously defaulted this value, so store the default we
1014 // used.
1015 defaulted_parameters_[key] = default_game_parameter;
1016 } else {
1017 // Already defaulted, so check we are being consistent.
1018 // Using different default values at different times means the game isn't
1019 // well-defined.
1020 if (default_game_parameter != iter->second) {
1021 SpielFatalError(absl::StrCat("Parameter ", key, " is defaulted to ",
1022 default_game_parameter.ToReprString(),
1023 " having previously been defaulted to ",
1024 iter->second.ToReprString(), " in game ",
1025 ToString()));
1026 }
1027 }
1028 return default_game_parameter.value<T>();
1029 }
