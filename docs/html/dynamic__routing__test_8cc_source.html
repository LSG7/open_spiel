306 // Test travel time update based on distribution is correct.
307 void TestCorrectTravelTimeUpdate() {
308  auto game = LoadGame(
309  "mfg_dynamic_routing(max_num_time_step=100,time_step_length=0.05,"
310  "network=braess)");
311  auto state = game->NewInitialState();
312 
313  SPIEL_CHECK_EQ(state->ToString(), "Before initial chance node.");
314  state->ApplyAction(0);
315  SPIEL_CHECK_EQ(state->ToString(),
316  "Location=O->A, waiting time=0, t=0, destination=D->E");
317  SPIEL_CHECK_EQ(state->LegalActions(), (std::vector<Action>{1, 2}));
318  state->ApplyAction(1);
319  SPIEL_CHECK_EQ(
320  state->ToString(),
321  "Location=A->B, waiting time=-1, t=1_mean_field, destination=D->E");
322 
323  std::vector<double> distribution{1};
324  state->UpdateDistribution({.5});
325  // Waiting time (in unit of time) = 1.0 (free flow travel time on A->B) +
326  // .5 (% player on A->B) * 5 (num of players) / 5 (capacity on A->B) = 1.5
327  // Waiting time (in time step) = 1.5 / 0.05 (time step lenght)
328  // - 1 (one time step for the current time running) = 29
329  SPIEL_CHECK_EQ(state->ToString(),
330  "Location=A->B, waiting time=29, t=1, destination=D->E");
331 }
