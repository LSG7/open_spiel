32def _eval_agent(env, agent, num_episodes):
33 """Evaluates `agent` for `num_episodes`."""
34 rewards = 0.0
35 for _ in range(num_episodes):
36 time_step = env.reset()
37 episode_reward = 0
38 while not time_step.last():
39 agent_output = agent.step(time_step, is_evaluation=True)
40 time_step = env.step([agent_output.action])
41 episode_reward += time_step.rewards[0]
42 rewards += episode_reward
43 return rewards / num_episodes

46def main_loop(unused_arg):
47 """Trains a Policy Gradient agent in the catch environment."""
48 env = catch.Environment()
49 info_state_size = env.observation_spec()["info_state"][0]
50 num_actions = env.action_spec()["num_actions"]
51
52 train_episodes = FLAGS.num_episodes
53
54 agent = policy_gradient.PolicyGradient(
55 player_id=0,
56 info_state_size=info_state_size,
57 num_actions=num_actions,
58 loss_str=FLAGS.algorithm,
59 hidden_layers_sizes=[128, 128],
60 lambda_=1.0,
61 entropy_cost=0.01,
62 critic_learning_rate=0.1,
63 pi_learning_rate=0.1,
64 num_critic_before_pi=3)
65
66 # Train agent
67 for ep in range(train_episodes):
68 time_step = env.reset()
69 while not time_step.last():
70 agent_output = agent.step(time_step)
71 action_list = [agent_output.action]
72 time_step = env.step(action_list)
73 # Episode is over, step agent with final info state.
74 agent.step(time_step)
75
76 if ep and ep % FLAGS.eval_every == 0:
77 logging.info("-" * 80)
78 logging.info("Episode %s", ep)
79 logging.info("Loss: %s", agent.loss)
80 avg_return = _eval_agent(env, agent, 100)
81 logging.info("Avg return: %s", avg_return)
82
83
84if __name__ == "__main__":
85 app.run(main_loop)
