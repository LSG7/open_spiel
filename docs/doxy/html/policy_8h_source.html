67// A general policy object. A policy is a mapping from states to list of
68// (action, prob) pairs for all the legal actions at the state.
69class Policy {
70 public:
71 virtual ~Policy() = default;
72
73 // A convenience method for callers that want to use arrays.
74 virtual std::pair<std::vector<Action>, std::vector<double>>
75 GetStatePolicyAsParallelVectors(const State& state) const {
76 std::pair<std::vector<Action>, std::vector<double>> parray;
77 for (const auto& action_and_prob : GetStatePolicy(state)) {
78 parray.first.push_back(action_and_prob.first);
79 parray.second.push_back(action_and_prob.second);
80 }
81 return parray;
82 }
83
84 // A convenience method for callers that want to use arrays.
85 virtual std::pair<std::vector<Action>, std::vector<double>>
86 GetStatePolicyAsParallelVectors(const std::string& info_state) const {
87 std::pair<std::vector<Action>, std::vector<double>> parray;
88 for (const auto& action_and_prob : GetStatePolicy(info_state)) {
89 parray.first.push_back(action_and_prob.first);
90 parray.second.push_back(action_and_prob.second);
91 }
92 return parray;
93 }
94
95 virtual std::unordered_map<Action, double> GetStatePolicyAsMap(
96 const State& state) const {
97 std::unordered_map<Action, double> pmap;
98 for (const auto& action_and_prob : GetStatePolicy(state)) {
99 pmap[action_and_prob.first] = action_and_prob.second;
100 }
101 return pmap;
102 }
103
104 virtual std::unordered_map<Action, double> GetStatePolicyAsMap(
105 const std::string& info_state) const {
106 std::unordered_map<Action, double> pmap;
107 for (const auto& action_and_prob : GetStatePolicy(info_state)) {
108 pmap[action_and_prob.first] = action_and_prob.second;
109 }
110 return pmap;
111 }
112
113 // Returns a list of (action, prob) pairs for the policy for the current
114 // player at this state. If the policy is not available at the state, returns
115 // an empty list.
116 virtual ActionsAndProbs GetStatePolicy(const State& state) const {
117 return GetStatePolicy(state, state.CurrentPlayer());
118 }
119
120 // Returns a list of (action, prob) pairs for the policy for the specified
121 // player at this state. If the policy is not available at the state, returns
122 // an empty list.
123 virtual ActionsAndProbs GetStatePolicy(const State& state,
124 Player player) const {
125 return GetStatePolicy(state.InformationStateString(player));
126 }
127
128 // Returns a list of (action, prob) pairs for the policy at this info state.
129 // If the policy is not available at the state, returns and empty list.
130 // It is sufficient for subclasses to override only this method, but not all
131 // forms of policies will be able to do so from just the information state.
132 virtual ActionsAndProbs GetStatePolicy(const std::string& info_state) const {
133 SpielFatalError("GetStatePolicy(const std::string&) unimplemented.");
134 }
135
136 // Each override must write out the class’s identity followed by ":" as the
137 // very first thing so that the DeserializePolicy method can then call the

139 // DeserializePolicy below for an example. The double_precision parameter

141 // formatting, value -1 formats doubles with lossless, non-portable bitwise
142 // representation hex strings.
143 virtual std::string Serialize(int double_precision = -1,
144 std::string delimiter = "<~>") const {
145 SpielFatalError("Serialize(std::string delimiter) unimplemented.");

152// A tabular policy represented internally as a map. Note that this
153// implementation is not directly compatible with the Python TabularPolicy
154// implementation; the latter is implemented as a table of size
155// [num_states, num_actions], while this is implemented as a map. It is
156// non-trivial to convert between the two, but we have a function that does so
157// in the open_spiel/python/policy.py file.
158class TabularPolicy : public Policy {
159 public:
160 TabularPolicy() = default;
161 TabularPolicy(const Game& game); // Construct a uniform random policy.
162 TabularPolicy(const TabularPolicy& other) = default;
163 TabularPolicy(const std::unordered_map<std::string, ActionsAndProbs>& table)
164 : policy_table_(table) {}
165
166 // Converts a policy to a TabularPolicy.
167 TabularPolicy(const Game& game, const Policy& policy) : TabularPolicy(game) {
168 for (auto& [infostate, is_policy] : policy_table_) {
169 is_policy = policy.GetStatePolicy(infostate);
170 }
171 }
172
173 // Creates a new TabularPolicy from a deterministic policy encoded as a
174 // {info_state_str -> action} dict. The dummy_policy is used to initialize
175 // the initial mapping.
176 TabularPolicy(const TabularPolicy& dummy_policy,
177 const std::unordered_map<std::string, Action>& action_map)
178 : policy_table_(dummy_policy.policy_table_) {
179 for (const auto& entry : action_map) {
180 std::string info_state = entry.first;
181 Action action_taken = action_map.at(entry.first);
182 for (auto& action_and_prob : policy_table_[info_state]) {
183 action_and_prob.second =
184 (action_and_prob.first == action_taken ? 1.0 : 0.0);
185 }
186 }
187 }
188
189 ActionsAndProbs GetStatePolicy(const std::string& info_state) const override {
190 auto iter = policy_table_.find(info_state);
191 if (iter == policy_table_.end()) {
192 return {};
193 } else {
194 return iter->second;
195 }
196 }
197
198 std::string Serialize(int double_precision = -1,
199 std::string delimiter = "<~>") const override {
200 SPIEL_CHECK_GE(double_precision, -1);
201 if (delimiter == "," || delimiter == "=") {
202 // The two delimiters are used for de/serialization of policy_table_
203 SpielFatalError(
204 "Please select a different delimiter,"
205 "invalid values are \",\" and \"=\".");
206 }
207 std::string str = "TabularPolicy:";
208 if (policy_table_.empty()) return str;
209
210 for (auto const& [info_state, policy] : policy_table_) {
211 if (info_state.find(delimiter) != std::string::npos) {
212 SpielFatalError(absl::StrCat(
213 "Info state contains delimiter \"", delimiter,
214 "\", please fix the info state or select a different delimiter."));
215 }
216
217 std::string policy_str;
218 if (double_precision == -1) {
219 policy_str =
220 absl::StrJoin(policy, ",",
221 absl::PairFormatter(absl::AlphaNumFormatter(), "=",
222 HexDoubleFormatter()));
223 } else {
224 policy_str = absl::StrJoin(
225 policy, ",",
226 absl::PairFormatter(absl::AlphaNumFormatter(), "=",
227 SimpleDoubleFormatter(double_precision)));
228 }
229 absl::StrAppend(&str, info_state, delimiter, policy_str, delimiter);
230 }
231 // Remove the trailing delimiter
232 str.erase(str.length() - delimiter.length());
233 return str;
234 }
235
236 // Set (overwrite) all the state policies contained in another policy within
237 // this policy. Does not change other state policies not contained in this
238 // policy.
239 void ImportPolicy(const TabularPolicy& other_policy) {
240 for (const auto& [info_state, actions_and_probs] :
241 other_policy.policy_table_) {
242 SetStatePolicy(info_state, actions_and_probs);
243 }
244 }
245
246 // Set the probability for action at the info state. If the info state is not
247 // in the policy, it is added. If the action is not in the info state policy,
248 // it is added. Otherwise it is modified.
249 void SetProb(const std::string& info_state, Action action, double prob) {
250 auto iter = policy_table_.find(info_state);
251 if (iter == policy_table_.end()) {
252 auto iter_and_bool = policy_table_.insert({info_state, {}});
253 iter = iter_and_bool.first;
254 }
255 open_spiel::SetProb(&(iter->second), action, prob);
256 }
257
258 void SetStatePolicy(const std::string& info_state,
259 const ActionsAndProbs& state_policy) {
260 policy_table_[info_state] = state_policy;
261 }
262
263 std::unordered_map<std::string, ActionsAndProbs>& PolicyTable() {
264 return policy_table_;
265 }
266
267 const std::unordered_map<std::string, ActionsAndProbs>& PolicyTable() const {
268 return policy_table_;
269 }
270
271 int size() const { return policy_table_.size(); }
272
273 std::string ToString() const;
274
275 // A ToString where the keys are sorted.
276 std::string ToStringSorted() const;
277
278 protected:
279 std::unordered_map<std::string, ActionsAndProbs> policy_table_;
280};

316// Chooses all legal actions with equal probability. This is equivalent to the
317// tabular version, except that this works for large games.
318class UniformPolicy : public Policy {
319 public:
320 ActionsAndProbs GetStatePolicy(const State& state,
321 Player player) const override {
322 if (state.IsSimultaneousNode()) {
323 return UniformStatePolicy(state, player);
324 } else {
325 SPIEL_CHECK_TRUE(state.IsPlayerActing(player));
326 return UniformStatePolicy(state);
327 }
328 }
329
330 std::string Serialize(int double_precision = -1,
331 std::string delimiter = "") const override {
332 return "UniformPolicy:";
333 }
334};
